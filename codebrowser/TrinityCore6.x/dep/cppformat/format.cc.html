109// Checks if a value fits in int - used to avoid warnings about comparing
110// signed and unsigned integers.
111template <bool IsSigned>
112struct IntChecker {
113 template <typename T>
114 static bool fits_in_int(T value) {
115 unsigned max = INT_MAX;
116 return value <= max;
117 }
118 static bool fits_in_int(bool) { return true; }
119};

121template <>
122struct IntChecker<true> {
123 template <typename T>
124 static bool fits_in_int(T value) {
125 return value >= INT_MIN && value <= INT_MAX;
126 }
127 static bool fits_in_int(int) { return true; }

134// Portable thread-safe version of strerror.
135// Sets buffer to point to a string describing the error code.
136// This can be either a pointer to a string stored in buffer,
137// or a pointer to some static immutable string.
138// Returns one of the following values:
139// 0 - success
140// ERANGE - buffer is not large enough to store the error message
141// other - failure
142// Buffer should be at least of size 1.
143int safe_strerror(
144 int error_code, char *&buffer, std::size_t buffer_size) FMT_NOEXCEPT {
145 FMT_ASSERT(buffer != 0 && buffer_size != 0, "invalid buffer");
146
147 class StrError {
148 private:
149 int error_code_;
150 char *&buffer_;
151 std::size_t buffer_size_;
152
153 // A noop assignment operator to avoid bogus warnings.
154 void operator=(const StrError &) {}
155
156 // Handle the result of XSI-compliant version of strerror_r.
157 int handle(int result) {
158 // glibc versions before 2.13 return result in errno.
159 return result == -1 ? errno : result;
160 }
161
162 // Handle the result of GNU-specific version of strerror_r.
163 int handle(char *message) {
164 // If the buffer is full then the message is probably truncated.
165 if (message == buffer_ && strlen(buffer_) == buffer_size_ - 1)
166 return ERANGE;
167 buffer_ = message;
168 return 0;
169 }
170
171 // Handle the case when strerror_r is not available.
172 int handle(fmt::internal::Null<>) {
173 return fallback(strerror_s(buffer_, buffer_size_, error_code_));
174 }
175
176 // Fallback to strerror_s when strerror_r is not available.
177 int fallback(int result) {
178 // If the buffer is full then the message is probably truncated.
179 return result == 0 && strlen(buffer_) == buffer_size_ - 1 ?
180 ERANGE : result;
181 }
182
183 // Fallback to strerror if strerror_r and strerror_s are not available.
184 int fallback(fmt::internal::Null<>) {
185 errno = 0;
186 buffer_ = strerror(error_code_);
187 return errno;
188 }
189
190 public:
191 StrError(int err_code, char *&buf, std::size_t buf_size)
192 : error_code_(err_code), buffer_(buf), buffer_size_(buf_size) {}
193
194 int run() {
195 strerror_r(0, 0, ""); // Suppress a warning about unused strerror_r.
196 return handle(strerror_r(error_code_, buffer_, buffer_size_));
197 }
198 };
199 return StrError(error_code, buffer, buffer_size).run();
200}

202void format_error_code(fmt::Writer &out, int error_code,
203 fmt::StringRef message) FMT_NOEXCEPT {
204 // Report error code making sure that the output fits into
205 // INLINE_BUFFER_SIZE to avoid dynamic memory allocation and potential
206 // bad_alloc.
207 out.clear();
208 static const char SEP[] = ": ";
209 static const char ERROR_STR[] = "error ";
210 fmt::internal::IntTraits<int>::MainType ec_value = error_code;
211 // Subtract 2 to account for terminating null characters in SEP and ERROR_STR.
212 std::size_t error_code_size = sizeof(SEP) + sizeof(ERROR_STR) - 2;
213 error_code_size += fmt::internal::count_digits(ec_value);
214 if (message.size() <= fmt::internal::INLINE_BUFFER_SIZE - error_code_size)
215 out << message << SEP;
216 out << ERROR_STR << error_code;
217 assert(out.size() <= fmt::internal::INLINE_BUFFER_SIZE);
218}

221 int error_code, fmt::StringRef message) FMT_NOEXCEPT {

237// Checks if an argument is a valid printf width specifier and sets
238// left alignment if it is negative.
239class WidthHandler : public fmt::internal::ArgVisitor<WidthHandler, unsigned> {
240 private:
241 fmt::FormatSpec &spec_;
242
243 FMT_DISALLOW_COPY_AND_ASSIGN(WidthHandler);
244
245 public:
246 explicit WidthHandler(fmt::FormatSpec &spec) : spec_(spec) {}
247
248 void report_unhandled_arg() {
249 FMT_THROW(fmt::FormatError("width is not integer"));
250 }
251
252 template <typename T>
253 unsigned visit_any_int(T value) {
254 typedef typename fmt::internal::IntTraits<T>::MainType UnsignedType;
255 UnsignedType width = value;
256 if (fmt::internal::is_negative(value)) {
257 spec_.align_ = fmt::ALIGN_LEFT;
258 width = 0 - width;
259 }
260 if (width > INT_MAX)
261 FMT_THROW(fmt::FormatError("number is too big"));
262 return static_cast<unsigned>(width);
263 }
264};

281// Converts an integer argument to an integral type T for printf.
282template <typename T>
283class ArgConverter : public fmt::internal::ArgVisitor<ArgConverter<T>, void> {
284 private:
285 fmt::internal::Arg &arg_;
286 wchar_t type_;
287
288 FMT_DISALLOW_COPY_AND_ASSIGN(ArgConverter);
289
290 public:
291 ArgConverter(fmt::internal::Arg &arg, wchar_t type)
292 : arg_(arg), type_(type) {}
293
294 void visit_bool(bool value) {
295 if (type_ != 's')
296 visit_any_int(value);
297 }
298
299 template <typename U>
300 void visit_any_int(U value) {
301 bool is_signed = type_ == 'd' || type_ == 'i';
302 using fmt::internal::Arg;
303 if (sizeof(T) <= sizeof(int)) {
304 // Extra casts are used to silence warnings.
305 if (is_signed) {
306 arg_.type = Arg::INT;
307 arg_.int_value = static_cast<int>(static_cast<T>(value));
308 } else {
309 arg_.type = Arg::UINT;
310 arg_.uint_value = static_cast<unsigned>(
311 static_cast<typename fmt::internal::MakeUnsigned<T>::Type>(value));
312 }
313 } else {
314 if (is_signed) {
315 arg_.type = Arg::LONG_LONG;
316 arg_.long_long_value =
317 static_cast<typename fmt::internal::MakeUnsigned<U>::Type>(value);
318 } else {
319 arg_.type = Arg::ULONG_LONG;
320 arg_.ulong_long_value =
321 static_cast<typename fmt::internal::MakeUnsigned<U>::Type>(value);
322 }
323 }
324 }
325};

347template <typename Char>
348class PrintfArgFormatter :
349 public ArgFormatterBase<PrintfArgFormatter<Char>, Char> {
350
351 void write_null_pointer() {
352 this->spec().type_ = 0;
353 this->write("(nil)");
354 }
355
356 typedef ArgFormatterBase<PrintfArgFormatter<Char>, Char> Base;
357
358 public:
359 PrintfArgFormatter(BasicWriter<Char> &w, FormatSpec &s)
360 : ArgFormatterBase<PrintfArgFormatter<Char>, Char>(w, s) {}
361
362 void visit_bool(bool value) {
363 FormatSpec &fmt_spec = this->spec();
364 if (fmt_spec.type_ != 's')
365 return this->visit_any_int(value);
366 fmt_spec.type_ = 0;
367 this->write(value);
368 }
369
370 void visit_char(int value) {
371 const FormatSpec &fmt_spec = this->spec();
372 BasicWriter<Char> &w = this->writer();
373 if (fmt_spec.type_ && fmt_spec.type_ != 'c')
374 w.write_int(value, fmt_spec);
375 typedef typename BasicWriter<Char>::CharPtr CharPtr;
376 CharPtr out = CharPtr();
377 if (fmt_spec.width_ > 1) {
378 Char fill = ' ';
379 out = w.grow_buffer(fmt_spec.width_);
380 if (fmt_spec.align_ != ALIGN_LEFT) {
381 std::fill_n(out, fmt_spec.width_ - 1, fill);
382 out += fmt_spec.width_ - 1;
383 } else {
384 std::fill_n(out + 1, fmt_spec.width_ - 1, fill);
385 }
386 } else {
387 out = w.grow_buffer(1);
388 }
389 *out = static_cast<Char>(value);
390 }
391
392 void visit_cstring(const char *value) {
393 if (value)
394 Base::visit_cstring(value);
395 else if (this->spec().type_ == 'p')
396 write_null_pointer();
397 else
398 this->write("(null)");
399 }
400
401 void visit_pointer(const void *value) {
402 if (value)
403 return Base::visit_pointer(value);
404 this->spec().type_ = 0;
405 write_null_pointer();
406 }
407
408 void visit_custom(Arg::CustomValue c) {
409 BasicFormatter<Char> formatter(ArgList(), this->writer());
410 const Char format_str[] = {'}', 0};
411 const Char *format = format_str;
412 c.format(&formatter, c.value, &format);
413 }
414};

502FMT_FUNC fmt::internal::UTF8ToUTF16::UTF8ToUTF16(fmt::StringRef s) {
503 static const char ERROR_MSG[] = "cannot convert string from UTF-8 to UTF-16";
504 if (s.size() > INT_MAX)
505 FMT_THROW(WindowsError(ERROR_INVALID_PARAMETER, ERROR_MSG));
506 int s_size = static_cast<int>(s.size());
507 int length = MultiByteToWideChar(
508 CP_UTF8, MB_ERR_INVALID_CHARS, s.data(), s_size, 0, 0);
509 if (length == 0)
510 FMT_THROW(WindowsError(GetLastError(), ERROR_MSG));
511 buffer_.resize(length + 1);
512 length = MultiByteToWideChar(
513 CP_UTF8, MB_ERR_INVALID_CHARS, s.data(), s_size, &buffer_[0], length);
514 if (length == 0)
515 FMT_THROW(WindowsError(GetLastError(), ERROR_MSG));
516 buffer_[length] = 0;
517}

526FMT_FUNC int fmt::internal::UTF16ToUTF8::convert(fmt::WStringRef s) {
527 if (s.size() > INT_MAX)
528 return ERROR_INVALID_PARAMETER;
529 int s_size = static_cast<int>(s.size());
530 int length = WideCharToMultiByte(CP_UTF8, 0, s.data(), s_size, 0, 0, 0, 0);
531 if (length == 0)
532 return GetLastError();
533 buffer_.resize(length + 1);
534 length = WideCharToMultiByte(
535 CP_UTF8, 0, s.data(), s_size, &buffer_[0], length, 0, 0);
536 if (length == 0)
537 return GetLastError();
538 buffer_[length] = 0;
539 return 0;
540}

551FMT_FUNC void fmt::internal::format_windows_error(
552 fmt::Writer &out, int error_code,
553 fmt::StringRef message) FMT_NOEXCEPT {
554 class String {
555 private:
556 LPWSTR str_;
557
558 public:
559 String() : str_() {}
560 ~String() { LocalFree(str_); }
561 LPWSTR *ptr() { return &str_; }
562 LPCWSTR c_str() const { return str_; }
563 };
564 FMT_TRY {
565 String system_message;
566 if (FormatMessageW(FORMAT_MESSAGE_ALLOCATE_BUFFER |
567 FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS, 0,
568 error_code, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
569 reinterpret_cast<LPWSTR>(system_message.ptr()), 0, 0)) {
570 UTF16ToUTF8 utf8_message;
571 if (utf8_message.convert(system_message.c_str()) == ERROR_SUCCESS) {
572 out << message << ": " << utf8_message;
573 return;
574 }
575 }
576 } FMT_CATCH(...) {}
577 fmt::format_error_code(out, error_code, message); // 'fmt::' is for bcc32.
578}

582FMT_FUNC void fmt::internal::format_system_error(
583 fmt::Writer &out, int error_code,
584 fmt::StringRef message) FMT_NOEXCEPT {
585 FMT_TRY {
586 MemoryBuffer<char, INLINE_BUFFER_SIZE> buffer;
587 buffer.resize(INLINE_BUFFER_SIZE);
588 for (;;) {
589 char *system_message = &buffer[0];
590 int result = safe_strerror(error_code, system_message, buffer.size());
591 if (result == 0) {
592 out << message << ": " << system_message;
593 return;
594 }
595 if (result != ERANGE)
596 break; // Can't get error message, report error code instead.
597 buffer.resize(buffer.size() * 2);
598 }
599 } FMT_CATCH(...) {}
600 fmt::format_error_code(out, error_code, message); // 'fmt::' is for bcc32.
601}

603template <typename Char>
604void fmt::internal::ArgMap<Char>::init(const ArgList &args) {
605 if (!map_.empty())
606 return;
607 typedef internal::NamedArg<Char> NamedArg;
608 const NamedArg *named_arg = 0;
609 bool use_values =
610 args.type(ArgList::MAX_PACKED_ARGS - 1) == internal::Arg::NONE;
611 if (use_values) {
612 for (unsigned i = 0;/*nothing*/; ++i) {
613 internal::Arg::Type arg_type = args.type(i);
614 switch (arg_type) {
615 case internal::Arg::NONE:
616 return;
617 case internal::Arg::NAMED_ARG:
618 named_arg = static_cast<const NamedArg*>(args.values_[i].pointer);
619 map_.insert(Pair(named_arg->name, *named_arg));
620 break;
621 default:
622 /*nothing*/;
623 }
624 }
625 return;
626 }
627 for (unsigned i = 0; i != ArgList::MAX_PACKED_ARGS; ++i) {
628 internal::Arg::Type arg_type = args.type(i);
629 if (arg_type == internal::Arg::NAMED_ARG) {
630 named_arg = static_cast<const NamedArg*>(args.args_[i].pointer);
631 map_.insert(Pair(named_arg->name, *named_arg));
632 }
633 }
634 for (unsigned i = ArgList::MAX_PACKED_ARGS;/*nothing*/; ++i) {
635 switch (args.args_[i].type) {
636 case internal::Arg::NONE:
637 return;
638 case internal::Arg::NAMED_ARG:
639 named_arg = static_cast<const NamedArg*>(args.args_[i].pointer);
640 map_.insert(Pair(named_arg->name, *named_arg));
641 break;
642 default:
643 /*nothing*/;
644 }
645 }
646}

668template <typename Char>
669void fmt::internal::PrintfFormatter<Char>::parse_flags(
670 FormatSpec &spec, const Char *&s) {
671 for (;;) {
672 switch (*s++) {
673 case '-':
674 spec.align_ = ALIGN_LEFT;
675 break;
676 case '+':
677 spec.flags_ |= SIGN_FLAG | PLUS_FLAG;
678 break;
679 case '0':
680 spec.fill_ = '0';
681 break;
682 case ' ':
683 spec.flags_ |= SIGN_FLAG;
684 break;
685 case '#':
686 spec.flags_ |= HASH_FLAG;
687 break;
688 default:
689 --s;
690 return;
691 }
692 }
693}

707template <typename Char>
708unsigned fmt::internal::PrintfFormatter<Char>::parse_header(
709 const Char *&s, FormatSpec &spec) {
710 unsigned arg_index = UINT_MAX;
711 Char c = *s;
712 if (c >= '0' && c <= '9') {
713 // Parse an argument index (if followed by '$') or a width possibly
714 // preceded with '0' flag(s).
715 unsigned value = parse_nonnegative_int(s);
716 if (*s == '$') { // value is an argument index
717 ++s;
718 arg_index = value;
719 } else {
720 if (c == '0')
721 spec.fill_ = '0';
722 if (value != 0) {
723 // Nonzero value means that we parsed width and don't need to
724 // parse it or flags again, so return now.
725 spec.width_ = value;
726 return arg_index;
727 }
728 }
729 }
730 parse_flags(spec, s);
731 // Parse width.
732 if (*s >= '0' && *s <= '9') {
733 spec.width_ = parse_nonnegative_int(s);
734 } else if (*s == '*') {
735 ++s;
736 spec.width_ = WidthHandler(spec).visit(get_arg(s));
737 }
738 return arg_index;
739}
740
741template <typename Char>
742void fmt::internal::PrintfFormatter<Char>::format(
743 BasicWriter<Char> &writer, BasicCStringRef<Char> format_str) {
744 const Char *start = format_str.c_str();
745 const Char *s = start;
746 while (*s) {
747 Char c = *s++;
748 if (c != '%') continue;
749 if (*s == c) {
750 write(writer, start, s);
751 start = ++s;
752 continue;
753 }
754 write(writer, start, s - 1);
755
756 FormatSpec spec;
757 spec.align_ = ALIGN_RIGHT;
758
759 // Parse argument index, flags and width.
760 unsigned arg_index = parse_header(s, spec);
761
762 // Parse precision.
763 if (*s == '.') {
764 ++s;
765 if ('0' <= *s && *s <= '9') {
766 spec.precision_ = parse_nonnegative_int(s);
767 } else if (*s == '*') {
768 ++s;
769 spec.precision_ = PrecisionHandler().visit(get_arg(s));
770 }
771 }
772
773 Arg arg = get_arg(s, arg_index);
774 if (spec.flag(HASH_FLAG) && IsZeroInt().visit(arg))
775 spec.flags_ &= ~HASH_FLAG;
776 if (spec.fill_ == '0') {
777 if (arg.type <= Arg::LAST_NUMERIC_TYPE)
778 spec.align_ = ALIGN_NUMERIC;
779 else
780 spec.fill_ = ' '; // Ignore '0' flag for non-numeric types.
781 }
782
783 // Parse length and convert the argument to the required type.
784 switch (*s++) {
785 case 'h':
786 if (*s == 'h')
787 ArgConverter<signed char>(arg, *++s).visit(arg);
788 else
789 ArgConverter<short>(arg, *s).visit(arg);
790 break;
791 case 'l':
792 if (*s == 'l')
793 ArgConverter<fmt::LongLong>(arg, *++s).visit(arg);
794 else
795 ArgConverter<long>(arg, *s).visit(arg);
796 break;
797 case 'j':
798 ArgConverter<intmax_t>(arg, *s).visit(arg);
799 break;
800 case 'z':
801 ArgConverter<std::size_t>(arg, *s).visit(arg);
802 break;
803 case 't':
804 ArgConverter<std::ptrdiff_t>(arg, *s).visit(arg);
805 break;
806 case 'L':
807 // printf produces garbage when 'L' is omitted for long double, no
808 // need to do the same.
809 break;
810 default:
811 --s;
812 ArgConverter<int>(arg, *s).visit(arg);
813 }
814
815 // Parse type.
816 if (!*s)
817 FMT_THROW(FormatError("invalid format string"));
818 spec.type_ = static_cast<char>(*s++);
819 if (arg.type <= Arg::LAST_INTEGER_TYPE) {
820 // Normalize type.
821 switch (spec.type_) {
822 case 'i': case 'u':
823 spec.type_ = 'd';
824 break;
825 case 'c':
826 // TODO: handle wchar_t
827 CharConverter(arg).visit(arg);
828 break;
829 }
830 }
831
832 start = s;
833
834 // Format argument.
835 internal::PrintfArgFormatter<Char>(writer, spec).visit(arg);
836 }
837 write(writer, start, s);
838}

87 _BitScanReverse64(&r, x);
88# else
89 // Scan the high 32 bits.

269// Standard permits specialization of std::numeric_limits. This specialization
270// is used to resolve ambiguity between isinf and std::isinf in glibc:
271// https://gcc.gnu.org/bugzilla/show_bug.cgi?id=48891
272// and the same for isnan and signbit.
273template <>
274class numeric_limits<fmt::internal::DummyInt> :
275 public std::numeric_limits<int> {
276 public:
277 // Portable version of isinf.
278 template <typename T>
279 static bool isinfinity(T x) {
280 using namespace fmt::internal;
281 // The resolution "priority" is:
282 // isinf macro > std::isinf > ::isinf > fmt::internal::isinf
283 if (check(sizeof(isinf(x)) == sizeof(bool) ||
284 sizeof(isinf(x)) == sizeof(int))) {
285 return isinf(x) != 0;
286 }
287 return !_finite(static_cast<double>(x));
288 }
289
290 // Portable version of isnan.
291 template <typename T>
292 static bool isnotanumber(T x) {
293 using namespace fmt::internal;
294 if (check(sizeof(isnan(x)) == sizeof(bool) ||
295 sizeof(isnan(x)) == sizeof(int))) {
296 return isnan(x) != 0;
297 }
298 return _isnan(static_cast<double>(x)) != 0;
299 }
300
301 // Portable version of signbit.
302 static bool isnegative(double x) {
303 using namespace fmt::internal;
304 if (check(sizeof(signbit(x)) == sizeof(int)))
305 return signbit(x) != 0;
306 if (x < 0) return true;
307 if (!isnotanumber(x)) return false;
308 int dec = 0, sign = 0;
309 char buffer[2]; // The buffer size must be >= 2 or _ecvt_s will fail.
310 _ecvt_s(buffer, sizeof(buffer), x, 0, &dec, &sign);
311 return sign != 0;
312 }
313};
314} // namespace std

339/**
340 \rst
341 A string reference. It can be constructed from a C string or ``std::string``.
342
343 You can use one of the following typedefs for common character types:
344
345 +------------+-------------------------+
346 | Type | Definition |
347 +============+=========================+
348 | StringRef | BasicStringRef<char> |
349 +------------+-------------------------+
350 | WStringRef | BasicStringRef<wchar_t> |
351 +------------+-------------------------+
352
353 This class is most useful as a parameter type to allow passing
354 different types of strings to a function, for example::
355
356 template <typename... Args>
357 std::string format(StringRef format_str, const Args & ... args);
358
359 format("{}", 42);
360 format(std::string("{}"), 42);
361 \endrst
362 */
363template <typename Char>
364class BasicStringRef {
365 private:
366 const Char *data_;
367 std::size_t size_;
368
369 public:
370 /** Constructs a string reference object from a C string and a size. */
371 BasicStringRef(const Char *s, std::size_t size) : data_(s), size_(size) {}
372
373 /**
374 \rst
375 Constructs a string reference object from a C string computing
376 the size with ``std::char_traits<Char>::length``.
377 \endrst
378 */
379 BasicStringRef(const Char *s)
380 : data_(s), size_(std::char_traits<Char>::length(s)) {}
381
382 /**
383 \rst
384 Constructs a string reference from an ``std::string`` object.
385 \endrst
386 */
387 BasicStringRef(const std::basic_string<Char> &s)
388 : data_(s.c_str()), size_(s.size()) {}
389
390 /**
391 \rst
392 Converts a string reference to an ``std::string`` object.
393 \endrst
394 */
395 std::basic_string<Char> to_string() const {
396 return std::basic_string<Char>(data_, size_);
397 }
398
399 /** Returns the pointer to a C string. */
400 const Char *data() const { return data_; }
401
402 /** Returns the string size. */
403 std::size_t size() const { return size_; }
404
405 // Lexicographically compare this string reference to other.
406 int compare(BasicStringRef other) const {
407 std::size_t size = size_ < other.size_ ? size_ : other.size_;
408 int result = std::char_traits<Char>::compare(data_, other.data_, size);
409 if (result == 0)
410 result = size_ == other.size_ ? 0 : (size_ < other.size_ ? -1 : 1);
411 return result;
412 }
413
414 friend bool operator==(BasicStringRef lhs, BasicStringRef rhs) {
415 return lhs.compare(rhs) == 0;
416 }
417 friend bool operator!=(BasicStringRef lhs, BasicStringRef rhs) {
418 return lhs.compare(rhs) != 0;
419 }
420 friend bool operator<(BasicStringRef lhs, BasicStringRef rhs) {
421 return lhs.compare(rhs) < 0;
422 }
423 friend bool operator<=(BasicStringRef lhs, BasicStringRef rhs) {
424 return lhs.compare(rhs) <= 0;
425 }
426 friend bool operator>(BasicStringRef lhs, BasicStringRef rhs) {
427 return lhs.compare(rhs) > 0;
428 }
429 friend bool operator>=(BasicStringRef lhs, BasicStringRef rhs) {
430 return lhs.compare(rhs) >= 0;
431 }
432};

511/**
512 \rst
513 A buffer supporting a subset of ``std::vector``'s operations.
514 \endrst
515 */
516template <typename T>
517class Buffer {
518 private:
519 FMT_DISALLOW_COPY_AND_ASSIGN(Buffer);
520
521 protected:
522 T *ptr_;
523 std::size_t size_;
524 std::size_t capacity_;
525
526 Buffer(T *ptr = 0, std::size_t capacity = 0)
527 : ptr_(ptr), size_(0), capacity_(capacity) {}
528
529 /**
530 \rst
531 Increases the buffer capacity to hold at least *size* elements updating
532 ``ptr_`` and ``capacity_``.
533 \endrst
534 */
535 virtual void grow(std::size_t size) = 0;
536
537 public:
538 virtual ~Buffer() {}
539
540 /** Returns the size of this buffer. */
541 std::size_t size() const { return size_; }
542
543 /** Returns the capacity of this buffer. */
544 std::size_t capacity() const { return capacity_; }
545
546 /**
547 Resizes the buffer. If T is a POD type new elements may not be initialized.
548 */
549 void resize(std::size_t new_size) {
550 if (new_size > capacity_)
551 grow(new_size);
552 size_ = new_size;
553 }
554
555 /**
556 \rst
557 Reserves space to store at least *capacity* elements.
558 \endrst
559 */
560 void reserve(std::size_t capacity) {
561 if (capacity > capacity_)
562 grow(capacity);
563 }
564
565 void clear() FMT_NOEXCEPT { size_ = 0; }
566
567 void push_back(const T &value) {
568 if (size_ == capacity_)
569 grow(size_ + 1);
570 ptr_[size_++] = value;
571 }
572
573 /** Appends data to the end of the buffer. */
574 template <typename U>
575 void append(const U *begin, const U *end);
576
577 T &operator[](std::size_t index) { return ptr_[index]; }
578 const T &operator[](std::size_t index) const { return ptr_[index]; }
579};
580
581template <typename T>
582template <typename U>
583void Buffer<T>::append(const U *begin, const U *end) {
584 assert(begin <= end);
585 std::size_t new_size = size_ + (end - begin);
586 if (new_size > capacity_)
587 grow(new_size);
588 std::uninitialized_copy(begin, end,
589 internal::make_ptr(ptr_, capacity_) + size_);
590 size_ = new_size;
591}

595// A memory buffer for POD types with the first SIZE elements stored in
596// the object itself.
597template <typename T, std::size_t SIZE, typename Allocator = std::allocator<T> >
598class MemoryBuffer : private Allocator, public Buffer<T> {
599 private:
600 T data_[SIZE];
601
602 // Deallocate memory allocated by the buffer.
603 void deallocate() {
604 if (this->ptr_ != data_) Allocator::deallocate(this->ptr_, this->capacity_);
605 }
606
607 protected:
608 void grow(std::size_t size);
609
610 public:
611 explicit MemoryBuffer(const Allocator &alloc = Allocator())
612 : Allocator(alloc), Buffer<T>(data_, SIZE) {}
613 ~MemoryBuffer() { deallocate(); }
614
615#if FMT_USE_RVALUE_REFERENCES
616 private:
617 // Move data from other to this buffer.
618 void move(MemoryBuffer &other) {
619 Allocator &this_alloc = *this, &other_alloc = other;
620 this_alloc = std::move(other_alloc);
621 this->size_ = other.size_;
622 this->capacity_ = other.capacity_;
623 if (other.ptr_ == other.data_) {
624 this->ptr_ = data_;
625 std::uninitialized_copy(other.data_, other.data_ + this->size_,
626 make_ptr(data_, this->capacity_));
627 } else {
628 this->ptr_ = other.ptr_;
629 // Set pointer to the inline array so that delete is not called
630 // when deallocating.
631 other.ptr_ = other.data_;
632 }
633 }
634
635 public:
636 MemoryBuffer(MemoryBuffer &&other) {
637 move(other);
638 }
639
640 MemoryBuffer &operator=(MemoryBuffer &&other) {
641 assert(this != &other);
642 deallocate();
643 move(other);
644 return *this;
645 }
646#endif
647
648 // Returns a copy of the allocator associated with this buffer.
649 Allocator get_allocator() const { return *this; }
650};
651
652template <typename T, std::size_t SIZE, typename Allocator>
653void MemoryBuffer<T, SIZE, Allocator>::grow(std::size_t size) {
654 std::size_t new_capacity = this->capacity_ + this->capacity_ / 2;
655 if (size > new_capacity)
656 new_capacity = size;
657 T *new_ptr = this->allocate(new_capacity);
658 // The following code doesn't throw, so the raw pointer above doesn't leak.
659 std::uninitialized_copy(this->ptr_, this->ptr_ + this->size_,
660 make_ptr(new_ptr, new_capacity));
661 std::size_t old_capacity = this->capacity_;
662 T *old_ptr = this->ptr_;
663 this->capacity_ = new_capacity;
664 this->ptr_ = new_ptr;
665 // deallocate may throw (at least in principle), but it doesn't matter since
666 // the buffer already uses the new storage and will deallocate it in case
667 // of exception.
668 if (old_ptr != data_)
669 Allocator::deallocate(old_ptr, old_capacity);
670}

794// Returns the number of decimal digits in n. Leading zeros are not counted
795// except for n == 0 in which case count_digits returns 1.
796inline unsigned count_digits(uint64_t n) {
797 // Based on http://graphics.stanford.edu/~seander/bithacks.html#IntegerLog10
798 // and the benchmark https://github.com/localvoid/cxx-benchmark-count-digits.
799 unsigned t = (64 - FMT_BUILTIN_CLZLL(n | 1)) * 1233 >> 12;
800 return t - (n < Data::POWERS_OF_10_64[t]) + 1;
801}
802#else
803// Fallback version of count_digits used when __builtin_clz is not available.
804inline unsigned count_digits(uint64_t n) {
805 unsigned count = 1;
806 for (;;) {
807 // Integer division is slow so do it for a group of four digits instead
808 // of for every digit. The idea comes from the talk by Alexandrescu
809 // "Three Optimization Tips for C++". See speed-test for a comparison.
810 if (n < 10) return count;
811 if (n < 100) return count + 1;
812 if (n < 1000) return count + 2;
813 if (n < 10000) return count + 3;
814 n /= 10000u;
815 count += 4;
816 }
817}
818#endif

828// Formats a decimal unsigned integer value writing into buffer.
829template <typename UInt, typename Char>
830inline void format_decimal(Char *buffer, UInt value, unsigned num_digits) {
831 buffer += num_digits;
832 while (value >= 100) {
833 // Integer division is slow so do it for a group of two digits instead
834 // of for every digit. The idea comes from the talk by Alexandrescu
835 // "Three Optimization Tips for C++". See speed-test for a comparison.
836 unsigned index = static_cast<unsigned>((value % 100) * 2);
837 value /= 100;
838 *--buffer = Data::DIGITS[index + 1];
839 *--buffer = Data::DIGITS[index];
840 }
841 if (value < 10) {
842 *--buffer = static_cast<char>('0' + value);
843 return;
844 }
845 unsigned index = static_cast<unsigned>(value * 2);
846 *--buffer = Data::DIGITS[index + 1];
847 *--buffer = Data::DIGITS[index];
848}

1217// An argument visitor.
1218// To use ArgVisitor define a subclass that implements some or all of the
1219// visit methods with the same signatures as the methods in ArgVisitor,
1220// for example, visit_int(int).
1221// Specify the subclass name as the Impl template parameter. Then calling
1222// ArgVisitor::visit for some argument will dispatch to a visit method
1223// specific to the argument type. For example, if the argument type is
1224// double then visit_double(double) method of a subclass will be called.
1225// If the subclass doesn't contain a method with this signature, then
1226// a corresponding method of ArgVisitor will be called.
1227//
1228// Example:
1229// class MyArgVisitor : public ArgVisitor<MyArgVisitor, void> {
1230// public:
1231// void visit_int(int value) { print("{}", value); }
1232// void visit_double(double value) { print("{}", value ); }
1233// };
1234//
1235// ArgVisitor uses the curiously recurring template pattern:
1236// http://en.wikipedia.org/wiki/Curiously_recurring_template_pattern
1237template <typename Impl, typename Result>
1238class ArgVisitor {
1239 public:
1240 void report_unhandled_arg() {}
1241
1242 Result visit_unhandled_arg() {
1243 FMT_DISPATCH(report_unhandled_arg());
1244 return Result();
1245 }
1246
1247 Result visit_int(int value) {
1248 return FMT_DISPATCH(visit_any_int(value));
1249 }
1250 Result visit_long_long(LongLong value) {
1251 return FMT_DISPATCH(visit_any_int(value));
1252 }
1253 Result visit_uint(unsigned value) {
1254 return FMT_DISPATCH(visit_any_int(value));
1255 }
1256 Result visit_ulong_long(ULongLong value) {
1257 return FMT_DISPATCH(visit_any_int(value));
1258 }
1259 Result visit_bool(bool value) {
1260 return FMT_DISPATCH(visit_any_int(value));
1261 }
1262 Result visit_char(int value) {
1263 return FMT_DISPATCH(visit_any_int(value));
1264 }
1265 template <typename T>
1266 Result visit_any_int(T) {
1267 return FMT_DISPATCH(visit_unhandled_arg());
1268 }
1269
1270 Result visit_double(double value) {
1271 return FMT_DISPATCH(visit_any_double(value));
1272 }
1273 Result visit_long_double(long double value) {
1274 return FMT_DISPATCH(visit_any_double(value));
1275 }
1276 template <typename T>
1277 Result visit_any_double(T) {
1278 return FMT_DISPATCH(visit_unhandled_arg());
1279 }
1280
1281 Result visit_cstring(const char *) {
1282 return FMT_DISPATCH(visit_unhandled_arg());
1283 }
1284 Result visit_string(Arg::StringValue<char>) {
1285 return FMT_DISPATCH(visit_unhandled_arg());
1286 }
1287 Result visit_wstring(Arg::StringValue<wchar_t>) {
1288 return FMT_DISPATCH(visit_unhandled_arg());
1289 }
1290 Result visit_pointer(const void *) {
1291 return FMT_DISPATCH(visit_unhandled_arg());
1292 }
1293 Result visit_custom(Arg::CustomValue) {
1294 return FMT_DISPATCH(visit_unhandled_arg());
1295 }
1296
1297 Result visit(const Arg &arg) {
1298 switch (arg.type) {
1299 default:
1300 FMT_ASSERT(false, "invalid argument type");
1301 return Result();
1302 case Arg::INT:
1303 return FMT_DISPATCH(visit_int(arg.int_value));
1304 case Arg::UINT:
1305 return FMT_DISPATCH(visit_uint(arg.uint_value));
1306 case Arg::LONG_LONG:
1307 return FMT_DISPATCH(visit_long_long(arg.long_long_value));
1308 case Arg::ULONG_LONG:
1309 return FMT_DISPATCH(visit_ulong_long(arg.ulong_long_value));
1310 case Arg::BOOL:
1311 return FMT_DISPATCH(visit_bool(arg.int_value != 0));
1312 case Arg::CHAR:
1313 return FMT_DISPATCH(visit_char(arg.int_value));
1314 case Arg::DOUBLE:
1315 return FMT_DISPATCH(visit_double(arg.double_value));
1316 case Arg::LONG_DOUBLE:
1317 return FMT_DISPATCH(visit_long_double(arg.long_double_value));
1318 case Arg::CSTRING:
1319 return FMT_DISPATCH(visit_cstring(arg.string.value));
1320 case Arg::STRING:
1321 return FMT_DISPATCH(visit_string(arg.string));
1322 case Arg::WSTRING:
1323 return FMT_DISPATCH(visit_wstring(arg.wstring));
1324 case Arg::POINTER:
1325 return FMT_DISPATCH(visit_pointer(arg.pointer));
1326 case Arg::CUSTOM:
1327 return FMT_DISPATCH(visit_custom(arg.custom));
1328 }

1344/** An argument list. */
1345class ArgList {
1346 private:
1347 // To reduce compiled code size per formatting function call, types of first
1348 // MAX_PACKED_ARGS arguments are passed in the types_ field.
1349 uint64_t types_;
1350 union {
1351 // If the number of arguments is less than MAX_PACKED_ARGS, the argument
1352 // values are stored in values_, otherwise they are stored in args_.
1353 // This is done to reduce compiled code size as storing larger objects
1354 // may require more code (at least on x86-64) even if the same amount of
1355 // data is actually copied to stack. It saves ~10% on the bloat test.
1356 const internal::Value *values_;
1357 const internal::Arg *args_;
1358 };
1359
1360 internal::Arg::Type type(unsigned index) const {
1361 unsigned shift = index * 4;
1362 uint64_t mask = 0xf;
1363 return static_cast<internal::Arg::Type>(
1364 (types_ & (mask << shift)) >> shift);
1365 }
1366
1367 template <typename Char>
1368 friend class internal::ArgMap;
1369
1370 public:
1371 // Maximum number of arguments with packed types.
1372 enum { MAX_PACKED_ARGS = 16 };
1373
1374 ArgList() : types_(0) {}
1375
1376 ArgList(ULongLong types, const internal::Value *values)
1377 : types_(types), values_(values) {}
1378 ArgList(ULongLong types, const internal::Arg *args)
1379 : types_(types), args_(args) {}
1380
1381 /** Returns the argument at specified index. */
1382 internal::Arg operator[](unsigned index) const {
1383 using internal::Arg;
1384 Arg arg;
1385 bool use_values = type(MAX_PACKED_ARGS - 1) == Arg::NONE;
1386 if (index < MAX_PACKED_ARGS) {
1387 Arg::Type arg_type = type(index);
1388 internal::Value &val = arg;
1389 if (arg_type != Arg::NONE)
1390 val = use_values ? values_[index] : args_[index];
1391 arg.type = arg_type;
1392 return arg;
1393 }
1394 if (use_values) {
1395 // The index is greater than the number of arguments that can be stored
1396 // in values, so return a "none" argument.
1397 arg.type = Arg::NONE;
1398 return arg;
1399 }
1400 for (unsigned i = MAX_PACKED_ARGS; i <= index; ++i) {
1401 if (args_[i].type == Arg::NONE)
1402 return args_[i];
1403 }
1404 return args_[index];
1405 }

1649class ArgFormatterBase : public ArgVisitor<Impl, void> {
1650 private:
1651 BasicWriter<Char> &writer_;
1652 FormatSpec &spec_;
1653
1654 FMT_DISALLOW_COPY_AND_ASSIGN(ArgFormatterBase);
1655
1656 void write_pointer(const void *p) {
1657 spec_.flags_ = HASH_FLAG;
1658 spec_.type_ = 'x';
1659 writer_.write_int(reinterpret_cast<uintptr_t>(p), spec_);
1660 }
1661
1662 protected:
1663 BasicWriter<Char> &writer() { return writer_; }
1664 FormatSpec &spec() { return spec_; }
1665
1666 void write(bool value) {
1667 const char *str_value = value ? "true" : "false";
1668 Arg::StringValue<char> str = { str_value, std::strlen(str_value) };
1669 writer_.write_str(str, spec_);
1670 }
1671
1672 void write(const char *value) {
1673 Arg::StringValue<char> str = {value, value != 0 ? std::strlen(value) : 0};
1674 writer_.write_str(str, spec_);
1675 }
1676
1677 public:
1678 ArgFormatterBase(BasicWriter<Char> &w, FormatSpec &s)
1679 : writer_(w), spec_(s) {}

1682 void visit_any_int(T value) { writer_.write_int(value, spec_); }
1683
1684 template <typename T>
1685 void visit_any_double(T value) { writer_.write_double(value, spec_); }
1686
1687 void visit_bool(bool value) {
1688 if (spec_.type_)
1689 return visit_any_int(value);
1690 write(value);
1691 }
1692
1693 void visit_char(int value) {
1694 if (spec_.type_ && spec_.type_ != 'c') {
1695 spec_.flags_ |= CHAR_FLAG;
1696 writer_.write_int(value, spec_);
1697 return;
1698 }
1699 if (spec_.align_ == ALIGN_NUMERIC || spec_.flags_ != 0)
1700 FMT_THROW(FormatError("invalid format specifier for char"));
1701 typedef typename BasicWriter<Char>::CharPtr CharPtr;
1702 Char fill = internal::CharTraits<Char>::cast(spec_.fill());
1703 CharPtr out = CharPtr();
1704 const unsigned CHAR_WIDTH = 1;
1705 if (spec_.width_ > CHAR_WIDTH) {
1706 out = writer_.grow_buffer(spec_.width_);
1707 if (spec_.align_ == ALIGN_RIGHT) {
1708 std::uninitialized_fill_n(out, spec_.width_ - CHAR_WIDTH, fill);
1709 out += spec_.width_ - CHAR_WIDTH;
1710 } else if (spec_.align_ == ALIGN_CENTER) {
1711 out = writer_.fill_padding(out, spec_.width_,
1712 internal::check(CHAR_WIDTH), fill);
1713 } else {
1714 std::uninitialized_fill_n(out + CHAR_WIDTH,
1715 spec_.width_ - CHAR_WIDTH, fill);
1716 }
1717 } else {
1718 out = writer_.grow_buffer(CHAR_WIDTH);
1719 }
1720 *out = internal::CharTraits<Char>::cast(value);
1721 }
1722
1723 void visit_cstring(const char *value) {
1724 if (spec_.type_ == 'p')
1725 return write_pointer(value);
1726 write(value);
1727 }
1728
1729 void visit_string(Arg::StringValue<char> value) {
1730 writer_.write_str(value, spec_);
1731 }
1732
1733 using ArgVisitor<Impl, void>::visit_wstring;
1734
1735 void visit_wstring(Arg::StringValue<Char> value) {
1736 writer_.write_str(value, spec_);
1737 }
1738
1739 void visit_pointer(const void *value) {
1740 if (spec_.type_ && spec_.type_ != 'p')
1741 report_unknown_type(spec_.type_, "pointer");
1742 write_pointer(value);
1743 }

2499template <typename Char>
2500template <typename Spec>
2501typename BasicWriter<Char>::CharPtr
2502 BasicWriter<Char>::prepare_int_buffer(
2503 unsigned num_digits, const Spec &spec,
2504 const char *prefix, unsigned prefix_size) {
2505 unsigned width = spec.width();
2506 Alignment align = spec.align();
2507 Char fill = internal::CharTraits<Char>::cast(spec.fill());
2508 if (spec.precision() > static_cast<int>(num_digits)) {
2509 // Octal prefix '0' is counted as a digit, so ignore it if precision
2510 // is specified.
2511 if (prefix_size > 0 && prefix[prefix_size - 1] == '0')
2512 --prefix_size;
2513 unsigned number_size = prefix_size + spec.precision();
2514 AlignSpec subspec(number_size, '0', ALIGN_NUMERIC);
2515 if (number_size >= width)
2516 return prepare_int_buffer(num_digits, subspec, prefix, prefix_size);
2517 buffer_.reserve(width);
2518 unsigned fill_size = width - number_size;
2519 if (align != ALIGN_LEFT) {
2520 CharPtr p = grow_buffer(fill_size);
2521 std::uninitialized_fill(p, p + fill_size, fill);
2522 }
2523 CharPtr result = prepare_int_buffer(
2524 num_digits, subspec, prefix, prefix_size);
2525 if (align == ALIGN_LEFT) {
2526 CharPtr p = grow_buffer(fill_size);
2527 std::uninitialized_fill(p, p + fill_size, fill);
2528 }
2529 return result;
2530 }
2531 unsigned size = prefix_size + num_digits;
2532 if (width <= size) {
2533 CharPtr p = grow_buffer(size);
2534 std::uninitialized_copy(prefix, prefix + prefix_size, p);
2535 return p + size - 1;
2536 }
2537 CharPtr p = grow_buffer(width);
2538 CharPtr end = p + width;
2539 if (align == ALIGN_LEFT) {
2540 std::uninitialized_copy(prefix, prefix + prefix_size, p);
2541 p += size;
2542 std::uninitialized_fill(p, end, fill);
2543 } else if (align == ALIGN_CENTER) {
2544 p = fill_padding(p, width, size, fill);
2545 std::uninitialized_copy(prefix, prefix + prefix_size, p);
2546 p += size;
2547 } else {
2548 if (align == ALIGN_NUMERIC) {
2549 if (prefix_size != 0) {
2550 p = std::uninitialized_copy(prefix, prefix + prefix_size, p);
2551 size -= prefix_size;
2552 }
2553 } else {
2554 std::uninitialized_copy(prefix, prefix + prefix_size, end - size);
2555 }
2556 std::uninitialized_fill(p, end - size, fill);
2557 p = end;
2558 }
2559 return p - 1;
2560}

2562template <typename Char>
2563template <typename T, typename Spec>
2564void BasicWriter<Char>::write_int(T value, Spec spec) {
2565 unsigned prefix_size = 0;
2566 typedef typename internal::IntTraits<T>::MainType UnsignedType;
2567 UnsignedType abs_value = value;
2568 char prefix[4] = "";
2569 if (internal::is_negative(value)) {
2570 prefix[0] = '-';
2571 ++prefix_size;
2572 abs_value = 0 - abs_value;
2573 } else if (spec.flag(SIGN_FLAG)) {
2574 prefix[0] = spec.flag(PLUS_FLAG) ? '+' : ' ';
2575 ++prefix_size;
2576 }
2577 switch (spec.type()) {
2578 case 0: case 'd': {
2579 unsigned num_digits = internal::count_digits(abs_value);
2580 CharPtr p = prepare_int_buffer(
2581 num_digits, spec, prefix, prefix_size) + 1 - num_digits;
2582 internal::format_decimal(get(p), abs_value, num_digits);
2583 break;
2584 }
2585 case 'x': case 'X': {
2586 UnsignedType n = abs_value;
2587 if (spec.flag(HASH_FLAG)) {
2588 prefix[prefix_size++] = '0';
2589 prefix[prefix_size++] = spec.type();
2590 }
2591 unsigned num_digits = 0;
2592 do {
2593 ++num_digits;
2594 } while ((n >>= 4) != 0);
2595 Char *p = get(prepare_int_buffer(
2596 num_digits, spec, prefix, prefix_size));
2597 n = abs_value;
2598 const char *digits = spec.type() == 'x' ?
2599 "0123456789abcdef" : "0123456789ABCDEF";
2600 do {
2601 *p-- = digits[n & 0xf];
2602 } while ((n >>= 4) != 0);
2603 break;
2604 }
2605 case 'b': case 'B': {
2606 UnsignedType n = abs_value;
2607 if (spec.flag(HASH_FLAG)) {
2608 prefix[prefix_size++] = '0';
2609 prefix[prefix_size++] = spec.type();
2610 }
2611 unsigned num_digits = 0;
2612 do {
2613 ++num_digits;
2614 } while ((n >>= 1) != 0);
2615 Char *p = get(prepare_int_buffer(num_digits, spec, prefix, prefix_size));
2616 n = abs_value;
2617 do {
2618 *p-- = static_cast<Char>('0' + (n & 1));
2619 } while ((n >>= 1) != 0);
2620 break;
2621 }
2622 case 'o': {
2623 UnsignedType n = abs_value;
2624 if (spec.flag(HASH_FLAG))
2625 prefix[prefix_size++] = '0';
2626 unsigned num_digits = 0;
2627 do {
2628 ++num_digits;
2629 } while ((n >>= 3) != 0);
2630 Char *p = get(prepare_int_buffer(num_digits, spec, prefix, prefix_size));
2631 n = abs_value;
2632 do {
2633 *p-- = static_cast<Char>('0' + (n & 7));
2634 } while ((n >>= 3) != 0);
2635 break;
2636 }
2637 default:
2638 internal::report_unknown_type(
2639 spec.type(), spec.flag(CHAR_FLAG) ? "char" : "integer");

2648 // Check type.
2649 char type = spec.type();
2650 bool upper = false;
2651 switch (type) {
2652 case 0:

2654 break;
2655 case 'e': case 'f': case 'g': case 'a':
2656 break;

2675 sign = '-';

2678 sign = spec.flag(PLUS_FLAG) ? '+' : ' ';

2683 // across platforms.
2684 std::size_t nan_size = 4;
2685 const char *nan = upper ? " NAN" : " nan";
2686 if (!sign) {
2687 --nan_size;
2688 ++nan;
2689 }
2690 CharPtr out = write_str(nan, nan_size, spec);
2691 if (sign)
2692 *out = sign;
2693 return;
2694 }
2695
2696 if (internal::FPUtil::isinfinity(value)) {
2697 // Format infinity ourselves because sprintf's output is not consistent
2698 // across platforms.
2699 std::size_t inf_size = 4;
2700 const char *inf = upper ? " INF" : " inf";
2701 if (!sign) {
2702 --inf_size;
2703 ++inf;
2704 }
2705 CharPtr out = write_str(inf, inf_size, spec);
2706 if (sign)
2707 *out = sign;
2708 return;
2709 }
2710
2711 std::size_t offset = buffer_.size();
2712 unsigned width = spec.width();
2713 if (sign) {
2714 buffer_.reserve(buffer_.size() + (width > 1u ? width : 1u));
2715 if (width > 0)
2716 --width;
2717 ++offset;
2718 }
2719
2720 // Build format string.
2721 enum { MAX_FORMAT_SIZE = 10}; // longest format: %#-*.*Lg
2722 Char format[MAX_FORMAT_SIZE];
2723 Char *format_ptr = format;
2724 *format_ptr++ = '%';
2725 unsigned width_for_sprintf = width;
2726 if (spec.flag(HASH_FLAG))
2727 *format_ptr++ = '#';
2728 if (spec.align() == ALIGN_CENTER) {
2729 width_for_sprintf = 0;
2730 } else {
2731 if (spec.align() == ALIGN_LEFT)
2732 *format_ptr++ = '-';
2733 if (width != 0)
2734 *format_ptr++ = '*';
2735 }
2736 if (spec.precision() >= 0) {
2737 *format_ptr++ = '.';
2738 *format_ptr++ = '*';
2739 }
2740
2741 append_float_length(format_ptr, value);
2742 *format_ptr++ = type;
2743 *format_ptr = '\0';
2744
2745 // Format using snprintf.
2746 Char fill = internal::CharTraits<Char>::cast(spec.fill());
2747 for (;;) {
2748 std::size_t buffer_size = buffer_.capacity() - offset;
2749#ifdef _MSC_VER
2750 // MSVC's vsnprintf_s doesn't work with zero size, so reserve
2751 // space for at least one extra character to make the size non-zero.
2752 // Note that the buffer's capacity will increase by more than 1.
2753 if (buffer_size == 0) {
2754 buffer_.reserve(offset + 1);
2755 buffer_size = buffer_.capacity() - offset;
2756 }
2757#endif
2758 Char *start = &buffer_[offset];
2759 int n = internal::CharTraits<Char>::format_float(
2760 start, buffer_size, format, width_for_sprintf, spec.precision(), value);
2761 if (n >= 0 && offset + n < buffer_.capacity()) {
2762 if (sign) {
2763 if ((spec.align() != ALIGN_RIGHT && spec.align() != ALIGN_DEFAULT) ||
2764 *start != ' ') {
2765 *(start - 1) = sign;
2766 sign = 0;
2767 } else {
2768 *(start - 1) = fill;
2769 }
2770 ++n;
2771 }
2772 if (spec.align() == ALIGN_CENTER &&
2773 spec.width() > static_cast<unsigned>(n)) {
2774 width = spec.width();
2775 CharPtr p = grow_buffer(width);
2776 std::memmove(get(p) + (width - n) / 2, get(p), n * sizeof(Char));
2777 fill_padding(p, spec.width(), n, fill);
2778 return;
2779 }
2780 if (spec.fill() != ' ' || sign) {
2781 while (*start == ' ')
2782 *start++ = fill;
2783 if (sign)
2784 *(start - 1) = sign;
2785 }
2786 grow_buffer(n);
2787 return;
2788 }
2789 // If n is negative we ask to increase the capacity by at least 1,
2790 // but as std::vector, the buffer grows exponentially.
2791 buffer_.reserve(n >= 0 ? offset + n + 1 : buffer_.capacity() + 1);
2792 }
2793}

3112 *--buffer_end = internal::Data::DIGITS[index + 1];

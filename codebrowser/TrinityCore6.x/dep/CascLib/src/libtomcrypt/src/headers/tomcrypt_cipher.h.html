261#ifdef LTC_CTR_MODE
262/** A block cipher CTR structure */
263typedef struct {
264 /** The index of the cipher chosen */
265 int cipher,
266 /** The block size of the given cipher */
267 blocklen,
268 /** The padding offset */
269 padlen,
270 /** The mode (endianess) of the CTR, 0==little, 1==big */
271 mode,
272 /** counter width */
273 ctrlen;
274
275 /** The counter */
276 unsigned char ctr[MAXBLOCKSIZE],
277 /** The pad used to encrypt/decrypt */
278 pad[MAXBLOCKSIZE];
279 /** The scheduled key */
280 symmetric_key key;
281} symmetric_CTR;

285#ifdef LTC_LRW_MODE
286/** A LRW structure */
287typedef struct {
288 /** The index of the cipher chosen (must be a 128-bit block cipher) */
289 int cipher;
290
291 /** The current IV */
292 unsigned char IV[16],
293
294 /** the tweak key */
295 tweak[16],
296
297 /** The current pad, it's the product of the first 15 bytes against the tweak key */
298 pad[16];
299
300 /** The scheduled symmetric key */
301 symmetric_key key;
302
303#ifdef LRW_TABLES
304 /** The pre-computed multiplication table */
305 unsigned char PC[16][256][16];
306#endif
307} symmetric_LRW;
308#endif

330/** cipher descriptor table, last entry has "name == NULL" to mark the end of table */
331extern struct ltc_cipher_descriptor {
332 /** name of cipher */
333 char *name;
334 /** internal ID */
335 unsigned char ID;
336 /** min keysize (octets) */
337 int min_key_length,
338 /** max keysize (octets) */
339 max_key_length,
340 /** block size (octets) */
341 block_length,
342 /** default number of rounds */
343 default_rounds;
344 /** Setup the cipher
345 @param key The input symmetric key
346 @param keylen The length of the input key (octets)
347 @param num_rounds The requested number of rounds (0==default)
348 @param skey [out] The destination of the scheduled key
349 @return CRYPT_OK if successful
350 */
351 int (*setup)(const unsigned char *key, int keylen, int num_rounds, symmetric_key *skey);
352 /** Encrypt a block
353 @param pt The plaintext
354 @param ct [out] The ciphertext
355 @param skey The scheduled key
356 @return CRYPT_OK if successful
357 */
358 int (*ecb_encrypt)(const unsigned char *pt, unsigned char *ct, symmetric_key *skey);
359 /** Decrypt a block
360 @param ct The ciphertext
361 @param pt [out] The plaintext
362 @param skey The scheduled key
363 @return CRYPT_OK if successful
364 */
365 int (*ecb_decrypt)(const unsigned char *ct, unsigned char *pt, symmetric_key *skey);
366 /** Test the block cipher
367 @return CRYPT_OK if successful, CRYPT_NOP if self-testing has been disabled
368 */
369 int (*test)(void);
370
371 /** Terminate the context
372 @param skey The scheduled key
373 */
374 void (*done)(symmetric_key *skey);
375
376 /** Determine a key size
377 @param keysize [in/out] The size of the key desired and the suggested size
378 @return CRYPT_OK if successful
379 */
380 int (*keysize)(int *keysize);
381
382/** Accelerators **/
383 /** Accelerated ECB encryption
384 @param pt Plaintext
385 @param ct Ciphertext
386 @param blocks The number of complete blocks to process
387 @param skey The scheduled key context
388 @return CRYPT_OK if successful
389 */
390 int (*accel_ecb_encrypt)(const unsigned char *pt, unsigned char *ct, unsigned long blocks, symmetric_key *skey);
391
392 /** Accelerated ECB decryption
393 @param pt Plaintext
394 @param ct Ciphertext
395 @param blocks The number of complete blocks to process
396 @param skey The scheduled key context
397 @return CRYPT_OK if successful
398 */
399 int (*accel_ecb_decrypt)(const unsigned char *ct, unsigned char *pt, unsigned long blocks, symmetric_key *skey);

401 /** Accelerated CBC encryption

454 /** Accelerated CCM packet (one-shot)
455 @param key The secret key to use
456 @param keylen The length of the secret key (octets)
457 @param uskey A previously scheduled key [optional can be NULL]
458 @param nonce The session nonce [use once]
459 @param noncelen The length of the nonce
460 @param header The header for the session
461 @param headerlen The length of the header (octets)
462 @param pt [out] The plaintext
463 @param ptlen The length of the plaintext (octets)
464 @param ct [out] The ciphertext
465 @param tag [out] The destination tag
466 @param taglen [in/out] The max size and resulting size of the authentication tag
467 @param direction Encrypt or Decrypt direction (0 or 1)
468 @return CRYPT_OK if successful
469 */
470 int (*accel_ccm_memory)(
471 const unsigned char *key, unsigned long keylen,
472 symmetric_key *uskey,
473 const unsigned char *nonce, unsigned long noncelen,
474 const unsigned char *header, unsigned long headerlen,
475 unsigned char *pt, unsigned long ptlen,
476 unsigned char *ct,
477 unsigned char *tag, unsigned long *taglen,
478 int direction);
479
480 /** Accelerated GCM packet (one shot)
481 @param key The secret key
482 @param keylen The length of the secret key
483 @param IV The initial vector
484 @param IVlen The length of the initial vector
485 @param adata The additional authentication data (header)
486 @param adatalen The length of the adata
487 @param pt The plaintext
488 @param ptlen The length of the plaintext (ciphertext length is the same)
489 @param ct The ciphertext
490 @param tag [out] The MAC tag
491 @param taglen [in/out] The MAC tag length
492 @param direction Encrypt or Decrypt mode (GCM_ENCRYPT or GCM_DECRYPT)
493 @return CRYPT_OK on success
494 */
495 int (*accel_gcm_memory)(
496 const unsigned char *key, unsigned long keylen,
497 const unsigned char *IV, unsigned long IVlen,
498 const unsigned char *adata, unsigned long adatalen,
499 unsigned char *pt, unsigned long ptlen,
500 unsigned char *ct,
501 unsigned char *tag, unsigned long *taglen,
502 int direction);

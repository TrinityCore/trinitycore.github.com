26static int CompareIndexEntries_FilePos(const void *, const void * pvIndexEntry1, const void * pvIndexEntry2)
27{
28 PCASC_INDEX_ENTRY pIndexEntry1 = (PCASC_INDEX_ENTRY)pvIndexEntry1;
29 PCASC_INDEX_ENTRY pIndexEntry2 = (PCASC_INDEX_ENTRY)pvIndexEntry2;
30 ULONGLONG FileOffset1 = ConvertBytesToInteger_5(pIndexEntry1->FileOffsetBE);
31 ULONGLONG FileOffset2 = ConvertBytesToInteger_5(pIndexEntry2->FileOffsetBE);
32 DWORD ArchIndex1 = (DWORD)(FileOffset1 >> 0x1E);
33 DWORD ArchIndex2 = (DWORD)(FileOffset2 >> 0x1E);
34
35 // First, compare the archive index
36 if(ArchIndex1 < ArchIndex2)
37 return -1;
38 if(ArchIndex1 > ArchIndex2)
39 return +1;
40
41 // Second, compare the archive offset
42 FileOffset1 &= 0x3FFFFFFF;
43 FileOffset2 &= 0x3FFFFFFF;
44 if(FileOffset1 < FileOffset2)
45 return -1;
46 if(FileOffset1 > FileOffset2)
47 return +1;
48
49 return 0;
50}

55void CascDumpSparseArray(const char * szFileName, void * pvSparseArray)
56{
57 TSparseArray * pSparseArray = (TSparseArray *)pvSparseArray;
58 FILE * fp;
59
60 // Create the dump file
61 fp = fopen(szFileName, "wt");
62 if(fp != NULL)
63 {
64 // Write header
65 fprintf(fp, "## Value\n-- -----\n");
66
67 // Write the values
68 for(DWORD i = 0; i < pSparseArray->TotalItemCount; i++)
69 {
70 DWORD Value = 0;
71
72 if(pSparseArray->IsItemPresent(i))
73 {
74 Value = pSparseArray->GetItemValue(i);
75 fprintf(fp, "%02X %02X\n", i, Value);
76 }
77 else
78 {
79 fprintf(fp, "%02X --\n", i);
80 }
81 }
82
83 fclose(fp);
84 }

87void CascDumpNameFragTable(const char * szFileName, void * pMarFile)
88{
89 TFileNameDatabase * pDB = ((PMAR_FILE)pMarFile)->pDatabasePtr->pDB;
90 FILE * fp;
91
92 // Create the dump file
93 fp = fopen(szFileName, "wt");
94 if(fp != NULL)
95 {
96 PNAME_FRAG pNameTable = pDB->NameFragTable.NameFragArray;
97 const char * szNames = pDB->IndexStruct_174.NameFragments.CharArray;
98 const char * szLastEntry;
99 char szMatchType[0x100];
100
101 // Dump the table header
102 fprintf(fp, "Indx ThisHash NextHash FragOffs\n");
103 fprintf(fp, "---- -------- -------- --------\n");
104
105 // Dump all name entries
106 for(DWORD i = 0; i < pDB->NameFragTable.ItemCount; i++)
107 {
108 // Reset both match types
109 szMatchType[0] = 0;
110 szLastEntry = "";
111
112 // Only if the table entry is not empty
113 if(pNameTable->ItemIndex != 0xFFFFFFFF)
114 {
115 // Prepare the entry
116 if(IS_SINGLE_CHAR_MATCH(pDB->NameFragTable, i))
117 sprintf(szMatchType, "SINGLE_CHAR (\'%c\')", (pNameTable->FragOffs & 0xFF));
118 else
119 sprintf(szMatchType, "NAME_FRAGMT (\"%s\")", szNames + pNameTable->FragOffs);
120 }
121
122 // Dump the entry
123 fprintf(fp, "0x%02X %08x %08x %08x %s%s\n", i, pNameTable->ItemIndex,
124 pNameTable->NextIndex,
125 pNameTable->FragOffs,
126 szMatchType,
127 szLastEntry);
128 pNameTable++;
129 }
130 fclose(fp);
131 }

134void CascDumpFileNames(const char * szFileName, void * pvMarFile)
135{
136 TMndxFindResult Struct1C;
137 PMAR_FILE pMarFile = (PMAR_FILE)pvMarFile;
138 FILE * fp;
139 char szNameBuff[0x200];
140 bool bFindResult;
141
142 // Create the dump file
143 fp = fopen(szFileName, "wt");
144 if(fp != NULL)
145 {
146 // Set an empty path as search mask (?)
147 Struct1C.SetSearchPath("", 0);
148
149 // Keep searching
150 for(;;)
151 {
152 // Search the next file name
153 pMarFile->pDatabasePtr->sub_1956CE0(&Struct1C, &bFindResult);
154
155 // Stop the search in case of failure
156 if(!bFindResult)
157 break;
158
159 // Printf the found file name
160 memcpy(szNameBuff, Struct1C.szFoundPath, Struct1C.cchFoundPath);
161 szNameBuff[Struct1C.cchFoundPath] = 0;
162 fprintf(fp, "%s\n", szNameBuff);
163 }
164
165 fclose(fp);
166 }
167
168 // Free the search structures
169 Struct1C.FreeStruct40();
170}

220void CascDumpEncodingEntry(
221 TCascStorage * hs,
222 TDumpContext * dc,
223 PCASC_ENCODING_ENTRY pEncodingEntry,
224 int nDumpLevel)
225{
226 PCASC_INDEX_ENTRY pIndexEntry;
227 QUERY_KEY QueryKey;
228 LPBYTE pbIndexKey;
229 char szMd5[MD5_STRING_SIZE+1];
230
231 // If the encoding key exists
232 if(pEncodingEntry != NULL)
233 {
234 dump_print(dc, " Size %lx Key Count: %u\n",
235 ConvertBytesToInteger_4(pEncodingEntry->FileSizeBE),
236 pEncodingEntry->KeyCount);
237
238 // Dump all index keys
239 pbIndexKey = pEncodingEntry->EncodingKey + MD5_HASH_SIZE;
240 for(DWORD j = 0; j < pEncodingEntry->KeyCount; j++, pbIndexKey += MD5_HASH_SIZE)
241 {
242 // Dump the index key
243 dump_print(dc, " %s\n", StringFromMD5(pbIndexKey, szMd5));
244
245 // Dump the index entry as well
246 if(nDumpLevel >= DUMP_LEVEL_INDEX_ENTRIES)
247 {
248 QueryKey.pbData = pbIndexKey;
249 QueryKey.cbData = MD5_HASH_SIZE;
250 pIndexEntry = FindIndexEntry(hs, &QueryKey);
251 CascDumpIndexEntry(hs, dc, pIndexEntry, nDumpLevel);
252 }
253 }
254 }
255 else
256 {
257 dump_print(dc, " NO ENCODING KEYS\n");
258 }

261void CascDumpIndexEntries(const char * szFileName, TCascStorage * hs)
262{
263 PCASC_INDEX_ENTRY * ppIndexEntries;
264 FILE * fp;
265 size_t nIndexEntries = hs->pIndexEntryMap->ItemCount;
266 char szIndexKey[0x40];
267
268 // Create the dump file
269 fp = fopen(szFileName, "wt");
270 if(fp != NULL)
271 {
272 // Create linear aray
273 ppIndexEntries = CASC_ALLOC(PCASC_INDEX_ENTRY, nIndexEntries);
274 if(ppIndexEntries != NULL)
275 {
276 // Obtain the linear array of index entries
277 Map_EnumObjects(hs->pIndexEntryMap, (void **)ppIndexEntries);
278
279 // Sort the array by archive number and archive offset
280 qsort_pointer_array((void **)ppIndexEntries, nIndexEntries, CompareIndexEntries_FilePos, NULL);
281
282 // Parse the array
283 fprintf(fp, "ArNo ArOffset FileSize IndexKey\n==== ======== ======== ================================\n");
284 for(size_t i = 0; i < nIndexEntries; i++)
285 {
286 PCASC_INDEX_ENTRY pIndexEntry = ppIndexEntries[i];
287 ULONGLONG ArchOffset = ConvertBytesToInteger_5(pIndexEntry->FileOffsetBE);
288 DWORD ArchIndex = (DWORD)(ArchOffset >> 0x1E);
289 DWORD FileSize;
290
291 FileSize = ConvertBytesToInteger_4_LE(pIndexEntry->FileSizeLE);
292 ArchOffset &= 0x3FFFFFFF;
293
294 fprintf(fp, " %02X %08X %08X %s\n", ArchIndex, (DWORD)ArchOffset, FileSize, StringFromBinary(pIndexEntry->IndexKey, CASC_FILE_KEY_SIZE, szIndexKey));
295 }
296
297 CASC_FREE(ppIndexEntries);
298 }
299
300 fclose(fp);
301 }

304void CascDumpFile(const char * szFileName, HANDLE hFile)
305{
306 FILE * fp;
307 DWORD dwBytesRead = 1;
308 DWORD dwFilePos = CascSetFilePointer(hFile, 0, NULL, FILE_BEGIN);
309 BYTE Buffer[0x1000];
310
311 // Create the dump file
312 fp = fopen(szFileName, "wb");
313 if(fp != NULL)
314 {
315 // Read data as long as we can, write as long as we can
316 while(dwBytesRead != 0)
317 {
318 // Read from the source file
319 if(!CascReadFile(hFile, Buffer, sizeof(Buffer), &dwBytesRead))
320 break;
321
322 // Write to the destination file
323 if(fwrite(Buffer, 1, dwBytesRead, fp) != dwBytesRead)
324 break;
325 }
326
327 // Close the local file
328 fclose(fp);
329
330 // Restore the file pointer
331 CascSetFilePointer(hFile, dwFilePos, NULL, FILE_BEGIN);
332 }

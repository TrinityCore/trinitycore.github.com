15//-----------------------------------------------------------------------------
16// Structure definitions for Overwatch root file
17
18typedef struct _CASC_FILE_ENTRY
19{
20 ENCODING_KEY EncodingKey; // Encoding key
21 ULONGLONG FileNameHash; // File name hash
22 DWORD dwFileName; // Offset of the file name in the name cache
23} CASC_FILE_ENTRY, *PCASC_FILE_ENTRY;
24
25struct TRootHandler_Ovr : public TRootHandler
26{
27 // Linear global list of file entries
28 DYNAMIC_ARRAY FileTable;
29
30 // Linear global list of names
31 DYNAMIC_ARRAY FileNames;
32
33 // Global map of FileName -> FileEntry
34 PCASC_MAP pRootMap;
35};
36
37//-----------------------------------------------------------------------------
38// Local functions
39
40static int InsertFileEntry(
41 TRootHandler_Ovr * pRootHandler,
42 const char * szFileName,
43 LPBYTE pbEncodingKey)
44{
45 PCASC_FILE_ENTRY pFileEntry;
46 size_t nLength = strlen(szFileName);
47
48 // Attempt to insert the file name to the global buffer
49 szFileName = (char *)Array_Insert(&pRootHandler->FileNames, szFileName, nLength + 1);
50 if(szFileName == NULL)
51 return ERROR_NOT_ENOUGH_MEMORY;
52
53 // Attempt to insert the entry to the array of file entries
54 pFileEntry = (PCASC_FILE_ENTRY)Array_Insert(&pRootHandler->FileTable, NULL, 1);
55 if(pFileEntry == NULL)
56 return ERROR_NOT_ENOUGH_MEMORY;
57
58 // Fill the file entry
59 pFileEntry->EncodingKey = *(PENCODING_KEY)pbEncodingKey;
60 pFileEntry->FileNameHash = CalcFileNameHash(szFileName);
61 pFileEntry->dwFileName = Array_IndexOf(&pRootHandler->FileNames, szFileName);
62
63 // Insert the file entry to the map
64 assert(Map_FindObject(pRootHandler->pRootMap, &pFileEntry->FileNameHash, NULL) == NULL);
65 Map_InsertObject(pRootHandler->pRootMap, pFileEntry, &pFileEntry->FileNameHash);
66 return ERROR_SUCCESS;
67}

69//-----------------------------------------------------------------------------
70// Implementation of Overwatch root file
71
72static int OvrHandler_Insert(
73 TRootHandler_Ovr * pRootHandler,
74 const char * szFileName,
75 LPBYTE pbEncodingKey)
76{
77 return InsertFileEntry(pRootHandler, szFileName, pbEncodingKey);
78}

80static LPBYTE OvrHandler_Search(TRootHandler_Ovr * pRootHandler, TCascSearch * pSearch, PDWORD /* PtrFileSize */, PDWORD /* PtrLocaleFlags */)
81{
82 PCASC_FILE_ENTRY pFileEntry;
83
84 // Are we still inside the root directory range?
85 while(pSearch->IndexLevel1 < pRootHandler->FileTable.ItemCount)
86 {
87 // Retrieve the file item
88 pFileEntry = (PCASC_FILE_ENTRY)Array_ItemAt(&pRootHandler->FileTable, pSearch->IndexLevel1);
89 strcpy(pSearch->szFileName, (char *)Array_ItemAt(&pRootHandler->FileNames, pFileEntry->dwFileName));
90
91 // Prepare the pointer to the next search
92 pSearch->IndexLevel1++;
93 return pFileEntry->EncodingKey.Value;
94 }
95
96 // No more entries
97 return NULL;
98}

100static void OvrHandler_EndSearch(TRootHandler_Ovr * /* pRootHandler */, TCascSearch * /* pSearch */)
101{
102 // Do nothing
103}

105static LPBYTE OvrHandler_GetKey(TRootHandler_Ovr * pRootHandler, const char * szFileName)
106{
107 ULONGLONG FileNameHash = CalcFileNameHash(szFileName);
108
109 return (LPBYTE)Map_FindObject(pRootHandler->pRootMap, &FileNameHash, NULL);
110}
111
112static void OvrHandler_Close(TRootHandler_Ovr * pRootHandler)
113{
114 if(pRootHandler != NULL)
115 {
116 // Free the file map
117 if(pRootHandler->pRootMap)
118 Map_Free(pRootHandler->pRootMap);
119 pRootHandler->pRootMap = NULL;
120
121 // Free the array of the file names and file items
122 Array_Free(&pRootHandler->FileTable);
123 Array_Free(&pRootHandler->FileNames);
124
125 // Free the root file itself
126 CASC_FREE(pRootHandler);
127 }
128}

133int RootHandler_CreateOverwatch(TCascStorage * hs, LPBYTE pbRootFile, DWORD cbRootFile)
134{
135 TRootHandler_Ovr * pRootHandler;
136 ENCODING_KEY KeyBuffer;
137 QUERY_KEY EncodingKey = {KeyBuffer.Value, MD5_HASH_SIZE};
138 void * pTextFile;
139 size_t nLength;
140 char szOneLine[0x200];
141 char szFileName[MAX_PATH+1];
142 DWORD dwFileCountMax = hs->pEncodingMap->TableSize;
143 int nError = ERROR_SUCCESS;
144
145 // Allocate the root handler object
146 hs->pRootHandler = pRootHandler = CASC_ALLOC(TRootHandler_Ovr, 1);
147 if(pRootHandler == NULL)
148 return ERROR_NOT_ENOUGH_MEMORY;
149
150 // Fill-in the handler functions
151 memset(pRootHandler, 0, sizeof(TRootHandler_Ovr));
152 pRootHandler->Insert = (ROOT_INSERT)OvrHandler_Insert;
153 pRootHandler->Search = (ROOT_SEARCH)OvrHandler_Search;
154 pRootHandler->EndSearch = (ROOT_ENDSEARCH)OvrHandler_EndSearch;
155 pRootHandler->GetKey = (ROOT_GETKEY)OvrHandler_GetKey;
156 pRootHandler->Close = (ROOT_CLOSE)OvrHandler_Close;
157
158 // Fill-in the flags
159 pRootHandler->dwRootFlags |= ROOT_FLAG_HAS_NAMES;
160
161 // Allocate the linear array of file entries
162 nError = Array_Create(&pRootHandler->FileTable, CASC_FILE_ENTRY, 0x10000);
163 if(nError != ERROR_SUCCESS)
164 return nError;
165
166 // Allocate the buffer for the file names
167 nError = Array_Create(&pRootHandler->FileNames, char, 0x10000);
168 if(nError != ERROR_SUCCESS)
169 return nError;
170
171 // Create map of ROOT_ENTRY -> FileEntry
172 pRootHandler->pRootMap = Map_Create(dwFileCountMax, sizeof(ULONGLONG), FIELD_OFFSET(CASC_FILE_ENTRY, FileNameHash));
173 if(pRootHandler->pRootMap == NULL)
174 return ERROR_NOT_ENOUGH_MEMORY;
175
176 // Parse the ROOT file
177 pTextFile = ListFile_FromBuffer(pbRootFile, cbRootFile);
178 if(pTextFile != NULL)
179 {
180 // Skip the first line, containing "#MD5|CHUNK_ID|FILENAME|INSTALLPATH"
181 ListFile_GetNextLine(pTextFile, szOneLine, _maxchars(szOneLine));
182
183 // Parse the next lines
184 while((nLength = ListFile_GetNextLine(pTextFile, szOneLine, _maxchars(szOneLine))) > 0)
185 {
186 // Parse the line
187 nError = ParseRootFileLine(szOneLine, szOneLine + nLength, &EncodingKey, szFileName, _maxchars(szFileName));
188 if(nError == ERROR_SUCCESS)
189 {
190 InsertFileEntry(pRootHandler, szFileName, KeyBuffer.Value);
191 }
192 }
193
194 ListFile_Free(pTextFile);
195 }
196
197 // Succeeded
198 return nError;
199}

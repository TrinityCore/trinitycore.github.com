19class bind_stdout : public initializer_base
20{
21public:
22 explicit bind_stdout(const boost::iostreams::file_descriptor_sink &sink)
23 : sink_(sink) {}
24
25 template <class PosixExecutor>

27 {

29 }

31private:

33};

10#ifndef BOOST_PROCESS_POSIX_SEARCH_PATH_HPP
11#define BOOST_PROCESS_POSIX_SEARCH_PATH_HPP
12
13#include <boost/process/config.hpp>
14#include <boost/filesystem.hpp>
15#include <boost/tokenizer.hpp>
16#include <string>
17#include <stdexcept>
18#include <stdlib.h>
19#include <unistd.h>
20
21namespace boost { namespace process { namespace posix {
22
23inline std::string search_path(const std::string &filename,
24 std::string path = "")
25{
26 if (path.empty())
27 {
28 path = ::getenv("PATH");
29 if (path.empty())
30 BOOST_PROCESS_THROW(std::runtime_error(
31 "Environment variable PATH not found"));
32 }
33
34 std::string result;
35 typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
36 boost::char_separator<char> sep(":");
37 tokenizer tok(path, sep);
38 for (tokenizer::iterator it = tok.begin(); it != tok.end(); ++it)
39 {
40 boost::filesystem::path p = *it;
41 p /= filename;
42 if (!::access(p.c_str(), X_OK))
43 {
44 result = p.string();
45 break;
46 }
47 }
48 return result;
49}

106/**
107 * Closes all file descriptors a predicate returns
108 * true for.
109 *
110 * This initializer doesn't close file descriptors
111 * immediately. Instead it sets the \c FD_CLOEXEC
112 * flag. File descriptors are closed when \c execve
113 * is called and the call succeeds.
114 *
115 * \remark <em>POSIX only.</em>
116 */
117class close_fds_if : public initializer_base
118{
119public:
120 /**
121 * Constructor.
122 *
123 * \c predicate_type must be a function or functor with
124 * this signature: <tt>bool(int)</tt>
125 */
126 explicit close_fds_if(const predicate_type &pred);
127};

346/**
347 * Specifies the executable to start.
348 *
349 * This initializer must always be used. The only exception is
350 * if you use \c set_args or a generic initializer which
351 * specifies the executable.
352 */
353class run_exe : public initializer_base
354{
355public:
356 /**
357 * Constructor.
358 *
359 * On Windows \c string_type must be <tt>const char*</tt>,
360 * <tt>std::string</tt> or <tt>boost::filesystem::path</tt>.
361 * If Unicode is used, \c string_type must be
362 * <tt>const wchar_t*</tt>, <tt>std::wstring</tt> or
363 * <tt>boost::filesystem::path</tt>.
364 *
365 * On POSIX \c string_type must be <tt>const char*</tt>,
366 * <tt>std::string</tt> or <tt>boost::filesystem::path</tt>.
367 */
368 explicit run_exe(const string_type &s);
369};

371/**
372 * Sets the command line arguments.
373 *
374 * The first argument specifies the executable to start unless
375 * \c run_exe is used.
376 *
377 * Use \c set_cmd_line if you don't want to pass a collection of
378 * command line arguments but set the command line as one string.
379 */
380class set_args : public initializer_base
381{
382public:
383 /**
384 * Constructor.
385 *
386 * On Windows \c range_type must be a <tt>std::string</tt>-range.
387 * If Unicode is used, \c range_type must be a
388 * <tt>std::wstring</tt>-range.
389 *
390 * On POSIX \c range_type must be a <tt>std::string</tt>-range.
391 */
392 explicit set_args(const range_type &r);
393};

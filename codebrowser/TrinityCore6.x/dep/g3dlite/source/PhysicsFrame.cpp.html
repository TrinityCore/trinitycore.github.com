45PhysicsFrame::PhysicsFrame(const Any& a) {
46 const std::string& n = toLower(a.name());
47 *this = PhysicsFrame();
48
49 if (beginsWith(n, "vector3") || beginsWith(n, "point3")) {
50 *this = PhysicsFrame(Point3(a));
51 } else if (beginsWith(n, "matrix3")) {
52 *this = PhysicsFrame(Matrix3(a));
53 } else if (beginsWith(n, "cframe") || beginsWith(n, "coordinateframe") || beginsWith(n, "matrix4")) {
54 *this = CoordinateFrame(a);
55 } else if (beginsWith(n, "uprightframe")) {
56 *this = UprightFrame(a).toCoordinateFrame();
57 } else if (beginsWith(n, "pframe") || beginsWith(n, "physicsframe")) {
58 if (a.type() == Any::ARRAY) {
59 a.verifySize(2);
60 rotation = a[0];
61 translation = a[1];
62 } else {
63 for (Any::AnyTable::Iterator it = a.table().begin(); it.isValid(); ++it) {
64 const std::string& n = toLower(it->key);
65 if (n == "translation") {
66 translation = it->value;
67 } else if (n == "rotation") {
68 rotation = it->value;
69 } else {
70 a.verify(false, "Illegal table key: " + it->key);
71 }
72 }
73 }
74 } else {
75 a.verify(false, "Unrecognized class name where a PhysicsFrame or equivalent was expected.");
76 }
77}

80PhysicsFrame PhysicsFrame::operator*(const PhysicsFrame& other) const {
81 PhysicsFrame result;
82
83 result.rotation = rotation * other.rotation;
84 result.translation = translation + rotation.toRotationMatrix() * other.translation;
85
86 return result;
87}

90PhysicsFrame::operator CoordinateFrame() const {
91 CoordinateFrame f;
92
93 f.translation = translation;
94 f.rotation = rotation.toRotationMatrix();
95
96 return f;

100PhysicsFrame PhysicsFrame::lerp(
101 const PhysicsFrame& other,
102 float alpha) const {
103
104 PhysicsFrame result;
105
106 result.translation = translation.lerp(other.translation, alpha);
107 result.rotation = rotation.slerp(other.rotation, alpha);
108
109 return result;
110}

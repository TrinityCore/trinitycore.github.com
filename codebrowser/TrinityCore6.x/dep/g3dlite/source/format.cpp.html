25std::string __cdecl format(const char* fmt,...) {
26 va_list argList;
27 va_start(argList,fmt);
28 std::string result = vformat(fmt, argList);
29 va_end(argList);
30
31 return result;
32}

34#if defined(_MSC_VER) && (_MSC_VER >= 1300)
35// Both MSVC seems to use the non-standard vsnprintf
36// so we are using vscprintf to determine buffer size, however
37// only MSVC7 and up headers include vscprintf for some reason.
38std::string vformat(const char *fmt, va_list argPtr) {
39 // We draw the line at a 1MB string.
40 const int maxSize = 1000000;
41
42 // If the string is less than 161 characters,
43 // allocate it on the stack because this saves
44 // the malloc/free time.
45 const int bufSize = 161;
46 char stackBuffer[bufSize];
47
48 // MSVC does not support va_copy
49 int actualSize = _vscprintf(fmt, argPtr) + 1;
50
51 if (actualSize > bufSize) {
52
53 // Now use the heap.
54 char* heapBuffer = NULL;
55
56 if (actualSize < maxSize) {
57
58 heapBuffer = (char*)System::malloc(maxSize + 1);
59 _vsnprintf(heapBuffer, maxSize, fmt, argPtr);
60 heapBuffer[maxSize] = '\0';
61 } else {
62 heapBuffer = (char*)System::malloc(actualSize);
63 vsprintf(heapBuffer, fmt, argPtr);
64 }
65
66 std::string formattedString(heapBuffer);
67 System::free(heapBuffer);
68 return formattedString;
69 } else {
70
71 vsprintf(stackBuffer, fmt, argPtr);
72 return std::string(stackBuffer);
73 }
74}

78std::string vformat(const char *fmt, va_list argPtr) {
79 // We draw the line at a 1MB string.
80 const int maxSize = 1000000;
81
82 // If the string is less than 161 characters,
83 // allocate it on the stack because this saves
84 // the malloc/free time.
85 const int bufSize = 161;
86 char stackBuffer[bufSize];
87
88 // MSVC6 doesn't support va_copy, however it also seems to compile
89 // correctly if we just pass our argument list along. Note that
90 // this whole code block is only compiled if we're on MSVC6 anyway
91 int actualWritten = _vsnprintf(stackBuffer, bufSize, fmt, argPtr);
92
93 // Not a big enough buffer, bufSize characters written
94 if (actualWritten == -1) {
95
96 int heapSize = 512;
97 double powSize = 1.0;
98 char* heapBuffer = (char*)System::malloc(heapSize);
99
100 while ((_vsnprintf(heapBuffer, heapSize, fmt, argPtr) == -1) &&
101 (heapSize < maxSize)) {
102
103 heapSize = iCeil(heapSize * ::pow((double)2.0, powSize++));
104 heapBuffer = (char*)System::realloc(heapBuffer, heapSize);
105 }
106
107 heapBuffer[heapSize-1] = '\0';
108
109 std::string heapString(heapBuffer);
110 System::free(heapBuffer);
111
112 return heapString;
113 } else {
114
115 return std::string(stackBuffer);
116 }
117}

121// glibc 2.1 has been updated to the C99 standard
122std::string vformat(const char* fmt, va_list argPtr) {
123 // If the string is less than 161 characters,
124 // allocate it on the stack because this saves
125 // the malloc/free time. The number 161 is chosen
126 // to support two lines of text on an 80 character
127 // console (plus the null terminator).
128 const int bufSize = 161;
129 char stackBuffer[bufSize];
130
131 va_list argPtrCopy;
132 va_copy(argPtrCopy, argPtr);
133 int numChars = vsnprintf(stackBuffer, bufSize, fmt, argPtrCopy);
134 va_end(argPtrCopy);
135
136 if (numChars >= bufSize) {
137 // We didn't allocate a big enough string.
138 char* heapBuffer = (char*)System::malloc((numChars + 1) * sizeof(char));
139
140 debugAssert(heapBuffer);
141 int numChars2 = vsnprintf(heapBuffer, numChars + 1, fmt, argPtr);
142 debugAssert(numChars2 == numChars);
143 (void)numChars2;
144
145 std::string result(heapBuffer);
146
147 System::free(heapBuffer);
148
149 return result;
150
151 } else {
152
153 return std::string(stackBuffer);
154
155 }

19class BasicThread: public GThread {
20public:
21 BasicThread(const std::string& name, void (*proc)(void*), void* param):
22 GThread(name), m_wrapperProc(proc), m_param(param) { }
23protected:
24 virtual void threadMain() {
25 m_wrapperProc(m_param);
26 }

29 void (*m_wrapperProc)(void*);

81#ifdef G3D_WINDOWS
82// From http://msdn.microsoft.com/en-us/library/xcb2z8hs.aspx
83const DWORD MS_VC_EXCEPTION=0x406D1388;
84
85#pragma pack(push,8)
86typedef struct tagTHREADNAME_INFO {
87 DWORD dwType; // Must be 0x1000.
88 LPCSTR szName; // Pointer to name (in user addr space).
89 DWORD dwThreadID; // Thread ID (-1=caller thread).
90 DWORD dwFlags; // Reserved for future use, must be zero.
91} THREADNAME_INFO;
92#pragma pack(pop)
93
94static void SetThreadName(DWORD dwThreadID, const char* threadName) {
95 THREADNAME_INFO info;
96 info.dwType = 0x1000;
97 info.szName = threadName;
98 info.dwThreadID = dwThreadID;
99 info.dwFlags = 0;
100
101 __try {
102 RaiseException( MS_VC_EXCEPTION, 0, sizeof(info)/sizeof(ULONG_PTR), (ULONG_PTR*)&info );
103 } __except(EXCEPTION_EXECUTE_HANDLER) {}
104}
105#endif

108bool GThread::start(SpawnBehavior behavior) {
109
110 debugAssertM(! started(), "Thread has already executed.");
111 if (started()) {
112 return false;
113 }
114
115 m_status = STATUS_STARTED;
116
117 if (behavior == USE_CURRENT_THREAD) {
118 // Run on this thread
119 m_status = STATUS_RUNNING;
120 threadMain();
121 m_status = STATUS_COMPLETED;
122 return true;
123 }
124
125# ifdef G3D_WINDOWS
126 DWORD threadId;
127
128 m_event = ::CreateEvent(NULL, TRUE, FALSE, NULL);
129 debugAssert(m_event);
130
131 m_handle = ::CreateThread(NULL, 0, &internalThreadProc, this, 0, &threadId);
132
133 if (m_handle == NULL) {
134 ::CloseHandle(m_event);
135 m_event = NULL;
136 }
137
138 SetThreadName(threadId, m_name.c_str());
139
140 return (m_handle != NULL);
141# else
142 if (!pthread_create(&m_handle, NULL, &internalThreadProc, this)) {
143 return true;
144 } else {
145 // system-independent clear of handle
146 System::memset(&m_handle, 0, sizeof(m_handle));
147
148 return false;
149 }
150# endif
151}

153void GThread::terminate() {
154 if (m_handle) {
155# ifdef G3D_WINDOWS
156 ::TerminateThread(m_handle, 0);
157# else
158 pthread_kill(m_handle, SIGSTOP);
159# endif
160 // system-independent clear of handle
161 System::memset(&m_handle, 0, sizeof(m_handle));
162 }
163}

176void GThread::waitForCompletion() {
177 if (m_status == STATUS_COMPLETED) {
178 // Must be done
179 return;
180 }
181
182# ifdef G3D_WINDOWS
183 debugAssert(m_event);
184 ::WaitForSingleObject(m_event, INFINITE);
185# else
186 debugAssert(m_handle);
187 pthread_join(m_handle, NULL);
188# endif
189}

192#ifdef G3D_WINDOWS
193DWORD WINAPI GThread::internalThreadProc(LPVOID param) {
194 GThread* current = reinterpret_cast<GThread*>(param);
195 debugAssert(current->m_event);
196 current->m_status = STATUS_RUNNING;
197 current->threadMain();
198 current->m_status = STATUS_COMPLETED;
199 ::SetEvent(current->m_event);
200 return 0;
201}
202#else
203void* GThread::internalThreadProc(void* param) {
204 GThread* current = reinterpret_cast<GThread*>(param);
205 current->m_status = STATUS_RUNNING;
206 current->threadMain();
207 current->m_status = STATUS_COMPLETED;
208 return (void*)NULL;
209}
210#endif

214//GMutex implementation
215GMutex::GMutex() {
216#ifdef G3D_WINDOWS
217 ::InitializeCriticalSection(&m_handle);
218#else
219 int ret = pthread_mutexattr_init(&m_attr);
220 debugAssert(ret == 0);
221 ret = pthread_mutexattr_settype(&m_attr, PTHREAD_MUTEX_RECURSIVE);
222 debugAssert(ret == 0);
223 ret = pthread_mutex_init(&m_handle, &m_attr);
224 debugAssert(ret == 0);
225#endif
226}
227
228GMutex::~GMutex() {
229 //TODO: Debug check for locked

234 debugAssert(ret == 0);
235 ret = pthread_mutexattr_destroy(&m_attr);
236 debugAssert(ret == 0);
237#endif
238}

240bool GMutex::tryLock() {
241#ifdef G3D_WINDOWS
242 return (::TryEnterCriticalSection(&m_handle) != 0);
243#else
244 return (pthread_mutex_trylock(&m_handle) == 0);
245#endif
246}

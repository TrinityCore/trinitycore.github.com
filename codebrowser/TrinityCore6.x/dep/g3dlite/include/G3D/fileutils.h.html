70/**
71 Parses a filename into four useful pieces.
72
73 Examples:
74
75 c:\\a\\b\\d.e
76 root = "c:\\"
77 path = "a" "b"
78 base = "d"
79 ext = "e"
80
81 /a/b/d.e
82 root = "/"
83 path = "a" "b"
84 base = "d"
85 ext = "e"
86
87 /a/b
88 root = "/"
89 path = "a"
90 base = "b"
91 ext = "e"
92
93 */
94void parseFilename(
95 const std::string& filename,
96 std::string& drive,
97 Array<std::string>& path,
98 std::string& base,
99 std::string& ext);
100
101
102/**
103 Returns the part of the filename that includes the base and ext from
104 parseFilename (i.e. everything to the right of the path).
105 */
106std::string filenameBaseExt(const std::string& filename);
107
108/**
109 Returns the extension on a filename.
110 */
111std::string filenameExt(const std::string& filename);
112
113
114/** Returns the portion of a filename to the left of the last period
115 and to the right of the last slash or colon.
116 */
117std::string filenameBase(const std::string& filename);

119/** Creates a unique filename base in the current directory using the

123/**
124 Returns the drive (if Win32) and path from a filename, including

128std::string filenamePath(const std::string& filename);

24/**
25\def G3D_DECLARE_ENUM_CLASS_METHODS
26
27 \brief Creates a series of methods that turn a class into a scoped enumeration.
28
29 Example of use:
30
31 \code
32 class Resource {
33 public:
34 enum Value {FUEL, FOOD, WATER} value;
35
36 // i is the position the enum value in Value (not the enum value itself)
37 static const char* toString(int i, Value& v) {
38 static const char* str[] = {"FUEL", "FOOD", "WATER", NULL}; // Whatever your enum values are
39 static const Value val[] = {FUEL, FOOD, WATER}; // Whatever your enum values are
40 const char* s = str[i];
41 if (s) {
42 v = val[i];
43 }
44 return s;
45 }
46
47 G3D_DECLARE_ENUM_CLASS_METHODS(Resource);
48 };
49 G3D_DECLARE_ENUM_CLASS_HASHCODE(Resource);
50 \endcode
51
52 Extends the "Intelligent Enum" design pattern
53 http://www.codeguru.com/cpp/cpp/cpp_mfc/article.php/c4001/
54
55 Enum classes are initialized to their zero value by default.
56
57 See GLG3D/GKey.h and G3D/WrapMode for an example.
58 \sa G3D_DECLARE_ENUM_CLASS_HASHCODE
59 */
60#define G3D_DECLARE_ENUM_CLASS_METHODS(Classname)\
61private: \
62 void fromString(const std::string& x) {\
63 Value v = (Value)0;\
64 const char* s;\
65 int i = 0;\
66\
67 do {\
68 s = toString(i, v);\
69 if (s == NULL) { return; /** Needed to get correct compilation on gcc */ } \
70 if (x == s) {\
71 value = v;\
72 return;\
73 }\
74 ++i;\
75 } while (true);\
76 }\
77\
78public:\
79 static const char* classname() {\
80 return #Classname;\
81 }\
82\
83 const char* toString() const {\
84 const char* s;\
85 int i = 0;\
86 Value v = (Value)0;\
87 while (true) {\
88 s = toString(i, v);\
89 if ((s == NULL) || (v == value)) {\
90 return s;\
91 }\
92 ++i;\
93 }\
94 }\
95\
96 explicit Classname(const std::string& x) : value((Value)0) {\
97 fromString(x);\
98 }\
99\
100 explicit Classname(const G3D::Any& a) : value((Value)0) { \
101 fromString(a.string());\
102 }\
103\
104 G3D::Any toAny() const { \
105 return G3D::Any(toString()); \
106 }\
107\
108 explicit Classname(char v) : value((Value)v) {}\
109\
110 Classname() : value((Value)0) {}\
111\
112 Classname(const Value v) : value(v) {}\
113\
114 explicit Classname(int v) : value((Value)v) {}\
115\
116 operator int() const {\
117 return (int)value;\

227/**
228\def G3D_DECLARE_ENUM_CLASS
229
230\code
231// Arguments may not have initializer expressions. Arguments may contain comments.
232// Namespaces aren't *required*, this example just shows how to use them.
233namespace Foo {
234 G3D_DECLARE_ENUM_CLASS(Day, SUNDAY, MONDAY, TUESDAY, WEDNESDAY, THURSDAY, FRIDAY, SATURDAY);
235}
236G3D_DECLARE_ENUM_CLASS_HASHCODE(Foo::Day);
237...
238
239using namespace Foo;
240// Example use of the smart enum
241Day d = Day::TUESDAY;
242Day d2("SATURDAY");
243Any a(d);
244d = a;
245printf("%s = %d\n", d.toString(), d.value);
246\endcode
247
248\sa G3D_DECLARE_ENUM_CLASS_METHODS, G3D_DECLARE_ENUM_CLASS_HASHCODE, G3D::enumToJavaScriptDeclaration, G3D_BEGIN_ENUM_CLASS_DECLARATION
249*/
250#define G3D_DECLARE_ENUM_CLASS(ClassName, ...)\
251 G3D_BEGIN_ENUM_CLASS_DECLARATION(ClassName, __VA_ARGS__);\
252 G3D_END_ENUM_CLASS_DECLARATION();
253
254/** \def G3D_BEGIN_ENUM_CLASS_DECLARATION
255
256\code
257G3D_BEGIN_ENUM_CLASS_DECLARATION(Day, SUNDAY, MONDAY, TUESDAY, WEDNESDAY, THURSDAY, FRIDAY, SATURDAY);
258 // Put extra methods here, e.g.,
259 Value nextValue() { ... }
260G3D_END_ENUM_CLASS_DECLARATION();
261}
262G3D_DECLARE_ENUM_CLASS_HASHCODE(Foo::Day);
263
264\endcode
265 \sa G3D_DECLARE_ENUM_CLASS, G3D_END_ENUM_CLASS_DECLARATION
266*/
267#define G3D_BEGIN_ENUM_CLASS_DECLARATION(ClassName, ...)\
268class ClassName {\
269public:\
270 enum Value {\
271 __VA_ARGS__\
272 } value;\
273\
274 /* The static variables here may be duplicated in different shared object binaries (DLLs), but that's ok--we only depend on their values, not their uniqueness. See also http://stackoverflow.com/questions/11962918/local-static-variable-is-instantiated-multiple-times-why */\
275 static const char* toString(int i, Value& v) {\
276 static const char** str = G3D::_internal::smartEnumParseNames(#__VA_ARGS__);\
277 static const Value val[] = {__VA_ARGS__};\
278 const char* s = str[i];\
279 if (s) { v = val[i]; }\
280 return s;\
281 }\
282\
283 G3D_DECLARE_ENUM_CLASS_METHODS(ClassName);
284
285/** \def G3D_END_ENUM_CLASS_DECLARATION
286 \sa G3D_BEGIN_ENUM_CLASS_DECLARATION */
287#define G3D_END_ENUM_CLASS_DECLARATION() }

292/**
293 \brief Generates JavaScript source code defining an enum equivalent to
294 EnumClass.
295
296 \code
297 TextOutput t("WrapMode.js");
298 enumToJavaScriptDeclaration<WrapMode, WrapMode::Value>(t);
299 t.commit();
300 \endcode
301 */
302template<class EnumClass, class EnumClassValue>
303void enumToJavaScriptDeclaration(TextOutput& t) {
304 t.printf("/* BEGIN GENERATED CODE. The following was automatically generated by G3D::enumToJavaScriptDeclaration(). Do not edit it manually. */\n\n");
305 t.printf("var %s = (function (propList) {", EnumClass::classname()); t.writeNewline();
306 t.pushIndent(); {
307 t.printf("// Define immutable properties"); t.writeNewline();
308 t.printf("var en = {};"); t.writeNewline();
309 t.printf("for (var i = 0; i < propList.length; i += 2)"); t.writeNewline();
310 t.pushIndent(); {
311 t.printf("Object.defineProperty(en, propList[i], {enumerable: true, value: propList[i + 1]});"); t.writeNewline();
312 } t.popIndent();
313 t.printf("return en;");
314 t.writeNewline();
315 } t.popIndent();
316 t.printf("})([");
317 int i = 0;
318 EnumClassValue m;
319 const char* s = EnumClass::toString(i, m);
320 while (notNull(s)) {
321 t.printf("\"%s\", %d, ", s, int(m));
322 ++i;
323 s = EnumClass::toString(i, m);
324 }
325 // JavaScript allows a trailing comma
326 t.printf("]);"); t.writeNewline();
327 t.printf("/* END GENERATED CODE */"); t.writeNewline();
328}

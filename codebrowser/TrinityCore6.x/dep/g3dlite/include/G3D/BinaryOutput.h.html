40/**
41 Sequential or random access byte-order independent binary file access.
42
43 The compress() call can be used to compress with zlib.
44
45 Any method call can trigger an out of memory error (thrown as char*)
46 when writing to "<memory>" instead of a file.
47
48 Compressed writing and seeking backwards is not supported for huge files
49 (i.e., BinaryOutput may have to dump the contents to disk if they
50 exceed available RAM).
51 */

53private:

97 /**
98 Make sure at least bytes can be written, resizing if
99 necessary.
100 */
101 inline void reserveBytes(size_t bytes) {
102 debugAssert(bytes > 0);
103 size_t oldBufferLen = m_bufferLen;
104
105 m_bufferLen = max(m_bufferLen, (size_t)(m_pos + bytes));
106 if (m_bufferLen > m_maxBufferLen) {
107 reallocBuffer(bytes, oldBufferLen);
108 }
109 }

167 /**
168 Write the bytes to disk. It is ok to call this
169 multiple times; it will just overwrite the previous file.
170
171 Parent directories are created as needed if they do
172 not exist.
173
174 <B>Not</B> called from the destructor; you must call
175 it yourself.
176
177 @param flush If true (default) the file is ready for reading when the method returns, otherwise

196 inline int64 length() const {
197 return m_bufferLen + m_alreadyWritten;
198 }
199
200 inline int64 size() const {
201 return length();
202 }
203
204 /**
205 Sets the length of the file to n, padding
206 with 0's past the current end. Does not
207 change the position of the next byte to be
208 written unless n < size().
209
210 Throws char* when resetting a huge file to be shorter
211 than its current length.
212 */
213 inline void setLength(int64 n) {
214 n = n - m_alreadyWritten;
215
216 if (n < 0) {
217 throw "Cannot resize huge files to be shorter.";
218 }
219
220 if (n < (int64)m_bufferLen) {
221 m_pos = n;
222 }
223 if (n > (int64)m_bufferLen) {
224 reserveBytes((size_t)(n - m_bufferLen));
225 }
226 }
227
228 /**
229 Returns the current byte position in the file,
230 where 0 is the beginning and getLength() - 1 is the end.
231 */
232 inline int64 position() const {
233 return m_pos + m_alreadyWritten;
234 }
235
236
237 /**
238 Sets the position. Can set past length, in which case
239 the file is padded with zeros up to one byte before the
240 next to be written.
241
242 May throw a char* exception when seeking backwards on a huge file.
243 */
244 inline void setPosition(int64 p) {
245 p = p - m_alreadyWritten;
246
247 if (p > (int64)m_bufferLen) {
248 setLength((int)(p + (int64)m_alreadyWritten));
249 }
250
251 if (p < 0) {
252 throw "Cannot seek more than 10 MB backwards on huge files.";
253 }
254
255 m_pos = (int)p;
256 }

258
259 void writeBytes
260 (const void* b,
261 size_t count) {
262
263 reserveBytes(count);
264 debugAssert(m_pos >= 0);
265 debugAssert(m_bufferLen >= count);
266 System::memcpy(m_buffer + m_pos, b, count);
267 m_pos += count;
268 }

270 /**
271 Writes a signed 8-bit integer to the current position.
272 */
273 inline void writeInt8(int8 i) {
274 reserveBytes(1);

338 /** Write a string that always consumes len bytes, truncating or padding as necessary*/
339 inline void writeString(const std::string& s, int len) {
340 const int pad = len - ((int)s.length() + 1);
341 if (pad >= 0) {
342 writeString(s.c_str());
343 for (int i = 0; i < pad; ++i) {
344 writeUInt8(0);
345 }
346 } else {
347 // Truncate
348 writeBytes(s.c_str(), len);
349 }
350 }

384 /**
385 Skips ahead n bytes.
386 */
387 inline void skip(int n) {
388 if (m_pos + n > (int64)m_bufferLen) {
389 setLength(m_pos + m_alreadyWritten + n);
390 }
391 m_pos += n;
392 }

394 /** Call before a series of BinaryOutput::writeBits calls. Only writeBits
395 can be called between beginBits and endBits without corrupting the stream.*/
396 void beginBits();
397
398 /** Write numBits from bitString to the output stream. Bits are numbered from
399 low to high.
400
401 Can only be
402 called between beginBits and endBits. Bits written are semantically
403 little-endian, regardless of the actual endian-ness of the system. That is,
404 <CODE>writeBits(0xABCD, 16)</CODE> writes 0xCD to the first byte and
405 0xAB to the second byte. However, if used with BinaryInput::readBits, the ordering
406 is transparent to the caller.
407 */
408 void writeBits(uint32 bitString, int numBits);
409
410 /** Call after a series of BinaryOutput::writeBits calls. This will
411 finish out with zeros the last byte into which bits were written.*/
412 void endBits();

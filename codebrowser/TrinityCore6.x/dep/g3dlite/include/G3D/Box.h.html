29/**
30 \brief An arbitrary (oriented) 3D box, useful as a bounding box.
31
32 To construct a box from a coordinate frame, center and extent, use the idiom:
33 <CODE>Box box = cframe.toObjectSpace(Box(center - extent/2, center + extent/2));</CODE>
34 */
35class Box {
36private:
37
38 static int32 dummy;
39
40 friend class CoordinateFrame;
41
42 /**
43 Axes with length equal to the 4 edges that run along each of them
44 */
45 Vector3 _edgeVector[3];
46
47 Point3 _center;
48
49 float _area;
50
51 float _volume;
52
53 void init(
54 const Vector3& min,
55 const Vector3& max);
56
57public:
58
59 Box();
60
61 explicit Box(const Any& a);

63 /**

66 Box(const Vector3& min,
67 const Vector3& max);

69 Box(const Vector3& osMin,
70 const Vector3& osMax,
71 const CoordinateFrame& frame);
72
73 Box(class BinaryInput& b);
74
75 Box(const class AABox& b);

77 explicit Box(const Point3& p);
78
79 static Box inf();
80
81 Any toAny() const;
82
83 void serialize(class BinaryOutput& b) const;
84 void deserialize(class BinaryInput& b);
85
86 /**
87 Returns the object to world transformation for
88 this box, where the origin is the center of the box. localFrame().worldToObject(...) takes
89 objects into the space where the box axes are
90 (1,0,0), (0,1,0), (0,0,1). Note that there
91 is no scaling in this transformation.
92 */
93 CoordinateFrame localFrame() const;
94
95 /** \sa localFrame */
96 void getLocalFrame(CoordinateFrame& frame) const;
97
98 Box operator*(float f) const;
99
100 /**
101 Returns the centroid of the box.
102 */
103 inline Vector3 center() const {
104 return _center;
105 }
106
107 /**
108 \htmlonly
109 <PRE>
110
111
112 2--------3
113 / : /|
114 / : / |
115 6--------7 |
116 | : | |
117 | 0....|..1
118 | / | /
119 |/ |/
120 4--------5
121
122 y
123 ^
124 |
125 |-->x
126 z/
127
128
129 </PRE>
130 \endhtmlonly
131 */
132 Vector3 corner(int i) const;
133
134 /**
135 Unit length.
136 */
137 inline Vector3 axis(int a) const {
138 debugAssert(a < 3);
139 return _edgeVector[a].direction();
140 }
141
142 /**
143 Distance from corner(0) to the next corner
144 along the box's local axis a.
145 */
146 inline float extent(int a) const {
147 debugAssert(a < 3);
148 return _edgeVector[a].length();
149 }
150
151 inline Vector3 extent() const {
152 return Vector3(_edgeVector[0].length(), _edgeVector[1].length(), _edgeVector[2].length());
153 }
154
155 /**
156 Returns the four corners of a face (0 <= f < 6).
157 The corners are returned to form a clockwise quad facing outwards.
158
159
160
161 +--------+
162 / : /|
163 / : / |
164 +--------+ |
165 | : | |
166 | +....|..+
167 | / | /
168 |/ |/
169 +--------+
170
171 y
172 ^
173 |
174 |-->x
175z/
176 Faces are in the following order:
177 0: -Z
178 1: X
179 2: Z
180 3: Y
181 4: -X
182 5: -Y
183
184 */
185 void getFaceCorners(
186 int f,
187 Vector3& v0,
188 Vector3& v1,
189 Vector3& v2,
190 Vector3& v3) const;

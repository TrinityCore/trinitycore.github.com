20/**
21 OS-independent file system layer that optimizes the performance
22 of queries by caching and prefetching.
23
24 This class uses the following definitions:
25 <ul>
26 <li> "file" = document that can be opened for reading or writing
27 <li> "directory" = folder containing files and other directories
28 <li> "node" = file or directory
29 <li> "path" = string identifying a (see the FSPath class)
30 <li> "zipfile" = a compressed file storing an archive of files and directories in the zip format
31 </ul>
32
33 In G3D, Zipfiles are transparently treated as if they were directories, provided:
34 <ul>
35 <li> The zipfile name contains an extension (e.g., map.pk3, files.zip)
36 <li> There are no nested zipfiles
37 </ul>
38
39 All FileSystem routines invoke FilePath::expandEnvironmentVariables if the input contains a '$'.
40
41 The extension requirement allows G3D to quickly identify whether a path could enter a
42 zipfile without forcing it to open all parent directories for reading.
43
44 \sa FilePath
45*/
46class FileSystem {

386 /** Returns true if a node named \a f exists.
387
388 \param f If \a f contains wildcards, the function returns true if any file
389 matches those wildcards. Wildcards may only appear in the base or ext, not the
390 path. Environment variables beginning with dollar signs (e.g., in "$G3DDATA/cubemap"),
391 with optional parens ("$(G3DDATA)") are
392 automatically expanded in \a f. Default share names on Windows (e.g., "\\mycomputer\c$")
393 are correctly distinguished from empty environment variables.
394
395 \param trustCache If true, uses the cache for optimizing repeated calls
396 in the same parent directory.
397
398 \param caseSensitive If true, the match must have exactly the same case for the base and extension. If false,
399 case is ignored. The default on Windows is false and the default on other operating systems is true.
400 */

404#else
405 true
406#endif

408 mutex.lock();

512/** \brief Parsing of file system paths.
513
514 None of these routines touch the disk--they are purely string manipulation.
515
516 In "/a/b/base.ext",
517
518 <ul>
519 <li> base = "base"
520 <li> ext = "ext"
521 <li> parentPath = "/a/b"
522 <li> baseExt = "base.ext"
523 </ul>
524
525*/
526class FilePath {
527public:
528
529 /** Appends file onto dirname, ensuring a / if needed. */
530 static std::string concat(const std::string& a, const std::string& b);
531
532 /** Returns true if \a f specifies a path that parses as root of the filesystem.
533 On OS X and other Unix-based operating systems, "/" is the only root.
534 On Windows, drive letters and shares are roots, e.g., "c:\", "\\foo\".
535 Does not check on Windows to see if the root is actually mounted or a legal
536 drive letter--this is a purely string based test. */
537 static bool isRoot(const std::string& f);
538
539 /** Removes the trailing slash unless \a f is a filesystem root */
540 static std::string removeTrailingSlash(const std::string& f);
541
542 /** Returns everything to the right of the last '.' */
543 static std::string ext(const std::string& path);
544
545 /** Returns everything to the right of the last slash (or, on Windows, the last ':') */
546 static std::string baseExt(const std::string& path);
547
548 /** Returns everything between the right-most slash and the following '.' */
549 static std::string base(const std::string& path);
550
551 /** Returns everything to the left of the right-most slash */
552 static std::string parent(const std::string& path);
553
554 /** Returns true if '*' or '?' appear in the filename */
555 static bool containsWildcards(const std::string& p);
556
557 /** Convert all slashes to '/' */
558 static std::string canonicalize(std::string x);
559
560 /** \brief Replaces <code>$VAR</code> and <code>$(VAR)</code> patterns with the corresponding environment variable.
561 Throws std::string if the environment variable is not defined.
562 */
563 static std::string expandEnvironmentVariables(const std::string& path);
564
565 /**
566 Parses a filename into four useful pieces.
567
568 Examples:
569
570 c:\\a\\b\\d.e
571 root = "c:\\"
572 path = "a" "b"
573 base = "d"
574 ext = "e"
575
576 /a/b/d.e
577 root = "/"
578 path = "a" "b"
579 base = "d"
580 ext = "e"
581
582 /a/b
583 root = "/"
584 path = "a"
585 base = "b"
586 ext = "e"
587
588 */
589 static void parse
590 (const std::string& filename,
591 std::string& drive,
592 Array<std::string>& path,
593 std::string& base,
594 std::string& ext);
595
596 /**
597 Returns true if \a path matches \a pattern, with standard filesystem wildcards.
598 */
599 static bool matches(const std::string& path, const std::string& pattern, bool caseSensitive = true);
600
601 /** Replaces characters that are illegal in a filename with legal equivalents.*/
602 static std::string makeLegalFilename(const std::string& f, size_t maxLength = 100000);
603};

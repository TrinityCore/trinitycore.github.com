14/** Manages a set of threads. All methods are threadsafe except for
15 the iterator begin/end.
16
17 @beta*/
18class ThreadSet : public ReferenceCountedObject {
19public:
20 /** Intended to allow future use with a template parameter.*/
21 typedef GThread Thread;
22
23 typedef shared_ptr<Thread> ThreadRef;
24 typedef shared_ptr<ThreadSet> Ref;
25 typedef Array<ThreadRef>::Iterator Iterator;
26 typedef Array<ThreadRef>::ConstIterator ConstIterator;
27
28private:
29
30 /** Protects m_thread */
31 GMutex m_lock;
32
33 /** Threads in the set */
34 Array<ThreadRef> m_thread;
35
36public:
37
38 /** Total number of threads (some of which may be completed). */
39 int size() const;
40
41 /** Number of threads that have been started */
42 int numStarted() const;
43
44 /** Start all threads that are not currently started.
45
46 @param lastThreadBehavior If USE_CURRENT_THREAD, takes the
47 last unstarted thread and executes it manually on the current
48 thread. This helps to take full advantage of the machine when
49 running a large number of jobs and avoids the overhead of a
50 thread start for single-thread groups. Note that this forces
51 start() to block until that thread is complete.
52 */
53 void start(SpawnBehavior lastThreadBehavior = USE_NEW_THREAD) const;
54
55 /** Terminate all threads that are currently started */
56 void terminate() const;
57
58 /** Waits until all started threads have completed. */
59 void waitForCompletion() const;
60
61 /** Remove all (not stopping them) */
62 void clear();
63
64 /** Removes completed threads and returns the new size.*/
65 int removeCompleted();
66
67 /** Inserts a new thread, if it is not already present, and
68 returns the new number of threads.*/
69 int insert(const ThreadRef& t);
70
71 /** Removes a thread. Returns true if the thread was present and
72 removed. */
73 bool remove(const ThreadRef& t);
74
75 bool contains(const ThreadRef& t) const;
76
77 /** It is an error to mutate the ThreadSet while iterating through it. */
78 Iterator begin();
79
80 Iterator end();
81
82 ConstIterator begin() const;
83
84 ConstIterator end() const;
85};

23/**
24 Arbitrary quaternion (not necessarily unit).
25
26 Unit quaternions (aka versors) are used in computer graphics to represent
27 rotation about an axis. Any 3x3 rotation matrix can
28 be stored as a quaternion.
29
30 A quaternion represents the sum of a real scalar and
31 an imaginary vector: ix + jy + kz + w. A unit quaternion
32 representing a rotation by A about axis v has the form
33 [sin(A/2)*v, cos(A/2)]. For a unit quaternion, q.conj() == q.inverse()
34 is a rotation by -A about v. -q is the same rotation as q
35 (negate both the axis and angle).
36
37 A non-unit quaterion q represents the same rotation as
38 q.unitize() (Dam98 pg 28).
39
40 Although quaternion-vector operations (eg. Quat + Vector3) are
41 well defined, they are not supported by this class because
42 they typically are bugs when they appear in code.
43
44 Do not subclass.
45
46 <B>BETA API -- subject to change</B>
47 \cite Erik B. Dam, Martin Koch, Martin Lillholm, Quaternions, Interpolation and Animation. Technical Report DIKU-TR-98/5, Department of Computer Science, University of Copenhagen, Denmark. 1998.
48 */

60 q = [sin(angle / 2) * axis, cos(angle / 2)]
61
62 In Watt & Watt's notation, s = w, v = (x, y, z)

64 */

166 float dot(const Quat& other) const {
167 return (x * other.x) + (y * other.y) + (z * other.z) + (w * other.w);

170 /** Note: two quats can represent the Quat::sameRotation and not be equal. */
171 bool fuzzyEq(const Quat& q) {
172 return G3D::fuzzyEq(x, q.x) && G3D::fuzzyEq(y, q.y) && G3D::fuzzyEq(z, q.z) && G3D::fuzzyEq(w, q.w);

176 represented by two values; q and -q).

179 return fuzzyEq(q) || fuzzyEq(-q);

218 /** \param maxAngle Maximum angle of rotation allowed. If a larger rotation is required, the angle of rotation applied is clamped to maxAngle */
219 Quat slerp
220 (const Quat& other,
221 float alpha,
222 float threshold,
223 float maxAngle) const;
224public:

226 /**
227 Spherical linear interpolation: linear interpolation along the
228 shortest (3D) great-circle route between two quaternions.
229
230 Assumes that both arguments are unit quaternions.
231
232 Note: Correct rotations are expected between 0 and PI in the right order.
233
234 \cite Based on Game Physics -- David Eberly pg 538-540
235
236 \param threshold Critical angle between between rotations (in radians) at which
237 the algorithm switches to normalized lerp, which is more
238 numerically stable in those situations. 0.0 will always slerp.
239
240 */
241 Quat slerp
242 (const Quat& other,
243 float alpha,
244 float threshold = 0.05f) const {
245 return slerp(other, alpha, threshold, finf());
246 }
247
248 /** Rotates towards \a other by at most \a maxAngle. */
249 Quat movedTowards
250 (const Quat& other,
251 float maxAngle) const {
252 return slerp(other, 1.0f, 0.05f, maxAngle);
253 }
254
255 /** Rotates towards \a other by at most \a maxAngle. */
256 void moveTowards

258 float maxAngle) {
259 *this = movedTowards(other, maxAngle);

271 /** Note that q<SUP>-1</SUP> = q.conj() for a unit quaternion.
272 @cite Dam99 page 13 */
273 inline Quat inverse() const {
274 return conj() / dot(*this);
275 }

277 /**
278 Quaternion multiplication (composition of rotations).
279 Note that this does not commute.

281 Quat operator*(const Quat& other) const;

283 /* (*this) * other.inverse() */
284 Quat operator/(const Quat& other) const {
285 return (*this) * other.inverse();
286 }

288 /** Is the magnitude nearly 1.0? */
289 bool isUnit(float tolerance = 1e-5) const {
290 return abs(dot(*this) - 1.0f) < tolerance;
291 }
292
293 float magnitude() const {
294 return sqrtf(dot(*this));
295 }
296
297 Quat log() const {
298 if ((x == 0) && (y == 0) && (z == 0)) {
299 if (w > 0) {
300 return Quat(0, 0, 0, ::logf(w));
301 } else if (w < 0) {
302 // Log of a negative number. Multivalued, any number of the form
303 // (PI * v, ln(-q.w))
304 return Quat((float)pi(), 0, 0, ::logf(-w));
305 } else {
306 // log of zero!
307 return Quat((float)nan(), (float)nan(), (float)nan(), (float)nan());
308 }
309 } else {
310 // Partly imaginary.
311 float imagLen = sqrtf(x * x + y * y + z * z);
312 float len = sqrtf(imagLen * imagLen + w * w);
313 float theta = atan2f(imagLen, (float)w);
314 float t = theta / imagLen;
315 return Quat(t * x, t * y, t * z, ::logf(len));
316 }
317 }

319 /** exp q = [sin(A) * v, cos(A)] where q = [Av, 0].
320 Only defined for pure-vector quaternions */
321 inline Quat exp() const {
322 debugAssertM(w == 0, "exp only defined for vector quaternions");
323 Vector3 u(x, y, z);
324 float A = u.magnitude();
325 Vector3 v = u / A;
326 return Quat(sinf(A) * v, cosf(A));
327 }

330 /**
331 Raise this quaternion to a power. For a rotation, this is
332 the effect of rotating x times as much as the original
333 quaterion.
334
335 Note that q.pow(a).pow(b) == q.pow(a + b)
336 @cite Dam98 pg 21
337 */
338 inline Quat pow(float r) const {
339 return (log() * r).exp();
340 }

343 void unitize() {
344 *this *= rsq(dot(*this));

351 Quat toUnit() const {
352 Quat copyOfThis = *this;
353 copyOfThis.unitize();
354 return copyOfThis;

358 The linear algebra 2-norm, sqrt(q dot q). This matches
359 the value used in Dam's 1998 tech report but differs from the
360 n(q) value used in Eberly's 1999 paper, which is the square of the
361 norm.
362 */

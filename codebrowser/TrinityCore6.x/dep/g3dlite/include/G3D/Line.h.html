12#ifndef G3D_LINE_H

15#include "G3D/platform.h"
16#include "G3D/Vector3.h"
17
18namespace G3D {
19
20class Plane;
21
22/**
23 An infinite 3D line.
24 */
25class Line {
26protected:
27
28 Vector3 _point;
29 Vector3 _direction;
30
31 Line(const Vector3& point, const Vector3& direction) {
32 _point = point;
33 _direction = direction.direction();
34 }
35
36public:
37
38 /** Undefined (provided for creating Array<Line> only) */
39 inline Line() {}
40
41 Line(class BinaryInput& b);
42
43 void serialize(class BinaryOutput& b) const;
44
45 void deserialize(class BinaryInput& b);
46
47 virtual ~Line() {}
48
49 /**
50 Constructs a line from two (not equal) points.
51 */
52 static Line fromTwoPoints(const Vector3 &point1, const Vector3 &point2) {
53 return Line(point1, point2 - point1);
54 }
55
56 /**
57 Creates a line from a point and a (nonzero) direction.
58 */
59 static Line fromPointAndDirection(const Vector3& point, const Vector3& direction) {
60 return Line(point, direction);
61 }
62
63 /**
64 Returns the closest point on the line to point.
65 */
66 Vector3 closestPoint(const Vector3& pt) const;
67
68 /**
69 Returns the distance between point and the line
70 */
71 double distance(const Vector3& point) const {
72 return (closestPoint(point) - point).magnitude();
73 }
74
75 /** Returns a point on the line */
76 Vector3 point() const;
77
78 /** Returns the direction (or negative direction) of the line */
79 Vector3 direction() const;
80
81 /**
82 Returns the point where the line and plane intersect. If there
83 is no intersection, returns a point at infinity.
84 */
85 Vector3 intersection(const Plane &plane) const;
86
87
88 /** Finds the closest point to the two lines.
89
90 @param minDist Returns the minimum distance between the lines.
91
92 @cite http://objectmix.com/graphics/133793-coordinates-closest-points-pair-skew-lines.html
93 */
94 Vector3 closestPoint(const Line& B, float& minDist) const;
95
96 inline Vector3 closestPoint(const Line& B) const {
97 float m;
98 return closestPoint(B, m);
99 }
100};
101
102};// namespace
103
104
105#endif

16/**
17 \brief Coordinate frame expressed in Euler angles.
18 Unlike a G3D::Quat, UprightFrame always keeps the reference frame from rolling about its own z axis.
19 Particularly useful for cameras.
20
21 \sa G3D::CoordinateFrame, G3D::Matrix4, G3D::PhysicsFrame, G3D::UprightSpline, G3D::UprightSplineManipulator
22 */
23class UprightFrame {
24public:
25
26 Vector3 translation;
27
28 /** -pi/2 < pitch < pi/2 in radians about the X-axis */
29 float pitch;
30
31 /** In radians about the Y-axis */
32 float yaw;
33
34 UprightFrame(const Vector3& t = Vector3::zero(), float p = 0, float y = 0)
35 : translation(t), pitch(p), yaw(y) {}
36
37 UprightFrame(const CoordinateFrame& cframe);
38
39 /** Constructs an UprightFrame from an Any object.
40
41 The Any format for UprightFrame is:
42
43 pitch = ##,
44 translation = Vector3(),
45 yaw = ##
46 */
47 explicit UprightFrame(const Any& any);
48
49 Any toAny() const;
50
51 UprightFrame& operator=(const Any& any);
52
53 /** Supports implicit cast to CoordinateFrame */
54 operator CoordinateFrame() const {
55 return toCoordinateFrame();
56 }
57
58 CoordinateFrame toCoordinateFrame() const;
59
60 /** Required for use with spline */
61 UprightFrame operator+(const UprightFrame& other) const;
62
63 /** Required for use with spline */
64 UprightFrame operator*(const float k) const;
65
66 /**
67 Unwraps the yaw values in the elements of the array such that
68 they still represent the same angles but strictly increase/decrease
69 without wrapping about zero. For use with Spline<UprightFrame>
70 */
71 static void unwrapYaw(UprightFrame* a, int N);
72
73 void serialize(class BinaryOutput& b) const;
74 void deserialize(class BinaryInput& b);
75};

77/**
78 \brief Shortest-path linear velocity spline for camera positions. Always keeps the camera from rolling.
79
80 \sa G3D::UprightSplineManipulator, G3D::UprightFrame
81*/
82class UprightSpline : public Spline<UprightFrame> {
83protected:
84
85 virtual void ensureShortestPath(UprightFrame* A, int N) const {
86 UprightFrame::unwrapYaw(A, N);
87 }
88
89public:
90 UprightSpline();
91
92
93 /** Constructs an UprightSpline from an Any object.
94
95 The Any format for UprightSpline is:
96
97 controls = (UprightFrame, ...),
98 times = (##, ...),
99 cyclic = bool
100
101 The controls and times arrays must have the same length.
102 */
103 explicit UprightSpline(const Any& any);
104
105 virtual Any toAny(const std::string& myName) const override;
106
107 Any toAny() const;
108
109 UprightSpline& operator=(const Any& any);
110
111 void serialize(class BinaryOutput& b) const;
112 void deserialize(class BinaryInput& b);
113};

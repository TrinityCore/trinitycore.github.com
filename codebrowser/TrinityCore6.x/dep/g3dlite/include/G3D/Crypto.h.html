21/** See G3D::Crypto::md5 */
22class MD5Hash {
23private:
24
25 uint8 value[16];
26
27public:
28
29 MD5Hash() {
30 for (int i = 0; i < 16; ++i) {
31 value[i] = 0;
32 }
33 }
34
35 explicit MD5Hash(class BinaryInput& b);
36
37 /** Rotates the bytes once */
38 void rotateBytes() {
39 uint8 temp = value[0];
40 for (int i = 0; i < 15; ++i) {
41 value[i] = value[i + 1];
42 }
43 value[15] = temp;
44 }
45
46 /** Rotates by n bytes */
47 void rotateBytes(int n) {
48 uint8 temp[16];
49 System::memcpy(temp, value, 16);
50 for (int i = 0; i < 16; ++i) {
51 value[i] = value[(i + n) & 15];
52 }
53 }
54
55 uint8& operator[](int i) {
56 return value[i];
57 }
58
59 const uint8& operator[](int i) const {
60 return value[i];
61 }
62
63 bool operator==(const MD5Hash& other) const {
64 bool match = true;
65 for (int i = 0; i < 16; ++i) {
66 match = match && (other.value[i] == value[i]);
67 }
68 return match;
69 }
70
71 inline bool operator!=(const MD5Hash& other) const {
72 return !(*this == other);
73 }
74
75 void deserialize(class BinaryInput& b);
76
77 void serialize(class BinaryOutput& b) const;
78
79 static size_t hashCode(const MD5Hash& key) {
80 size_t h = 0;
81 for (int i = 0; i < 4; ++i) {
82 const int x = i * 4;
83 h ^= (((uint32)key.value[x + 0]) << 24) |
84 (((uint32)key.value[x + 1]) << 16) |
85 (((uint32)key.value[x + 2]) << 8) |
86 ((uint32)key.value[x + 3]);
87 }
88 return h;
89 }
90};

93/** Cryptography and hashing helper functions */
94class Crypto {
95public:
96
97 /**
98 Computes the CRC32 value of a byte array. CRC32 is designed to be a hash
99 function that produces different values for similar strings.
100
101 This implementation is compatible with PKZIP and GZIP.
102
103 Based on http://www.gamedev.net/reference/programming/features/crc32/
104 */
105 static uint32 crc32(const void* bytes, size_t numBytes);
106
107 /**
108 Computes the MD5 hash (message digest) of a byte stream, as defined by
109 http://www.ietf.org/rfc/rfc1321.txt.
110
111 @cite Based on implementation by L. Peter Deutsch, ghost@aladdin.com
112 */
113 static MD5Hash md5(const void* bytes, size_t numBytes);
114
115 /**
116 Returns the nth prime less than 2000 in constant time. The first prime has index
117 0 and is the number 2.
118 */
119 static int smallPrime(int n);
120
121 /** Returns 1 + the largest value that can be passed to smallPrime. */
122 static int numSmallPrimes();
123};

13#ifndef G3D_CAPSULE_H
14#define G3D_CAPSULE_H
15
16#include "G3D/platform.h"
17#include "G3D/g3dmath.h"
18#include "G3D/Vector3.h"
19
20namespace G3D {
21
22class Line;
23class AABox;
24/**
25 A shape formed by extruding a sphere along a line segment.
26 */
27class Capsule {
28private:
29 Vector3 p1;
30 Vector3 p2;
31
32 float _radius;
33public:
34
35
36 /** Uninitialized */
37 Capsule();
38 Capsule(class BinaryInput& b);
39 Capsule(const Vector3& _p1, const Vector3& _p2, float _r);
40 void serialize(class BinaryOutput& b) const;
41 void deserialize(class BinaryInput& b);
42
43 /** The line down the center of the capsule */
44 Line axis() const;
45
46 inline float radius() const {
47 return _radius;
48 }
49
50 /** Argument may be 0 or 1 */
51 inline Vector3 point(int i) const {
52 debugAssert(i == 0 || i == 1);
53 return (i == 0) ? p1 : p2;
54 }
55
56 /** Distance between the sphere centers. The total extent of the cylinder is
57 2r + h. */
58 inline float height() const {
59 return (p1 - p2).magnitude();
60 }
61
62 inline Vector3 center() const {
63 return (p1 + p2) / 2.0;
64 }
65
66 /** Get a reference frame in which the center of mass is the origin and Y is the axis of the capsule.*/
67 void getReferenceFrame(class CoordinateFrame& cframe) const;
68
69 /**
70 Returns true if the point is inside the capsule or on its surface.
71 */
72 bool contains(const Vector3& p) const;
73
74 float volume() const;
75
76 float area() const;
77
78 /** Get axis aligned bounding box */
79 void getBounds(AABox& out) const;
80
81 /** Random world space point with outward facing normal. */
82 void getRandomSurfacePoint(Vector3& P, Vector3& N) const;
83
84 /** Point selected uniformly at random over the volume. */
85 Vector3 randomInteriorPoint() const;
86};
87
88} // namespace
89
90#endif

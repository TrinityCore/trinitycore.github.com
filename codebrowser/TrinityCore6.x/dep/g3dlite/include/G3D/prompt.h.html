19/**
20 Prints a prompt to stdout and waits for user input. The return value is
21 the number of the user's choice (the first is 0, if there are no
22 choices, returns 0).
23
24 @param useGui Under Win32, use a GUI, not stdout prompt.
25 @param windowTitle The title for the prompt window
26 @param promptx The text string to prompt the user with
27 @param choice An array of strings that are the choices the user may make
28 @param numChoices The length of choice.
29
30 @cite Windows dialog interface by Max McGuire, mmcguire@ironlore.com
31 @cite Font setting code by Kurt Miller, kurt@flipcode.com
32 */
33int prompt(
34 const char* windowTitle,
35 const char* promptx,
36 const char** choice,
37 int numChoices,
38 bool useGui);
39
40/**
41 Prints a prompt and waits for user input. The return value is
42 the number of the user's choice (the first is 0, if there are no
43 choices, returns 0).
44 <P>Uses GUI under Win32, stdout prompt otherwise.
45 */
46inline int prompt(
47 const char* windowTitle,
48 const char* promptx,
49 const char** choice,
50 int numChoices) {
51
52 return prompt(windowTitle, promptx, choice, numChoices, true);
53}

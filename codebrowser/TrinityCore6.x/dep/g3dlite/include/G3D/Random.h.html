21/** Random number generator.
22
23 Threadsafe.
24
25 Useful for generating consistent random numbers across platforms
26 and when multiple threads are involved.
27
28 Uses the Fast Mersenne Twister (FMT-19937) algorithm.
29
30 On average, uniform() runs about 2x-3x faster than rand().
31
32 @cite http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/SFMT/index.html
33
34 On OS X, Random is about 10x faster than drand48() (which is
35 threadsafe) and 4x faster than rand() (which is not threadsafe).
36
37 \sa Noise
38 */
39class Random {
40protected:
41
42 /** Constants (important for the algorithm; do not modify) */
43 enum {
44 N = 624,
45 M = 397,
46 R = 31,
47 U = 11,
48 S = 7,
49 T = 15,
50 L = 18,
51 A = 0x9908B0DF,
52 B = 0x9D2C5680,
53 C = 0xEFC60000};
54
55 /**
56 Prevents multiple overlapping calls to generate().
57 */
58 Spinlock lock;
59
60 /** State vector (these are the next N values that will be returned) */
61 uint32* state;
62
63 /** Index into state */
64 int index;
65
66 bool m_threadsafe;
67
68 /** Generate the next N ints, and store them for readback later.
69 Called from bits() */
70 virtual void generate();
71
72 /** For subclasses. The void* parameter is just to distinguish this from the
73 public constructor.*/
74 Random(void*);
75
76
77private:
78
79 Random& operator=(const Random&) {
80 alwaysAssertM(false,
81 "There is no copy constructor or assignment operator for Random because you "
82 "probably didn't actually want to copy the state--it would "
83 "be slow and duplicate the state of a pseudo-random sequence. Maybe you could "
84 "provide arguments to a member variable in the constructor, "
85 "or pass the Random by reference?");
86 return *this;
87 }
88
89 Random(const Random& r) {
90 *this = r;
91 }
92
93public:
94
95 /** \param threadsafe Set to false if you know that this random
96 will only be used on a single thread. This eliminates the
97 lock and improves performance on some platforms.
98 */
99 Random(uint32 seed = 0xF018A4D2, bool threadsafe = true);
100
101 virtual ~Random();
102
103 virtual void reset(uint32 seed = 0xF018A4D2, bool threadsafe = true);
104
105 /** Each bit is random. Subclasses can choose to override just
106 this method and the other methods will all work automatically. */
107 virtual uint32 bits();
108
109 /** Uniform random integer on the range [min, max] */
110 virtual int integer(int min, int max);
111
112 /** Uniform random float on the range [min, max] */
113 virtual inline float uniform(float low, float high) {
114 // We could compute the ratio in double precision here for
115 // about 1.5x slower performance and slightly better
116 // precision.
117 return low + (high - low) * ((float)bits() / (float)0xFFFFFFFFUL);
118 }
119
120 /** Uniform random float on the range [0, 1] */
121 virtual inline float uniform() {
122 // We could compute the ratio in double precision here for
123 // about 1.5x slower performance and slightly better
124 // precision.
125 const float norm = 1.0f / (float)0xFFFFFFFFUL;
126 return (float)bits() * norm;
127 }
128
129 /** Normally distributed reals. */
130 virtual float gaussian(float mean, float stdev);
131
132 /** Returns 3D unit vectors distributed according to
133 a cosine distribution about the positive z-axis. */
134 virtual void cosHemi(float& x, float& y, float& z);
135
136 /** Returns 3D unit vectors distributed according to
137 a cosine distribution about the z-axis. */
138 virtual void cosSphere(float& x, float& y, float& z);
139
140 /** Returns 3D unit vectors distributed according to a cosine
141 power distribution (\f$ \cos^k \theta \f$) about
142 the z-axis. */
143 virtual void cosPowHemi(const float k, float& x, float& y, float& z);
144
145 /** Returns 3D unit vectors uniformly distributed on the
146 hemisphere about the z-axis. */
147 virtual void hemi(float& x, float& y, float& z);
148
149 /** Returns 3D unit vectors uniformly distributed on the sphere */
150 virtual void sphere(float& x, float& y, float& z);
151
152 /**
153 A shared instance for when the performance and features but not
154 consistency of the class are desired. It is slightly (10%)
155 faster to use a distinct instance than to use the common one.
156
157 Threadsafe.
158 */
159 static Random& common();
160};

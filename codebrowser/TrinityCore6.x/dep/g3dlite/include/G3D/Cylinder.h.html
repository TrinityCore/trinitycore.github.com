13#ifndef G3D_Cylinder_H
14#define G3D_Cylinder_H
15
16#include "G3D/platform.h"
17#include "G3D/g3dmath.h"
18#include "G3D/Vector3.h"
19
20namespace G3D {
21
22class Line;
23class AABox;
24/**
25 Right cylinder
26 */
27class Cylinder {
28private:
29 Vector3 p1;
30 Vector3 p2;
31
32 float mRadius;
33
34public:
35
36 /** Uninitialized */
37 Cylinder();
38 Cylinder(class BinaryInput& b);
39 Cylinder(const Vector3& _p1, const Vector3& _p2, float _r);
40 void serialize(class BinaryOutput& b) const;
41 void deserialize(class BinaryInput& b);
42
43 /** The line down the center of the Cylinder */
44 Line axis() const;
45
46 /**
47 A reference frame in which the center of mass is at the origin and
48 the Y-axis is the cylinder's axis. If the cylinder is transformed, this reference frame
49 may freely rotate around its axis.*/
50 void getReferenceFrame(class CoordinateFrame& cframe) const;
51
52 /** Returns point 0 or 1 */
53 inline const Vector3& point(int i) const {
54 debugAssert(i >= 0 && i <= 1);
55 return (i == 0) ? p1 : p2;
56 }
57
58 /**
59 Returns true if the point is inside the Cylinder or on its surface.
60 */
61 bool contains(const Vector3& p) const;
62
63 float area() const;
64
65 float volume() const;
66
67 float radius() const;
68
69 /** Center of mass */
70 inline Vector3 center() const {
71 return (p1 + p2) / 2.0f;
72 }
73
74 inline float height() const {
75 return (p1 - p2).magnitude();
76 }
77
78 /**
79 Get close axis aligned bounding box.
80 With vertical world orientation, the top and bottom might not be very tight. */
81 void getBounds(AABox& out) const;
82
83 /** Random world space point with outward facing normal. */
84 void getRandomSurfacePoint(Vector3& P, Vector3& N) const;
85
86 /** Point selected uniformly at random over the volume. */
87 Vector3 randomInteriorPoint() const;
88};
89
90} // namespace
91
92#endif

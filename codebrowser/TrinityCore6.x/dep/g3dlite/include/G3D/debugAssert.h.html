51 @def debugBreak()
52
53 Break at the current location (i.e. don't push a procedure stack frame
54 before breaking).
55 */
56
57/**
58 @def debugAssert(exp)
59 Breaks if the expression is false. If G3D_DEBUG_NOGUI is defined, prompts at
60 the console, otherwise pops up a dialog. The user may then break (debug),
61 ignore, or halt the program.
62
63 The assertion is also posted to the clipboard under Win32.
64 */
65
66/**
67 @def debugAssertM(exp, msg)
68 Breaks if the expression is false and displays a message. If G3D_DEBUG_NOGUI
69 is defined, prompts at the console, otherwise pops up a dialog. The user may
70 then break (debug), ignore, or halt the program.
71
72 The assertion is also posted to the clipboard under Win32.
73 */
74
75/**
76 @def alwaysAssertM(exp, msg)
77 Same as debugAssertM except that it asserts in release builds as well.

80namespace G3D {
81typedef bool (*AssertionHook)(
82 const char* _expression,
83 const std::string& message,
84 const char* filename,
85 int lineNumber,
86 bool useGuiPrompt);

88/**
89 Allows customization of the global function invoked when a debugAssert fails.
90 The initial value is G3D::_internal::_handleDebugAssert_. G3D will invoke
91 rawBreak if the hook returns true. If NULL, assertions are not handled.
92*/
93void setAssertionHook(AssertionHook hook);
94
95AssertionHook assertionHook();
96
97/**
98 Called by alwaysAssertM in case of failure in release mode. If returns
99 true then the program exits with -1 (you can replace this with your own
100 version that throws an exception or has other failure modes).
101 */
102void setFailureHook(AssertionHook hook);
103AssertionHook failureHook();

116#ifdef G3D_DEBUG
117
118# if defined(_MSC_VER)
119# define rawBreak() ::DebugBreak();
120# elif defined(__i386__)
121 // gcc on intel
122# define rawBreak() __asm__ __volatile__ ( "int $3" );
123# else
124 // some other gcc
125# define rawBreak() ::abort()
126# endif
127
128
129# define debugBreak() G3D::_internal::_releaseInputGrab_(); rawBreak(); G3D::_internal::_restoreInputGrab_();
130# define debugAssert(exp) debugAssertM(exp, "Debug assertion failure")
131
132 #ifdef G3D_DEBUG_NOGUI
133 #define __debugPromptShowDialog__ false
134 #else
135 #define __debugPromptShowDialog__ true
136 #endif
137
138 #define debugAssertM(exp, message) do { \
139 if (!(exp)) { \
140 G3D::_internal::_releaseInputGrab_(); \
141 if ((G3D::_internal::_debugHook != NULL) && \
142 G3D::_internal::_debugHook((const char*)(#exp), message, __FILE__, __LINE__, __debugPromptShowDialog__)) { \
143 rawBreak(); \
144 } \
145 G3D::_internal::_restoreInputGrab_(); \
146 } \
147 } while (0)
148
149 #define alwaysAssertM debugAssertM
150
151#else // Release
152 #ifdef G3D_DEBUG_NOGUI
153 #define __debugPromptShowDialog__ false
154 #else
155 #define __debugPromptShowDialog__ true
156 #endif
157
158 // In the release build, just define away assertions.
159 #define rawBreak() do {} while (0)
160 #define debugAssert(exp) do {} while (0)
161 #define debugAssertM(exp, message) do {} while (0)
162 #define debugBreak() do {} while (0)
163
164 // But keep the 'always' assertions
165 #define alwaysAssertM(exp, message) { \
166 if (!(exp)) { \
167 G3D::_internal::_releaseInputGrab_(); \
168 if ((G3D::_internal::_failureHook != NULL) && \
169 G3D::_internal::_failureHook(#exp, message, __FILE__, __LINE__, __debugPromptShowDialog__)) { \
170 ::exit(-1); \
171 } \
172 G3D::_internal::_restoreInputGrab_(); \
173 } \
174 }
175
176#endif // if debug

13#ifndef G3D_DEBUGPRINTF_H
14#define G3D_DEBUGPRINTF_H
15
16#include "G3D/platform.h"
17#include <stdio.h>
18#include <cstdarg>
19#include "G3D/format.h"
20#include <string>
21
22namespace G3D {
23
24typedef void (*ConsolePrintHook)(const std::string&);
25
26namespace _internal {
27 extern ConsolePrintHook _consolePrintHook;
28}
29
30/** Called by consolePrintf after the log and terminal have been written to.
31 Used by GConsole to intercept printing routines.*/
32void setConsolePrintHook(ConsolePrintHook h);
33
34ConsolePrintHook consolePrintHook();
35
36/**
37 Sends output to the log and to the last GConsole instantiated.
38
39 Guarantees that the output has been flushed by the time the routine
40 returns.
41 @sa G3D::logPrintf, G3D::screenPrintf
42 @return The string that was printed
43 */
44std::string __cdecl consolePrintf(const char* fmt ...) G3D_CHECK_PRINTF_ARGS;
45std::string consolePrint(const std::string&);
46
47/**
48 Under visual studio, appears in the VS debug pane.
49 On unix-based operating systems the output is sent to stderr.
50
51 Also sends output to the console (G3D::consolePrintf) if there is a consolePrintHook,
52 and log (G3D::logPrintf), and flushes before returning.
53
54 @return The string that was printed
55*/
56std::string __cdecl debugPrintf(const char* fmt ...) G3D_CHECK_PRINTF_ARGS;
57std::string debugPrint(const std::string&);
58
59} // namespace G3D

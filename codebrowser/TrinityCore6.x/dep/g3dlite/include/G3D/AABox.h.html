29/**
30 An axis-aligned box.
31 */
32class AABox {
33private:
34 friend class Intersect;
35
36 /** Optional argument placeholder */
37 static int dummy;
38
39 /** NaN if empty */
40 Point3 lo;
41
42 /** NaN if empty */
43 Point3 hi;
44
45public:
46
47 /** Creates the empty bounds, i.e., an empty set of points. */
48 AABox() : lo(fnan(), fnan(), fnan()), hi(fnan(), fnan(), fnan()) {}
49
50 /**
51 Constructs a zero-volume AABox at v.
52 */
53 explicit AABox(const Point3& v) {
54 lo = hi = v;
55 }
56
57 /** Format is one of:
58 - AABox(lowpoint, highpoint)
59 - AABox(point)
60 - AABox::empty()
61 - AABox::inf()
62 */
63 explicit AABox(const class Any& a);
64
65 Any toAny() const;
66
67 bool isEmpty() const {
68 return lo.isNaN();
69 }
70
71 /** Assumes that low is less than or equal to high along each dimension.
72 To have this automatically enforced, use
73 <code>AABox(low.min(high), low.max(high));</code>
74 */
75 AABox(const Point3& low, const Point3& high) {
76 set(low, high);
77 }
78
79 AABox operator*(float f) const {
80 if (f < 0) {
81 return AABox(hi * f, lo * f);
82 } else {
83 return AABox(lo * f, hi * f);
84 }
85 }
86
87 AABox operator/(float f) const {
88 return *this * (1.0f / f);
89 }
90
91 /** Assumes that low is less than or equal to high along each dimension.
92 */
93 inline void set(const Point3& low, const Point3& high) {
94 debugAssert(
95 (low.x <= high.x) &&
96 (low.y <= high.y) &&
97 (low.z <= high.z));
98 debugAssert(! low.isNaN() && ! high.isNaN());
99 lo = low;
100 hi = high;
101 }
102
103 /**
104 Grows to include the bounds of \a a
105 */
106 inline void merge(const AABox& a) {
107 if (isEmpty()) {
108 lo = a.lo;
109 hi = a.hi;
110 } else if (! a.isEmpty()) {
111 lo = lo.min(a.lo);
112 hi = hi.max(a.hi);
113 }
114 }
115
116 inline void merge(const Point3& a) {
117 if (isEmpty()) {
118 lo = hi = a;
119 } else {
120 lo = lo.min(a);
121 hi = hi.max(a);
122 }
123 }
124
125 void merge(const class Box& b);
126
127 void serialize(class BinaryOutput& b) const;
128
129 void deserialize(class BinaryInput& b);
130
131 inline bool isFinite() const {
132 return isEmpty() || (lo.isFinite() && hi.isFinite());
133 }
134
135 /** Returns not-a-number if empty */
136 inline const Point3& low() const {
137 return lo;
138 }
139
140 /** Returns not-a-number if empty */
141 inline const Point3& high() const {
142 return hi;
143 }

146 The largest possible finite box.
147 */
148 static const AABox& maxFinite();

150 /** A large finite box. This is smaller than FLT_MAX

152 static const AABox& large();

154 static const AABox& inf();

156 static const AABox& zero();
157
158 static const AABox& empty();

160 /**
161 Returns the centroid of the box (NaN if empty)
162 */
163 inline Point3 center() const {
164 return (lo + hi) * 0.5;
165 }
166
167 Point3 corner(int index) const;
168
169 /**
170 Distance from corner(0) to the next corner along axis a.
171 */
172 inline float extent(int a) const {
173 if (isEmpty()) {
174 return 0.0f;
175 }
176 debugAssert(a < 3);
177 return hi[a] - lo[a];
178 }
179
180
181 inline Vector3 extent() const {
182 if (isEmpty()) {
183 return Vector3::zero();
184 }
185 return hi - lo;
186 }

188
189 /**
190 Splits the box into two AABoxes along the specified axis. low contains
191 the part that was closer to negative infinity along axis, high contains
192 the other part. Either may have zero volume.
193 */
194 void split(const Vector3::Axis& axis, float location, AABox& low, AABox& high) const;

196 /**
197 Conservative culling test for up to 32 planes.
198 Returns true if there exists a <CODE>plane[p]</CODE> for
199 which the entire object is in the negative half space
200 (opposite the plane normal).
201
202 <CODE>testMask</CODE> and <CODE>childMask</CODE>
203 are used for optimizing bounding volume hierarchies.
204 The version of this method that produces childMask
205 is slower than the version without; it should only
206 be used for parent nodes.
207
208 @param cullingPlaneIndex The index of the first plane for which
209 the entire object is in the negative half-space. The function
210 exits early when one plane is found. -1 when the function
211 returns false (i.e. when no plane culls the whole object).
212
213 @param testMask If bit <I>p</I> is 0, the
214 bounding volume automatically passes the culling test for
215 <CODE>plane[p]</CODE> (i.e. it is known that the volume
216 is entirely within the positive half space). The function
217 must return false if testMask is 0 and test all planes
218 when testMask is -1 (0xFFFFFFFF).
219
220 @param childMask Test mask for the children of this volume.
221
222 */
223 bool culledBy
224 (const Array<Plane>& plane,
225 int32& cullingPlaneIndex,
226 const uint32 testMask,
227 uint32& childMask) const;
228
229 /**
230 Conservative culling test that does not produce a mask for children.
231 */
232 bool culledBy
233 (const Array<Plane>& plane,
234 int32& cullingPlaneIndex = dummy,
235 const uint32 testMask = 0xFFFFFFFF) const;

237 /** less than or equal to containment */
238 inline bool contains(const AABox& other) const {
239 return
240 (other.hi.x <= hi.x) &&
241 (other.hi.y <= hi.y) &&
242 (other.hi.z <= hi.z) &&
243 (other.lo.x >= lo.x) &&
244 (other.lo.y >= lo.y) &&
245 (other.lo.z >= lo.z);
246 }
247
248 inline bool contains(const Point3& point) const {
249 return
250 (point.x >= lo.x) &&
251 (point.y >= lo.y) &&
252 (point.z >= lo.z) &&
253 (point.x <= hi.x) &&
254 (point.y <= hi.y) &&
255 (point.z <= hi.z);
256 }
257
258 inline float area() const {
259 if (isEmpty()) { return 0; }
260 Vector3 diag = hi - lo;
261 return 2.0f * (diag.x * diag.y + diag.y * diag.z + diag.x * diag.z);
262 }
263
264 inline float volume() const {
265 if (isEmpty()) { return 0; }
266 Vector3 diag = hi - lo;
267 return diag.x * diag.y * diag.z;
268 }

274 /** Returns true if there is any overlap */
275 bool intersects(const AABox& other) const;
276
277 /** Returns true if there is any overlap.
278 @cite Jim Arvo's algorithm from Graphics Gems II*/
279 bool intersects(const Sphere& other) const;
280
281 /** Return the intersection of the two boxes */
282 AABox intersect(const AABox& other) const {
283 if (isEmpty() || other.isEmpty()) {
284 return empty();
285 }
286
287 const Point3& H = hi.min(other.hi);
288 const Point3& L = lo.max(other.lo).min(H);
289
290 if (H.x < L.x && H.y < L.y && H.z < L.z) {
291 return empty();
292 } else {
293 return AABox(L, H);
294 }
295 }
296
297 inline size_t hashCode() const {
298 return lo.hashCode() + hi.hashCode();
299 }
300
301 inline bool operator==(const AABox& b) const {
302 if (isEmpty() && b.isEmpty()) {
303 return true;
304 } else {
305 return (lo == b.lo) && (hi == b.hi);
306 }
307 }
308
309 inline bool operator!=(const AABox& b) const {
310 if (isEmpty()) {
311 return b.isEmpty();
312 } else {
313 return !((lo == b.lo) && (hi == b.hi));
314 }
315 }
316
317 inline AABox operator+(const Vector3& v) const {
318 AABox out;
319 out.lo = lo + v;
320 out.hi = hi + v;
321 return out;
322 }
323
324 inline AABox operator-(const Vector3& v) const {
325 AABox out;
326 out.lo = lo - v;
327 out.hi = hi - v;
328 return out;
329 }

331 void getBounds(AABox& out) const {

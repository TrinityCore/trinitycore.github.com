20/** Common implementation code for all G3D::Spline template parameters */
21class SplineBase {
22public:
23
24 /** Times at which control points occur. Must have the same
25 number of elements as Spline::control. */
26 Array<float> time;
27
28 /** If CYCLIC, then the control points will be assumed to wrap around.
29 If LINEAR, then the tangents at the ends of the spline
30 point to the final control points. If CONSTANT, the end control
31 points will be treated as multiple contol points (so the value remains constant at the ends)
32 */
33 SplineExtrapolationMode extrapolationMode;
34
35 /** For a cyclic spline, this is the time elapsed between the last
36 control point and the first. If less than or equal to zero this is
37 assumed to be:
38
39 (time[0] - time[1] + .
40 time[time.size() - 1] - time[time.size() - 2]) / 2.
41 */
42 float finalInterval;
43
44 SplineInterpolationMode interpolationMode;
45
46 SplineBase() :
47 extrapolationMode(SplineExtrapolationMode::CYCLIC),
48 finalInterval(-1),
49 interpolationMode(SplineInterpolationMode::CUBIC) {}
50
51 virtual ~SplineBase() {}
52
53 /** See specification for Spline::finalInterval; this handles the
54 non-positive case. Returns 0 if not cyclic. */
55 float getFinalInterval() const;
56
57 /** Returns the amount of time covered by this spline in one
58 period. For a cyclic spline, this contains the final
59 interval.*/
60 float duration() const;
61
62 /** Computes the derivative spline basis from the control point version. */
63 static Matrix4 computeBasis();
64
65protected:
66
67 /** Assumes that t0 <= s < tn. called by computeIndex. */
68 void computeIndexInBounds(float s, int& i, float& u) const;
69
70public:
71
72 /**
73 Given a time @a s, finds @a i and 0 <= @a u < 1 such that
74 @a s = time[@a i] * @a u + time[@a i + 1] * (1 - @a u). Note that
75 @a i may be outside the bounds of the time and control arrays;
76 use getControl to handle wraparound and extrapolation issues.
77
78 This function takes expected O(1) time for control points with
79 uniform time sampled control points or for uniformly
80 distributed random time samples, but may take O( log time.size() ) time
81 in the worst case.
82
83 Called from evaluate().
84 */
85 void computeIndex(float s, int& i, float& u) const;
86};

89/**
90 Smooth parameteric curve implemented using a piecewise 3rd-order
91 Catmull-Rom spline curve. The spline is considered infinite and may
92 either continue linearly from the specified control points or cycle
93 through them. Control points are spaced uniformly in time at unit
94 intervals by default, but irregular spacing may be explicitly
95 specified.
96
97 The dimension of the spline can be set by varying the Control
98 template parameter. For a 1D function, use Spline<float>. For a
99 curve in the plane, Spline<Vector2>. Note that <i>any</i> template
100 parameter that supports operator+(Control) and operator*(float) can
101 be used; you can make splines out of G3D::Vector4, G3D::Matrix3, or
102 your own classes.
103
104 To provide shortest-path interpolation, subclass G3D::Spline and
105 override ensureShortestPath(). To provide normalization of
106 interpolated points (e.g., projecting Quats onto the unit
107 hypersphere) override correct().
108
109 See Real Time Rendering, 2nd edition, ch 12 for a general discussion
110 of splines and their properties.
111
112 \sa G3D::UprightSpline
113 */
114template<typename Control>
115class Spline : public SplineBase {
116protected:
117 /** The additive identity control point. */
118 Control zero;
119
120public:
121
122 /** Control points. Must have the same number of elements as
123 Spline::time.*/
124 Array<Control> control;
125
126 Spline() {
127 static Control x;
128 // Hide the fact from C++ that we are using an
129 // uninitialized variable here by pointer arithmetic.
130 // This is ok because any type that is a legal control
131 // point also supports multiplication by float.
132 zero = *(&x) * 0.0f;
133 }
134
135 /** Appends a control point at a specific time that must be
136 greater than that of the previous point. */
137 void append(float t, const Control& c) {
138 debugAssertM((time.size() == 0) || (t > time.last()),
139 "Control points must have monotonically increasing times.");
140 time.append(t);
141 control.append(c);
142 debugAssert(control.size() == time.size());
143 }
144
145
146 /** Appends control point spaced in time based on the previous
147 control point, or spaced at unit intervals if this is the
148 first control point. */
149 void append(const Control& c) {
150 switch (time.size()) {
151 case 0:
152 append(0, c);
153 break;
154
155 case 1:
156 append(time[0] + 1, c);
157 break;
158
159 default:
160 append(2 * time[time.size() - 1] - time[time.size() - 2], c);
161 }
162 debugAssert(control.size() == time.size());
163 }
164
165 /** Erases all control points and times, but retains the state of
166 cyclic and finalInterval.
167 */
168 void clear() {
169 control.clear();
170 time.clear();
171 }
172
173
174 /** Number of control points */
175 int size() const {
176 debugAssert(time.size() == control.size());
177 return control.size();
178 }
179
180
181 /** Returns the requested control point and time sample based on
182 array index. If the array index is out of bounds, wraps (for
183 a cyclic spline) or linearly extrapolates (for a non-cyclic
184 spline), assuming time intervals follow the first or last
185 sample recorded.
186
187 Calls correct() on the control point if it was extrapolated.
188
189 Returns 0 if there are no control points.
190
191 @sa Spline::control and Spline::time for the underlying
192 control point array; Spline::computeIndex to find the index
193 given a time.
194 */
195 void getControl(int i, float& t, Control& c) const {
196 int N = control.size();
197 if (N == 0) {
198 c = zero;
199 t = 0;
200 } else if (extrapolationMode == SplineExtrapolationMode::CYCLIC) {
201 c = control[iWrap(i, N)];
202
203 if (i < 0) {
204 // Wrapped around bottom
205
206 // Number of times we wrapped around the cyclic array
207 int wraps = (N + 1 - i) / N;
208 int j = (i + wraps * N) % N;
209 t = time[j] - wraps * duration();
210
211 } else if (i < N) {
212
213 t = time[i];
214
215 } else {
216 // Wrapped around top
217
218 // Number of times we wrapped around the cyclic array
219 int wraps = i / N;
220 int j = i % N;
221 t = time[j] + wraps * duration();
222 }
223
224 } else if (i < 0) {
225 // Are there enough points to extrapolate?
226 if (N >= 2) {
227 // Step away from control point 0
228 float dt = time[1] - time[0];
229
230 if (extrapolationMode == SplineExtrapolationMode::LINEAR) {
231 // Extrapolate (note; i is negative)
232 c = control[1] * float(i) + control[0] * float(1 - i);
233 correct(c);
234 } else if (extrapolationMode == SplineExtrapolationMode::CLAMP){
235 // Return the first, clamping
236 c = control[0];
237 } else {
238 alwaysAssertM(false, "Invalid extrapolation mode");
239 }
240 t = dt * i + time[0];
241
242 } else {
243 // Just clamp
244 c = control[0];
245
246 // Only 1 time; assume 1s intervals
247 t = time[0] + i;
248 }
249
250 } else if (i >= N) {
251 if (N >= 2) {
252 float dt = time[N - 1] - time[N - 2];
253
254 if (extrapolationMode == SplineExtrapolationMode::LINEAR) {
255 // Extrapolate (note; i is negative)
256 c = control[N - 1] * float(i - N + 2) + control[N - 2] * -float(i - N + 1);
257 correct(c);
258 } else if (extrapolationMode == SplineExtrapolationMode::CLAMP){
259 // Return the last, clamping
260 c = control.last();
261 } else {
262 alwaysAssertM(false, "Invalid extrapolation mode");
263 }
264 // Extrapolate
265 t = time[N - 1] + dt * (i - N + 1);
266
267 } else {
268 // Return the last, clamping
269 c = control.last();
270 // Only 1 time; assume 1s intervals
271 t = time[0] + i;
272 }
273 } else {
274 // In bounds
275 c = control[i];
276 t = time[i];
277 }
278 }

280protected:
281
282 /** Returns a series of N control points and times, fixing
283 boundary issues. The indices may be assumed to be treated
284 cyclically. */
285 void getControls(int i, float* T, Control* A, int N) const {
286 for (int j = 0; j < N; ++j) {
287 getControl(i + j, T[j], A[j]);
288 }
289 ensureShortestPath(A, N);
290 }

292 /**
293 Mutates the array of N control points that begins at \a A. It is useful to override this
294 method by one that wraps the values if they are angles or quaternions
295 for which "shortest path" interpolation is significant.
296 */
297 virtual void ensureShortestPath(Control* A, int N) const { (void)A; (void) N;}

299 /** Normalize or otherwise adjust this interpolated Control. */

302 /** Does not invoke verifyDone() on the propertyTable because subclasses may have more properties */
303 virtual void init(AnyTableReader& propertyTable) {
304 propertyTable.getIfPresent("extrapolationMode", extrapolationMode);
305 propertyTable.getIfPresent("interpolationMode", interpolationMode);
306 propertyTable.getIfPresent("finalInterval", finalInterval);
307
308 const bool hasTime = propertyTable.getIfPresent("time", time);
309
310 if (propertyTable.getIfPresent("control", control)) {
311 if (! hasTime) {
312 // Assign unit times
313 time.resize(control.size());
314 for (int i = 0; i < time.size(); ++i) {
315 time[i] = float(i);
316 }
317 } // if has time
318 } // if has control
319 } // init

344 /**
345 Return the position at time s. The spline is defined outside
346 of the time samples by extrapolation or cycling.
347 */
348 Control evaluate(float s) const {
349 debugAssertM(control.size() == time.size(), "Corrupt spline: wrong number of control points.");
350
351 /*
352 @cite http://www.gamedev.net/reference/articles/article1497.asp
353 Derivation of basis matrix follows.
354
355 Given control points with positions p[i] at times t[i], 0 <= i <= 3, find the position
356 at time t[1] <= s <= t[2].
357
358 Let u = s - t[0]
359 Let U = [u^0 u^1 u^2 u^3] = [1 u u^2 u^3]
360 Let dt0 = t[0] - t[-1]
361 Let dt1 = t[1] - t[0]
362 Let dt2 = t[2] - t[1]
363 */
364
365 // Index of the first control point (i.e., the u = 0 point)
366 int i = 0;
367 // Fractional part of the time
368 float u = 0;
369
370 computeIndex(s, i, u);
371
372 Control p[4];
373 float t[4];
374 getControls(i - 1, t, p, 4);
375
376 const Control& p0 = p[0];
377 const Control& p1 = p[1];
378 const Control& p2 = p[2];
379 const Control& p3 = p[3];
380
381 // Compute the weighted sum of the neighboring control points.
382 Control sum;
383
384 if (interpolationMode == SplineInterpolationMode::LINEAR) {
385 const float a = (s - t[1]) / (t[2] - t[1]);
386 sum = p1 * (1.0f - a) + p2 * a;
387 correct(sum);
388 return sum;
389 }
390
391 float dt0 = t[1] - t[0];
392 float dt1 = t[2] - t[1];
393 float dt2 = t[3] - t[2];
394
395 static const Matrix4 basis = computeBasis();
396
397 // Powers of u
398 Vector4 uvec((float)(u*u*u), (float)(u*u), (float)u, 1.0f);
399
400 // Compute the weights on each of the control points.
401 const Vector4& weights = uvec * basis;
402
403
404 // The factor of 1/2 from averaging two time intervals is
405 // already factored into the basis
406
407 // tan1 = (dp0 / dt0 + dp1 / dt1) * ((dt0 + dt1) * 0.5);
408 // The last term normalizes for unequal time intervals
409 float x = (dt0 + dt1) * 0.5f;
410 float n0 = x / dt0;
411 float n1 = x / dt1;
412 float n2 = x / dt2;
413
414 const Control& dp0 = p1 + (p0*-1.0f);
415 const Control& dp1 = p2 + (p1*-1.0f);
416 const Control& dp2 = p3 + (p2*-1.0f);
417
418 const Control& dp1n1 = dp1 * n1;
419 const Control& tan1 = dp0 * n0 + dp1n1;
420 const Control& tan2 = dp1n1 + dp2 * n2;
421
422 sum =
423 tan1 * weights[0] +
424 p1 * weights[1] +
425 p2 * weights[2] +
426 tan2 * weights[3];
427
428
429 correct(sum);
430 return sum;
431 }

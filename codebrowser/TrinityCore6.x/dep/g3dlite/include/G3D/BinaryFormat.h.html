31/**
32 Some values like float16 and int128 have no current CPU data structure that implements them but are useful
33 for file formats and for GPUs.
34
35 CHUNK_BINFMT data follows the protocol.
36 */
37// Must be packed int 16 bits for the chunk reader
38// We can't name these just "INT8" etc. because some libraries #define names like that
39enum BinaryFormat {
40 FIRST_BINFMT = 1000,
41
42 BOOL8_BINFMT,
43 UINT8_BINFMT, INT8_BINFMT, UINT16_BINFMT, INT16_BINFMT, UINT32_BINFMT, INT32_BINFMT, UINT64_BINFMT, INT64_BINFMT, UINT128_BINFMT, INT128_BINFMT,
44 FLOAT16_BINFMT, FLOAT32_BINFMT, FLOAT64_BINFMT,
45 VECTOR2_BINFMT, VECTOR2INT16_BINFMT,
46 VECTOR3_BINFMT, VECTOR3INT16_BINFMT,
47 VECTOR4_BINFMT, VECTOR4INT16_BINFMT,
48 COLOR3_BINFMT, COLOR3UINT8_BINFMT, COLOR3INT16_BINFMT,
49 COLOR4_BINFMT, COLOR4UINT8_BINFMT, COLOR4INT16_BINFMT,
50 STRING_BINFMT, STRINGEVEN_BINFMT, STRING8_BINFMT, STRING16_BINFMT, STRING32_BINFMT,
51
52 CHUNK_BINFMT,
53
54 CUSTOM_BINFMT,
55
56 LAST_BINFMT
57};

61/** A macro that maps G3D types to format constants.
62 (e.g. binaryFormatOf(Vector3) == VECTOR3_BINFMT).
63*/
64// This implementation is designed to meet the following constraints:
65// 1. Work around the many MSVC++ partial template bugs
66// 2. Work for primitive types (e.g. int)
67#define binaryFormatOf(T) (G3D::_internal::_BinaryFormat<T>::x())
68
69namespace G3D {
70namespace _internal {
71
72
73template<class T> class _BinaryFormat {
74public:
75 static BinaryFormat x() {
76 return CUSTOM_BINFMT;
77 }
78};
79}}

82/**
83 Macro to declare the underlying format (as will be returned by glFormatOf)
84 of a type. For example,
85
86 <PRE>
87 DECLARE_BINARYFORMATOF(Vector4, VECTOR4_BINFMT)
88 </PRE>
89
90 Use this so you can make vertex arrays of your own classes and not just
91 the standard ones.
92 */
93#define DECLARE_BINARYFORMATOF(CType, EnumType) \
94namespace G3D { \
95 namespace _internal { \
96 template<> class _BinaryFormat<CType> { \
97 public: \
98 static BinaryFormat x() { \
99 return EnumType; \
100 } \
101 }; \
102 } \
103}

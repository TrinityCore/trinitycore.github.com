150/**
151 \brief A simple tokenizer for parsing text files.
152
153 TextInput handles a
154 superset of C++,Java, Matlab, and Bash code text including single
155 line comments, block comments, quoted strings with escape sequences,
156 and operators. TextInput recognizes several categories of tokens,
157 which are separated by white space, quotation marks, or the end of a
158 recognized operator:
159
160 <ul>
161 <li><CODE>Token::SINGLE_QUOTED_TYPE</CODE> string of characters surrounded by single quotes, e.g., 'x', '\\0', 'foo'.
162 <li><CODE>Token::DOUBLE_QUOTED_TYPE</CODE> string of characters surrounded by double quotes, e.g., "x", "abc\txyz", "b o b".
163 <li><CODE>Token::SYMBOL_TYPE</CODE> legal C++ operators, keywords, and identifiers. e.g., >=, Foo, _X, class, {
164 <li><CODE>Token::INTEGER_TYPE</CODE> numbers without decimal places or exponential notation. e.g., 10, 0x17F, 32, 0, -155
165 <li><CODE>Token::FLOATING_POINT_TYPE</CODE> numbers with decimal places or exponential notation. e.g., 1e3, -1.2, .4, 0.5
166 <li><CODE>Token::BOOLEAN_TYPE</CODE> special symbols like "true" and "false"; the exact details can be configured in TextInput::Settings
167 <li><CODE>Token::LINE_COMMENT_TYPE</CODE> (disabled by default); generated for line comments as specified by TextInput::Settings
168 <li><CODE>Token::BLOCK_COMMENT_TYPE</CODE> (disabled by default); generated for c-style block comments as specified by TextInput::Settings
169 <li><CODE>Token::NEWLINE_TYPE</CODE> (disabled by default); generated for any of "\\r", "\\n" or "\\r\\n"
170 </ul>
171
172 <P>The special ".." and "..." tokens are always recognized in
173 addition to normal C++ operators. Additional tokens can be made
174 available by changing the Settings.
175
176 Negative numbers are handled specially because of the ambiguity between unary minus and negative numbers--
177 see the note on TextInput::read.
178
179 TextInput does not have helper functions for types with non-obvious
180 formatting, or helpers that would be redundant. Use the serialize
181 methods instead for parsing specific types like int, Vector3, and
182 Color3.
183
184 Inside quoted strings escape sequences are converted. Thus the
185 string token for ["a\\nb"] is 'a', followed by a newline, followed by
186 'b'. Outside of quoted strings, escape sequences are not converted,
187 so the token sequence for [a\\nb] is symbol 'a', symbol '\\', symbol
188 'nb' (this matches what a C++ parser would do). The exception is
189 that a specified TextInput::Settings::otherCommentCharacter preceeded
190 by a backslash is assumed to be an escaped comment character and is
191 returned as a symbol token instead of being parsed as a comment
192 (this is what a LaTex or VRML parser would do).
193
194 <B>Examples</B>
195
196 \code
197 TextInput ti(TextInput::FROM_STRING, "name = \"Max\", height = 6");
198
199 Token t;
200
201 t = ti.read();
202 debugAssert(t.type == Token::SYMBOL);
203 debugAssert(t.sval == "name");
204
205 ti.read();
206 debugAssert(t.type == Token::SYMBOL);
207 debugAssert(t.sval == "=");
208
209 std::string name = ti.read().sval;
210 ti.read();
211 \endcode
212
213 \code
214 TextInput ti(TextInput::FROM_STRING, "name = \"Max\", height = 6");
215 ti.readSymbols("name", "=");
216 std::string name = ti.readString();
217 ti.readSymbols(",", "height", "=");
218 double height = ti. readNumber();
219 \endcode
220
221 Assumes that the file is not modified once opened.
222 */
223class TextInput {
224public:
225 /** Includes MSVC specials parsing */
226 static double parseNumber(const std::string& _string);

229 static bool parseBoolean(const std::string& _string);

235 /** If true, C-style slash-star marks a multi-line comment.
236
237 See generateCommentTokens for rules on how this is applied.
238
239 Default is true.
240 */
241 bool cppBlockComments;
242
243 /** If true, // begins a single line comment.
244
245 See generateCommentTokens for rules on how this is applied.
246
247 Default is true.
248 */
249 bool cppLineComments;
250
251 /** If true, otherCommentCharacter and otherCommentCharacter2
252 are used to begin single line comments in the same way
253 cppLineComments is.
254
255 See generateCommentTokens for rules on how this is applied.
256
257 Default is true.
258 */
259 bool otherLineComments;
260
261 /** If true, \\r, \\n, \\t, \\0, \\\\ and other escape sequences inside
262 strings are converted to the equivalent C++ escaped character.
263 If false, backslashes are treated literally. It is convenient to
264 set to false if reading Windows paths, for example, like
265 c:\\foo\\bar.
266
267 Default is true.
268 */
269 bool escapeSequencesInStrings;
270
271 /** If not '\\0', specifies a character that begins single line
272 comments ('#' and '%' are popular choices). This is independent
273 of the cppLineComments flag. If the character appears in text with
274 a backslash in front of it, it is considered escaped and is not
275 treated as a comment character.
276
277 Default is '\\0'.
278 */
279 char otherCommentCharacter;
280
281 /** Another (optional) 1-comment character. Useful for files that
282 support multiple comment syntaxes. Default is '\\0'.
283 */
284 char otherCommentCharacter2;
285
286 /** If true, comments enabled by cppBlockComments, cppLineComments
287 and otherLineComments will generate their respective tokens.
288 If false, the same settings will enable parsing and ignoring
289 comments
290
291 Default is false.
292 */
293 bool generateCommentTokens;
294
295 /** If true, newlines will generate tokens.
296 If false, newlines will be discarded as whitespace when parsed
297 outside of other tokens.
298
299 Default is false.
300 */
301 bool generateNewlineTokens;
302
303 /** If true, "-1" parses as the number -1 instead of the
304 symbol "-" followed by the number 1. Default is true.*/
305 bool signedNumbers;
306
307 /** If true, strings can be marked with single quotes (e.g.,
308 'aaa'). If false, the quote character is parsed as a
309 symbol. Default is true. Backquote (`) is always parsed

311 bool singleQuotedStrings;
312
313 /** The character to use as a single quote. Defaults to "'" (backquote),
314 occasionally useful to set to "`" (forward quote) or to "," (comma) for
315 reading CSV files. */
316 char singleQuoteCharacter;
317
318 /** If set to a non-empty string, that string will be used in
319 place of the real file name (or in place of a pseudonym
320 constructed from the buffer if given FROM_STRING) in
321 tokens and exceptions.
322
323 Default is empty.
324 */
325 std::string sourceFileName;
326
327
328 /** Added to the line number reported by peekLineNumber and in
329 exceptions. Useful for concatenating files that are
330 parsed separately. Default is zero. */
331 int startingLineNumberOffset;
332
333 /**
334 Parse "-1.#IND00" as the floating point number returned by
335 G3D::nan(), "-1.#INF00" as - G3D::inf(), and "1.#INF00" as G3D::inf().
336
337 Note that the C99 standard specifies that a variety of formats
338 like "nan" are to be used; these are supported by
339 G3D::TextInput::Settings::simpleFloatSpecials.
340
341 An alternative to specifying msvcFloatSpecials is to read numbers as:
342 \htmlonly
343 <pre>
344 Token x = t.read();
345 Token y = t.peek();
346 if ((x.string() == "-1.") &&
347 (y.string() == "#INF00") &&
348 (y.character() == x.character() + 3) &&
349 (y.line() == x.line()) {
350 t.read();
351 return nan();
352 }
353 // ... similar cases for inf
354 </pre>
355 \endhtmlonly
356
357 If the single-comment character was #, the floating point
358 special format overrides the comment and will be parsed
359 instead.
360
361 If signedNumbers is false msvcFloatSpecials will not be parsed.
362
363 Default is true. */
364 bool msvcFloatSpecials;
365
366 /** Parses "+inf', "-inf", "inf", "nan" as floats instead of symbols.
367 Defaults to true.*/
368 bool simpleFloatSpecials;
369
370 /**
371 Parse the following set of useful proof symbols:
372
373 =>
374 ::>
375 <::
376 :>
377 <:
378 |-
379 ::=
380 :=
381 <-
382
383 Default is false.
384 */
385 bool proofSymbols;
386
387 /**
388 When parsing booleans and msvcFloatSpecials, is case significant?
389 Default is {true}
390 */
391 bool caseSensitive;
392
393 /** All symbols that will become the 'true' boolean token. See also caseSensitive.
394 Clear this value to disable parsing of true booleans.
395
396 Default is {true}.
397 */
398 Set<std::string> trueSymbols;
399
400 /** See trueSymbols. Default is {false}*/
401 Set<std::string> falseSymbols;

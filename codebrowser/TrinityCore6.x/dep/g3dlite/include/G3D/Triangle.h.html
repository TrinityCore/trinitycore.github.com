28/**
29 A generic triangle representation. This should not be used
30 as the underlying triangle for creating models; it is intended
31 for providing fast property queries but requires a lot of
32 storage and is mostly immutable.
33 */
34class Triangle {
35private:
36 friend class CollisionDetection;
37 friend class Ray;
38
39 Vector3 _vertex[3];
40
41 /** edgeDirection[i] is the normalized vector v[i+1] - v[i] */
42 Vector3 edgeDirection[3];
43 float edgeMagnitude[3];
44 Plane _plane;
45 Vector3::Axis _primaryAxis;
46
47 /** vertex[1] - vertex[0] */
48 Vector3 _edge01;
49
50 /** vertex[2] - vertex[0] */
51 Vector3 _edge02;
52
53 float _area;
54
55 void init(const Vector3& v0, const Vector3& v1, const Vector3& v2);
56
57public:
58
59 Triangle(class BinaryInput& b);
60 void serialize(class BinaryOutput& b);
61 void deserialize(class BinaryInput& b);
62
63 Triangle();
64
65 Triangle(const Point3& v0, const Point3& v1, const Point3& v2);
66
67 ~Triangle();
68
69 /** 0, 1, or 2 */
70 inline const Point3& vertex(int n) const {
71 debugAssert((n >= 0) && (n < 3));
72 return _vertex[n];
73 }
74
75 /** vertex[1] - vertex[0] */
76 inline const Vector3& edge01() const {
77 return _edge01;
78 }
79
80 /** vertex[2] - vertex[0] */
81 inline const Vector3& edge02() const {
82 return _edge02;
83 }
84
85 float area() const;
86
87 Vector3::Axis primaryAxis() const {
88 return _primaryAxis;
89 }
90
91 const Vector3& normal() const;
92
93 /** Barycenter */
94 Point3 center() const;
95
96 const Plane& plane() const;
97
98 /** Returns a random point in the triangle. */
99 Point3 randomPoint() const;
100
101 inline void getRandomSurfacePoint
102 (Point3& P,
103 Vector3& N = Vector3::ignore()) const {
104 P = randomPoint();
105 N = normal();
106 }
107
108 /**
109 For two triangles to be equal they must have
110 the same vertices <I>in the same order</I>.
111 That is, vertex[0] == vertex[0], etc.
112 */
113 inline bool operator==(const Triangle& other) const {
114 for (int i = 0; i < 3; ++i) {
115 if (_vertex[i] != other._vertex[i]) {
116 return false;
117 }
118 }
119
120 return true;
121 }
122
123 inline size_t hashCode() const {
124 return
125 _vertex[0].hashCode() +
126 (_vertex[1].hashCode() >> 2) +
127 (_vertex[2].hashCode() >> 3);
128 }
129
130 void getBounds(class AABox&) const;
131
132 /**
133 @brief Intersect the ray at distance less than @a distance.
134
135 @param distance Set to the maximum distance (can be G3D::inf())
136 to search for an intersection. On return, this is the smaller
137 of the distance to the intersection, if one exists, and the original
138 value.
139
140 @param baryCoord If a triangle is hit before @a distance, a
141 the barycentric coordinates of the hit location on the triangle.
142 Otherwise, unmodified.
143
144 @return True if there was an intersection before the original distance.
145 */
146 bool intersect(const class Ray& ray, float& distance, float baryCoord[3]) const;
147};
148
149} // namespace G3D
150
151template <> struct HashTrait<G3D::Triangle> {
152 static size_t hashCode(const G3D::Triangle& key) { return key.hashCode(); }
153};
154
155
156template<> struct BoundsTrait<class G3D::Triangle> {
157 static void getBounds(const G3D::Triangle& t, G3D::AABox& out) { t.getBounds(out); }
158};

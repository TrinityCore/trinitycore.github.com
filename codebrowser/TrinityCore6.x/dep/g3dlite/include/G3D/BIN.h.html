33/* 8-bit conversion function */
34#define B8__(x) ((x&0x0000000FLU)?1:0) \
35+((x&0x000000F0LU)?2:0) \
36+((x&0x00000F00LU)?4:0) \

43/**
44 \def BIN8()
45 \brief Generate a 16-bit constant in binary notation, in 8-bit strings.
46
47 The most significant byte is first.
48
49 \code
50 unsigned int i = BIN8(01010101); // 85

56#define BIN8(d) ((unsigned char)B8__(HEX__(d)))

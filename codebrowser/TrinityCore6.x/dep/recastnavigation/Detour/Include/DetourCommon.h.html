80/// Derives the cross product of two vectors. (@p v1 x @p v2)
81/// @param[out] dest The cross product. [(x, y, z)]
82/// @param[in] v1 A Vector [(x, y, z)]
83/// @param[in] v2 A vector [(x, y, z)]
84inline void dtVcross(float* dest, const float* v1, const float* v2)
85{
86 dest[0] = v1[1]*v2[2] - v1[2]*v2[1];
87 dest[1] = v1[2]*v2[0] - v1[0]*v2[2];
88 dest[2] = v1[0]*v2[1] - v1[1]*v2[0];
89}

91/// Derives the dot product of two vectors. (@p v1 . @p v2)
92/// @param[in] v1 A Vector [(x, y, z)]
93/// @param[in] v2 A vector [(x, y, z)]
94/// @return The dot product.
95inline float dtVdot(const float* v1, const float* v2)
96{
97 return v1[0]*v2[0] + v1[1]*v2[1] + v1[2]*v2[2];
98}

100/// Performs a scaled vector addition. (@p v1 + (@p v2 * @p s))
101/// @param[out] dest The result vector. [(x, y, z)]
102/// @param[in] v1 The base vector. [(x, y, z)]
103/// @param[in] v2 The vector to scale and add to @p v1. [(x, y, z)]
104/// @param[in] s The amount to scale @p v2 by before adding to @p v1.
105inline void dtVmad(float* dest, const float* v1, const float* v2, const float s)
106{
107 dest[0] = v1[0]+v2[0]*s;
108 dest[1] = v1[1]+v2[1]*s;
109 dest[2] = v1[2]+v2[2]*s;
110}

112/// Performs a linear interpolation between two vectors. (@p v1 toward @p v2)
113/// @param[out] dest The result vector. [(x, y, x)]
114/// @param[in] v1 The starting vector.
115/// @param[in] v2 The destination vector.
116/// @param[in] t The interpolation factor. [Limits: 0 <= value <= 1.0]
117inline void dtVlerp(float* dest, const float* v1, const float* v2, const float t)
118{
119 dest[0] = v1[0]+(v2[0]-v1[0])*t;
120 dest[1] = v1[1]+(v2[1]-v1[1])*t;
121 dest[2] = v1[2]+(v2[2]-v1[2])*t;
122}

213/// Returns the distance between two points.
214/// @param[in] v1 A point. [(x, y, z)]
215/// @param[in] v2 A point. [(x, y, z)]
216/// @return The distance between the two points.
217inline float dtVdist(const float* v1, const float* v2)
218{
219 const float dx = v2[0] - v1[0];
220 const float dy = v2[1] - v1[1];
221 const float dz = v2[2] - v1[2];
222 return dtMathSqrtf(dx*dx + dy*dy + dz*dz);
223}
224
225/// Returns the square of the distance between two points.
226/// @param[in] v1 A point. [(x, y, z)]
227/// @param[in] v2 A point. [(x, y, z)]
228/// @return The square of the distance between the two points.
229inline float dtVdistSqr(const float* v1, const float* v2)
230{
231 const float dx = v2[0] - v1[0];
232 const float dy = v2[1] - v1[1];
233 const float dz = v2[2] - v1[2];
234 return dx*dx + dy*dy + dz*dz;
235}

237/// Derives the distance between the specified points on the xz-plane.
238/// @param[in] v1 A point. [(x, y, z)]
239/// @param[in] v2 A point. [(x, y, z)]
240/// @return The distance between the point on the xz-plane.
241///
242/// The vectors are projected onto the xz-plane, so the y-values are ignored.
243inline float dtVdist2D(const float* v1, const float* v2)
244{
245 const float dx = v2[0] - v1[0];
246 const float dz = v2[2] - v1[2];
247 return dtMathSqrtf(dx*dx + dz*dz);
248}

250/// Derives the square of the distance between the specified points on the xz-plane.
251/// @param[in] v1 A point. [(x, y, z)]
252/// @param[in] v2 A point. [(x, y, z)]
253/// @return The square of the distance between the point on the xz-plane.
254inline float dtVdist2DSqr(const float* v1, const float* v2)
255{
256 const float dx = v2[0] - v1[0];
257 const float dz = v2[2] - v1[2];
258 return dx*dx + dz*dz;
259}

261/// Normalizes the vector.
262/// @param[in,out] v The vector to normalize. [(x, y, z)]
263inline void dtVnormalize(float* v)
264{
265 float d = 1.0f / dtMathSqrtf(dtSqr(v[0]) + dtSqr(v[1]) + dtSqr(v[2]));
266 v[0] *= d;
267 v[1] *= d;
268 v[2] *= d;
269}

271/// Performs a 'sloppy' colocation check of the specified points.
272/// @param[in] p0 A point. [(x, y, z)]
273/// @param[in] p1 A point. [(x, y, z)]
274/// @return True if the points are considered to be at the same location.
275///
276/// Basically, this function will return true if the specified points are
277/// close enough to eachother to be considered colocated.
278inline bool dtVequal(const float* p0, const float* p1)
279{
280 static const float thr = dtSqr(1.0f/16384.0f);
281 const float d = dtVdistSqr(p0, p1);
282 return d < thr;
283}

311/// Derives the signed xz-plane area of the triangle ABC, or the relationship of line AB to point C.
312/// @param[in] a Vertex A. [(x, y, z)]
313/// @param[in] b Vertex B. [(x, y, z)]
314/// @param[in] c Vertex C. [(x, y, z)]
315/// @return The signed xz-plane area of the triangle.
316inline float dtTriArea2D(const float* a, const float* b, const float* c)
317{
318 const float abx = b[0] - a[0];
319 const float abz = b[2] - a[2];
320 const float acx = c[0] - a[0];
321 const float acz = c[2] - a[2];
322 return acx*abz - abx*acz;
323}

325/// Determines if two axis-aligned bounding boxes overlap.
326/// @param[in] amin Minimum bounds of box A. [(x, y, z)]
327/// @param[in] amax Maximum bounds of box A. [(x, y, z)]
328/// @param[in] bmin Minimum bounds of box B. [(x, y, z)]
329/// @param[in] bmax Maximum bounds of box B. [(x, y, z)]
330/// @return True if the two AABB's overlap.
331/// @see dtOverlapBounds
332inline bool dtOverlapQuantBounds(const unsigned short amin[3], const unsigned short amax[3],
333 const unsigned short bmin[3], const unsigned short bmax[3])
334{
335 bool overlap = true;
336 overlap = (amin[0] > bmax[0] || amax[0] < bmin[0]) ? false : overlap;
337 overlap = (amin[1] > bmax[1] || amax[1] < bmin[1]) ? false : overlap;
338 overlap = (amin[2] > bmax[2] || amax[2] < bmin[2]) ? false : overlap;
339 return overlap;
340}

417inline unsigned int dtNextPow2(unsigned int v)
418{
419 v--;
420 v |= v >> 1;
421 v |= v >> 2;
422 v |= v >> 4;
423 v |= v >> 8;
424 v |= v >> 16;
425 v++;
426 return v;
427}

429inline unsigned int dtIlog2(unsigned int v)
430{
431 unsigned int r;
432 unsigned int shift;
433 r = (v > 0xffff) << 4; v >>= r;
434 shift = (v > 0xff) << 3; v >>= shift; r |= shift;
435 shift = (v > 0xf) << 2; v >>= shift; r |= shift;
436 shift = (v > 0x3) << 1; v >>= shift; r |= shift;
437 r |= (v >> 1);
438 return r;
439}

494/**
495
496@fn float dtTriArea2D(const float* a, const float* b, const float* c)
497@par
498
499The vertices are projected onto the xz-plane, so the y-values are ignored.
500
501This is a low cost function than can be used for various purposes. Its main purpose
502is for point/line relationship testing.
503
504In all cases: A value of zero indicates that all vertices are collinear or represent the same point.
505(On the xz-plane.)
506
507When used for point/line relationship tests, AB usually represents a line against which
508the C point is to be tested. In this case:
509
510A positive value indicates that point C is to the left of line AB, looking from A toward B.<br/>
511A negative value indicates that point C is to the right of lineAB, looking from A toward B.
512
513When used for evaluating a triangle:
514
515The absolute value of the return value is two times the area of the triangle when it is
516projected onto the xz-plane.
517
518A positive return value indicates:
519
520<ul>
521<li>The vertices are wrapped in the normal Detour wrap direction.</li>
522<li>The triangle's 3D face normal is in the general up direction.</li>
523</ul>
524
525A negative return value indicates:
526
527<ul>
528<li>The vertices are reverse wrapped. (Wrapped opposite the normal Detour wrap direction.)</li>
529<li>The triangle's 3D face normal is in the general down direction.</li>
530</ul>
531
532*/

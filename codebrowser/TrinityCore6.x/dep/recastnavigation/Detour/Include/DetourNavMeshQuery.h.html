33/// Defines polygon filtering and traversal costs for navigation mesh query operations.
34/// @ingroup detour
35class dtQueryFilter
36{
37 float m_areaCost[DT_MAX_AREAS]; ///< Cost per area type. (Used by default implementation.)
38 unsigned short m_includeFlags; ///< Flags for polygons that can be visited. (Used by default implementation.)
39 unsigned short m_excludeFlags; ///< Flags for polygons that should not be visted. (Used by default implementation.)
40
41public:
42 dtQueryFilter();
43
44#ifdef DT_VIRTUAL_QUERYFILTER
45 virtual ~dtQueryFilter() { }
46#endif
47
48 /// Returns true if the polygon can be visited. (I.e. Is traversable.)
49 /// @param[in] ref The reference id of the polygon test.
50 /// @param[in] tile The tile containing the polygon.
51 /// @param[in] poly The polygon to test.
52#ifdef DT_VIRTUAL_QUERYFILTER
53 virtual bool passFilter(const dtPolyRef ref,
54 const dtMeshTile* tile,
55 const dtPoly* poly) const;
56#else
57 bool passFilter(const dtPolyRef ref,
58 const dtMeshTile* tile,
59 const dtPoly* poly) const;
60#endif
61
62 /// Returns cost to move from the beginning to the end of a line segment
63 /// that is fully contained within a polygon.
64 /// @param[in] pa The start position on the edge of the previous and current polygon. [(x, y, z)]
65 /// @param[in] pb The end position on the edge of the current and next polygon. [(x, y, z)]
66 /// @param[in] prevRef The reference id of the previous polygon. [opt]
67 /// @param[in] prevTile The tile containing the previous polygon. [opt]
68 /// @param[in] prevPoly The previous polygon. [opt]
69 /// @param[in] curRef The reference id of the current polygon.
70 /// @param[in] curTile The tile containing the current polygon.
71 /// @param[in] curPoly The current polygon.
72 /// @param[in] nextRef The refernece id of the next polygon. [opt]
73 /// @param[in] nextTile The tile containing the next polygon. [opt]
74 /// @param[in] nextPoly The next polygon. [opt]
75#ifdef DT_VIRTUAL_QUERYFILTER
76 virtual float getCost(const float* pa, const float* pb,
77 const dtPolyRef prevRef, const dtMeshTile* prevTile, const dtPoly* prevPoly,
78 const dtPolyRef curRef, const dtMeshTile* curTile, const dtPoly* curPoly,
79 const dtPolyRef nextRef, const dtMeshTile* nextTile, const dtPoly* nextPoly) const;
80#else
81 float getCost(const float* pa, const float* pb,
82 const dtPolyRef prevRef, const dtMeshTile* prevTile, const dtPoly* prevPoly,
83 const dtPolyRef curRef, const dtMeshTile* curTile, const dtPoly* curPoly,
84 const dtPolyRef nextRef, const dtMeshTile* nextTile, const dtPoly* nextPoly) const;
85#endif
86
87 /// @name Getters and setters for the default implementation data.
88 ///@{
89
90 /// Returns the traversal cost of the area.
91 /// @param[in] i The id of the area.
92 /// @returns The traversal cost of the area.
93 inline float getAreaCost(const int i) const { return m_areaCost[i]; }
94
95 /// Sets the traversal cost of the area.
96 /// @param[in] i The id of the area.
97 /// @param[in] cost The new cost of traversing the area.
98 inline void setAreaCost(const int i, const float cost) { m_areaCost[i] = cost; }
99
100 /// Returns the include flags for the filter.
101 /// Any polygons that include one or more of these flags will be
102 /// included in the operation.
103 inline unsigned short getIncludeFlags() const { return m_includeFlags; }
104
105 /// Sets the include flags for the filter.
106 /// @param[in] flags The new flags.
107 inline void setIncludeFlags(const unsigned short flags) { m_includeFlags = flags; }
108
109 /// Returns the exclude flags for the filter.
110 /// Any polygons that include one ore more of these flags will be
111 /// excluded from the operation.
112 inline unsigned short getExcludeFlags() const { return m_excludeFlags; }
113
114 /// Sets the exclude flags for the filter.
115 /// @param[in] flags The new flags.
116 inline void setExcludeFlags(const unsigned short flags) { m_excludeFlags = flags; }
117
118 ///@}
119
120};

124/// Provides information about raycast hit
125/// filled by dtNavMeshQuery::raycast
126/// @ingroup detour
127struct dtRaycastHit
128{
129 /// The hit parameter. (FLT_MAX if no wall hit.)
130 float t;
131
132 /// hitNormal The normal of the nearest wall hit. [(x, y, z)]
133 float hitNormal[3];
134
135 /// Pointer to an array of reference ids of the visited polygons. [opt]
136 dtPolyRef* path;
137
138 /// The number of visited polygons. [opt]
139 int pathCount;
140
141 /// The maximum number of polygons the @p path array can hold.
142 int maxPath;
143
144 /// The cost of the path until hit.
145 float pathCost;
146};

150/// Provides the ability to perform pathfinding related queries against
151/// a navigation mesh.
152/// @ingroup detour
153class dtNavMeshQuery
154{
155public:
156 dtNavMeshQuery();
157 ~dtNavMeshQuery();
158
159 /// Initializes the query object.
160 /// @param[in] nav Pointer to the dtNavMesh object to use for all queries.
161 /// @param[in] maxNodes Maximum number of search nodes. [Limits: 0 < value <= 65536]
162 /// @returns The status flags for the query.
163 dtStatus init(const dtNavMesh* nav, const int maxNodes);
164
165 /// @name Standard Pathfinding Functions
166 // /@{
167
168 /// Finds a path from the start polygon to the end polygon.
169 /// @param[in] startRef The refrence id of the start polygon.
170 /// @param[in] endRef The reference id of the end polygon.
171 /// @param[in] startPos A position within the start polygon. [(x, y, z)]
172 /// @param[in] endPos A position within the end polygon. [(x, y, z)]
173 /// @param[in] filter The polygon filter to apply to the query.
174 /// @param[out] path An ordered list of polygon references representing the path. (Start to end.)
175 /// [(polyRef) * @p pathCount]
176 /// @param[out] pathCount The number of polygons returned in the @p path array.
177 /// @param[in] maxPath The maximum number of polygons the @p path array can hold. [Limit: >= 1]
178 dtStatus findPath(dtPolyRef startRef, dtPolyRef endRef,
179 const float* startPos, const float* endPos,
180 const dtQueryFilter* filter,
181 dtPolyRef* path, int* pathCount, const int maxPath) const;
182
183 /// Finds the straight path from the start to the end position within the polygon corridor.
184 /// @param[in] startPos Path start position. [(x, y, z)]
185 /// @param[in] endPos Path end position. [(x, y, z)]
186 /// @param[in] path An array of polygon references that represent the path corridor.
187 /// @param[in] pathSize The number of polygons in the @p path array.
188 /// @param[out] straightPath Points describing the straight path. [(x, y, z) * @p straightPathCount].
189 /// @param[out] straightPathFlags Flags describing each point. (See: #dtStraightPathFlags) [opt]
190 /// @param[out] straightPathRefs The reference id of the polygon that is being entered at each point. [opt]
191 /// @param[out] straightPathCount The number of points in the straight path.
192 /// @param[in] maxStraightPath The maximum number of points the straight path arrays can hold. [Limit: > 0]
193 /// @param[in] options Query options. (see: #dtStraightPathOptions)
194 /// @returns The status flags for the query.
195 dtStatus findStraightPath(const float* startPos, const float* endPos,
196 const dtPolyRef* path, const int pathSize,
197 float* straightPath, unsigned char* straightPathFlags, dtPolyRef* straightPathRefs,
198 int* straightPathCount, const int maxStraightPath, const int options = 0) const;
199
200 ///@}
201 /// @name Sliced Pathfinding Functions
202 /// Common use case:
203 /// -# Call initSlicedFindPath() to initialize the sliced path query.
204 /// -# Call updateSlicedFindPath() until it returns complete.
205 /// -# Call finalizeSlicedFindPath() to get the path.
206 ///@{
207
208 /// Intializes a sliced path query.
209 /// @param[in] startRef The refrence id of the start polygon.
210 /// @param[in] endRef The reference id of the end polygon.
211 /// @param[in] startPos A position within the start polygon. [(x, y, z)]
212 /// @param[in] endPos A position within the end polygon. [(x, y, z)]
213 /// @param[in] filter The polygon filter to apply to the query.
214 /// @param[in] options query options (see: #dtFindPathOptions)
215 /// @returns The status flags for the query.
216 dtStatus initSlicedFindPath(dtPolyRef startRef, dtPolyRef endRef,
217 const float* startPos, const float* endPos,
218 const dtQueryFilter* filter, const unsigned int options = 0);
219
220 /// Updates an in-progress sliced path query.
221 /// @param[in] maxIter The maximum number of iterations to perform.
222 /// @param[out] doneIters The actual number of iterations completed. [opt]
223 /// @returns The status flags for the query.
224 dtStatus updateSlicedFindPath(const int maxIter, int* doneIters);
225
226 /// Finalizes and returns the results of a sliced path query.
227 /// @param[out] path An ordered list of polygon references representing the path. (Start to end.)
228 /// [(polyRef) * @p pathCount]
229 /// @param[out] pathCount The number of polygons returned in the @p path array.
230 /// @param[in] maxPath The max number of polygons the path array can hold. [Limit: >= 1]
231 /// @returns The status flags for the query.
232 dtStatus finalizeSlicedFindPath(dtPolyRef* path, int* pathCount, const int maxPath);
233
234 /// Finalizes and returns the results of an incomplete sliced path query, returning the path to the furthest
235 /// polygon on the existing path that was visited during the search.
236 /// @param[in] existing An array of polygon references for the existing path.
237 /// @param[in] existingSize The number of polygon in the @p existing array.
238 /// @param[out] path An ordered list of polygon references representing the path. (Start to end.)
239 /// [(polyRef) * @p pathCount]
240 /// @param[out] pathCount The number of polygons returned in the @p path array.
241 /// @param[in] maxPath The max number of polygons the @p path array can hold. [Limit: >= 1]
242 /// @returns The status flags for the query.
243 dtStatus finalizeSlicedFindPathPartial(const dtPolyRef* existing, const int existingSize,
244 dtPolyRef* path, int* pathCount, const int maxPath);
245
246 ///@}
247 /// @name Dijkstra Search Functions
248 /// @{
249
250 /// Finds the polygons along the navigation graph that touch the specified circle.
251 /// @param[in] startRef The reference id of the polygon where the search starts.
252 /// @param[in] centerPos The center of the search circle. [(x, y, z)]
253 /// @param[in] radius The radius of the search circle.
254 /// @param[in] filter The polygon filter to apply to the query.
255 /// @param[out] resultRef The reference ids of the polygons touched by the circle. [opt]
256 /// @param[out] resultParent The reference ids of the parent polygons for each result.
257 /// Zero if a result polygon has no parent. [opt]
258 /// @param[out] resultCost The search cost from @p centerPos to the polygon. [opt]
259 /// @param[out] resultCount The number of polygons found. [opt]
260 /// @param[in] maxResult The maximum number of polygons the result arrays can hold.
261 /// @returns The status flags for the query.
262 dtStatus findPolysAroundCircle(dtPolyRef startRef, const float* centerPos, const float radius,
263 const dtQueryFilter* filter,
264 dtPolyRef* resultRef, dtPolyRef* resultParent, float* resultCost,
265 int* resultCount, const int maxResult) const;
266
267 /// Finds the polygons along the naviation graph that touch the specified convex polygon.
268 /// @param[in] startRef The reference id of the polygon where the search starts.
269 /// @param[in] verts The vertices describing the convex polygon. (CCW)
270 /// [(x, y, z) * @p nverts]
271 /// @param[in] nverts The number of vertices in the polygon.
272 /// @param[in] filter The polygon filter to apply to the query.
273 /// @param[out] resultRef The reference ids of the polygons touched by the search polygon. [opt]
274 /// @param[out] resultParent The reference ids of the parent polygons for each result. Zero if a
275 /// result polygon has no parent. [opt]
276 /// @param[out] resultCost The search cost from the centroid point to the polygon. [opt]
277 /// @param[out] resultCount The number of polygons found.
278 /// @param[in] maxResult The maximum number of polygons the result arrays can hold.
279 /// @returns The status flags for the query.
280 dtStatus findPolysAroundShape(dtPolyRef startRef, const float* verts, const int nverts,
281 const dtQueryFilter* filter,
282 dtPolyRef* resultRef, dtPolyRef* resultParent, float* resultCost,
283 int* resultCount, const int maxResult) const;

300 /// Finds polygons that overlap the search box.

312 /// Finds the non-overlapping navigation polygons in the local neighbourhood around the center position.
313 /// @param[in] startRef The reference id of the polygon where the search starts.
314 /// @param[in] centerPos The center of the query circle. [(x, y, z)]
315 /// @param[in] radius The radius of the query circle.
316 /// @param[in] filter The polygon filter to apply to the query.
317 /// @param[out] resultRef The reference ids of the polygons touched by the circle.
318 /// @param[out] resultParent The reference ids of the parent polygons for each result.
319 /// Zero if a result polygon has no parent. [opt]
320 /// @param[out] resultCount The number of polygons found.
321 /// @param[in] maxResult The maximum number of polygons the result arrays can hold.
322 /// @returns The status flags for the query.
323 dtStatus findLocalNeighbourhood(dtPolyRef startRef, const float* centerPos, const float radius,

325 dtPolyRef* resultRef, dtPolyRef* resultParent,
326 int* resultCount, const int maxResult) const;

328 /// Moves from the start to the end position constrained to the navigation mesh.
329 /// @param[in] startRef The reference id of the start polygon.
330 /// @param[in] startPos A position of the mover within the start polygon. [(x, y, x)]
331 /// @param[in] endPos The desired end position of the mover. [(x, y, z)]
332 /// @param[in] filter The polygon filter to apply to the query.
333 /// @param[out] resultPos The result position of the mover. [(x, y, z)]

342 /// Casts a 'walkability' ray along the surface of the navigation mesh from
343 /// the start position toward the end position.
344 /// @note A wrapper around raycast(..., RaycastHit*). Retained for backward compatibility.
345 /// @param[in] startRef The reference id of the start polygon.
346 /// @param[in] startPos A position within the start polygon representing
347 /// the start of the ray. [(x, y, z)]
348 /// @param[in] endPos The position to cast the ray toward. [(x, y, z)]
349 /// @param[out] t The hit parameter. (FLT_MAX if no wall hit.)
350 /// @param[out] hitNormal The normal of the nearest wall hit. [(x, y, z)]
351 /// @param[in] filter The polygon filter to apply to the query.
352 /// @param[out] path The reference ids of the visited polygons. [opt]
353 /// @param[out] pathCount The number of visited polygons. [opt]
354 /// @param[in] maxPath The maximum number of polygons the @p path array can hold.
355 /// @returns The status flags for the query.
356 dtStatus raycast(dtPolyRef startRef, const float* startPos, const float* endPos,
357 const dtQueryFilter* filter,
358 float* t, float* hitNormal, dtPolyRef* path, int* pathCount, const int maxPath) const;

360 /// Casts a 'walkability' ray along the surface of the navigation mesh from
361 /// the start position toward the end position.
362 /// @param[in] startRef The reference id of the start polygon.
363 /// @param[in] startPos A position within the start polygon representing
364 /// the start of the ray. [(x, y, z)]
365 /// @param[in] endPos The position to cast the ray toward. [(x, y, z)]
366 /// @param[in] filter The polygon filter to apply to the query.
367 /// @param[in] flags govern how the raycast behaves. See dtRaycastOptions
368 /// @param[out] hit Pointer to a raycast hit structure which will be filled by the results.
369 /// @param[in] prevRef parent of start ref. Used during for cost calculation [opt]
370 /// @returns The status flags for the query.
371 dtStatus raycast(dtPolyRef startRef, const float* startPos, const float* endPos,
372 const dtQueryFilter* filter, const unsigned int options,

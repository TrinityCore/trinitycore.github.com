151/// Defines a polyogn within a dtMeshTile object.
152/// @ingroup detour
153struct dtPoly
154{
155 /// Index to first link in linked list. (Or #DT_NULL_LINK if there is no link.)
156 unsigned int firstLink;
157
158 /// The indices of the polygon's vertices.
159 /// The actual vertices are located in dtMeshTile::verts.
160 unsigned short verts[DT_VERTS_PER_POLYGON];
161
162 /// Packed data representing neighbor polygons references and flags for each edge.
163 unsigned short neis[DT_VERTS_PER_POLYGON];
164
165 /// The user defined polygon flags.
166 unsigned short flags;
167
168 /// The number of vertices in the polygon.
169 unsigned char vertCount;
170
171 /// The bit packed area id and polygon type.
172 /// @note Use the structure's set and get methods to acess this value.
173 unsigned char areaAndtype;
174
175 /// Sets the user defined area id. [Limit: < #DT_MAX_AREAS]
176 inline void setArea(unsigned char a) { areaAndtype = (areaAndtype & 0xc0) | (a & 0x3f); }
177
178 /// Sets the polygon type. (See: #dtPolyTypes.)
179 inline void setType(unsigned char t) { areaAndtype = (areaAndtype & 0x3f) | (t << 6); }
180
181 /// Gets the user defined area id.
182 inline unsigned char getArea() const { return areaAndtype & 0x3f; }
183
184 /// Gets the polygon type. (See: #dtPolyTypes)
185 inline unsigned char getType() const { return areaAndtype >> 6; }
186};

277/// Defines a navigation mesh tile.
278/// @ingroup detour
279struct dtMeshTile
280{

283 unsigned int linksFreeList; ///< Index to the next free link.

285 dtPoly* polys; ///< The tile polygons. [Size: dtMeshHeader::polyCount]
286 float* verts; ///< The tile vertices. [Size: dtMeshHeader::vertCount]
287 dtLink* links; ///< The tile links. [Size: dtMeshHeader::maxLinkCount]
288 dtPolyDetail* detailMeshes; ///< The tile's detail sub-meshes. [Size: dtMeshHeader::detailMeshCount]
289
290 /// The detail mesh's unique vertices. [(x, y, z) * dtMeshHeader::detailVertCount]
291 float* detailVerts;
292
293 /// The detail mesh's triangles. [(vertA, vertB, vertC) * dtMeshHeader::detailTriCount]
294 unsigned char* detailTris;
295
296 /// The tile bounding volume nodes. [Size: dtMeshHeader::bvNodeCount]
297 /// (Will be null if bounding volumes are disabled.)
298 dtBVNode* bvTree;
299
300 dtOffMeshConnection* offMeshCons; ///< The tile off-mesh connections. [Size: dtMeshHeader::offMeshConCount]
301
302 unsigned char* data; ///< The tile data. (Not directly accessed under normal situations.)
303 int dataSize; ///< Size of the tile data.
304 int flags; ///< Tile flags. (See: #dtTileFlags)
305 dtMeshTile* next; ///< The next free tile, or the next tile in the spatial grid.

505 /// @{
506 /// @name Encoding and Decoding
507 /// These functions are generally meant for internal use only.
508
509 /// Derives a standard polygon reference.
510 /// @note This function is generally meant for internal use only.
511 /// @param[in] salt The tile's salt value.
512 /// @param[in] it The index of the tile.
513 /// @param[in] ip The index of the polygon within the tile.
514 inline dtPolyRef encodePolyId(unsigned int salt, unsigned int it, unsigned int ip) const
515 {
516 return ((dtPolyRef)salt << (m_polyBits+m_tileBits)) | ((dtPolyRef)it << m_polyBits) | (dtPolyRef)ip;
517 }
518
519 /// Decodes a standard polygon reference.
520 /// @note This function is generally meant for internal use only.
521 /// @param[in] ref The polygon reference to decode.
522 /// @param[out] salt The tile's salt value.
523 /// @param[out] it The index of the tile.
524 /// @param[out] ip The index of the polygon within the tile.
525 /// @see #encodePolyId
526 inline void decodePolyId(dtPolyRef ref, unsigned int& salt, unsigned int& it, unsigned int& ip) const
527 {
528 const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
529 const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;
530 const dtPolyRef polyMask = ((dtPolyRef)1<<m_polyBits)-1;
531 salt = (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
532 it = (unsigned int)((ref >> m_polyBits) & tileMask);
533 ip = (unsigned int)(ref & polyMask);
534 }
535
536 /// Extracts a tile's salt value from the specified polygon reference.
537 /// @note This function is generally meant for internal use only.
538 /// @param[in] ref The polygon reference.
539 /// @see #encodePolyId
540 inline unsigned int decodePolyIdSalt(dtPolyRef ref) const
541 {
542 const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
543 return (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
544 }
545
546 /// Extracts the tile's index from the specified polygon reference.
547 /// @note This function is generally meant for internal use only.
548 /// @param[in] ref The polygon reference.
549 /// @see #encodePolyId
550 inline unsigned int decodePolyIdTile(dtPolyRef ref) const
551 {
552 const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;
553 return (unsigned int)((ref >> m_polyBits) & tileMask);
554 }
555
556 /// Extracts the polygon's index (within its tile) from the specified polygon reference.
557 /// @note This function is generally meant for internal use only.
558 /// @param[in] ref The polygon reference.
559 /// @see #encodePolyId
560 inline unsigned int decodePolyIdPoly(dtPolyRef ref) const
561 {
562 const dtPolyRef polyMask = ((dtPolyRef)1<<m_polyBits)-1;
563 return (unsigned int)(ref & polyMask);
564 }

649Polygon references are subject to the same invalidate/preserve/restore

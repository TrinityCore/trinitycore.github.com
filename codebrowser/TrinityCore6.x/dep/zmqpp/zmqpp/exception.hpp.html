21/*!
22 * Represents the base zmqpp exception.
23 *
24 * All zmqpp runtime exceptions are children of this class.
25 * The class itself does not provide any special access fields but it only
26 * for convince when catching exceptions.
27 *
28 * The class extends std::runtime_error.
29 *
30 */
31class exception : public std::runtime_error
32{
33public:
34 /*!
35 * Standard exception constructor.
36 *
37 * \param message a string representing the error message.
38 */
39 exception(std::string const& message)
40 : std::runtime_error(message)
41 { }
42};
43
44/*!
45 * Represents an attempt to use an invalid object.
46 *
47 * Objects may be invalid initially or after a shutdown or close.
48 */
49class invalid_instance : public exception
50{
51public:
52 invalid_instance(std::string const& message)
53 : exception(message)
54 { }
55};
56
57/*!
58 * Represents internal zmq errors.
59 *
60 * Any error response from the zmq bindings will be wrapped in this error.
61 *
62 * The class provides access to the zmq error number via zmq_error().
63 */
64class zmq_internal_exception : public exception
65{
66public:
67 /*!
68 * Uses the zmq functions to pull out error messages and numbers.
69 */
70 zmq_internal_exception()
71 : exception(zmq_strerror(zmq_errno()))
72 , _error(zmq_errno())
73 { }
74
75 /*!
76 * Retrieve the zmq error number associated with this exception.
77 * \return zmq error number
78 */
79 int zmq_error() const { return _error; }
80
81private:
82 int _error;
83};

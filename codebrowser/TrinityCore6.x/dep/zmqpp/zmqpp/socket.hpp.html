32/*!
33 * The socket class represents the zmq sockets.
34 *
35 * A socket can be bound and/or connected to as many endpoints as required
36 * with the sole exception of a ::pair socket.
37 *
38 * The routing is handled by zmq based on the type set.
39 *
40 * The bound side of an inproc connection must occur first and inproc can only
41 * connect to other inproc sockets of the same context.
42 *
43 * This class is c++0x move supporting and cannot be copied.
44 */
45class socket
46{
47public:
48 static const int normal; /*!< /brief default send type, no flags set */
49#if (ZMQ_VERSION_MAJOR == 2)
50 static const int dont_wait; /*!< /brief don't block if sending is not currently possible */
51#else
52 static const int dont_wait; /*!< /brief don't block if sending is not currently possible */
53#endif
54 static const int send_more; /*!< /brief more parts will follow this one */
55#ifdef ZMQ_EXPERIMENTAL_LABELS
56 static const int send_label; /*!< /brief this message part is an internal zmq label */
57#endif

94 /*!
95 * Asynchronously connects to an endpoint.
96 * If the endpoint is not inproc then zmq will happily keep trying
97 * to connect until there is something there.
98 *
99 * Inproc sockets must have a valid target already bound before connection
100 * will work.
101 *
102 * \param endpoint the zmq endpoint to connect to
103 */
104 void connect(endpoint_t const& endpoint);
105
106 /*!
107 * Asynchronously connects to multiple endpoints.
108 * If the endpoint is not inproc then zmq will happily keep trying
109 * to connect until there is something there.
110 *
111 * Inproc sockets must have a valid target already bound before connection
112 * will work.
113 *
114 * This is a helper function that wraps the single item connect in a loop
115 *
116 * \param connections_begin the starting iterator for zmq endpoints.
117 * \param connections_end the final iterator for zmq endpoints.
118 */
119 template<typename InputIterator>
120 void connect(InputIterator const& connections_begin, InputIterator const& connections_end)
121 {
122 for(InputIterator it = connections_begin; it != connections_end; ++it)
123 {
124 connect(*it);
125 }
126 }
127

129 /*!
130 * Disconnects a previously connected endpoint.

132 * \param endpoint the zmq endpoint to disconnect from

134 void disconnect(endpoint_t const& endpoint);

136 /*!
137 * Disconnects from multiple previously connected endpoints.
138 *
139 * This is a helper function that wraps the single item disconnect in a loop
140 *
141 * \param disconnections_begin the starting iterator for zmq endpoints.
142 * \param disconnections_end the final iterator for zmq endpoints.
143 */
144 template<typename InputIterator>
145 void disconnect(InputIterator const& disconnections_begin, InputIterator const& disconnections_end)
146 {
147 for(InputIterator it = disconnections_begin; it != disconnections_end; ++it)
148 {
149 disconnect(*it);
150 }

159 /*!
160 * Sends the message over the connection, this may be a multipart message.
161 *
162 * If dont_block is true and we are unable to add a new message then this
163 * function will return false.
164 *
165 * \param message message to send
166 * \param dont_block boolean to dictate if we wait while sending.
167 * \return true if message sent, false if it would have blocked
168 */
169 bool send(message_t& message, bool const dont_block = false);
170
171 /*!
172 * Gets a message from the connection, this may be a multipart message.
173 *
174 * If dont_block is true and we are unable to get a message then this
175 * function will return false.
176 *
177 * \param message reference to fill with received data
178 * \param dont_block boolean to dictate if we wait for data.
179 * \return true if message sent, false if it would have blocked
180 */
181 bool receive(message_t& message, bool const dont_block = false);
182
183 /*!
184 * Sends the byte data held by the string as the next message part.
185 *
186 * If the socket::DONT_WAIT flag and we are unable to add a new message to
187 * socket then this function will return false.
188 *
189 * \param string message part to send
190 * \param flags message send flags
191 * \return true if message part sent, false if it would have blocked
192 */
193 bool send(std::string const& string, int const flags = normal);
194
195 /*!
196 * If there is a message ready then get the next part as a string
197 *
198 * If the socket::DONT_WAIT flag and there is no message ready to receive
199 * then this function will return false.
200 *
201 * \param string message part to receive into
202 * \param flags message receive flags
203 * \return true if message part received, false if it would have blocked
204 */
205 bool receive(std::string& string, int const flags = normal);
206
207 /*!
208 * Sends the byte data pointed to by buffer as the next part of the message.
209 *
210 * If the socket::DONT_WAIT flag and we are unable to add a new message to
211 * socket then this function will return false.
212 *
213 * \param buffer byte buffer pointer to start writing from
214 * \param length max length of the buffer
215 * \param flags message send flags
216 * \return true if message part sent, false if it would have blocked
217 */
218 bool send_raw(char const* buffer, int const length, int const flags = normal);
219
220 /*!
221 * \warning If the buffer is not large enough for the message part then the
222 * data will be truncated. The rest of the part is lost forever.
223 *
224 * If there is a message ready then get the next part of it as a raw
225 * byte buffer.
226 *
227 * If the socket::DONT_WAIT flag and there is no message ready to receive
228 * then this function will return false.
229 *
230 * \param buffer byte buffer pointer to start writing to
231 * \param length max length of the buffer
232 * \param flags message receive flags
233 * \return true if message part received, false if it would have blocked
234 */
235 bool receive_raw(char* buffer, int& length, int const flags = normal);

239 * Subscribe to a topic

243 * set(zmqpp::socket_option::subscribe, topic);

245 *
246 * This method is only useful for subscribe type sockets.

248 * \param topic the topic to subscribe to.

252 /*!
253 * Subscribe to a topic
254 *
255 * Helper function that is equivalent of a loop calling
256 * \code
257 * set(zmqpp::socket_option::subscribe, topic);
258 * \endcode
259 *
260 * This method is only useful for subscribe type sockets.
261 *
262 * Generally this will be used with stl collections using begin() and
263 * end() functions to get the iterators.
264 * For this reason the end loop runs until the end iterator, not inclusive
265 * of it.
266 *
267 * \param topics_begin the starting iterator for topics.
268 * \param topics_end the final iterator for topics.
269 */
270 template<typename InputIterator>
271 void subscribe(InputIterator const& topics_begin, InputIterator const& topics_end)
272 {
273 for(InputIterator it = topics_begin; it != topics_end; ++it)
274 {
275 subscribe(*it);
276 }
277 }

428 /*!
429 * For those that don't want to get into a referenced value this templated method
430 * will return the value instead.
431 *
432 * \param option a valid ::socket_option
433 * \return socket option value
434 */
435 template<typename Type>
436 Type get(socket_option const option) const
437 {
438 Type value = Type();
439 get(option, value);
440 return value;

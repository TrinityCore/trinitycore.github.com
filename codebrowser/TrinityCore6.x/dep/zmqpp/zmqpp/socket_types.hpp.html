9#define ZMQPP_SOCKET_TYPES_HPP_
10
11namespace zmqpp
12{
13
14/*!
15 * \brief Socket types allowed by zmq
16 *
17 * The socket type choose at creation must be one of these types.
18 *
19 * Each is designed for a different use and has different limitations.
20 */
21enum class socket_type {
22 /*!
23 * One to one - two way connection.\n
24 * Connect to ::pair.\n
25 * A \c pair socket has to be connected only one other pair socket and allows
26 * two way communication between them.
27 */
28 pair = ZMQ_PAIR,
29
30 /*!
31 * One to many - fan out.\n
32 * Connect to ::subscribe or ::xsubscribe.\n
33 * Socket is send only.\n
34 * Socket will drop messages and not block.\n
35 * \c publish sockets allow sending of the same message to many subscribers
36 * each subscriber can limit what is sent through the socket_option::subscribe
37 * settings.
38 */
39 publish = ZMQ_PUB,
40
41 /*!
42 * \note It seems doxygen can't work out which data is for the socket type and
43 * which is for the socket option so both get listed for both.
44 *
45 * One to many - fair-queued.\n
46 * Connect to ::publish or ::xpublish.\n
47 * Socket is receive only.\n
48 * The \c subscribe socket can connection to any number of publishers and will
49 * fairly pull messages from each. The socket_option::subscribe settings can
50 * be use to limit which messages are received and by default none are.
51 */
52 subscribe = ZMQ_SUB,
53
54 /*!
55 * One to many - fair-queued.\n
56 * Connect to ::push.\n
57 * Socket is receive only.\n
58 * The \c pull socket fairly pulls messages from all pushers it is connected
59 * to.
60 */
61 pull = ZMQ_PULL,
62
63 /*!
64 * One to many - load-balanced.\n
65 * Connect to ::pull.\n
66 * Socket is send only.\n
67 * Socket will block if unable to send.\n
68 * The \c push socket fairly distributes messages between any connected
69 * puller sockets.
70 */
71 push = ZMQ_PUSH,
72
73 /*!
74 * One to many - fair-queued outgoing, last peer incoming.\n
75 * Connect to ::reply or ::xreply.\n
76 * Socket flips between send and receive only.\n
77 * Socket will block if unable to send.\n
78 * The \c request socket will fairly balance requests sent out to a
79 * replier and then can only be used to receive until that replier
80 * sends a reply.
81 */
82 request = ZMQ_REQ,
83
84 /*!
85 * One to many - load-balanced incoming, last peer outgoing.\n
86 * Connect to ::request or ::xrequest.\n
87 * Socket flips between send and receive only.\n
88 * Socket will drop messages and not block.\n
89 * The \c reply socket can only receive until it pulls a message from a
90 * requester at which point it can only send until the reply is sent.
91 */
92 reply = ZMQ_REP,
93
94 /*!
95 * One to many - fan out.\n
96 * Connect to ::subscribe or ::xsubscribe.\n
97 * Socket is send only with the exception of special subscription messages.\n
98 * Socket will drop messages and not block.\n
99 * \c xpublish act the same as ::publish sockets however also allow special
100 * subscription messages to be received from subscribers.
101 */
102 xpublish = ZMQ_XPUB,
103
104 /*!
105 * One to many - fair-queued.\n
106 * Connect to ::publish or ::xpublish.\n
107 * Socket is receive only with the exception of special subscription messages\n
108 * \c xsubscribe act the same as ::subscribe sockets however also allow special
109 * subscription messages to be send to connected publishers.
110 */
111 xsubscribe = ZMQ_XSUB,
112
113 /*!
114 * One to many - fair-queued incoming, load-balanced outgoing.\n
115 * Connect to ::reply or ::xreply.\n
116 * Socket will block if unable to send.\n
117 * An \c xrequest socket balances requests between repliers and pulls replies
118 * back in a fair manner. Each request is expected to have exactly one reply.
119 */
120 xrequest = ZMQ_XREQ,
121
122 /*!
123 * One to many - fair-queued incoming, targeted outgoing.\n
124 * Connect to ::request or ::xrequest.\n
125 * Socket will drop messages and not block.\n
126 * An \c xreply socket fairly pulls in requests from requesters and will
127 * label requests so it can return replies back to the correct target.
128 */
129 xreply = ZMQ_XREP,
130
131 // To match for people who prefer the shorter versions
132 pub = ZMQ_PUB, /*!< version of ::publish to match zmq name convention */
133 sub = ZMQ_SUB, /*!< version of ::subscribe to match zmq name convention */
134 req = ZMQ_REQ, /*!< version of ::request to match zmq name convention */
135 rep = ZMQ_REP, /*!< version of ::reply to match zmq name convention */
136 xpub = ZMQ_XPUB, /*!< version of ::xpublish to match zmq name convention */
137 xsub = ZMQ_XSUB, /*!< version of ::xsubscribe to match zmq name convention */
138 xreq = ZMQ_XREQ, /*!< version of ::xrequest to match zmq name convention */
139 xrep = ZMQ_XREP, /*!< version of ::xreply to match zmq name convention */
140
141 // For completion
142 router = ZMQ_ROUTER, /*!< \deprecated Matches zmq 2.x xrep functionality. */
143 dealer = ZMQ_DEALER /*!< \deprecated Matches zmq 2.x xreq functionality. */
144};
145
146}

148#endif /* ZMQPP_SOCKET_TYPES_HPP_ */

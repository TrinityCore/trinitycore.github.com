8#ifndef ZMQPP_POLLER_HPP_
9#define ZMQPP_POLLER_HPP_
10
11#include <unordered_map>
12#include <vector>
13
14#include "compatibility.hpp"
15
16namespace zmqpp
17{
18
19class socket;
20typedef socket socket_t;
21
22/*!
23 * Polling wrapper.
24 *
25 * Allows access to polling for any number of sockets or file descriptors.
26 */
27class poller
28{
29public:
30 static const long wait_forever; /*!< Block forever flag, default setting. */
31
32 static const short poll_none; /*!< No polling flags set. */
33 static const short poll_in; /*!< Monitor inbound flag. */
34 static const short poll_out; /*!< Monitor output flag. */
35 static const short poll_error; /*!< Monitor error flag.\n Only for file descriptors. */
36
37 /*!
38 * Construct an empty polling model.
39 */
40 poller();
41
42 /*!
43 * Cleanup poller.
44 *
45 * Any sockets will need to be closed separately.
46 */
47 ~poller();
48
49 /*!
50 * Add a socket to the polling model and set which events to monitor.
51 *
52 * \param socket the socket to monitor.
53 * \param event the event flags to monitor on the socket.
54 */
55 void add(socket_t& socket, short const event = poll_in);
56
57 /*!
58 * Add a file descriptor to the polling model and set which events to monitor.
59 *
60 * \param descriptor the file descriptor to monitor.
61 * \param event the event flags to monitor.
62 */
63 void add(int const descriptor, short const event = poll_in | poll_error);
64
65 /*!
66 * Check if we are monitoring a given socket with this poller.
67 *
68 * \param socket the socket to check.
69 * \return true if it is there.
70 */
71 bool has(socket_t const& socket);
72
73 /*!
74 * Check if we are monitoring a given file descriptor with this poller.
75 *
76 * \param descriptor the file descriptor to check.
77 * \return true if it is there.
78 */
79 bool has(int const descriptor);
80
81 /*!
82 * Stop monitoring a socket.
83 *
84 * \param socket the socket to stop monitoring.
85 */
86 void remove(socket_t const& socket);
87
88 /*!
89 * Stop monitoring a file descriptor.
90 *
91 * \param descriptor the file descriptor to stop monitoring.
92 */
93 void remove(int const descriptor);
94
95 /*!
96 * Update the monitored event flags for a given socket.
97 *
98 * \param socket the socket to update event flags.
99 * \param event the event flags to monitor on the socket.
100 */
101 void check_for(socket_t const& socket, short const event);
102
103 /*!
104 * Update the monitored event flags for a given file descriptor.
105 *
106 * \param descriptor the file descriptor to update event flags.
107 * \param event the event flags to monitor on the socket.
108 */
109 void check_for(int const descriptor, short const event);
110
111 /*!
112 * Poll for monitored events.
113 *
114 * By default this method will block forever or until at least one of the monitored
115 * sockets or file descriptors has events.
116 *
117 * If a timeout is set and was reached then this function returns false.
118 *
119 * \param timeout milliseconds to timeout.
120 * \return true if there is an event..
121 */
122 bool poll(long timeout = wait_forever);
123
124 /*!
125 * Get the event flags triggered for a socket.
126 *
127 * \param socket the socket to get triggered event flags for.
128 * \return the event flags.
129 */
130 short events(socket_t const& socket) const;
131
132 /*!
133 * Get the event flags triggered for a file descriptor.
134 *
135 * \param descriptor the file descriptor to get triggered event flags for.
136 * \return the event flags.
137 */
138 short events(int const descriptor) const;
139
140 /*!
141 * Check either a file descriptor or socket for input events.
142 *
143 * Templated helper method that calls through to event and checks for a given flag
144 *
145 * \param watchable either a file descriptor or socket known to the poller.
146 * \return true if there is input.
147 */
148 template<typename Watched>
149 bool has_input(Watched const& watchable) const { return events(watchable) & poll_in; }
150
151 /*!
152 * Check either a file descriptor or socket for output events.
153 *
154 * Templated helper method that calls through to event and checks for a given flag
155 *
156 * \param watchable either a file descriptor or socket known to the poller.
157 * \return true if there is output.
158 */
159 template<typename Watched>
160 bool has_output(Watched const& watchable) const { return events(watchable) & poll_out; }
161
162 /*!
163 * Check a file descriptor.
164 *
165 * Templated helper method that calls through to event and checks for a given flag
166 *
167 * Technically this template works for sockets as well but the error flag is never set for
168 * sockets so I have no idea why someone would call it.
169 *
170 * \param watchable a file descriptor know to the poller.
171 * \return true if there is an error.
172 */
173 template<typename Watched>
174 bool has_error(Watched const& watchable) const { return events(watchable) & poll_error; }
175
176private:
177 std::vector<zmq_pollitem_t> _items;
178 std::unordered_map<void *, size_t> _index;
179 std::unordered_map<int, size_t> _fdindex;
180
181 void reindex(size_t const index);
182};
183
184}
185
186#endif /* ZMQPP_POLLER_HPP_ */

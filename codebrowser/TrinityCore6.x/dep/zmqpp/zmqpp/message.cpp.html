42 * The two const_casts in size and raw_data are a little bit hacky
43 * but neither of these methods called this way actually modify data
44 * so accurately represent the intent of these calls.

47size_t message::size(size_t const part /* = 0 */) const
48{
49 if(part >= _parts.size())
50 {
51 throw exception("attempting to request a message part outside the valid range");
52 }
53
54 return _parts[part].size();
55}

57void const* message::raw_data(size_t const part /* = 0 */) const
58{
59 if(part >= _parts.size())

97// Move operators will take ownership of message parts without copying
98void message::move(void* part, size_t const size, release_function const& release)
99{
100 callback_releaser* hint = new callback_releaser();
101 hint->func = release;
102
103 _parts.push_back( frame( part, size, &message::release_callback, hint ) );
104}

422void message::copy(message const& source)
423{
424 _parts.resize( source._parts.size() );
425 for(size_t i = 0; i < source._parts.size(); ++i)
426 {
427 _parts[i] = source._parts[i].copy();
428 }
429
430 // we don't need a copy of the releasers as we did data copies of the internal data,
431 //_releasers = source._releasers;
432 //_strings = source._strings
433}

436void message::sent(size_t const part)
437{
438 // sanity check

440 _parts[part].mark_sent();
441}

443// Note that these releasers are not thread safe, the only safety is provided by
444// the socket class taking ownership so no updates can happen while zmq does it's thing
445// If used in a custom class this has to be dealt with.
446void message::release_callback(void* data, void* hint)
447{
448 callback_releaser* releaser = static_cast<callback_releaser*>(hint);
449 releaser->func(data);
450
451 delete releaser;
452}

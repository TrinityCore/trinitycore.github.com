121bool socket::send(message& message, bool const dont_block /* = false */)
122{
123 size_t parts = message.parts();
124 if (parts == 0)
125 {
126 throw std::invalid_argument("sending requires messages have at least one part");
127 }
128
129 bool dont_wait = dont_block;
130 for(size_t i = 0; i < parts; ++i)
131 {
132 int flag = socket::normal;
133 if(dont_wait) { flag |= socket::dont_wait; }
134 if(i < (parts - 1)) { flag |= socket::send_more; }
135
136#if (ZMQ_VERSION_MAJOR == 2)
137 int result = zmq_send( _socket, &message.raw_msg(i), flag );
138#elif (ZMQ_VERSION_MAJOR < 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR < 2))
139 int result = zmq_sendmsg( _socket, &message.raw_msg(i), flag );
140#else
141 int result = zmq_msg_send( &message.raw_msg(i), _socket, flag );
142#endif
143
144 if (result < 0)
145 {
146 // the zmq framework should not block if the first part is accepted
147 // so we should only ever get this error on the first part
148 if((0 == i) && (EAGAIN == zmq_errno()))
149 {
150 return false;
151 }
152
153 if(EINTR == zmq_errno())
154 {
155 if (0 == message.parts())
156 {
157 return false;
158 }
159
160 // If we have an interrupt but it's not on the first part then we
161 // know we can safely send out the rest of the message as we can
162 // enforce that it won't wait on a blocking action
163 dont_wait = true;
164 continue;
165 }
166
167 // sanity checking
168 assert(EAGAIN != zmq_errno());
169
170 throw zmq_internal_exception();
171 }
172
173 message.sent(i);
174 }
175
176 // Leave message reference in a stable state
177 zmqpp::message local;
178 std::swap(local, message);
179 return true;
180}

182bool socket::receive(message& message, bool const dont_block /* = false */)
183{
184 if (message.parts() > 0)
185 {
186 // swap and discard old message
187 zmqpp::message local;
188 std::swap(local, message);
189 }
190
191 int flags = (dont_block) ? socket::dont_wait : socket::normal;
192 bool more = true;
193
194 while(more)
195 {
196#if (ZMQ_VERSION_MAJOR == 2)
197 int result = zmq_recv( _socket, &_recv_buffer, flags );
198#elif (ZMQ_VERSION_MAJOR < 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR < 2))
199 int result = zmq_recvmsg( _socket, &_recv_buffer, flags );
200#else
201 int result = zmq_msg_recv( &_recv_buffer, _socket, flags );
202#endif
203
204 if(result < 0)
205 {
206 if ((0 == message.parts()) && (EAGAIN == zmq_errno()))
207 {
208 return false;
209 }
210
211 if(EINTR == zmq_errno())
212 {
213 if (0 == message.parts())
214 {
215 return false;
216 }
217
218 // If we have an interrupt but it's not on the first part then we
219 // know we can safely pull out the rest of the message as it will
220 // not be blocking
221 continue;
222 }
223
224 assert(EAGAIN != zmq_errno());
225
226 throw zmq_internal_exception();
227 }
228
229 zmq_msg_t& dest = message.raw_new_msg();
230 zmq_msg_move(&dest, &_recv_buffer);
231
232 get(socket_option::receive_more, more);
233 }
234
235 return true;
236}

244bool socket::receive(std::string& string, int const flags /* = NORMAL */)
245{
246#if (ZMQ_VERSION_MAJOR == 2)
247 int result = zmq_recv( _socket, &_recv_buffer, flags );
248#elif (ZMQ_VERSION_MAJOR < 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR < 2))
249 int result = zmq_recvmsg( _socket, &_recv_buffer, flags );
250#else
251 int result = zmq_msg_recv( &_recv_buffer, _socket, flags );
252#endif
253
254 if(result >= 0)
255 {
256 string.reserve(zmq_msg_size(&_recv_buffer));
257 string.assign(static_cast<char*>(zmq_msg_data(&_recv_buffer)), zmq_msg_size(&_recv_buffer));
258
259 return true;
260 }
261
262 if (EAGAIN == zmq_errno() || EINTR == zmq_errno())
263 {
264 return false;
265 }
266
267 throw zmq_internal_exception();
268}

271bool socket::send_raw(char const* buffer, int const length, int const flags /* = NORMAL */)
272{
273#if (ZMQ_VERSION_MAJOR == 2)
274 zmq_msg_t msg;
275 int result = zmq_msg_init_size(&msg, length);
276 if (result != 0)
277 {
278 zmq_internal_exception();
279 }
280
281 memcpy(zmq_msg_data(&msg), buffer, length);
282 result = zmq_send(_socket, &msg, flags);
283#else
284 int result = zmq_send(_socket, buffer, length, flags);
285#endif
286 if(result >= 0)
287 {
288 return true;
289 }
290
291#if (ZMQ_VERSION_MAJOR == 2)
292 // only actually need to close this on error
293 zmq_msg_close(&msg);
294#endif
295
296 if (EAGAIN == zmq_errno() || EINTR == zmq_errno())
297 {
298 return false;
299 }
300
301 throw zmq_internal_exception();
302}

304bool socket::receive_raw(char* buffer, int& length, int const flags /* = NORMAL */)
305{
306#if (ZMQ_VERSION_MAJOR == 2)
307 int result = zmq_recv( _socket, &_recv_buffer, flags );
308#elif (ZMQ_VERSION_MAJOR < 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR < 2))
309 int result = zmq_recvmsg( _socket, &_recv_buffer, flags );
310#else
311 int result = zmq_msg_recv( &_recv_buffer, _socket, flags );
312#endif
313
314 if(result >= 0)
315 {
316 length = zmq_msg_size(&_recv_buffer);
317 memcpy(buffer, zmq_msg_data(&_recv_buffer), length);
318
319 return true;
320 }
321
322 if (EAGAIN == zmq_errno() || EINTR == zmq_errno())
323 {
324 return false;
325 }
326
327 throw zmq_internal_exception();
328}

348// Set socket options for different types of option
349void socket::set(socket_option const option, int const value)
350{
351 switch(option)
352 {
353 // unsigned 64bit Integers
354#if (ZMQ_VERSION_MAJOR == 2)
355 case socket_option::high_water_mark:
356 case socket_option::send_buffer_size:
357 case socket_option::receive_buffer_size:
358#endif
359 case socket_option::affinity:
360 if (value < 0) { throw exception("attempting to set an unsigned 64 bit integer option with a negative integer"); }
361 set(option, static_cast<uint64_t>(value));
362 break;
363
364 // 64bit Integers
365#if (ZMQ_VERSION_MAJOR == 2)
366 case socket_option::rate:
367 case socket_option::recovery_interval:
368 case socket_option::recovery_interval_seconds:
369 case socket_option::swap_size:
370#else
371 case socket_option::max_messsage_size:
372#endif
373 set(option, static_cast<int64_t>(value));
374 break;
375
376 // Boolean
377#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 1))
378 case socket_option::ipv4_only:
379#endif
380#if (ZMQ_VERSION_MAJOR == 2)
381 case socket_option::multicast_loopback:
382#endif
383#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 2))
384#if (ZMQ_VERSION_MINOR == 2)
385 case socket_option::delay_attach_on_connect:
386#else
387 case socket_option::immediate:
388#endif
389 case socket_option::router_mandatory:
390 case socket_option::xpub_verbose:
391#endif
392 if (value == 0) { set(option, false); }
393 else if (value == 1) { set(option, true); }
394 else { throw exception("attempting to set a boolean option with a non 0 or 1 integer"); }
395 break;
396
397 // Default or Boolean
398#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 2))
399 case socket_option::tcp_keepalive:
400 if (value < -1 || value > 1) { throw exception("attempting to set a default or boolean option with a non -1, 0 or 1 integer"); }
401 if (0 != zmq_setsockopt(_socket, static_cast<int>(option), &value, sizeof(value)))
402 {
403 throw zmq_internal_exception();
404 }
405 break;
406#endif
407
408 // Integers that require +ve numbers
409#if (ZMQ_VERSION_MAJOR == 2)
410 case socket_option::reconnect_interval_max:
411#else
412 case socket_option::reconnect_interval_max:
413 case socket_option::send_buffer_size:
414 case socket_option::recovery_interval:
415 case socket_option::receive_buffer_size:
416 case socket_option::send_high_water_mark:
417 case socket_option::receive_high_water_mark:
418 case socket_option::multicast_hops:
419 case socket_option::rate:
420#endif
421 case socket_option::backlog:
422 if (value < 0) { throw exception("attempting to set a positive only integer option with a negative integer"); }
423 // Integers
424 case socket_option::reconnect_interval:
425 case socket_option::linger:
426 case socket_option::receive_timeout:
427 case socket_option::send_timeout:
428#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 2))
429 case socket_option::tcp_keepalive_idle:
430 case socket_option::tcp_keepalive_count:
431 case socket_option::tcp_keepalive_interval:
432#endif
433 if (0 != zmq_setsockopt(_socket, static_cast<int>(option), &value, sizeof(value)))
434 {
435 throw zmq_internal_exception();
436 }
437 break;
438 default:
439 throw exception("attempting to set a non signed integer option with a signed integer value");
440 }

541void socket::get(socket_option const option, int& value) const
542{
543 size_t value_size = sizeof(int);
544
545 switch(option)
546 {
547#if (ZMQ_VERSION_MAJOR == 2)
548 case socket_option::receive_more:
549 case socket_option::multicast_loopback:
550 value = static_cast<int>(get<int64_t>(option));
551 break;
552#endif
553 case socket_option::type:
554 case socket_option::linger:
555 case socket_option::backlog:
556 case socket_option::reconnect_interval:
557 case socket_option::reconnect_interval_max:
558 case socket_option::receive_timeout:
559 case socket_option::send_timeout:
560 case socket_option::file_descriptor:
561 case socket_option::events:
562#if (ZMQ_VERSION_MAJOR > 2)
563 case socket_option::receive_more:
564 case socket_option::send_buffer_size:
565 case socket_option::receive_buffer_size:
566 case socket_option::rate:
567 case socket_option::recovery_interval:
568 case socket_option::send_high_water_mark:
569 case socket_option::receive_high_water_mark:
570 case socket_option::multicast_hops:
571#endif
572#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 1))
573 case socket_option::ipv4_only:
574#endif
575#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 2))
576#if (ZMQ_VERSION_MINOR == 2)
577 case socket_option::delay_attach_on_connect:
578#else
579 case socket_option::immediate:
580#endif
581 case socket_option::tcp_keepalive:
582 case socket_option::tcp_keepalive_idle:
583 case socket_option::tcp_keepalive_count:
584 case socket_option::tcp_keepalive_interval:
585#endif
586#ifdef ZMQ_EXPERIMENTAL_LABELS
587 case socket_option::receive_label:
588#endif
589 if (0 != zmq_getsockopt(_socket, static_cast<int>(option), &value, &value_size))
590 {
591 throw zmq_internal_exception();
592 }
593
594 // sanity check
595 assert(value_size <= sizeof(int));
596 break;
597 default:
598 throw exception("attempting to get a non integer option with an integer value");
599 }

714socket::socket(socket&& source) NOEXCEPT
715 : _socket(source._socket)
716 , _type(source._type)
717 , _recv_buffer()
718{
719 // we steal the zmq_msg_t from the valid socket, we only init our own because it's cheap
720 // and zmq_msg_move does a valid check
721 zmq_msg_init(&_recv_buffer);
722 zmq_msg_move(&_recv_buffer, &source._recv_buffer);
723
724 // Clean up source a little, we will handle the deinit, it doesn't need to
725 source._socket = nullptr;
726}

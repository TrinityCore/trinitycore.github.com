14namespace zmqpp {
15
16frame::frame()
17 : _sent( false )
18{
19 if( 0 != zmq_msg_init( &_msg ) )
20 {
21 throw zmq_internal_exception();
22 }
23}
24
25frame::frame(size_t const size)
26 : _sent( false )
27{
28 if( 0 != zmq_msg_init_size( &_msg, size ) )
29 {
30 throw zmq_internal_exception();
31 }
32}
33
34frame::frame(void const* part, size_t const size)
35 : _sent( false )
36{
37 if( 0 != zmq_msg_init_size( &_msg, size ) )
38 {
39 throw zmq_internal_exception();
40 }
41
42 void* msg_data = zmq_msg_data( &_msg );
43 memcpy( msg_data, part, size );
44}
45
46frame::frame(void* part, size_t const size, zmq_free_fn *ffn, void *hint)
47 : _sent( false )
48{
49 if( 0 != zmq_msg_init_data( &_msg, part, size, ffn, hint ) )
50 {
51 throw zmq_internal_exception();
52 }
53}
54
55frame::~frame()
56{
57#ifndef NDEBUG // unused assert variable in release
58 int result = zmq_msg_close( &_msg );
59 assert(0 == result);
60#else
61 zmq_msg_close( &_msg );
62#endif // NDEBUG
63}
64
65frame::frame(frame&& other)
66 : _sent( other._sent )
67{
68 zmq_msg_init( &_msg );
69 zmq_msg_move( &_msg, &other._msg );
70 other._sent = false;
71}
72
73frame& frame::operator=(frame&& other)
74{
75 zmq_msg_init( &_msg );
76 zmq_msg_move( &_msg, &other._msg );
77 std::swap( _sent, other._sent );
78
79 return *this;
80}
81
82frame frame::copy() const
83{
84 frame other( size() );
85 other._sent = _sent;
86
87 if( 0 != zmq_msg_copy( &other._msg, const_cast<zmq_msg_t*>(&_msg) ) )
88 {
89 throw zmq_internal_exception();
90 }
91
92 return other;
93}

8#ifndef ZMQPP_MESSAGE_FRAME_HPP_
9#define ZMQPP_MESSAGE_FRAME_HPP_
10
11#include <zmq.h>
12
13#include "compatibility.hpp"
14
15namespace zmqpp {
16
17/*!
18 * \brief an internal frame wrapper for a single zmq message
19 *
20 * This frame wrapper consists of a zmq message and meta data it is used
21 * by the zmqpp message class to keep track of parts in the internal
22 * queue. It is unlikely you need to use this class.
23 */
24class frame
25{
26public:
27 frame();
28 frame(size_t const size);
29 frame(void const* part, size_t const size);
30 frame(void* part, size_t const size, zmq_free_fn *ffn, void *hint);
31
32 ~frame();
33
34 bool is_sent() const { return _sent; }
35 void const* data() const { return zmq_msg_data( const_cast<zmq_msg_t*>(&_msg) ); }
36 size_t size() const { return zmq_msg_size( const_cast<zmq_msg_t*>(&_msg) ); }
37
38 void mark_sent() { _sent = true; }
39 zmq_msg_t& msg() { return _msg; }
40
41 // Move operators
42 frame(frame&& other);
43 frame& operator=(frame&& other);
44
45 frame copy() const;
46
47private:
48 bool _sent;
49 zmq_msg_t _msg;
50
51 // Disable implicit copy support, code must request a copy to clone
52 frame(frame const&) NOEXCEPT ZMQPP_EXPLICITLY_DELETED;
53 frame& operator=(frame const&) NOEXCEPT ZMQPP_EXPLICITLY_DELETED;
54};
55
56} // namespace zmqpp
57
58#endif /* ZMQPP_MESSAGE_FRAME_HPP_ */

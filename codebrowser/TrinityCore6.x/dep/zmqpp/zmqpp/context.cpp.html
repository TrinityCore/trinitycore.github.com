13void context::terminate()
14{
15 int result;
16 do
17 {
18#if (ZMQ_VERSION_MAJOR < 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR < 2))
19 result = zmq_term(_context);
20#else
21 result = zmq_ctx_destroy(_context);
22#endif
23 } while (result != 0 && zmq_errno() == EINTR);
24 if (result != 0) { throw zmq_internal_exception(); }
25 _context = nullptr;
26}

28#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 2))
29void context::set(context_option const option, int const value)
30{
31 if (nullptr == _context) { throw invalid_instance("context is invalid"); }
32
33 if (0 != zmq_ctx_set(_context, static_cast<int>(option), value))
34 {
35 throw zmq_internal_exception();
36 }
37}
38
39int context::get(context_option const option)
40{
41 if (nullptr == _context) { throw invalid_instance("context is invalid"); }
42
43 int result = zmq_ctx_get(_context, static_cast<int>(option));
44
45 if (result < 0)
46 {
47 throw zmq_internal_exception();
48 }
49
50 return result;
51}
52#endif

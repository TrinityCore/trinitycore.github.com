11namespace zmqpp

13
14/** \todo Expand the information on the options to make it actually useful. */
15/*!
16 * \brief possible Socket options in zmq
17 */
18
19enum class socket_option {
20 affinity = ZMQ_AFFINITY, /*!< I/O thread affinity */
21 identity = ZMQ_IDENTITY, /*!< Socket identity */
22 subscribe = ZMQ_SUBSCRIBE, /*!< Add topic subscription - set only */
23 unsubscribe = ZMQ_UNSUBSCRIBE, /*!< Remove topic subscription - set only */
24 rate = ZMQ_RATE, /*!< Multicast data rate */
25 send_buffer_size = ZMQ_SNDBUF, /*!< Kernel transmission buffer size */
26 receive_buffer_size = ZMQ_RCVBUF, /*!< Kernel receive buffer size */
27 receive_more = ZMQ_RCVMORE, /*!< Can receive more parts - get only */
28 file_descriptor = ZMQ_FD, /*!< Socket file descriptor - get only */
29 events = ZMQ_EVENTS, /*!< Socket event flags - get only */
30 type = ZMQ_TYPE, /*!< Socket type - get only */
31 linger = ZMQ_LINGER, /*!< Socket linger timeout */
32 backlog = ZMQ_BACKLOG, /*!< Maximum length of outstanding connections - get only */
33#if (ZMQ_VERSION_MAJOR == 2)
34 // Note that this is inverse of the zmq names for version 2.x
35 recovery_interval_seconds = ZMQ_RECOVERY_IVL, /*!< Multicast recovery interval in seconds */
36 recovery_interval = ZMQ_RECOVERY_IVL_MSEC, /*!< Multicast recovery interval in milliseconds */
37#else
38 recovery_interval = ZMQ_RECOVERY_IVL, /*!< Multicast recovery interval in milliseconds */
39#endif
40 reconnect_interval = ZMQ_RECONNECT_IVL, /*!< Reconnection interval */
41 reconnect_interval_max = ZMQ_RECONNECT_IVL_MAX, /*!< Maximum reconnection interval */
42 receive_timeout = ZMQ_RCVTIMEO, /*!< Maximum inbound block timeout */
43 send_timeout = ZMQ_SNDTIMEO, /*!< Maximum outbound block timeout */
44#if (ZMQ_VERSION_MAJOR == 2)
45 high_water_mark = ZMQ_HWM, /*!< High-water mark for all messages */
46 swap_size = ZMQ_SWAP, /*!< Maximum socket swap size in bytes */
47 multicast_loopback = ZMQ_MCAST_LOOP, /*!< Allow multicast packet loopback */
48#else
49 max_messsage_size = ZMQ_MAXMSGSIZE, /*!< Maximum inbound message size */
50 send_high_water_mark = ZMQ_SNDHWM, /*!< High-water mark for outbound messages */
51 receive_high_water_mark = ZMQ_RCVHWM, /*!< High-water mark for inbound messages */
52 multicast_hops = ZMQ_MULTICAST_HOPS, /*!< Maximum number of multicast hops */
53#endif
54#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 1))
55 ipv4_only = ZMQ_IPV4ONLY,
56#endif
57#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 2))
58#if (ZMQ_VERSION_MINOR == 2)
59 delay_attach_on_connect = ZMQ_DELAY_ATTACH_ON_CONNECT, /*!< Delay buffer attachment until connect complete */
60#else
61 // ZMQ_DELAY_ATTACH_ON_CONNECT is renamed in ZeroMQ starting 3.3.x
62 immediate = ZMQ_IMMEDIATE, /*!< Block message sending until connect complete */
63#endif
64 last_endpoint = ZMQ_LAST_ENDPOINT, /*!< Last bound endpoint - get only */
65 router_mandatory = ZMQ_ROUTER_MANDATORY, /*!< Require routable messages - set only */
66 xpub_verbose = ZMQ_XPUB_VERBOSE, /*!< Pass on existing subscriptions - set only */
67 tcp_keepalive = ZMQ_TCP_KEEPALIVE, /*!< Enable TCP keepalives */
68 tcp_keepalive_idle = ZMQ_TCP_KEEPALIVE_IDLE, /*!< TCP keepalive idle count (generally retry count) */
69 tcp_keepalive_count = ZMQ_TCP_KEEPALIVE_CNT, /*!< TCP keepalive retry count */
70 tcp_keepalive_interval = ZMQ_TCP_KEEPALIVE_INTVL, /*!< TCP keepalive interval */
71 tcp_accept_filter = ZMQ_TCP_ACCEPT_FILTER, /*!< Filter inbound connections - set only */
72#endif
73#ifdef ZMQ_EXPERIMENTAL_LABELS
74 receive_label = ZMQ_RCVLABEL, /*!< Received label part - get only */
75#endif
76};
77
78}

80#endif /* ZMQPP_SOCKET_OPTIONS_HPP_ */

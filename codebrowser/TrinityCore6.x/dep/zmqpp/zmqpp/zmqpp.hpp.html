60/*!
61 * \brief C++ wrapper around zmq
62 *
63 * All zmq++ / zmqpp functions, constants and classes live within this namespace,
64 */
65namespace zmqpp
66{
67
68/*!
69 * Returns the current major.minor.revision version number as a string.
70 *
71 * \return string version number.
72 */
73std::string version();
74
75/*!
76 * Retrieve the parts of the zmqpp version number.
77 *
78 * Set the three parameters to values representing the zmqpp version number.
79 * These values are generated at library compile time.
80 *
81 * \param major an unsigned 8 bit reference to set to the major version.
82 * \param minor an unsigned 8 bit reference to set to the minor version.
83 * \param revision an unsigned 8 bit reference to set the current revision.
84 */
85void version(uint8_t& major, uint8_t& minor, uint8_t& revision);
86
87/*!
88 * Retrieve the parts of the 0mq version this library was built against.
89 *
90 * Because sections of the library are optionally compiled in or ignored
91 * depending on the version of 0mq it was compiled against this method is
92 * provided to allow sanity checking for usage.
93 *
94 * Set the three parameters to values representing the 0mq version number.
95 * These values are generated at library compile time.
96 *
97 * \param major an unsigned 8 bit reference to set to the major version.
98 * \param minor an unsigned 8 bit reference to set to the minor version.
99 * \param revision an unsigned 8 bit reference to set the current revision.
100 */
101void zmq_version(uint8_t& major, uint8_t& minor, uint8_t& patch);
102
103typedef context context_t; /*!< \brief context type */
104typedef std::string endpoint_t; /*!< \brief endpoint type */
105typedef message message_t; /*!< \brief message type */
106typedef poller poller_t; /*!< \brief poller type */
107typedef socket socket_t; /*!< \brief socket type */
108
109}

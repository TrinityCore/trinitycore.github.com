26/*!
27 * \brief a zmq message with optional multipart support
28 *
29 * A zmq message is made up of one or more parts which are sent together to
30 * the target endpoints. zmq guarantees either the whole message or none
31 * of the message will be delivered.
32 */
33class message
34{
35public:
36 /*!
37 * \brief callback to release user allocated data.
38 *
39 * The release function will be called on any void* moved part.
40 * It must be thread safe to the extent that the callback may occur on
41 * one of the context threads.
42 *
43 * The function called will be passed a single variable which is the
44 * pointer to the memory allocated.
45 */
46 typedef std::function<void (void*)> release_function;
47
48 message();
49 ~message();
50
51 template <typename T, typename ...Args>
52 message(T const &part, Args &&...args)
53 : message()
54 {
55 add(part, std::forward<Args>(args)...);
56 }
57
58 size_t parts() const;
59 size_t size(size_t const part) const;
60 std::string get(size_t const part) const;
61
62 void get(int8_t& integer, size_t const part) const;
63 void get(int16_t& integer, size_t const part) const;
64 void get(int32_t& integer, size_t const part) const;
65 void get(int64_t& integer, size_t const part) const;
66
67 void get(uint8_t& unsigned_integer, size_t const part) const;
68 void get(uint16_t& unsigned_integer, size_t const part) const;
69 void get(uint32_t& unsigned_integer, size_t const part) const;
70 void get(uint64_t& unsigned_integer, size_t const part) const;
71
72 void get(float& floating_point, size_t const part) const;
73 void get(double& double_precision, size_t const part) const;
74 void get(bool& boolean, size_t const part) const;
75
76 void get(std::string& string, size_t const part) const;
77
78 // Warn: If a pointer type is requested the message (well zmq) still 'owns'
79 // the data and will release it when the message object is freed.
80 template<typename Type>
81 Type get(size_t const part)
82 {
83 Type value;
84 get(value, part);
85 return value;
86 }
87
88 template<int part=0, typename T, typename ...Args>
89 void extract(T &nextpart, Args &...args)
90 {
91 assert(part < parts());
92 get(nextpart,part);
93 extract<part+1>(args...);
94 }
95
96 template<int part=0, typename T>
97 void extract(T &nextpart)
98 {
99 assert(part < parts());
100 get(nextpart,part);
101 }
102
103 // Raw get data operations, useful with data structures more than anything else
104 // Warn: The message (well zmq) still 'owns' the data and will release it
105 // when the message object is freed.
106 template<typename Type>
107 void get(Type*& value, size_t const part) const
108 {
109 value = static_cast<Type*>(raw_data(part));
110 }
111
112 // Warn: The message (well zmq) still 'owns' the data and will release it
113 // when the message object is freed.
114 template<typename Type>
115 void get(Type** value, size_t const part) const
116 {
117 *value = static_cast<Type*>(raw_data(part));
118 }
119
120 // Move operators will take ownership of message parts without copying
121 void move(void* part, size_t const size, release_function const& release);
122
123 // Raw move data operation, useful with data structures more than anything else
124 template<typename Object>
125 void move(Object *part)
126 {
127 move(part, sizeof(Object), &deleter_callback<Object>);
128 }
129
130 // Copy operators will take copies of any data
131 template<typename Type>
132 void add(Type *part, size_t const size)
133 {
134 _parts.push_back( frame( part, size ) );
135 }
136
137
138 template<typename Type, typename ...Args>
139 void add(Type const& part, Args &&...args)
140 {
141 *this << part;
142 add(std::forward<Args>(args)...);
143 }
144
145 template<typename Type>
146 void add(Type const part)
147 {
148 *this << part;
149 }
150
151 // Stream reader style
152 void reset_read_cursor();
153
154 template<typename Type>
155 message& operator>>(Type& value)
156 {
157 get(value, _read_cursor++);
158 return *this;
159 }

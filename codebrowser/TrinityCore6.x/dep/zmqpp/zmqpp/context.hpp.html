21namespace zmqpp
22{
23
24/*!
25 * The context class represents internal zmq context and io threads.
26 *
27 * By default the context class will create one thread, however this can be
28 * overridden in the constructor.
29 *
30 * The context class is the only object that can be considered thread safe.
31 *
32 * All sockets using endpoints other than inproc require the context to have
33 * at least one thread.
34 *
35 * This class is c++0x move supporting and cannot be copied.
36 */
37class context
38{
39public:
40
41#if (ZMQ_VERSION_MAJOR < 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR < 2))
42 /*!
43 * Initialise the 0mq context.
44 *
45 * If only inproc is used then the context may be created with zero threads.
46 * Any inproc endpoint using sockets must be created using the same context.
47 *
48 * The context is thread safe an may be used anywhere in your application,
49 * however there is no requirement (other than inproc restrictions) for you
50 * to do this.
51 *
52 * \param threads an integer argument for the number of required threads. Defaults to 1.
53 */
54 context(int const& threads = 1)
55#else
56 /*!
57 * Initialise the 0mq context.
58 *
59 * The context is thread safe an may be used anywhere in your application,
60 * however there is no requirement (other than inproc restrictions) for you
61 * to do this.
62 */
63 context()
64#endif
65 : _context(nullptr)
66 {
67#if (ZMQ_VERSION_MAJOR < 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR < 2))
68 _context = zmq_init(threads);
69#else
70 _context = zmq_ctx_new();
71#endif
72
73 if (nullptr == _context)
74 {
75 throw zmq_internal_exception();
76 }
77 }
78
79 /*!
80 * Closes the 0mq context.
81 *
82 * Any blocking calls other than a socket close will return with an error.
83 *
84 * If there are open sockets will block while zmq internal buffers are
85 * processed up to a limit specified by that sockets linger option.
86 */
87 ~context() NOEXCEPT
88 {
89 if (nullptr != _context)
90 {
91 terminate();
92 }
93 }
94
95 /*!
96 * Move supporting constructor.
97 *
98 * Allows zero-copy move semantics to be used with this class.
99 *
100 * \param source a rvalue instance of the object who's internals we wish to steal.
101 */
102 context(context&& source) NOEXCEPT
103 : _context(source._context)
104 {
105 source._context = nullptr;
106 }
107
108 /*!
109 * Move supporting operator.
110 *
111 * Allows zero-copy move semantics to be used with this class.
112 *
113 * \param source an rvalue instance of the context who's internals we wish to steal.
114 */
115 context& operator=(context&& source) NOEXCEPT
116 {
117 std::swap( _context, source._context );
118 return *this;
119 }
120
121 /*!
122 * Terminate the current context.
123 *
124 * Any blocking calls other than a socket close will return with an error.
125 *
126 * If there are open sockets will block while zmq internal buffers are
127 * processed up to a limit specified by that sockets linger option.
128 */
129 void terminate();
130
131#if (ZMQ_VERSION_MAJOR > 3) || ((ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR >= 2))
132 /*!
133 * Set the value of an option in the underlaying zmq context.
134 *
135 * \param option a valid ::context_option
136 * \param value to set the option to
137 */
138 void set(context_option const option, int const value);
139
140 /*!
141 * Get a context option from the underlaying zmq context.
142 *
143 * \param option a valid ::context_option
144 * \return context option value
145 */
146 int get(context_option const option);
147#endif
148
149 /*!
150 * Validity checking of the context
151 *
152 * Checks if the underlying 0mq context for this instance is valid.
153 *
154 * Contexts should always be valid unless people are doing 'fun' things with
155 * std::move.
156 *
157 * \return boolean true if the object is valid.
158 */
159 operator bool() const NOEXCEPT
160 {
161 return nullptr != _context;
162 }
163
164 /*!
165 * Access to the raw 0mq context
166 *
167 * \return void pointer to the underlying 0mq context.
168 */
169 operator void*() const NOEXCEPT
170 {
171 return _context;
172 }
173
174private:
175 void* _context;
176
177 // No copy - private and not implemented
178 context(context const&) ZMQPP_EXPLICITLY_DELETED;
179 context& operator=(context const&) NOEXCEPT ZMQPP_EXPLICITLY_DELETED;
180};
181
182}
183
184#endif /* ZMQPP_CONTEXT_HPP_ */

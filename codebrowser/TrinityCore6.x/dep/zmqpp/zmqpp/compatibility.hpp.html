2 * \file

4 * \date 10 Sep 2011
5 * \author ron
6 * \author Ben Gray (\@benjamg)
7 *
8 * A fair number of C++0x (or more accurately C++11) features are used in this
9 * library and as this project is used where I work on older compilers this
10 * file was created to help.
11 *
12 * C++ features and their workaround where not supported:
13 * \li lambda functions - disabled, these are only used in the test anyway.
14 * \li typesafe enums - replaced with enum where comparisons needed.
15 * \li nullptr - defined to null.
16 *
17 * As of the port to version 3.1 (libzmqpp version 1.1.0) this file will also
18 * be used to maintain compatablity with multiple versions of 0mq
19 */
20
21#ifndef ZMQPP_COMPATIBILITY_HPP_
22#define ZMQPP_COMPATIBILITY_HPP_
23
24#include <zmq.h>
25#include <cstdint>
26
27// Currently we require at least 0mq version 2.2.x
28#define ZMQPP_REQUIRED_ZMQ_MAJOR 2
29#define ZMQPP_REQUIRED_ZMQ_MINOR 2
30
31#if (ZMQ_VERSION_MAJOR < ZMQPP_REQUIRED_ZMQ_MAJOR) || ((ZMQ_VERSION_MAJOR == ZMQPP_REQUIRED_ZMQ_MAJOR) && (ZMQ_VERSION_MINOR < ZMQPP_REQUIRED_ZMQ_MINOR))
32#error zmqpp requires a later version of 0mq
33#endif
34
35// Experimental feature support
36#if (ZMQ_VERSION_MAJOR == 3) && (ZMQ_VERSION_MINOR == 0)
37#define ZMQ_EXPERIMENTAL_LABELS
38#endif
39
40// Deal with older versions of gcc
41#if defined(__GNUC__) && !defined(__clang__)
42#if __GNUC__ == 4
43
44// Deal with older gcc not supporting C++0x typesafe enum class name {} comparison
45#if __GNUC_MINOR__ < 4
46#define ZMQPP_COMPARABLE_ENUM enum
47#endif
48
49#if __GNUC_MINOR__ == 4
50#if __GNUC_PATCHLEVEL__ < 1
51#undef ZMQPP_COMPARABLE_ENUM
52#define ZMQPP_COMPARABLE_ENUM enum
53#endif // if __GNUC_PATCHLEVEL__ < 1
54#endif // if __GNUC_MINOR__ == 4
55
56// Deal with older gcc not supporting C++0x lambda function
57#if __GNUC_MINOR__ < 5
58#define ZMQPP_IGNORE_LAMBDA_FUNCTION_TESTS
59#define ZMQPP_EXPLICITLY_DELETED
60#endif // if __GNUC_MINOR__ < 5
61
62// Deal with older gcc not supporting C++0x nullptr
63#if __GNUC_MINOR__ < 6
64#define nullptr NULL
65#define NOEXCEPT
66#endif // if __GNUC_MINOR__ < 6
67
68#endif // if __GNUC_ == 4
69#endif // if defined(__GNUC__) && !defined(__clang__)
70
71#if defined(_MSC_VER)
72#define NOEXCEPT throw()
73#if _MSC_VER < 1800
74#define ZMQPP_EXPLICITLY_DELETED
75#endif // if _MSC_VER < 1800
76#if _MSC_VER < 1600
77#define nullptr NULL
78#define ZMQPP_IGNORE_LAMBDA_FUNCTION_TESTS
79#define ZMQPP_COMPARABLE_ENUM enum
80#endif // if _MSC_VER < 1600
81#endif // if defined(_MSC_VER)
82
83// Generic state, assume a modern compiler
84#ifndef ZMQPP_COMPARABLE_ENUM
85#define ZMQPP_COMPARABLE_ENUM enum class
86#endif
87
88#ifndef ZMQPP_EXPLICITLY_DELETED
89#define ZMQPP_EXPLICITLY_DELETED = delete
90#endif
91
92#ifndef NOEXCEPT
93#define NOEXCEPT noexcept
94#endif
95
96#endif /* ZMQPP_COMPATIBILITY_HPP_ */

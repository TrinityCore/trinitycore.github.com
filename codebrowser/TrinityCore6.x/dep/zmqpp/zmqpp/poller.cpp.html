36void poller::add(socket& socket, short const event /* = POLL_IN */)
37{
38 zmq_pollitem_t item { socket, 0, event, 0 };
39
40 size_t index = _items.size();
41 _items.push_back(item);
42 _index[socket] = index;

64void poller::reindex(size_t const index)
65{
66 if ( nullptr != _items[index].socket )
67 {
68 auto found = _index.find( _items[index].socket );
69 if (_index.end() == found) { throw exception("unable to reindex socket in poller"); }
70 found->second = index;
71 }
72 else
73 {
74 auto found = _fdindex.find( _items[index].fd );
75 if (_fdindex.end() == found) { throw exception("unable to reindex file descriptor in poller"); }
76 found->second = index;
77 }
78}

80void poller::remove(socket_t const& socket)
81{
82 auto found = _index.find(socket);
83 if (_index.end() == found) { return; }
84
85 if ( _items.size() - 1 == found->second )
86 {
87 _items.pop_back();
88 _index.erase(found);
89 return;
90 }
91
92 std::swap(_items[found->second], _items.back());
93 _items.pop_back();
94
95 auto index = found->second;
96 _index.erase(found);
97
98 reindex( index );
99}

144bool poller::poll(long timeout /* = WAIT_FOREVER */)
145{
146 int result = zmq_poll(_items.data(), _items.size(), timeout);
147 if (result < 0)
148 {
149 if(EINTR == zmq_errno())
150 {
151 return false;
152 }
153
154 throw zmq_internal_exception();
155 }
156
157 return (result > 0);
158}

160short poller::events(socket const& socket) const
161{
162 auto found = _index.find(socket);
163 if (_index.end() == found)
164 {
165 throw exception("this socket is not represented within this poller");

167

23 Patcher::Patcher(boost::filesystem::path file)
24 : filePath(file)
25 {
26 ReadFile();
27 binaryType = Helper::GetBinaryType(binary);
28 }
29
30 void Patcher::ReadFile()
31 {
32 std::ifstream ifs(filePath.string(), std::ifstream::binary);
33 if (!ifs)
34 throw std::runtime_error("could not open " + filePath.string());
35
36 binary.clear();
37 ifs >> std::noskipws;
38 ifs.seekg(0, std::ios_base::end);
39 binary.reserve(ifs.tellg());
40 ifs.seekg(0, std::ios_base::beg);
41
42 std::copy(std::istream_iterator<unsigned char>(ifs), std::istream_iterator<unsigned char>(), std::back_inserter(binary));
43 }
44
45 void Patcher::WriteFile(boost::filesystem::path const& path)
46 {
47 std::ofstream ofs(path.string(), std::ofstream::binary);
48 if (!ofs)
49 throw std::runtime_error("could not open " + path.string());
50
51 ofs << std::noskipws;
52
53 std::copy(binary.begin(), binary.end(), std::ostream_iterator<unsigned char>(ofs));
54 }
55
56 void Patcher::Patch(std::vector<unsigned char> const& bytes, std::vector<unsigned char> const& pattern)
57 {
58 if (binary.size() < pattern.size())
59 throw std::logic_error("pattern larger than binary");
60
61 if (pattern.empty())
62 return;
63
64 for (size_t const offset : Helper::SearchOffset(binary, pattern))
65 {
66 std::cout << "Found offset " << offset << std::endl;
67
68 if (offset != 0 && binary.size() >= bytes.size())
69 for (size_t i = 0; i < bytes.size(); i++)
70 binary[offset + i] = bytes[i];
71 }
72 }
73
74 void Patcher::Finish(boost::filesystem::path out)
75 {
76 if (boost::filesystem::exists(out))
77 boost::filesystem::remove(out);
78
79 WriteFile(out);
80 }

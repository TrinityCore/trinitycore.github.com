47 template<typename PATCH, typename PATTERN>
48 void PatchModule(boost::filesystem::path file, boost::filesystem::path path)
49 {
50 namespace fs = boost::filesystem;
51
52 std::cout << "Patching module...\n";
53
54 Patcher patcher(file);
55
56 std::cout << "patching Password\n";
57 // if (Authentication::ServerSignature::ClientValidateProof(x)) to if (true)
58 patcher.Patch(PATCH::Password(), PATTERN::Password());
59
60 std::string const moduleName(Helper::GetFileChecksum(patcher.GetBinary()) + ".auth");
61 fs::path const modulePath
62 (path / std::string(&moduleName[0], 2) / std::string(&moduleName[2], 2));
63
64 if (!fs::exists(modulePath))
65 fs::create_directories(modulePath);
66
67 if (fs::exists(modulePath / moduleName))
68 fs::permissions(modulePath / moduleName, fs::add_perms | fs::others_write | fs::group_write | fs::owner_write);
69 patcher.Finish(modulePath / moduleName);
70 fs::permissions(modulePath / moduleName, fs::remove_perms | fs::others_write | fs::group_write | fs::owner_write);
71
72 std::cout << "Patching module finished.\n";
73 }

75 template<typename PATCH, typename PATTERN>
76 void do_module(const std::string& moduleName, const boost::filesystem::path& path)
77 {
78 boost::filesystem::path const modulePath
79 (path / std::string(&moduleName[0], 2) / std::string(&moduleName[2], 2));
80 boost::filesystem::path const module(modulePath / moduleName);
81
82 if (!boost::filesystem::exists(module))
83 {
84 std::cout << "Base module doesn't exist, downloading it...\n";
85
86 if (!boost::filesystem::exists(modulePath))
87 boost::filesystem::create_directories(modulePath);
88
89 std::ofstream outFile(module.string(), std::ofstream::out | std::ofstream::binary);
90 Helper::DownloadFile("xx.depot.battle.net", 1119, "/" + moduleName, outFile);
91 outFile.close();
92 std::cout << "Done.\n";
93 }
94
95 PatchModule<PATCH, PATTERN>(module, path);
96 }

98 template<typename PATCH, typename PATTERN>
99 void do_patches(Patcher* patcher, boost::filesystem::path output, bool patchVersionPath, uint32_t buildNumber)
100 {
101 std::cout << "patching Portal\n";
102 // '.logon.battle.net' -> '' to allow for set portal 'host'
103 patcher->Patch(Patches::Common::Portal(), Patterns::Common::Portal());
104
105 std::cout << "patching redirect RSA Modulus\n";
106 // public component of connection signing key to use known key pair
107 patcher->Patch(Patches::Common::Modulus(), Patterns::Common::Modulus());
108
109 std::cout << "patching BNet\n";
110 // hardcode 213.248.127.130 in IP6::Address::Address(IP4::Address::Address const&)
111 // used in Creep::Layer::Authentication::Online(), which overwrites GameStream::Connection::GetAddressRemote()
112 // to avoid CRYPT_SERVER_ADDRESS_IPV6 check in module
113 patcher->Patch(PATCH::BNet(), PATTERN::BNet());
114
115 std::cout << "patching Signature\n";
116 // if (Authentication::ModuleSignature::Validator::IsValid(x)) to if (true) in
117 // Creep::Instance::LoadModule() to allow for unsigned auth module
118 patcher->Patch(PATCH::Signature(), PATTERN::Signature());
119
120 if (patchVersionPath)
121 {
122 std::cout << "patching Versions\n";
123 // sever the connection to blizzard's versions file to stop it from updating and replace with custom version
124 // hardcode %s.patch.battle.net:1119/%s/versions to trinity6.github.io/%s/%s/build/versi
125 std::string verPatch(Patches::Common::VersionsFile());
126 std::string buildPattern = "build";
127
128 boost::algorithm::replace_all(verPatch, buildPattern, std::to_string(buildNumber));
129 std::vector<unsigned char> verVec(verPatch.begin(), verPatch.end());
130 patcher->Patch(verVec, Patterns::Common::VersionsFile());
131 }
132
133 patcher->Finish(output);
134
135 std::cout << "Patching done.\n";
136 }
137 }

139 po::variables_map GetConsoleArguments(int argc, char** argv)
140 {
141 po::options_description all("Allowed options");
142 all.add_options()
143 ("help,h", "print usage message")
144 ("path", po::value<std::string>()->required(), "Path to the Wow.exe")
145 ("extra,e", po::value<uint32_t>()->implicit_value(0), "Enable patching of versions file download path. Version can be specified explicitly.")
146 ("modulePath,m", po::value<std::string>(), "Path to the Battle.net module download destination.")
147 ;
148
149 po::positional_options_description pos;
150 pos.add("path", 1);
151
152 po::variables_map vm;
153 try
154 {
155 po::store(po::command_line_parser(argc, argv).options(all).positional(pos).run(), vm);
156 po::notify(vm);
157 }
158 catch (std::exception& e)
159 {
160 std::cerr << e.what() << "\n";
161 }
162
163 if (vm.count("help"))
164 std::cout << all << "\n";
165
166 if (!vm.count("path"))
167 throw std::invalid_argument("Wrong number of arguments: Missing client file.");
168
169 return vm;
170 }

173int main(int argc, char** argv)
174{
175 using namespace Connection_Patcher;
176
177 try
178 {
179 bool patchVersionPath = false;
180 int wowBuild = 0;
181
182 auto vm = GetConsoleArguments(argc, argv);
183
184 // exit if help is enabled
185 if (vm.count("help"))
186 {
187 std::cin.get();
188 return 0;
189 }
190
191 std::string const binary_path(std::move(vm["path"].as<std::string>()));
192 std::string renamed_binary_path(binary_path);
193 std::wstring appDataPath;
194
195#if PLATFORM == PLATFORM_WINDOWS
196 wchar_t* tempPath(nullptr);
197 SHGetKnownFolderPath(FOLDERID_ProgramData, 0, NULL, &tempPath);
198 appDataPath = std::wstring(tempPath);
199#elif PLATFORM == PLATFORM_UNIX
200 char* tempPath(nullptr);
201 if ((tempPath = getenv("HOME")) == nullptr)
202 tempPath = getpwuid(getuid())->pw_dir;
203 std::string tempPathStr(tempPath);
204 appDataPath.assign(tempPathStr.begin(), tempPathStr.end());
205 appDataPath += std::wstring(L"/.wine/drive_c/users/Public/Application Data");
206#endif
207 if (vm.count("modulePath"))
208 appDataPath.assign(vm["modulePath"].as<std::string>().begin(), vm["modulePath"].as<std::string>().end());
209
210 std::cout << "Creating patched binary..." << std::endl;
211
212 Patcher patcher(binary_path);
213
214 if (vm.count("extra"))
215 {
216 patchVersionPath = true;
217 wowBuild = vm["extra"].as<uint32_t>();
218
219 if (wowBuild == 0)
220 wowBuild = Helper::GetBuildNumber(patcher.GetBinary());
221
222 if (wowBuild == 0 || wowBuild < 10000 || wowBuild > 65535) // Build number has to be exactly 5 characters long
223 throw std::runtime_error("Could not retrieve build number or it was out of range. Build: " + std::to_string(wowBuild));
224
225 std::cout << "Determined build number: " << std::to_string(wowBuild) << std::endl;
226 }
227
228 switch (patcher.GetType())
229 {
230 case Constants::BinaryTypes::Pe32:
231 std::cout << "Win32 client...\n";
232
233 boost::algorithm::replace_all(renamed_binary_path, ".exe", "_Patched.exe");
234 do_patches<Patches::Windows::x86, Patterns::Windows::x86>
235 (&patcher, renamed_binary_path, patchVersionPath, wowBuild);
236
237 do_module<Patches::Windows::x86, Patterns::Windows::x86>
238 ( "8f52906a2c85b416a595702251570f96d3522f39237603115f2f1ab24962043c.auth"
239 , std::wstring(appDataPath) + std::wstring(L"/Blizzard Entertainment/Battle.net/Cache/")
240 );
241
242 break;
243 case Constants::BinaryTypes::Pe64:
244 std::cout << "Win64 client...\n";
245
246 boost::algorithm::replace_all(renamed_binary_path, ".exe", "_Patched.exe");
247 do_patches<Patches::Windows::x64, Patterns::Windows::x64>
248 (&patcher, renamed_binary_path, patchVersionPath, wowBuild);
249
250 do_module<Patches::Windows::x64, Patterns::Windows::x64>
251 ( "0a3afee2cade3a0e8b458c4b4660104cac7fc50e2ca9bef0d708942e77f15c1d.auth"
252 , std::wstring(appDataPath) + std::wstring(L"/Blizzard Entertainment/Battle.net/Cache/")
253 );
254
255 break;
256 case Constants::BinaryTypes::Mach64:
257 std::cout << "Mac client...\n";
258
259 boost::algorithm::replace_all (renamed_binary_path, ".app", " Patched.app");
260 Helper::CopyDir(boost::filesystem::path(binary_path).parent_path()/*MacOS*/.parent_path()/*Contents*/.parent_path()
261 , boost::filesystem::path(renamed_binary_path).parent_path()/*MacOS*/.parent_path()/*Contents*/.parent_path()
262 );
263
264 do_patches<Patches::Mac::x64, Patterns::Mac::x64>
265 (&patcher, renamed_binary_path, patchVersionPath, wowBuild);
266
267 {
268 namespace fs = boost::filesystem;
269 fs::permissions(renamed_binary_path, fs::add_perms | fs::others_exe | fs::group_exe | fs::owner_exe);
270 }
271
272 do_module<Patches::Mac::x64, Patterns::Mac::x64>
273 ( "97eeb2e28e9e56ed6a22d09f44e2ff43c93315e006bbad43bafc0defaa6f50ae.auth"
274 , "/Users/Shared/Blizzard/Battle.net/Cache/"
275 );
276
277 break;
278 default:
279 throw std::runtime_error("Type: " + std::to_string(static_cast<uint32_t>(patcher.GetType())) + " not supported!");
280 }
281
282 std::cout << "Successfully created your patched binaries.\n";
283
284 return 0;
285 }
286 catch (std::exception const& ex)
287 {
288 std::cerr << "EX: " << ex.what() << std::endl;
289 std::cerr << "An error occurred. Press ENTER to continue...";
290 std::cin.get();
291 return 1;
292 }

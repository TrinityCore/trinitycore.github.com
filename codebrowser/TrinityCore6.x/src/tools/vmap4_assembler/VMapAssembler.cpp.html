19#include <string>
20#include <iostream>
21
22#include "TileAssembler.h"
23
24int main(int argc, char* argv[])
25{
26 if (argc != 3)
27 {
28 std::cout << "usage: " << argv[0] << " <raw data dir> <vmap dest dir>" << std::endl;
29 return 1;
30 }
31
32 std::string src = argv[1];
33 std::string dest = argv[2];
34
35 std::cout << "using " << src << " as source directory and writing output to " << dest << std::endl;
36
37 VMAP::TileAssembler* ta = new VMAP::TileAssembler(src, dest);
38
39 if (!ta->convertWorld2())
40 {
41 std::cout << "exit with errors" << std::endl;
42 delete ta;
43 return 1;
44 }
45
46 delete ta;
47 std::cout << "Ok, all done" << std::endl;
48 return 0;
49}

23DBCFile::DBCFile(HANDLE mpq, const char* filename) :
24 _mpq(mpq), _filename(filename), _file(NULL), _recordSize(0), _recordCount(0),
25 _fieldCount(0), _stringSize(0), _data(NULL), _stringTable(NULL)
26{
27}
28
29bool DBCFile::open()
30{
31 if (!CascOpenFile(_mpq, _filename, CASC_LOCALE_NONE, 0, &_file))
32 return false;
33
34 char header[4];
35 unsigned int na, nb, es, ss;
36
37 DWORD readBytes = 0;
38 CascReadFile(_file, header, 4, &readBytes);
39 if (readBytes != 4) // Number of records
40 return false;
41
42 if (header[0] != 'W' || header[1] != 'D' || header[2] != 'B' || header[3] != 'C')
43 return false;
44
45 readBytes = 0;
46 CascReadFile(_file, &na, 4, &readBytes);
47 if (readBytes != 4) // Number of records
48 return false;
49
50 readBytes = 0;
51 CascReadFile(_file, &nb, 4, &readBytes);
52 if (readBytes != 4) // Number of fields
53 return false;
54
55 readBytes = 0;
56 CascReadFile(_file, &es, 4, &readBytes);
57 if (readBytes != 4) // Size of a record
58 return false;
59
60 readBytes = 0;
61 CascReadFile(_file, &ss, 4, &readBytes);
62 if (readBytes != 4) // String size
63 return false;
64
65 _recordSize = es;
66 _recordCount = na;
67 _fieldCount = nb;
68 _stringSize = ss;
69 if (_fieldCount * 4 != _recordSize)
70 return false;
71
72 _data = new unsigned char[_recordSize * _recordCount + _stringSize];
73 _stringTable = _data + _recordSize*_recordCount;
74
75 size_t data_size = _recordSize * _recordCount + _stringSize;
76 readBytes = 0;
77 CascReadFile(_file, _data, data_size, &readBytes);
78 if (readBytes != data_size)
79 return false;
80
81 return true;
82}

84DBCFile::~DBCFile()
85{
86 delete [] _data;
87 if (_file != NULL)
88 CascCloseFile(_file);
89}

91DBCFile::Record DBCFile::getRecord(size_t id)
92{
93 assert(_data);
94 return Record(*this, _data + id*_recordSize);
95}
96
97size_t DBCFile::getMaxId()
98{
99 assert(_data);
100
101 size_t maxId = 0;
102 for(size_t i = 0; i < getRecordCount(); ++i)
103 if (maxId < getRecord(i).getUInt(0))
104 maxId = getRecord(i).getUInt(0);
105
106 return maxId;
107}

109DBCFile::Iterator DBCFile::begin()
110{
111 assert(_data);
112 return Iterator(*this, _data);

115DBCFile::Iterator DBCFile::end()
116{
117 assert(_data);
118 return Iterator(*this, _stringTable);

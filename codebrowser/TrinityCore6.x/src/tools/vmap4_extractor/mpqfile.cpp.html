52size_t MPQFile::read(void* dest, size_t bytes)
53{
54 if (eof) return 0;
55
56 size_t rpos = pointer + bytes;
57 if (rpos > size) {
58 bytes = size - pointer;
59 eof = true;
60 }
61
62 memcpy(dest, &(buffer[pointer]), bytes);
63
64 pointer = rpos;
65

19#ifndef _REFMANAGER_H
20#define _REFMANAGER_H
21//=====================================================
22
23#include "Dynamic/LinkedList.h"
24#include "Dynamic/LinkedReference/Reference.h"
25
26template <class TO, class FROM> class RefManager : public LinkedListHead
27{
28 public:
29 typedef LinkedListHead::Iterator< Reference<TO, FROM> > iterator;
30 RefManager() { }
31 virtual ~RefManager() { clearReferences(); }
32
33 Reference<TO, FROM>* getFirst() { return ((Reference<TO, FROM>*) LinkedListHead::getFirst()); }
34 Reference<TO, FROM> const* getFirst() const { return ((Reference<TO, FROM> const*) LinkedListHead::getFirst()); }
35 Reference<TO, FROM>* getLast() { return ((Reference<TO, FROM>*) LinkedListHead::getLast()); }
36 Reference<TO, FROM> const* getLast() const { return ((Reference<TO, FROM> const*) LinkedListHead::getLast()); }
37
38 iterator begin() { return iterator(getFirst()); }
39 iterator end() { return iterator(NULL); }
40 iterator rbegin() { return iterator(getLast()); }
41 iterator rend() { return iterator(NULL); }
42
43 void clearReferences()
44 {
45 LinkedListElement* ref;
46 while ((ref = getFirst()) != NULL)
47 {
48 ((Reference<TO, FROM>*) ref)->invalidate();
49 ref->delink(); // the delink might be already done by invalidate(), but doing it here again does not hurt and insures an empty list
50 }
51 }
52};
53
54//=====================================================
55#endif

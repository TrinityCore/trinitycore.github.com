34/*
35 * @class ContainerMapList is a mulit-type container for map elements
36 * By itself its meaningless but collaborate along with TypeContainers,
37 * it become the most powerfully container in the whole system.
38 */
39template<class OBJECT>
40struct ContainerMapList
41{
42 //std::map<OBJECT_HANDLE, OBJECT *> _element;
43 GridRefManager<OBJECT> _element;
44};
45
46template<>
47struct ContainerMapList<TypeNull> /* nothing is in type null */
48{
49};
50
51template<class H, class T>
52struct ContainerMapList<TypeList<H, T> >
53{
54 ContainerMapList<H> _elements;
55 ContainerMapList<T> _TailElements;
56};

58template<class OBJECT, class KEY_TYPE>
59struct ContainerUnorderedMap
60{
61 std::unordered_map<KEY_TYPE, OBJECT*> _element;
62};
63
64template<class KEY_TYPE>
65struct ContainerUnorderedMap<TypeNull, KEY_TYPE>
66{
67};

69template<class H, class T, class KEY_TYPE>
70struct ContainerUnorderedMap<TypeList<H, T>, KEY_TYPE>
71{
72 ContainerUnorderedMap<H, KEY_TYPE> _elements;
73 ContainerUnorderedMap<T, KEY_TYPE> _TailElements;
74};

78/*
79 * @class TypeMapContainer contains a fixed number of types and is
80 * determined at compile time. This is probably the most complicated
81 * class and do its simplest thing, that is, holds objects
82 * of different types.
83 */
84
85template<class OBJECT_TYPES>
86class TypeMapContainer
87{
88 public:
89 template<class SPECIFIC_TYPE> size_t Count() const { return Trinity::Count(i_elements, (SPECIFIC_TYPE*)NULL); }
90
91 /// inserts a specific object into the container
92 template<class SPECIFIC_TYPE>
93 bool insert(SPECIFIC_TYPE *obj)
94 {
95 SPECIFIC_TYPE* t = Trinity::Insert(i_elements, obj);
96 return (t != NULL);
97 }
98
99 /// Removes the object from the container, and returns the removed object
100 //template<class SPECIFIC_TYPE>
101 //bool remove(SPECIFIC_TYPE* obj)
102 //{
103 // SPECIFIC_TYPE* t = Trinity::Remove(i_elements, obj);
104 // return (t != NULL);
105 //}
106
107 ContainerMapList<OBJECT_TYPES> & GetElements(void) { return i_elements; }
108 const ContainerMapList<OBJECT_TYPES> & GetElements(void) const { return i_elements;}
109
110 private:
111 ContainerMapList<OBJECT_TYPES> i_elements;
112};
113
114template<class OBJECT_TYPES, class KEY_TYPE>
115class TypeUnorderedMapContainer
116{
117public:
118 template<class SPECIFIC_TYPE>
119 bool Insert(KEY_TYPE const& handle, SPECIFIC_TYPE* obj)
120 {
121 return Trinity::Insert(_elements, handle, obj);
122 }
123
124 template<class SPECIFIC_TYPE>
125 bool Remove(KEY_TYPE const& handle)
126 {
127 return Trinity::Remove(_elements, handle, (SPECIFIC_TYPE*)NULL);
128 }
129
130 template<class SPECIFIC_TYPE>
131 SPECIFIC_TYPE* Find(KEY_TYPE const& handle)
132 {
133 return Trinity::Find(_elements, handle, (SPECIFIC_TYPE*)NULL);
134 }
135
136 ContainerUnorderedMap<OBJECT_TYPES, KEY_TYPE>& GetElements() { return _elements; }
137 ContainerUnorderedMap<OBJECT_TYPES, KEY_TYPE> const& GetElements() const { return _elements; }
138
139private:
140 ContainerUnorderedMap<OBJECT_TYPES, KEY_TYPE> _elements;
141};

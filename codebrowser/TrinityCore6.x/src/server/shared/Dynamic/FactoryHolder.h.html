20#define TRINITY_FACTORY_HOLDER

22#include "Define.h"
23#include "Dynamic/TypeList.h"
24#include "ObjectRegistry.h"

26/** FactoryHolder holds a factory object of a specific type
27 */
28template<class T, class Key = std::string>
29class FactoryHolder
30{
31 public:
32 typedef ObjectRegistry<FactoryHolder<T, Key >, Key > FactoryHolderRegistry;
33
34 FactoryHolder(Key k) : i_key(k) { }
35 virtual ~FactoryHolder() { }
36 inline Key key() const { return i_key; }
37
38 void RegisterSelf(void) { FactoryHolderRegistry::instance()->InsertItem(this, i_key); }
39 void DeregisterSelf(void) { FactoryHolderRegistry::instance()->RemoveItem(this, false); }
40
41 /// Abstract Factory create method
42 virtual T* Create(void *data = NULL) const = 0;
43 private:
44 Key i_key;
45};

47/** Permissible is a classic way of letting the object decide
48 * whether how good they handle things. This is not retricted
49 * to factory selectors.
50 */
51template<class T>
52class Permissible
53{
54 public:
55 virtual ~Permissible() { }
56 virtual int Permit(const T *) const = 0;
57};

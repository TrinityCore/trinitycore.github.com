18#ifndef __SOCKET_H__
19#define __SOCKET_H__
20
21#include "MessageBuffer.h"
22#include "Log.h"
23#include <atomic>
24#include <queue>
25#include <memory>
26#include <functional>
27#include <type_traits>
28#include <boost/asio/ip/tcp.hpp>
29
30using boost::asio::ip::tcp;
31
32#define READ_BLOCK_SIZE 4096
33#ifdef BOOST_ASIO_HAS_IOCP
34#define TC_SOCKET_USE_IOCP
35#endif
36
37template<class T>
38class Socket : public std::enable_shared_from_this<T>
39{
40public:
41 explicit Socket(tcp::socket&& socket) : _socket(std::move(socket)), _remoteAddress(_socket.remote_endpoint().address()),
42 _remotePort(_socket.remote_endpoint().port()), _readBuffer(), _closed(false), _closing(false), _isWritingAsync(false)
43 {
44 _readBuffer.Resize(READ_BLOCK_SIZE);
45 }
46
47 virtual ~Socket()
48 {
49 _closed = true;
50 boost::system::error_code error;
51 _socket.close(error);
52 }
53
54 virtual void Start() = 0;
55
56 virtual bool Update()
57 {
58 if (!IsOpen())
59 return false;
60
61#ifndef TC_SOCKET_USE_IOCP
62 if (_isWritingAsync || _writeQueue.empty())
63 return true;
64
65 for (; HandleQueue();)
66 ;
67#endif
68
69 return true;
70 }
71
72 boost::asio::ip::address GetRemoteIpAddress() const
73 {
74 return _remoteAddress;
75 }
76
77 uint16 GetRemotePort() const
78 {
79 return _remotePort;
80 }
81
82 void AsyncRead()
83 {
84 if (!IsOpen())
85 return;
86
87 _readBuffer.Normalize();
88 _readBuffer.EnsureFreeSpace();
89 _socket.async_read_some(boost::asio::buffer(_readBuffer.GetWritePointer(), _readBuffer.GetRemainingSpace()),
90 std::bind(&Socket<T>::ReadHandlerInternal, this->shared_from_this(), std::placeholders::_1, std::placeholders::_2));
91 }
92
93 void QueuePacket(MessageBuffer&& buffer)
94 {
95 _writeQueue.push(std::move(buffer));
96
97#ifdef TC_SOCKET_USE_IOCP
98 AsyncProcessQueue();
99#endif
100 }
101
102 bool IsOpen() const { return !_closed && !_closing; }
103
104 void CloseSocket()
105 {
106 if (_closed.exchange(true))
107 return;
108
109 boost::system::error_code shutdownError;
110 _socket.shutdown(boost::asio::socket_base::shutdown_send, shutdownError);
111 if (shutdownError)
112 TC_LOG_DEBUG("network", "Socket::CloseSocket: %s errored when shutting down socket: %i (%s)", GetRemoteIpAddress().to_string().c_str(),
113 shutdownError.value(), shutdownError.message().c_str());
114
115 OnClose();
116 }
117
118 /// Marks the socket for closing after write buffer becomes empty
119 void DelayedCloseSocket() { _closing = true; }
120
121 MessageBuffer& GetReadBuffer() { return _readBuffer; }
122
123protected:
124 virtual void OnClose() { }
125
126 virtual void ReadHandler() = 0;
127
128 bool AsyncProcessQueue()
129 {
130 if (_isWritingAsync)
131 return false;
132
133 _isWritingAsync = true;
134
135#ifdef TC_SOCKET_USE_IOCP
136 MessageBuffer& buffer = _writeQueue.front();
137 _socket.async_write_some(boost::asio::buffer(buffer.GetReadPointer(), buffer.GetActiveSize()), std::bind(&Socket<T>::WriteHandler,
138 this->shared_from_this(), std::placeholders::_1, std::placeholders::_2));
139#else
140 _socket.async_write_some(boost::asio::null_buffers(), std::bind(&Socket<T>::WriteHandlerWrapper,
141 this->shared_from_this(), std::placeholders::_1, std::placeholders::_2));
142#endif
143
144 return false;
145 }
146
147 void SetNoDelay(bool enable)
148 {
149 boost::system::error_code err;
150 _socket.set_option(tcp::no_delay(enable), err);
151 if (err)
152 TC_LOG_DEBUG("network", "Socket::SetNoDelay: failed to set_option(boost::asio::ip::tcp::no_delay) for %s - %d (%s)",
153 GetRemoteIpAddress().to_string().c_str(), err.value(), err.message().c_str());
154 }
155
156
157private:
158 void ReadHandlerInternal(boost::system::error_code error, size_t transferredBytes)
159 {
160 if (error)
161 {
162 CloseSocket();
163 return;
164 }
165
166 _readBuffer.WriteCompleted(transferredBytes);
167 ReadHandler();
168 }
169
170#ifdef TC_SOCKET_USE_IOCP
171
172 void WriteHandler(boost::system::error_code error, std::size_t transferedBytes)
173 {
174 if (!error)
175 {
176 _isWritingAsync = false;
177 _writeQueue.front().ReadCompleted(transferedBytes);
178 if (!_writeQueue.front().GetActiveSize())
179 _writeQueue.pop();
180
181 if (!_writeQueue.empty())
182 AsyncProcessQueue();
183 else if (_closing)
184 CloseSocket();
185 }
186 else
187 CloseSocket();
188 }
189
190#else
191
192 void WriteHandlerWrapper(boost::system::error_code /*error*/, std::size_t /*transferedBytes*/)
193 {
194 _isWritingAsync = false;
195 HandleQueue();
196 }
197
198 bool HandleQueue()
199 {
200 if (!IsOpen())
201 return false;
202
203 if (_writeQueue.empty())
204 return false;
205
206 MessageBuffer& queuedMessage = _writeQueue.front();
207
208 std::size_t bytesToSend = queuedMessage.GetActiveSize();
209
210 boost::system::error_code error;
211 std::size_t bytesSent = _socket.write_some(boost::asio::buffer(queuedMessage.GetReadPointer(), bytesToSend), error);
212
213 if (error)
214 {
215 if (error == boost::asio::error::would_block || error == boost::asio::error::try_again)
216 return AsyncProcessQueue();
217
218 _writeQueue.pop();
219 return false;
220 }
221 else if (bytesSent == 0)
222 {
223 _writeQueue.pop();
224 return false;
225 }
226 else if (bytesSent < bytesToSend) // now n > 0
227 {
228 queuedMessage.ReadCompleted(bytesSent);
229 return AsyncProcessQueue();
230 }
231
232 _writeQueue.pop();
233 return !_writeQueue.empty();
234 }
235
236#endif
237
238 tcp::socket _socket;
239
240 boost::asio::ip::address _remoteAddress;
241 uint16 _remotePort;
242
243 MessageBuffer _readBuffer;
244 std::queue<MessageBuffer> _writeQueue;
245
246 std::atomic<bool> _closed;
247 std::atomic<bool> _closing;
248
249 bool _isWritingAsync;
250};
251
252#endif // __SOCKET_H__

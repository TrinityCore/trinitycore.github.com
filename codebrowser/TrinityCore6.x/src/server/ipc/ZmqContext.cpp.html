20ZmqContext::ZmqContext() : _inproc(nullptr)
21{
22}
23
24ZmqContext::~ZmqContext()
25{
26}
27
28zmqpp::socket* ZmqContext::CreateNewSocket(zmqpp::socket_type type)
29{
30 std::unique_lock<std::mutex> lock(_mutex);
31 zmqpp::socket* socket = new zmqpp::socket(_context, type);
32 socket->set(zmqpp::socket_option::linger, 0);
33 return socket;
34}
35
36void ZmqContext::Initialize()
37{
38 _inproc = new zmqpp::socket(_context, zmqpp::socket_type::pub);
39 _inproc->bind("inproc://workers");
40}
41
42zmqpp::socket* ZmqContext::CreateInprocSubscriber()
43{
44 zmqpp::socket* sub = CreateNewSocket(zmqpp::socket_type::sub);
45 sub->connect("inproc://workers");
46 sub->subscribe("internalmq.");
47 return sub;
48}
49
50void ZmqContext::Close()
51{
52 _inproc->send("internalmq.kill");
53 delete _inproc;
54 _inproc = nullptr;
55}

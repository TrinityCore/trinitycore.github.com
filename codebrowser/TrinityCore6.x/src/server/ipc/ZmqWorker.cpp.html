18#include "ZmqWorker.h"
19#include "ZmqContext.h"
20
21ZmqWorker::ZmqWorker(std::string const& taskUri, std::string const& resUri) :
22 _taskUri(taskUri), _resultsUri(resUri)
23{
24}
25
26ZmqWorker::~ZmqWorker()
27{
28 delete _taskQueue;
29 delete _results;
30}
31
32void ZmqWorker::HandleOpen()
33{
34 _taskQueue = sIpcContext->CreateNewSocket(zmqpp::socket_type::pull);
35 _results = sIpcContext->CreateNewSocket(zmqpp::socket_type::push);
36
37 _taskQueue->connect(_taskUri);
38 _results->connect(_resultsUri);
39
40 _poller->add(*_taskQueue);
41}
42
43void ZmqWorker::HandleClose()
44{
45 _taskQueue->close();
46 _results->close();
47}
48
49void ZmqWorker::Run()
50{
51 while (!ProcessExit())
52 {
53 _poller->poll();
54 if (_poller->events(*_taskQueue) & zmqpp::poller::poll_in)
55 PerformWork();
56 }
57}
58
59void ZmqWorker::PerformWork()
60{
61 int32 op1;
62 do
63 {
64 zmqpp::message msg;
65 _taskQueue->receive(msg);
66 Dispatch(msg);
67 _taskQueue->get(zmqpp::socket_option::events, op1);
68 } while (op1 & zmqpp::poller::poll_in);
69}

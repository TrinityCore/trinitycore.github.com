18#include "ZMQTask.h"
19#include "ZmqContext.h"

22ZMQTask::ZMQTask()
23{
24 _poller = new zmqpp::poller();
25}
26
27ZMQTask::~ZMQTask()
28{
29 delete _poller;
30 _poller = NULL;
31 delete _inproc;
32 delete _thread;
33}
34
35void ZMQTask::Start()
36{
37 _inproc = sIpcContext->CreateInprocSubscriber();
38 _poller->add(*_inproc);
39
40 HandleOpen();
41 _thread = new std::thread(&ZMQTask::Run, this);
42}
43
44void ZMQTask::End()
45{
46 _thread->join();
47 _poller->remove(*_inproc);
48 _inproc->close();
49 HandleClose();
50}

52bool ZMQTask::ProcessExit()
53{
54 if (_poller->events(*_inproc) == zmqpp::poller::poll_in)
55 {
56 int op1;
57 do
58 {
59 zmqpp::message msg;
60 if (!_inproc->receive(msg, true))
61 return false; //No more messages to read from sock. This shouldn't happen.
62
63 // strip 'internalmq.' from message
64 std::string cmd = msg.get(0).substr(11);
65 if (cmd == "kill")
66 return true;
67
68 _inproc->get(zmqpp::socket_option::events, op1);
69 } while (op1 & zmqpp::poller::poll_in);
70 }
71
72 return false;
73}
74
75void ZMQTask::Pipeline(zmqpp::socket* from, zmqpp::socket* to)
76{
77 /*
78 Push messages from socket to socket.
79 */
80 if (_poller->events(*from) == zmqpp::poller::poll_in)
81 {
82 int32 op1, op2;
83 do
84 {
85 zmqpp::message msg;
86 if (!from->receive(msg, true))
87 return; //No more messages to read from socket. This shouldn't happen.
88
89 to->send(msg);
90 from->get(zmqpp::socket_option::events, op1);
91 to->get(zmqpp::socket_option::events, op2);
92 } while(op1 & zmqpp::poller::poll_in && op2 & zmqpp::poller::poll_out);
93 }
94}

18#ifndef __ZMQLISTENER_H
19#define __ZMQLISTENER_H
20
21#include "ZMQTask.h"
22#include <zmqpp/zmqpp.hpp>
23
24class ZmqListener : public ZMQTask
25{
26/*
27 * Read broadcasts from remote PUB socket, and forward them to
28 * another socket.
29 *
30 * from - client SUB socket
31 * to - listen PUSH socket
32 *
33 */
34public:
35 ZmqListener(std::string const& from, std::string const& to);
36 ~ZmqListener();
37 void Run() override;
38
39 void Subscribe(std::string const& keyword);
40 void Unsubscribe(std::string const& keyword);
41
42protected:
43 void HandleOpen() override;
44 void HandleClose() override;
45
46private:
47 zmqpp::socket* _from;
48 zmqpp::socket* _to;
49};

18#include "ZmqMux.h"
19#include "ZmqContext.h"
20
21ZmqMux::ZmqMux(std::string const& fromUri, std::string const& toUri):
22 _fromAddress(fromUri)
23{
24 printf("Opening muxer thread from %s to %s\n", fromUri.c_str(), toUri.c_str());
25 _from = sIpcContext->CreateNewSocket(zmqpp::socket_type::pull);
26 _to = sIpcContext->CreateNewSocket(zmqpp::socket_type::push);
27
28 _from->bind(fromUri);
29 _to->connect(toUri);
30}
31
32ZmqMux::~ZmqMux()
33{
34 delete _from;
35 delete _to;
36}
37
38void ZmqMux::HandleOpen()
39{
40 _poller->add(*_from);
41 _poller->add(*_to, zmqpp::poller::poll_out);
42}
43
44bool ZmqMux::Send(zmqpp::message* m, bool dont_block)
45{
46 if (_socket.get() == nullptr)
47 {
48 _socket.reset(sIpcContext->CreateNewSocket(zmqpp::socket_type::push));
49 _socket->connect(_fromAddress);
50 }
51
52 return _socket->send(*m, dont_block);
53}
54
55void ZmqMux::Run()
56{
57 for (;;)
58 {
59 std::this_thread::sleep_for(std::chrono::milliseconds(50));
60 if (!_poller->poll())
61 break;
62
63 if (ProcessExit())
64 break;
65
66 Pipeline(_from, _to);
67 }
68}

18#include "ZmqListener.h"
19#include "ZmqContext.h"
20
21ZmqListener::ZmqListener(std::string const& from, std::string const& to)
22{
23 _from = sIpcContext->CreateNewSocket(zmqpp::socket_type::sub);
24 _to = sIpcContext->CreateNewSocket(zmqpp::socket_type::push);
25 _from->connect(from);
26 _to->bind(to);
27}
28
29ZmqListener::~ZmqListener()
30{
31 delete _from;
32 delete _to;
33}
34
35void ZmqListener::HandleOpen()
36{
37}
38
39void ZmqListener::HandleClose()
40{
41 _from->close();
42 _to->close();
43}
44
45void ZmqListener::Run()
46{
47 while (!ProcessExit())
48 {
49 _poller->poll();
50
51 while (_poller->events(*_from) & zmqpp::poller::poll_in &&
52 _poller->events(*_to) & zmqpp::poller::poll_out)
53 {
54 zmqpp::message msg;
55 _from->receive(msg);
56 _to->send(msg);
57 }
58 }
59}
60
61void ZmqListener::Subscribe(std::string const& keyword)
62{
63 _from->subscribe(keyword);
64}
65
66void ZmqListener::Unsubscribe(std::string const& keyword)
67{
68 _from->unsubscribe(keyword);
69}

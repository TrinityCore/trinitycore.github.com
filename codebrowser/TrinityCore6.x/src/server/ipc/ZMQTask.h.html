18#ifndef __ZMQTASK_H
19#define __ZMQTASK_H
20
21#include "Define.h"
22#include <thread>
23#include <zmqpp/poller.hpp>
24#include <zmqpp/socket.hpp>
25
26/*
27 This class serves as a base for all long running tasks
28 It is set up to terminate its running task upon receiving "kill" command
29*/
30class ZMQTask
31{
32public:
33 ZMQTask();
34 virtual ~ZMQTask();
35
36 void Start();
37 void End();
38 virtual void Run() = 0;
39
40protected:
41 virtual void HandleOpen() { }
42 virtual void HandleClose() { }
43 void Pipeline(zmqpp::socket* from, zmqpp::socket* to);
44 bool ProcessExit();
45
46 zmqpp::poller* _poller;
47
48 zmqpp::socket* _inproc = nullptr;
49 std::thread* _thread = nullptr;
50};
51
52#endif // __ZMQTASK_H

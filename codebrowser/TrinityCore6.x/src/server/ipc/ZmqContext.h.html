18#ifndef __ZMQCONTEX_H
19#define __ZMQCONTEX_H
20
21#include <zmqpp/zmqpp.hpp>
22#include <mutex>
23
24/*
25 * We need to serialize access to zmq context otherwise stuff blows up.
26 */
27class ZmqContext
28{
29public:
30 ~ZmqContext();
31
32 static ZmqContext* Instance()
33 {
34 static ZmqContext instance;
35 return &instance;
36 }
37
38 zmqpp::socket* CreateNewSocket(zmqpp::socket_type);
39 void Initialize();
40 zmqpp::socket* CreateInprocSubscriber();
41 void Close();
42
43private:
44 ZmqContext();
45 ZmqContext(ZmqContext const&) = delete;
46 ZmqContext& operator=(ZmqContext const&) = delete;
47
48 zmqpp::context _context;
49 std::mutex _mutex;
50 zmqpp::socket* _inproc;
51};
52
53#define sIpcContext ZmqContext::Instance()
54
55#endif

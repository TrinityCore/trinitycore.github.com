47template <class T>
48class DatabaseWorkerPool
49{
50 private:
51 enum InternalIndex
52 {
53 IDX_ASYNC,
54 IDX_SYNCH,
55 IDX_SIZE
56 };
57
58 public:
59 /* Activity state */
60 DatabaseWorkerPool();
61
62 ~DatabaseWorkerPool()
63 {
64 _queue->Cancel();
65 }
66
67 void SetConnectionInfo(std::string const& infoString, uint8 const asyncThreads, uint8 const synchThreads);
68
69 uint32 Open();
70
71 void Close();
72
73 //! Prepares all prepared statements
74 bool PrepareStatements();
75
76 inline MySQLConnectionInfo const* GetConnectionInfo() const
77 {
78 return _connectionInfo.get();
79 }
80
81 /**
82 Delayed one-way statement methods.
83 */
84
85 //! Enqueues a one-way SQL operation in string format that will be executed asynchronously.
86 //! This method should only be used for queries that are only executed once, e.g during startup.
87 void Execute(const char* sql)
88 {
89 if (Trinity::IsFormatEmptyOrNull(sql))
90 return;
91
92 BasicStatementTask* task = new BasicStatementTask(sql);
93 Enqueue(task);
94 }
95
96 //! Enqueues a one-way SQL operation in string format -with variable args- that will be executed asynchronously.
97 //! This method should only be used for queries that are only executed once, e.g during startup.
98 template<typename Format, typename... Args>
99 void PExecute(Format&& sql, Args&&... args)
100 {
101 if (Trinity::IsFormatEmptyOrNull(sql))
102 return;
103
104 Execute(Trinity::StringFormat(std::forward<Format>(sql), std::forward<Args>(args)...).c_str());
105 }
106
107 //! Enqueues a one-way SQL operation in prepared statement format that will be executed asynchronously.
108 //! Statement must be prepared with CONNECTION_ASYNC flag.
109 void Execute(PreparedStatement* stmt)
110 {
111 PreparedStatementTask* task = new PreparedStatementTask(stmt);
112 Enqueue(task);
113 }
114
115 /**
116 Direct synchronous one-way statement methods.
117 */
118
119 //! Directly executes a one-way SQL operation in string format, that will block the calling thread until finished.
120 //! This method should only be used for queries that are only executed once, e.g during startup.
121 void DirectExecute(const char* sql)
122 {
123 if (!sql)
124 return;
125
126 T* connection = GetFreeConnection();
127 connection->Execute(sql);
128 connection->Unlock();
129 }
130
131 //! Directly executes a one-way SQL operation in string format -with variable args-, that will block the calling thread until finished.
132 //! This method should only be used for queries that are only executed once, e.g during startup.
133 template<typename Format, typename... Args>
134 void DirectPExecute(Format&& sql, Args&&... args)
135 {
136 if (Trinity::IsFormatEmptyOrNull(sql))
137 return;
138
139 DirectExecute(Trinity::StringFormat(std::forward<Format>(sql), std::forward<Args>(args)...).c_str());
140 }
141
142 //! Directly executes a one-way SQL operation in prepared statement format, that will block the calling thread until finished.
143 //! Statement must be prepared with the CONNECTION_SYNCH flag.
144 void DirectExecute(PreparedStatement* stmt)
145 {
146 T* connection = GetFreeConnection();
147 connection->Execute(stmt);
148 connection->Unlock();
149
150 //! Delete proxy-class. Not needed anymore
151 delete stmt;
152 }
153
154 /**
155 Synchronous query (with resultset) methods.
156 */
157
158 //! Directly executes an SQL query in string format that will block the calling thread until finished.
159 //! Returns reference counted auto pointer, no need for manual memory management in upper level code.
160 QueryResult Query(const char* sql, T* connection = nullptr);
161
162 //! Directly executes an SQL query in string format -with variable args- that will block the calling thread until finished.
163 //! Returns reference counted auto pointer, no need for manual memory management in upper level code.
164 template<typename Format, typename... Args>
165 QueryResult PQuery(Format&& sql, T* conn, Args&&... args)
166 {
167 if (Trinity::IsFormatEmptyOrNull(sql))
168 return QueryResult(nullptr);
169
170 return Query(Trinity::StringFormat(std::forward<Format>(sql), std::forward<Args>(args)...).c_str(), conn);
171 }
172
173 //! Directly executes an SQL query in string format -with variable args- that will block the calling thread until finished.
174 //! Returns reference counted auto pointer, no need for manual memory management in upper level code.
175 template<typename Format, typename... Args>
176 QueryResult PQuery(Format&& sql, Args&&... args)
177 {
178 if (!sql)
179 return QueryResult(nullptr);
180
181 return Query(Trinity::StringFormat(std::forward<Format>(sql), std::forward<Args>(args)...).c_str());
182 }
183
184 //! Directly executes an SQL query in prepared format that will block the calling thread until finished.
185 //! Returns reference counted auto pointer, no need for manual memory management in upper level code.
186 //! Statement must be prepared with CONNECTION_SYNCH flag.
187 PreparedQueryResult Query(PreparedStatement* stmt);
188
189 /**
190 Asynchronous query (with resultset) methods.
191 */
192
193 //! Enqueues a query in string format that will set the value of the QueryResultFuture return object as soon as the query is executed.
194 //! The return value is then processed in ProcessQueryCallback methods.
195 QueryResultFuture AsyncQuery(const char* sql);
196
197 //! Enqueues a query in string format -with variable args- that will set the value of the QueryResultFuture return object as soon as the query is executed.
198 //! The return value is then processed in ProcessQueryCallback methods.
199 template<typename Format, typename... Args>
200 QueryResultFuture AsyncPQuery(Format&& sql, Args&&... args)
201 {
202 return AsyncQuery(Trinity::StringFormat(std::forward<Format>(sql), std::forward<Args>(args)...).c_str());
203 }
204
205 //! Enqueues a query in prepared format that will set the value of the PreparedQueryResultFuture return object as soon as the query is executed.
206 //! The return value is then processed in ProcessQueryCallback methods.
207 //! Statement must be prepared with CONNECTION_ASYNC flag.
208 PreparedQueryResultFuture AsyncQuery(PreparedStatement* stmt);
209
210 //! Enqueues a vector of SQL operations (can be both adhoc and prepared) that will set the value of the QueryResultHolderFuture
211 //! return object as soon as the query is executed.
212 //! The return value is then processed in ProcessQueryCallback methods.
213 //! Any prepared statements added to this holder need to be prepared with the CONNECTION_ASYNC flag.
214 QueryResultHolderFuture DelayQueryHolder(SQLQueryHolder* holder);
215
216 /**
217 Transaction context methods.
218 */
219
220 //! Begins an automanaged transaction pointer that will automatically rollback if not commited. (Autocommit=0)
221 SQLTransaction BeginTransaction()
222 {
223 return SQLTransaction(new Transaction);
224 }
225
226 //! Enqueues a collection of one-way SQL operations (can be both adhoc and prepared). The order in which these operations
227 //! were appended to the transaction will be respected during execution.
228 void CommitTransaction(SQLTransaction transaction);
229
230 //! Directly executes a collection of one-way SQL operations (can be both adhoc and prepared). The order in which these operations
231 //! were appended to the transaction will be respected during execution.
232 void DirectCommitTransaction(SQLTransaction& transaction);
233
234 //! Method used to execute prepared statements in a diverse context.
235 //! Will be wrapped in a transaction if valid object is present, otherwise executed standalone.
236 void ExecuteOrAppend(SQLTransaction& trans, PreparedStatement* stmt)
237 {
238 if (!trans)
239 Execute(stmt);
240 else
241 trans->Append(stmt);
242 }
243
244 //! Method used to execute ad-hoc statements in a diverse context.
245 //! Will be wrapped in a transaction if valid object is present, otherwise executed standalone.
246 void ExecuteOrAppend(SQLTransaction& trans, const char* sql)
247 {
248 if (!trans)
249 Execute(sql);
250 else
251 trans->Append(sql);
252 }

254 /**
255 Other

257
258 typedef typename T::Statements PreparedStatementIndex;
259
260 //! Automanaged (internally) pointer to a prepared statement object for usage in upper level code.
261 //! Pointer is deleted in this->DirectExecute(PreparedStatement*), this->Query(PreparedStatement*) or PreparedStatementTask::~PreparedStatementTask.
262 //! This object is not tied to the prepared statement on the MySQL context yet until execution.
263 PreparedStatement* GetPreparedStatement(PreparedStatementIndex index)
264 {
265 return new PreparedStatement(index);
266 }
267
268 //! Apply escape string'ing for current collation. (utf8)
269 void EscapeString(std::string& str);
270
271 //! Keeps all our MySQL connections alive, prevent the server from disconnecting us.
272 void KeepAlive();
273
274 private:
275 uint32 OpenConnections(InternalIndex type, uint8 numConnections);
276
277 unsigned long EscapeString(char *to, const char *from, unsigned long length)
278 {
279 if (!to || !from || !length)
280 return 0;
281
282 return mysql_real_escape_string(
283 _connections[IDX_SYNCH].front()->GetHandle(), to, from, length);
284 }
285
286 void Enqueue(SQLOperation* op)
287 {
288 _queue->Push(op);
289 }
290
291 //! Gets a free connection in the synchronous connection pool.
292 //! Caller MUST call t->Unlock() after touching the MySQL context to prevent deadlocks.
293 T* GetFreeConnection();
294
295 char const* GetDatabaseName() const
296 {
297 return _connectionInfo->database.c_str();
298 }
299
300 //! Queue shared by async worker threads.
301 std::unique_ptr<ProducerConsumerQueue<SQLOperation*>> _queue;
302 std::array<std::vector<std::unique_ptr<T>>, IDX_SIZE> _connections;
303 std::unique_ptr<MySQLConnectionInfo> _connectionInfo;
304 uint8 _async_threads, _synch_threads;
305};

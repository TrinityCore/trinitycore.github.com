62void MySQLConnection::Close()
63{
64 // Stop the worker thread before the statements are cleared
65 m_worker.reset();
66
67 m_stmts.clear();
68
69 if (m_Mysql)
70 {
71 mysql_close(m_Mysql);
72 m_Mysql = nullptr;
73 }

76uint32 MySQLConnection::Open()
77{
78 MYSQL *mysqlInit;
79 mysqlInit = mysql_init(NULL);
80 if (!mysqlInit)
81 {
82 TC_LOG_ERROR("sql.sql", "Could not initialize Mysql connection to database `%s`", m_connectionInfo.database.c_str());
83 return CR_UNKNOWN_ERROR;
84 }
85
86 int port;
87 char const* unix_socket;
88 //unsigned int timeout = 10;
89
90 mysql_options(mysqlInit, MYSQL_SET_CHARSET_NAME, "utf8");
91 //mysql_options(mysqlInit, MYSQL_OPT_READ_TIMEOUT, (char const*)&timeout);
92 #ifdef _WIN32
93 if (m_connectionInfo.host == ".") // named pipe use option (Windows)
94 {
95 unsigned int opt = MYSQL_PROTOCOL_PIPE;
96 mysql_options(mysqlInit, MYSQL_OPT_PROTOCOL, (char const*)&opt);
97 port = 0;
98 unix_socket = 0;
99 }
100 else // generic case
101 {
102 port = atoi(m_connectionInfo.port_or_socket.c_str());
103 unix_socket = 0;
104 }
105 #else
106 if (m_connectionInfo.host == ".") // socket use option (Unix/Linux)
107 {
108 unsigned int opt = MYSQL_PROTOCOL_SOCKET;
109 mysql_options(mysqlInit, MYSQL_OPT_PROTOCOL, (char const*)&opt);
110 m_connectionInfo.host = "localhost";
111 port = 0;
112 unix_socket = m_connectionInfo.port_or_socket.c_str();
113 }
114 else // generic case
115 {
116 port = atoi(m_connectionInfo.port_or_socket.c_str());
117 unix_socket = nullptr;
118 }
119 #endif
120
121 m_Mysql = mysql_real_connect(mysqlInit, m_connectionInfo.host.c_str(), m_connectionInfo.user.c_str(),
122 m_connectionInfo.password.c_str(), m_connectionInfo.database.c_str(), port, unix_socket, 0);
123
124 if (m_Mysql)
125 {
126 if (!m_reconnecting)
127 {
128 TC_LOG_INFO("sql.sql", "MySQL client library: %s", mysql_get_client_info());
129 TC_LOG_INFO("sql.sql", "MySQL server ver: %s ", mysql_get_server_info(m_Mysql));
130 // MySQL version above 5.1 IS required in both client and server and there is no known issue with different versions above 5.1
131 // if (mysql_get_server_version(m_Mysql) != mysql_get_client_version())
132 // TC_LOG_INFO("sql.sql", "[WARNING] MySQL client/server version mismatch; may conflict with behaviour of prepared statements.");
133 }
134
135 TC_LOG_INFO("sql.sql", "Connected to MySQL database at %s", m_connectionInfo.host.c_str());
136 mysql_autocommit(m_Mysql, 1);
137
138 // set connection properties to UTF8 to properly handle locales for different
139 // server configs - core sends data in UTF8, so MySQL must expect UTF8 too
140 mysql_set_character_set(m_Mysql, "utf8");
141 return 0;
142 }
143 else
144 {
145 TC_LOG_ERROR("sql.sql", "Could not connect to MySQL database at %s: %s", m_connectionInfo.host.c_str(), mysql_error(mysqlInit));
146 mysql_close(mysqlInit);
147 return mysql_errno(mysqlInit);
148 }
149}

157bool MySQLConnection::Execute(const char* sql)
158{
159 if (!m_Mysql)
160 return false;
161
162 {
163 uint32 _s = getMSTime();
164
165 if (mysql_query(m_Mysql, sql))
166 {
167 uint32 lErrno = mysql_errno(m_Mysql);
168
169 TC_LOG_INFO("sql.sql", "SQL: %s", sql);
170 TC_LOG_ERROR("sql.sql", "[%u] %s", lErrno, mysql_error(m_Mysql));
171
172 if (_HandleMySQLErrno(lErrno)) // If it returns true, an error was handled successfully (i.e. reconnection)
173 return Execute(sql); // Try again
174
175 return false;
176 }
177 else
178 TC_LOG_DEBUG("sql.sql", "[%u ms] SQL: %s", getMSTimeDiff(_s, getMSTime()), sql);
179 }
180
181 return true;
182}

184bool MySQLConnection::Execute(PreparedStatement* stmt)
185{
186 if (!m_Mysql)
187 return false;
188
189 uint32 index = stmt->m_index;
190 {
191 MySQLPreparedStatement* m_mStmt = GetPreparedStatement(index);
192 ASSERT(m_mStmt); // Can only be null if preparation failed, server side error or bad query
193 m_mStmt->m_stmt = stmt; // Cross reference them for debug output
194 stmt->m_stmt = m_mStmt; /// @todo Cleaner way
195
196 stmt->BindParameters();
197
198 MYSQL_STMT* msql_STMT = m_mStmt->GetSTMT();
199 MYSQL_BIND* msql_BIND = m_mStmt->GetBind();
200
201 uint32 _s = getMSTime();
202
203 if (mysql_stmt_bind_param(msql_STMT, msql_BIND))
204 {
205 uint32 lErrno = mysql_errno(m_Mysql);
206 TC_LOG_ERROR("sql.sql", "SQL(p): %s\n [ERROR]: [%u] %s", m_mStmt->getQueryString(m_queries[index].first).c_str(), lErrno, mysql_stmt_error(msql_STMT));
207
208 if (_HandleMySQLErrno(lErrno)) // If it returns true, an error was handled successfully (i.e. reconnection)
209 return Execute(stmt); // Try again
210
211 m_mStmt->ClearParameters();
212 return false;
213 }
214
215 if (mysql_stmt_execute(msql_STMT))
216 {
217 uint32 lErrno = mysql_errno(m_Mysql);
218 TC_LOG_ERROR("sql.sql", "SQL(p): %s\n [ERROR]: [%u] %s", m_mStmt->getQueryString(m_queries[index].first).c_str(), lErrno, mysql_stmt_error(msql_STMT));
219
220 if (_HandleMySQLErrno(lErrno)) // If it returns true, an error was handled successfully (i.e. reconnection)
221 return Execute(stmt); // Try again
222
223 m_mStmt->ClearParameters();
224 return false;
225 }
226
227 TC_LOG_DEBUG("sql.sql", "[%u ms] SQL(p): %s", getMSTimeDiff(_s, getMSTime()), m_mStmt->getQueryString(m_queries[index].first).c_str());
228
229 m_mStmt->ClearParameters();
230 return true;
231 }

292{

307bool MySQLConnection::_Query(const char *sql, MYSQL_RES **pResult, MYSQL_FIELD **pFields, uint64* pRowCount, uint32* pFieldCount)
308{
309 if (!m_Mysql)
310 return false;
311
312 {
313 uint32 _s = getMSTime();
314
315 if (mysql_query(m_Mysql, sql))
316 {
317 uint32 lErrno = mysql_errno(m_Mysql);
318 TC_LOG_INFO("sql.sql", "SQL: %s", sql);
319 TC_LOG_ERROR("sql.sql", "[%u] %s", lErrno, mysql_error(m_Mysql));
320
321 if (_HandleMySQLErrno(lErrno)) // If it returns true, an error was handled successfully (i.e. reconnection)
322 return _Query(sql, pResult, pFields, pRowCount, pFieldCount); // We try again
323
324 return false;
325 }
326 else
327 TC_LOG_DEBUG("sql.sql", "[%u ms] SQL: %s", getMSTimeDiff(_s, getMSTime()), sql);
328
329 *pResult = mysql_store_result(m_Mysql);
330 *pRowCount = mysql_affected_rows(m_Mysql);
331 *pFieldCount = mysql_field_count(m_Mysql);
332 }
333
334 if (!*pResult )
335 return false;
336
337 if (!*pRowCount)
338 {
339 mysql_free_result(*pResult);
340 return false;
341 }
342
343 *pFields = mysql_fetch_fields(*pResult);
344
345 return true;
346}

363int MySQLConnection::ExecuteTransaction(SQLTransaction& transaction)
364{
365 std::list<SQLElementData> const& queries = transaction->m_queries;
366 if (queries.empty())
367 return -1;
368
369 BeginTransaction();
370
371 std::list<SQLElementData>::const_iterator itr;
372 for (itr = queries.begin(); itr != queries.end(); ++itr)
373 {
374 SQLElementData const& data = *itr;
375 switch (itr->type)
376 {
377 case SQL_ELEMENT_PREPARED:
378 {
379 PreparedStatement* stmt = data.element.stmt;
380 ASSERT(stmt);
381 if (!Execute(stmt))
382 {
383 TC_LOG_WARN("sql.sql", "Transaction aborted. %u queries not executed.", (uint32)queries.size());
384 int errorCode = GetLastError();
385 RollbackTransaction();
386 return errorCode;
387 }
388 }
389 break;
390 case SQL_ELEMENT_RAW:
391 {
392 const char* sql = data.element.query;
393 ASSERT(sql);
394 if (!Execute(sql))
395 {
396 TC_LOG_WARN("sql.sql", "Transaction aborted. %u queries not executed.", (uint32)queries.size());
397 int errorCode = GetLastError();
398 RollbackTransaction();
399 return errorCode;
400 }
401 }
402 break;
403 }
404 }
405
406 // we might encounter errors during certain queries, and depending on the kind of error
407 // we might want to restart the transaction. So to prevent data loss, we only clean up when it's all done.
408 // This is done in calling functions DatabaseWorkerPool<T>::DirectCommitTransaction and TransactionTask::Execute,
409 // and not while iterating over every element.
410
411 CommitTransaction();
412 return 0;
413}

415MySQLPreparedStatement* MySQLConnection::GetPreparedStatement(uint32 index)
416{
417 ASSERT(index < m_stmts.size());
418 MySQLPreparedStatement* ret = m_stmts[index].get();
419 if (!ret)
420 TC_LOG_ERROR("sql.sql", "Could not fetch prepared statement %u on database `%s`, connection type: %s.",
421 index, m_connectionInfo.database.c_str(), (m_connectionFlags & CONNECTION_ASYNC) ? "asynchronous" : "synchronous");
422
423 return ret;

426void MySQLConnection::PrepareStatement(uint32 index, const char* sql, ConnectionFlags flags)
427{
428 m_queries.insert(PreparedStatementMap::value_type(index, std::make_pair(sql, flags)));
429
430 // Check if specified query should be prepared on this connection
431 // i.e. don't prepare async statements on synchronous connections
432 // to save memory that will not be used.
433 if (!(m_connectionFlags & flags))
434 {
435 m_stmts[index].reset();
436 return;
437 }
438
439 MYSQL_STMT* stmt = mysql_stmt_init(m_Mysql);
440 if (!stmt)
441 {
442 TC_LOG_ERROR("sql.sql", "In mysql_stmt_init() id: %u, sql: \"%s\"", index, sql);
443 TC_LOG_ERROR("sql.sql", "%s", mysql_error(m_Mysql));
444 m_prepareError = true;
445 }
446 else
447 {
448 if (mysql_stmt_prepare(stmt, sql, static_cast<unsigned long>(strlen(sql))))
449 {
450 TC_LOG_ERROR("sql.sql", "In mysql_stmt_prepare() id: %u, sql: \"%s\"", index, sql);
451 TC_LOG_ERROR("sql.sql", "%s", mysql_stmt_error(stmt));
452 mysql_stmt_close(stmt);
453 m_prepareError = true;
454 }
455 else
456 {
457 m_stmts[index] = Trinity::make_unique<MySQLPreparedStatement>(stmt);
458 }
459 }
460}

469 return NULL;

474 }
475 return new PreparedResultSet(stmt->m_stmt->GetSTMT(), result, rowCount, fieldCount);
476}

478bool MySQLConnection::_HandleMySQLErrno(uint32 errNo, uint8 attempts /*= 5*/)
479{
480 switch (errNo)
481 {
482 case CR_SERVER_GONE_ERROR:
483 case CR_SERVER_LOST:
484 case CR_INVALID_CONN_HANDLE:
485 case CR_SERVER_LOST_EXTENDED:
486 {
487 if (m_Mysql)
488 {
489 TC_LOG_ERROR("sql.sql", "Lost the connection to the MySQL server!");
490
491 mysql_close(GetHandle());
492 m_Mysql = nullptr;
493 }
494
495 /*no break*/
496 }
497 case CR_CONN_HOST_ERROR:
498 {
499 TC_LOG_INFO("sql.sql", "Attempting to reconnect to the MySQL server...");
500
501 m_reconnecting = true;
502
503 uint32 const lErrno = Open();
504 if (!lErrno)
505 {
506 // Don't remove 'this' pointer unless you want to skip loading all prepared statements...
507 if (!this->PrepareStatements())
508 {
509 TC_LOG_FATAL("sql.sql", "Could not re-prepare statements!");
510 std::this_thread::sleep_for(std::chrono::seconds(10));
511 std::abort();
512 }
513
514 TC_LOG_INFO("sql.sql", "Successfully reconnected to %s @%s:%s (%s).",
515 m_connectionInfo.database.c_str(), m_connectionInfo.host.c_str(), m_connectionInfo.port_or_socket.c_str(),
516 (m_connectionFlags & CONNECTION_ASYNC) ? "asynchronous" : "synchronous");
517
518 m_reconnecting = false;
519 return true;
520 }
521
522 if ((--attempts) == 0)
523 {
524 // Shut down the server when the mysql server isn't
525 // reachable for some time
526 TC_LOG_FATAL("sql.sql", "Failed to reconnect to the MySQL server, "
527 "terminating the server to prevent data corruption!");
528
529 // We could also initiate a shutdown through using std::raise(SIGTERM)
530 std::this_thread::sleep_for(std::chrono::seconds(10));
531 std::abort();
532 }
533 else
534 {
535 // It's possible this attempted reconnect throws 2006 at us.
536 // To prevent crazy recursive calls, sleep here.
537 std::this_thread::sleep_for(std::chrono::seconds(3)); // Sleep 3 seconds
538 return _HandleMySQLErrno(lErrno, attempts); // Call self (recursive)
539 }
540 }
541
542 case ER_LOCK_DEADLOCK:
543 return false; // Implemented in TransactionTask::Execute and DatabaseWorkerPool<T>::DirectCommitTransaction
544 // Query related errors - skip query
545 case ER_WRONG_VALUE_COUNT:
546 case ER_DUP_ENTRY:
547 return false;
548
549 // Outdated table or database structure - terminate core
550 case ER_BAD_FIELD_ERROR:
551 case ER_NO_SUCH_TABLE:
552 TC_LOG_ERROR("sql.sql", "Your database structure is not up to date. Please make sure you've executed all queries in the sql/updates folders.");
553 std::this_thread::sleep_for(std::chrono::seconds(10));
554 std::abort();
555 return false;
556 case ER_PARSE_ERROR:
557 TC_LOG_ERROR("sql.sql", "Error while parsing SQL. Core fix required.");
558 std::this_thread::sleep_for(std::chrono::seconds(10));
559 std::abort();
560 return false;
561 default:
562 TC_LOG_ERROR("sql.sql", "Unhandled MySQL errno %u. Unexpected behaviour possible.", errNo);
563 return false;
564 }

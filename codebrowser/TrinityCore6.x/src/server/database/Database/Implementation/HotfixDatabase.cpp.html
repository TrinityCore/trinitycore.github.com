20// Force locale statments to appear exactly in locale declaration order, right after normal data fetch statement
21#define PREPARE_LOCALE_STMT(stmtBase, sql, con) \
22 static_assert(stmtBase + 1 == stmtBase##_LOCALE, "Invalid prepared statement index for " #stmtBase "_LOCALE"); \
23 PrepareStatement(stmtBase##_LOCALE, sql, con);

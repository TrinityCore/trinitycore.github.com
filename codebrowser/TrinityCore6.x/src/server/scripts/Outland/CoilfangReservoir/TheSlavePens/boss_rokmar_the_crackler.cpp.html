37enum Events

39 EVENT_GRIEVOUS_WOUND = 1,
40 EVENT_ENSNARING_MOSS = 2,

44class boss_rokmar_the_crackler : public CreatureScript
45{
46 public:
47 boss_rokmar_the_crackler() : CreatureScript("boss_rokmar_the_crackler") { }
48
49 struct boss_rokmar_the_cracklerAI : public BossAI
50 {
51 boss_rokmar_the_cracklerAI(Creature* creature) : BossAI(creature, DATA_MENNU_THE_BETRAYER)
52 {
53 Initialize();
54 }
55
56 void Initialize()
57 {
58 rokmarFrenzy = false;
59 }
60
61 void Reset() override
62 {
63 _Reset();
64 Initialize();
65 }
66
67 void JustDied(Unit* /*killer*/) override
68 {
69 _JustDied();
70 }
71
72 void EnterCombat(Unit* /*who*/) override
73 {
74 _EnterCombat();
75 events.ScheduleEvent(EVENT_GRIEVOUS_WOUND, 10000);
76 events.ScheduleEvent(EVENT_ENSNARING_MOSS, 20000);
77 events.ScheduleEvent(EVENT_WATER_SPIT, 14000);
78 }
79
80 void KilledUnit(Unit* /*victim*/) override { }
81
82 void UpdateAI(uint32 diff) override
83 {
84 if (!UpdateVictim())
85 return;
86
87 events.Update(diff);
88
89 if (me->HasUnitState(UNIT_STATE_CASTING))
90 return;
91
92 while (uint32 eventId = events.ExecuteEvent())
93 {
94 switch (eventId)
95 {
96 case EVENT_GRIEVOUS_WOUND:
97 DoCastVictim(SPELL_GRIEVOUS_WOUND, true);
98 events.ScheduleEvent(EVENT_GRIEVOUS_WOUND, urand(20000, 30000));
99 break;
100 case EVENT_ENSNARING_MOSS:
101 DoCastAOE(SPELL_ENSNARING_MOSS);
102 events.ScheduleEvent(EVENT_ENSNARING_MOSS, urand(20000, 30000));
103 break;
104 case EVENT_WATER_SPIT:
105 DoCastAOE(SPELL_WATER_SPIT);
106 events.ScheduleEvent(EVENT_WATER_SPIT, urand(14000, 18000));
107 break;
108 default:
109 break;
110 }
111 }
112
113 if (HealthBelowPct(10) && !rokmarFrenzy)
114 {
115 DoCast(me, SPELL_FRENZY);
116 rokmarFrenzy = true;
117 }
118
119 DoMeleeAttackIfReady();
120 }
121
122 private:
123 bool rokmarFrenzy;
124 };
125
126 CreatureAI* GetAI(Creature* creature) const override
127 {
128 return new boss_rokmar_the_cracklerAI(creature);
129 }
130};

132void AddSC_boss_rokmar_the_crackler()

134 new boss_rokmar_the_crackler();
135}

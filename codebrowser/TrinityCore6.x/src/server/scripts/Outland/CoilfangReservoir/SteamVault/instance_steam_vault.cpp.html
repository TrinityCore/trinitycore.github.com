22class go_main_chambers_access_panel : public GameObjectScript
23{
24 public:
25 go_main_chambers_access_panel() : GameObjectScript("go_main_chambers_access_panel") { }
26
27 bool OnGossipHello(Player* /*player*/, GameObject* go) override
28 {
29 InstanceScript* instance = go->GetInstanceScript();
30 if (!instance)
31 return false;
32
33 if (go->GetEntry() == GO_ACCESS_PANEL_HYDRO && (instance->GetBossState(DATA_HYDROMANCER_THESPIA) == DONE || instance->GetBossState(DATA_HYDROMANCER_THESPIA) == SPECIAL))
34 instance->SetBossState(DATA_HYDROMANCER_THESPIA, SPECIAL);
35
36 if (go->GetEntry() == GO_ACCESS_PANEL_MEK && (instance->GetBossState(DATA_MEKGINEER_STEAMRIGGER) == DONE || instance->GetBossState(DATA_MEKGINEER_STEAMRIGGER) == SPECIAL))
37 instance->SetBossState(DATA_MEKGINEER_STEAMRIGGER, SPECIAL);
38
39 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
40 go->SetGoState(GO_STATE_ACTIVE);
41
42 return true;
43 }
44};
45
46ObjectData const gameObjectData[] =
47{
48 { GO_ACCESS_PANEL_HYDRO, DATA_ACCESS_PANEL_HYDRO },
49 { GO_ACCESS_PANEL_MEK, DATA_ACCESS_PANEL_MEK },
50 { 0, 0 } // END
51};

53class instance_steam_vault : public InstanceMapScript
54{
55 public:
56 instance_steam_vault() : InstanceMapScript(SteamVaultScriptName, 545) { }
57
58 struct instance_steam_vault_InstanceMapScript : public InstanceScript
59 {
60 instance_steam_vault_InstanceMapScript(Map* map) : InstanceScript(map)
61 {
62 SetHeaders(DataHeader);
63 SetBossNumber(EncounterCount);
64 LoadObjectData(nullptr, gameObjectData);
65
66 DistillerState = 0;
67 }
68
69 void OnCreatureCreate(Creature* creature) override
70 {
71 switch (creature->GetEntry())
72 {
73 case NPC_HYDROMANCER_THESPIA:
74 ThespiaGUID = creature->GetGUID();
75 break;
76 case NPC_MEKGINEER_STEAMRIGGER:
77 MekgineerGUID = creature->GetGUID();
78 break;
79 case NPC_WARLORD_KALITHRESH:
80 KalithreshGUID = creature->GetGUID();
81 break;
82 default:
83 break;
84 }
85 }
86
87 void OnGameObjectCreate(GameObject* go) override
88 {
89 switch (go->GetEntry())
90 {
91 case GO_MAIN_CHAMBERS_DOOR:
92 MainChambersDoorGUID = go->GetGUID();
93 break;
94 default:
95 break;
96 }
97
98 InstanceScript::OnGameObjectCreate(go);
99 }
100
101 ObjectGuid GetGuidData(uint32 type) const override
102 {
103 switch (type)
104 {
105 case DATA_HYDROMANCER_THESPIA:
106 return ThespiaGUID;
107 case DATA_MEKGINEER_STEAMRIGGER:
108 return MekgineerGUID;
109 case DATA_WARLORD_KALITHRESH:
110 return KalithreshGUID;
111 default:
112 break;
113 }
114 return ObjectGuid::Empty;
115 }
116
117 void SetData(uint32 type, uint32 data) override
118 {
119 if (type == DATA_DISTILLER)
120 DistillerState = data;
121 }
122
123 uint32 GetData(uint32 type) const override
124 {
125 if (type == DATA_DISTILLER)
126 return DistillerState;
127 return 0;
128 }
129
130 bool SetBossState(uint32 type, EncounterState state) override
131 {
132 if (!InstanceScript::SetBossState(type, state))
133 return false;
134
135 switch (type)
136 {
137 case DATA_HYDROMANCER_THESPIA:
138 if (state == DONE)
139 if (GameObject* panel = GetGameObject(DATA_ACCESS_PANEL_HYDRO))
140 panel->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
141 if (state == SPECIAL)
142 {
143 if (GetBossState(DATA_MEKGINEER_STEAMRIGGER) == SPECIAL)
144 HandleGameObject(MainChambersDoorGUID, true);
145
146 TC_LOG_DEBUG("scripts", "Instance Steamvault: Access panel used.");
147 }
148 break;
149 case DATA_MEKGINEER_STEAMRIGGER:
150 if (state == DONE)
151 if (GameObject* panel = GetGameObject(DATA_ACCESS_PANEL_MEK))
152 panel->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
153 if (state == SPECIAL)
154 {
155 if (GetBossState(DATA_HYDROMANCER_THESPIA) == SPECIAL)
156 HandleGameObject(MainChambersDoorGUID, true);
157
158 TC_LOG_DEBUG("scripts", "Instance Steamvault: Access panel used.");
159 }
160 break;
161 default:
162 break;
163 }
164
165 return true;
166 }
167
168 protected:
169 ObjectGuid ThespiaGUID;
170 ObjectGuid MekgineerGUID;
171 ObjectGuid KalithreshGUID;
172
173 ObjectGuid MainChambersDoorGUID;
174 uint8 DistillerState;
175 };
176
177 InstanceScript* GetInstanceScript(InstanceMap* map) const override
178 {
179 return new instance_steam_vault_InstanceMapScript(map);
180 }
181};
182
183void AddSC_instance_steam_vault()
184{
185 new go_main_chambers_access_panel();
186 new instance_steam_vault();
187}

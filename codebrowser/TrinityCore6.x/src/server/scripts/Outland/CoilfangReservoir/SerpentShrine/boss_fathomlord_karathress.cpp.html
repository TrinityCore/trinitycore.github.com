99//Fathom-Lord Karathress AI
100class boss_fathomlord_karathress : public CreatureScript
101{
102public:
103 boss_fathomlord_karathress() : CreatureScript("boss_fathomlord_karathress") { }

105 CreatureAI* GetAI(Creature* creature) const override

107 return GetInstanceAI<boss_fathomlord_karathressAI>(creature);

110 struct boss_fathomlord_karathressAI : public ScriptedAI
111 {
112 boss_fathomlord_karathressAI(Creature* creature) : ScriptedAI(creature)
113 {
114 Initialize();
115 instance = creature->GetInstanceScript();
116 }
117
118 void Initialize()
119 {
120 CataclysmicBolt_Timer = 10000;
121 Enrage_Timer = 600000; //10 minutes
122 SearNova_Timer = 20000 + rand32() % 40000; // 20 - 60 seconds
123
124 BlessingOfTides = false;
125 }
126
127 InstanceScript* instance;
128
129 uint32 CataclysmicBolt_Timer;
130 uint32 Enrage_Timer;
131 uint32 SearNova_Timer;
132
133 bool BlessingOfTides;
134
135 ObjectGuid Advisors[MAX_ADVISORS];
136
137 void Reset() override
138 {
139 Initialize();
140
141 ObjectGuid RAdvisors[MAX_ADVISORS];
142 RAdvisors[0] = instance->GetGuidData(DATA_SHARKKIS);
143 RAdvisors[1] = instance->GetGuidData(DATA_TIDALVESS);
144 RAdvisors[2] = instance->GetGuidData(DATA_CARIBDIS);
145 // Respawn of the 3 Advisors
146 for (uint8 i = 0; i < MAX_ADVISORS; ++i)
147 {
148 if (!RAdvisors[i].IsEmpty())
149 {
150 Creature* advisor = ObjectAccessor::GetCreature(*me, RAdvisors[i]);
151 if (advisor && !advisor->IsAlive())
152 {
153 advisor->Respawn();
154 advisor->AI()->EnterEvadeMode();
155 advisor->GetMotionMaster()->MoveTargetedHome();
156 }
157 }
158 }
159
160 instance->SetData(DATA_KARATHRESSEVENT, NOT_STARTED);
161 }
162
163 void EventSharkkisDeath()
164 {
165 Talk(SAY_GAIN_ABILITY1);
166 DoCast(me, SPELL_POWER_OF_SHARKKIS);
167 }
168
169 void EventTidalvessDeath()
170 {
171 Talk(SAY_GAIN_ABILITY2);
172 DoCast(me, SPELL_POWER_OF_TIDALVESS);
173 }
174
175 void EventCaribdisDeath()
176 {
177 Talk(SAY_GAIN_ABILITY3);
178 DoCast(me, SPELL_POWER_OF_CARIBDIS);
179 }
180
181 void GetAdvisors()
182 {
183 Advisors[0] = instance->GetGuidData(DATA_SHARKKIS);
184 Advisors[1] = instance->GetGuidData(DATA_TIDALVESS);
185 Advisors[2] = instance->GetGuidData(DATA_CARIBDIS);
186 }
187
188 void StartEvent(Unit* who)
189 {
190 GetAdvisors();
191
192 Talk(SAY_AGGRO);
193 DoZoneInCombat();
194
195 instance->SetGuidData(DATA_KARATHRESSEVENT_STARTER, who->GetGUID());
196 instance->SetData(DATA_KARATHRESSEVENT, IN_PROGRESS);
197 }
198
199 void KilledUnit(Unit* /*victim*/) override
200 {
201 Talk(SAY_SLAY);
202 }
203
204 void JustDied(Unit* /*killer*/) override
205 {
206 Talk(SAY_DEATH);
207
208 instance->SetData(DATA_FATHOMLORDKARATHRESSEVENT, DONE);
209
210 //support for quest 10944
211 me->SummonCreature(SEER_OLUM, OLUM_X, OLUM_Y, OLUM_Z, OLUM_O, TEMPSUMMON_TIMED_DESPAWN, 3600000);
212 }
213
214 void EnterCombat(Unit* who) override
215 {
216 StartEvent(who);
217 }
218
219 void UpdateAI(uint32 diff) override
220 {
221 //Only if not incombat check if the event is started
222 if (!me->IsInCombat() && instance->GetData(DATA_KARATHRESSEVENT))
223 {
224 if (Unit* target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_KARATHRESSEVENT_STARTER)))
225 {
226 AttackStart(target);
227 GetAdvisors();
228 }
229 }
230
231 //Return since we have no target
232 if (!UpdateVictim())
233 return;
234
235 //someone evaded!
236 if (!instance->GetData(DATA_KARATHRESSEVENT))
237 {
238 EnterEvadeMode();
239 return;
240 }
241
242 //CataclysmicBolt_Timer
243 if (CataclysmicBolt_Timer <= diff)
244 {
245 //select a random unit other than the main tank
246 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1);
247
248 //if there aren't other units, cast on the tank
249 if (!target)
250 target = me->GetVictim();
251
252 if (target)
253 DoCast(target, SPELL_CATACLYSMIC_BOLT);
254 CataclysmicBolt_Timer = 10000;
255 } else CataclysmicBolt_Timer -= diff;
256
257 //SearNova_Timer
258 if (SearNova_Timer <= diff)
259 {
260 DoCastVictim(SPELL_SEAR_NOVA);
261 SearNova_Timer = 20000 + rand32() % 40000;
262 } else SearNova_Timer -= diff;
263
264 //Enrage_Timer
265 if (Enrage_Timer <= diff)
266 {
267 DoCast(me, SPELL_ENRAGE);
268 Enrage_Timer = 90000;
269 } else Enrage_Timer -= diff;
270
271 //Blessing of Tides Trigger
272 if (!HealthAbovePct(75) && !BlessingOfTides)
273 {
274 BlessingOfTides = true;
275 bool continueTriggering = false;
276 for (uint8 i = 0; i < MAX_ADVISORS; ++i)
277 {
278 if (!Advisors[i].IsEmpty())
279 {
280 Creature* advisor = ObjectAccessor::GetCreature(*me, Advisors[i]);
281 if (advisor && advisor->IsAlive())
282 {
283 continueTriggering = true;
284 break;
285 }
286 }
287 }
288
289 if (continueTriggering)
290 {
291 DoCast(me, SPELL_BLESSING_OF_THE_TIDES);
292 me->Yell(SAY_GAIN_BLESSING_OF_TIDES, LANG_UNIVERSAL);
293 DoPlaySoundToSet(me, SOUND_GAIN_BLESSING_OF_TIDES);
294 }
295 }
296
297 DoMeleeAttackIfReady();
298 }
299 };

369 {
370 //Only if not incombat check if the event is started
371 if (!me->IsInCombat() && instance->GetData(DATA_KARATHRESSEVENT))
372 {
373 if (Unit* target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_KARATHRESSEVENT_STARTER)))
374 AttackStart(target);
375 }
376
377 //Return since we have no target
378 if (!UpdateVictim())
379 return;
380
381 //someone evaded!
382 if (!instance->GetData(DATA_KARATHRESSEVENT))
383 {
384 EnterEvadeMode();
385 return;
386 }
387
388 //LeechingThrow_Timer
389 if (LeechingThrow_Timer <= diff)
390 {
391 DoCastVictim(SPELL_LEECHING_THROW);
392 LeechingThrow_Timer = 20000;
393 } else LeechingThrow_Timer -= diff;
394
395 //Multishot_Timer
396 if (Multishot_Timer <= diff)
397 {
398 DoCastVictim(SPELL_MULTISHOT);
399 Multishot_Timer = 20000;
400 } else Multishot_Timer -= diff;
401
402 //TheBeastWithin_Timer
403 if (TheBeastWithin_Timer <= diff)
404 {
405 DoCast(me, SPELL_THE_BEAST_WITHIN);
406
407 Creature* Pet = ObjectAccessor::GetCreature(*me, SummonedPet);
408 if (Pet && Pet->IsAlive())
409 Pet->CastSpell(Pet, SPELL_PET_ENRAGE, true);
410
411 TheBeastWithin_Timer = 30000;
412 } else TheBeastWithin_Timer -= diff;
413
414 //Pet_Timer
415 if (Pet_Timer < diff && pet == false)
416 {
417 pet = true;
418 //uint32 spell_id;
419 uint32 pet_id;
420 if (!urand(0, 1))
421 {
422 //spell_id = SPELL_SUMMON_FATHOM_LURKER;
423 pet_id = CREATURE_FATHOM_LURKER;
424 }
425 else
426 {
427 //spell_id = SPELL_SUMMON_FATHOM_SPOREBAT;
428 pet_id = CREATURE_FATHOM_SPOREBAT;
429 }
430 //DoCast(me, spell_id, true);
431 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
432 {
433 if (Creature* Pet = DoSpawnCreature(pet_id, 0, 0, 0, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
434 {
435 Pet->AI()->AttackStart(target);
436 SummonedPet = Pet->GetGUID();
437 }
438 }
439 } else Pet_Timer -= diff;
440
441 DoMeleeAttackIfReady();
442 }

670 Cyclone->CastSpell(Cyclone, SPELL_CYCLONE_CYCLONE, true);
671 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
672 Cyclone->AI()->AttackStart(target);
673 }
674 }
675 else
676 Cyclone_Timer -= diff;
677
678 //Heal_Timer
679 if (Heal_Timer <= diff)
680 {
681 // It can be cast on any of the mobs
682 Unit* unit = NULL;
683
684 while (unit == NULL || !unit->IsAlive())
685 unit = selectAdvisorUnit();
686
687 if (unit && unit->IsAlive())
688 DoCast(unit, SPELL_HEAL);
689 Heal_Timer = 60000;
690 }
691 else
692 Heal_Timer -= diff;
693
694 DoMeleeAttackIfReady();
695 }
696
697 Unit* selectAdvisorUnit()
698 {
699 Unit* unit = NULL;
700 switch (rand32() % 4)
701 {
702 case 0:
703 unit = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_KARATHRESS));
704 break;
705 case 1:
706 unit = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_SHARKKIS));
707 break;
708 case 2:
709 unit = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_TIDALVESS));
710 break;
711 case 3:
712 unit = me;
713 break;
714 }
715 return unit;
716 }

29enum Spells
30{
31 SPELL_LEVITATE = 31704,
32 SPELL_SUSPENSION = 31719,
33 SPELL_LEVITATION_PULSE = 31701,
34 SPELL_MAGNETIC_PULL = 31705,
35 SPELL_CHAIN_LIGHTNING = 31717,
36 SPELL_STATIC_CHARGE = 31715,
37 SPELL_SUMMON_SPORE_STRIDER = 38755
38};

40enum CreatureIdS
41{
42 ENTRY_SPORE_STRIDER = 22299

45class boss_the_black_stalker : public CreatureScript
46{
47public:
48 boss_the_black_stalker() : CreatureScript("boss_the_black_stalker") { }
49
50 CreatureAI* GetAI(Creature* creature) const override
51 {
52 return new boss_the_black_stalkerAI(creature);
53 }
54
55 struct boss_the_black_stalkerAI : public ScriptedAI
56 {
57 boss_the_black_stalkerAI(Creature* creature) : ScriptedAI(creature), Striders(creature)
58 {
59 Initialize();
60 InAir = false;
61 }
62
63 void Initialize()
64 {
65 Levitate_Timer = 12000;
66 ChainLightning_Timer = 6000;
67 StaticCharge_Timer = 10000;
68 SporeStriders_Timer = 10000 + rand32() % 5000;
69 check_Timer = 5000;
70 LevitatedTarget.Clear();
71 LevitatedTarget_Timer = 0;
72 }
73
74 uint32 SporeStriders_Timer;
75 uint32 Levitate_Timer;
76 uint32 ChainLightning_Timer;
77 uint32 StaticCharge_Timer;
78 ObjectGuid LevitatedTarget;
79 uint32 LevitatedTarget_Timer;
80 bool InAir;
81 uint32 check_Timer;
82 SummonList Striders;
83
84 void Reset() override
85 {
86 Initialize();
87 Striders.DespawnAll();
88 }
89
90 void EnterCombat(Unit* /*who*/) override { }
91
92 void JustSummoned(Creature* summon) override
93 {
94 if (summon && summon->GetEntry() == ENTRY_SPORE_STRIDER)
95 {
96 Striders.Summon(summon);
97 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
98 summon->AI()->AttackStart(target);
99 else
100 if (me->GetVictim())
101 summon->AI()->AttackStart(me->GetVictim());
102 }
103 }
104
105 void JustDied(Unit* /*killer*/) override
106 {
107 Striders.DespawnAll();
108 }
109
110 void UpdateAI(uint32 diff) override
111 {
112 if (!UpdateVictim())
113 return;
114
115 // Evade if too far
116 if (check_Timer <= diff)
117 {
118 float x, y, z, o;
119 me->GetHomePosition(x, y, z, o);
120 if (!me->IsWithinDist3d(x, y, z, 60))
121 {
122 EnterEvadeMode();
123 return;
124 }
125 check_Timer = 1000;
126 } else check_Timer -= diff;
127
128 // Spore Striders
129 if (IsHeroic() && SporeStriders_Timer <= diff)
130 {
131 DoCast(me, SPELL_SUMMON_SPORE_STRIDER);
132 SporeStriders_Timer = 10000 + rand32() % 5000;
133 } else SporeStriders_Timer -= diff;
134
135 // Levitate
136 if (!LevitatedTarget.IsEmpty())
137 {
138 if (LevitatedTarget_Timer <= diff)
139 {
140 if (Unit* target = ObjectAccessor::GetUnit(*me, LevitatedTarget))
141 {
142 if (!target->HasAura(SPELL_LEVITATE))
143 {
144 LevitatedTarget.Clear();
145 return;
146 }
147 if (InAir)
148 {
149 target->AddAura(SPELL_SUSPENSION, target);
150 LevitatedTarget.Clear();
151 }
152 else
153 {
154 target->CastSpell(target, SPELL_MAGNETIC_PULL, true);
155 InAir = true;
156 LevitatedTarget_Timer = 1500;
157 }
158 }
159 else
160 LevitatedTarget.Clear();
161 } else LevitatedTarget_Timer -= diff;
162 }
163 if (Levitate_Timer <= diff)
164 {
165 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
166 {
167 DoCast(target, SPELL_LEVITATE);
168 LevitatedTarget = target->GetGUID();
169 LevitatedTarget_Timer = 2000;
170 InAir = false;
171 }
172 Levitate_Timer = 12000 + rand32() % 3000;
173 } else Levitate_Timer -= diff;
174
175 // Chain Lightning
176 if (ChainLightning_Timer <= diff)
177 {
178 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
179 DoCast(target, SPELL_CHAIN_LIGHTNING);
180 ChainLightning_Timer = 7000;
181 } else ChainLightning_Timer -= diff;
182
183 // Static Charge
184 if (StaticCharge_Timer <= diff)
185 {
186 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30, true))
187 DoCast(target, SPELL_STATIC_CHARGE);
188 StaticCharge_Timer = 10000;
189 } else StaticCharge_Timer -= diff;
190
191 DoMeleeAttackIfReady();
192 }
193 };
194
195};

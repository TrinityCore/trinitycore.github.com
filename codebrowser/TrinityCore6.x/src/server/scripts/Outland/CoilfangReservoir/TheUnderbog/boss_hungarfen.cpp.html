29enum Spells
30{
31 SPELL_FOUL_SPORES = 31673,
32 SPELL_ACID_GEYSER = 38739,
33
34 SPELL_SPORE_CLOUD = 34168,
35 SPELL_PUTRID_MUSHROOM = 31690,
36 SPELL_GROW = 31698
37};
38
39class boss_hungarfen : public CreatureScript
40{
41public:
42 boss_hungarfen() : CreatureScript("boss_hungarfen") { }
43
44 CreatureAI* GetAI(Creature* creature) const override
45 {
46 return new boss_hungarfenAI(creature);
47 }
48
49 struct boss_hungarfenAI : public ScriptedAI
50 {
51 boss_hungarfenAI(Creature* creature) : ScriptedAI(creature)
52 {
53 Initialize();
54 }
55
56 void Initialize()
57 {
58 Root = false;
59 Mushroom_Timer = 5000; // 1 mushroom after 5s, then one per 10s. This should be different in heroic mode
60 AcidGeyser_Timer = 10000;
61 }
62
63 bool Root;
64 uint32 Mushroom_Timer;
65 uint32 AcidGeyser_Timer;
66
67 void Reset() override
68 {
69 Initialize();
70 }
71
72 void EnterCombat(Unit* /*who*/) override
73 {
74 }
75
76 void UpdateAI(uint32 diff) override
77 {
78 if (!UpdateVictim())
79 return;
80
81 if (!HealthAbovePct(20))
82 {
83 if (!Root)
84 {
85 DoCast(me, SPELL_FOUL_SPORES);
86 Root = true;
87 }
88 }
89
90 if (Mushroom_Timer <= diff)
91 {
92 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
93 me->SummonCreature(17990, target->GetPositionX() + (rand32() % 8), target->GetPositionY() + (rand32() % 8), target->GetPositionZ(), float(rand32() % 5), TEMPSUMMON_TIMED_DESPAWN, 22000);
94 else
95 me->SummonCreature(17990, me->GetPositionX() + (rand32() % 8), me->GetPositionY() + (rand32() % 8), me->GetPositionZ(), float(rand32() % 5), TEMPSUMMON_TIMED_DESPAWN, 22000);
96
97 Mushroom_Timer = 10000;
98 } else Mushroom_Timer -= diff;
99
100 if (AcidGeyser_Timer <= diff)
101 {
102 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
103 DoCast(target, SPELL_ACID_GEYSER);
104 AcidGeyser_Timer = 10000 + rand32() % 7500;
105 } else AcidGeyser_Timer -= diff;
106
107 DoMeleeAttackIfReady();
108 }
109 };
110
111};

113class npc_underbog_mushroom : public CreatureScript
114{
115public:
116 npc_underbog_mushroom() : CreatureScript("npc_underbog_mushroom") { }
117
118 CreatureAI* GetAI(Creature* creature) const override
119 {
120 return new npc_underbog_mushroomAI(creature);
121 }
122
123 struct npc_underbog_mushroomAI : public ScriptedAI
124 {
125 npc_underbog_mushroomAI(Creature* creature) : ScriptedAI(creature)
126 {
127 Initialize();
128 }
129
130 void Initialize()
131 {
132 Stop = false;
133 Grow_Timer = 0;
134 Shrink_Timer = 20000;
135 }
136
137 bool Stop;
138 uint32 Grow_Timer;
139 uint32 Shrink_Timer;
140
141 void Reset() override
142 {
143 Initialize();
144
145 DoCast(me, SPELL_PUTRID_MUSHROOM, true);
146 DoCast(me, SPELL_SPORE_CLOUD, true);
147 }
148
149 void MoveInLineOfSight(Unit* /*who*/) override { }
150
151
152 void AttackStart(Unit* /*who*/) override { }
153
154 void EnterCombat(Unit* /*who*/) override { }
155
156 void UpdateAI(uint32 diff) override
157 {
158 if (Stop)
159 return;
160
161 if (Grow_Timer <= diff)
162 {
163 DoCast(me, SPELL_GROW);
164 Grow_Timer = 3000;
165 } else Grow_Timer -= diff;
166
167 if (Shrink_Timer <= diff)
168 {
169 me->RemoveAurasDueToSpell(SPELL_GROW);
170 Stop = true;
171 } else Shrink_Timer -= diff;
172 }
173 };
174
175};
176
177void AddSC_boss_hungarfen()
178{
179 new boss_hungarfen();
180 new npc_underbog_mushroom();
181}

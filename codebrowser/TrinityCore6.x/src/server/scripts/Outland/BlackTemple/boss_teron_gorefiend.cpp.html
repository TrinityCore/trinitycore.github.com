108 Despawn();
109 }
110 else
111 Despawn();
112
113 CheckTeronTimer = 5000;
114 } else CheckTeronTimer -= diff;
115
116 if (ShadowBoltTimer < diff && me->IsInCombat())
117 {
118 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0), SPELL_SHADOWBOLT);
119 ShadowBoltTimer = 10000;
120 } else ShadowBoltTimer -= diff;
121 return;
122 }

187 void CheckPlayers()
188 {
189 ThreatContainer::StorageType const &threatlist = me->getThreatManager().getThreatList();
190 if (threatlist.empty())
191 return; // No threat list. Don't continue.
192 ThreatContainer::StorageType::const_iterator itr = threatlist.begin();
193 std::list<Unit*> targets;
194 for (; itr != threatlist.end(); ++itr)
195 {
196 Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid());
197 if (unit && unit->IsAlive())
198 targets.push_back(unit);
199 }
200 targets.sort(Trinity::ObjectDistanceOrderPred(me));
201 Unit* target = targets.front();
202 if (target && me->IsWithinDistInMap(target, me->GetAttackDistance(target)))
203 {
204 DoCast(target, SPELL_ATROPHY);
205 AttackStart(target);
206 }
207 }
208
209 void UpdateAI(uint32 diff) override
210 {
211 if (CheckPlayerTimer <= diff)
212 {
213 CheckPlayers();
214 CheckPlayerTimer = 3000;
215 } else CheckPlayerTimer -= diff;
216
217 if (CheckTeronTimer <= diff)
218 {
219 Creature* Teron = (ObjectAccessor::GetCreature((*me), TeronGUID));
220 if (!Teron || !Teron->IsAlive() || Teron->IsInEvadeMode())
221 me->DealDamage(me, me->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
222
223 CheckTeronTimer = 5000;
224 } else CheckTeronTimer -= diff;
225 }

324 float CalculateRandomLocation(float Loc, uint32 radius)
325 {
326 float coord = Loc;
327 switch (urand(0, 1))
328 {
329 case 0:
330 coord += rand32() % radius;
331 break;
332 case 1:
333 coord -= rand32() % radius;
334 break;
335 }
336 return coord;
337 }

339 void SetThreatList(Creature* blossom)
340 {
341 if (!blossom)
342 return;
343
344 ThreatContainer::StorageType const &threatlist = me->getThreatManager().getThreatList();
345 ThreatContainer::StorageType::const_iterator i = threatlist.begin();
346 for (i = threatlist.begin(); i != threatlist.end(); ++i)
347 {
348 Unit* unit = ObjectAccessor::GetUnit(*me, (*i)->getUnitGuid());
349 if (unit && unit->IsAlive())
350 {
351 float threat = DoGetThreat(unit);
352 blossom->AddThreat(unit, threat);
353 }
354 }

404 if (Intro && !Done)
405 {

410 Talk(SAY_AGGRO);
411 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_NONE);
412 Done = true;
413 if (!AggroTargetGUID.IsEmpty())
414 {
415 Unit* unit = ObjectAccessor::GetUnit(*me, AggroTargetGUID);
416 if (unit)
417 AttackStart(unit);
418
419 DoZoneInCombat();
420 }
421 else
422 {
423 EnterEvadeMode();
424 return;
425 }
426 } else AggroTimer -= diff;
427 }
428
429 if (!UpdateVictim() || !Done)
430 return;

432 if (SummonShadowsTimer <= diff)

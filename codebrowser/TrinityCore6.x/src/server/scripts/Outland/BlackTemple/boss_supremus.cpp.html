86class boss_supremus : public CreatureScript
87{
88public:
89 boss_supremus() : CreatureScript("boss_supremus") { }
90
91 struct boss_supremusAI : public BossAI
92 {
93 boss_supremusAI(Creature* creature) : BossAI(creature, DATA_SUPREMUS)
94 {
95 phase = 0;
96 }
97
98 void Reset() override
99 {
100 _Reset();
101 phase = 0;
102 }
103
104 void EnterCombat(Unit* /*who*/) override
105 {
106 _EnterCombat();
107 ChangePhase();
108 events.ScheduleEvent(EVENT_BERSERK, 900000, GCD_CAST);
109 events.ScheduleEvent(EVENT_FLAME, 20000, GCD_CAST);
110 }
111
112 void ChangePhase()
113 {
114 if (!phase || phase == PHASE_CHASE)
115 {
116 phase = PHASE_STRIKE;
117 DummyEntryCheckPredicate pred;
118 summons.DoAction(EVENT_VOLCANO, pred);
119 events.ScheduleEvent(EVENT_HATEFUL_STRIKE, 5000, GCD_CAST, PHASE_STRIKE);
120 me->SetSpeed(MOVE_RUN, 1.2f);
121 me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, false);
122 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, false);
123 }
124 else
125 {
126 phase = PHASE_CHASE;
127 events.ScheduleEvent(EVENT_VOLCANO, 5000, GCD_CAST, PHASE_CHASE);
128 events.ScheduleEvent(EVENT_SWITCH_TARGET, 10000, 0, PHASE_CHASE);
129 me->SetSpeed(MOVE_RUN, 0.9f);
130 me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, true);
131 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, true);
132 }
133 DoResetThreat();
134 DoZoneInCombat();
135 events.SetPhase(phase);
136 events.ScheduleEvent(EVENT_SWITCH_PHASE, 60000, GCD_CAST);
137 }
138
139 Unit* CalculateHatefulStrikeTarget()
140 {
141 uint32 health = 0;
142 Unit* target = NULL;
143
144 ThreatContainer::StorageType const &threatlist = me->getThreatManager().getThreatList();
145 ThreatContainer::StorageType::const_iterator i = threatlist.begin();
146 for (i = threatlist.begin(); i != threatlist.end(); ++i)
147 {
148 Unit* unit = ObjectAccessor::GetUnit(*me, (*i)->getUnitGuid());
149 if (unit && me->IsWithinMeleeRange(unit))
150 {
151 if (unit->GetHealth() > health)
152 {
153 health = unit->GetHealth();
154 target = unit;
155 }
156 }
157 }
158
159 return target;
160 }
161
162 void ExecuteEvent(uint32 eventId) override
163 {
164 switch (eventId)
165 {
166 case EVENT_BERSERK:
167 DoCast(me, SPELL_BERSERK, true);
168 break;
169 case EVENT_FLAME:
170 DoCast(me, SPELL_MOLTEN_PUNCH);
171 events.DelayEvents(1500, GCD_CAST);
172 events.ScheduleEvent(EVENT_FLAME, 20000, GCD_CAST);
173 break;
174 case EVENT_HATEFUL_STRIKE:
175 if (Unit* target = CalculateHatefulStrikeTarget())
176 DoCast(target, SPELL_HATEFUL_STRIKE);
177 events.DelayEvents(1000, GCD_CAST);
178 events.ScheduleEvent(EVENT_HATEFUL_STRIKE, 5000, GCD_CAST, PHASE_STRIKE);
179 break;
180 case EVENT_SWITCH_TARGET:
181 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true))
182 {
183 DoResetThreat();
184 me->AddThreat(target, 5000000.0f);
185 Talk(EMOTE_NEW_TARGET);
186 }
187 events.ScheduleEvent(EVENT_SWITCH_TARGET, 10000, 0, PHASE_CHASE);
188 break;
189 case EVENT_VOLCANO:
190 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 999, true))
191 {
192 //DoCast(target, SPELL_VOLCANIC_SUMMON);//movement bugged
193 me->SummonCreature(NPC_SUPREMUS_VOLCANO, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 30000);
194 Talk(EMOTE_GROUND_CRACK);
195 events.DelayEvents(1500, GCD_CAST);
196 }
197 events.ScheduleEvent(EVENT_VOLCANO, 10000, GCD_CAST, PHASE_CHASE);
198 return;
199 case EVENT_SWITCH_PHASE:
200 ChangePhase();
201 break;
202 default:
203 break;
204 }
205 }
206
207 private:
208 uint32 phase;
209 };
210
211 CreatureAI* GetAI(Creature* creature) const override
212 {
213 return GetBlackTempleAI<boss_supremusAI>(creature);
214 }

217class npc_volcano : public CreatureScript
218{
219public:
220 npc_volcano() : CreatureScript("npc_volcano") { }
221
222 struct npc_volcanoAI : public ScriptedAI
223 {
224 npc_volcanoAI(Creature* creature) : ScriptedAI(creature)
225 {
226 Initialize();
227 SetCombatMovement(false);
228 }
229
230 void Initialize()
231 {
232 wait = 3000;
233 }
234
235 void Reset() override
236 {
237 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
238 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
239 //DoCast(me, SPELL_VOLCANIC_ERUPTION);
240 me->SetReactState(REACT_PASSIVE);
241 Initialize();
242 }
243 uint32 wait;
244
245 void EnterCombat(Unit* /*who*/) override { }
246
247 void MoveInLineOfSight(Unit* /*who*/) override { }
248
249 void DoAction(int32 /*info*/) override
250 {
251 me->RemoveAura(SPELL_VOLCANIC_ERUPTION);
252 }
253
254 void UpdateAI(uint32 diff) override
255 {
256 if (wait <= diff)//wait 3secs before casting
257 {
258 DoCast(me, SPELL_VOLCANIC_ERUPTION);
259 wait = 60000;
260 }
261 else wait -= diff;
262 }
263 };
264
265 CreatureAI* GetAI(Creature* creature) const override
266 {
267 return new npc_volcanoAI(creature);
268 }

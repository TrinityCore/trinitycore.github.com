153class boss_shade_of_akama : public CreatureScript
154{
155public:
156 boss_shade_of_akama() : CreatureScript("boss_shade_of_akama") { }
157
158 struct boss_shade_of_akamaAI : public BossAI
159 {
160 boss_shade_of_akamaAI(Creature* creature) : BossAI(creature, DATA_SHADE_OF_AKAMA)
161 {
162 Initialize();
163 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
164 }
165
166 void Initialize()
167 {
168 combatStarted = false;
169 akamaReached = false;
170 HasKilledAkama = false;
171 HasKilledAkamaAndReseting = false;
172 }
173
174 void Reset() override
175 {
176 _Reset();
177 if (!HasKilledAkamaAndReseting)
178 {
179 for (GuidList::const_iterator itr = Channelers.begin(); itr != Channelers.end(); ++itr)
180 if (Creature* Channeler = ObjectAccessor::GetCreature(*me, *itr))
181 Channeler->DespawnOrUnsummon();
182
183 for (GuidList::const_iterator itr = Spawners.begin(); itr != Spawners.end(); ++itr)
184 if (Creature* Spawner = ObjectAccessor::GetCreature(*me, *itr))
185 Spawner->AI()->DoAction(ACTION_DESPAWN_ALL_SPAWNS);
186
187 events.ScheduleEvent(EVENT_FIND_CHANNELERS_SPAWNERS, 3000);
188 events.ScheduleEvent(EVENT_RESET_ENCOUNTER, 5000);
189 }
190
191 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
192
193 me->SetWalk(true);
194 Initialize();
195 }
196
197 void EnterCombat(Unit* /*who*/) override { }
198
199 void AttackStart(Unit* who) override
200 {
201 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE))
202 {
203 if (Creature* Akama = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_AKAMA_SHADE)))
204 if (Akama->IsAlive())
205 ScriptedAI::AttackStart(Akama);
206 }
207 else
208 ScriptedAI::AttackStart(who);
209 }
210
211 void DoAction(int32 actionId) override
212 {
213 if (actionId == ACTION_CHANNELER_DIED)
214 me->RemoveAuraFromStack(SPELL_SHADE_SOUL_CHANNEL_2);
215
216 UpdateSpeed();
217 }
218
219 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
220 {
221 if (spell->Id == SPELL_AKAMA_SOUL_CHANNEL)
222 {
223 combatStarted = true;
224 events.ScheduleEvent(EVENT_START_ATTACK_AKAMA, 500);
225 events.ScheduleEvent(EVENT_SET_CHANNELERS_SPAWNERS, 1000);
226 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_NONE);
227 if (Creature* Akama = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_AKAMA_SHADE)))
228 me->AddThreat(Akama, 10000000.0f);
229 }
230 else if (spell->Id == SPELL_SHADE_SOUL_CHANNEL_2)
231 UpdateSpeed();
232 }
233
234 void UpdateSpeed()
235 {
236 float moveSpeed = 0.2f;
237
238 if (me->GetAuraCount(SPELL_SHADE_SOUL_CHANNEL_2) <= 3)
239 {
240 moveSpeed = (2.0f - (0.6f * me->GetAuraCount(SPELL_SHADE_SOUL_CHANNEL_2)));
241 me->SetSpeed(MOVE_WALK, moveSpeed / 2.5f);
242 me->SetSpeed(MOVE_RUN, (moveSpeed * 2) / 7);
243 me->ClearUnitState(UNIT_STATE_ROOT);
244 }
245 else
246 me->AddUnitState(UNIT_STATE_ROOT);
247 }
248
249 void UpdateAI(uint32 diff) override
250 {
251 if (HasKilledAkamaAndReseting)
252 return;
253
254 events.Update(diff);
255
256 if (!combatStarted)
257 {
258 while (uint32 eventId = events.ExecuteEvent())
259 {
260 switch (eventId)
261 {
262 case EVENT_RESET_ENCOUNTER:
263 if (Creature* Akama = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_AKAMA_SHADE)))
264 if (!Akama->IsAlive())
265 Akama->Respawn();
266 break;
267 case EVENT_FIND_CHANNELERS_SPAWNERS:
268 {
269 std::list<Creature*> ChannelerList;
270 me->GetCreatureListWithEntryInGrid(ChannelerList, NPC_ASHTONGUE_CHANNELER, 15.0f);
271
272 if (!ChannelerList.empty())
273 for (std::list<Creature*>::const_iterator itr = ChannelerList.begin(); itr != ChannelerList.end(); ++itr)
274 {
275 Channelers.push_back((*itr)->GetGUID());
276 if ((*itr)->isDead())
277 (*itr)->Respawn();
278 }
279
280 std::list<Creature*> SpawnerList;
281 me->GetCreatureListWithEntryInGrid(SpawnerList, NPC_CREATURE_SPAWNER_AKAMA, 90.0f);
282
283 if (!SpawnerList.empty())
284 for (std::list<Creature*>::const_iterator itr = SpawnerList.begin(); itr != SpawnerList.end(); ++itr)
285 Spawners.push_back((*itr)->GetGUID());
286
287 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_STUN);
288 break;
289 }
290 default:
291 break;
292 }
293 }
294 }
295 else
296 {
297 while (uint32 eventId = events.ExecuteEvent())
298 {
299 switch (eventId)
300 {
301 case EVENT_SET_CHANNELERS_SPAWNERS:
302 {
303 for (GuidList::const_iterator itr = Channelers.begin(); itr != Channelers.end(); ++itr)
304 if (Creature* Channeler = ObjectAccessor::GetCreature(*me, *itr))
305 Channeler->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
306
307 for (GuidList::const_iterator itr = Spawners.begin(); itr != Spawners.end(); ++itr)
308 if (Creature* Spawner = ObjectAccessor::GetCreature(*me, *itr))
309 Spawner->AI()->DoAction(ACTION_START_SPAWNING);
310 break;
311 }
312 case EVENT_START_ATTACK_AKAMA:
313 me->GetMotionMaster()->MovePoint(0, ShadeWP.x, ShadeWP.y, ShadeWP.z, false);
314 events.ScheduleEvent(EVENT_START_ATTACK_AKAMA, 1000);
315 break;
316 case EVENT_ADD_THREAT:
317 DoCast(SPELL_THREAT);
318 events.ScheduleEvent(EVENT_ADD_THREAT, 3500);
319 break;
320 default:
321 break;
322 }
323 }
324
325 if (HasKilledAkama)
326 {
327 if (!HasKilledAkamaAndReseting)
328 {
329 HasKilledAkamaAndReseting = true;
330 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
331 instance->SetBossState(DATA_SHADE_OF_AKAMA, NOT_STARTED);
332 me->RemoveAllAurasExceptType(SPELL_AURA_DUMMY);
333 me->DeleteThreatList();
334 me->CombatStop();
335 me->GetMotionMaster()->MoveTargetedHome();
336 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
337 combatStarted = false;
338
339 if (Creature* Akama = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_AKAMA_SHADE)))
340 Akama->DespawnOrUnsummon();
341
342 for (GuidList::const_iterator itr = Channelers.begin(); itr != Channelers.end(); ++itr)
343 if (Creature* Channeler = ObjectAccessor::GetCreature(*me, *itr))
344 Channeler->DespawnOrUnsummon();
345
346 for (GuidList::const_iterator itr = Spawners.begin(); itr != Spawners.end(); ++itr)
347 if (Creature* Spawner = ObjectAccessor::GetCreature(*me, *itr))
348 Spawner->AI()->DoAction(ACTION_DESPAWN_ALL_SPAWNS);
349
350 events.ScheduleEvent(EVENT_FIND_CHANNELERS_SPAWNERS, 10000);
351 events.ScheduleEvent(EVENT_RESET_ENCOUNTER, 20000);
352 }
353 }
354
355 if (!akamaReached)
356 {
357 if (Creature* Akama = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_AKAMA_SHADE)))
358 {
359 if (me->IsWithinDist(Akama, 2.0f, false))
360 {
361 akamaReached = true;
362 me->GetMotionMaster()->Clear(true);
363 me->GetMotionMaster()->MoveIdle();
364 me->SetWalk(false);
365
366 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
367
368 events.CancelEvent(EVENT_START_ATTACK_AKAMA);
369 events.ScheduleEvent(EVENT_ADD_THREAT, 100);
370
371 for (GuidList::const_iterator itr = Spawners.begin(); itr != Spawners.end(); ++itr)
372 if (Creature* Spawner = ObjectAccessor::GetCreature(*me, *itr))
373 Spawner->AI()->DoAction(ACTION_STOP_SPAWNING);
374 }
375 }
376 }
377 else
378 DoMeleeAttackIfReady();
379 }
380 }
381
382 public:
383 bool HasKilledAkama;
384 private:
385 GuidList Channelers;
386 GuidList Spawners;
387 bool akamaReached;
388 bool combatStarted;
389 bool HasKilledAkamaAndReseting;
390 };
391
392 CreatureAI* GetAI(Creature* creature) const override
393 {
394 return GetInstanceAI<boss_shade_of_akamaAI>(creature);
395 }
396};

402class npc_akama_shade : public CreatureScript
403{
404public:
405 npc_akama_shade() : CreatureScript("npc_akama_shade") { }

407 struct npc_akamaAI : public ScriptedAI
408 {
409 npc_akamaAI(Creature* creature) : ScriptedAI(creature)
410 {
411 Initialize();
412 instance = creature->GetInstanceScript();
413 }
414
415 void Initialize()
416 {
417 StartChannel = false;
418 StartCombat = false;
419 HasYelledOnce = false;
420 ShadeHasDied = false;
421 }
422
423 void Reset() override
424 {
425 me->setFaction(FACTION_FRIENDLY);
426 DoCast(me, SPELL_STEALTH);
427 Initialize();
428
429 if (instance->GetBossState(DATA_SHADE_OF_AKAMA) != DONE)
430 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
431 }
432
433 void JustDied(Unit* /*killer*/) override
434 {
435 if (Creature* Shade = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SHADE_OF_AKAMA)))
436 if (Shade->IsAlive())
437 ENSURE_AI(boss_shade_of_akama::boss_shade_of_akamaAI, Shade->AI())->HasKilledAkama = true;
438 me->GetMotionMaster()->Clear(true);
439 me->GetMotionMaster()->MoveIdle();
440 }
441
442 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
443 {
444 if (spell->Id == SPELL_THREAT && !StartCombat)
445 {
446 me->ClearUnitState(UNIT_STATE_ROOT);
447 me->RemoveAura(SPELL_AKAMA_SOUL_CHANNEL);
448 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED);
449 if (Creature* Shade = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SHADE_OF_AKAMA)))
450 Shade->RemoveAura(SPELL_AKAMA_SOUL_CHANNEL);
451 StartCombat = true;
452 }
453 }
454
455 void EnterCombat(Unit* /*who*/) override
456 {
457 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 2000);
458 events.ScheduleEvent(EVENT_DESTRUCTIVE_POISON, 5000);
459 }
460
461 void UpdateAI(uint32 diff) override
462 {
463 if (StartChannel)
464 {
465 events.Update(diff);
466
467 while (uint32 eventId = events.ExecuteEvent())
468 {
469 switch (eventId)
470 {
471 case EVENT_SHADE_START:
472 instance->SetBossState(DATA_SHADE_OF_AKAMA, IN_PROGRESS);
473 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
474 me->RemoveAura(SPELL_STEALTH);
475 me->SetWalk(true);
476 me->GetMotionMaster()->MovePoint(0, AkamaWP[0].x, AkamaWP[0].y, AkamaWP[0].z, false);
477 events.ScheduleEvent(EVENT_SHADE_CHANNEL, 10000);
478 break;
479 case EVENT_SHADE_CHANNEL:
480 me->AddUnitState(UNIT_STATE_ROOT);
481 me->SetFacingTo(3.118662f);
482 DoCast(me, SPELL_AKAMA_SOUL_CHANNEL);
483 me->setFaction(FACTION_COMBAT);
484 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED);
485 events.ScheduleEvent(EVENT_FIXATE, 5000);
486 break;
487 case EVENT_FIXATE:
488 DoCast(SPELL_FIXATE);
489 StartChannel = false;
490 break;
491 default:
492 break;
493 }
494 }
495 }
496
497 if (!UpdateVictim())
498 return;
499
500 events.Update(diff);
501
502 while (uint32 eventId = events.ExecuteEvent())
503 {
504 switch (eventId)
505 {
506 case EVENT_CHAIN_LIGHTNING:
507 DoCastVictim(SPELL_CHAIN_LIGHTNING);
508 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, urand(10000, 15000));
509 break;
510 case EVENT_DESTRUCTIVE_POISON:
511 DoCast(me, SPELL_DESTRUCTIVE_POISON);
512 events.ScheduleEvent(EVENT_DESTRUCTIVE_POISON, urand(4000, 5000));
513 break;
514 default:
515 break;
516 }
517 }
518
519 DoMeleeAttackIfReady();
520 }
521
522 void sGossipSelect(Player* player, uint32 /*menuId*/, uint32 gossipListId) override
523 {
524 if (gossipListId == 0)
525 {
526 player->CLOSE_GOSSIP_MENU();
527 StartChannel = true;
528 events.ScheduleEvent(EVENT_SHADE_START, 500);
529 }
530 }
531
532 private:
533 InstanceScript* instance;
534 EventMap events;
535 bool StartChannel;
536 bool ShadeHasDied;
537 bool StartCombat;
538 bool HasYelledOnce;
539
540 };

544 return GetInstanceAI<npc_akamaAI>(creature);

629 struct npc_creature_generator_akamaAI : public ScriptedAI
630 {
631 npc_creature_generator_akamaAI(Creature* creature) : ScriptedAI(creature), Summons(me)
632 {
633 Initialize();
634 instance = creature->GetInstanceScript();
635 }
636
637 void Initialize()
638 {
639 doSpawning = false;
640 leftSide = false;
641 }
642
643 void Reset() override
644 {
645 Summons.DespawnAll();
646
647 Initialize();
648
649 if (me->GetPositionY() < 400.0f)
650 leftSide = true;
651 }
652
653 void JustSummoned(Creature* summon) override
654 {
655 Summons.Summon(summon);
656 }
657
658 void DoAction(int32 actionId) override
659 {
660 doSpawning = true;
661
662 switch (actionId)
663 {
664 case ACTION_START_SPAWNING:
665 if (leftSide)
666 {
667 events.ScheduleEvent(EVENT_SPAWN_WAVE_B, 100);
668 events.ScheduleEvent(EVENT_SUMMON_ASHTONGUE_SORCERER, urand(2000, 5000));
669 }
670 else
671 {
672 events.ScheduleEvent(EVENT_SPAWN_WAVE_B, 10000);
673 events.ScheduleEvent(EVENT_SUMMON_ASHTONGUE_DEFENDER, urand(2000, 5000));
674 }
675 break;
676 case ACTION_STOP_SPAWNING:
677 doSpawning = false;
678 break;
679 case ACTION_DESPAWN_ALL_SPAWNS:
680 doSpawning = false;
681 Summons.DespawnAll();
682 break;
683 default:
684 break;
685 }
686 }
687
688 void UpdateAI(uint32 diff) override
689 {
690 if (doSpawning)
691 {
692 events.Update(diff);
693
694 while (uint32 eventId = events.ExecuteEvent())
695 {
696 switch (eventId)
697 {
698 case EVENT_SPAWN_WAVE_B:
699 DoCast(me, SPELL_ASHTONGUE_WAVE_B);
700 events.ScheduleEvent(EVENT_SPAWN_WAVE_B, urand(45000, 50000));
701 break;
702 case EVENT_SUMMON_ASHTONGUE_SORCERER: // left
703 DoCast(me, SPELL_SUMMON_ASHTONGUE_SORCERER);
704 events.ScheduleEvent(EVENT_SUMMON_ASHTONGUE_SORCERER, urand(30000, 35000));
705 break;
706 case EVENT_SUMMON_ASHTONGUE_DEFENDER: // right
707 DoCast(me, SPELL_SUMMON_ASHTONGUE_DEFENDER);
708 events.ScheduleEvent(EVENT_SUMMON_ASHTONGUE_DEFENDER, urand(30000, 35000));
709 break;
710 default:
711 break;
712 }
713 }
714 }
715 }
716
717 private:
718 InstanceScript* instance;
719 EventMap events;
720 SummonList Summons;
721 bool leftSide;
722 bool doSpawning;
723 };

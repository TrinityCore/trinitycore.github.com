366class npc_flame_of_azzinoth : public CreatureScript
367{
368public:
369 npc_flame_of_azzinoth() : CreatureScript("npc_flame_of_azzinoth") { }
370
371 struct flame_of_azzinothAI : public ScriptedAI
372 {
373 flame_of_azzinothAI(Creature* creature) : ScriptedAI(creature)
374 {
375 Initialize();
376 }
377
378 void Initialize()
379 {
380 FlameBlastTimer = 15000;
381 CheckTimer = 5000;
382 GlaiveGUID.Clear();
383 }
384
385 void Reset() override
386 {
387 Initialize();
388 }
389
390 void EnterCombat(Unit* /*who*/) override
391 {
392 DoZoneInCombat();
393 }
394
395 void ChargeCheck()
396 {
397 Unit* target = SelectTarget(SELECT_TARGET_FARTHEST, 0, 200, false);
398 if (target && (!me->IsWithinCombatRange(target, FLAME_CHARGE_DISTANCE)))
399 {
400 me->AddThreat(target, 5000000.0f);
401 AttackStart(target);
402 DoCast(target, SPELL_CHARGE);
403 Talk(EMOTE_AZZINOTH_GAZE);
404 }
405 }
406
407 void EnrageCheck()
408 {
409 if (Creature* glaive = ObjectAccessor::GetCreature(*me, GlaiveGUID))
410 {
411 if (!me->IsWithinDistInMap(glaive, FLAME_ENRAGE_DISTANCE))
412 {
413 glaive->InterruptNonMeleeSpells(true);
414 DoCast(me, SPELL_FLAME_ENRAGE, true);
415 DoResetThreat();
416 if (SelectTarget(SELECT_TARGET_RANDOM, 0))
417 {
418 me->AddThreat(me->GetVictim(), 5000000.0f);
419 AttackStart(me->GetVictim());
420 }
421 }
422 else if (!me->HasAura(SPELL_AZZINOTH_CHANNEL))
423 {
424 glaive->CastSpell(me, SPELL_AZZINOTH_CHANNEL, false);
425 me->RemoveAurasDueToSpell(SPELL_FLAME_ENRAGE);
426 }
427 }
428 }
429
430 void SetGlaiveGUID(ObjectGuid guid)
431 {
432 GlaiveGUID = guid;
433 }
434
435 void UpdateAI(uint32 diff) override
436 {
437 if (!UpdateVictim())
438 return;
439
440 if (FlameBlastTimer <= diff)
441 {
442 DoCastVictim(SPELL_BLAZE_SUMMON, true); // appear at victim
443 DoCastVictim(SPELL_FLAME_BLAST);
444 FlameBlastTimer = 15000; // 10000 is official-like?
445 DoZoneInCombat(); // in case someone is revived
446 } else FlameBlastTimer -= diff;
447
448 if (CheckTimer <= diff)
449 {
450 ChargeCheck();
451 EnrageCheck();
452 CheckTimer = 1000;
453 } else CheckTimer -= diff;
454
455 DoMeleeAttackIfReady();
456 }
457
458 private:
459 uint32 FlameBlastTimer;
460 uint32 CheckTimer;
461 ObjectGuid GlaiveGUID;
462 };
463
464 CreatureAI* GetAI(Creature* creature) const override
465 {
466 return new flame_of_azzinothAI(creature);
467 }

57class boss_gurtogg_bloodboil : public CreatureScript
58{
59public:
60 boss_gurtogg_bloodboil() : CreatureScript("boss_gurtogg_bloodboil") { }
61
62 CreatureAI* GetAI(Creature* creature) const override
63 {
64 return GetInstanceAI<boss_gurtogg_bloodboilAI>(creature);
65 }
66
67 struct boss_gurtogg_bloodboilAI : public ScriptedAI
68 {
69 boss_gurtogg_bloodboilAI(Creature* creature) : ScriptedAI(creature)
70 {
71 Initialize();
72 instance = creature->GetInstanceScript();
73 }
74
75 void Initialize()
76 {
77 TargetGUID.Clear();
78 TargetThreat = 0;
79
80 BloodboilTimer = 10000;
81 BloodboilCount = 0;
82 AcidGeyserTimer = 1000;
83 AcidicWoundTimer = 6000;
84 ArcingSmashTimer = 19000;
85 EnrageTimer = 600000;
86 FelAcidTimer = 25000;
87 EjectTimer = 10000;
88 BewilderingStrikeTimer = 15000;
89 PhaseChangeTimer = 60000;
90
91 Phase1 = true;
92 }
93
94 InstanceScript* instance;
95
96 ObjectGuid TargetGUID;
97
98 float TargetThreat;
99
100 uint32 BloodboilTimer;
101 uint32 BloodboilCount;
102 uint32 AcidGeyserTimer;
103 uint32 AcidicWoundTimer;
104 uint32 ArcingSmashTimer;
105 uint32 EnrageTimer;
106 uint32 FelAcidTimer;
107 uint32 EjectTimer;
108 uint32 BewilderingStrikeTimer;
109 uint32 PhaseChangeTimer;
110
111 bool Phase1;
112
113 void Reset() override
114 {
115 instance->SetBossState(DATA_GURTOGG_BLOODBOIL, NOT_STARTED);
116
117 Initialize();
118
119 me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, false);
120 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, false);
121 }
122
123 void EnterCombat(Unit* /*who*/) override
124 {
125 DoZoneInCombat();
126 Talk(SAY_AGGRO);
127 instance->SetBossState(DATA_GURTOGG_BLOODBOIL, IN_PROGRESS);
128 }
129
130 void KilledUnit(Unit* /*victim*/) override
131 {
132 Talk(SAY_SLAY);
133 }
134
135 void JustDied(Unit* /*killer*/) override
136 {
137 instance->SetBossState(DATA_GURTOGG_BLOODBOIL, DONE);
138
139 Talk(SAY_DEATH);
140 }
141
142 void RevertThreatOnTarget(ObjectGuid guid)
143 {
144 if (Unit* unit = ObjectAccessor::GetUnit(*me, guid))
145 {
146 if (DoGetThreat(unit))
147 DoModifyThreatPercent(unit, -100);
148 if (TargetThreat)
149 me->AddThreat(unit, TargetThreat);
150 }
151 }
152
153 void UpdateAI(uint32 diff) override
154 {
155 if (!UpdateVictim())
156 return;
157
158 if (ArcingSmashTimer <= diff)
159 {
160 DoCastVictim(SPELL_ARCING_SMASH);
161 ArcingSmashTimer = 10000;
162 } else ArcingSmashTimer -= diff;
163
164 if (FelAcidTimer <= diff)
165 {
166 DoCastVictim(SPELL_FEL_ACID);
167 FelAcidTimer = 25000;
168 } else FelAcidTimer -= diff;
169
170 if (!me->HasAura(SPELL_BERSERK))
171 {
172 if (EnrageTimer <= diff)
173 {
174 DoCast(me, SPELL_BERSERK);
175 Talk(SAY_ENRAGE);
176 } else EnrageTimer -= diff;
177 }
178
179 if (Phase1)
180 {
181 if (BewilderingStrikeTimer <= diff)
182 {
183 DoCastVictim(SPELL_BEWILDERING_STRIKE);
184 float mt_threat = DoGetThreat(me->GetVictim());
185 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 1))
186 me->AddThreat(target, mt_threat);
187 BewilderingStrikeTimer = 20000;
188 } else BewilderingStrikeTimer -= diff;
189
190 if (EjectTimer <= diff)
191 {
192 DoCastVictim(SPELL_EJECT1);
193 DoModifyThreatPercent(me->GetVictim(), -40);
194 EjectTimer = 15000;
195 } else EjectTimer -= diff;
196
197 if (AcidicWoundTimer <= diff)
198 {
199 DoCastVictim(SPELL_ACIDIC_WOUND);
200 AcidicWoundTimer = 10000;
201 } else AcidicWoundTimer -= diff;
202
203 if (BloodboilTimer <= diff)
204 {
205 if (BloodboilCount < 5) // Only cast it five times.
206 {
207 DoCastAOE(SPELL_BLOODBOIL);
208 ++BloodboilCount;
209 BloodboilTimer = 10000*BloodboilCount;
210 }
211 } else BloodboilTimer -= diff;
212 }
213
214 if (!Phase1)
215 {
216 if (AcidGeyserTimer <= diff)
217 {
218 DoCastVictim(SPELL_ACID_GEYSER);
219 AcidGeyserTimer = 30000;
220 } else AcidGeyserTimer -= diff;
221
222 if (EjectTimer <= diff)
223 {
224 DoCastVictim(SPELL_EJECT2);
225 EjectTimer = 15000;
226 } else EjectTimer -= diff;
227 }
228
229 if (PhaseChangeTimer <= diff)
230 {
231 if (Phase1)
232 {
233 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
234 {
235 Phase1 = false;
236
237 TargetThreat = DoGetThreat(target);
238 TargetGUID = target->GetGUID();
239 target->CastSpell(me, SPELL_TAUNT_GURTOGG, true);
240 if (DoGetThreat(target))
241 DoModifyThreatPercent(target, -100);
242 me->AddThreat(target, 50000000.0f);
243 me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, true);
244 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, true);
245 // If VMaps are disabled, this spell can call the whole instance
246 DoCast(me, SPELL_INSIGNIFIGANCE, true);
247 DoCast(target, SPELL_FEL_RAGE_TARGET, true);
248 DoCast(target, SPELL_FEL_RAGE_2, true);
249 /* These spells do not work, comment them out for now.
250 DoCast(target, SPELL_FEL_RAGE_2, true);
251 DoCast(target, SPELL_FEL_RAGE_3, true);*/
252
253 //Cast this without triggered so that it appears in combat logs and shows visual.
254 DoCast(me, SPELL_FEL_RAGE_SELF);
255
256 Talk(SAY_SPECIAL);
257
258 AcidGeyserTimer = 1000;
259 PhaseChangeTimer = 30000;
260 }
261 }
262 else // Encounter is a loop pretty much. Phase 1 -> Phase 2 -> Phase 1 -> Phase 2 till death or enrage
263 {
264 if (!TargetGUID.IsEmpty())
265 RevertThreatOnTarget(TargetGUID);
266 TargetGUID.Clear();
267 Phase1 = true;
268 BloodboilTimer = 10000;
269 BloodboilCount = 0;
270 AcidicWoundTimer += 2000;
271 ArcingSmashTimer += 2000;
272 FelAcidTimer += 2000;
273 EjectTimer += 2000;
274 PhaseChangeTimer = 60000;
275 me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, false);
276 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, false);
277 }
278 } else PhaseChangeTimer -= diff;
279
280 DoMeleeAttackIfReady();
281 }
282 };

296 void FilterTargets(std::list<WorldObject*>& targets)
297 {
298 if (targets.size() <= 5)
299 return;
300
301 // Sort the list of players
302 targets.sort(Trinity::ObjectDistanceOrderPred(GetCaster(), false));
303 // Resize so we only get top 5
304 targets.resize(5);
305 }
306
307 void Register() override
308 {
309 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_gurtogg_bloodboil_bloodboil_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
310 }
311 };

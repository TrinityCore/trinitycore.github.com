22enum Spells
23{
24 // Wrathbone Flayer
25 SPELL_CLEAVE = 15496,
26 SPELL_IGNORED = 39544,
27 SPELL_SUMMON_CHANNEL = 40094
28};
29
30enum Creatures
31{
32 NPC_BLOOD_MAGE = 22945,
33 NPC_DEATHSHAPER = 22882
34};
35
36enum Events
37{
38 // Wrathbone Flayer
39 EVENT_GET_CHANNELERS = 1,
40 EVENT_SET_CHANNELERS = 2,
41 EVENT_CLEAVE = 3,
42 EVENT_IGNORED = 4,
43};

45// ########################################################
46// Wrathbone Flayer
47// ########################################################
48
49class npc_wrathbone_flayer : public CreatureScript
50{
51public:
52 npc_wrathbone_flayer() : CreatureScript("npc_wrathbone_flayer") { }
53
54 struct npc_wrathbone_flayerAI : public ScriptedAI
55 {
56 npc_wrathbone_flayerAI(Creature* creature) : ScriptedAI(creature)
57 {
58 Initialize();
59 _instance = creature->GetInstanceScript();
60 }
61
62 void Initialize()
63 {
64 _enteredCombat = false;
65 }
66
67 void Reset() override
68 {
69 _events.ScheduleEvent(EVENT_GET_CHANNELERS, 3000);
70 Initialize();
71 _bloodmageList.clear();
72 _deathshaperList.clear();
73 }
74
75 void JustDied(Unit* /*killer*/) override { }
76
77 void EnterCombat(Unit* /*who*/) override
78 {
79 _events.ScheduleEvent(EVENT_CLEAVE, 5000);
80 _events.ScheduleEvent(EVENT_IGNORED, 7000);
81 _enteredCombat = true;
82 }
83
84 void UpdateAI(uint32 diff) override
85 {
86 if (!_enteredCombat)
87 {
88 _events.Update(diff);
89
90 while (uint32 eventId = _events.ExecuteEvent())
91 {
92 switch (eventId)
93 {
94 case EVENT_GET_CHANNELERS:
95 {
96 std::list<Creature*> BloodMageList;
97 me->GetCreatureListWithEntryInGrid(BloodMageList, NPC_BLOOD_MAGE, 15.0f);
98
99 if (!BloodMageList.empty())
100 for (std::list<Creature*>::const_iterator itr = BloodMageList.begin(); itr != BloodMageList.end(); ++itr)
101 {
102 _bloodmageList.push_back((*itr)->GetGUID());
103 if ((*itr)->isDead())
104 (*itr)->Respawn();
105 }
106
107 std::list<Creature*> DeathShaperList;
108 me->GetCreatureListWithEntryInGrid(DeathShaperList, NPC_DEATHSHAPER, 15.0f);
109
110 if (!DeathShaperList.empty())
111 for (std::list<Creature*>::const_iterator itr = DeathShaperList.begin(); itr != DeathShaperList.end(); ++itr)
112 {
113 _deathshaperList.push_back((*itr)->GetGUID());
114 if ((*itr)->isDead())
115 (*itr)->Respawn();
116 }
117
118 _events.ScheduleEvent(EVENT_SET_CHANNELERS, 3000);
119
120 break;
121 }
122 case EVENT_SET_CHANNELERS:
123 {
124 for (ObjectGuid guid : _bloodmageList)
125 if (Creature* bloodmage = ObjectAccessor::GetCreature(*me, guid))
126 bloodmage->CastSpell((Unit*)NULL, SPELL_SUMMON_CHANNEL);
127
128 for (ObjectGuid guid : _deathshaperList)
129 if (Creature* deathshaper = ObjectAccessor::GetCreature(*me, guid))
130 deathshaper->CastSpell((Unit*)NULL, SPELL_SUMMON_CHANNEL);
131
132 _events.ScheduleEvent(EVENT_SET_CHANNELERS, 12000);
133
134 break;
135 }
136 default:
137 break;
138 }
139 }
140 }
141
142 if (!UpdateVictim())
143 return;
144
145 _events.Update(diff);
146
147 while (uint32 eventId = _events.ExecuteEvent())
148 {
149 switch (eventId)
150 {
151 case EVENT_CLEAVE:
152 DoCastVictim(SPELL_CLEAVE);
153 _events.ScheduleEvent(EVENT_CLEAVE, urand (1000, 2000));
154 break;
155 case EVENT_IGNORED:
156 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
157 DoCast(target, SPELL_IGNORED);
158 _events.ScheduleEvent(EVENT_IGNORED, 10000);
159 break;
160 default:
161 break;
162 }
163 }
164 DoMeleeAttackIfReady();
165 }
166
167 private:
168 InstanceScript* _instance;
169 EventMap _events;
170 GuidList _bloodmageList;
171 GuidList _deathshaperList;
172 bool _enteredCombat;
173 };
174
175 CreatureAI* GetAI(Creature* creature) const override
176 {
177 return GetBlackTempleAI<npc_wrathbone_flayerAI>(creature);
178 }
179};
180
181void AddSC_black_temple()
182{
183 new npc_wrathbone_flayer();
184}

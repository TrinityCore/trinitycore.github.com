34class instance_blood_furnace : public InstanceMapScript
35{
36 public:
37 instance_blood_furnace() : InstanceMapScript(BFScriptName, 542) { }

39 struct instance_blood_furnace_InstanceMapScript : public InstanceScript
40 {
41 instance_blood_furnace_InstanceMapScript(Map* map) : InstanceScript(map)
42 {
43 SetHeaders(DataHeader);
44 SetBossNumber(EncounterCount);
45 LoadDoorData(doorData);
46
47 PrisonerCounter5 = 0;
48 PrisonerCounter6 = 0;
49 PrisonerCounter7 = 0;
50 PrisonerCounter8 = 0;
51 }
52
53 void OnCreatureCreate(Creature* creature) override
54 {
55 switch (creature->GetEntry())
56 {
57 case NPC_THE_MAKER:
58 TheMakerGUID = creature->GetGUID();
59 break;
60 case NPC_BROGGOK:
61 BroggokGUID = creature->GetGUID();
62 break;
63 case NPC_KELIDAN_THE_BREAKER:
64 KelidanTheBreakerGUID = creature->GetGUID();
65 break;
66 case NPC_PRISONER:
67 StorePrisoner(creature);
68 break;
69 default:
70 break;
71 }
72 }
73
74 void OnUnitDeath(Unit* unit) override
75 {
76 if (unit->GetTypeId() == TYPEID_UNIT && unit->GetEntry() == NPC_PRISONER)
77 PrisonerDied(unit->GetGUID());
78 }
79
80 void OnGameObjectCreate(GameObject* go) override
81 {
82 switch (go->GetEntry())
83 {
84 case GO_PRISON_DOOR_04:
85 PrisonDoor4GUID = go->GetGUID();
86 // no break
87 case GO_PRISON_DOOR_01:
88 case GO_PRISON_DOOR_02:
89 case GO_PRISON_DOOR_03:
90 case GO_PRISON_DOOR_05:
91 case GO_SUMMON_DOOR:
92 AddDoor(go, true);
93 break;
94 case GO_BROGGOK_LEVER:
95 BroggokLeverGUID = go->GetGUID();
96 break;
97 case GO_PRISON_CELL_DOOR_1:
98 PrisonCellGUIDs[DATA_PRISON_CELL1 - DATA_PRISON_CELL1] = go->GetGUID();
99 break;
100 case GO_PRISON_CELL_DOOR_2:
101 PrisonCellGUIDs[DATA_PRISON_CELL2 - DATA_PRISON_CELL1] = go->GetGUID();
102 break;
103 case GO_PRISON_CELL_DOOR_3:
104 PrisonCellGUIDs[DATA_PRISON_CELL3 - DATA_PRISON_CELL1] = go->GetGUID();
105 break;
106 case GO_PRISON_CELL_DOOR_4:
107 PrisonCellGUIDs[DATA_PRISON_CELL4 - DATA_PRISON_CELL1] = go->GetGUID();
108 break;
109 case GO_PRISON_CELL_DOOR_5:
110 PrisonCellGUIDs[DATA_PRISON_CELL5 - DATA_PRISON_CELL1] = go->GetGUID();
111 break;
112 case GO_PRISON_CELL_DOOR_6:
113 PrisonCellGUIDs[DATA_PRISON_CELL6 - DATA_PRISON_CELL1] = go->GetGUID();
114 break;
115 case GO_PRISON_CELL_DOOR_7:
116 PrisonCellGUIDs[DATA_PRISON_CELL7 - DATA_PRISON_CELL1] = go->GetGUID();
117 break;
118 case GO_PRISON_CELL_DOOR_8:
119 PrisonCellGUIDs[DATA_PRISON_CELL8 - DATA_PRISON_CELL1] = go->GetGUID();
120 break;
121 default:
122 break;
123 }
124 }
125

143 ObjectGuid GetGuidData(uint32 type) const override
144 {
145 switch (type)

150 return BroggokGUID;
151 case DATA_KELIDAN_THE_BREAKER:

159
160 bool SetBossState(uint32 type, EncounterState state) override
161 {
162 if (!InstanceScript::SetBossState(type, state))
163 return false;
164
165 switch (type)
166 {
167 case DATA_BROGGOK:
168 switch (state)
169 {
170 case IN_PROGRESS:
171 ActivateCell(DATA_PRISON_CELL5);
172 break;
173 case NOT_STARTED:
174 ResetPrisons();
175 if (GameObject* lever = instance->GetGameObject(BroggokLeverGUID))
176 lever->Respawn();
177 break;
178 default:
179 break;
180 }
181 break;
182 default:
183 break;
184 }
185
186 return true;
187 }
188
189 void ResetPrisons()
190 {
191 PrisonerCounter5 = PrisonersCell5.size();
192 ResetPrisoners(PrisonersCell5);
193 HandleGameObject(PrisonCellGUIDs[DATA_PRISON_CELL5 - DATA_PRISON_CELL1], false);
194
195 PrisonerCounter6 = PrisonersCell6.size();
196 ResetPrisoners(PrisonersCell6);
197 HandleGameObject(PrisonCellGUIDs[DATA_PRISON_CELL6 - DATA_PRISON_CELL1], false);
198
199 PrisonerCounter7 = PrisonersCell7.size();
200 ResetPrisoners(PrisonersCell7);
201 HandleGameObject(PrisonCellGUIDs[DATA_PRISON_CELL7 - DATA_PRISON_CELL1], false);
202
203 PrisonerCounter8 = PrisonersCell8.size();
204 ResetPrisoners(PrisonersCell8);
205 HandleGameObject(PrisonCellGUIDs[DATA_PRISON_CELL8 - DATA_PRISON_CELL1], false);
206 }
207
208 void ResetPrisoners(GuidSet const& prisoners)
209 {
210 for (GuidSet::const_iterator i = prisoners.begin(); i != prisoners.end(); ++i)
211 if (Creature* prisoner = instance->GetCreature(*i))
212 ResetPrisoner(prisoner);
213 }
214
215 void ResetPrisoner(Creature* prisoner)
216 {
217 if (!prisoner->IsAlive())
218 prisoner->Respawn(true);
219 prisoner->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_NON_ATTACKABLE);
220 }
221
222 void StorePrisoner(Creature* creature)
223 {
224 float posX = creature->GetPositionX();
225 float posY = creature->GetPositionY();
226
227 if (posX >= 405.0f && posX <= 423.0f)
228 {
229 if (posY >= 106.0f && posY <= 123.0f)
230 {
231 PrisonersCell5.insert(creature->GetGUID());
232 ++PrisonerCounter5;
233 }
234 else if (posY >= 76.0f && posY <= 91.0f)
235 {
236 PrisonersCell6.insert(creature->GetGUID());
237 ++PrisonerCounter6;
238 }
239 else return;
240 }
241 else if (posX >= 490.0f && posX <= 506.0f)
242 {
243 if (posY >= 106.0f && posY <= 123.0f)
244 {
245 PrisonersCell7.insert(creature->GetGUID());
246 ++PrisonerCounter7;
247 }
248 else if (posY >= 76.0f && posY <= 91.0f)
249 {
250 PrisonersCell8.insert(creature->GetGUID());
251 ++PrisonerCounter8;
252 }
253 else
254 return;
255 }
256 else
257 return;
258
259 ResetPrisoner(creature);
260 }
261
262 void PrisonerDied(ObjectGuid guid)
263 {
264 if (PrisonersCell5.find(guid) != PrisonersCell5.end() && --PrisonerCounter5 <= 0)
265 ActivateCell(DATA_PRISON_CELL6);
266 else if (PrisonersCell6.find(guid) != PrisonersCell6.end() && --PrisonerCounter6 <= 0)
267 ActivateCell(DATA_PRISON_CELL7);
268 else if (PrisonersCell7.find(guid) != PrisonersCell7.end() && --PrisonerCounter7 <= 0)
269 ActivateCell(DATA_PRISON_CELL8);
270 else if (PrisonersCell8.find(guid) != PrisonersCell8.end() && --PrisonerCounter8 <= 0)
271 ActivateCell(DATA_DOOR_4);
272 }
273
274 void ActivateCell(uint8 id)
275 {
276 switch (id)
277 {
278 case DATA_PRISON_CELL5:
279 HandleGameObject(PrisonCellGUIDs[id - DATA_PRISON_CELL1], true);
280 ActivatePrisoners(PrisonersCell5);
281 break;
282 case DATA_PRISON_CELL6:
283 HandleGameObject(PrisonCellGUIDs[id - DATA_PRISON_CELL1], true);
284 ActivatePrisoners(PrisonersCell6);
285 break;
286 case DATA_PRISON_CELL7:
287 HandleGameObject(PrisonCellGUIDs[id - DATA_PRISON_CELL1], true);
288 ActivatePrisoners(PrisonersCell7);
289 break;
290 case DATA_PRISON_CELL8:
291 HandleGameObject(PrisonCellGUIDs[id - DATA_PRISON_CELL1], true);
292 ActivatePrisoners(PrisonersCell8);
293 break;
294 case DATA_DOOR_4:
295 HandleGameObject(PrisonDoor4GUID, true);
296 if (Creature* broggok = instance->GetCreature(BroggokGUID))
297 broggok->AI()->DoAction(ACTION_ACTIVATE_BROGGOK);
298 break;
299 }
300 }
301
302 void ActivatePrisoners(GuidSet const& prisoners)
303 {
304 for (GuidSet::const_iterator i = prisoners.begin(); i != prisoners.end(); ++i)
305 if (Creature* prisoner = instance->GetCreature(*i))
306 {
307 prisoner->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_NON_ATTACKABLE);
308 prisoner->SetInCombatWithZone();
309 }
310 }

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "blood_furnace.h"
21
22enum Yells
23{
24 SAY_AGGRO = 0,
25 SAY_KILL = 1,
26 SAY_DIE = 2
27};
28
29enum Spells
30{
31 SPELL_ACID_SPRAY = 38153,
32 SPELL_EXPLODING_BREAKER = 30925,
33 SPELL_KNOCKDOWN = 20276,
34 SPELL_DOMINATION = 25772
35};
36
37enum Events
38{
39 EVENT_ACID_SPRAY = 1,
40 EVENT_EXPLODING_BREAKER,
41 EVENT_DOMINATION,
42 EVENT_KNOCKDOWN
43};
44
45class boss_the_maker : public CreatureScript
46{
47 public:
48 boss_the_maker() : CreatureScript("boss_the_maker") { }
49
50 struct boss_the_makerAI : public BossAI
51 {
52 boss_the_makerAI(Creature* creature) : BossAI(creature, DATA_THE_MAKER) { }
53
54 void EnterCombat(Unit* /*who*/) override
55 {
56 _EnterCombat();
57 Talk(SAY_AGGRO);
58
59 events.ScheduleEvent(EVENT_ACID_SPRAY, 15000);
60 events.ScheduleEvent(EVENT_EXPLODING_BREAKER, 6000);
61 events.ScheduleEvent(EVENT_DOMINATION, 120000);
62 events.ScheduleEvent(EVENT_KNOCKDOWN, 10000);
63 }
64
65 void KilledUnit(Unit* who) override
66 {
67 if (who->GetTypeId() == TYPEID_PLAYER)
68 Talk(SAY_KILL);
69 }
70
71 void JustDied(Unit* /*killer*/) override
72 {
73 _JustDied();
74 Talk(SAY_DIE);
75 }
76
77 void ExecuteEvent(uint32 eventId) override
78 {
79 switch (eventId)
80 {
81 case EVENT_ACID_SPRAY:
82 DoCastVictim(SPELL_ACID_SPRAY);
83 events.ScheduleEvent(EVENT_ACID_SPRAY, urand(15000, 23000));
84 break;
85 case EVENT_EXPLODING_BREAKER:
86 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30.0f, true))
87 DoCast(target, SPELL_EXPLODING_BREAKER);
88 events.ScheduleEvent(EVENT_EXPLODING_BREAKER, urand(4000, 12000));
89 break;
90 case EVENT_DOMINATION:
91 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
92 DoCast(target, SPELL_DOMINATION);
93 events.ScheduleEvent(EVENT_DOMINATION, 120000);
94 break;
95 case EVENT_KNOCKDOWN:
96 DoCastVictim(SPELL_KNOCKDOWN);
97 events.ScheduleEvent(EVENT_KNOCKDOWN, urand(4000, 12000));
98 break;
99 default:
100 break;
101 }
102 }
103 };
104
105 CreatureAI* GetAI(Creature* creature) const override
106 {
107 return GetBloodFurnaceAI<boss_the_makerAI>(creature);
108 }
109};
110
111void AddSC_boss_the_maker()
112{
113 new boss_the_maker();
114}

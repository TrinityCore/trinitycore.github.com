45class boss_broggok : public CreatureScript
46{
47 public:
48 boss_broggok() : CreatureScript("boss_broggok") { }
49
50 struct boss_broggokAI : public BossAI
51 {
52 boss_broggokAI(Creature* creature) : BossAI(creature, DATA_BROGGOK) { }
53
54 void Reset() override
55 {
56 _Reset();
57 DoAction(ACTION_RESET_BROGGOK);
58 }
59
60 void EnterCombat(Unit* /*who*/) override
61 {
62 _EnterCombat();
63 Talk(SAY_AGGRO);
64 }
65
66 void JustSummoned(Creature* summoned) override
67 {
68 if (summoned->GetEntry() == NPC_BROGGOK_POISON_CLOUD)
69 {
70 summoned->SetReactState(REACT_PASSIVE);
71 summoned->CastSpell(summoned, SPELL_POISON_CLOUD_PASSIVE, true);
72 summons.Summon(summoned);
73 }
74 }
75
76 void ExecuteEvent(uint32 eventId) override
77 {
78 switch (eventId)
79 {
80 case EVENT_SLIME_SPRAY:
81 DoCastVictim(SPELL_SLIME_SPRAY);
82 events.ScheduleEvent(EVENT_SLIME_SPRAY, urand(4000, 12000));
83 break;
84 case EVENT_POISON_BOLT:
85 DoCastVictim(SPELL_POISON_BOLT);
86 events.ScheduleEvent(EVENT_POISON_BOLT, urand(4000, 12000));
87 break;
88 case EVENT_POISON_CLOUD:
89 DoCast(me, SPELL_POISON_CLOUD);
90 events.ScheduleEvent(EVENT_POISON_CLOUD, 20000);
91 break;
92 default:
93 break;
94 }
95 }
96
97 void DoAction(int32 action) override
98 {
99 switch (action)
100 {
101 case ACTION_PREPARE_BROGGOK:
102 me->SetInCombatWithZone();
103 break;
104 case ACTION_ACTIVATE_BROGGOK:
105 me->SetReactState(REACT_AGGRESSIVE);
106 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_NON_ATTACKABLE);
107 events.ScheduleEvent(EVENT_SLIME_SPRAY, 10000);
108 events.ScheduleEvent(EVENT_POISON_BOLT, 7000);
109 events.ScheduleEvent(EVENT_POISON_CLOUD, 5000);
110 break;
111 case ACTION_RESET_BROGGOK:
112 me->SetReactState(REACT_PASSIVE);
113 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_NON_ATTACKABLE);
114 break;
115 }
116 }
117 };
118
119 CreatureAI* GetAI(Creature* creature) const override
120 {
121 return GetBloodFurnaceAI<boss_broggokAI>(creature);
122 }
123};

125class go_broggok_lever : public GameObjectScript
126{
127 public:
128 go_broggok_lever() : GameObjectScript("go_broggok_lever") { }
129
130 bool OnGossipHello(Player* /*player*/, GameObject* go) override
131 {
132 if (InstanceScript* instance = go->GetInstanceScript())
133 if (instance->GetBossState(DATA_BROGGOK) != DONE && instance->GetBossState(DATA_BROGGOK) != IN_PROGRESS)
134 {
135 instance->SetBossState(DATA_BROGGOK, IN_PROGRESS);
136 if (Creature* broggok = ObjectAccessor::GetCreature(*go, instance->GetGuidData(DATA_BROGGOK)))
137 broggok->AI()->DoAction(ACTION_PREPARE_BROGGOK);
138 }
139
140 go->UseDoorOrButton();
141 return false;
142 }
143};

145// 30914, 38462 - Poison (Broggok)
146class spell_broggok_poison_cloud : public SpellScriptLoader
147{
148 public:
149 spell_broggok_poison_cloud() : SpellScriptLoader("spell_broggok_poison_cloud") { }
150
151 class spell_broggok_poison_cloud_AuraScript : public AuraScript
152 {
153 PrepareAuraScript(spell_broggok_poison_cloud_AuraScript);
154
155 bool Validate(SpellInfo const* spellInfo) override
156 {
157 if (!sSpellMgr->GetSpellInfo(spellInfo->GetEffect(EFFECT_0)->TriggerSpell))
158 return false;
159 return true;
160 }
161
162 void PeriodicTick(AuraEffect const* aurEff)
163 {
164 PreventDefaultAction();
165
166 uint32 triggerSpell = GetSpellInfo()->GetEffect(aurEff->GetEffIndex())->TriggerSpell;
167 int32 mod = int32(((float(aurEff->GetTickNumber()) / aurEff->GetTotalTicks()) * 0.9f + 0.1f) * 10000 * 2 / 3);
168 GetTarget()->CastCustomSpell(triggerSpell, SPELLVALUE_RADIUS_MOD, mod, (Unit*)NULL, TRIGGERED_FULL_MASK, NULL, aurEff);
169 }
170
171 void Register() override
172 {
173 OnEffectPeriodic += AuraEffectPeriodicFn(spell_broggok_poison_cloud_AuraScript::PeriodicTick, EFFECT_0, SPELL_AURA_PERIODIC_TRIGGER_SPELL);
174 }
175 };
176
177 AuraScript* GetAuraScript() const override
178 {
179 return new spell_broggok_poison_cloud_AuraScript();
180 }
181};

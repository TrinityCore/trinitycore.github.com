59class boss_kelidan_the_breaker : public CreatureScript
60{
61 public:
62 boss_kelidan_the_breaker() : CreatureScript("boss_kelidan_the_breaker") { }
63
64 struct boss_kelidan_the_breakerAI : public BossAI
65 {
66 boss_kelidan_the_breakerAI(Creature* creature) : BossAI(creature, DATA_KELIDAN_THE_BREAKER)
67 {
68 Initialize();
69 Firenova_Timer = 0;
70 }
71
72 void Initialize()
73 {
74 ShadowVolley_Timer = 1000;
75 BurningNova_Timer = 15000;
76 Corruption_Timer = 5000;
77 check_Timer = 0;
78 Firenova = false;
79 addYell = false;
80 }
81
82 uint32 ShadowVolley_Timer;
83 uint32 BurningNova_Timer;
84 uint32 Firenova_Timer;
85 uint32 Corruption_Timer;
86 uint32 check_Timer;
87 bool Firenova;
88 bool addYell;
89 ObjectGuid Channelers[5];
90
91 void Reset() override
92 {
93 _Reset();
94 Initialize();
95 SummonChannelers();
96 me->SetReactState(REACT_PASSIVE);
97 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_NON_ATTACKABLE);
98 }
99
100 void EnterCombat(Unit* who) override
101 {
102 _EnterCombat();
103 Talk(SAY_WAKE);
104 if (me->IsNonMeleeSpellCast(false))
105 me->InterruptNonMeleeSpells(true);
106 DoStartMovement(who);
107 }
108
109 void KilledUnit(Unit* /*victim*/) override
110 {
111 if (rand32() % 2)
112 return;
113
114 Talk(SAY_KILL);
115 }
116
117 void ChannelerEngaged(Unit* who)
118 {
119 if (who && !addYell)
120 {
121 addYell = true;
122 Talk(SAY_ADD_AGGRO);
123 }
124 for (uint8 i = 0; i<5; ++i)
125 {
126 Creature* channeler = ObjectAccessor::GetCreature(*me, Channelers[i]);
127 if (who && channeler && !channeler->IsInCombat())
128 channeler->AI()->AttackStart(who);
129 }
130 }
131
132 void ChannelerDied(Unit* killer)
133 {
134 for (uint8 i = 0; i < 5; ++i)
135 {
136 Creature* channeler = ObjectAccessor::GetCreature(*me, Channelers[i]);
137 if (channeler && channeler->IsAlive())
138 return;
139 }
140 me->SetReactState(REACT_AGGRESSIVE);
141 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_NON_ATTACKABLE);
142 if (killer)
143 AttackStart(killer);
144 }
145
146 ObjectGuid GetChanneled(Creature* channeler1)
147 {
148 SummonChannelers();
149 if (!channeler1)
150 return ObjectGuid::Empty;
151
152 uint8 i;
153 for (i = 0; i < 5; ++i)
154 {
155 Creature* channeler = ObjectAccessor::GetCreature(*me, Channelers[i]);
156 if (channeler && channeler->GetGUID() == channeler1->GetGUID())
157 break;
158 }
159 return Channelers[(i + 2) % 5];
160 }
161
162 void SummonChannelers()
163 {
164 for (uint8 i = 0; i < 5; ++i)
165 {
166 Creature* channeler = ObjectAccessor::GetCreature(*me, Channelers[i]);
167 if (!channeler || channeler->isDead())
168 channeler = me->SummonCreature(ENTRY_CHANNELER, ShadowmoonChannelers[i][0], ShadowmoonChannelers[i][1], ShadowmoonChannelers[i][2], ShadowmoonChannelers[i][3], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
169 if (channeler)
170 Channelers[i] = channeler->GetGUID();
171 else
172 Channelers[i].Clear();
173 }
174 }
175
176 void JustDied(Unit* /*killer*/) override
177 {
178 _JustDied();
179 Talk(SAY_DIE);
180 }
181
182 void UpdateAI(uint32 diff) override
183 {
184 if (!UpdateVictim())
185 {
186 if (check_Timer <= diff)
187 {
188 if (!me->IsNonMeleeSpellCast(false))
189 DoCast(me, SPELL_EVOCATION);
190 check_Timer = 5000;
191 }
192 else
193 check_Timer -= diff;
194 return;
195 }
196
197 if (Firenova)
198 {
199 if (Firenova_Timer <= diff)
200 {
201 DoCast(me, SPELL_FIRE_NOVA, true);
202 Firenova = false;
203 ShadowVolley_Timer = 2000;
204 }
205 else
206 Firenova_Timer -=diff;
207
208 return;
209 }
210
211 if (ShadowVolley_Timer <= diff)
212 {
213 DoCast(me, SPELL_SHADOW_BOLT_VOLLEY);
214 ShadowVolley_Timer = 5000 + rand32() % 8000;
215 }
216 else
217 ShadowVolley_Timer -=diff;
218
219 if (Corruption_Timer <= diff)
220 {
221 DoCast(me, SPELL_CORRUPTION);
222 Corruption_Timer = 30000 + rand32() % 20000;
223 }
224 else
225 Corruption_Timer -=diff;
226
227 if (BurningNova_Timer <= diff)
228 {
229 if (me->IsNonMeleeSpellCast(false))
230 me->InterruptNonMeleeSpells(true);
231
232 Talk(SAY_NOVA);
233
234 if (SpellInfo const* nova = sSpellMgr->GetSpellInfo(SPELL_BURNING_NOVA))
235 {
236 if (Aura* aura = Aura::TryRefreshStackOrCreate(nova, MAX_EFFECT_MASK, me, me))
237 aura->ApplyForTargets();
238 }
239
240 if (IsHeroic())
241 DoTeleportAll(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
242
243 BurningNova_Timer = 20000 + rand32() % 8000;
244 Firenova_Timer= 5000;
245 Firenova = true;
246 }
247 else
248 BurningNova_Timer -=diff;
249
250 DoMeleeAttackIfReady();
251 }
252 };
253
254 CreatureAI* GetAI(Creature* creature) const override
255 {
256 return GetBloodFurnaceAI<boss_kelidan_the_breakerAI>(creature);
257 }

273class npc_shadowmoon_channeler : public CreatureScript
274{
275 public:
276
277 npc_shadowmoon_channeler() : CreatureScript("npc_shadowmoon_channeler") { }
278
279 struct npc_shadowmoon_channelerAI : public ScriptedAI
280 {
281 npc_shadowmoon_channelerAI(Creature* creature) : ScriptedAI(creature)
282 {
283 Initialize();
284 }
285
286 void Initialize()
287 {
288 ShadowBolt_Timer = 1000 + rand32() % 1000;
289 MarkOfShadow_Timer = 5000 + rand32() % 2000;
290 check_Timer = 0;
291 }
292
293 uint32 ShadowBolt_Timer;
294 uint32 MarkOfShadow_Timer;
295 uint32 check_Timer;
296
297 void Reset() override
298 {
299 Initialize();
300 if (me->IsNonMeleeSpellCast(false))
301 me->InterruptNonMeleeSpells(true);
302 }
303
304 void EnterCombat(Unit* who) override
305 {
306 if (Creature* Kelidan = me->FindNearestCreature(ENTRY_KELIDAN, 100))
307 ENSURE_AI(boss_kelidan_the_breaker::boss_kelidan_the_breakerAI, Kelidan->AI())->ChannelerEngaged(who);
308 if (me->IsNonMeleeSpellCast(false))
309 me->InterruptNonMeleeSpells(true);
310 DoStartMovement(who);
311 }
312
313 void JustDied(Unit* killer) override
314 {
315 if (Creature* Kelidan = me->FindNearestCreature(ENTRY_KELIDAN, 100))
316 ENSURE_AI(boss_kelidan_the_breaker::boss_kelidan_the_breakerAI, Kelidan->AI())->ChannelerDied(killer);
317 }
318
319 void UpdateAI(uint32 diff) override
320 {
321 if (!UpdateVictim())
322 {
323 if (check_Timer <= diff)
324 {
325 if (!me->IsNonMeleeSpellCast(false))
326 {
327 if (Creature* Kelidan = me->FindNearestCreature(ENTRY_KELIDAN, 100))
328 {
329 ObjectGuid channeler = ENSURE_AI(boss_kelidan_the_breaker::boss_kelidan_the_breakerAI, Kelidan->AI())->GetChanneled(me);
330 if (Unit* channeled = ObjectAccessor::GetUnit(*me, channeler))
331 DoCast(channeled, SPELL_CHANNELING);
332 }
333 }
334 check_Timer = 5000;
335 }
336 else
337 check_Timer -= diff;
338
339 return;
340 }
341
342 if (MarkOfShadow_Timer <= diff)
343 {
344 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
345 DoCast(target, SPELL_MARK_OF_SHADOW);
346 MarkOfShadow_Timer = 15000 + rand32() % 5000;
347 }
348 else
349 MarkOfShadow_Timer -=diff;
350
351 if (ShadowBolt_Timer <= diff)
352 {
353 DoCastVictim(SPELL_SHADOW_BOLT);
354 ShadowBolt_Timer = 5000 + rand32() % 1000;
355 }
356 else
357 ShadowBolt_Timer -=diff;
358
359 DoMeleeAttackIfReady();
360 }
361 };
362
363 CreatureAI* GetAI(Creature* creature) const override
364 {
365 return GetBloodFurnaceAI<npc_shadowmoon_channelerAI>(creature);
366 }
367};

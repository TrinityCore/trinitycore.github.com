45class instance_magtheridons_lair : public InstanceMapScript

53 struct instance_magtheridons_lair_InstanceMapScript : public InstanceScript
54 {
55 instance_magtheridons_lair_InstanceMapScript(Map* map) : InstanceScript(map)
56 {
57 SetHeaders(DataHeader);
58 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
59
60 CageTimer = 0;
61 RespawnTimer = 0;

64 uint32 m_auiEncounter[MAX_ENCOUNTER];
65
66 ObjectGuid MagtheridonGUID;
67 GuidSet ChannelerGUID;
68 ObjectGuid DoorGUID;
69 GuidSet ColumnGUID;
70
71 uint32 CageTimer;
72 uint32 RespawnTimer;
73
74 bool IsEncounterInProgress() const override
75 {
76 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
77 if (m_auiEncounter[i] == IN_PROGRESS)
78 return true;
79
80 return false;
81 }
82
83 void OnCreatureCreate(Creature* creature) override
84 {
85 switch (creature->GetEntry())
86 {
87 case 17257:
88 MagtheridonGUID = creature->GetGUID();
89 break;
90 case 17256:
91 ChannelerGUID.insert(creature->GetGUID());
92 break;
93 }
94 }
95
96 void OnGameObjectCreate(GameObject* go) override
97 {
98 switch (go->GetEntry())
99 {
100 case 181713:
101 go->SetUInt32Value(GAMEOBJECT_FLAGS, 0);
102 break;
103 case 183847:
104 DoorGUID = go->GetGUID();
105 break;
106 case 184653: // hall
107 case 184634: // six columns
108 case 184635:
109 case 184636:
110 case 184637:
111 case 184638:
112 case 184639:
113 ColumnGUID.insert(go->GetGUID());
114 break;
115 }
116 }
117
118 ObjectGuid GetGuidData(uint32 type) const override
119 {
120 switch (type)
121 {
122 case DATA_MAGTHERIDON:
123 return MagtheridonGUID;
124 }
125 return ObjectGuid::Empty;
126 }
127
128 void SetData(uint32 type, uint32 data) override
129 {
130 switch (type)
131 {
132 case DATA_MAGTHERIDON_EVENT:
133 m_auiEncounter[0] = data;
134 if (data == NOT_STARTED)
135 RespawnTimer = 10000;
136 if (data != IN_PROGRESS)
137 HandleGameObject(DoorGUID, true);
138 break;
139 case DATA_CHANNELER_EVENT:
140 switch (data)
141 {
142 case NOT_STARTED: // Reset all channelers once one is reset.
143 if (m_auiEncounter[1] != NOT_STARTED)
144 {
145 m_auiEncounter[1] = NOT_STARTED;
146 for (GuidSet::const_iterator i = ChannelerGUID.begin(); i != ChannelerGUID.end(); ++i)
147 {
148 if (Creature* Channeler = instance->GetCreature(*i))
149 {
150 if (Channeler->IsAlive())
151 Channeler->AI()->EnterEvadeMode();
152 else
153 Channeler->Respawn();
154 }
155 }
156 CageTimer = 0;
157 HandleGameObject(DoorGUID, true);
158 }
159 break;
160 case IN_PROGRESS: // Event start.
161 if (m_auiEncounter[1] != IN_PROGRESS)
162 {
163 m_auiEncounter[1] = IN_PROGRESS;
164 // Let all five channelers aggro.
165 for (GuidSet::const_iterator i = ChannelerGUID.begin(); i != ChannelerGUID.end(); ++i)
166 {
167 Creature* Channeler = instance->GetCreature(*i);
168 if (Channeler && Channeler->IsAlive())
169 Channeler->AI()->AttackStart(Channeler->SelectNearestTarget(999));
170 }
171 // Release Magtheridon after two minutes.
172 Creature* Magtheridon = instance->GetCreature(MagtheridonGUID);
173 if (Magtheridon && Magtheridon->IsAlive())
174 {
175 Magtheridon->TextEmote(EMOTE_BONDS_WEAKEN);
176 CageTimer = 120000;
177 }
178 HandleGameObject(DoorGUID, false);
179 }
180 break;
181 case DONE: // Add buff and check if all channelers are dead.
182 for (GuidSet::const_iterator i = ChannelerGUID.begin(); i != ChannelerGUID.end(); ++i)
183 {
184 Creature* Channeler = instance->GetCreature(*i);
185 if (Channeler && Channeler->IsAlive())
186 {
187 //Channeler->CastSpell(Channeler, SPELL_SOUL_TRANSFER, true);
188 data = IN_PROGRESS;
189 break;
190 }
191 }
192 break;
193 }
194 m_auiEncounter[1] = data;
195 break;
196 case DATA_COLLAPSE:
197 // true - collapse / false - reset
198 for (GuidSet::const_iterator i = ColumnGUID.begin(); i != ColumnGUID.end(); ++i)
199 DoUseDoorOrButton(*i);
200 break;
201 default:
202 break;
203 }
204 }
205
206 uint32 GetData(uint32 type) const override
207 {
208 if (type == DATA_MAGTHERIDON_EVENT)
209 return m_auiEncounter[0];
210 return 0;
211 }
212
213 void Update(uint32 diff) override
214 {
215 if (CageTimer)
216 {
217 if (CageTimer <= diff)
218 {
219 Creature* Magtheridon = instance->GetCreature(MagtheridonGUID);
220 if (Magtheridon && Magtheridon->IsAlive())
221 {
222 Magtheridon->ClearUnitState(UNIT_STATE_STUNNED);
223 Magtheridon->AI()->AttackStart(Magtheridon->SelectNearestTarget(999));
224 }
225 CageTimer = 0;
226 } else CageTimer -= diff;
227 }
228
229 if (RespawnTimer)
230 {
231 if (RespawnTimer <= diff)
232 {
233 for (GuidSet::const_iterator i = ChannelerGUID.begin(); i != ChannelerGUID.end(); ++i)
234 {
235 if (Creature* Channeler = instance->GetCreature(*i))
236 {
237 if (Channeler->IsAlive())
238 Channeler->AI()->EnterEvadeMode();
239 else
240 Channeler->Respawn();
241 }
242 }
243 RespawnTimer = 0;
244 } else RespawnTimer -= diff;
245 }
246 }
247 };

27class at_nethekurse_exit : public AreaTriggerScript
28{
29 public:
30 at_nethekurse_exit() : AreaTriggerScript("at_nethekurse_exit") { };
31
32 bool OnTrigger(Player* player, AreaTriggerEntry const*, bool /*entered*/) override
33 {
34 if (InstanceScript* is = player->GetInstanceScript())
35 {
36 if (is->instance->IsHeroic())
37 {
38 Creature* executioner = nullptr;
39
40 is->instance->LoadGrid(Executioner.GetPositionX(), Executioner.GetPositionY());
41 if (Creature* kargath = ObjectAccessor::GetCreature(*player, is->GetGuidData(NPC_KARGATH_BLADEFIST)))
42 {
43 if (is->GetGuidData(NPC_SHATTERED_EXECUTIONER).IsEmpty())
44 {
45 executioner = kargath->SummonCreature(NPC_SHATTERED_EXECUTIONER, Executioner);
46 kargath->AI()->DoAction(ACTION_EXECUTIONER_TAUNT);
47 }
48 }
49
50 if (executioner)
51 for (uint8 i = 0; i < VictimCount; ++i)
52 executioner->SummonCreature(executionerVictims[i](is->GetData(DATA_TEAM_IN_INSTANCE)), executionerVictims[i].GetPos());
53 }
54 }
55
56 return false;
57 }
58};

65class boss_shattered_executioner : public CreatureScript
66{
67 public:
68 boss_shattered_executioner() : CreatureScript("boss_shattered_executioner") { }
69
70 struct boss_shattered_executionerAI : public BossAI
71 {
72 boss_shattered_executionerAI(Creature* creature) : BossAI(creature, DATA_SHATTERED_EXECUTIONER)
73 {
74 Initialize();
75 };
76
77 void Initialize()
78 {
79 cleaveTimer = 500;
80 }
81
82 void Reset() override
83 {
84 _Reset();
85
86 // _Reset() resets the loot mode, so we add them again, if any
87 uint32 prisonersExecuted = instance->GetData(DATA_PRISONERS_EXECUTED);
88 if (prisonersExecuted == 0)
89 me->AddLootMode(LOOT_MODE_HARD_MODE_3);
90 if (prisonersExecuted <= 1)
91 me->AddLootMode(LOOT_MODE_HARD_MODE_2);
92 if (prisonersExecuted <= 2)
93 me->AddLootMode(LOOT_MODE_HARD_MODE_1);
94
95 if (instance->GetBossState(DATA_KARGATH) == DONE)
96 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
97 else
98 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
99
100 Initialize();
101 }
102
103 void JustSummoned(Creature*) override { } // avoid despawn of prisoners on death/reset
104
105 void JustDied(Unit*) override
106 {
107 _JustDied();
108
109 if (instance->GetData(DATA_PRISONERS_EXECUTED) > 0)
110 return;
111
112 Map::PlayerList const &players = instance->instance->GetPlayers();
113 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
114 {
115 Player* pl = itr->GetSource();
116 uint32 qId = pl->GetTeam() == ALLIANCE ? QUEST_IMPRISONED_A : QUEST_IMPRISONED_H;
117 if (pl->GetQuestStatus(qId) == QUEST_STATUS_INCOMPLETE)
118 pl->CompleteQuest(qId);
119 }
120 }
121
122 void SetData(uint32 type, uint32 data) override
123 {
124 if (type == DATA_PRISONERS_EXECUTED && data <= 3)
125 {
126 if (Creature* victim = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FIRST_PRISONER + data - 1)))
127 me->Kill(victim);
128
129 if (data == 1)
130 {
131 Map::PlayerList const &players = instance->instance->GetPlayers();
132 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
133 {
134 Player* pl = itr->GetSource();
135 uint32 qId = pl->GetTeam() == ALLIANCE ? QUEST_IMPRISONED_A : QUEST_IMPRISONED_H;
136 if (pl->GetQuestStatus(qId) == QUEST_STATUS_INCOMPLETE)
137 pl->FailQuest(qId);
138 }
139 }
140
141 switch (data)
142 {
143 case 3:
144 me->RemoveLootMode(LOOT_MODE_HARD_MODE_1);
145 case 2:
146 me->RemoveLootMode(LOOT_MODE_HARD_MODE_2);
147 case 1:
148 me->RemoveLootMode(LOOT_MODE_HARD_MODE_3);
149 }
150 }
151 }
152
153 void UpdateAI(uint32 diff) override
154 {
155 if (!UpdateVictim())
156 return;
157
158 if (cleaveTimer <= diff)
159 {
160 DoCast(SPELL_CLEAVE);
161 cleaveTimer = urand(5000, 7000);
162 }
163 else
164 cleaveTimer -= diff;
165
166 DoMeleeAttackIfReady();
167 }
168 private:
169 uint32 cleaveTimer;
170 };
171
172 CreatureAI* GetAI(Creature* creature) const override
173 {
174 return GetInstanceAI<boss_shattered_executionerAI>(creature);
175 }
176};

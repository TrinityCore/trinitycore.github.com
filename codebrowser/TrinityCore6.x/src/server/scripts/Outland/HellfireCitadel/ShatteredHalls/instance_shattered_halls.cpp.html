42class instance_shattered_halls : public InstanceMapScript
43{
44 public:
45 instance_shattered_halls() : InstanceMapScript("instance_shattered_halls", 540) { }

47 InstanceScript* GetInstanceScript(InstanceMap* map) const override

49 return new instance_shattered_halls_InstanceMapScript(map);

52 struct instance_shattered_halls_InstanceMapScript : public InstanceScript
53 {
54 instance_shattered_halls_InstanceMapScript(Map* map) : InstanceScript(map)
55 {
56 SetHeaders(DataHeader);
57 SetBossNumber(EncounterCount);
58 LoadDoorData(doorData);
59 executionTimer = 0;
60 executed = 0;
61 _team = 0;
62 }
63
64 void OnPlayerEnter(Player* player) override
65 {
66 Aura* ex = nullptr;
67
68 if (!_team)
69 _team = player->GetTeam();
70
71 player->CastSpell(player, SPELL_REMOVE_KARGATH_EXECUTIONER, true);
72
73 if (!executionTimer || executionerGUID.IsEmpty())
74 return;
75
76 switch (executed)
77 {
78 case 0:
79 ex = player->AddAura(SPELL_KARGATH_EXECUTIONER_1, player);
80 break;
81 case 1:
82 ex = player->AddAura(SPELL_KARGATH_EXECUTIONER_2, player);
83 break;
84 case 2:
85 ex = player->AddAura(SPELL_KARGATH_EXECUTIONER_3, player);
86 break;
87 default:
88 break;
89 }
90
91 if (ex)
92 ex->SetDuration(executionTimer);
93 }
94
95 void OnGameObjectCreate(GameObject* go) override
96 {
97 switch (go->GetEntry())
98 {
99 case GO_GRAND_WARLOCK_CHAMBER_DOOR_1:
100 case GO_GRAND_WARLOCK_CHAMBER_DOOR_2:
101 AddDoor(go, true);
102 default:
103 break;
104 }
105 }
106
107 void OnGameObjectRemove(GameObject* go) override
108 {
109 switch (go->GetEntry())
110 {
111 case GO_GRAND_WARLOCK_CHAMBER_DOOR_1:
112 case GO_GRAND_WARLOCK_CHAMBER_DOOR_2:
113 AddDoor(go, false);
114 default:
115 break;
116 }
117 }
118
119 void OnCreatureCreate(Creature* creature) override
120 {
121 if (!_team)
122 {
123 Map::PlayerList const &players = instance->GetPlayers();
124 if (!players.isEmpty())
125 if (Player* player = players.begin()->GetSource())
126 _team = player->GetTeam();
127 }
128
129 switch (creature->GetEntry())
130 {
131 case NPC_GRAND_WARLOCK_NETHEKURSE:
132 nethekurseGUID = creature->GetGUID();
133 break;
134 case NPC_KARGATH_BLADEFIST:
135 kargathGUID = creature->GetGUID();
136 break;
137 case NPC_RANDY_WHIZZLESPROCKET:
138 if (_team == HORDE)
139 creature->UpdateEntry(NPC_DRISELLA);
140 break;
141 case NPC_SHATTERED_EXECUTIONER:
142 executionTimer = 55 * MINUTE * IN_MILLISECONDS;
143 DoCastSpellOnPlayers(SPELL_KARGATH_EXECUTIONER_1);
144 executionerGUID = creature->GetGUID();
145 SaveToDB();
146 break;
147 case NPC_CAPTAIN_ALINA:
148 case NPC_CAPTAIN_BONESHATTER:
149 victimsGUID[0] = creature->GetGUID();
150 break;
151 case NPC_ALLIANCE_VICTIM_1:
152 case NPC_HORDE_VICTIM_1:
153 victimsGUID[1] = creature->GetGUID();
154 break;
155 case NPC_ALLIANCE_VICTIM_2:
156 case NPC_HORDE_VICTIM_2:
157 victimsGUID[2] = creature->GetGUID();
158 break;
159 }
160 }
161
162 bool SetBossState(uint32 type, EncounterState state) override
163 {
164 if (!InstanceScript::SetBossState(type, state))
165 return false;
166
167 switch (type)
168 {
169 case DATA_SHATTERED_EXECUTIONER:
170 if (state == DONE)
171 {
172 DoCastSpellOnPlayers(SPELL_REMOVE_KARGATH_EXECUTIONER);
173 executionTimer = 0;
174 SaveToDB();
175 }
176 break;
177 case DATA_KARGATH:
178 if (Creature* executioner = instance->GetCreature(executionerGUID))
179 executioner->AI()->Reset(); // trigger removal of IMMUNE_TO_PC flag
180 break;
181 case DATA_OMROGG:
182 break;
183 }
184 return true;
185 }
186
187 ObjectGuid GetGuidData(uint32 data) const override
188 {
189 switch (data)
190 {
191 case NPC_GRAND_WARLOCK_NETHEKURSE:
192 return nethekurseGUID;
193 case NPC_KARGATH_BLADEFIST:
194 return kargathGUID;
195 case NPC_SHATTERED_EXECUTIONER:
196 return executionerGUID;
197 case DATA_FIRST_PRISONER:
198 case DATA_SECOND_PRISONER:
199 case DATA_THIRD_PRISONER:
200 return victimsGUID[data - DATA_FIRST_PRISONER];
201 default:
202 return ObjectGuid::Empty;
203 }
204 }

206 void WriteSaveDataMore(std::ostringstream& data) override
207 {
208 if (!instance->IsHeroic())
209 return;
210
211 data << uint32(executed) << ' '
212 << executionTimer << ' ';
213 }
214
215 void ReadSaveDataMore(std::istringstream& data) override
216 {
217 if (!instance->IsHeroic())
218 return;
219
220 uint32 readbuff;
221 data >> readbuff;
222 executed = uint8(readbuff);
223 data >> readbuff;
224
225 if (executed > VictimCount)
226 {
227 executed = VictimCount;
228 executionTimer = 0;
229 return;
230 }
231
232 if (!readbuff)
233 return;
234
235 Creature* executioner = nullptr;
236
237 instance->LoadGrid(Executioner.GetPositionX(), Executioner.GetPositionY());
238 if (Creature* kargath = instance->GetCreature(kargathGUID))
239 if (executionerGUID.IsEmpty())
240 executioner = kargath->SummonCreature(NPC_SHATTERED_EXECUTIONER, Executioner);
241
242 if (executioner)
243 for (uint8 i = executed; i < VictimCount; ++i)
244 executioner->SummonCreature(executionerVictims[i](GetData(DATA_TEAM_IN_INSTANCE)), executionerVictims[i].GetPos());
245
246 executionTimer = readbuff;
247 }
248
249 uint32 GetData(uint32 type) const override
250 {
251 switch (type)
252 {
253 case DATA_PRISONERS_EXECUTED:
254 return executed;
255 case DATA_TEAM_IN_INSTANCE:
256 return _team;
257 default:
258 return 0;
259 }
260 }
261
262 void Update(uint32 diff) override
263 {
264 if (!executionTimer)
265 return;
266
267 if (executionTimer <= diff)
268 {
269 DoCastSpellOnPlayers(SPELL_REMOVE_KARGATH_EXECUTIONER);
270 switch (++executed)
271 {
272 case 1:
273 DoCastSpellOnPlayers(SPELL_KARGATH_EXECUTIONER_2);
274 executionTimer = 10 * MINUTE * IN_MILLISECONDS;
275 break;
276 case 2:
277 DoCastSpellOnPlayers(SPELL_KARGATH_EXECUTIONER_3);
278 executionTimer = 15 * MINUTE * IN_MILLISECONDS;
279 break;
280 default:
281 executionTimer = 0;
282 break;
283 }
284
285 if (Creature* executioner = instance->GetCreature(executionerGUID))
286 executioner->AI()->SetData(DATA_PRISONERS_EXECUTED, executed);
287
288 SaveToDB();
289 }
290 else
291 executionTimer -= diff;
292 }
293
294 private:
295 ObjectGuid nethekurseGUID;
296 ObjectGuid kargathGUID;
297 ObjectGuid executionerGUID;
298 ObjectGuid victimsGUID[3];
299
300 uint8 executed;
301 uint32 executionTimer;
302 uint32 _team;
303 };

79class boss_grand_warlock_nethekurse : public CreatureScript
80{
81 public:
82 boss_grand_warlock_nethekurse() : CreatureScript("boss_grand_warlock_nethekurse") { }
83
84 struct boss_grand_warlock_nethekurseAI : public BossAI
85 {
86 boss_grand_warlock_nethekurseAI(Creature* creature) : BossAI(creature, DATA_NETHEKURSE)
87 {
88 Initialize();
89 }
90
91 void Initialize()
92 {
93 IsIntroEvent = false;
94 IntroOnce = false;
95 IsMainEvent = false;
96 //HasTaunted = false;
97 SpinOnce = false;
98 Phase = false;
99
100 PeonEngagedCount = 0;
101 PeonKilledCount = 0;
102
103 IntroEvent_Timer = 90000; // how long before getting bored and kills his minions?
104 DeathCoil_Timer = 20000;
105 ShadowFissure_Timer = 8000;
106 Cleave_Timer = 5000;
107 }
108
109 void Reset() override
110 {
111 _Reset();
112 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
113
114 Initialize();
115 }
116
117 void JustDied(Unit* /*killer*/) override
118 {
119 _JustDied();
120 Talk(SAY_DIE);
121 }
122
123 void SetData(uint32 data, uint32 value) override
124 {
125 if (data != SETDATA_DATA)
126 return;
127
128 switch (value)
129 {
130 case SETDATA_PEON_AGGRO:
131 if (PeonEngagedCount >= 4)
132 return;
133
134 Talk(SAY_PEON_ATTACKED);
135 ++PeonEngagedCount;
136 break;
137 case SETDATA_PEON_DEATH:
138 if (PeonKilledCount >= 4)
139 return;
140
141 Talk(SAY_PEON_DIES);
142 ++PeonKilledCount;
143
144 if (PeonKilledCount == 4)
145 {
146 IsIntroEvent = false;
147 IsMainEvent = true;
148 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
149 }
150 break;
151 default:
152 break;
153 }
154 }
155
156 void DoTauntPeons()
157 {
158 Talk(SAY_TAUNT);
159
160 /// @todo kill the peons first
161 IsIntroEvent = false;
162 PeonEngagedCount = 4;
163 PeonKilledCount = 4;
164 IsMainEvent = true;
165 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
166 }
167
168 void AttackStart(Unit* who) override
169 {
170 if (IsIntroEvent || !IsMainEvent)
171 return;
172
173 if (me->Attack(who, true))
174 {
175 if (Phase)
176 DoStartNoMovement(who);
177 else
178 DoStartMovement(who);
179 }
180 }
181
182 void MoveInLineOfSight(Unit* who) override
183
184 {
185 if (!IntroOnce && me->IsWithinDistInMap(who, 30.0f))
186 {
187 if (who->GetTypeId() != TYPEID_PLAYER)
188 return;
189
190 Talk(SAY_INTRO);
191 IntroOnce = true;
192 IsIntroEvent = true;
193
194 instance->SetBossState(DATA_NETHEKURSE, IN_PROGRESS);
195 }
196
197 if (IsIntroEvent || !IsMainEvent)
198 return;
199
200 ScriptedAI::MoveInLineOfSight(who);
201 }
202
203 void EnterCombat(Unit* /*who*/) override
204 {
205 Talk(SAY_AGGRO);
206 }
207
208 void JustSummoned(Creature* summoned) override
209 {
210 summoned->setFaction(16);
211 summoned->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
212 summoned->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
213
214 //triggered spell of consumption does not properly show it's SpellVisual, wrong spellid?
215 summoned->CastSpell(summoned, SPELL_TEMPORARY_VISUAL, true);
216 summoned->CastSpell(summoned, SPELL_CONSUMPTION, false, 0, 0, me->GetGUID());
217 }
218
219 void KilledUnit(Unit* /*victim*/) override
220 {
221 Talk(SAY_SLAY);
222 }
223
224 void UpdateAI(uint32 diff) override
225 {
226 if (IsIntroEvent)
227 {
228 if (instance->GetBossState(DATA_NETHEKURSE) == IN_PROGRESS)
229 {
230 if (IntroEvent_Timer <= diff)
231 DoTauntPeons();
232 else
233 IntroEvent_Timer -= diff;
234 }
235 }
236
237 if (!UpdateVictim())
238 return;
239
240 if (!IsMainEvent)
241 return;
242
243 if (Phase)
244 {
245 if (!SpinOnce)
246 {
247 DoCastVictim(SPELL_DARK_SPIN);
248 SpinOnce = true;
249 }
250
251 if (Cleave_Timer <= diff)
252 {
253 DoCastVictim(SPELL_SHADOW_CLEAVE);
254 Cleave_Timer = 6000 + rand32() % 2500;
255 }
256 else
257 Cleave_Timer -= diff;
258 }
259 else
260 {
261 if (ShadowFissure_Timer <= diff)
262 {
263 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
264 DoCast(target, SPELL_SHADOW_FISSURE);
265 ShadowFissure_Timer = urand(7500, 15000);
266 }
267 else
268 ShadowFissure_Timer -= diff;
269
270 if (DeathCoil_Timer <= diff)
271 {
272 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
273 DoCast(target, SPELL_DEATH_COIL);
274 DeathCoil_Timer = urand(15000, 20000);
275 }
276 else
277 DeathCoil_Timer -= diff;
278
279 if (!HealthAbovePct(20))
280 Phase = true;
281
282 DoMeleeAttackIfReady();
283 }
284 }
285
286 private:
287 uint32 PeonEngagedCount;
288 uint32 PeonKilledCount;
289 uint32 IntroEvent_Timer;
290 uint32 DeathCoil_Timer;
291 uint32 ShadowFissure_Timer;
292 uint32 Cleave_Timer;
293 bool IntroOnce;
294 bool IsIntroEvent;
295 bool IsMainEvent;
296 bool SpinOnce;
297 //bool HasTaunted;
298 bool Phase;
299 };
300
301 CreatureAI* GetAI(Creature* creature) const override
302 {
303 return GetInstanceAI<boss_grand_warlock_nethekurseAI>(creature);
304 }
305};

311class npc_fel_orc_convert : public CreatureScript
312{
313 public:
314 npc_fel_orc_convert() : CreatureScript("npc_fel_orc_convert") { }
315
316 struct npc_fel_orc_convertAI : public ScriptedAI
317 {
318 npc_fel_orc_convertAI(Creature* creature) : ScriptedAI(creature)
319 {
320 instance = creature->GetInstanceScript();
321 }
322
323 void Reset() override
324 {
325 me->SetNoCallAssistance(true); //we don't want any assistance (WE R HEROZ!)
326 }
327
328 void MoveInLineOfSight(Unit* /*who*/) override { }
329
330 void EnterCombat(Unit* /*who*/) override
331 {
332 events.ScheduleEvent(EVENT_HEMORRHAGE, 3000);
333
334 if (Creature* Kurse = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_GRAND_WARLOCK_NETHEKURSE)))
335 if (me->IsWithinDist(Kurse, 45.0f))
336 Kurse->AI()->SetData(SETDATA_DATA, SETDATA_PEON_AGGRO);
337 }
338
339 void JustDied(Unit* /*killer*/) override
340 {
341 if (instance->GetBossState(DATA_NETHEKURSE) != IN_PROGRESS)
342 return;
343
344 if (Creature* Kurse = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_GRAND_WARLOCK_NETHEKURSE)))
345 Kurse->AI()->SetData(SETDATA_DATA, SETDATA_PEON_DEATH);
346 }
347
348 void UpdateAI(uint32 diff) override
349 {
350 if (!UpdateVictim())
351 return;
352
353 events.Update(diff);
354
355 if (events.ExecuteEvent() == EVENT_HEMORRHAGE)
356 {
357 DoCastVictim(SPELL_HEMORRHAGE);
358 events.ScheduleEvent(EVENT_HEMORRHAGE, 15000);
359 }
360
361 DoMeleeAttackIfReady();
362 }
363
364 private:
365 InstanceScript* instance;
366 EventMap events;
367 };
368
369 CreatureAI* GetAI(Creature* creature) const override
370 {
371 return GetInstanceAI<npc_fel_orc_convertAI>(creature);
372 }
373};

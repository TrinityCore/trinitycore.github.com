65{
66 public:
67 boss_nazan() : CreatureScript("boss_nazan") { }

69 struct boss_nazanAI : public BossAI
70 {
71 boss_nazanAI(Creature* creature) : BossAI(creature, DATA_NAZAN)
72 {
73 Initialize();
74 flight = true;
75 BellowingRoar_Timer = 0;
76 ConeOfFire_Timer = 0;
77 }
78
79 void Initialize()
80 {
81 Fireball_Timer = 4000;
82 Fly_Timer = 45000;
83 Turn_Timer = 0;
84 }
85
86 void Reset() override
87 {
88 Initialize();
89 _Reset();
90 }
91
92 void EnterCombat(Unit* /*who*/) override { }
93
94 void IsSummonedBy(Unit* summoner) override
95 {
96 if (summoner->GetEntry() == NPC_VAZRUDEN_HERALD)
97 VazrudenGUID = summoner->GetGUID();
98 }
99
100 void JustSummoned(Creature* summoned) override
101 {
102 if (summoned && summoned->GetEntry() == NPC_LIQUID_FIRE)
103 {
104 summoned->SetLevel(me->getLevel());
105 summoned->setFaction(me->getFaction());
106 summoned->CastSpell(summoned, DUNGEON_MODE(SPELL_SUMMON_LIQUID_FIRE, SPELL_SUMMON_LIQUID_FIRE_H), true);
107 summoned->CastSpell(summoned, SPELL_FIRE_NOVA_VISUAL, true);
108 }
109 }
110
111 void SpellHitTarget(Unit* target, const SpellInfo* entry) override
112 {
113 if (target && entry->Id == uint32(SPELL_FIREBALL))
114 me->SummonCreature(NPC_LIQUID_FIRE, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), target->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN, 30000);
115 }
116
117 void UpdateAI(uint32 diff) override
118 {
119 if (!UpdateVictim())
120 return;
121
122 if (Fireball_Timer <= diff)
123 {
124 if (Unit* victim = SelectTarget(SELECT_TARGET_RANDOM, 0))
125 DoCast(victim, SPELL_FIREBALL, true);
126 Fireball_Timer = urand(4000, 7000);
127 }
128 else
129 Fireball_Timer -= diff;
130
131 if (flight) // phase 1 - the flight
132 {
133 Creature* Vazruden = ObjectAccessor::GetCreature(*me, VazrudenGUID);
134 if (Fly_Timer < diff || !(Vazruden && Vazruden->IsAlive() && Vazruden->HealthAbovePct(20)))
135 {
136 flight = false;
137 BellowingRoar_Timer = 6000;
138 ConeOfFire_Timer = 12000;
139 me->SetDisableGravity(false);
140 me->SetWalk(true);
141 me->GetMotionMaster()->Clear();
142 if (Unit* victim = SelectTarget(SELECT_TARGET_NEAREST, 0))
143 AttackStart(victim);
144 DoStartMovement(me->GetVictim());
145 Talk(EMOTE);
146 return;
147 }
148 else
149 Fly_Timer -= diff;
150
151 if (Turn_Timer <= diff)
152 {
153 uint32 waypoint = (Fly_Timer/10000)%2;
154 if (!me->IsWithinDist3d(VazrudenRing[waypoint][0], VazrudenRing[waypoint][1], VazrudenRing[waypoint][2], 5))
155 me->GetMotionMaster()->MovePoint(0, VazrudenRing[waypoint][0], VazrudenRing[waypoint][1], VazrudenRing[waypoint][2]);
156 Turn_Timer = 10000;
157 }
158 else
159 Turn_Timer -= diff;
160 }
161 else // phase 2 - land fight
162 {
163 if (ConeOfFire_Timer <= diff)
164 {
165 DoCast(me, SPELL_CONE_OF_FIRE);
166 ConeOfFire_Timer = 12000;
167 Fireball_Timer = 4000;
168 }
169 else
170 ConeOfFire_Timer -= diff;
171
172 if (IsHeroic())
173 {
174 if (BellowingRoar_Timer <= diff)
175 {
176 DoCast(me, SPELL_BELLOWING_ROAR);
177 BellowingRoar_Timer = 45000;
178 }
179 else
180 BellowingRoar_Timer -= diff;
181 }
182
183 DoMeleeAttackIfReady();
184 }
185 }
186
187 private:
188 uint32 Fireball_Timer;
189 uint32 ConeOfFire_Timer;
190 uint32 BellowingRoar_Timer;
191 uint32 Fly_Timer;
192 uint32 Turn_Timer;
193 bool flight;
194 ObjectGuid VazrudenGUID;
195 };
196
197 CreatureAI* GetAI(Creature* creature) const override
198 {
199 return new boss_nazanAI(creature);
200 }

203class boss_vazruden : public CreatureScript
204{
205 public:
206 boss_vazruden() : CreatureScript("boss_vazruden") { }
207
208 struct boss_vazrudenAI : public BossAI
209 {
210 boss_vazrudenAI(Creature* creature) : BossAI(creature, DATA_VAZRUDEN)
211 {
212 Initialize();
213 }
214
215 void Initialize()
216 {
217 Revenge_Timer = 4000;
218 UnsummonCheck = 2000;
219 WipeSaid = false;
220 }
221
222 void Reset() override
223 {
224 Initialize();
225 _Reset();
226 }
227
228 void EnterCombat(Unit* /*who*/) override
229 {
230 Talk(SAY_AGGRO);
231 _EnterCombat();
232 }
233
234 void KilledUnit(Unit* who) override
235 {
236 if (who && who->GetEntry() != NPC_VAZRUDEN)
237 Talk(SAY_KILL);
238 }
239
240 void JustDied(Unit* killer) override
241 {
242 if (killer && killer != me)
243 Talk(SAY_DIE);
244 _JustDied();
245 }
246
247 void UpdateAI(uint32 diff) override
248 {
249 if (!UpdateVictim())
250 {
251 if (UnsummonCheck < diff && me->IsAlive())
252 {
253 if (!WipeSaid)
254 {
255 Talk(SAY_WIPE);
256 WipeSaid = true;
257 }
258 me->DisappearAndDie();
259 }
260 else
261 UnsummonCheck -= diff;
262 return;
263 }
264
265 if (Revenge_Timer <= diff)
266 {
267 if (Unit* victim = me->GetVictim())
268 DoCast(victim, DUNGEON_MODE(SPELL_REVENGE, SPELL_REVENGE_H));
269 Revenge_Timer = 5000;
270 }
271 else
272 Revenge_Timer -= diff;
273
274 DoMeleeAttackIfReady();
275 }
276
277 private:
278 uint32 Revenge_Timer;
279 bool WipeSaid;
280 uint32 UnsummonCheck;
281 };
282
283 CreatureAI* GetAI(Creature* creature) const override
284 {
285 return new boss_vazrudenAI(creature);
286 }

289class boss_vazruden_the_herald : public CreatureScript
290{
291 public:
292 boss_vazruden_the_herald() : CreatureScript("boss_vazruden_the_herald") { }
293
294 struct boss_vazruden_the_heraldAI : public ScriptedAI
295 {
296 boss_vazruden_the_heraldAI(Creature* creature) : ScriptedAI(creature)
297 {
298 Initialize();
299 summoned = false;
300 sentryDown = false;
301 }
302
303 void Initialize()
304 {
305 phase = 0;
306 waypoint = 0;
307 check = 0;
308 }
309
310 void Reset() override
311 {
312 Initialize();
313 UnsummonAdds();
314 }
315
316 void UnsummonAdds()
317 {
318 if (summoned)
319 {
320 Creature* Nazan = ObjectAccessor::GetCreature(*me, NazanGUID);
321 if (!Nazan)
322 Nazan = me->FindNearestCreature(NPC_NAZAN, 5000);
323 if (Nazan)
324 {
325 Nazan->DisappearAndDie();
326 NazanGUID.Clear();
327 }
328
329 Creature* Vazruden = ObjectAccessor::GetCreature(*me, VazrudenGUID);
330 if (!Vazruden)
331 Vazruden = me->FindNearestCreature(NPC_VAZRUDEN, 5000);
332 if (Vazruden)
333 {
334 Vazruden->DisappearAndDie();
335 VazrudenGUID.Clear();
336 }
337 summoned = false;
338 me->ClearUnitState(UNIT_STATE_ROOT);
339 me->SetVisible(true);
340 }
341 }
342
343 void SummonAdds()
344 {
345 if (!summoned)
346 {
347 if (Creature* Vazruden = me->SummonCreature(NPC_VAZRUDEN, VazrudenMiddle[0], VazrudenMiddle[1], VazrudenMiddle[2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 6000000))
348 VazrudenGUID = Vazruden->GetGUID();
349 if (Creature* Nazan = me->SummonCreature(NPC_NAZAN, VazrudenMiddle[0], VazrudenMiddle[1], VazrudenMiddle[2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 6000000))
350 NazanGUID = Nazan->GetGUID();
351 summoned = true;
352 me->SetVisible(false);
353 me->AddUnitState(UNIT_STATE_ROOT);
354 }
355 }
356
357 void EnterCombat(Unit* /*who*/) override
358 {
359 if (phase == 0)
360 {
361 phase = 1;
362 check = 0;
363 Talk(SAY_INTRO);
364 }
365 }
366
367 void JustSummoned(Creature* summon) override
368 {
369 if (!summon)
370 return;
371 Unit* victim = me->GetVictim();
372 if (summon->GetEntry() == NPC_NAZAN)
373 {
374 summon->SetDisableGravity(true);
375 summon->SetSpeed(MOVE_FLIGHT, 2.5f);
376 if (victim)
377 AttackStartNoMove(victim);
378 }
379 else
380 if (victim)
381 summon->AI()->AttackStart(victim);
382 }
383
384 void SentryDownBy(Unit* killer)
385 {
386 if (sentryDown)
387 {
388 AttackStartNoMove(killer);
389 sentryDown = false;
390 }
391 else
392 sentryDown = true;
393 }
394
395 void UpdateAI(uint32 diff) override
396 {
397 switch (phase)
398 {
399 case 0: // circle around the platform
400 return;
401 break;
402 case 1: // go to the middle and begin the fight
403 if (check <= diff)
404 {
405 if (!me->IsWithinDist3d(VazrudenMiddle[0], VazrudenMiddle[1], VazrudenMiddle[2], 5))
406 {
407 me->GetMotionMaster()->Clear();
408 me->GetMotionMaster()->MovePoint(0, VazrudenMiddle[0], VazrudenMiddle[1], VazrudenMiddle[2]);
409 check = 1000;
410 }
411 else
412 {
413 SummonAdds();
414 phase = 2;
415 return;
416 }
417 }
418 else
419 check -= diff;
420 break;
421 default: // adds do the job now
422 if (check <= diff)
423 {
424 Creature* Nazan = ObjectAccessor::GetCreature(*me, NazanGUID);
425 Creature* Vazruden = ObjectAccessor::GetCreature(*me, VazrudenGUID);
426 if ((Nazan && Nazan->IsAlive()) || (Vazruden && Vazruden->IsAlive()))
427 {
428 if ((Nazan && Nazan->GetVictim()) || (Vazruden && Vazruden->GetVictim()))
429 return;
430 else
431 {
432 UnsummonAdds();
433 EnterEvadeMode();
434 return;
435 }
436 }
437 check = 2000;
438 }
439 else
440 check -= diff;
441 break;
442 }
443 }
444
445 private:
446 uint32 phase;
447 uint32 waypoint;
448 uint32 check;
449 bool sentryDown;
450 ObjectGuid NazanGUID;
451 ObjectGuid VazrudenGUID;
452 bool summoned;
453 };
454
455 CreatureAI* GetAI(Creature* creature) const override
456 {
457 return new boss_vazruden_the_heraldAI(creature);
458 }

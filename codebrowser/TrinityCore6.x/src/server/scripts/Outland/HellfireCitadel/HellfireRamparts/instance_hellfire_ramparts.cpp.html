26#include "ScriptMgr.h"
27#include "InstanceScript.h"
28#include "hellfire_ramparts.h"
29
30class instance_ramparts : public InstanceMapScript
31{
32 public:
33 instance_ramparts() : InstanceMapScript("instance_ramparts", 543) { }
34
35 struct instance_ramparts_InstanceMapScript : public InstanceScript
36 {
37 instance_ramparts_InstanceMapScript(Map* map) : InstanceScript(map)
38 {
39 SetHeaders(DataHeader);
40 SetBossNumber(EncounterCount);
41 }
42
43 void OnGameObjectCreate(GameObject* go) override
44 {
45 switch (go->GetEntry())
46 {
47 case GO_FEL_IRON_CHEST_NORMAL:
48 case GO_FEL_IRON_CHEST_HEROIC:
49 felIronChestGUID = go->GetGUID();
50 break;
51 }
52 }
53
54 bool SetBossState(uint32 type, EncounterState state) override
55 {
56 if (!InstanceScript::SetBossState(type, state))
57 return false;
58
59 switch (type)
60 {
61 case DATA_VAZRUDEN:
62 case DATA_NAZAN:
63 if (GetBossState(DATA_VAZRUDEN) == DONE && GetBossState(DATA_NAZAN) == DONE)
64 if (GameObject* chest = instance->GetGameObject(felIronChestGUID))
65 chest->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
66 break;
67 default:
68 break;
69 }
70 return true;
71 }
72
73 protected:
74 ObjectGuid felIronChestGUID;
75 };
76
77 InstanceScript* GetInstanceScript(InstanceMap* map) const override
78 {
79 return new instance_ramparts_InstanceMapScript(map);
80 }
81};
82
83void AddSC_instance_ramparts()
84{
85 new instance_ramparts;
86}

69class boss_pathaleon_the_calculator : public CreatureScript
70{
71 public:
72 boss_pathaleon_the_calculator(): CreatureScript("boss_pathaleon_the_calculator") { }
73
74 struct boss_pathaleon_the_calculatorAI : public BossAI
75 {
76 boss_pathaleon_the_calculatorAI(Creature* creature) : BossAI(creature, DATA_PATHALEON_THE_CALCULATOR) { }
77
78 void EnterCombat(Unit* /*who*/) override
79 {
80 _EnterCombat();
81 events.ScheduleEvent(EVENT_SUMMON, 30000);
82 events.ScheduleEvent(EVENT_MANA_TAP, urand(12000, 20000));
83 events.ScheduleEvent(EVENT_ARCANE_TORRENT, urand(16000, 25000));
84 events.ScheduleEvent(EVENT_DOMINATION, urand(25000, 40000));
85 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(8000, 13000));
86 Talk(SAY_AGGRO);
87 }
88
89 void KilledUnit(Unit* /*victim*/) override
90 {
91 Talk(SAY_SLAY);
92 }
93
94 void JustDied(Unit* /*killer*/) override
95 {
96 _JustDied();
97 Talk(SAY_DEATH);
98 }
99
100 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
101 {
102 if (me->HealthBelowPctDamaged(20, damage) && !me->HasAura(SPELL_FRENZY))
103 {
104 DoCast(me, SPELL_FRENZY);
105 Talk(SAY_ENRAGE);
106 }
107 }
108
109 void UpdateAI(uint32 diff) override
110 {
111 if (!UpdateVictim())
112 return;
113
114 events.Update(diff);
115
116 if (me->HasUnitState(UNIT_STATE_CASTING))
117 return;
118
119 while (uint32 eventId = events.ExecuteEvent())
120 {
121 switch (eventId)
122 {
123 case EVENT_SUMMON:
124 for (uint8 i = 0; i < 3; ++i)
125 {
126 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
127 {
128 if (Creature* Wraith = me->SummonCreature(NPC_NETHER_WRAITH, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000))
129 Wraith->AI()->AttackStart(target);
130 }
131 }
132 Talk(SAY_SUMMON);
133 events.ScheduleEvent(EVENT_SUMMON, urand(30000, 45000));
134 break;
135 case EVENT_MANA_TAP:
136 DoCastVictim(SPELL_MANA_TAP, true);
137 events.ScheduleEvent(EVENT_MANA_TAP, urand(14000, 22000));
138 break;
139 case EVENT_ARCANE_TORRENT:
140 DoCastVictim(SPELL_ARCANE_TORRENT, true);
141 events.ScheduleEvent(EVENT_ARCANE_TORRENT, urand(12000, 18000));
142 break;
143 case EVENT_DOMINATION:
144 Talk(SAY_DOMINATION);
145 DoCastVictim(SPELL_DOMINATION, true);
146 events.ScheduleEvent(EVENT_DOMINATION, urand(25000, 30000));
147 break;
148 case EVENT_ARCANE_EXPLOSION: // Heroic only
149 DoCastVictim(H_SPELL_ARCANE_EXPLOSION, true);
150 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(10000, 14000));
151 break;
152 default:
153 break;
154 }
155 }
156
157 DoMeleeAttackIfReady();
158 }
159 };
160
161 CreatureAI* GetAI(Creature* creature) const override
162 {
163 return new boss_pathaleon_the_calculatorAI(creature);
164 }

167class npc_nether_wraith : public CreatureScript
168{
169 public:
170
171 npc_nether_wraith() : CreatureScript("npc_nether_wraith") { }
172
173 struct npc_nether_wraithAI : public ScriptedAI
174 {
175 npc_nether_wraithAI(Creature* creature) : ScriptedAI(creature)
176 {
177 Initialize();
178 }
179
180 void Initialize()
181 {
182 ArcaneMissiles_Timer = urand(1000, 4000);
183 Detonation_Timer = 20000;
184 Die_Timer = 2200;
185 Detonation = false;
186 }
187
188 uint32 ArcaneMissiles_Timer;
189 uint32 Detonation_Timer;
190 uint32 Die_Timer;
191 bool Detonation;
192
193 void Reset() override
194 {
195 Initialize();
196 }
197
198 void EnterCombat(Unit* /*who*/) override { }
199
200 void UpdateAI(uint32 diff) override
201 {
202 if (!UpdateVictim())
203 return;
204
205 if (ArcaneMissiles_Timer <= diff)
206 {
207 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
208 DoCast(target, SPELL_ARCANE_MISSILES);
209 else
210 DoCastVictim(SPELL_ARCANE_MISSILES);
211 ArcaneMissiles_Timer = urand(5000, 10000);
212 }
213 else
214 ArcaneMissiles_Timer -=diff;
215
216 if (!Detonation)
217 {
218 if (Detonation_Timer <= diff)
219 {
220 DoCast(me, SPELL_DETONATION);
221 Detonation = true;
222 }
223 else
224 Detonation_Timer -= diff;
225 }
226
227 if (Detonation)
228 {
229 if (Die_Timer <= diff)
230 {
231 me->setDeathState(JUST_DIED);
232 me->RemoveCorpse();
233 }
234 else
235 Die_Timer -= diff;
236 }
237 DoMeleeAttackIfReady();
238 }
239 };
240
241 CreatureAI* GetAI(Creature* creature) const override
242 {
243 return new npc_nether_wraithAI(creature);
244 }
245};

19#include "ScriptMgr.h"
20#include "InstanceScript.h"
21#include "mechanar.h"

23static DoorData const doorData[] =
24{
25 { GO_DOOR_MOARG_1, DATA_GATEWATCHER_IRON_HAND, DOOR_TYPE_PASSAGE, BOUNDARY_NONE },
26 { GO_DOOR_MOARG_2, DATA_GATEWATCHER_GYROKILL, DOOR_TYPE_PASSAGE, BOUNDARY_NONE },
27 { GO_DOOR_NETHERMANCER, DATA_NETHERMANCER_SEPRETHREA, DOOR_TYPE_ROOM, BOUNDARY_NONE },
28 { 0, 0, DOOR_TYPE_ROOM, BOUNDARY_NONE }
29};
30
31class instance_mechanar : public InstanceMapScript
32{
33 public:
34 instance_mechanar(): InstanceMapScript("instance_mechanar", 554) { }
35
36 struct instance_mechanar_InstanceMapScript : public InstanceScript
37 {
38 instance_mechanar_InstanceMapScript(Map* map) : InstanceScript(map)
39 {
40 SetHeaders(DataHeader);
41 SetBossNumber(EncounterCount);
42 LoadDoorData(doorData);
43 }
44
45 void OnGameObjectCreate(GameObject* gameObject) override
46 {
47 switch (gameObject->GetEntry())
48 {
49 case GO_DOOR_MOARG_1:
50 case GO_DOOR_MOARG_2:
51 case GO_DOOR_NETHERMANCER:
52 AddDoor(gameObject, true);
53 break;
54 default:
55 break;
56 }
57 }
58
59 void OnGameObjectRemove(GameObject* gameObject) override
60 {
61 switch (gameObject->GetEntry())
62 {
63 case GO_DOOR_MOARG_1:
64 case GO_DOOR_MOARG_2:
65 case GO_DOOR_NETHERMANCER:
66 AddDoor(gameObject, false);
67 break;
68 default:
69 break;
70 }
71 }
72
73 bool SetBossState(uint32 type, EncounterState state) override
74 {
75 if (!InstanceScript::SetBossState(type, state))
76 return false;
77
78 switch (type)
79 {
80 case DATA_GATEWATCHER_GYROKILL:
81 case DATA_GATEWATCHER_IRON_HAND:
82 case DATA_MECHANOLORD_CAPACITUS:
83 case DATA_NETHERMANCER_SEPRETHREA:
84 case DATA_PATHALEON_THE_CALCULATOR:
85 break;
86 default:
87 break;
88 }
89
90 return true;
91 }
92 };
93
94 InstanceScript* GetInstanceScript(InstanceMap* map) const override
95 {
96 return new instance_mechanar_InstanceMapScript(map);
97 }
98};
99
100void AddSC_instance_mechanar()
101{
102 new instance_mechanar();
103}

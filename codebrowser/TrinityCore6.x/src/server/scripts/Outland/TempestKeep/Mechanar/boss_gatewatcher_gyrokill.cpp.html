53class boss_gatewatcher_gyrokill : public CreatureScript
54{
55 public:
56 boss_gatewatcher_gyrokill() : CreatureScript("boss_gatewatcher_gyrokill") { }
57
58 struct boss_gatewatcher_gyrokillAI : public BossAI
59 {
60 boss_gatewatcher_gyrokillAI(Creature* creature) : BossAI(creature, DATA_GATEWATCHER_GYROKILL) { }
61
62 void JustDied(Unit* /*killer*/) override
63 {
64 _JustDied();
65 Talk(SAY_DEATH);
66 }
67
68 void EnterCombat(Unit* /*who*/) override
69 {
70 _EnterCombat();
71 events.ScheduleEvent(EVENT_STREAM_OF_MACHINE_FLUID, 10000);
72 events.ScheduleEvent(EVENT_SAW_BLADE, 20000);
73 events.ScheduleEvent(EVENT_SHADOW_POWER, 25000);
74 Talk(SAY_AGGRO);
75 }
76
77 void KilledUnit(Unit* /*victim*/) override
78 {
79 Talk(SAY_SLAY);
80 }
81
82 void UpdateAI(uint32 diff) override
83 {
84 if (!UpdateVictim())
85 return;
86
87 events.Update(diff);
88
89 if (me->HasUnitState(UNIT_STATE_CASTING))
90 return;
91
92 while (uint32 eventId = events.ExecuteEvent())
93 {
94 switch (eventId)
95 {
96 case EVENT_STREAM_OF_MACHINE_FLUID:
97 DoCastVictim(SPELL_STREAM_OF_MACHINE_FLUID, true);
98 events.ScheduleEvent(EVENT_STREAM_OF_MACHINE_FLUID, urand(13000, 17000));
99 break;
100 case EVENT_SAW_BLADE:
101 DoCast(me, SPELL_SAW_BLADE);
102 Talk(SAY_SAW_BLADEs);
103 events.ScheduleEvent(EVENT_SAW_BLADE, urand(20000, 30000));
104 break;
105 case EVENT_SHADOW_POWER:
106 DoCast(me, SPELL_SHADOW_POWER);
107 events.ScheduleEvent(EVENT_SAW_BLADE, urand(25000, 35000));
108 break;
109 default:
110 break;
111 }
112 }
113
114 DoMeleeAttackIfReady();
115 }
116 };
117
118 CreatureAI* GetAI(Creature* creature) const override
119 {
120 return new boss_gatewatcher_gyrokillAI(creature);
121 }
122};

124void AddSC_boss_gatewatcher_gyrokill()
125{
126 new boss_gatewatcher_gyrokill();
127}

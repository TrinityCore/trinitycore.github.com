46class boss_void_reaver : public CreatureScript
47{
48 public:
49 boss_void_reaver() : CreatureScript("boss_void_reaver") { }
50
51 struct boss_void_reaverAI : public BossAI
52 {
53 boss_void_reaverAI(Creature* creature) : BossAI(creature, DATA_VOID_REAVER)
54 {
55 Initialize();
56 }
57
58 void Initialize()
59 {
60 Enraged = false;
61 }
62
63 void Reset() override
64 {
65 Initialize();
66 _Reset();
67 }
68
69 void KilledUnit(Unit* /*victim*/) override
70 {
71 Talk(SAY_SLAY);
72 }
73
74 void JustDied(Unit* /*killer*/) override
75 {
76 Talk(SAY_DEATH);
77 DoZoneInCombat();
78 _JustDied();
79 }
80
81 void EnterCombat(Unit* /*who*/) override
82 {
83 Talk(SAY_AGGRO);
84 _EnterCombat();
85
86 events.ScheduleEvent(EVENT_POUNDING, 15000);
87 events.ScheduleEvent(EVENT_ARCANE_ORB, 3000);
88 events.ScheduleEvent(EVENT_KNOCK_AWAY, 30000);
89 events.ScheduleEvent(EVENT_BERSERK, 600000);
90 }
91
92 void UpdateAI(uint32 diff) override
93 {
94 if (!UpdateVictim())
95 return;
96
97 if (me->HasUnitState(UNIT_STATE_CASTING))
98 return;
99
100 while (uint32 eventId = events.ExecuteEvent())
101 {
102 switch (eventId)
103 {
104 case EVENT_POUNDING:
105 DoCastVictim(SPELL_POUNDING);
106 Talk(SAY_POUNDING);
107 events.ScheduleEvent(EVENT_POUNDING, 15000);
108 break;
109 case EVENT_ARCANE_ORB:
110 {
111 Unit* target = NULL;
112 std::list<HostileReference*> t_list = me->getThreatManager().getThreatList();
113 std::vector<Unit*> target_list;
114 for (std::list<HostileReference*>::const_iterator itr = t_list.begin(); itr != t_list.end(); ++itr)
115 {
116 target = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid());
117 if (!target)
118 continue;
119 // exclude pets & totems, 18 yard radius minimum
120 if (target->GetTypeId() == TYPEID_PLAYER && target->IsAlive() && !target->IsWithinDist(me, 18, false))
121 target_list.push_back(target);
122 target = NULL;
123 }
124
125 if (!target_list.empty())
126 target = *(target_list.begin() + rand32() % target_list.size());
127 else
128 target = me->GetVictim();
129
130 if (target)
131 me->CastSpell(target, SPELL_ARCANE_ORB, false, NULL, NULL);
132
133 events.ScheduleEvent(EVENT_ARCANE_ORB, 3000);
134 break;
135 }
136 case EVENT_KNOCK_AWAY:
137 DoCastVictim(SPELL_KNOCK_AWAY);
138 // Drop 25% aggro
139 if (DoGetThreat(me->GetVictim()))
140 DoModifyThreatPercent(me->GetVictim(), -25);
141
142 events.ScheduleEvent(EVENT_KNOCK_AWAY, 30000);
143 break;
144 case EVENT_BERSERK:
145 if (!Enraged)
146 {
147 DoCast(me, SPELL_BERSERK);
148 Enraged = true;
149 }
150 break;
151 default:
152 break;
153 }
154 }
155
156 DoMeleeAttackIfReady();
157
158 EnterEvadeIfOutOfCombatArea(diff);
159 }
160
161 private:
162 bool Enraged;
163 };
164
165 CreatureAI* GetAI(Creature* creature) const override
166 {
167 return GetInstanceAI<boss_void_reaverAI>(creature);
168 }
169};

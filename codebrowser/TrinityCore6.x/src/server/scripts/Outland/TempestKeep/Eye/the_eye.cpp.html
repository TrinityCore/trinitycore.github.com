34enum Spells
35{
36 SPELL_COUNTERCHARGE = 35035,
37 SPELL_KNOCKAWAY = 22893,
38};
39
40class npc_crystalcore_devastator : public CreatureScript
41{
42 public:
43
44 npc_crystalcore_devastator()
45 : CreatureScript("npc_crystalcore_devastator")
46 {
47 }
48 struct npc_crystalcore_devastatorAI : public ScriptedAI
49 {
50 npc_crystalcore_devastatorAI(Creature* creature) : ScriptedAI(creature)
51 {
52 Initialize();
53 }
54
55 void Initialize()
56 {
57 Countercharge_Timer = 9000;
58 Knockaway_Timer = 25000;
59 }
60
61 uint32 Knockaway_Timer;
62 uint32 Countercharge_Timer;
63
64 void Reset() override
65 {
66 Initialize();
67 }
68
69 void EnterCombat(Unit* /*who*/) override
70 {
71 }
72
73 void UpdateAI(uint32 diff) override
74 {
75 if (!UpdateVictim())
76 return;
77
78 //Check if we have a current target
79 //Knockaway_Timer
80 if (Knockaway_Timer <= diff)
81 {
82 DoCastVictim(SPELL_KNOCKAWAY, true);
83
84 // current aggro target is knocked away pick new target
85 Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0);
86
87 if (!target || target == me->GetVictim())
88 target = SelectTarget(SELECT_TARGET_TOPAGGRO, 1);
89
90 if (target)
91 me->TauntApply(target);
92
93 Knockaway_Timer = 23000;
94 }
95 else
96 Knockaway_Timer -= diff;
97
98 //Countercharge_Timer
99 if (Countercharge_Timer <= diff)
100 {
101 DoCast(me, SPELL_COUNTERCHARGE);
102 Countercharge_Timer = 45000;
103 }
104 else
105 Countercharge_Timer -= diff;
106
107 DoMeleeAttackIfReady();
108 }
109 };
110
111 CreatureAI* GetAI(Creature* creature) const override
112 {
113 return new npc_crystalcore_devastatorAI(creature);
114 }
115};
116void AddSC_the_eye()
117{
118 new npc_crystalcore_devastator();
119}

52class instance_the_eye : public InstanceMapScript
53{
54 public:
55 instance_the_eye() : InstanceMapScript("instance_the_eye", 550) { }
56
57 struct instance_the_eye_InstanceMapScript : public InstanceScript
58 {
59 instance_the_eye_InstanceMapScript(Map* map) : InstanceScript(map)
60 {
61 SetHeaders(DataHeader);
62 SetBossNumber(EncounterCount);
63 LoadDoorData(doorData);
64 LoadObjectData(nullptr, gameObjectData);
65 }
66
67 ObjectGuid ThaladredTheDarkener;
68 ObjectGuid LordSanguinar;
69 ObjectGuid GrandAstromancerCapernian;
70 ObjectGuid MasterEngineerTelonicus;
71 ObjectGuid Kaelthas;
72 ObjectGuid Astromancer;
73 ObjectGuid Alar;
74
75 void OnCreatureCreate(Creature* creature) override
76 {
77 switch (creature->GetEntry())
78 {
79 case NPC_SANGUINAR:
80 LordSanguinar = creature->GetGUID();
81 break;
82 case NPC_CAPERNIAN:
83 GrandAstromancerCapernian = creature->GetGUID();
84 break;
85 case NPC_TELONICUS:
86 MasterEngineerTelonicus = creature->GetGUID();
87 break;
88 case NPC_THALADRED:
89 ThaladredTheDarkener = creature->GetGUID();
90 break;
91 case NPC_KAELTHAS:
92 Kaelthas = creature->GetGUID();
93 break;
94 case NPC_HIGH_ASTROMANCER_SOLARIAN:
95 Astromancer = creature->GetGUID();
96 break;
97 case NPC_ALAR:
98 Alar = creature->GetGUID();
99 break;
100 default:
101 break;
102 }
103 }
104
105 ObjectGuid GetGuidData(uint32 identifier) const override
106 {
107 switch (identifier)
108 {
109 case DATA_THALADREDTHEDARKENER: return ThaladredTheDarkener;
110 case DATA_LORDSANGUINAR: return LordSanguinar;
111 case DATA_GRANDASTROMANCERCAPERNIAN: return GrandAstromancerCapernian;
112 case DATA_MASTERENGINEERTELONICUS: return MasterEngineerTelonicus;
113 case DATA_KAELTHAS: return Kaelthas;
114 case DATA_HIGH_ASTROMANCER_SOLARIAN: return Astromancer;
115 case DATA_ALAR: return Alar;
116 }
117 return ObjectGuid::Empty;
118 }
119 };
120
121 InstanceScript* GetInstanceScript(InstanceMap* map) const override
122 {
123 return new instance_the_eye_InstanceMapScript(map);
124 }
125};
126void AddSC_instance_the_eye()

128 new instance_the_eye;
129}

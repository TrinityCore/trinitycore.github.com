312struct advisorbase_ai : public ScriptedAI
313{
314 advisorbase_ai(Creature* creature) : ScriptedAI(creature)
315 {
316 Initialize();
317 instance = creature->GetInstanceScript();
318 }
319
320 void Initialize()
321 {
322 _hasRessurrected = false;
323 _inFakeDeath = false;
324 DelayRes_Target.Clear();
325 }
326
327 void Reset() override
328 {
329 Initialize();
330
331 me->SetStandState(UNIT_STAND_STATE_STAND);
332 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
333 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_STUNNED);
334
335 //reset encounter
336 if (instance->GetBossState(DATA_KAELTHAS) == IN_PROGRESS)
337 if (Creature* Kaelthas = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_KAELTHAS)))
338 Kaelthas->AI()->EnterEvadeMode();
339 }
340
341 void MoveInLineOfSight(Unit* who) override
342 {
343 if (!who || _inFakeDeath || me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
344 return;
345
346 ScriptedAI::MoveInLineOfSight(who);
347 }
348
349 void AttackStart(Unit* who) override
350 {
351 if (!who || _inFakeDeath || me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
352 return;
353
354 ScriptedAI::AttackStart(who);
355 }
356
357 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
358 {
359 if (spell->Id == SPELL_RESSURECTION)
360 {
361 _hasRessurrected = true;
362 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_STUNNED);
363 me->SetStandState(UNIT_STAND_STATE_STAND);
364 events.ScheduleEvent(EVENT_DELAYED_RESSURECTION, 2000);
365 }
366 }
367
368 void DamageTaken(Unit* killer, uint32 &damage) override
369 {
370 if (damage >= me->GetHealth() && !_inFakeDeath && !_hasRessurrected)
371 {
372 //prevent death
373 damage = 0;
374 _inFakeDeath = true;
375
376 me->InterruptNonMeleeSpells(false);
377 me->SetHealth(0);
378 me->RemoveAllAurasOnDeath();
379 me->ModifyAuraState(AURA_STATE_HEALTHLESS_20_PERCENT, false);
380 me->ModifyAuraState(AURA_STATE_HEALTHLESS_35_PERCENT, false);
381 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_STUNNED);
382 me->SetTarget(ObjectGuid::Empty);
383 me->SetStandState(UNIT_STAND_STATE_DEAD);
384 me->GetMotionMaster()->Clear();
385 JustDied(killer);
386 }
387 }
388
389 void JustDied(Unit* /*killer*/) override
390 {
391 if (Creature* kael = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_KAELTHAS)))
392 kael->AI()->DoAction(ACTION_ACTIVE_ADVISOR);
393 }
394
395 void UpdateAI(uint32 diff) override
396 {
397 if (_hasRessurrected)
398 events.Update(diff);
399
400 while (uint32 eventId = events.ExecuteEvent())
401 {
402 if (eventId == EVENT_DELAYED_RESSURECTION)
403 {
404 _inFakeDeath = false;
405
406 Unit* Target = ObjectAccessor::GetUnit(*me, DelayRes_Target);
407 if (!Target)
408 Target = me->GetVictim();
409
410 DoResetThreat();
411 AttackStart(Target);
412 me->GetMotionMaster()->Clear();
413 me->GetMotionMaster()->MoveChase(Target);
414 me->AddThreat(Target, 0.0f);
415 }
416 }
417 }
418 public:
419 EventMap events;
420 InstanceScript* instance;
421 bool _hasRessurrected;
422 bool _inFakeDeath;
423 ObjectGuid DelayRes_Target;
424};

475
476 void DoAction(int32 action) override
477 {
478 switch (action)
479 {
480 case ACTION_START_ENCOUNTER:
481 Talk(SAY_INTRO);
482 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
483
484 _advisorGuid[ADVISOR_THALADRED] = instance->GetGuidData(DATA_THALADREDTHEDARKENER);
485 _advisorGuid[ADVISOR_SANGUINAR] = instance->GetGuidData(DATA_LORDSANGUINAR);
486 _advisorGuid[ADVISOR_CAPERNIAN] = instance->GetGuidData(DATA_GRANDASTROMANCERCAPERNIAN);
487 _advisorGuid[ADVISOR_TELONICUS] = instance->GetGuidData(DATA_MASTERENGINEERTELONICUS);
488
489 _phase = PHASE_INTRO;
490 instance->SetBossState(DATA_KAELTHAS, IN_PROGRESS);
491 events.ScheduleEvent(EVENT_START_ENCOUNTER, 23000);
492 break;
493 case ACTION_PREPARE_ADVISORS:
494 for (uint8 i = 0; i < MAX_ADVISORS; ++i)
495 {
496 if (Creature* creature = ObjectAccessor::GetCreature(*me, _advisorGuid[i]))
497 {
498 creature->Respawn(true);
499 creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
500 creature->AI()->EnterEvadeMode();
501 }
502 }
503 break;
504 case ACTION_ACTIVE_ADVISOR:
505 // They have already been active, so we are not handling new ones, just counting their death.
506 if (_phase == PHASE_REVIVED_ADVISORS)
507 ++_advisorCounter;
508
509 switch (_advisorCounter)
510 {
511 case ADVISOR_THALADRED:
512 Talk(SAY_INTRO_THALADRED);
513 events.ScheduleEvent(EVENT_ACTIVE_ADVISOR, 7000);
514 break;
515 case ADVISOR_SANGUINAR:
516 Talk(SAY_INTRO_SANGUINAR);
517 events.ScheduleEvent(EVENT_ACTIVE_ADVISOR, 12500);
518 break;
519 case ADVISOR_CAPERNIAN:
520 Talk(SAY_INTRO_CAPERNIAN);
521 events.ScheduleEvent(EVENT_ACTIVE_ADVISOR, 7000);
522 break;
523 case ADVISOR_TELONICUS:
524 Talk(SAY_INTRO_TELONICUS);
525 events.ScheduleEvent(EVENT_ACTIVE_ADVISOR, 8400);
526 break;
527 case MAX_DEFEATED_ADVISORS:
528 // Every advisor defeated - Phase 2 starts.
529 Talk(SAY_PHASE2_WEAPON);
530 events.ScheduleEvent(EVENT_SUMMON_WEAPONS, 3500);
531 break;
532 case MAX_KILLED_ADVISORS:
533 // Every advisor killed - Phase 3 starts.
534 events.ScheduleEvent(EVENT_ENGAGE_COMBAT, 5000);
535 break;
536 default:
537 break;
538 }
539 break;
540 case ACTION_SCHEDULE_COMBAT_EVENTS:
541 _phase = PHASE_COMBAT;
542 events.SetPhase(PHASE_COMBAT);
543 events.ScheduleEvent(EVENT_FIREBALL, 1000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
544 events.ScheduleEvent(EVENT_ARCANE_DISRUPTION, 45000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
545 events.ScheduleEvent(EVENT_FLAMESTRIKE, 30000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
546 events.ScheduleEvent(EVENT_MIND_CONTROL, 40000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
547 events.ScheduleEvent(EVENT_SUMMON_PHOENIX, 50000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
548 break;
549 default:
550 break;
551 }
552 }

564
565 void DamageTaken(Unit* attacker, uint32& damage) override
566 {
567 if (_phase == PHASE_NONE)
568 {
569 DoAction(ACTION_START_ENCOUNTER);
570 me->SetTarget(attacker->GetGUID());
571 }
572
573 if (!_hasFullPower && me->HealthBelowPctDamaged(50, damage))
574 {
575 _hasFullPower = true;
576 me->AttackStop();
577 me->InterruptNonMeleeSpells(false);
578 events.CancelEventGroup(EVENT_GROUP_COMBAT);
579 events.CancelEventGroup(EVENT_GROUP_SPECIAL);
580 events.SetPhase(PHASE_TRANSITION);
581 me->SetReactState(REACT_PASSIVE);
582 me->GetMotionMaster()->MovePoint(POINT_START_TRANSITION, TransitionPos[0]);
583 }
584 }
585
586 void MovementInform(uint32 type, uint32 point) override
587 {
588 if (type != POINT_MOTION_TYPE)
589 return;
590
591 switch (point)
592 {
593 case POINT_START_TRANSITION:
594 events.ScheduleEvent(EVENT_TRANSITION_1, 1000);
595 break;
596 case POINT_TRANSITION_CENTER_ASCENDING:
597 me->SetFacingTo(float(M_PI));
598 Talk(SAY_PHASE5_NUTS);
599 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
600 me->SetDisableGravity(true);
601 //me->SetHover(true); -- Set in sniffs, but breaks his visual.
602 events.ScheduleEvent(EVENT_TRANSITION_2, 2000);
603 events.ScheduleEvent(EVENT_SIZE_INCREASE, 5000);
604 break;
605 case POINT_TRANSITION_HALFWAY_ASCENDING:
606 DoCast(me, SPELL_NETHER_BEAM_VISUAL3, true);
607 events.ScheduleEvent(EVENT_TRANSITION_3, 1000);
608 break;
609 case POINT_TRANSITION_TOP:
610 events.ScheduleEvent(EVENT_EXPLODE, 10000);
611 break;
612 case POINT_TRANSITION_HALFWAY_DESCENDING:
613 events.ScheduleEvent(EVENT_TRANSITION_5, 2000);
614 break;
615 case POINT_END_TRANSITION:
616 me->SetReactState(REACT_AGGRESSIVE);
617 me->InterruptNonMeleeSpells(false);
618 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
619 me->RemoveAurasDueToSpell(SPELL_FULLPOWER);
620
621 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0))

655 void UpdateAI(uint32 diff) override
656 {
657 if (_phase == PHASE_COMBAT)
658 if (!UpdateVictim())
659 return;
660
661 events.Update(diff);
662
663 // SPELL_KAEL_GAINING_POWER and SPELL_KAEL_STUNNED are channeling spells that need to be interrupted during his transition.
664 if (me->HasUnitState(UNIT_STATE_CASTING) && !me->FindCurrentSpellBySpellId(SPELL_KAEL_GAINING_POWER) && !me->FindCurrentSpellBySpellId(SPELL_KAEL_STUNNED))
665 return;
666
667 while (uint32 eventId = events.ExecuteEvent())
668 {
669 switch (eventId)
670 {
671 case EVENT_START_ENCOUNTER:
672 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PACIFIED);
673 DoAction(ACTION_ACTIVE_ADVISOR);
674 break;
675 case EVENT_ACTIVE_ADVISOR:
676 if (Creature* advisor = ObjectAccessor::GetCreature(*me, _advisorGuid[_advisorCounter]))
677 {
678 advisor->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
679
680 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
681 advisor->AI()->AttackStart(target);
682 }
683 ++_advisorCounter;
684 break;
685 case EVENT_SUMMON_WEAPONS:
686 {
687 DoCast(me, SPELL_SUMMON_WEAPONS, false);
688
689 uint8 uiMaxWeapon = sizeof(m_auiSpellSummonWeapon) / sizeof(uint32);
690
691 for (uint32 i = 0; i < uiMaxWeapon; ++i)
692 DoCast(me, m_auiSpellSummonWeapon[i], true);
693
694 events.ScheduleEvent(EVENT_REVIVE_ADVISORS, 120000);
695 break;
696 }
697 case EVENT_REVIVE_ADVISORS:
698 _phase = PHASE_REVIVED_ADVISORS;
699 Talk(SAY_PHASE3_ADVANCE);
700 DoCast(me, SPELL_RESSURECTION);
701 break;
702 case EVENT_ENGAGE_COMBAT:
703 Talk(SAY_PHASE4_INTRO2);
704
705 // Sometimes people can collect Aggro in Phase 1-3. Reset threat before releasing Kael.
706 DoResetThreat();
707
708 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_PACIFIED);
709
710 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
711 AttackStart(target);
712
713 DoAction(ACTION_SCHEDULE_COMBAT_EVENTS);
714 events.ScheduleEvent(EVENT_PYROBLAST, 60000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
715 break;
716 case EVENT_FIREBALL:
717 DoCastVictim(SPELL_FIREBALL);
718 events.ScheduleEvent(EVENT_FIREBALL, 2500, EVENT_GROUP_COMBAT, PHASE_COMBAT);
719 break;
720 case EVENT_ARCANE_DISRUPTION:
721 DoCastVictim(SPELL_ARCANE_DISRUPTION, true);
722 events.ScheduleEvent(EVENT_ARCANE_DISRUPTION, 60000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
723 break;
724 case EVENT_FLAMESTRIKE:
725 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
726 DoCast(target, SPELL_FLAME_STRIKE);
727
728 events.ScheduleEvent(EVENT_FLAMESTRIKE, 30000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
729 break;
730 case EVENT_MIND_CONTROL:
731 Talk(SAY_MIND_CONTROL);
732 DoCastAOE(SPELL_MIND_CONTROL, true);
733 events.ScheduleEvent(EVENT_MIND_CONTROL, 60000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
734 break;
735 case EVENT_SUMMON_PHOENIX:
736 DoCast(me, SPELL_PHOENIX_ANIMATION);
737 Talk(SAY_SUMMON_PHOENIX);
738 events.ScheduleEvent(EVENT_SUMMON_PHOENIX, urand(45000, 60000), EVENT_GROUP_COMBAT, PHASE_COMBAT);
739 break;
740 case EVENT_END_TRANSITION:
741 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_NONE);
742 DoCast(SPELL_FULLPOWER);
743 events.ScheduleEvent(EVENT_TRANSITION_4, 2000);
744 break;
745 case EVENT_PYROBLAST:
746 _pyrosCast = 0;
747 Talk(EMOTE_PYROBLAST);
748 DoCast(me, SPELL_SHOCK_BARRIER);
749 events.DelayEvents(10000, EVENT_GROUP_COMBAT);
750 events.ScheduleEvent(EVENT_PYROBLAST_CAST, 1000, EVENT_GROUP_SPECIAL, PHASE_COMBAT);
751 break;
752 case EVENT_PYROBLAST_CAST:
753 if (_pyrosCast < 3)
754 {
755 DoCastVictim(SPELL_PYROBLAST);
756 events.ScheduleEvent(EVENT_PYROBLAST_CAST, 3000);
757 _pyrosCast++;
758 }
759 else
760 events.ScheduleEvent(EVENT_PYROBLAST, 60000, EVENT_GROUP_COMBAT, PHASE_COMBAT);
761 break;
762 case EVENT_GRAVITY_LAPSE:
763 Talk(SAY_GRAVITY_LAPSE);
764 DoCastAOE(SPELL_GRAVITY_LAPSE);
765 DoCast(me, SPELL_NETHER_VAPOR);
766 events.DelayEvents(24000, EVENT_GROUP_COMBAT);
767 events.ScheduleEvent(EVENT_NETHER_BEAM, 3000, EVENT_GROUP_SPECIAL, PHASE_COMBAT);
768 events.ScheduleEvent(EVENT_SHOCK_BARRIER, 1000, EVENT_GROUP_SPECIAL, PHASE_COMBAT);
769 events.ScheduleEvent(EVENT_GRAVITY_LAPSE, 30000, EVENT_GROUP_SPECIAL, PHASE_COMBAT);
770 break;
771 case EVENT_NETHER_BEAM:
772 if (_netherbeamsCast <= 8)
773 {
774 if (Unit* unit = SelectTarget(SELECT_TARGET_RANDOM, 0))
775 DoCast(unit, SPELL_NETHER_BEAM);
776
777 _netherbeamsCast++;
778 events.ScheduleEvent(EVENT_NETHER_BEAM, 3000);
779 }
780 else
781 _netherbeamsCast = 0;
782 break;
783 case EVENT_TRANSITION_1:
784 me->GetMotionMaster()->MovePoint(POINT_TRANSITION_CENTER_ASCENDING, TransitionPos[1]);
785 break;
786 case EVENT_TRANSITION_2:
787 DoCast(me, SPELL_KAEL_GAINING_POWER);
788 me->GetMotionMaster()->Clear();
789 me->RemoveUnitMovementFlag(MOVEMENTFLAG_ROOT);
790 me->GetMotionMaster()->MovePoint(POINT_TRANSITION_HALFWAY_ASCENDING, TransitionPos[2], false);
791 break;
792 case EVENT_TRANSITION_3:
793 me->GetMotionMaster()->MovePoint(POINT_TRANSITION_TOP, TransitionPos[3], false);
794 break;
795 case EVENT_TRANSITION_4:
796 me->GetMotionMaster()->MovePoint(POINT_TRANSITION_HALFWAY_DESCENDING, TransitionPos[4], false);
797 break;
798 case EVENT_TRANSITION_5:
799 me->GetMotionMaster()->MovePoint(POINT_END_TRANSITION, TransitionPos[5], false);
800 break;
801 case EVENT_EXPLODE:
802 me->InterruptNonMeleeSpells(false);
803 me->RemoveAurasDueToSpell(SPELL_NETHER_BEAM_VISUAL3);
804 DoCast(me, SPELL_KAEL_EXPLODES3, true);
805 DoCast(me, SPELL_KAEL_STUNNED); // Core doesn't handle the emote properly while flying.
806 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_DROWNED);
807
808 // Destroy the surrounding environment.
809 if (GameObject* statue = instance->GetGameObject(DATA_KAEL_STATUE_LEFT))
810 statue->UseDoorOrButton();
811
812 if (GameObject* statue = instance->GetGameObject(DATA_KAEL_STATUE_RIGHT))
813 statue->UseDoorOrButton();
814
815 if (GameObject* window = instance->GetGameObject(DATA_TEMPEST_BRIDGE_WINDOW))
816 window->UseDoorOrButton();
817
818 events.ScheduleEvent(EVENT_END_TRANSITION, 10000);
819 break;
820 case EVENT_SIZE_INCREASE:
821 switch (_scaleStage)
822 {
823 case 0:
824 me->SetObjectScale(1.4f);
825 events.ScheduleEvent(EVENT_SIZE_INCREASE, 5000);
826 break;
827 case 1:
828 me->SetObjectScale(1.8f);
829 events.ScheduleEvent(EVENT_SIZE_INCREASE, 3000);
830 break;
831 case 2:
832 me->SetObjectScale(2.0f);
833 events.ScheduleEvent(EVENT_SIZE_INCREASE, 1000);
834 break;
835 case 3:
836 me->SetObjectScale(2.2f);
837 break;
838 default:
839 break;
840 }
841 ++_scaleStage;
842 break;
843 default:
844 break;
845 }
846 }
847
848 if (events.IsInPhase(PHASE_COMBAT))
849 DoMeleeAttackIfReady();
850 }
851 private:
852 uint8 _advisorCounter;
853 uint8 _phase;
854 uint8 _pyrosCast;
855 uint8 _scaleStage;
856 uint8 _netherbeamsCast;
857 bool _hasFullPower;
858 ObjectGuid _advisorGuid[MAX_ADVISORS];
859 };

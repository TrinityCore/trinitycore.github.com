22enum Say
23{
24 // Dalliah the Doomsayer
25 SAY_AGGRO = 1,
26 SAY_SLAY = 2,
27 SAY_WHIRLWIND = 3,
28 SAY_HEAL = 4,
29 SAY_DEATH = 5,
30 SAY_SOCCOTHRATES_DEATH = 7,
31
32 // Wrath-Scryer Soccothrates
33 SAY_AGGRO_DALLIAH_FIRST = 0,
34 SAY_DALLIAH_25_PERCENT = 5
35};
36
37enum Spells
38{
39 SPELL_GIFT_OF_THE_DOOMSAYER = 36173,
40 SPELL_WHIRLWIND = 36142,
41 SPELL_HEAL = 36144,
42 SPELL_SHADOW_WAVE = 39016 // Heroic only
43};
44
45enum Events
46{
47 EVENT_GIFT_OF_THE_DOOMSAYER = 1,
48 EVENT_WHIRLWIND = 2,
49 EVENT_HEAL = 3,
50 EVENT_SHADOW_WAVE = 4, // Heroic only
51 EVENT_ME_FIRST = 5,
52 EVENT_SOCCOTHRATES_DEATH = 6
53};
54
55class boss_dalliah_the_doomsayer : public CreatureScript
56{
57 public:
58 boss_dalliah_the_doomsayer() : CreatureScript("boss_dalliah_the_doomsayer") { }
59
60 struct boss_dalliah_the_doomsayerAI : public BossAI
61 {
62 boss_dalliah_the_doomsayerAI(Creature* creature) : BossAI(creature, DATA_DALLIAH)
63 {
64 soccothratesTaunt = false;
65 soccothratesDeath = false;
66 }
67
68 void Reset() override
69 {
70 _Reset();
71 soccothratesTaunt = false;
72 soccothratesDeath = false;
73 }
74
75 void JustDied(Unit* /*killer*/) override
76 {
77 _JustDied();
78 Talk(SAY_DEATH);
79
80 if (Creature* soccothrates = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SOCCOTHRATES)))
81 if (soccothrates->IsAlive() && !soccothrates->IsInCombat())
82 soccothrates->AI()->SetData(1, 1);
83 }
84
85 void EnterCombat(Unit* /*who*/) override
86 {
87 _EnterCombat();
88 events.ScheduleEvent(EVENT_GIFT_OF_THE_DOOMSAYER, urand(1000, 4000));
89 events.ScheduleEvent(EVENT_WHIRLWIND, urand(7000, 9000));
90 if (IsHeroic())
91 events.ScheduleEvent(EVENT_SHADOW_WAVE, urand(11000, 16000));
92 events.ScheduleEvent(EVENT_ME_FIRST, 6000);
93 Talk(SAY_AGGRO);
94 }
95
96 void KilledUnit(Unit* /*victim*/) override
97 {
98 Talk(SAY_SLAY);
99 }
100
101 void SetData(uint32 /*type*/, uint32 data) override
102 {
103 switch (data)
104 {
105 case 1:
106 events.ScheduleEvent(EVENT_SOCCOTHRATES_DEATH, 6000);
107 soccothratesDeath = true;
108 break;
109 default:
110 break;
111 }
112 }
113
114 void UpdateAI(uint32 diff) override
115 {
116 if (!UpdateVictim())
117 {
118 if (soccothratesDeath)
119 {
120 events.Update(diff);
121
122 while (uint32 eventId = events.ExecuteEvent())
123 {
124 switch (eventId)
125 {
126 case EVENT_SOCCOTHRATES_DEATH:
127 Talk(SAY_SOCCOTHRATES_DEATH);
128 break;
129 default:
130 break;
131 }
132 }
133 }
134
135 return;
136 }
137
138 events.Update(diff);
139
140 if (me->HasUnitState(UNIT_STATE_CASTING))
141 return;
142
143 while (uint32 eventId = events.ExecuteEvent())
144 {
145 switch (eventId)
146 {
147 case EVENT_GIFT_OF_THE_DOOMSAYER:
148 DoCastVictim(SPELL_GIFT_OF_THE_DOOMSAYER, true);
149 events.ScheduleEvent(EVENT_GIFT_OF_THE_DOOMSAYER, urand(16000, 21000));
150 break;
151 case EVENT_WHIRLWIND:
152 DoCast(me, SPELL_WHIRLWIND);
153 Talk(SAY_WHIRLWIND);
154 events.ScheduleEvent(EVENT_WHIRLWIND, urand(19000, 21000));
155 events.ScheduleEvent(EVENT_HEAL, 6000);
156 break;
157 case EVENT_HEAL:
158 DoCast(me, SPELL_HEAL);
159 Talk(SAY_HEAL);
160 break;
161 case EVENT_SHADOW_WAVE:
162 DoCastVictim(SPELL_SHADOW_WAVE, true);
163 events.ScheduleEvent(EVENT_SHADOW_WAVE, urand(11000, 16000));
164 break;
165 case EVENT_ME_FIRST:
166 if (Creature* soccothrates = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SOCCOTHRATES)))
167 if (soccothrates->IsAlive() && !soccothrates->IsInCombat())
168 soccothrates->AI()->Talk(SAY_AGGRO_DALLIAH_FIRST);
169 break;
170 default:
171 break;
172 }
173 }
174
175 if (HealthBelowPct(25) && !soccothratesTaunt)
176 {
177 if (Creature* soccothrates = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SOCCOTHRATES)))
178 soccothrates->AI()->Talk(SAY_DALLIAH_25_PERCENT);
179 soccothratesTaunt = true;
180 }
181
182 DoMeleeAttackIfReady();
183 }
184
185 private:
186 bool soccothratesTaunt;
187 bool soccothratesDeath;
188 };
189
190 CreatureAI* GetAI(Creature* creature) const override
191 {
192 return GetArcatrazAI<boss_dalliah_the_doomsayerAI>(creature);
193 }
194};
195
196void AddSC_boss_dalliah_the_doomsayer()
197{
198 new boss_dalliah_the_doomsayer();
199}

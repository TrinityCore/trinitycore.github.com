66 struct boss_harbinger_skyrissAI : public BossAI
67 {
68 boss_harbinger_skyrissAI(Creature* creature) : BossAI(creature, DATA_HARBINGER_SKYRISS)
69 {
70 Initialize();
71 Intro = false;
72 }

74 void Initialize()
75 {
76 IsImage33 = false;
77 IsImage66 = false;
78
79 Intro_Phase = 1;
80 Intro_Timer = 5000;
81 MindRend_Timer = 3000;
82 Fear_Timer = 15000;
83 Domination_Timer = 30000;
84 ManaBurn_Timer = 25000;
85 }
86
87 bool Intro;
88 bool IsImage33;
89 bool IsImage66;
90
91 uint32 Intro_Phase;
92 uint32 Intro_Timer;
93 uint32 MindRend_Timer;
94 uint32 Fear_Timer;
95 uint32 Domination_Timer;
96 uint32 ManaBurn_Timer;
97
98 void Reset() override
99 {
100 if (!Intro)
101 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
102
103 Initialize();
104 }
105
106 void MoveInLineOfSight(Unit* who) override
107 {
108 if (!Intro)
109 return;
110
111 ScriptedAI::MoveInLineOfSight(who);
112 }
113
114 void EnterCombat(Unit* /*who*/) override { }
115
116 void JustDied(Unit* /*killer*/) override
117 {
118 Talk(SAY_DEATH);
119 _JustDied();
120 }
121
122 void JustSummoned(Creature* summon) override
123 {
124 if (!summon)
125 return;
126 if (IsImage66)
127 summon->SetHealth(summon->CountPctFromMaxHealth(33));
128 else
129 summon->SetHealth(summon->CountPctFromMaxHealth(66));
130 if (me->GetVictim())
131 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
132 summon->AI()->AttackStart(target);
133 }
134
135 void KilledUnit(Unit* victim) override
136 {
137 //won't yell killing pet/other unit
138 if (victim->GetEntry() == NPC_ALPHA_POD_TARGET)
139 return;
140
141 Talk(SAY_KILL);
142 }
143
144 void DoSplit(uint32 val)
145 {
146 if (me->IsNonMeleeSpellCast(false))
147 me->InterruptNonMeleeSpells(false);
148
149 Talk(SAY_IMAGE);
150
151 if (val == 66)
152 DoCast(me, SPELL_66_ILLUSION);
153 else
154 DoCast(me, SPELL_33_ILLUSION);
155 }
156
157 void UpdateAI(uint32 diff) override
158 {
159 if (!Intro)
160 {
161 if (Intro_Timer <= diff)
162 {
163 switch (Intro_Phase)
164 {
165 case 1:
166 Talk(SAY_INTRO);
167 instance->HandleGameObject(instance->GetGuidData(DATA_WARDENS_SHIELD), true);
168 ++Intro_Phase;
169 Intro_Timer = 25000;
170 break;
171 case 2:
172 Talk(SAY_AGGRO);
173 if (Unit* mellic = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_MELLICHAR)))
174 {
175 //should have a better way to do this. possibly spell exist.
176 mellic->setDeathState(JUST_DIED);
177 mellic->SetHealth(0);
178 instance->HandleGameObject(instance->GetGuidData(DATA_WARDENS_SHIELD), false);
179 }
180 ++Intro_Phase;
181 Intro_Timer = 3000;
182 break;
183 case 3:
184 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
185 Intro = true;
186 break;
187 }
188 }
189 else
190 Intro_Timer -=diff;
191 }
192 if (!UpdateVictim())
193 return;
194
195 if (!IsImage66 && !HealthAbovePct(66))
196 {
197 DoSplit(66);
198 IsImage66 = true;
199 }
200 if (!IsImage33 && !HealthAbovePct(33))
201 {
202 DoSplit(33);
203 IsImage33 = true;
204 }
205
206 if (MindRend_Timer <= diff)
207 {
208 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
209 DoCast(target, SPELL_MIND_REND);
210 else
211 DoCastVictim(SPELL_MIND_REND);
212
213 MindRend_Timer = 8000;
214 }
215 else
216 MindRend_Timer -=diff;
217
218 if (Fear_Timer <= diff)
219 {
220 if (me->IsNonMeleeSpellCast(false))
221 return;
222
223 Talk(SAY_FEAR);
224
225 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
226 DoCast(target, SPELL_FEAR);
227 else
228 DoCastVictim(SPELL_FEAR);
229
230 Fear_Timer = 25000;
231 }
232 else
233 Fear_Timer -=diff;
234
235 if (Domination_Timer <= diff)
236 {
237 if (me->IsNonMeleeSpellCast(false))
238 return;
239
240 Talk(SAY_MIND);
241
242 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
243 DoCast(target, SPELL_DOMINATION);
244 else
245 DoCastVictim(SPELL_DOMINATION);
246
247 Domination_Timer = 16000 + rand32() % 16000;
248 }
249 else
250 Domination_Timer -=diff;
251
252 if (IsHeroic())
253 {
254 if (ManaBurn_Timer <= diff)
255 {
256 if (me->IsNonMeleeSpellCast(false))
257 return;
258
259 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
260 DoCast(target, H_SPELL_MANA_BURN);
261
262 ManaBurn_Timer = 16000 + rand32() % 16000;
263 }
264 else
265 ManaBurn_Timer -=diff;
266 }
267 DoMeleeAttackIfReady();
268 }
269 };

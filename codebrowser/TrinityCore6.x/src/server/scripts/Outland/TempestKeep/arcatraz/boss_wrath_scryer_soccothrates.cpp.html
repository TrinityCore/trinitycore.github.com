78class boss_wrath_scryer_soccothrates : public CreatureScript
79{
80 public:
81 boss_wrath_scryer_soccothrates() : CreatureScript("boss_wrath_scryer_soccothrates") { }
82
83 struct boss_wrath_scryer_soccothratesAI : public BossAI
84 {
85 boss_wrath_scryer_soccothratesAI(Creature* creature) : BossAI(creature, DATA_SOCCOTHRATES)
86 {
87 preFight = false;
88 dalliahTaunt = false;
89 dalliahDeath = false;
90 }
91
92 void Reset() override
93 {
94 _Reset();
95 preFight = false;
96 dalliahTaunt = false;
97 dalliahDeath = false;
98 DoCast(me, SPELL_FEL_IMMOLATION);
99 }
100
101 void JustDied(Unit* /*killer*/) override
102 {
103 _JustDied();
104 Talk(SAY_DEATH);
105
106 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
107 if (dalliah->IsAlive() && !dalliah->IsInCombat())
108 dalliah->AI()->SetData(1, 1);
109 }
110
111 void EnterCombat(Unit* /*who*/) override
112 {
113 _EnterCombat();
114 events.ScheduleEvent(EVENT_FELFIRE_SHOCK, urand(12000, 14000));
115 events.ScheduleEvent(EVENT_KNOCK_AWAY, urand(11000, 12000));
116 events.ScheduleEvent(EVENT_ME_FIRST, 6000);
117 Talk(SAY_AGGRO);
118 preFight = false;
119 }
120
121 void KilledUnit(Unit* /*victim*/) override
122 {
123 Talk(SAY_SLAY);
124 }
125
126 void MoveInLineOfSight(Unit* who) override
127 {
128 if (instance->GetData(DATA_CONVERSATION) == NOT_STARTED && who->GetTypeId() == TYPEID_PLAYER && me->IsWithinDistInMap(who, 70.0f))
129 {
130 Talk(SAY_SOCCOTHRATES_CONVO_1);
131 instance->SetData(DATA_CONVERSATION, DONE);
132
133 preFight = true;
134 events.ScheduleEvent(EVENT_PREFIGHT_1, 2000);
135 }
136 }
137
138 void SetData(uint32 /*type*/, uint32 data) override
139 {
140 switch (data)
141 {
142 case 1:
143 events.ScheduleEvent(EVENT_DALLIAH_DEATH, 6000);
144 dalliahDeath = true;
145 break;
146 default:
147 break;
148 }
149 }
150
151 void UpdateAI(uint32 diff) override
152 {
153 if (!UpdateVictim())
154 {
155 if (preFight)
156 {
157 events.Update(diff);
158
159 while (uint32 eventId = events.ExecuteEvent())
160 {
161 switch (eventId)
162 {
163 case EVENT_PREFIGHT_1:
164 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
165 dalliah->AI()->Talk(SAY_DALLIAH_CONVO_1);
166 events.ScheduleEvent(EVENT_PREFIGHT_2, 3000);
167 break;
168 case EVENT_PREFIGHT_2:
169 Talk(SAY_SOCCOTHRATES_CONVO_2);
170 events.ScheduleEvent(EVENT_PREFIGHT_3, 3000);
171 break;
172 case EVENT_PREFIGHT_3:
173 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
174 dalliah->AI()->Talk(SAY_DALLIAH_CONVO_2);
175 events.ScheduleEvent(EVENT_PREFIGHT_4, 6000);
176 break;
177 case EVENT_PREFIGHT_4:
178 Talk(SAY_SOCCOTHRATES_CONVO_3);
179 events.ScheduleEvent(EVENT_PREFIGHT_5, 2000);
180 break;
181 case EVENT_PREFIGHT_5:
182 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
183 dalliah->AI()->Talk(SAY_DALLIAH_CONVO_3);
184 events.ScheduleEvent(EVENT_PREFIGHT_6, 3000);
185 break;
186 case EVENT_PREFIGHT_6:
187 Talk(SAY_SOCCOTHRATES_CONVO_4);
188 events.ScheduleEvent(EVENT_PREFIGHT_7, 2000);
189 break;
190 case EVENT_PREFIGHT_7:
191 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
192 dalliah->GetMotionMaster()->MovePoint(0, 118.6048f, 96.84852f, 22.44115f);
193 events.ScheduleEvent(EVENT_PREFIGHT_8, 4000);
194 break;
195 case EVENT_PREFIGHT_8:
196 me->GetMotionMaster()->MovePoint(0, 122.1035f, 192.7203f, 22.44115f);
197 events.ScheduleEvent(EVENT_PREFIGHT_9, 4000);
198 break;
199 case EVENT_PREFIGHT_9:
200 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
201 {
202 dalliah->SetFacingToObject(me);
203 me->SetFacingToObject(dalliah);
204 dalliah->SetHomePosition(dalliah->GetPositionX(), dalliah->GetPositionY(), dalliah->GetPositionZ(), 1.51737f);
205 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 4.725722f);
206 preFight = false;
207 }
208 break;
209 default:
210 break;
211 }
212 }
213 }
214
215 if (dalliahDeath)
216 {
217 events.Update(diff);
218
219 while (uint32 eventId = events.ExecuteEvent())
220 {
221 switch (eventId)
222 {
223 case EVENT_DALLIAH_DEATH:
224 Talk(SAY_DALLIAH_DEATH);
225 break;
226 default:
227 break;
228 }
229 }
230 }
231
232 return;
233 }
234
235 events.Update(diff);
236
237 if (me->HasUnitState(UNIT_STATE_CASTING))
238 return;
239
240 while (uint32 eventId = events.ExecuteEvent())
241 {
242 switch (eventId)
243 {
244 case EVENT_FELFIRE_SHOCK:
245 DoCastVictim(SPELL_FELFIRE_SHOCK, true);
246 events.ScheduleEvent(EVENT_FELFIRE_SHOCK, urand(12000, 14000));
247 break;
248 case EVENT_KNOCK_AWAY:
249 DoCast(me, SPELL_KNOCK_AWAY);
250 Talk(SAY_KNOCK_AWAY);
251 events.ScheduleEvent(EVENT_KNOCK_AWAY, urand(11000, 12000));
252 break;
253 case EVENT_ME_FIRST:
254 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
255 if (dalliah->IsAlive() && !dalliah->IsInCombat())
256 dalliah->AI()->Talk(SAY_AGGRO_SOCCOTHRATES_FIRST);
257 break;
258 default:
259 break;
260 }
261 }
262
263 if (HealthBelowPct(25) && !dalliahTaunt)
264 {
265 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
266 dalliah->AI()->Talk(SAY_SOCCOTHRATES_25_PERCENT);
267 dalliahTaunt = true;
268 }
269
270 DoMeleeAttackIfReady();
271 }
272
273 private:
274 bool preFight;
275 bool dalliahTaunt;
276 bool dalliahDeath;
277 };
278
279 CreatureAI* GetAI(Creature* creature) const override
280 {
281 return GetArcatrazAI<boss_wrath_scryer_soccothratesAI>(creature);
282 }
283};

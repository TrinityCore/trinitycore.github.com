144 void DoSummons()
145 {
146 switch (rand32() % 4)
147 {
148 case 0:
149 DoCast(me, SPELL_SUMMON_LASHER_1, true);
150 DoCast(me, SPELL_SUMMON_FLAYER_1, true);
151 break;
152 case 1:
153 DoCast(me, SPELL_SUMMON_LASHER_2, true);
154 DoCast(me, SPELL_SUMMON_FLAYER_2, true);
155 break;
156 case 2:
157 DoCast(me, SPELL_SUMMON_LASHER_3, true);
158 DoCast(me, SPELL_SUMMON_FLAYER_3, true);
159 break;
160 case 3:
161 DoCast(me, SPELL_SUMMON_LASHER_4, true);
162 DoCast(me, SPELL_SUMMON_FLAYER_4, true);
163 break;
164 }
165 CanSummon = false;
166 }

178 void UpdateAI(uint32 diff) override
179 {
180 if (!UpdateVictim())
181 return;
182
183 if (CanSummon)
184 {
185 if (Summon_Timer <= diff)
186 {
187 Talk(EMOTE_SUMMON);
188 DoSummons();
189 Summon_Timer = 2500;
190 }
191 else
192 Summon_Timer -= diff;
193 }
194
195 if (Allergic_Timer <= diff)
196 {
197 DoCastVictim(SPELL_ALLERGIC_REACTION);
198 Allergic_Timer = 25000 + rand32() % 15000;
199 }
200 else
201 Allergic_Timer -= diff;
202
203 if (Teleport_Timer <= diff)
204 {
205 DoCast(me, SPELL_TELEPORT_SELF);
206 Teleport_Timer = 30000 + rand32() % 10000;
207 CanSummon = true;
208 }
209 else
210 Teleport_Timer -= diff;
211
212 if (Transform_Timer <= diff)
213 {
214 DoTransform();
215 Transform_Timer = 25000 + rand32() % 15000;
216 }
217 else
218 Transform_Timer -= diff;
219
220 DoMeleeAttackIfReady();
221 }
222 };

227 }

47class boss_commander_sarannis : public CreatureScript
48{
49 public: boss_commander_sarannis() : CreatureScript("boss_commander_sarannis") { }
50
51 struct boss_commander_sarannisAI : public BossAI
52 {
53 boss_commander_sarannisAI(Creature* creature) : BossAI(creature, DATA_COMMANDER_SARANNIS)
54 {
55 Initialize();
56 }
57
58 void Initialize()
59 {
60 _phase = true;
61 }
62
63 void Reset() override
64 {
65 _Reset();
66 Initialize();
67 }
68
69 void EnterCombat(Unit* /*who*/) override
70 {
71 _EnterCombat();
72 Talk(SAY_AGGRO);
73 events.ScheduleEvent(EVENT_ARCANE_RESONANCE, 42700);
74 events.ScheduleEvent(EVENT_ARCANE_DEVASTATION, 15200);
75 }
76
77 void KilledUnit(Unit* /*victim*/) override
78 {
79 Talk(SAY_KILL);
80 }
81
82 void JustDied(Unit* /*killer*/) override
83 {
84 _JustDied();
85 Talk(SAY_DEATH);
86 }
87
88 void DamageTaken(Unit* /*killer*/, uint32 &damage) override
89 {
90 if (me->HealthBelowPctDamaged(50, damage) && _phase)
91 {
92 _phase = false;
93 Talk(EMOTE_SUMMON);
94 Talk(SAY_SUMMON);
95 DoCast(me, SPELL_SUMMON_REINFORCEMENTS);
96 }
97 }
98
99 void JustSummoned(Creature* summon) override
100 {
101 BossAI::JustSummoned(summon);
102 }
103
104 void UpdateAI(uint32 diff) override
105 {
106 if (!UpdateVictim())
107 return;
108
109 events.Update(diff);
110
111 if (me->HasUnitState(UNIT_STATE_CASTING))
112 return;
113
114 while (uint32 eventId = events.ExecuteEvent())
115 {
116 switch (eventId)
117 {
118 case EVENT_ARCANE_RESONANCE:
119 Talk(SAY_ARCANE_RESONANCE);
120 DoCastVictim(SPELL_ARCANE_RESONANCE, true);
121 events.ScheduleEvent(EVENT_ARCANE_RESONANCE, 42700);
122 break;
123 case EVENT_ARCANE_DEVASTATION:
124 Talk(SAY_ARCANE_DEVASTATION);
125 DoCastVictim(SPELL_ARCANE_DEVASTATION, true);
126 events.ScheduleEvent(EVENT_ARCANE_DEVASTATION, urand(11000, 19200));
127 break;
128 default:
129 break;
130 }
131 }
132
133 DoMeleeAttackIfReady();
134 }
135
136 private:
137 bool _phase;
138 };
139
140 CreatureAI* GetAI(Creature* creature) const override
141 {
142 return new boss_commander_sarannisAI(creature);
143 }
144};

160class spell_commander_sarannis_summon_reinforcements : public SpellScriptLoader
161{
162 public:
163 spell_commander_sarannis_summon_reinforcements() : SpellScriptLoader("spell_commander_sarannis_summon_reinforcements") { }
164
165 class spell_commander_sarannis_summon_reinforcements_SpellScript : public SpellScript
166 {
167 PrepareSpellScript(spell_commander_sarannis_summon_reinforcements_SpellScript);
168
169 void HandleCast(SpellEffIndex /*effIndex*/)
170 {
171 GetCaster()->SummonCreature(NPC_SUMMONED_BLOODWARDER_MENDER, PosSummonReinforcements[0], TEMPSUMMON_CORPSE_DESPAWN);
172 GetCaster()->SummonCreature(NPC_SUMMONED_BLOODWARDER_RESERVIST, PosSummonReinforcements[1], TEMPSUMMON_CORPSE_DESPAWN);
173 GetCaster()->SummonCreature(NPC_SUMMONED_BLOODWARDER_RESERVIST, PosSummonReinforcements[2], TEMPSUMMON_CORPSE_DESPAWN);
174 if (GetCaster()->GetMap()->IsHeroic())
175 GetCaster()->SummonCreature(NPC_SUMMONED_BLOODWARDER_RESERVIST, PosSummonReinforcements[3], TEMPSUMMON_CORPSE_DESPAWN);
176 }
177
178 void Register() override
179 {
180 OnEffectHitTarget += SpellEffectFn(spell_commander_sarannis_summon_reinforcements_SpellScript::HandleCast, EFFECT_0, SPELL_EFFECT_DUMMY);
181 }
182 };
183
184 SpellScript* GetSpellScript() const override
185 {
186 return new spell_commander_sarannis_summon_reinforcements_SpellScript();
187 }
188};

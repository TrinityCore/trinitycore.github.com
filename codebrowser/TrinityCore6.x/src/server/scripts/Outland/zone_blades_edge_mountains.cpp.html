131 void SpellHit(Unit* caster, const SpellInfo* spell) override
132 {
133 if (spell->Id == SPELL_T_PHASE_MODULATOR && caster->GetTypeId() == TYPEID_PLAYER)
134 {
135 const uint32 entry_list[4] = {ENTRY_PROTO, ENTRY_ADOLE, ENTRY_MATUR, ENTRY_NIHIL};
136 int cid = rand32() % (4 - 1);
137
138 if (entry_list[cid] == me->GetEntry())
139 ++cid;
140
141 //we are nihil, so say before transform
142 if (me->GetEntry() == ENTRY_NIHIL)
143 {
144 Talk(SAY_NIHIL_INTERRUPT);
145 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
146 IsNihil = false;
147 }
148
149 if (me->UpdateEntry(entry_list[cid]))
150 {
151 if (entry_list[cid] == ENTRY_NIHIL)
152 {
153 EnterEvadeMode();
154 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
155 IsNihil = true;
156 }else
157 AttackStart(caster);
158 }
159 }
160 }

162 void UpdateAI(uint32 diff) override
163 {
164 if (IsNihil)
165 {
166 if (NihilSpeech_Timer <= diff)
167 {
168 switch (NihilSpeech_Phase)
169 {
170 case 0:
171 Talk(SAY_NIHIL_1);
172 ++NihilSpeech_Phase;
173 break;
174 case 1:
175 Talk(SAY_NIHIL_2);
176 ++NihilSpeech_Phase;
177 break;
178 case 2:
179 Talk(SAY_NIHIL_3);
180 ++NihilSpeech_Phase;
181 break;
182 case 3:
183 Talk(SAY_NIHIL_4);
184 ++NihilSpeech_Phase;
185 break;
186 case 4:
187 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
188 //take off to location above
189 me->GetMotionMaster()->MovePoint(0, me->GetPositionX()+50.0f, me->GetPositionY(), me->GetPositionZ()+50.0f);
190 ++NihilSpeech_Phase;
191 break;
192 }
193 NihilSpeech_Timer = 5000;
194 } else NihilSpeech_Timer -=diff;
195
196 //anything below here is not interesting for Nihil, so skip it
197 return;
198 }
199
200 if (!UpdateVictim())
201 return;
202
203 if (IntangiblePresence_Timer <= diff)
204 {
205 DoCastVictim(SPELL_INTANGIBLE_PRESENCE);
206 IntangiblePresence_Timer = 15000 + rand32() % 15000;
207 } else IntangiblePresence_Timer -= diff;
208
209 if (ManaBurn_Timer <= diff)
210 {
211 Unit* target = me->GetVictim();
212 if (target && target->getPowerType() == POWER_MANA)
213 DoCast(target, SPELL_MANA_BURN);
214 ManaBurn_Timer = 8000 + rand32() % 8000;
215 } else ManaBurn_Timer -= diff;
216
217 if (ArcaneBlast_Timer <= diff)
218 {
219 DoCastVictim(SPELL_ARCANE_BLAST);
220 ArcaneBlast_Timer = 2500 + rand32() % 5000;
221 } else ArcaneBlast_Timer -= diff;
222
223 DoMeleeAttackIfReady();
224 }
225 };

230 }

413class npc_simon_bunny : public CreatureScript

418 struct npc_simon_bunnyAI : public ScriptedAI
419 {
420 npc_simon_bunnyAI(Creature* creature) : ScriptedAI(creature)
421 {
422 large = false;
423 listening = false;
424 gameLevel = 0;
425 fails = 0;
426 gameTicks = 0;
427 memset(clusterIds, 0, sizeof(clusterIds));
428 zCoordCorrection = 0.f;
429 searchDistance = 0.f;
430 }
431
432 bool large;
433 bool listening;
434 uint8 gameLevel;
435 uint8 fails;
436 uint8 gameTicks;
437 ObjectGuid playerGUID;
438 uint32 clusterIds[SIMON_MAX_COLORS];
439 float zCoordCorrection;
440 float searchDistance;
441 EventMap _events;
442 std::list<uint8> colorSequence, playableSequence, playerSequence;
443
444 void UpdateAI(uint32 diff) override
445 {
446 _events.Update(diff);
447
448 switch (_events.ExecuteEvent())
449 {
450 case EVENT_SIMON_PERIODIC_PLAYER_CHECK:
451 if (!CheckPlayer())
452 ResetNode();
453 else
454 _events.ScheduleEvent(EVENT_SIMON_PERIODIC_PLAYER_CHECK, 2000);
455 break;
456 case EVENT_SIMON_SETUP_PRE_GAME:
457 SetUpPreGame();
458 _events.CancelEvent(EVENT_SIMON_GAME_TICK);
459 _events.ScheduleEvent(EVENT_SIMON_PLAY_SEQUENCE, 1000);
460 break;
461 case EVENT_SIMON_PLAY_SEQUENCE:
462 if (!playableSequence.empty())
463 {
464 PlayNextColor();
465 _events.ScheduleEvent(EVENT_SIMON_PLAY_SEQUENCE, 1500);
466 }
467 else
468 {
469 listening = true;
470 DoCast(SPELL_VISUAL_START_PLAYER_LEVEL);
471 playerSequence.clear();
472 PrepareClusters();
473 gameTicks = 0;
474 _events.ScheduleEvent(EVENT_SIMON_GAME_TICK, 3000);
475 }
476 break;
477 case EVENT_SIMON_GAME_TICK:
478 DoCast(SPELL_AUDIBLE_GAME_TICK);
479
480 if (gameTicks > gameLevel)
481 _events.ScheduleEvent(EVENT_SIMON_TOO_LONG_TIME, 500);
482 else
483 _events.ScheduleEvent(EVENT_SIMON_GAME_TICK, 3000);
484 gameTicks++;
485 break;
486 case EVENT_SIMON_RESET_CLUSTERS:
487 PrepareClusters(true);
488 break;
489 case EVENT_SIMON_TOO_LONG_TIME:
490 DoAction(ACTION_SIMON_WRONG_SEQUENCE);
491 break;
492 case EVENT_SIMON_ROUND_FINISHED:
493 DoAction(ACTION_SIMON_ROUND_FINISHED);
494 break;
495 }
496 }
497
498 void DoAction(int32 action) override
499 {
500 switch (action)
501 {
502 case ACTION_SIMON_ROUND_FINISHED:
503 listening = false;
504 DoCast(SPELL_VISUAL_START_AI_LEVEL);
505 GiveRewardForLevel(gameLevel);
506 _events.CancelEventGroup(0);
507 if (gameLevel == 10)
508 ResetNode();
509 else
510 _events.ScheduleEvent(EVENT_SIMON_SETUP_PRE_GAME, 1000);
511 break;
512 case ACTION_SIMON_CORRECT_FULL_SEQUENCE:
513 gameLevel++;
514 DoAction(ACTION_SIMON_ROUND_FINISHED);
515 break;
516 case ACTION_SIMON_WRONG_SEQUENCE:
517 GivePunishment();
518 DoAction(ACTION_SIMON_ROUND_FINISHED);
519 break;
520 }
521 }

523 // Called by color clusters script (go_simon_cluster) and used for knowing the button pressed by player
524 void SetData(uint32 type, uint32 /*data*/) override
525 {
526 if (!listening)
527 return;

529 uint8 pressedColor = SIMON_MAX_COLORS;

542 _events.ScheduleEvent(EVENT_SIMON_RESET_CLUSTERS, 500);
543 CheckPlayerSequence();
544 }

556 /*
557 Resets all variables and also find the ids of the four closests color clusters, since every simon
558 node have diferent ids for clusters this is absolutely NECESSARY.
559 */
560 void StartGame()
561 {
562 listening = false;
563 gameLevel = 0;
564 fails = 0;
565 gameTicks = 0;
566 zCoordCorrection = large ? 8.0f : 2.75f;
567 searchDistance = large ? 13.0f : 5.0f;
568 colorSequence.clear();
569 playableSequence.clear();
570 playerSequence.clear();
571 me->SetObjectScale(large ? 2.0f : 1.0f);
572
573 std::list<WorldObject*> ClusterList;
574 Trinity::AllWorldObjectsInRange objects(me, searchDistance);
575 Trinity::WorldObjectListSearcher<Trinity::AllWorldObjectsInRange> searcher(me, ClusterList, objects);
576 me->VisitNearbyObject(searchDistance, searcher);
577
578 for (std::list<WorldObject*>::const_iterator i = ClusterList.begin(); i != ClusterList.end(); ++i)
579 {
580 if (GameObject* go = (*i)->ToGameObject())
581 {
582 // We are checking for displayid because all simon nodes have 4 clusters with different entries
583 if (large)
584 {
585 switch (go->GetGOInfo()->displayId)
586 {
587 case GO_BLUE_CLUSTER_DISPLAY_LARGE:
588 clusterIds[SIMON_BLUE] = go->GetEntry();
589 break;
590
591 case GO_RED_CLUSTER_DISPLAY_LARGE:
592 clusterIds[SIMON_RED] = go->GetEntry();
593 break;
594
595 case GO_GREEN_CLUSTER_DISPLAY_LARGE:
596 clusterIds[SIMON_GREEN] = go->GetEntry();
597 break;
598
599 case GO_YELLOW_CLUSTER_DISPLAY_LARGE:
600 clusterIds[SIMON_YELLOW] = go->GetEntry();
601 break;
602 }
603 }
604 else
605 {
606 switch (go->GetGOInfo()->displayId)
607 {
608 case GO_BLUE_CLUSTER_DISPLAY:
609 clusterIds[SIMON_BLUE] = go->GetEntry();
610 break;
611
612 case GO_RED_CLUSTER_DISPLAY:
613 clusterIds[SIMON_RED] = go->GetEntry();
614 break;
615
616 case GO_GREEN_CLUSTER_DISPLAY:
617 clusterIds[SIMON_GREEN] = go->GetEntry();
618 break;
619
620 case GO_YELLOW_CLUSTER_DISPLAY:
621 clusterIds[SIMON_YELLOW] = go->GetEntry();
622 break;
623 }
624 }
625 }
626 }
627
628 _events.Reset();
629 _events.ScheduleEvent(EVENT_SIMON_ROUND_FINISHED, 1000);
630 _events.ScheduleEvent(EVENT_SIMON_PERIODIC_PLAYER_CHECK, 2000);
631
632 if (GameObject* relic = me->FindNearestGameObject(large ? GO_APEXIS_MONUMENT : GO_APEXIS_RELIC, searchDistance))
633 relic->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
634 }

636 // Called when despawning the bunny. Sets all the node GOs to their default states.
637 void ResetNode()
638 {
639 DoPlaySoundToSet(me, SOUND_DISABLE_NODE);
640
641 for (uint32 clusterId = SIMON_BLUE; clusterId < SIMON_MAX_COLORS; clusterId++)
642 if (GameObject* cluster = me->FindNearestGameObject(clusterIds[clusterId], searchDistance))
643 cluster->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
644
645 for (uint32 auraId = GO_AURA_BLUE; auraId <= GO_AURA_YELLOW; auraId++)
646 if (GameObject* auraGo = me->FindNearestGameObject(auraId, searchDistance))
647 auraGo->RemoveFromWorld();
648
649 if (GameObject* relic = me->FindNearestGameObject(large ? GO_APEXIS_MONUMENT : GO_APEXIS_RELIC, searchDistance))
650 relic->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
651
652 me->DespawnOrUnsummon(1000);
653 }
654
655 /*
656 Called on every button click of player. Adds the clicked color to the player created sequence and
657 checks if it corresponds to the AI created sequence. If so, incremente gameLevel and start a new
658 round, if not, give punishment and restart current level.
659 */
660 void CheckPlayerSequence()
661 {
662 bool correct = true;
663 if (playerSequence.size() <= colorSequence.size())
664 for (std::list<uint8>::const_iterator i = playerSequence.begin(), j = colorSequence.begin(); i != playerSequence.end(); ++i, ++j)
665 if ((*i) != (*j))
666 correct = false;
667
668 if (correct && (playerSequence.size() == colorSequence.size()))
669 DoAction(ACTION_SIMON_CORRECT_FULL_SEQUENCE);
670 else if (!correct)
671 DoAction(ACTION_SIMON_WRONG_SEQUENCE);
672 }
673
674 /*
675 Generates a random sequence of colors depending on the gameLevel. We also copy this sequence to
676 the playableSequence wich will be used when playing the sequence to the player.
677 */
678 void GenerateColorSequence()
679 {
680 colorSequence.clear();
681 for (uint8 i = 0; i <= gameLevel; i++)
682 colorSequence.push_back(RAND(SIMON_BLUE, SIMON_RED, SIMON_GREEN, SIMON_YELLOW));
683
684 for (std::list<uint8>::const_iterator i = colorSequence.begin(); i != colorSequence.end(); ++i)
685 playableSequence.push_back(*i);
686 }

700 if (GameObject* auraGo = me->FindNearestGameObject(auraId, searchDistance))
701 auraGo->RemoveFromWorld();
702 }

738 /*
739 Creates the transparent glowing auras on every cluster of this node.
740 After calling this function bunny is teleported to the center of the node.
741 */
742 void SetUpPreGame()
743 {
744 for (uint32 clusterId = SIMON_BLUE; clusterId < SIMON_MAX_COLORS; clusterId++)
745 {
746 if (GameObject* cluster = me->FindNearestGameObject(clusterIds[clusterId], 2.0f*searchDistance))
747 {
748 cluster->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
749
750 // break since we don't need glowing auras for large clusters
751 if (large)
752 break;
753
754 float x, y, z, o;
755 cluster->GetPosition(x, y, z, o);
756 me->NearTeleportTo(x, y, z, o);
757
758 uint32 preGameSpellId;
759 if (cluster->GetEntry() == clusterIds[SIMON_RED])
760 preGameSpellId = SPELL_PRE_GAME_RED;
761 else if (cluster->GetEntry() == clusterIds[SIMON_BLUE])
762 preGameSpellId = SPELL_PRE_GAME_BLUE;
763 else if (cluster->GetEntry() == clusterIds[SIMON_GREEN])
764 preGameSpellId = SPELL_PRE_GAME_GREEN;
765 else if (cluster->GetEntry() == clusterIds[SIMON_YELLOW])
766 preGameSpellId = SPELL_PRE_GAME_YELLOW;
767 else break;
768
769 me->CastSpell(cluster, preGameSpellId, true);
770 }
771 }
772
773 if (GameObject* relic = me->FindNearestGameObject(large ? GO_APEXIS_MONUMENT : GO_APEXIS_RELIC, searchDistance))
774 {
775 float x, y, z, o;
776 relic->GetPosition(x, y, z, o);
777 me->NearTeleportTo(x, y, z + zCoordCorrection, o);
778 }
779
780 GenerateColorSequence();
781 }

783 // Handles the spell rewards. The spells also have the QuestCompleteEffect, so quests credits are working.
784 void GiveRewardForLevel(uint8 level)
785 {
786 uint32 rewSpell = 0;
787 switch (level)
788 {
789 case 6:
790 if (large)
791 GivePunishment();
792 else
793 rewSpell = SPELL_REWARD_BUFF_1;
794 break;
795 case 8:
796 rewSpell = SPELL_REWARD_BUFF_2;
797 break;
798 case 10:
799 rewSpell = SPELL_REWARD_BUFF_3;
800 break;
801 }
802
803 if (rewSpell)
804 if (Player* player = ObjectAccessor::GetPlayer(*me, playerGUID))
805 DoCast(player, rewSpell, true);
806 }
807
808 /*
809 Depending on the number of failed pushes for player the damage of the spell scales, so we first
810 cast the spell on the target that hits for 50 and shows the visual and then forces the player
811 to cast the damaging spell on it self with the modified basepoints.
812 4 fails = death.
813 On large nodes punishment and reward are the same, summoning the Apexis Guardian.
814 */
815 void GivePunishment()
816 {
817 if (large)
818 {
819 if (Player* player = ObjectAccessor::GetPlayer(*me, playerGUID))
820 if (Creature* guardian = me->SummonCreature(NPC_APEXIS_GUARDIAN, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ() - zCoordCorrection, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 20000))
821 guardian->AI()->AttackStart(player);
822
823 ResetNode();
824 }
825 else
826 {
827 fails++;
828
829 if (Player* player = ObjectAccessor::GetPlayer(*me, playerGUID))
830 DoCast(player, SPELL_BAD_PRESS_TRIGGER, true);
831
832 if (fails >= 4)
833 ResetNode();
834 }
835 }

837 void SpellHitTarget(Unit* target, const SpellInfo* spell) override
838 {
839 // Cast SPELL_BAD_PRESS_DAMAGE with scaled basepoints when the visual hits the target.
840 // Need Fix: When SPELL_BAD_PRESS_TRIGGER hits target it triggers spell SPELL_BAD_PRESS_DAMAGE by itself
841 // so player gets damage equal to calculated damage dbc basepoints for SPELL_BAD_PRESS_DAMAGE (~50)
842 if (spell->Id == SPELL_BAD_PRESS_TRIGGER)
843 {
844 int32 bp = (int32)((float)(fails)*0.33f*target->GetMaxHealth());
845 target->CastCustomSpell(target, SPELL_BAD_PRESS_DAMAGE, &bp, NULL, NULL, true);
846 }
847 }

849 // Checks if player has already die or has get too far from the current node
850 bool CheckPlayer()
851 {
852 if (Player* player = ObjectAccessor::GetPlayer(*me, playerGUID))
853 {
854 if (player->isDead())
855 return false;
856 if (player->GetDistance2d(me) >= 2.0f*searchDistance)
857 {
858 GivePunishment();
859 return false;
860 }
861 }
862 else
863 return false;
864
865 return true;
866 }

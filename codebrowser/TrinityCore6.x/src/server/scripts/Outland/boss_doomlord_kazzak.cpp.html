61class boss_doomlord_kazzak : public CreatureScript
62{
63 public:
64 boss_doomlord_kazzak() : CreatureScript("boss_doomlord_kazzak") { }
65
66 struct boss_doomlordkazzakAI : public ScriptedAI
67 {
68 boss_doomlordkazzakAI(Creature* creature) : ScriptedAI(creature)
69 {
70 }
71
72 void Reset() override
73 {
74 _events.Reset();
75 _events.ScheduleEvent(EVENT_SHADOW_VOLLEY, urand(6000, 10000));
76 _events.ScheduleEvent(EVENT_CLEAVE, 7000);
77 _events.ScheduleEvent(EVENT_THUNDERCLAP, urand(14000, 18000));
78 _events.ScheduleEvent(EVENT_VOID_BOLT, 30000);
79 _events.ScheduleEvent(EVENT_MARK_OF_KAZZAK, 25000);
80 _events.ScheduleEvent(EVENT_ENRAGE, 60000);
81 _events.ScheduleEvent(EVENT_TWISTED_REFLECTION, 33000);
82 _events.ScheduleEvent(EVENT_BERSERK, 180000);
83 }
84
85 void JustRespawned() override
86 {
87 Talk(SAY_INTRO);
88 }
89
90 void EnterCombat(Unit* /*who*/) override
91 {
92 Talk(SAY_AGGRO);
93 }
94
95 void KilledUnit(Unit* victim) override
96 {
97 // When Kazzak kills a player (not pets/totems), he regens some health
98 if (victim->GetTypeId() != TYPEID_PLAYER)
99 return;
100
101 DoCast(me, SPELL_CAPTURE_SOUL);
102
103 Talk(SAY_KILL);
104 }
105
106 void JustDied(Unit* /*killer*/) override
107 {
108 Talk(SAY_DEATH);
109 }
110
111 void UpdateAI(uint32 diff) override
112 {
113 // Return since we have no target
114 if (!UpdateVictim())
115 return;
116
117 _events.Update(diff);
118
119 if (me->HasUnitState(UNIT_STATE_CASTING))
120 return;
121
122 while (uint32 eventId = _events.ExecuteEvent())
123 {
124 switch (eventId)
125 {
126 case EVENT_SHADOW_VOLLEY:
127 DoCastVictim(SPELL_SHADOW_VOLLEY);
128 _events.ScheduleEvent(EVENT_SHADOW_VOLLEY, urand(4000, 6000));
129 break;
130 case EVENT_CLEAVE:
131 DoCastVictim(SPELL_CLEAVE);
132 _events.ScheduleEvent(EVENT_CLEAVE, urand(8000, 12000));
133 break;
134 case EVENT_THUNDERCLAP:
135 DoCastVictim(SPELL_THUNDERCLAP);
136 _events.ScheduleEvent(EVENT_THUNDERCLAP, urand(10000, 14000));
137 break;
138 case EVENT_VOID_BOLT:
139 DoCastVictim(SPELL_VOID_BOLT);
140 _events.ScheduleEvent(EVENT_VOID_BOLT, urand(15000, 18000));
141 break;
142 case EVENT_MARK_OF_KAZZAK:
143 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
144 DoCast(target, SPELL_MARK_OF_KAZZAK);
145 _events.ScheduleEvent(EVENT_MARK_OF_KAZZAK, 20000);
146 break;
147 case EVENT_ENRAGE:
148 Talk(EMOTE_FRENZY);
149 DoCast(me, SPELL_ENRAGE);
150 _events.ScheduleEvent(EVENT_ENRAGE, 30000);
151 break;
152 case EVENT_TWISTED_REFLECTION:
153 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
154 DoCast(target, SPELL_TWISTED_REFLECTION);
155 _events.ScheduleEvent(EVENT_TWISTED_REFLECTION, 15000);
156 break;
157 case EVENT_BERSERK:
158 DoCast(me, SPELL_BERSERK);
159 break;
160 default:
161 break;
162 }
163 }
164
165 DoMeleeAttackIfReady();
166 }
167
168 private:
169 EventMap _events;
170 };
171
172 CreatureAI* GetAI(Creature* creature) const override
173 {
174 return new boss_doomlordkazzakAI(creature);
175 }

178class spell_mark_of_kazzak : public SpellScriptLoader
179{
180 public:
181 spell_mark_of_kazzak() : SpellScriptLoader("spell_mark_of_kazzak") { }
182
183 class spell_mark_of_kazzak_AuraScript : public AuraScript
184 {
185 PrepareAuraScript(spell_mark_of_kazzak_AuraScript);
186
187 bool Validate(SpellInfo const* /*spell*/) override
188 {
189 if (!sSpellMgr->GetSpellInfo(SPELL_MARK_OF_KAZZAK_DAMAGE))
190 return false;
191 return true;
192 }
193
194 void CalculateAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
195 {
196 if (Unit* owner = GetUnitOwner())
197 amount = CalculatePct(owner->GetPower(POWER_MANA), 5);
198 }
199
200 void OnPeriodic(AuraEffect const* aurEff)
201 {
202 Unit* target = GetTarget();
203
204 if (target->GetPower(POWER_MANA) == 0)
205 {
206 target->CastSpell(target, SPELL_MARK_OF_KAZZAK_DAMAGE, true, NULL, aurEff);
207 // Remove aura
208 SetDuration(0);
209 }
210 }
211
212 void Register() override
213 {
214 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_mark_of_kazzak_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_PERIODIC_MANA_LEECH);
215 OnEffectPeriodic += AuraEffectPeriodicFn(spell_mark_of_kazzak_AuraScript::OnPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_MANA_LEECH);
216 }
217 };
218
219 AuraScript* GetAuraScript() const override
220 {
221 return new spell_mark_of_kazzak_AuraScript();
222 }
223};

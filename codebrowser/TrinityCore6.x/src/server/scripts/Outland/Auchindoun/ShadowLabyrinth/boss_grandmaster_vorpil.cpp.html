80class boss_grandmaster_vorpil : public CreatureScript
81{
82 public:
83 boss_grandmaster_vorpil() : CreatureScript("boss_grandmaster_vorpil") { }
84
85 struct boss_grandmaster_vorpilAI : public BossAI
86 {
87 boss_grandmaster_vorpilAI(Creature* creature) : BossAI(creature, DATA_GRANDMASTER_VORPIL)
88 {
89 Initialize();
90 _intro = false;
91 }
92
93 void Initialize()
94 {
95 _helpYell = false;
96 }
97
98 void Reset() override
99 {
100 _Reset();
101 Initialize();
102 }
103
104 void SummonPortals()
105 {
106 for (uint8 i = 0; i < 5; ++i)
107 if (Creature* portal = me->SummonCreature(NPC_VOID_PORTAL, VoidPortalCoords[i][0], VoidPortalCoords[i][1], VoidPortalCoords[i][2], 0, TEMPSUMMON_CORPSE_DESPAWN, 3000000))
108 portal->CastSpell(portal, SPELL_VOID_PORTAL_VISUAL, true);
109
110 events.ScheduleEvent(EVENT_SUMMON_TRAVELER, 5000);
111 }
112
113 void spawnVoidTraveler()
114 {
115 uint8 pos = urand(0, 4);
116 me->SummonCreature(NPC_VOID_TRAVELER, VoidPortalCoords[pos][0], VoidPortalCoords[pos][1], VoidPortalCoords[pos][2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000);
117 if (!_helpYell)
118 {
119 Talk(SAY_HELP);
120 _helpYell = true;
121 }
122 }
123
124 void KilledUnit(Unit* who) override
125 {
126 if (who->GetTypeId() == TYPEID_PLAYER)
127 Talk(SAY_SLAY);
128 }
129
130 void JustDied(Unit* /*killer*/) override
131 {
132 _JustDied();
133 Talk(SAY_DEATH);
134 }
135
136 void EnterCombat(Unit* /*who*/) override
137 {
138 _EnterCombat();
139 events.ScheduleEvent(EVENT_SHADOWBOLT_VOLLEY, urand(7000, 14000));
140 if (IsHeroic())
141 events.ScheduleEvent(EVENT_BANISH, 17000);
142 events.ScheduleEvent(EVENT_DRAW_SHADOWS, 45000);
143 events.ScheduleEvent(EVENT_SUMMON_TRAVELER, 90000);
144
145 Talk(SAY_AGGRO);
146 SummonPortals();
147 }
148
149 void MoveInLineOfSight(Unit* who) override
150 {
151 BossAI::MoveInLineOfSight(who);
152
153 if (!_intro && me->IsWithinLOSInMap(who) && me->IsWithinDistInMap(who, 100) && me->IsValidAttackTarget(who))
154 {
155 Talk(SAY_INTRO);
156 _intro = true;
157 }
158 }
159
160 void UpdateAI(uint32 diff) override
161 {
162 if (!UpdateVictim())
163 return;
164
165 events.Update(diff);
166
167 if (me->HasUnitState(UNIT_STATE_CASTING))
168 return;
169
170 while (uint32 eventId = events.ExecuteEvent())
171 {
172 switch (eventId)
173 {
174 case EVENT_SHADOWBOLT_VOLLEY:
175 DoCast(me, SPELL_SHADOWBOLT_VOLLEY);
176 events.ScheduleEvent(EVENT_SHADOWBOLT_VOLLEY, urand(15000, 30000));
177 break;
178 case EVENT_BANISH:
179 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30.0f, false))
180 DoCast(target, SPELL_BANISH);
181 events.ScheduleEvent(EVENT_BANISH, 16000);
182 break;
183 case EVENT_DRAW_SHADOWS:
184 {
185 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
186 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
187 if (Player* i_pl = i->GetSource())
188 if (i_pl->IsAlive() && !i_pl->HasAura(SPELL_BANISH))
189 i_pl->TeleportTo(me->GetMapId(), VorpilPosition.GetPositionX(), VorpilPosition.GetPositionY(), VorpilPosition.GetPositionZ(), VorpilPosition.GetOrientation(), TELE_TO_NOT_LEAVE_COMBAT);
190
191 me->SetPosition(VorpilPosition);
192 DoCast(me, SPELL_DRAW_SHADOWS, true);
193 DoCast(me, SPELL_RAIN_OF_FIRE);
194 events.ScheduleEvent(EVENT_SHADOWBOLT_VOLLEY, 6000);
195 events.ScheduleEvent(EVENT_DRAW_SHADOWS, 30000);
196 break;
197 }
198 case EVENT_SUMMON_TRAVELER:
199 spawnVoidTraveler();
200 events.ScheduleEvent(EVENT_SUMMON_TRAVELER, 10000);
201 // enrage at 20%
202 if (HealthBelowPct(20))
203 events.ScheduleEvent(EVENT_SUMMON_TRAVELER, 5000);
204 break;
205 }
206 }
207
208 DoMeleeAttackIfReady();
209 }
210
211 private:
212 bool _intro;
213 bool _helpYell;
214 };
215
216 CreatureAI* GetAI(Creature* creature) const override
217 {
218 return GetShadowLabyrinthAI<boss_grandmaster_vorpilAI>(creature);
219 }

222class npc_voidtraveler : public CreatureScript
223{
224 public:
225 npc_voidtraveler() : CreatureScript("npc_voidtraveler") { }
226
227 struct npc_voidtravelerAI : public ScriptedAI
228 {
229 npc_voidtravelerAI(Creature* creature) : ScriptedAI(creature)
230 {
231 Initialize();
232 _instance = creature->GetInstanceScript();
233 }
234
235 void Initialize()
236 {
237 _moveTimer = 0;
238 _sacrificed = false;
239 }
240
241 void Reset() override
242 {
243 Initialize();
244 }
245
246 void EnterCombat(Unit* /*who*/) override { }
247
248 void UpdateAI(uint32 diff) override
249 {
250 if (_moveTimer <= diff)
251 {
252 Creature* Vorpil = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_GRANDMASTER_VORPIL));
253 if (!Vorpil)
254 return;
255
256 if (_sacrificed)
257 {
258 DoCastAOE(SPELL_EMPOWERING_SHADOWS, true);
259 DoCast(me, SPELL_SHADOW_NOVA, true);
260 me->KillSelf();
261 return;
262 }
263 me->GetMotionMaster()->MoveFollow(Vorpil, 0, 0);
264 if (me->IsWithinDist(Vorpil, 3))
265 {
266 DoCast(me, SPELL_SACRIFICE, false);
267 _sacrificed = true;
268 _moveTimer = 500;
269 return;
270 }
271 _moveTimer = 1000;
272 }
273 else
274 _moveTimer -= diff;
275 }
276
277 private:
278 InstanceScript* _instance;
279 uint32 _moveTimer;
280 bool _sacrificed;
281 };
282
283 CreatureAI* GetAI(Creature* creature) const override
284 {
285 return GetShadowLabyrinthAI<npc_voidtravelerAI>(creature);
286 }

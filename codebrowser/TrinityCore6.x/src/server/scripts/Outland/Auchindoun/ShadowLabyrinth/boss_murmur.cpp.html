55class boss_murmur : public CreatureScript
56{
57 public:
58 boss_murmur() : CreatureScript("boss_murmur") { }
59
60 struct boss_murmurAI : public BossAI
61 {
62 boss_murmurAI(Creature* creature) : BossAI(creature, DATA_MURMUR)
63 {
64 SetCombatMovement(false);
65 }
66
67 void Reset() override
68 {
69 _Reset();
70 events.ScheduleEvent(EVENT_SONIC_BOOM, 30000);
71 events.ScheduleEvent(EVENT_MURMURS_TOUCH, urand(8000, 20000));
72 events.ScheduleEvent(EVENT_RESONANCE, 5000);
73 events.ScheduleEvent(EVENT_MAGNETIC_PULL, urand(15000, 30000));
74 if (IsHeroic())
75 {
76 events.ScheduleEvent(EVENT_THUNDERING_STORM, 15000);
77 events.ScheduleEvent(EVENT_SONIC_SHOCK, 10000);
78 }
79
80 // database should have `RegenHealth`=0 to prevent regen
81 uint32 hp = me->CountPctFromMaxHealth(40);
82 if (hp)
83 me->SetHealth(hp);
84 me->ResetPlayerDamageReq();
85 }
86
87 void EnterCombat(Unit* /*who*/) override
88 {
89 _EnterCombat();
90 }
91
92 void JustDied(Unit* /*killer*/) override
93 {
94 _JustDied();
95 }
96
97 void UpdateAI(uint32 diff) override
98 {
99 if (!UpdateVictim())
100 return;
101
102 events.Update(diff);
103
104 if (me->HasUnitState(UNIT_STATE_CASTING))
105 return;
106
107 while (uint32 eventId = events.ExecuteEvent())
108 {
109 switch (eventId)
110 {
111 case EVENT_SONIC_BOOM:
112 Talk(EMOTE_SONIC_BOOM);
113 DoCast(me, SPELL_SONIC_BOOM_CAST);
114 events.ScheduleEvent(EVENT_SONIC_BOOM, 30000);
115 events.ScheduleEvent(EVENT_RESONANCE, 1500);
116 break;
117 case EVENT_MURMURS_TOUCH:
118 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 80.0f, true))
119 DoCast(target, SPELL_MURMURS_TOUCH);
120 events.ScheduleEvent(EVENT_MURMURS_TOUCH, urand(25000, 35000));
121 break;
122 case EVENT_RESONANCE:
123 if (!(me->IsWithinMeleeRange(me->GetVictim())))
124 {
125 DoCast(me, SPELL_RESONANCE);
126 events.ScheduleEvent(EVENT_RESONANCE, 5000);
127 }
128 break;
129 case EVENT_MAGNETIC_PULL:
130 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
131 {
132 DoCast(target, SPELL_MAGNETIC_PULL);
133 events.ScheduleEvent(EVENT_MAGNETIC_PULL, urand(15000, 30000));
134 break;
135 }
136 events.ScheduleEvent(EVENT_MAGNETIC_PULL, 500);
137 break;
138 case EVENT_THUNDERING_STORM:
139 DoCastAOE(SPELL_THUNDERING_STORM, true);
140 events.ScheduleEvent(EVENT_THUNDERING_STORM, 15000);
141 break;
142 case EVENT_SONIC_SHOCK:
143 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 20.0f, false))
144 DoCast(target, SPELL_SONIC_SHOCK);
145 events.ScheduleEvent(EVENT_SONIC_SHOCK, urand(10000, 20000));
146 break;
147 }
148 }
149
150 // Select nearest most aggro target if top aggro too far
151 if (!me->isAttackReady())
152 return;
153
154 if (!me->IsWithinMeleeRange(me->GetVictim()))
155 {
156 ThreatContainer::StorageType threatlist = me->getThreatManager().getThreatList();
157 for (ThreatContainer::StorageType::const_iterator i = threatlist.begin(); i != threatlist.end(); ++i)
158 if (Unit* target = ObjectAccessor::GetUnit(*me, (*i)->getUnitGuid()))
159 if (me->IsWithinMeleeRange(target))
160 {
161 me->TauntApply(target);
162 break;
163 }
164 }
165
166 DoMeleeAttackIfReady();
167 }
168 };
169
170 CreatureAI* GetAI(Creature* creature) const override
171 {
172 return GetShadowLabyrinthAI<boss_murmurAI>(creature);
173 }

238class ThunderingStormCheck
239{
240 public:
241 ThunderingStormCheck(WorldObject* source) : _source(source) { }
242
243 bool operator()(WorldObject* obj)
244 {
245 float distSq = _source->GetExactDist2dSq(obj);
246 return distSq < (25.0f * 25.0f) || distSq > (100.0f * 100.0f);
247 }
248
249 private:
250 WorldObject const* _source;
251};

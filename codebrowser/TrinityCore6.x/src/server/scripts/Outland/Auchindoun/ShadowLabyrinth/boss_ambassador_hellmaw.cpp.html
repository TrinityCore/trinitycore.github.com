31enum Yells
32{
33 SAY_INTRO = 0,
34 SAY_AGGRO = 1,
35 SAY_HELP = 2,
36 SAY_SLAY = 3,
37 SAY_DEATH = 4
38};
39
40enum Spells
41{
42 SPELL_BANISH = 30231,
43 SPELL_CORROSIVE_ACID = 33551,
44 SPELL_FEAR = 33547,
45 SPELL_ENRAGE = 34970
46};
47
48enum Events
49{
50 EVENT_CORROSIVE_ACID = 1,
51 EVENT_FEAR,
52 EVENT_BERSERK
53};

55class boss_ambassador_hellmaw : public CreatureScript
56{
57 public:
58 boss_ambassador_hellmaw() : CreatureScript("boss_ambassador_hellmaw") { }
59
60 struct boss_ambassador_hellmawAI : public npc_escortAI
61 {
62 boss_ambassador_hellmawAI(Creature* creature) : npc_escortAI(creature)
63 {
64 _instance = creature->GetInstanceScript();
65 _intro = false;
66 }
67
68 void Reset() override
69 {
70 if (!me->IsAlive())
71 return;
72
73 _events.Reset();
74 _instance->SetBossState(DATA_AMBASSADOR_HELLMAW, NOT_STARTED);
75
76 _events.ScheduleEvent(EVENT_CORROSIVE_ACID, urand(5000, 10000));
77 _events.ScheduleEvent(EVENT_FEAR, urand(25000, 30000));
78 if (IsHeroic())
79 _events.ScheduleEvent(EVENT_BERSERK, 180000);
80
81 DoAction(ACTION_AMBASSADOR_HELLMAW_BANISH);
82 }
83
84 void MoveInLineOfSight(Unit* who) override
85 {
86 if (me->HasAura(SPELL_BANISH))
87 return;
88
89 npc_escortAI::MoveInLineOfSight(who);
90 }
91
92 void WaypointReached(uint32 /*waypointId*/) override
93 {
94 }
95
96 void DoAction(int32 actionId) override
97 {
98 if (actionId == ACTION_AMBASSADOR_HELLMAW_INTRO)
99 DoIntro();
100 else if (actionId == ACTION_AMBASSADOR_HELLMAW_BANISH)
101 {
102 if (_instance->GetData(DATA_FEL_OVERSEER) && me->HasAura(SPELL_BANISH))
103 DoCast(me, SPELL_BANISH, true); // this will not work, because he is immune to banish
104 }
105 }
106
107 void DoIntro()
108 {
109 if (_intro)
110 return;
111
112 _intro = true;
113
114 if (me->HasAura(SPELL_BANISH))
115 me->RemoveAurasDueToSpell(SPELL_BANISH);
116
117 Talk(SAY_INTRO);
118 Start(true, false, ObjectGuid::Empty, NULL, false, true);
119 }
120
121 void EnterCombat(Unit* /*who*/) override
122 {
123 _instance->SetBossState(DATA_AMBASSADOR_HELLMAW, IN_PROGRESS);
124 Talk(SAY_AGGRO);
125 }
126
127 void KilledUnit(Unit* who) override
128 {
129 if (who->GetTypeId() == TYPEID_PLAYER)
130 Talk(SAY_SLAY);
131 }
132
133 void JustDied(Unit* /*killer*/) override
134 {
135 _instance->SetBossState(DATA_AMBASSADOR_HELLMAW, DONE);
136 Talk(SAY_DEATH);
137 }
138
139 void UpdateEscortAI(uint32 diff) override
140 {
141 if (!UpdateVictim())
142 return;
143
144 if (me->HasAura(SPELL_BANISH))
145 {
146 EnterEvadeMode();
147 return;
148 }
149
150 _events.Update(diff);
151
152 if (me->HasUnitState(UNIT_STATE_CASTING))
153 return;
154
155 while (uint32 eventId = _events.ExecuteEvent())
156 {
157 switch (eventId)
158 {
159 case EVENT_CORROSIVE_ACID:
160 DoCastVictim(SPELL_CORROSIVE_ACID);
161 _events.ScheduleEvent(EVENT_CORROSIVE_ACID, urand(15000, 25000));
162 break;
163 case EVENT_FEAR:
164 DoCastAOE(SPELL_FEAR);
165 _events.ScheduleEvent(EVENT_FEAR, urand(20000, 35000));
166 break;
167 case EVENT_BERSERK:
168 DoCast(me, SPELL_ENRAGE, true);
169 break;
170 default:
171 break;
172 }
173 }
174
175 DoMeleeAttackIfReady();
176 }
177
178 private:
179 InstanceScript* _instance;
180 EventMap _events;
181 bool _intro;
182 };
183
184 CreatureAI* GetAI(Creature* creature) const override
185 {
186 return GetShadowLabyrinthAI<boss_ambassador_hellmawAI>(creature);
187 }
188};
189
190void AddSC_boss_ambassador_hellmaw()
191{
192 new boss_ambassador_hellmaw();
193}

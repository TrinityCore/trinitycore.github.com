58class boss_blackheart_the_inciter : public CreatureScript
59{
60 public:
61 boss_blackheart_the_inciter() : CreatureScript("boss_blackheart_the_inciter") { }
62
63 struct boss_blackheart_the_inciterAI : public BossAI
64 {
65 boss_blackheart_the_inciterAI(Creature* creature) : BossAI(creature, DATA_BLACKHEART_THE_INCITER) { }
66
67 void Reset() override
68 {
69 _Reset();
70 }
71
72 void EnterCombat(Unit* /*who*/) override
73 {
74 _EnterCombat();
75 events.ScheduleEvent(EVENT_INCITE_CHAOS, 20000);
76 events.ScheduleEvent(EVENT_CHARGE_ATTACK, 5000);
77 events.ScheduleEvent(EVENT_WAR_STOMP, 15000);
78
79 Talk(SAY_AGGRO);
80 }
81
82 void KilledUnit(Unit* who) override
83 {
84 if (who->GetTypeId() == TYPEID_PLAYER)
85 Talk(SAY_SLAY);
86 }
87
88 void JustDied(Unit* /*killer*/) override
89 {
90 _JustDied();
91 Talk(SAY_DEATH);
92 }
93
94 void UpdateAI(uint32 diff) override
95 {
96 if (!UpdateVictim())
97 return;
98
99 events.Update(diff);
100
101 if (me->HasUnitState(UNIT_STATE_CASTING))
102 return;
103
104 while (uint32 eventId = events.ExecuteEvent())
105 {
106 switch (eventId)
107 {
108 case EVENT_INCITE_CHAOS:
109 {
110 DoCast(me, SPELL_INCITE_CHAOS);
111
112 std::list<HostileReference*> t_list = me->getThreatManager().getThreatList();
113 for (std::list<HostileReference*>::const_iterator itr = t_list.begin(); itr!= t_list.end(); ++itr)
114 {
115 if (Unit* target = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid()))
116 if (target->GetTypeId() == TYPEID_PLAYER)
117 me->CastSpell(target, SPELL_INCITE_CHAOS_B, true);
118 }
119
120 DoResetThreat();
121 events.ScheduleEvent(EVENT_INCITE_CHAOS, 40000);
122 break;
123 }
124 case EVENT_CHARGE_ATTACK:
125 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
126 DoCast(target, SPELL_CHARGE);
127 events.ScheduleEvent(EVENT_CHARGE, urand(15000, 25000));
128 break;
129 case EVENT_WAR_STOMP:
130 DoCast(me, SPELL_WAR_STOMP);
131 events.ScheduleEvent(EVENT_WAR_STOMP, urand(18000, 24000));
132 break;
133 }
134 }
135
136 DoMeleeAttackIfReady();
137 }
138 };
139
140 CreatureAI* GetAI(Creature* creature) const override
141 {
142 return GetShadowLabyrinthAI<boss_blackheart_the_inciterAI>(creature);
143 }
144};

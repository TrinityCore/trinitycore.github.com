53class boss_talon_king_ikiss : public CreatureScript
54{
55public:
56 boss_talon_king_ikiss() : CreatureScript("boss_talon_king_ikiss") { }
57
58 struct boss_talon_king_ikissAI : public BossAI
59 {
60 boss_talon_king_ikissAI(Creature* creature) : BossAI(creature, DATA_TALON_KING_IKISS)
61 {
62 Intro = false;
63 ManaShield = false;
64 }
65
66 void Reset() override
67 {
68 _Reset();
69 Intro = false;
70 ManaShield = false;
71 }
72
73 void MoveInLineOfSight(Unit* who) override
74 {
75 if (!Intro && who->GetTypeId() == TYPEID_PLAYER && me->IsWithinDistInMap(who, 100.0f))
76 {
77 Intro = true;
78 Talk(SAY_INTRO);
79 }
80
81 BossAI::MoveInLineOfSight(who);
82 }
83
84 void EnterCombat(Unit* /*who*/) override
85 {
86 _EnterCombat();
87 Talk(SAY_AGGRO);
88 events.ScheduleEvent(EVENT_ARCANE_VOLLEY, 5000);
89 events.ScheduleEvent(EVENT_POLYMORPH, 8000);
90 events.ScheduleEvent(EVENT_BLINK, 35000);
91 if (IsHeroic())
92 events.ScheduleEvent(EVENT_SLOW, urand(15000, 30000));
93 }
94
95 void ExecuteEvent(uint32 eventId) override
96 {
97 switch (eventId)
98 {
99 case EVENT_POLYMORPH:
100 // Second top aggro in normal, random target in heroic.
101 if (IsHeroic())
102 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0), SPELL_POLYMORPH);
103 else
104 DoCast(SelectTarget(SELECT_TARGET_TOPAGGRO, 1), SPELL_POLYMORPH);
105 events.ScheduleEvent(EVENT_POLYMORPH, urand(15000, 17500));
106 break;
107 case EVENT_ARCANE_VOLLEY:
108 DoCast(me, SPELL_ARCANE_VOLLEY);
109 events.ScheduleEvent(EVENT_ARCANE_VOLLEY, urand(7000, 12000));
110 break;
111 case EVENT_SLOW:
112 DoCast(me, SPELL_SLOW);
113 events.ScheduleEvent(EVENT_SLOW, urand(15000, 40000));
114 break;
115 case EVENT_BLINK:
116 if (me->IsNonMeleeSpellCast(false))
117 me->InterruptNonMeleeSpells(false);
118 Talk(EMOTE_ARCANE_EXPLOSION);
119 DoCastAOE(SPELL_BLINK);
120 events.ScheduleEvent(EVENT_BLINK, urand(35000, 40000));
121 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, 1000);
122 break;
123 case EVENT_ARCANE_EXPLOSION:
124 DoCast(me, SPELL_ARCANE_EXPLOSION);
125 DoCast(me, SPELL_ARCANE_BUBBLE, true);
126 break;
127 default:
128 break;
129 }
130 }
131
132 void DamageTaken(Unit* /*who*/, uint32& damage) override
133 {
134 if (!ManaShield && me->HealthBelowPctDamaged(20, damage))
135 {
136 DoCast(me, SPELL_MANA_SHIELD);
137 ManaShield = true;
138 }
139 }
140
141 void JustDied(Unit* /*killer*/) override
142 {
143 _JustDied();
144 Talk(SAY_DEATH);
145 }
146
147 void KilledUnit(Unit* who) override
148 {
149 if (who->GetTypeId() == TYPEID_PLAYER)
150 Talk(SAY_SLAY);
151 }
152
153 private:
154 bool ManaShield;
155 bool Intro;
156 };
157
158 CreatureAI* GetAI(Creature* creature) const override
159 {
160 return GetSethekkHallsAI<boss_talon_king_ikissAI>(creature);
161 }
162};

164// 38194 - Blink
165class spell_talon_king_ikiss_blink : public SpellScriptLoader
166{
167 public:
168 spell_talon_king_ikiss_blink() : SpellScriptLoader("spell_talon_king_ikiss_blink") { }
169
170 class spell_talon_king_ikiss_blink_SpellScript : public SpellScript
171 {
172 PrepareSpellScript(spell_talon_king_ikiss_blink_SpellScript);
173
174 bool Validate(SpellInfo const* /*spellInfo*/) override
175 {
176 if (!sSpellMgr->GetSpellInfo(SPELL_BLINK_TELEPORT))
177 return false;
178 return true;
179 }
180
181 void FilterTargets(std::list<WorldObject*>& targets)
182 {
183 WorldObject* target = Trinity::Containers::SelectRandomContainerElement(targets);
184 targets.clear();
185 targets.push_back(target);
186 }
187
188 void HandleDummyHitTarget(SpellEffIndex effIndex)
189 {
190 PreventHitDefaultEffect(effIndex);
191 GetHitUnit()->CastSpell(GetCaster(), SPELL_BLINK_TELEPORT, true);
192 }
193
194 void Register() override
195 {
196 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_talon_king_ikiss_blink_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
197 OnEffectHitTarget += SpellEffectFn(spell_talon_king_ikiss_blink_SpellScript::HandleDummyHitTarget, EFFECT_0, SPELL_EFFECT_DUMMY);
198 }
199 };
200
201 SpellScript* GetSpellScript() const override
202 {
203 return new spell_talon_king_ikiss_blink_SpellScript();
204 }
205};

67class boss_darkweaver_syth : public CreatureScript
68{
69public:
70 boss_darkweaver_syth() : CreatureScript("boss_darkweaver_syth") { }
71
72 struct boss_darkweaver_sythAI : public BossAI
73 {
74 boss_darkweaver_sythAI(Creature* creature) : BossAI(creature, DATA_DARKWEAVER_SYTH)
75 {
76 Initialize();
77 }
78
79 void Initialize()
80 {
81 _summon90 = false;
82 _summon50 = false;
83 _summon10 = false;
84 }
85
86 void Reset() override
87 {
88 Initialize();
89 _Reset();
90 }
91
92 void EnterCombat(Unit* /*who*/) override
93 {
94 _EnterCombat();
95 events.ScheduleEvent(EVENT_FLAME_SHOCK, 2000);
96 events.ScheduleEvent(EVENT_ARCANE_SHOCK, 4000);
97 events.ScheduleEvent(EVENT_FROST_SHOCK, 6000);
98 events.ScheduleEvent(EVENT_SHADOW_SHOCK, 8000);
99 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 15000);
100
101 Talk(SAY_AGGRO);
102 }
103
104 void JustDied(Unit* /*killer*/) override
105 {
106 _JustDied();
107 Talk(SAY_DEATH);
108 }
109
110 void KilledUnit(Unit* who) override
111 {
112 if (who->GetTypeId() == TYPEID_PLAYER)
113 Talk(SAY_SLAY);
114 }
115
116 void JustSummoned(Creature* summoned) override
117 {
118 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
119 summoned->AI()->AttackStart(target);
120
121 summons.Summon(summoned);
122 }
123
124 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
125 {
126 if (me->HealthBelowPctDamaged(90, damage) && !_summon90)
127 {
128 SythSummoning();
129 _summon90 = true;
130 }
131
132 if (me->HealthBelowPctDamaged(50, damage) && !_summon50)
133 {
134 SythSummoning();
135 _summon50 = true;
136 }
137
138 if (me->HealthBelowPctDamaged(10, damage) && !_summon10)
139 {
140 SythSummoning();
141 _summon10 = true;
142 }
143 }
144
145 void SythSummoning()
146 {
147 Talk(SAY_SUMMON);
148
149 if (me->IsNonMeleeSpellCast(false))
150 me->InterruptNonMeleeSpells(false);
151
152 DoCast(me, SPELL_SUMMON_SYTH_ARCANE, true); //front
153 DoCast(me, SPELL_SUMMON_SYTH_FIRE, true); //back
154 DoCast(me, SPELL_SUMMON_SYTH_FROST, true); //left
155 DoCast(me, SPELL_SUMMON_SYTH_SHADOW, true); //right
156 }
157
158 void ExecuteEvent(uint32 eventId) override
159 {
160 switch (eventId)
161 {
162 case EVENT_FLAME_SHOCK:
163 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
164 DoCast(target, SPELL_FLAME_SHOCK);
165 events.ScheduleEvent(EVENT_FLAME_SHOCK, urand(10000, 15000));
166 break;
167 case EVENT_ARCANE_SHOCK:
168 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
169 DoCast(target, SPELL_ARCANE_SHOCK);
170 events.ScheduleEvent(EVENT_ARCANE_SHOCK, urand(10000, 15000));
171 break;
172 case EVENT_FROST_SHOCK:
173 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
174 DoCast(target, SPELL_FROST_SHOCK);
175 events.ScheduleEvent(EVENT_FROST_SHOCK, urand(10000, 15000));
176 break;
177 case EVENT_SHADOW_SHOCK:
178 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
179 DoCast(target, SPELL_SHADOW_SHOCK);
180 events.ScheduleEvent(EVENT_SHADOW_SHOCK, urand(10000, 15000));
181 break;
182 case EVENT_CHAIN_LIGHTNING:
183 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
184 DoCast(target, SPELL_CHAIN_LIGHTNING);
185 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 25000);
186 break;
187 default:
188 break;
189 }
190 }
191
192 private:
193 bool _summon90;
194 bool _summon50;
195 bool _summon10;
196 };
197
198 CreatureAI* GetAI(Creature* creature) const override
199 {
200 return GetSethekkHallsAI<boss_darkweaver_sythAI>(creature);
201 }
202};

205class npc_syth_fire : public CreatureScript
206{
207public:
208 npc_syth_fire() : CreatureScript("npc_syth_fire") { }
209
210 struct npc_syth_fireAI : public ScriptedAI
211 {
212 npc_syth_fireAI(Creature* creature) : ScriptedAI(creature)
213 {
214 Initialize();
215 }
216
217 void Initialize()
218 {
219 flameshock_timer = 2500;
220 flamebuffet_timer = 5000;
221 }
222
223 void Reset() override
224 {
225 Initialize();
226 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FIRE, true);
227 }
228
229 void UpdateAI(uint32 diff) override
230 {
231 if (!UpdateVictim())
232 return;
233
234 if (flameshock_timer <= diff)
235 {
236 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
237 DoCast(target, SPELL_FLAME_SHOCK);
238
239 flameshock_timer = 5000;
240 } else flameshock_timer -= diff;
241
242 if (flamebuffet_timer <= diff)
243 {
244 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
245 DoCast(target, SPELL_FLAME_BUFFET);
246
247 flamebuffet_timer = 5000;
248 } else flamebuffet_timer -= diff;
249
250 DoMeleeAttackIfReady();
251 }
252
253 private:
254 uint32 flameshock_timer;
255 uint32 flamebuffet_timer;
256 };
257
258 CreatureAI* GetAI(Creature* creature) const override
259 {
260 return new npc_syth_fireAI(creature);
261 }
262};

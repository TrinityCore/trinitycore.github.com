33
34class instance_sethekk_halls : public InstanceMapScript
35{
36 public:
37 instance_sethekk_halls() : InstanceMapScript(SHScriptName, 556) { }
38
39 struct instance_sethekk_halls_InstanceMapScript : public InstanceScript
40 {
41 instance_sethekk_halls_InstanceMapScript(Map* map) : InstanceScript(map)
42 {
43 SetHeaders(DataHeader);
44 SetBossNumber(EncounterCount);
45 LoadDoorData(doorData);
46 LoadObjectData(nullptr, gameObjectData);
47 }
48
49 void OnCreatureCreate(Creature* creature) override
50 {
51 if (creature->GetEntry() == NPC_ANZU)
52 {
53 if (GetBossState(DATA_ANZU) == DONE)
54 creature->DisappearAndDie();
55 else
56 SetBossState(DATA_ANZU, IN_PROGRESS);
57 }
58 }
59
60 bool SetBossState(uint32 type, EncounterState state) override
61 {
62 if (!InstanceScript::SetBossState(type, state))
63 return false;
64
65 switch (type)
66 {
67 case DATA_TALON_KING_IKISS:
68 if (state == DONE)
69 {
70 /// @workaround: GO_FLAG_INTERACT_COND remains on the gob, but it is not handled correctly in this case
71 /// gameobject should have GO_DYNFLAG_LO_ACTIVATE too, which makes gobs interactable with GO_FLAG_INTERACT_COND
72 /// so just removed GO_FLAG_INTERACT_COND
73 if (GameObject* coffer = GetGameObject(DATA_TALON_KING_COFFER))
74 coffer->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_INTERACT_COND | GO_FLAG_NOT_SELECTABLE);
75 }
76 break;
77 default:
78 break;
79 }
80 return true;
81 }
82 };
83
84 InstanceScript* GetInstanceScript(InstanceMap* map) const override
85 {
86 return new instance_sethekk_halls_InstanceMapScript(map);
87 }
88};
89
90void AddSC_instance_sethekk_halls()
91{
92 new instance_sethekk_halls();
93}

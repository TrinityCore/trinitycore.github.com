73class boss_nexusprince_shaffar : public CreatureScript
74{
75 public:
76 boss_nexusprince_shaffar() : CreatureScript("boss_nexusprince_shaffar") { }
77
78 struct boss_nexusprince_shaffarAI : public BossAI
79 {
80 boss_nexusprince_shaffarAI(Creature* creature) : BossAI(creature, DATA_NEXUSPRINCE_SHAFFAR)
81 {
82 _hasTaunted = false;
83 }
84
85 void Reset() override
86 {
87 _Reset();
88
89 float dist = 8.0f;
90 float posX, posY, posZ, angle;
91 me->GetHomePosition(posX, posY, posZ, angle);
92
93 me->SummonCreature(NPC_BEACON, posX - dist, posY - dist, posZ, angle, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 7200000);
94 me->SummonCreature(NPC_BEACON, posX - dist, posY + dist, posZ, angle, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 7200000);
95 me->SummonCreature(NPC_BEACON, posX + dist, posY, posZ, angle, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 7200000);
96 }
97
98 void MoveInLineOfSight(Unit* who) override
99 {
100 if (!_hasTaunted && who->GetTypeId() == TYPEID_PLAYER && me->IsWithinDistInMap(who, 100.0f))
101 {
102 Talk(SAY_INTRO);
103 _hasTaunted = true;
104 }
105 }
106
107 void EnterCombat(Unit* /*who*/) override
108 {
109 Talk(SAY_AGGRO);
110 _EnterCombat();
111
112 events.ScheduleEvent(EVENT_BEACON, 10000);
113 events.ScheduleEvent(EVENT_FIREBALL, 8000);
114 events.ScheduleEvent(EVENT_FROSTBOLT, 4000);
115 events.ScheduleEvent(EVENT_FROST_NOVA, 15000);
116 }
117
118 void JustSummoned(Creature* summoned) override
119 {
120 if (summoned->GetEntry() == NPC_BEACON)
121 {
122 summoned->CastSpell(summoned, SPELL_ETHEREAL_BEACON_VISUAL, false);
123
124 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
125 summoned->AI()->AttackStart(target);
126 }
127
128 summons.Summon(summoned);
129 }
130
131 void KilledUnit(Unit* victim) override
132 {
133 if (victim->GetTypeId() == TYPEID_PLAYER)
134 Talk(SAY_SLAY);
135 }
136
137 void JustDied(Unit* /*killer*/) override
138 {
139 Talk(SAY_DEAD);
140 _JustDied();
141 }
142
143 void ExecuteEvent(uint32 eventId) override
144 {
145 switch (eventId)
146 {
147 case EVENT_BLINK:
148 if (me->IsNonMeleeSpellCast(false))
149 me->InterruptNonMeleeSpells(true);
150
151 // expire movement, will prevent from running right back to victim after cast
152 // (but should MoveChase be used again at a certain time or should he not move?)
153 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
154 me->GetMotionMaster()->MovementExpired();
155
156 DoCast(me, SPELL_BLINK);
157 break;
158 case EVENT_BEACON:
159 if (!urand(0, 3))
160 Talk(SAY_SUMMON);
161
162 DoCast(me, SPELL_ETHEREAL_BEACON, true);
163 events.ScheduleEvent(EVENT_BEACON, 10000);
164 break;
165 case EVENT_FIREBALL:
166 DoCastVictim(SPELL_FROSTBOLT);
167 events.ScheduleEvent(EVENT_FIREBALL, urand(4500, 6000));
168 break;
169 case EVENT_FROSTBOLT:
170 DoCastVictim(SPELL_FROSTBOLT);
171 events.ScheduleEvent(EVENT_FROSTBOLT, urand(4500, 6000));
172 break;
173 case EVENT_FROST_NOVA:
174 DoCast(me, SPELL_FROSTNOVA);
175 events.ScheduleEvent(EVENT_FROST_NOVA, urand(17500, 25000));
176 events.ScheduleEvent(EVENT_BLINK, 1500);
177 break;
178 default:
179 break;
180 }
181 }
182
183 private:
184 bool _hasTaunted;
185 };
186
187 CreatureAI* GetAI(Creature* creature) const override
188 {
189 return GetManaTombsAI<boss_nexusprince_shaffarAI>(creature);
190 }

201 public:
202 npc_ethereal_beacon() : CreatureScript("npc_ethereal_beacon") { }
203
204 struct npc_ethereal_beaconAI : public ScriptedAI
205 {
206 npc_ethereal_beaconAI(Creature* creature) : ScriptedAI(creature) { }
207
208 void Reset() override
209 {
210 _events.Reset();
211 }
212
213 void EnterCombat(Unit* who) override
214 {
215 if (Creature* shaffar = me->FindNearestCreature(NPC_SHAFFAR, 100.0f))
216 if (!shaffar->IsInCombat())
217 shaffar->AI()->AttackStart(who);
218
219 _events.ScheduleEvent(EVENT_APPRENTICE, DUNGEON_MODE(20000, 10000));
220 _events.ScheduleEvent(EVENT_ARCANE_BOLT, 1000);
221 }
222
223 void JustSummoned(Creature* summoned) override
224 {
225 summoned->AI()->AttackStart(me->GetVictim());
226 }
227
228 void UpdateAI(uint32 diff) override
229 {
230 if (!UpdateVictim())
231 return;
232
233 _events.Update(diff);
234
235 if (me->HasUnitState(UNIT_STATE_CASTING))
236 return;
237
238 while (uint32 eventId = _events.ExecuteEvent())
239 {
240 switch (eventId)
241 {
242 case EVENT_APPRENTICE:
243 DoCast(me, SPELL_ETHEREAL_APPRENTICE, true);
244 me->DespawnOrUnsummon();
245 break;
246 case EVENT_ARCANE_BOLT:
247 DoCastVictim(SPELL_ARCANE_BOLT);
248 _events.ScheduleEvent(EVENT_ARCANE_BOLT, urand(2000, 4500));
249 break;
250 default:
251 break;
252 }
253 }
254 }
255
256 private:
257 EventMap _events;
258 };
259
260 CreatureAI* GetAI(Creature* creature) const override
261 {
262 return new npc_ethereal_beaconAI(creature);
263 }

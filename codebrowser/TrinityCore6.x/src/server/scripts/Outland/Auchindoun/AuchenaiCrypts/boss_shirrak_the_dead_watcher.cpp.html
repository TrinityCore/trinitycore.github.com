61 struct boss_shirrak_the_dead_watcherAI : public ScriptedAI
62 {
63 boss_shirrak_the_dead_watcherAI(Creature* creature) : ScriptedAI(creature)
64 {
65 Initialize();
66 }
67
68 void Initialize()
69 {
70 Inhibitmagic_Timer = 0;
71 Attractmagic_Timer = 28000;
72 Carnivorousbite_Timer = 10000;
73 FocusFire_Timer = 17000;
74 FocusedTargetGUID.Clear();
75 }
76
77 uint32 Inhibitmagic_Timer;
78 uint32 Attractmagic_Timer;
79 uint32 Carnivorousbite_Timer;
80 uint32 FocusFire_Timer;
81
82 ObjectGuid FocusedTargetGUID;
83
84 void Reset() override
85 {
86 Initialize();
87 }
88
89 void EnterCombat(Unit* /*who*/) override
90 { }
91
92 void JustSummoned(Creature* summoned) override
93 {
94 if (summoned && summoned->GetEntry() == NPC_FOCUS_FIRE)
95 {
96 summoned->CastSpell(summoned, SPELL_FOCUS_FIRE_VISUAL, false);
97 summoned->setFaction(me->getFaction());
98 summoned->SetLevel(me->getLevel());
99 summoned->AddUnitState(UNIT_STATE_ROOT);
100
101 if (Unit* pFocusedTarget = ObjectAccessor::GetUnit(*me, FocusedTargetGUID))
102 summoned->AI()->AttackStart(pFocusedTarget);
103 }
104 }
105
106 void UpdateAI(uint32 diff) override
107 {
108 //Inhibitmagic_Timer
109 if (Inhibitmagic_Timer <= diff)
110 {
111 float dist;
112 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
113 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
114 if (Player* i_pl = i->GetSource())
115 if (i_pl->IsAlive() && (dist = i_pl->GetDistance(me)) < 45)
116 {
117 i_pl->RemoveAurasDueToSpell(SPELL_INHIBITMAGIC);
118 me->AddAura(SPELL_INHIBITMAGIC, i_pl);
119 if (dist < 35)
120 me->AddAura(SPELL_INHIBITMAGIC, i_pl);
121 if (dist < 25)
122 me->AddAura(SPELL_INHIBITMAGIC, i_pl);
123 if (dist < 15)
124 me->AddAura(SPELL_INHIBITMAGIC, i_pl);
125 }
126 Inhibitmagic_Timer = 3000 + (rand32() % 1000);
127 } else Inhibitmagic_Timer -= diff;
128
129 //Return since we have no target
130 if (!UpdateVictim())
131 return;
132
133 //Attractmagic_Timer
134 if (Attractmagic_Timer <= diff)
135 {
136 DoCast(me, SPELL_ATTRACTMAGIC);
137 Attractmagic_Timer = 30000;
138 Carnivorousbite_Timer = 1500;
139 } else Attractmagic_Timer -= diff;
140
141 //Carnivorousbite_Timer
142 if (Carnivorousbite_Timer <= diff)
143 {
144 DoCast(me, SPELL_CARNIVOROUSBITE);
145 Carnivorousbite_Timer = 10000;
146 } else Carnivorousbite_Timer -= diff;
147
148 //FocusFire_Timer
149 if (FocusFire_Timer <= diff)
150 {
151 // Summon Focus Fire & Emote
152 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1);
153 if (target && target->GetTypeId() == TYPEID_PLAYER && target->IsAlive())
154 {
155 FocusedTargetGUID = target->GetGUID();
156 me->SummonCreature(NPC_FOCUS_FIRE, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 5500);
157 Talk(EMOTE_FOCUSED, target);
158 }
159 FocusFire_Timer = 15000 + (rand32() % 5000);
160 } else FocusFire_Timer -= diff;
161
162 DoMeleeAttackIfReady();
163 }
164 };
165
166};

203 {

205 if (!UpdateVictim())
206 return;
207
208 //FieryBlast_Timer
209 if (fiery2 && FieryBlast_Timer <= diff)
210 {
211 DoCast(me, SPELL_FIERY_BLAST);
212
213 if (fiery1) fiery1 = false;
214 else if (fiery2) fiery2 = false;
215
216 FieryBlast_Timer = 1000;
217 } else FieryBlast_Timer -= diff;

141 break;
142 case 16:
143 Talk(SAY_MAG_COMPLETE);
144
145 if (Player* player = GetPlayerForEscort())

147
148 SetRun();
149 break;
150 }
151 }
152
153 void JustSummoned(Creature* summoned) override
154 {
155 if (summoned->GetEntry() == NPC_MURK_BRUTE)
156 summoned->AI()->Talk(SAY_MAG_NO_ESCAPE);
157
158 if (summoned->IsTotem())
159 return;
160
161 summoned->SetWalk(false);
162 summoned->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
163 summoned->AI()->AttackStart(me);
164
165 }
166
167 void SpellHitTarget(Unit* /*target*/, const SpellInfo* spell) override
168 {
169 if (spell->Id == SPELL_CHAIN_LIGHTNING)
170 {
171 if (rand32() % 10)
172 return;
173
174 Talk(SAY_MAG_LIGHTNING);
175 }
176 }
177
178 void UpdateAI(uint32 diff) override
179 {
180 npc_escortAI::UpdateAI(diff);
181
182 if (!UpdateVictim())
183 return;
184
185 if (me->HasUnitState(UNIT_STATE_CASTING))
186 return;
187
188 if (ChainLightningTimer <= diff)
189 {
190 DoCastVictim(SPELL_CHAIN_LIGHTNING);
191 ChainLightningTimer = urand(7000, 14000);
192 }
193 else
194 ChainLightningTimer -= diff;
195
196 if (HealthBelowPct(30))
197 {
198 if (HealTimer <= diff)
199 {
200 DoCast(me, SPELL_HEALING_WAVE);
201 HealTimer = 5000;
202 }
203 else
204 HealTimer -= diff;
205 }
206
207 if (FrostShockTimer <= diff)
208 {
209 DoCastVictim(SPELL_FROST_SHOCK);
210 FrostShockTimer = urand(7500, 15000);
211 }
212 else
213 FrostShockTimer -= diff;
214
215 DoMeleeAttackIfReady();
216 }
217 };

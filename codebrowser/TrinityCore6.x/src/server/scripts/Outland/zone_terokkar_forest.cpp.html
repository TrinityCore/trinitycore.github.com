60public:
61 npc_unkor_the_ruthless() : CreatureScript("npc_unkor_the_ruthless") { }

68 struct npc_unkor_the_ruthlessAI : public ScriptedAI
69 {
70 npc_unkor_the_ruthlessAI(Creature* creature) : ScriptedAI(creature)
71 {
72 Initialize();
73 }
74
75 void Initialize()
76 {
77 CanDoQuest = false;
78 UnkorUnfriendly_Timer = 0;
79 Pulverize_Timer = 3000;
80 }
81
82 bool CanDoQuest;
83 uint32 UnkorUnfriendly_Timer;
84 uint32 Pulverize_Timer;
85
86 void Reset() override
87 {
88 Initialize();
89 me->SetStandState(UNIT_STAND_STATE_STAND);
90 me->setFaction(FACTION_HOSTILE);
91 }
92
93 void EnterCombat(Unit* /*who*/) override { }
94
95 void DoNice()
96 {
97 Talk(SAY_SUBMIT);
98 me->setFaction(FACTION_FRIENDLY);
99 me->SetStandState(UNIT_STAND_STATE_SIT);
100 me->RemoveAllAuras();
101 me->DeleteThreatList();
102 me->CombatStop(true);
103 UnkorUnfriendly_Timer = 60000;
104 }
105
106 void DamageTaken(Unit* done_by, uint32 &damage) override
107 {
108 Player* player = done_by->ToPlayer();
109
110 if (player && me->HealthBelowPctDamaged(30, damage))
111 {
112 if (Group* group = player->GetGroup())
113 {
114 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
115 {
116 Player* groupie = itr->GetSource();
117 if (groupie &&
118 groupie->GetQuestStatus(QUEST_DONTKILLTHEFATONE) == QUEST_STATUS_INCOMPLETE &&
119 groupie->GetReqKillOrCastCurrentCount(QUEST_DONTKILLTHEFATONE, 18260) == 10)
120 {
121 groupie->AreaExploredOrEventHappens(QUEST_DONTKILLTHEFATONE);
122 if (!CanDoQuest)
123 CanDoQuest = true;
124 }
125 }
126 }
127 else if (player->GetQuestStatus(QUEST_DONTKILLTHEFATONE) == QUEST_STATUS_INCOMPLETE &&
128 player->GetReqKillOrCastCurrentCount(QUEST_DONTKILLTHEFATONE, 18260) == 10)
129 {
130 player->AreaExploredOrEventHappens(QUEST_DONTKILLTHEFATONE);
131 CanDoQuest = true;
132 }
133 }
134 }
135
136 void UpdateAI(uint32 diff) override
137 {
138 if (CanDoQuest)
139 {
140 if (!UnkorUnfriendly_Timer)
141 {
142 //DoCast(me, SPELL_QUID9889); //not using spell for now
143 DoNice();
144 }
145 else
146 {
147 if (UnkorUnfriendly_Timer <= diff)
148 {
149 EnterEvadeMode();
150 return;
151 } else UnkorUnfriendly_Timer -= diff;
152 }
153 }
154
155 if (!UpdateVictim())
156 return;
157
158 if (Pulverize_Timer <= diff)
159 {
160 DoCast(me, SPELL_PULVERIZE);
161 Pulverize_Timer = 9000;
162 } else Pulverize_Timer -= diff;
163
164 DoMeleeAttackIfReady();
165 }
166 };

392 DoCastVictim(SPELL_FROSTBOLT);
393 Frostbolt_Timer = 5000;
394 } else Frostbolt_Timer -= diff;

422 struct npc_isla_starmaneAI : public npc_escortAI
423 {
424 npc_isla_starmaneAI(Creature* creature) : npc_escortAI(creature) { }
425
426 void WaypointReached(uint32 waypointId) override
427 {
428 Player* player = GetPlayerForEscort();
429 if (!player)
430 return;
431
432 switch (waypointId)
433 {
434 case 0:
435 if (GameObject* Cage = me->FindNearestGameObject(GO_CAGE, 10))
436 Cage->SetGoState(GO_STATE_ACTIVE);
437 break;
438 case 2:
439 Talk(SAY_PROGRESS_1, player);
440 break;
441 case 5:
442 Talk(SAY_PROGRESS_2, player);
443 break;
444 case 6:
445 Talk(SAY_PROGRESS_3, player);
446 break;
447 case 29:
448 Talk(SAY_PROGRESS_4, player);
449 if (player->GetTeam() == ALLIANCE)
450 player->GroupEventHappens(QUEST_EFTW_A, me);
451 else if (player->GetTeam() == HORDE)
452 player->GroupEventHappens(QUEST_EFTW_H, me);
453 me->SetInFront(player);
454 break;
455 case 30:
456 me->HandleEmoteCommand(EMOTE_ONESHOT_WAVE);
457 break;
458 case 31:
459 DoCast(me, SPELL_CAT);
460 me->SetWalk(false);
461 break;
462 }
463 }
464
465 void Reset() override
466 {
467 me->RestoreFaction();
468 }
469
470 void JustDied(Unit* /*killer*/) override
471 {
472 if (Player* player = GetPlayerForEscort())
473 {
474 if (player->GetTeam() == ALLIANCE)
475 player->FailQuest(QUEST_EFTW_A);
476 else if (player->GetTeam() == HORDE)
477 player->FailQuest(QUEST_EFTW_H);
478 }
479 }
480 };
481
482 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
483 {
484 if (quest->GetQuestId() == QUEST_EFTW_H || quest->GetQuestId() == QUEST_EFTW_A)
485 {
486 ENSURE_AI(npc_escortAI, (creature->AI()))->Start(true, false, player->GetGUID());
487 creature->setFaction(113);
488 }
489 return true;
490 }
491
492 CreatureAI* GetAI(Creature* creature) const override
493 {
494 return new npc_isla_starmaneAI(creature);
495 }

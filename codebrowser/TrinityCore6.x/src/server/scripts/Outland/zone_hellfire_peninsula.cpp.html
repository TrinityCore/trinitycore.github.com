54class npc_aeranas : public CreatureScript
55{

57 npc_aeranas() : CreatureScript("npc_aeranas") { }
58
59 struct npc_aeranasAI : public ScriptedAI
60 {
61 npc_aeranasAI(Creature* creature) : ScriptedAI(creature)
62 {
63 Initialize();
64 }
65
66 void Initialize()
67 {
68 faction_Timer = 8000;
69 envelopingWinds_Timer = 9000;
70 shock_Timer = 5000;
71 }
72
73 void Reset() override
74 {
75 Initialize();
76
77 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
78 me->setFaction(FACTION_FRIENDLY);
79
80 Talk(SAY_SUMMON);
81 }
82
83 void UpdateAI(uint32 diff) override
84 {
85 if (faction_Timer)
86 {
87 if (faction_Timer <= diff)
88 {
89 me->setFaction(FACTION_HOSTILE);
90 faction_Timer = 0;
91 } else faction_Timer -= diff;
92 }
93
94 if (!UpdateVictim())
95 return;
96
97 if (HealthBelowPct(30))
98 {
99 me->setFaction(FACTION_FRIENDLY);
100 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
101 me->RemoveAllAuras();
102 me->DeleteThreatList();
103 me->CombatStop(true);
104 Talk(SAY_FREE);
105 return;
106 }
107
108 if (shock_Timer <= diff)
109 {
110 DoCastVictim(SPELL_SHOCK);
111 shock_Timer = 10000;
112 } else shock_Timer -= diff;
113
114 if (envelopingWinds_Timer <= diff)
115 {
116 DoCastVictim(SPELL_ENVELOPING_WINDS);
117 envelopingWinds_Timer = 25000;
118 } else envelopingWinds_Timer -= diff;
119
120 DoMeleeAttackIfReady();
121 }
122
123 private:
124 uint32 faction_Timer;
125 uint32 envelopingWinds_Timer;
126 uint32 shock_Timer;
127 };
128
129 CreatureAI* GetAI(Creature* creature) const override
130 {
131 return new npc_aeranasAI(creature);
132 }
133};

799 break;

803 me->GetMotionMaster()->MovePoint(11, exorcismPos[2]);

805 events.CancelEvent(EVENT_SUMMON_SKULL);

844 wpreached = false;

857 me->SummonCreature(NPC_DARKNESS_RELEASED, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ() + 1.5f, 0, TEMPSUMMON_MANUAL_DESPAWN);

861 }
862 }

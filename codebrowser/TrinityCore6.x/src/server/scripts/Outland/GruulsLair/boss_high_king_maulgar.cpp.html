71class boss_high_king_maulgar : public CreatureScript
72{
73public:
74 boss_high_king_maulgar() : CreatureScript("boss_high_king_maulgar") { }
75
76 struct boss_high_king_maulgarAI : public ScriptedAI
77 {
78 boss_high_king_maulgarAI(Creature* creature) : ScriptedAI(creature)
79 {
80 Initialize();
81 instance = creature->GetInstanceScript();
82 }
83
84 void Initialize()
85 {
86 ArcingSmash_Timer = 10000;
87 MightyBlow_Timer = 40000;
88 Whirlwind_Timer = 30000;
89 Charging_Timer = 0;
90 Roar_Timer = 0;
91
92 Phase2 = false;
93 }
94
95 InstanceScript* instance;
96
97 uint32 ArcingSmash_Timer;
98 uint32 MightyBlow_Timer;
99 uint32 Whirlwind_Timer;
100 uint32 Charging_Timer;
101 uint32 Roar_Timer;
102
103 bool Phase2;
104
105 void Reset() override
106 {
107 Initialize();
108
109 DoCast(me, SPELL_DUAL_WIELD, false);
110
111 instance->SetBossState(DATA_MAULGAR, NOT_STARTED);
112 }
113
114 void KilledUnit(Unit* /*victim*/) override
115 {
116 Talk(SAY_SLAY);
117 }
118
119 void JustDied(Unit* /*killer*/) override
120 {
121 Talk(SAY_DEATH);
122
123 instance->SetBossState(DATA_MAULGAR, DONE);
124 }
125
126 void DoAction(int32 actionId) override
127 {
128 if (actionId == ACTION_ADD_DEATH)
129 Talk(SAY_OGRE_DEATH);
130 }
131
132 void EnterCombat(Unit* /*who*/) override
133 {
134 DoZoneInCombat();
135 instance->SetBossState(DATA_MAULGAR, IN_PROGRESS);
136 Talk(SAY_AGGRO);
137 }
138
139 void UpdateAI(uint32 diff) override
140 {
141 if (!UpdateVictim())
142 return;
143
144 //ArcingSmash_Timer
145 if (ArcingSmash_Timer <= diff)
146 {
147 DoCastVictim(SPELL_ARCING_SMASH);
148 ArcingSmash_Timer = 10000;
149 } else ArcingSmash_Timer -= diff;
150
151 //Whirlwind_Timer
152 if (Whirlwind_Timer <= diff)
153 {
154 DoCastVictim(SPELL_WHIRLWIND);
155 Whirlwind_Timer = 55000;
156 } else Whirlwind_Timer -= diff;
157
158 //MightyBlow_Timer
159 if (MightyBlow_Timer <= diff)
160 {
161 DoCastVictim(SPELL_MIGHTY_BLOW);
162 MightyBlow_Timer = 30000 + rand32() % 10000;
163 } else MightyBlow_Timer -= diff;
164
165 //Entering Phase 2
166 if (!Phase2 && HealthBelowPct(50))
167 {
168 Phase2 = true;
169 Talk(SAY_ENRAGE);
170
171 DoCast(me, SPELL_DUAL_WIELD, true);
172 me->SetVirtualItem(0, 0);
173 me->SetVirtualItem(1, 0);
174 }
175
176 if (Phase2)
177 {
178 //Charging_Timer
179 if (Charging_Timer <= diff)
180 {
181 Unit* target = NULL;
182 target = SelectTarget(SELECT_TARGET_RANDOM, 0);
183 if (target)
184 {
185 AttackStart(target);
186 DoCast(target, SPELL_BERSERKER_C);
187 }
188 Charging_Timer = 20000;
189 } else Charging_Timer -= diff;
190
191 //Intimidating Roar
192 if (Roar_Timer <= diff)
193 {
194 DoCast(me, SPELL_ROAR);
195 Roar_Timer = 40000 + (rand32() % 10000);
196 } else Roar_Timer -= diff;
197 }
198
199 DoMeleeAttackIfReady();
200 }
201 };
202
203 CreatureAI* GetAI(Creature* creature) const override
204 {
205 return GetGruulsLairAI<boss_high_king_maulgarAI>(creature);
206 }

493 struct boss_krosh_firehandAI : public ScriptedAI
494 {
495 boss_krosh_firehandAI(Creature* creature) : ScriptedAI(creature)
496 {
497 Initialize();
498 instance = creature->GetInstanceScript();
499 }
500
501 void Initialize()
502 {
503 GreaterFireball_Timer = 1000;
504 SpellShield_Timer = 5000;
505 BlastWave_Timer = 20000;
506 }
507
508 uint32 GreaterFireball_Timer;
509 uint32 SpellShield_Timer;
510 uint32 BlastWave_Timer;
511
512 InstanceScript* instance;
513
514 void Reset() override
515 {
516 Initialize();
517
518 instance->SetBossState(DATA_MAULGAR, NOT_STARTED);
519 }
520
521 void EnterCombat(Unit* /*who*/) override
522 {
523 DoZoneInCombat();
524 instance->SetBossState(DATA_MAULGAR, IN_PROGRESS);
525 }
526
527 void JustDied(Unit* /*killer*/) override
528 {
529 if (Creature* maulgar = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MAULGAR)))
530 maulgar->AI()->DoAction(ACTION_ADD_DEATH);
531
532 instance->SetBossState(DATA_MAULGAR, DONE);
533 }
534
535 void UpdateAI(uint32 diff) override
536 {
537 if (!UpdateVictim())
538 return;
539
540 //GreaterFireball_Timer
541 if (GreaterFireball_Timer < diff || me->IsWithinDist(me->GetVictim(), 30))
542 {
543 DoCastVictim(SPELL_GREATER_FIREBALL);
544 GreaterFireball_Timer = 2000;
545 } else GreaterFireball_Timer -= diff;
546
547 //SpellShield_Timer
548 if (SpellShield_Timer <= diff)
549 {
550 me->InterruptNonMeleeSpells(false);
551 DoCastVictim(SPELL_SPELLSHIELD);
552 SpellShield_Timer = 30000;
553 } else SpellShield_Timer -= diff;
554
555 //BlastWave_Timer
556 if (BlastWave_Timer <= diff)
557 {
558 Unit* target = NULL;
559 std::list<HostileReference*> t_list = me->getThreatManager().getThreatList();
560 std::vector<Unit*> target_list;
561 for (std::list<HostileReference*>::const_iterator itr = t_list.begin(); itr!= t_list.end(); ++itr)
562 {
563 target = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid());
564 //15 yard radius minimum
565 if (target && target->IsWithinDist(me, 15, false))
566 target_list.push_back(target);
567 target = NULL;
568 }
569 if (!target_list.empty())
570 target = *(target_list.begin() + rand32() % target_list.size());
571
572 me->InterruptNonMeleeSpells(false);
573 DoCast(target, SPELL_BLAST_WAVE);
574 BlastWave_Timer = 60000;
575 } else BlastWave_Timer -= diff;
576 }
577 };

582 }

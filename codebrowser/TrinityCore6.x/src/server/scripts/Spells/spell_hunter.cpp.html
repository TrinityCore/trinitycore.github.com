68// 90355 - Ancient Hysteria
69class spell_hun_ancient_hysteria : public SpellScriptLoader
70{
71 public:
72 spell_hun_ancient_hysteria() : SpellScriptLoader("spell_hun_ancient_hysteria") { }
73
74 class spell_hun_ancient_hysteria_SpellScript : public SpellScript
75 {
76 PrepareSpellScript(spell_hun_ancient_hysteria_SpellScript);
77
78 bool Validate(SpellInfo const* /*spellInfo*/) override
79 {
80 if (!sSpellMgr->GetSpellInfo(SPELL_HUNTER_INSANITY)
81 || !sSpellMgr->GetSpellInfo(SPELL_MAGE_TEMPORAL_DISPLACEMENT)
82 || !sSpellMgr->GetSpellInfo(SPELL_SHAMAN_EXHAUSTION)
83 || !sSpellMgr->GetSpellInfo(SPELL_SHAMAN_SATED))
84 return false;
85 return true;
86 }
87
88 void RemoveInvalidTargets(std::list<WorldObject*>& targets)
89 {
90 targets.remove_if(Trinity::UnitAuraCheck(true, SPELL_HUNTER_INSANITY));
91 targets.remove_if(Trinity::UnitAuraCheck(true, SPELL_MAGE_TEMPORAL_DISPLACEMENT));
92 targets.remove_if(Trinity::UnitAuraCheck(true, SPELL_SHAMAN_EXHAUSTION));
93 targets.remove_if(Trinity::UnitAuraCheck(true, SPELL_SHAMAN_SATED));
94 }
95
96 void ApplyDebuff()
97 {
98 if (Unit* target = GetHitUnit())
99 target->CastSpell(target, SPELL_HUNTER_INSANITY, true);
100 }
101
102 void Register() override
103 {
104 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_hun_ancient_hysteria_SpellScript::RemoveInvalidTargets, EFFECT_ALL, TARGET_UNIT_CASTER_AREA_RAID);
105 AfterHit += SpellHitFn(spell_hun_ancient_hysteria_SpellScript::ApplyDebuff);
106 }
107 };
108
109 SpellScript* GetSpellScript() const override
110 {
111 return new spell_hun_ancient_hysteria_SpellScript();
112 }

499
500 SpellCastResult CheckIfCorpseNear()
501 {
502 Unit* caster = GetCaster();
503 float max_range = GetSpellInfo()->GetMaxRange(false);
504 WorldObject* result = NULL;
505 // search for nearby enemy corpse in range
506 Trinity::AnyDeadUnitSpellTargetInRangeCheck check(caster, max_range, GetSpellInfo(), TARGET_CHECK_ENEMY);
507 Trinity::WorldObjectSearcher<Trinity::AnyDeadUnitSpellTargetInRangeCheck> searcher(caster, result, check);
508 caster->GetMap()->VisitFirstFound(caster->m_positionX, caster->m_positionY, max_range, searcher);
509 if (!result)
510 return SPELL_FAILED_NO_EDIBLE_CORPSES;
511 return SPELL_CAST_OK;
512 }
513
514 void HandleDummy(SpellEffIndex /*effIndex*/)
515 {
516 Unit* caster = GetCaster();
517 caster->CastSpell(caster, SPELL_HUNTER_PET_CARRION_FEEDER_TRIGGERED, false);
518 }
519
520 void Register() override
521 {
522 OnEffectHit += SpellEffectFn(spell_hun_pet_carrion_feeder_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);

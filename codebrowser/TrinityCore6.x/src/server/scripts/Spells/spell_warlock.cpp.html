134// 710 - Banish
135class spell_warl_banish : public SpellScriptLoader
136{
137 public:
138 spell_warl_banish() : SpellScriptLoader("spell_warl_banish") { }
139
140 class spell_warl_banish_SpellScript : public SpellScript
141 {
142 PrepareSpellScript(spell_warl_banish_SpellScript);
143
144 public:
145 spell_warl_banish_SpellScript()
146 {
147 _removed = false;
148 }
149
150 private:
151 void HandleBanish()
152 {
153 /// Casting Banish on a banished target will cancel the effect
154 /// Check if the target already has Banish, if so, do nothing.
155 if (Unit* target = GetHitUnit())
156 {
157 if (target->GetAuraEffect(SPELL_AURA_SCHOOL_IMMUNITY, SPELLFAMILY_WARLOCK, flag128(0, 0x08000000, 0)))
158 {
159 // No need to remove old aura since its removed due to not stack by current Banish aura
160 PreventHitDefaultEffect(EFFECT_0);
161 PreventHitDefaultEffect(EFFECT_1);
162 PreventHitDefaultEffect(EFFECT_2);
163 _removed = true;
164 }
165 }
166 }
167
168 void RemoveAura()
169 {
170 if (_removed)
171 PreventHitAura();
172 }
173
174 void Register() override
175 {
176 BeforeHit += SpellHitFn(spell_warl_banish_SpellScript::HandleBanish);
177 AfterHit += SpellHitFn(spell_warl_banish_SpellScript::RemoveAura);
178 }
179
180 bool _removed;
181 };
182
183 SpellScript* GetSpellScript() const override
184 {
185 return new spell_warl_banish_SpellScript();
186 }

312// 48018 - Demonic Circle: Summon
313/// Updated 4.3.4
314class spell_warl_demonic_circle_summon : public SpellScriptLoader
315{
316 public:
317 spell_warl_demonic_circle_summon() : SpellScriptLoader("spell_warl_demonic_circle_summon") { }
318
319 class spell_warl_demonic_circle_summon_AuraScript : public AuraScript
320 {
321 PrepareAuraScript(spell_warl_demonic_circle_summon_AuraScript);
322
323 void HandleRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes mode)
324 {
325 // If effect is removed by expire remove the summoned demonic circle too.
326 if (!(mode & AURA_EFFECT_HANDLE_REAPPLY))
327 GetTarget()->RemoveGameObject(GetId(), true);
328
329 GetTarget()->RemoveAura(SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST);
330 }
331
332 void HandleDummyTick(AuraEffect const* /*aurEff*/)
333 {
334 if (GameObject* circle = GetTarget()->GetGameObject(GetId()))
335 {
336 // Here we check if player is in demonic circle teleport range, if so add
337 // WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST; allowing him to cast the WARLOCK_DEMONIC_CIRCLE_TELEPORT.
338 // If not in range remove the WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST.
339
340 SpellInfo const* spellInfo = sSpellMgr->AssertSpellInfo(SPELL_WARLOCK_DEMONIC_CIRCLE_TELEPORT);
341
342 if (GetTarget()->IsWithinDist(circle, spellInfo->GetMaxRange(true)))
343 {
344 if (!GetTarget()->HasAura(SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST))
345 GetTarget()->CastSpell(GetTarget(), SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST, true);
346 }
347 else
348 GetTarget()->RemoveAura(SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST);
349 }
350 }
351
352 void Register() override
353 {
354 OnEffectRemove += AuraEffectApplyFn(spell_warl_demonic_circle_summon_AuraScript::HandleRemove, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY, AURA_EFFECT_HANDLE_REAL_OR_REAPPLY_MASK);
355 OnEffectPeriodic += AuraEffectPeriodicFn(spell_warl_demonic_circle_summon_AuraScript::HandleDummyTick, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
356 }
357 };
358
359 AuraScript* GetAuraScript() const override
360 {
361 return new spell_warl_demonic_circle_summon_AuraScript();
362 }

1296class spell_warl_soul_swap_exhale : public SpellScriptLoader
1297{
1298public:

1300
1301 class spell_warl_soul_swap_exhale_SpellScript : public SpellScript
1302 {
1303 PrepareSpellScript(spell_warl_soul_swap_exhale_SpellScript);
1304
1305 bool Validate(SpellInfo const* /*spellInfo*/) override
1306 {
1307 if (!sSpellMgr->GetSpellInfo(SPELL_WARLOCK_SOUL_SWAP_MOD_COST) ||
1308 !sSpellMgr->GetSpellInfo(SPELL_WARLOCK_SOUL_SWAP_OVERRIDE))
1309 return false;
1310 return true;
1311 }
1312
1313 SpellCastResult CheckCast()
1314 {
1315 Unit* currentTarget = GetExplTargetUnit();
1316 Unit* swapTarget = nullptr;
1317 if (Aura const* swapOverride = GetCaster()->GetAura(SPELL_WARLOCK_SOUL_SWAP_OVERRIDE))
1318 if (SoulSwapOverrideAuraScript* swapScript = dynamic_cast<SoulSwapOverrideAuraScript*>(swapOverride->GetScriptByName(SoulSwapOverrideScriptName)))
1319 swapTarget = swapScript->GetOriginalSwapSource();
1320
1321 // Soul Swap Exhale can't be cast on the same target than Soul Swap
1322 if (swapTarget && currentTarget && swapTarget == currentTarget)
1323 return SPELL_FAILED_BAD_TARGETS;
1324
1325 return SPELL_CAST_OK;
1326 }
1327
1328 void OnEffectHit(SpellEffIndex /*effIndex*/)
1329 {
1330 GetCaster()->CastSpell(GetCaster(), SPELL_WARLOCK_SOUL_SWAP_MOD_COST, true);
1331 bool hasGlyph = GetCaster()->HasAura(SPELL_WARLOCK_GLYPH_OF_SOUL_SWAP);
1332
1333 std::list<uint32> dotList;
1334 Unit* swapSource = nullptr;
1335 if (Aura const* swapOverride = GetCaster()->GetAura(SPELL_WARLOCK_SOUL_SWAP_OVERRIDE))
1336 {
1337 SoulSwapOverrideAuraScript* swapScript = dynamic_cast<SoulSwapOverrideAuraScript*>(swapOverride->GetScriptByName(SoulSwapOverrideScriptName));
1338 if (!swapScript)
1339 return;
1340 dotList = swapScript->GetDotList();
1341 swapSource = swapScript->GetOriginalSwapSource();
1342 }
1343
1344 if (dotList.empty())
1345 return;
1346
1347 for (std::list<uint32>::const_iterator itr = dotList.begin(); itr != dotList.end(); ++itr)
1348 {
1349 GetCaster()->AddAura(*itr, GetHitUnit());
1350 if (!hasGlyph && swapSource)
1351 swapSource->RemoveAurasDueToSpell(*itr);
1352 }
1353
1354 // Remove Soul Swap Exhale buff
1355 GetCaster()->RemoveAurasDueToSpell(SPELL_WARLOCK_SOUL_SWAP_OVERRIDE);
1356
1357 if (hasGlyph) // Add a cooldown on Soul Swap if caster has the glyph
1358 GetCaster()->CastSpell(GetCaster(), SPELL_WARLOCK_SOUL_SWAP_CD_MARKER, false);
1359 }
1360
1361 void Register() override
1362 {
1363 OnCheckCast += SpellCheckCastFn(spell_warl_soul_swap_exhale_SpellScript::CheckCast);
1364 OnEffectHitTarget += SpellEffectFn(spell_warl_soul_swap_exhale_SpellScript::OnEffectHit, EFFECT_0, SPELL_EFFECT_SCHOOL_DAMAGE);
1365 }
1366 };

86class spell_warr_bloodthirst : public SpellScriptLoader
87{
88 public:
89 spell_warr_bloodthirst() : SpellScriptLoader("spell_warr_bloodthirst") { }
90
91 class spell_warr_bloodthirst_SpellScript : public SpellScript
92 {
93 PrepareSpellScript(spell_warr_bloodthirst_SpellScript);
94
95 void HandleDamage(SpellEffIndex effIndex)
96 {
97 int32 damage = GetEffectValue();
98 ApplyPct(damage, GetCaster()->GetTotalAttackPowerValue(BASE_ATTACK));
99
100 if (Unit* target = GetHitUnit())
101 {
102 damage = GetCaster()->SpellDamageBonusDone(target, GetSpellInfo(), uint32(damage), SPELL_DIRECT_DAMAGE, GetEffectInfo(effIndex));
103 damage = target->SpellDamageBonusTaken(GetCaster(), GetSpellInfo(), uint32(damage), SPELL_DIRECT_DAMAGE, GetEffectInfo(effIndex));
104 }
105 SetHitDamage(damage);
106 }
107
108 void HandleDummy(SpellEffIndex /*effIndex*/)
109 {
110 int32 damage = GetEffectValue();
111 GetCaster()->CastCustomSpell(GetCaster(), SPELL_WARRIOR_BLOODTHIRST, &damage, NULL, NULL, true, NULL);
112 }
113
114 void Register() override
115 {
116 OnEffectHitTarget += SpellEffectFn(spell_warr_bloodthirst_SpellScript::HandleDamage, EFFECT_0, SPELL_EFFECT_SCHOOL_DAMAGE);
117 OnEffectHit += SpellEffectFn(spell_warr_bloodthirst_SpellScript::HandleDummy, EFFECT_1, SPELL_EFFECT_DUMMY);
118 }
119 };
120
121 SpellScript* GetSpellScript() const override
122 {
123 return new spell_warr_bloodthirst_SpellScript();
124 }

454// 94009 - Rend
455class spell_warr_rend : public SpellScriptLoader
456{
457 public:
458 spell_warr_rend() : SpellScriptLoader("spell_warr_rend") { }
459
460 class spell_warr_rend_AuraScript : public AuraScript
461 {
462 PrepareAuraScript(spell_warr_rend_AuraScript);
463
464 void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool& canBeRecalculated)
465 {
466 if (Unit* caster = GetCaster())
467 {
468 canBeRecalculated = false;
469
470 // $0.25 * (($MWB + $mwb) / 2 + $AP / 14 * $MWS) bonus per tick
471 float ap = caster->GetTotalAttackPowerValue(BASE_ATTACK);
472 int32 mws = caster->GetAttackTime(BASE_ATTACK);
473 float mwbMin = caster->GetWeaponDamageRange(BASE_ATTACK, MINDAMAGE);
474 float mwbMax = caster->GetWeaponDamageRange(BASE_ATTACK, MAXDAMAGE);
475 float mwb = ((mwbMin + mwbMax) / 2 + ap * mws / 14000) * 0.25f;
476 amount += int32(caster->ApplyEffectModifiers(GetSpellInfo(), aurEff->GetEffIndex(), mwb));
477 }
478 }
479
480 void Register() override
481 {
482 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_warr_rend_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE);
483 }
484 };
485
486 AuraScript* GetAuraScript() const override
487 {
488 return new spell_warr_rend_AuraScript();
489 }

972 SpellCastResult CheckElevation()
973 {
974 if (WorldLocation const* dest = GetExplTargetDest())
975 {
976 if (GetCaster()->HasUnitMovementFlag(MOVEMENTFLAG_ROOT))
977 return SPELL_FAILED_ROOTED;
978
979 if (GetCaster()->GetMap()->Instanceable())
980 {
981 float range = GetSpellInfo()->GetMaxRange(true, GetCaster()) * 1.5f;
982
983 PathGenerator generatedPath(GetCaster());
984 generatedPath.SetPathLengthLimit(range);
985
986 bool result = generatedPath.CalculatePath(dest->GetPositionX(), dest->GetPositionY(), dest->GetPositionZ(), false, true);
987 if (generatedPath.GetPathType() & PATHFIND_SHORT)
988 return SPELL_FAILED_OUT_OF_RANGE;
989 else if (!result || generatedPath.GetPathType() & PATHFIND_NOPATH)
990 {
991 result = generatedPath.CalculatePath(dest->GetPositionX(), dest->GetPositionY(), dest->GetPositionZ(), false, false);
992 if (generatedPath.GetPathType() & PATHFIND_SHORT)
993 return SPELL_FAILED_OUT_OF_RANGE;
994 else if (!result || generatedPath.GetPathType() & PATHFIND_NOPATH)
995 return SPELL_FAILED_NOPATH;
996 }
997 }
998 else if (dest->GetPositionZ() > GetCaster()->GetPositionZ() + 4.0f)
999 return SPELL_FAILED_NOPATH;
1000
1001 return SPELL_CAST_OK;
1002 }
1003
1004 return SPELL_FAILED_NO_VALID_TARGETS;
1005 }

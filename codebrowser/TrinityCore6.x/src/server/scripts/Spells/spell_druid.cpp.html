139// 2912, 5176, 78674 - Starfire, Wrath, and Starsurge
140class spell_dru_eclipse_energize : public SpellScriptLoader
141{
142 public:
143 spell_dru_eclipse_energize() : SpellScriptLoader("spell_dru_eclipse_energize") { }
144
145 class spell_dru_eclipse_energize_SpellScript : public SpellScript
146 {
147 PrepareSpellScript(spell_dru_eclipse_energize_SpellScript);
148
149 bool Load() override
150 {
151 if (GetCaster()->GetTypeId() != TYPEID_PLAYER)
152 return false;
153
154 if (GetCaster()->ToPlayer()->getClass() != CLASS_DRUID)
155 return false;
156
157 return true;
158 }
159
160 void HandleEnergize(SpellEffIndex /*effIndex*/)
161 {
162 Player* caster = GetCaster()->ToPlayer();
163
164 // No boomy, no deal.
165 if (caster->GetSpecId(caster->GetActiveTalentGroup()) != TALENT_SPEC_DRUID_BALANCE)
166 return;
167
168 switch (GetSpellInfo()->Id)
169 {
170 case SPELL_DRUID_WRATH:
171 {
172 int32 energizeAmount = -GetEffectValue(); // -13
173 // If we are set to fill the lunar side or we've just logged in with 0 power..
174 if ((!caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE_MARKER) && caster->HasAura(SPELL_DRUID_LUNAR_ECLIPSE_MARKER))
175 || caster->GetPower(POWER_ECLIPSE) == 0)
176 {
177 caster->CastCustomSpell(caster, SPELL_DRUID_ECLIPSE_GENERAL_ENERGIZE, &energizeAmount, 0, 0, true);
178 // If the energize was due to 0 power, cast the eclipse marker aura
179 if (!caster->HasAura(SPELL_DRUID_LUNAR_ECLIPSE_MARKER))
180 caster->CastSpell(caster, SPELL_DRUID_LUNAR_ECLIPSE_MARKER, true);
181 }
182 // The energizing effect brought us out of the solar eclipse, remove the aura
183 if (caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE) && caster->GetPower(POWER_ECLIPSE) <= 0)
184 caster->RemoveAurasDueToSpell(SPELL_DRUID_SOLAR_ECLIPSE);
185 break;
186 }
187 case SPELL_DRUID_STARFIRE:
188 {
189 int32 energizeAmount = GetEffectValue(); // 20
190 // If we are set to fill the solar side or we've just logged in with 0 power..
191 if ((!caster->HasAura(SPELL_DRUID_LUNAR_ECLIPSE_MARKER) && caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE_MARKER))
192 || caster->GetPower(POWER_ECLIPSE) == 0)
193 {
194 caster->CastCustomSpell(caster, SPELL_DRUID_ECLIPSE_GENERAL_ENERGIZE, &energizeAmount, 0, 0, true);
195 // If the energize was due to 0 power, cast the eclipse marker aura
196 if (!caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE_MARKER))
197 caster->CastSpell(caster, SPELL_DRUID_SOLAR_ECLIPSE_MARKER, true);
198 }
199 // The energizing effect brought us out of the lunar eclipse, remove the aura
200 if (caster->HasAura(SPELL_DRUID_LUNAR_ECLIPSE) && caster->GetPower(POWER_ECLIPSE) >= 0)
201 caster->RemoveAura(SPELL_DRUID_LUNAR_ECLIPSE);
202 break;
203 }
204 case SPELL_DRUID_STARSURGE:
205 {
206 // If we are set to fill the solar side or we've just logged in with 0 power (confirmed with sniffs)
207 if ((!caster->HasAura(SPELL_DRUID_LUNAR_ECLIPSE_MARKER) && caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE_MARKER))
208 || caster->GetPower(POWER_ECLIPSE) == 0)
209 {
210 int32 energizeAmount = GetEffectValue(); // 15
211 caster->CastCustomSpell(caster, SPELL_DRUID_STARSURGE_ENERGIZE, &energizeAmount, 0, 0, true);
212
213 // If the energize was due to 0 power, cast the eclipse marker aura
214 if (!caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE_MARKER))
215 caster->CastSpell(caster, SPELL_DRUID_SOLAR_ECLIPSE_MARKER, true);
216 }
217 else if (!caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE_MARKER) && caster->HasAura(SPELL_DRUID_LUNAR_ECLIPSE_MARKER))
218 {
219 int32 energizeAmount = -GetEffectValue(); // -15
220 caster->CastCustomSpell(caster, SPELL_DRUID_STARSURGE_ENERGIZE, &energizeAmount, 0, 0, true);
221 }
222 // The energizing effect brought us out of the lunar eclipse, remove the aura
223 if (caster->HasAura(SPELL_DRUID_LUNAR_ECLIPSE) && caster->GetPower(POWER_ECLIPSE) >= 0)
224 caster->RemoveAura(SPELL_DRUID_LUNAR_ECLIPSE);
225 // The energizing effect brought us out of the solar eclipse, remove the aura
226 else if (caster->HasAura(SPELL_DRUID_SOLAR_ECLIPSE) && caster->GetPower(POWER_ECLIPSE) <= 0)
227 caster->RemoveAura(SPELL_DRUID_SOLAR_ECLIPSE);
228 break;
229 }
230 }
231 }
232
233 void Register() override
234 {
235 OnEffectHitTarget += SpellEffectFn(spell_dru_eclipse_energize_SpellScript::HandleEnergize, EFFECT_1, SPELL_EFFECT_DUMMY);
236 }
237 };
238
239 SpellScript* GetSpellScript() const override
240 {
241 return new spell_dru_eclipse_energize_SpellScript;
242 }

306 Unit* caster = GetCaster();
307 if (Unit* unitTarget = GetHitUnit())
308 if (AuraEffect const* aurEff = unitTarget->GetAuraEffect(SPELL_AURA_PERIODIC_DAMAGE, SPELLFAMILY_DRUID, flag128(0x2, 0, 0), caster->GetGUID()))

310 Aura* aura = aurEff->GetBase();
311

315 countMax += 3000;
316
317 if (countMin < countMax)
318 {

320 aura->SetMaxDuration(countMin + 3000);
321 }
322 }
323 }
324

467// 33763 - Lifebloom
468class spell_dru_lifebloom : public SpellScriptLoader
469{
470 public:
471 spell_dru_lifebloom() : SpellScriptLoader("spell_dru_lifebloom") { }
472
473 class spell_dru_lifebloom_AuraScript : public AuraScript
474 {
475 PrepareAuraScript(spell_dru_lifebloom_AuraScript);
476
477 bool Validate(SpellInfo const* /*spell*/) override
478 {
479 if (!sSpellMgr->GetSpellInfo(SPELL_DRUID_LIFEBLOOM_FINAL_HEAL))
480 return false;
481 if (!sSpellMgr->GetSpellInfo(SPELL_DRUID_LIFEBLOOM_ENERGIZE))
482 return false;
483 return true;
484 }
485
486 void AfterRemove(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
487 {
488 // Final heal only on duration end
489 if (GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_EXPIRE)
490 return;
491
492 // final heal
493 int32 stack = GetStackAmount();
494 int32 healAmount = aurEff->GetAmount();
495 if (Unit* caster = GetCaster())
496 {
497 healAmount = caster->SpellHealingBonusDone(GetTarget(), GetSpellInfo(), healAmount, HEAL, aurEff->GetSpellEffectInfo(), stack);
498 healAmount = GetTarget()->SpellHealingBonusTaken(caster, GetSpellInfo(), healAmount, HEAL, aurEff->GetSpellEffectInfo(), stack);
499
500 GetTarget()->CastCustomSpell(GetTarget(), SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, aurEff, GetCasterGUID());
501
502 // restore mana
503 std::vector<SpellInfo::CostData> costs = GetSpellInfo()->CalcPowerCost(caster, GetSpellInfo()->GetSchoolMask());
504 auto m = std::find_if(costs.begin(), costs.end(), [](SpellInfo::CostData const& cost) { return cost.Power == POWER_MANA; });
505 if (m != costs.end())
506 {
507 int32 returnMana = m->Amount * stack / 2;
508 caster->CastCustomSpell(caster, SPELL_DRUID_LIFEBLOOM_ENERGIZE, &returnMana, NULL, NULL, true, NULL, aurEff, GetCasterGUID());
509 }
510 return;
511 }
512
513 GetTarget()->CastCustomSpell(GetTarget(), SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, aurEff, GetCasterGUID());
514 }
515
516 void HandleDispel(DispelInfo* dispelInfo)
517 {
518 if (Unit* target = GetUnitOwner())
519 {
520 if (AuraEffect const* aurEff = GetEffect(EFFECT_1))
521 {
522 // final heal
523 int32 healAmount = aurEff->GetAmount();
524 if (Unit* caster = GetCaster())
525 {
526 healAmount = caster->SpellHealingBonusDone(target, GetSpellInfo(), healAmount, HEAL, aurEff->GetSpellEffectInfo(), dispelInfo->GetRemovedCharges());
527 healAmount = target->SpellHealingBonusTaken(caster, GetSpellInfo(), healAmount, HEAL, aurEff->GetSpellEffectInfo(), dispelInfo->GetRemovedCharges());
528 target->CastCustomSpell(target, SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, NULL, GetCasterGUID());
529
530 // restore mana
531 std::vector<SpellInfo::CostData> costs = GetSpellInfo()->CalcPowerCost(caster, GetSpellInfo()->GetSchoolMask());
532 auto m = std::find_if(costs.begin(), costs.end(), [](SpellInfo::CostData const& cost) { return cost.Power == POWER_MANA; });
533 if (m != costs.end())
534 {
535 int32 returnMana = m->Amount * dispelInfo->GetRemovedCharges() / 2;
536 caster->CastCustomSpell(caster, SPELL_DRUID_LIFEBLOOM_ENERGIZE, &returnMana, NULL, NULL, true, NULL, NULL, GetCasterGUID());
537 }
538 return;
539 }
540
541 target->CastCustomSpell(target, SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, NULL, GetCasterGUID());
542 }
543 }
544 }
545
546 void Register() override
547 {
548 AfterEffectRemove += AuraEffectRemoveFn(spell_dru_lifebloom_AuraScript::AfterRemove, EFFECT_1, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
549 AfterDispel += AuraDispelFn(spell_dru_lifebloom_AuraScript::HandleDispel);
550 }
551 };

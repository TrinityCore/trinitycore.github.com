80// 48707 - Anti-Magic Shell
81/// 6.x
82class spell_dk_anti_magic_shell : public SpellScriptLoader
83{
84 public:
85 spell_dk_anti_magic_shell() : SpellScriptLoader("spell_dk_anti_magic_shell") { }
86
87 class spell_dk_anti_magic_shell_AuraScript : public AuraScript
88 {
89 PrepareAuraScript(spell_dk_anti_magic_shell_AuraScript);
90
91 public:
92 spell_dk_anti_magic_shell_AuraScript()
93 {
94 absorbPct = 0;
95 maxHealth = 0;
96 absorbedAmount = 0;
97 }
98
99 bool Validate(SpellInfo const* /*spellInfo*/) override
100 {
101 if (!sSpellMgr->GetSpellInfo(SPELL_DK_RUNIC_POWER_ENERGIZE))
102 return false;
103 if (!sSpellMgr->GetSpellInfo(SPELL_DK_GLYPH_OF_ABSORB_MAGIC))
104 return false;
105 if (!sSpellMgr->GetSpellInfo(SPELL_DK_GLYPH_OF_REGENERATIVE_MAGIC))
106 return false;
107 return true;
108 }
109
110 bool Load() override
111 {
112 absorbPct = GetSpellInfo()->GetEffect(EFFECT_0)->CalcValue(GetCaster());
113 maxHealth = GetCaster()->GetMaxHealth();
114 absorbedAmount = 0;
115 return true;
116 }
117
118 void CalculateAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
119 {
120 amount = maxHealth;
121
122 /// todo, check if AMS has basepoints for EFFECT_2. in that case, this function should be rewritten.
123 if (!GetUnitOwner()->HasAura(SPELL_DK_GLYPH_OF_ABSORB_MAGIC))
124 amount /= 2;
125 }
126
127 void Absorb(AuraEffect* /*aurEff*/, DamageInfo& dmgInfo, uint32& absorbAmount)
128 {
129 // we may only absorb a certain percentage of incoming damage.
130 absorbAmount = dmgInfo.GetDamage() * uint32(absorbPct) / 100;
131 }
132
133 void Trigger(AuraEffect* aurEff, DamageInfo& /*dmgInfo*/, uint32& absorbAmount)
134 {
135 absorbedAmount += absorbAmount;
136
137 if (!GetTarget()->HasAura(SPELL_DK_GLYPH_OF_ABSORB_MAGIC))
138 {
139 // Patch 6.0.2 (October 14, 2014): Anti-Magic Shell now restores 2 Runic Power per 1% of max health absorbed.
140 int32 bp = 2 * absorbAmount * 100 / maxHealth;
141 GetTarget()->CastCustomSpell(SPELL_DK_RUNIC_POWER_ENERGIZE, SPELLVALUE_BASE_POINT0, bp, GetTarget(), true, nullptr, aurEff);
142 }
143 }
144
145 void HandleEffectRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
146 {
147 if (Player* player = GetTarget()->ToPlayer())
148 if (AuraEffect const* glyph = player->GetAuraEffect(SPELL_DK_GLYPH_OF_REGENERATIVE_MAGIC, EFFECT_0)) // reduce cooldown of AMS if player has glyph
149 {
150 // Cannot reduce cooldown by more than 50%
151 int32 val = std::min(glyph->GetAmount(), int32(absorbedAmount) * 100 / maxHealth);
152 player->GetSpellHistory()->ModifyCooldown(GetId(), -int32(player->GetSpellHistory()->GetRemainingCooldown(GetSpellInfo()) * val / 100));
153 }
154 }
155
156 void Register() override
157 {
158 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_dk_anti_magic_shell_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);
159 OnEffectAbsorb += AuraEffectAbsorbFn(spell_dk_anti_magic_shell_AuraScript::Absorb, EFFECT_0);
160 AfterEffectAbsorb += AuraEffectAbsorbFn(spell_dk_anti_magic_shell_AuraScript::Trigger, EFFECT_0);
161 AfterEffectRemove += AuraEffectRemoveFn(spell_dk_anti_magic_shell_AuraScript::HandleEffectRemove, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB, AURA_EFFECT_HANDLE_REAL);
162 }
163
164 private:
165 int32 absorbPct;
166 int32 maxHealth;
167 uint32 absorbedAmount;
168 };
169
170 AuraScript* GetAuraScript() const override
171 {
172 return new spell_dk_anti_magic_shell_AuraScript();
173 }

266// 50842 - Blood Boil
267/// 6.x
268class spell_dk_blood_boil : public SpellScriptLoader
269{
270 public:
271 spell_dk_blood_boil() : SpellScriptLoader("spell_dk_blood_boil") { }
272
273 class spell_dk_blood_boil_SpellScript : public SpellScript
274 {
275 PrepareSpellScript(spell_dk_blood_boil_SpellScript);
276
277 public:
278 spell_dk_blood_boil_SpellScript()
279 {
280 bpDuration = 0;
281 ffDuration = 0;
282 }
283
284 bool Validate(SpellInfo const* /*spellInfo*/) override
285 {
286 if (!sSpellMgr->GetSpellInfo(SPELL_DK_SCENT_OF_BLOOD_TRIGGERED))
287 return false;
288 if (!sSpellMgr->GetSpellInfo(SPELL_DK_SCENT_OF_BLOOD))
289 return false;
290 if (!sSpellMgr->GetSpellInfo(SPELL_DK_BLOOD_PLAGUE))
291 return false;
292 if (!sSpellMgr->GetSpellInfo(SPELL_DK_FROST_FEVER))
293 return false;
294 return true;
295 }
296
297 void FilterTargets(std::list<WorldObject*>& targets)
298 {
299 if (targets.empty())
300 return;
301
302 Unit* caster = GetCaster();
303
304 if (caster->HasAura(SPELL_DK_SCENT_OF_BLOOD))
305 return;
306
307 for (auto target : targets)
308 {
309 if (bpDuration && ffDuration)
310 break;
311
312 if (Unit* unit = target->ToUnit())
313 {
314 if (Aura const* bp = unit->GetAura(SPELL_DK_BLOOD_PLAGUE, caster->GetGUID()))
315 bpDuration = bp->GetDuration();
316 if (Aura const* ff = unit->GetAura(SPELL_DK_FROST_FEVER, caster->GetGUID()))
317 ffDuration = ff->GetDuration();
318 }
319 }
320 }
321
322 void HandleDummy(SpellEffIndex /*effIndex*/)
323 {
324 Unit* caster = GetCaster();
325 Unit* target = GetHitUnit();
326
327 if (ffDuration)
328 caster->CastSpell(target, SPELL_DK_FROST_FEVER, true);
329 if (bpDuration)
330 caster->CastSpell(target, SPELL_DK_BLOOD_PLAGUE, true);
331
332 if (!caster->HasAura(SPELL_DK_SCENT_OF_BLOOD))
333 {
334 if (Aura* bp = target->GetAura(SPELL_DK_BLOOD_PLAGUE, caster->GetGUID()))
335 {
336 bp->SetDuration(bpDuration);
337 bp->SetMaxDuration(bpDuration);
338 }
339 if (Aura* ff = target->GetAura(SPELL_DK_FROST_FEVER, caster->GetGUID()))
340 {
341 ff->SetDuration(ffDuration);
342 ff->SetMaxDuration(ffDuration);
343 }
344 }
345 }
346
347 void HandleEnergize(SpellEffIndex /*effIndex*/)
348 {
349 Unit* caster = GetCaster();
350
351 if (caster->HasAura(SPELL_DK_SCENT_OF_BLOOD))
352 caster->CastSpell(caster, SPELL_DK_SCENT_OF_BLOOD_TRIGGERED, true);
353 }
354
355 void Register() override
356 {
357 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_dk_blood_boil_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_DEST_AREA_ENEMY);
358 OnEffectHitTarget += SpellEffectFn(spell_dk_blood_boil_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
359 OnEffectHitTarget += SpellEffectFn(spell_dk_blood_boil_SpellScript::HandleEnergize, EFFECT_1, SPELL_EFFECT_ENERGIZE);
360 }
361
362 private:
363 int32 bpDuration;
364 int32 ffDuration;
365 };
366
367 SpellScript* GetSpellScript() const override
368 {
369 return new spell_dk_blood_boil_SpellScript();
370 }

424// 47541 - Death Coil
425/// 6.x
426class spell_dk_death_coil : public SpellScriptLoader
427{
428 public:
429 spell_dk_death_coil() : SpellScriptLoader("spell_dk_death_coil") { }
430
431 class spell_dk_death_coil_SpellScript : public SpellScript
432 {
433 PrepareSpellScript(spell_dk_death_coil_SpellScript);
434
435 bool Validate(SpellInfo const* /*spell*/) override
436 {
437 if (!sSpellMgr->GetSpellInfo(SPELL_DK_DEATH_COIL_DAMAGE) || !sSpellMgr->GetSpellInfo(SPELL_DK_DEATH_COIL_HEAL) || !sSpellMgr->GetSpellInfo(SPELL_DK_GLYPH_OF_DEATH_COIL))
438 return false;
439 return true;
440 }
441
442 void HandleDummy(SpellEffIndex /*effIndex*/)
443 {
444 Unit* caster = GetCaster();
445 if (Unit* target = GetHitUnit())
446 {
447 if (caster->IsFriendlyTo(target))
448 {
449 if (target->GetCreatureType() == CREATURE_TYPE_UNDEAD) // Any undead ally, including caster if he has lichborne.
450 {
451 caster->CastSpell(target, SPELL_DK_DEATH_COIL_HEAL, true);
452 }
453 else if (target != caster) // Any non undead ally except caster and only if caster has glyph of death coil.
454 {
455 SpellInfo const* DCD = sSpellMgr->AssertSpellInfo(SPELL_DK_DEATH_COIL_DAMAGE);
456 SpellEffectInfo const* eff = DCD->GetEffect(EFFECT_0);
457 int32 bp = caster->SpellDamageBonusDone(target, DCD, eff->CalcValue(caster), SPELL_DIRECT_DAMAGE, eff);
458
459 caster->CastCustomSpell(target, SPELL_DK_DEATH_COIL_BARRIER, &bp, nullptr, nullptr, true);
460 }
461 }
462 else // Any enemy target.
463 {
464 caster->CastSpell(target, SPELL_DK_DEATH_COIL_DAMAGE, true);
465 }
466 }
467
468 if (caster->HasAura(SPELL_DK_ENHANCED_DEATH_COIL))
469 caster->CastSpell(caster, SPELL_DK_SHADOW_OF_DEATH, true);
470 }
471
472 SpellCastResult CheckCast()
473 {
474 Unit* caster = GetCaster();
475 if (Unit* target = GetExplTargetUnit())
476 {
477 if (!caster->IsFriendlyTo(target) && !caster->isInFront(target))
478 return SPELL_FAILED_UNIT_NOT_INFRONT;
479
480 if (target->IsFriendlyTo(caster) && target->GetCreatureType() != CREATURE_TYPE_UNDEAD && !caster->HasAura(SPELL_DK_GLYPH_OF_DEATH_COIL))
481 return SPELL_FAILED_BAD_TARGETS;
482 }
483 else
484 return SPELL_FAILED_BAD_TARGETS;
485
486 return SPELL_CAST_OK;
487 }
488
489 void Register() override
490 {
491 OnCheckCast += SpellCheckCastFn(spell_dk_death_coil_SpellScript::CheckCast);
492 OnEffectHitTarget += SpellEffectFn(spell_dk_death_coil_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
493 }
494 };
495
496 SpellScript* GetSpellScript() const override
497 {
498 return new spell_dk_death_coil_SpellScript();
499 }

217{
218 public:
219 spell_sha_chain_heal() : SpellScriptLoader("spell_sha_chain_heal") { }
220
221 class spell_sha_chain_heal_SpellScript : public SpellScript
222 {
223 PrepareSpellScript(spell_sha_chain_heal_SpellScript);
224
225 public:
226 spell_sha_chain_heal_SpellScript()
227 {
228 firstHeal = true;
229 riptide = false;
230 amount = 0;
231 }
232
233 private:
234 void HandleHeal(SpellEffIndex /*effIndex*/)
235 {
236 if (firstHeal)
237 {
238 // Check if the target has Riptide
239 if (AuraEffect* aurEff = GetHitUnit()->GetAuraEffect(SPELL_AURA_PERIODIC_HEAL, SPELLFAMILY_SHAMAN, flag128(0, 0, 0x10), GetCaster()->GetGUID()))
240 {
241 riptide = true;
242 amount = aurEff->GetSpellInfo()->GetEffect(DIFFICULTY_NONE, EFFECT_2)->CalcValue();
243 // Consume it
244 GetHitUnit()->RemoveAura(aurEff->GetBase());
245 }
246 firstHeal = false;
247 }
248 // Riptide increases the Chain Heal effect by 25%
249 if (riptide)
250 {
251 uint32 bonus = CalculatePct(GetHitHeal(), amount);
252 SetHitHeal(GetHitHeal() + bonus);
253 }
254 }
255
256 void Register() override
257 {
258 OnEffectHitTarget += SpellEffectFn(spell_sha_chain_heal_SpellScript::HandleHeal, EFFECT_0, SPELL_EFFECT_HEAL);
259 }
260
261 bool firstHeal;
262 bool riptide;
263 uint32 amount;
264 };
265
266 SpellScript* GetSpellScript() const override
267 {
268 return new spell_sha_chain_heal_SpellScript();
269 }

272// 974 - Earth Shield
273class spell_sha_earth_shield : public SpellScriptLoader
274{
275 public:
276 spell_sha_earth_shield() : SpellScriptLoader("spell_sha_earth_shield") { }
277
278 class spell_sha_earth_shield_AuraScript : public AuraScript
279 {
280 PrepareAuraScript(spell_sha_earth_shield_AuraScript);
281
282 bool Validate(SpellInfo const* /*spellInfo*/) override
283 {
284 if (!sSpellMgr->GetSpellInfo(SPELL_SHAMAN_EARTH_SHIELD_HEAL) ||
285 !sSpellMgr->GetSpellInfo(SPELL_SHAMAN_GLYPH_OF_EARTH_SHIELD))
286 return false;
287 return true;
288 }
289
290 void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool & /*canBeRecalculated*/)
291 {
292 if (Unit* caster = GetCaster())
293 {
294 amount = caster->SpellHealingBonusDone(GetUnitOwner(), GetSpellInfo(), amount, HEAL, aurEff->GetSpellEffectInfo());
295 amount = GetUnitOwner()->SpellHealingBonusTaken(caster, GetSpellInfo(), amount, HEAL, aurEff->GetSpellEffectInfo());
296
297 //! WORKAROUND
298 // If target is affected by healing reduction, modifier is guaranteed to be negative
299 // value (e.g. -50). To revert the effect, multiply amount with reciprocal of relative value:
300 // (100 / ((-1) * modifier)) * 100 = (-1) * 100 * 100 / modifier = -10000 / modifier
301 if (int32 modifier = GetUnitOwner()->GetMaxNegativeAuraModifier(SPELL_AURA_MOD_HEALING_PCT))
302 ApplyPct(amount, -10000.0f / float(modifier));
303
304 // Glyph of Earth Shield
305 //! WORKAROUND
306 //! this glyph is a proc
307 if (AuraEffect* glyph = caster->GetAuraEffect(SPELL_SHAMAN_GLYPH_OF_EARTH_SHIELD, EFFECT_0))
308 AddPct(amount, glyph->GetAmount());
309 }
310 }
311
312 bool CheckProc(ProcEventInfo& /*eventInfo*/)
313 {
314 //! HACK due to currenct proc system implementation
315 if (Player* player = GetTarget()->ToPlayer())
316 if (player->GetSpellHistory()->HasCooldown(SPELL_SHAMAN_EARTH_SHIELD_HEAL))
317 return false;
318 return true;
319 }
320
321 void HandleProc(AuraEffect const* aurEff, ProcEventInfo& /*eventInfo*/)
322 {
323 PreventDefaultAction();
324
325 GetTarget()->CastCustomSpell(SPELL_SHAMAN_EARTH_SHIELD_HEAL, SPELLVALUE_BASE_POINT0, aurEff->GetAmount(), GetTarget(), true, NULL, aurEff, GetCasterGUID());
326
327 /// @hack: due to currenct proc system implementation
328 if (Player* player = GetTarget()->ToPlayer())
329 player->GetSpellHistory()->AddCooldown(SPELL_SHAMAN_EARTH_SHIELD_HEAL, 0, std::chrono::seconds(3));
330 }
331
332 void Register() override
333 {
334 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_sha_earth_shield_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_DUMMY);
335 DoCheckProc += AuraCheckProcFn(spell_sha_earth_shield_AuraScript::CheckProc);
336 OnEffectProc += AuraEffectProcFn(spell_sha_earth_shield_AuraScript::HandleProc, EFFECT_0, SPELL_AURA_DUMMY);
337 }
338 };
339
340 AuraScript* GetAuraScript() const override
341 {
342 return new spell_sha_earth_shield_AuraScript();
343 }

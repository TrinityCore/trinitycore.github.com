28enum MageSpells
29{
30 SPELL_MAGE_CLONE_ME = 45204,
31 SPELL_MAGE_MASTERS_THREAT_LIST = 58838

33
34class npc_pet_mage_mirror_image : public CreatureScript
35{
36 public:
37 npc_pet_mage_mirror_image() : CreatureScript("npc_pet_mage_mirror_image") { }
38
39 struct npc_pet_mage_mirror_imageAI : CasterAI
40 {
41 npc_pet_mage_mirror_imageAI(Creature* creature) : CasterAI(creature) { }
42
43 void InitializeAI() override
44 {
45 CasterAI::InitializeAI();
46 Unit* owner = me->GetOwner();
47 if (!owner)
48 return;
49 // Inherit Master's Threat List (not yet implemented)
50 owner->CastSpell((Unit*)NULL, SPELL_MAGE_MASTERS_THREAT_LIST, true);
51 // here mirror image casts on summoner spell (not present in client dbc) 49866
52 // here should be auras (not present in client dbc): 35657, 35658, 35659, 35660 selfcast by mirror images (stats related?)
53 // Clone Me!
54 owner->CastSpell(me, SPELL_MAGE_CLONE_ME, false);
55 }
56
57 // Do not reload Creature templates on evade mode enter - prevent visual lost
58 void EnterEvadeMode() override
59 {
60 if (me->IsInEvadeMode() || !me->IsAlive())
61 return;
62
63 Unit* owner = me->GetCharmerOrOwner();
64
65 me->CombatStop(true);
66 if (owner && !me->HasUnitState(UNIT_STATE_FOLLOW))
67 {
68 me->GetMotionMaster()->Clear(false);
69 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, me->GetFollowAngle(), MOTION_SLOT_ACTIVE);
70 }
71 }
72 };
73
74 CreatureAI* GetAI(Creature* creature) const override
75 {
76 return new npc_pet_mage_mirror_imageAI(creature);
77 }
78};
79
80void AddSC_mage_pet_scripts()
81{
82 new npc_pet_mage_mirror_image();
83}

70class boss_saviana_ragefire : public CreatureScript
71{
72 public:
73 boss_saviana_ragefire() : CreatureScript("boss_saviana_ragefire") { }
74
75 struct boss_saviana_ragefireAI : public BossAI
76 {
77 boss_saviana_ragefireAI(Creature* creature) : BossAI(creature, DATA_SAVIANA_RAGEFIRE)
78 {
79 }
80
81 void Reset() override
82 {
83 _Reset();
84 me->SetReactState(REACT_AGGRESSIVE);
85 }
86
87 void EnterCombat(Unit* /*who*/) override
88 {
89 _EnterCombat();
90 Talk(SAY_AGGRO);
91 events.Reset();
92 events.ScheduleEvent(EVENT_ENRAGE, 20000, EVENT_GROUP_LAND_PHASE);
93 events.ScheduleEvent(EVENT_FLAME_BREATH, 14000, EVENT_GROUP_LAND_PHASE);
94 events.ScheduleEvent(EVENT_FLIGHT, 60000);
95 }
96
97 void JustDied(Unit* /*killer*/) override
98 {
99 _JustDied();
100 me->PlayDirectSound(SOUND_ID_DEATH);
101 }
102
103 void MovementInform(uint32 type, uint32 point) override
104 {
105 if (type != POINT_MOTION_TYPE && type != EFFECT_MOTION_TYPE)
106 return;
107
108 switch (point)
109 {
110 case POINT_FLIGHT:
111 events.ScheduleEvent(EVENT_CONFLAGRATION, 1000);
112 Talk(SAY_CONFLAGRATION);
113 break;
114 case POINT_LAND:
115 events.ScheduleEvent(EVENT_LAND_GROUND, 1);
116 break;
117 case POINT_LAND_GROUND:
118 me->SetCanFly(false);
119 me->SetDisableGravity(false);
120 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
121 me->SetReactState(REACT_AGGRESSIVE);
122 DoStartMovement(me->GetVictim());
123 break;
124 case POINT_TAKEOFF:
125 events.ScheduleEvent(EVENT_AIR_MOVEMENT, 1);
126 break;
127 default:
128 break;
129 }
130 }
131
132 void JustReachedHome() override
133 {
134 _JustReachedHome();
135 me->SetCanFly(false);
136 me->SetDisableGravity(false);
137 }
138
139 void KilledUnit(Unit* victim) override
140 {
141 if (victim->GetTypeId() == TYPEID_PLAYER)
142 Talk(SAY_KILL);
143 }
144
145 void UpdateAI(uint32 diff) override
146 {
147 if (!UpdateVictim())
148 return;
149
150 events.Update(diff);
151
152 if (me->HasUnitState(UNIT_STATE_CASTING))
153 return;
154
155 while (uint32 eventId = events.ExecuteEvent())
156 {
157 switch (eventId)
158 {
159 case EVENT_FLIGHT:
160 {
161 me->SetCanFly(true);
162 me->SetDisableGravity(true);
163 me->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
164 me->SetReactState(REACT_PASSIVE);
165 me->AttackStop();
166 Position pos;
167 pos.Relocate(me);
168 pos.m_positionZ += 10.0f;
169 me->GetMotionMaster()->MoveTakeoff(POINT_TAKEOFF, pos);
170 events.ScheduleEvent(EVENT_FLIGHT, 50000);
171 events.DelayEvents(12500, EVENT_GROUP_LAND_PHASE);
172 break;
173 }
174 case EVENT_CONFLAGRATION:
175 DoCast(me, SPELL_CONFLAGRATION, true);
176 break;
177 case EVENT_ENRAGE:
178 DoCast(me, SPELL_ENRAGE);
179 Talk(EMOTE_ENRAGED);
180 events.ScheduleEvent(EVENT_ENRAGE, urand(15000, 20000), EVENT_GROUP_LAND_PHASE);
181 break;
182 case EVENT_FLAME_BREATH:
183 DoCastVictim(SPELL_FLAME_BREATH);
184 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(20000, 30000), EVENT_GROUP_LAND_PHASE);
185 break;
186 case EVENT_AIR_MOVEMENT:
187 me->GetMotionMaster()->MovePoint(POINT_FLIGHT, SavianaRagefireFlyOutPos);
188 break;
189 case EVENT_LAND_GROUND:
190 me->GetMotionMaster()->MoveLand(POINT_LAND_GROUND, SavianaRagefireLandPos);
191 break;
192 default:
193 break;
194 }
195 }
196
197 DoMeleeAttackIfReady();
198 }
199 };
200
201 CreatureAI* GetAI(Creature* creature) const override
202 {
203 return GetRubySanctumAI<boss_saviana_ragefireAI>(creature);
204 }

218class spell_saviana_conflagration_init : public SpellScriptLoader
219{
220 public:
221 spell_saviana_conflagration_init() : SpellScriptLoader("spell_saviana_conflagration_init") { }
222
223 class spell_saviana_conflagration_init_SpellScript : public SpellScript
224 {
225 PrepareSpellScript(spell_saviana_conflagration_init_SpellScript);
226
227 void FilterTargets(std::list<WorldObject*>& targets)
228 {
229 targets.remove_if(ConflagrationTargetSelector());
230 uint8 maxSize = uint8(GetCaster()->GetMap()->GetSpawnMode() & 1 ? 6 : 3);
231 if (targets.size() > maxSize)
232 Trinity::Containers::RandomResizeList(targets, maxSize);
233 }
234
235 void HandleDummy(SpellEffIndex effIndex)
236 {
237 PreventHitDefaultEffect(effIndex);
238 GetCaster()->CastSpell(GetHitUnit(), SPELL_FLAME_BEACON, true);
239 GetCaster()->CastSpell(GetHitUnit(), SPELL_CONFLAGRATION_2, false);
240 }
241
242 void Register() override
243 {
244 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_saviana_conflagration_init_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
245 OnEffectHitTarget += SpellEffectFn(spell_saviana_conflagration_init_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
246 }
247 };
248
249 SpellScript* GetSpellScript() const override
250 {
251 return new spell_saviana_conflagration_init_SpellScript();
252 }

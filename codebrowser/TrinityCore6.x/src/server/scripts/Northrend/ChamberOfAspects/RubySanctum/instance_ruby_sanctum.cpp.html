46
47 void OnPlayerEnter(Player* /*player*/) override
48 {
49 if (!GetGuidData(DATA_HALION_CONTROLLER) && GetBossState(DATA_HALION) != DONE && GetBossState(DATA_GENERAL_ZARITHRIAN) == DONE)
50 {
51 instance->LoadGrid(HalionControllerSpawnPos.GetPositionX(), HalionControllerSpawnPos.GetPositionY());
52 if (Creature* halionController = instance->SummonCreature(NPC_HALION_CONTROLLER, HalionControllerSpawnPos))
53 halionController->AI()->DoAction(ACTION_INTRO_HALION);
54 }
55 }
56
57 void OnCreatureCreate(Creature* creature) override
58 {
59 switch (creature->GetEntry())
60 {
61 case NPC_BALTHARUS_THE_WARBORN:
62 BaltharusTheWarbornGUID = creature->GetGUID();
63 break;
64 case NPC_GENERAL_ZARITHRIAN:
65 GeneralZarithrianGUID = creature->GetGUID();
66 break;
67 case NPC_SAVIANA_RAGEFIRE:
68 SavianaRagefireGUID = creature->GetGUID();
69 break;
70 case NPC_HALION:
71 HalionGUID = creature->GetGUID();
72 break;
73 case NPC_TWILIGHT_HALION:
74 TwilightHalionGUID = creature->GetGUID();
75 break;
76 case NPC_HALION_CONTROLLER:
77 HalionControllerGUID = creature->GetGUID();
78 break;
79 case NPC_ORB_CARRIER:
80 OrbCarrierGUID = creature->GetGUID();
81 break;
82 case NPC_ORB_ROTATION_FOCUS:
83 OrbRotationFocusGUID = creature->GetGUID();
84 break;
85 case NPC_BALTHARUS_TARGET:
86 CrystalChannelTargetGUID = creature->GetGUID();
87 break;
88 case NPC_XERESTRASZA:
89 XerestraszaGUID = creature->GetGUID();
90 break;
91 case NPC_ZARITHRIAN_SPAWN_STALKER:
92 if (!ZarithrianSpawnStalkerGUID[0])
93 ZarithrianSpawnStalkerGUID[0] = creature->GetGUID();
94 else
95 ZarithrianSpawnStalkerGUID[1] = creature->GetGUID();
96 break;
97 default:
98 break;
99 }
100 }

102 void OnGameObjectCreate(GameObject* go) override
103 {
104 switch (go->GetEntry())
105 {
106 case GO_FIRE_FIELD:
107 AddDoor(go, true);
108 break;
109 case GO_FLAME_WALLS:
110 FlameWallsGUID = go->GetGUID();
111 if (GetBossState(DATA_SAVIANA_RAGEFIRE) == DONE && GetBossState(DATA_BALTHARUS_THE_WARBORN) == DONE)
112 HandleGameObject(FlameWallsGUID, true, go);
113 break;
114 case GO_FLAME_RING:
115 FlameRingGUID = go->GetGUID();
116 break;
117 case GO_TWILIGHT_FLAME_RING:
118 TwilightFlameRingGUID = go->GetGUID();
119 break;
120 case GO_BURNING_TREE_1:
121 BurningTreeGUID[0] = go->GetGUID();
122 if (GetBossState(DATA_GENERAL_ZARITHRIAN) == DONE)
123 HandleGameObject(BurningTreeGUID[0], true);
124 break;
125 case GO_BURNING_TREE_2:
126 BurningTreeGUID[1] = go->GetGUID();
127 if (GetBossState(DATA_GENERAL_ZARITHRIAN) == DONE)
128 HandleGameObject(BurningTreeGUID[1], true);
129 break;
130 case GO_BURNING_TREE_3:
131 BurningTreeGUID[2] = go->GetGUID();
132 if (GetBossState(DATA_GENERAL_ZARITHRIAN) == DONE)
133 HandleGameObject(BurningTreeGUID[2], true);
134 break;
135 case GO_BURNING_TREE_4:
136 BurningTreeGUID[3] = go->GetGUID();
137 if (GetBossState(DATA_GENERAL_ZARITHRIAN) == DONE)
138 HandleGameObject(BurningTreeGUID[3], true);
139 break;
140 default:
141 break;
142 }
143 }

157 void OnUnitDeath(Unit* unit) override
158 {
159 Creature* creature = unit->ToCreature();
160 if (!creature)
161 return;
162
163 if (creature->GetEntry() == NPC_GENERAL_ZARITHRIAN && GetBossState(DATA_HALION) != DONE)
164 {
165 instance->LoadGrid(HalionControllerSpawnPos.GetPositionX(), HalionControllerSpawnPos.GetPositionY());
166 if (Creature* halionController = instance->SummonCreature(NPC_HALION_CONTROLLER, HalionControllerSpawnPos))
167 halionController->AI()->DoAction(ACTION_INTRO_HALION);
168 }
169 }

175 case DATA_BALTHARUS_THE_WARBORN:

179 case DATA_XERESTRASZA:

194 case DATA_ORB_ROTATION_FOCUS:
195 return OrbRotationFocusGUID;
196 case DATA_HALION_CONTROLLER:
197 return HalionControllerGUID;
198 case DATA_BURNING_TREE_1:
199 case DATA_BURNING_TREE_2:
200 case DATA_BURNING_TREE_3:
201 case DATA_BURNING_TREE_4:

214 bool SetBossState(uint32 type, EncounterState state) override
215 {
216 if (!InstanceScript::SetBossState(type, state))
217 return false;
218
219 switch (type)
220 {
221 case DATA_BALTHARUS_THE_WARBORN:
222 {
223 if (state == DONE && GetBossState(DATA_SAVIANA_RAGEFIRE) == DONE)
224 {
225 HandleGameObject(FlameWallsGUID, true);
226 if (Creature* zarithrian = instance->GetCreature(GeneralZarithrianGUID))
227 zarithrian->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
228 }
229 break;
230 }
231 case DATA_SAVIANA_RAGEFIRE:
232 {
233 if (state == DONE && GetBossState(DATA_BALTHARUS_THE_WARBORN) == DONE)
234 {
235 HandleGameObject(FlameWallsGUID, true);
236 if (Creature* zarithrian = instance->GetCreature(GeneralZarithrianGUID))
237 zarithrian->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
238 }
239 break;
240 }
241 case DATA_GENERAL_ZARITHRIAN:
242 {
243 if (GetBossState(DATA_SAVIANA_RAGEFIRE) == DONE && GetBossState(DATA_BALTHARUS_THE_WARBORN) == DONE)
244 HandleGameObject(FlameWallsGUID, state != IN_PROGRESS);
245 break;
246 }
247 case DATA_HALION:
248 {
249 DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TOGGLE, 0);
250 DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TWILIGHT, 0);
251 DoUpdateWorldState(WORLDSTATE_CORPOREALITY_MATERIAL, 0);
252
253 // Reopen rings on wipe or success
254 if (state == DONE || state == FAIL)
255 {
256 HandleGameObject(FlameRingGUID, true);
257 HandleGameObject(TwilightFlameRingGUID, true);
258 }
259 break;
260 }
261 default:
262 break;
263 }
264
265 return true;
266 }

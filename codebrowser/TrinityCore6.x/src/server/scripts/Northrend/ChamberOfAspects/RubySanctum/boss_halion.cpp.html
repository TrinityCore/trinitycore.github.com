214struct generic_halionAI : public BossAI
215{
216 generic_halionAI(Creature* creature, uint32 bossId) : BossAI(creature, bossId), _canEvade(false) { }
217
218 void EnterCombat(Unit* /*who*/) override
219 {
220 _EnterCombat();
221 me->AddAura(SPELL_TWILIGHT_PRECISION, me);
222 _canEvade = false;
223 events.ScheduleEvent(EVENT_CLEAVE, urand(8000, 10000));
224 events.ScheduleEvent(EVENT_TAIL_LASH, 10000);
225 events.ScheduleEvent(EVENT_BREATH, urand(10000, 15000));
226 }
227
228 void Reset() override
229 {
230 _canEvade = false;
231 _Reset();
232 }
233
234 void JustReachedHome() override
235 {
236 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
237 _JustReachedHome();
238 }
239
240 void ExecuteEvent(uint32 eventId) override
241 {
242 switch (eventId)
243 {
244 case EVENT_CLEAVE:
245 DoCastVictim(SPELL_CLEAVE);
246 events.ScheduleEvent(EVENT_CLEAVE, urand(8000, 10000));
247 break;
248 case EVENT_TAIL_LASH:
249 DoCastAOE(SPELL_TAIL_LASH);
250 events.ScheduleEvent(EVENT_TAIL_LASH, 10000);
251 break;
252 case EVENT_BREATH:
253 DoCast(me, me->GetEntry() == NPC_HALION ? SPELL_FLAME_BREATH : SPELL_DARK_BREATH);
254 events.ScheduleEvent(EVENT_BREATH, urand(10000, 12000));
255 break;
256 }
257 }
258
259 bool CheckInRoom() override
260 {
261 // Rough radius, it is not an exactly perfect circle
262 if (me->GetDistance2d(HalionControllerSpawnPos.GetPositionX(), HalionControllerSpawnPos.GetPositionY()) > 48.5f)
263 {
264 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HALION_CONTROLLER)))
265 controller->AI()->EnterEvadeMode();
266 return false;
267 }
268 return true;
269 }
270
271 void UpdateAI(uint32 diff) override
272 {
273 if (!UpdateVictim() || !CheckInRoom() || me->HasUnitState(UNIT_STATE_CASTING))
274 return;
275
276 events.Update(diff);
277
278 while (uint32 eventId = events.ExecuteEvent())
279 ExecuteEvent(eventId);
280
281 DoMeleeAttackIfReady();
282 }
283
284 void SetData(uint32 index, uint32 dataValue) override
285 {
286 switch (index)
287 {
288 case DATA_EVADE_METHOD:
289 _canEvade = (dataValue == 1);
290 break;
291 default:
292 break;
293 }
294 }
295
296 void SpellHit(Unit* /*who*/, SpellInfo const* spellInfo) override
297 {
298 if (spellInfo->Id == SPELL_TWILIGHT_MENDING)
299 Talk(SAY_REGENERATE);
300 }
301
302protected:
303 bool _canEvade;
304};

306class boss_halion : public CreatureScript
307{
308 public:
309 boss_halion() : CreatureScript("boss_halion") { }
310
311 struct boss_halionAI : public generic_halionAI
312 {
313 boss_halionAI(Creature* creature) : generic_halionAI(creature, DATA_HALION)
314 {
315 me->SetHomePosition(HalionSpawnPos);
316 }
317
318 void Reset() override
319 {
320 generic_halionAI::Reset();
321 me->SetReactState(REACT_DEFENSIVE);
322 me->RemoveAurasDueToSpell(SPELL_TWILIGHT_PHASING);
323 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
324 }
325
326 void EnterEvadeMode() override
327 {
328 // Phase 1: We always can evade. Phase 2 & 3: We can evade if and only if the controller tells us to.
329 if (events.IsInPhase(PHASE_ONE) || _canEvade)
330 generic_halionAI::EnterEvadeMode();
331 }
332
333 void EnterCombat(Unit* who) override
334 {
335 Talk(SAY_AGGRO);
336
337 events.Reset();
338 events.SetPhase(PHASE_ONE);
339
340 generic_halionAI::EnterCombat(who);
341
342 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, me, 1);
343 instance->SetBossState(DATA_HALION, IN_PROGRESS);
344
345 events.ScheduleEvent(EVENT_ACTIVATE_FIREWALL, 5000);
346 events.ScheduleEvent(EVENT_METEOR_STRIKE, urand(20000, 25000));
347 events.ScheduleEvent(EVENT_FIERY_COMBUSTION, urand(15000, 18000));
348
349 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HALION_CONTROLLER)))
350 controller->AI()->SetData(DATA_FIGHT_PHASE, PHASE_ONE);
351 }
352
353 void JustDied(Unit* /*killer*/) override
354 {
355 _JustDied();
356
357 Talk(SAY_DEATH);
358 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
359
360 if (Creature* twilightHalion = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_TWILIGHT_HALION)))
361 if (twilightHalion->IsAlive())
362 twilightHalion->KillSelf();
363
364 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HALION_CONTROLLER)))
365 if (controller->IsAlive())
366 controller->KillSelf();
367 }
368
369 Position const* GetMeteorStrikePosition() const { return &_meteorStrikePos; }
370
371 void DamageTaken(Unit* attacker, uint32& damage) override
372 {
373 if (me->HealthBelowPctDamaged(75, damage) && events.IsInPhase(PHASE_ONE))
374 {
375 events.SetPhase(PHASE_TWO);
376 Talk(SAY_PHASE_TWO);
377
378 me->CastStop();
379 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
380 DoCast(me, SPELL_TWILIGHT_PHASING);
381
382 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HALION_CONTROLLER)))
383 controller->AI()->SetData(DATA_FIGHT_PHASE, PHASE_TWO);
384 return;
385 }
386
387 if (events.IsInPhase(PHASE_THREE))
388 {
389 // Don't consider copied damage.
390 if (!me->IsInPhase(attacker))
391 return;
392
393 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HALION_CONTROLLER)))
394 controller->AI()->SetData(DATA_MATERIAL_DAMAGE_TAKEN, damage);
395 }
396 }
397
398 void UpdateAI(uint32 diff) override
399 {
400 if (events.IsInPhase(PHASE_TWO))
401 return;
402
403 generic_halionAI::UpdateAI(diff);
404 }
405
406 void ExecuteEvent(uint32 eventId) override
407 {
408 switch (eventId)
409 {
410 case EVENT_ACTIVATE_FIREWALL:
411 // Flame ring is activated 5 seconds after starting encounter, DOOR_TYPE_ROOM is only instant.
412 for (uint8 i = DATA_FLAME_RING; i <= DATA_TWILIGHT_FLAME_RING; ++i)
413 if (GameObject* flameRing = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(i)))
414 instance->HandleGameObject(instance->GetGuidData(DATA_FLAME_RING), false, flameRing);
415 break;
416 case EVENT_METEOR_STRIKE:
417 {
418 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_TWILIGHT_REALM))
419 {
420 _meteorStrikePos = target->GetPosition();
421 me->CastSpell(_meteorStrikePos.GetPositionX(), _meteorStrikePos.GetPositionY(), _meteorStrikePos.GetPositionZ(), SPELL_METEOR_STRIKE, true, NULL, NULL, me->GetGUID());
422 Talk(SAY_METEOR_STRIKE);
423 }
424 events.ScheduleEvent(EVENT_METEOR_STRIKE, 40000);
425 break;
426 }
427 case EVENT_FIERY_COMBUSTION:
428 {
429 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true, -SPELL_TWILIGHT_REALM))
430 DoCast(target, SPELL_FIERY_COMBUSTION);
431 events.ScheduleEvent(EVENT_FIERY_COMBUSTION, 25000);
432 break;
433 }
434 default:
435 generic_halionAI::ExecuteEvent(eventId);
436 break;
437 }
438 }
439
440 void SetData(uint32 index, uint32 value) override
441 {
442 switch (index)
443 {
444 case DATA_FIGHT_PHASE:
445 events.SetPhase(value);
446 break;
447 default:
448 generic_halionAI::SetData(index, value);
449 }
450 }
451
452 private:
453 Position _meteorStrikePos;
454 };
455
456 CreatureAI* GetAI(Creature* creature) const override
457 {
458 return GetRubySanctumAI<boss_halionAI>(creature);
459 }

523 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HALION_CONTROLLER)))

533 {
534 events.SetPhase(PHASE_THREE);
535 me->CastStop();
536 DoCast(me, SPELL_TWILIGHT_DIVISION);
537 Talk(SAY_PHASE_THREE);
538 return;
539 }
540
541 if (events.IsInPhase(PHASE_THREE))
542 {
543 // Don't consider copied damage.
544 if (!me->IsInPhase(attacker))
545 return;
546
547 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HALION_CONTROLLER)))

639 void JustReachedHome() override
640 {
641 if (Creature* twilightHalion = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_TWILIGHT_HALION)))
642 twilightHalion->DespawnOrUnsummon();
643
644 if (Creature* halion = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_HALION)))
645 {
646 halion->AI()->SetData(DATA_EVADE_METHOD, 1);
647 halion->AI()->EnterEvadeMode();
648 }
649
650 _instance->SetBossState(DATA_HALION, FAIL);
651 }
652
653 void DoAction(int32 action) override
654 {
655 switch (action)
656 {
657 case ACTION_INTRO_HALION:
658 _events.Reset();
659 _events.SetPhase(PHASE_INTRO);
660 _events.ScheduleEvent(EVENT_START_INTRO, 2000);
661 break;
662 case ACTION_MONITOR_CORPOREALITY:
663 {
664 for (uint8 itr = DATA_HALION; itr <= DATA_TWILIGHT_HALION; itr++)
665 {
666 Creature* halion = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(itr));
667 if (!halion)
668 continue;
669
670 halion->CastSpell(halion, GetSpell(_materialCorporealityValue, itr == DATA_TWILIGHT_HALION), false);
671 halion->AI()->SetData(DATA_FIGHT_PHASE, PHASE_THREE);
672
673 if (itr == DATA_TWILIGHT_HALION)
674 continue;
675
676 halion->RemoveAurasDueToSpell(SPELL_TWILIGHT_PHASING);
677 halion->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
678 }
679
680 // Summon Twilight portals
681 DoCast(me, SPELL_SUMMON_EXIT_PORTALS);
682
683 _instance->DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TOGGLE, 1);
684 // Hardcoding doesn't really matter here.
685 _instance->DoUpdateWorldState(WORLDSTATE_CORPOREALITY_MATERIAL, 50);
686 _instance->DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TWILIGHT, 50);
687
688 _events.ScheduleEvent(EVENT_CHECK_CORPOREALITY, 7500);
689 }
690 default:
691 break;
692 }
693 }

798 void UpdateCorporeality()
799 {

813 action = CORPOREALITY_DECREASE;
814 else if (0.99f < damageRatio && damageRatio < 1.01f) // ]0.99, 1.01[: Twilight Mending
815 action = CORPOREALITY_TWILIGHT_MENDING;
816 else if (1.02f < damageRatio) // ]1.02, +oo [: Corporeality goes up
817 action = CORPOREALITY_INCREASE;
818
819 switch (action)
820 {
821 case CORPOREALITY_NONE:
822 {
823 _materialDamageTaken = 0;
824 _twilightDamageTaken = 0;
825 return;
826 }
827 case CORPOREALITY_INCREASE:
828 {
829 if (_materialCorporealityValue >= (MAX_CORPOREALITY_STATE - 1))
830 return;
831 ++_materialCorporealityValue;
832 break;
833 }
834 case CORPOREALITY_DECREASE:
835 {
836 if (_materialCorporealityValue <= 0)
837 return;
838 --_materialCorporealityValue;
839 break;
840 }
841 case CORPOREALITY_TWILIGHT_MENDING:
842 {
843 _events.ScheduleEvent(EVENT_TWILIGHT_MENDING, 100);
844 _materialDamageTaken = 0;
845 _twilightDamageTaken = 0;
846 return;
847 }
848 }

850 _materialDamageTaken = 0;
851 _twilightDamageTaken = 0;
852
853 _instance->DoUpdateWorldState(WORLDSTATE_CORPOREALITY_MATERIAL, _materialCorporealityValue * 10);
854 _instance->DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TWILIGHT, 100 - _materialCorporealityValue * 10);

864 else // if (itr == DATA_HALION)
865 halion->AI()->Talk(oldValue > _materialCorporealityValue ? EMOTE_CORPOREALITY_POT : EMOTE_CORPOREALITY_PIP, halion);
866 }
867 }

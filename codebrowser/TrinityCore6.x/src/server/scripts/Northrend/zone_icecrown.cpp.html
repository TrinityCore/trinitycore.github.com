38class npc_argent_valiant : public CreatureScript
39{
40public:
41 npc_argent_valiant() : CreatureScript("npc_argent_valiant") { }
42
43 struct npc_argent_valiantAI : public ScriptedAI
44 {
45 npc_argent_valiantAI(Creature* creature) : ScriptedAI(creature)
46 {
47 Initialize();
48 creature->GetMotionMaster()->MovePoint(0, 8599.258f, 963.951f, 547.553f);
49 creature->setFaction(35); //wrong faction in db?
50 }
51
52 void Initialize()
53 {
54 uiChargeTimer = 7000;
55 uiShieldBreakerTimer = 10000;
56 }
57
58 uint32 uiChargeTimer;
59 uint32 uiShieldBreakerTimer;
60
61 void Reset() override
62 {
63 Initialize();
64 }
65
66 void MovementInform(uint32 uiType, uint32 /*uiId*/) override
67 {
68 if (uiType != POINT_MOTION_TYPE)
69 return;
70
71 me->setFaction(14);
72 }
73
74 void DamageTaken(Unit* pDoneBy, uint32& uiDamage) override
75 {
76 if (uiDamage > me->GetHealth() && pDoneBy->GetTypeId() == TYPEID_PLAYER)
77 {
78 uiDamage = 0;
79 pDoneBy->CastSpell(pDoneBy, SPELL_KILL_CREDIT, true);
80 me->setFaction(35);
81 me->DespawnOrUnsummon(5000);
82 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
83 EnterEvadeMode();
84 }
85 }
86
87 void UpdateAI(uint32 uiDiff) override
88 {
89 if (!UpdateVictim())
90 return;
91
92 if (uiChargeTimer <= uiDiff)
93 {
94 DoCastVictim(SPELL_CHARGE);
95 uiChargeTimer = 7000;
96 } else uiChargeTimer -= uiDiff;
97
98 if (uiShieldBreakerTimer <= uiDiff)
99 {
100 DoCastVictim(SPELL_SHIELD_BREAKER);
101 uiShieldBreakerTimer = 10000;
102 } else uiShieldBreakerTimer -= uiDiff;
103
104 DoMeleeAttackIfReady();
105 }
106 };
107
108 CreatureAI* GetAI(Creature* creature) const override
109 {
110 return new npc_argent_valiantAI(creature);
111 }

139 void MoveInLineOfSight(Unit* who) override
140
141 {
142 if (me->GetAreaId() != AREA_SUNREAVER_PAVILION && me->GetAreaId() != AREA_SILVER_COVENANT_PAVILION)
143 return;
144
145 if (!who || who->GetTypeId() != TYPEID_PLAYER || !me->IsHostileTo(who) || !me->isInBackInMap(who, 5.0f))

147

149 return;
150
151 if (who->ToPlayer()->GetTeamId() == TEAM_ALLIANCE)
152 who->CastSpell(who, SPELL_TRESPASSER_H, true);
153 else
154 who->CastSpell(who, SPELL_TRESPASSER_A, true);
155
156 }

192class npc_tournament_training_dummy : public CreatureScript
193{
194 public:
195 npc_tournament_training_dummy(): CreatureScript("npc_tournament_training_dummy"){ }
196
197 struct npc_tournament_training_dummyAI : ScriptedAI
198 {
199 npc_tournament_training_dummyAI(Creature* creature) : ScriptedAI(creature)
200 {
201 Initialize();
202 SetCombatMovement(false);
203 }
204
205 void Initialize()
206 {
207 isVulnerable = false;
208 }
209
210 EventMap events;
211 bool isVulnerable;
212
213 void Reset() override
214 {
215 me->SetControlled(true, UNIT_STATE_STUNNED);
216 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_KNOCK_BACK, true);
217 Initialize();
218
219 // Cast Defend spells to max stack size
220 switch (me->GetEntry())
221 {
222 case NPC_CHARGE_TARGET:
223 DoCast(SPELL_CHARGE_DEFEND);
224 break;
225 case NPC_RANGED_TARGET:
226 me->CastCustomSpell(SPELL_RANGED_DEFEND, SPELLVALUE_AURA_STACK, 3, me);
227 break;
228 }
229
230 events.Reset();
231 events.ScheduleEvent(EVENT_DUMMY_RECAST_DEFEND, 5000);
232 }
233
234 void EnterEvadeMode() override
235 {
236 if (!_EnterEvadeMode())
237 return;
238
239 Reset();
240 }
241
242 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
243 {
244 damage = 0;
245 events.RescheduleEvent(EVENT_DUMMY_RESET, 10000);
246 }
247
248 void SpellHit(Unit* caster, SpellInfo const* spell) override
249 {
250 switch (me->GetEntry())
251 {
252 case NPC_CHARGE_TARGET:
253 if (spell->Id == SPELL_PLAYER_CHARGE)
254 if (isVulnerable)
255 DoCast(caster, SPELL_CHARGE_CREDIT, true);
256 break;
257 case NPC_MELEE_TARGET:
258 if (spell->Id == SPELL_PLAYER_THRUST)
259 {
260 DoCast(caster, SPELL_MELEE_CREDIT, true);
261
262 if (Unit* target = caster->GetVehicleBase())
263 DoCast(target, SPELL_COUNTERATTACK, true);
264 }
265 break;
266 case NPC_RANGED_TARGET:
267 if (spell->Id == SPELL_PLAYER_BREAK_SHIELD)
268 if (isVulnerable)
269 DoCast(caster, SPELL_RANGED_CREDIT, true);
270 break;
271 }
272
273 if (spell->Id == SPELL_PLAYER_BREAK_SHIELD)
274 if (!me->HasAura(SPELL_CHARGE_DEFEND) && !me->HasAura(SPELL_RANGED_DEFEND))
275 isVulnerable = true;
276 }
277
278 void UpdateAI(uint32 diff) override
279 {
280 events.Update(diff);
281
282 switch (events.ExecuteEvent())
283 {
284 case EVENT_DUMMY_RECAST_DEFEND:
285 switch (me->GetEntry())
286 {
287 case NPC_CHARGE_TARGET:
288 {
289 if (!me->HasAura(SPELL_CHARGE_DEFEND))
290 DoCast(SPELL_CHARGE_DEFEND);
291 break;
292 }
293 case NPC_RANGED_TARGET:
294 {
295 Aura* defend = me->GetAura(SPELL_RANGED_DEFEND);
296 if (!defend || defend->GetStackAmount() < 3 || defend->GetDuration() <= 8000)
297 DoCast(SPELL_RANGED_DEFEND);
298 break;
299 }
300 }
301 isVulnerable = false;
302 events.ScheduleEvent(EVENT_DUMMY_RECAST_DEFEND, 5000);
303 break;
304 case EVENT_DUMMY_RESET:
305 if (UpdateVictim())
306 {
307 EnterEvadeMode();
308 events.ScheduleEvent(EVENT_DUMMY_RESET, 10000);
309 }
310 break;
311 }
312
313 if (!UpdateVictim())
314 return;
315
316 if (!me->HasUnitState(UNIT_STATE_STUNNED))
317 me->SetControlled(true, UNIT_STATE_STUNNED);
318 }
319
320 void MoveInLineOfSight(Unit* /*who*/) override { }
321
322 };
323
324 CreatureAI* GetAI(Creature* creature) const override
325 {
326 return new npc_tournament_training_dummyAI(creature);
327 }

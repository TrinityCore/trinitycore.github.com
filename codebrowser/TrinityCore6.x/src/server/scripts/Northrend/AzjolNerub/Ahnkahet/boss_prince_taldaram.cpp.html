81class boss_prince_taldaram : public CreatureScript
82{
83 public:
84 boss_prince_taldaram() : CreatureScript("boss_prince_taldaram") { }
85
86 struct boss_prince_taldaramAI : public BossAI
87 {
88 boss_prince_taldaramAI(Creature* creature) : BossAI(creature, DATA_PRINCE_TALDARAM)
89 {
90 me->SetDisableGravity(true);
91 _embraceTakenDamage = 0;
92 }
93
94 void Reset() override
95 {
96 _Reset();
97 _flameSphereTargetGUID.Clear();
98 _embraceTargetGUID.Clear();
99 _embraceTakenDamage = 0;
100 }
101
102 void EnterCombat(Unit* /*who*/) override
103 {
104 _EnterCombat();
105 Talk(SAY_AGGRO);
106 events.ScheduleEvent(EVENT_BLOODTHIRST, 10000);
107 events.ScheduleEvent(EVENT_VANISH, urand(25000, 35000));
108 events.ScheduleEvent(EVENT_CONJURE_FLAME_SPHERES, 5000);
109 }
110
111 void JustSummoned(Creature* summon) override
112 {
113 BossAI::JustSummoned(summon);
114
115 switch (summon->GetEntry())
116 {
117 case NPC_FLAME_SPHERE_1:
118 case NPC_FLAME_SPHERE_2:
119 case NPC_FLAME_SPHERE_3:
120 summon->AI()->SetGUID(_flameSphereTargetGUID);
121 default:
122 return;
123 }
124 }
125
126 void UpdateAI(uint32 diff) override
127 {
128 if (!UpdateVictim())
129 return;
130
131 events.Update(diff);
132
133 if (me->HasUnitState(UNIT_STATE_CASTING))
134 return;
135
136 while (uint32 eventId = events.ExecuteEvent())
137 {
138 switch (eventId)
139 {
140 case EVENT_BLOODTHIRST:
141 DoCast(me, SPELL_BLOODTHIRST);
142 events.ScheduleEvent(EVENT_BLOODTHIRST, 10000);
143 break;
144 case EVENT_CONJURE_FLAME_SPHERES:
145 // random target?
146 if (Unit* victim = me->GetVictim())
147 {
148 _flameSphereTargetGUID = victim->GetGUID();
149 DoCast(victim, SPELL_CONJURE_FLAME_SPHERE);
150 }
151 events.ScheduleEvent(EVENT_CONJURE_FLAME_SPHERES, 15000);
152 break;
153 case EVENT_VANISH:
154 {
155 Map::PlayerList const& players = me->GetMap()->GetPlayers();
156 uint32 targets = 0;
157 for (Map::PlayerList::const_iterator i = players.begin(); i != players.end(); ++i)
158 {
159 Player* player = i->GetSource();
160 if (player && player->IsAlive())
161 ++targets;
162 }
163
164 if (targets > 2)
165 {
166 Talk(SAY_VANISH);
167 DoCast(me, SPELL_VANISH);
168 me->SetInCombatState(true); // Prevents the boss from resetting
169 events.DelayEvents(500);
170 events.ScheduleEvent(EVENT_JUST_VANISHED, 500);
171 if (Unit* embraceTarget = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
172 _embraceTargetGUID = embraceTarget->GetGUID();
173 }
174 events.ScheduleEvent(EVENT_VANISH, urand(25000, 35000));
175 break;
176 }
177 case EVENT_JUST_VANISHED:
178 if (Unit* embraceTarget = GetEmbraceTarget())
179 {
180 me->GetMotionMaster()->Clear();
181 me->SetSpeed(MOVE_WALK, 2.0f, true);
182 me->GetMotionMaster()->MoveChase(embraceTarget);
183 }
184 events.ScheduleEvent(EVENT_VANISHED, 1300);
185 break;
186 case EVENT_VANISHED:
187 if (Unit* embraceTarget = GetEmbraceTarget())
188 DoCast(embraceTarget, SPELL_EMBRACE_OF_THE_VAMPYR);
189 Talk(SAY_FEED);
190 me->GetMotionMaster()->Clear();
191 me->SetSpeed(MOVE_WALK, 1.0f, true);
192 me->GetMotionMaster()->MoveChase(me->GetVictim());
193 events.ScheduleEvent(EVENT_FEEDING, 20000);
194 break;
195 case EVENT_FEEDING:
196 _embraceTargetGUID.Clear();
197 break;
198 default:
199 break;
200 }
201 }
202
203 DoMeleeAttackIfReady();
204 }
205
206 void DamageTaken(Unit* /*doneBy*/, uint32& damage) override
207 {
208 Unit* embraceTarget = GetEmbraceTarget();
209
210 if (embraceTarget && embraceTarget->IsAlive())
211 {
212 _embraceTakenDamage += damage;
213 if (_embraceTakenDamage > DUNGEON_MODE<uint32>(DATA_EMBRACE_DMG, H_DATA_EMBRACE_DMG))
214 {
215 _embraceTargetGUID.Clear();
216 me->CastStop();
217 }
218 }
219 }
220
221 void JustDied(Unit* /*killer*/) override
222 {
223 Talk(SAY_DEATH);
224 _JustDied();
225 }
226
227 void KilledUnit(Unit* victim) override
228 {
229 if (victim->GetTypeId() != TYPEID_PLAYER)
230 return;
231
232 if (victim->GetGUID() == _embraceTargetGUID)
233 _embraceTargetGUID.Clear();
234
235 Talk(SAY_SLAY);
236 }
237
238 bool CheckSpheres()
239 {
240 for (uint8 i = 0; i < 2; ++i)
241 if (!instance->GetData(DATA_SPHERE_1 + i))
242 return false;
243
244 RemovePrison();
245 return true;
246 }
247
248 Unit* GetEmbraceTarget()
249 {
250 if (!_embraceTargetGUID.IsEmpty())
251 return ObjectAccessor::GetUnit(*me, _embraceTargetGUID);
252
253 return NULL;
254 }
255
256 void RemovePrison()
257 {
258 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
259 me->RemoveAurasDueToSpell(SPELL_BEAM_VISUAL);
260 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), DATA_GROUND_POSITION_Z, me->GetOrientation());
261 DoCast(SPELL_HOVER_FALL);
262 me->SetDisableGravity(false);
263 me->GetMotionMaster()->MoveLand(0, me->GetHomePosition());
264 Talk(SAY_WARNING);
265 instance->HandleGameObject(instance->GetGuidData(DATA_PRINCE_TALDARAM_PLATFORM), true);
266 }
267
268 private:
269 ObjectGuid _flameSphereTargetGUID;
270 ObjectGuid _embraceTargetGUID;
271 uint32 _embraceTakenDamage;
272 };
273
274 CreatureAI* GetAI(Creature* creature) const override
275 {
276 return GetAhnKahetAI<boss_prince_taldaramAI>(creature);
277 }
278};

280// 30106, 31686, 31687 - Flame Sphere
281class npc_prince_taldaram_flame_sphere : public CreatureScript
282{
283 public:
284 npc_prince_taldaram_flame_sphere() : CreatureScript("npc_prince_taldaram_flame_sphere") { }
285
286 struct npc_prince_taldaram_flame_sphereAI : public ScriptedAI
287 {
288 npc_prince_taldaram_flame_sphereAI(Creature* creature) : ScriptedAI(creature)
289 {
290 }
291
292 void Reset() override
293 {
294 DoCast(me, SPELL_FLAME_SPHERE_SPAWN_EFFECT, true);
295 DoCast(me, SPELL_FLAME_SPHERE_VISUAL, true);
296
297 _flameSphereTargetGUID.Clear();
298 _events.Reset();
299 _events.ScheduleEvent(EVENT_START_MOVE, 3 * IN_MILLISECONDS);
300 _events.ScheduleEvent(EVENT_DESPAWN, 13 * IN_MILLISECONDS);
301 }
302
303 void SetGUID(ObjectGuid guid, int32 /*id = 0*/) override
304 {
305 _flameSphereTargetGUID = guid;
306 }
307
308 void EnterCombat(Unit* /*who*/) override { }
309 void MoveInLineOfSight(Unit* /*who*/) override { }
310
311 void UpdateAI(uint32 diff) override
312 {
313 _events.Update(diff);
314
315 while (uint32 eventId = _events.ExecuteEvent())
316 {
317 switch (eventId)
318 {
319 case EVENT_START_MOVE:
320 {
321 DoCast(me, SPELL_FLAME_SPHERE_PERIODIC, true);
322
323 /// @todo: find correct values
324 float angleOffset = 0.0f;
325 float distOffset = DATA_SPHERE_DISTANCE;
326
327 switch (me->GetEntry())
328 {
329 case NPC_FLAME_SPHERE_1:
330 break;
331 case NPC_FLAME_SPHERE_2:
332 angleOffset = DATA_SPHERE_ANGLE_OFFSET;
333 break;
334 case NPC_FLAME_SPHERE_3:
335 angleOffset = -DATA_SPHERE_ANGLE_OFFSET;
336 break;
337 default:
338 return;
339 }
340
341 Unit* sphereTarget = ObjectAccessor::GetUnit(*me, _flameSphereTargetGUID);
342 if (!sphereTarget)
343 return;
344
345 float angle = me->GetAngle(sphereTarget) + angleOffset;
346 float x = me->GetPositionX() + distOffset * std::cos(angle);
347 float y = me->GetPositionY() + distOffset * std::sin(angle);
348
349 /// @todo: correct speed
350 me->GetMotionMaster()->MovePoint(0, x, y, me->GetPositionZ());
351 break;
352 }
353 case EVENT_DESPAWN:
354 DoCast(me, SPELL_FLAME_SPHERE_DEATH_EFFECT, true);
355 me->DespawnOrUnsummon(1000);
356 break;
357 default:
358 break;
359 }
360 }
361 }
362
363 private:
364 EventMap _events;
365 ObjectGuid _flameSphereTargetGUID;
366 };
367
368 CreatureAI* GetAI(Creature* creature) const override
369 {
370 return new npc_prince_taldaram_flame_sphereAI(creature);
371 }

374// 193093, 193094 - Ancient Nerubian Device
375class go_prince_taldaram_sphere : public GameObjectScript
376{
377 public:
378 go_prince_taldaram_sphere() : GameObjectScript("go_prince_taldaram_sphere") { }
379
380 bool OnGossipHello(Player* /*player*/, GameObject* go) override
381 {
382 InstanceScript* instance = go->GetInstanceScript();
383 if (!instance)
384 return false;
385
386 Creature* PrinceTaldaram = ObjectAccessor::GetCreature(*go, instance->GetGuidData(DATA_PRINCE_TALDARAM));
387 if (PrinceTaldaram && PrinceTaldaram->IsAlive())
388 {
389 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
390 go->SetGoState(GO_STATE_ACTIVE);
391
392 switch (go->GetEntry())
393 {
394 case GO_SPHERE_1:
395 instance->SetData(DATA_SPHERE_1, IN_PROGRESS);
396 PrinceTaldaram->AI()->Talk(SAY_1);
397 break;
398 case GO_SPHERE_2:
399 instance->SetData(DATA_SPHERE_2, IN_PROGRESS);
400 PrinceTaldaram->AI()->Talk(SAY_1);
401 break;
402 }
403
404 ENSURE_AI(boss_prince_taldaram::boss_prince_taldaramAI, PrinceTaldaram->AI())->CheckSpheres();
405 }
406 return true;
407 }

417 {
418 PrepareSpellScript(spell_prince_taldaram_conjure_flame_sphere_SpellScript);
419
420 bool Validate(SpellInfo const* /*spellInfo*/) override
421 {
422 if (!sSpellMgr->GetSpellInfo(SPELL_FLAME_SPHERE_SUMMON_1)
423 || !sSpellMgr->GetSpellInfo(SPELL_FLAME_SPHERE_SUMMON_2)

430 {
431 Unit* caster = GetCaster();
432 caster->CastSpell(caster, SPELL_FLAME_SPHERE_SUMMON_1, true);
433
434 if (caster->GetMap()->IsHeroic())
435 {
436 caster->CastSpell(caster, SPELL_FLAME_SPHERE_SUMMON_2, true);
437 caster->CastSpell(caster, SPELL_FLAME_SPHERE_SUMMON_3, true);
438 }
439 }
440
441 void Register() override
442 {
443 OnEffectHitTarget += SpellEffectFn(spell_prince_taldaram_conjure_flame_sphere_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_DUMMY);

445 };

88 boss_anub_arakAI(Creature* creature) : ScriptedAI(creature), Summons(me)

95 UndergroundTimer = 0;

101 void Initialize()
102 {
103 CarrionBeetlesTimer = 8 * IN_MILLISECONDS;
104 LeechingSwarmTimer = 20 * IN_MILLISECONDS;
105 ImpaleTimer = 9 * IN_MILLISECONDS;
106 PoundTimer = 15 * IN_MILLISECONDS;
107
108 Phase = PHASE_MELEE;
109 UndergroundPhase = 0;
110 Channeling = false;
111 ImpalePhase = IMPALE_PHASE_TARGET;
112 ImpaleTarget.Clear();
113 }

137 void Reset() override
138 {
139 Initialize();
140
141 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
142 me->RemoveAura(SPELL_SUBMERGE);
143
144 Summons.DespawnAll();
145
146 instance->SetBossState(DATA_ANUBARAK, NOT_STARTED);
147 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMED_START_EVENT);
148 }
149
150 Creature* DoSummonImpaleTarget(Unit* target)
151 {
152 Position targetPos = target->GetPosition();
153
154 if (TempSummon* impaleTarget = me->SummonCreature(CREATURE_IMPALE_TARGET, targetPos, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 6*IN_MILLISECONDS))
155 {
156 ImpaleTarget = impaleTarget->GetGUID();
157 impaleTarget->SetReactState(REACT_PASSIVE);
158 impaleTarget->SetDisplayId(DISPLAY_INVISIBLE);
159 impaleTarget->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE|UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
160 return impaleTarget;
161 }
162
163 return NULL;
164 }

178 void UpdateAI(uint32 diff) override
179 {
180 if (!UpdateVictim())
181 return;
182
183 if (DelayTimer && DelayTimer > 5000)
184 DelayEventStart();
185 else DelayTimer+=diff;
186
187 switch (Phase)
188 {
189 case PHASE_UNDERGROUND:
190 if (ImpaleTimer <= diff)
191 {
192 switch (ImpalePhase)
193 {
194 case IMPALE_PHASE_TARGET:
195 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
196 {
197 if (Creature* impaleTarget = DoSummonImpaleTarget(target))
198 impaleTarget->CastSpell(impaleTarget, SPELL_IMPALE_SHAKEGROUND, true);
199 ImpaleTimer = 3*IN_MILLISECONDS;
200 ImpalePhase = IMPALE_PHASE_ATTACK;
201 }
202 break;
203 case IMPALE_PHASE_ATTACK:
204 if (Creature* impaleTarget = ObjectAccessor::GetCreature(*me, ImpaleTarget))
205 {
206 impaleTarget->CastSpell(impaleTarget, SPELL_IMPALE_SPIKE, false);
207 impaleTarget->RemoveAurasDueToSpell(SPELL_IMPALE_SHAKEGROUND);
208 }
209 ImpalePhase = IMPALE_PHASE_DMG;
210 ImpaleTimer = 1*IN_MILLISECONDS;
211 break;
212 case IMPALE_PHASE_DMG:
213 if (Creature* impaleTarget = ObjectAccessor::GetCreature(*me, ImpaleTarget))
214 me->CastSpell(impaleTarget, SPELL_IMPALE_DMG, true);
215 ImpalePhase = IMPALE_PHASE_TARGET;
216 ImpaleTimer = 9*IN_MILLISECONDS;
217 break;
218 }
219 } else ImpaleTimer -= diff;
220
221 if (!GuardianSummoned)
222 {
223 for (uint8 i = 0; i < 2; ++i)
224 {
225 if (Creature* Guardian = me->SummonCreature(CREATURE_GUARDIAN, SpawnPointGuardian[i], TEMPSUMMON_CORPSE_DESPAWN, 0))
226 {
227 Guardian->AddThreat(me->GetVictim(), 0.0f);
228 DoZoneInCombat(Guardian);
229 }
230 }
231 GuardianSummoned = true;
232 }
233
234 if (!VenomancerSummoned)
235 {
236 if (VenomancerTimer <= diff)
237 {
238 if (UndergroundPhase > 1)
239 {
240 for (uint8 i = 0; i < 2; ++i)
241 {
242 if (Creature* Venomancer = me->SummonCreature(CREATURE_VENOMANCER, SpawnPoint[i], TEMPSUMMON_CORPSE_DESPAWN, 0))
243 {
244 Venomancer->AddThreat(me->GetVictim(), 0.0f);
245 DoZoneInCombat(Venomancer);
246 }
247 }
248 VenomancerSummoned = true;
249 }
250 } else VenomancerTimer -= diff;
251 }
252
253 if (!DatterSummoned)
254 {
255 if (DatterTimer <= diff)
256 {
257 if (UndergroundPhase > 2)
258 {
259 for (uint8 i = 0; i < 2; ++i)
260 {
261 if (Creature* Datter = me->SummonCreature(CREATURE_DATTER, SpawnPoint[i], TEMPSUMMON_CORPSE_DESPAWN, 0))
262 {
263 Datter->AddThreat(me->GetVictim(), 0.0f);
264 DoZoneInCombat(Datter);
265 }
266 }
267 DatterSummoned = true;
268 }
269 } else DatterTimer -= diff;
270
271 if (me->HasAura(SPELL_LEECHING_SWARM))
272 me->RemoveAurasDueToSpell(SPELL_LEECHING_SWARM);
273 }
274
275 if (UndergroundTimer <= diff)
276 {
277 me->RemoveAura(SPELL_SUBMERGE);
278 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
279 Phase = PHASE_MELEE;
280 } else UndergroundTimer -= diff;
281 break;
282
283 case PHASE_MELEE:
284 if (((UndergroundPhase == 0 && HealthBelowPct(75))
285 || (UndergroundPhase == 1 && HealthBelowPct(50))
286 || (UndergroundPhase == 2 && HealthBelowPct(25)))
287 && !me->HasUnitState(UNIT_STATE_CASTING))
288 {
289 GuardianSummoned = false;
290 VenomancerSummoned = false;
291 DatterSummoned = false;
292
293 UndergroundTimer = 40*IN_MILLISECONDS;
294 VenomancerTimer = 25*IN_MILLISECONDS;
295 DatterTimer = 32*IN_MILLISECONDS;
296
297 ImpalePhase = 0;
298 ImpaleTimer = 9*IN_MILLISECONDS;
299
300 DoCast(me, SPELL_SUBMERGE, false);
301 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
302
303 Phase = PHASE_UNDERGROUND;
304 ++UndergroundPhase;
305 }
306
307 if (Channeling == true)
308 {
309 for (uint8 i = 0; i < 8; ++i)
310 DoCastVictim(SPELL_SUMMON_CARRION_BEETLES, true);
311 Channeling = false;
312 }
313 else if (CarrionBeetlesTimer <= diff)
314 {
315 Channeling = true;
316 DoCastVictim(SPELL_CARRION_BEETLES);
317 CarrionBeetlesTimer = 25*IN_MILLISECONDS;
318 } else CarrionBeetlesTimer -= diff;
319
320 if (LeechingSwarmTimer <= diff)
321 {
322 DoCast(me, SPELL_LEECHING_SWARM, true);
323 LeechingSwarmTimer = 19*IN_MILLISECONDS;
324 } else LeechingSwarmTimer -= diff;
325
326 if (PoundTimer <= diff)
327 {
328 if (Unit* target = me->GetVictim())
329 {
330 if (Creature* pImpaleTarget = DoSummonImpaleTarget(target))
331 me->CastSpell(pImpaleTarget, SPELL_POUND, false);
332 }
333 PoundTimer = 16500;
334 } else PoundTimer -= diff;
335
336 DoMeleeAttackIfReady();
337 break;
338 }
339 }

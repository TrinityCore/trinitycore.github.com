40 SPELL_FREE_RAGECLAW = 55223
41};
42
43class npc_drakuru_shackles : public CreatureScript
44{
45public:
46 npc_drakuru_shackles() : CreatureScript("npc_drakuru_shackles") { }
47
48 struct npc_drakuru_shacklesAI : public ScriptedAI
49 {
50 npc_drakuru_shacklesAI(Creature* creature) : ScriptedAI(creature) { }
51
52 void Reset() override
53 {
54 _rageclawGUID.Clear();
55 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
56
57 float x, y, z;
58 me->GetClosePoint(x, y, z, me->GetObjectSize() / 3, 0.1f);
59
60 if (Creature* summon = me->SummonCreature(NPC_RAGECLAW, x, y, z, 0, TEMPSUMMON_DEAD_DESPAWN, 1000))
61 {
62 _rageclawGUID = summon->GetGUID();
63 LockRageclaw(summon);
64 }
65 }
66
67 void LockRageclaw(Creature* rageclaw)
68 {
69 // pointer check not needed
70 me->SetInFront(rageclaw);
71 rageclaw->SetInFront(me);
72
73 DoCast(rageclaw, SPELL_LEFT_CHAIN, true);
74 DoCast(rageclaw, SPELL_RIGHT_CHAIN, true);
75 }
76
77 void UnlockRageclaw(Unit* who, Creature* rageclaw)
78 {
79 if (!who)
80 return;
81
82 // pointer check not needed
83 DoCast(rageclaw, SPELL_FREE_RAGECLAW, true);
84
85 me->setDeathState(DEAD);
86 }
87
88 void SpellHit(Unit* caster, const SpellInfo* spell) override
89 {
90 if (spell->Id == SPELL_UNLOCK_SHACKLE)
91 {
92 if (caster->ToPlayer()->GetQuestStatus(QUEST_TROLLS_IS_GONE_CRAZY) == QUEST_STATUS_INCOMPLETE)
93 {
94 if (Creature* rageclaw = ObjectAccessor::GetCreature(*me, _rageclawGUID))
95 {
96 UnlockRageclaw(caster, rageclaw);
97 caster->ToPlayer()->KilledMonster(rageclaw->GetCreatureTemplate(), _rageclawGUID);
98 me->DespawnOrUnsummon();
99 }
100 else
101 me->setDeathState(JUST_DIED);
102 }
103 }
104 }
105
106 private:
107 ObjectGuid _rageclawGUID;
108 };
109
110 CreatureAI* GetAI(Creature* creature) const override
111 {
112 return new npc_drakuru_shacklesAI(creature);
113 }
114};

218class npc_crusade_recruit : public CreatureScript
219{
220public:
221 npc_crusade_recruit() : CreatureScript("npc_crusade_recruit") { }
222
223 struct npc_crusade_recruitAI : public ScriptedAI
224 {
225 npc_crusade_recruitAI(Creature* creature) : ScriptedAI(creature)
226 {
227 Initialize();
228 }
229
230 void Initialize()
231 {
232 _heading = me->GetOrientation();
233 }
234
235 void Reset() override
236 {
237 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
238 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_COWER);
239 Initialize();
240 }
241
242 void UpdateAI(uint32 diff) override
243 {
244 _events.Update(diff);
245
246 while (uint32 eventId = _events.ExecuteEvent())
247 {
248 switch (eventId)
249 {
250 case EVENT_RECRUIT_1:
251 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
252 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_NONE);
253 Talk(SAY_RECRUIT);
254 _events.ScheduleEvent(EVENT_RECRUIT_2, 3000);
255 break;
256 case EVENT_RECRUIT_2:
257 me->SetWalk(true);
258 me->GetMotionMaster()->MovePoint(0, me->GetPositionX() + (std::cos(_heading) * 10), me->GetPositionY() + (std::sin(_heading) * 10), me->GetPositionZ());
259 me->DespawnOrUnsummon(5000);
260 break;
261 default:
262 break;
263 }
264 }
265
266 if (!UpdateVictim())
267 return;
268 }
269
270 void sGossipSelect(Player* player, uint32 /*menuId*/, uint32 /*gossipListId*/) override
271 {
272 _events.ScheduleEvent(EVENT_RECRUIT_1, 100);
273 player->CLOSE_GOSSIP_MENU();
274 me->CastSpell(player, SPELL_QUEST_CREDIT, true);
275 me->SetFacingToObject(player);
276 }
277
278 private:
279 EventMap _events;
280 float _heading; // Store creature heading
281 };
282
283 CreatureAI* GetAI(Creature* creature) const override
284 {
285 return new npc_crusade_recruitAI(creature);
286 }

485 case 4:
486 _events.ScheduleEvent(EVENT_MEDIUM_4, 100);
487 break;
488 case 5:
489 _events.ScheduleEvent(EVENT_MEDIUM_5, 100);

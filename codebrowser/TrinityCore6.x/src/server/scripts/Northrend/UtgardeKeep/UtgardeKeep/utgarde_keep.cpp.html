36class npc_dragonflayer_forge_master : public CreatureScript
37{
38 public:
39 npc_dragonflayer_forge_master() : CreatureScript("npc_dragonflayer_forge_master") { }
40
41 struct npc_dragonflayer_forge_masterAI : public ScriptedAI
42 {
43 npc_dragonflayer_forge_masterAI(Creature* creature) : ScriptedAI(creature)
44 {
45 _instance = creature->GetInstanceScript();
46 _forgeId = 0;
47 }
48
49 void Reset() override
50 {
51 if (!_forgeId)
52 _forgeId = GetForgeMasterType();
53
54 if (!me->IsAlive())
55 return;
56
57 if (_forgeId)
58 _instance->SetData(DATA_FORGE_1 + _forgeId - 1, NOT_STARTED);
59 }
60
61 void JustDied(Unit* /*killer*/) override
62 {
63 if (!_forgeId)
64 _forgeId = GetForgeMasterType();
65
66 if (_forgeId)
67 _instance->SetData(DATA_FORGE_1 + _forgeId - 1, DONE);
68 }
69
70 void EnterCombat(Unit* /*who*/) override
71 {
72 if (!_forgeId)
73 _forgeId = GetForgeMasterType();
74
75 if (_forgeId)
76 _instance->SetData(DATA_FORGE_1 + _forgeId - 1, IN_PROGRESS);
77
78 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_NONE);
79 }
80
81 void UpdateAI(uint32 /*diff*/) override
82 {
83 if (!_forgeId)
84 _forgeId = GetForgeMasterType();
85
86 if (!UpdateVictim())
87 return;
88
89 DoMeleeAttackIfReady();
90 }
91
92 private:
93 uint8 GetForgeMasterType()
94 {
95 float diff = 30.0f;
96 uint8 id = 0;
97
98 for (uint8 i = 0; i < 3; ++i)
99 {
100 if (GameObject* go = me->FindNearestGameObject(ForgeSearch[i], 30))
101 {
102 if (me->IsWithinDist(go, diff, false))
103 {
104 id = i + 1;
105 diff = me->GetDistance2d(go);
106 }
107 }
108 }
109 return id > 0 && id < 4 ? id : 0;
110 }
111
112 InstanceScript* _instance;
113 uint8 _forgeId;
114 };
115
116 CreatureAI* GetAI(Creature* creature) const override
117 {
118 return GetUtgardeKeepAI<npc_dragonflayer_forge_masterAI>(creature);
119 }

217const Position protodrakeCheckPos = {206.24f, -190.28f, 200.11f, 0.f};

219class npc_enslaved_proto_drake : public CreatureScript
220{
221 public:
222 npc_enslaved_proto_drake() : CreatureScript("npc_enslaved_proto_drake") { }
223
224 struct npc_enslaved_proto_drakeAI : public ScriptedAI
225 {
226 npc_enslaved_proto_drakeAI(Creature* creature) : ScriptedAI(creature)
227 {
228 _setData = false;
229 }
230
231 void Reset() override
232 {
233 _events.Reset();
234 _events.ScheduleEvent(EVENT_REND, urand(2000, 3000));
235 _events.ScheduleEvent(EVENT_FLAME_BREATH, urand(5500, 7000));
236 _events.ScheduleEvent(EVENT_KNOCKAWAY, urand(3500, 6000));
237 }
238
239 void MovementInform(uint32 type, uint32 id) override
240 {
241 if (type == WAYPOINT_MOTION_TYPE && id == POINT_LAST)
242 {
243 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
244 }
245 }
246
247 void SetData(uint32 type, uint32 data) override
248 {
249 if (type == TYPE_PROTODRAKE_AT && data == DATA_PROTODRAKE_MOVE && !_setData && me->GetDistance(protodrakeCheckPos) < 5.0f)
250 {
251 _setData = true;
252 me->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
253 me->GetMotionMaster()->MovePath(PATH_PROTODRAKE, false);
254 }
255 }
256
257 void UpdateAI(uint32 diff) override
258 {
259 if (!UpdateVictim())
260 return;
261
262 _events.Update(diff);
263
264 if (me->HasUnitState(UNIT_STATE_CASTING))
265 return;
266
267 while (uint32 eventid = _events.ExecuteEvent())
268 {
269 switch (eventid)
270 {
271 case EVENT_REND:
272 DoCast(SPELL_REND);
273 _events.ScheduleEvent(EVENT_REND, urand(15000, 20000));
274 break;
275 case EVENT_FLAME_BREATH:
276 DoCast(SPELL_FLAME_BREATH);
277 _events.ScheduleEvent(EVENT_FLAME_BREATH, urand(11000, 12000));
278 break;
279 case EVENT_KNOCKAWAY:
280 DoCast(SPELL_KNOCK_AWAY);
281 _events.ScheduleEvent(EVENT_KNOCKAWAY, urand(7000, 8500));
282 break;
283 default:
284 break;
285 }
286 }
287
288 DoMeleeAttackIfReady();
289 }
290
291 private:
292 bool _setData;
293 EventMap _events;
294
295 };
296
297 CreatureAI* GetAI(Creature* creature) const override
298 {
299 return new npc_enslaved_proto_drakeAI(creature);
300 }

303class spell_uk_second_wind_proc : public SpellScriptLoader
304{
305 public:
306 spell_uk_second_wind_proc() : SpellScriptLoader("spell_uk_second_wind_proc") { }
307
308 class spell_uk_second_wind_proc_AuraScript : public AuraScript
309 {
310 PrepareAuraScript(spell_uk_second_wind_proc_AuraScript);
311
312 bool Validate(SpellInfo const* /*spellInfo*/) override
313 {
314 if (!sSpellMgr->GetSpellInfo(SPELL_UK_SECOUND_WIND_TRIGGER))
315 return false;
316 return true;
317 }
318
319 bool CheckProc(ProcEventInfo& eventInfo)
320 {
321 if (eventInfo.GetProcTarget() == GetTarget())
322 return false;
323 if (!(eventInfo.GetDamageInfo() || eventInfo.GetDamageInfo()->GetSpellInfo()->GetAllEffectsMechanicMask() & ((1 << MECHANIC_ROOT) | (1 << MECHANIC_STUN))))
324 return false;
325 return true;
326 }
327
328 void HandleProc(AuraEffect const* aurEff, ProcEventInfo& /*eventInfo*/)
329 {
330 PreventDefaultAction();
331 GetTarget()->CastCustomSpell(SPELL_UK_SECOUND_WIND_TRIGGER, SPELLVALUE_BASE_POINT0, 5, GetTarget(), true, NULL, aurEff);
332 }
333
334 void Register() override
335 {
336 DoCheckProc += AuraCheckProcFn(spell_uk_second_wind_proc_AuraScript::CheckProc);
337 OnEffectProc += AuraEffectProcFn(spell_uk_second_wind_proc_AuraScript::HandleProc, EFFECT_0, SPELL_AURA_DUMMY);
338 }
339
340 };
341
342 AuraScript* GetAuraScript() const override
343 {
344 return new spell_uk_second_wind_proc_AuraScript();
345 }

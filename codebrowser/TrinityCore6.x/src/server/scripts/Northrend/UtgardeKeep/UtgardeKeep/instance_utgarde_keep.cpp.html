35class instance_utgarde_keep : public InstanceMapScript
36{
37 public:
38 instance_utgarde_keep() : InstanceMapScript(UKScriptName, 574) { }
39
40 struct instance_utgarde_keep_InstanceMapScript : public InstanceScript
41 {
42 instance_utgarde_keep_InstanceMapScript(Map* map) : InstanceScript(map)
43 {
44 SetHeaders(DataHeader);
45 SetBossNumber(EncounterCount);
46 LoadDoorData(doorData);
47 LoadMinionData(minionData);
48 }
49
50 void OnCreatureCreate(Creature* creature) override
51 {
52 switch (creature->GetEntry())
53 {
54 case NPC_PRINCE_KELESETH:
55 PrinceKelesethGUID = creature->GetGUID();
56 break;
57 case NPC_SKARVALD:
58 SkarvaldGUID = creature->GetGUID();
59 AddMinion(creature, true);
60 break;
61 case NPC_DALRONN:
62 DalronnGUID = creature->GetGUID();
63 AddMinion(creature, true);
64 break;
65 case NPC_INGVAR:
66 IngvarGUID = creature->GetGUID();
67 break;
68 default:
69 break;
70 }
71 }
72
73 void OnCreatureRemove(Creature* creature) override
74 {
75 switch (creature->GetEntry())
76 {
77 case NPC_SKARVALD:
78 case NPC_DALRONN:
79 AddMinion(creature, false);
80 break;
81 default:
82 break;
83 }
84 }
85
86 void OnGameObjectCreate(GameObject* go) override
87 {
88 switch (go->GetEntry())
89 {
90 case GO_BELLOW_1:
91 Forges[0].BellowGUID = go->GetGUID();
92 HandleGameObject(ObjectGuid::Empty, Forges[0].Event != NOT_STARTED, go);
93 break;
94 case GO_BELLOW_2:
95 Forges[1].BellowGUID = go->GetGUID();
96 HandleGameObject(ObjectGuid::Empty, Forges[1].Event != NOT_STARTED, go);
97 break;
98 case GO_BELLOW_3:
99 Forges[2].BellowGUID = go->GetGUID();
100 HandleGameObject(ObjectGuid::Empty, Forges[2].Event != NOT_STARTED, go);
101 break;
102 case GO_FORGEFIRE_1:
103 Forges[0].FireGUID = go->GetGUID();
104 HandleGameObject(ObjectGuid::Empty, Forges[0].Event != NOT_STARTED, go);
105 break;
106 case GO_FORGEFIRE_2:
107 Forges[1].FireGUID = go->GetGUID();
108 HandleGameObject(ObjectGuid::Empty, Forges[1].Event != NOT_STARTED, go);
109 break;
110 case GO_FORGEFIRE_3:
111 Forges[2].FireGUID = go->GetGUID();
112 HandleGameObject(ObjectGuid::Empty, Forges[2].Event != NOT_STARTED, go);
113 break;
114 case GO_GLOWING_ANVIL_1:
115 Forges[0].AnvilGUID = go->GetGUID();
116 HandleGameObject(ObjectGuid::Empty, Forges[0].Event != NOT_STARTED, go);
117 break;
118 case GO_GLOWING_ANVIL_2:
119 Forges[1].AnvilGUID = go->GetGUID();
120 HandleGameObject(ObjectGuid::Empty, Forges[1].Event != NOT_STARTED, go);
121 break;
122 case GO_GLOWING_ANVIL_3:
123 Forges[2].AnvilGUID = go->GetGUID();
124 HandleGameObject(ObjectGuid::Empty, Forges[2].Event != NOT_STARTED, go);
125 break;
126 case GO_GIANT_PORTCULLIS_1:
127 case GO_GIANT_PORTCULLIS_2:
128 AddDoor(go, true);
129 break;
130 default:
131 break;
132 }
133 }
134
135 void OnGameObjectRemove(GameObject* go) override
136 {
137 switch (go->GetEntry())
138 {
139 case GO_GIANT_PORTCULLIS_1:
140 case GO_GIANT_PORTCULLIS_2:
141 AddDoor(go, false);
142 break;
143 default:
144 break;
145 }
146 }
147
148 ObjectGuid GetGuidData(uint32 type) const override
149 {
150 switch (type)
151 {
152 case DATA_PRINCE_KELESETH:
153 return PrinceKelesethGUID;
154 case DATA_SKARVALD:
155 return SkarvaldGUID;
156 case DATA_DALRONN:
157 return DalronnGUID;
158 case DATA_INGVAR:
159 return IngvarGUID;
160 default:
161 break;
162 }
163
164 return ObjectGuid::Empty;
165 }
166
167 void SetData(uint32 type, uint32 data) override
168 {
169 switch (type)
170 {
171 case DATA_FORGE_1:
172 case DATA_FORGE_2:
173 case DATA_FORGE_3:
174 {
175 uint8 i = type - DATA_FORGE_1;
176 HandleGameObject(Forges[i].AnvilGUID, data != NOT_STARTED);
177 HandleGameObject(Forges[i].BellowGUID, data != NOT_STARTED);
178 HandleGameObject(Forges[i].FireGUID, data != NOT_STARTED);
179 Forges[i].Event = data;
180
181 if (data == DONE)
182 SaveToDB();
183 break;
184 }
185 default:
186 break;
187 }
188 }
189
190 void WriteSaveDataMore(std::ostringstream& data) override
191 {
192 for (uint8 i = 0; i < 3; ++i)
193 data << Forges[i].Event << ' ';
194 }
195
196 void ReadSaveDataMore(std::istringstream& data) override
197 {
198 for (uint8 i = 0; i < 3; ++i)
199 data >> Forges[i].Event;
200 }
201
202 protected:
203 ForgeInfo Forges[3];
204
205 ObjectGuid PrinceKelesethGUID;
206 ObjectGuid SkarvaldGUID;
207 ObjectGuid DalronnGUID;
208 ObjectGuid IngvarGUID;
209 };
210
211 InstanceScript* GetInstanceScript(InstanceMap* map) const override
212 {
213 return new instance_utgarde_keep_InstanceMapScript(map);
214 }
215};

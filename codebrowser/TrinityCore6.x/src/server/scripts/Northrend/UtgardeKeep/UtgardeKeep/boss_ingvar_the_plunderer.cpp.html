100class boss_ingvar_the_plunderer : public CreatureScript
101{
102 public:
103 boss_ingvar_the_plunderer() : CreatureScript("boss_ingvar_the_plunderer") { }
104
105 struct boss_ingvar_the_plundererAI : public BossAI
106 {
107 boss_ingvar_the_plundererAI(Creature* creature) : BossAI(creature, DATA_INGVAR) { }
108
109 void Reset() override
110 {
111 if (me->GetEntry() != NPC_INGVAR)
112 me->UpdateEntry(NPC_INGVAR);
113
114 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
115
116 _Reset();
117 events.SetPhase(PHASE_HUMAN);
118
119 events.ScheduleEvent(EVENT_CLEAVE, urand(6, 12)*IN_MILLISECONDS, 0, PHASE_HUMAN);
120 events.ScheduleEvent(EVENT_STAGGERING_ROAR, urand(18, 21)*IN_MILLISECONDS, 0, PHASE_HUMAN);
121 events.ScheduleEvent(EVENT_ENRAGE, urand(7, 14)*IN_MILLISECONDS, 0, PHASE_HUMAN);
122 events.ScheduleEvent(EVENT_SMASH, urand(12, 17)*IN_MILLISECONDS, 0, PHASE_HUMAN);
123 }
124
125 void DamageTaken(Unit* /*doneBy*/, uint32& damage) override
126 {
127 if (damage >= me->GetHealth() && events.IsInPhase(PHASE_HUMAN))
128 {
129 me->RemoveAllAuras();
130 DoCast(me, SPELL_INGVAR_FEIGN_DEATH, true);
131
132 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
133
134 events.SetPhase(PHASE_EVENT);
135 events.ScheduleEvent(EVENT_SUMMON_BANSHEE, 3 * IN_MILLISECONDS, 0, PHASE_EVENT);
136
137 Talk(SAY_DEATH);
138 }
139
140 if (events.IsInPhase(PHASE_EVENT))
141 damage = 0;
142 }
143
144 void DoAction(int32 actionId) override
145 {
146 if (actionId == ACTION_START_PHASE_2)
147 StartZombiePhase();
148 }
149
150 void StartZombiePhase()
151 {
152 me->RemoveAura(SPELL_INGVAR_FEIGN_DEATH);
153 DoCast(me, SPELL_INGVAR_TRANSFORM, true);
154 me->UpdateEntry(NPC_INGVAR_UNDEAD);
155 events.ScheduleEvent(EVENT_JUST_TRANSFORMED, 2 * IN_MILLISECONDS, 0, PHASE_EVENT);
156 }
157
158 void EnterCombat(Unit* /*who*/) override
159 {
160 _EnterCombat();
161 Talk(SAY_AGGRO);
162 }
163
164 void JustDied(Unit* /*killer*/) override
165 {
166 _JustDied();
167 Talk(SAY_DEATH);
168 }
169
170 void ScheduleSecondPhase()
171 {
172 events.SetPhase(PHASE_UNDEAD);
173 events.ScheduleEvent(EVENT_DARK_SMASH, urand(14, 18)*IN_MILLISECONDS, 0, PHASE_UNDEAD);
174 events.ScheduleEvent(EVENT_DREADFUL_ROAR, urand(18, 22)*IN_MILLISECONDS, 0, PHASE_UNDEAD);
175 events.ScheduleEvent(EVENT_WOE_STRIKE, urand(10, 14)*IN_MILLISECONDS, 0, PHASE_UNDEAD);
176 events.ScheduleEvent(EVENT_SHADOW_AXE, 30*IN_MILLISECONDS, 0, PHASE_UNDEAD);
177 }
178
179 void KilledUnit(Unit* who) override
180 {
181 if (who->GetTypeId() == TYPEID_PLAYER)
182 Talk(SAY_SLAY);
183 }
184
185 void UpdateAI(uint32 diff) override
186 {
187 if (!UpdateVictim() && !events.IsInPhase(PHASE_EVENT))
188 return;
189
190 events.Update(diff);
191
192 if (me->HasUnitState(UNIT_STATE_CASTING))
193 return;
194
195 while (uint32 eventId = events.ExecuteEvent())
196 {
197 switch (eventId)
198 {
199 // PHASE ONE
200 case EVENT_CLEAVE:
201 DoCastVictim(SPELL_CLEAVE);
202 events.ScheduleEvent(EVENT_CLEAVE, urand(6, 12)*IN_MILLISECONDS, 0, PHASE_HUMAN);
203 break;
204 case EVENT_STAGGERING_ROAR:
205 DoCast(me, SPELL_STAGGERING_ROAR);
206 events.ScheduleEvent(EVENT_STAGGERING_ROAR, urand(18, 22)*IN_MILLISECONDS, 0, PHASE_HUMAN);
207 break;
208 case EVENT_ENRAGE:
209 DoCast(me, SPELL_ENRAGE);
210 events.ScheduleEvent(EVENT_ENRAGE, urand(7, 14)*IN_MILLISECONDS, 0, PHASE_HUMAN);
211 break;
212 case EVENT_SMASH:
213 DoCastAOE(SPELL_SMASH);
214 events.ScheduleEvent(EVENT_SMASH, urand(12, 16)*IN_MILLISECONDS, 0, PHASE_HUMAN);
215 break;
216 case EVENT_JUST_TRANSFORMED:
217 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
218 DoZoneInCombat();
219 ScheduleSecondPhase();
220 return;
221 case EVENT_SUMMON_BANSHEE:
222 DoCast(me, SPELL_SUMMON_BANSHEE);
223 return;
224 // PHASE TWO
225 case EVENT_DARK_SMASH:
226 DoCastVictim(SPELL_DARK_SMASH);
227 events.ScheduleEvent(EVENT_DARK_SMASH, urand(12, 16)*IN_MILLISECONDS, 0, PHASE_UNDEAD);
228 break;
229 case EVENT_DREADFUL_ROAR:
230 DoCast(me, SPELL_DREADFUL_ROAR);
231 events.ScheduleEvent(EVENT_DREADFUL_ROAR, urand(18, 22)*IN_MILLISECONDS, 0, PHASE_UNDEAD);
232 break;
233 case EVENT_WOE_STRIKE:
234 DoCastVictim(SPELL_WOE_STRIKE);
235 events.ScheduleEvent(EVENT_WOE_STRIKE, urand(10, 14)*IN_MILLISECONDS, 0, PHASE_UNDEAD);
236 break;
237 case EVENT_SHADOW_AXE:
238 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
239 DoCast(target, SPELL_SHADOW_AXE_SUMMON);
240 events.ScheduleEvent(EVENT_SHADOW_AXE, 30*IN_MILLISECONDS, 0, PHASE_UNDEAD);
241 break;
242 default:
243 break;
244 }
245 }
246
247 if (!events.IsInPhase(PHASE_EVENT))
248 DoMeleeAttackIfReady();
249 }
250 };
251
252 CreatureAI* GetAI(Creature* creature) const override
253 {
254 return GetUtgardeKeepAI<boss_ingvar_the_plundererAI>(creature);
255 }

258class npc_annhylde_the_caller : public CreatureScript
259{
260 public:
261 npc_annhylde_the_caller() : CreatureScript("npc_annhylde_the_caller") { }
262
263 struct npc_annhylde_the_callerAI : public ScriptedAI
264 {
265 npc_annhylde_the_callerAI(Creature* creature) : ScriptedAI(creature)
266 {
267 x = 0.f;
268 y = 0.f;
269 z = 0.f;
270 _instance = creature->GetInstanceScript();
271 }
272
273 void Reset() override
274 {
275 _events.Reset();
276
277 me->GetPosition(x, y, z);
278 me->GetMotionMaster()->MovePoint(1, x, y, z - 15.0f);
279 }
280
281 void MovementInform(uint32 type, uint32 id) override
282 {
283 if (type != POINT_MOTION_TYPE)
284 return;
285
286 switch (id)
287 {
288 case 1:
289 Talk(YELL_RESURRECT);
290 if (Creature* ingvar = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_INGVAR)))
291 {
292 ingvar->RemoveAura(SPELL_SUMMON_BANSHEE);
293 ingvar->CastSpell(ingvar, SPELL_SCOURG_RESURRECTION_DUMMY, true);
294 DoCast(ingvar, SPELL_SCOURG_RESURRECTION_BEAM);
295 }
296 _events.ScheduleEvent(EVENT_RESURRECT_1, 8000);
297 break;
298 case 2:
299 me->DespawnOrUnsummon();
300 break;
301 default:
302 break;
303 }
304 }
305
306 void AttackStart(Unit* /*who*/) override { }
307 void MoveInLineOfSight(Unit* /*who*/) override { }
308 void EnterCombat(Unit* /*who*/) override { }
309
310 void UpdateAI(uint32 diff) override
311 {
312 _events.Update(diff);
313
314 while (uint32 eventId = _events.ExecuteEvent())
315 {
316 switch (eventId)
317 {
318 case EVENT_RESURRECT_1:
319 if (Creature* ingvar = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_INGVAR)))
320 {
321 ingvar->RemoveAura(SPELL_INGVAR_FEIGN_DEATH);
322 ingvar->CastSpell(ingvar, SPELL_SCOURG_RESURRECTION_HEAL, false);
323 }
324 _events.ScheduleEvent(EVENT_RESURRECT_2, 3000);
325 break;
326 case EVENT_RESURRECT_2:
327 if (Creature* ingvar = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_INGVAR)))
328 {
329 ingvar->RemoveAurasDueToSpell(SPELL_SCOURG_RESURRECTION_DUMMY);
330 ingvar->AI()->DoAction(ACTION_START_PHASE_2);
331 }
332
333 me->GetMotionMaster()->MovePoint(2, x, y, z + 15.0f);
334 break;
335 default:
336 break;
337 }
338 }
339 }
340
341 private:
342 InstanceScript* _instance;
343 EventMap _events;
344 float x, y, z;
345 };
346
347 CreatureAI* GetAI(Creature* creature) const override
348 {
349 return GetUtgardeKeepAI<npc_annhylde_the_callerAI>(creature);
350 }

353class npc_ingvar_throw_dummy : public CreatureScript
354{
355 public:
356 npc_ingvar_throw_dummy() : CreatureScript("npc_ingvar_throw_dummy") { }
357
358 struct npc_ingvar_throw_dummyAI : public ScriptedAI
359 {
360 npc_ingvar_throw_dummyAI(Creature* creature) : ScriptedAI(creature) { }
361
362 void Reset() override
363 {
364 me->SetReactState(REACT_PASSIVE);
365
366 if (Creature* target = me->FindNearestCreature(NPC_THROW_TARGET, 200.0f))
367 {
368 float x, y, z;
369 target->GetPosition(x, y, z);
370 me->GetMotionMaster()->MoveCharge(x, y, z);
371 target->DespawnOrUnsummon();
372 }
373 else
374 me->DespawnOrUnsummon();
375 }
376
377 void MovementInform(uint32 type, uint32 id) override
378 {
379 if (type == EFFECT_MOTION_TYPE && id == EVENT_CHARGE)
380 {
381 me->CastSpell(me, SPELL_SHADOW_AXE_PERIODIC_DAMAGE, true);
382 me->DespawnOrUnsummon(10000);
383 }
384 }
385 };
386
387 CreatureAI* GetAI(Creature* creature) const override
388 {
389 return new npc_ingvar_throw_dummyAI(creature);
390 }

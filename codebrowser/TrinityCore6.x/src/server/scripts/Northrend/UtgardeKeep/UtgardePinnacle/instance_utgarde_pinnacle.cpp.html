22DoorData const doorData[] =
23{
24 { GO_SKADI_THE_RUTHLESS_DOOR, DATA_SKADI_THE_RUTHLESS, DOOR_TYPE_PASSAGE, BOUNDARY_W },
25 { GO_KING_YMIRON_DOOR, DATA_KING_YMIRON, DOOR_TYPE_PASSAGE, BOUNDARY_N },

29class instance_utgarde_pinnacle : public InstanceMapScript
30{
31 public:
32 instance_utgarde_pinnacle() : InstanceMapScript(UPScriptName, 575) { }
33
34 struct instance_utgarde_pinnacle_InstanceMapScript : public InstanceScript
35 {
36 instance_utgarde_pinnacle_InstanceMapScript(Map* map) : InstanceScript(map)
37 {
38 SetHeaders(DataHeader);
39 SetBossNumber(EncounterCount);
40 LoadDoorData(doorData);
41 }
42
43 void OnCreatureCreate(Creature* creature) override
44 {
45 switch (creature->GetEntry())
46 {
47 case NPC_SVALA_SORROWGRAVE:
48 SvalaSorrowgraveGUID = creature->GetGUID();
49 break;
50 case NPC_GORTOK_PALEHOOF:
51 GortokPalehoofGUID = creature->GetGUID();
52 break;
53 case NPC_SKADI_THE_RUTHLESS:
54 SkadiTheRuthlessGUID = creature->GetGUID();
55 break;
56 case NPC_KING_YMIRON:
57 KingYmironGUID = creature->GetGUID();
58 break;
59 case NPC_FRENZIED_WORGEN:
60 FrenziedWorgenGUID = creature->GetGUID();
61 break;
62 case NPC_RAVENOUS_FURBOLG:
63 RavenousFurbolgGUID = creature->GetGUID();
64 break;
65 case NPC_MASSIVE_JORMUNGAR:
66 MassiveJormungarGUID = creature->GetGUID();
67 break;
68 case NPC_FEROCIOUS_RHINO:
69 FerociousRhinoGUID = creature->GetGUID();
70 break;
71 case NPC_SVALA:
72 SvalaGUID = creature->GetGUID();
73 break;
74 case NPC_PALEHOOF_ORB:
75 PalehoofOrbGUID = creature->GetGUID();
76 break;
77 default:
78 break;
79 }
80 }
81
82 void OnGameObjectCreate(GameObject* go) override
83 {
84 switch (go->GetEntry())
85 {
86 case GO_UTGARDE_MIRROR:
87 UtgardeMirrorGUID = go->GetGUID();
88 break;
89 case GO_GORTOK_PALEHOOF_SPHERE:
90 GortokPalehoofSphereGUID = go->GetGUID();
91 if (GetBossState(DATA_GORTOK_PALEHOOF) == DONE)
92 {
93 HandleGameObject(ObjectGuid::Empty, true, go);
94 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
95 }
96 break;
97 case GO_SKADI_THE_RUTHLESS_DOOR:
98 case GO_KING_YMIRON_DOOR:
99 AddDoor(go, true);
100 break;
101 default:
102 break;
103 }
104 }
105
106 void OnGameObjectRemove(GameObject* go) override
107 {
108 switch (go->GetEntry())
109 {
110 case GO_SKADI_THE_RUTHLESS_DOOR:
111 case GO_KING_YMIRON_DOOR:
112 AddDoor(go, false);
113 break;
114 default:
115 break;
116 }
117 }
118
119 void SetGuidData(uint32 type, ObjectGuid data) override
120 {
121 switch (type)
122 {
123 case DATA_SACRIFICED_PLAYER:
124 SacrificedPlayerGUID = data;
125 break;
126 default:
127 break;
128 }
129 }
130
131 ObjectGuid GetGuidData(uint32 type) const override
132 {
133 switch (type)
134 {
135 case DATA_SVALA_SORROWGRAVE:
136 return SvalaSorrowgraveGUID;
137 case DATA_GORTOK_PALEHOOF:
138 return GortokPalehoofGUID;
139 case DATA_SKADI_THE_RUTHLESS:
140 return SkadiTheRuthlessGUID;
141 case DATA_KING_YMIRON:
142 return KingYmironGUID;
143 case DATA_FRENZIED_WORGEN:
144 return FrenziedWorgenGUID;
145 case DATA_RAVENOUS_FURBOLG:
146 return RavenousFurbolgGUID;
147 case DATA_MASSIVE_JORMUNGAR:
148 return MassiveJormungarGUID;
149 case DATA_FEROCIOUS_RHINO:
150 return FerociousRhinoGUID;
151 case DATA_GORTOK_ORB:
152 return PalehoofOrbGUID;
153 case DATA_GORTOK_PALEHOOF_SPHERE:
154 return GortokPalehoofSphereGUID;
155 case DATA_UTGARDE_MIRROR:
156 return UtgardeMirrorGUID;
157 case DATA_SVALA:
158 return SvalaGUID;
159 case DATA_SACRIFICED_PLAYER:
160 return SacrificedPlayerGUID;
161 default:
162 break;
163 }
164
165 return ObjectGuid::Empty;
166 }
167
168 protected:
169 ObjectGuid SvalaSorrowgraveGUID;
170 ObjectGuid GortokPalehoofGUID;
171 ObjectGuid SkadiTheRuthlessGUID;
172 ObjectGuid KingYmironGUID;

180 ObjectGuid MassiveJormungarGUID;
181
182 ObjectGuid PalehoofOrbGUID;
183
184 ObjectGuid SvalaGUID;
185 ObjectGuid SacrificedPlayerGUID;
186 };

192};

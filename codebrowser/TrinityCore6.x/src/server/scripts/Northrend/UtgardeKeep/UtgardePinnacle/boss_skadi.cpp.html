207 void Reset() override
208 {
209 Initialize();
210
211 Summons.DespawnAll();
212 me->SetSpeed(MOVE_FLIGHT, 3.0f);

247 void JustSummoned(Creature* summoned) override
248 {
249 switch (summoned->GetEntry())
250 {
251 case NPC_GRAUF:
252 m_uiGraufGUID = summoned->GetGUID();
253 break;
254 case NPC_YMIRJAR_WARRIOR:
255 case NPC_YMIRJAR_WITCH_DOCTOR:
256 case NPC_YMIRJAR_HARPOONER:
257 summoned->setActive(true);
258 summoned->SetInCombatWithZone();
259 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
260 summoned->AI()->AttackStart(target);
261 break;
262 case NPC_TRIGGER:
263 summoned->CastSpell((Unit*)NULL, SPELL_FREEZING_CLOUD, true);
264 summoned->DespawnOrUnsummon(10*IN_MILLISECONDS);
265 break;
266 }
267 Summons.Summon(summoned);
268 }

270 void SummonedCreatureDespawn(Creature* summoned) override
271 {
272 if (summoned->GetEntry() == NPC_GRAUF)
273 m_uiGraufGUID.Clear();
274 Summons.Despawn(summoned);
275 }

277 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
278 {
279 if (spell->Id == SPELL_HARPOON_DAMAGE)
280 {
281 m_uiSpellHitCount++;
282 if (m_uiSpellHitCount >= 3)
283 {
284 Phase = SKADI;
285 me->SetCanFly(false);
286 me->Dismount();
287 if (Creature* pGrauf = me->SummonCreature(NPC_GRAUF, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 3*IN_MILLISECONDS))
288 {
289 pGrauf->GetMotionMaster()->MoveFall();
290 pGrauf->HandleEmoteCommand(EMOTE_ONESHOT_FLYDEATH);
291 }
292 me->GetMotionMaster()->MoveJump(Location[4].GetPositionX(), Location[4].GetPositionY(), Location[4].GetPositionZ(), 5.0f, 10.0f);
293 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
294 Talk(SAY_DRAKE_DEATH);
295 m_uiCrushTimer = 8000;
296 m_uiPoisonedSpearTimer = 10000;
297 m_uiWhirlwindTimer = 20000;
298 AttackStart(SelectTarget(SELECT_TARGET_RANDOM));
299 }
300 }

303 void UpdateAI(uint32 diff) override
304 {
305 switch (Phase)
306 {
307 case FLYING:
308 if (!UpdateVictim())
309 return;
310
311 if (me->GetPositionX() >= 519)
312 {
313 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
314 if (!m_bSaidEmote)
315 {
316 Talk(EMOTE_RANGE);
317 m_bSaidEmote = true;
318 }
319 }
320 else
321 {
322 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
323 m_bSaidEmote = false;
324 }
325
326 if (m_uiMountTimer && m_uiMountTimer <= diff)
327 {
328 me->Mount(DATA_MOUNT);
329 me->SetCanFly(true);
330 m_uiMountTimer = 0;
331 } else m_uiMountTimer -= diff;
332
333 if (m_uiSummonTimer <= diff)
334 {
335 SpawnMobs();
336 m_uiSummonTimer = 25000;
337 } else m_uiSummonTimer -= diff;
338
339 if (m_uiMovementTimer <= diff)
340 {
341 switch (m_uiWaypointId)
342 {
343 case 0:
344 me->GetMotionMaster()->MovePoint(0, Location[1].GetPositionX(), Location[1].GetPositionY(), Location[1].GetPositionZ());
345 m_uiMovementTimer = 5000;
346 break;
347 case 1:
348 me->GetMotionMaster()->MovePoint(0, Location[2].GetPositionX(), Location[2].GetPositionY(), Location[2].GetPositionZ());
349 m_uiMovementTimer = 2000;
350 break;
351 case 2:
352 me->GetMotionMaster()->MovePoint(0, Location[3].GetPositionX(), Location[3].GetPositionY(), Location[3].GetPositionZ());
353 m_uiMovementTimer = 15000;
354 break;
355 case 3:
356 me->GetMotionMaster()->MovePoint(0, Location[69].GetPositionX(), Location[69].GetPositionY(), Location[69].GetPositionZ());
357 Talk(SAY_DRAKE_BREATH);
358 Talk(EMOTE_BREATH);
359 m_uiMovementTimer = 2500;
360 break;
361 case 4:
362 me->GetMotionMaster()->MovePoint(0, Location[70].GetPositionX(), Location[70].GetPositionY(), Location[70].GetPositionZ());
363 m_uiMovementTimer = 2000;
364 SpawnTrigger();
365 break;
366 case 5:
367 me->GetMotionMaster()->MovePoint(0, Location[71].GetPositionX(), Location[71].GetPositionY(), Location[71].GetPositionZ());
368 m_uiMovementTimer = 3000;
369 break;
370 case 6:
371 me->GetMotionMaster()->MovePoint(0, Location[3].GetPositionX(), Location[3].GetPositionY(), Location[3].GetPositionZ());
372 m_uiWaypointId = 2;
373 m_uiMovementTimer = 15000;
374 break;
375 }
376 m_uiWaypointId++;
377 } else m_uiMovementTimer -= diff;
378 break;
379 case SKADI:
380 //Return since we have no target
381 if (!UpdateVictim())
382 return;
383
384 if (m_uiCrushTimer <= diff)
385 {
386 DoCastVictim(SPELL_CRUSH);
387 m_uiCrushTimer = 8000;
388 } else m_uiCrushTimer -= diff;
389
390 if (m_uiPoisonedSpearTimer <= diff)
391 {
392 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
393 DoCast(target, SPELL_POISONED_SPEAR);
394 m_uiPoisonedSpearTimer = 10000;
395 } else m_uiPoisonedSpearTimer -= diff;
396
397 if (m_uiWhirlwindTimer <= diff)
398 {
399 DoCastAOE(SPELL_WHIRLWIND);
400 m_uiWhirlwindTimer = 20000;
401 } else m_uiWhirlwindTimer -= diff;
402
403 DoMeleeAttackIfReady();
404 break;
405 }
406 }

421 void SpawnMobs()
422 {
423 for (uint8 i = 0; i < DUNGEON_MODE(5, 6); ++i)
424 {
425 switch (urand(0, 2))
426 {
427 case 0:
428 me->SummonCreature(NPC_YMIRJAR_WARRIOR, SpawnLoc.GetPositionX() + rand32() % 5, SpawnLoc.GetPositionY() + rand32() % 5, SpawnLoc.GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
429 break;
430
431 case 1:
432 me->SummonCreature(NPC_YMIRJAR_WITCH_DOCTOR, SpawnLoc.GetPositionX() + rand32() % 5, SpawnLoc.GetPositionY() + rand32() % 5, SpawnLoc.GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
433 break;
434
435 case 2:
436 me->SummonCreature(NPC_YMIRJAR_HARPOONER, SpawnLoc.GetPositionX() + rand32() % 5, SpawnLoc.GetPositionY() + rand32() % 5, SpawnLoc.GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
437 break;
438 }

442 void SpawnTrigger()
443 {
444 uint8 iStart = 0, iEnd = 0;
445 switch (urand(0, 1))
446 {
447 case 0:
448 iStart = 8;
449 iEnd = 37;
450 break;
451 case 1:
452 iStart = 38;
453 iEnd = 68;
454 break;
455 default:
456 break;
457 }
458 for (uint32 i = iStart; i < iEnd; ++i)
459 me->SummonCreature(NPC_TRIGGER, Location[i]);
460 }

58class npc_emily : public CreatureScript
59{
60public:
61 npc_emily() : CreatureScript("npc_emily") { }
62
63 struct npc_emilyAI : public npc_escortAI
64 {
65 npc_emilyAI(Creature* creature) : npc_escortAI(creature) { }
66
67 void JustSummoned(Creature* summoned) override
68 {
69 if (Creature* Mrfloppy = GetClosestCreatureWithEntry(me, NPC_MRFLOPPY, 50.0f))
70 summoned->AI()->AttackStart(Mrfloppy);
71 else
72 summoned->AI()->AttackStart(me->GetVictim());
73 }
74
75 void WaypointReached(uint32 waypointId) override
76 {
77 Player* player = GetPlayerForEscort();
78 if (!player)
79 return;
80
81 switch (waypointId)
82 {
83 case 9:
84 if (Creature* Mrfloppy = GetClosestCreatureWithEntry(me, NPC_MRFLOPPY, 100.0f))
85 _mrfloppyGUID = Mrfloppy->GetGUID();
86 break;
87 case 10:
88 if (ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
89 {
90 Talk(SAY_WORGHAGGRO1);
91 me->SummonCreature(NPC_HUNGRY_WORG, me->GetPositionX()+5, me->GetPositionY()+2, me->GetPositionZ()+1, 3.229f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 120000);
92 }
93 break;
94 case 11:
95 if (Creature* Mrfloppy = ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
96 Mrfloppy->GetMotionMaster()->MoveFollow(me, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
97 break;
98 case 17:
99 if (Creature* Mrfloppy = ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
100 Mrfloppy->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
101 Talk(SAY_WORGRAGGRO3);
102 if (Creature* RWORG = me->SummonCreature(NPC_RAVENOUS_WORG, me->GetPositionX()+10, me->GetPositionY()+8, me->GetPositionZ()+2, 3.229f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 120000))
103 {
104 RWORG->setFaction(35);
105 _RavenousworgGUID = RWORG->GetGUID();
106 }
107 break;
108 case 18:
109 if (Creature* Mrfloppy = ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
110 {
111 if (Creature* RWORG = ObjectAccessor::GetCreature(*me, _RavenousworgGUID))
112 RWORG->GetMotionMaster()->MovePoint(0, Mrfloppy->GetPositionX(), Mrfloppy->GetPositionY(), Mrfloppy->GetPositionZ());
113 DoCast(Mrfloppy, SPELL_MRFLOPPY);
114 }
115 break;
116 case 19:
117 if (Creature* Mrfloppy = ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
118 {
119 if (Mrfloppy->HasAura(SPELL_MRFLOPPY))
120 {
121 if (Creature* RWORG = ObjectAccessor::GetCreature(*me, _RavenousworgGUID))
122 Mrfloppy->EnterVehicle(RWORG);
123 }
124 }
125 break;
126 case 20:
127 if (Creature* RWORG = ObjectAccessor::GetCreature(*me, _RavenousworgGUID))
128 RWORG->HandleEmoteCommand(34);
129 break;
130 case 21:
131 if (Creature* Mrfloppy = ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
132 {
133 if (Creature* RWORG = ObjectAccessor::GetCreature(*me, _RavenousworgGUID))
134 {
135 RWORG->Kill(Mrfloppy);
136 Mrfloppy->ExitVehicle();
137 RWORG->setFaction(14);
138 RWORG->GetMotionMaster()->MovePoint(0, RWORG->GetPositionX()+10, RWORG->GetPositionY()+80, RWORG->GetPositionZ());
139 Talk(SAY_VICTORY2);
140 }
141 }
142 break;
143 case 22:
144 if (Creature* Mrfloppy = ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
145 {
146 if (Mrfloppy->isDead())
147 {
148 if (Creature* RWORG = ObjectAccessor::GetCreature(*me, _RavenousworgGUID))
149 RWORG->DisappearAndDie();
150 me->GetMotionMaster()->MovePoint(0, Mrfloppy->GetPositionX(), Mrfloppy->GetPositionY(), Mrfloppy->GetPositionZ());
151 Mrfloppy->setDeathState(ALIVE);
152 Mrfloppy->GetMotionMaster()->MoveFollow(me, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
153 Talk(SAY_VICTORY3);
154 }
155 }
156 break;
157 case 24:
158 if (player)
159 {
160 player->GroupEventHappens(QUEST_PERILOUS_ADVENTURE, me);
161 Talk(SAY_QUEST_COMPLETE, player);
162 }
163 me->SetWalk(false);
164 break;
165 case 25:
166 Talk(SAY_VICTORY4);
167 break;
168 case 27:
169 me->DisappearAndDie();
170 if (Creature* Mrfloppy = ObjectAccessor::GetCreature(*me, _mrfloppyGUID))
171 Mrfloppy->DisappearAndDie();
172 break;
173 }
174 }
175
176 void EnterCombat(Unit* /*Who*/) override
177 {
178 Talk(SAY_RANDOMAGGRO);
179 }
180
181 void Reset() override
182 {
183 _mrfloppyGUID.Clear();
184 _RavenousworgGUID.Clear();
185 }
186
187 private:
188 ObjectGuid _RavenousworgGUID;
189 ObjectGuid _mrfloppyGUID;
190 };

198 Mrfloppy->GetMotionMaster()->MoveFollow(creature, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
199

201 pEscortAI->Start(true, false, player->GetGUID());
202 }
203 return true;
204 }

735 void ReceiveEmote(Player* player, uint32 emote) override
736 {
737 if (_following || _runningScript)
738 return;
739
740 if (emote == TEXT_EMOTE_KISS && me->IsWithinDistInMap(player, 30.0f) && player->HasItemCount(ITEM_WARTS_B_GONE_LIP_BALM, 1, false))
741 {
742 if (!player->HasAura(SPELL_WARTSBGONE_LIP_BALM))
743 player->AddAura(SPELL_WARTS, player);

126class boss_tyrannus : public CreatureScript
127{
128 public:
129 boss_tyrannus() : CreatureScript("boss_tyrannus") { }
130
131 struct boss_tyrannusAI : public BossAI
132 {
133 boss_tyrannusAI(Creature* creature) : BossAI(creature, DATA_TYRANNUS)
134 {
135 }
136
137 void InitializeAI() override
138 {
139 if (instance->GetBossState(DATA_TYRANNUS) != DONE)
140 Reset();
141 else
142 me->DespawnOrUnsummon();
143 }
144
145 void Reset() override
146 {
147 events.Reset();
148 events.SetPhase(PHASE_NONE);
149 me->SetReactState(REACT_PASSIVE);
150 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
151 instance->SetBossState(DATA_TYRANNUS, NOT_STARTED);
152 }
153
154 Creature* GetRimefang()
155 {
156 return ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_RIMEFANG));
157 }
158
159 void EnterCombat(Unit* /*who*/) override
160 {
161 Talk(SAY_AGGRO);
162 }
163
164 void AttackStart(Unit* victim) override
165 {
166 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
167 return;
168
169 if (victim && me->Attack(victim, true) && !events.IsInPhase(PHASE_INTRO))
170 me->GetMotionMaster()->MoveChase(victim);
171 }
172
173 void EnterEvadeMode() override
174 {
175 instance->SetBossState(DATA_TYRANNUS, FAIL);
176 if (Creature* rimefang = GetRimefang())
177 rimefang->AI()->EnterEvadeMode();
178
179 me->DespawnOrUnsummon();
180 }
181
182 void KilledUnit(Unit* victim) override
183 {
184 if (victim->GetTypeId() == TYPEID_PLAYER)
185 Talk(SAY_SLAY);
186 }
187
188 void JustDied(Unit* /*killer*/) override
189 {
190 Talk(SAY_DEATH);
191 instance->SetBossState(DATA_TYRANNUS, DONE);
192
193 // Prevent corpse despawning
194 if (TempSummon* summ = me->ToTempSummon())
195 summ->SetTempSummonType(TEMPSUMMON_DEAD_DESPAWN);
196
197 // Stop combat for Rimefang
198 if (Creature* rimefang = GetRimefang())
199 rimefang->AI()->DoAction(ACTION_END_COMBAT);
200 }
201
202 void DoAction(int32 actionId) override
203 {
204 if (actionId == ACTION_START_INTRO)
205 {
206 Talk(SAY_TYRANNUS_INTRO_1);
207 events.SetPhase(PHASE_INTRO);
208 events.ScheduleEvent(EVENT_INTRO_1, 14000, 0, PHASE_INTRO);
209 events.ScheduleEvent(EVENT_INTRO_2, 22000, 0, PHASE_INTRO);
210 events.ScheduleEvent(EVENT_INTRO_3, 34000, 0, PHASE_INTRO);
211 events.ScheduleEvent(EVENT_COMBAT_START, 36000, 0, PHASE_INTRO);
212 instance->SetBossState(DATA_TYRANNUS, IN_PROGRESS);
213 }
214 }
215
216 void UpdateAI(uint32 diff) override
217 {
218 if (!UpdateVictim() && !events.IsInPhase(PHASE_INTRO))
219 return;
220
221 events.Update(diff);
222
223 while (uint32 eventId = events.ExecuteEvent())
224 {
225 switch (eventId)
226 {
227 case EVENT_INTRO_1:
228 //Talk(SAY_GORKUN_INTRO_2, pGorkunOrVictus);
229 break;
230 case EVENT_INTRO_2:
231 Talk(SAY_TYRANNUS_INTRO_3);
232 break;
233 case EVENT_INTRO_3:
234 me->ExitVehicle();
235 me->GetMotionMaster()->MovePoint(0, miscPos);
236 break;
237 case EVENT_COMBAT_START:
238 if (Creature* rimefang = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_RIMEFANG)))
239 rimefang->AI()->DoAction(ACTION_START_RIMEFANG); //set rimefang also infight
240 events.SetPhase(PHASE_COMBAT);
241 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
242 me->SetReactState(REACT_AGGRESSIVE);
243 DoCast(me, SPELL_FULL_HEAL);
244 DoZoneInCombat();
245 events.ScheduleEvent(EVENT_OVERLORD_BRAND, urand(5000, 7000));
246 events.ScheduleEvent(EVENT_FORCEFUL_SMASH, urand(14000, 16000));
247 events.ScheduleEvent(EVENT_MARK_OF_RIMEFANG, urand(25000, 27000));
248 break;
249 case EVENT_OVERLORD_BRAND:
250 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
251 DoCast(target, SPELL_OVERLORD_BRAND);
252 events.ScheduleEvent(EVENT_OVERLORD_BRAND, urand(11000, 12000));
253 break;
254 case EVENT_FORCEFUL_SMASH:
255 DoCastVictim(SPELL_FORCEFUL_SMASH);
256 events.ScheduleEvent(EVENT_UNHOLY_POWER, 1000);
257 break;
258 case EVENT_UNHOLY_POWER:
259 Talk(SAY_DARK_MIGHT_1);
260 Talk(SAY_DARK_MIGHT_2);
261 DoCast(me, SPELL_UNHOLY_POWER);
262 events.ScheduleEvent(EVENT_FORCEFUL_SMASH, urand(40000, 48000));
263 break;
264 case EVENT_MARK_OF_RIMEFANG:
265 Talk(SAY_MARK_RIMEFANG_1);
266 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
267 {
268 Talk(SAY_MARK_RIMEFANG_2, target);
269 DoCast(target, SPELL_MARK_OF_RIMEFANG);
270 }
271 events.ScheduleEvent(EVENT_MARK_OF_RIMEFANG, urand(24000, 26000));
272 break;
273 }
274 }
275
276 DoMeleeAttackIfReady();
277 }
278 };
279
280 CreatureAI* GetAI(Creature* creature) const override
281 {
282 return GetPitOfSaronAI<boss_tyrannusAI>(creature);
283 }

286class boss_rimefang : public CreatureScript
287{
288 public:
289 boss_rimefang() : CreatureScript("boss_rimefang") { }
290
291 struct boss_rimefangAI : public ScriptedAI
292 {
293 boss_rimefangAI(Creature* creature) : ScriptedAI(creature)
294 {
295 Initialize();
296 }
297
298 void Initialize()
299 {
300 _currentWaypoint = 0;
301 _hoarfrostTargetGUID.Clear();
302 }
303
304 void Reset() override
305 {
306 _events.Reset();
307 _events.SetPhase(PHASE_NONE);
308 Initialize();
309 me->SetCanFly(true);
310 me->SetReactState(REACT_PASSIVE);
311 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
312 }
313
314 void JustReachedHome() override
315 {
316 if (Vehicle* _vehicle = me->GetVehicleKit())
317 _vehicle->InstallAllAccessories(false);
318 }
319
320 void DoAction(int32 actionId) override
321 {
322 if (actionId == ACTION_START_RIMEFANG)
323 {
324 _events.SetPhase(PHASE_COMBAT);
325 DoZoneInCombat();
326 _events.ScheduleEvent(EVENT_MOVE_NEXT, 500, 0, PHASE_COMBAT);
327 _events.ScheduleEvent(EVENT_ICY_BLAST, 15000, 0, PHASE_COMBAT);
328 }
329 else if (actionId == ACTION_END_COMBAT)
330 _EnterEvadeMode();
331 }
332
333 void SetGUID(ObjectGuid guid, int32 type) override
334 {
335 if (type == GUID_HOARFROST)
336 {
337 _hoarfrostTargetGUID = guid;
338 _events.ScheduleEvent(EVENT_HOARFROST, 1000);
339 }
340 }
341
342 void UpdateAI(uint32 diff) override
343 {
344 if (!UpdateVictim() && !_events.IsInPhase(PHASE_COMBAT))
345 return;
346
347 _events.Update(diff);
348
349 while (uint32 eventId = _events.ExecuteEvent())
350 {
351 switch (eventId)
352 {
353 case EVENT_MOVE_NEXT:
354 if (_currentWaypoint >= 10 || _currentWaypoint == 0)
355 _currentWaypoint = 1;
356 me->GetMotionMaster()->MovePoint(0, rimefangPos[_currentWaypoint]);
357 ++_currentWaypoint;
358 _events.ScheduleEvent(EVENT_MOVE_NEXT, 2000, 0, PHASE_COMBAT);
359 break;
360 case EVENT_ICY_BLAST:
361 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
362 DoCast(target, SPELL_ICY_BLAST);
363 _events.ScheduleEvent(EVENT_ICY_BLAST, 15000, 0, PHASE_COMBAT);
364 break;
365 case EVENT_HOARFROST:
366 if (Unit* target = ObjectAccessor::GetUnit(*me, _hoarfrostTargetGUID))
367 {
368 DoCast(target, SPELL_HOARFROST);
369 _hoarfrostTargetGUID.Clear();
370 }
371 break;
372 default:
373 break;
374 }
375 }
376 }
377
378 private:
379 ObjectGuid _hoarfrostTargetGUID;
380 EventMap _events;
381 uint8 _currentWaypoint;
382 };
383
384 CreatureAI* GetAI(Creature* creature) const override
385 {
386 return new boss_rimefangAI(creature);
387 }

414class spell_tyrannus_overlord_brand : public SpellScriptLoader
415{
416 public:
417 spell_tyrannus_overlord_brand() : SpellScriptLoader("spell_tyrannus_overlord_brand") { }
418
419 class spell_tyrannus_overlord_brand_AuraScript : public AuraScript
420 {
421 PrepareAuraScript(spell_tyrannus_overlord_brand_AuraScript);
422
423 public:
424 spell_tyrannus_overlord_brand_AuraScript()
425 {
426 oldAI = nullptr;
427 oldAIState = false;
428 }
429
430 private:
431 bool Load() override
432 {
433 return GetCaster() && GetCaster()->GetEntry() == NPC_TYRANNUS;
434 }
435
436 void OnApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
437 {
438 if (GetTarget()->GetTypeId() != TYPEID_PLAYER)
439 return;
440
441 oldAI = GetTarget()->GetAI();
442 oldAIState = GetTarget()->IsAIEnabled;
443 GetTarget()->SetAI(new player_overlord_brandAI(GetTarget()->ToPlayer(), GetCasterGUID()));
444 GetTarget()->IsAIEnabled = true;
445 }
446
447 void OnRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
448 {
449 if (GetTarget()->GetTypeId() != TYPEID_PLAYER)
450 return;
451
452 GetTarget()->IsAIEnabled = oldAIState;
453 UnitAI* thisAI = GetTarget()->GetAI();
454 GetTarget()->SetAI(oldAI);
455 delete thisAI;
456 }
457
458 void Register() override
459 {
460 AfterEffectApply += AuraEffectApplyFn(spell_tyrannus_overlord_brand_AuraScript::OnApply, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
461 AfterEffectRemove += AuraEffectRemoveFn(spell_tyrannus_overlord_brand_AuraScript::OnRemove, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
462 }
463
464 UnitAI* oldAI;
465 bool oldAIState;
466 };
467
468 AuraScript* GetAuraScript() const override
469 {
470 return new spell_tyrannus_overlord_brand_AuraScript();
471 }

549 {
550 InstanceScript* instance = player->GetInstanceScript();
551 if (player->IsGameMaster() || !instance)
552 return false;
553
554 if (instance->GetBossState(DATA_TYRANNUS) != IN_PROGRESS && instance->GetBossState(DATA_TYRANNUS) != DONE)
555 if (Creature* tyrannus = ObjectAccessor::GetCreature(*player, instance->GetGuidData(DATA_TYRANNUS)))
556 {
557 tyrannus->AI()->DoAction(ACTION_START_INTRO);
558 return true;
559 }
560
561 return false;
562 }

133class boss_ick : public CreatureScript
134{
135 public:
136 boss_ick() : CreatureScript("boss_ick") { }
137
138 struct boss_ickAI : public BossAI
139 {
140 boss_ickAI(Creature* creature) : BossAI(creature, DATA_ICK)
141 {
142 _tempThreat = 0;
143 }
144
145 void Reset() override
146 {
147 events.Reset();
148 instance->SetBossState(DATA_ICK, NOT_STARTED);
149 }
150
151 Creature* GetKrick()
152 {
153 return ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_KRICK));
154 }
155
156 void EnterCombat(Unit* /*who*/) override
157 {
158 _EnterCombat();
159
160 if (Creature* krick = GetKrick())
161 krick->AI()->Talk(SAY_KRICK_AGGRO);
162
163 events.ScheduleEvent(EVENT_MIGHTY_KICK, 20000);
164 events.ScheduleEvent(EVENT_TOXIC_WASTE, 5000);
165 events.ScheduleEvent(EVENT_SHADOW_BOLT, 10000);
166 events.ScheduleEvent(EVENT_SPECIAL, urand(30000, 35000));
167 }
168
169 void EnterEvadeMode() override
170 {
171 me->GetMotionMaster()->Clear();
172 ScriptedAI::EnterEvadeMode();
173 }
174
175 void JustDied(Unit* /*killer*/) override
176 {
177 if (Creature* krick = GetKrick())
178 {
179 if (Vehicle* _vehicle = me->GetVehicleKit())
180 _vehicle->RemoveAllPassengers();
181 if (krick->AI())
182 krick->AI()->DoAction(ACTION_OUTRO);
183 }
184
185 instance->SetBossState(DATA_ICK, DONE);
186 }
187
188 void SetTempThreat(float threat)
189 {
190 _tempThreat = threat;
191 }
192
193 void _ResetThreat(Unit* target)
194 {
195 DoModifyThreatPercent(target, -100);
196 me->AddThreat(target, _tempThreat);
197 }
198
199 void UpdateAI(uint32 diff) override
200 {
201 if (!me->IsInCombat())
202 return;
203
204 if (!me->GetVictim() && me->getThreatManager().isThreatListEmpty())
205 {
206 EnterEvadeMode();
207 return;
208 }
209
210 events.Update(diff);
211
212 if (me->HasUnitState(UNIT_STATE_CASTING))
213 return;
214
215 while (uint32 eventId = events.ExecuteEvent())
216 {
217 switch (eventId)
218 {
219 case EVENT_TOXIC_WASTE:
220 if (Creature* krick = GetKrick())
221 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
222 krick->CastSpell(target, SPELL_TOXIC_WASTE);
223 events.ScheduleEvent(EVENT_TOXIC_WASTE, urand(7000, 10000));
224 break;
225 case EVENT_SHADOW_BOLT:
226 if (Creature* krick = GetKrick())
227 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
228 krick->CastSpell(target, SPELL_SHADOW_BOLT);
229 events.ScheduleEvent(EVENT_SHADOW_BOLT, 15000);
230 return;
231 case EVENT_MIGHTY_KICK:
232 DoCastVictim(SPELL_MIGHTY_KICK);
233 events.ScheduleEvent(EVENT_MIGHTY_KICK, 25000);
234 return;
235 case EVENT_SPECIAL:
236 //select one of these three special events
237 events.ScheduleEvent(RAND(EVENT_EXPLOSIVE_BARRAGE, EVENT_POISON_NOVA, EVENT_PURSUIT), 1000);
238 events.ScheduleEvent(EVENT_SPECIAL, urand(23000, 28000));
239 break;
240 case EVENT_EXPLOSIVE_BARRAGE:
241 if (Creature* krick = GetKrick())
242 {
243 krick->AI()->Talk(SAY_KRICK_BARRAGE_1);
244 krick->AI()->Talk(SAY_KRICK_BARRAGE_2);
245 krick->CastSpell(krick, SPELL_EXPLOSIVE_BARRAGE_KRICK, true);
246 DoCast(me, SPELL_EXPLOSIVE_BARRAGE_ICK);
247 }
248 events.DelayEvents(20000);
249 break;
250 case EVENT_POISON_NOVA:
251 if (Creature* krick = GetKrick())
252 krick->AI()->Talk(SAY_KRICK_POISON_NOVA);
253
254 Talk(SAY_ICK_POISON_NOVA);
255 DoCast(me, SPELL_POISON_NOVA);
256 break;
257 case EVENT_PURSUIT:
258 if (Creature* krick = GetKrick())
259 krick->AI()->Talk(SAY_KRICK_CHASE);
260 DoCast(me, SPELL_PURSUIT);
261 break;
262 default:
263 break;
264 }
265 }
266
267 DoMeleeAttackIfReady();
268 }
269
270 private:
271 float _tempThreat;
272 };
273
274 CreatureAI* GetAI(Creature* creature) const override
275 {
276 return GetPitOfSaronAI<boss_ickAI>(creature);
277 }

285 struct boss_krickAI : public ScriptedAI
286 {
287 boss_krickAI(Creature* creature) : ScriptedAI(creature), _instanceScript(creature->GetInstanceScript()), _summons(creature)
288 {
289 Initialize();
290 }
291
292 void Initialize()
293 {
294 _phase = PHASE_COMBAT;
295 _outroNpcGUID.Clear();
296 _tyrannusGUID.Clear();
297 }
298
299 void Reset() override
300 {
301 _events.Reset();
302 Initialize();
303
304 me->SetReactState(REACT_PASSIVE);
305 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
306 }
307
308 Creature* GetIck()
309 {
310 return ObjectAccessor::GetCreature(*me, _instanceScript->GetGuidData(DATA_ICK));
311 }
312
313 void KilledUnit(Unit* victim) override
314 {
315 if (victim->GetTypeId() != TYPEID_PLAYER)
316 return;
317
318 Talk(SAY_KRICK_SLAY);
319 }
320
321 void JustSummoned(Creature* summon) override
322 {
323 _summons.Summon(summon);
324 if (summon->GetEntry() == NPC_EXPLODING_ORB)
325 {
326 summon->CastSpell(summon, SPELL_EXPLODING_ORB, true);
327 summon->CastSpell(summon, SPELL_AUTO_GROW, true);
328 }
329 }
330
331 void DoAction(int32 actionId) override
332 {
333 if (actionId == ACTION_OUTRO)
334 {
335 Creature* tyrannusPtr = ObjectAccessor::GetCreature(*me, _instanceScript->GetGuidData(DATA_TYRANNUS_EVENT));
336 if (tyrannusPtr)
337 tyrannusPtr->NearTeleportTo(outroPos[1].GetPositionX(), outroPos[1].GetPositionY(), outroPos[1].GetPositionZ(), outroPos[1].GetOrientation());
338 else
339 tyrannusPtr = me->SummonCreature(NPC_TYRANNUS_EVENTS, outroPos[1], TEMPSUMMON_MANUAL_DESPAWN);
340
341 tyrannusPtr->SetCanFly(true);
342 me->GetMotionMaster()->MovePoint(POINT_KRICK_INTRO, outroPos[0].GetPositionX(), outroPos[0].GetPositionY(), outroPos[0].GetPositionZ());
343 tyrannusPtr->SetFacingToObject(me);
344 }
345 }
346
347 void MovementInform(uint32 type, uint32 id) override
348 {
349 if (type != POINT_MOTION_TYPE || id != POINT_KRICK_INTRO)
350 return;
351
352 Talk(SAY_KRICK_OUTRO_1);
353 _phase = PHASE_OUTRO;
354 _events.Reset();
355 _events.ScheduleEvent(EVENT_OUTRO_1, 1000);
356 }
357
358 void UpdateAI(uint32 diff) override
359 {
360 if (_phase != PHASE_OUTRO)
361 return;
362
363 _events.Update(diff);
364
365 while (uint32 eventId = _events.ExecuteEvent())
366 {
367 switch (eventId)
368 {
369 case EVENT_OUTRO_1:
370 {
371 if (Creature* temp = ObjectAccessor::GetCreature(*me, _instanceScript->GetGuidData(DATA_JAINA_SYLVANAS_1)))
372 temp->DespawnOrUnsummon();
373
374 Creature* jainaOrSylvanas = NULL;
375 if (_instanceScript->GetData(DATA_TEAM_IN_INSTANCE) == ALLIANCE)
376 jainaOrSylvanas = me->SummonCreature(NPC_JAINA_PART1, outroPos[2], TEMPSUMMON_MANUAL_DESPAWN);
377 else
378 jainaOrSylvanas = me->SummonCreature(NPC_SYLVANAS_PART1, outroPos[2], TEMPSUMMON_MANUAL_DESPAWN);
379
380 if (jainaOrSylvanas)
381 {
382 jainaOrSylvanas->GetMotionMaster()->MovePoint(0, outroPos[3]);
383 _outroNpcGUID = jainaOrSylvanas->GetGUID();
384 }
385 _events.ScheduleEvent(EVENT_OUTRO_2, 6000);
386 break;
387 }
388 case EVENT_OUTRO_2:
389 if (Creature* jainaOrSylvanas = ObjectAccessor::GetCreature(*me, _outroNpcGUID))
390 {
391 jainaOrSylvanas->SetFacingToObject(me);
392 me->SetFacingToObject(jainaOrSylvanas);
393 if (_instanceScript->GetData(DATA_TEAM_IN_INSTANCE) == ALLIANCE)
394 jainaOrSylvanas->AI()->Talk(SAY_JAYNA_OUTRO_2);
395 else
396 jainaOrSylvanas->AI()->Talk(SAY_SYLVANAS_OUTRO_2);
397 }
398 _events.ScheduleEvent(EVENT_OUTRO_3, 5000);
399 break;
400 case EVENT_OUTRO_3:
401 Talk(SAY_KRICK_OUTRO_3);
402 _events.ScheduleEvent(EVENT_OUTRO_4, 18000);
403 break;
404 case EVENT_OUTRO_4:
405 if (Creature* jainaOrSylvanas = ObjectAccessor::GetCreature(*me, _outroNpcGUID))
406 {
407 if (_instanceScript->GetData(DATA_TEAM_IN_INSTANCE) == ALLIANCE)
408 jainaOrSylvanas->AI()->Talk(SAY_JAYNA_OUTRO_4);
409 else
410 jainaOrSylvanas->AI()->Talk(SAY_SYLVANAS_OUTRO_4);
411 }
412 _events.ScheduleEvent(EVENT_OUTRO_5, 5000);
413 break;
414 case EVENT_OUTRO_5:
415 Talk(SAY_KRICK_OUTRO_5);
416 _events.ScheduleEvent(EVENT_OUTRO_6, 1000);
417 break;
418 case EVENT_OUTRO_6:
419 if (Creature* tyrannus = ObjectAccessor::GetCreature(*me, _instanceScript->GetGuidData(DATA_TYRANNUS_EVENT)))
420 {
421 tyrannus->SetSpeed(MOVE_FLIGHT, 3.5f, true);
422 tyrannus->GetMotionMaster()->MovePoint(1, outroPos[4]);
423 _tyrannusGUID = tyrannus->GetGUID();
424 }
425 _events.ScheduleEvent(EVENT_OUTRO_7, 5000);
426 break;
427 case EVENT_OUTRO_7:
428 if (Creature* tyrannus = ObjectAccessor::GetCreature(*me, _tyrannusGUID))
429 tyrannus->AI()->Talk(SAY_TYRANNUS_OUTRO_7);
430 _events.ScheduleEvent(EVENT_OUTRO_8, 5000);
431 break;
432 case EVENT_OUTRO_8:
433 //! HACK: Creature's can't have MOVEMENTFLAG_FLYING
434 me->AddUnitMovementFlag(MOVEMENTFLAG_FLYING);
435 me->GetMotionMaster()->MovePoint(0, outroPos[5]);
436 DoCast(me, SPELL_STRANGULATING);
437 _events.ScheduleEvent(EVENT_OUTRO_9, 2000);
438 break;
439 case EVENT_OUTRO_9:
440 Talk(SAY_KRICK_OUTRO_8);
441 /// @todo Tyrannus starts killing Krick.
442 // there shall be some visual spell effect
443 if (Creature* tyrannus = ObjectAccessor::GetCreature(*me, _tyrannusGUID))
444 tyrannus->CastSpell(me, SPELL_NECROMANTIC_POWER, true); //not sure if it's the right spell :/
445 _events.ScheduleEvent(EVENT_OUTRO_10, 1000);
446 break;
447 case EVENT_OUTRO_10:
448 //! HACK: Creature's can't have MOVEMENTFLAG_FLYING
449 me->RemoveUnitMovementFlag(MOVEMENTFLAG_FLYING);
450 me->AddUnitMovementFlag(MOVEMENTFLAG_FALLING_FAR);
451 me->GetMotionMaster()->MovePoint(0, outroPos[6]);
452 _events.ScheduleEvent(EVENT_OUTRO_11, 2000);
453 break;
454 case EVENT_OUTRO_11:
455 DoCast(me, SPELL_KRICK_KILL_CREDIT); // don't really know if we need it
456 me->SetStandState(UNIT_STAND_STATE_DEAD);
457 me->SetHealth(0);
458 _events.ScheduleEvent(EVENT_OUTRO_12, 3000);
459 break;
460 case EVENT_OUTRO_12:
461 if (Creature* tyrannus = ObjectAccessor::GetCreature(*me, _tyrannusGUID))
462 tyrannus->AI()->Talk(SAY_TYRANNUS_OUTRO_9);
463 _events.ScheduleEvent(EVENT_OUTRO_13, 2000);
464 break;
465 case EVENT_OUTRO_13:
466 if (Creature* jainaOrSylvanas = ObjectAccessor::GetCreature(*me, _outroNpcGUID))
467 {
468 if (_instanceScript->GetData(DATA_TEAM_IN_INSTANCE) == ALLIANCE)
469 jainaOrSylvanas->AI()->Talk(SAY_JAYNA_OUTRO_10);
470 else
471 jainaOrSylvanas->AI()->Talk(SAY_SYLVANAS_OUTRO_10);
472 }
473 // End of OUTRO. for now...
474 _events.ScheduleEvent(EVENT_OUTRO_END, 3000);
475 if (Creature* tyrannus = ObjectAccessor::GetCreature(*me, _tyrannusGUID))
476 tyrannus->GetMotionMaster()->MovePoint(0, outroPos[7]);
477 break;
478 case EVENT_OUTRO_END:
479 if (Creature* tyrannus = ObjectAccessor::GetCreature(*me, _tyrannusGUID))
480 tyrannus->DespawnOrUnsummon();
481
482 me->DisappearAndDie();
483 break;
484 default:
485 break;
486 }
487 }
488 }

494

496 ObjectGuid _outroNpcGUID;
497 ObjectGuid _tyrannusGUID;
498 };

526 }

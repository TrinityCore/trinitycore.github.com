25enum Spells
26{
27 SPELL_FIREBALL = 69583, //Ymirjar Flamebearer
28 SPELL_HELLFIRE = 69586,
29 SPELL_TACTICAL_BLINK = 69584,
30 SPELL_FROST_BREATH = 69527, //Iceborn Proto-Drake
31 SPELL_LEAPING_FACE_MAUL = 69504, // Geist Ambusher
32};

34enum Events
35{
36 // Ymirjar Flamebearer
37 EVENT_FIREBALL = 1,
38 EVENT_TACTICAL_BLINK = 2,
39};

41class npc_ymirjar_flamebearer : public CreatureScript
42{
43 public:
44 npc_ymirjar_flamebearer() : CreatureScript("npc_ymirjar_flamebearer") { }
45
46 struct npc_ymirjar_flamebearerAI: public ScriptedAI
47 {
48 npc_ymirjar_flamebearerAI(Creature* creature) : ScriptedAI(creature)
49 {
50 }
51
52 void Reset() override
53 {
54 _events.Reset();
55 }
56
57 void EnterCombat(Unit* /*who*/) override
58 {
59 _events.ScheduleEvent(EVENT_FIREBALL, 4000);
60 _events.ScheduleEvent(EVENT_TACTICAL_BLINK, 15000);
61 }
62
63 void UpdateAI(uint32 diff) override
64 {
65 if (!UpdateVictim())
66 return;
67
68 _events.Update(diff);
69
70 if (me->HasUnitState(UNIT_STATE_CASTING))
71 return;
72
73 while (uint32 eventId = _events.ExecuteEvent())
74 {
75 switch (eventId)
76 {
77 case EVENT_FIREBALL:
78 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
79 DoCast(target, SPELL_FIREBALL);
80 _events.RescheduleEvent(EVENT_FIREBALL, 5000);
81 break;
82 case EVENT_TACTICAL_BLINK:
83 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
84 DoCast(target, SPELL_TACTICAL_BLINK);
85 DoCast(me, SPELL_HELLFIRE);
86 _events.RescheduleEvent(EVENT_TACTICAL_BLINK, 12000);
87 break;
88 default:
89 break;
90 }
91 }
92
93 DoMeleeAttackIfReady();
94 }
95
96 private:
97 EventMap _events;
98 };
99
100 CreatureAI* GetAI(Creature* creature) const override
101 {
102 return new npc_ymirjar_flamebearerAI(creature);
103 }
104};

106class npc_iceborn_protodrake : public CreatureScript
107{
108 public:
109 npc_iceborn_protodrake() : CreatureScript("npc_iceborn_protodrake") { }
110
111 struct npc_iceborn_protodrakeAI: public ScriptedAI
112 {
113 npc_iceborn_protodrakeAI(Creature* creature) : ScriptedAI(creature)
114 {
115 Initialize();
116 }
117
118 void Initialize()
119 {
120 _frostBreathCooldown = 5000;
121 }
122
123 void Reset() override
124 {
125 Initialize();
126 }
127
128 void EnterCombat(Unit* /*who*/) override
129 {
130 if (Vehicle* _vehicle = me->GetVehicleKit())
131 _vehicle->RemoveAllPassengers();
132 }
133
134 void UpdateAI(uint32 diff) override
135 {
136 if (!UpdateVictim())
137 return;
138
139 if (_frostBreathCooldown < diff)
140 {
141 DoCastVictim(SPELL_FROST_BREATH);
142 _frostBreathCooldown = 10000;
143 }
144 else
145 _frostBreathCooldown -= diff;
146
147 DoMeleeAttackIfReady();
148 }
149
150 private:
151 uint32 _frostBreathCooldown;
152 };
153
154 CreatureAI* GetAI(Creature* creature) const override
155 {
156 return new npc_iceborn_protodrakeAI(creature);
157 }
158};

160class npc_geist_ambusher : public CreatureScript
161{
162 public:
163 npc_geist_ambusher() : CreatureScript("npc_geist_ambusher") { }
164
165 struct npc_geist_ambusherAI: public ScriptedAI
166 {
167 npc_geist_ambusherAI(Creature* creature) : ScriptedAI(creature)
168 {
169 Initialize();
170 }
171
172 void Initialize()
173 {
174 _leapingFaceMaulCooldown = 9000;
175 }
176
177 void Reset() override
178 {
179 Initialize();
180 }
181
182 void EnterCombat(Unit* who) override
183 {
184 if (who->GetTypeId() != TYPEID_PLAYER)
185 return;
186
187 // the max range is determined by aggro range
188 if (me->GetDistance(who) > 5.0f)
189 DoCast(who, SPELL_LEAPING_FACE_MAUL);
190 }
191
192 void UpdateAI(uint32 diff) override
193 {
194 if (!UpdateVictim())
195 return;
196
197 if (_leapingFaceMaulCooldown < diff)
198 {
199 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 5.0f, true))
200 DoCast(target, SPELL_LEAPING_FACE_MAUL);
201 _leapingFaceMaulCooldown = urand(9000, 14000);
202 }
203 else
204 _leapingFaceMaulCooldown -= diff;
205
206 DoMeleeAttackIfReady();
207 }
208
209 private:
210 uint32 _leapingFaceMaulCooldown;
211 };
212
213 CreatureAI* GetAI(Creature* creature) const override
214 {
215 return new npc_geist_ambusherAI(creature);
216 }
217};

219class spell_trash_npc_glacial_strike : public SpellScriptLoader
220{
221 public:
222 spell_trash_npc_glacial_strike() : SpellScriptLoader("spell_trash_npc_glacial_strike") { }
223
224 class spell_trash_npc_glacial_strike_AuraScript : public AuraScript
225 {
226 PrepareAuraScript(spell_trash_npc_glacial_strike_AuraScript);
227
228 void PeriodicTick(AuraEffect const* /*aurEff*/)
229 {
230 if (GetTarget()->IsFullHealth())
231 {
232 GetTarget()->RemoveAura(GetId(), ObjectGuid::Empty, 0, AURA_REMOVE_BY_ENEMY_SPELL);
233 PreventDefaultAction();
234 }
235 }
236
237 void Register() override
238 {
239 OnEffectPeriodic += AuraEffectPeriodicFn(spell_trash_npc_glacial_strike_AuraScript::PeriodicTick, EFFECT_2, SPELL_AURA_PERIODIC_DAMAGE_PERCENT);
240 }
241 };
242
243 AuraScript* GetAuraScript() const override
244 {
245 return new spell_trash_npc_glacial_strike_AuraScript();
246 }

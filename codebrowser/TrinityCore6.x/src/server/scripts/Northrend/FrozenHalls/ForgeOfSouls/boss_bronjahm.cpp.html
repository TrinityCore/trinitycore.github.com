67class boss_bronjahm : public CreatureScript
68{
69 public:
70 boss_bronjahm() : CreatureScript("boss_bronjahm") { }
71
72 struct boss_bronjahmAI : public BossAI
73 {
74 boss_bronjahmAI(Creature* creature) : BossAI(creature, DATA_BRONJAHM)
75 {
76 DoCast(me, SPELL_SOULSTORM_CHANNEL, true);
77 }
78
79 void Reset() override
80 {
81 _Reset();
82 events.SetPhase(PHASE_1);
83 events.ScheduleEvent(EVENT_SHADOW_BOLT, 2000);
84 events.ScheduleEvent(EVENT_MAGIC_BANE, urand(8000, 20000));
85 events.ScheduleEvent(EVENT_CORRUPT_SOUL, urand(25000, 35000), 0, PHASE_1);
86 }
87
88 void JustReachedHome() override
89 {
90 _JustReachedHome();
91 DoCast(me, SPELL_SOULSTORM_CHANNEL, true);
92 }
93
94 void EnterCombat(Unit* /*who*/) override
95 {
96 _EnterCombat();
97 Talk(SAY_AGGRO);
98 me->RemoveAurasDueToSpell(SPELL_SOULSTORM_CHANNEL);
99 }
100
101 void JustDied(Unit* /*killer*/) override
102 {
103 _JustDied();
104 Talk(SAY_DEATH);
105 }
106
107 void KilledUnit(Unit* who) override
108 {
109 if (who->GetTypeId() == TYPEID_PLAYER)
110 Talk(SAY_SLAY);
111 }
112
113 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
114 {
115 if (events.IsInPhase(PHASE_1) && !HealthAbovePct(30))
116 {
117 events.SetPhase(PHASE_2);
118 DoCast(me, SPELL_TELEPORT);
119 events.ScheduleEvent(EVENT_FEAR, urand(12000, 16000), 0, PHASE_2);
120 events.ScheduleEvent(EVENT_SOULSTORM, 100, 0, PHASE_2);
121 }
122 }
123
124 void JustSummoned(Creature* summon) override
125 {
126 if (summon->GetEntry() == NPC_CORRUPTED_SOUL_FRAGMENT)
127 {
128 summons.Summon(summon);
129 summon->SetReactState(REACT_PASSIVE);
130 summon->GetMotionMaster()->MoveFollow(me, me->GetObjectSize(), 0.0f);
131 summon->CastSpell(summon, SPELL_PURPLE_BANISH_VISUAL, true);
132 }
133 }
134
135 uint32 GetData(uint32 type) const override
136 {
137 if (type == DATA_SOUL_POWER)
138 {
139 uint32 count = 0;
140 for (ObjectGuid const& guid : summons)
141 {
142 if (Creature* summon = ObjectAccessor::GetCreature(*me, guid))
143 if (summon->GetEntry() == NPC_CORRUPTED_SOUL_FRAGMENT && summon->IsAlive())
144 ++count;
145 }
146 return count;
147 }
148
149 return 0;
150 }
151
152 void UpdateAI(uint32 diff) override
153 {
154 if (!UpdateVictim())
155 return;
156
157 events.Update(diff);
158
159 if (me->HasUnitState(UNIT_STATE_CASTING))
160 return;
161
162 while (uint32 eventId = events.ExecuteEvent())
163 {
164 switch (eventId)
165 {
166 case EVENT_MAGIC_BANE:
167 DoCastAOE(SPELL_MAGIC_S_BANE);
168 events.ScheduleEvent(EVENT_MAGIC_BANE, urand(8000, 20000));
169 break;
170 case EVENT_SHADOW_BOLT:
171 if (events.IsInPhase(PHASE_2))
172 {
173 DoCastVictim(SPELL_SHADOW_BOLT);
174 events.ScheduleEvent(EVENT_SHADOW_BOLT, urand(1, 2) * IN_MILLISECONDS);
175 }
176 else
177 {
178 if (!me->IsWithinMeleeRange(me->GetVictim()))
179 DoCastVictim(SPELL_SHADOW_BOLT);
180 events.ScheduleEvent(EVENT_SHADOW_BOLT, 2 * IN_MILLISECONDS);
181 }
182 break;
183 case EVENT_CORRUPT_SOUL:
184 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
185 {
186 Talk(SAY_CORRUPT_SOUL);
187 DoCast(target, SPELL_CORRUPT_SOUL);
188 }
189 events.ScheduleEvent(EVENT_CORRUPT_SOUL, urand(25000, 35000), 0, PHASE_1);
190 break;
191 case EVENT_SOULSTORM:
192 Talk(SAY_SOUL_STORM);
193 me->CastSpell(me, SPELL_SOULSTORM_VISUAL, true);
194 me->CastSpell(me, SPELL_SOULSTORM, false);
195 break;
196 case EVENT_FEAR:
197 me->CastCustomSpell(SPELL_FEAR, SPELLVALUE_MAX_TARGETS, 1, nullptr, false);
198 events.ScheduleEvent(EVENT_FEAR, urand(8000, 12000), 0, PHASE_2);
199 break;
200 default:
201 break;
202 }
203 }
204
205 if (!events.IsInPhase(PHASE_2))
206 DoMeleeAttackIfReady();
207 }
208 };
209
210 CreatureAI* GetAI(Creature* creature) const override
211 {
212 return GetInstanceAI<boss_bronjahmAI>(creature, FoSScriptName);
213 }

216class npc_corrupted_soul_fragment : public CreatureScript
217{
218 public:
219 npc_corrupted_soul_fragment() : CreatureScript("npc_corrupted_soul_fragment") { }
220
221 struct npc_corrupted_soul_fragmentAI : public ScriptedAI
222 {
223 npc_corrupted_soul_fragmentAI(Creature* creature) : ScriptedAI(creature)
224 {
225 instance = me->GetInstanceScript();
226 }
227
228 void IsSummonedBy(Unit* /*summoner*/) override
229 {
230 if (Creature* bronjahm = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BRONJAHM)))
231 bronjahm->AI()->JustSummoned(me);
232 }
233
234 void MovementInform(uint32 type, uint32 id) override
235 {
236 if (type != FOLLOW_MOTION_TYPE)
237 return;
238
239 if (instance->GetGuidData(DATA_BRONJAHM).GetCounter() != id)
240 return;
241
242 me->CastSpell((Unit*)nullptr, SPELL_CONSUME_SOUL, true);
243 me->DespawnOrUnsummon();
244 }
245
246 private:
247 InstanceScript* instance;
248 };
249
250 CreatureAI* GetAI(Creature* creature) const override
251 {
252 return GetInstanceAI<npc_corrupted_soul_fragmentAI>(creature, FoSScriptName);
253 }

256class spell_bronjahm_magic_bane : public SpellScriptLoader
257{
258 public:
259 spell_bronjahm_magic_bane() : SpellScriptLoader("spell_bronjahm_magic_bane") { }
260
261 class spell_bronjahm_magic_bane_SpellScript : public SpellScript
262 {
263 PrepareSpellScript(spell_bronjahm_magic_bane_SpellScript);
264
265 void RecalculateDamage()
266 {
267 if (GetHitUnit()->getPowerType() != POWER_MANA)
268 return;
269
270 int32 const maxDamage = GetCaster()->GetMap()->IsHeroic() ? 15000 : 10000;
271 int32 newDamage = GetHitDamage() + (GetHitUnit()->GetMaxPower(POWER_MANA) / 2);
272
273 SetHitDamage(std::min<int32>(maxDamage, newDamage));
274 }
275
276 void Register() override
277 {
278 OnHit += SpellHitFn(spell_bronjahm_magic_bane_SpellScript::RecalculateDamage);
279 }
280 };
281
282 SpellScript* GetSpellScript() const override
283 {
284 return new spell_bronjahm_magic_bane_SpellScript();
285 }

362
363 void FilterTargets(std::list<WorldObject*>& targets)
364 {
365 Unit* caster = GetCaster();
366 targets.remove_if([caster](WorldObject* target)
367 {
368 return caster->GetExactDist2d(target) <= 10.0f;
369 });
370 }
371
372 void Register() override
373 {
374 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_bronjahm_soulstorm_targeting_SpellScript::FilterTargets, EFFECT_ALL, TARGET_UNIT_DEST_AREA_ENEMY);
375 }
376 };

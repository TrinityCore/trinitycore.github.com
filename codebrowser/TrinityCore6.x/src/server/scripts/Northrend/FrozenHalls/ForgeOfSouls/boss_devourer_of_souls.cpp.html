129 struct boss_devourer_of_soulsAI : public BossAI
130 {
131 boss_devourer_of_soulsAI(Creature* creature) : BossAI(creature, DATA_DEVOURER_OF_SOULS)
132 {
133 Initialize();
134 beamAngle = 0.f;
135 beamAngleDiff = 0.f;
136 wailingSoulTick = 0;
137 }
138
139 void Initialize()
140 {
141 threeFaced = true;
142 }
143
144 void Reset() override
145 {
146 _Reset();
147 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
148 me->SetDisplayId(DISPLAY_ANGER);
149 me->SetReactState(REACT_AGGRESSIVE);
150
151 Initialize();
152 }
153
154 void EnterCombat(Unit* /*who*/) override
155 {
156 _EnterCombat();
157 Talk(SAY_FACE_AGGRO);
158
159 if (!me->FindNearestCreature(NPC_CRUCIBLE_OF_SOULS, 60)) // Prevent double spawn
160 instance->instance->SummonCreature(NPC_CRUCIBLE_OF_SOULS, CrucibleSummonPos);
161 events.ScheduleEvent(EVENT_PHANTOM_BLAST, 5000);
162 events.ScheduleEvent(EVENT_MIRRORED_SOUL, 8000);
163 events.ScheduleEvent(EVENT_WELL_OF_SOULS, 30000);
164 events.ScheduleEvent(EVENT_UNLEASHED_SOULS, 20000);
165 events.ScheduleEvent(EVENT_WAILING_SOULS, urand(60000, 70000));
166 }
167
168 void KilledUnit(Unit* victim) override
169 {
170 if (victim->GetTypeId() != TYPEID_PLAYER)
171 return;
172
173 uint8 textId = 0;
174 switch (me->GetDisplayId())
175 {
176 case DISPLAY_ANGER:
177 textId = SAY_FACE_ANGER_SLAY;
178 break;
179 case DISPLAY_SORROW:
180 textId = SAY_FACE_SORROW_SLAY;
181 break;
182 case DISPLAY_DESIRE:
183 textId = SAY_FACE_DESIRE_SLAY;
184 break;
185 default:
186 break;
187 }
188
189 if (textId)
190 Talk(textId);
191 }
192
193 void JustDied(Unit* /*killer*/) override
194 {
195 _JustDied();
196
197 Position spawnPoint = {5618.139f, 2451.873f, 705.854f, 0};
198
199 Talk(SAY_FACE_DEATH);
200
201 int32 entryIndex;
202 if (instance->GetData(DATA_TEAM_IN_INSTANCE) == ALLIANCE)
203 entryIndex = 0;
204 else
205 entryIndex = 1;
206
207 for (int8 i = 0; outroPositions[i].entry[entryIndex] != 0; ++i)
208 {
209 if (Creature* summon = me->SummonCreature(outroPositions[i].entry[entryIndex], spawnPoint, TEMPSUMMON_DEAD_DESPAWN))
210 {
211 summon->GetMotionMaster()->MovePoint(0, outroPositions[i].movePosition);
212 if (summon->GetEntry() == NPC_JAINA_PART2)
213 summon->AI()->Talk(SAY_JAINA_OUTRO);
214 else if (summon->GetEntry() == NPC_SYLVANAS_PART2)
215 summon->AI()->Talk(SAY_SYLVANAS_OUTRO);
216 }
217 }
218 }

220 void SpellHitTarget(Unit* /*target*/, const SpellInfo* spell) override
221 {
222 if (spell->Id == H_SPELL_PHANTOM_BLAST)
223 threeFaced = false;
224 }

228 if (type == DATA_THREE_FACED)
229 return threeFaced;
230
231 return 0;
232 }
233
234 void UpdateAI(uint32 diff) override
235 {
236 // Return since we have no target
237 if (!UpdateVictim())
238 return;
239
240 events.Update(diff);
241
242 if (me->HasUnitState(UNIT_STATE_CASTING))
243 return;
244
245 while (uint32 eventId = events.ExecuteEvent())
246 {
247 switch (eventId)
248 {
249 case EVENT_PHANTOM_BLAST:
250 DoCastVictim(SPELL_PHANTOM_BLAST);
251 events.ScheduleEvent(EVENT_PHANTOM_BLAST, 5000);
252 break;
253 case EVENT_MIRRORED_SOUL:
254 DoCastAOE(SPELL_MIRRORED_SOUL_TARGET_SELECTOR);
255 Talk(EMOTE_MIRRORED_SOUL);
256 events.ScheduleEvent(EVENT_MIRRORED_SOUL, urand(15000, 30000));
257 break;
258 case EVENT_WELL_OF_SOULS:
259 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
260 DoCast(target, SPELL_WELL_OF_SOULS);
261 events.ScheduleEvent(EVENT_WELL_OF_SOULS, 20000);
262 break;
263 case EVENT_UNLEASHED_SOULS:
264 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
265 DoCast(target, SPELL_UNLEASHED_SOULS);
266 me->SetDisplayId(DISPLAY_SORROW);
267 Talk(SAY_FACE_UNLEASH_SOUL);
268 Talk(EMOTE_UNLEASH_SOUL);
269 events.ScheduleEvent(EVENT_UNLEASHED_SOULS, 30000);
270 events.ScheduleEvent(EVENT_FACE_ANGER, 5000);
271 break;
272 case EVENT_FACE_ANGER:
273 me->SetDisplayId(DISPLAY_ANGER);
274 break;
275
276 case EVENT_WAILING_SOULS:
277 me->SetDisplayId(DISPLAY_DESIRE);
278 Talk(SAY_FACE_WAILING_SOUL);
279 Talk(EMOTE_WAILING_SOUL);
280 DoCast(me, SPELL_WAILING_SOULS_STARTING);
281 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
282 {
283 me->SetFacingToObject(target);
284 DoCast(me, SPELL_WAILING_SOULS_BEAM);
285 }
286
287 beamAngle = me->GetOrientation();
288
289 beamAngleDiff = float(M_PI)/30.0f; // PI/2 in 15 sec = PI/30 per tick
290 if (RAND(true, false))
291 beamAngleDiff = -beamAngleDiff;
292
293 me->InterruptNonMeleeSpells(false);
294 me->SetReactState(REACT_PASSIVE);
295
296 //Remove any target
297 me->SetTarget(ObjectGuid::Empty);
298
299 me->GetMotionMaster()->Clear();
300 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
301
302 wailingSoulTick = 15;
303 events.DelayEvents(18000); // no other events during wailing souls
304 events.ScheduleEvent(EVENT_WAILING_SOULS_TICK, 3000); // first one after 3 secs.
305 break;
306
307 case EVENT_WAILING_SOULS_TICK:
308 beamAngle += beamAngleDiff;
309 me->SetFacingTo(beamAngle);
310 me->StopMoving();
311
312 DoCast(me, SPELL_WAILING_SOULS);
313
314 if (--wailingSoulTick)
315 events.ScheduleEvent(EVENT_WAILING_SOULS_TICK, 1000);
316 else
317 {
318 me->SetReactState(REACT_AGGRESSIVE);
319 me->SetDisplayId(DISPLAY_ANGER);
320 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
321 me->GetMotionMaster()->MoveChase(me->GetVictim());
322 events.ScheduleEvent(EVENT_WAILING_SOULS, urand(60000, 70000));
323 }
324 break;
325 }
326 }
327
328 DoMeleeAttackIfReady();
329 }
330
331 private:
332 bool threeFaced;
333
334 // wailing soul event
335 float beamAngle;
336 float beamAngleDiff;
337 int8 wailingSoulTick;
338 };

18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "vault_of_archavon.h"
21
22/* Vault of Archavon encounters:
231 - Archavon the Stone Watcher event
242 - Emalon the Storm Watcher event
253 - Koralon the Flame Watcher event
264 - Toravon the Ice Watcher event
27*/
28
29class instance_vault_of_archavon : public InstanceMapScript
30{
31 public:
32 instance_vault_of_archavon() : InstanceMapScript("instance_vault_of_archavon", 624) { }
33
34 struct instance_vault_of_archavon_InstanceMapScript : public InstanceScript
35 {
36 instance_vault_of_archavon_InstanceMapScript(Map* map) : InstanceScript(map)
37 {
38 SetHeaders(DataHeader);
39 SetBossNumber(EncounterCount);
40
41 ArchavonDeath = 0;
42 EmalonDeath = 0;
43 KoralonDeath = 0;
44 }
45
46 void OnCreatureCreate(Creature* creature) override
47 {
48 switch (creature->GetEntry())
49 {
50 case NPC_EMALON:
51 EmalonGUID = creature->GetGUID();
52 break;
53 case NPC_TORAVON:
54 ToravonGUID = creature->GetGUID();
55 break;
56 default:
57 break;
58 }
59 }
60
61 ObjectGuid GetGuidData(uint32 identifier) const override
62 {
63 switch (identifier)
64 {
65 case DATA_EMALON:
66 return EmalonGUID;
67 case DATA_TORAVON:
68 return ToravonGUID;
69 default:
70 break;
71 }
72
73 return ObjectGuid::Empty;
74 }
75
76 bool SetBossState(uint32 type, EncounterState state) override
77 {
78 if (!InstanceScript::SetBossState(type, state))
79 return false;
80
81 if (state != DONE)
82 return true;
83
84 switch (type)
85 {
86 case DATA_ARCHAVON:
87 ArchavonDeath = time(NULL);
88 break;
89 case DATA_EMALON:
90 EmalonDeath = time(NULL);
91 break;
92 case DATA_KORALON:
93 KoralonDeath = time(NULL);
94 break;
95 default:
96 return true;
97 }
98
99 // on every death of Archavon, Emalon and Koralon check our achievement
100 DoCastSpellOnPlayers(SPELL_EARTH_WIND_FIRE_ACHIEVEMENT_CHECK);
101
102 return true;
103 }
104
105 bool CheckAchievementCriteriaMeet(uint32 criteria_id, Player const* /*source*/, Unit const* /*target*/, uint32 /*miscvalue1*/) override
106 {
107 switch (criteria_id)
108 {
109 case CRITERIA_EARTH_WIND_FIRE_10:
110 case CRITERIA_EARTH_WIND_FIRE_25:
111 if (ArchavonDeath && EmalonDeath && KoralonDeath)
112 {
113 // instance difficulty check is already done in db (achievement_criteria_data)
114 // int() for Visual Studio, compile errors with abs(time_t)
115 return (abs(int(ArchavonDeath-EmalonDeath)) < MINUTE && \
116 abs(int(EmalonDeath-KoralonDeath)) < MINUTE && \
117 abs(int(KoralonDeath-ArchavonDeath)) < MINUTE);
118 }
119 break;
120 default:
121 break;
122 }
123
124 return false;
125 }
126
127 private:
128 ObjectGuid EmalonGUID;
129 ObjectGuid ToravonGUID;
130 time_t ArchavonDeath;
131 time_t EmalonDeath;
132 time_t KoralonDeath;
133 };
134
135 InstanceScript* GetInstanceScript(InstanceMap* map) const override
136 {
137 return new instance_vault_of_archavon_InstanceMapScript(map);
138 }
139};
140
141void AddSC_instance_vault_of_archavon()
142{
143 new instance_vault_of_archavon();
144}

68class boss_archavon : public CreatureScript
69{
70 public:
71 boss_archavon() : CreatureScript("boss_archavon") { }
72
73 struct boss_archavonAI : public BossAI
74 {
75 boss_archavonAI(Creature* creature) : BossAI(creature, DATA_ARCHAVON)
76 {
77 }
78
79 void EnterCombat(Unit* /*who*/) override
80 {
81 events.ScheduleEvent(EVENT_ROCK_SHARDS, 15000);
82 events.ScheduleEvent(EVENT_CHOKING_CLOUD, 30000);
83 events.ScheduleEvent(EVENT_STOMP, 45000);
84 events.ScheduleEvent(EVENT_BERSERK, 300000);
85
86 _EnterCombat();
87 }
88
89 // Below UpdateAI may need review/debug.
90 void UpdateAI(uint32 diff) override
91 {
92 if (!UpdateVictim())
93 return;
94
95 events.Update(diff);
96
97 if (me->HasUnitState(UNIT_STATE_CASTING))
98 return;
99
100 while (uint32 eventId = events.ExecuteEvent())
101 {
102 switch (eventId)
103 {
104 case EVENT_ROCK_SHARDS:
105 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
106 DoCast(target, SPELL_ROCK_SHARDS);
107 events.ScheduleEvent(EVENT_ROCK_SHARDS, 15000);
108 break;
109 case EVENT_CHOKING_CLOUD:
110 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
111 DoCast(target, SPELL_CRUSHING_LEAP, true); //10y~80y, ignore range
112 events.ScheduleEvent(EVENT_CHOKING_CLOUD, 30000);
113 break;
114 case EVENT_STOMP:
115 DoCastVictim(SPELL_STOMP);
116 events.ScheduleEvent(EVENT_IMPALE, 3000);
117 events.ScheduleEvent(EVENT_STOMP, 45000);
118 break;
119 case EVENT_IMPALE:
120 DoCastVictim(SPELL_IMPALE);
121 break;
122 case EVENT_BERSERK:
123 DoCast(me, SPELL_BERSERK);
124 Talk(EMOTE_BERSERK);
125 break;
126 default:
127 break;
128 }
129 }
130
131 DoMeleeAttackIfReady();
132 }
133 };
134
135 CreatureAI* GetAI(Creature* creature) const override
136 {
137 return new boss_archavonAI(creature);
138 }
139};

166 {
167 DoZoneInCombat();
168 }
169
170 void UpdateAI(uint32 diff) override
171 {
172 if (!UpdateVictim())
173 return;
174
175 events.Update(diff);
176
177 if (me->HasUnitState(UNIT_STATE_CASTING))
178 return;
179
180 while (uint32 eventId = events.ExecuteEvent())
181 {
182 switch (eventId)
183 {
184 case EVENT_ROCK_SHOWER:
185 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
186 DoCast(target, SPELL_ROCK_SHOWER);
187 events.ScheduleEvent(EVENT_ROCK_SHARDS, 6000);
188 break;
189 case EVENT_SHIELD_CRUSH:
190 DoCastVictim(SPELL_SHIELD_CRUSH);
191 events.ScheduleEvent(EVENT_SHIELD_CRUSH, 20000);
192 break;
193 case EVENT_WHIRL:
194 DoCastVictim(SPELL_WHIRL);
195 events.ScheduleEvent(EVENT_WHIRL, 8000);
196 break;
197 default:
198 break;
199 }
200 }
201
202 DoMeleeAttackIfReady();
203 }
204 };

213class spell_archavon_rock_shards : public SpellScriptLoader
214{
215 public:
216 spell_archavon_rock_shards() : SpellScriptLoader("spell_archavon_rock_shards") { }
217
218 class spell_archavon_rock_shards_SpellScript : public SpellScript
219 {
220 PrepareSpellScript(spell_archavon_rock_shards_SpellScript);
221
222 bool Validate(SpellInfo const* /*spellInfo*/) override
223 {
224 if (!sSpellMgr->GetSpellInfo(SPELL_ROCK_SHARDS_VISUAL_L)
225 || !sSpellMgr->GetSpellInfo(SPELL_ROCK_SHARDS_VISUAL_R)
226 || !sSpellMgr->GetSpellInfo(SPELL_ROCK_SHARDS_DAMAGE_L)
227 || !sSpellMgr->GetSpellInfo(SPELL_ROCK_SHARDS_DAMAGE_R))
228 return false;
229 return true;
230 }
231
232 void HandleScript(SpellEffIndex /*effIndex*/)
233 {
234 Unit* caster = GetCaster();
235
236 for (uint8 i = 0; i < 3; ++i)
237 {
238 caster->CastSpell((Unit*)NULL, SPELL_ROCK_SHARDS_VISUAL_L, true);
239 caster->CastSpell((Unit*)NULL, SPELL_ROCK_SHARDS_VISUAL_R, true);
240 }
241
242 caster->CastSpell((Unit*)NULL, SPELL_ROCK_SHARDS_DAMAGE_L, true);
243 caster->CastSpell((Unit*)NULL, SPELL_ROCK_SHARDS_DAMAGE_R, true);
244 }
245
246 void Register() override
247 {
248 OnEffectHit += SpellEffectFn(spell_archavon_rock_shards_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
249 }
250 };
251
252 SpellScript* GetSpellScript() const override
253 {
254 return new spell_archavon_rock_shards_SpellScript();
255 }

69/*######
70## Emalon the Storm Watcher
71######*/
72class boss_emalon : public CreatureScript
73{
74 public:
75 boss_emalon() : CreatureScript("boss_emalon") { }
76
77 struct boss_emalonAI : public BossAI
78 {
79 boss_emalonAI(Creature* creature) : BossAI(creature, DATA_EMALON)
80 {
81 }
82
83 void Reset() override
84 {
85 _Reset();
86
87 for (uint8 i = 0; i < MAX_TEMPEST_MINIONS; ++i)
88 me->SummonCreature(NPC_TEMPEST_MINION, TempestMinions[i], TEMPSUMMON_CORPSE_DESPAWN, 0);
89 }
90
91 void JustSummoned(Creature* summoned) override
92 {
93 BossAI::JustSummoned(summoned);
94
95 // AttackStart has NULL-check for victim
96 if (summoned->AI())
97 summoned->AI()->AttackStart(me->GetVictim());
98 }
99
100 void EnterCombat(Unit* who) override
101 {
102 if (!summons.empty())
103 {
104 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
105 {
106 Creature* minion = ObjectAccessor::GetCreature(*me, *itr);
107 if (minion && minion->IsAlive() && !minion->GetVictim() && minion->AI())
108 minion->AI()->AttackStart(who);
109 }
110 }
111
112 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 5000);
113 events.ScheduleEvent(EVENT_LIGHTNING_NOVA, 40000);
114 events.ScheduleEvent(EVENT_BERSERK, 360000);
115 events.ScheduleEvent(EVENT_OVERCHARGE, 45000);
116
117 _EnterCombat();
118 }
119
120 void UpdateAI(uint32 diff) override
121 {
122 if (!UpdateVictim())
123 return;
124
125 events.Update(diff);
126
127 if (me->HasUnitState(UNIT_STATE_CASTING))
128 return;
129
130 while (uint32 eventId = events.ExecuteEvent())
131 {
132 switch (eventId)
133 {
134 case EVENT_CHAIN_LIGHTNING:
135 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
136 DoCast(target, SPELL_CHAIN_LIGHTNING);
137 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 25000);
138 break;
139 case EVENT_LIGHTNING_NOVA:
140 DoCastAOE(SPELL_LIGHTNING_NOVA);
141 events.ScheduleEvent(EVENT_LIGHTNING_NOVA, 40000);
142 break;
143 case EVENT_OVERCHARGE:
144 if (!summons.empty())
145 {
146 Creature* minion = ObjectAccessor::GetCreature(*me, Trinity::Containers::SelectRandomContainerElement(summons));
147 if (minion && minion->IsAlive())
148 {
149 minion->CastSpell(me, SPELL_OVERCHARGED, true);
150 minion->SetFullHealth();
151 Talk(EMOTE_OVERCHARGE);
152 events.ScheduleEvent(EVENT_OVERCHARGE, 45000);
153 }
154 }
155 break;
156 case EVENT_BERSERK:
157 DoCast(me, SPELL_BERSERK);
158 Talk(EMOTE_BERSERK);
159 break;
160 default:
161 break;
162 }
163 }
164
165 DoMeleeAttackIfReady();
166 }
167 };
168
169 CreatureAI* GetAI(Creature* creature) const override
170 {
171 return new boss_emalonAI(creature);
172 }
173};

178class npc_tempest_minion : public CreatureScript
179{
180 public:
181 npc_tempest_minion() : CreatureScript("npc_tempest_minion") { }
182
183 struct npc_tempest_minionAI : public ScriptedAI
184 {
185 npc_tempest_minionAI(Creature* creature) : ScriptedAI(creature)
186 {
187 Initialize();
188 instance = creature->GetInstanceScript();
189 }
190
191 void Initialize()
192 {
193 OverchargedTimer = 0;
194 }
195
196 void Reset() override
197 {
198 events.Reset();
199 Initialize();
200 }
201
202 void JustDied(Unit* /*killer*/) override
203 {
204 if (Creature* emalon = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_EMALON)))
205 {
206 if (emalon->IsAlive())
207 {
208 emalon->SummonCreature(NPC_TEMPEST_MINION, 0, 0, 0, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
209 Talk(EMOTE_MINION_RESPAWN);
210 }
211 }
212 }
213
214 void EnterCombat(Unit* who) override
215 {
216 DoZoneInCombat();
217 events.ScheduleEvent(EVENT_SHOCK, 20000);
218
219 if (Creature* pEmalon = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_EMALON)))
220 {
221 if (!pEmalon->GetVictim() && pEmalon->AI())
222 pEmalon->AI()->AttackStart(who);
223 }
224 }
225
226 void UpdateAI(uint32 diff) override
227 {
228 //Return since we have no target
229 if (!UpdateVictim())
230 return;
231
232 events.Update(diff);
233
234 if (me->HasUnitState(UNIT_STATE_CASTING))
235 return;
236
237 if (Aura const* overchargedAura = me->GetAura(SPELL_OVERCHARGED))
238 {
239 if (overchargedAura->GetStackAmount() < 10)
240 {
241 if (OverchargedTimer <= diff)
242 {
243 DoCast(me, SPELL_OVERCHARGED);
244 OverchargedTimer = 2000;
245 }
246 else
247 OverchargedTimer -= diff;
248 }
249 else
250 {
251 if (overchargedAura->GetStackAmount() == 10)
252 {
253 DoCast(me, SPELL_OVERCHARGED_BLAST);
254 me->DespawnOrUnsummon();
255 Talk(EMOTE_MINION_RESPAWN);
256 }
257 }
258 }
259
260 if (events.ExecuteEvent() == EVENT_SHOCK)
261 {
262 DoCastVictim(SPELL_SHOCK);
263 events.ScheduleEvent(EVENT_SHOCK, 20000);
264 }
265
266 DoMeleeAttackIfReady();
267 }
268
269 private:
270 InstanceScript* instance;
271 EventMap events;
272 uint32 OverchargedTimer;
273 };
274
275 CreatureAI* GetAI(Creature* creature) const override
276 {
277 return GetInstanceAI<npc_tempest_minionAI>(creature);
278 }
279};

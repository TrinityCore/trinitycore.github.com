39class boss_moragg : public CreatureScript
40{
41 public:
42 boss_moragg() : CreatureScript("boss_moragg") { }
43
44 struct boss_moraggAI : public BossAI
45 {
46 boss_moraggAI(Creature* creature) : BossAI(creature, DATA_MORAGG) { }
47
48 void Reset() override
49 {
50 BossAI::Reset();
51 }
52
53 void EnterCombat(Unit* who) override
54 {
55 BossAI::EnterCombat(who);
56 }
57
58 void JustReachedHome() override
59 {
60 BossAI::JustReachedHome();
61 instance->SetData(DATA_HANDLE_CELLS, DATA_MORAGG);
62 }
63
64 void JustDied(Unit* killer) override
65 {
66 BossAI::JustDied(killer);
67 }
68
69 void UpdateAI(uint32 diff) override
70 {
71 if (!UpdateVictim())
72 return;
73
74 scheduler.Update(diff,
75 std::bind(&BossAI::DoMeleeAttackIfReady, this));
76 }
77
78 void ScheduleTasks() override
79 {
80 scheduler.Async([this]
81 {
82 DoCast(me, DUNGEON_MODE(SPELL_RAY_OF_PAIN, SPELL_RAY_OF_PAIN_H));
83 DoCast(me, DUNGEON_MODE(SPELL_RAY_OF_SUFFERING, SPELL_RAY_OF_SUFFERING_H));
84 });
85
86 scheduler.Schedule(Seconds(15), [this](TaskContext task)
87 {
88 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true))
89 DoCast(target, SPELL_OPTIC_LINK);
90 task.Repeat(Seconds(25));
91 });
92
93 scheduler.Schedule(Seconds(5), [this](TaskContext task)
94 {
95 DoCastVictim(SPELL_CORROSIVE_SALIVA);
96 task.Repeat(Seconds(10));
97 });
98 }
99 };
100
101 CreatureAI* GetAI(Creature* creature) const override
102 {
103 return GetVioletHoldAI<boss_moraggAI>(creature);
104 }
105};

107class spell_moragg_ray : public SpellScriptLoader
108{
109 public:
110 spell_moragg_ray() : SpellScriptLoader("spell_moragg_ray") { }
111
112 class spell_moragg_ray_AuraScript : public AuraScript
113 {
114 PrepareAuraScript(spell_moragg_ray_AuraScript);
115
116 void OnPeriodic(AuraEffect const* aurEff)
117 {
118 PreventDefaultAction();
119
120 if (!GetTarget()->IsAIEnabled)
121 return;
122
123 if (Unit* target = GetTarget()->GetAI()->SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
124 {
125 uint32 triggerSpell = aurEff->GetSpellEffectInfo()->TriggerSpell;
126 GetTarget()->CastSpell(target, triggerSpell, TRIGGERED_FULL_MASK, nullptr, aurEff);
127 }
128 }
129
130 void Register() override
131 {
132 OnEffectPeriodic += AuraEffectPeriodicFn(spell_moragg_ray_AuraScript::OnPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_TRIGGER_SPELL);
133 }
134 };
135
136 AuraScript* GetAuraScript() const override
137 {
138 return new spell_moragg_ray_AuraScript();
139 }
140};

142class spell_moragg_optic_link : public SpellScriptLoader
143{
144public:
145 spell_moragg_optic_link() : SpellScriptLoader("spell_moragg_optic_link") { }
146
147 class spell_moragg_optic_link_AuraScript : public AuraScript
148 {
149 PrepareAuraScript(spell_moragg_optic_link_AuraScript);
150
151 void OnPeriodic(AuraEffect const* aurEff)
152 {
153 if (Unit* caster = GetCaster())
154 {
155 if (aurEff->GetTickNumber() >= 8)
156 caster->CastSpell(GetTarget(), SPELL_OPTIC_LINK_LEVEL_3, TRIGGERED_FULL_MASK, nullptr, aurEff);
157
158 if (aurEff->GetTickNumber() >= 4)
159 caster->CastSpell(GetTarget(), SPELL_OPTIC_LINK_LEVEL_2, TRIGGERED_FULL_MASK, nullptr, aurEff);
160
161 caster->CastSpell(GetTarget(), SPELL_OPTIC_LINK_LEVEL_1, TRIGGERED_FULL_MASK, nullptr, aurEff);
162 }
163 }
164
165 void OnUpdate(AuraEffect* aurEff)
166 {
167 switch (aurEff->GetTickNumber())
168 {
169 case 1:
170 aurEff->SetAmount(aurEff->GetAmount() + 250); // base amount is 500
171 break;
172 case 4:
173 aurEff->SetAmount(aurEff->GetAmount() * 2); // goes to 1500
174 break;
175 case 8:
176 aurEff->SetAmount(aurEff->GetAmount() * 2); // goes to 3000
177 break;
178 default:
179 break;
180 }
181 }
182
183 void Register() override
184 {
185 OnEffectPeriodic += AuraEffectPeriodicFn(spell_moragg_optic_link_AuraScript::OnPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE);
186 OnEffectUpdatePeriodic += AuraEffectUpdatePeriodicFn(spell_moragg_optic_link_AuraScript::OnUpdate, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE);
187 }
188 };
189
190 AuraScript* GetAuraScript() const override
191 {
192 return new spell_moragg_optic_link_AuraScript();
193 }
194};

79class boss_xevozz : public CreatureScript
80{
81 public:
82 boss_xevozz() : CreatureScript("boss_xevozz") { }
83
84 struct boss_xevozzAI : public BossAI
85 {
86 boss_xevozzAI(Creature* creature) : BossAI(creature, DATA_XEVOZZ) { }
87
88 void Reset() override
89 {
90 BossAI::Reset();
91 }
92
93 void EnterCombat(Unit* who) override
94 {
95 BossAI::EnterCombat(who);
96 Talk(SAY_AGGRO);
97 }
98
99 void JustReachedHome() override
100 {
101 BossAI::JustReachedHome();
102 instance->SetData(DATA_HANDLE_CELLS, DATA_XEVOZZ);
103 }
104
105 void JustSummoned(Creature* summon) override
106 {
107 BossAI::JustSummoned(summon);
108 summon->GetMotionMaster()->MoveFollow(me, 0.0f, 0.0f);
109 }
110
111 void KilledUnit(Unit* victim) override
112 {
113 if (victim->GetTypeId() == TYPEID_PLAYER)
114 Talk(SAY_SLAY);
115 }
116
117 void JustDied(Unit* killer) override
118 {
119 BossAI::JustDied(killer);
120 Talk(SAY_DEATH);
121 }
122
123 void SpellHit(Unit* /*who*/, SpellInfo const* spell) override
124 {
125 if (spell->Id == SPELL_ARCANE_POWER || spell->Id == H_SPELL_ARCANE_POWER)
126 Talk(SAY_SUMMON_ENERGY);
127 }
128
129 void UpdateAI(uint32 diff) override
130 {
131 if (!UpdateVictim())
132 return;
133
134 scheduler.Update(diff,
135 std::bind(&BossAI::DoMeleeAttackIfReady, this));
136 }
137
138 void ScheduleTasks() override
139 {
140 scheduler.Schedule(Seconds(8), Seconds(10), [this](TaskContext task)
141 {
142 DoCastAOE(SPELL_ARCANE_BARRAGE_VOLLEY);
143 task.Repeat(Seconds(8), Seconds(10));
144 });
145
146 scheduler.Schedule(Seconds(10), Seconds(11), [this](TaskContext task)
147 {
148 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
149 DoCast(target, SPELL_ARCANE_BUFFET);
150 task.Repeat(Seconds(15), Seconds(20));
151 });
152
153 scheduler.Schedule(Seconds(5), [this](TaskContext task)
154 {
155 Talk(SAY_REPEAT_SUMMON);
156
157 std::list<uint8> summonSpells = { 0, 1, 2 };
158
159 uint8 spell = Trinity::Containers::SelectRandomContainerElement(summonSpells);
160 DoCast(me, EtherealSphereSummonSpells[spell]);
161 summonSpells.remove(spell);
162
163 if (IsHeroic())
164 {
165 spell = Trinity::Containers::SelectRandomContainerElement(summonSpells);
166 task.Schedule(Milliseconds(2500), [this, spell](TaskContext /*task*/)
167 {
168 DoCast(me, EtherealSphereHeroicSummonSpells[spell]);
169 });
170 }
171
172 task.Schedule(Seconds(33), Seconds(35), [this](TaskContext /*task*/)
173 {
174 DummyEntryCheckPredicate pred;
175 summons.DoAction(ACTION_SUMMON, pred);
176 });
177
178 task.Repeat(Seconds(45), Seconds(47));
179 });
180 }
181 };
182
183 CreatureAI* GetAI(Creature* creature) const override
184 {
185 return GetVioletHoldAI<boss_xevozzAI>(creature);
186 }

189class npc_ethereal_sphere : public CreatureScript
190{
191 public:
192 npc_ethereal_sphere() : CreatureScript("npc_ethereal_sphere") { }
193
194 struct npc_ethereal_sphereAI : public ScriptedAI
195 {
196 npc_ethereal_sphereAI(Creature* creature) : ScriptedAI(creature)
197 {
198 instance = creature->GetInstanceScript();
199 }
200
201 void Reset() override
202 {
203 scheduler.CancelAll();
204 ScheduledTasks();
205
206 DoCast(me, SPELL_POWER_BALL_VISUAL);
207 DoCast(me, DUNGEON_MODE(SPELL_POWER_BALL_DAMAGE_TRIGGER, SPELL_POWER_BALL_DAMAGE_TRIGGER_H));
208
209 me->DespawnOrUnsummon(40000);
210 }
211
212 void DoAction(int32 action) override
213 {
214 if (action == ACTION_SUMMON)
215 {
216 Talk(SAY_ETHEREAL_SPHERE_SUMMON);
217 DoCastAOE(SPELL_SUMMON_PLAYERS);
218 }
219 }
220
221 void UpdateAI(uint32 diff) override
222 {
223 scheduler.Update(diff);
224 }
225
226 void ScheduledTasks()
227 {
228 scheduler.Schedule(Seconds(1), [this](TaskContext task)
229 {
230 if (Creature* xevozz = instance->GetCreature(DATA_XEVOZZ))
231 {
232 if (me->IsWithinDist(xevozz, 3.0f))
233 {
234 DoCastAOE(SPELL_ARCANE_POWER);
235 me->DespawnOrUnsummon(8000);
236 return;
237 }
238 }
239 task.Repeat();
240 });
241 }
242
243 private:
244 InstanceScript* instance;
245 TaskScheduler scheduler;
246 };
247
248 CreatureAI* GetAI(Creature* creature) const override
249 {
250 return GetVioletHoldAI<npc_ethereal_sphereAI>(creature);
251 }
252};

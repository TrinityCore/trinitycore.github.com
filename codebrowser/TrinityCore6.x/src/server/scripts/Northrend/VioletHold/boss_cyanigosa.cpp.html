45class boss_cyanigosa : public CreatureScript
46{
47 public:
48 boss_cyanigosa() : CreatureScript("boss_cyanigosa") { }
49
50 struct boss_cyanigosaAI : public BossAI
51 {
52 boss_cyanigosaAI(Creature* creature) : BossAI(creature, DATA_CYANIGOSA) { }
53
54 void EnterCombat(Unit* who) override
55 {
56 BossAI::EnterCombat(who);
57 Talk(SAY_AGGRO);
58 }
59
60 void KilledUnit(Unit* victim) override
61 {
62 if (victim->GetTypeId() == TYPEID_PLAYER)
63 Talk(SAY_SLAY);
64 }
65
66 void JustDied(Unit* killer) override
67 {
68 BossAI::JustDied(killer);
69 Talk(SAY_DEATH);
70 }
71
72 void MoveInLineOfSight(Unit* /*who*/) override { }
73
74 void UpdateAI(uint32 diff) override
75 {
76 if (!UpdateVictim())
77 return;
78
79 scheduler.Update(diff,
80 std::bind(&BossAI::DoMeleeAttackIfReady, this));
81 }
82
83 void ScheduleTasks() override
84 {
85 scheduler.Schedule(Seconds(10), [this](TaskContext task)
86 {
87 DoCastAOE(SPELL_ARCANE_VACUUM);
88 task.Repeat();
89 });
90
91 scheduler.Schedule(Seconds(15), [this](TaskContext task)
92 {
93 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
94 DoCast(target, SPELL_BLIZZARD);
95 task.Repeat();
96 });
97
98 scheduler.Schedule(Seconds(20), [this](TaskContext task)
99 {
100 DoCastVictim(SPELL_TAIL_SWEEP);
101 task.Repeat();
102 });
103
104 scheduler.Schedule(Seconds(25), [this](TaskContext task)
105 {
106 DoCastVictim(SPELL_UNCONTROLLABLE_ENERGY);
107 task.Repeat();
108 });
109
110 if (IsHeroic())
111 {
112 scheduler.Schedule(Seconds(30), [this](TaskContext task)
113 {
114 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true))
115 DoCast(target, SPELL_MANA_DESTRUCTION);
116 task.Repeat();
117 });
118 }
119 }
120 };
121
122 CreatureAI* GetAI(Creature* creature) const override
123 {
124 return GetVioletHoldAI<boss_cyanigosaAI>(creature);
125 }
126};

128class achievement_defenseless : public AchievementCriteriaScript
129{
130 public:
131 achievement_defenseless() : AchievementCriteriaScript("achievement_defenseless") { }
132
133 bool OnCheck(Player* /*player*/, Unit* target) override
134 {
135 if (!target)
136 return false;
137
138 InstanceScript* instance = target->GetInstanceScript();
139 if (!instance)
140 return false;
141
142 return instance->GetData(DATA_DEFENSELESS) != 0;
143 }
144};

146class spell_cyanigosa_arcane_vacuum : public SpellScriptLoader
147{
148 public:
149 spell_cyanigosa_arcane_vacuum() : SpellScriptLoader("spell_cyanigosa_arcane_vacuum") { }
150
151 class spell_cyanigosa_arcane_vacuum_SpellScript : public SpellScript
152 {
153 PrepareSpellScript(spell_cyanigosa_arcane_vacuum_SpellScript);
154
155 bool Validate(SpellInfo const* /*spellInfo*/) override
156 {
157 if (!sSpellMgr->GetSpellInfo(SPELL_SUMMON_PLAYER))
158 return false;
159 return true;
160 }
161
162 void HandleScript(SpellEffIndex /*effIndex*/)
163 {
164 GetCaster()->CastSpell(GetHitUnit(), SPELL_SUMMON_PLAYER, true);
165 }
166
167 void Register() override
168 {
169 OnEffectHitTarget += SpellEffectFn(spell_cyanigosa_arcane_vacuum_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_DUMMY);
170 }
171 };
172
173 SpellScript* GetSpellScript() const override
174 {
175 return new spell_cyanigosa_arcane_vacuum_SpellScript();
176 }
177};

179void AddSC_boss_cyanigosa()

182 new achievement_defenseless();
183 new spell_cyanigosa_arcane_vacuum();

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "violet_hold.h"
21
22enum Spells
23{
24 SPELL_CAUTERIZING_FLAMES = 59466, // Only in heroic
25 SPELL_FIREBOLT = 54235,
26 SPELL_FLAME_BREATH = 54282,
27 SPELL_LAVA_BURN = 54249
28};
29
30class boss_lavanthor : public CreatureScript
31{
32 public:
33 boss_lavanthor() : CreatureScript("boss_lavanthor") { }
34
35 struct boss_lavanthorAI : public BossAI
36 {
37 boss_lavanthorAI(Creature* creature) : BossAI(creature, DATA_LAVANTHOR) { }
38
39 void Reset() override
40 {
41 BossAI::Reset();
42 }
43
44 void EnterCombat(Unit* who) override
45 {
46 BossAI::EnterCombat(who);
47 }
48
49 void JustReachedHome() override
50 {
51 BossAI::JustReachedHome();
52 instance->SetData(DATA_HANDLE_CELLS, DATA_LAVANTHOR);
53 }
54
55 void JustDied(Unit* killer) override
56 {
57 BossAI::JustDied(killer);
58 }
59
60 void UpdateAI(uint32 diff) override
61 {
62 if (!UpdateVictim())
63 return;
64
65 scheduler.Update(diff,
66 std::bind(&BossAI::DoMeleeAttackIfReady, this));
67 }
68
69 void ScheduleTasks() override
70 {
71 scheduler.Schedule(Seconds(1), [this](TaskContext task)
72 {
73 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40.0f, true))
74 DoCast(target, SPELL_FIREBOLT);
75 task.Repeat(Seconds(5), Seconds(13));
76 });
77
78 scheduler.Schedule(Seconds(5), [this](TaskContext task)
79 {
80 DoCastVictim(SPELL_FLAME_BREATH);
81 task.Repeat(Seconds(10), Seconds(15));
82 });
83
84 scheduler.Schedule(Seconds(10), [this](TaskContext task)
85 {
86 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f))
87 DoCast(target, SPELL_LAVA_BURN);
88 task.Repeat(Seconds(15), Seconds(23));
89 });
90
91 if (IsHeroic())
92 {
93 scheduler.Schedule(Seconds(3), [this](TaskContext task)
94 {
95 DoCastAOE(SPELL_CAUTERIZING_FLAMES);
96 task.Repeat(Seconds(10), Seconds(16));
97 });
98 }
99 }
100 };
101
102 CreatureAI* GetAI(Creature* creature) const override
103 {
104 return GetVioletHoldAI<boss_lavanthorAI>(creature);
105 }
106};
107
108void AddSC_boss_lavanthor()
109{
110 new boss_lavanthor();
111}

72struct WebTargetSelector : public std::unary_function<Unit*, bool>
73{
74 WebTargetSelector(Unit* maexxna) : _maexxna(maexxna) {}
75 bool operator()(Unit const* target) const
76 {
77 if (_maexxna->GetVictim() == target) // never target tank
78 return false;
79 if (target->HasAura(SPELL_WEB_WRAP)) // never target targets that are already webbed
80 return false;
81 return true;
82 }
83
84 private:
85 const Unit* _maexxna;
86};

88class boss_maexxna : public CreatureScript
89{
90public:
91 boss_maexxna() : CreatureScript("boss_maexxna") { }

97
98 struct boss_maexxnaAI : public BossAI
99 {
100 boss_maexxnaAI(Creature* creature) : BossAI(creature, BOSS_MAEXXNA) { }
101
102 void EnterCombat(Unit* /*who*/) override
103 {
104 _EnterCombat();
105 events.ScheduleEvent(EVENT_WRAP, 20 * IN_MILLISECONDS);
106 events.ScheduleEvent(EVENT_SPRAY, 40 * IN_MILLISECONDS);
107 events.ScheduleEvent(EVENT_SHOCK, urandms(5, 10));
108 events.ScheduleEvent(EVENT_POISON, urandms(10, 15));
109 events.ScheduleEvent(EVENT_SUMMON, 30 * IN_MILLISECONDS);
110 }
111
112 void Reset() override
113 {
114 _Reset();
115 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_WEB_WRAP);
116 }
117
118 void UpdateAI(uint32 diff) override
119 {
120 if (!UpdateVictim() || !CheckInRoom())
121 return;
122
123 if (HealthBelowPct(30) && !me->HasAura(SPELL_FRENZY_HELPER))
124 {
125 DoCast(SPELL_FRENZY);
126 }
127
128 events.Update(diff);
129
130 while (uint32 eventId = events.ExecuteEvent())
131 {
132 switch (eventId)
133 {
134 case EVENT_WRAP:
135 {
136 std::list<Unit*> targets;
137 SelectTargetList(targets, WebTargetSelector(me), RAID_MODE(1, 2), SELECT_TARGET_RANDOM);
138 if (!targets.empty())
139 {
140 Talk(EMOTE_WEB_WRAP);
141 int8 wrapPos = -1;
142 for (Unit* target : targets)
143 {
144 if (wrapPos == -1) // allow all positions on the first target
145 wrapPos = urand(0, MAX_WRAP_POSITION - 1);
146 else // on subsequent iterations, only allow positions that are not equal to the previous one (this is sufficient since we should only have two targets at most, ever)
147 wrapPos = (wrapPos + urand(1, MAX_WRAP_POSITION - 1)) % MAX_WRAP_POSITION;
148
149 target->RemoveAura(RAID_MODE(SPELL_WEB_SPRAY_10, SPELL_WEB_SPRAY_25));
150 if (Creature* wrap = DoSummon(NPC_WEB_WRAP, WrapPositions[wrapPos], 70 * IN_MILLISECONDS, TEMPSUMMON_TIMED_DESPAWN))
151 {
152 wrap->AI()->SetGUID(target->GetGUID()); // handles application of debuff
153 target->GetMotionMaster()->MoveJump(WrapPositions[wrapPos], WEB_WRAP_MOVE_SPEED, WEB_WRAP_MOVE_SPEED); // move after stun to avoid stun cancelling move
154 }
155 }
156 }
157 events.ScheduleEvent(EVENT_WRAP, 40000);
158 break;
159 }
160 case EVENT_SPRAY:
161 Talk(EMOTE_WEB_SPRAY);
162 DoCastAOE(RAID_MODE(SPELL_WEB_SPRAY_10, SPELL_WEB_SPRAY_25));
163 events.ScheduleEvent(EVENT_SPRAY, 40000);
164 break;
165 case EVENT_SHOCK:
166 DoCastAOE(SPELL_POISON_SHOCK);
167 events.ScheduleEvent(EVENT_SHOCK, urandms(10, 20));
168 break;
169 case EVENT_POISON:
170 DoCastVictim(SPELL_NECROTIC_POISON);
171 events.ScheduleEvent(EVENT_POISON, urandms(10, 20));
172 break;
173 case EVENT_SUMMON:
174 Talk(EMOTE_SPIDERS);
175 uint8 amount = urand(8, 10);
176 for (uint8 i = 0; i < amount; ++i)
177 DoSummon(NPC_SPIDERLING, me, 4.0f, 5 * IN_MILLISECONDS, TEMPSUMMON_CORPSE_TIMED_DESPAWN);
178 events.ScheduleEvent(EVENT_SUMMON, 40000);
179 break;
180 }
181 }
182
183 DoMeleeAttackIfReady();
184 }
185 };

198
199 struct npc_webwrapAI : public NullCreatureAI
200 {
201 npc_webwrapAI(Creature* creature) : NullCreatureAI(creature), visibleTimer(0) { }
202
203 ObjectGuid victimGUID;
204 uint32 visibleTimer;
205
206 void InitializeAI() override
207 {
208 me->SetVisible(false);
209 }
210
211 void SetGUID(ObjectGuid guid, int32 /*param*/) override
212 {
213 if (!guid)
214 return;
215 victimGUID = guid;
216 if (Unit* victim = ObjectAccessor::GetUnit(*me, victimGUID))
217 {
218 visibleTimer = (me->GetDistance2d(victim)/WEB_WRAP_MOVE_SPEED + 0.5f) * IN_MILLISECONDS;
219 victim->CastSpell(victim, SPELL_WEB_WRAP, true, NULL, NULL, me->GetGUID());
220 }
221 }
222
223 void UpdateAI(uint32 diff) override
224 {
225 if (!visibleTimer)
226 return;
227
228 if (diff >= visibleTimer)
229 {
230 visibleTimer = 0;
231 me->SetVisible(true);
232 }
233 else
234 visibleTimer -= diff;
235 }
236
237 void JustDied(Unit* /*killer*/) override
238 {
239 if (!victimGUID.IsEmpty())
240 if (Unit* victim = ObjectAccessor::GetUnit(*me, victimGUID))
241 victim->RemoveAurasDueToSpell(SPELL_WEB_WRAP, me->GetGUID());
242
243 me->DespawnOrUnsummon(5 * IN_MILLISECONDS);
244 }
245 };

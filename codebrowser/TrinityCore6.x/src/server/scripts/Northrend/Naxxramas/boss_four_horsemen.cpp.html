97 boss_four_horsemenAI(Creature* creature) : BossAI(creature, BOSS_HORSEMEN)
98 {
99 Initialize();
100 id = Horsemen(0);
101 for (uint8 i = 0; i < 4; ++i)
102 if (me->GetEntry() == NPC_HORSEMEN[i])
103 id = Horsemen(i);
104 caster = (id == HORSEMEN_LADY || id == HORSEMEN_SIR);
105 }

108 {
109 uiEventStarterGUID.Clear();

144 bool DoEncounteraction(Unit* who, bool attack, bool reset, bool checkAllDead)
145 {
146 Creature* Thane = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_THANE));
147 Creature* Lady = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_LADY));
148 Creature* Baron = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BARON));
149 Creature* Sir = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SIR));
150
151 if (Thane && Lady && Baron && Sir)
152 {
153 if (attack && who)
154 {
155 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Thane->AI())->encounterActionAttack = true;
156 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Lady->AI())->encounterActionAttack = true;
157 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Baron->AI())->encounterActionAttack = true;
158 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Sir->AI())->encounterActionAttack = true;
159
160 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Thane->AI())->AttackStart(who);
161 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Lady->AI())->AttackStart(who);
162 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Baron->AI())->AttackStart(who);
163 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Sir->AI())->AttackStart(who);
164 }
165
166 if (reset)
167 {
168 if (instance->GetBossState(BOSS_HORSEMEN) != NOT_STARTED)
169 {
170 if (!Thane->IsAlive())
171 Thane->Respawn();
172
173 if (!Lady->IsAlive())
174 Lady->Respawn();
175
176 if (!Baron->IsAlive())
177 Baron->Respawn();
178
179 if (!Sir->IsAlive())
180 Sir->Respawn();
181
182 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Thane->AI())->encounterActionReset = true;
183 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Lady->AI())->encounterActionReset = true;
184 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Baron->AI())->encounterActionReset = true;
185 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Sir->AI())->encounterActionReset = true;
186
187 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Thane->AI())->EnterEvadeMode();
188 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Lady->AI())->EnterEvadeMode();
189 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Baron->AI())->EnterEvadeMode();
190 ENSURE_AI(boss_four_horsemen::boss_four_horsemenAI, Sir->AI())->EnterEvadeMode();
191 }
192 }
193
194 if (checkAllDead)
195 return !Thane->IsAlive() && !Lady->IsAlive() && !Baron->IsAlive() && !Sir->IsAlive();
196 }
197 return false;
198 }

200 void BeginFourHorsemenMovement()
201 {
202 movementStarted = true;
203 me->SetReactState(REACT_PASSIVE);
204 me->SetWalk(false);
205 me->SetSpeed(MOVE_RUN, me->GetSpeedRate(MOVE_RUN), true);
206
207 switch (id)
208 {
209 case HORSEMEN_THANE:
210 me->GetMotionMaster()->MovePoint(0, WaypointPositions[0]);
211 break;
212 case HORSEMEN_LADY:
213 me->GetMotionMaster()->MovePoint(3, WaypointPositions[3]);
214 break;
215 case HORSEMEN_BARON:
216 me->GetMotionMaster()->MovePoint(6, WaypointPositions[6]);
217 break;
218 case HORSEMEN_SIR:
219 me->GetMotionMaster()->MovePoint(9, WaypointPositions[9]);
220 break;
221 }
222 }

224 void MovementInform(uint32 type, uint32 point) override
225 {
226 if (type != POINT_MOTION_TYPE)
227 return;
228
229 if (point == 2 || point == 5 || point == 8 || point == 11)
230 {
231 movementCompleted = true;
232 me->SetReactState(REACT_AGGRESSIVE);
233
234 Unit* eventStarter = ObjectAccessor::GetUnit(*me, uiEventStarterGUID);
235
236 if (eventStarter && me->IsValidAttackTarget(eventStarter))
237 AttackStart(eventStarter);
238 else if (!UpdateVictim())
239 {
240 EnterEvadeMode();
241 return;
242 }
243
244 if (caster)
245 {
246 me->GetMotionMaster()->Clear();
247 me->GetMotionMaster()->MoveIdle();
248 }
249
250 return;
251 }
252
253 nextMovementStarted = false;
254 nextWP = point + 1;
255 }

262 me->getThreatManager().modifyThreatPercent(me->GetVictim(), -100);
263 me->AddThreat(who, 1000000.0f);
264 }

275 void AttackStart(Unit* who) override
276 {
277 if (!movementCompleted && !movementStarted)
278 {
279 uiEventStarterGUID = who->GetGUID();
280 BeginFourHorsemenMovement();
281
282 if (!encounterActionAttack)
283 DoEncounteraction(who, true, false, false);
284 }
285 else if (movementCompleted && movementStarted)
286 {
287 if (caster)
288 me->Attack(who, false);
289 else
290 BossAI::AttackStart(who);
291 }
292 }

300 void JustDied(Unit* /*killer*/) override
301 {
302 events.Reset();
303 summons.DespawnAll();
304
305 instance->SetData(DATA_HORSEMEN0 + id, DONE);
306
307 if (DoEncounteraction(NULL, false, false, true))
308 {
309 instance->SetBossState(BOSS_HORSEMEN, DONE);
310 instance->SaveToDB();
311
312 // Achievements related to the 4-horsemen are given through spell 59450 which does not exist.
313 // There is thus no way it can be given by casting the spell on the players.
314 instance->DoUpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, 59450);
315 }
316
317 Talk(SAY_DEATH);
318 }

322 _EnterCombat();

329
330 void UpdateAI(uint32 diff) override
331 {
332 if (nextWP && movementStarted && !movementCompleted && !nextMovementStarted)
333 {
334 nextMovementStarted = true;
335 me->GetMotionMaster()->MovePoint(nextWP, WaypointPositions[nextWP]);
336 }
337
338 if (!UpdateVictim() || !CheckInRoom() || !movementCompleted)
339 return;
340
341 events.Update(diff);
342
343 if (me->HasUnitState(UNIT_STATE_CASTING))
344 return;
345
346 while (uint32 eventId = events.ExecuteEvent())
347 {
348 switch (eventId)
349 {
350 case EVENT_MARK:
351 if (!(rand32() % 5))
352 Talk(SAY_SPECIAL);
353 DoCastAOE(SPELL_MARK[id]);
354 events.ScheduleEvent(EVENT_MARK, 15000);
355 break;
356 case EVENT_CAST:
357 if (!(rand32() % 5))
358 Talk(SAY_TAUNT);
359
360 if (caster)
361 {
362 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
363 DoCast(target, SPELL_PRIMARY(id));
364 }
365 else
366 DoCastVictim(SPELL_PRIMARY(id));
367
368 events.ScheduleEvent(EVENT_CAST, 15000);
369 break;
370 case EVENT_BERSERK:
371 Talk(SAY_SPECIAL);
372 DoCast(me, EVENT_BERSERK);
373 break;
374 }
375 }
376
377 if (punishTimer <= diff)
378 {
379 if (doDelayPunish)
380 {
381 DoCastAOE(SPELL_PUNISH[id], true);
382 doDelayPunish = false;
383 }
384 punishTimer = 2000;
385 } else punishTimer -= diff;
386
387 if (!caster)
388 DoMeleeAttackIfReady();
389 else if ((!DoSpellAttackIfReady(SPELL_SECONDARY(id)) || !me->IsWithinLOSInMap(me->GetVictim())) && movementCompleted && !doDelayPunish)
390 doDelayPunish = true;
391 }
392 };

396class spell_four_horsemen_mark : public SpellScriptLoader

398 public:

401 class spell_four_horsemen_mark_AuraScript : public AuraScript
402 {
403 PrepareAuraScript(spell_four_horsemen_mark_AuraScript);
404
405 void OnApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
406 {
407 if (Unit* caster = GetCaster())
408 {
409 int32 damage;
410 switch (GetStackAmount())
411 {
412 case 1:
413 damage = 0;
414 break;
415 case 2:
416 damage = 500;
417 break;
418 case 3:
419 damage = 1000;
420 break;
421 case 4:
422 damage = 1500;
423 break;
424 case 5:
425 damage = 4000;
426 break;
427 case 6:
428 damage = 12000;
429 break;
430 default:
431 damage = 20000 + 1000 * (GetStackAmount() - 7);
432 break;
433 }
434 if (damage)
435 caster->CastCustomSpell(SPELL_MARK_DAMAGE, SPELLVALUE_BASE_POINT0, damage, GetTarget());
436 }
437 }
438
439 void Register() override
440 {
441 AfterEffectApply += AuraEffectApplyFn(spell_four_horsemen_mark_AuraScript::OnApply, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL_OR_REAPPLY_MASK);
442 }
443 };

446 {
447 return new spell_four_horsemen_mark_AuraScript();

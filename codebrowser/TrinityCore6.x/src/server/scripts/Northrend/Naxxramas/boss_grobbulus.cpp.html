48class boss_grobbulus : public CreatureScript
49{
50 public:
51 boss_grobbulus() : CreatureScript("boss_grobbulus") { }
52
53 struct boss_grobbulusAI : public BossAI
54 {
55 boss_grobbulusAI(Creature* creature) : BossAI(creature, BOSS_GROBBULUS) { }
56
57 void EnterCombat(Unit* /*who*/) override
58 {
59 _EnterCombat();
60 events.ScheduleEvent(EVENT_CLOUD, 15000);
61 events.ScheduleEvent(EVENT_INJECT, 20000);
62 events.ScheduleEvent(EVENT_SPRAY, urand(15000, 30000)); // not sure
63 events.ScheduleEvent(EVENT_BERSERK, 12 * 60000);
64 }
65
66 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
67 {
68 if (spell->Id == SPELL_SLIME_SPRAY)
69 me->SummonCreature(NPC_FALLOUT_SLIME, *target, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT);
70 }
71
72 void UpdateAI(uint32 diff) override
73 {
74 if (!UpdateVictim())
75 return;
76
77 events.Update(diff);
78
79 while (uint32 eventId = events.ExecuteEvent())
80 {
81 switch (eventId)
82 {
83 case EVENT_CLOUD:
84 DoCastAOE(SPELL_POISON_CLOUD);
85 events.ScheduleEvent(EVENT_CLOUD, 15000);
86 return;
87 case EVENT_BERSERK:
88 DoCastAOE(SPELL_BERSERK, true);
89 return;
90 case EVENT_SPRAY:
91 DoCastAOE(SPELL_SLIME_SPRAY);
92 events.ScheduleEvent(EVENT_SPRAY, urand(15000, 30000));
93 return;
94 case EVENT_INJECT:
95 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true, -SPELL_MUTATING_INJECTION))
96 DoCast(target, SPELL_MUTATING_INJECTION);
97 events.ScheduleEvent(EVENT_INJECT, 8000 + uint32(120 * me->GetHealthPct()));
98 return;
99 default:
100 break;
101 }
102 }
103
104 DoMeleeAttackIfReady();
105 }
106 };
107
108 CreatureAI* GetAI(Creature* creature) const override
109 {
110 return new boss_grobbulusAI(creature);
111 }
112};

142// 28169 - Mutating Injection
143class spell_grobbulus_mutating_injection : public SpellScriptLoader
144{
145 public:
146 spell_grobbulus_mutating_injection() : SpellScriptLoader("spell_grobbulus_mutating_injection") { }
147
148 class spell_grobbulus_mutating_injection_AuraScript : public AuraScript
149 {
150 PrepareAuraScript(spell_grobbulus_mutating_injection_AuraScript);
151
152 bool Validate(SpellInfo const* /*spellInfo*/) override
153 {
154 if (!sSpellMgr->GetSpellInfo(SPELL_MUTATING_EXPLOSION)
155 || !sSpellMgr->GetSpellInfo(SPELL_POISON_CLOUD))
156 return false;
157 return true;
158 }
159
160 void HandleRemove(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
161 {
162 AuraRemoveMode removeMode = GetTargetApplication()->GetRemoveMode();
163 if (removeMode != AURA_REMOVE_BY_ENEMY_SPELL && removeMode != AURA_REMOVE_BY_EXPIRE)
164 return;
165
166 if (Unit* caster = GetCaster())
167 {
168 caster->CastSpell(GetTarget(), SPELL_MUTATING_EXPLOSION, true);
169 GetTarget()->CastSpell(GetTarget(), SPELL_POISON_CLOUD, true, NULL, aurEff, GetCasterGUID());
170 }
171 }
172
173 void Register() override
174 {
175 AfterEffectRemove += AuraEffectRemoveFn(spell_grobbulus_mutating_injection_AuraScript::HandleRemove, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
176 }
177 };
178
179 AuraScript* GetAuraScript() const override
180 {
181 return new spell_grobbulus_mutating_injection_AuraScript();
182 }
183};

185// 28158, 54362 - Poison (Grobbulus)
186class spell_grobbulus_poison_cloud : public SpellScriptLoader
187{
188 public:
189 spell_grobbulus_poison_cloud() : SpellScriptLoader("spell_grobbulus_poison_cloud") { }
190
191 class spell_grobbulus_poison_cloud_AuraScript : public AuraScript
192 {
193 PrepareAuraScript(spell_grobbulus_poison_cloud_AuraScript);
194
195 bool Validate(SpellInfo const* spellInfo) override
196 {
197 if (!sSpellMgr->GetSpellInfo(spellInfo->GetEffect(EFFECT_0)->TriggerSpell))
198 return false;
199 return true;
200 }
201
202 void PeriodicTick(AuraEffect const* aurEff)
203 {
204 PreventDefaultAction();
205
206 uint32 triggerSpell = GetSpellInfo()->GetEffect(aurEff->GetEffIndex())->TriggerSpell;
207 int32 mod = int32(((float(aurEff->GetTickNumber()) / aurEff->GetTotalTicks()) * 0.9f + 0.1f) * 10000 * 2 / 3);
208 GetTarget()->CastCustomSpell(triggerSpell, SPELLVALUE_RADIUS_MOD, mod, (Unit*)NULL, TRIGGERED_FULL_MASK, NULL, aurEff);
209 }
210
211 void Register() override
212 {
213 OnEffectPeriodic += AuraEffectPeriodicFn(spell_grobbulus_poison_cloud_AuraScript::PeriodicTick, EFFECT_0, SPELL_AURA_PERIODIC_TRIGGER_SPELL);
214 }
215 };
216
217 AuraScript* GetAuraScript() const override
218 {
219 return new spell_grobbulus_poison_cloud_AuraScript();
220 }

144// Predicate function to check that the r efzr unit is NOT on the same side as the source.
145struct NotOnSameSide : public std::unary_function<Unit*, bool>
146{
147 NotOnSameSide(Unit* source) : _onLiveSide(IN_LIVE_SIDE(source)) { }
148
149 bool operator() (Unit const* target)
150 {
151 return (_onLiveSide != IN_LIVE_SIDE(target));
152 }
153
154 private:
155 bool _onLiveSide;

197 void EnterCombat(Unit* /*who*/) override
198 {
199 for (uint32 i = 0; i < POS_LIVE; ++i)
200 if (Creature* trigger = DoSummon(WORLD_TRIGGER, PosSummonLive[i]))
201 LiveTriggerGUID.push_back(trigger->GetGUID());
202 for (uint32 i = 0; i < POS_DEAD; ++i)
203 if (Creature* trigger = DoSummon(WORLD_TRIGGER, PosSummonDead[i]))
204 DeadTriggerGUID.push_back(trigger->GetGUID());
205
206 if (LiveTriggerGUID.size() < POS_LIVE || DeadTriggerGUID.size() < POS_DEAD)
207 {
208 TC_LOG_ERROR("scripts", "Script Gothik: cannot summon triggers!");
209 EnterEvadeMode();
210 return;
211 }
212
213 _EnterCombat();
214 waveCount = 0;
215 events.ScheduleEvent(EVENT_SUMMON, 30000);
216 DoTeleportTo(PosPlatform);
217 Talk(SAY_SPEECH);
218 instance->SetData(DATA_GOTHIK_GATE, GO_STATE_READY);
219 }

258 void DoGothikSummon(uint32 entry)
259 {
260 if (GetDifficulty() == DIFFICULTY_25_N)
261 {
262 switch (entry)
263 {
264 case NPC_LIVE_TRAINEE:
265 {
266 if (Creature* liveTrigger = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[0]))
267 DoSummon(NPC_LIVE_TRAINEE, liveTrigger, 1);
268 if (Creature* liveTrigger1 = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[1]))
269 DoSummon(NPC_LIVE_TRAINEE, liveTrigger1, 1);
270 if (Creature* liveTrigger2 = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[2]))
271 DoSummon(NPC_LIVE_TRAINEE, liveTrigger2, 1);
272 break;
273 }
274 case NPC_LIVE_KNIGHT:
275 {
276 if (Creature* liveTrigger3 = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[3]))
277 DoSummon(NPC_LIVE_KNIGHT, liveTrigger3, 1);
278 if (Creature* liveTrigger5 = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[5]))
279 DoSummon(NPC_LIVE_KNIGHT, liveTrigger5, 1);
280 break;
281 }
282 case NPC_LIVE_RIDER:
283 {
284 if (Creature* liveTrigger4 = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[4]))
285 DoSummon(NPC_LIVE_RIDER, liveTrigger4, 1);
286 break;
287 }
288 }
289 }
290 else
291 {
292 switch (entry)
293 {
294 case NPC_LIVE_TRAINEE:
295 {
296 if (Creature* liveTrigger = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[4]))
297 DoSummon(NPC_LIVE_TRAINEE, liveTrigger, 1);
298 if (Creature* liveTrigger2 = ObjectAccessor::GetCreature(*me, LiveTriggerGUID[4]))
299 DoSummon(NPC_LIVE_TRAINEE, liveTrigger2, 1);

302 case NPC_LIVE_KNIGHT:

311 DoSummon(NPC_LIVE_RIDER, liveTrigger4, 1);

313 }
314 }
315 }

318 bool CheckGroupSplitted()
319 {
320 bool checklife = false;
321 bool checkdead = false;
322 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
323 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
324 {
325 if (i->GetSource() && i->GetSource()->IsAlive() &&
326 i->GetSource()->GetPositionX() <= POS_X_NORTH &&
327 i->GetSource()->GetPositionX() >= POS_X_SOUTH &&
328 i->GetSource()->GetPositionY() <= POS_Y_GATE &&
329 i->GetSource()->GetPositionY() >= POS_Y_EAST)
330 {
331 checklife = true;
332 }
333 else if (i->GetSource() && i->GetSource()->IsAlive() &&
334 i->GetSource()->GetPositionX() <= POS_X_NORTH &&
335 i->GetSource()->GetPositionX() >= POS_X_SOUTH &&
336 i->GetSource()->GetPositionY() >= POS_Y_GATE &&
337 i->GetSource()->GetPositionY() <= POS_Y_WEST)
338 {
339 checkdead = true;
340 }
341
342 if (checklife && checkdead)
343 return true;
344 }
345
346 return false;
347 }

349 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
350 {
351 uint32 spellId = 0;
352 switch (spell->Id)
353 {
354 case SPELL_INFORM_LIVE_TRAINEE: spellId = SPELL_INFORM_DEAD_TRAINEE; break;
355 case SPELL_INFORM_LIVE_KNIGHT: spellId = SPELL_INFORM_DEAD_KNIGHT; break;
356 case SPELL_INFORM_LIVE_RIDER: spellId = SPELL_INFORM_DEAD_RIDER; break;
357 }
358 if (spellId && me->IsInCombat())
359 {
360 me->HandleEmoteCommand(EMOTE_ONESHOT_SPELL_CAST);
361 if (Creature* pRandomDeadTrigger = ObjectAccessor::GetCreature(*me, DeadTriggerGUID[rand32() % POS_DEAD]))
362 me->CastSpell(pRandomDeadTrigger, spellId, true);
363 }
364 }

369 damage = 0;

372 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
373 {
374 if (!me->IsInCombat())
375 return;
376
377 switch (spell->Id)
378 {
379 case SPELL_INFORM_DEAD_TRAINEE:
380 DoSummon(NPC_DEAD_TRAINEE, target, 0);
381 break;
382 case SPELL_INFORM_DEAD_KNIGHT:
383 DoSummon(NPC_DEAD_KNIGHT, target, 0);
384 break;
385 case SPELL_INFORM_DEAD_RIDER:
386 DoSummon(NPC_DEAD_RIDER, target, 1.0f);
387 DoSummon(NPC_DEAD_HORSE, target, 1.0f);
388 break;
389 }
390 }

392 void UpdateAI(uint32 diff) override
393 {
394 if (!UpdateVictim() || !CheckInRoom())
395 return;
396
397 events.Update(diff);
398
399 if (!thirtyPercentReached && HealthBelowPct(30) && phaseTwo)
400 {
401 thirtyPercentReached = true;
402 instance->SetData(DATA_GOTHIK_GATE, GO_STATE_ACTIVE);
403 }
404
405 if (me->HasUnitState(UNIT_STATE_CASTING))
406 return;
407
408 while (uint32 eventId = events.ExecuteEvent())
409 {
410 switch (eventId)
411 {
412 case EVENT_SUMMON:
413 if (waves[waveCount].entry)
414 {
415 if ((waves[waveCount].mode == 2) && (GetDifficulty() == DIFFICULTY_25_N))
416 DoGothikSummon(waves[waveCount].entry);
417 else if ((waves[waveCount].mode == 0) && (GetDifficulty() == DIFFICULTY_10_N))
418 DoGothikSummon(waves[waveCount].entry);
419 else if (waves[waveCount].mode == 1)
420 DoGothikSummon(waves[waveCount].entry);
421
422 // if group is not splitted, open gate and merge both sides at ~ 2 minutes (wave 11)
423 if (waveCount == 11)
424 {
425 if (!CheckGroupSplitted())
426 {
427 instance->SetData(DATA_GOTHIK_GATE, GO_STATE_ACTIVE);
428 DummyEntryCheckPredicate pred;
429 summons.DoAction(0, pred); //! Magic numbers fail
430 summons.DoZoneInCombat();
431 mergedSides = true;
432 }
433 }
434
435 if (waves[waveCount].mode == 1)
436 events.ScheduleEvent(EVENT_SUMMON, waves[waveCount].time);
437 else if ((waves[waveCount].mode == 2) && (GetDifficulty() == DIFFICULTY_25_N))
438 events.ScheduleEvent(EVENT_SUMMON, waves[waveCount].time);
439 else if ((waves[waveCount].mode == 0) && (GetDifficulty() == DIFFICULTY_10_N))
440 events.ScheduleEvent(EVENT_SUMMON, waves[waveCount].time);
441 else
442 events.ScheduleEvent(EVENT_SUMMON, 0);
443
444 ++waveCount;
445 }
446 else
447 {
448 phaseTwo = true;
449 Talk(SAY_TELEPORT);
450 DoTeleportTo(PosGroundLiveSide);
451 me->SetReactState(REACT_AGGRESSIVE);
452 DummyEntryCheckPredicate pred;
453 summons.DoAction(0, pred); //! Magic numbers fail
454 summons.DoZoneInCombat();
455 events.ScheduleEvent(EVENT_BOLT, 1000);
456 events.ScheduleEvent(EVENT_HARVEST, urand(3000, 15000));
457 events.ScheduleEvent(EVENT_TELEPORT, 20000);
458 }
459 break;
460 case EVENT_BOLT:
461 DoCastVictim(SPELL_SHADOW_BOLT);
462 events.ScheduleEvent(EVENT_BOLT, 1000);
463 break;
464 case EVENT_HARVEST:
465 DoCastVictim(SPELL_HARVEST_SOUL, true);
466 events.ScheduleEvent(EVENT_HARVEST, urand(20000, 25000));
467 break;
468 case EVENT_TELEPORT:
469 if (!thirtyPercentReached)
470 {
471 me->AttackStop();
472 if (IN_LIVE_SIDE(me))
473 DoTeleportTo(PosGroundDeadSide);
474 else
475 DoTeleportTo(PosGroundLiveSide);
476
477 me->getThreatManager().resetAggro(NotOnSameSide(me));
478 if (Unit* target = SelectTarget(SELECT_TARGET_NEAREST, 0))
479 {
480 me->getThreatManager().addThreat(target, 100.0f);
481 AttackStart(target);
482 }
483
484 events.ScheduleEvent(EVENT_TELEPORT, 20000);
485 }
486 break;
487 }
488 }
489
490 if (!phaseTwo)
491 DoMeleeAttackIfReady();
492 }
493 };

529 if (gateClose && !isOnSameSide(attacker))

531 }
532
533 void JustDied(Unit* /*killer*/) override
534 {
535 if (me->IsSummon())
536 if (Unit* owner = me->ToTempSummon()->GetSummoner())
537 CombatAI::JustDied(owner);
538 }
539
540 void EnterEvadeMode() override
541 {
542 if (!gateClose)
543 {
544 CombatAI::EnterEvadeMode();
545 return;
546 }
547
548 if (!_EnterEvadeMode())
549 return;
550
551 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
552 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
553 {
554 if (i->GetSource() && i->GetSource()->IsAlive() && isOnSameSide(i->GetSource()))
555 {
556 AttackStart(i->GetSource());
557 return;
558 }
559 }
560
561 me->GetMotionMaster()->MoveIdle();
562 Reset();
563 }
564
565 void UpdateAI(uint32 diff) override
566 {
567 if (gateClose && (!isOnSameSide(me) || (me->GetVictim() && !isOnSameSide(me->GetVictim()))))
568 {
569 EnterEvadeMode();
570 return;
571 }
572
573 CombatAI::UpdateAI(diff);
574 }
575 };

579 return new npc_gothik_minionAI(creature);

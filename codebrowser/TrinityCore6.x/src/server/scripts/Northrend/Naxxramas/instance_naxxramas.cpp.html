79// from P2 teleport spell stored target
80float const HeiganPos[2] = { 2793.86f, -3707.38f };
81float const HeiganEruptionSlope[3] =
82{
83 (-3703.303223f - HeiganPos[1]) / (2777.494141f - HeiganPos[0]), // between right center and far right
84 (-3696.948242f - HeiganPos[1]) / (2785.624268f - HeiganPos[0]), // between left and right halves
85 (-3691.880615f - HeiganPos[1]) / (2790.280029f - HeiganPos[0]) // between far left and left center
86};
87
88// 0 H x
89// 1 ^
90// 2 |
91// 3 y<--o
92inline uint32 GetEruptionSection(float x, float y)
93{
94 y -= HeiganPos[1];
95 if (y < 1.0f)
96 return 0;
97
98 x -= HeiganPos[0];
99 if (x > -1.0f)
100 return 3;
101
102 float slope = y/x;
103 for (uint32 i = 0; i < 3; ++i)
104 if (slope > HeiganEruptionSlope[i])
105 return i;
106 return 3;
107}

352 break;

420 bool SetBossState(uint32 id, EncounterState state) override
421 {
422 if (!InstanceScript::SetBossState(id, state))
423 return false;
424
425 switch (id)
426 {
427 case BOSS_MAEXXNA:
428 if (state == DONE)
429 {
430 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_ARACHNID))
431 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
432
433 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
434 }
435 break;
436 case BOSS_LOATHEB:
437 if (state == DONE)
438 {
439 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_PLAGUE))
440 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
441
442 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
443 }
444 break;
445 case BOSS_THADDIUS:
446 if (state == DONE)
447 {
448 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_CONSTRUCT))
449 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
450
451 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
452 }
453 break;
454 case BOSS_GOTHIK:
455 if (state == DONE)
456 events.ScheduleEvent(EVENT_DIALOGUE_GOTHIK_KORTHAZZ, 10000);
457 break;
458 case BOSS_HORSEMEN:
459 if (state == DONE)
460 {
461 if (GameObject* horsemenChest = instance->GetGameObject(HorsemenChestGUID))
462 {
463 horsemenChest->SetRespawnTime(horsemenChest->GetRespawnDelay());
464 horsemenChest->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
465 }
466
467 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_MILITARY))
468 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
469
470 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
471 }
472 break;
473 case BOSS_SAPPHIRON:
474 if (state == DONE)
475 events.ScheduleEvent(EVENT_DIALOGUE_SAPPHIRON_KELTHUZAD, 6000);
476 break;
477 default:
478 break;
479 }
480
481 return true;
482 }

580 void HeiganErupt(uint32 section)
581 {
582 for (uint32 i = 0; i < 4; ++i)
583 {
584 if (i == section)
585 continue;
586
587 for (ObjectGuid guid : HeiganEruptionGUID[i])
588 {

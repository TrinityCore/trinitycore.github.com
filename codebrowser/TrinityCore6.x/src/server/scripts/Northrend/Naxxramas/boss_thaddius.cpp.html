168 public:
169 boss_thaddiusAI(Creature* creature) : BossAI(creature, BOSS_THADDIUS), stalaggAlive(true), feugenAlive(true), ballLightningEnabled(false), shockingEligibility(true)
170 {
171 if (instance->GetBossState(BOSS_THADDIUS) != DONE)
172 {
173 events.SetPhase(PHASE_NOT_ENGAGED);
174 SetCombatMovement(false);

176 // initialize everything properly, and ensure that the coils are loaded by the time we initialize

190 BeginResetEncounter();

193 void JustDied(Unit* /*killer*/) override
194 {
195 _JustDied();
196 me->setActive(false);
197 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
198 stalagg->setActive(false);
199 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
200 feugen->setActive(false);
201 Talk(SAY_DEATH);

204 void DoAction(int32 action) override
205 {
206 switch (action)
207 {
208 case ACTION_RESET_ENCOUNTER_TIMER:
209 if (events.IsInPhase(PHASE_RESETTING))
210 ResetEncounter();
211 break;
212 case ACTION_FEUGEN_RESET:
213 case ACTION_STALAGG_RESET:
214 if (!events.IsInPhase(PHASE_NOT_ENGAGED) && !events.IsInPhase(PHASE_RESETTING))
215 BeginResetEncounter();
216 break;
217 case ACTION_FEUGEN_AGGRO:
218 case ACTION_STALAGG_AGGRO:
219 if (events.IsInPhase(PHASE_RESETTING))
220 return BeginResetEncounter();
221 if (!events.IsInPhase(PHASE_NOT_ENGAGED))
222 return;
223 events.SetPhase(PHASE_PETS);
224
225 shockingEligibility = true;
226
227 if (!instance->CheckRequiredBosses(BOSS_THADDIUS))
228 return BeginResetEncounter();
229 instance->SetBossState(BOSS_THADDIUS, IN_PROGRESS);
230
231 me->setActive(true);
232 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
233 stalagg->setActive(true);
234 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
235 feugen->setActive(true);
236 break;
237 case ACTION_FEUGEN_DIED:
238 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
239 feugen->AI()->DoAction(ACTION_FEUGEN_REVIVING_FX);
240 feugenAlive = false;
241 if (stalaggAlive)
242 events.ScheduleEvent(EVENT_REVIVE_FEUGEN, 5 * IN_MILLISECONDS, 0, PHASE_PETS);
243 else
244 Transition();
245
246 break;
247 case ACTION_STALAGG_DIED:
248 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
249 stalagg->AI()->DoAction(ACTION_STALAGG_REVIVING_FX);
250 stalaggAlive = false;
251 if (feugenAlive)
252 events.ScheduleEvent(EVENT_REVIVE_STALAGG, 5 * IN_MILLISECONDS, 0, PHASE_PETS);
253 else
254 Transition();
255
256 break;
257
258 case ACTION_POLARITY_CROSSED:
259 shockingEligibility = false;
260 break;
261 default:
262 break;
263 }
264 }

281
282 void BeginResetEncounter(bool initial = false)
283 {
284 if (instance->GetBossState(BOSS_THADDIUS) == DONE)
285 return;
286 if (events.IsInPhase(PHASE_RESETTING))
287 return;
288
289 if (initial) // signal shorter spawn timer to instance script
290 instance->SetBossState(BOSS_THADDIUS, SPECIAL);
291 instance->ProcessEvent(me, EVENT_THADDIUS_BEGIN_RESET);
292 instance->SetBossState(BOSS_THADDIUS, NOT_STARTED);
293
294 // remove polarity shift debuffs on reset
295 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_POSITIVE_CHARGE_APPLY);
296 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_NEGATIVE_CHARGE_APPLY);
297
298 me->DespawnOrUnsummon();
299
300 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_STUNNED);
301 events.SetPhase(PHASE_RESETTING);
302 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
303 feugen->AI()->DoAction(ACTION_BEGIN_RESET_ENCOUNTER);
304 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
305 stalagg->AI()->DoAction(ACTION_BEGIN_RESET_ENCOUNTER);
306
307 me->setActive(false);
308 }
309
310 void ResetEncounter()
311 {
312 feugenAlive = true;
313 stalaggAlive = true;
314
315 me->Respawn(true);
316 _Reset();
317 events.SetPhase(PHASE_NOT_ENGAGED);
318
319 me->CastSpell(me, SPELL_THADDIUS_INACTIVE_VISUAL, true);
320
321 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
322 feugen->AI()->DoAction(ACTION_RESET_ENCOUNTER);
323 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
324 stalagg->AI()->DoAction(ACTION_RESET_ENCOUNTER);
325 }

327 void UpdateAI(uint32 diff) override
328 {
329 if (events.IsInPhase(PHASE_NOT_ENGAGED))
330 return;
331 if (events.IsInPhase(PHASE_THADDIUS) && !UpdateVictim())
332 return;
333
334 events.Update(diff);
335 while (uint32 eventId = events.ExecuteEvent())
336 {
337 switch (eventId)
338 {
339 case EVENT_REVIVE_FEUGEN:
340 feugenAlive = true;
341 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
342 feugen->AI()->DoAction(ACTION_FEUGEN_REVIVED);
343 break;
344 case EVENT_REVIVE_STALAGG:
345 stalaggAlive = true;
346 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
347 stalagg->AI()->DoAction(ACTION_STALAGG_REVIVED);
348 break;
349 case EVENT_TRANSITION_1: // tesla coils overload
350 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
351 feugen->AI()->DoAction(ACTION_TRANSITION);
352 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
353 stalagg->AI()->DoAction(ACTION_TRANSITION);
354 break;
355 case EVENT_TRANSITION_2: // tesla coils shock thaddius
356 me->CastSpell(me, SPELL_THADDIUS_SPARK_VISUAL, true);
357 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
358 feugen->AI()->DoAction(ACTION_TRANSITION_2);
359 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
360 stalagg->AI()->DoAction(ACTION_TRANSITION_2);
361 break;
362 case EVENT_TRANSITION_3: // thaddius becomes active
363 me->CastSpell(me, SPELL_THADDIUS_SPARK_VISUAL, true);
364 ballLightningEnabled = false;
365 me->RemoveAura(SPELL_THADDIUS_INACTIVE_VISUAL);
366 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED);
367 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
368
369 DoZoneInCombat();
370 if (Unit* closest = SelectTarget(SELECT_TARGET_NEAREST, 0, 500.0f))
371 AttackStart(closest);
372 else // if there is no nearest target, then there is no target, meaning we should reset
373 return BeginResetEncounter();
374
375 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
376 feugen->AI()->DoAction(ACTION_TRANSITION_3);
377 if (Creature* stalagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_STALAGG)))
378 stalagg->AI()->DoAction(ACTION_TRANSITION_3);
379
380 events.SetPhase(PHASE_THADDIUS);
381
382 Talk(SAY_AGGRO);
383
384 events.ScheduleEvent(EVENT_ENABLE_BALL_LIGHTNING, 5 * IN_MILLISECONDS, 0, PHASE_THADDIUS);
385 events.ScheduleEvent(EVENT_SHIFT, 10 * IN_MILLISECONDS, 0, PHASE_THADDIUS);
386 events.ScheduleEvent(EVENT_CHAIN, urand(10, 20) * IN_MILLISECONDS, 0, PHASE_THADDIUS);
387 events.ScheduleEvent(EVENT_BERSERK, 6 * MINUTE * IN_MILLISECONDS, 0, PHASE_THADDIUS);
388
389 break;
390 case EVENT_ENABLE_BALL_LIGHTNING:
391 ballLightningEnabled = true;
392 break;
393 case EVENT_SHIFT:
394 me->CastStop(); // shift overrides all other spells
395 DoCastAOE(SPELL_POLARITY_SHIFT);
396 events.ScheduleEvent(EVENT_SHIFT_TALK, 3 * IN_MILLISECONDS, PHASE_THADDIUS);
397 events.ScheduleEvent(EVENT_SHIFT, 30 * IN_MILLISECONDS, PHASE_THADDIUS);
398 break;
399 case EVENT_SHIFT_TALK:
400 Talk(SAY_ELECT);
401 Talk(EMOTE_POLARITY_SHIFTED);
402 break;
403 case EVENT_CHAIN:
404 if (me->FindCurrentSpellBySpellId(SPELL_POLARITY_SHIFT)) // delay until shift is over
405 events.ScheduleEvent(EVENT_CHAIN, 3 * IN_MILLISECONDS, 0, PHASE_THADDIUS);
406 else
407 {
408 me->CastStop();
409 DoCastVictim(SPELL_CHAIN_LIGHTNING);
410 events.ScheduleEvent(EVENT_CHAIN, urand(10, 20) * IN_MILLISECONDS, PHASE_THADDIUS);
411 }
412 break;
413 case EVENT_BERSERK:
414 me->CastStop();
415 DoCast(me, SPELL_BERSERK);
416 break;
417 default:
418 break;
419 }
420 }
421
422 if (events.IsInPhase(PHASE_THADDIUS))
423 {
424 if (me->IsWithinMeleeRange(me->GetVictim()))
425 DoMeleeAttackIfReady();
426 else
427 if (ballLightningEnabled)
428 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
429 DoCast(target, SPELL_BALL_LIGHTNING);
430 }
431 }
432
433 private:
434 bool stalaggAlive;
435 bool feugenAlive;
436 bool ballLightningEnabled;
437 bool shockingEligibility;
438 };

461 void Initialize()
462 {
463 if (GameObject* coil = myCoilGO())
464 coil->SetGoState(GO_STATE_ACTIVE);
465
466 // if the encounter reset while feigning death
467 me->SetStandState(UNIT_STAND_STATE_STAND);
468 me->SetReactState(REACT_AGGRESSIVE);
469 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
470 isOverloading = false;
471 isFeignDeath = false;
472
473 // force tesla coil state refresh
474 refreshBeam = true;
475
476 powerSurgeTimer = 10 * IN_MILLISECONDS;
477 }

479 void Reset() override
480 {
481 if (isFeignDeath || !me->IsAlive())
482 return;
483 if (Creature* thaddius = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_THADDIUS)))
484 thaddius->AI()->DoAction(ACTION_STALAGG_RESET);
485 }

488 {
489 if (GameObject* coil = myCoilGO())
490 coil->SetGoState(GO_STATE_READY);
491 me->DespawnOrUnsummon();
492 me->setActive(false);
493 }

495 void ResetEncounter()
496 {
497 me->Respawn(true);
498 Initialize();
499 }

501 void DoAction(int32 action) override
502 {
503 switch (action)
504 {
505 case ACTION_BEGIN_RESET_ENCOUNTER:
506 BeginResetEncounter();
507 break;
508 case ACTION_RESET_ENCOUNTER:
509 ResetEncounter();
510 break;
511 case ACTION_STALAGG_REVIVING_FX:
512 break;
513 case ACTION_STALAGG_REVIVED:
514 if (!isFeignDeath)
515 break;
516
517 me->SetFullHealth();
518 me->SetStandState(UNIT_STAND_STATE_STAND);
519 me->SetReactState(REACT_AGGRESSIVE);
520 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
521 Talk(EMOTE_FEIGN_REVIVE);
522 isFeignDeath = false;
523
524 refreshBeam = true; // force beam refresh
525
526 if (Creature* feugen = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_FEUGEN)))
527 if (feugen->GetVictim())
528 {
529 me->AddThreat(feugen->EnsureVictim(), 0.0f);
530 me->SetInCombatWith(feugen->EnsureVictim());
531 }
532 break;
533 case ACTION_TRANSITION:
534 me->KillSelf(); // true death
535 me->DespawnOrUnsummon();
536
537 if (Creature* coil = myCoil())
538 {
539 coil->CastStop();
540 coil->AI()->Talk(EMOTE_TESLA_OVERLOAD);
541 }
542 break;
543 case ACTION_TRANSITION_2:
544 if (Creature* coil = myCoil())
545 if (Creature* thaddius = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_THADDIUS)))
546 coil->CastSpell(thaddius, SPELL_SHOCK_VISUAL);
547 break;
548 case ACTION_TRANSITION_3:
549 if (GameObject* coil = myCoilGO())
550 coil->SetGoState(GO_STATE_READY);
551 break;
552 default:
553 break;
554 }
555 }

578 void DamageTaken(Unit* /*who*/, uint32& damage) override
579 {
580 if (damage < me->GetHealth())
581 return;
582
583 if (isFeignDeath) // don't take damage while feigning death
584 {
585 damage = 0;
586 return;
587 }
588
589 isFeignDeath = true;
590 isOverloading = false;
591
592 Talk(EMOTE_FEIGN_DEATH);
593 if (Creature* thaddius = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_THADDIUS)))
594 thaddius->AI()->DoAction(ACTION_STALAGG_DIED);
595
596 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
597 me->RemoveAllAuras();
598 me->SetReactState(REACT_PASSIVE);
599 me->AttackStop();
600 me->StopMoving();
601 me->SetStandState(UNIT_STAND_STATE_DEAD);
602
603 damage = 0;
604
605 // force beam refresh as we just removed auras
606 refreshBeam = true;
607 }
608
609 void SpellHit(Unit* caster, SpellInfo const* spell) override
610 {
611 if (!caster)
612 return;
613 if (spell->Id != SPELL_STALAGG_TESLA_PERIODIC)
614 return;
615 if (!isFeignDeath && me->IsInCombat() && !me->GetHomePosition().IsInDist(me, OVERLOAD_DISTANCE))
616 {
617 if (!isOverloading)
618 {
619 isOverloading = true;
620 caster->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
621 if (Creature* creatureCaster = caster->ToCreature())
622 creatureCaster->AI()->Talk(EMOTE_TESLA_LINK_BREAKS);
623 me->RemoveAura(SPELL_STALAGG_CHAIN_VISUAL);
624 }
625 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
626 {
627 caster->CastStop(SPELL_TESLA_SHOCK);
628 caster->CastSpell(target, SPELL_TESLA_SHOCK,true);
629 }
630 }
631 else if (isOverloading || refreshBeam)
632 {
633 isOverloading = false;
634 refreshBeam = false;
635 caster->CastStop();
636 caster->CastSpell(me, SPELL_STALAGG_CHAIN_VISUAL, true);
637 caster->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
638 }
639 }

641 void UpdateAI(uint32 uiDiff) override
642 {
643 if (!isFeignDeath)
644 if (!UpdateVictim())
645 return;
646
647 if (powerSurgeTimer <= uiDiff)
648 {
649 if (isFeignDeath) // delay until potential revive
650 powerSurgeTimer = 0u;
651 else
652 {
653 DoCast(me, SPELL_STALAGG_POWERSURGE);
654 powerSurgeTimer = urand(25, 30) * IN_MILLISECONDS;
655 }
656 }
657 else
658 powerSurgeTimer -= uiDiff;
659
660 if (!isFeignDeath)
661 DoMeleeAttackIfReady();
662 }

664 private:
665 Creature* myCoil()
666 {
667 Creature* coil = nullptr;

1022 void HandleTargets(std::list<WorldObject*>& targetList)
1023 {
1024 if (!GetTriggeringSpell())
1025 return;
1026
1027 uint32 triggeringId = GetTriggeringSpell()->Id;
1028 uint32 ampId;
1029 switch (triggeringId)
1030 {
1031 case SPELL_POSITIVE_CHARGE_APPLY:
1032 ampId = SPELL_POSITIVE_CHARGE_AMP;
1033 break;
1034 case SPELL_NEGATIVE_CHARGE_APPLY:
1035 ampId = SPELL_NEGATIVE_CHARGE_AMP;
1036 break;
1037 default:
1038 return;
1039 }
1040
1041 uint8 maxStacks = 0;
1042 if (GetCaster())
1043 switch (GetCaster()->GetMap()->GetDifficultyID())
1044 {
1045 case DIFFICULTY_10_N:
1046 maxStacks = MAX_POLARITY_10M;
1047 break;
1048 case DIFFICULTY_25_N:
1049 maxStacks = MAX_POLARITY_25M;
1050 break;
1051 default:
1052 break;
1053 }
1054
1055 uint8 stacksCount = 1; // do we get a stack for our own debuff?
1056 std::list<WorldObject*>::iterator it = targetList.begin();
1057 while(it != targetList.end())
1058 {
1059 if ((*it)->GetTypeId() != TYPEID_PLAYER)
1060 {
1061 it = targetList.erase(it);
1062 continue;
1063 }
1064 if ((*it)->ToPlayer()->HasAura(triggeringId))
1065 {
1066 it = targetList.erase(it);
1067 if (stacksCount < maxStacks)
1068 stacksCount++;
1069 continue;
1070 }
1071
1072 // this guy will get hit - achievement failure trigger
1073 if (Creature* thaddius = (*it)->FindNearestCreature(NPC_THADDIUS,200.0f))
1074 thaddius->AI()->DoAction(ACTION_POLARITY_CROSSED);
1075
1076 ++it;
1077 }
1078
1079 if (GetCaster() && GetCaster()->ToPlayer())
1080 {
1081 if (!GetCaster()->ToPlayer()->HasAura(ampId))
1082 GetCaster()->ToPlayer()->AddAura(ampId, GetCaster());
1083 GetCaster()->ToPlayer()->SetAuraStack(ampId, GetCaster(), stacksCount);
1084 }
1085 }

1088 {
1089 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_thaddius_polarity_charge_SpellScript::HandleTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ALLY);

1171 Unit* stalagg = ObjectAccessor::GetCreature(*feugen, feugen->GetInstanceScript()->GetGuidData(DATA_STALAGG));

1177

1190 stalaggThreat.addThreat(stalaggTank, feugenTankThreat - stalaggTankThreat);
1191
1192 feugen->CastSpell(stalaggTank, SPELL_MAGNETIC_PULL_EFFECT, true);

1198 float stalaggTankThreat = stalaggThreat.getThreat(stalaggTank);

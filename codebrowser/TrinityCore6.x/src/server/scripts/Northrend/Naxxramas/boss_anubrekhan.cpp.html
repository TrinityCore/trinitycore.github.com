75class boss_anubrekhan : public CreatureScript
76{
77public:
78 boss_anubrekhan() : CreatureScript("boss_anubrekhan") { }
79
80 CreatureAI* GetAI(Creature* creature) const override
81 {
82 return GetInstanceAI<boss_anubrekhanAI>(creature);
83 }
84
85 struct boss_anubrekhanAI : public BossAI
86 {
87 boss_anubrekhanAI(Creature* creature) : BossAI(creature, BOSS_ANUBREKHAN) { }
88
89 void SummonGuards()
90 {
91 if (Is25ManRaid())
92 me->SummonCreatureGroup(GROUP_INITIAL_25M);
93 }
94
95 void InitializeAI() override
96 {
97 if (!me->isDead())
98 {
99 Reset();
100 SummonGuards();
101 }
102 }
103
104 void Reset() override
105 {
106 _Reset();
107 guardCorpses.clear();
108 }
109
110 void JustReachedHome() override
111 {
112 _JustReachedHome();
113 SummonGuards();
114 }
115
116 void JustSummoned(Creature* summon) override
117 {
118 BossAI::JustSummoned(summon);
119
120 if (me->IsInCombat())
121 if (summon->GetEntry() == NPC_CRYPT_GUARD)
122 summon->AI()->Talk(EMOTE_SPAWN, me);
123 }
124
125 void SummonedCreatureDies(Creature* summon, Unit* killer) override
126 {
127 BossAI::SummonedCreatureDies(summon, killer);
128
129 if (summon->GetEntry() == NPC_CRYPT_GUARD)
130 guardCorpses.insert(summon->GetGUID());
131 }
132
133 void SummonedCreatureDespawn(Creature* summon) override
134 {
135 BossAI::SummonedCreatureDespawn(summon);
136
137 if (summon->GetEntry() == NPC_CRYPT_GUARD)
138 guardCorpses.erase(summon->GetGUID());
139 }
140
141 void KilledUnit(Unit* victim) override
142 {
143 if (victim->GetTypeId() == TYPEID_PLAYER)
144 victim->CastSpell(victim, SPELL_SUMMON_CORPSE_SCARABS_PLR, true, nullptr, nullptr, me->GetGUID());
145
146 Talk(SAY_SLAY);
147 }
148
149 void JustDied(Unit* /*killer*/) override
150 {
151 _JustDied();
152
153 // start achievement timer (kill Maexna within 20 min)
154 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMED_START_EVENT);
155 }
156
157 void EnterCombat(Unit* /*who*/) override
158 {
159 _EnterCombat();
160 Talk(SAY_AGGRO);
161
162 summons.DoZoneInCombat();
163
164 events.SetPhase(PHASE_NORMAL);
165 events.ScheduleEvent(EVENT_IMPALE, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, PHASE_NORMAL);
166 events.ScheduleEvent(EVENT_SCARABS, urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS), 0, PHASE_NORMAL);
167 events.ScheduleEvent(EVENT_LOCUST, urand(80,120) * IN_MILLISECONDS, 0, PHASE_NORMAL);
168 events.ScheduleEvent(EVENT_BERSERK, 10 * MINUTE * IN_MILLISECONDS);
169
170 if (!Is25ManRaid())
171 events.ScheduleEvent(EVENT_SPAWN_GUARD, urand(15, 20) * IN_MILLISECONDS);
172 }
173
174 void UpdateAI(uint32 diff) override
175 {
176 if (!UpdateVictim() || !CheckInRoom())
177 return;
178
179 events.Update(diff);
180
181 while (uint32 eventId = events.ExecuteEvent())
182 {
183 switch (eventId)
184 {
185 case EVENT_IMPALE:
186 if (events.GetTimeUntilEvent(EVENT_LOCUST) < 5 * IN_MILLISECONDS) break; // don't chain impale tank -> locust swarm
187 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
188 DoCast(target, SPELL_IMPALE);
189 else
190 EnterEvadeMode();
191
192 events.ScheduleEvent(EVENT_IMPALE, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, PHASE_NORMAL);
193 break;
194 case EVENT_SCARABS:
195 events.ScheduleEvent(EVENT_SCARABS, urand(40 * IN_MILLISECONDS, 60 * IN_MILLISECONDS), 0, PHASE_NORMAL);
196
197 if (!guardCorpses.empty())
198 {
199 if (Creature* creatureTarget = ObjectAccessor::GetCreature(*me, Trinity::Containers::SelectRandomContainerElement(guardCorpses)))
200 {
201 creatureTarget->CastSpell(creatureTarget, SPELL_SUMMON_CORPSE_SCARABS_MOB, true, nullptr, nullptr, me->GetGUID());
202 creatureTarget->AI()->Talk(EMOTE_SCARAB);
203 creatureTarget->DespawnOrUnsummon();
204 }
205 }
206 break;
207 case EVENT_LOCUST:
208 Talk(EMOTE_LOCUST);
209 DoCast(me, SPELL_LOCUST_SWARM);
210 events.ScheduleEvent(EVENT_SPAWN_GUARD, 3 * IN_MILLISECONDS);
211
212 events.ScheduleEvent(EVENT_LOCUST_ENDS, RAID_MODE(19, 23) * IN_MILLISECONDS);
213 events.ScheduleEvent(EVENT_LOCUST, 90000);
214 events.SetPhase(PHASE_SWARM);
215 break;
216 case EVENT_LOCUST_ENDS:
217 events.ScheduleEvent(EVENT_IMPALE, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, PHASE_NORMAL);
218 events.ScheduleEvent(EVENT_SCARABS, urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS), 0, PHASE_NORMAL);
219 events.SetPhase(PHASE_NORMAL);
220 break;
221 case EVENT_SPAWN_GUARD:
222 me->SummonCreatureGroup(GROUP_SINGLE_SPAWN);
223 break;
224 case EVENT_BERSERK:
225 DoCast(me, SPELL_BERSERK, true);
226 events.ScheduleEvent(EVENT_BERSERK, 600000);
227 break;
228 }
229 }
230
231 if (events.IsInPhase(PHASE_NORMAL))
232 DoMeleeAttackIfReady();
233 }
234 private:
235 GuidSet guardCorpses;
236 };
237
238};

240class at_anubrekhan_entrance : public AreaTriggerScript
241{
242 public:
243 at_anubrekhan_entrance() : AreaTriggerScript("at_anubrekhan_entrance") { }
244
245 bool OnTrigger(Player* player, AreaTriggerEntry const* /*areaTrigger*/, bool /*entered*/) override
246 {
247 InstanceScript* instance = player->GetInstanceScript();
248 if (!instance || instance->GetData(DATA_HAD_ANUBREKHAN_GREET) || instance->GetBossState(BOSS_ANUBREKHAN) != NOT_STARTED)
249 return true;
250
251 if (Creature* anub = ObjectAccessor::GetCreature(*player, instance->GetGuidData(DATA_ANUBREKHAN)))
252 anub->AI()->Talk(SAY_GREET);
253 instance->SetData(DATA_HAD_ANUBREKHAN_GREET, 1u);
254
255 return true;
256 }
257};

57class boss_patchwerk : public CreatureScript
58{
59public:
60 boss_patchwerk() : CreatureScript("boss_patchwerk") { }

62 CreatureAI* GetAI(Creature* creature) const override
63 {
64 return GetInstanceAI<boss_patchwerkAI>(creature);
65 }

67 struct boss_patchwerkAI : public BossAI
68 {
69 boss_patchwerkAI(Creature* creature) : BossAI(creature, BOSS_PATCHWERK)
70 {
71 Enraged = false;
72 }
73
74 bool Enraged;
75
76 void Reset() override
77 {
78 _Reset();
79
80 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_MAKE_QUICK_WERK_OF_HIM_STARTING_EVENT);
81 }
82
83 void KilledUnit(Unit* /*Victim*/) override
84 {
85 if (!(rand32() % 5))
86 Talk(SAY_SLAY);
87 }
88
89 void JustDied(Unit* /*killer*/) override
90 {
91 _JustDied();
92 Talk(SAY_DEATH);
93 }
94
95 void EnterCombat(Unit* /*who*/) override
96 {
97 _EnterCombat();
98 Enraged = false;
99 Talk(SAY_AGGRO);
100 events.ScheduleEvent(EVENT_HATEFUL, 1 * IN_MILLISECONDS);
101 events.ScheduleEvent(EVENT_BERSERK, 6 * MINUTE * IN_MILLISECONDS);
102
103 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_MAKE_QUICK_WERK_OF_HIM_STARTING_EVENT);
104 }
105
106 void UpdateAI(uint32 diff) override
107 {
108 if (!UpdateVictim())
109 return;
110
111 events.Update(diff);
112
113 while (uint32 eventId = events.ExecuteEvent())
114 {
115 switch (eventId)
116 {
117 case EVENT_HATEFUL:
118 {
119 // Hateful Strike targets the highest non-MT threat in melee range on 10man
120 // and the higher HP target out of the two highest non-MT threats in melee range on 25man
121 float MostThreat = 0.0f;
122 Unit* secondThreatTarget = NULL;
123 Unit* thirdThreatTarget = NULL;
124
125 std::list<HostileReference*>::const_iterator i = me->getThreatManager().getThreatList().begin();
126 for (; i != me->getThreatManager().getThreatList().end(); ++i)
127 { // find second highest
128 Unit* target = (*i)->getTarget();
129 if (target->IsAlive() && target != me->GetVictim() && (*i)->getThreat() >= MostThreat && me->IsWithinMeleeRange(target))
130 {
131 MostThreat = (*i)->getThreat();
132 secondThreatTarget = target;
133 }
134 }
135
136 if (secondThreatTarget && Is25ManRaid())
137 { // find third highest
138 MostThreat = 0.0f;
139 i = me->getThreatManager().getThreatList().begin();
140 for (; i != me->getThreatManager().getThreatList().end(); ++i)
141 {
142 Unit* target = (*i)->getTarget();
143 if (target->IsAlive() && target != me->GetVictim() && target != secondThreatTarget && (*i)->getThreat() >= MostThreat && me->IsWithinMeleeRange(target))
144 {
145 MostThreat = (*i)->getThreat();
146 thirdThreatTarget = target;
147 }
148 }
149 }
150
151 Unit* pHatefulTarget = NULL;
152 if (!thirdThreatTarget)
153 pHatefulTarget = secondThreatTarget;
154 else if (secondThreatTarget)
155 pHatefulTarget = (secondThreatTarget->GetHealth() < thirdThreatTarget->GetHealth()) ? thirdThreatTarget : secondThreatTarget;
156
157 if (!pHatefulTarget)
158 pHatefulTarget = me->GetVictim();
159
160 DoCast(pHatefulTarget, SPELL_HATEFUL_STRIKE, true);
161
162 // add threat to highest threat targets
163 if (me->GetVictim() && me->IsWithinMeleeRange(me->GetVictim()))
164 me->getThreatManager().addThreat(me->GetVictim(), HATEFUL_THREAT_AMT);
165 if (secondThreatTarget)
166 me->getThreatManager().addThreat(secondThreatTarget, HATEFUL_THREAT_AMT);
167 if (thirdThreatTarget)
168 me->getThreatManager().addThreat(thirdThreatTarget, HATEFUL_THREAT_AMT); // this will only ever be used in 25m
169
170 events.ScheduleEvent(EVENT_HATEFUL, 1 * IN_MILLISECONDS);
171 break;
172 }
173 case EVENT_BERSERK:
174 DoCast(me, SPELL_BERSERK, true);
175 Talk(EMOTE_BERSERK);
176 events.ScheduleEvent(EVENT_SLIME, 2 * IN_MILLISECONDS);
177 break;
178 case EVENT_SLIME:
179 DoCastVictim(SPELL_SLIME_BOLT, true);
180 events.ScheduleEvent(EVENT_SLIME, 2 * IN_MILLISECONDS);
181 break;
182 }
183 }
184
185 if (!Enraged && HealthBelowPct(5))
186 {
187 DoCast(me, SPELL_FRENZY, true);
188 Talk(EMOTE_ENRAGE);
189 Enraged = true;
190 }
191
192 DoMeleeAttackIfReady();
193 }
194 };
195
196};

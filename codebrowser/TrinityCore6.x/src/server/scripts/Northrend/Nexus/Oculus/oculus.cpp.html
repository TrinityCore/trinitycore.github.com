165 else if (gossipListId == 3)
166 RemoveEssence(player, ITEM_RUBY_ESSENCE);
167
168 StoreEssence(player, ITEM_AMBER_ESSENCE);
169 break;
170 }
171 return;
172 case GOSSIP_MENU_BELGARISTRASZ:
173 if (gossipListId <= 2)
174 {
175 if (gossipListId == 1)
176 RemoveEssence(player, ITEM_AMBER_ESSENCE);

241class npc_ruby_emerald_amber_drake : public CreatureScript
242{
243 public:
244 npc_ruby_emerald_amber_drake() : CreatureScript("npc_ruby_emerald_amber_drake") { }
245
246 struct npc_ruby_emerald_amber_drakeAI : public VehicleAI
247 {
248 npc_ruby_emerald_amber_drakeAI(Creature* creature) : VehicleAI(creature)
249 {
250 Initialize();
251 _instance = creature->GetInstanceScript();
252 }
253
254 void Initialize()
255 {
256 _healthWarning = true;
257 }
258
259 void Reset() override
260 {
261 _events.Reset();
262 Initialize();
263 }
264
265 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
266 {
267 if (Unit* creator = ObjectAccessor::GetUnit(*me, me->GetCreatorGUID()))
268 if (spell->Id == SPELL_GPS)
269 {
270 if (_instance->GetBossState(DATA_EREGOS) == DONE)
271 Talk(WHISPER_GPS_END, creator);
272 else if (_instance->GetBossState(DATA_UROM) == DONE)
273 Talk(WHISPER_GPS_EREGOS, creator);
274 else if (_instance->GetBossState(DATA_VAROS) == DONE)
275 Talk(WHISPER_GPS_UROM, creator);
276 else if (_instance->GetData(DATA_CONSTRUCTS) == KILL_NO_CONSTRUCT)
277 Talk(WHISPER_GPS_VAROS, creator);
278 else if (_instance->GetData(DATA_CONSTRUCTS) == KILL_ONE_CONSTRUCT)
279 Talk(WHISPER_GPS_1_CONSTRUCT, creator);
280 else if (_instance->GetData(DATA_CONSTRUCTS) == KILL_MORE_CONSTRUCT)
281 Talk(WHISPER_GPS_10_CONSTRUCTS, creator);
282 }
283 }
284
285 void IsSummonedBy(Unit* summoner) override
286 {
287 if (_instance->GetBossState(DATA_EREGOS) == IN_PROGRESS)
288 if (Creature* eregos = me->FindNearestCreature(NPC_EREGOS, 450.0f, true))
289 eregos->DespawnOrUnsummon(); // On retail this kills abusive call of drake during engaged Eregos
290
291 me->SetFacingToObject(summoner);
292
293 switch (me->GetEntry())
294 {
295 case NPC_RUBY_DRAKE_VEHICLE:
296 me->CastSpell(summoner, SPELL_RIDE_RUBY_DRAKE_QUE);
297 break;
298 case NPC_EMERALD_DRAKE_VEHICLE:
299 me->CastSpell(summoner, SPELL_RIDE_EMERALD_DRAKE_QUE);
300 break;
301 case NPC_AMBER_DRAKE_VEHICLE:
302 me->CastSpell(summoner, SPELL_RIDE_AMBER_DRAKE_QUE);
303 break;
304 default:
305 return;
306 }
307
308 Position pos = summoner->GetPosition();
309 me->GetMotionMaster()->MovePoint(POINT_LAND, pos);
310 }
311
312 void MovementInform(uint32 type, uint32 id) override
313 {
314 if (type == POINT_MOTION_TYPE && id == POINT_LAND)
315 me->SetDisableGravity(false); // Needed this for proper animation after spawn, the summon in air fall to ground bug leave no other option for now, if this isn't used the drake will only walk on move.
316 }
317
318 void PassengerBoarded(Unit* passenger, int8 /*seatId*/, bool apply) override
319 {
320 if (passenger->GetTypeId() != TYPEID_PLAYER)
321 return;
322
323 if (apply)
324 {
325 if (_instance->GetBossState(DATA_VAROS) != DONE)
326 _events.ScheduleEvent(EVENT_WELCOME, 10 * IN_MILLISECONDS);
327
328 else if (_instance->GetBossState(DATA_UROM) == DONE)
329 _events.ScheduleEvent(EVENT_SPECIAL_ATTACK, 10 * IN_MILLISECONDS);
330 }
331 else
332 {
333 _events.Reset();
334 _events.ScheduleEvent(EVENT_TAKE_OFF, 2 * IN_MILLISECONDS);
335 }
336 }
337
338 void UpdateAI(uint32 diff) override
339 {
340 if (_healthWarning)
341 {
342 if (me->GetHealthPct() <= 40.0f)
343 _events.ScheduleEvent(EVENT_LOW_HEALTH, 0);
344 }
345
346 _events.Update(diff);
347
348 while (uint32 eventId = _events.ExecuteEvent())
349 {
350 switch (eventId)
351 {
352 case EVENT_WELCOME:
353 if (Unit* creator = ObjectAccessor::GetUnit(*me, me->GetCreatorGUID()))
354 Talk(WHISPER_DRAKES_WELCOME, creator);
355 _events.ScheduleEvent(EVENT_ABILITIES, 5 * IN_MILLISECONDS);
356 break;
357 case EVENT_ABILITIES:
358 if (Unit* creator = ObjectAccessor::GetUnit(*me, me->GetCreatorGUID()))
359 Talk(WHISPER_DRAKES_ABILITIES, creator);
360 break;
361 case EVENT_SPECIAL_ATTACK:
362 if (Unit* creator = ObjectAccessor::GetUnit(*me, me->GetCreatorGUID()))
363 Talk(WHISPER_DRAKES_SPECIAL, creator);
364 break;
365 case EVENT_LOW_HEALTH:
366 if (Unit* creator = ObjectAccessor::GetUnit(*me, me->GetCreatorGUID()))
367 Talk(WHISPER_DRAKES_LOWHEALTH, creator);
368 _healthWarning = false;
369 _events.ScheduleEvent(EVENT_RESET_LOW_HEALTH, 25000);
370 break;
371 case EVENT_RESET_LOW_HEALTH:
372 _healthWarning = true;
373 break;
374 case EVENT_TAKE_OFF:
375 {
376 me->DespawnOrUnsummon(2050);
377 me->SetOrientation(2.5f);
378 me->SetSpeed(MOVE_FLIGHT, 1.0f, true);
379 Talk(SAY_DRAKES_TAKEOFF);
380 Position pos = me->GetPosition();
381 Position offset = { 10.0f, 10.0f, 12.0f, 0.0f };
382 pos.RelocateOffset(offset);
383 me->SetDisableGravity(true);
384 me->GetMotionMaster()->MovePoint(POINT_TAKE_OFF, pos);
385 break;
386 }
387 default:
388 break;
389 }
390 }
391 };
392
393 private:
394 InstanceScript* _instance;
395 EventMap _events;
396 bool _healthWarning;
397 };
398
399 CreatureAI* GetAI(Creature* creature) const override
400 {
401 return GetOculusAI<npc_ruby_emerald_amber_drakeAI>(creature);
402 }

81class boss_eregos : public CreatureScript
82{
83 public:
84 boss_eregos() : CreatureScript("boss_eregos") { }
85
86 struct boss_eregosAI : public BossAI
87 {
88 boss_eregosAI(Creature* creature) : BossAI(creature, DATA_EREGOS)
89 {
90 Initialize();
91 }
92
93 void Initialize()
94 {
95 _phase = PHASE_NORMAL;
96
97 _rubyVoid = true;
98 _emeraldVoid = true;
99 _amberVoid = true;
100 }
101
102 void Reset() override
103 {
104 _Reset();
105 Initialize();
106
107 DoAction(ACTION_SET_NORMAL_EVENTS);
108 }
109
110 void KilledUnit(Unit* who) override
111 {
112 if (who->GetTypeId() == TYPEID_PLAYER)
113 Talk(SAY_KILL);
114 }
115
116 void EnterCombat(Unit* /*who*/) override
117 {
118 _EnterCombat();
119
120 Talk(SAY_AGGRO);
121 /* Checks for present drakes vehicles from each type and deactivate achievement that corresponds to each found
122 The checks are so big in case some party try weird things like pulling boss down or hiding out of check range, the only thing player need is to get the boss kill credit after the check /even if he or his drake die/
123 Drakes mechanic would despawn all after unmount and also drakes should be auto mounted after item use, item use after Eregos is engaged leads to his despawn - based on retail data. */
124 if (me->FindNearestCreature(NPC_RUBY_DRAKE_VEHICLE, 500.0f, true))
125 _rubyVoid = false;
126 if (me->FindNearestCreature(NPC_EMERALD_DRAKE_VEHICLE, 500.0f, true))
127 _emeraldVoid = false;
128 if (me->FindNearestCreature(NPC_AMBER_DRAKE_VEHICLE, 500.0f, true))
129 _amberVoid = false;
130 }
131
132 uint32 GetData(uint32 type) const override
133 {
134 switch (type)
135 {
136 case DATA_RUBY_VOID:
137 return _rubyVoid;
138 case DATA_EMERALD_VOID:
139 return _emeraldVoid;
140 case DATA_AMBER_VOID:
141 return _amberVoid;
142 default:
143 break;
144 }
145 return 0;
146 }
147
148 void DoAction(int32 action) override
149 {
150 if (action != ACTION_SET_NORMAL_EVENTS)
151 return;
152
153 events.SetPhase(PHASE_NORMAL);
154 events.ScheduleEvent(EVENT_ARCANE_BARRAGE, urand(3, 10) * IN_MILLISECONDS, 0, PHASE_NORMAL);
155 events.ScheduleEvent(EVENT_ARCANE_VOLLEY, urand(10, 25) * IN_MILLISECONDS, 0, PHASE_NORMAL);
156 events.ScheduleEvent(EVENT_ENRAGED_ASSAULT, urand(35, 50) * IN_MILLISECONDS, 0, PHASE_NORMAL);
157 events.ScheduleEvent(EVENT_SUMMON_LEY_WHELP, urand(15, 30) * IN_MILLISECONDS, 0, PHASE_NORMAL);
158 }
159
160 void JustSummoned(Creature* summon) override
161 {
162 BossAI::JustSummoned(summon);
163
164 if (summon->GetEntry() != NPC_PLANAR_ANOMALY)
165 return;
166
167 summon->CombatStop(true);
168 summon->SetReactState(REACT_PASSIVE);
169 summon->GetMotionMaster()->MoveRandom(100.0f);
170 }
171
172 void SummonedCreatureDespawn(Creature* summon) override
173 {
174 if (summon->GetEntry() != NPC_PLANAR_ANOMALY)
175 return;
176
177 /// @todo: See why the spell is not cast
178 summon->CastSpell(summon, SPELL_PLANAR_BLAST, true);
179 }
180
181 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
182 {
183 if (!IsHeroic())
184 return;
185
186 if ( (me->GetHealthPct() < 60.0f && me->GetHealthPct() > 20.0f && _phase < PHASE_FIRST_PLANAR)
187 || (me->GetHealthPct() < 20.0f && _phase < PHASE_SECOND_PLANAR) )
188 {
189 events.Reset();
190 _phase = (me->GetHealthPct() < 60.0f && me->GetHealthPct() > 20.0f) ? PHASE_FIRST_PLANAR : PHASE_SECOND_PLANAR;
191
192 Talk(SAY_SHIELD);
193 DoCast(SPELL_PLANAR_SHIFT);
194
195 // not sure about the amount, and if we should despawn previous spawns (dragon trashs)
196 summons.DespawnAll();
197 for (uint8 i = 0; i < 6; i++)
198 DoCast(SPELL_PLANAR_ANOMALIES);
199 }
200 }
201
202 void UpdateAI(uint32 diff) override
203 {
204 if (!UpdateVictim())
205 return;
206
207 events.Update(diff);
208
209 if (me->HasUnitState(UNIT_STATE_CASTING))
210 return;
211
212 while (uint32 eventId = events.ExecuteEvent())
213 {
214 switch (eventId)
215 {
216 case EVENT_ARCANE_BARRAGE:
217 DoCastVictim(SPELL_ARCANE_BARRAGE);
218 events.ScheduleEvent(EVENT_ARCANE_BARRAGE, urand(3, 10) * IN_MILLISECONDS, 0, PHASE_NORMAL);
219 break;
220 case EVENT_ARCANE_VOLLEY:
221 DoCastAOE(SPELL_ARCANE_VOLLEY);
222 events.ScheduleEvent(EVENT_ARCANE_VOLLEY, urand(10, 25) * IN_MILLISECONDS, 0, PHASE_NORMAL);
223 break;
224 case EVENT_ENRAGED_ASSAULT:
225 Talk(SAY_ENRAGE);
226 DoCast(SPELL_ENRAGED_ASSAULT);
227 events.ScheduleEvent(EVENT_ENRAGED_ASSAULT, urand(35, 50) * IN_MILLISECONDS, 0, PHASE_NORMAL);
228 break;
229 case EVENT_SUMMON_LEY_WHELP:
230 for (uint8 i = 0; i < 3; i++)
231 DoCast(SPELL_SUMMON_LEY_WHELP);
232 events.ScheduleEvent(EVENT_SUMMON_LEY_WHELP, urand(15, 30) * IN_MILLISECONDS, 0, PHASE_NORMAL);
233 break;
234 default:
235 break;
236 }
237 }
238
239 DoMeleeAttackIfReady();
240 }
241
242 void JustDied(Unit* /*killer*/) override
243 {
244 Talk(SAY_DEATH);
245
246 _JustDied();
247 }
248
249 private:
250 uint8 _phase;
251 bool _rubyVoid;
252 bool _emeraldVoid;
253 bool _amberVoid;
254 };
255
256 CreatureAI* GetAI(Creature* creature) const override
257 {
258 return new boss_eregosAI(creature);
259 }

267 class spell_eregos_planar_shift_AuraScript : public AuraScript
268 {
269 PrepareAuraScript(spell_eregos_planar_shift_AuraScript);
270
271 void OnRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
272 {
273 if (Creature* creature = GetTarget()->ToCreature())
274 creature->AI()->DoAction(ACTION_SET_NORMAL_EVENTS);
275 }
276
277 void Register() override
278 {
279 AfterEffectRemove += AuraEffectRemoveFn(spell_eregos_planar_shift_AuraScript::OnRemove, EFFECT_0, SPELL_AURA_SCHOOL_IMMUNITY, AURA_EFFECT_HANDLE_REAL);
280 }
281 };
282

284 {
285 return new spell_eregos_planar_shift_AuraScript();
286 }

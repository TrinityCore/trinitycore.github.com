55class boss_varos : public CreatureScript
56{
57 public:
58 boss_varos() : CreatureScript("boss_varos") { }
59
60 struct boss_varosAI : public BossAI
61 {
62 boss_varosAI(Creature* creature) : BossAI(creature, DATA_VAROS)
63 {
64 Initialize();
65 }
66
67 void Initialize()
68 {
69 firstCoreEnergize = false;
70 coreEnergizeOrientation = 0.0f;
71 }
72
73 void InitializeAI() override
74 {
75 BossAI::InitializeAI();
76 if (instance->GetBossState(DATA_DRAKOS) != DONE)
77 DoCast(me, SPELL_CENTRIFUGE_SHIELD);
78 }
79
80 void Reset() override
81 {
82 _Reset();
83
84 events.ScheduleEvent(EVENT_AMPLIFY_MAGIC, urand(20, 25) * IN_MILLISECONDS);
85 events.ScheduleEvent(EVENT_ENERGIZE_CORES_VISUAL, 5000);
86 // not sure if this is handled by a timer or hp percentage
87 events.ScheduleEvent(EVENT_CALL_AZURE, urand(15, 30) * IN_MILLISECONDS);
88
89 Initialize();
90 }
91
92 void EnterCombat(Unit* /*who*/) override
93 {
94 _EnterCombat();
95
96 Talk(SAY_AGGRO);
97 }
98
99 float GetCoreEnergizeOrientation()
100 {
101 return coreEnergizeOrientation;
102 }
103
104 void UpdateAI(uint32 diff) override
105 {
106 if (!UpdateVictim())
107 return;
108
109 events.Update(diff);
110
111 if (me->HasUnitState(UNIT_STATE_CASTING))
112 return;
113
114 while (uint32 eventId = events.ExecuteEvent())
115 {
116 switch (eventId)
117 {
118 case EVENT_ENERGIZE_CORES:
119 DoCast(me, SPELL_ENERGIZE_CORES);
120 events.CancelEvent(EVENT_ENERGIZE_CORES);
121 break;
122 case EVENT_ENERGIZE_CORES_VISUAL:
123 if (!firstCoreEnergize)
124 {
125 coreEnergizeOrientation = me->GetOrientation();
126 firstCoreEnergize = true;
127 }
128 else
129 coreEnergizeOrientation = Position::NormalizeOrientation(coreEnergizeOrientation - 2.0f);
130
131 DoCast(me, SPELL_ENERGIZE_CORES_VISUAL);
132 events.ScheduleEvent(EVENT_ENERGIZE_CORES_VISUAL, 5000);
133 events.ScheduleEvent(EVENT_ENERGIZE_CORES, 4000);
134 break;
135 case EVENT_CALL_AZURE:
136 // not sure how blizz handles this, i cant see any pattern between the differnt spells
137 DoCast(me, SPELL_CALL_AZURE_RING_CAPTAIN);
138 Talk(SAY_AZURE);
139 Talk(SAY_AZURE_EMOTE);
140 events.ScheduleEvent(EVENT_CALL_AZURE, urand(20, 25) * IN_MILLISECONDS);
141 break;
142 case EVENT_AMPLIFY_MAGIC:
143 DoCastVictim(SPELL_CALL_AMPLIFY_MAGIC);
144 events.ScheduleEvent(EVENT_AMPLIFY_MAGIC, urand(17, 20) * IN_MILLISECONDS);
145 break;
146 default:
147 break;
148 }
149 }
150
151 DoMeleeAttackIfReady();
152 }
153
154 void JustDied(Unit* /*killer*/) override
155 {
156 _JustDied();
157 Talk(SAY_DEATH);
158 DoCast(me, SPELL_DEATH_SPELL, true); // we cast the spell as triggered or the summon effect does not occur
159 }
160
161 private:
162 bool firstCoreEnergize;
163 float coreEnergizeOrientation;
164 };
165
166 CreatureAI* GetAI(Creature* creature) const override
167 {
168 return GetOculusAI<boss_varosAI>(creature);
169 }
170};

172class npc_azure_ring_captain : public CreatureScript
173{
174 public:
175 npc_azure_ring_captain() : CreatureScript("npc_azure_ring_captain") { }
176
177 struct npc_azure_ring_captainAI : public ScriptedAI
178 {
179 npc_azure_ring_captainAI(Creature* creature) : ScriptedAI(creature)
180 {
181 Initialize();
182 instance = creature->GetInstanceScript();
183 }
184
185 void Initialize()
186 {
187 targetGUID.Clear();
188 }
189
190 void Reset() override
191 {
192 Initialize();
193
194 me->SetWalk(true);
195 //! HACK: Creature's can't have MOVEMENTFLAG_FLYING
196 me->AddUnitMovementFlag(MOVEMENTFLAG_FLYING);
197 me->SetReactState(REACT_AGGRESSIVE);
198 }
199
200 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
201 {
202 if (spell->Id == SPELL_ICE_BEAM)
203 {
204 target->CastSpell(target, SPELL_SUMMON_ARCANE_BEAM, true);
205 me->DespawnOrUnsummon();
206 }
207 }
208
209 void UpdateAI(uint32 /*diff*/) override
210 {
211 if (!UpdateVictim())
212 return;
213
214 DoMeleeAttackIfReady();
215 }
216
217 void MovementInform(uint32 type, uint32 id) override
218 {
219 if (type != POINT_MOTION_TYPE ||
220 id != ACTION_CALL_DRAGON_EVENT)
221 return;
222
223 me->GetMotionMaster()->MoveIdle();
224
225 if (Unit* target = ObjectAccessor::GetUnit(*me, targetGUID))
226 DoCast(target, SPELL_ICE_BEAM);
227 }
228
229 void DoAction(int32 action) override
230 {
231 switch (action)
232 {
233 case ACTION_CALL_DRAGON_EVENT:
234 if (Creature* varos = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_VAROS)))
235 {
236 if (Unit* victim = varos->AI()->SelectTarget(SELECT_TARGET_RANDOM, 0))
237 {
238 me->SetReactState(REACT_PASSIVE);
239 me->SetWalk(false);
240 me->GetMotionMaster()->MovePoint(ACTION_CALL_DRAGON_EVENT, victim->GetPositionX(), victim->GetPositionY(), victim->GetPositionZ() + 20.0f);
241 targetGUID = victim->GetGUID();
242 }
243 }
244 break;
245 }
246 }
247
248 private:
249 ObjectGuid targetGUID;
250 InstanceScript* instance;
251 };
252
253 CreatureAI* GetAI(Creature* creature) const override
254 {
255 return GetInstanceAI<npc_azure_ring_captainAI>(creature);
256 }

259class spell_varos_centrifuge_shield : public SpellScriptLoader
260{
261 public:
262 spell_varos_centrifuge_shield() : SpellScriptLoader("spell_varos_centrifuge_shield") { }
263
264 class spell_varos_centrifuge_shield_AuraScript : public AuraScript
265 {
266 PrepareAuraScript(spell_varos_centrifuge_shield_AuraScript);
267
268 bool Load() override
269 {
270 Unit* caster = GetCaster();
271 return (caster && caster->ToCreature());
272 }
273
274 void OnApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
275 {
276 if (Unit* caster = GetCaster())
277 {
278 // flags taken from sniffs
279 if (caster->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNK_15|UNIT_FLAG_IMMUNE_TO_NPC|UNIT_FLAG_IMMUNE_TO_PC|UNIT_FLAG_UNK_6))
280 {
281 caster->ToCreature()->SetReactState(REACT_PASSIVE);
282 caster->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNK_15|UNIT_FLAG_IMMUNE_TO_NPC|UNIT_FLAG_IMMUNE_TO_PC|UNIT_FLAG_UNK_6);
283 }
284 }
285 }
286
287 void OnRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
288 {
289 if (Unit* caster = GetCaster())
290 {
291 caster->ToCreature()->SetReactState(REACT_AGGRESSIVE);
292 caster->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNK_15|UNIT_FLAG_IMMUNE_TO_NPC|UNIT_FLAG_IMMUNE_TO_PC|UNIT_FLAG_UNK_6);
293 }
294 }
295
296 void Register() override
297 {
298 OnEffectRemove += AuraEffectRemoveFn(spell_varos_centrifuge_shield_AuraScript::OnRemove, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
299 OnEffectApply += AuraEffectApplyFn(spell_varos_centrifuge_shield_AuraScript::OnApply, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
300 }
301 };
302
303 AuraScript* GetAuraScript() const override
304 {
305 return new spell_varos_centrifuge_shield_AuraScript();
306 }

309class spell_varos_energize_core_area_enemy : public SpellScriptLoader
310{
311 public:
312 spell_varos_energize_core_area_enemy() : SpellScriptLoader("spell_varos_energize_core_area_enemy") { }
313
314 class spell_varos_energize_core_area_enemySpellScript : public SpellScript
315 {
316 PrepareSpellScript(spell_varos_energize_core_area_enemySpellScript);
317
318 void FilterTargets(std::list<WorldObject*>& targets)
319 {
320 Creature* varos = GetCaster()->ToCreature();
321 if (!varos)
322 return;
323
324 if (varos->GetEntry() != NPC_VAROS)
325 return;
326
327 float orientation = ENSURE_AI(boss_varos::boss_varosAI, varos->AI())->GetCoreEnergizeOrientation();
328
329 for (std::list<WorldObject*>::iterator itr = targets.begin(); itr != targets.end();)
330 {
331 float angle = varos->GetAngle((*itr)->GetPositionX(), (*itr)->GetPositionY());
332 float diff = std::fabs(orientation - angle);
333
334 if (diff > 1.0f)
335 itr = targets.erase(itr);
336 else
337 ++itr;
338 }
339 }
340
341 void Register() override
342 {
343 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_varos_energize_core_area_enemySpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
344 }
345 };
346
347 SpellScript* GetSpellScript() const override
348 {
349 return new spell_varos_energize_core_area_enemySpellScript();
350 }

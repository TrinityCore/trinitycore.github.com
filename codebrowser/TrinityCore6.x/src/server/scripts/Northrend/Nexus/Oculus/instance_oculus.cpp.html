24DoorData const doorData[] =
25{
26 { GO_DRAGON_CAGE_DOOR, DATA_DRAKOS, DOOR_TYPE_PASSAGE, BOUNDARY_NONE },

30Position const VerdisaMove = { 949.188f, 1032.91f, 359.967f, 1.093027f };
31Position const BelgaristraszMove = { 941.453f, 1044.1f, 359.967f, 0.1984709f };
32Position const EternosMove = { 943.202f, 1059.35f, 359.967f, 5.757278f };

34class instance_oculus : public InstanceMapScript
35{
36 public:
37 instance_oculus() : InstanceMapScript(OculusScriptName, 578) { }

39 struct instance_oculus_InstanceMapScript : public InstanceScript
40 {
41 instance_oculus_InstanceMapScript(Map* map) : InstanceScript(map)
42 {
43 SetHeaders(DataHeader);
44 SetBossNumber(EncounterCount);
45 LoadDoorData(doorData);
46
47 CentrifugueConstructCounter = 0;

49
50 void OnCreatureCreate(Creature* creature) override
51 {
52 switch (creature->GetEntry())
53 {
54 case NPC_DRAKOS:
55 DrakosGUID = creature->GetGUID();
56 break;
57 case NPC_VAROS:
58 VarosGUID = creature->GetGUID();
59 if (GetBossState(DATA_DRAKOS) == DONE)
60 creature->SetPhaseMask(1, true);
61 break;
62 case NPC_UROM:
63 UromGUID = creature->GetGUID();
64 if (GetBossState(DATA_VAROS) == DONE)
65 creature->SetPhaseMask(1, true);
66 break;
67 case NPC_EREGOS:
68 EregosGUID = creature->GetGUID();
69 if (GetBossState(DATA_UROM) == DONE)
70 creature->SetPhaseMask(1, true);
71 break;
72 case NPC_CENTRIFUGE_CONSTRUCT:
73 if (creature->IsAlive())
74 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT, ++CentrifugueConstructCounter);
75 break;
76 case NPC_BELGARISTRASZ:
77 BelgaristraszGUID = creature->GetGUID();
78 if (GetBossState(DATA_DRAKOS) == DONE)
79 {
80 creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
81 creature->Relocate(BelgaristraszMove);
82 }
83 break;
84 case NPC_ETERNOS:
85 EternosGUID = creature->GetGUID();
86 if (GetBossState(DATA_DRAKOS) == DONE)
87 {
88 creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
89 creature->Relocate(EternosMove);
90 }
91 break;
92 case NPC_VERDISA:
93 VerdisaGUID = creature->GetGUID();
94 if (GetBossState(DATA_DRAKOS) == DONE)
95 {
96 creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
97 creature->Relocate(VerdisaMove);
98 }
99 break;
100 case NPC_GREATER_WHELP:
101 if (GetBossState(DATA_UROM) == DONE)
102 {
103 creature->SetPhaseMask(1, true);
104 GreaterWhelpList.push_back(creature->GetGUID());
105 }
106 break;
107 default:
108 break;
109 }
110 }

112 void OnGameObjectCreate(GameObject* go) override
113 {
114 switch (go->GetEntry())
115 {
116 case GO_DRAGON_CAGE_DOOR:
117 AddDoor(go, true);
118 break;
119 case GO_EREGOS_CACHE_N:
120 case GO_EREGOS_CACHE_H:
121 EregosCacheGUID = go->GetGUID();
122 break;
123 default:
124 break;
125 }
126 }
127
128 void OnGameObjectRemove(GameObject* go) override
129 {
130 switch (go->GetEntry())
131 {
132 case GO_DRAGON_CAGE_DOOR:
133 AddDoor(go, false);
134 break;
135 default:
136 break;
137 }
138 }
139
140 void OnUnitDeath(Unit* unit) override
141 {
142 Creature* creature = unit->ToCreature();
143 if (!creature)
144 return;
145
146 if (creature->GetEntry() == NPC_CENTRIFUGE_CONSTRUCT)
147 {
148 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT, --CentrifugueConstructCounter);
149
150 if (!CentrifugueConstructCounter)
151 if (Creature* varos = instance->GetCreature(VarosGUID))
152 varos->RemoveAllAuras();
153 }
154 }
155
156 void FillInitialWorldStates(WorldPackets::WorldState::InitWorldStates& packet) override
157 {
158 if (GetBossState(DATA_DRAKOS) == DONE && GetBossState(DATA_VAROS) != DONE)
159 {
160 packet.Worldstates.emplace_back(uint32(WORLD_STATE_CENTRIFUGE_CONSTRUCT_SHOW), 1);
161 packet.Worldstates.emplace_back(uint32(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT), int32(CentrifugueConstructCounter));
162 }
163 else
164 {
165 packet.Worldstates.emplace_back(uint32(WORLD_STATE_CENTRIFUGE_CONSTRUCT_SHOW), 0);
166 packet.Worldstates.emplace_back(uint32(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT), 0);
167 }
168 }
169
170 void ProcessEvent(WorldObject* /*unit*/, uint32 eventId) override
171 {
172 if (eventId != EVENT_CALL_DRAGON)
173 return;
174
175 if (Creature* varos = instance->GetCreature(VarosGUID))
176 if (Creature* drake = varos->SummonCreature(NPC_AZURE_RING_GUARDIAN, varos->GetPositionX(), varos->GetPositionY(), varos->GetPositionZ() + 40))
177 drake->AI()->DoAction(ACTION_CALL_DRAGON_EVENT);
178 }
179
180 bool SetBossState(uint32 type, EncounterState state) override
181 {
182 if (!InstanceScript::SetBossState(type, state))
183 return false;
184
185 switch (type)
186 {
187 case DATA_DRAKOS:
188 if (state == DONE)
189 {
190 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_SHOW, 1);
191 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT, CentrifugueConstructCounter);
192 FreeDragons();
193 if (Creature* varos = instance->GetCreature(VarosGUID))
194 varos->SetPhaseMask(1, true);
195 events.ScheduleEvent(EVENT_VAROS_INTRO, 15000);
196 }
197 break;
198 case DATA_VAROS:
199 if (state == DONE)
200 {
201 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_SHOW, 0);
202 if (Creature* urom = instance->GetCreature(UromGUID))
203 urom->SetPhaseMask(1, true);
204 }
205 break;
206 case DATA_UROM:
207 if (state == DONE)
208 {
209 if (Creature* eregos = instance->GetCreature(EregosGUID))
210 {
211 eregos->SetPhaseMask(1, true);
212 GreaterWhelps();
213 events.ScheduleEvent(EVENT_EREGOS_INTRO, 5000);
214 }
215 }
216 break;
217 case DATA_EREGOS:
218 if (state == DONE)
219 {
220 if (GameObject* cache = instance->GetGameObject(EregosCacheGUID))
221 {
222 cache->SetRespawnTime(cache->GetRespawnDelay());
223 cache->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
224 }
225 }
226 break;
227 }
228
229 return true;
230 }
231
232 uint32 GetData(uint32 type) const override
233 {
234 if (type == DATA_CONSTRUCTS)
235 {
236 if (CentrifugueConstructCounter == 0)
237 return KILL_NO_CONSTRUCT;
238 else if (CentrifugueConstructCounter == 1)
239 return KILL_ONE_CONSTRUCT;
240 else if (CentrifugueConstructCounter > 1)
241 return KILL_MORE_CONSTRUCT;
242 }
243
244 return KILL_NO_CONSTRUCT;
245 }
246
247 ObjectGuid GetGuidData(uint32 type) const override
248 {
249 switch (type)
250 {
251 case DATA_DRAKOS:
252 return DrakosGUID;
253 case DATA_VAROS:
254 return VarosGUID;
255 case DATA_UROM:
256 return UromGUID;
257 case DATA_EREGOS:
258 return EregosGUID;
259 default:
260 break;
261 }
262
263 return ObjectGuid::Empty;
264 }

266 void FreeDragons()
267 {
268 if (Creature* belgaristrasz = instance->GetCreature(BelgaristraszGUID))
269 {
270 belgaristrasz->SetWalk(true);
271 belgaristrasz->GetMotionMaster()->MovePoint(POINT_MOVE_OUT, BelgaristraszMove);
272 }
273
274 if (Creature* eternos = instance->GetCreature(EternosGUID))
275 {
276 eternos->SetWalk(true);
277 eternos->GetMotionMaster()->MovePoint(POINT_MOVE_OUT, EternosMove);
278 }
279
280 if (Creature* verdisa = instance->GetCreature(VerdisaGUID))
281 {
282 verdisa->SetWalk(true);
283 verdisa->GetMotionMaster()->MovePoint(POINT_MOVE_OUT, VerdisaMove);
284 }
285 }
286
287 void Update(uint32 diff) override
288 {
289 events.Update(diff);
290
291 while (uint32 eventId = events.ExecuteEvent())
292 {
293 switch (eventId)
294 {
295 case EVENT_VAROS_INTRO:
296 if (Creature* varos = instance->GetCreature(VarosGUID))
297 varos->AI()->Talk(SAY_VAROS_INTRO_TEXT);
298 break;
299 case EVENT_EREGOS_INTRO:
300 if (Creature* eregos = instance->GetCreature(EregosGUID))
301 eregos->AI()->Talk(SAY_EREGOS_INTRO_TEXT);
302 break;
303 default:
304 break;
305 }
306 }
307 }
308
309 void GreaterWhelps()
310 {
311 for (ObjectGuid guid : GreaterWhelpList)
312 if (Creature* gwhelp = instance->GetCreature(guid))
313 gwhelp->SetPhaseMask(1, true);
314 }

328 ObjectGuid EregosCacheGUID;
329
330 GuidList GreaterWhelpList;

332 EventMap events;
333 };

71class boss_anomalus : public CreatureScript
72{
73 public:
74 boss_anomalus() : CreatureScript("boss_anomalus") { }
75
76 struct boss_anomalusAI : public ScriptedAI
77 {
78 boss_anomalusAI(Creature* creature) : ScriptedAI(creature)
79 {
80 Initialize();
81 instance = me->GetInstanceScript();
82 }
83
84 void Initialize()
85 {
86 Phase = 0;
87 uiSparkTimer = 5000;
88 uiChaoticRiftGUID.Clear();
89 chaosTheory = true;
90 }
91
92 InstanceScript* instance;
93
94 uint8 Phase;
95 uint32 uiSparkTimer;
96 ObjectGuid uiChaoticRiftGUID;
97 bool chaosTheory;
98
99 void Reset() override
100 {
101 Initialize();
102
103 instance->SetBossState(DATA_ANOMALUS, NOT_STARTED);
104 }
105
106 void EnterCombat(Unit* /*who*/) override
107 {
108 Talk(SAY_AGGRO);
109
110 instance->SetBossState(DATA_ANOMALUS, IN_PROGRESS);
111 }
112
113 void JustDied(Unit* /*killer*/) override
114 {
115 Talk(SAY_DEATH);
116
117 instance->SetBossState(DATA_ANOMALUS, DONE);
118 }
119
120 uint32 GetData(uint32 type) const override
121 {
122 if (type == DATA_CHAOS_THEORY)
123 return chaosTheory ? 1 : 0;
124
125 return 0;
126 }
127
128 void SummonedCreatureDies(Creature* summoned, Unit* /*who*/) override
129 {
130 if (summoned->GetEntry() == NPC_CHAOTIC_RIFT)
131 chaosTheory = false;
132 }
133
134 void UpdateAI(uint32 diff) override
135 {
136 if (!UpdateVictim())
137 return;
138
139 if (me->GetDistance(me->GetHomePosition()) > 60.0f)
140 {
141 // Not blizzlike, hack to avoid an exploit
142 EnterEvadeMode();
143 return;
144 }
145
146 if (me->HasAura(SPELL_RIFT_SHIELD))
147 {
148 if (!uiChaoticRiftGUID.IsEmpty())
149 {
150 Creature* Rift = ObjectAccessor::GetCreature(*me, uiChaoticRiftGUID);
151 if (Rift && Rift->isDead())
152 {
153 me->RemoveAurasDueToSpell(SPELL_RIFT_SHIELD);
154 uiChaoticRiftGUID.Clear();
155 }
156 return;
157 }
158 }
159 else
160 uiChaoticRiftGUID.Clear();
161
162 if ((Phase == 0) && HealthBelowPct(50))
163 {
164 Phase = 1;
165 Talk(SAY_SHIELD);
166 DoCast(me, SPELL_RIFT_SHIELD);
167 if (Creature* Rift = me->SummonCreature(NPC_CHAOTIC_RIFT, RiftLocation[urand(0, 5)], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 1000))
168 {
169 //DoCast(Rift, SPELL_CHARGE_RIFT);
170 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
171 Rift->AI()->AttackStart(target);
172 uiChaoticRiftGUID = Rift->GetGUID();
173 Talk(SAY_RIFT);
174 }
175 }
176
177 if (uiSparkTimer <= diff)
178 {
179 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
180 DoCast(target, SPELL_SPARK);
181 uiSparkTimer = 5000;
182 }
183 else
184 uiSparkTimer -= diff;
185
186 DoMeleeAttackIfReady();
187 }
188 };
189
190 CreatureAI* GetAI(Creature* creature) const override
191 {
192 return GetInstanceAI<boss_anomalusAI>(creature);
193 }
194};

196class npc_chaotic_rift : public CreatureScript
197{
198 public:
199 npc_chaotic_rift() : CreatureScript("npc_chaotic_rift") { }
200
201 struct npc_chaotic_riftAI : public ScriptedAI
202 {
203 npc_chaotic_riftAI(Creature* creature) : ScriptedAI(creature)
204 {
205 Initialize();
206 instance = me->GetInstanceScript();
207 SetCombatMovement(false);
208 }
209
210 void Initialize()
211 {
212 uiChaoticEnergyBurstTimer = 1000;
213 uiSummonCrazedManaWraithTimer = 5000;
214 }
215
216 InstanceScript* instance;
217
218 uint32 uiChaoticEnergyBurstTimer;
219 uint32 uiSummonCrazedManaWraithTimer;
220
221 void Reset() override
222 {
223 Initialize();
224 me->SetDisplayId(me->GetCreatureTemplate()->Modelid2);
225 DoCast(me, SPELL_ARCANEFORM, false);
226 }
227
228 void UpdateAI(uint32 diff) override
229 {
230 if (!UpdateVictim())
231 return;
232
233 if (uiChaoticEnergyBurstTimer <= diff)
234 {
235 Creature* Anomalus = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ANOMALUS));
236 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
237 {
238 if (Anomalus && Anomalus->HasAura(SPELL_RIFT_SHIELD))
239 DoCast(target, SPELL_CHARGED_CHAOTIC_ENERGY_BURST);
240 else
241 DoCast(target, SPELL_CHAOTIC_ENERGY_BURST);
242 }
243 uiChaoticEnergyBurstTimer = 1000;
244 }
245 else
246 uiChaoticEnergyBurstTimer -= diff;
247
248 if (uiSummonCrazedManaWraithTimer <= diff)
249 {
250 if (Creature* Wraith = me->SummonCreature(NPC_CRAZED_MANA_WRAITH, me->GetPositionX() + 1, me->GetPositionY() + 1, me->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 1000))
251 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
252 Wraith->AI()->AttackStart(target);
253 Creature* Anomalus = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ANOMALUS));
254 if (Anomalus && Anomalus->HasAura(SPELL_RIFT_SHIELD))
255 uiSummonCrazedManaWraithTimer = 5000;
256 else
257 uiSummonCrazedManaWraithTimer = 10000;
258 }
259 else
260 uiSummonCrazedManaWraithTimer -= diff;
261 }
262 };
263
264 CreatureAI* GetAI(Creature* creature) const override
265 {
266 return GetInstanceAI<npc_chaotic_riftAI>(creature);
267 }
268};

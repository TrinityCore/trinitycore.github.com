64class boss_ormorok : public CreatureScript
65{
66public:
67 boss_ormorok() : CreatureScript("boss_ormorok") { }
68
69 struct boss_ormorokAI : public BossAI
70 {
71 boss_ormorokAI(Creature* creature) : BossAI(creature, DATA_ORMOROK)
72 {
73 Initialize();
74 }
75
76 void Initialize()
77 {
78 frenzy = false;
79 }
80
81 void Reset() override
82 {
83 BossAI::Reset();
84 Initialize();
85 }
86
87 void EnterCombat(Unit* /*who*/) override
88 {
89 _EnterCombat();
90
91 events.ScheduleEvent(EVENT_CRYSTAL_SPIKES, 12000);
92 events.ScheduleEvent(EVENT_TRAMPLE, 10000);
93 events.ScheduleEvent(EVENT_SPELL_REFLECTION, 30000);
94 if (IsHeroic())
95 events.ScheduleEvent(EVENT_CRYSTALLINE_TANGLER, 17000);
96
97 Talk(SAY_AGGRO);
98 }
99
100 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
101 {
102 if (!frenzy && HealthBelowPct(25))
103 {
104 Talk(SAY_FRENZY);
105 DoCast(me, SPELL_FRENZY);
106 frenzy = true;
107 }
108 }
109
110 void JustDied(Unit* /*killer*/) override
111 {
112 _JustDied();
113 Talk(SAY_DEATH);
114 }
115
116 void KilledUnit(Unit* who) override
117 {
118 if (who->GetTypeId() == TYPEID_PLAYER)
119 Talk(SAY_KILL);
120 }
121
122 void UpdateAI(uint32 diff) override
123 {
124 if (!UpdateVictim())
125 return;
126
127 events.Update(diff);
128
129 if (me->HasUnitState(UNIT_STATE_CASTING))
130 return;
131
132 while (uint32 eventId = events.ExecuteEvent())
133 {
134 switch (eventId)
135 {
136 case EVENT_TRAMPLE:
137 DoCast(me, SPELL_TRAMPLE);
138 events.ScheduleEvent(EVENT_TRAMPLE, 10000);
139 break;
140 case EVENT_SPELL_REFLECTION:
141 Talk(SAY_REFLECT);
142 DoCast(me, SPELL_SPELL_REFLECTION);
143 events.ScheduleEvent(EVENT_SPELL_REFLECTION, 30000);
144 break;
145 case EVENT_CRYSTAL_SPIKES:
146 Talk(SAY_CRYSTAL_SPIKES);
147 DoCast(SPELL_CRYSTAL_SPIKES);
148 events.ScheduleEvent(EVENT_CRYSTAL_SPIKES, 12000);
149 break;
150 case EVENT_CRYSTALLINE_TANGLER:
151 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, OrmorokTanglerPredicate(me)))
152 DoCast(target, SPELL_SUMMON_CRYSTALLINE_TANGLER);
153 events.ScheduleEvent(EVENT_CRYSTALLINE_TANGLER, 17000);
154 break;
155 default:
156 break;
157 }
158 }
159
160 DoMeleeAttackIfReady();
161 }
162
163 private:
164 bool frenzy;
165
166 };
167
168 CreatureAI* GetAI(Creature* creature) const override
169 {
170 return GetInstanceAI<boss_ormorokAI>(creature);
171 }

194class npc_crystal_spike_trigger : public CreatureScript
195{
196public:
197 npc_crystal_spike_trigger() : CreatureScript("npc_crystal_spike_trigger") { }
198
199 struct npc_crystal_spike_triggerAI : public ScriptedAI
200 {
201 npc_crystal_spike_triggerAI(Creature* creature) : ScriptedAI(creature)
202 {
203 _count = 0;
204 _despawntimer = 0;
205 }
206
207 void IsSummonedBy(Unit* owner) override
208 {
209 switch (me->GetEntry())
210 {
211 case NPC_CRYSTAL_SPIKE_INITIAL:
212 _count = 0;
213 me->SetFacingToObject(owner);
214 break;
215 case NPC_CRYSTAL_SPIKE_TRIGGER:
216 if (Creature* trigger = owner->ToCreature())
217 _count = trigger->AI()->GetData(DATA_COUNT) + 1;
218 break;
219 default:
220 _count = MAX_COUNT;
221 break;
222 }
223
224 if (me->GetEntry() == NPC_CRYSTAL_SPIKE_TRIGGER)
225 if (GameObject* trap = me->FindNearestGameObject(GO_CRYSTAL_SPIKE_TRAP, 1.0f))
226 trap->Use(me);
227
228 _despawntimer = 2000;
229 }
230
231 uint32 GetData(uint32 type) const override
232 {
233 return type == DATA_COUNT ? _count : 0;
234 }
235
236 void UpdateAI(uint32 diff) override
237 {
238 if (_despawntimer <= diff)
239 {
240 if (me->GetEntry() == NPC_CRYSTAL_SPIKE_TRIGGER)
241 if (GameObject* trap = me->FindNearestGameObject(GO_CRYSTAL_SPIKE_TRAP, 1.0f))
242 trap->Delete();
243
244 me->DespawnOrUnsummon();
245 }
246 else
247 _despawntimer -= diff;
248 }
249
250 private:
251 uint32 _count;
252 uint32 _despawntimer;
253
254 };
255
256 CreatureAI* GetAI(Creature* creature) const override
257 {
258 return new npc_crystal_spike_triggerAI(creature);
259 }

262class spell_crystal_spike : public SpellScriptLoader
263{
264 public:
265 spell_crystal_spike() : SpellScriptLoader("spell_crystal_spike") { }
266
267 class spell_crystal_spike_AuraScript : public AuraScript
268 {
269 PrepareAuraScript(spell_crystal_spike_AuraScript);
270
271 void HandlePeriodic(AuraEffect const* /*aurEff*/)
272 {
273 Unit* target = GetTarget();
274 if (target->GetEntry() == NPC_CRYSTAL_SPIKE_INITIAL || target->GetEntry() == NPC_CRYSTAL_SPIKE_TRIGGER)
275 if (Creature* trigger = target->ToCreature())
276 {
277 uint32 spell = target->GetEntry() == NPC_CRYSTAL_SPIKE_INITIAL ? crystalSpikeSummon[0] : crystalSpikeSummon[urand(0, 2)];
278 if (trigger->AI()->GetData(DATA_COUNT) < MAX_COUNT)
279 trigger->CastSpell(trigger, spell, true);
280 }
281 }
282
283 void Register() override
284 {
285 OnEffectPeriodic += AuraEffectPeriodicFn(spell_crystal_spike_AuraScript::HandlePeriodic, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
286 }
287 };
288
289 AuraScript* GetAuraScript() const override
290 {
291 return new spell_crystal_spike_AuraScript();
292 }

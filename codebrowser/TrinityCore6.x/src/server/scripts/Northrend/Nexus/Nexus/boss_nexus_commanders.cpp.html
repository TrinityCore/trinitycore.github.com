18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "nexus.h"

22enum Spells
23{
24 SPELL_BATTLE_SHOUT = 31403,
25 SPELL_CHARGE = 60067,
26 SPELL_FRIGHTENING_SHOUT = 19134,
27 SPELL_WHIRLWIND = 38618,
28 SPELL_FROZEN_PRISON = 47543
29};
30
31enum Yells
32{
33 SAY_AGGRO = 0,
34 SAY_KILL = 1,
35 SAY_DEATH = 2
36};
37
38enum Events
39{
40 EVENT_CHARGE_COMMANDER = 1,
41 EVENT_WHIRLWIND,
42 EVENT_FRIGHTENING_SHOUT
43};
44
45class boss_nexus_commanders : public CreatureScript
46{
47 public:
48 boss_nexus_commanders() : CreatureScript("boss_nexus_commanders") { }
49
50 struct boss_nexus_commandersAI : public BossAI
51 {
52 boss_nexus_commandersAI(Creature* creature) : BossAI(creature, DATA_COMMANDER) { }
53
54 void EnterCombat(Unit* /*who*/) override
55 {
56 _EnterCombat();
57 Talk(SAY_AGGRO);
58 me->RemoveAurasDueToSpell(SPELL_FROZEN_PRISON);
59 DoCast(me, SPELL_BATTLE_SHOUT);
60
61 events.ScheduleEvent(EVENT_CHARGE_COMMANDER, urand(3000, 4000));
62 events.ScheduleEvent(EVENT_WHIRLWIND, urand(6000, 8000));
63 events.ScheduleEvent(EVENT_FRIGHTENING_SHOUT, urand(13000, 15000));
64 }
65
66 void ExecuteEvent(uint32 eventId) override
67 {
68 switch (eventId)
69 {
70 case EVENT_CHARGE_COMMANDER:
71 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
72 DoCast(target, SPELL_CHARGE);
73 events.ScheduleEvent(EVENT_CHARGE_COMMANDER, urand(11000, 15000));
74 break;
75 case EVENT_WHIRLWIND:
76 DoCast(me, SPELL_WHIRLWIND);
77 events.ScheduleEvent(EVENT_WHIRLWIND, urand(19500, 25000));
78 break;
79 case EVENT_FRIGHTENING_SHOUT:
80 DoCastAOE(SPELL_FRIGHTENING_SHOUT);
81 events.ScheduleEvent(EVENT_FRIGHTENING_SHOUT, urand(45000, 55000));
82 break;
83 default:
84 break;
85 }
86 }
87
88 void JustDied(Unit* /*killer*/) override
89 {
90 _JustDied();
91 Talk(SAY_DEATH);
92 }
93
94 void KilledUnit(Unit* who) override
95 {
96 if (who->GetTypeId() == TYPEID_PLAYER)
97 Talk(SAY_KILL);
98 }
99 };
100
101 CreatureAI* GetAI(Creature* creature) const override
102 {
103 return GetInstanceAI<boss_nexus_commandersAI>(creature);
104 }
105};
106
107void AddSC_boss_nexus_commanders()
108{
109 new boss_nexus_commanders();
110}

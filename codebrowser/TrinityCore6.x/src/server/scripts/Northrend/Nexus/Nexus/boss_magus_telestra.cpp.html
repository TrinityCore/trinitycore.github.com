66class boss_magus_telestra : public CreatureScript
67{
68public:
69 boss_magus_telestra() : CreatureScript("boss_magus_telestra") { }

76 struct boss_magus_telestraAI : public ScriptedAI
77 {
78 boss_magus_telestraAI(Creature* creature) : ScriptedAI(creature)
79 {
80 Initialize();
81 instance = creature->GetInstanceScript();
82 bFireMagusDead = false;
83 bFrostMagusDead = false;
84 bArcaneMagusDead = false;
85 uiIsWaitingToAppearTimer = 0;
86 }
87
88 void Initialize()
89 {
90 Phase = 0;
91 //These times are probably wrong
92 uiIceNovaTimer = 7 * IN_MILLISECONDS;
93 uiFireBombTimer = 0;
94 uiGravityWellTimer = 15 * IN_MILLISECONDS;
95 uiCooldown = 0;
96
97 uiFireMagusGUID.Clear();
98 uiFrostMagusGUID.Clear();
99 uiArcaneMagusGUID.Clear();
100
101 for (uint8 n = 0; n < 3; ++n)
102 time[n] = 0;
103
104 splitPersonality = 0;
105 bIsWaitingToAppear = false;
106 }
107
108 InstanceScript* instance;
109
110 ObjectGuid uiFireMagusGUID;
111 ObjectGuid uiFrostMagusGUID;
112 ObjectGuid uiArcaneMagusGUID;
113
114 bool bFireMagusDead;
115 bool bFrostMagusDead;
116 bool bArcaneMagusDead;
117 bool bIsWaitingToAppear;
118
119 uint32 uiIsWaitingToAppearTimer;
120 uint32 uiIceNovaTimer;
121 uint32 uiFireBombTimer;
122 uint32 uiGravityWellTimer;
123 uint32 uiCooldown;
124
125 uint8 Phase;
126 uint8 splitPersonality;
127 time_t time[3];
128
129 void Reset() override
130 {
131 Initialize();
132
133 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
134 me->SetVisible(true);
135
136 instance->SetBossState(DATA_MAGUS_TELESTRA, NOT_STARTED);
137
138 if (IsHeroic() && sGameEventMgr->IsActiveEvent(GAME_EVENT_WINTER_VEIL) && !me->HasAura(SPELL_WEAR_CHRISTMAS_HAT))
139 me->AddAura(SPELL_WEAR_CHRISTMAS_HAT, me);
140 }
141
142 void EnterCombat(Unit* /*who*/) override
143 {
144 Talk(SAY_AGGRO);
145
146 instance->SetBossState(DATA_MAGUS_TELESTRA, IN_PROGRESS);
147 }
148
149 void JustDied(Unit* /*killer*/) override
150 {
151 Talk(SAY_DEATH);
152
153 instance->SetBossState(DATA_MAGUS_TELESTRA, DONE);
154 }
155
156 void KilledUnit(Unit* who) override
157 {
158 if (who->GetTypeId() == TYPEID_PLAYER)
159 Talk(SAY_KILL);
160 }
161
162 void DoAction(int32 action) override
163 {
164 if (action == ACTION_MAGUS_DEAD)
165 {
166 uint8 i = 0;
167 while (time[i] != 0)
168 ++i;
169
170 time[i] = sWorld->GetGameTime();
171 if (i == 2 && (time[2] - time[1] < 5) && (time[1] - time[0] < 5))
172 ++splitPersonality;
173 }
174 }
175
176 uint32 GetData(uint32 type) const override
177 {
178 if (type == DATA_SPLIT_PERSONALITY)
179 return splitPersonality;
180
181 return 0;
182 }
183
184 ObjectGuid SplitPersonality(uint32 entry)
185 {
186 if (Creature* Summoned = me->SummonCreature(entry, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 1*IN_MILLISECONDS))
187 {
188 switch (entry)
189 {
190 case NPC_FIRE_MAGUS:
191 {
192 Summoned->CastSpell(Summoned, SPELL_FIRE_MAGUS_VISUAL, false);
193 break;
194 }
195 case NPC_FROST_MAGUS:
196 {
197 Summoned->CastSpell(Summoned, SPELL_FROST_MAGUS_VISUAL, false);
198 break;
199 }
200 case NPC_ARCANE_MAGUS:
201 {
202 Summoned->CastSpell(Summoned, SPELL_ARCANE_MAGUS_VISUAL, false);
203 break;
204 }
205 }
206 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
207 Summoned->AI()->AttackStart(target);
208 return Summoned->GetGUID();
209 }
210 return ObjectGuid::Empty;
211 }
212
213 void SummonedCreatureDespawn(Creature* summon) override
214 {
215 if (summon->IsAlive())
216 return;
217
218 if (summon->GetGUID() == uiFireMagusGUID)
219 {
220 DoAction(ACTION_MAGUS_DEAD);
221 bFireMagusDead = true;
222 }
223 else if (summon->GetGUID() == uiFrostMagusGUID)
224 {
225 DoAction(ACTION_MAGUS_DEAD);
226 bFrostMagusDead = true;
227 }
228 else if (summon->GetGUID() == uiArcaneMagusGUID)
229 {
230 DoAction(ACTION_MAGUS_DEAD);
231 bArcaneMagusDead = true;
232 }
233 }
234
235 void UpdateAI(uint32 diff) override
236 {
237 //Return since we have no target
238 if (!UpdateVictim())
239 return;
240
241 if (bIsWaitingToAppear)
242 {
243 me->StopMoving();
244 me->AttackStop();
245 if (uiIsWaitingToAppearTimer <= diff)
246 {
247 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
248 bIsWaitingToAppear = false;
249 } else uiIsWaitingToAppearTimer -= diff;
250 return;
251 }
252
253 if ((Phase == 1) ||(Phase == 3))
254 {
255 if (bFireMagusDead && bFrostMagusDead && bArcaneMagusDead)
256 {
257 for (uint8 n = 0; n < 3; ++n)
258 time[n] = 0;
259 me->GetMotionMaster()->Clear();
260 me->SetPosition(CenterOfRoom.GetPositionX(), CenterOfRoom.GetPositionY(), CenterOfRoom.GetPositionZ(), CenterOfRoom.GetOrientation());
261 DoCast(me, SPELL_TELESTRA_BACK);
262 me->SetVisible(true);
263 if (Phase == 1)
264 Phase = 2;
265 if (Phase == 3)
266 Phase = 4;
267 uiFireMagusGUID.Clear();
268 uiFrostMagusGUID.Clear();
269 uiArcaneMagusGUID.Clear();
270 bIsWaitingToAppear = true;
271 uiIsWaitingToAppearTimer = 4*IN_MILLISECONDS;
272 Talk(SAY_MERGE);
273 }
274 else
275 return;
276 }
277
278 if ((Phase == 0) && HealthBelowPct(50))
279 {
280 Phase = 1;
281 me->CastStop();
282 me->RemoveAllAuras();
283 me->SetVisible(false);
284 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
285 uiFireMagusGUID = SplitPersonality(NPC_FIRE_MAGUS);
286 uiFrostMagusGUID = SplitPersonality(NPC_FROST_MAGUS);
287 uiArcaneMagusGUID = SplitPersonality(NPC_ARCANE_MAGUS);
288 bFireMagusDead = false;
289 bFrostMagusDead = false;
290 bArcaneMagusDead = false;
291 Talk(SAY_SPLIT);
292 return;
293 }
294
295 if (IsHeroic() && (Phase == 2) && HealthBelowPct(10))
296 {
297 Phase = 3;
298 me->CastStop();
299 me->RemoveAllAuras();
300 me->SetVisible(false);
301 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
302 uiFireMagusGUID = SplitPersonality(NPC_FIRE_MAGUS);
303 uiFrostMagusGUID = SplitPersonality(NPC_FROST_MAGUS);
304 uiArcaneMagusGUID = SplitPersonality(NPC_ARCANE_MAGUS);
305 bFireMagusDead = false;
306 bFrostMagusDead = false;
307 bArcaneMagusDead = false;
308 Talk(SAY_SPLIT);
309 return;
310 }
311
312 if (uiCooldown)
313 {
314 if (uiCooldown <= diff)
315 uiCooldown = 0;
316 else
317 {
318 uiCooldown -= diff;
319 return;
320 }
321 }
322
323 if (uiIceNovaTimer <= diff)
324 {
325 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
326 {
327 DoCast(target, SPELL_ICE_NOVA, false);
328 uiCooldown = 1500;
329 }
330 uiIceNovaTimer = 15*IN_MILLISECONDS;
331 } else uiIceNovaTimer -= diff;
332
333 if (uiGravityWellTimer <= diff)
334 {
335 if (Unit* target = me->GetVictim())
336 {
337 DoCast(target, SPELL_GRAVITY_WELL);
338 uiCooldown = 6*IN_MILLISECONDS;
339 }
340 uiGravityWellTimer = 15*IN_MILLISECONDS;
341 } else uiGravityWellTimer -= diff;
342
343 if (uiFireBombTimer <= diff)
344 {
345 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
346 {
347 DoCast(target, SPELL_FIREBOMB, false);
348 uiCooldown = 2*IN_MILLISECONDS;
349 }
350 uiFireBombTimer = 2*IN_MILLISECONDS;
351 } else uiFireBombTimer -=diff;
352
353 DoMeleeAttackIfReady();
354 }
355 };

357};

359class achievement_split_personality : public AchievementCriteriaScript
360{
361 public:
362 achievement_split_personality() : AchievementCriteriaScript("achievement_split_personality")
363 {
364 }
365
366 bool OnCheck(Player* /*player*/, Unit* target) override
367 {
368 if (!target)
369 return false;
370
371 if (Creature* Telestra = target->ToCreature())
372 if (Telestra->AI()->GetData(DATA_SPLIT_PERSONALITY) == 2)
373 return true;
374
375 return false;
376 }
377};

455 void DoAction(int32 action) override
456 {
457 switch (action)
458 {
459 case ACTION_LAND_ENCOUNTER_START:
460 events.CancelEventGroup(1);
461 if (Creature* alexstraszaBunny = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ALEXSTRASZA_BUNNY_GUID)))
462 {
463 Position pos;
464 pos.m_positionZ = alexstraszaBunny->GetPositionZ();
465 alexstraszaBunny->GetNearPoint2D(pos.m_positionX, pos.m_positionY, 30.0f, alexstraszaBunny->GetAngle(me));
466 me->GetMotionMaster()->MoveLand(POINT_LAND_P_ONE, pos);
467 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC);
468 me->SetReactState(REACT_AGGRESSIVE);
469 me->SetInCombatWithZone();
470 events.ScheduleEvent(EVENT_LAND_START_ENCOUNTER, 7*IN_MILLISECONDS, 1, PHASE_NOT_STARTED);
471 }
472 break;
473 case ACTION_EXECUTE_VORTEX:
474 DoCast(me, SPELL_VORTEX_1, true);
475 DoCast(me, SPELL_VORTEX_2, true);
476 // the vortex execution continues in the dummy effect of this spell (see it's script)
477 DoCast(me, SPELL_VORTEX_3, true);
478 break;
479 case ACTION_LIFT_IN_AIR:
480 {
481 Position _zToLift = me->GetPosition();
482 if (_phase == PHASE_ONE)
483 {
484 _zToLift.m_positionZ += 20.0f;
485 me->GetMotionMaster()->MoveTakeoff(POINT_LIFT_IN_AIR_P_ONE, _zToLift);
486 }
487 else if (_phase == PHASE_TWO)
488 {
489 _zToLift.m_positionZ = 300.1f;
490 me->GetMotionMaster()->MoveTakeoff(POINT_PHASE_ONE_TO_TWO_TRANSITION, _zToLift);
491 }
492 break;
493 }
494 case ACTION_HANDLE_P_THREE_INTRO:
495 events.CancelEventGroup(0);
496 events.CancelEventGroup(1);
497 events.CancelEventGroup(2);
498 // Vehicles shouldn't be despawned with 0 delay if the call comes from virtual function that overrides PassengerBoarded.
499 // Aside from that he doesn't despawn both vehicles and arcane overloads right away, but with some delay.
500 DummyEntryCheckPredicate pred;
501 summons.DoAction(ACTION_DELAYED_DESPAWN, pred);
502 Talk(SAY_END_P_TWO);
503 me->GetMotionMaster()->Clear(false);
504 me->StopMoving();
505 if (me->GetPositionZ() > 300.0f)
506 events.ScheduleEvent(EVENT_DELAY_MOVE_TO_DESTROY_P, 5*IN_MILLISECONDS, 0, PHASE_TWO);
507 else
508 me->GetMotionMaster()->MovePoint(POINT_DESTROY_PLATFORM_P_TWO, MalygosPositions[0]);
509
510 events.ScheduleEvent(EVENT_LIGHT_DIMENSION_CHANGE, 1*IN_MILLISECONDS, 0, PHASE_TWO);
511 break;
512 case ACTION_HANDLE_RESPAWN:
513 // Teleport to spawn position, we can't use normal relocate
514 float x, y, z, o;
515 me->GetRespawnPosition(x, y, z, &o);
516 me->NearTeleportTo(x, y, z, o);
517 // Respawn Iris
518 instance->SetData(DATA_RESPAWN_IRIS, 0);
519 _despawned = false;
520 break;
521 case ACTION_CYCLIC_MOVEMENT:
522 me->GetMotionMaster()->MoveCirclePath(MalygosPositions[3].GetPositionX(), MalygosPositions[3].GetPositionY(), 283.2763f, 120.0f, true, 16);
523 break;
524 }
525 }

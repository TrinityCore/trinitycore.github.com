69class boss_trollgore : public CreatureScript
70{
71 public:
72 boss_trollgore() : CreatureScript("boss_trollgore") { }
73
74 struct boss_trollgoreAI : public BossAI
75 {
76 boss_trollgoreAI(Creature* creature) : BossAI(creature, DATA_TROLLGORE)
77 {
78 Initialize();
79 }
80
81 void Initialize()
82 {
83 _consumptionJunction = true;
84 }
85
86 void Reset() override
87 {
88 _Reset();
89 Initialize();
90 }
91
92 void EnterCombat(Unit* /*who*/) override
93 {
94 _EnterCombat();
95 Talk(SAY_AGGRO);
96
97 events.ScheduleEvent(EVENT_CONSUME, 15000);
98 events.ScheduleEvent(EVENT_CRUSH, urand(1000, 5000));
99 events.ScheduleEvent(EVENT_INFECTED_WOUND, urand(10000, 60000));
100 events.ScheduleEvent(EVENT_CORPSE_EXPLODE, 3000);
101 events.ScheduleEvent(EVENT_SPAWN, urand(30000, 40000));
102 }
103
104 void UpdateAI(uint32 diff) override
105 {
106 if (!UpdateVictim())
107 return;
108
109 events.Update(diff);
110
111 if (me->HasUnitState(UNIT_STATE_CASTING))
112 return;
113
114 while (uint32 eventId = events.ExecuteEvent())
115 {
116 switch (eventId)
117 {
118 case EVENT_CONSUME:
119 Talk(SAY_CONSUME);
120 DoCastAOE(SPELL_CONSUME);
121 events.ScheduleEvent(EVENT_CONSUME, 15000);
122 break;
123 case EVENT_CRUSH:
124 DoCastVictim(SPELL_CRUSH);
125 events.ScheduleEvent(EVENT_CRUSH, urand(10000, 15000));
126 break;
127 case EVENT_INFECTED_WOUND:
128 DoCastVictim(SPELL_INFECTED_WOUND);
129 events.ScheduleEvent(EVENT_INFECTED_WOUND, urand(25000, 35000));
130 break;
131 case EVENT_CORPSE_EXPLODE:
132 Talk(SAY_EXPLODE);
133 DoCastAOE(SPELL_CORPSE_EXPLODE);
134 events.ScheduleEvent(EVENT_CORPSE_EXPLODE, urand(15000, 19000));
135 break;
136 case EVENT_SPAWN:
137 for (uint8 i = 0; i < 3; ++i)
138 if (Creature* trigger = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_TROLLGORE_INVADER_SUMMONER_1 + i)))
139 trigger->CastSpell(trigger, RAND(SPELL_SUMMON_INVADER_A, SPELL_SUMMON_INVADER_B, SPELL_SUMMON_INVADER_C), true, NULL, NULL, me->GetGUID());
140
141 events.ScheduleEvent(EVENT_SPAWN, urand(30000, 40000));
142 break;
143 default:
144 break;
145 }
146 }
147
148 if (_consumptionJunction)
149 {
150 Aura* ConsumeAura = me->GetAura(SPELL_CONSUME_BUFF_HELPER);
151 if (ConsumeAura && ConsumeAura->GetStackAmount() > 9)
152 _consumptionJunction = false;
153 }
154
155 DoMeleeAttackIfReady();
156 }
157
158 void JustDied(Unit* /*killer*/) override
159 {
160 _JustDied();
161 Talk(SAY_DEATH);
162 }
163
164 uint32 GetData(uint32 type) const override
165 {
166 if (type == DATA_CONSUMPTION_JUNCTION)
167 return _consumptionJunction ? 1 : 0;
168
169 return 0;
170 }
171
172 void KilledUnit(Unit* victim) override
173 {
174 if (victim->GetTypeId() != TYPEID_PLAYER)
175 return;
176
177 Talk(SAY_KILL);
178 }
179
180 void JustSummoned(Creature* summon) override
181 {
182 summon->GetMotionMaster()->MovePoint(POINT_LANDING, Landing);
183 summons.Summon(summon);
184 }
185
186 private:
187 bool _consumptionJunction;
188 };
189
190 CreatureAI* GetAI(Creature* creature) const override
191 {
192 return GetDrakTharonKeepAI<boss_trollgoreAI>(creature);
193 }
194};

263 class spell_trollgore_corpse_explode_AuraScript : public AuraScript
264 {
265 PrepareAuraScript(spell_trollgore_corpse_explode_AuraScript);
266
267 bool Validate(SpellInfo const* /*spellInfo*/) override
268 {
269 if (!sSpellMgr->GetSpellInfo(SPELL_CORPSE_EXPLODE_DAMAGE))
270 return false;
271 return true;
272 }
273
274 void PeriodicTick(AuraEffect const* aurEff)
275 {
276 if (aurEff->GetTickNumber() == 2)
277 if (Unit* caster = GetCaster())
278 caster->CastSpell(GetTarget(), SPELL_CORPSE_EXPLODE_DAMAGE, true, NULL, aurEff);
279 }
280
281 void HandleRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
282 {
283 if (Creature* target = GetTarget()->ToCreature())
284 target->DespawnOrUnsummon();
285 }
286
287 void Register() override
288 {
289 OnEffectPeriodic += AuraEffectPeriodicFn(spell_trollgore_corpse_explode_AuraScript::PeriodicTick, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
290 AfterEffectRemove += AuraEffectRemoveFn(spell_trollgore_corpse_explode_AuraScript::HandleRemove, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY, AURA_EFFECT_HANDLE_REAL);
291 }
292 };
293
294 AuraScript* GetAuraScript() const override
295 {
296 return new spell_trollgore_corpse_explode_AuraScript();

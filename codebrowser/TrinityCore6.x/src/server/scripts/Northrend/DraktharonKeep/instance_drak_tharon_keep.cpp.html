18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "InstanceScript.h"
21#include "drak_tharon_keep.h"
22
23class instance_drak_tharon_keep : public InstanceMapScript
24{
25 public:
26 instance_drak_tharon_keep() : InstanceMapScript(DrakTharonKeepScriptName, 600) { }
27
28 struct instance_drak_tharon_keep_InstanceScript : public InstanceScript
29 {
30 instance_drak_tharon_keep_InstanceScript(Map* map) : InstanceScript(map)
31 {
32 SetHeaders(DataHeader);
33 SetBossNumber(EncounterCount);
34 }
35
36 void OnCreatureCreate(Creature* creature) override
37 {
38 switch (creature->GetEntry())
39 {
40 case NPC_TROLLGORE:
41 TrollgoreGUID = creature->GetGUID();
42 break;
43 case NPC_NOVOS:
44 NovosGUID = creature->GetGUID();
45 break;
46 case NPC_KING_DRED:
47 KingDredGUID = creature->GetGUID();
48 break;
49 case NPC_THARON_JA:
50 TharonJaGUID = creature->GetGUID();
51 break;
52 case NPC_WORLD_TRIGGER:
53 InitializeTrollgoreInvaderSummoner(creature);
54 break;
55 case NPC_CRYSTAL_CHANNEL_TARGET:
56 InitializeNovosSummoner(creature);
57 break;
58 default:
59 break;
60 }
61 }
62
63 void OnGameObjectCreate(GameObject* go) override
64 {
65 switch (go->GetEntry())
66 {
67 case GO_NOVOS_CRYSTAL_1:
68 NovosCrystalGUIDs[0] = go->GetGUID();
69 break;
70 case GO_NOVOS_CRYSTAL_2:
71 NovosCrystalGUIDs[1] = go->GetGUID();
72 break;
73 case GO_NOVOS_CRYSTAL_3:
74 NovosCrystalGUIDs[2] = go->GetGUID();
75 break;
76 case GO_NOVOS_CRYSTAL_4:
77 NovosCrystalGUIDs[3] = go->GetGUID();
78 break;
79 default:
80 break;
81 }
82 }
83
84 void InitializeTrollgoreInvaderSummoner(Creature* creature)
85 {
86 float y = creature->GetPositionY();
87 float z = creature->GetPositionZ();
88
89 if (z < 50.0f)
90 return;
91
92 if (y < -650.0f && y > -660.0f)
93 TrollgoreInvaderSummonerGuids[0] = creature->GetGUID();
94 else if (y < -660.0f && y > -670.0f)
95 TrollgoreInvaderSummonerGuids[1] = creature->GetGUID();
96 else if (y < -675.0f && y > -685.0f)
97 TrollgoreInvaderSummonerGuids[2] = creature->GetGUID();
98 }
99
100 void InitializeNovosSummoner(Creature* creature)
101 {
102 float x = creature->GetPositionX();
103 float y = creature->GetPositionY();
104 float z = creature->GetPositionZ();
105
106 if (x < -374.0f && x > -379.0f && y > -820.0f && y < -815.0f && z < 60.0f && z > 58.0f)
107 NovosSummonerGUIDs[0] = creature->GetGUID();
108 else if (x < -379.0f && x > -385.0f && y > -820.0f && y < -815.0f && z < 60.0f && z > 58.0f)
109 NovosSummonerGUIDs[1] = creature->GetGUID();
110 else if (x < -374.0f && x > -385.0f && y > -827.0f && y < -820.0f && z < 60.0f && z > 58.0f)
111 NovosSummonerGUIDs[2] = creature->GetGUID();
112 else if (x < -338.0f && x > -344.0f && y > -727.0f && y < 721.0f && z < 30.0f && z > 26.0f)
113 NovosSummonerGUIDs[3] = creature->GetGUID();
114 }
115
116 ObjectGuid GetGuidData(uint32 type) const override
117 {
118 switch (type)
119 {
120 case DATA_TROLLGORE:
121 return TrollgoreGUID;
122 case DATA_NOVOS:
123 return NovosGUID;
124 case DATA_KING_DRED:
125 return KingDredGUID;
126 case DATA_THARON_JA:
127 return TharonJaGUID;
128 case DATA_TROLLGORE_INVADER_SUMMONER_1:
129 case DATA_TROLLGORE_INVADER_SUMMONER_2:
130 case DATA_TROLLGORE_INVADER_SUMMONER_3:
131 return TrollgoreInvaderSummonerGuids[type - DATA_TROLLGORE_INVADER_SUMMONER_1];
132 case DATA_NOVOS_CRYSTAL_1:
133 case DATA_NOVOS_CRYSTAL_2:
134 case DATA_NOVOS_CRYSTAL_3:
135 case DATA_NOVOS_CRYSTAL_4:
136 return NovosCrystalGUIDs[type - DATA_NOVOS_CRYSTAL_1];
137 case DATA_NOVOS_SUMMONER_1:
138 case DATA_NOVOS_SUMMONER_2:
139 case DATA_NOVOS_SUMMONER_3:
140 case DATA_NOVOS_SUMMONER_4:
141 return NovosSummonerGUIDs[type - DATA_NOVOS_SUMMONER_1];
142 }
143
144 return ObjectGuid::Empty;
145 }
146
147 void OnUnitDeath(Unit* unit) override
148 {
149 if (unit->GetEntry() == NPC_CRYSTAL_HANDLER)
150 if (Creature* novos = instance->GetCreature(NovosGUID))
151 novos->AI()->DoAction(ACTION_CRYSTAL_HANDLER_DIED);
152 }
153
154 protected:
155 ObjectGuid TrollgoreGUID;
156 ObjectGuid NovosGUID;
157 ObjectGuid KingDredGUID;
158 ObjectGuid TharonJaGUID;
159
160 ObjectGuid TrollgoreInvaderSummonerGuids[3];
161 ObjectGuid NovosCrystalGUIDs[4];
162 ObjectGuid NovosSummonerGUIDs[4];
163 };
164
165 InstanceScript* GetInstanceScript(InstanceMap* map) const override
166 {
167 return new instance_drak_tharon_keep_InstanceScript(map);
168 }
169};
170
171void AddSC_instance_drak_tharon_keep()
172{
173 new instance_drak_tharon_keep();
174}

78class boss_tharon_ja : public CreatureScript
79{
80 public:
81 boss_tharon_ja() : CreatureScript("boss_tharon_ja") { }
82
83 struct boss_tharon_jaAI : public BossAI
84 {
85 boss_tharon_jaAI(Creature* creature) : BossAI(creature, DATA_THARON_JA) { }
86
87 void Reset() override
88 {
89 _Reset();
90 me->RestoreDisplayId();
91 }
92
93 void EnterCombat(Unit* /*who*/) override
94 {
95 Talk(SAY_AGGRO);
96 _EnterCombat();
97
98 events.ScheduleEvent(EVENT_DECAY_FLESH, 20000);
99 events.ScheduleEvent(EVENT_CURSE_OF_LIFE, 1000);
100 events.ScheduleEvent(EVENT_RAIN_OF_FIRE, urand(14000, 18000));
101 events.ScheduleEvent(EVENT_SHADOW_VOLLEY, urand(8000, 10000));
102 }
103
104 void KilledUnit(Unit* who) override
105 {
106 if (who->GetTypeId() == TYPEID_PLAYER)
107 Talk(SAY_KILL);
108 }
109
110 void JustDied(Unit* /*killer*/) override
111 {
112 _JustDied();
113
114 Talk(SAY_DEATH);
115 DoCastAOE(SPELL_CLEAR_GIFT_OF_THARON_JA, true);
116 DoCastAOE(SPELL_ACHIEVEMENT_CHECK, true);
117 }
118
119 void UpdateAI(uint32 diff) override
120 {
121 if (!UpdateVictim())
122 return;
123
124 events.Update(diff);
125
126 if (me->HasUnitState(UNIT_STATE_CASTING))
127 return;
128
129 while (uint32 eventId = events.ExecuteEvent())
130 {
131 switch (eventId)
132 {
133 case EVENT_CURSE_OF_LIFE:
134 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
135 DoCast(target, SPELL_CURSE_OF_LIFE);
136 events.ScheduleEvent(EVENT_CURSE_OF_LIFE, urand(10000, 15000));
137 return;
138 case EVENT_SHADOW_VOLLEY:
139 DoCastVictim(SPELL_SHADOW_VOLLEY);
140 events.ScheduleEvent(EVENT_SHADOW_VOLLEY, urand(8000, 10000));
141 return;
142 case EVENT_RAIN_OF_FIRE:
143 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
144 DoCast(target, SPELL_RAIN_OF_FIRE);
145 events.ScheduleEvent(EVENT_RAIN_OF_FIRE, urand(14000, 18000));
146 return;
147 case EVENT_LIGHTNING_BREATH:
148 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
149 DoCast(target, SPELL_LIGHTNING_BREATH);
150 events.ScheduleEvent(EVENT_LIGHTNING_BREATH, urand(6000, 7000));
151 return;
152 case EVENT_EYE_BEAM:
153 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
154 DoCast(target, SPELL_EYE_BEAM);
155 events.ScheduleEvent(EVENT_EYE_BEAM, urand(4000, 6000));
156 return;
157 case EVENT_POISON_CLOUD:
158 DoCastAOE(SPELL_POISON_CLOUD);
159 events.ScheduleEvent(EVENT_POISON_CLOUD, urand(10000, 12000));
160 return;
161 case EVENT_DECAY_FLESH:
162 DoCastAOE(SPELL_DECAY_FLESH);
163 events.ScheduleEvent(EVENT_GOING_FLESH, 6000);
164 return;
165 case EVENT_GOING_FLESH:
166 Talk(SAY_FLESH);
167 me->SetDisplayId(MODEL_FLESH);
168 DoCastAOE(SPELL_GIFT_OF_THARON_JA, true);
169 DoCast(me, SPELL_FLESH_VISUAL, true);
170 DoCast(me, SPELL_DUMMY, true);
171
172 events.Reset();
173 events.ScheduleEvent(EVENT_RETURN_FLESH, 20000);
174 events.ScheduleEvent(EVENT_LIGHTNING_BREATH, urand(3000, 4000));
175 events.ScheduleEvent(EVENT_EYE_BEAM, urand(4000, 8000));
176 events.ScheduleEvent(EVENT_POISON_CLOUD, urand(6000, 7000));
177 break;
178 case EVENT_RETURN_FLESH:
179 DoCastAOE(SPELL_RETURN_FLESH);
180 events.ScheduleEvent(EVENT_GOING_SKELETAL, 6000);
181 return;
182 case EVENT_GOING_SKELETAL:
183 Talk(SAY_SKELETON);
184 me->RestoreDisplayId();
185 DoCastAOE(SPELL_CLEAR_GIFT_OF_THARON_JA, true);
186
187 events.Reset();
188 events.ScheduleEvent(EVENT_DECAY_FLESH, 20000);
189 events.ScheduleEvent(EVENT_CURSE_OF_LIFE, 1000);
190 events.ScheduleEvent(EVENT_RAIN_OF_FIRE, urand(14000, 18000));
191 events.ScheduleEvent(EVENT_SHADOW_VOLLEY, urand(8000, 10000));
192 break;
193 default:
194 break;
195 }
196 }
197
198 DoMeleeAttackIfReady();
199 }
200 };
201
202 CreatureAI* GetAI(Creature* creature) const override
203 {
204 return GetDrakTharonKeepAI<boss_tharon_jaAI>(creature);
205 }

208class spell_tharon_ja_clear_gift_of_tharon_ja : public SpellScriptLoader
209{
210 public:
211 spell_tharon_ja_clear_gift_of_tharon_ja() : SpellScriptLoader("spell_tharon_ja_clear_gift_of_tharon_ja") { }
212
213 class spell_tharon_ja_clear_gift_of_tharon_ja_SpellScript : public SpellScript
214 {
215 PrepareSpellScript(spell_tharon_ja_clear_gift_of_tharon_ja_SpellScript);
216
217 bool Validate(SpellInfo const* /*spellInfo*/) override
218 {
219 if (!sSpellMgr->GetSpellInfo(SPELL_GIFT_OF_THARON_JA))
220 return false;
221 return true;
222 }
223
224 void HandleScript(SpellEffIndex /*effIndex*/)
225 {
226 if (Unit* target = GetHitUnit())
227 target->RemoveAura(SPELL_GIFT_OF_THARON_JA);
228 }
229
230 void Register() override
231 {
232 OnEffectHitTarget += SpellEffectFn(spell_tharon_ja_clear_gift_of_tharon_ja_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
233 }
234 };
235
236 SpellScript* GetSpellScript() const override
237 {
238 return new spell_tharon_ja_clear_gift_of_tharon_ja_SpellScript();
239 }

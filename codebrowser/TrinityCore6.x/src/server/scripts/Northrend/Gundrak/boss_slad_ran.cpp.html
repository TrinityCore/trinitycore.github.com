71class boss_slad_ran : public CreatureScript
72{
73public:
74 boss_slad_ran() : CreatureScript("boss_slad_ran") { }
75
76 struct boss_slad_ranAI : public BossAI
77 {
78 boss_slad_ranAI(Creature* creature) : BossAI(creature, DATA_SLAD_RAN)
79 {
80 Initialize();
81 }
82
83 void Initialize()
84 {
85 uiPoisonNovaTimer = 10 * IN_MILLISECONDS;
86 uiPowerfullBiteTimer = 3 * IN_MILLISECONDS;
87 uiVenomBoltTimer = 15 * IN_MILLISECONDS;
88 uiSpawnTimer = 5 * IN_MILLISECONDS;
89 uiPhase = 0;
90 }
91
92 uint32 uiPoisonNovaTimer;
93 uint32 uiPowerfullBiteTimer;
94 uint32 uiVenomBoltTimer;
95 uint32 uiSpawnTimer;
96
97 uint8 uiPhase;
98
99 GuidSet lWrappedPlayers;
100
101 void Reset() override
102 {
103 Initialize();
104 _Reset();
105 lWrappedPlayers.clear();
106 }
107
108 void EnterCombat(Unit* /*who*/) override
109 {
110 _EnterCombat();
111 Talk(SAY_AGGRO);
112 }
113
114 void UpdateAI(uint32 diff) override
115 {
116 //Return since we have no target
117 if (!UpdateVictim())
118 return;
119
120 if (uiPoisonNovaTimer <= diff)
121 {
122 DoCastVictim(SPELL_POISON_NOVA);
123 Talk(EMOTE_NOVA);
124 uiPoisonNovaTimer = 15*IN_MILLISECONDS;
125 } else uiPoisonNovaTimer -= diff;
126
127 if (uiPowerfullBiteTimer <= diff)
128 {
129 DoCastVictim(SPELL_POWERFULL_BITE);
130 uiPowerfullBiteTimer = 10*IN_MILLISECONDS;
131 } else uiPowerfullBiteTimer -= diff;
132
133 if (uiVenomBoltTimer <= diff)
134 {
135 DoCastVictim(SPELL_VENOM_BOLT);
136 uiVenomBoltTimer = 10*IN_MILLISECONDS;
137 } else uiVenomBoltTimer -= diff;
138
139 if (uiPhase)
140 {
141 if (uiSpawnTimer <= diff)
142 {
143 if (uiPhase == 1)
144 for (uint8 i = 0; i < DUNGEON_MODE(3, 5); ++i)
145 me->SummonCreature(CREATURE_SNAKE, SpawnLoc[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20*IN_MILLISECONDS);
146 if (uiPhase == 2)
147 for (uint8 i = 0; i < DUNGEON_MODE(3, 5); ++i)
148 me->SummonCreature(CREATURE_CONSTRICTORS, SpawnLoc[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20*IN_MILLISECONDS);
149 uiSpawnTimer = 5*IN_MILLISECONDS;
150 } else uiSpawnTimer -= diff;
151 }
152
153 if (uiPhase == 0 && HealthBelowPct(30))
154 {
155 Talk(SAY_SUMMON_SNAKES);
156 uiPhase = 1;
157 }
158
159 if (uiPhase == 1 && HealthBelowPct(25))
160 {
161 Talk(SAY_SUMMON_CONSTRICTORS);
162 uiPhase = 2;
163 }
164
165 DoMeleeAttackIfReady();
166 }
167
168 void JustDied(Unit* /*killer*/) override
169 {
170 _JustDied();
171 Talk(SAY_DEATH);
172 Talk(EMOTE_ACTIVATE_ALTAR);
173 }
174
175 void KilledUnit(Unit* who) override
176 {
177 if (who->GetTypeId() == TYPEID_PLAYER)
178 Talk(SAY_SLAY);
179 }
180
181 void JustSummoned(Creature* summon) override
182 {
183 summon->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
184 summons.Summon(summon);
185 }
186
187 void SetGUID(ObjectGuid guid, int32 type) override
188 {
189 if (type == DATA_SNAKES_WHYD_IT_HAVE_TO_BE_SNAKES)
190 lWrappedPlayers.insert(guid);
191 }
192
193 bool WasWrapped(ObjectGuid guid)
194 {
195 return lWrappedPlayers.count(guid) != 0;
196 }
197 };
198
199 CreatureAI* GetAI(Creature* creature) const override
200 {
201 return GetGundrakAI<boss_slad_ranAI>(creature);
202 }
203};

206{
207public:
208 npc_slad_ran_constrictor() : CreatureScript("npc_slad_ran_constrictor") { }

214
215 struct npc_slad_ran_constrictorAI : public ScriptedAI
216 {
217 npc_slad_ran_constrictorAI(Creature* creature) : ScriptedAI(creature)
218 {
219 uiGripOfSladRanTimer = 1 * IN_MILLISECONDS;
220 }
221
222 uint32 uiGripOfSladRanTimer;
223
224 void Reset() override
225 {
226 uiGripOfSladRanTimer = 1*IN_MILLISECONDS;
227 }
228
229 void UpdateAI(uint32 diff) override
230 {
231 if (!UpdateVictim())
232 return;
233
234 if (uiGripOfSladRanTimer <= diff)
235 {
236 Unit* target = me->GetVictim();
237
238 DoCast(target, SPELL_GRIP_OF_SLAD_RAN);
239 uiGripOfSladRanTimer = urand(3, 6)*IN_MILLISECONDS;
240
241 Aura* grip = target->GetAura(SPELL_GRIP_OF_SLAD_RAN, me->GetGUID());
242 if (grip && grip->GetStackAmount() == 5)
243 {
244 target->RemoveAurasDueToSpell(SPELL_GRIP_OF_SLAD_RAN, me->GetGUID());
245 target->CastSpell(target, SPELL_SNAKE_WRAP, true);
246
247 if (TempSummon* _me = me->ToTempSummon())
248 if (Unit* summoner = _me->GetSummoner())
249 if (Creature* sladran = summoner->ToCreature())
250 sladran->AI()->SetGUID(target->GetGUID(), DATA_SNAKES_WHYD_IT_HAVE_TO_BE_SNAKES);
251
252 me->DespawnOrUnsummon();
253 }
254 } else uiGripOfSladRanTimer -= diff;
255 }
256 };

299class achievement_snakes_whyd_it_have_to_be_snakes : public AchievementCriteriaScript
300{
301 public:
302 achievement_snakes_whyd_it_have_to_be_snakes() : AchievementCriteriaScript("achievement_snakes_whyd_it_have_to_be_snakes") { }
303
304 bool OnCheck(Player* player, Unit* target) override
305 {
306 if (!target)
307 return false;
308
309 if (boss_slad_ran::boss_slad_ranAI* sladRanAI = CAST_AI(boss_slad_ran::boss_slad_ranAI, target->GetAI()))
310 return !sladRanAI->WasWrapped(player->GetGUID());
311 return false;
312 }

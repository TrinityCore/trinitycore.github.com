56class instance_gundrak : public InstanceMapScript

59 instance_gundrak() : InstanceMapScript(GundrakScriptName, 604) { }

61 struct instance_gundrak_InstanceMapScript : public InstanceScript
62 {
63 instance_gundrak_InstanceMapScript(Map* map) : InstanceScript(map)
64 {
65 SetHeaders(DataHeader);
66 SetBossNumber(EncounterCount);
67 LoadDoorData(doorData);
68 LoadObjectData(creatureData, gameObjectData);
69
70 SladRanStatueState = GO_STATE_ACTIVE;
71 DrakkariColossusStatueState = GO_STATE_ACTIVE;
72 MoorabiStatueState = GO_STATE_ACTIVE;
73 }
74
75 void OnCreatureCreate(Creature* creature) override
76 {
77 switch (creature->GetEntry())
78 {
79 case NPC_RUIN_DWELLER:
80 if (creature->IsAlive())
81 DwellerGUIDs.insert(creature->GetGUID());
82 break;
83 default:
84 break;
85 }
86
87 InstanceScript::OnCreatureCreate(creature);
88 }
89
90 void OnGameObjectCreate(GameObject* go) override
91 {
92 switch (go->GetEntry())
93 {
94 case GO_SLAD_RAN_ALTAR:
95 if (GetBossState(DATA_SLAD_RAN) == DONE)
96 {
97 if (SladRanStatueState == GO_STATE_ACTIVE)
98 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
99 else
100 go->SetGoState(GO_STATE_ACTIVE);
101 }
102 break;
103 case GO_MOORABI_ALTAR:
104 if (GetBossState(DATA_MOORABI) == DONE)
105 {
106 if (MoorabiStatueState == GO_STATE_ACTIVE)
107 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
108 else
109 go->SetGoState(GO_STATE_ACTIVE);
110 }
111 break;
112 case GO_DRAKKARI_COLOSSUS_ALTAR:
113 if (GetBossState(DATA_DRAKKARI_COLOSSUS) == DONE)
114 {
115 if (DrakkariColossusStatueState == GO_STATE_ACTIVE)
116 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
117 else
118 go->SetGoState(GO_STATE_ACTIVE);
119 }
120 break;
121 case GO_SLAD_RAN_STATUE:
122 go->SetGoState(SladRanStatueState);
123 break;
124 case GO_MOORABI_STATUE:
125 go->SetGoState(MoorabiStatueState);
126 break;
127 case GO_GAL_DARAH_STATUE:
128 go->SetGoState(CheckRequiredBosses(DATA_GAL_DARAH) ? GO_STATE_ACTIVE_ALTERNATIVE : GO_STATE_READY);
129 break;
130 case GO_DRAKKARI_COLOSSUS_STATUE:
131 go->SetGoState(DrakkariColossusStatueState);
132 break;
133 case GO_ECK_THE_FEROCIOUS_DOOR:
134 // Don't store door on non-heroic
135 if (!instance->IsHeroic())
136 return;
137 break;
138 case GO_TRAPDOOR:
139 go->SetGoState(CheckRequiredBosses(DATA_GAL_DARAH) ? GO_STATE_READY : GO_STATE_ACTIVE);
140 break;
141 case GO_COLLISION:
142 go->SetGoState(CheckRequiredBosses(DATA_GAL_DARAH) ? GO_STATE_ACTIVE : GO_STATE_READY);
143 break;
144 default:
145 break;
146 }
147
148 InstanceScript::OnGameObjectCreate(go);
149 }
150
151 void OnUnitDeath(Unit* unit) override
152 {
153 if (unit->GetEntry() == NPC_RUIN_DWELLER)
154 {
155 DwellerGUIDs.erase(unit->GetGUID());
156
157 if (DwellerGUIDs.empty())
158 unit->SummonCreature(NPC_ECK_THE_FEROCIOUS, EckSpawnPoint, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300 * IN_MILLISECONDS);
159 }
160 }
161
162 bool SetBossState(uint32 type, EncounterState state) override
163 {
164 if (!InstanceScript::SetBossState(type, state))
165 return false;
166
167 switch (type)
168 {
169 case DATA_SLAD_RAN:
170 if (state == DONE)
171 if (GameObject* go = GetGameObject(DATA_SLAD_RAN_ALTAR))
172 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
173 break;
174 case DATA_DRAKKARI_COLOSSUS:
175 if (state == DONE)
176 if (GameObject* go = GetGameObject(DATA_DRAKKARI_COLOSSUS_ALTAR))
177 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
178 break;
179 case DATA_MOORABI:
180 if (state == DONE)
181 if (GameObject* go = GetGameObject(DATA_MOORABI_ALTAR))
182 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
183 break;
184 default:
185 break;
186 }
187
188 return true;
189 }
190
191 bool CheckRequiredBosses(uint32 bossId, Player const* player = nullptr) const override
192 {
193 if (_SkipCheckRequiredBosses(player))
194 return true;
195
196 switch (bossId)
197 {
198 case DATA_ECK_THE_FEROCIOUS:
199 if (!instance->IsHeroic() || GetBossState(DATA_MOORABI) != DONE)
200 return false;
201 break;
202 case DATA_GAL_DARAH:
203 if (SladRanStatueState != GO_STATE_ACTIVE_ALTERNATIVE
204 || DrakkariColossusStatueState != GO_STATE_ACTIVE_ALTERNATIVE
205 || MoorabiStatueState != GO_STATE_ACTIVE_ALTERNATIVE)
206 return false;
207 break;
208 default:
209 break;
210 }
211
212 return true;
213 }
214
215 bool IsBridgeReady() const
216 {
217 return SladRanStatueState == GO_STATE_READY && DrakkariColossusStatueState == GO_STATE_READY && MoorabiStatueState == GO_STATE_READY;
218 }

220 void SetData(uint32 type, uint32 data) override
221 {
222 if (type == DATA_STATUE_ACTIVATE)
223 {
224 switch (data)
225 {
226 case GO_SLAD_RAN_ALTAR:
227 Events.ScheduleEvent(DATA_SLAD_RAN_STATUE, TIMER_STATUE_ACTIVATION);
228 break;
229 case GO_DRAKKARI_COLOSSUS_ALTAR:
230 Events.ScheduleEvent(DATA_DRAKKARI_COLOSSUS_STATUE, TIMER_STATUE_ACTIVATION);
231 break;
232 case GO_MOORABI_ALTAR:
233 Events.ScheduleEvent(DATA_MOORABI_STATUE, TIMER_STATUE_ACTIVATION);
234 break;
235 default:
236 break;
237 }
238 }
239 }

241 void WriteSaveDataMore(std::ostringstream& data) override
242 {
243 data << uint32(SladRanStatueState) << ' ';
244 data << uint32(DrakkariColossusStatueState) << ' ';
245 data << uint32(MoorabiStatueState) << ' ';

247
248 void ReadSaveDataMore(std::istringstream& data) override
249 {
250 uint32 temp;
251
252 data >> temp;
253 SladRanStatueState = GOState(temp);
254
255 data >> temp;
256 DrakkariColossusStatueState = GOState(temp);
257
258 data >> temp;
259 MoorabiStatueState = GOState(temp);
260
261 if (IsBridgeReady())
262 Events.ScheduleEvent(DATA_BRIDGE, TIMER_STATUE_ACTIVATION);
263 }
264
265 void ToggleGameObject(uint32 type, GOState state)
266 {
267 if (GameObject* go = GetGameObject(type))
268 go->SetGoState(state);
269
270 switch (type)
271 {
272 case DATA_SLAD_RAN_STATUE:
273 SladRanStatueState = state;
274 break;
275 case DATA_DRAKKARI_COLOSSUS_STATUE:
276 DrakkariColossusStatueState = state;
277 break;
278 case DATA_MOORABI_STATUE:
279 MoorabiStatueState = state;
280 break;
281 default:
282 break;
283 }
284 }
285
286 void Update(uint32 diff) override
287 {
288 Events.Update(diff);
289
290 while (uint32 eventId = Events.ExecuteEvent())
291 {
292 uint32 spellId = 0;
293 uint32 altarId = 0;
294 switch (eventId)
295 {
296 case DATA_SLAD_RAN_STATUE:
297 spellId = SPELL_FIRE_BEAM_SNAKE;
298 altarId = DATA_SLAD_RAN_ALTAR;
299 break;
300 case DATA_DRAKKARI_COLOSSUS_STATUE:
301 spellId = SPELL_FIRE_BEAM_ELEMENTAL;
302 altarId = DATA_DRAKKARI_COLOSSUS_ALTAR;
303 break;
304 case DATA_MOORABI_STATUE:
305 spellId = SPELL_FIRE_BEAM_MAMMOTH;
306 altarId = DATA_MOORABI_ALTAR;
307 break;
308 case DATA_BRIDGE:
309 for (uint32 type = DATA_SLAD_RAN_STATUE; type <= DATA_GAL_DARAH_STATUE; ++type)
310 ToggleGameObject(type, GO_STATE_ACTIVE_ALTERNATIVE);
311 ToggleGameObject(DATA_TRAPDOOR, GO_STATE_READY);
312 ToggleGameObject(DATA_COLLISION, GO_STATE_ACTIVE);
313 SaveToDB();
314 return;
315 default:
316 return;
317 }
318
319 if (GameObject* altar = GetGameObject(altarId))
320 if (Creature* trigger = altar->FindNearestCreature(NPC_ALTAR_TRIGGER, 10.0f))
321 trigger->CastSpell((Unit*)nullptr, spellId, true);
322
323 // eventId equals statueId
324 ToggleGameObject(eventId, GO_STATE_READY);
325
326 if (IsBridgeReady())
327 Events.ScheduleEvent(DATA_BRIDGE, TIMER_STATUE_ACTIVATION);
328
329 SaveToDB();
330 }
331 }
332
333 protected:
334 EventMap Events;
335 GuidSet DwellerGUIDs;
336
337 GOState SladRanStatueState;
338 GOState DrakkariColossusStatueState;
339 GOState MoorabiStatueState;
340 };
341
342 InstanceScript* GetInstanceScript(InstanceMap* map) const override
343 {
344 return new instance_gundrak_InstanceMapScript(map);
345 }

353 bool OnGossipHello(Player* /*player*/, GameObject* go) override
354 {
355 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
356 go->SetGoState(GO_STATE_ACTIVE);
357
358 if (InstanceScript* instance = go->GetInstanceScript())
359 {
360 instance->SetData(DATA_STATUE_ACTIVATE, go->GetEntry());
361 return true;
362 }
363
364 return false;

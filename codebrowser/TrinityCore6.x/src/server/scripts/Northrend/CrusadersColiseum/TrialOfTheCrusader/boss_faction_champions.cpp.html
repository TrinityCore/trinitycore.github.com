368 std::vector<uint32> SelectChampions(Team playerTeam)
369 {
370 std::vector<uint32> vHealersEntries;
371 vHealersEntries.clear();
372 vHealersEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_DRUID_RESTORATION : NPC_ALLIANCE_DRUID_RESTORATION);
373 vHealersEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_PALADIN_HOLY : NPC_ALLIANCE_PALADIN_HOLY);
374 vHealersEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_PRIEST_DISCIPLINE : NPC_ALLIANCE_PRIEST_DISCIPLINE);
375 vHealersEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_SHAMAN_RESTORATION : NPC_ALLIANCE_SHAMAN_RESTORATION);
376
377 std::vector<uint32> vOtherEntries;
378 vOtherEntries.clear();
379 vOtherEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_DEATH_KNIGHT : NPC_ALLIANCE_DEATH_KNIGHT);
380 vOtherEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_HUNTER : NPC_ALLIANCE_HUNTER);
381 vOtherEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_MAGE : NPC_ALLIANCE_MAGE);
382 vOtherEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_ROGUE : NPC_ALLIANCE_ROGUE);
383 vOtherEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_WARLOCK : NPC_ALLIANCE_WARLOCK);
384 vOtherEntries.push_back(playerTeam == ALLIANCE ? NPC_HORDE_WARRIOR : NPC_ALLIANCE_WARRIOR);
385
386 uint8 healersSubtracted = 2;
387 if (_instance->instance->GetSpawnMode() == DIFFICULTY_25_N || _instance->instance->GetSpawnMode() == DIFFICULTY_25_HC)
388 healersSubtracted = 1;
389 for (uint8 i = 0; i < healersSubtracted; ++i)
390 {
391 uint8 pos = urand(0, vHealersEntries.size() - 1);
392 switch (vHealersEntries[pos])
393 {
394 case NPC_ALLIANCE_DRUID_RESTORATION:
395 vOtherEntries.push_back(NPC_ALLIANCE_DRUID_BALANCE);
396 break;
397 case NPC_HORDE_DRUID_RESTORATION:
398 vOtherEntries.push_back(NPC_HORDE_DRUID_BALANCE);
399 break;
400 case NPC_ALLIANCE_PALADIN_HOLY:
401 vOtherEntries.push_back(NPC_ALLIANCE_PALADIN_RETRIBUTION);
402 break;
403 case NPC_HORDE_PALADIN_HOLY:
404 vOtherEntries.push_back(NPC_HORDE_PALADIN_RETRIBUTION);
405 break;
406 case NPC_ALLIANCE_PRIEST_DISCIPLINE:
407 vOtherEntries.push_back(NPC_ALLIANCE_PRIEST_SHADOW);
408 break;
409 case NPC_HORDE_PRIEST_DISCIPLINE:
410 vOtherEntries.push_back(NPC_HORDE_PRIEST_SHADOW);
411 break;
412 case NPC_ALLIANCE_SHAMAN_RESTORATION:
413 vOtherEntries.push_back(NPC_ALLIANCE_SHAMAN_ENHANCEMENT);
414 break;
415 case NPC_HORDE_SHAMAN_RESTORATION:
416 vOtherEntries.push_back(NPC_HORDE_SHAMAN_ENHANCEMENT);
417 break;
418 default:
419 break;
420 }
421 vHealersEntries.erase(vHealersEntries.begin() + pos);
422 }
423
424 if (_instance->instance->GetSpawnMode() == DIFFICULTY_10_N || _instance->instance->GetSpawnMode() == DIFFICULTY_10_HC)
425 for (uint8 i = 0; i < 4; ++i)
426 vOtherEntries.erase(vOtherEntries.begin() + urand(0, vOtherEntries.size() - 1));
427
428 std::vector<uint32> vChampionEntries;
429 vChampionEntries.clear();
430 for (uint8 i = 0; i < vHealersEntries.size(); ++i)
431 vChampionEntries.push_back(vHealersEntries[i]);
432 for (uint8 i = 0; i < vOtherEntries.size(); ++i)
433 vChampionEntries.push_back(vOtherEntries[i]);
434
435 return vChampionEntries;
436 }

438 void SummonChampions(Team playerTeam)
439 {
440 std::vector<Position> vChampionJumpOrigin;
441 if (playerTeam == ALLIANCE)
442 for (uint8 i = 0; i < 5; i++)
443 vChampionJumpOrigin.push_back(FactionChampionLoc[i]);
444 else
445 for (uint8 i = 5; i < 10; i++)
446 vChampionJumpOrigin.push_back(FactionChampionLoc[i]);
447
448 std::vector<Position> vChampionJumpTarget;
449 for (uint8 i = 10; i < 20; i++)
450 vChampionJumpTarget.push_back(FactionChampionLoc[i]);
451 std::vector<uint32> vChampionEntries = SelectChampions(playerTeam);
452
453 for (uint8 i = 0; i < vChampionEntries.size(); ++i)
454 {
455 uint8 pos = urand(0, vChampionJumpTarget.size()-1);
456 if (Creature* temp = me->SummonCreature(vChampionEntries[i], vChampionJumpOrigin[urand(0, vChampionJumpOrigin.size()-1)], TEMPSUMMON_MANUAL_DESPAWN))
457 {
458 _summons.Summon(temp);
459 temp->SetReactState(REACT_PASSIVE);
460 temp->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_IMMUNE_TO_PC);
461 if (playerTeam == ALLIANCE)
462 {
463 temp->SetHomePosition(vChampionJumpTarget[pos].GetPositionX(), vChampionJumpTarget[pos].GetPositionY(), vChampionJumpTarget[pos].GetPositionZ(), 0);
464 temp->GetMotionMaster()->MoveJump(vChampionJumpTarget[pos].GetPositionX(), vChampionJumpTarget[pos].GetPositionY(), vChampionJumpTarget[pos].GetPositionZ(), 20.0f, 20.0f);
465 temp->SetOrientation(0);
466 }
467 else
468 {
469 temp->SetHomePosition((ToCCommonLoc[1].GetPositionX()*2)-vChampionJumpTarget[pos].GetPositionX(), vChampionJumpTarget[pos].GetPositionY(), vChampionJumpTarget[pos].GetPositionZ(), 3);
470 temp->GetMotionMaster()->MoveJump((ToCCommonLoc[1].GetPositionX()*2)-vChampionJumpTarget[pos].GetPositionX(), vChampionJumpTarget[pos].GetPositionY(), vChampionJumpTarget[pos].GetPositionZ(), 20.0f, 20.0f);
471 temp->SetOrientation(3);
472 }
473 }
474 vChampionJumpTarget.erase(vChampionJumpTarget.begin()+pos);
475 }

478 void SetData(uint32 uiType, uint32 uiData) override
479 {
480 switch (uiType)
481 {
482 case 0:
483 SummonChampions((Team)uiData);
484 break;
485 case 1:
486 for (SummonList::iterator i = _summons.begin(); i != _summons.end(); ++i)
487 {
488 if (Creature* temp = ObjectAccessor::GetCreature(*me, *i))
489 {
490 temp->SetReactState(REACT_AGGRESSIVE);
491 temp->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_IMMUNE_TO_PC);
492 }
493 }
494 break;
495 case 2:
496 switch (uiData)
497 {
498 case FAIL:
499 _championsFailed++;
500 if (_championsFailed + _championsKilled >= _summons.size())
501 {
502 _instance->SetBossState(BOSS_CRUSADERS, FAIL);
503 _summons.DespawnAll();
504 me->DespawnOrUnsummon();
505 }
506 break;
507 case IN_PROGRESS:
508 if (!_inProgress)
509 {
510 _championsNotStarted = 0;
511 _championsFailed = 0;
512 _championsKilled = 0;
513 _inProgress = true;
514 _summons.DoZoneInCombat();
515 _instance->SetBossState(BOSS_CRUSADERS, IN_PROGRESS);
516 }
517 break;
518 case DONE:
519 _championsKilled++;
520 if (_championsKilled == 1)
521 _instance->SetBossState(BOSS_CRUSADERS, SPECIAL);
522 else if (_championsKilled >= _summons.size())
523 {
524 _instance->SetBossState(BOSS_CRUSADERS, DONE);
525 _summons.DespawnAll();
526 me->DespawnOrUnsummon();
527 }
528 break;
529 default:
530 break;
531 }
532 break;
533 default:
534 break;
535 }
536 }

541 uint32 _championsFailed;
542 uint32 _championsKilled;
543 bool _inProgress;
544 };

573 float CalculateThreat(float distance, float armor, uint32 health)
574 {
575 float dist_mod = (_aiType == AI_MELEE || _aiType == AI_PET) ? 15.0f / (15.0f + distance) : 1.0f;
576 float armor_mod = (_aiType == AI_MELEE || _aiType == AI_PET) ? armor / 16635.0f : 0.0f;
577 float eh = (health + 1) * (1.0f + armor_mod);
578 return dist_mod * 30000.0f / eh;
579 }

581 void UpdateThreat()
582 {
583 std::list<HostileReference*> const& tList = me->getThreatManager().getThreatList();
584 for (std::list<HostileReference*>::const_iterator itr = tList.begin(); itr != tList.end(); ++itr)
585 {
586 Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid());
587 if (unit && me->getThreatManager().getThreat(unit))
588 {
589 if (unit->GetTypeId() == TYPEID_PLAYER)
590 {
591 float threat = CalculateThreat(me->GetDistance2d(unit), (float)unit->GetArmor(), unit->GetHealth());
592 me->getThreatManager().modifyThreatPercent(unit, -100);
593 me->AddThreat(unit, 1000000.0f * threat);
594 }
595 }
596 }

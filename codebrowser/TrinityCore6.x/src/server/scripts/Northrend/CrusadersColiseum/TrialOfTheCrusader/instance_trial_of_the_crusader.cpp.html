48 bool IsEncounterInProgress() const override
49 {
50 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
51 if (GetBossState(i) == IN_PROGRESS)
52 return true;
53
54 // Special state is set at Faction Champions after first champ dead, encounter is still in combat
55 if (GetBossState(BOSS_CRUSADERS) == SPECIAL)
56 return true;
57
58 return false;
59 }

200 bool SetBossState(uint32 type, EncounterState state) override
201 {
202 if (!InstanceScript::SetBossState(type, state))
203 return false;
204
205 switch (type)
206 {
207 case BOSS_BEASTS:
208 break;
209 case BOSS_JARAXXUS:
210 // Cleanup Icehowl
211 if (Creature* icehowl = instance->GetCreature(IcehowlGUID))
212 icehowl->DespawnOrUnsummon();
213 if (state == DONE)
214 EventStage = 2000;
215 break;
216 case BOSS_CRUSADERS:
217 // Cleanup Jaraxxus
218 if (Creature* jaraxxus = instance->GetCreature(JaraxxusGUID))
219 jaraxxus->DespawnOrUnsummon();
220 if (Creature* fizzlebang = instance->GetCreature(FizzlebangGUID))
221 fizzlebang->DespawnOrUnsummon();
222 switch (state)
223 {
224 case IN_PROGRESS:
225 ResilienceWillFixItTimer = 0;
226 break;
227 case SPECIAL: //Means the first blood
228 ResilienceWillFixItTimer = 60*IN_MILLISECONDS;
229 state = IN_PROGRESS;
230 break;
231 case DONE:
232 DoUpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_DEFEAT_FACTION_CHAMPIONS);
233 if (ResilienceWillFixItTimer > 0)
234 DoUpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_CHAMPIONS_KILLED_IN_MINUTE);
235 DoRespawnGameObject(CrusadersCacheGUID, 7*DAY);
236 if (GameObject* cache = instance->GetGameObject(CrusadersCacheGUID))
237 cache->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
238 EventStage = 3100;
239 break;
240 default:
241 break;
242 }
243 break;
244 case BOSS_VALKIRIES:
245 // Cleanup chest
246 if (GameObject* cache = instance->GetGameObject(CrusadersCacheGUID))
247 cache->Delete();
248 switch (state)
249 {
250 case FAIL:
251 if (GetBossState(BOSS_VALKIRIES) == NOT_STARTED)
252 state = NOT_STARTED;
253 break;
254 case SPECIAL:
255 if (GetBossState(BOSS_VALKIRIES) == SPECIAL)
256 state = DONE;
257 break;
258 default:
259 break;
260 }
261 break;
262 case BOSS_LICH_KING:
263 break;
264 case BOSS_ANUBARAK:
265 switch (state)
266 {
267 case DONE:
268 {
269 EventStage = 6000;
270 uint32 tributeChest = 0;
271 if (instance->GetSpawnMode() == DIFFICULTY_10_HC)
272 {
273 if (TrialCounter >= 50)
274 tributeChest = GO_TRIBUTE_CHEST_10H_99;
275 else
276 {
277 if (TrialCounter >= 45)
278 tributeChest = GO_TRIBUTE_CHEST_10H_50;
279 else
280 {
281 if (TrialCounter >= 25)
282 tributeChest = GO_TRIBUTE_CHEST_10H_45;
283 else
284 tributeChest = GO_TRIBUTE_CHEST_10H_25;
285 }
286 }
287 }
288 else if (instance->GetSpawnMode() == DIFFICULTY_25_HC)
289 {
290 if (TrialCounter >= 50)
291 tributeChest = GO_TRIBUTE_CHEST_25H_99;
292 else
293 {
294 if (TrialCounter >= 45)
295 tributeChest = GO_TRIBUTE_CHEST_25H_50;
296 else
297 {
298 if (TrialCounter >= 25)
299 tributeChest = GO_TRIBUTE_CHEST_25H_45;
300 else
301 tributeChest = GO_TRIBUTE_CHEST_25H_25;
302 }
303 }
304 }
305
306 if (tributeChest)
307 if (Creature* tirion = instance->GetCreature(TirionGUID))
308 if (GameObject* chest = tirion->SummonGameObject(tributeChest, 805.62f, 134.87f, 142.16f, 3.27f, 0, 0, 0, 0, WEEK))
309 chest->SetRespawnTime(chest->GetRespawnDelay());
310 break;
311 }
312 default:
313 break;
314 }
315 break;
316 default:
317 break;
318 }
319
320 if (IsEncounterInProgress())
321 {
322 CloseDoor(GetGuidData(GO_EAST_PORTCULLIS));
323 CloseDoor(GetGuidData(GO_WEB_DOOR));
324 }
325 else
326 {
327 OpenDoor(GetGuidData(GO_EAST_PORTCULLIS));
328 OpenDoor(GetGuidData(GO_WEB_DOOR));
329 }
330
331 if (type < MAX_ENCOUNTERS)
332 {
333 TC_LOG_DEBUG("scripts", "[ToCr] BossState(type %u) %u = state %u;", type, GetBossState(type), state);
334 if (state == FAIL)
335 {
336 if (instance->IsHeroic())
337 {
338 --TrialCounter;
339 // decrease attempt counter at wipe
340 Map::PlayerList const &PlayerList = instance->GetPlayers();
341 for (Map::PlayerList::const_iterator itr = PlayerList.begin(); itr != PlayerList.end(); ++itr)
342 if (Player* player = itr->GetSource())
343 player->SendUpdateWorldState(UPDATE_STATE_UI_COUNT, TrialCounter);
344
345 // if theres no more attemps allowed
346 if (!TrialCounter)
347 {
348 if (Unit* announcer = instance->GetCreature(GetGuidData(NPC_BARRENT)))
349 announcer->ToCreature()->DespawnOrUnsummon();
350
351 if (Creature* anubArak = instance->GetCreature(GetGuidData(NPC_ANUBARAK)))
352 anubArak->DespawnOrUnsummon();
353 }
354 }
355 NeedSave = true;
356 EventStage = (type == BOSS_BEASTS ? 666 : 0);
357 state = NOT_STARTED;
358 }
359
360 if (state == DONE || NeedSave)
361 {
362 if (Unit* announcer = instance->GetCreature(GetGuidData(NPC_BARRENT)))
363 announcer->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
364 Save();
365 }
366 }
367 return true;
368 }

370 void SetData(uint32 type, uint32 data) override
371 {
372 switch (type)
373 {
374 case TYPE_COUNTER:
375 TrialCounter = data;
376 data = DONE;
377 break;
378 case TYPE_EVENT:
379 EventStage = data;
380 data = NOT_STARTED;
381 break;
382 case TYPE_EVENT_TIMER:
383 EventTimer = data;
384 data = NOT_STARTED;
385 break;
386 case TYPE_NORTHREND_BEASTS:
387 NorthrendBeasts = data;
388 switch (data)
389 {
390 case GORMOK_DONE:
391 EventStage = 200;
392 SetData(TYPE_NORTHREND_BEASTS, IN_PROGRESS);
393 break;
394 case SNAKES_IN_PROGRESS:
395 NotOneButTwoJormungarsTimer = 0;
396 break;
397 case SNAKES_SPECIAL:
398 NotOneButTwoJormungarsTimer = 10*IN_MILLISECONDS;
399 break;
400 case SNAKES_DONE:
401 if (NotOneButTwoJormungarsTimer > 0)
402 DoUpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_WORMS_KILLED_IN_10_SECONDS);
403 EventStage = 300;
404 SetData(TYPE_NORTHREND_BEASTS, IN_PROGRESS);
405 break;
406 case ICEHOWL_DONE:
407 EventStage = 400;
408 SetData(TYPE_NORTHREND_BEASTS, DONE);
409 SetBossState(BOSS_BEASTS, DONE);
410 break;
411 case FAIL:
412 SetBossState(BOSS_BEASTS, FAIL);
413 break;
414 default:
415 break;
416 }
417 break;
418 //Achievements
419 case DATA_SNOBOLD_COUNT:
420 if (data == INCREASE)
421 ++SnoboldCount;
422 else if (data == DECREASE)
423 --SnoboldCount;
424 break;
425 case DATA_MISTRESS_OF_PAIN_COUNT:
426 if (data == INCREASE)
427 ++MistressOfPainCount;
428 else if (data == DECREASE)
429 --MistressOfPainCount;
430 break;
431 default:
432 break;
433 }

593 void Update(uint32 diff) override
594 {
595 if (GetData(TYPE_NORTHREND_BEASTS) == SNAKES_SPECIAL && NotOneButTwoJormungarsTimer)
596 {
597 if (NotOneButTwoJormungarsTimer <= diff)
598 NotOneButTwoJormungarsTimer = 0;
599 else
600 NotOneButTwoJormungarsTimer -= diff;
601 }
602
603 if (GetBossState(BOSS_CRUSADERS) == SPECIAL && ResilienceWillFixItTimer)
604 {
605 if (ResilienceWillFixItTimer <= diff)
606 ResilienceWillFixItTimer = 0;
607 else

634 void Load(const char* strIn) override
635 {
636 if (!strIn)
637 {
638 OUT_LOAD_INST_DATA_FAIL;
639 return;
640 }
641
642 OUT_LOAD_INST_DATA(strIn);
643
644 std::istringstream loadStream(strIn);
645
646 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
647 {
648 uint32 tmpState;
649 loadStream >> tmpState;
650 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
651 tmpState = NOT_STARTED;
652 SetBossState(i, EncounterState(tmpState));
653 }
654
655 loadStream >> TrialCounter;
656 EventStage = 0;
657
658 OUT_LOAD_INST_DATA_COMPLETE;
659 }

661 bool CheckAchievementCriteriaMeet(uint32 criteria_id, Player const* /*source*/, Unit const* /*target*/, uint32 /*miscvalue1*/) override
662 {
663 switch (criteria_id)
664 {
665 case UPPER_BACK_PAIN_10_PLAYER:
666 case UPPER_BACK_PAIN_10_PLAYER_HEROIC:
667 return SnoboldCount >= 2;
668 case UPPER_BACK_PAIN_25_PLAYER:
669 case UPPER_BACK_PAIN_25_PLAYER_HEROIC:
670 return SnoboldCount >= 4;
671 case THREE_SIXTY_PAIN_SPIKE_10_PLAYER:
672 case THREE_SIXTY_PAIN_SPIKE_10_PLAYER_HEROIC:
673 case THREE_SIXTY_PAIN_SPIKE_25_PLAYER:
674 case THREE_SIXTY_PAIN_SPIKE_25_PLAYER_HEROIC:
675 return MistressOfPainCount >= 2;
676 case A_TRIBUTE_TO_SKILL_10_PLAYER:
677 case A_TRIBUTE_TO_SKILL_25_PLAYER:
678 return TrialCounter >= 25;
679 case A_TRIBUTE_TO_MAD_SKILL_10_PLAYER:
680 case A_TRIBUTE_TO_MAD_SKILL_25_PLAYER:
681 return TrialCounter >= 45;
682 case A_TRIBUTE_TO_INSANITY_10_PLAYER:
683 case A_TRIBUTE_TO_INSANITY_25_PLAYER:
684 case REALM_FIRST_GRAND_CRUSADER:
685 return TrialCounter == 50;
686 case A_TRIBUTE_TO_IMMORTALITY_HORDE:
687 case A_TRIBUTE_TO_IMMORTALITY_ALLIANCE:
688 return TrialCounter == 50 && TributeToImmortalityEligible;
689 case A_TRIBUTE_TO_DEDICATED_INSANITY:
690 return false/*uiGrandCrusaderAttemptsLeft == 50 && !bHasAtAnyStagePlayerEquippedTooGoodItem*/;
691 default:
692 break;
693 }
694
695 return false;
696 }

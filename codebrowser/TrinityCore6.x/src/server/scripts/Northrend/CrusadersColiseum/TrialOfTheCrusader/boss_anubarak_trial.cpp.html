178 void Reset() override
179 {
180 _Reset();
181 events.SetPhase(PHASE_MELEE);
182 events.ScheduleEvent(EVENT_FREEZE_SLASH, 15*IN_MILLISECONDS, 0, PHASE_MELEE);
183 events.ScheduleEvent(EVENT_PENETRATING_COLD, 20*IN_MILLISECONDS, PHASE_MELEE);
184 events.ScheduleEvent(EVENT_SUMMON_NERUBIAN, 10*IN_MILLISECONDS, 0, PHASE_MELEE);
185 events.ScheduleEvent(EVENT_SUBMERGE, 80*IN_MILLISECONDS, 0, PHASE_MELEE);
186 events.ScheduleEvent(EVENT_BERSERK, 10*MINUTE*IN_MILLISECONDS);
187 if (IsHeroic())
188 events.ScheduleEvent(EVENT_NERUBIAN_SHADOW_STRIKE, 30*IN_MILLISECONDS, 0, PHASE_MELEE);
189
190 if (!IsHeroic())
191 events.ScheduleEvent(EVENT_SUMMON_FROST_SPHERE, 20*IN_MILLISECONDS);
192
193 Initialize();
194 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
195 // clean up spawned Frost Spheres
196 std::list<Creature*> FrostSphereList;
197 me->GetCreatureListWithEntryInGrid(FrostSphereList, NPC_FROST_SPHERE, 150.0f);
198 if (!FrostSphereList.empty())
199 for (std::list<Creature*>::iterator itr = FrostSphereList.begin(); itr != FrostSphereList.end(); ++itr)
200 (*itr)->DespawnOrUnsummon();
201
202 _burrowGUID.clear();
203 }

246 void JustSummoned(Creature* summoned) override
247 {
248 switch (summoned->GetEntry())
249 {
250 case NPC_BURROW:
251 _burrowGUID.push_back(summoned->GetGUID());
252 summoned->SetReactState(REACT_PASSIVE);
253 summoned->CastSpell(summoned, SPELL_CHURNING_GROUND, false);
254 summoned->SetDisplayId(summoned->GetCreatureTemplate()->Modelid2);
255 break;
256 case NPC_SPIKE:
257 summoned->SetDisplayId(summoned->GetCreatureTemplate()->Modelid1);
258 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
259 {
260 summoned->CombatStart(target);
261 Talk(EMOTE_SPIKE, target);
262 }
263 break;
264 default:
265 break;
266 }
267 summons.Summon(summoned);

270 void EnterCombat(Unit* /*who*/) override
271 {
272 _EnterCombat();
273 Talk(SAY_AGGRO);
274 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
275
276 // Despawn Scarab Swarms neutral
277 EntryCheckPredicate pred(NPC_SCARAB);
278 summons.DoAction(ACTION_SCARAB_SUBMERGE, pred);
279
280 // Spawn Burrow
281 for (int i = 0; i < 4; i++)
282 me->SummonCreature(NPC_BURROW, AnubarakLoc[i + 2]);
283
284 // Spawn 6 Frost Spheres at start
285 for (int i = 0; i < 6; i++)
286 if (Unit* summoned = me->SummonCreature(NPC_FROST_SPHERE, SphereSpawn[i]))
287 _sphereGUID[i] = summoned->GetGUID();
288 }

290 void UpdateAI(uint32 diff) override
291 {
292 if (!UpdateVictim())
293 return;
294
295 if (me->HasUnitState(UNIT_STATE_CASTING))
296 return;
297
298 events.Update(diff);
299
300 while (uint32 eventId = events.ExecuteEvent())
301 {
302 switch (eventId)
303 {
304 case EVENT_FREEZE_SLASH:
305 DoCastVictim(SPELL_FREEZE_SLASH);
306 events.ScheduleEvent(EVENT_FREEZE_SLASH, 15*IN_MILLISECONDS, 0, PHASE_MELEE);
307 return;
308 case EVENT_PENETRATING_COLD:
309 me->CastCustomSpell(SPELL_PENETRATING_COLD, SPELLVALUE_MAX_TARGETS, RAID_MODE(2, 5, 2, 5));
310 events.ScheduleEvent(EVENT_PENETRATING_COLD, 20*IN_MILLISECONDS, 0, PHASE_MELEE);
311 return;
312 case EVENT_SUMMON_NERUBIAN:
313 if (IsHeroic() || !_reachedPhase3)
314 me->CastCustomSpell(SPELL_SUMMON_BURROWER, SPELLVALUE_MAX_TARGETS, RAID_MODE(1, 2, 2, 4));
315 events.ScheduleEvent(EVENT_SUMMON_NERUBIAN, 45*IN_MILLISECONDS, 0, PHASE_MELEE);
316 return;
317 case EVENT_NERUBIAN_SHADOW_STRIKE:
318 {
319 EntryCheckPredicate pred(NPC_BURROWER);
320 summons.DoAction(ACTION_SHADOW_STRIKE, pred);
321 events.ScheduleEvent(EVENT_NERUBIAN_SHADOW_STRIKE, 30*IN_MILLISECONDS, 0, PHASE_MELEE);
322 break;
323 }
324 case EVENT_SUBMERGE:
325 if (!_reachedPhase3 && !me->HasAura(SPELL_BERSERK))
326 {
327 DoCast(me, SPELL_SUBMERGE_ANUBARAK);
328 DoCast(me, SPELL_CLEAR_ALL_DEBUFFS);
329 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
330 Talk(EMOTE_BURROWER);
331 events.SetPhase(PHASE_SUBMERGED);
332 events.ScheduleEvent(EVENT_PURSUING_SPIKE, 2*IN_MILLISECONDS, 0, PHASE_SUBMERGED);
333 events.ScheduleEvent(EVENT_SUMMON_SCARAB, 4*IN_MILLISECONDS, 0, PHASE_SUBMERGED);
334 events.ScheduleEvent(EVENT_EMERGE, 1*MINUTE*IN_MILLISECONDS, 0, PHASE_SUBMERGED);
335 }
336 break;
337 case EVENT_PURSUING_SPIKE:
338 DoCast(SPELL_SPIKE_CALL);
339 break;
340 case EVENT_SUMMON_SCARAB:
341 {
342 /* WORKAROUND
343 * - The correct implementation is more likely the comment below but it needs spell knowledge
344 */
345 GuidList::iterator i = _burrowGUID.begin();
346 uint32 at = urand(0, _burrowGUID.size()-1);
347 for (uint32 k = 0; k < at; k++)
348 ++i;
349 if (Creature* pBurrow = ObjectAccessor::GetCreature(*me, *i))
350 pBurrow->CastSpell(pBurrow, 66340, false);
351
352 events.ScheduleEvent(EVENT_SUMMON_SCARAB, 4*IN_MILLISECONDS, 0, PHASE_SUBMERGED);
353
354 /*It seems that this spell have something more that needs to be taken into account
355 //Need more sniff info
356 DoCast(SPELL_SUMMON_BEATLES);
357 // Just to make sure it won't happen again in this phase
358 m_uiSummonScarabTimer = 90*IN_MILLISECONDS;*/
359 break;
360 }
361 case EVENT_EMERGE:
362 events.ScheduleEvent(EVENT_SUBMERGE, 80*IN_MILLISECONDS, 0, PHASE_MELEE);
363 DoCast(SPELL_SPIKE_TELE);
364 summons.DespawnEntry(NPC_SPIKE);
365 me->RemoveAurasDueToSpell(SPELL_SUBMERGE_ANUBARAK);
366 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
367 DoCast(me, SPELL_EMERGE_ANUBARAK);
368 events.SetPhase(PHASE_MELEE);
369 events.ScheduleEvent(EVENT_FREEZE_SLASH, 15*IN_MILLISECONDS, 0, PHASE_MELEE);
370 events.ScheduleEvent(EVENT_PENETRATING_COLD, 20*IN_MILLISECONDS, PHASE_MELEE);
371 events.ScheduleEvent(EVENT_SUMMON_NERUBIAN, 10*IN_MILLISECONDS, 0, PHASE_MELEE);
372 events.ScheduleEvent(EVENT_SUBMERGE, 80*IN_MILLISECONDS, 0, PHASE_MELEE);
373 if (IsHeroic())
374 events.ScheduleEvent(EVENT_NERUBIAN_SHADOW_STRIKE, 30*IN_MILLISECONDS, 0, PHASE_MELEE);
375 return;
376 case EVENT_SUMMON_FROST_SPHERE:
377 {
378 uint8 startAt = urand(0, 5);
379 uint8 i = startAt;
380 do
381 {
382 if (Unit* pSphere = ObjectAccessor::GetCreature(*me, _sphereGUID[i]))
383 {
384 if (!pSphere->HasAura(SPELL_FROST_SPHERE))
385 {
386 if (Creature* summon = me->SummonCreature(NPC_FROST_SPHERE, SphereSpawn[i]))
387 _sphereGUID[i] = summon->GetGUID();
388 break;
389 }
390 }
391 i = (i + 1) % 6;
392 }
393 while
394 (i != startAt);
395 events.ScheduleEvent(EVENT_SUMMON_FROST_SPHERE, urand(20*IN_MILLISECONDS, 30*IN_MILLISECONDS));
396 break;
397 }
398 case EVENT_BERSERK:
399 DoCast(me, SPELL_BERSERK);
400 break;
401 default:
402 break;
403 }
404
405 }
406
407 if (HealthBelowPct(30) && events.IsInPhase(PHASE_MELEE) && !_reachedPhase3)
408 {
409 _reachedPhase3 = true;
410 DoCastAOE(SPELL_LEECHING_SWARM);
411 Talk(EMOTE_LEECHING_SWARM);
412 Talk(SAY_LEECHING_SWARM);
413 }
414
415 if (events.IsInPhase(PHASE_MELEE))
416 DoMeleeAttackIfReady();
417 }
418
419 private:
420 GuidList _burrowGUID;
421 ObjectGuid _sphereGUID[6];
422 bool _intro;
423 bool _reachedPhase3;
424 };

426 CreatureAI* GetAI(Creature* creature) const override
427 {
428 return GetInstanceAI<boss_anubarak_trialAI>(creature);

562

564 return;

567 {

569 {
570 me->RemoveAurasDueToSpell(SPELL_SUBMERGE_EFFECT);
571 DoCast(me, SPELL_EMERGE_EFFECT);

573 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
574 }
575 else
576 {

578 {
579 DoCast(me, SPELL_SUBMERGE_EFFECT);
580 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
581 DoCast(me, SPELL_PERSISTENT_DIRT, true);
582 }
583 }
584 _submergeTimer = 20*IN_MILLISECONDS;
585 }
586 else
587 _submergeTimer -= diff;
588
589 DoMeleeAttackIfReady();
590 }
591

594 EventMap _events;
595 InstanceScript* _instance;
596 };

621 void DamageTaken(Unit* /*who*/, uint32& damage) override
622 {
623 if (me->GetHealth() <= damage)
624 {
625 damage = 0;
626 float floorZ = me->GetMap()->GetHeight(me->GetPhaseMask(), me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
627 if (fabs(me->GetPositionZ() - floorZ) < 0.1f)
628 {
629 // we are close to the ground
630 me->GetMotionMaster()->MoveIdle();
631 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
632 me->RemoveAurasDueToSpell(SPELL_FROST_SPHERE);
633 DoCast(SPELL_PERMAFROST_MODEL);
634 DoCast(SPELL_PERMAFROST);
635 me->SetObjectScale(2.0f);
636 }
637 else
638 {
639 // we are in air
640 me->GetMotionMaster()->MoveIdle();
641 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
642 //At hit the ground
643 me->HandleEmoteCommand(EMOTE_ONESHOT_FLYDEATH);
644 me->GetMotionMaster()->MoveFall(POINT_FALL_GROUND);
645 }
646 }
647 }

658 DoCast(SPELL_PERMAFROST_MODEL);
659 DoCast(SPELL_PERMAFROST_VISUAL);

719 void UpdateAI(uint32 diff) override
720 {
721 if (!UpdateVictim())
722 {
723 me->DisappearAndDie();
724 return;
725 }
726
727 if (_phaseSwitchTimer)
728 {
729 if (_phaseSwitchTimer <= diff)
730 {
731 switch (_phase)
732 {
733 case PHASE_NO_MOVEMENT:
734 DoCast(me, SPELL_SPIKE_SPEED1);
735 DoCast(me, SPELL_SPIKE_TRAIL);
736 _phase = PHASE_IMPALE_NORMAL;
737 if (Unit* target2 = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
738 {
739 StartChase(target2);
740 Talk(EMOTE_SPIKE, target2);
741 }
742 _phaseSwitchTimer = 7*IN_MILLISECONDS;
743 return;
744 case PHASE_IMPALE_NORMAL:
745 DoCast(me, SPELL_SPIKE_SPEED2);
746 _phase = PHASE_IMPALE_MIDDLE;
747 _phaseSwitchTimer = 7*IN_MILLISECONDS;
748 return;
749 case PHASE_IMPALE_MIDDLE:
750 DoCast(me, SPELL_SPIKE_SPEED3);
751 _phase = PHASE_IMPALE_FAST;
752 _phaseSwitchTimer = 0;
753 return;
754 default:
755 return;
756 }
757 }
758 else
759 _phaseSwitchTimer -= diff;
760 }
761 }
762
763 void MoveInLineOfSight(Unit* pWho) override
764 {
765 if (!pWho)
766 return;
767
768 if (pWho->GetEntry() != NPC_FROST_SPHERE)
769 return;
770
771 if (_phase == PHASE_NO_MOVEMENT)
772 return;
773
774 if (me->IsWithinDist(pWho, 7.0f))
775 {
776 switch (_phase)
777 {
778 case PHASE_IMPALE_NORMAL:
779 me->RemoveAurasDueToSpell(SPELL_SPIKE_SPEED1);
780 break;
781 case PHASE_IMPALE_MIDDLE:
782 me->RemoveAurasDueToSpell(SPELL_SPIKE_SPEED2);
783 break;
784 case PHASE_IMPALE_FAST:
785 me->RemoveAurasDueToSpell(SPELL_SPIKE_SPEED3);
786 break;
787 default:
788 break;
789 }
790
791 me->CastSpell(me, SPELL_SPIKE_FAIL, true);
792
793 pWho->ToCreature()->DespawnOrUnsummon(3*IN_MILLISECONDS);
794
795 // After the spikes hit the icy surface they can't move for about ~5 seconds
796 _phase = PHASE_NO_MOVEMENT;
797 _phaseSwitchTimer = 5*IN_MILLISECONDS;
798 SetCombatMovement(false);
799 me->GetMotionMaster()->MoveIdle();
800 me->GetMotionMaster()->Clear();
801 }
802 }
803
804 void StartChase(Unit* who)
805 {
806 DoCast(who, SPELL_MARK);
807 me->SetSpeed(MOVE_RUN, 0.5f);
808 // make sure the Spine will really follow the one he should
809 me->getThreatManager().clearReferences();
810 me->SetInCombatWithZone();
811 me->getThreatManager().addThreat(who, std::numeric_limits<float>::max());
812 me->GetMotionMaster()->Clear(true);
813 me->GetMotionMaster()->MoveChase(who);
814 me->TauntApply(who);
815 }
816
817 private:
818 uint32 _phaseSwitchTimer;
819 PursuingSpikesPhases _phase;
820 };

823 {
824 return new npc_anubarak_spikeAI(creature);

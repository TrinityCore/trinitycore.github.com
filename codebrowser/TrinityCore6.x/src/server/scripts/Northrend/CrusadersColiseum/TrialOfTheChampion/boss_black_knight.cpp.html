77class boss_black_knight : public CreatureScript
78{
79public:
80 boss_black_knight() : CreatureScript("boss_black_knight") { }
81
82 struct boss_black_knightAI : public ScriptedAI
83 {
84 boss_black_knightAI(Creature* creature) : ScriptedAI(creature), summons(creature)
85 {
86 Initialize();
87 instance = creature->GetInstanceScript();
88 }
89
90 void Initialize()
91 {
92 bEventInProgress = false;
93 bEvent = false;
94 bSummonArmy = false;
95 bDeathArmyDone = false;
96
97 uiPhase = PHASE_UNDEAD;
98
99 uiIcyTouchTimer = urand(5000, 9000);
100 uiPlagueStrikeTimer = urand(10000, 13000);
101 uiDeathRespiteTimer = urand(15000, 16000);
102 uiObliterateTimer = urand(17000, 19000);
103 uiDesecration = urand(15000, 16000);
104 uiDeathArmyCheckTimer = 7000;
105 uiResurrectTimer = 4000;
106 uiGhoulExplodeTimer = 8000;
107 uiDeathBiteTimer = urand(2000, 4000);
108 uiMarkedDeathTimer = urand(5000, 7000);
109 }
110
111 InstanceScript* instance;
112
113 SummonList summons;
114
115 bool bEventInProgress;
116 bool bEvent;
117 bool bSummonArmy;
118 bool bDeathArmyDone;
119
120 uint8 uiPhase;
121
122 uint32 uiPlagueStrikeTimer;
123 uint32 uiIcyTouchTimer;
124 uint32 uiDeathRespiteTimer;
125 uint32 uiObliterateTimer;
126 uint32 uiDesecration;
127 uint32 uiResurrectTimer;
128 uint32 uiDeathArmyCheckTimer;
129 uint32 uiGhoulExplodeTimer;
130 uint32 uiDeathBiteTimer;
131 uint32 uiMarkedDeathTimer;
132
133 void Reset() override
134 {
135 summons.DespawnAll();
136 me->SetDisplayId(me->GetNativeDisplayId());
137 me->ClearUnitState(UNIT_STATE_ROOT | UNIT_STATE_STUNNED);
138
139 Initialize();
140 }
141
142 void JustSummoned(Creature* summon) override
143 {
144 summons.Summon(summon);
145 summon->AI()->AttackStart(me->GetVictim());
146 }
147
148 void SummonedCreatureDespawn(Creature* summon) override
149 {
150 summons.Despawn(summon);
151 }
152
153 void UpdateAI(uint32 uiDiff) override
154 {
155 //Return since we have no target
156 if (!UpdateVictim())
157 return;
158
159 if (bEventInProgress)
160 {
161 if (uiResurrectTimer <= uiDiff)
162 {
163 me->SetFullHealth();
164 DoCast(me, SPELL_BLACK_KNIGHT_RES, true);
165 uiPhase++;
166 uiResurrectTimer = 4000;
167 bEventInProgress = false;
168 me->ClearUnitState(UNIT_STATE_ROOT | UNIT_STATE_STUNNED);
169 } else uiResurrectTimer -= uiDiff;
170 }
171
172 switch (uiPhase)
173 {
174 case PHASE_UNDEAD:
175 case PHASE_SKELETON:
176 {
177 if (uiIcyTouchTimer <= uiDiff)
178 {
179 DoCastVictim(SPELL_ICY_TOUCH);
180 uiIcyTouchTimer = urand(5000, 7000);
181 } else uiIcyTouchTimer -= uiDiff;
182 if (uiPlagueStrikeTimer <= uiDiff)
183 {
184 DoCastVictim(SPELL_ICY_TOUCH);
185 uiPlagueStrikeTimer = urand(12000, 15000);
186 } else uiPlagueStrikeTimer -= uiDiff;
187 if (uiObliterateTimer <= uiDiff)
188 {
189 DoCastVictim(SPELL_OBLITERATE);
190 uiObliterateTimer = urand(17000, 19000);
191 } else uiObliterateTimer -= uiDiff;
192 switch (uiPhase)
193 {
194 case PHASE_UNDEAD:
195 {
196 if (uiDeathRespiteTimer <= uiDiff)
197 {
198 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
199 {
200 if (target && target->IsAlive())
201 DoCast(target, SPELL_DEATH_RESPITE);
202 }
203 uiDeathRespiteTimer = urand(15000, 16000);
204 } else uiDeathRespiteTimer -= uiDiff;
205 break;
206 }
207 case PHASE_SKELETON:
208 {
209 if (!bSummonArmy)
210 {
211 bSummonArmy = true;
212 me->AddUnitState(UNIT_STATE_ROOT | UNIT_STATE_STUNNED);
213 DoCast(me, SPELL_ARMY_DEAD);
214 }
215 if (!bDeathArmyDone)
216 {
217 if (uiDeathArmyCheckTimer <= uiDiff)
218 {
219 me->ClearUnitState(UNIT_STATE_ROOT | UNIT_STATE_STUNNED);
220 uiDeathArmyCheckTimer = 0;
221 bDeathArmyDone = true;
222 } else uiDeathArmyCheckTimer -= uiDiff;
223 }
224 if (uiDesecration <= uiDiff)
225 {
226 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
227 {
228 if (target && target->IsAlive())
229 DoCast(target, SPELL_DESECRATION);
230 }
231 uiDesecration = urand(15000, 16000);
232 } else uiDesecration -= uiDiff;
233 if (uiGhoulExplodeTimer <= uiDiff)
234 {
235 DoCast(me, SPELL_GHOUL_EXPLODE);
236 uiGhoulExplodeTimer = 8000;
237 } else uiGhoulExplodeTimer -= uiDiff;
238 break;
239 }
240 break;
241 }
242 break;
243 }
244 case PHASE_GHOST:
245 {
246 if (uiDeathBiteTimer <= uiDiff)
247 {
248 DoCastAOE(SPELL_DEATH_BITE);
249 uiDeathBiteTimer = urand(2000, 4000);
250 } else uiDeathBiteTimer -= uiDiff;
251 if (uiMarkedDeathTimer <= uiDiff)
252 {
253 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
254 {
255 if (target && target->IsAlive())
256 DoCast(target, SPELL_MARKED_DEATH);
257 }
258 uiMarkedDeathTimer = urand(5000, 7000);
259 } else uiMarkedDeathTimer -= uiDiff;
260 break;
261 }
262 }
263
264 if (!me->HasUnitState(UNIT_STATE_ROOT) && !me->HealthBelowPct(1))
265 DoMeleeAttackIfReady();
266 }
267
268 void DamageTaken(Unit* /*pDoneBy*/, uint32& uiDamage) override
269 {
270 if (uiDamage > me->GetHealth() && uiPhase <= PHASE_SKELETON)
271 {
272 uiDamage = 0;
273 me->SetHealth(0);
274 me->AddUnitState(UNIT_STATE_ROOT | UNIT_STATE_STUNNED);
275 summons.DespawnAll();
276 switch (uiPhase)
277 {
278 case PHASE_UNDEAD:
279 me->SetDisplayId(MODEL_SKELETON);
280 break;
281 case PHASE_SKELETON:
282 me->SetDisplayId(MODEL_GHOST);
283 break;
284 }
285 bEventInProgress = true;
286 }
287 }
288
289 void JustDied(Unit* /*killer*/) override
290 {
291 DoCast(me, SPELL_KILL_CREDIT);
292
293 instance->SetData(BOSS_BLACK_KNIGHT, DONE);
294 }
295 };
296
297 CreatureAI* GetAI(Creature* creature) const override
298 {
299 return GetInstanceAI<boss_black_knightAI>(creature);
300 }

370 void UpdateAI(uint32 diff) override
371 {
372 if (!UpdateVictim() || !CheckInRoom())
373 return;
374
375 events.Update(diff);
376
377 if (Shutdown == RAID_MODE(TWO_SEATS, FOUR_SEATS))
378 {
379 Shutdown = 0;
380 events.ScheduleEvent(EVENT_SHUTDOWN, 4000);
381 me->RemoveAurasDueToSpell(SPELL_OVERLOAD_CIRCUIT);
382 me->InterruptNonMeleeSpells(true);
383 return;
384 }
385
386 if (me->HasUnitState(UNIT_STATE_CASTING))
387 return;
388
389 while (uint32 eventId = events.ExecuteEvent())
390 {
391 switch (eventId)
392 {
393 case EVENT_PURSUE:
394 Talk(SAY_TARGET);
395 DoCast(SPELL_PURSUED); // Will select target in spellscript
396 events.ScheduleEvent(EVENT_PURSUE, 35*IN_MILLISECONDS);
397 break;
398 case EVENT_MISSILE:
399 DoCast(me, SPELL_MISSILE_BARRAGE, true);
400 events.ScheduleEvent(EVENT_MISSILE, 2*IN_MILLISECONDS);
401 break;
402 case EVENT_VENT:
403 DoCastAOE(SPELL_FLAME_VENTS);
404 events.ScheduleEvent(EVENT_VENT, 20*IN_MILLISECONDS);
405 break;
406 case EVENT_SPEED:
407 DoCastAOE(SPELL_GATHERING_SPEED);
408 events.ScheduleEvent(EVENT_SPEED, 15*IN_MILLISECONDS);
409 break;
410 case EVENT_SUMMON:
411 if (summons.size() < 15)
412 if (Creature* lift = DoSummonFlyer(NPC_MECHANOLIFT, me, 30.0f, 50.0f, 0))
413 lift->GetMotionMaster()->MoveRandom(100);
414 events.ScheduleEvent(EVENT_SUMMON, 2*IN_MILLISECONDS);
415 break;
416 case EVENT_SHUTDOWN:
417 Talk(SAY_OVERLOAD);
418 Talk(EMOTE_OVERLOAD);
419 me->CastSpell(me, SPELL_SYSTEMS_SHUTDOWN, true);
420 if (Shutout)
421 Shutout = false;
422 events.ScheduleEvent(EVENT_REPAIR, 4000);
423 events.DelayEvents(20 * IN_MILLISECONDS, 0);
424 break;
425 case EVENT_REPAIR:
426 Talk(EMOTE_REPAIR);
427 me->ClearUnitState(UNIT_STATE_STUNNED | UNIT_STATE_ROOT);
428 events.ScheduleEvent(EVENT_SHUTDOWN, 150*IN_MILLISECONDS);
429 events.CancelEvent(EVENT_REPAIR);
430 break;
431 case EVENT_THORIM_S_HAMMER: // Tower of Storms
432 for (uint8 i = 0; i < 7; ++i)
433 {
434 if (Creature* thorim = DoSummon(NPC_THORIM_BEACON, me, float(urand(20, 60)), 20000, TEMPSUMMON_TIMED_DESPAWN))
435 thorim->GetMotionMaster()->MoveRandom(100);
436 }
437 Talk(SAY_TOWER_STORM);
438 events.CancelEvent(EVENT_THORIM_S_HAMMER);
439 break;
440 case EVENT_MIMIRON_S_INFERNO: // Tower of Flames
441 me->SummonCreature(NPC_MIMIRON_BEACON, InfernoStart);
442 Talk(SAY_TOWER_FLAME);
443 events.CancelEvent(EVENT_MIMIRON_S_INFERNO);
444 break;
445 case EVENT_HODIR_S_FURY: // Tower of Frost
446 for (uint8 i = 0; i < 7; ++i)
447 {
448 if (Creature* hodir = DoSummon(NPC_HODIR_BEACON, me, 50, 0))
449 hodir->GetMotionMaster()->MoveRandom(100);
450 }
451 Talk(SAY_TOWER_FROST);
452 events.CancelEvent(EVENT_HODIR_S_FURY);
453 break;
454 case EVENT_FREYA_S_WARD: // Tower of Nature
455 Talk(SAY_TOWER_NATURE);
456 for (int32 i = 0; i < 4; ++i)
457 me->SummonCreature(NPC_FREYA_BEACON, FreyaBeacons[i]);
458
459 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
460 DoCast(target, SPELL_FREYA_S_WARD);
461 events.CancelEvent(EVENT_FREYA_S_WARD);
462 break;
463 }
464 }

475 void DoAction(int32 action) override
476 {
477 if (action && action <= 4) // Tower destruction, debuff leviathan loot and reduce active tower count
478 {
479 if (me->HasLootMode(LOOT_MODE_DEFAULT | LOOT_MODE_HARD_MODE_1 | LOOT_MODE_HARD_MODE_2 | LOOT_MODE_HARD_MODE_3 | LOOT_MODE_HARD_MODE_4) && ActiveTowersCount == 4)
480 me->RemoveLootMode(LOOT_MODE_HARD_MODE_4);
481
482 if (me->HasLootMode(LOOT_MODE_DEFAULT | LOOT_MODE_HARD_MODE_1 | LOOT_MODE_HARD_MODE_2 | LOOT_MODE_HARD_MODE_3) && ActiveTowersCount == 3)
483 me->RemoveLootMode(LOOT_MODE_HARD_MODE_3);
484
485 if (me->HasLootMode(LOOT_MODE_DEFAULT | LOOT_MODE_HARD_MODE_1 | LOOT_MODE_HARD_MODE_2) && ActiveTowersCount == 2)
486 me->RemoveLootMode(LOOT_MODE_HARD_MODE_2);
487
488 if (me->HasLootMode(LOOT_MODE_DEFAULT | LOOT_MODE_HARD_MODE_1) && ActiveTowersCount == 1)
489 me->RemoveLootMode(LOOT_MODE_HARD_MODE_1);
490 }
491
492 switch (action)
493 {
494 case ACTION_TOWER_OF_STORM_DESTROYED:
495 if (towerOfStorms)
496 {
497 towerOfStorms = false;
498 --ActiveTowersCount;
499 }
500 break;
501 case ACTION_TOWER_OF_FROST_DESTROYED:
502 if (towerOfFrost)
503 {
504 towerOfFrost = false;
505 --ActiveTowersCount;
506 }
507 break;
508 case ACTION_TOWER_OF_FLAMES_DESTROYED:
509 if (towerOfFlames)
510 {
511 towerOfFlames = false;
512 --ActiveTowersCount;
513 }
514 break;
515 case ACTION_TOWER_OF_LIFE_DESTROYED:
516 if (towerOfLife)
517 {
518 towerOfLife = false;
519 --ActiveTowersCount;
520 }
521 break;
522 case ACTION_START_HARD_MODE: // Activate hard-mode enable all towers, apply buffs on leviathan
523 ActiveTowers = true;
524 towerOfStorms = true;
525 towerOfLife = true;
526 towerOfFlames = true;
527 towerOfFrost = true;
528 me->SetLootMode(LOOT_MODE_DEFAULT | LOOT_MODE_HARD_MODE_1 | LOOT_MODE_HARD_MODE_2 | LOOT_MODE_HARD_MODE_3 | LOOT_MODE_HARD_MODE_4);
529 break;
530 case ACTION_MOVE_TO_CENTER_POSITION: // Triggered by 2 Collossus near door
531 if (!me->isDead())
532 {
533 me->SetHomePosition(Center);
534 me->GetMotionMaster()->MoveCharge(Center.GetPositionX(), Center.GetPositionY(), Center.GetPositionZ()); // position center
535 me->SetReactState(REACT_AGGRESSIVE);
536 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_STUNNED);
537 return;
538 }
539 break;
540 default:
541 break;
542 }
543 }

548 void DoBatteringRamIfReady()

550 if (me->isAttackReady())
551 {
552 Unit* target = ObjectAccessor::GetUnit(*me, _pursueTarget);
553
554 // Pursue was unable to acquire a valid target, so get the current victim as target.
555 if (!target && me->GetVictim())
556 target = me->GetVictim();
557
558 if (me->IsWithinCombatRange(target, 30.0f))
559 {
560 DoCast(target, SPELL_BATTERING_RAM);
561 me->resetAttackTimer();
562 }
563 }
564 }

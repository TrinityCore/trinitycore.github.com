101class boss_auriaya : public CreatureScript
102{
103 public:
104 boss_auriaya() : CreatureScript("boss_auriaya") { }
105
106 struct boss_auriayaAI : public BossAI
107 {
108 boss_auriayaAI(Creature* creature) : BossAI(creature, BOSS_AURIAYA)
109 {
110 Initialize();
111 }
112
113 void Initialize()
114 {
115 DefenderGUID.Clear();
116 defenderLives = 8;
117 crazyCatLady = true;
118 nineLives = false;
119 }
120
121 void Reset() override
122 {
123 _Reset();
124 Initialize();
125 }
126
127 void EnterCombat(Unit* /*who*/) override
128 {
129 _EnterCombat();
130 Talk(SAY_AGGRO);
131
132 events.ScheduleEvent(EVENT_SCREECH, urand(45000, 65000));
133 events.ScheduleEvent(EVENT_BLAST, urand(20000, 25000));
134 events.ScheduleEvent(EVENT_TERRIFYING, urand(20000, 30000));
135 events.ScheduleEvent(EVENT_DEFENDER, urand(40000, 55000));
136 events.ScheduleEvent(EVENT_SUMMON, urand(45000, 55000));
137 events.ScheduleEvent(EVENT_BERSERK, 600000);
138 }
139
140 void KilledUnit(Unit* who) override
141 {
142 if (who->GetTypeId() == TYPEID_PLAYER)
143 Talk(SAY_SLAY);
144 }
145
146 void JustSummoned(Creature* summoned) override
147 {
148 summons.Summon(summoned);
149
150 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
151 {
152 summoned->AI()->AttackStart(target);
153 summoned->AddThreat(target, 250.0f);
154 DoZoneInCombat(summoned);
155 }
156
157 if (summoned->GetEntry() == NPC_FERAL_DEFENDER)
158 {
159 if (!summoned->IsInCombat() && me->GetVictim())
160 summoned->AI()->AttackStart(me->GetVictim());
161 summoned->SetAuraStack(SPELL_FERAL_ESSENCE, summoned, 9);
162 DefenderGUID = summoned->GetGUID();
163 DoZoneInCombat(summoned);
164 }
165 }
166
167 void DoAction(int32 action) override
168 {
169 switch (action)
170 {
171 case ACTION_CRAZY_CAT_LADY:
172 SetData(DATA_CRAZY_CAT_LADY, 0);
173 break;
174 case ACTION_RESPAWN_DEFENDER:
175 --defenderLives;
176 if (!defenderLives)
177 {
178 SetData(DATA_NINE_LIVES, 1);
179 break;
180 }
181 events.ScheduleEvent(EVENT_RESPAWN_DEFENDER, 30000);
182 break;
183 default:
184 break;
185 }
186 }
187
188 uint32 GetData(uint32 type) const override
189 {
190 switch (type)
191 {
192 case DATA_NINE_LIVES:
193 return nineLives ? 1 : 0;
194 case DATA_CRAZY_CAT_LADY:
195 return crazyCatLady ? 1 : 0;
196 }
197
198 return 0;
199 }
200
201 void SetData(uint32 id, uint32 data) override
202 {
203 switch (id)
204 {
205 case DATA_NINE_LIVES:
206 nineLives = data ? true : false;
207 break;
208 case DATA_CRAZY_CAT_LADY:
209 crazyCatLady = data ? true : false;
210 break;
211 }
212 }
213
214 void JustDied(Unit* /*killer*/) override
215 {
216 _JustDied();
217 Talk(SAY_DEATH);
218 }
219
220 void UpdateAI(uint32 diff) override
221 {
222 if (!UpdateVictim())
223 return;
224
225 events.Update(diff);
226
227 if (me->HasUnitState(UNIT_STATE_CASTING))
228 return;
229
230 while (uint32 eventId = events.ExecuteEvent())
231 {
232 switch (eventId)
233 {
234 case EVENT_SCREECH:
235 DoCast(SPELL_SONIC_SCREECH);
236 events.ScheduleEvent(EVENT_SCREECH, urand(40000, 60000));
237 break;
238 case EVENT_TERRIFYING:
239 Talk(EMOTE_FEAR);
240 DoCast(SPELL_TERRIFYING_SCREECH);
241 events.ScheduleEvent(EVENT_TERRIFYING, urand(20000, 30000));
242 break;
243 case EVENT_BLAST:
244 DoCastAOE(SPELL_SENTINEL_BLAST);
245 events.ScheduleEvent(EVENT_BLAST, urand(25000, 35000));
246 break;
247 case EVENT_DEFENDER:
248 Talk(EMOTE_DEFENDER);
249 DoCast(SPELL_DEFENDER_TRIGGER);
250 if (Creature* trigger = me->FindNearestCreature(NPC_FERAL_DEFENDER_TRIGGER, 15.0f, true))
251 DoCast(trigger, SPELL_ACTIVATE_DEFENDER, true);
252 break;
253 case EVENT_RESPAWN_DEFENDER:
254 if (Creature* Defender = ObjectAccessor::GetCreature(*me, DefenderGUID))
255 {
256 Defender->Respawn();
257 if (defenderLives)
258 Defender->SetAuraStack(SPELL_FERAL_ESSENCE, Defender, defenderLives);
259 Defender->SetInCombatWithZone();
260 if (!Defender->IsInCombat())
261 Defender->AI()->AttackStart(me->GetVictim());
262 events.CancelEvent(EVENT_RESPAWN_DEFENDER);
263 }
264 break;
265 case EVENT_SUMMON:
266 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
267 DoCast(target, SPELL_SUMMON_SWARMING_GUARDIAN);
268 events.ScheduleEvent(EVENT_SUMMON, urand(30000, 45000));
269 break;
270 case EVENT_BERSERK:
271 DoCast(me, SPELL_BERSERK, true);
272 Talk(SAY_BERSERK);
273 events.CancelEvent(EVENT_BERSERK);
274 break;
275 }
276 }
277
278 DoMeleeAttackIfReady();
279 }
280
281 private:
282 ObjectGuid DefenderGUID;
283 uint8 defenderLives;
284 bool crazyCatLady;
285 bool nineLives;
286 };
287
288 CreatureAI* GetAI(Creature* creature) const override
289 {
290 return GetUlduarAI<boss_auriayaAI>(creature);
291 }

329{
330 public:
331 npc_sanctum_sentry() : CreatureScript("npc_sanctum_sentry") { }
332
333 struct npc_sanctum_sentryAI : public ScriptedAI
334 {
335 npc_sanctum_sentryAI(Creature* creature) : ScriptedAI(creature)
336 {
337 instance = me->GetInstanceScript();
338 }
339
340 void Reset() override
341 {
342 events.ScheduleEvent(EVENT_RIP, urand(4000, 8000));
343 events.ScheduleEvent(EVENT_POUNCE, urand(12000, 15000));
344 }
345
346 void EnterCombat(Unit* /*who*/) override
347 {
348 DoCast(me, SPELL_STRENGHT_PACK, true);
349 }
350
351 void UpdateAI(uint32 diff) override
352 {
353 if (!UpdateVictim())
354 return;
355
356 events.Update(diff);
357
358 if (me->HasUnitState(UNIT_STATE_CASTING))
359 return;
360
361 while (uint32 eventId = events.ExecuteEvent())
362 {
363 switch (eventId)
364 {
365 case EVENT_RIP:
366 DoCastVictim(SPELL_RIP_FLESH);
367 events.ScheduleEvent(EVENT_RIP, urand(12000, 15000));
368 break;
369 case EVENT_POUNCE:
370 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
371 {
372 me->AddThreat(target, 100.0f);
373 AttackStart(target);
374 DoCast(target, SPELL_SAVAGE_POUNCE);
375 }
376 events.ScheduleEvent(EVENT_POUNCE, urand(12000, 17000));
377 break;
378 default:
379 break;
380 }
381 }
382
383 DoMeleeAttackIfReady();
384 }
385
386 void JustDied(Unit* /*killer*/) override
387 {
388 if (Creature* Auriaya = ObjectAccessor::GetCreature(*me, instance->GetGuidData(BOSS_AURIAYA)))
389 Auriaya->AI()->DoAction(ACTION_CRAZY_CAT_LADY);
390 }
391
392 private:
393 InstanceScript* instance;
394 EventMap events;
395 };
396
397 CreatureAI* GetAI(Creature* creature) const override
398 {
399 return GetInstanceAI<npc_sanctum_sentryAI>(creature);
400 }

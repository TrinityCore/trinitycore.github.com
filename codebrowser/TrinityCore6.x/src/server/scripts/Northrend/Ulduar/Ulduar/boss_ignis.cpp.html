113class boss_ignis : public CreatureScript
114{
115 public:
116 boss_ignis() : CreatureScript("boss_ignis") { }
117
118 struct boss_ignis_AI : public BossAI
119 {
120 boss_ignis_AI(Creature* creature) : BossAI(creature, BOSS_IGNIS)
121 {
122 Initialize();
123 }
124
125 void Initialize()
126 {
127 _slagPotGUID.Clear();
128 _shattered = false;
129 _firstConstructKill = 0;
130 }
131
132 void Reset() override
133 {
134 _Reset();
135 if (Vehicle* _vehicle = me->GetVehicleKit())
136 _vehicle->RemoveAllPassengers();
137
138 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEVEMENT_IGNIS_START_EVENT);
139 }
140
141 void EnterCombat(Unit* /*who*/) override
142 {
143 _EnterCombat();
144 Talk(SAY_AGGRO);
145 events.ScheduleEvent(EVENT_JET, 30000);
146 events.ScheduleEvent(EVENT_SCORCH, 25000);
147 events.ScheduleEvent(EVENT_SLAG_POT, 35000);
148 events.ScheduleEvent(EVENT_CONSTRUCT, 15000);
149 events.ScheduleEvent(EVENT_END_POT, 40000);
150 events.ScheduleEvent(EVENT_BERSERK, 480000);
151 Initialize();
152 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEVEMENT_IGNIS_START_EVENT);
153 }
154
155 void JustDied(Unit* /*killer*/) override
156 {
157 _JustDied();
158 Talk(SAY_DEATH);
159 }
160
161 uint32 GetData(uint32 type) const override
162 {
163 if (type == DATA_SHATTERED)
164 return _shattered ? 1 : 0;
165
166 return 0;
167 }
168
169 void KilledUnit(Unit* who) override
170 {
171 if (who->GetTypeId() == TYPEID_PLAYER)
172 Talk(SAY_SLAY);
173 }
174
175 void JustSummoned(Creature* summon) override
176 {
177 if (summon->GetEntry() == NPC_IRON_CONSTRUCT)
178 {
179 summon->setFaction(16);
180 summon->SetReactState(REACT_AGGRESSIVE);
181 summon->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_PACIFIED | UNIT_FLAG_STUNNED | UNIT_FLAG_DISABLE_MOVE);
182 }
183
184 summon->AI()->AttackStart(me->GetVictim());
185 summon->AI()->DoZoneInCombat();
186 summons.Summon(summon);
187 }
188
189 void DoAction(int32 action) override
190 {
191 if (action != ACTION_REMOVE_BUFF)
192 return;
193
194 me->RemoveAuraFromStack(SPELL_STRENGHT);
195 // Shattered Achievement
196 time_t secondKill = sWorld->GetGameTime();
197 if ((secondKill - _firstConstructKill) < 5)
198 _shattered = true;
199 _firstConstructKill = secondKill;
200 }
201
202 void UpdateAI(uint32 diff) override
203 {
204 if (!UpdateVictim())
205 return;
206
207 events.Update(diff);
208
209 if (me->HasUnitState(UNIT_STATE_CASTING))
210 return;
211
212 while (uint32 eventId = events.ExecuteEvent())
213 {
214 switch (eventId)
215 {
216 case EVENT_JET:
217 Talk(EMOTE_JETS);
218 DoCast(me, SPELL_FLAME_JETS);
219 events.ScheduleEvent(EVENT_JET, urand(35000, 40000));
220 break;
221 case EVENT_SLAG_POT:
222 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true))
223 {
224 Talk(SAY_SLAG_POT);
225 _slagPotGUID = target->GetGUID();
226 DoCast(target, SPELL_GRAB);
227 events.DelayEvents(3000);
228 events.ScheduleEvent(EVENT_GRAB_POT, 500);
229 }
230 events.ScheduleEvent(EVENT_SLAG_POT, RAID_MODE(30000, 15000));
231 break;
232 case EVENT_GRAB_POT:
233 if (Unit* slagPotTarget = ObjectAccessor::GetUnit(*me, _slagPotGUID))
234 {
235 slagPotTarget->EnterVehicle(me, 0);
236 events.CancelEvent(EVENT_GRAB_POT);
237 events.ScheduleEvent(EVENT_CHANGE_POT, 1000);
238 }
239 break;
240 case EVENT_CHANGE_POT:
241 if (Unit* slagPotTarget = ObjectAccessor::GetUnit(*me, _slagPotGUID))
242 {
243 DoCast(slagPotTarget, SPELL_SLAG_POT, true);
244 slagPotTarget->EnterVehicle(me, 1);
245 events.CancelEvent(EVENT_CHANGE_POT);
246 events.ScheduleEvent(EVENT_END_POT, 10000);
247 }
248 break;
249 case EVENT_END_POT:
250 if (Unit* slagPotTarget = ObjectAccessor::GetUnit(*me, _slagPotGUID))
251 {
252 slagPotTarget->ExitVehicle();
253 slagPotTarget = NULL;
254 _slagPotGUID.Clear();
255 events.CancelEvent(EVENT_END_POT);
256 }
257 break;
258 case EVENT_SCORCH:
259 Talk(SAY_SCORCH);
260 if (Unit* target = me->GetVictim())
261 me->SummonCreature(NPC_GROUND_SCORCH, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 45000);
262 DoCast(SPELL_SCORCH);
263 events.ScheduleEvent(EVENT_SCORCH, 25000);
264 break;
265 case EVENT_CONSTRUCT:
266 Talk(SAY_SUMMON);
267 DoSummon(NPC_IRON_CONSTRUCT, ConstructSpawnPosition[urand(0, CONSTRUCT_SPAWN_POINTS - 1)], 30000, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT);
268 DoCast(SPELL_STRENGHT);
269 DoCast(me, SPELL_ACTIVATE_CONSTRUCT);
270 events.ScheduleEvent(EVENT_CONSTRUCT, RAID_MODE(40000, 30000));
271 break;
272 case EVENT_BERSERK:
273 DoCast(me, SPELL_BERSERK, true);
274 Talk(SAY_BERSERK);
275 break;
276 }
277 }
278
279 DoMeleeAttackIfReady();
280
281 EnterEvadeIfOutOfCombatArea(diff);
282 }
283
284 private:
285 ObjectGuid _slagPotGUID;
286 time_t _firstConstructKill;
287 bool _shattered;
288
289 };
290
291 CreatureAI* GetAI(Creature* creature) const override
292 {
293 return GetUlduarAI<boss_ignis_AI>(creature);
294 }

297class npc_iron_construct : public CreatureScript
298{
299 public:
300 npc_iron_construct() : CreatureScript("npc_iron_construct") { }
301
302 struct npc_iron_constructAI : public ScriptedAI
303 {
304 npc_iron_constructAI(Creature* creature) : ScriptedAI(creature), _instance(creature->GetInstanceScript())
305 {
306 Initialize();
307 creature->SetReactState(REACT_PASSIVE);
308 }
309
310 void Initialize()
311 {
312 _brittled = false;
313 }
314
315 void Reset() override
316 {
317 Initialize();
318 }
319
320 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
321 {
322 if (me->HasAura(RAID_MODE(SPELL_BRITTLE, SPELL_BRITTLE_25)) && damage >= 5000)
323 {
324 DoCast(SPELL_SHATTER);
325 if (Creature* ignis = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(BOSS_IGNIS)))
326 if (ignis->AI())
327 ignis->AI()->DoAction(ACTION_REMOVE_BUFF);
328
329 me->DespawnOrUnsummon(1000);
330 }
331 }
332
333 void UpdateAI(uint32 /*uiDiff*/) override
334 {
335 if (!UpdateVictim())
336 return;
337
338 if (Aura* aur = me->GetAura(SPELL_HEAT))
339 {
340 if (aur->GetStackAmount() >= 10)
341 {
342 me->RemoveAura(SPELL_HEAT);
343 DoCast(SPELL_MOLTEN);
344 _brittled = false;
345 }
346 }
347
348 // Water pools
349 if (me->IsInWater() && !_brittled && me->HasAura(SPELL_MOLTEN))
350 {
351 DoCast(SPELL_BRITTLE);
352 me->RemoveAura(SPELL_MOLTEN);
353 _brittled = true;
354 }
355
356 DoMeleeAttackIfReady();
357 }
358
359 private:
360 InstanceScript* _instance;
361 bool _brittled;
362 };
363
364 CreatureAI* GetAI(Creature* creature) const override
365 {
366 return GetUlduarAI<npc_iron_constructAI>(creature);
367 }

370class npc_scorch_ground : public CreatureScript
371{
372 public:
373 npc_scorch_ground() : CreatureScript("npc_scorch_ground") { }
374
375 struct npc_scorch_groundAI : public ScriptedAI
376 {
377 npc_scorch_groundAI(Creature* creature) : ScriptedAI(creature)
378 {
379 Initialize();
380 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE |UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_PACIFIED);
381 creature->SetDisplayId(16925); //model 2 in db cannot overwrite wdb fields
382 }
383
384 void Initialize()
385 {
386 _heat = false;
387 _constructGUID.Clear();
388 _heatTimer = 0;
389 }
390
391 void MoveInLineOfSight(Unit* who) override
392 {
393 if (!_heat)
394 {
395 if (who->GetEntry() == NPC_IRON_CONSTRUCT)
396 {
397 if (!who->HasAura(SPELL_HEAT) || !who->HasAura(SPELL_MOLTEN))
398 {
399 _constructGUID = who->GetGUID();
400 _heat = true;
401 }
402 }
403 }
404 }
405
406 void Reset() override
407 {
408 Initialize();
409 DoCast(me, SPELL_GROUND);
410 }
411
412 void UpdateAI(uint32 uiDiff) override
413 {
414 if (_heat)
415 {
416 if (_heatTimer <= uiDiff)
417 {
418 Creature* construct = ObjectAccessor::GetCreature(*me, _constructGUID);
419 if (construct && !construct->HasAura(SPELL_MOLTEN))
420 {
421 me->AddAura(SPELL_HEAT, construct);
422 _heatTimer = 1000;
423 }
424 }
425 else
426 _heatTimer -= uiDiff;
427 }
428 }
429
430 private:
431 ObjectGuid _constructGUID;
432 uint32 _heatTimer;
433 bool _heat;
434 };
435
436 CreatureAI* GetAI(Creature* creature) const override
437 {
438 return GetUlduarAI<npc_scorch_groundAI>(creature);
439 }

442class spell_ignis_slag_pot : public SpellScriptLoader
443{
444 public:
445 spell_ignis_slag_pot() : SpellScriptLoader("spell_ignis_slag_pot") { }
446
447 class spell_ignis_slag_pot_AuraScript : public AuraScript
448 {
449 PrepareAuraScript(spell_ignis_slag_pot_AuraScript);
450
451 bool Validate(SpellInfo const* /*spellInfo*/) override
452 {
453 if (!sSpellMgr->GetSpellInfo(SPELL_SLAG_POT_DAMAGE)
454 || !sSpellMgr->GetSpellInfo(SPELL_SLAG_IMBUED))
455 return false;
456 return true;
457 }
458
459 void HandleEffectPeriodic(AuraEffect const* /*aurEff*/)
460 {
461 if (Unit* caster = GetCaster())
462 {
463 Unit* target = GetTarget();
464 caster->CastSpell(target, SPELL_SLAG_POT_DAMAGE, true);
465 }
466 }
467
468 void OnRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
469 {
470 if (GetTarget()->IsAlive())
471 GetTarget()->CastSpell(GetTarget(), SPELL_SLAG_IMBUED, true);
472 }
473
474 void Register() override
475 {
476 OnEffectPeriodic += AuraEffectPeriodicFn(spell_ignis_slag_pot_AuraScript::HandleEffectPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
477 AfterEffectRemove += AuraEffectRemoveFn(spell_ignis_slag_pot_AuraScript::OnRemove, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY, AURA_EFFECT_HANDLE_REAL);
478 }
479 };
480
481 AuraScript* GetAuraScript() const override
482 {
483 return new spell_ignis_slag_pot_AuraScript();
484 }

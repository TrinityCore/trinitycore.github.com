134class boss_steelbreaker : public CreatureScript
135{
136 public:
137 boss_steelbreaker() : CreatureScript("boss_steelbreaker") { }
138
139 struct boss_steelbreakerAI : public BossAI
140 {
141 boss_steelbreakerAI(Creature* creature) : BossAI(creature, BOSS_ASSEMBLY_OF_IRON)
142 {
143 Initialize();
144 }
145
146 void Initialize()
147 {
148 phase = 0;
149 }
150
151 uint32 phase;
152
153 void Reset() override
154 {
155 _Reset();
156 Initialize();
157 me->RemoveAllAuras();
158 }
159
160 void EnterCombat(Unit* /*who*/) override
161 {
162 _EnterCombat();
163 Talk(SAY_STEELBREAKER_AGGRO);
164 DoCast(me, SPELL_HIGH_VOLTAGE);
165 events.SetPhase(++phase);
166 events.ScheduleEvent(EVENT_BERSERK, 900000);
167 events.ScheduleEvent(EVENT_FUSION_PUNCH, 15000);
168 }
169
170 uint32 GetData(uint32 type) const override
171 {
172 if (type == DATA_PHASE_3)
173 return (phase >= 3) ? 1 : 0;
174
175 return 0;
176 }
177
178 void DoAction(int32 action) override
179 {
180 switch (action)
181 {
182 case ACTION_SUPERCHARGE:
183 me->SetFullHealth();
184 me->AddAura(SPELL_SUPERCHARGE, me);
185 events.SetPhase(++phase);
186 events.RescheduleEvent(EVENT_FUSION_PUNCH, 15000);
187 if (phase >= 2)
188 events.RescheduleEvent(EVENT_STATIC_DISRUPTION, 30000);
189 if (phase >= 3)
190 events.RescheduleEvent(EVENT_OVERWHELMING_POWER, urand(2000, 5000));
191 break;
192 case ACTION_ADD_CHARGE:
193 DoCast(me, SPELL_ELECTRICAL_CHARGE, true);
194 break;
195 }
196 }
197
198 void JustDied(Unit* /*killer*/) override
199 {
200 _JustDied();
201
202 if (instance->GetBossState(BOSS_ASSEMBLY_OF_IRON) == DONE)
203 {
204 DoCastAOE(SPELL_KILL_CREDIT, true);
205 Talk(SAY_STEELBREAKER_ENCOUNTER_DEFEATED);
206 }
207 else
208 {
209 me->SetLootRecipient(NULL);
210 Talk(SAY_STEELBREAKER_DEATH);
211 //DoCastAOE(SPELL_SUPERCHARGE, true);
212
213 if (Creature* Brundir = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BRUNDIR)))
214 if (Brundir->IsAlive())
215 Brundir->AI()->DoAction(ACTION_SUPERCHARGE);
216
217 if (Creature* Molgeim = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MOLGEIM)))
218 if (Molgeim->IsAlive())
219 Molgeim->AI()->DoAction(ACTION_SUPERCHARGE);
220 }
221 }
222
223 void KilledUnit(Unit* who) override
224 {
225 if (who->GetTypeId() == TYPEID_PLAYER)
226 Talk(SAY_STEELBREAKER_SLAY);
227
228 if (phase == 3)
229 DoCast(me, SPELL_ELECTRICAL_CHARGE);
230 }
231
232 void UpdateAI(uint32 diff) override
233 {
234 if (!UpdateVictim())
235 return;
236
237 events.Update(diff);
238
239 if (me->HasUnitState(UNIT_STATE_CASTING))
240 return;
241
242 while (uint32 eventId = events.ExecuteEvent())
243 {
244 switch (eventId)
245 {
246 case EVENT_BERSERK:
247 Talk(SAY_STEELBREAKER_BERSERK);
248 DoCast(SPELL_BERSERK);
249 events.CancelEvent(EVENT_BERSERK);
250 break;
251 case EVENT_FUSION_PUNCH:
252 if (me->IsWithinMeleeRange(me->GetVictim()))
253 DoCastVictim(SPELL_FUSION_PUNCH);
254 events.ScheduleEvent(EVENT_FUSION_PUNCH, urand(13000, 22000));
255 break;
256 case EVENT_STATIC_DISRUPTION:
257 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
258 DoCast(target, SPELL_STATIC_DISRUPTION);
259 events.ScheduleEvent(EVENT_STATIC_DISRUPTION, urand(20000, 40000));
260 break;
261 case EVENT_OVERWHELMING_POWER:
262 Talk(SAY_STEELBREAKER_POWER);
263 DoCastVictim(SPELL_OVERWHELMING_POWER);
264 events.ScheduleEvent(EVENT_OVERWHELMING_POWER, RAID_MODE(60000, 35000));
265 break;
266 }
267 }
268
269 DoMeleeAttackIfReady();
270 }
271 };
272
273 CreatureAI* GetAI(Creature* creature) const override
274 {
275 return GetUlduarAI<boss_steelbreakerAI>(creature);
276 }

583 events.ScheduleEvent(EVENT_LIGHTNING_WHIRL, urand(15000, 20000));
584 break;

588 DoCast(me, SPELL_LIGHTNING_TENDRILS_VISUAL);

590 //me->SetLevitate(true);
591 me->GetMotionMaster()->Initialize();
592 me->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), FINAL_FLIGHT_Z);
593 events.DelayEvents(35000);
594 events.ScheduleEvent(EVENT_FLIGHT, 2500);
595 events.ScheduleEvent(EVENT_ENDFLIGHT, 32500);
596 events.ScheduleEvent(EVENT_LIGHTNING_TENDRILS, 90000);
597 break;
598 case EVENT_FLIGHT:

600 me->GetMotionMaster()->MovePoint(0, target->GetPositionX(), target->GetPositionY(), FINAL_FLIGHT_Z);
601 events.ScheduleEvent(EVENT_FLIGHT, 6000);
602 break;
603 case EVENT_ENDFLIGHT:
604 me->GetMotionMaster()->Initialize();
605 me->GetMotionMaster()->MovePoint(0, 1586.920166f, 119.848984f, FINAL_FLIGHT_Z);
606 events.CancelEvent(EVENT_FLIGHT);
607 events.CancelEvent(EVENT_ENDFLIGHT);
608 events.ScheduleEvent(EVENT_LAND, 4000);
609 break;
610 case EVENT_LAND:
611 me->GetMotionMaster()->Initialize();
612 me->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), FLOOR_Z);
613 events.CancelEvent(EVENT_LAND);
614 events.ScheduleEvent(EVENT_GROUND, 2500);
615 break;
616 case EVENT_GROUND:
617 //me->SetLevitate(false);
618 me->RemoveAurasDueToSpell(SPELL_LIGHTNING_TENDRILS);
619 me->RemoveAurasDueToSpell(SPELL_LIGHTNING_TENDRILS_VISUAL);
620 DoStartMovement(me->GetVictim());
621 events.CancelEvent(EVENT_GROUND);
622 me->getThreatManager().resetAllAggro();
623 break;
624 case EVENT_MOVE_POSITION:
625 if (me->IsWithinMeleeRange(me->GetVictim()))
626 {
627 float x = float(irand(-25, 25));
628 float y = float(irand(-25, 25));
629 me->GetMotionMaster()->MovePoint(0, me->GetPositionX() + x, me->GetPositionY() + y, FLOOR_Z);
630 // Prevention to go outside the room or into the walls
631 if (Creature* trigger = me->FindNearestCreature(NPC_WORLD_TRIGGER, 100.0f, true))
632 if (me->GetDistance(trigger) >= 50.0f)
633 me->GetMotionMaster()->MovePoint(0, trigger->GetPositionX(), trigger->GetPositionY(), FLOOR_Z);
634 }
635 events.ScheduleEvent(EVENT_MOVE_POSITION, urand(7500, 10000));
636 break;
637 default:
638 break;
639 }
640 }
641
642 DoMeleeAttackIfReady();
643 }
644 };

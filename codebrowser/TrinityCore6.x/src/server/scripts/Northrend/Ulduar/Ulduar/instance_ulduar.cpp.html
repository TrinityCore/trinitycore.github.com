163 void FillInitialWorldStates(WorldPackets::WorldState::InitWorldStates& packet) override
164 {
165 packet.Worldstates.emplace_back(uint32(WORLD_STATE_ALGALON_TIMER_ENABLED), int32(_algalonTimer && _algalonTimer <= 60));
166 packet.Worldstates.emplace_back(uint32(WORLD_STATE_ALGALON_DESPAWN_TIMER), int32(std::min<uint32>(_algalonTimer, 60)));
167 }

169 void OnPlayerEnter(Player* player) override
170 {
171 if (!TeamInInstance)
172 TeamInInstance = player->GetTeam();
173
174 if (_summonAlgalon)
175 {
176 _summonAlgalon = false;
177 TempSummon* algalon = instance->SummonCreature(NPC_ALGALON, AlgalonLandPos);
178 if (_algalonTimer && _algalonTimer <= 60)
179 algalon->AI()->DoAction(ACTION_INIT_ALGALON);
180 else
181 algalon->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
182 }
183
184 // Keepers at Observation Ring
185 if (GetBossState(BOSS_FREYA) == DONE && _summonObservationRingKeeper[0] && !KeeperGUIDs[0])
186 {
187 _summonObservationRingKeeper[0] = false;
188 instance->SummonCreature(NPC_FREYA_OBSERVATION_RING, ObservationRingKeepersPos[0]);
189 }
190 if (GetBossState(BOSS_HODIR) == DONE && _summonObservationRingKeeper[1] && !KeeperGUIDs[1])
191 {
192 _summonObservationRingKeeper[1] = false;
193 instance->SummonCreature(NPC_HODIR_OBSERVATION_RING, ObservationRingKeepersPos[1]);
194 }
195 if (GetBossState(BOSS_THORIM) == DONE && _summonObservationRingKeeper[2] && !KeeperGUIDs[2])
196 {
197 _summonObservationRingKeeper[2] = false;
198 instance->SummonCreature(NPC_THORIM_OBSERVATION_RING, ObservationRingKeepersPos[2]);
199 }
200 if (GetBossState(BOSS_MIMIRON) == DONE && _summonObservationRingKeeper[3] && !KeeperGUIDs[3])
201 {
202 _summonObservationRingKeeper[3] = false;
203 instance->SummonCreature(NPC_MIMIRON_OBSERVATION_RING, ObservationRingKeepersPos[3]);
204 }
205
206 // Keepers in Yogg-Saron's room
207 if (_summonYSKeeper[0])
208 instance->SummonCreature(NPC_FREYA_YS, YSKeepersPos[0]);
209 if (_summonYSKeeper[1])
210 instance->SummonCreature(NPC_HODIR_YS, YSKeepersPos[1]);
211 if (_summonYSKeeper[2])
212 instance->SummonCreature(NPC_THORIM_YS, YSKeepersPos[2]);
213 if (_summonYSKeeper[3])
214 instance->SummonCreature(NPC_MIMIRON_YS, YSKeepersPos[3]);
215 }

221 Map::PlayerList const& Players = instance->GetPlayers();

228 {
229 case NPC_LEVIATHAN:

232 case NPC_SALVAGED_DEMOLISHER:
233 case NPC_SALVAGED_SIEGE_ENGINE:
234 case NPC_SALVAGED_CHOPPER:
235 LeviathanVehicleGUIDs.push_back(creature->GetGUID());
236 break;
237 case NPC_IGNIS:

241 // Razorscale

244 break;
245 case NPC_RAZORSCALE_CONTROLLER:

247 break;
248 case NPC_EXPEDITION_COMMANDER:
249 ExpeditionCommanderGUID = creature->GetGUID();
250 break;

252 // XT-002 Deconstructor

255 break;
256 case NPC_XT_TOY_PILE:
257 for (uint8 i = 0; i < 4; ++i)

261 break;

263 break;
264
265 // Assembly of Iron

437 void OnCreatureRemove(Creature* creature) override
438 {
439 switch (creature->GetEntry())
440 {
441 case NPC_XT_TOY_PILE:
442 for (uint8 i = 0; i < 4; ++i)
443 if (XTToyPileGUIDs[i] == creature->GetGUID())
444 {
445 XTToyPileGUIDs[i].Clear();
446 break;
447 }
448 break;
449 case NPC_STEELBREAKER:
450 case NPC_MOLGEIM:
451 case NPC_BRUNDIR:
452 AddMinion(creature, false);
453 break;
454 case NPC_BRANN_BRONZBEARD_ALG:
455 if (BrannBronzebeardAlgGUID == creature->GetGUID())
456 BrannBronzebeardAlgGUID.Clear();
457 break;
458 default:
459 break;
460 }
461
462 InstanceScript::OnCreatureRemove(creature);
463 }

615 void OnUnitDeath(Unit* unit) override
616 {
617 // Champion/Conqueror of Ulduar
618 if (unit->GetTypeId() == TYPEID_PLAYER)
619 {
620 for (uint8 i = 0; i < BOSS_ALGALON; i++)
621 {

646 DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, CRITERIA_CON_SPEED_ATORY);
647 conSpeedAtory = true;
648 }

651 case NPC_STONEBARK:
652 case NPC_BRIGHTLEAF:
653 if (!lumberjacked)
654 {
655 DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, CRITERIA_LUMBERJACKED);
656 lumberjacked = true;
657 }
658 break;
659 default:
660 break;
661 }

741 }
742 break;
743 case BOSS_HODIR:
744 if (state == DONE)
745 {
746 if (GameObject* HodirRareCache = instance->GetGameObject(HodirRareCacheGUID))
747 if (GetData(DATA_HODIR_RARE_CACHE))
748 HodirRareCache->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);

750 HodirChest->SetRespawnTime(HodirChest->GetRespawnDelay());
751
752 instance->SummonCreature(NPC_HODIR_OBSERVATION_RING, ObservationRingKeepersPos[1]);
753 }
754 break;
755 case BOSS_THORIM:
756 if (state == DONE)
757 {
758 if (GameObject* gameObject = instance->GetGameObject(ThorimChestGUID))
759 gameObject->SetRespawnTime(gameObject->GetRespawnDelay());
760
761 instance->SummonCreature(NPC_THORIM_OBSERVATION_RING, ObservationRingKeepersPos[2]);
762 }
763 break;
764 case BOSS_ALGALON:
765 if (state == DONE)
766 {
767 _events.CancelEvent(EVENT_UPDATE_ALGALON_TIMER);
768 _events.CancelEvent(EVENT_DESPAWN_ALGALON);
769 DoUpdateWorldState(WORLD_STATE_ALGALON_TIMER_ENABLED, 0);
770 _algalonTimer = 61;
771 if (GameObject* gameObject = instance->GetGameObject(GiftOfTheObserverGUID))
772 gameObject->SetRespawnTime(gameObject->GetRespawnDelay());
773 // get item level (recheck weapons)
774 Map::PlayerList const& players = instance->GetPlayers();
775 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
776 if (Player* player = itr->GetSource())
777 for (uint8 slot = EQUIPMENT_SLOT_MAINHAND; slot <= EQUIPMENT_SLOT_OFFHAND; ++slot)
778 if (Item* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, slot))
779 if (item->GetItemLevel(player) > _maxWeaponItemLevel)
780 _maxWeaponItemLevel = item->GetItemLevel(player);
781 }
782 else if (state == IN_PROGRESS)
783 {
784 // get item level (armor cannot be swapped in combat)
785 Map::PlayerList const& players = instance->GetPlayers();
786 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
787 {
788 if (Player* player = itr->GetSource())
789 {
790 for (uint8 slot = EQUIPMENT_SLOT_START; slot < EQUIPMENT_SLOT_END; ++slot)
791 {
792 if (slot == EQUIPMENT_SLOT_TABARD || slot == EQUIPMENT_SLOT_BODY)
793 continue;
794
795 if (Item* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, slot))
796 {
797 if (slot >= EQUIPMENT_SLOT_MAINHAND && slot <= EQUIPMENT_SLOT_OFFHAND)
798 {
799 if (item->GetItemLevel(player) > _maxWeaponItemLevel)
800 _maxWeaponItemLevel = item->GetItemLevel(player);
801 }
802 else if (item->GetItemLevel(player) > _maxArmorItemLevel)
803 _maxArmorItemLevel = item->GetItemLevel(player);
804 }
805 }
806 }
807 }
808 }
809 break;
810 }
811
812 return true;

1142 void Update(uint32 diff) override
1143 {
1144 if (_events.Empty())
1145 return;
1146
1147 _events.Update(diff);
1148
1149 while (uint32 eventId = _events.ExecuteEvent())
1150 {
1151 switch (eventId)
1152 {
1153 case EVENT_UPDATE_ALGALON_TIMER:
1154 SaveToDB();
1155 DoUpdateWorldState(WORLD_STATE_ALGALON_DESPAWN_TIMER, --_algalonTimer);
1156 if (_algalonTimer)
1157 _events.ScheduleEvent(EVENT_UPDATE_ALGALON_TIMER, 60000);
1158 else
1159 {
1160 DoUpdateWorldState(WORLD_STATE_ALGALON_TIMER_ENABLED, 0);
1161 _events.CancelEvent(EVENT_UPDATE_ALGALON_TIMER);
1162 if (Creature* algalon = instance->GetCreature(AlgalonGUID))
1163 algalon->AI()->DoAction(EVENT_DESPAWN_ALGALON);
1164 }
1165 break;
1166 case EVENT_DESPAWN_LEVIATHAN_VEHICLES:
1167 // Eject all players from vehicles and make them untargetable.
1168 // They will be despawned after a while
1169 for (auto const& vehicleGuid : LeviathanVehicleGUIDs)
1170 {
1171 if (Creature* vehicleCreature = instance->GetCreature(vehicleGuid))
1172 {
1173 if (Vehicle* vehicle = vehicleCreature->GetVehicleKit())
1174 {
1175 vehicle->RemoveAllPassengers();
1176 vehicleCreature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
1177 vehicleCreature->DespawnOrUnsummon(5 * MINUTE * IN_MILLISECONDS);
1178 }
1179 }
1180 }
1181 break;
1182 }
1183 }

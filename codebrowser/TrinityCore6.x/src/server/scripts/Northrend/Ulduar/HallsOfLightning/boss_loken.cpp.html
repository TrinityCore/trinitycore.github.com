78class boss_loken : public CreatureScript
79{
80public:
81 boss_loken() : CreatureScript("boss_loken") { }
82
83 struct boss_lokenAI : public BossAI
84 {
85 boss_lokenAI(Creature* creature) : BossAI(creature, DATA_LOKEN)
86 {
87 Initialize();
88 _isIntroDone = false;
89 }
90
91 void Initialize()
92 {
93 _healthAmountModifier = 1;
94 }
95
96 void Reset() override
97 {
98 Initialize();
99 _Reset();
100 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMELY_DEATH_START_EVENT);
101 }
102
103 void EnterCombat(Unit* /*who*/) override
104 {
105 _EnterCombat();
106 Talk(SAY_AGGRO);
107 events.SetPhase(PHASE_NORMAL);
108 events.ScheduleEvent(EVENT_ARC_LIGHTNING, 15000);
109 events.ScheduleEvent(EVENT_LIGHTNING_NOVA, 20000);
110 events.ScheduleEvent(EVENT_RESUME_PULSING_SHOCKWAVE, 1000);
111 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMELY_DEATH_START_EVENT);
112 }
113
114 void JustDied(Unit* /*killer*/) override
115 {
116 Talk(SAY_DEATH);
117 _JustDied();
118 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_PULSING_SHOCKWAVE_AURA);
119 }
120
121 void KilledUnit(Unit* who) override
122 {
123 if (who->GetTypeId() == TYPEID_PLAYER)
124 Talk(SAY_SLAY);
125 }
126
127 void MoveInLineOfSight(Unit* who) override
128 {
129 if (!_isIntroDone && me->IsValidAttackTarget(who) && me->IsWithinDistInMap(who, 40.0f))
130 {
131 _isIntroDone = true;
132 Talk(SAY_INTRO_1);
133 events.ScheduleEvent(EVENT_INTRO_DIALOGUE, 20000, 0, PHASE_INTRO);
134 }
135 BossAI::MoveInLineOfSight(who);
136 }
137
138 void UpdateAI(uint32 diff) override
139 {
140 if (events.IsInPhase(PHASE_NORMAL) && !UpdateVictim())
141 return;
142
143 events.Update(diff);
144
145 while (uint32 eventId = events.ExecuteEvent())
146 {
147 switch (eventId)
148 {
149 case EVENT_ARC_LIGHTNING:
150 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
151 DoCast(target, SPELL_ARC_LIGHTNING);
152 events.ScheduleEvent(EVENT_ARC_LIGHTNING, urand(15000, 16000));
153 break;
154 case EVENT_LIGHTNING_NOVA:
155 Talk(SAY_NOVA);
156 Talk(EMOTE_NOVA);
157 DoCastAOE(SPELL_LIGHTNING_NOVA);
158 me->RemoveAurasDueToSpell(SPELL_PULSING_SHOCKWAVE);
159 events.ScheduleEvent(EVENT_RESUME_PULSING_SHOCKWAVE, DUNGEON_MODE(5000, 4000)); // Pause Pulsing Shockwave aura
160 events.ScheduleEvent(EVENT_LIGHTNING_NOVA, urand(20000, 21000));
161 break;
162 case EVENT_RESUME_PULSING_SHOCKWAVE:
163 DoCast(me, SPELL_PULSING_SHOCKWAVE_AURA, true);
164 me->ClearUnitState(UNIT_STATE_CASTING); // This flag breaks movement.
165 DoCast(me, SPELL_PULSING_SHOCKWAVE, true);
166 break;
167 case EVENT_INTRO_DIALOGUE:
168 Talk(SAY_INTRO_2);
169 events.SetPhase(PHASE_NORMAL);
170 break;
171 default:
172 break;
173 }
174 }
175
176 DoMeleeAttackIfReady();
177 }
178
179 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
180 {
181 if (me->HealthBelowPctDamaged(100 - 25 * _healthAmountModifier, damage))
182 {
183 switch (_healthAmountModifier)
184 {
185 case 1:
186 Talk(SAY_75HEALTH);
187 break;
188 case 2:
189 Talk(SAY_50HEALTH);
190 break;
191 case 3:
192 Talk(SAY_25HEALTH);
193 break;
194 default:
195 break;
196 }
197 ++_healthAmountModifier;
198 }
199 }
200
201 private:
202 uint32 _healthAmountModifier;
203 bool _isIntroDone;
204 };
205
206 CreatureAI* GetAI(Creature* creature) const override
207 {
208 return GetInstanceAI<boss_lokenAI>(creature);
209 }
210};

212class spell_loken_pulsing_shockwave : public SpellScriptLoader
213{
214 public:
215 spell_loken_pulsing_shockwave() : SpellScriptLoader("spell_loken_pulsing_shockwave") { }
216
217 class spell_loken_pulsing_shockwave_SpellScript : public SpellScript
218 {
219 PrepareSpellScript(spell_loken_pulsing_shockwave_SpellScript);
220
221 void CalculateDamage(SpellEffIndex /*effIndex*/)
222 {
223 if (!GetHitUnit())
224 return;
225
226 float distance = GetCaster()->GetDistance2d(GetHitUnit());
227 if (distance > 1.0f)
228 SetHitDamage(int32(GetHitDamage() * distance));
229 }
230
231 void Register() override
232 {
233 OnEffectHitTarget += SpellEffectFn(spell_loken_pulsing_shockwave_SpellScript::CalculateDamage, EFFECT_0, SPELL_EFFECT_SCHOOL_DAMAGE);
234 }
235 };
236
237 SpellScript* GetSpellScript() const override
238 {
239 return new spell_loken_pulsing_shockwave_SpellScript();
240 }
241};

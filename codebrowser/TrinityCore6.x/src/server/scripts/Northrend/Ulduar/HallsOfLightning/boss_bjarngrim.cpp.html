95class boss_bjarngrim : public CreatureScript
96{
97public:
98 boss_bjarngrim() : CreatureScript("boss_bjarngrim") { }

105 struct boss_bjarngrimAI : public ScriptedAI
106 {
107 boss_bjarngrimAI(Creature* creature) : ScriptedAI(creature)
108 {
109 Initialize();
110 instance = creature->GetInstanceScript();
111 m_uiStance = STANCE_DEFENSIVE;
112 canBuff = true;
113 }
114
115 void Initialize()
116 {
117 m_bIsChangingStance = false;
118
119 m_uiChargingStatus = 0;
120 m_uiCharge_Timer = 1000;
121
122 m_uiChangeStance_Timer = urand(20000, 25000);
123
124 m_uiReflection_Timer = 8000;
125 m_uiKnockAway_Timer = 20000;
126 m_uiPummel_Timer = 10000;
127 m_uiIronform_Timer = 25000;
128
129 m_uiIntercept_Timer = 5000;
130 m_uiWhirlwind_Timer = 10000;
131 m_uiCleave_Timer = 8000;
132
133 m_uiMortalStrike_Timer = 8000;
134 m_uiSlam_Timer = 10000;
135 }
136
137 InstanceScript* instance;
138
139 bool m_bIsChangingStance;
140 bool canBuff;
141
142 uint8 m_uiChargingStatus;
143 uint8 m_uiStance;
144
145 uint32 m_uiCharge_Timer;
146 uint32 m_uiChangeStance_Timer;
147
148 uint32 m_uiReflection_Timer;
149 uint32 m_uiKnockAway_Timer;
150 uint32 m_uiPummel_Timer;
151 uint32 m_uiIronform_Timer;
152
153 uint32 m_uiIntercept_Timer;
154 uint32 m_uiWhirlwind_Timer;
155 uint32 m_uiCleave_Timer;
156
157 uint32 m_uiMortalStrike_Timer;
158 uint32 m_uiSlam_Timer;
159
160 ObjectGuid m_auiStormforgedLieutenantGUID[2];
161
162 void Reset() override
163 {
164 if (canBuff)
165 if (!me->HasAura(SPELL_TEMPORARY_ELECTRICAL_CHARGE))
166 me->AddAura(SPELL_TEMPORARY_ELECTRICAL_CHARGE, me);
167
168 Initialize();
169
170 for (uint8 i = 0; i < 2; ++i)
171 {
172 // Something isn't right here - m_auiStormforgedLieutenantGUID is never assinged to
173 if (Creature* pStormforgedLieutenant = ObjectAccessor::GetCreature(*me, m_auiStormforgedLieutenantGUID[i]))
174 if (!pStormforgedLieutenant->IsAlive())
175 pStormforgedLieutenant->Respawn();
176 }
177
178 if (m_uiStance != STANCE_DEFENSIVE)
179 {
180 DoRemoveStanceAura(m_uiStance);
181 DoCast(me, SPELL_DEFENSIVE_STANCE);
182 m_uiStance = STANCE_DEFENSIVE;
183 }
184
185 SetEquipmentSlots(false, EQUIP_SWORD, EQUIP_SHIELD, EQUIP_NO_CHANGE);
186
187 instance->SetBossState(DATA_BJARNGRIM, NOT_STARTED);
188 }
189
190 void EnterEvadeMode() override
191 {
192 if (me->HasAura(SPELL_TEMPORARY_ELECTRICAL_CHARGE))
193 canBuff = true;
194 else
195 canBuff = false;
196
197 ScriptedAI::EnterEvadeMode();
198 }
199
200 void EnterCombat(Unit* /*who*/) override
201 {
202 Talk(SAY_AGGRO);
203
204 //must get both lieutenants here and make sure they are with him
205 me->CallForHelp(30.0f);
206
207 instance->SetBossState(DATA_BJARNGRIM, IN_PROGRESS);
208 }
209
210 void KilledUnit(Unit* /*victim*/) override
211 {
212 Talk(SAY_SLAY);
213 }
214
215 void JustDied(Unit* /*killer*/) override
216 {
217 Talk(SAY_DEATH);
218
219 instance->SetBossState(DATA_BJARNGRIM, DONE);
220 }
221
222 /// @todo remove when removal is done by the core
223 void DoRemoveStanceAura(uint8 uiStance)
224 {
225 switch (uiStance)
226 {
227 case STANCE_DEFENSIVE:
228 me->RemoveAurasDueToSpell(SPELL_DEFENSIVE_STANCE);
229 break;
230 case STANCE_BERSERKER:
231 me->RemoveAurasDueToSpell(SPELL_BERSEKER_STANCE);
232 break;
233 case STANCE_BATTLE:
234 me->RemoveAurasDueToSpell(SPELL_BATTLE_STANCE);
235 break;
236 }
237 }
238
239 void UpdateAI(uint32 uiDiff) override
240 {
241 //Return since we have no target
242 if (!UpdateVictim())
243 return;
244
245 // Change stance
246 if (m_uiChangeStance_Timer <= uiDiff)
247 {
248 //wait for current spell to finish before change stance
249 if (me->IsNonMeleeSpellCast(false))
250 return;
251
252 DoRemoveStanceAura(m_uiStance);
253
254 int uiTempStance = rand32() % (3 - 1);
255
256 if (uiTempStance >= m_uiStance)
257 ++uiTempStance;
258
259 m_uiStance = uiTempStance;
260
261 switch (m_uiStance)
262 {
263 case STANCE_DEFENSIVE:
264 Talk(SAY_DEFENSIVE_STANCE);
265 Talk(EMOTE_DEFENSIVE_STANCE);
266 DoCast(me, SPELL_DEFENSIVE_STANCE);
267 SetEquipmentSlots(false, EQUIP_SWORD, EQUIP_SHIELD, EQUIP_NO_CHANGE);
268 break;
269 case STANCE_BERSERKER:
270 Talk(SAY_BERSEKER_STANCE);
271 Talk(EMOTE_BERSEKER_STANCE);
272 DoCast(me, SPELL_BERSEKER_STANCE);
273 SetEquipmentSlots(false, EQUIP_SWORD, EQUIP_SWORD, EQUIP_NO_CHANGE);
274 break;
275 case STANCE_BATTLE:
276 Talk(SAY_BATTLE_STANCE);
277 Talk(EMOTE_BATTLE_STANCE);
278 DoCast(me, SPELL_BATTLE_STANCE);
279 SetEquipmentSlots(false, EQUIP_MACE, EQUIP_UNEQUIP, EQUIP_NO_CHANGE);
280 break;
281 }
282
283 m_uiChangeStance_Timer = urand(20000, 25000);
284 return;
285 }
286 else
287 m_uiChangeStance_Timer -= uiDiff;
288
289 switch (m_uiStance)
290 {
291 case STANCE_DEFENSIVE:
292 {
293 if (m_uiReflection_Timer <= uiDiff)
294 {
295 DoCast(me, SPELL_SPELL_REFLECTION);
296 m_uiReflection_Timer = urand(8000, 9000);
297 }
298 else
299 m_uiReflection_Timer -= uiDiff;
300
301 if (m_uiKnockAway_Timer <= uiDiff)
302 {
303 DoCast(me, SPELL_KNOCK_AWAY);
304 m_uiKnockAway_Timer = urand(20000, 21000);
305 }
306 else
307 m_uiKnockAway_Timer -= uiDiff;
308
309 if (m_uiPummel_Timer <= uiDiff)
310 {
311 DoCastVictim(SPELL_PUMMEL);
312 m_uiPummel_Timer = urand(10000, 11000);
313 }
314 else
315 m_uiPummel_Timer -= uiDiff;
316
317 if (m_uiIronform_Timer <= uiDiff)
318 {
319 DoCast(me, SPELL_IRONFORM);
320 m_uiIronform_Timer = urand(25000, 26000);
321 }
322 else
323 m_uiIronform_Timer -= uiDiff;
324
325 break;
326 }
327 case STANCE_BERSERKER:
328 {
329 if (m_uiIntercept_Timer <= uiDiff)
330 {
331 //not much point is this, better random target and more often?
332 DoCastVictim(SPELL_INTERCEPT);
333 m_uiIntercept_Timer = urand(45000, 46000);
334 }
335 else
336 m_uiIntercept_Timer -= uiDiff;
337
338 if (m_uiWhirlwind_Timer <= uiDiff)
339 {
340 DoCast(me, SPELL_WHIRLWIND);
341 m_uiWhirlwind_Timer = urand(10000, 11000);
342 }
343 else
344 m_uiWhirlwind_Timer -= uiDiff;
345
346 if (m_uiCleave_Timer <= uiDiff)
347 {
348 DoCastVictim(SPELL_CLEAVE);
349 m_uiCleave_Timer = urand(8000, 9000);
350 }
351 else
352 m_uiCleave_Timer -= uiDiff;
353
354 break;
355 }
356 case STANCE_BATTLE:
357 {
358 if (m_uiMortalStrike_Timer <= uiDiff)
359 {
360 DoCastVictim(SPELL_MORTAL_STRIKE);
361 m_uiMortalStrike_Timer = urand(20000, 21000);
362 }
363 else
364 m_uiMortalStrike_Timer -= uiDiff;
365
366 if (m_uiSlam_Timer <= uiDiff)
367 {
368 DoCastVictim(SPELL_SLAM);
369 m_uiSlam_Timer = urand(15000, 16000);
370 }
371 else
372 m_uiSlam_Timer -= uiDiff;
373
374 break;
375 }
376 }
377
378 DoMeleeAttackIfReady();
379 }
380 };

382};

398 struct npc_stormforged_lieutenantAI : public ScriptedAI
399 {
400 npc_stormforged_lieutenantAI(Creature* creature) : ScriptedAI(creature)
401 {
402 Initialize();
403 instance = creature->GetInstanceScript();
404 }
405
406 void Initialize()
407 {
408 m_uiArcWeld_Timer = urand(20000, 21000);
409 m_uiRenewSteel_Timer = urand(10000, 11000);
410 }
411
412 InstanceScript* instance;
413
414 uint32 m_uiArcWeld_Timer;
415 uint32 m_uiRenewSteel_Timer;
416
417 void Reset() override
418 {
419 Initialize();
420 }
421
422 void EnterCombat(Unit* who) override
423 {
424 if (Creature* pBjarngrim = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BJARNGRIM)))
425 {
426 if (pBjarngrim->IsAlive() && !pBjarngrim->GetVictim())
427 pBjarngrim->AI()->AttackStart(who);
428 }
429 }
430
431 void UpdateAI(uint32 uiDiff) override
432 {
433 //Return since we have no target
434 if (!UpdateVictim())
435 return;
436
437 if (m_uiArcWeld_Timer <= uiDiff)
438 {
439 DoCastVictim(SPELL_ARC_WELD);
440 m_uiArcWeld_Timer = urand(20000, 21000);
441 }
442 else
443 m_uiArcWeld_Timer -= uiDiff;
444
445 if (m_uiRenewSteel_Timer <= uiDiff)
446 {
447 if (Creature* pBjarngrim = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BJARNGRIM)))
448 {
449 if (pBjarngrim->IsAlive())
450 DoCast(pBjarngrim, SPELL_RENEW_STEEL_N);
451 }
452 m_uiRenewSteel_Timer = urand(10000, 14000);
453 }
454 else
455 m_uiRenewSteel_Timer -= uiDiff;
456
457 DoMeleeAttackIfReady();
458 }
459 };

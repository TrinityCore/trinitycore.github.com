18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "halls_of_lightning.h"

22DoorData const doorData[] =
23{
24 { GO_VOLKHAN_DOOR, DATA_VOLKHAN, DOOR_TYPE_PASSAGE, BOUNDARY_NONE },
25 { GO_IONAR_DOOR, DATA_IONAR, DOOR_TYPE_PASSAGE, BOUNDARY_NONE },
26 { GO_LOKEN_DOOR, DATA_LOKEN, DOOR_TYPE_PASSAGE, BOUNDARY_NONE },
27 { 0, 0, DOOR_TYPE_ROOM, BOUNDARY_NONE } // END
28};
29
30class instance_halls_of_lightning : public InstanceMapScript
31{
32 public:
33 instance_halls_of_lightning() : InstanceMapScript(HoLScriptName, 602) { }
34
35 struct instance_halls_of_lightning_InstanceMapScript : public InstanceScript
36 {
37 instance_halls_of_lightning_InstanceMapScript(Map* map) : InstanceScript(map)
38 {
39 SetHeaders(DataHeader);
40 SetBossNumber(EncounterCount);
41 LoadDoorData(doorData);
42 }
43
44 void OnCreatureCreate(Creature* creature) override
45 {
46 switch (creature->GetEntry())
47 {
48 case NPC_BJARNGRIM:
49 GeneralBjarngrimGUID = creature->GetGUID();
50 break;
51 case NPC_VOLKHAN:
52 VolkhanGUID = creature->GetGUID();
53 break;
54 case NPC_IONAR:
55 IonarGUID = creature->GetGUID();
56 break;
57 case NPC_LOKEN:
58 LokenGUID = creature->GetGUID();
59 break;
60 default:
61 break;
62 }
63 }
64
65 void OnGameObjectCreate(GameObject* go) override
66 {
67 switch (go->GetEntry())
68 {
69 case GO_VOLKHAN_DOOR:
70 case GO_IONAR_DOOR:
71 case GO_LOKEN_DOOR:
72 AddDoor(go, true);
73 break;
74 case GO_LOKEN_THRONE:
75 LokenGlobeGUID = go->GetGUID();
76 break;
77 default:
78 break;
79 }
80 }
81
82 void OnGameObjectRemove(GameObject* go) override
83 {
84 switch (go->GetEntry())
85 {
86 case GO_VOLKHAN_DOOR:
87 case GO_IONAR_DOOR:
88 case GO_LOKEN_DOOR:
89 AddDoor(go, false);
90 break;
91 default:
92 break;
93 }
94 }
95
96 bool SetBossState(uint32 type, EncounterState state) override
97 {
98 if (!InstanceScript::SetBossState(type, state))
99 return false;
100
101 switch (type)
102 {
103 case DATA_LOKEN:
104 if (state == DONE)
105 if (GameObject* globe = instance->GetGameObject(LokenGlobeGUID))
106 globe->SendCustomAnim(0);
107 break;
108 default:
109 break;
110 }
111
112 return true;
113 }
114
115 ObjectGuid GetGuidData(uint32 type) const override
116 {
117 switch (type)
118 {
119 case DATA_BJARNGRIM:
120 return GeneralBjarngrimGUID;
121 case DATA_VOLKHAN:
122 return VolkhanGUID;
123 case DATA_IONAR:
124 return IonarGUID;
125 case DATA_LOKEN:
126 return LokenGUID;
127 default:
128 break;
129 }
130 return ObjectGuid::Empty;
131 }
132
133 protected:
134 ObjectGuid GeneralBjarngrimGUID;
135 ObjectGuid VolkhanGUID;
136 ObjectGuid IonarGUID;
137 ObjectGuid LokenGUID;
138
139 ObjectGuid LokenGlobeGUID;
140 };
141
142 InstanceScript* GetInstanceScript(InstanceMap* map) const override
143 {
144 return new instance_halls_of_lightning_InstanceMapScript(map);
145 }
146};
147
148void AddSC_instance_halls_of_lightning()
149{
150 new instance_halls_of_lightning();
151}

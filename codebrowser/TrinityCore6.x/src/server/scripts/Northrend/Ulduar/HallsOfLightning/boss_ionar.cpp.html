61/*######
62## Boss Ionar
63######*/
64
65class boss_ionar : public CreatureScript
66{
67public:
68 boss_ionar() : CreatureScript("boss_ionar") { }
69
70 CreatureAI* GetAI(Creature* creature) const override
71 {
72 return GetInstanceAI<boss_ionarAI>(creature);
73 }
74
75 struct boss_ionarAI : public ScriptedAI
76 {
77 boss_ionarAI(Creature* creature) : ScriptedAI(creature), lSparkList(creature)
78 {
79 Initialize();
80 instance = creature->GetInstanceScript();
81 }
82
83 void Initialize()
84 {
85 bIsSplitPhase = true;
86 bHasDispersed = false;
87
88 uiSplitTimer = 25 * IN_MILLISECONDS;
89
90 uiStaticOverloadTimer = urand(5 * IN_MILLISECONDS, 6 * IN_MILLISECONDS);
91 uiBallLightningTimer = urand(10 * IN_MILLISECONDS, 11 * IN_MILLISECONDS);
92
93 uiDisperseHealth = 45 + urand(0, 10);
94 }
95
96 InstanceScript* instance;
97
98 SummonList lSparkList;
99
100 bool bIsSplitPhase;
101 bool bHasDispersed;
102
103 uint32 uiSplitTimer;
104
105 uint32 uiStaticOverloadTimer;
106 uint32 uiBallLightningTimer;
107
108 uint32 uiDisperseHealth;
109
110 void Reset() override
111 {
112 lSparkList.DespawnAll();
113
114 Initialize();
115
116 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE|UNIT_FLAG_DISABLE_MOVE);
117
118 if (!me->IsVisible())
119 me->SetVisible(true);
120
121 instance->SetBossState(DATA_IONAR, NOT_STARTED);
122 }
123
124 void EnterCombat(Unit* /*who*/) override
125 {
126 Talk(SAY_AGGRO);
127
128 instance->SetBossState(DATA_IONAR, IN_PROGRESS);
129 }
130
131 void JustDied(Unit* /*killer*/) override
132 {
133 Talk(SAY_DEATH);
134
135 lSparkList.DespawnAll();
136
137 instance->SetBossState(DATA_IONAR, DONE);
138 }
139
140 void KilledUnit(Unit* who) override
141 {
142 if (who->GetTypeId() == TYPEID_PLAYER)
143 Talk(SAY_SLAY);
144 }
145
146 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
147 {
148 if (spell->Id == SPELL_DISPERSE)
149 {
150 for (uint8 i = 0; i < DATA_MAX_SPARKS; ++i)
151 me->CastSpell(me, SPELL_SUMMON_SPARK, true);
152
153 me->AttackStop();
154 me->SetVisible(false);
155 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE|UNIT_FLAG_DISABLE_MOVE);
156
157 me->GetMotionMaster()->Clear();
158 me->GetMotionMaster()->MoveIdle();
159 }
160 }
161
162 //make sparks come back
163 void CallBackSparks()
164 {
165 //should never be empty here, but check
166 if (lSparkList.empty())
167 return;
168
169 Position pos = me->GetPosition();
170
171 for (ObjectGuid guid : lSparkList)
172 {
173 if (Creature* pSpark = ObjectAccessor::GetCreature(*me, guid))
174 {
175 if (pSpark->IsAlive())
176 {
177 pSpark->SetSpeed(MOVE_RUN, 2.0f);
178 pSpark->GetMotionMaster()->Clear();
179 pSpark->GetMotionMaster()->MovePoint(DATA_POINT_CALLBACK, pos);
180 }
181 else
182 pSpark->DespawnOrUnsummon();
183 }
184 }
185 }
186
187 void DamageTaken(Unit* /*pDoneBy*/, uint32 &uiDamage) override
188 {
189 if (!me->IsVisible())
190 uiDamage = 0;
191 }
192
193 void JustSummoned(Creature* summoned) override
194 {
195 if (summoned->GetEntry() == NPC_SPARK_OF_IONAR)
196 {
197 lSparkList.Summon(summoned);
198
199 summoned->CastSpell(summoned, SPELL_SPARK_VISUAL_TRIGGER, true);
200
201 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
202 {
203 summoned->SetInCombatWith(target);
204 summoned->GetMotionMaster()->Clear();
205 summoned->GetMotionMaster()->MoveFollow(target, 0.0f, 0.0f);
206 }
207 }
208 }
209
210 void SummonedCreatureDespawn(Creature* summoned) override
211 {
212 if (summoned->GetEntry() == NPC_SPARK_OF_IONAR)
213 lSparkList.Despawn(summoned);
214 }
215
216 void UpdateAI(uint32 uiDiff) override
217 {
218 //Return since we have no target
219 if (!UpdateVictim())
220 return;
221
222 // Splitted
223 if (!me->IsVisible())
224 {
225 if (uiSplitTimer <= uiDiff)
226 {
227 uiSplitTimer = 2500;
228
229 // Return sparks to where Ionar splitted
230 if (bIsSplitPhase)
231 {
232 CallBackSparks();
233 bIsSplitPhase = false;
234 }
235 // Lightning effect and restore Ionar
236 else if (lSparkList.empty())
237 {
238 me->SetVisible(true);
239 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE|UNIT_FLAG_DISABLE_MOVE);
240
241 DoCast(me, SPELL_SPARK_DESPAWN, false);
242
243 uiSplitTimer = 25*IN_MILLISECONDS;
244 bIsSplitPhase = true;
245
246 if (me->GetVictim())
247 me->GetMotionMaster()->MoveChase(me->GetVictim());
248 }
249 }
250 else
251 uiSplitTimer -= uiDiff;
252
253 return;
254 }
255
256 if (uiStaticOverloadTimer <= uiDiff)
257 {
258 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
259 DoCast(target, SPELL_STATIC_OVERLOAD);
260
261 uiStaticOverloadTimer = urand(5*IN_MILLISECONDS, 6*IN_MILLISECONDS);
262 }
263 else
264 uiStaticOverloadTimer -= uiDiff;
265
266 if (uiBallLightningTimer <= uiDiff)
267 {
268 DoCastVictim(SPELL_BALL_LIGHTNING);
269 uiBallLightningTimer = urand(10*IN_MILLISECONDS, 11*IN_MILLISECONDS);
270 }
271 else
272 uiBallLightningTimer -= uiDiff;
273
274 // Health check
275 if (!bHasDispersed && HealthBelowPct(uiDisperseHealth))
276 {
277 bHasDispersed = true;
278
279 Talk(SAY_SPLIT);
280
281 if (me->IsNonMeleeSpellCast(false))
282 me->InterruptNonMeleeSpells(false);
283
284 DoCast(me, SPELL_DISPERSE, false);
285 }
286
287 DoMeleeAttackIfReady();
288 }
289 };
290
291};

297class npc_spark_of_ionar : public CreatureScript
298{
299public:
300 npc_spark_of_ionar() : CreatureScript("npc_spark_of_ionar") { }
301
302 struct npc_spark_of_ionarAI : public ScriptedAI
303 {
304 npc_spark_of_ionarAI(Creature* creature) : ScriptedAI(creature)
305 {
306 Initialize();
307 instance = creature->GetInstanceScript();
308 }
309
310 void Initialize()
311 {
312 uiCheckTimer = 2 * IN_MILLISECONDS;
313 }
314
315 InstanceScript* instance;
316
317 uint32 uiCheckTimer;
318
319 void Reset() override
320 {
321 Initialize();
322 me->SetReactState(REACT_PASSIVE);
323 }
324
325 void MovementInform(uint32 uiType, uint32 uiPointId) override
326 {
327 if (uiType != POINT_MOTION_TYPE || !instance)
328 return;
329
330 if (uiPointId == DATA_POINT_CALLBACK)
331 me->DespawnOrUnsummon();
332 }
333
334 void DamageTaken(Unit* /*pDoneBy*/, uint32 &uiDamage) override
335 {
336 uiDamage = 0;
337 }
338
339 void UpdateAI(uint32 uiDiff) override
340 {
341 // Despawn if the encounter is not running
342 if (instance->GetBossState(DATA_IONAR) != IN_PROGRESS)
343 {
344 me->DespawnOrUnsummon();
345 return;
346 }
347
348 // Prevent them to follow players through the whole instance
349 if (uiCheckTimer <= uiDiff)
350 {
351 Creature* ionar = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_IONAR));
352 if (ionar && ionar->IsAlive())
353 {
354 if (me->GetDistance(ionar) > DATA_MAX_SPARK_DISTANCE)
355 {
356 Position pos = ionar->GetPosition();
357
358 me->SetSpeed(MOVE_RUN, 2.0f);
359 me->GetMotionMaster()->Clear();
360 me->GetMotionMaster()->MovePoint(DATA_POINT_CALLBACK, pos);
361 }
362 }
363 else
364 me->DespawnOrUnsummon();
365 uiCheckTimer = 2*IN_MILLISECONDS;
366 }
367 else
368 uiCheckTimer -= uiDiff;
369
370 // No melee attack at all!
371 }
372 };
373
374 CreatureAI* GetAI(Creature* creature) const override
375 {
376 return GetInstanceAI<npc_spark_of_ionarAI>(creature);
377 }
378};
379

382 new boss_ionar();
383 new npc_spark_of_ionar();
384}

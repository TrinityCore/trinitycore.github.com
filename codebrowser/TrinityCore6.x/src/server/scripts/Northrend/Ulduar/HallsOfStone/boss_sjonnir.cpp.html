72class boss_sjonnir : public CreatureScript
73{
74 public:
75 boss_sjonnir() : CreatureScript("boss_sjonnir") { }
76
77 struct boss_sjonnirAI : public BossAI
78 {
79 boss_sjonnirAI(Creature* creature) : BossAI(creature, DATA_SJONNIR)
80 {
81 Initialize();
82 }
83
84 void Initialize()
85 {
86 abuseTheOoze = 0;
87 }
88
89 void Reset() override
90 {
91 _Reset();
92 Initialize();
93 }
94
95 void EnterCombat(Unit* who) override
96 {
97 if (!instance->CheckRequiredBosses(DATA_SJONNIR, who->ToPlayer()))
98 {
99 EnterEvadeMode();
100 return;
101 }
102
103 _EnterCombat();
104 Talk(SAY_AGGRO);
105
106 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, urand(3000, 8000));
107 events.ScheduleEvent(EVENT_LIGHTNING_SHIELD, urand(20000, 25000));
108 events.ScheduleEvent(EVENT_STATIC_CHARGE, urand(20000, 25000));
109 events.ScheduleEvent(EVENT_LIGHTNING_RING, urand(30000, 35000));
110 events.ScheduleEvent(EVENT_SUMMON, 5000);
111 events.ScheduleEvent(EVENT_FRENZY, 300000);
112 }
113
114 void JustSummoned(Creature* summon) override
115 {
116 summon->GetMotionMaster()->MovePoint(0, CenterPoint);
117 /*if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
118 summon->AI()->AttackStart(target);*/
119 summons.Summon(summon);
120 }
121
122 void JustDied(Unit* /*killer*/) override
123 {
124 _JustDied();
125 Talk(SAY_DEATH);
126 }
127
128 void KilledUnit(Unit* who) override
129 {
130 if (who->GetTypeId() == TYPEID_PLAYER)
131 Talk(SAY_SLAY);
132 }
133
134 void DoAction(int32 action) override
135 {
136 if (action == ACTION_OOZE_DEAD)
137 ++abuseTheOoze;
138 }
139
140 uint32 GetData(uint32 type) const override
141 {
142 if (type == DATA_ABUSE_THE_OOZE)
143 return abuseTheOoze;
144
145 return 0;
146 }
147
148 void UpdateAI(uint32 diff) override
149 {
150 if (!UpdateVictim())
151 return;
152
153 events.Update(diff);
154
155 if (me->HasUnitState(UNIT_STATE_CASTING))
156 return;
157
158 while (uint32 eventId = events.ExecuteEvent())
159 {
160 switch (eventId)
161 {
162 case EVENT_CHAIN_LIGHTNING:
163 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
164 DoCast(target, SPELL_CHAIN_LIGHTING);
165 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, urand(10000, 15000));
166 break;
167 case EVENT_LIGHTNING_SHIELD:
168 DoCast(me, SPELL_LIGHTING_SHIELD);
169 break;
170 case EVENT_STATIC_CHARGE:
171 DoCastVictim(SPELL_STATIC_CHARGE);
172 events.ScheduleEvent(EVENT_STATIC_CHARGE, urand(20000, 25000));
173 break;
174 case EVENT_LIGHTNING_RING:
175 DoCast(me, SPELL_LIGHTING_RING);
176 events.ScheduleEvent(EVENT_LIGHTNING_RING, urand(30000, 35000));
177 break;
178 case EVENT_SUMMON:
179 {
180 uint8 summonPipe = urand(0, 1);
181 if (HealthAbovePct(75))
182 me->SummonCreature(NPC_FORGED_IRON_DWARF, PipeLocations[summonPipe], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 30000);
183 else if (HealthAbovePct(50))
184 me->SummonCreature(NPC_FORGED_IRON_TROGG, PipeLocations[summonPipe], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 30000);
185 else if (HealthAbovePct(25))
186 me->SummonCreature(NPC_MALFORMED_OOZE, PipeLocations[summonPipe], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 30000);
187 else
188 me->SummonCreature(NPC_EARTHEN_DWARF, PipeLocations[summonPipe], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 30000);
189
190 events.ScheduleEvent(EVENT_SUMMON, 20000);
191 break;
192 }
193 case EVENT_FRENZY:
194 /// @todo: add emote
195 DoCast(me, SPELL_FRENZY, true);
196 break;
197 default:
198 break;
199 }
200 }
201
202 DoMeleeAttackIfReady();
203 }
204
205 private:
206 uint8 abuseTheOoze;
207 };
208
209 CreatureAI* GetAI(Creature* creature) const override
210 {
211 return GetHallsOfStoneAI<boss_sjonnirAI>(creature);
212 }
213};

215class npc_malformed_ooze : public CreatureScript
216{
217 public:
218 npc_malformed_ooze() : CreatureScript("npc_malformed_ooze") { }
219
220 struct npc_malformed_oozeAI : public ScriptedAI
221 {
222 npc_malformed_oozeAI(Creature* creature) : ScriptedAI(creature)
223 {
224 Initialize();
225 }
226
227 void Initialize()
228 {
229 _mergeTimer = 10000;
230 }
231
232 void Reset() override
233 {
234 Initialize();
235 }
236
237 void UpdateAI(uint32 diff) override
238 {
239 if (_mergeTimer <= diff)
240 {
241 if (Creature* temp = me->FindNearestCreature(NPC_MALFORMED_OOZE, 3.0f, true))
242 {
243 DoSpawnCreature(NPC_IRON_SLUDGE, 0, 0, 0, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 20000);
244 temp->DisappearAndDie();
245 me->DisappearAndDie();
246 }
247 _mergeTimer = 3000;
248 }
249 else
250 _mergeTimer -= diff;
251
252 if (!UpdateVictim())
253 return;
254
255 DoMeleeAttackIfReady();
256 }
257
258 private:
259 uint32 _mergeTimer;
260 };
261
262 CreatureAI* GetAI(Creature* creature) const override
263 {
264 return GetHallsOfStoneAI<npc_malformed_oozeAI>(creature);
265 }
266};

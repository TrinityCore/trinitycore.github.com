339 if (lDwarfGUIDList.empty())
340 return;
341 for (GuidList::const_iterator itr = lDwarfGUIDList.begin(); itr != lDwarfGUIDList.end(); ++itr)
342 {
343 Creature* temp = ObjectAccessor::GetCreature(*me, instance ? (*itr) : ObjectGuid::Empty);
344 if (temp && temp->IsAlive())
345 temp->DespawnOrUnsummon();
346 }
347 lDwarfGUIDList.clear();

351 {
352 switch (waypointId)
353 {
354 case 7:

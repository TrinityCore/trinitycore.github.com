23DoorData const doorData[] =
24{
25 { GO_SJONNIR_DOOR, DATA_TRIBUNAL_OF_AGES, DOOR_TYPE_PASSAGE, BOUNDARY_NONE },
26 { 0, 0, DOOR_TYPE_ROOM, BOUNDARY_NONE } // END
27};
28
29class instance_halls_of_stone : public InstanceMapScript
30{
31 public:
32 instance_halls_of_stone() : InstanceMapScript(HoSScriptName, 599) { }
33
34 struct instance_halls_of_stone_InstanceMapScript : public InstanceScript
35 {
36 instance_halls_of_stone_InstanceMapScript(Map* map) : InstanceScript(map)
37 {
38 SetHeaders(DataHeader);
39 SetBossNumber(EncounterCount);
40 LoadDoorData(doorData);
41 }
42
43 void OnCreatureCreate(Creature* creature) override
44 {
45 switch (creature->GetEntry())
46 {
47 case NPC_KRYSTALLUS:
48 KrystallusGUID = creature->GetGUID();
49 break;
50 case NPC_MAIDEN:
51 MaidenOfGriefGUID = creature->GetGUID();
52 break;
53 case NPC_SJONNIR:
54 SjonnirGUID = creature->GetGUID();
55 break;
56 case NPC_MARNAK:
57 MarnakGUID = creature->GetGUID();
58 break;
59 case NPC_KADDRAK:
60 KaddrakGUID = creature->GetGUID();
61 break;
62 case NPC_ABEDNEUM:
63 AbedneumGUID = creature->GetGUID();
64 break;
65 case NPC_BRANN:
66 BrannGUID = creature->GetGUID();
67 break;
68 default:
69 break;
70 }
71 }
72
73 void OnGameObjectCreate(GameObject* go) override
74 {
75 switch (go->GetEntry())
76 {
77 case GO_ABEDNEUM:
78 AbedneumGoGUID = go->GetGUID();
79 break;
80 case GO_MARNAK:
81 MarnakGoGUID = go->GetGUID();
82 break;
83 case GO_KADDRAK:
84 KaddrakGoGUID = go->GetGUID();
85 break;
86 case GO_TRIBUNAL_CONSOLE:
87 TribunalConsoleGUID = go->GetGUID();
88 break;
89 case GO_TRIBUNAL_CHEST:
90 case GO_TRIBUNAL_CHEST_HERO:
91 TribunalChestGUID = go->GetGUID();
92 if (GetBossState(DATA_TRIBUNAL_OF_AGES) == DONE)
93 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
94 break;
95 case GO_TRIBUNAL_SKY_FLOOR:
96 TribunalSkyFloorGUID = go->GetGUID();
97 break;
98 case GO_SJONNIR_DOOR:
99 AddDoor(go, true);
100 break;
101 default:
102 break;
103 }
104 }
105
106 void OnGameObjectRemove(GameObject* go) override
107 {
108 switch (go->GetEntry())
109 {
110 case GO_SJONNIR_DOOR:
111 AddDoor(go, false);
112 break;
113 default:
114 break;
115 }
116 }
117
118 ObjectGuid GetGuidData(uint32 type) const override
119 {
120 switch (type)
121 {
122 case DATA_MAIDEN_OF_GRIEF:
123 return MaidenOfGriefGUID;
124 case DATA_KRYSTALLUS:
125 return KrystallusGUID;
126 case DATA_SJONNIR:
127 return SjonnirGUID;
128 case DATA_KADDRAK:
129 return KaddrakGUID;
130 case DATA_MARNAK:
131 return MarnakGUID;
132 case DATA_ABEDNEUM:
133 return AbedneumGUID;
134 case DATA_GO_TRIBUNAL_CONSOLE:
135 return TribunalConsoleGUID;
136 case DATA_GO_KADDRAK:
137 return KaddrakGoGUID;
138 case DATA_GO_ABEDNEUM:
139 return AbedneumGoGUID;
140 case DATA_GO_MARNAK:
141 return MarnakGoGUID;
142 case DATA_GO_SKY_FLOOR:
143 return TribunalSkyFloorGUID;
144 default:
145 break;
146 }
147
148 return ObjectGuid::Empty;
149 }
150
151 bool SetBossState(uint32 type, EncounterState state) override
152 {
153 if (!InstanceScript::SetBossState(type, state))
154 return false;
155
156 switch (type)
157 {
158 case DATA_TRIBUNAL_OF_AGES:
159 if (state == DONE)
160 {
161 if (GameObject* go = instance->GetGameObject(TribunalChestGUID))
162 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
163 }
164 break;
165 default:
166 break;
167 }
168
169 return true;
170 }
171
172 bool CheckRequiredBosses(uint32 bossId, Player const* player = nullptr) const override
173 {
174 if (_SkipCheckRequiredBosses(player))
175 return true;
176
177 switch (bossId)
178 {
179 case DATA_SJONNIR:
180 if (GetBossState(DATA_TRIBUNAL_OF_AGES) != DONE)
181 return false;
182 break;
183 default:
184 break;
185 }
186
187 return true;
188 }
189
190 protected:
191 ObjectGuid KrystallusGUID;
192 ObjectGuid MaidenOfGriefGUID;
193 ObjectGuid SjonnirGUID;
194
195 ObjectGuid KaddrakGUID;
196 ObjectGuid AbedneumGUID;
197 ObjectGuid MarnakGUID;
198 ObjectGuid BrannGUID;
199
200 ObjectGuid TribunalConsoleGUID;
201 ObjectGuid TribunalChestGUID;
202 ObjectGuid TribunalSkyFloorGUID;
203 ObjectGuid KaddrakGoGUID;
204 ObjectGuid AbedneumGoGUID;
205 ObjectGuid MarnakGoGUID;
206 };
207
208 InstanceScript* GetInstanceScript(InstanceMap* map) const override
209 {
210 return new instance_halls_of_stone_InstanceMapScript(map);
211 }
212};

214void AddSC_instance_halls_of_stone()

216 new instance_halls_of_stone();
217}

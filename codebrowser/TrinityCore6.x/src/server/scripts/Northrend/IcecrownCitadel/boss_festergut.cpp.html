80class boss_festergut : public CreatureScript
81{
82 public:
83 boss_festergut() : CreatureScript("boss_festergut") { }

85 struct boss_festergutAI : public BossAI
86 {
87 boss_festergutAI(Creature* creature) : BossAI(creature, DATA_FESTERGUT)
88 {
89 _maxInoculatedStack = 0;
90 _inhaleCounter = 0;
91 }
92
93 void Reset() override
94 {
95 _Reset();
96 events.ScheduleEvent(EVENT_BERSERK, 300000);
97 events.ScheduleEvent(EVENT_INHALE_BLIGHT, urand(25000, 30000));
98 events.ScheduleEvent(EVENT_GAS_SPORE, urand(20000, 25000));
99 events.ScheduleEvent(EVENT_GASTRIC_BLOAT, urand(12500, 15000));
100 _maxInoculatedStack = 0;
101 _inhaleCounter = 0;
102 me->RemoveAurasDueToSpell(SPELL_BERSERK2);
103 if (Creature* gasDummy = me->FindNearestCreature(NPC_GAS_DUMMY, 100.0f, true))
104 {
105 _gasDummyGUID = gasDummy->GetGUID();
106 for (uint8 i = 0; i < 3; ++i)
107 {
108 me->RemoveAurasDueToSpell(gaseousBlight[i]);
109 gasDummy->RemoveAurasDueToSpell(gaseousBlightVisual[i]);
110 }
111 }
112 }
113
114 void EnterCombat(Unit* who) override
115 {
116 if (!instance->CheckRequiredBosses(DATA_FESTERGUT, who->ToPlayer()))
117 {
118 EnterEvadeMode();
119 instance->DoCastSpellOnPlayers(LIGHT_S_HAMMER_TELEPORT);
120 return;
121 }
122
123 me->setActive(true);
124 Talk(SAY_AGGRO);
125 if (Creature* gasDummy = me->FindNearestCreature(NPC_GAS_DUMMY, 100.0f, true))
126 _gasDummyGUID = gasDummy->GetGUID();
127 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
128 professor->AI()->DoAction(ACTION_FESTERGUT_COMBAT);
129 DoZoneInCombat();
130 }
131
132 void JustDied(Unit* /*killer*/) override
133 {
134 _JustDied();
135 Talk(SAY_DEATH);
136 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
137 professor->AI()->DoAction(ACTION_FESTERGUT_DEATH);
138
139 RemoveBlight();
140 }
141
142 void JustReachedHome() override
143 {
144 _JustReachedHome();
145 instance->SetBossState(DATA_FESTERGUT, FAIL);
146 }
147
148 void EnterEvadeMode() override
149 {
150 ScriptedAI::EnterEvadeMode();
151 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
152 professor->AI()->EnterEvadeMode();
153 }
154
155 void KilledUnit(Unit* victim) override
156 {
157 if (victim->GetTypeId() == TYPEID_PLAYER)
158 Talk(SAY_KILL);
159 }
160
161 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
162 {
163 if (spell->Id == PUNGENT_BLIGHT_HELPER)
164 target->RemoveAurasDueToSpell(INOCULATED_HELPER);
165 }
166
167 void UpdateAI(uint32 diff) override
168 {
169 if (!UpdateVictim() || !CheckInRoom())
170 return;
171
172 events.Update(diff);
173
174 if (me->HasUnitState(UNIT_STATE_CASTING))
175 return;
176
177 while (uint32 eventId = events.ExecuteEvent())
178 {
179 switch (eventId)
180 {
181 case EVENT_INHALE_BLIGHT:
182 {
183 RemoveBlight();
184 if (_inhaleCounter == 3)
185 {
186 Talk(EMOTE_WARN_PUNGENT_BLIGHT);
187 Talk(SAY_PUNGENT_BLIGHT);
188 DoCast(me, SPELL_PUNGENT_BLIGHT);
189 _inhaleCounter = 0;
190 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
191 professor->AI()->DoAction(ACTION_FESTERGUT_GAS);
192 events.RescheduleEvent(EVENT_GAS_SPORE, urand(20000, 25000));
193 }
194 else
195 {
196 DoCast(me, SPELL_INHALE_BLIGHT);
197 // just cast and dont bother with target, conditions will handle it
198 ++_inhaleCounter;
199 if (_inhaleCounter < 3)
200 me->CastSpell(me, gaseousBlight[_inhaleCounter], true, NULL, NULL, me->GetGUID());
201 }
202
203 events.ScheduleEvent(EVENT_INHALE_BLIGHT, urand(33500, 35000));
204 break;
205 }
206 case EVENT_VILE_GAS:
207 {
208 std::list<Unit*> ranged, melee;
209 uint32 minTargets = RAID_MODE<uint32>(3, 8, 3, 8);
210 SelectTargetList(ranged, 25, SELECT_TARGET_RANDOM, -5.0f, true);
211 SelectTargetList(melee, 25, SELECT_TARGET_RANDOM, 5.0f, true);
212 while (ranged.size() < minTargets)
213 {
214 if (melee.empty())
215 break;
216
217 Unit* target = Trinity::Containers::SelectRandomContainerElement(melee);
218 ranged.push_back(target);
219 melee.remove(target);
220 }
221
222 if (!ranged.empty())
223 {
224 Trinity::Containers::RandomResizeList(ranged, RAID_MODE<uint32>(1, 3, 1, 3));
225 for (std::list<Unit*>::iterator itr = ranged.begin(); itr != ranged.end(); ++itr)
226 DoCast(*itr, SPELL_VILE_GAS);
227 }
228
229 events.ScheduleEvent(EVENT_VILE_GAS, urand(28000, 35000));
230 break;
231 }
232 case EVENT_GAS_SPORE:
233 Talk(EMOTE_WARN_GAS_SPORE);
234 Talk(EMOTE_GAS_SPORE);
235 me->CastCustomSpell(SPELL_GAS_SPORE, SPELLVALUE_MAX_TARGETS, RAID_MODE<int32>(2, 3, 2, 3), me);
236 events.ScheduleEvent(EVENT_GAS_SPORE, urand(40000, 45000));
237 events.RescheduleEvent(EVENT_VILE_GAS, urand(28000, 35000));
238 break;
239 case EVENT_GASTRIC_BLOAT:
240 DoCastVictim(SPELL_GASTRIC_BLOAT);
241 events.ScheduleEvent(EVENT_GASTRIC_BLOAT, urand(15000, 17500));
242 break;
243 case EVENT_BERSERK:
244 DoCast(me, SPELL_BERSERK2);
245 Talk(SAY_BERSERK);
246 break;
247 default:
248 break;
249 }
250 }
251
252 DoMeleeAttackIfReady();
253 }
254
255 void SetData(uint32 type, uint32 data) override
256 {
257 if (type == DATA_INOCULATED_STACK && data > _maxInoculatedStack)
258 _maxInoculatedStack = data;
259 }
260
261 uint32 GetData(uint32 type) const override
262 {
263 if (type == DATA_INOCULATED_STACK)
264 return uint32(_maxInoculatedStack);
265
266 return 0;
267 }
268
269 void RemoveBlight()
270 {
271 if (Creature* gasDummy = ObjectAccessor::GetCreature(*me, _gasDummyGUID))
272 for (uint8 i = 0; i < 3; ++i)
273 {
274 me->RemoveAurasDueToSpell(gaseousBlight[i]);
275 gasDummy->RemoveAurasDueToSpell(gaseousBlightVisual[i]);
276 }
277 }
278
279 private:
280 ObjectGuid _gasDummyGUID;
281 uint32 _maxInoculatedStack;
282 uint32 _inhaleCounter;
283 };
284
285 CreatureAI* GetAI(Creature* creature) const override
286 {
287 return GetIcecrownCitadelAI<boss_festergutAI>(creature);
288 }

314
315 void UpdateAI(uint32 diff) override
316 {
317 if (!UpdateVictim())
318 return;
319
320 _events.Update(diff);
321
322 if (me->HasUnitState(UNIT_STATE_CASTING))
323 return;
324
325 while (uint32 eventId = _events.ExecuteEvent())
326 {
327 switch (eventId)
328 {
329 case EVENT_DECIMATE:
330 DoCastVictim(SPELL_DECIMATE);
331 _events.ScheduleEvent(EVENT_DECIMATE, urand(20000, 25000));

333 case EVENT_MORTAL_WOUND:
334 DoCastVictim(SPELL_MORTAL_WOUND);
335 _events.ScheduleEvent(EVENT_MORTAL_WOUND, urand(10000, 12500));
336 break;
337 default:
338 break;
339 }
340 }
341
342 DoMeleeAttackIfReady();
343 }
344
345 void JustDied(Unit* /*killer*/) override
346 {
347 if (Creature* festergut = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_FESTERGUT)))
348 if (festergut->IsAlive())
349 festergut->AI()->Talk(SAY_STINKY_DEAD);
350 }
351
352 private:
353 EventMap _events;
354 InstanceScript* _instance;
355 };

438 class spell_festergut_blighted_spores_AuraScript : public AuraScript
439 {
440 PrepareAuraScript(spell_festergut_blighted_spores_AuraScript);
441
442 bool Validate(SpellInfo const* /*spell*/) override
443 {
444 if (!sSpellMgr->GetSpellInfo(SPELL_INOCULATED))
445 return false;
446 return true;
447 }
448
449 void ExtraEffect(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
450 {
451 GetTarget()->CastSpell(GetTarget(), SPELL_INOCULATED, true);
452 if (InstanceScript* instance = GetTarget()->GetInstanceScript())
453 if (Creature* festergut = ObjectAccessor::GetCreature(*GetTarget(), instance->GetGuidData(DATA_FESTERGUT)))
454 festergut->AI()->SetData(DATA_INOCULATED_STACK, GetStackAmount());
455
456 HandleResidue();
457 }
458
459 void HandleResidue()
460 {
461 Player* target = GetUnitOwner()->ToPlayer();
462 if (!target)
463 return;
464
465 if (target->HasAura(SPELL_ORANGE_BLIGHT_RESIDUE))
466 return;
467
468 uint32 questId = target->GetMap()->Is25ManRaid() ? QUEST_RESIDUE_RENDEZVOUS_25 : QUEST_RESIDUE_RENDEZVOUS_10;
469 if (target->GetQuestStatus(questId) != QUEST_STATUS_INCOMPLETE)
470 return;
471
472 target->CastSpell(target, SPELL_ORANGE_BLIGHT_RESIDUE, TRIGGERED_FULL_MASK);
473 }
474
475 void Register() override
476 {
477 OnEffectRemove += AuraEffectApplyFn(spell_festergut_blighted_spores_AuraScript::ExtraEffect, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE, AURA_EFFECT_HANDLE_REAL_OR_REAPPLY_MASK);
478 }
479 };
480
481 AuraScript* GetAuraScript() const override
482 {
483 return new spell_festergut_blighted_spores_AuraScript();
484 }

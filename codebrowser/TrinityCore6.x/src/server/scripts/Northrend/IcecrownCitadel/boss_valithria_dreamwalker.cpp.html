215class ValithriaDespawner : public BasicEvent
216{
217 public:
218 explicit ValithriaDespawner(Creature* creature) : _creature(creature)
219 {
220 }
221
222 bool Execute(uint64 /*currTime*/, uint32 /*diff*/) override
223 {
224 Trinity::CreatureWorker<ValithriaDespawner> worker(_creature, *this);
225 _creature->VisitNearbyGridObject(333.0f, worker);
226 return true;
227 }
228
229 void operator()(Creature* creature) const
230 {
231 switch (creature->GetEntry())
232 {
233 case NPC_VALITHRIA_DREAMWALKER:
234 if (InstanceScript* instance = creature->GetInstanceScript())
235 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, creature);
236 break;
237 case NPC_BLAZING_SKELETON:
238 case NPC_SUPPRESSER:
239 case NPC_BLISTERING_ZOMBIE:
240 case NPC_GLUTTONOUS_ABOMINATION:
241 case NPC_MANA_VOID:
242 case NPC_COLUMN_OF_FROST:
243 case NPC_ROT_WORM:
244 creature->DespawnOrUnsummon();
245 return;
246 case NPC_RISEN_ARCHMAGE:
247 if (!creature->GetSpawnId())
248 {
249 creature->DespawnOrUnsummon();
250 return;
251 }
252 creature->Respawn(true);
253 break;
254 default:
255 return;
256 }
257
258 uint32 corpseDelay = creature->GetCorpseDelay();
259 uint32 respawnDelay = creature->GetRespawnDelay();
260 creature->SetCorpseDelay(1);
261 creature->SetRespawnDelay(10);
262
263 if (CreatureData const* data = creature->GetCreatureData())
264 creature->SetPosition(data->posX, data->posY, data->posZ, data->orientation);
265 creature->DespawnOrUnsummon();
266
267 creature->SetCorpseDelay(corpseDelay);
268 creature->SetRespawnDelay(respawnDelay);
269 }
270
271 private:
272 Creature* _creature;

289 void Initialize()

292 _under25PercentTalkDone = false;

297
298 void InitializeAI() override
299 {
300 if (CreatureData const* data = sObjectMgr->GetCreatureData(me->GetSpawnId()))
301 if (data->curhealth)
302 _spawnHealth = data->curhealth;
303
304 if (!me->isDead())
305 Reset();
306 }
307
308 void Reset() override
309 {
310 me->SetHealth(_spawnHealth);
311 me->SetReactState(REACT_PASSIVE);
312 me->LoadCreaturesAddon();
313 // immune to percent heals
314 me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_OBS_MOD_HEALTH, true);
315 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_HEAL_PCT, true);
316 // Glyph of Dispel Magic - not a percent heal by effect, its cast with custom basepoints
317 me->ApplySpellImmune(0, IMMUNITY_ID, 56131, true);
318 _instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
319 Initialize();
320 }
321
322 void AttackStart(Unit* /*target*/) override
323 {
324 }
325
326 void DoAction(int32 action) override
327 {
328 if (action != ACTION_ENTER_COMBAT)
329 return;
330
331 DoCast(me, SPELL_COPY_DAMAGE);
332 _instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, me);
333 _events.ScheduleEvent(EVENT_INTRO_TALK, 15000);
334 _events.ScheduleEvent(EVENT_DREAM_PORTAL, urand(45000, 48000));
335 if (IsHeroic())
336 _events.ScheduleEvent(EVENT_BERSERK, 420000);
337 }
338
339 void HealReceived(Unit* healer, uint32& heal) override
340 {
341 if (!me->hasLootRecipient())
342 me->SetLootRecipient(healer);
343
344 me->LowerPlayerDamageReq(heal);
345
346 // encounter complete
347 if (me->HealthAbovePctHealed(100, heal) && !_done)
348 {
349 _done = true;
350 Talk(SAY_VALITHRIA_SUCCESS);
351 _instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
352 me->RemoveAurasDueToSpell(SPELL_CORRUPTION_VALITHRIA);
353 DoCast(me, SPELL_ACHIEVEMENT_CHECK);
354 DoCastAOE(SPELL_DREAMWALKERS_RAGE);
355 _events.ScheduleEvent(EVENT_DREAM_SLIP, 3500);
356 if (Creature* lichKing = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_VALITHRIA_LICH_KING)))
357 lichKing->AI()->EnterEvadeMode();
358 }
359 else if (!_over75PercentTalkDone && me->HealthAbovePctHealed(75, heal))
360 {
361 _over75PercentTalkDone = true;
362 Talk(SAY_VALITHRIA_75_PERCENT);
363 }
364 else if (_instance->GetBossState(DATA_VALITHRIA_DREAMWALKER) == NOT_STARTED)
365 if (Creature* archmage = me->FindNearestCreature(NPC_RISEN_ARCHMAGE, 30.0f))
366 archmage->AI()->DoZoneInCombat(); // call EnterCombat on one of them, that will make it all start
367 }
368
369 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
370 {
371 if (me->HealthBelowPctDamaged(25, damage))
372 {
373 if (!_under25PercentTalkDone)
374 {
375 _under25PercentTalkDone = true;
376 Talk(SAY_VALITHRIA_25_PERCENT);
377 }
378
379 if (damage > me->GetHealth())
380 {
381 damage = 0;
382 if (!_justDied)
383 {
384 _justDied = true;
385 Talk(SAY_VALITHRIA_DEATH);
386 _instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
387 if (Creature* trigger = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_VALITHRIA_TRIGGER)))
388 trigger->AI()->DoAction(ACTION_DEATH);
389 }
390 }
391 }
392 }
393
394 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
395 {
396 if (spell->Id == SPELL_DREAM_SLIP)
397 {
398 DoCast(me, SPELL_CLEAR_ALL);
399 DoCast(me, SPELL_AWARD_REPUTATION_BOSS_KILL);
400 // this display id was found in sniff instead of the one on aura
401 me->SetDisplayId(11686);
402 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
403 me->DespawnOrUnsummon(4000);
404 if (Creature* lichKing = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_VALITHRIA_LICH_KING)))
405 lichKing->CastSpell(lichKing, SPELL_SPAWN_CHEST, false);
406
407 if (Creature* trigger = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_VALITHRIA_TRIGGER)))
408 me->Kill(trigger);
409 }
410 }
411
412 void JustSummoned(Creature* summon) override
413 {
414 if (summon->GetEntry() == NPC_DREAM_PORTAL_PRE_EFFECT)
415 {
416 summon->m_Events.AddEvent(new DelayedCastEvent(summon, SPELL_SUMMON_DREAM_PORTAL, me->GetGUID(), 6000), summon->m_Events.CalculateTime(15000));
417 summon->m_Events.AddEvent(new AuraRemoveEvent(summon, SPELL_DREAM_PORTAL_VISUAL_PRE), summon->m_Events.CalculateTime(15000));
418 }
419 else if (summon->GetEntry() == NPC_NIGHTMARE_PORTAL_PRE_EFFECT)
420 {
421 summon->m_Events.AddEvent(new DelayedCastEvent(summon, SPELL_SUMMON_NIGHTMARE_PORTAL, me->GetGUID(), 6000), summon->m_Events.CalculateTime(15000));
422 summon->m_Events.AddEvent(new AuraRemoveEvent(summon, SPELL_NIGHTMARE_PORTAL_VISUAL_PRE), summon->m_Events.CalculateTime(15000));
423 }
424 }
425
426 void SummonedCreatureDespawn(Creature* summon) override
427 {
428 if (summon->GetEntry() == NPC_DREAM_PORTAL || summon->GetEntry() == NPC_NIGHTMARE_PORTAL)
429 if (summon->AI()->GetData(MISSED_PORTALS))
430 ++_missedPortals;
431 }
432
433 void UpdateAI(uint32 diff) override
434 {
435 // does not enter combat
436 if (_instance->GetBossState(DATA_VALITHRIA_DREAMWALKER) != IN_PROGRESS)
437 return;
438
439 _events.Update(diff);
440
441 if (me->HasUnitState(UNIT_STATE_CASTING))
442 return;
443
444 while (uint32 eventId = _events.ExecuteEvent())
445 {
446 switch (eventId)
447 {
448 case EVENT_INTRO_TALK:
449 Talk(SAY_VALITHRIA_ENTER_COMBAT);
450 break;
451 case EVENT_BERSERK:
452 Talk(SAY_VALITHRIA_BERSERK);
453 break;
454 case EVENT_DREAM_PORTAL:
455 if (!IsHeroic())
456 Talk(SAY_VALITHRIA_DREAM_PORTAL);
457 for (uint32 i = 0; i < _portalCount; ++i)
458 DoCast(me, SUMMON_PORTAL);
459 _events.ScheduleEvent(EVENT_DREAM_PORTAL, urand(45000, 48000));
460 break;
461 case EVENT_DREAM_SLIP:
462 DoCast(me, SPELL_DREAM_SLIP);
463 break;
464 default:
465 break;
466 }
467 }
468 }
469
470 uint32 GetData(uint32 type) const override
471 {
472 if (type == MISSED_PORTALS)
473 return _missedPortals;
474
475 return 0;
476 }

478 private:
479 EventMap _events;
480 InstanceScript* _instance;
481 uint32 _spawnHealth;
482 uint32 const _portalCount;
483 uint32 _missedPortals;
484 bool _under25PercentTalkDone;
485 bool _over75PercentTalkDone;
486 bool _justDied;
487 bool _done;
488 };

509 _Reset();

512
513 void EnterCombat(Unit* target) override
514 {
515 if (!instance->CheckRequiredBosses(DATA_VALITHRIA_DREAMWALKER, target->ToPlayer()))
516 {
517 EnterEvadeMode();
518 instance->DoCastSpellOnPlayers(LIGHT_S_HAMMER_TELEPORT);
519 return;
520 }
521
522 me->setActive(true);
523 DoZoneInCombat();
524 instance->SetBossState(DATA_VALITHRIA_DREAMWALKER, IN_PROGRESS);
525 if (Creature* valithria = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_VALITHRIA_DREAMWALKER)))
526 valithria->AI()->DoAction(ACTION_ENTER_COMBAT);
527 }

530 {
531 if (target->GetTypeId() == TYPEID_PLAYER)
532 BossAI::AttackStart(target);
533 }
534
535 bool CanAIAttack(Unit const* target) const override
536 {
537 return target->GetTypeId() == TYPEID_PLAYER;
538 }
539
540 void JustReachedHome() override
541 {
542 _JustReachedHome();
543 DoAction(ACTION_DEATH);
544 }
545
546 void DoAction(int32 action) override
547 {
548 if (action == ACTION_DEATH)
549 {
550 instance->SetBossState(DATA_VALITHRIA_DREAMWALKER, NOT_STARTED);
551 me->m_Events.AddEvent(new ValithriaDespawner(me), me->m_Events.CalculateTime(5000));
552 }
553 }
554
555 void UpdateAI(uint32 /*diff*/) override
556 {
557 if (!me->IsInCombat())
558 return;
559
560 std::list<HostileReference*> const& threatList = me->getThreatManager().getThreatList();
561 if (threatList.empty())
562 {
563 EnterEvadeMode();
564 return;
565 }
566
567 // check evade every second tick
568 _evadeCheck ^= true;
569 if (!_evadeCheck)
570 return;
571
572 // check if there is any player on threatlist, if not - evade
573 for (std::list<HostileReference*>::const_iterator itr = threatList.begin(); itr != threatList.end(); ++itr)
574 if (Unit* target = (*itr)->getTarget())
575 if (target->GetTypeId() == TYPEID_PLAYER)
576 return; // found any player, return
577
578 EnterEvadeMode();
579 }
580
581 private:
582 bool _evadeCheck;
583 };
584
585 CreatureAI* GetAI(Creature* creature) const override
586 {
587 return GetIcecrownCitadelAI<npc_green_dragon_combat_triggerAI>(creature);
588 }

720 me->VisitNearbyGridObject(100.0f, searcher);
721 for (std::list<Creature*>::iterator itr = archmages.begin(); itr != archmages.end(); ++itr)

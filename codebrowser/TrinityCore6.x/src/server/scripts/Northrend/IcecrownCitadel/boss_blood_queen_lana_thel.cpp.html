334 void UpdateAI(uint32 diff) override
335 {
336 if (!UpdateVictim() || !CheckInRoom())
337 return;
338
339 events.Update(diff);
340
341 if (me->HasUnitState(UNIT_STATE_CASTING))
342 return;
343
344 while (uint32 eventId = events.ExecuteEvent())
345 {
346 switch (eventId)
347 {
348 case EVENT_BERSERK:
349 Talk(EMOTE_BERSERK_RAID);
350 Talk(SAY_BERSERK);
351 DoCast(me, SPELL_BERSERK);
352 break;
353 case EVENT_VAMPIRIC_BITE:
354 {
355 std::list<Player*> targets;
356 SelectRandomTarget(false, &targets);
357 if (!targets.empty())
358 {
359 Unit* target = targets.front();
360 DoCast(target, SPELL_VAMPIRIC_BITE);
361 DoCastAOE(SPELL_VAMPIRIC_BITE_DUMMY, true);
362 Talk(SAY_VAMPIRIC_BITE);
363 _vampires.insert(target->GetGUID());
364 target->CastSpell(target, SPELL_PRESENCE_OF_THE_DARKFALLEN, TRIGGERED_FULL_MASK);
365 target->CastSpell(target, SPELL_PRESENCE_OF_THE_DARKFALLEN_2, TRIGGERED_FULL_MASK);
366 }
367 break;
368 }
369 case EVENT_BLOOD_MIRROR:
370 {
371 // victim can be NULL when this is processed in the same update tick as EVENT_AIR_PHASE
372 if (me->GetVictim())
373 {
374 Player* newOfftank = SelectRandomTarget(true);
375 if (newOfftank)
376 {
377 if (_offtankGUID != newOfftank->GetGUID())
378 {
379 _offtankGUID = newOfftank->GetGUID();
380
381 // both spells have SPELL_ATTR5_SINGLE_TARGET_SPELL, no manual removal needed
382 newOfftank->CastSpell(me->GetVictim(), SPELL_BLOOD_MIRROR_DAMAGE, true);
383 me->EnsureVictim()->CastSpell(newOfftank, SPELL_BLOOD_MIRROR_DUMMY, true);
384 DoCastVictim(SPELL_BLOOD_MIRROR_VISUAL);
385 if (Is25ManRaid() && newOfftank->GetQuestStatus(QUEST_BLOOD_INFUSION) == QUEST_STATUS_INCOMPLETE &&
386 newOfftank->HasAura(SPELL_UNSATED_CRAVING) && !newOfftank->HasAura(SPELL_THIRST_QUENCHED) &&
387 !newOfftank->HasAura(SPELL_GUSHING_WOUND))
388 newOfftank->CastSpell(newOfftank, SPELL_GUSHING_WOUND, TRIGGERED_FULL_MASK);
389
390 }
391 }
392 else
393 _offtankGUID.Clear();
394 }
395 events.ScheduleEvent(EVENT_BLOOD_MIRROR, 2500, EVENT_GROUP_CANCELLABLE);
396 break;
397 }
398 case EVENT_DELIRIOUS_SLASH:
399 if (!_offtankGUID.IsEmpty() && !me->HasByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER))
400 if (Player* _offtank = ObjectAccessor::GetPlayer(*me, _offtankGUID))
401 DoCast(_offtank, SPELL_DELIRIOUS_SLASH);
402 events.ScheduleEvent(EVENT_DELIRIOUS_SLASH, urand(20000, 24000), EVENT_GROUP_NORMAL);
403 break;
404 case EVENT_PACT_OF_THE_DARKFALLEN:
405 {
406 std::list<Player*> targets;
407 SelectRandomTarget(false, &targets);
408 Trinity::Containers::RandomResizeList(targets, Is25ManRaid() ? 3 : 2);
409 if (targets.size() > 1)
410 {
411 Talk(SAY_PACT_OF_THE_DARKFALLEN);
412 for (std::list<Player*>::iterator itr = targets.begin(); itr != targets.end(); ++itr)
413 DoCast(*itr, SPELL_PACT_OF_THE_DARKFALLEN);
414 }
415 events.ScheduleEvent(EVENT_PACT_OF_THE_DARKFALLEN, 30500, EVENT_GROUP_NORMAL);
416 break;
417 }
418 case EVENT_SWARMING_SHADOWS:
419 if (Player* target = SelectRandomTarget(false))
420 {
421 Talk(EMOTE_SWARMING_SHADOWS, target);
422 Talk(SAY_SWARMING_SHADOWS);
423 DoCast(target, SPELL_SWARMING_SHADOWS);
424 }
425 events.ScheduleEvent(EVENT_SWARMING_SHADOWS, 30500, EVENT_GROUP_NORMAL);
426 break;
427 case EVENT_TWILIGHT_BLOODBOLT:
428 {
429 std::list<Player*> targets;
430 SelectRandomTarget(false, &targets);
431 Trinity::Containers::RandomResizeList<Player*>(targets, uint32(Is25ManRaid() ? 4 : 2));
432 for (std::list<Player*>::iterator itr = targets.begin(); itr != targets.end(); ++itr)
433 DoCast(*itr, SPELL_TWILIGHT_BLOODBOLT);
434 DoCast(me, SPELL_TWILIGHT_BLOODBOLT_TARGET);
435 events.ScheduleEvent(EVENT_TWILIGHT_BLOODBOLT, urand(10000, 15000), EVENT_GROUP_NORMAL);
436 break;
437 }
438 case EVENT_AIR_PHASE:
439 DoStopAttack();
440 me->SetReactState(REACT_PASSIVE);
441 events.DelayEvents(10000, EVENT_GROUP_NORMAL);
442 events.CancelEventGroup(EVENT_GROUP_CANCELLABLE);
443 me->GetMotionMaster()->MovePoint(POINT_CENTER, centerPos);
444 break;
445 case EVENT_AIR_START_FLYING:
446 me->SetDisableGravity(true);
447 me->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND);
448 me->GetMotionMaster()->MovePoint(POINT_AIR, airPos);
449 break;
450 case EVENT_AIR_FLY_DOWN:
451 me->GetMotionMaster()->MovePoint(POINT_GROUND, centerPos);
452 break;
453 default:
454 break;
455 }
456 }
457
458 DoMeleeAttackIfReady();
459 }

471 private:
472 // offtank for this encounter is the player standing closest to main tank
473 Player* SelectRandomTarget(bool includeOfftank, std::list<Player*>* targetList = NULL)
474 {
475 std::list<HostileReference*> const& threatlist = me->getThreatManager().getThreatList();
476 std::list<Player*> tempTargets;
477
478 if (threatlist.empty())
479 return NULL;
480
481 for (std::list<HostileReference*>::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
482 if (Unit* refTarget = (*itr)->getTarget())
483 if (refTarget != me->GetVictim() && refTarget->GetTypeId() == TYPEID_PLAYER && (includeOfftank || (refTarget->GetGUID() != _offtankGUID)))
484 tempTargets.push_back(refTarget->ToPlayer());
485
486 if (tempTargets.empty())
487 return NULL;
488
489 if (targetList)
490 {
491 *targetList = tempTargets;
492 return NULL;
493 }
494
495 if (includeOfftank)
496 {
497 tempTargets.sort(Trinity::ObjectDistanceOrderPred(me->GetVictim()));
498 return tempTargets.front();
499 }
500
501 return Trinity::Containers::SelectRandomContainerElement(tempTargets);
502 }

525 class spell_blood_queen_vampiric_bite_SpellScript : public SpellScript
526 {
527 PrepareSpellScript(spell_blood_queen_vampiric_bite_SpellScript);
528
529 bool Validate(SpellInfo const* /*spell*/) override
530 {
531 if (!sSpellMgr->GetSpellInfo(SPELL_ESSENCE_OF_THE_BLOOD_QUEEN_PLR))
532 return false;
533 if (!sSpellMgr->GetSpellInfo(SPELL_FRENZIED_BLOODTHIRST))
534 return false;
535 if (!sSpellMgr->GetSpellInfo(SPELL_PRESENCE_OF_THE_DARKFALLEN))
536 return false;
537 return true;
538 }
539
540 SpellCastResult CheckTarget()
541 {
542 if (IsVampire(GetExplTargetUnit()))
543 {
544 SetCustomCastResultMessage(SPELL_CUSTOM_ERROR_CANT_TARGET_VAMPIRES);
545 return SPELL_FAILED_CUSTOM_ERROR;
546 }
547
548 return SPELL_CAST_OK;
549 }
550
551 void OnCast()
552 {
553 if (GetCaster()->GetTypeId() != TYPEID_PLAYER)
554 return;
555
556 GetCaster()->RemoveAura(SPELL_FRENZIED_BLOODTHIRST, ObjectGuid::Empty, 0, AURA_REMOVE_BY_ENEMY_SPELL);
557 GetCaster()->CastSpell(GetCaster(), SPELL_ESSENCE_OF_THE_BLOOD_QUEEN_PLR, TRIGGERED_FULL_MASK);
558
559 // Shadowmourne questline
560 if (Aura* aura = GetCaster()->GetAura(SPELL_GUSHING_WOUND))
561 {
562 if (aura->GetStackAmount() == 3)
563 {
564 GetCaster()->CastSpell(GetCaster(), SPELL_THIRST_QUENCHED, TRIGGERED_FULL_MASK);
565 GetCaster()->RemoveAura(aura);
566 }
567 else
568 GetCaster()->CastSpell(GetCaster(), SPELL_GUSHING_WOUND, TRIGGERED_FULL_MASK);
569 }
570
571 if (InstanceScript* instance = GetCaster()->GetInstanceScript())
572 if (Creature* bloodQueen = ObjectAccessor::GetCreature(*GetCaster(), instance->GetGuidData(DATA_BLOOD_QUEEN_LANA_THEL)))
573 bloodQueen->AI()->SetGUID(GetHitUnit()->GetGUID(), GUID_VAMPIRE);
574 }
575
576 void HandlePresence(SpellEffIndex /*effIndex*/)
577 {
578 GetHitUnit()->CastSpell(GetHitUnit(), SPELL_PRESENCE_OF_THE_DARKFALLEN, TRIGGERED_FULL_MASK);
579 GetHitUnit()->CastSpell(GetHitUnit(), SPELL_PRESENCE_OF_THE_DARKFALLEN_2, TRIGGERED_FULL_MASK);
580 }
581
582 void Register() override
583 {
584 OnCheckCast += SpellCheckCastFn(spell_blood_queen_vampiric_bite_SpellScript::CheckTarget);
585 BeforeHit += SpellHitFn(spell_blood_queen_vampiric_bite_SpellScript::OnCast);
586 OnEffectHitTarget += SpellEffectFn(spell_blood_queen_vampiric_bite_SpellScript::HandlePresence, EFFECT_1, SPELL_EFFECT_TRIGGER_SPELL);
587 }
588 };

764 remove = false;

767 if (remove)
768 {
769 if (InstanceScript* instance = GetCaster()->GetInstanceScript())

327 void MovementInform(uint32 type, uint32 point) override
328 {
329 if (type != POINT_MOTION_TYPE && type != EFFECT_MOTION_TYPE)
330 return;
331
332 switch (point)
333 {
334 case POINT_FROSTWYRM_LAND:
335 me->setActive(false);
336 me->SetCanFly(false);
337 me->SetDisableGravity(false);
338 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
339 me->SetHomePosition(SindragosaLandPos);
340 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
341 me->SetSpeed(MOVE_FLIGHT, 2.5f);
342
343 // Sindragosa enters combat as soon as she lands
344 DoZoneInCombat();
345 break;
346 case POINT_TAKEOFF:
347 events.ScheduleEvent(EVENT_AIR_MOVEMENT, 1);
348 break;
349 case POINT_AIR_PHASE:
350 me->CastCustomSpell(SPELL_ICE_TOMB_TARGET, SPELLVALUE_MAX_TARGETS, RAID_MODE<int32>(2, 5, 2, 6), NULL, TRIGGERED_FULL_MASK);
351 me->SetFacingTo(float(M_PI));
352 events.ScheduleEvent(EVENT_AIR_MOVEMENT_FAR, 1);
353 events.ScheduleEvent(EVENT_FROST_BOMB, 9000);
354 break;
355 case POINT_AIR_PHASE_FAR:
356 me->SetFacingTo(float(M_PI));
357 events.ScheduleEvent(EVENT_LAND, 30000);
358 break;
359 case POINT_LAND:
360 events.ScheduleEvent(EVENT_LAND_GROUND, 1);
361 break;
362 case POINT_LAND_GROUND:
363 {
364 me->SetCanFly(false);
365 me->SetDisableGravity(false);
366 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
367 me->SetReactState(REACT_DEFENSIVE);
368 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == POINT_MOTION_TYPE)
369 me->GetMotionMaster()->MovementExpired();
370 _isInAirPhase = false;
371 // trigger Asphyxiation
372 EntryCheckPredicate pred(NPC_ICE_TOMB);
373 summons.DoAction(ACTION_TRIGGER_ASPHYXIATION, pred);
374 break;
375 }
376 default:
377 break;
378 }

417 void UpdateAI(uint32 diff) override
418 {
419 if (!UpdateVictim() || !CheckInRoom())
420 return;
421
422 events.Update(diff);
423
424 if (me->HasUnitState(UNIT_STATE_CASTING))
425 return;
426
427 while (uint32 eventId = events.ExecuteEvent())
428 {
429 switch (eventId)
430 {
431 case EVENT_BERSERK:
432 Talk(EMOTE_BERSERK_RAID);
433 Talk(SAY_BERSERK);
434 DoCast(me, SPELL_BERSERK);
435 break;
436 case EVENT_CLEAVE:
437 DoCastVictim(SPELL_CLEAVE);
438 events.ScheduleEvent(EVENT_CLEAVE, urand(15000, 20000), EVENT_GROUP_LAND_PHASE);
439 break;
440 case EVENT_TAIL_SMASH:
441 DoCast(me, SPELL_TAIL_SMASH);
442 events.ScheduleEvent(EVENT_TAIL_SMASH, urand(27000, 32000), EVENT_GROUP_LAND_PHASE);
443 break;
444 case EVENT_FROST_BREATH:
445 DoCastVictim(_isThirdPhase ? SPELL_FROST_BREATH_P2 : SPELL_FROST_BREATH_P1);
446 events.ScheduleEvent(EVENT_FROST_BREATH, urand(20000, 25000), EVENT_GROUP_LAND_PHASE);
447 break;
448 case EVENT_UNCHAINED_MAGIC:
449 Talk(SAY_UNCHAINED_MAGIC);
450 DoCast(me, SPELL_UNCHAINED_MAGIC);
451 events.ScheduleEvent(EVENT_UNCHAINED_MAGIC, urand(30000, 35000), EVENT_GROUP_LAND_PHASE);
452 break;
453 case EVENT_ICY_GRIP:
454 DoCast(me, SPELL_ICY_GRIP);
455 events.ScheduleEvent(EVENT_BLISTERING_COLD, 1000, EVENT_GROUP_LAND_PHASE);
456 break;
457 case EVENT_BLISTERING_COLD:
458 Talk(EMOTE_WARN_BLISTERING_COLD);
459 DoCast(me, SPELL_BLISTERING_COLD);
460 events.ScheduleEvent(EVENT_BLISTERING_COLD_YELL, 5000, EVENT_GROUP_LAND_PHASE);
461 break;
462 case EVENT_BLISTERING_COLD_YELL:
463 Talk(SAY_BLISTERING_COLD);
464 break;
465 case EVENT_AIR_PHASE:
466 {
467 _isInAirPhase = true;
468 Talk(SAY_AIR_PHASE);
469 me->SetCanFly(true);
470 me->SetDisableGravity(true);
471 me->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
472 me->SetReactState(REACT_PASSIVE);
473 me->AttackStop();
474 Position pos;
475 pos.Relocate(me);
476 pos.m_positionZ += 17.0f;
477 me->GetMotionMaster()->MoveTakeoff(POINT_TAKEOFF, pos);
478 events.CancelEventGroup(EVENT_GROUP_LAND_PHASE);
479 events.ScheduleEvent(EVENT_AIR_PHASE, 110000);
480 break;
481 }
482 case EVENT_AIR_MOVEMENT:
483 me->GetMotionMaster()->MovePoint(POINT_AIR_PHASE, SindragosaAirPos);
484 break;
485 case EVENT_AIR_MOVEMENT_FAR:
486 me->GetMotionMaster()->MovePoint(POINT_AIR_PHASE_FAR, SindragosaAirPosFar);
487 break;
488 case EVENT_ICE_TOMB:
489 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, FrostBeaconSelector(me)))
490 {
491 Talk(EMOTE_WARN_FROZEN_ORB, target);
492 me->CastCustomSpell(SPELL_ICE_TOMB_TARGET, SPELLVALUE_MAX_TARGETS, 1, nullptr, TRIGGERED_FULL_MASK);
493 }
494 events.ScheduleEvent(EVENT_ICE_TOMB, urand(16000, 23000));
495 break;
496 case EVENT_FROST_BOMB:
497 {
498 float destX, destY, destZ;
499 destX = float(rand_norm()) * 75.0f + 4350.0f;
500 destY = float(rand_norm()) * 75.0f + 2450.0f;
501 destZ = 205.0f; // random number close to ground, get exact in next call
502 me->UpdateGroundPositionZ(destX, destY, destZ);
503 me->CastSpell(destX, destY, destZ, SPELL_FROST_BOMB_TRIGGER, false);
504 events.ScheduleEvent(EVENT_FROST_BOMB, urand(6000, 8000));
505 break;
506 }
507 case EVENT_LAND:
508 {
509 events.CancelEvent(EVENT_FROST_BOMB);
510 me->GetMotionMaster()->MovePoint(POINT_LAND, SindragosaFlyInPos);
511 break;
512 }
513 case EVENT_LAND_GROUND:
514 events.ScheduleEvent(EVENT_CLEAVE, urand(13000, 15000), EVENT_GROUP_LAND_PHASE);
515 events.ScheduleEvent(EVENT_TAIL_SMASH, urand(19000, 23000), EVENT_GROUP_LAND_PHASE);
516 events.ScheduleEvent(EVENT_FROST_BREATH, urand(10000, 15000), EVENT_GROUP_LAND_PHASE);
517 events.ScheduleEvent(EVENT_UNCHAINED_MAGIC, urand(12000, 17000), EVENT_GROUP_LAND_PHASE);
518 events.ScheduleEvent(EVENT_ICY_GRIP, urand(35000, 40000), EVENT_GROUP_LAND_PHASE);
519 me->GetMotionMaster()->MoveLand(POINT_LAND_GROUND, SindragosaLandPos);
520 break;
521 case EVENT_THIRD_PHASE_CHECK:
522 {
523 if (!_isInAirPhase)
524 {
525 Talk(SAY_PHASE_2);
526 events.ScheduleEvent(EVENT_ICE_TOMB, urand(7000, 10000));
527 events.RescheduleEvent(EVENT_ICY_GRIP, urand(35000, 40000));
528 DoCast(me, SPELL_MYSTIC_BUFFET, true);
529 }
530 else
531 events.ScheduleEvent(EVENT_THIRD_PHASE_CHECK, 5000);
532 break;
533 }
534 default:
535 break;
536 }
537 }
538
539 DoMeleeAttackIfReady();
540 }

612 JustDied(me);

614 return;
615 }

617 }

620 }

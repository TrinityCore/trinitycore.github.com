25#define GOSSIP_SENDER_ICC_PORT 631
26
27class icecrown_citadel_teleport : public GameObjectScript
28{
29 public:
30 icecrown_citadel_teleport() : GameObjectScript("icecrown_citadel_teleport") { }
31
32 bool OnGossipHello(Player* player, GameObject* go) override
33 {
34 if (InstanceScript* instance = go->GetInstanceScript())
35 {
36 if (instance->GetBossState(DATA_LORD_MARROWGAR) == DONE)
37 {
38 if (go->GetEntry() != GO_SCOURGE_TRANSPORTER_LIGHTSHAMMER)
39 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, "Teleport to Light's Hammer.", GOSSIP_SENDER_ICC_PORT, LIGHT_S_HAMMER_TELEPORT);
40 if (go->GetEntry() != GO_SCOURGE_TRANSPORTER_ORATORY)
41 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, "Teleport to the Oratory of the Damned.", GOSSIP_SENDER_ICC_PORT, ORATORY_OF_THE_DAMNED_TELEPORT);
42 }
43 if (instance->GetBossState(DATA_LADY_DEATHWHISPER) == DONE && go->GetEntry() != GO_SCOURGE_TRANSPORTER_RAMPART)
44 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, "Teleport to the Rampart of Skulls.", GOSSIP_SENDER_ICC_PORT, RAMPART_OF_SKULLS_TELEPORT);
45 if (instance->GetBossState(DATA_ICECROWN_GUNSHIP_BATTLE) == DONE && go->GetEntry() != GO_SCOURGE_TRANSPORTER_DEATHBRINGER)
46 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, "Teleport to the Deathbringer's Rise.", GOSSIP_SENDER_ICC_PORT, DEATHBRINGER_S_RISE_TELEPORT);
47 if (instance->GetData(DATA_UPPERSPIRE_TELE_ACT) == DONE && go->GetEntry() != GO_SCOURGE_TRANSPORTER_UPPERSPIRE)
48 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, "Teleport to the Upper Spire.", GOSSIP_SENDER_ICC_PORT, UPPER_SPIRE_TELEPORT);
49 /// @todo Gauntlet event before Sindragosa
50 if (instance->GetBossState(DATA_VALITHRIA_DREAMWALKER) == DONE && go->GetEntry() != GO_SCOURGE_TRANSPORTER_SINDRAGOSA)
51 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, "Teleport to Sindragosa's Lair", GOSSIP_SENDER_ICC_PORT, SINDRAGOSA_S_LAIR_TELEPORT);
52 }
53
54 player->SEND_GOSSIP_MENU(player->GetGossipTextId(go), go->GetGUID());
55 return true;
56 }
57
58 bool OnGossipSelect(Player* player, GameObject* /*go*/, uint32 sender, uint32 action) override
59 {
60 player->PlayerTalkClass->ClearMenus();
61 player->CLOSE_GOSSIP_MENU();
62 SpellInfo const* spell = sSpellMgr->GetSpellInfo(action);
63 if (!spell)
64 return false;
65
66 if (player->IsInCombat())
67 {
68 Spell::SendCastResult(player, spell, 0, SPELL_FAILED_AFFECTING_COMBAT);
69 return true;
70 }
71
72 if (sender == GOSSIP_SENDER_ICC_PORT)
73 player->CastSpell(player, spell, true);
74
75 return true;
76 }
77};

79class at_frozen_throne_teleport : public AreaTriggerScript
80{
81 public:
82 at_frozen_throne_teleport() : AreaTriggerScript("at_frozen_throne_teleport") { }
83
84 bool OnTrigger(Player* player, AreaTriggerEntry const* /*areaTrigger*/, bool /*entered*/) override
85 {
86 if (player->IsInCombat())
87 {
88 if (SpellInfo const* spell = sSpellMgr->GetSpellInfo(FROZEN_THRONE_TELEPORT))
89 Spell::SendCastResult(player, spell, 0, SPELL_FAILED_AFFECTING_COMBAT);
90 return true;
91 }
92
93 if (InstanceScript* instance = player->GetInstanceScript())
94 if (instance->GetBossState(DATA_PROFESSOR_PUTRICIDE) == DONE &&
95 instance->GetBossState(DATA_BLOOD_QUEEN_LANA_THEL) == DONE &&
96 instance->GetBossState(DATA_SINDRAGOSA) == DONE &&
97 instance->GetBossState(DATA_THE_LICH_KING) != IN_PROGRESS)
98 player->CastSpell(player, FROZEN_THRONE_TELEPORT, true);
99
100 return true;
101 }
102};
103
104void AddSC_icecrown_citadel_teleport()
105{
106 new icecrown_citadel_teleport();
107 new at_frozen_throne_teleport();
108}

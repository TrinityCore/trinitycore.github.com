172class AbominationDespawner
173{
174 public:
175 explicit AbominationDespawner(Unit* owner) : _owner(owner) { }
176
177 bool operator()(ObjectGuid guid)
178 {
179 if (Unit* summon = ObjectAccessor::GetUnit(*_owner, guid))
180 {
181 if (summon->GetEntry() == NPC_MUTATED_ABOMINATION_10 || summon->GetEntry() == NPC_MUTATED_ABOMINATION_25)
182 {
183 if (Vehicle* veh = summon->GetVehicleKit())
184 veh->RemoveAllPassengers(); // also despawns the vehicle
185
186 // Found unit is Mutated Abomination, remove it
187 return true;
188 }
189
190 // Found unit is not Mutated Abomintaion, leave it
191 return false;
192 }
193
194 // No unit found, remove from SummonList
195 return true;
196 }
197
198 private:
199 Unit* _owner;
200};

417 void DoAction(int32 action) override
418 {
419 switch (action)
420 {
421 case ACTION_FESTERGUT_COMBAT:
422 SetPhase(PHASE_FESTERGUT);
423 me->SetSpeed(MOVE_RUN, _baseSpeed*2.0f, true);
424 me->GetMotionMaster()->MovePoint(POINT_FESTERGUT, festergutWatchPos);
425 me->SetReactState(REACT_PASSIVE);
426 DoZoneInCombat(me);
427 if (IsHeroic())
428 events.ScheduleEvent(EVENT_FESTERGUT_GOO, urand(13000, 18000), 0, PHASE_FESTERGUT);
429 break;
430 case ACTION_FESTERGUT_GAS:
431 Talk(SAY_FESTERGUT_GASEOUS_BLIGHT);
432 DoCast(me, SPELL_RELEASE_GAS_VISUAL, true);
433 break;
434 case ACTION_FESTERGUT_DEATH:
435 events.ScheduleEvent(EVENT_FESTERGUT_DIES, 4000, 0, PHASE_FESTERGUT);
436 break;
437 case ACTION_ROTFACE_COMBAT:
438 {
439 SetPhase(PHASE_ROTFACE);
440 me->SetSpeed(MOVE_RUN, _baseSpeed*2.0f, true);
441 me->GetMotionMaster()->MovePoint(POINT_ROTFACE, rotfaceWatchPos);
442 me->SetReactState(REACT_PASSIVE);
443 _oozeFloodStage = 0;
444 DoZoneInCombat(me);
445 // init random sequence of floods
446 if (Creature* rotface = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ROTFACE)))
447 {
448 std::list<Creature*> list;
449 GetCreatureListWithEntryInGrid(list, rotface, NPC_PUDDLE_STALKER, 50.0f);
450 list.remove_if(RotfaceHeightCheck(rotface));
451 if (list.size() > 4)
452 {
453 list.sort(Trinity::ObjectDistanceOrderPred(rotface));
454 do
455 {
456 list.pop_back();
457 } while (list.size() > 4);
458 }
459
460 uint8 i = 0;
461 while (!list.empty())
462 {
463 std::list<Creature*>::iterator itr = list.begin();
464 std::advance(itr, urand(0, list.size()-1));
465 _oozeFloodDummyGUIDs[i++] = (*itr)->GetGUID();
466 list.erase(itr);
467 }
468 }
469 break;
470 }
471 case ACTION_ROTFACE_OOZE:
472 Talk(SAY_ROTFACE_OOZE_FLOOD);
473 if (Creature* dummy = ObjectAccessor::GetCreature(*me, _oozeFloodDummyGUIDs[_oozeFloodStage]))
474 dummy->CastSpell(dummy, oozeFloodSpells[_oozeFloodStage], true, NULL, NULL, me->GetGUID()); // cast from self for LoS (with prof's GUID for logs)
475 if (++_oozeFloodStage == 4)
476 _oozeFloodStage = 0;
477 break;
478 case ACTION_ROTFACE_DEATH:
479 events.ScheduleEvent(EVENT_ROTFACE_DIES, 4500, 0, PHASE_ROTFACE);
480 break;
481 case ACTION_CHANGE_PHASE:
482 me->SetSpeed(MOVE_RUN, _baseSpeed*2.0f, true);
483 events.DelayEvents(30000);
484 me->AttackStop();
485 if (!IsHeroic())
486 {
487 DoCast(me, SPELL_TEAR_GAS);
488 events.ScheduleEvent(EVENT_TEAR_GAS, 2500);
489 }
490 else
491 {
492 Talk(SAY_PHASE_TRANSITION_HEROIC);
493 DoCast(me, SPELL_UNSTABLE_EXPERIMENT, true);
494 DoCast(me, SPELL_UNSTABLE_EXPERIMENT, true);
495 // cast variables
496 if (Is25ManRaid())
497 {
498 std::list<Unit*> targetList;
499 {
500 const std::list<HostileReference*>& threatlist = me->getThreatManager().getThreatList();
501 for (std::list<HostileReference*>::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
502 if ((*itr)->getTarget()->GetTypeId() == TYPEID_PLAYER)
503 targetList.push_back((*itr)->getTarget());
504 }
505
506 size_t half = targetList.size()/2;
507 // half gets ooze variable
508 while (half < targetList.size())
509 {
510 std::list<Unit*>::iterator itr = targetList.begin();
511 advance(itr, urand(0, targetList.size() - 1));
512 (*itr)->CastSpell(*itr, SPELL_OOZE_VARIABLE, true);
513 targetList.erase(itr);
514 }
515 // and half gets gas
516 for (std::list<Unit*>::iterator itr = targetList.begin(); itr != targetList.end(); ++itr)
517 (*itr)->CastSpell(*itr, SPELL_GAS_VARIABLE, true);
518 }
519 me->GetMotionMaster()->MovePoint(POINT_TABLE, tablePos);
520 }
521 switch (_phase)
522 {
523 case PHASE_COMBAT_1:
524 SetPhase(PHASE_COMBAT_2);
525 events.ScheduleEvent(EVENT_MALLEABLE_GOO, urand(21000, 26000));
526 events.ScheduleEvent(EVENT_CHOKING_GAS_BOMB, urand(35000, 40000));
527 break;
528 case PHASE_COMBAT_2:
529 SetPhase(PHASE_COMBAT_3);
530 events.ScheduleEvent(EVENT_MUTATED_PLAGUE, 25000);
531 events.CancelEvent(EVENT_UNSTABLE_EXPERIMENT);
532 break;
533 default:
534 break;
535 }
536 break;
537 default:
538 break;
539 }

565 void UpdateAI(uint32 diff) override
566 {
567 if ((!(events.IsInPhase(PHASE_ROTFACE) || events.IsInPhase(PHASE_FESTERGUT)) && !UpdateVictim()) || !CheckInRoom())
568 return;
569
570 events.Update(diff);
571
572 if (me->HasUnitState(UNIT_STATE_CASTING))
573 return;
574
575 while (uint32 eventId = events.ExecuteEvent())
576 {
577 switch (eventId)
578 {
579 case EVENT_FESTERGUT_DIES:
580 Talk(SAY_FESTERGUT_DEATH);
581 EnterEvadeMode();
582 break;
583 case EVENT_FESTERGUT_GOO:
584 me->CastCustomSpell(SPELL_MALLEABLE_GOO_SUMMON, SPELLVALUE_MAX_TARGETS, 1, NULL, true);
585 events.ScheduleEvent(EVENT_FESTERGUT_GOO, (Is25ManRaid() ? 10000 : 30000) + urand(0, 5000), 0, PHASE_FESTERGUT);
586 break;
587 case EVENT_ROTFACE_DIES:
588 Talk(SAY_ROTFACE_DEATH);
589 EnterEvadeMode();
590 break;
591 case EVENT_ROTFACE_OOZE_FLOOD:
592 DoAction(ACTION_ROTFACE_OOZE);
593 events.ScheduleEvent(EVENT_ROTFACE_OOZE_FLOOD, 25000, 0, PHASE_ROTFACE);
594 break;
595 case EVENT_BERSERK:
596 Talk(SAY_BERSERK);
597 DoCast(me, SPELL_BERSERK2);
598 break;
599 case EVENT_SLIME_PUDDLE:
600 {
601 std::list<Unit*> targets;
602 SelectTargetList(targets, 2, SELECT_TARGET_RANDOM, 0.0f, true);
603 if (!targets.empty())
604 for (std::list<Unit*>::iterator itr = targets.begin(); itr != targets.end(); ++itr)
605 DoCast(*itr, SPELL_SLIME_PUDDLE_TRIGGER);
606 events.ScheduleEvent(EVENT_SLIME_PUDDLE, 35000);
607 break;
608 }
609 case EVENT_UNSTABLE_EXPERIMENT:
610 Talk(EMOTE_UNSTABLE_EXPERIMENT);
611 DoCast(me, SPELL_UNSTABLE_EXPERIMENT);
612 events.ScheduleEvent(EVENT_UNSTABLE_EXPERIMENT, urand(35000, 40000));
613 break;
614 case EVENT_TEAR_GAS:
615 me->GetMotionMaster()->MovePoint(POINT_TABLE, tablePos);
616 DoCast(me, SPELL_TEAR_GAS_PERIODIC_TRIGGER, true);
617 break;
618 case EVENT_RESUME_ATTACK:
619 me->SetReactState(REACT_AGGRESSIVE);
620 AttackStart(me->GetVictim());
621 // remove Tear Gas
622 me->RemoveAurasDueToSpell(SPELL_TEAR_GAS_PERIODIC_TRIGGER);
623 instance->DoRemoveAurasDueToSpellOnPlayers(71615);
624 DoCastAOE(SPELL_TEAR_GAS_CANCEL);
625 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_GAS_VARIABLE);

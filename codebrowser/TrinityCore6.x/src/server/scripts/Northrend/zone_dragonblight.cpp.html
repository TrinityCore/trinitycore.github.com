232 void StoreTargets()
233 {
234 uint8 creaturecount;
235
236 creaturecount = 0;
237
238 for (uint8 ii = 0; ii < 3; ++ii)
239 {
240 std::list<Creature*> creatureList;
241 GetCreatureListWithEntryInGrid(creatureList, me, AudienceMobs[ii], 15.0f);
242 for (Creature* creature : creatureList)
243 {
244 audienceList[creaturecount] = creature->GetGUID();
245 ++creaturecount;
246 }
247 }
248
249 if (Creature* creature = me->FindNearestCreature(NPC_IMAGE_OF_KELTHUZAD, 20.0f, true))
250 imageList[0] = creature->GetGUID();
251 if (Creature* creature = me->FindNearestCreature(NPC_IMAGE_OF_RAZUVIOUS, 20.0f, true))
252 imageList[1] = creature->GetGUID();
253 if (Creature* creature = me->FindNearestCreature(NPC_IMAGE_OF_PATCHWERK, 20.0f, true))
254 imageList[2] = creature->GetGUID();
255 if (Creature* creature = me->FindNearestCreature(NPC_IMAGE_OF_ANUBREKHAN, 20.0f, true))
256 imageList[3] = creature->GetGUID();
257 if (Creature* creature = me->FindNearestCreature(NPC_IMAGE_OF_NOTH, 20.0f, true))
258 imageList[4] = creature->GetGUID();
259 }

261 void ChangeImage(uint32 entry, uint32 model, uint8 text)
262 {
263 if (Creature* creature = ObjectAccessor::GetCreature(*me, imageList[talkWing]))
264 {
265 Talk(text);
266 creature->SetEntry(entry);
267 creature->SetDisplayId(model);
268 creature->CastSpell(creature, SPELL_HEROIC_IMAGE_CHANNEL);
269 _events.ScheduleEvent(EVENT_TALK_COMPLETE, 40000);
270 }
271 }
272
273 void TurnAudience()
274 {
275 for (uint8 i = 0; i < 10; ++i)
276 {
277 if (Creature* creature = ObjectAccessor::GetCreature(*me, audienceList[i]))
278 creature->SetFacingToObject(me);
279 }
280 }
281
282 void UpdateAI(uint32 diff) override
283 {
284 _events.Update(diff);
285
286 while (uint32 eventId = _events.ExecuteEvent())
287 {
288 switch (eventId)
289 {
290 case EVENT_START_RANDOM:
291 talkWing = urand (0, 4);
292 Talk(talkWing);
293 _events.ScheduleEvent(EVENT_MOVE_TO_POINT, 8000);
294 break;
295 case EVENT_MOVE_TO_POINT:
296 me->SetWalk(true);
297 me->GetMotionMaster()->Clear();
298 me->GetMotionMaster()->MovePoint(1, PosTalkLocations[talkWing].m_positionX, PosTalkLocations[talkWing].m_positionY, PosTalkLocations[talkWing].m_positionZ);
299 break;
300 case EVENT_TALK_COMPLETE:
301 talkWing = 5;
302 Talk(talkWing);
303 _events.ScheduleEvent(EVENT_MOVE_TO_POINT, 5000);
304 break;
305 case EVENT_GET_TARGETS:
306 StoreTargets();
307 break;
308 case EVENT_KELTHUZAD_2:
309 Talk(SAY_KELTHUZAD_2);
310 _events.ScheduleEvent(EVENT_KELTHUZAD_3, 8000);
311 break;
312 case EVENT_KELTHUZAD_3:
313 Talk(SAY_KELTHUZAD_3);
314 break;
315 case EVENT_DEATH_KNIGHTS_2:
316 Talk(SAY_DEATH_KNIGHTS_2);
317 if (Creature* creature = ObjectAccessor::GetCreature(*me, imageList[talkWing]))
318 {
319 creature->SetEntry(NPC_IMAGE_OF_BLAUMEUX);
320 creature->SetDisplayId(MODEL_IMAGE_OF_BLAUMEUX);
321 }
322 _events.ScheduleEvent(EVENT_DEATH_KNIGHTS_3, 10000);
323 break;
324 case EVENT_DEATH_KNIGHTS_3:
325 Talk(SAY_DEATH_KNIGHTS_3);
326 if (Creature* creature = ObjectAccessor::GetCreature(*me, imageList[talkWing]))
327 {
328 creature->SetEntry(NPC_IMAGE_OF_ZELIEK);
329 creature->SetDisplayId(MODEL_IMAGE_OF_ZELIEK);
330 }
331 _events.ScheduleEvent(EVENT_DEATH_KNIGHTS_4, 10000);
332 break;
333 case EVENT_DEATH_KNIGHTS_4:
334 Talk(SAY_DEATH_KNIGHTS_4);
335 break;
336 case EVENT_HEIGAN_2:
337 Talk(SAY_HEIGAN_2);
338 break;
339 default:
340 break;
341 }
342 }
343 DoMeleeAttackIfReady();
344 }
345 private:
346 EventMap _events;
347 ObjectGuid audienceList[10];
348 ObjectGuid imageList[5];
349 uint8 talkWing;
350 };

354 return new npc_commander_eligor_dawnbringerAI(creature);
355 }

529
530 void Initialize()
531 {
532 hpWarningReady = true;
533 renewRecoveryCanCheck = false;
534
535 RenewRecoveryChecker = 0;
536 }
537
538 bool hpWarningReady;
539 bool renewRecoveryCanCheck;
540
541 uint32 RenewRecoveryChecker;
542
543 void Reset() override
544 {
545 Initialize();
546 }
547
548 void UpdateAI(uint32 diff) override
549 {
550 // Check system for Health Warning should happen first time whenever get under 30%,
551 // after it should be able to happen only after recovery of last renew is fully done (20 sec),
552 // next one used won't interfere
553 if (hpWarningReady && me->GetHealthPct() <= 30.0f)
554 {
555 me->CastSpell(me, SPELL_DEFENDER_ON_LOW_HEALTH_EMOTE);
556 hpWarningReady = false;
557 }
558
559 if (renewRecoveryCanCheck)
560 {
561 if (RenewRecoveryChecker <= diff)
562 {
563 renewRecoveryCanCheck = false;
564 hpWarningReady = true;
565 }
566 else RenewRecoveryChecker -= diff;
567 }
568 }
569
570 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
571 {
572 switch (spell->Id)
573 {
574 case SPELL_WYRMREST_DEFENDER_MOUNT:
575 Talk(WHISPER_MOUNTED, me->GetCharmerOrOwner());
576 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
577 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PVP_ATTACKABLE);
578 break;
579 // Both below are for checking low hp warning
580 case SPELL_DEFENDER_ON_LOW_HEALTH_EMOTE:
581 Talk(BOSS_EMOTE_ON_LOW_HEALTH, me->GetCharmerOrOwner());
582 break;
583 case SPELL_RENEW:
584 if (!hpWarningReady && RenewRecoveryChecker <= 100)
585 {
586 RenewRecoveryChecker = 20000;
587 }
588 renewRecoveryCanCheck = true;
589 break;
590 }
591 }
592 };

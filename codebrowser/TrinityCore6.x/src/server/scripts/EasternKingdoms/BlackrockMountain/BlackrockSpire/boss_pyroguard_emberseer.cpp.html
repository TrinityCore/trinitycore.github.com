76 struct boss_pyroguard_emberseerAI : public BossAI
77 {
78 boss_pyroguard_emberseerAI(Creature* creature) : BossAI(creature, DATA_PYROGAURD_EMBERSEER) { }
79
80 void Reset() override
81 {
82 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
83 events.Reset();
84 // Apply auras on spawn and reset
85 // DoCast(me, SPELL_FIRE_SHIELD_TRIGGER); // Need to find this in old DBC if possible
86 me->RemoveAura(SPELL_EMBERSEER_FULL_STRENGTH);
87 me->RemoveAura(SPELL_EMBERSEER_GROWING);
88 me->RemoveAura(SPELL_EMBERSEER_GROWING_TRIGGER);
89 events.ScheduleEvent(EVENT_RESPAWN, 5000);
90 // Hack for missing trigger spell
91 events.ScheduleEvent(EVENT_FIRE_SHIELD, 3000);
92
93 // Open doors on reset
94 if (instance->GetBossState(DATA_PYROGAURD_EMBERSEER) == IN_PROGRESS)
95 OpenDoors(false); // Opens 2 entrance doors
96 }
97
98 void SetData(uint32 /*type*/, uint32 data) override
99 {
100 switch (data)
101 {
102 case 1:
103 events.ScheduleEvent(EVENT_PLAYER_CHECK, 5000);
104 break;
105 case 2:
106 // Close these two doors on Blackhand Incarcerators aggro
107 if (GameObject* door1 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_IN)))
108 if (door1->GetGoState() == GO_STATE_ACTIVE)
109 door1->SetGoState(GO_STATE_READY);
110 if (GameObject* door2 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_DOORS)))
111 if (door2->GetGoState() == GO_STATE_ACTIVE)
112 door2->SetGoState(GO_STATE_READY);
113 break;
114 case 3:
115 Reset();
116 break;
117 default:
118 break;
119 }
120 }
121
122 void EnterCombat(Unit* /*who*/) override
123 {
124 // ### TODO Check combat timing ###
125 events.ScheduleEvent(EVENT_FIRENOVA, 6000);
126 events.ScheduleEvent(EVENT_FLAMEBUFFET, 3000);
127 events.ScheduleEvent(EVENT_PYROBLAST, 14000);
128 }
129
130 void JustDied(Unit* /*killer*/) override
131 {
132 // Activate all the runes
133 UpdateRunes(GO_STATE_READY);
134 // Opens all 3 doors
135 OpenDoors(true);
136 // Complete encounter
137 instance->SetBossState(DATA_PYROGAURD_EMBERSEER, DONE);
138 }
139
140 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
141 {
142 if (spell->Id == SPELL_ENCAGE_EMBERSEER)
143 {
144 if (!me->GetAuraCount(SPELL_ENCAGED_EMBERSEER))
145 me->CastSpell(me, SPELL_ENCAGED_EMBERSEER);
146 }
147
148 if (spell->Id == SPELL_EMBERSEER_GROWING_TRIGGER)
149 {
150 if (me->GetAuraCount(SPELL_EMBERSEER_GROWING_TRIGGER) == 10)
151 Talk(EMOTE_TEN_STACK);
152
153 if (me->GetAuraCount(SPELL_EMBERSEER_GROWING_TRIGGER) == 20)
154 {
155 me->RemoveAura(SPELL_ENCAGED_EMBERSEER);
156 me->RemoveAura(SPELL_FREEZE_ANIM);
157 me->CastSpell(me, SPELL_EMBERSEER_FULL_STRENGTH);
158 Talk(EMOTE_FREE_OF_BONDS);
159 Talk(YELL_FREE_OF_BONDS);
160 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
161 events.ScheduleEvent(EVENT_ENTER_COMBAT, 2000);
162 }
163 }
164 }
165
166 void OpenDoors(bool Boss_Killed)
167 {
168 // These two doors reopen on reset or boss kill
169 if (GameObject* door1 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_IN)))
170 door1->SetGoState(GO_STATE_ACTIVE);
171 if (GameObject* door2 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_DOORS)))
172 door2->SetGoState(GO_STATE_ACTIVE);
173
174 // This door opens on boss kill
175 if (Boss_Killed)
176 if (GameObject* door3 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_OUT)))
177 door3->SetGoState(GO_STATE_ACTIVE);
178 }

182 // update all runes
183 if (GameObject* rune1 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_RUNE_1)))

188 rune3->SetGoState(state);
189 if (GameObject* rune4 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_RUNE_4)))
190 rune4->SetGoState(state);
191 if (GameObject* rune5 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_RUNE_5)))
192 rune5->SetGoState(state);
193 if (GameObject* rune6 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_RUNE_6)))
194 rune6->SetGoState(state);
195 if (GameObject* rune7 = me->GetMap()->GetGameObject(instance->GetGuidData(GO_EMBERSEER_RUNE_7)))
196 rune7->SetGoState(state);
197 }

199 void UpdateAI(uint32 diff) override
200 {
201 if (!UpdateVictim())
202 {
203 events.Update(diff);
204
205 while (uint32 eventId = events.ExecuteEvent())
206 {
207 switch (eventId)
208 {
209 case EVENT_RESPAWN:
210 {
211 // Respawn all Blackhand Incarcerators
212 std::list<Creature*> creatureList;
213 GetCreatureListWithEntryInGrid(creatureList, me, NPC_BLACKHAND_INCARCERATOR, 35.0f);
214 for (std::list<Creature*>::iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
215 if (Creature* creature = *itr)
216 {
217 if (!creature->IsAlive())
218 creature->Respawn();
219
220 creature->AI()->SetData(1, 1);
221 }
222 me->AddAura(SPELL_ENCAGED_EMBERSEER, me);
223 instance->SetBossState(DATA_PYROGAURD_EMBERSEER, NOT_STARTED);
224 break;
225 }
226 case EVENT_PRE_FIGHT_1:
227 {
228 // Set data on all Blackhand Incarcerators
229 std::list<Creature*> creatureList;
230 GetCreatureListWithEntryInGrid(creatureList, me, NPC_BLACKHAND_INCARCERATOR, 35.0f);
231 for (std::list<Creature*>::iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
232 {
233 if (Creature* creature = *itr)
234 creature->AI()->SetData(1, 1);
235 }
236 events.ScheduleEvent(EVENT_PRE_FIGHT_2, 32000);
237 break;
238 }
239 case EVENT_PRE_FIGHT_2:
240 me->CastSpell(me, SPELL_FREEZE_ANIM);
241 me->CastSpell(me, SPELL_EMBERSEER_GROWING);
242 Talk(EMOTE_ONE_STACK);
243 break;
244 case EVENT_FIRE_SHIELD:
245 // #### Spell isn't doing any damage ??? ####
246 DoCast(me, SPELL_FIRE_SHIELD);
247 events.ScheduleEvent(EVENT_FIRE_SHIELD, 3000);
248 break;
249 case EVENT_PLAYER_CHECK:
250 {
251 // Check to see if all players in instance have aura SPELL_EMBERSEER_START before starting event
252 bool _hasAura = true;
253 Map::PlayerList const &players = me->GetMap()->GetPlayers();
254 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
255 if (Player* player = itr->GetSource()->ToPlayer())
256 if (!player->HasAura(SPELL_EMBERSEER_OBJECT_VISUAL))
257 _hasAura = false;
258
259 if (_hasAura)
260 {
261 events.ScheduleEvent(EVENT_PRE_FIGHT_1, 1000);
262 instance->SetBossState(DATA_PYROGAURD_EMBERSEER, IN_PROGRESS);
263 }
264 break;
265 }
266 case EVENT_ENTER_COMBAT:
267 AttackStart(me->SelectNearestPlayer(30.0f));
268 break;
269 default:
270 break;
271 }
272 }
273 return;
274 }
275
276 events.Update(diff);
277
278 while (uint32 eventId = events.ExecuteEvent())
279 {
280 switch (eventId)
281 {
282 case EVENT_FIRE_SHIELD:
283 DoCast(me, SPELL_FIRE_SHIELD);
284 events.ScheduleEvent(EVENT_FIRE_SHIELD, 3000);
285 break;
286 case EVENT_FIRENOVA:
287 DoCast(me, SPELL_FIRENOVA);
288 events.ScheduleEvent(EVENT_FIRENOVA, 6000);
289 break;
290 case EVENT_FLAMEBUFFET:
291 DoCast(me, SPELL_FLAMEBUFFET);
292 events.ScheduleEvent(EVENT_FLAMEBUFFET, 14000);
293 break;
294 case EVENT_PYROBLAST:
295 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
296 DoCast(target, SPELL_PYROBLAST);
297 events.ScheduleEvent(EVENT_PYROBLAST, 15000);
298 break;
299 default:
300 break;
301 }
302 }
303 DoMeleeAttackIfReady();
304 }
305 };

326class npc_blackhand_incarcerator : public CreatureScript
327{
328public:
329 npc_blackhand_incarcerator() : CreatureScript("npc_blackhand_incarcerator") { }
330
331 struct npc_blackhand_incarceratorAI : public ScriptedAI
332 {
333 npc_blackhand_incarceratorAI(Creature* creature) : ScriptedAI(creature) { }
334
335 void Reset() override
336 {
337 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC|UNIT_FLAG_IMMUNE_TO_NPC);
338 if (Creature* Emberseer = me->FindNearestCreature(NPC_PYROGAURD_EMBERSEER, 30.0f, true))
339 Emberseer->AI()->SetData(1, 3);
340 }
341
342 void JustDied(Unit* /*killer*/) override
343 {
344 me->DespawnOrUnsummon(10000);
345 }
346
347 void SetData(uint32 data, uint32 value) override
348 {
349 if (data == 1 && value == 1)
350 {
351 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC|UNIT_FLAG_IMMUNE_TO_NPC);
352 me->InterruptSpell(CURRENT_CHANNELED_SPELL);
353 _events.CancelEvent(EVENT_ENCAGED_EMBERSEER);
354 }
355
356 if (data == 1 && value == 2)
357 _events.ScheduleEvent(EVENT_ENCAGED_EMBERSEER, 1000);
358 }
359
360 void EnterCombat(Unit* /*who*/) override
361 {
362 // Used to close doors
363 if (Creature* Emberseer = me->FindNearestCreature(NPC_PYROGAURD_EMBERSEER, 30.0f, true))
364 Emberseer->AI()->SetData(1, 2);
365
366 // Had to do this because CallForHelp will ignore any npcs without LOS
367 std::list<Creature*> creatureList;
368 GetCreatureListWithEntryInGrid(creatureList, me, NPC_BLACKHAND_INCARCERATOR, 60.0f);
369 for (std::list<Creature*>::iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
370 {
371 if (Creature* creature = *itr)
372 creature->SetInCombatWithZone(); // AI()->AttackStart(me->GetVictim());
373 }
374
375 _events.ScheduleEvent(EVENT_STRIKE, urand(8000, 16000));
376 _events.ScheduleEvent(EVENT_ENCAGE, urand(10000, 20000));
377 }
378
379 void UpdateAI(uint32 diff) override
380 {
381
382
383 if (!UpdateVictim())
384 {
385 _events.Update(diff);
386
387 while (uint32 eventId = _events.ExecuteEvent())
388 {
389 switch (eventId)
390 {
391 case EVENT_ENCAGED_EMBERSEER:
392 {
393 if (me->GetPositionX() == me->GetHomePosition().GetPositionX())
394 if (!me->HasAura(SPELL_ENCAGE_EMBERSEER))
395 if (Creature* Emberseer = me->FindNearestCreature(NPC_PYROGAURD_EMBERSEER, 30.0f, true))
396 DoCast(Emberseer, SPELL_ENCAGE_EMBERSEER);
397 break;
398
399 }
400 }
401 }
402 return;
403 }
404
405 _events.Update(diff);
406
407 while (uint32 eventId = _events.ExecuteEvent())
408 {
409 switch (eventId)
410 {
411 case EVENT_STRIKE:
412 DoCastVictim(SPELL_STRIKE, true);
413 _events.ScheduleEvent(EVENT_STRIKE, urand(14000, 23000));
414 break;
415 case EVENT_ENCAGE:
416 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), EVENT_ENCAGE, true);
417 _events.ScheduleEvent(EVENT_ENCAGE, urand(6000, 12000));
418 break;
419 default:
420 break;
421 }
422 }
423
424 DoMeleeAttackIfReady();
425 }
426
427 private:
428 EventMap _events;
429 };
430
431 CreatureAI* GetAI(Creature* creature) const override
432 {
433 return new npc_blackhand_incarceratorAI(creature);
434 }

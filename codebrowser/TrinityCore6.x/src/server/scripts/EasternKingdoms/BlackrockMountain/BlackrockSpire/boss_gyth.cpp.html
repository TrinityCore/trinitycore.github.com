23{
24 SPELL_REND_MOUNTS = 16167, // Change model
25 SPELL_CORROSIVE_ACID = 16359, // Combat (self cast)
26 SPELL_FLAMEBREATH = 16390, // Combat (Self cast)
27 SPELL_FREEZE = 16350, // Combat (Self cast)
28 SPELL_KNOCK_AWAY = 10101, // Combat
29 SPELL_SUMMON_REND = 16328 // Summons Rend near death
30};

39enum Events
40{
41 EVENT_CORROSIVE_ACID = 1,
42 EVENT_FREEZE = 2,
43 EVENT_FLAME_BREATH = 3,
44 EVENT_KNOCK_AWAY = 4,
45 EVENT_SUMMONED_1 = 5,
46 EVENT_SUMMONED_2 = 6
47};
48
49class boss_gyth : public CreatureScript
50{
51public:
52 boss_gyth() : CreatureScript("boss_gyth") { }
53
54 struct boss_gythAI : public BossAI
55 {
56 boss_gythAI(Creature* creature) : BossAI(creature, DATA_GYTH)
57 {
58 Initialize();
59 }
60
61 void Initialize()
62 {
63 SummonedRend = false;
64 }
65
66 bool SummonedRend;
67
68 void Reset() override
69 {
70 Initialize();
71 if (instance->GetBossState(DATA_GYTH) == IN_PROGRESS)
72 {
73 instance->SetBossState(DATA_GYTH, DONE);
74 me->DespawnOrUnsummon();
75 }
76 }
77
78 void EnterCombat(Unit* /*who*/) override
79 {
80 _EnterCombat();
81
82 events.ScheduleEvent(EVENT_CORROSIVE_ACID, urand(8000, 16000));
83 events.ScheduleEvent(EVENT_FREEZE, urand(8000, 16000));
84 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(8000, 16000));
85 events.ScheduleEvent(EVENT_KNOCK_AWAY, urand(12000, 18000));
86 }
87
88 void JustDied(Unit* /*killer*/) override
89 {
90 instance->SetBossState(DATA_GYTH, DONE);
91 }
92
93 void SetData(uint32 /*type*/, uint32 data) override
94 {
95 switch (data)
96 {
97 case 1:
98 events.ScheduleEvent(EVENT_SUMMONED_1, 1000);
99 break;
100 default:
101 break;
102 }
103 }
104
105 void UpdateAI(uint32 diff) override
106 {
107
108 if (!SummonedRend && HealthBelowPct(5))
109 {
110 DoCast(me, SPELL_SUMMON_REND);
111 me->RemoveAura(SPELL_REND_MOUNTS);
112 SummonedRend = true;
113 }
114
115 if (!UpdateVictim())
116 {
117 events.Update(diff);
118
119 while (uint32 eventId = events.ExecuteEvent())
120 {
121 switch (eventId)
122 {
123 case EVENT_SUMMONED_1:
124 me->AddAura(SPELL_REND_MOUNTS, me);
125 if (GameObject* portcullis = me->FindNearestGameObject(GO_DR_PORTCULLIS, 40.0f))
126 portcullis->UseDoorOrButton();
127 if (Creature* victor = me->FindNearestCreature(NPC_LORD_VICTOR_NEFARIUS, 75.0f, true))
128 victor->AI()->SetData(1, 1);
129 events.ScheduleEvent(EVENT_SUMMONED_2, 2000);
130 break;
131 case EVENT_SUMMONED_2:
132 me->GetMotionMaster()->MovePath(GYTH_PATH_1, false);
133 break;
134 default:
135 break;
136 }
137 }
138 return;
139 }
140
141 events.Update(diff);
142
143 while (uint32 eventId = events.ExecuteEvent())
144 {
145 switch (eventId)
146 {
147 case EVENT_CORROSIVE_ACID:
148 DoCast(me, SPELL_CORROSIVE_ACID);
149 events.ScheduleEvent(EVENT_CORROSIVE_ACID, urand(10000, 16000));
150 break;
151 case EVENT_FREEZE:
152 DoCast(me, SPELL_FREEZE);
153 events.ScheduleEvent(EVENT_FREEZE, urand(10000, 16000));
154 break;
155 case EVENT_FLAME_BREATH:
156 DoCast(me, SPELL_FLAMEBREATH);
157 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(10000, 16000));
158 break;
159 case EVENT_KNOCK_AWAY:
160 DoCastVictim(SPELL_KNOCK_AWAY);
161 events.ScheduleEvent(EVENT_KNOCK_AWAY, urand(14000, 20000));
162 break;
163 default:
164 break;
165 }
166 }
167 DoMeleeAttackIfReady();
168 }
169 };
170
171 CreatureAI* GetAI(Creature* creature) const override
172 {
173 return GetInstanceAI<boss_gythAI>(creature);
174 }
175};
176
177void AddSC_boss_gyth()
178{
179 new boss_gyth();
180}

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "blackrock_spire.h"
21
22enum Spells
23{
24 SPELL_REND = 16509,
25 SPELL_STRIKE = 15580,
26 SPELL_INTIMIDATING_ROAR = 16508
27};
28
29enum Says
30{
31 SAY_SUMMON = 0,
32 SAY_AGGRO = 1,
33};
34
35enum Events
36{
37 EVENT_REND = 1,
38 EVENT_STRIKE = 2,
39 EVENT_INTIMIDATING_ROAR = 3
40};
41
42class boss_urok_doomhowl : public CreatureScript
43{
44public:
45 boss_urok_doomhowl() : CreatureScript("boss_urok_doomhowl") { }
46
47 struct boss_urok_doomhowlAI : public BossAI
48 {
49 boss_urok_doomhowlAI(Creature* creature) : BossAI(creature, DATA_UROK_DOOMHOWL) { }
50
51 void Reset() override
52 {
53 _Reset();
54 }
55
56 void EnterCombat(Unit* /*who*/) override
57 {
58 _EnterCombat();
59 events.ScheduleEvent(SPELL_REND, urand(17000,20000));
60 events.ScheduleEvent(SPELL_STRIKE, urand(10000,12000));
61 Talk(SAY_AGGRO);
62 }
63
64 void JustDied(Unit* /*killer*/) override
65 {
66 _JustDied();
67 }
68
69 void UpdateAI(uint32 diff) override
70 {
71 if (!UpdateVictim())
72 return;
73
74 events.Update(diff);
75
76 if (me->HasUnitState(UNIT_STATE_CASTING))
77 return;
78
79 while (uint32 eventId = events.ExecuteEvent())
80 {
81 switch (eventId)
82 {
83 case SPELL_REND:
84 DoCastVictim(SPELL_REND);
85 events.ScheduleEvent(SPELL_REND, urand(8000,10000));
86 break;
87 case SPELL_STRIKE:
88 DoCastVictim(SPELL_STRIKE);
89 events.ScheduleEvent(SPELL_STRIKE, urand(8000,10000));
90 break;
91 default:
92 break;
93 }
94 }
95 DoMeleeAttackIfReady();
96 }
97 };
98
99 CreatureAI* GetAI(Creature* creature) const override
100 {
101 return new boss_urok_doomhowlAI(creature);
102 }
103};
104
105void AddSC_boss_urok_doomhowl()
106{
107 new boss_urok_doomhowl();
108}

74 boss_chromaggusAI(Creature* creature) : BossAI(creature, DATA_CHROMAGGUS)

77
78 Breath1_Spell = 0;
79 Breath2_Spell = 0;
80
81 // Select the 2 breaths that we are going to use until despawned
82 // 5 possiblities for the first breath, 4 for the second, 20 total possiblites
83 // This way we don't end up casting 2 of the same breath
84 // TL TL would be stupid
85 switch (urand(0, 19))
86 {
87 // B1 - Incin
88 case 0:
89 Breath1_Spell = SPELL_INCINERATE;
90 Breath2_Spell = SPELL_TIMELAPSE;
91 break;
92 case 1:
93 Breath1_Spell = SPELL_INCINERATE;
94 Breath2_Spell = SPELL_CORROSIVEACID;
95 break;

102 Breath2_Spell = SPELL_FROSTBURN;
103 break;

194 void EnterCombat(Unit* /*who*/) override
195 {
196 _EnterCombat();
197
198 events.ScheduleEvent(EVENT_SHIMMER, 0);
199 events.ScheduleEvent(EVENT_BREATH_1, 30000);
200 events.ScheduleEvent(EVENT_BREATH_2, 60000);
201 events.ScheduleEvent(EVENT_AFFLICTION, 10000);
202 events.ScheduleEvent(EVENT_FRENZY, 15000);
203 }
204
205 void UpdateAI(uint32 diff) override
206 {
207 if (!UpdateVictim())
208 return;
209
210 events.Update(diff);
211
212 if (me->HasUnitState(UNIT_STATE_CASTING))
213 return;
214
215 while (uint32 eventId = events.ExecuteEvent())
216 {
217 switch (eventId)
218 {
219 case EVENT_SHIMMER:
220 {
221 // Remove old vulnerabilty spell
222 if (CurrentVurln_Spell)
223 me->RemoveAurasDueToSpell(CurrentVurln_Spell);
224
225 // Cast new random vulnerabilty on self
226 uint32 spell = RAND(SPELL_FIRE_VULNERABILITY, SPELL_FROST_VULNERABILITY, SPELL_SHADOW_VULNERABILITY, SPELL_NATURE_VULNERABILITY, SPELL_ARCANE_VULNERABILITY);
227 DoCast(me, spell);
228 CurrentVurln_Spell = spell;
229 Talk(EMOTE_SHIMMER);
230 events.ScheduleEvent(EVENT_SHIMMER, 45000);
231 break;
232 }
233 case EVENT_BREATH_1:
234 DoCastVictim(Breath1_Spell);
235 events.ScheduleEvent(EVENT_BREATH_1, 60000);
236 break;
237 case EVENT_BREATH_2:
238 DoCastVictim(Breath2_Spell);
239 events.ScheduleEvent(EVENT_BREATH_2, 60000);
240 break;
241 case EVENT_AFFLICTION:
242 {
243 Map::PlayerList const &players = me->GetMap()->GetPlayers();
244 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
245 {
246 if (Player* player = itr->GetSource()->ToPlayer())
247 {
248 DoCast(player, RAND(SPELL_BROODAF_BLUE, SPELL_BROODAF_BLACK, SPELL_BROODAF_RED, SPELL_BROODAF_BRONZE, SPELL_BROODAF_GREEN), true);
249
250 if (player->HasAura(SPELL_BROODAF_BLUE) &&
251 player->HasAura(SPELL_BROODAF_BLACK) &&
252 player->HasAura(SPELL_BROODAF_RED) &&
253 player->HasAura(SPELL_BROODAF_BRONZE) &&
254 player->HasAura(SPELL_BROODAF_GREEN))
255 {
256 DoCast(player, SPELL_CHROMATIC_MUT_1);
257 }
258
259 }
260 }
261 }
262 events.ScheduleEvent(EVENT_AFFLICTION, 10000);
263 break;
264 case EVENT_FRENZY:
265 DoCast(me, SPELL_FRENZY);
266 events.ScheduleEvent(EVENT_FRENZY, urand(10000, 15000));
267 break;
268 }
269 }
270
271 // Enrage if not already enraged and below 20%
272 if (!Enraged && HealthBelowPct(20))
273 {
274 DoCast(me, SPELL_ENRAGE);
275 Enraged = true;
276 }
277
278 DoMeleeAttackIfReady();
279 }
280
281 private:
282 uint32 Breath1_Spell;
283 uint32 Breath2_Spell;
284 uint32 CurrentVurln_Spell;
285 bool Enraged;
286 };

64class boss_vaelastrasz : public CreatureScript
65{
66public:
67 boss_vaelastrasz() : CreatureScript("boss_vaelastrasz") { }
68
69 struct boss_vaelAI : public BossAI
70 {
71 boss_vaelAI(Creature* creature) : BossAI(creature, DATA_VAELASTRAZ_THE_CORRUPT)
72 {
73 Initialize();
74 creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
75 creature->setFaction(35);
76 creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
77 }
78
79 void Initialize()
80 {
81 PlayerGUID.Clear();
82 HasYelled = false;
83 }
84
85 void Reset() override
86 {
87 _Reset();
88
89 me->SetStandState(UNIT_STAND_STATE_DEAD);
90 Initialize();
91 }
92
93 void EnterCombat(Unit* /*who*/) override
94 {
95 _EnterCombat();
96
97 DoCast(me, SPELL_ESSENCEOFTHERED);
98 me->SetHealth(me->CountPctFromMaxHealth(30));
99 // now drop damage requirement to be able to take loot
100 me->ResetPlayerDamageReq();
101
102 events.ScheduleEvent(EVENT_CLEAVE, 10000);
103 events.ScheduleEvent(EVENT_FLAMEBREATH, 15000);
104 events.ScheduleEvent(EVENT_FIRENOVA, 20000);
105 events.ScheduleEvent(EVENT_TAILSWIPE, 11000);
106 events.ScheduleEvent(EVENT_BURNINGADRENALINE_CASTER, 15000);
107 events.ScheduleEvent(EVENT_BURNINGADRENALINE_TANK, 45000);
108 }
109
110 void BeginSpeech(Unit* target)
111 {
112 PlayerGUID = target->GetGUID();
113 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
114 events.ScheduleEvent(EVENT_SPEECH_1, 1000);
115 }
116
117 void KilledUnit(Unit* victim) override
118 {
119 if (rand32() % 5)
120 return;
121
122 Talk(SAY_KILLTARGET, victim);
123 }
124
125 void UpdateAI(uint32 diff) override
126 {
127 events.Update(diff);
128
129 // Speech
130 if (!UpdateVictim())
131 {
132 while (uint32 eventId = events.ExecuteEvent())
133 {
134 switch (eventId)
135 {
136 case EVENT_SPEECH_1:
137 Talk(SAY_LINE1);
138 me->SetStandState(UNIT_STAND_STATE_STAND);
139 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
140 events.ScheduleEvent(EVENT_SPEECH_2, 12000);
141 break;
142 case EVENT_SPEECH_2:
143 Talk(SAY_LINE2);
144 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
145 events.ScheduleEvent(EVENT_SPEECH_3, 12000);
146 break;
147 case EVENT_SPEECH_3:
148 Talk(SAY_LINE3);
149 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
150 events.ScheduleEvent(EVENT_SPEECH_4, 16000);
151 break;
152 case EVENT_SPEECH_4:
153 me->setFaction(103);
154 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
155 AttackStart(player);
156 break;
157 }
158 }
159 return;
160 }
161
162 if (me->HasUnitState(UNIT_STATE_CASTING))
163 return;
164
165 while (uint32 eventId = events.ExecuteEvent())
166 {
167 switch (eventId)
168 {
169 case EVENT_CLEAVE:
170 events.ScheduleEvent(EVENT_CLEAVE, 15000);
171 DoCastVictim(SPELL_CLEAVE);
172 break;
173 case EVENT_FLAMEBREATH:
174 DoCastVictim(SPELL_FLAMEBREATH);
175 events.ScheduleEvent(EVENT_FLAMEBREATH, urand(8000, 14000));
176 break;
177 case EVENT_FIRENOVA:
178 DoCastVictim(SPELL_FIRENOVA);
179 events.ScheduleEvent(EVENT_FIRENOVA, 15000);
180 break;
181 case EVENT_TAILSWIPE:
182 //Only cast if we are behind
183 /*if (!me->HasInArc(M_PI, me->GetVictim()))
184 {
185 DoCast(me->GetVictim(), SPELL_TAILSWIPE);
186 }*/
187 events.ScheduleEvent(EVENT_TAILSWIPE, 15000);
188 break;
189 case EVENT_BURNINGADRENALINE_CASTER:
190 {
191 Unit* target = NULL;
192
193 uint8 i = 0;
194 while (i < 3) // max 3 tries to get a random target with power_mana
195 {
196 ++i;
197 target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true); // not aggro leader
198 if (target && target->getPowerType() == POWER_MANA)
199 i = 3;
200 }
201 if (target) // cast on self (see below)
202 target->CastSpell(target, SPELL_BURNINGADRENALINE, true);
203 }
204 events.ScheduleEvent(EVENT_BURNINGADRENALINE_CASTER, 15000);
205 break;
206 case EVENT_BURNINGADRENALINE_TANK:
207 // have the victim cast the spell on himself otherwise the third effect aura will be applied to Vael instead of the player
208 me->EnsureVictim()->CastSpell(me->GetVictim(), SPELL_BURNINGADRENALINE, true);
209 events.ScheduleEvent(EVENT_BURNINGADRENALINE_TANK, 45000);
210 break;
211 }
212 }
213
214 // Yell if hp lower than 15%
215 if (HealthBelowPct(15) && !HasYelled)
216 {
217 Talk(SAY_HALFLIFE);
218 HasYelled = true;
219 }
220
221 DoMeleeAttackIfReady();
222 }
223
224 void sGossipSelect(Player* player, uint32 menuId, uint32 gossipListId) override
225 {
226 if (menuId == GOSSIP_ID && gossipListId == 0)
227 {
228 player->CLOSE_GOSSIP_MENU();
229 BeginSpeech(player);
230 }
231 }
232
233 private:
234 ObjectGuid PlayerGUID;
235 bool HasYelled;
236 };
237
238 CreatureAI* GetAI(Creature* creature) const override
239 {
240 return new boss_vaelAI(creature);
241 }
242};

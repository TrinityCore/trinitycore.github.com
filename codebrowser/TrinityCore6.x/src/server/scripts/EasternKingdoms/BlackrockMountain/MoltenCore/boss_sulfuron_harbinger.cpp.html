59class boss_sulfuron : public CreatureScript
60{
61 public:
62 boss_sulfuron() : CreatureScript("boss_sulfuron") { }
63
64 struct boss_sulfuronAI : public BossAI
65 {
66 boss_sulfuronAI(Creature* creature) : BossAI(creature, BOSS_SULFURON_HARBINGER)
67 {
68 }
69
70 void EnterCombat(Unit* victim) override
71 {
72 BossAI::EnterCombat(victim);
73 events.ScheduleEvent(EVENT_DARK_STRIKE, 10000);
74 events.ScheduleEvent(EVENT_DEMORALIZING_SHOUT, 15000);
75 events.ScheduleEvent(EVENT_INSPIRE, 13000);
76 events.ScheduleEvent(EVENT_KNOCKDOWN, 6000);
77 events.ScheduleEvent(EVENT_FLAMESPEAR, 2000);
78 }
79
80 void UpdateAI(uint32 diff) override
81 {
82 if (!UpdateVictim())
83 return;
84
85 events.Update(diff);
86
87 if (me->HasUnitState(UNIT_STATE_CASTING))
88 return;
89
90 while (uint32 eventId = events.ExecuteEvent())
91 {
92 switch (eventId)
93 {
94 case EVENT_DARK_STRIKE:
95 DoCast(me, SPELL_DARK_STRIKE);
96 events.ScheduleEvent(EVENT_DARK_STRIKE, urand(15000, 18000));
97 break;
98 case EVENT_DEMORALIZING_SHOUT:
99 DoCastVictim(SPELL_DEMORALIZING_SHOUT);
100 events.ScheduleEvent(EVENT_DEMORALIZING_SHOUT, urand(15000, 20000));
101 break;
102 case EVENT_INSPIRE:
103 {
104 std::list<Creature*> healers = DoFindFriendlyMissingBuff(45.0f, SPELL_INSPIRE);
105 if (!healers.empty())
106 DoCast(Trinity::Containers::SelectRandomContainerElement(healers), SPELL_INSPIRE);
107
108 DoCast(me, SPELL_INSPIRE);
109 events.ScheduleEvent(EVENT_INSPIRE, urand(20000, 26000));
110 break;
111 }
112 case EVENT_KNOCKDOWN:
113 DoCastVictim(SPELL_KNOCKDOWN);
114 events.ScheduleEvent(EVENT_KNOCKDOWN, urand(12000, 15000));
115 break;
116 case EVENT_FLAMESPEAR:
117 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
118 DoCast(target, SPELL_FLAMESPEAR);
119 events.ScheduleEvent(EVENT_FLAMESPEAR, urand(12000, 16000));
120 break;
121 default:
122 break;
123 }
124 }
125
126 DoMeleeAttackIfReady();
127 }
128 };
129
130 CreatureAI* GetAI(Creature* creature) const override
131 {
132 return new boss_sulfuronAI(creature);
133 }
134};

136class npc_flamewaker_priest : public CreatureScript
137{
138 public:
139 npc_flamewaker_priest() : CreatureScript("npc_flamewaker_priest") { }
140
141 struct npc_flamewaker_priestAI : public ScriptedAI
142 {
143 npc_flamewaker_priestAI(Creature* creature) : ScriptedAI(creature)
144 {
145 }
146
147 void Reset() override
148 {
149 events.Reset();
150 }
151
152 void JustDied(Unit* /*killer*/) override
153 {
154 events.Reset();
155 }
156
157 void EnterCombat(Unit* victim) override
158 {
159 ScriptedAI::EnterCombat(victim);
160 events.ScheduleEvent(EVENT_HEAL, urand(15000, 30000));
161 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, 2000);
162 events.ScheduleEvent(EVENT_IMMOLATE, 8000);
163 }
164
165 void UpdateAI(uint32 diff) override
166 {
167 if (!UpdateVictim())
168 return;
169
170 events.Update(diff);
171
172 if (me->HasUnitState(UNIT_STATE_CASTING))
173 return;
174
175 while (uint32 eventId = events.ExecuteEvent())
176 {
177 switch (eventId)
178 {
179 case EVENT_HEAL:
180 if (Unit* target = DoSelectLowestHpFriendly(60.0f, 1))
181 DoCast(target, SPELL_HEAL);
182 events.ScheduleEvent(EVENT_HEAL, urand(15000, 20000));
183 break;
184 case EVENT_SHADOW_WORD_PAIN:
185 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_SHADOWWORDPAIN))
186 DoCast(target, SPELL_SHADOWWORDPAIN);
187 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(18000, 26000));
188 break;
189 case EVENT_IMMOLATE:
190 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_IMMOLATE))
191 DoCast(target, SPELL_IMMOLATE);
192 events.ScheduleEvent(EVENT_IMMOLATE, urand(15000, 25000));
193 break;
194 default:
195 break;
196 }
197 }
198
199 DoMeleeAttackIfReady();
200 }
201
202 private:
203 EventMap events;
204 };
205
206 CreatureAI* GetAI(Creature* creature) const override
207 {
208 return new npc_flamewaker_priestAI(creature);
209 }
210};

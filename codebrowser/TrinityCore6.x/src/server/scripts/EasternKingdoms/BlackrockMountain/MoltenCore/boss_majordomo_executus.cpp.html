70class boss_majordomo : public CreatureScript
71{
72 public:
73 boss_majordomo() : CreatureScript("boss_majordomo") { }
74
75 struct boss_majordomoAI : public BossAI
76 {
77 boss_majordomoAI(Creature* creature) : BossAI(creature, BOSS_MAJORDOMO_EXECUTUS)
78 {
79 }
80
81 void KilledUnit(Unit* /*victim*/) override
82 {
83 if (urand(0, 99) < 25)
84 Talk(SAY_SLAY);
85 }
86
87 void EnterCombat(Unit* who) override
88 {
89 BossAI::EnterCombat(who);
90 Talk(SAY_AGGRO);
91 events.ScheduleEvent(EVENT_MAGIC_REFLECTION, 30000);
92 events.ScheduleEvent(EVENT_DAMAGE_REFLECTION, 15000);
93 events.ScheduleEvent(EVENT_BLAST_WAVE, 10000);
94 events.ScheduleEvent(EVENT_TELEPORT, 20000);
95 }
96
97 void UpdateAI(uint32 diff) override
98 {
99 if (instance->GetBossState(BOSS_MAJORDOMO_EXECUTUS) != DONE)
100 {
101 if (!UpdateVictim())
102 return;
103
104 events.Update(diff);
105
106 if (!me->FindNearestCreature(NPC_FLAMEWAKER_HEALER, 100.0f) && !me->FindNearestCreature(NPC_FLAMEWAKER_ELITE, 100.0f))
107 {
108 instance->UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, me->GetEntry(), me);
109 me->setFaction(35);
110 EnterEvadeMode();
111 Talk(SAY_DEFEAT);
112 _JustDied();
113 events.ScheduleEvent(EVENT_OUTRO_1, 32000);
114 return;
115 }
116
117 if (me->HasUnitState(UNIT_STATE_CASTING))
118 return;
119
120 if (HealthBelowPct(50))
121 DoCast(me, SPELL_AEGIS_OF_RAGNAROS, true);
122
123 while (uint32 eventId = events.ExecuteEvent())
124 {
125 switch (eventId)
126 {
127 case EVENT_MAGIC_REFLECTION:
128 DoCast(me, SPELL_MAGIC_REFLECTION);
129 events.ScheduleEvent(EVENT_MAGIC_REFLECTION, 30000);
130 break;
131 case EVENT_DAMAGE_REFLECTION:
132 DoCast(me, SPELL_DAMAGE_REFLECTION);
133 events.ScheduleEvent(EVENT_DAMAGE_REFLECTION, 30000);
134 break;
135 case EVENT_BLAST_WAVE:
136 DoCastVictim(SPELL_BLAST_WAVE);
137 events.ScheduleEvent(EVENT_BLAST_WAVE, 10000);
138 break;
139 case EVENT_TELEPORT:
140 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
141 DoCast(target, SPELL_TELEPORT);
142 events.ScheduleEvent(EVENT_TELEPORT, 20000);
143 break;
144 default:
145 break;
146 }
147 }
148
149 DoMeleeAttackIfReady();
150 }
151 else
152 {
153 events.Update(diff);
154
155 while (uint32 eventId = events.ExecuteEvent())
156 {
157 switch (eventId)
158 {
159 case EVENT_OUTRO_1:
160 me->NearTeleportTo(RagnarosTelePos.GetPositionX(), RagnarosTelePos.GetPositionY(), RagnarosTelePos.GetPositionZ(), RagnarosTelePos.GetOrientation());
161 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
162 break;
163 case EVENT_OUTRO_2:
164 instance->instance->SummonCreature(NPC_RAGNAROS, RagnarosSummonPos);
165 break;
166 case EVENT_OUTRO_3:
167 Talk(SAY_ARRIVAL2_MAJ);
168 break;
169 default:
170 break;
171 }
172 }
173 }
174 }
175
176 void DoAction(int32 action) override
177 {
178 if (action == ACTION_START_RAGNAROS)
179 {
180 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
181 Talk(SAY_SUMMON_MAJ);
182 events.ScheduleEvent(EVENT_OUTRO_2, 8000);
183 events.ScheduleEvent(EVENT_OUTRO_3, 24000);
184 }
185 else if (action == ACTION_START_RAGNAROS_ALT)
186 {
187 me->setFaction(35);
188 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
189 }
190 }
191 };
192
193 bool OnGossipHello(Player* player, Creature* creature) override
194 {
195 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_SELECT, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
196 player->SEND_GOSSIP_MENU(GOSSIP_HELLO, creature->GetGUID());
197 return true;
198 }
199
200 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 /*action*/) override
201 {
202 player->CLOSE_GOSSIP_MENU();
203 creature->AI()->DoAction(ACTION_START_RAGNAROS);
204 return true;
205 }
206
207 CreatureAI* GetAI(Creature* creature) const override
208 {
209 return GetInstanceAI<boss_majordomoAI>(creature);
210 }
211};

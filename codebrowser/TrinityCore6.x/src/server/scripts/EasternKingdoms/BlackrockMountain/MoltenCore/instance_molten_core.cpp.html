37class instance_molten_core : public InstanceMapScript
38{
39 public:
40 instance_molten_core() : InstanceMapScript("instance_molten_core", 409) { }
41
42 struct instance_molten_core_InstanceMapScript : public InstanceScript
43 {
44 instance_molten_core_InstanceMapScript(Map* map) : InstanceScript(map)
45 {
46 SetHeaders(DataHeader);
47 SetBossNumber(MAX_ENCOUNTER);
48 _executusSchedule = false;
49 _ragnarosAddDeaths = 0;
50 }
51
52 void OnPlayerEnter(Player* /*player*/) override
53 {
54 if (_executusSchedule)
55 SummonMajordomoExecutus();
56 }
57
58 void OnCreatureCreate(Creature* creature) override
59 {
60 switch (creature->GetEntry())
61 {
62 case NPC_GOLEMAGG_THE_INCINERATOR:
63 _golemaggTheIncineratorGUID = creature->GetGUID();
64 break;
65 case NPC_MAJORDOMO_EXECUTUS:
66 _majordomoExecutusGUID = creature->GetGUID();
67 break;
68 default:
69 break;
70 }
71 }
72
73 void OnGameObjectCreate(GameObject* go) override
74 {
75 switch (go->GetEntry())
76 {
77 case GO_CACHE_OF_THE_FIRELORD:
78 _cacheOfTheFirelordGUID = go->GetGUID();
79 break;
80 default:
81 break;
82 }
83 }
84
85 void SetData(uint32 type, uint32 data) override
86 {
87 if (type == DATA_RAGNAROS_ADDS)
88 {
89 if (data == 1)
90 ++_ragnarosAddDeaths;
91 else if (data == 0)
92 _ragnarosAddDeaths = 0;
93 }
94 }
95
96 uint32 GetData(uint32 type) const override
97 {
98 switch (type)
99 {
100 case DATA_RAGNAROS_ADDS:
101 return _ragnarosAddDeaths;
102 }
103
104 return 0;
105 }
106
107 ObjectGuid GetGuidData(uint32 type) const override
108 {
109 switch (type)
110 {
111 case BOSS_GOLEMAGG_THE_INCINERATOR:
112 return _golemaggTheIncineratorGUID;
113 case BOSS_MAJORDOMO_EXECUTUS:
114 return _majordomoExecutusGUID;
115 }
116
117 return ObjectGuid::Empty;
118 }
119
120 bool SetBossState(uint32 bossId, EncounterState state) override
121 {
122 if (!InstanceScript::SetBossState(bossId, state))
123 return false;
124
125 if (state == DONE && bossId < BOSS_MAJORDOMO_EXECUTUS)
126 if (CheckMajordomoExecutus())
127 SummonMajordomoExecutus();
128
129 if (bossId == BOSS_MAJORDOMO_EXECUTUS && state == DONE)
130 DoRespawnGameObject(_cacheOfTheFirelordGUID, 7 * DAY);
131
132 return true;
133 }
134
135 void SummonMajordomoExecutus()
136 {
137 _executusSchedule = false;
138 if (!_majordomoExecutusGUID.IsEmpty())
139 return;
140
141 if (GetBossState(BOSS_MAJORDOMO_EXECUTUS) != DONE)
142 {
143 instance->SummonCreature(NPC_MAJORDOMO_EXECUTUS, SummonPositions[0]);
144 instance->SummonCreature(NPC_FLAMEWAKER_HEALER, SummonPositions[1]);
145 instance->SummonCreature(NPC_FLAMEWAKER_HEALER, SummonPositions[2]);
146 instance->SummonCreature(NPC_FLAMEWAKER_HEALER, SummonPositions[3]);
147 instance->SummonCreature(NPC_FLAMEWAKER_HEALER, SummonPositions[4]);
148 instance->SummonCreature(NPC_FLAMEWAKER_ELITE, SummonPositions[5]);
149 instance->SummonCreature(NPC_FLAMEWAKER_ELITE, SummonPositions[6]);
150 instance->SummonCreature(NPC_FLAMEWAKER_ELITE, SummonPositions[7]);
151 instance->SummonCreature(NPC_FLAMEWAKER_ELITE, SummonPositions[8]);
152 }
153 else if (TempSummon* summon = instance->SummonCreature(NPC_MAJORDOMO_EXECUTUS, RagnarosTelePos))
154 summon->AI()->DoAction(ACTION_START_RAGNAROS_ALT);
155 }
156
157 bool CheckMajordomoExecutus() const
158 {
159 if (GetBossState(BOSS_RAGNAROS) == DONE)
160 return false;
161
162 for (uint8 i = 0; i < BOSS_MAJORDOMO_EXECUTUS; ++i)
163 if (GetBossState(i) != DONE)
164 return false;
165
166 return true;
167 }
168
169 void ReadSaveDataMore(std::istringstream& /*data*/) override
170 {
171 if (CheckMajordomoExecutus())
172 _executusSchedule = true;
173 }
174
175 private:
176 ObjectGuid _golemaggTheIncineratorGUID;
177 ObjectGuid _majordomoExecutusGUID;
178 ObjectGuid _cacheOfTheFirelordGUID;
179 bool _executusSchedule;
180 uint8 _ragnarosAddDeaths;
181 };
182
183 InstanceScript* GetInstanceScript(InstanceMap* map) const override
184 {
185 return new instance_molten_core_InstanceMapScript(map);
186 }
187};
188
189void AddSC_instance_molten_core()
190{
191 new instance_molten_core();
192}

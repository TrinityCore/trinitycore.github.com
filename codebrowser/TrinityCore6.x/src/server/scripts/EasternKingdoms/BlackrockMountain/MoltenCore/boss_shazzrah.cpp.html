23enum Spells
24{
25 SPELL_ARCANE_EXPLOSION = 19712,
26 SPELL_SHAZZRAH_CURSE = 19713,
27 SPELL_MAGIC_GROUNDING = 19714,
28 SPELL_COUNTERSPELL = 19715,
29 SPELL_SHAZZRAH_GATE_DUMMY = 23138, // Teleports to and attacks a random target.
30 SPELL_SHAZZRAH_GATE = 23139,
31};
32
33enum Events
34{
35 EVENT_ARCANE_EXPLOSION = 1,
36 EVENT_ARCANE_EXPLOSION_TRIGGERED = 2,
37 EVENT_SHAZZRAH_CURSE = 3,
38 EVENT_MAGIC_GROUNDING = 4,
39 EVENT_COUNTERSPELL = 5,
40 EVENT_SHAZZRAH_GATE = 6,
41};
42
43class boss_shazzrah : public CreatureScript
44{
45 public:
46 boss_shazzrah() : CreatureScript("boss_shazzrah") { }
47
48 struct boss_shazzrahAI : public BossAI
49 {
50 boss_shazzrahAI(Creature* creature) : BossAI(creature, BOSS_SHAZZRAH) { }
51
52 void EnterCombat(Unit* target) override
53 {
54 BossAI::EnterCombat(target);
55 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, 6000);
56 events.ScheduleEvent(EVENT_SHAZZRAH_CURSE, 10000);
57 events.ScheduleEvent(EVENT_MAGIC_GROUNDING, 24000);
58 events.ScheduleEvent(EVENT_COUNTERSPELL, 15000);
59 events.ScheduleEvent(EVENT_SHAZZRAH_GATE, 45000);
60 }
61
62 void UpdateAI(uint32 diff) override
63 {
64 if (!UpdateVictim())
65 return;
66
67 events.Update(diff);
68
69 if (me->HasUnitState(UNIT_STATE_CASTING))
70 return;
71
72 while (uint32 eventId = events.ExecuteEvent())
73 {
74 switch (eventId)
75 {
76 case EVENT_ARCANE_EXPLOSION:
77 DoCastVictim(SPELL_ARCANE_EXPLOSION);
78 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(4000, 7000));
79 break;
80 // Triggered subsequent to using "Gate of Shazzrah".
81 case EVENT_ARCANE_EXPLOSION_TRIGGERED:
82 DoCastVictim(SPELL_ARCANE_EXPLOSION);
83 break;
84 case EVENT_SHAZZRAH_CURSE:
85 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_SHAZZRAH_CURSE))
86 DoCast(target, SPELL_SHAZZRAH_CURSE);
87 events.ScheduleEvent(EVENT_SHAZZRAH_CURSE, urand(25000, 30000));
88 break;
89 case EVENT_MAGIC_GROUNDING:
90 DoCast(me, SPELL_MAGIC_GROUNDING);
91 events.ScheduleEvent(EVENT_MAGIC_GROUNDING, 35000);
92 break;
93 case EVENT_COUNTERSPELL:
94 DoCastVictim(SPELL_COUNTERSPELL);
95 events.ScheduleEvent(EVENT_COUNTERSPELL, urand(16000, 20000));
96 break;
97 case EVENT_SHAZZRAH_GATE:
98 DoResetThreat();
99 DoCastAOE(SPELL_SHAZZRAH_GATE_DUMMY);
100 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION_TRIGGERED, 2000);
101 events.RescheduleEvent(EVENT_ARCANE_EXPLOSION, urand(3000, 6000));
102 events.ScheduleEvent(EVENT_SHAZZRAH_GATE, 45000);
103 break;
104 default:
105 break;
106 }
107 }
108
109 DoMeleeAttackIfReady();
110 }
111 };
112
113 CreatureAI* GetAI(Creature* creature) const override
114 {
115 return new boss_shazzrahAI(creature);
116 }
117};

119// 23138 - Gate of Shazzrah
120class spell_shazzrah_gate_dummy : public SpellScriptLoader
121{
122 public:
123 spell_shazzrah_gate_dummy() : SpellScriptLoader("spell_shazzrah_gate_dummy") { }
124
125 class spell_shazzrah_gate_dummy_SpellScript : public SpellScript
126 {
127 PrepareSpellScript(spell_shazzrah_gate_dummy_SpellScript);
128
129 bool Validate(SpellInfo const* /*spellInfo*/) override
130 {
131 if (!sSpellMgr->GetSpellInfo(SPELL_SHAZZRAH_GATE))
132 return false;
133 return true;
134 }
135
136 void FilterTargets(std::list<WorldObject*>& targets)
137 {
138 if (targets.empty())
139 return;
140
141 WorldObject* target = Trinity::Containers::SelectRandomContainerElement(targets);
142 targets.clear();
143 targets.push_back(target);
144 }
145
146 void HandleScript(SpellEffIndex /*effIndex*/)
147 {
148 if (Unit* target = GetHitUnit())
149 {
150 target->CastSpell(GetCaster(), SPELL_SHAZZRAH_GATE, true);
151 if (Creature* creature = GetCaster()->ToCreature())
152 creature->AI()->AttackStart(target); // Attack the target which caster will teleport to.
153 }
154 }
155
156 void Register() override
157 {
158 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_shazzrah_gate_dummy_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
159 OnEffectHitTarget += SpellEffectFn(spell_shazzrah_gate_dummy_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_DUMMY);
160 }
161 };
162
163 SpellScript* GetSpellScript() const override
164 {
165 return new spell_shazzrah_gate_dummy_SpellScript();
166 }
167};

169void AddSC_boss_shazzrah()

171 new boss_shazzrah();
172 new spell_shazzrah_gate_dummy();
173}

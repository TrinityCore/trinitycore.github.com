22enum Romogg
23{
24 YELL_AGGRO = 0,
25 YELL_KILL = 1,
26 YELL_SKULLCRACKER = 2,
27 YELL_DEATH = 3,
28 EMOTE_CALL_FOR_HELP = 4,
29 EMOTE_SKULLCRACKER = 5,
30 SPELL_CALL_FOR_HELP = 82137, // Needs Scripting
31 SPELL_CHAINS_OF_WOE = 75539,
32 SPELL_QUAKE = 75272,
33 SPELL_SKULLCRACKER = 75543,
34 SPELL_WOUNDING_STRIKE = 75571,
35 EVENT_CHAINS_OF_WOE = 1,
36 EVENT_QUAKE = 2, // Not yet sure of timing
37 EVENT_SKULLCRACKER = 3,
38 EVENT_WOUNDING_STRIKE = 4,
39 TYPE_RAZ = 1,
40 DATA_ROMOGG_DEAD = 1
41};
42
43Position const SummonPos = { 249.2639f, 949.1614f, 191.7866f, 3.141593f };
44
45class boss_romogg_bonecrusher : public CreatureScript
46{
47 public:
48 boss_romogg_bonecrusher() : CreatureScript("boss_romogg_bonecrusher") { }
49
50 struct boss_romogg_bonecrusherAI : public BossAI
51 {
52 boss_romogg_bonecrusherAI(Creature* creature) : BossAI(creature, DATA_ROMOGG_BONECRUSHER)
53 {
54 me->SummonCreature(NPC_RAZ_THE_CRAZED, SummonPos, TEMPSUMMON_MANUAL_DESPAWN, 200000);
55 }
56
57 void Reset() override
58 {
59 _Reset();
60 }
61
62 void JustDied(Unit* /*killer*/) override
63 {
64 _JustDied();
65 Talk(YELL_DEATH);
66
67 if (Creature* raz = instance->GetCreature(DATA_RAZ_THE_CRAZED))
68 raz->AI()->SetData(TYPE_RAZ, DATA_ROMOGG_DEAD);
69 }
70
71 void KilledUnit(Unit* who) override
72 {
73 if (who->GetTypeId() == TYPEID_PLAYER)
74 Talk(YELL_KILL);
75 }
76
77 void EnterCombat(Unit* /*who*/) override
78 {
79 _EnterCombat();
80 events.ScheduleEvent(EVENT_CHAINS_OF_WOE, urand(22000, 32000));
81 events.ScheduleEvent(EVENT_WOUNDING_STRIKE, urand(26000, 32000));
82 events.ScheduleEvent(EVENT_QUAKE, 45000);
83 Talk(YELL_AGGRO);
84 Talk(EMOTE_CALL_FOR_HELP);
85 DoCast(me, SPELL_CALL_FOR_HELP);
86 }
87
88 void UpdateAI(uint32 diff) override
89 {
90 if (!UpdateVictim())
91 return;
92
93 events.Update(diff);
94
95 if (me->HasUnitState(UNIT_STATE_CASTING))
96 return;
97
98 while (uint32 eventId = events.ExecuteEvent())
99 {
100 switch (eventId)
101 {
102 case EVENT_CHAINS_OF_WOE:
103 Talk(YELL_SKULLCRACKER);
104 DoCast(me, SPELL_CHAINS_OF_WOE);
105 events.ScheduleEvent(EVENT_CHAINS_OF_WOE, urand(22000, 32000));
106 events.ScheduleEvent(EVENT_SKULLCRACKER, 3000);
107 break;
108 case EVENT_SKULLCRACKER:
109 Talk(EMOTE_SKULLCRACKER);
110 DoCast(me, SPELL_SKULLCRACKER);
111 break;
112 case EVENT_QUAKE:
113 DoCast(me, SPELL_QUAKE);
114 events.ScheduleEvent(EVENT_QUAKE, urand(32000, 40000));
115 break;
116 case EVENT_WOUNDING_STRIKE:
117 DoCastVictim(SPELL_WOUNDING_STRIKE, true);
118 events.ScheduleEvent(EVENT_WOUNDING_STRIKE, urand(26000, 32000));
119 break;
120 default:
121 break;
122 }
123 }
124
125 DoMeleeAttackIfReady();
126 }
127 };
128
129 CreatureAI* GetAI(Creature* creature) const override
130 {
131 return GetBlackrockCavernsAI<boss_romogg_bonecrusherAI>(creature);
132 }
133};
134
135void AddSC_boss_romogg_bonecrusher()
136{
137 new boss_romogg_bonecrusher();
138}

18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "blackrock_caverns.h"
21
22ObjectData const creatureData[] =
23{
24 { NPC_RAZ_THE_CRAZED, DATA_RAZ_THE_CRAZED },
25 { 0, 0 }
26};
27
28class instance_blackrock_caverns : public InstanceMapScript
29{
30 public:
31 instance_blackrock_caverns(): InstanceMapScript(BCScriptName, 645) { }
32
33 struct instance_blackrock_caverns_InstanceMapScript : public InstanceScript
34 {
35 instance_blackrock_caverns_InstanceMapScript(Map* map) : InstanceScript(map)
36 {
37 SetHeaders(DataHeader);
38 SetBossNumber(EncounterCount);
39 LoadObjectData(creatureData, nullptr);
40 }
41
42 bool SetBossState(uint32 type, EncounterState state) override
43 {
44 if (!InstanceScript::SetBossState(type, state))
45 return false;
46
47 switch (type)
48 {
49 case DATA_ROMOGG_BONECRUSHER:
50 case DATA_CORLA:
51 case DATA_KARSH_STEELBENDER:
52 case DATA_BEAUTY:
53 case DATA_ASCENDANT_LORD_OBSIDIUS:
54 break;
55 default:
56 break;
57 }
58
59 return true;
60 }
61 };
62
63 InstanceScript* GetInstanceScript(InstanceMap* map) const override
64 {
65 return new instance_blackrock_caverns_InstanceMapScript(map);
66 }
67};
68
69void AddSC_instance_blackrock_caverns()
70{
71 new instance_blackrock_caverns();
72}

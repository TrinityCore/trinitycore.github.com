105class npc_twilight_flame_caller : public CreatureScript
106{
107public: npc_twilight_flame_caller() : CreatureScript("npc_twilight_flame_caller") { }
108
109 struct npc_twilight_flame_callerAI : public ScriptedAI
110 {
111 npc_twilight_flame_callerAI(Creature* creature) : ScriptedAI(creature), _instance(creature->GetInstanceScript()), _summons(me)
112 {
113 Initialize();
114 }
115
116 void Initialize()
117 {
118 _combatPhase = false;
119 }
120
121 void Reset() override
122 {
123 Initialize();
124 _flamecaller1GUID.Clear();
125 _flamecaller2GUID.Clear();
126
127 if (me->GetPositionX() > 172 && me->GetPositionX() < 173 && me->GetPositionY() > 1086 && me->GetPositionY() < 1087)
128 {
129 _flamecaller1GUID = me->GetGUID();
130 me->SummonCreature(NPC_FIRE_CYCLONE, SummonPos[0], TEMPSUMMON_CORPSE_DESPAWN, 0);
131 me->SummonCreature(NPC_FIRE_CYCLONE, SummonPos[1], TEMPSUMMON_CORPSE_DESPAWN, 0);
132 me->SummonCreature(NPC_FIRE_CYCLONE, SummonPos[2], TEMPSUMMON_CORPSE_DESPAWN, 0);
133 }
134 if (me->GetPositionX() > 247 && me->GetPositionX() < 248 && me->GetPositionY() > 1081 && me->GetPositionY() < 1082)
135 {
136 _flamecaller2GUID = me->GetGUID();
137 me->SummonCreature(NPC_FIRE_CYCLONE, SummonPos[3], TEMPSUMMON_CORPSE_DESPAWN, 0);
138 me->SummonCreature(NPC_FIRE_CYCLONE, SummonPos[4], TEMPSUMMON_CORPSE_DESPAWN, 0);
139 me->SummonCreature(NPC_FIRE_CYCLONE, SummonPos[5], TEMPSUMMON_CORPSE_DESPAWN, 0);
140 }
141
142 _events.ScheduleEvent(EVENT_CHANNEL, 100);
143 }
144
145 void JustSummoned(Creature* summoned) override
146 {
147 _summons.Summon(summoned);
148 }
149
150 void JustDied(Unit* /*killer*/) override
151 {
152 _summons.DespawnAll();
153 }
154
155 void EnterCombat(Unit* /*who*/) override
156 {
157 _events.Reset();
158 _combatPhase = true;
159 _events.ScheduleEvent(EVENT_BLAST_WAVE, urand(8000, 10000));
160 _events.ScheduleEvent(EVENT_CALL_FLAMES, urand(10000, 14000));
161 }
162
163 void UpdateAI(uint32 diff) override
164 {
165 _events.Update(diff);
166
167 if (!_combatPhase)
168 {
169 while (uint32 eventId = _events.ExecuteEvent())
170 {
171 switch (eventId)
172 {
173 case EVENT_CHANNEL:
174 if (me->GetGUID() == _flamecaller1GUID)
175 DoCast(me, SPELL_FIRE_CHANNELING_1);
176 if (me->GetGUID() == _flamecaller2GUID)
177 DoCast(me, SPELL_FIRE_CHANNELING_2);
178 _events.ScheduleEvent(EVENT_CHANNEL, 12000);
179 break;
180 default:
181 break;
182 }
183 }
184 return;
185 }
186
187 if (!UpdateVictim())
188 return;
189
190 while (uint32 eventId = _events.ExecuteEvent())
191 {
192 switch (eventId)
193 {
194 case EVENT_BLAST_WAVE:
195 DoCast(me, SPELL_BLAST_WAVE);
196 _events.ScheduleEvent(EVENT_BLAST_WAVE, 16000, 20000);
197 break;
198 case EVENT_CALL_FLAMES:
199 DoCast(me, SPELL_CALL_FLAMES);
200 _events.ScheduleEvent(EVENT_CALL_FLAMES, 12000, 15000);
201 break;
202 default:
203 break;
204 }
205 }
206
207 DoMeleeAttackIfReady();
208 }
209
210 private:
211 EventMap _events;
212 InstanceScript* _instance;
213 ObjectGuid _flamecaller1GUID;
214 ObjectGuid _flamecaller2GUID;
215 SummonList _summons;
216 bool _combatPhase;
217 };
218
219 CreatureAI* GetAI(Creature* creature) const override
220 {
221 return GetBlackrockCavernsAI<npc_twilight_flame_callerAI>(creature);
222 }
223};

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"

21enum Spells
22{
23 SPELL_HEAL = 10917,
24 SPELL_RENEW = 10929,
25 SPELL_SHIELD = 10901,
26 SPELL_MINDBLAST = 10947,
27 SPELL_SHADOWWORDPAIN = 10894,
28 SPELL_SMITE = 10934
29};
30
31enum Events
32{
33 EVENT_MINDBLAST = 1,
34 EVENT_SHADOW_WORD_PAIN = 2,
35 EVENT_SMITE = 3,
36 EVENT_HEAL = 4 // not used atm
37};
38
39class boss_moira_bronzebeard : public CreatureScript
40{
41 public:
42 boss_moira_bronzebeard() : CreatureScript("boss_moira_bronzebeard") { }
43
44 struct boss_moira_bronzebeardAI : public ScriptedAI
45 {
46 boss_moira_bronzebeardAI(Creature* creature) : ScriptedAI(creature) { }
47
48 void Reset() override
49 {
50 _events.Reset();
51 }
52
53 void EnterCombat(Unit* /*who*/) override
54 {
55 //_events.ScheduleEvent(EVENT_HEAL, 12000); // not used atm // These times are probably wrong
56 _events.ScheduleEvent(EVENT_MINDBLAST, 16000);
57 _events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, 2000);
58 _events.ScheduleEvent(EVENT_SMITE, 8000);
59 }
60
61 void UpdateAI(uint32 diff) override
62 {
63 if (!UpdateVictim())
64 return;
65
66 _events.Update(diff);
67
68 while (uint32 eventId = _events.ExecuteEvent())
69 {
70 switch (eventId)
71 {
72 case EVENT_MINDBLAST:
73 DoCastVictim(SPELL_MINDBLAST);
74 _events.ScheduleEvent(EVENT_MINDBLAST, 14000);
75 break;
76 case EVENT_SHADOW_WORD_PAIN:
77 DoCastVictim(SPELL_SHADOWWORDPAIN);
78 _events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, 18000);
79 break;
80 case EVENT_SMITE:
81 DoCastVictim(SPELL_SMITE);
82 _events.ScheduleEvent(EVENT_SMITE, 10000);
83 break;
84 default:
85 break;
86 }
87 }
88 }
89
90 private:
91 EventMap _events;
92 };
93
94 CreatureAI* GetAI(Creature* creature) const override
95 {
96 return new boss_moira_bronzebeardAI(creature);
97 }
98};
99
100void AddSC_boss_moira_bronzebeard()
101{
102 new boss_moira_bronzebeard();
103}

183 void SetGuidData(uint32 type, ObjectGuid data) override
184 {
185 TC_LOG_DEBUG("scripts", "Instance Blackrock Depths: SetGuidData update (Type: %u Data %s)", type, data.ToString().c_str());
186
187 switch (type)
188 {
189 case DATA_EVENSTARTER:
190 TombEventStarterGUID = data;
191 if (!TombEventStarterGUID)
192 TombOfSevenReset();//reset
193 else
194 TombOfSevenStart();//start
195 break;
196 }

199 void SetData(uint32 type, uint32 data) override
200 {
201 TC_LOG_DEBUG("scripts", "Instance Blackrock Depths: SetData update (Type: %u Data %u)", type, data);
202
203 switch (type)
204 {
205 case TYPE_RING_OF_LAW:
206 encounter[0] = data;
207 break;
208 case TYPE_VAULT:
209 encounter[1] = data;
210 break;
211 case TYPE_BAR:
212 if (data == SPECIAL)
213 ++BarAleCount;
214 else
215 encounter[2] = data;
216 break;
217 case TYPE_TOMB_OF_SEVEN:
218 encounter[3] = data;
219 break;
220 case TYPE_LYCEUM:
221 encounter[4] = data;
222 break;
223 case TYPE_IRON_HALL:
224 encounter[5] = data;
225 break;
226 case DATA_GHOSTKILL:
227 GhostKillCount += data;
228 break;
229 }
230
231 if (data == DONE || GhostKillCount >= 7)
232 {
233 OUT_SAVE_INST_DATA;
234
235 std::ostringstream saveStream;
236 saveStream << encounter[0] << ' ' << encounter[1] << ' ' << encounter[2] << ' '
237 << encounter[3] << ' ' << encounter[4] << ' ' << encounter[5] << ' ' << GhostKillCount;
238
239 str_data = saveStream.str();
240
241 SaveToDB();
242 OUT_SAVE_INST_DATA_COMPLETE;
243 }
244 }

318 void Load(const char* in) override
319 {
320 if (!in)
321 {
322 OUT_LOAD_INST_DATA_FAIL;
323 return;
324 }
325
326 OUT_LOAD_INST_DATA(in);
327
328 std::istringstream loadStream(in);
329 loadStream >> encounter[0] >> encounter[1] >> encounter[2] >> encounter[3]
330 >> encounter[4] >> encounter[5] >> GhostKillCount;
331
332 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
333 if (encounter[i] == IN_PROGRESS)
334 encounter[i] = NOT_STARTED;
335 if (GhostKillCount > 0 && GhostKillCount < 7)
336 GhostKillCount = 0;//reset tomb of seven event
337 if (GhostKillCount >= 7)
338 GhostKillCount = 7;
339
340 OUT_LOAD_INST_DATA_COMPLETE;
341 }

343 void TombOfSevenEvent()
344 {
345 if (GhostKillCount < 7 && !TombBossGUIDs[TombEventCounter].IsEmpty())
346 {
347 if (Creature* boss = instance->GetCreature(TombBossGUIDs[TombEventCounter]))
348 {
349 boss->setFaction(FACTION_HOSTILE);
350 boss->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
351 if (Unit* target = boss->SelectNearestTarget(500))
352 boss->AI()->AttackStart(target);
353 }
354 }
355 }
356
357 void TombOfSevenReset()
358 {
359 HandleGameObject(GoTombExitGUID, false);//event reseted, close exit door
360 HandleGameObject(GoTombEnterGUID, true);//event reseted, open entrance door
361 for (uint8 i = 0; i < 7; ++i)
362 {
363 if (Creature* boss = instance->GetCreature(TombBossGUIDs[i]))
364 {
365 if (!boss->IsAlive())
366 {//do not call EnterEvadeMode(), it will create infinit loops
367 boss->Respawn();
368 boss->RemoveAllAuras();
369 boss->DeleteThreatList();
370 boss->CombatStop(true);
371 boss->LoadCreaturesAddon();
372 boss->GetMotionMaster()->MoveTargetedHome();
373 boss->SetLootRecipient(NULL);
374 }
375 boss->setFaction(FACTION_FRIEND);
376 }
377 }
378 GhostKillCount = 0;
379 TombEventStarterGUID.Clear();
380 TombEventCounter = 0;
381 TombTimer = TIMER_TOMBOFTHESEVEN;
382 SetData(TYPE_TOMB_OF_SEVEN, NOT_STARTED);
383 }
384
385 void TombOfSevenStart()
386 {
387 HandleGameObject(GoTombExitGUID, false);//event started, close exit door
388 HandleGameObject(GoTombEnterGUID, false);//event started, close entrance door
389 SetData(TYPE_TOMB_OF_SEVEN, IN_PROGRESS);
390 }
391
392 void TombOfSevenEnd()
393 {
394 DoRespawnGameObject(GoChestGUID, DAY);
395 HandleGameObject(GoTombExitGUID, true);//event done, open exit door
396 HandleGameObject(GoTombEnterGUID, true);//event done, open entrance door
397 TombEventStarterGUID.Clear();
398 SetData(TYPE_TOMB_OF_SEVEN, DONE);
399 }
400 void Update(uint32 diff) override
401 {
402 if (!TombEventStarterGUID.IsEmpty() && GhostKillCount < 7)
403 {
404 if (TombTimer <= diff)
405 {
406 TombTimer = TIMER_TOMBOFTHESEVEN;
407 ++TombEventCounter;
408 TombOfSevenEvent();
409 // Check Killed bosses
410 for (uint8 i = 0; i < 7; ++i)
411 {
412 if (Creature* boss = instance->GetCreature(TombBossGUIDs[i]))
413 {
414 if (!boss->IsAlive())
415 {
416 GhostKillCount = i+1;
417 }
418 }
419 }
420 } else TombTimer -= diff;
421 }
422 if (GhostKillCount >= 7 && !TombEventStarterGUID.IsEmpty())
423 TombOfSevenEnd();
424 }
425 };

190 void WaypointReached(uint32 waypointId) override
191 {
192 switch (waypointId)
193 {
194 case 0:
195 Talk(SAY_TEXT1);
196 CanWalk = false;
197 Event_Timer = 5000;
198 break;
199 case 1:
200 Talk(SAY_TEXT2);
201 CanWalk = false;
202 Event_Timer = 5000;
203 break;
204 case 2:
205 CanWalk = false;
206 break;
207 case 3:
208 Talk(SAY_TEXT3);
209 break;
210 case 4:
211 Talk(SAY_TEXT4);
212 CanWalk = false;
213 Event_Timer = 5000;
214 break;
215 case 5:
216 instance->UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, NPC_GRIMSTONE, me);
217 instance->SetData(TYPE_RING_OF_LAW, DONE);
218 TC_LOG_DEBUG("scripts", "npc_grimstone: event reached end and set complete.");

220 }

224 {
225 instance->HandleGameObject(instance->GetGuidData(id), open);
226 }
227
228 void UpdateAI(uint32 diff) override
229 {
230 if (MobDeath_Timer)
231 {
232 if (MobDeath_Timer <= diff)
233 {
234 MobDeath_Timer = 2500;
235
236 if (!RingBossGUID.IsEmpty())
237 {
238 Creature* boss = ObjectAccessor::GetCreature(*me, RingBossGUID);
239 if (boss && !boss->IsAlive() && boss->isDead())
240 {
241 RingBossGUID.Clear();
242 Event_Timer = 5000;
243 MobDeath_Timer = 0;
244 return;
245 }
246 return;
247 }
248
249 for (uint8 i = 0; i < MAX_NPC_AMOUNT; ++i)
250 {
251 Creature* mob = ObjectAccessor::GetCreature(*me, RingMobGUID[i]);
252 if (mob && !mob->IsAlive() && mob->isDead())
253 {
254 RingMobGUID[i].Clear();
255 --MobCount;
256
257 //seems all are gone, so set timer to continue and discontinue this
258 if (!MobCount)
259 {
260 Event_Timer = 5000;
261 MobDeath_Timer = 0;
262 }
263 }
264 }
265 } else MobDeath_Timer -= diff;
266 }
267
268 if (Event_Timer)
269 {
270 if (Event_Timer <= diff)
271 {
272 switch (EventPhase)
273 {
274 case 0:
275 Talk(SAY_TEXT5);
276 HandleGameObject(DATA_ARENA4, false);
277 Start(false, false);
278 CanWalk = true;
279 Event_Timer = 0;
280 break;
281 case 1:
282 CanWalk = true;
283 Event_Timer = 0;
284 break;
285 case 2:
286 Event_Timer = 2000;
287 break;
288 case 3:
289 HandleGameObject(DATA_ARENA1, true);
290 Event_Timer = 3000;
291 break;
292 case 4:
293 CanWalk = true;
294 me->SetVisible(false);
295 SummonRingMob();
296 Event_Timer = 8000;
297 break;
298 case 5:
299 SummonRingMob();
300 SummonRingMob();
301 Event_Timer = 8000;
302 break;
303 case 6:
304 SummonRingMob();
305 Event_Timer = 0;
306 break;
307 case 7:
308 me->SetVisible(true);
309 HandleGameObject(DATA_ARENA1, false);
310 Talk(SAY_TEXT6);
311 CanWalk = true;
312 Event_Timer = 0;
313 break;
314 case 8:
315 HandleGameObject(DATA_ARENA2, true);
316 Event_Timer = 5000;
317 break;
318 case 9:
319 me->SetVisible(false);
320 SummonRingBoss();
321 Event_Timer = 0;
322 break;
323 case 10:
324 //if quest, complete
325 HandleGameObject(DATA_ARENA2, false);
326 HandleGameObject(DATA_ARENA3, true);
327 HandleGameObject(DATA_ARENA4, true);
328 CanWalk = true;
329 Event_Timer = 0;
330 break;
331 }
332 ++EventPhase;
333 } else Event_Timer -= diff;
334 }
335
336 if (CanWalk)
337 npc_escortAI::UpdateAI(diff);
338 }
339 };
340};

641 bool OnQuestReward(Player* /*player*/, Creature* creature, Quest const* quest, uint32 /*item*/) override
642 {
643 InstanceScript* instance = creature->GetInstanceScript();
644 if (!instance)
645 return true;
646
647 if (instance->GetData(TYPE_BAR) == DONE || instance->GetData(TYPE_BAR) == SPECIAL)
648 return true;
649
650 if (quest->GetQuestId() == QUEST_ALE)
651 {
652 if (instance->GetData(TYPE_BAR) != IN_PROGRESS)
653 instance->SetData(TYPE_BAR, IN_PROGRESS);
654
655 instance->SetData(TYPE_BAR, SPECIAL);
656
657 //keep track of amount in instance script, returns SPECIAL if amount ok and event in progress
658 if (instance->GetData(TYPE_BAR) == SPECIAL)
659 {
660 creature->AI()->Talk(SAY_GOT_BEER);
661 creature->CastSpell(creature, SPELL_DRUNKEN_RAGE, false);
662
663 if (npc_escortAI* escortAI = CAST_AI(npc_rocknot::npc_rocknotAI, creature->AI()))
664 escortAI->Start(false, false);
665 }
666 }
667
668 return true;
669 }

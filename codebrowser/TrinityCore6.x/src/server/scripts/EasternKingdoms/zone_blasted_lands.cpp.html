41class spell_razelikh_teleport_group : public SpellScriptLoader
42{
43 public: spell_razelikh_teleport_group() : SpellScriptLoader("spell_razelikh_teleport_group") { }
44
45 class spell_razelikh_teleport_group_SpellScript : public SpellScript
46 {
47 PrepareSpellScript(spell_razelikh_teleport_group_SpellScript);
48
49 bool Validate(SpellInfo const* /*spell*/) override
50 {
51 if (!sSpellMgr->GetSpellInfo(SPELL_TELEPORT_SINGLE) && !sSpellMgr->GetSpellInfo(SPELL_TELEPORT_SINGLE_IN_GROUP))
52 return false;
53 return true;
54 }
55
56 void HandleScriptEffect(SpellEffIndex /* effIndex */)
57 {
58 if (Player* player = GetHitPlayer())
59 {
60 if (Group* group = player->GetGroup())
61 {
62 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
63 if (Player* member = itr->GetSource())
64 if (member->IsWithinDistInMap(player, 20.0f) && !member->isDead())
65 member->CastSpell(member, SPELL_TELEPORT_SINGLE_IN_GROUP, true);
66 }
67 else
68 player->CastSpell(player, SPELL_TELEPORT_SINGLE, true);
69 }
70 }
71
72 void Register() override
73 {
74 OnEffectHitTarget += SpellEffectFn(spell_razelikh_teleport_group_SpellScript::HandleScriptEffect, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
75 }
76 };
77
78 SpellScript* GetSpellScript() const override
79 {
80 return new spell_razelikh_teleport_group_SpellScript();
81 }
82};
83
84void AddSC_blasted_lands()
85{
86 new spell_razelikh_teleport_group();
87}

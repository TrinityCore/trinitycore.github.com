21enum Spells
22{
23 SPELL_CLEAVE = 15284,
24 SPELL_FRIGHTENING_SHOUT = 19134,
25 SPELL_WHIRLWIND1 = 15589,
26 SPELL_WHIRLWIND2 = 13736,
27 SPELL_MORTAL_STRIKE = 16856
28};
29
30enum Texts
31{
32 SAY_AGGRO = 0,
33 SAY_EVADE = 1,
34 SAY_BUFF = 2
35};
36
37enum Events
38{
39 EVENT_CLEAVE = 1,
40 EVENT_FRIGHTENING_SHOUT,
41 EVENT_WHIRLWIND1,
42 EVENT_WHIRLWIND2,
43 EVENT_MORTAL_STRIKE
44};
45
46enum Action
47{
48 ACTION_BUFF_YELL = -30001 // shared from Battleground
49};
50
51class boss_galvangar : public CreatureScript
52{
53public:
54 boss_galvangar() : CreatureScript("boss_galvangar") { }
55
56 struct boss_galvangarAI : public ScriptedAI
57 {
58 boss_galvangarAI(Creature* creature) : ScriptedAI(creature) { }
59
60 void Reset() override
61 {
62 events.Reset();
63 }
64
65 void EnterCombat(Unit* /*who*/) override
66 {
67 Talk(SAY_AGGRO);
68 events.ScheduleEvent(EVENT_CLEAVE, urand(1 * IN_MILLISECONDS, 9 * IN_MILLISECONDS));
69 events.ScheduleEvent(EVENT_FRIGHTENING_SHOUT, urand(2 * IN_MILLISECONDS, 19 * IN_MILLISECONDS));
70 events.ScheduleEvent(EVENT_WHIRLWIND1, urand(1 * IN_MILLISECONDS, 13 * IN_MILLISECONDS));
71 events.ScheduleEvent(EVENT_WHIRLWIND2, urand(5 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
72 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(5 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
73 }
74
75 void DoAction(int32 actionId) override
76 {
77 if (actionId == ACTION_BUFF_YELL)
78 Talk(SAY_BUFF);
79 }
80
81 bool CheckInRoom()
82 {
83 if (me->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 50)
84 {
85 EnterEvadeMode();
86 Talk(SAY_EVADE);
87 return false;
88 }
89
90 return true;
91 }
92
93 void UpdateAI(uint32 diff) override
94 {
95 if (!UpdateVictim() || !CheckInRoom())
96 return;
97
98 events.Update(diff);
99
100 if (me->HasUnitState(UNIT_STATE_CASTING))
101 return;
102
103 while (uint32 eventId = events.ExecuteEvent())
104 {
105 switch (eventId)
106 {
107 case EVENT_CLEAVE:
108 DoCastVictim(SPELL_CLEAVE);
109 events.ScheduleEvent(EVENT_CLEAVE, urand(10 * IN_MILLISECONDS, 16 * IN_MILLISECONDS));
110 break;
111 case EVENT_FRIGHTENING_SHOUT:
112 DoCastVictim(SPELL_FRIGHTENING_SHOUT);
113 events.ScheduleEvent(EVENT_FRIGHTENING_SHOUT, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
114 break;
115 case EVENT_WHIRLWIND1:
116 DoCastVictim(SPELL_WHIRLWIND1);
117 events.ScheduleEvent(EVENT_WHIRLWIND1, urand(6 * IN_MILLISECONDS, 10 * IN_MILLISECONDS));
118 break;
119 case EVENT_WHIRLWIND2:
120 DoCastVictim(SPELL_WHIRLWIND2);
121 events.ScheduleEvent(EVENT_WHIRLWIND2, urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
122 break;
123 case EVENT_MORTAL_STRIKE:
124 DoCastVictim(SPELL_MORTAL_STRIKE);
125 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(10 * IN_MILLISECONDS, 30 * IN_MILLISECONDS));
126 break;
127 default:
128 break;
129 }
130 }
131
132 DoMeleeAttackIfReady();
133 }
134
135 private:
136 EventMap events;
137 };
138
139 CreatureAI* GetAI(Creature* creature) const override
140 {
141 return new boss_galvangarAI(creature);
142 }
143};
144
145void AddSC_boss_galvangar()
146{
147 new boss_galvangar;
148}

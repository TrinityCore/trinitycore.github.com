21enum Spells
22{
23 SPELL_ARCANE_EXPLOSION = 46608,
24 SPELL_CONE_OF_COLD = 38384,
25 SPELL_FIREBALL = 46988,
26 SPELL_FROSTBOLT = 46987,
27 SPELL_SUMMON_WATER_ELEMENTAL = 45067,
28 SPELL_ICEBLOCK = 46604
29};
30
31enum Texts
32{
33 SAY_AGGRO = 0,
34 SAY_EVADE = 1,
35 SAY_SALVATION = 2,
36};
37
38enum Action
39{
40 ACTION_BUFF_YELL = -30001 // shared from Battleground
41};
42
43enum Events
44{
45 // Balinda
46 EVENT_ARCANE_EXPLOSION = 1,
47 EVENT_CONE_OF_COLD,
48 EVENT_FIREBOLT,
49 EVENT_FROSTBOLT,
50 EVENT_SUMMON_WATER_ELEMENTAL,
51 EVENT_CHECK_RESET, // Checks if Balinda or the Water Elemental are outside of building.
52};
53
54class boss_balinda : public CreatureScript
55{
56public:
57 boss_balinda() : CreatureScript("boss_balinda") { }
58
59 struct boss_balindaAI : public ScriptedAI
60 {
61 boss_balindaAI(Creature* creature) : ScriptedAI(creature), summons(me)
62 {
63 Initialize();
64 }
65
66 void Initialize()
67 {
68 WaterElementalGUID.Clear();
69 HasCastIceblock = false;
70 }
71
72 void Reset() override
73 {
74 Initialize();
75 events.Reset();
76 summons.DespawnAll();
77 }
78
79 void EnterCombat(Unit* /*who*/) override
80 {
81 Talk(SAY_AGGRO);
82 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(5 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
83 events.ScheduleEvent(EVENT_CONE_OF_COLD, 8 * IN_MILLISECONDS);
84 events.ScheduleEvent(EVENT_FIREBOLT, 1 * IN_MILLISECONDS);
85 events.ScheduleEvent(EVENT_FROSTBOLT, 4 * IN_MILLISECONDS);
86 events.ScheduleEvent(EVENT_SUMMON_WATER_ELEMENTAL, 3 * IN_MILLISECONDS);
87 events.ScheduleEvent(EVENT_CHECK_RESET, 5 * IN_MILLISECONDS);
88 }
89
90 void JustSummoned(Creature* summoned) override
91 {
92 summoned->AI()->AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0, 50, true));
93 summoned->setFaction(me->getFaction());
94 WaterElementalGUID = summoned->GetGUID();
95 summons.Summon(summoned);
96 }
97
98 void SummonedCreatureDespawn(Creature* summoned) override
99 {
100 summons.Despawn(summoned);
101 }
102
103 void JustDied(Unit* /*killer*/) override
104 {
105 summons.DespawnAll();
106 }
107
108 void DoAction(int32 actionId) override
109 {
110 if (actionId == ACTION_BUFF_YELL)
111 Talk(SAY_AGGRO);
112 }
113
114 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
115 {
116 if (me->HealthBelowPctDamaged(40, damage) && !HasCastIceblock)
117 {
118 DoCast(SPELL_ICEBLOCK);
119 HasCastIceblock = true;
120 }
121 }
122
123 void UpdateAI(uint32 diff) override
124 {
125 if (!UpdateVictim())
126 return;
127
128 events.Update(diff);
129
130 if (me->HasUnitState(UNIT_STATE_CASTING))
131 return;
132
133 while (uint32 eventId = events.ExecuteEvent())
134 {
135 switch (eventId)
136 {
137 case EVENT_ARCANE_EXPLOSION:
138 DoCastVictim(SPELL_ARCANE_EXPLOSION);
139 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(5 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
140 break;
141 case EVENT_CONE_OF_COLD:
142 DoCastVictim(SPELL_CONE_OF_COLD);
143 events.ScheduleEvent(EVENT_CONE_OF_COLD, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
144 break;
145 case EVENT_FIREBOLT:
146 DoCastVictim(SPELL_FIREBALL);
147 events.ScheduleEvent(EVENT_FIREBOLT, urand(5 * IN_MILLISECONDS, 9 * IN_MILLISECONDS));
148 break;
149 case EVENT_FROSTBOLT:
150 DoCastVictim(SPELL_FROSTBOLT);
151 events.ScheduleEvent(EVENT_FROSTBOLT, urand(4 * IN_MILLISECONDS, 12 * IN_MILLISECONDS));
152 break;
153 case EVENT_SUMMON_WATER_ELEMENTAL:
154 if (summons.empty())
155 DoCast(SPELL_SUMMON_WATER_ELEMENTAL);
156 events.ScheduleEvent(EVENT_SUMMON_WATER_ELEMENTAL, 50 * IN_MILLISECONDS);
157 break;
158 case EVENT_CHECK_RESET:
159 if (me->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 50)
160 {
161 EnterEvadeMode();
162 Talk(SAY_EVADE);
163 }
164 if (Creature* elemental = ObjectAccessor::GetCreature(*me, WaterElementalGUID))
165 if (elemental->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 50)
166 elemental->AI()->EnterEvadeMode();
167 events.ScheduleEvent(EVENT_CHECK_RESET, 5 * IN_MILLISECONDS);
168 break;
169 default:
170 break;
171 }
172 }
173
174 DoMeleeAttackIfReady();
175 }
176
177 private:
178 EventMap events;
179 SummonList summons;
180 ObjectGuid WaterElementalGUID;
181 bool HasCastIceblock;
182 };
183
184 CreatureAI* GetAI(Creature* creature) const override
185 {
186 return new boss_balindaAI(creature);
187 }
188};
189
190void AddSC_boss_balinda()
191{
192 new boss_balinda;
193}

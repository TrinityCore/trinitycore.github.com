80class npc_av_marshal_or_warmaster : public CreatureScript
81{
82 public:
83 npc_av_marshal_or_warmaster() : CreatureScript("npc_av_marshal_or_warmaster") { }
84
85 struct npc_av_marshal_or_warmasterAI : public ScriptedAI
86 {
87 npc_av_marshal_or_warmasterAI(Creature* creature) : ScriptedAI(creature)
88 {
89 Initialize();
90 }
91
92 void Initialize()
93 {
94 _hasAura = false;
95 }
96
97 void Reset() override
98 {
99 Initialize();
100
101 events.Reset();
102 events.ScheduleEvent(EVENT_CHARGE_TARGET, urand(2 * IN_MILLISECONDS, 12 * IN_MILLISECONDS));
103 events.ScheduleEvent(EVENT_CLEAVE, urand(1 * IN_MILLISECONDS, 11 * IN_MILLISECONDS));
104 events.ScheduleEvent(EVENT_DEMORALIZING_SHOUT, 2000);
105 events.ScheduleEvent(EVENT_WHIRLWIND, urand(5 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
106 events.ScheduleEvent(EVENT_ENRAGE, urand(5 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
107 events.ScheduleEvent(EVENT_CHECK_RESET, 5000);
108 }
109
110 void JustRespawned() override
111 {
112 Reset();
113 }
114
115 void UpdateAI(uint32 diff) override
116 {
117 // I have a feeling this isn't blizzlike, but owell, I'm only passing by and cleaning up.
118 if (!_hasAura)
119 {
120 for (uint8 i = 0; i < MAX_SPELL_PAIRS; ++i)
121 if (_auraPairs[i].npcEntry == me->GetEntry())
122 DoCast(me, _auraPairs[i].spellId);
123
124 _hasAura = true;
125 }
126
127 if (!UpdateVictim())
128 return;
129
130 events.Update(diff);
131
132 if (me->HasUnitState(UNIT_STATE_CASTING))
133 return;
134
135 while (uint32 eventId = events.ExecuteEvent())
136 {
137 switch (eventId)
138 {
139 case EVENT_CHARGE_TARGET:
140 DoCastVictim(SPELL_CHARGE);
141 events.ScheduleEvent(EVENT_CHARGE, urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
142 break;
143 case EVENT_CLEAVE:
144 DoCastVictim(SPELL_CLEAVE);
145 events.ScheduleEvent(EVENT_CLEAVE, urand(10 * IN_MILLISECONDS, 16 * IN_MILLISECONDS));
146 break;
147 case EVENT_DEMORALIZING_SHOUT:
148 DoCast(me, SPELL_DEMORALIZING_SHOUT);
149 events.ScheduleEvent(EVENT_DEMORALIZING_SHOUT, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
150 break;
151 case EVENT_WHIRLWIND:
152 DoCast(me, SPELL_WHIRLWIND);
153 events.ScheduleEvent(EVENT_WHIRLWIND, urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
154 break;
155 case EVENT_ENRAGE:
156 DoCast(me, SPELL_ENRAGE);
157 events.ScheduleEvent(EVENT_ENRAGE, urand(10 * IN_MILLISECONDS, 30 * IN_MILLISECONDS));
158 break;
159 case EVENT_CHECK_RESET:
160 {
161 Position const& _homePosition = me->GetHomePosition();
162 if (me->GetDistance2d(_homePosition.GetPositionX(), _homePosition.GetPositionY()) > 50.0f)
163 {
164 EnterEvadeMode();
165 return;
166 }
167 events.ScheduleEvent(EVENT_CHECK_RESET, 5000);
168 break;
169 }
170 }
171 }
172
173 DoMeleeAttackIfReady();
174 }
175
176 private:
177 EventMap events;
178 bool _hasAura;
179 };
180
181 CreatureAI* GetAI(Creature* creature) const override
182 {
183 return new npc_av_marshal_or_warmasterAI(creature);
184 }
185};

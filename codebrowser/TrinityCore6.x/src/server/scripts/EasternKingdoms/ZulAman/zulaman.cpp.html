81class npc_voljin_zulaman : public CreatureScript
82{
83 public:
84 npc_voljin_zulaman() : CreatureScript("npc_voljin_zulaman") { }
85
86 struct npc_voljin_zulamanAI : public ScriptedAI
87 {
88 npc_voljin_zulamanAI(Creature* creature) : ScriptedAI(creature), _instance(creature->GetInstanceScript())
89 {
90 me->SetDisplayId(me->GetCreatureTemplate()->Modelid1);
91 if (_instance->GetData(DATA_ZULAMAN_STATE) == NOT_STARTED)
92 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
93 }
94
95 void Reset() override
96 {
97 _gongCount = 0;
98 }
99
100 void sGossipSelect(Player* player, uint32 menuId, uint32 gossipListId) override
101 {
102 if (_instance->GetData(DATA_ZULAMAN_STATE) != NOT_STARTED)
103 return;
104
105 if (me->GetCreatureTemplate()->GossipMenuId == menuId && !gossipListId)
106 {
107 _events.Reset();
108 me->SetUInt32Value(UNIT_FIELD_MOUNTDISPLAYID, 0);
109 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
110 me->SetUInt32Value(OBJECT_DYNAMIC_FLAGS, UNIT_DYNFLAG_NONE);
111 _events.ScheduleEvent(EVENT_INTRO_MOVEPOINT_1, 1000);
112 Talk(SAY_INTRO_1, player);
113 me->SetWalk(true);
114 }
115 }
116
117 void DoAction(int32 action) override
118 {
119 if (action == ACTION_START_ZULAMAN)
120 {
121 if (++_gongCount == 10)
122 _events.ScheduleEvent(EVENT_START_DOOR_OPENING_1, 500);
123 }
124 }
125
126 void UpdateAI(uint32 diff) override
127 {
128 _events.Update(diff);
129 while (uint32 eventId = _events.ExecuteEvent())
130 {
131 switch (eventId)
132 {
133 case EVENT_INTRO_MOVEPOINT_1:
134 me->GetMotionMaster()->MovePoint(POINT_INTRO, VoljinIntroWaypoint[0]);
135 _events.ScheduleEvent(EVENT_INTRO_MOVEPOINT_2, 1000);
136 break;
137 case EVENT_INTRO_MOVEPOINT_2:
138 me->GetMotionMaster()->MovePoint(POINT_STRANGE_GONG, VoljinIntroWaypoint[1]);
139 _events.ScheduleEvent(EVENT_INTRO_MOVEPOINT_3, 4000);
140 break;
141 case EVENT_INTRO_MOVEPOINT_3:
142 Talk(SAY_INTRO_2);
143 _events.ScheduleEvent(EVENT_BANGING_THE_GONG, 3000);
144 break;
145 case EVENT_BANGING_THE_GONG:
146 DoCast(me, SPELL_BANGING_THE_GONG);
147 if (GameObject* strangeGong = ObjectAccessor::GetGameObject(*me, _instance->GetGuidData(DATA_STRANGE_GONG)))
148 strangeGong->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
149 me->SetVirtualItem(0, uint32(ITEM_VIRTUAL_ITEM));
150 break;
151 case EVENT_START_DOOR_OPENING_1:
152 me->RemoveAura(SPELL_BANGING_THE_GONG);
153 _events.ScheduleEvent(EVENT_START_DOOR_OPENING_2, 500);
154 break;
155 case EVENT_START_DOOR_OPENING_2:
156 me->SetVirtualItem(0, uint32(0));
157 if (GameObject* strangeGong = ObjectAccessor::GetGameObject(*me, _instance->GetGuidData(DATA_STRANGE_GONG)))
158 strangeGong->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
159 _events.ScheduleEvent(EVENT_START_DOOR_OPENING_3, 500);
160 break;
161 case EVENT_START_DOOR_OPENING_3:
162 me->GetMotionMaster()->MovePoint(POINT_START_DOOR_OPENING_1, VoljinIntroWaypoint[2]);
163 break;
164 case EVENT_START_DOOR_OPENING_4:
165 _instance->SetData(DATA_ZULAMAN_STATE, IN_PROGRESS);
166 if (GameObject* masiveGate = ObjectAccessor::GetGameObject(*me, _instance->GetGuidData(DATA_MASSIVE_GATE)))
167 masiveGate->SetGoState(GO_STATE_ACTIVE);
168 _events.ScheduleEvent(EVENT_START_DOOR_OPENING_5, 3000);
169 break;
170 case EVENT_START_DOOR_OPENING_5:
171 Talk(SAY_INTRO_4);
172 _events.ScheduleEvent(EVENT_START_DOOR_OPENING_6, 6000);
173 break;
174 case EVENT_START_DOOR_OPENING_6:
175 _events.ScheduleEvent(EVENT_START_DOOR_OPENING_7, 6000);
176 break;
177 case EVENT_START_DOOR_OPENING_7:
178 if (Creature* hexLordTrigger = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_HEXLORD_TRIGGER)))
179 sCreatureTextMgr->SendChat(hexLordTrigger, SAY_HEXLOR_INTRO, 0, CHAT_MSG_ADDON, LANG_ADDON, TEXT_RANGE_MAP);
180 break;
181 default:
182 break;
183 }
184 }
185 }
186
187 void MovementInform(uint32 movementType, uint32 pointId) override
188 {
189 if (movementType != POINT_MOTION_TYPE)
190 return;
191
192 switch (pointId)
193 {
194 case POINT_STRANGE_GONG:
195 if (GameObject* strangeGong = ObjectAccessor::GetGameObject(*me, _instance->GetGuidData(DATA_STRANGE_GONG)))
196 me->SetFacingToObject(strangeGong); // setInFront
197 break;
198 case POINT_START_DOOR_OPENING_1:
199 me->SetFacingTo(4.747295f);
200 me->GetMotionMaster()->MovePoint(POINT_START_DOOR_OPENING_2, VoljinIntroWaypoint[3]);
201 Talk(SAY_INTRO_3);
202 _events.ScheduleEvent(EVENT_START_DOOR_OPENING_4, 4500);
203 break;
204 default:
205 break;
206 }
207 }
208
209 private:
210 InstanceScript* _instance;
211 EventMap _events;
212 uint8 _gongCount = 0;
213 };
214
215 CreatureAI* GetAI(Creature* creature) const override
216 {
217 return GetInstanceAI<npc_voljin_zulamanAI>(creature);
218 }

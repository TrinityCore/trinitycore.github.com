44class boss_hexlord_malacrass : public CreatureScript
45{
46 public:
47
48 boss_hexlord_malacrass() : CreatureScript("boss_hexlord_malacrass") { }
49
50 struct boss_hex_lord_malacrassAI : public BossAI
51 {
52 boss_hex_lord_malacrassAI(Creature* creature) : BossAI(creature, DATA_HEXLORD) { }
53
54 void Reset() override
55 {
56 _Reset();
57 }
58
59 void EnterCombat(Unit* /*who*/) override
60 {
61 Talk(SAY_AGGRO);
62 _EnterCombat();
63 }
64
65 void JustDied(Unit* /*killer*/) override
66 {
67 Talk(SAY_DEATH);
68 _JustDied();
69 }
70
71 void KilledUnit(Unit* victim) override
72 {
73 if (victim->GetTypeId() == TYPEID_PLAYER)
74 Talk(SAY_PLAYER_KILL);
75 }
76
77 void UpdateAI(uint32 diff) override
78 {
79 if (!UpdateVictim())
80 return;
81
82 events.Update(diff);
83
84 if (me->HasUnitState(UNIT_STATE_CASTING))
85 return;
86 /*
87 while (uint32 eventId = events.ExecuteEvent())
88 {
89 switch (eventId)
90 {
91 default:
92 break;
93 }
94 }
95 */
96
97 DoMeleeAttackIfReady();
98 }
99 };
100
101 CreatureAI* GetAI(Creature* creature) const override
102 {
103 return GetInstanceAI<boss_hex_lord_malacrassAI>(creature);
104 }
105};

107class spell_hexlord_unstable_affliction : public SpellScriptLoader
108{
109 public:
110 spell_hexlord_unstable_affliction() : SpellScriptLoader("spell_hexlord_unstable_affliction") { }
111
112 class spell_hexlord_unstable_affliction_AuraScript : public AuraScript
113 {
114 PrepareAuraScript(spell_hexlord_unstable_affliction_AuraScript);
115
116 bool Validate(SpellInfo const* /*spell*/) override
117 {
118 if (!sSpellMgr->GetSpellInfo(SPELL_WL_UNSTABLE_AFFL_DISPEL))
119 return false;
120 return true;
121 }
122
123 void HandleDispel(DispelInfo* dispelInfo)
124 {
125 if (Unit* caster = GetCaster())
126 caster->CastSpell(dispelInfo->GetDispeller(), SPELL_WL_UNSTABLE_AFFL_DISPEL, true, NULL, GetEffect(EFFECT_0));
127 }
128
129 void Register() override
130 {
131 AfterDispel += AuraDispelFn(spell_hexlord_unstable_affliction_AuraScript::HandleDispel);
132 }
133 };
134
135 AuraScript* GetAuraScript() const override
136 {
137 return new spell_hexlord_unstable_affliction_AuraScript();
138 }
139};

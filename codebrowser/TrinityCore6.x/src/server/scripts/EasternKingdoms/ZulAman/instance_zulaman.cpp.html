23class instance_zulaman : public InstanceMapScript
24{
25 public:
26 instance_zulaman() : InstanceMapScript(ZulAmanScriptName, 568) { }
27
28 struct instance_zulaman_InstanceScript : public InstanceScript
29 {
30 instance_zulaman_InstanceScript(InstanceMap* map) : InstanceScript(map)
31 {
32 SetHeaders(DataHeader);
33 SetBossNumber(EncounterCount);
34
35 SpeedRunTimer = 16;
36 ZulAmanState = NOT_STARTED;
37 ZulAmanBossCount = 0;
38 }
39
40 void FillInitialWorldStates(WorldPackets::WorldState::InitWorldStates& packet) override
41 {
42 packet.Worldstates.emplace_back(uint32(WORLD_STATE_ZULAMAN_TIMER_ENABLED), int32(ZulAmanState ? 1 : 0));
43 packet.Worldstates.emplace_back(uint32(WORLD_STATE_ZULAMAN_TIMER), int32(SpeedRunTimer));
44 }

46 void OnCreatureCreate(Creature* creature) override
47 {
48 switch (creature->GetEntry())
49 {
50 case NPC_AKILZON:
51 AkilzonGUID = creature->GetGUID();
52 break;
53 case NPC_NALORAKK:
54 NalorakkGUID = creature->GetGUID();
55 break;
56 case NPC_JANALAI:
57 JanalaiGUID = creature->GetGUID();
58 break;
59 case NPC_HALAZZI:
60 HalazziGUID = creature->GetGUID();
61 break;
62 case NPC_HEXLORD:
63 HexLordMalacrassGUID = creature->GetGUID();
64 break;
65 case NPC_DAAKARA:
66 DaakaraGUID = creature->GetGUID();
67 break;
68 case NPC_VOLJIN:
69 VoljinGUID = creature->GetGUID();
70 break;
71 case NPC_HEXLORD_TRIGGER:
72 HexLordTriggerGUID = creature->GetGUID();
73 break;
74 default:
75 break;
76 }
77 }
78
79 void OnGameObjectCreate(GameObject* go) override
80 {
81 switch (go->GetEntry())
82 {
83 case GO_STRANGE_GONG:
84 StrangeGongGUID = go->GetGUID();
85 break;
86 case GO_MASSIVE_GATE:
87 MasiveGateGUID = go->GetGUID();
88 AddDoor(go, true);
89 if (ZulAmanState != NOT_STARTED)
90 go->SetGoState(GO_STATE_ACTIVE);
91 break;
92 default:
93 break;
94 }
95 }
96
97 void OnGameObjectRemove(GameObject* go) override
98 {
99 switch (go->GetEntry())
100 {
101 case GO_MASSIVE_GATE:
102 AddDoor(go, false);
103 break;
104 default:
105 break;
106 }
107 }

137
138 void SetData(uint32 type, uint32 data) override
139 {
140 switch (type)
141 {
142 case DATA_ZULAMAN_STATE:
143 {
144 if (data == IN_PROGRESS)
145 {
146 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER_ENABLED, 1);
147 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER, 15);
148 events.ScheduleEvent(EVENT_UPDATE_ZULAMAN_TIMER, 60000);
149 SpeedRunTimer = 15;
150 ZulAmanState = data;
151 SaveToDB();
152 }
153 break;
154 }
155 default:
156 break;
157 }
158 }
159
160 uint32 GetData(uint32 type) const override
161 {
162 switch (type)
163 {
164 case DATA_ZULAMAN_STATE:
165 return ZulAmanState;
166 default:
167 break;
168 }
169
170 return 0;
171 }
172
173 bool SetBossState(uint32 type, EncounterState state) override
174 {
175 if (!InstanceScript::SetBossState(type, state))
176 return false;
177
178 if (state == DONE)
179 {
180 if (ZulAmanState == IN_PROGRESS && SpeedRunTimer)
181 {
182 ++ZulAmanBossCount;
183
184 if (ZulAmanBossCount < 2)
185 {
186 SpeedRunTimer = SpeedRunTimer + 5;
187 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER, SpeedRunTimer);
188 }
189 else if (ZulAmanBossCount == 4)
190 {
191 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER_ENABLED, 0);
192 events.CancelEvent(EVENT_UPDATE_ZULAMAN_TIMER);
193 ZulAmanState = DONE;
194 }
195 }
196 }
197
198 switch (type)
199 {
200 case DATA_AKILZON:
201 break;
202 case DATA_NALORAKK:
203 break;
204 case DATA_JANALAI:
205 break;
206 case DATA_HALAZZI:
207 case DATA_HEXLORD:
208 case DATA_DAAKARA:
209 break;
210 default:
211 break;
212 }
213
214 return true;
215 }
216
217 void ProcessEvent(WorldObject* /*obj*/, uint32 eventId) override
218 {
219 switch (eventId)
220 {
221 case EVENT_START_ZULAMAN:
222 if (Creature* voljin = instance->GetCreature(VoljinGUID))
223 {
224 if (voljin->IsAIEnabled)
225 voljin->AI()->DoAction(ACTION_START_ZULAMAN);
226 }
227 break;
228 default:
229 break;
230 }
231 }
232
233 void Update(uint32 diff) override
234 {
235 if (events.Empty())
236 return;
237
238 events.Update(diff);
239
240 while (uint32 eventId = events.ExecuteEvent())
241 {
242 switch (eventId)
243 {
244 case EVENT_UPDATE_ZULAMAN_TIMER:
245 SaveToDB();
246 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER, --SpeedRunTimer);
247 if (SpeedRunTimer)
248 events.ScheduleEvent(EVENT_UPDATE_ZULAMAN_TIMER, 60000);
249 else
250 {
251 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER_ENABLED, 0);
252 events.CancelEvent(EVENT_UPDATE_ZULAMAN_TIMER);
253 ZulAmanState = FAIL;
254 }
255 break;
256 default:
257 break;
258 }
259 }
260 }
261
262 void WriteSaveDataMore(std::ostringstream& data) override
263 {
264 data << ZulAmanState << ' '
265 << SpeedRunTimer << ' '
266 << ZulAmanBossCount;
267 }
268
269 void ReadSaveDataMore(std::istringstream& data) override
270 {
271 data >> ZulAmanState;
272 data >> SpeedRunTimer;
273 data >> ZulAmanBossCount;
274
275 if (ZulAmanState == IN_PROGRESS && SpeedRunTimer && SpeedRunTimer <= 15)
276 {
277 events.ScheduleEvent(EVENT_UPDATE_ZULAMAN_TIMER, 60000);
278 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER_ENABLED, 1);
279 DoUpdateWorldState(WORLD_STATE_ZULAMAN_TIMER, SpeedRunTimer);
280 }
281 }

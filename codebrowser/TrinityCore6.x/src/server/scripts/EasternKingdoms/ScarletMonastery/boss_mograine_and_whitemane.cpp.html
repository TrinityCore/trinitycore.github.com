61class boss_scarlet_commander_mograine : public CreatureScript
62{
63public:
64 boss_scarlet_commander_mograine() : CreatureScript("boss_scarlet_commander_mograine") { }
65
66 CreatureAI* GetAI(Creature* creature) const override
67 {
68 return GetInstanceAI<boss_scarlet_commander_mograineAI>(creature);
69 }
70
71 struct boss_scarlet_commander_mograineAI : public ScriptedAI
72 {
73 boss_scarlet_commander_mograineAI(Creature* creature) : ScriptedAI(creature)
74 {
75 Initialize();
76 instance = creature->GetInstanceScript();
77 }
78
79 void Initialize()
80 {
81 CrusaderStrike_Timer = 10000;
82 HammerOfJustice_Timer = 10000;
83 _bHasDied = false;
84 _bHeal = false;
85 _bFakeDeath = false;
86 }
87
88 InstanceScript* instance;
89
90 uint32 CrusaderStrike_Timer;
91 uint32 HammerOfJustice_Timer;
92
93 bool _bHasDied;
94 bool _bHeal;
95 bool _bFakeDeath;
96
97 void Reset() override
98 {
99 Initialize();
100
101 //Incase wipe during phase that mograine fake death
102 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
103 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
104 me->SetStandState(UNIT_STAND_STATE_STAND);
105
106 if (me->IsAlive())
107 instance->SetBossState(DATA_MOGRAINE_AND_WHITE_EVENT, NOT_STARTED);
108 }
109
110 void JustReachedHome() override
111 {
112 if (instance->GetBossState(DATA_MOGRAINE_AND_WHITE_EVENT) != NOT_STARTED)
113 instance->SetBossState(DATA_MOGRAINE_AND_WHITE_EVENT, FAIL);
114 }
115
116 void EnterCombat(Unit* /*who*/) override
117 {
118 Talk(SAY_MO_AGGRO);
119 DoCast(me, SPELL_RETRIBUTIONAURA);
120
121 me->CallForHelp(VISIBLE_RANGE);
122 }
123
124 void KilledUnit(Unit* /*victim*/) override
125 {
126 Talk(SAY_MO_KILL);
127 }
128
129 void DamageTaken(Unit* /*doneBy*/, uint32 &damage) override
130 {
131 if (damage < me->GetHealth() || _bHasDied || _bFakeDeath)
132 return;
133
134 //On first death, fake death and open door, as well as initiate whitemane if exist
135 if (Unit* Whitemane = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_WHITEMANE)))
136 {
137 instance->SetBossState(DATA_MOGRAINE_AND_WHITE_EVENT, IN_PROGRESS);
138
139 Whitemane->GetMotionMaster()->MovePoint(1, 1163.113370f, 1398.856812f, 32.527786f);
140
141 me->GetMotionMaster()->MovementExpired();
142 me->GetMotionMaster()->MoveIdle();
143
144 me->SetHealth(0);
145
146 if (me->IsNonMeleeSpellCast(false))
147 me->InterruptNonMeleeSpells(false);
148
149 me->RemoveAllAuras();
150 me->ClearAllReactives();
151
152 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
153 me->SetStandState(UNIT_STAND_STATE_DEAD);
154
155 _bHasDied = true;
156 _bFakeDeath = true;
157
158 damage = 0;
159 }
160 }
161
162 void SpellHit(Unit* /*who*/, const SpellInfo* spell) override
163 {
164 //When hit with resurrection say text
165 if (spell->Id == SPELL_SCARLETRESURRECTION)
166 {
167 Talk(SAY_MO_RESURRECTED);
168 _bFakeDeath = false;
169
170 instance->SetBossState(DATA_MOGRAINE_AND_WHITE_EVENT, SPECIAL);
171 }
172 }
173
174 void UpdateAI(uint32 diff) override
175 {
176 if (!UpdateVictim())
177 return;
178
179 if (_bHasDied && !_bHeal && instance->GetBossState(DATA_MOGRAINE_AND_WHITE_EVENT) == SPECIAL)
180 {
181 //On resurrection, stop fake death and heal whitemane and resume fight
182 if (Unit* Whitemane = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_WHITEMANE)))
183 {
184 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
185 me->SetStandState(UNIT_STAND_STATE_STAND);
186 DoCast(Whitemane, SPELL_LAYONHANDS);
187
188 CrusaderStrike_Timer = 10000;
189 HammerOfJustice_Timer = 10000;
190
191 if (me->GetVictim())
192 me->GetMotionMaster()->MoveChase(me->GetVictim());
193
194 _bHeal = true;
195 }
196 }
197
198 //This if-check to make sure mograine does not attack while fake death
199 if (_bFakeDeath)
200 return;
201
202 //CrusaderStrike_Timer
203 if (CrusaderStrike_Timer <= diff)
204 {
205 DoCastVictim(SPELL_CRUSADERSTRIKE);
206 CrusaderStrike_Timer = 10000;
207 }
208 else CrusaderStrike_Timer -= diff;
209
210 //HammerOfJustice_Timer
211 if (HammerOfJustice_Timer <= diff)
212 {
213 DoCastVictim(SPELL_HAMMEROFJUSTICE);
214 HammerOfJustice_Timer = 60000;
215 }
216 else HammerOfJustice_Timer -= diff;
217
218 DoMeleeAttackIfReady();
219 }
220 };
221};

223class boss_high_inquisitor_whitemane : public CreatureScript
224{
225public:
226 boss_high_inquisitor_whitemane() : CreatureScript("boss_high_inquisitor_whitemane") { }
227
228 CreatureAI* GetAI(Creature* creature) const override
229 {
230 return GetInstanceAI<boss_high_inquisitor_whitemaneAI>(creature);
231 }
232
233 struct boss_high_inquisitor_whitemaneAI : public ScriptedAI
234 {
235 boss_high_inquisitor_whitemaneAI(Creature* creature) : ScriptedAI(creature)
236 {
237 Initialize();
238 instance = creature->GetInstanceScript();
239 }
240
241 void Initialize()
242 {
243 Wait_Timer = 7000;
244 Heal_Timer = 10000;
245 PowerWordShield_Timer = 15000;
246 HolySmite_Timer = 6000;
247
248 _bCanResurrectCheck = false;
249 _bCanResurrect = false;
250 }
251
252 InstanceScript* instance;
253
254 uint32 Heal_Timer;
255 uint32 PowerWordShield_Timer;
256 uint32 HolySmite_Timer;
257 uint32 Wait_Timer;
258
259 bool _bCanResurrectCheck;
260 bool _bCanResurrect;
261
262 void Reset() override
263 {
264 Initialize();
265
266 if (me->IsAlive())
267 instance->SetBossState(DATA_MOGRAINE_AND_WHITE_EVENT, NOT_STARTED);
268 }
269
270 void AttackStart(Unit* who) override
271 {
272 if (instance->GetBossState(DATA_MOGRAINE_AND_WHITE_EVENT) == NOT_STARTED)
273 return;
274
275 ScriptedAI::AttackStart(who);
276 }
277
278 void EnterCombat(Unit* /*who*/) override
279 {
280 Talk(SAY_WH_INTRO);
281 }
282
283 void KilledUnit(Unit* /*victim*/) override
284 {
285 Talk(SAY_WH_KILL);
286 }
287
288 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
289 {
290 if (!_bCanResurrectCheck && damage >= me->GetHealth())
291 damage = me->GetHealth() - 1;
292 }
293
294 void UpdateAI(uint32 diff) override
295 {
296 if (!UpdateVictim())
297 return;
298
299 if (_bCanResurrect)
300 {
301 //When casting resuruction make sure to delay so on rez when reinstate battle deepsleep runs out
302 if (Wait_Timer <= diff)
303 {
304 if (Creature* mograine = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MOGRAINE)))
305 {
306 DoCast(mograine, SPELL_SCARLETRESURRECTION);
307 Talk(SAY_WH_RESURRECT);
308 _bCanResurrect = false;
309 }
310 }
311 else Wait_Timer -= diff;
312 }
313
314 //Cast Deep sleep when health is less than 50%
315 if (!_bCanResurrectCheck && !HealthAbovePct(50))
316 {
317 if (me->IsNonMeleeSpellCast(false))
318 me->InterruptNonMeleeSpells(false);
319
320 DoCastVictim(SPELL_DEEPSLEEP);
321 _bCanResurrectCheck = true;
322 _bCanResurrect = true;
323 return;
324 }
325
326 //while in "resurrect-mode", don't do anything
327 if (_bCanResurrect)
328 return;
329
330 //If we are <75% hp cast healing spells at self or Mograine
331 if (Heal_Timer <= diff)
332 {
333 Creature* target = NULL;
334
335 if (!HealthAbovePct(75))
336 target = me;
337
338 if (Creature* mograine = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MOGRAINE)))
339 {
340 // checking _bCanResurrectCheck prevents her healing Mograine while he is "faking death"
341 if (_bCanResurrectCheck && mograine->IsAlive() && !mograine->HealthAbovePct(75))
342 target = mograine;
343 }
344
345 if (target)
346 DoCast(target, SPELL_HEAL);
347
348 Heal_Timer = 13000;
349 }
350 else Heal_Timer -= diff;
351
352 //PowerWordShield_Timer
353 if (PowerWordShield_Timer <= diff)
354 {
355 DoCast(me, SPELL_POWERWORDSHIELD);
356 PowerWordShield_Timer = 15000;
357 }
358 else PowerWordShield_Timer -= diff;
359
360 //HolySmite_Timer
361 if (HolySmite_Timer <= diff)
362 {
363 DoCastVictim(SPELL_HOLYSMITE);
364 HolySmite_Timer = 6000;
365 }
366 else HolySmite_Timer -= diff;
367
368 DoMeleeAttackIfReady();
369 }
370 };
371};

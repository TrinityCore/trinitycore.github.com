60Position const ScarletTraineePos = { 1939.18f, -431.58f, 17.09f, 6.22f };

62class boss_herod : public CreatureScript
63{
64 public:
65 boss_herod() : CreatureScript("boss_herod") { }
66
67 struct boss_herodAI : public BossAI
68 {
69 boss_herodAI(Creature* creature) : BossAI(creature, DATA_HEROD)
70 {
71 _enrage = false;
72 }
73
74 void Reset() override
75 {
76 _enrage = false;
77 _Reset();
78 }
79
80 void EnterCombat(Unit* /*who*/) override
81 {
82 Talk(SAY_AGGRO);
83 DoCast(me, SPELL_RUSHINGCHARGE);
84 _EnterCombat();
85
86 events.ScheduleEvent(EVENT_CLEAVE, 12000);
87 events.ScheduleEvent(EVENT_WHIRLWIND, 60000);
88 }
89
90 void KilledUnit(Unit* /*victim*/) override
91 {
92 Talk(SAY_KILL);
93 }
94
95 void JustDied(Unit* /*killer*/) override
96 {
97 _JustDied();
98
99 for (uint8 i = 0; i < 20; ++i)
100 me->SummonCreature(NPC_SCARLET_TRAINEE, ScarletTraineePos, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
101 }
102
103 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
104 {
105 if (!_enrage && me->HealthBelowPctDamaged(30, damage))
106 {
107 Talk(EMOTE_ENRAGE);
108 Talk(SAY_ENRAGE);
109 DoCast(me, SPELL_FRENZY);
110 _enrage = true;
111 }
112 }
113
114 void ExecuteEvent(uint32 eventId) override
115 {
116 switch (eventId)
117 {
118 case EVENT_CLEAVE:
119 DoCastVictim(SPELL_CLEAVE);
120 events.ScheduleEvent(EVENT_CLEAVE, 12000);
121 break;
122 case EVENT_WHIRLWIND:
123 Talk(SAY_WHIRLWIND);
124 DoCastVictim(SPELL_WHIRLWIND);
125 events.ScheduleEvent(EVENT_WHIRLWIND, 30000);
126 break;
127 default:
128 break;
129 }
130 }
131
132 private:
133 bool _enrage;
134 };
135
136 CreatureAI* GetAI(Creature* creature) const override
137 {
138 return GetScarletMonasteryAI<boss_herodAI>(creature);
139 }
140};

142class npc_scarlet_trainee : public CreatureScript
143{
144public:
145 npc_scarlet_trainee() : CreatureScript("npc_scarlet_trainee") { }
146
147 CreatureAI* GetAI(Creature* creature) const override
148 {
149 return new npc_scarlet_traineeAI(creature);
150 }
151
152 struct npc_scarlet_traineeAI : public npc_escortAI
153 {
154 npc_scarlet_traineeAI(Creature* creature) : npc_escortAI(creature)
155 {
156 Start_Timer = urand(1000, 6000);
157 }
158
159 uint32 Start_Timer;
160
161 void Reset() override { }
162 void WaypointReached(uint32 /*waypointId*/) override { }
163 void EnterCombat(Unit* /*who*/) override { }
164
165 void UpdateAI(uint32 diff) override
166 {
167 if (Start_Timer)
168 {
169 if (Start_Timer <= diff)
170 {
171 Start(true, true);
172 Start_Timer = 0;
173 } else Start_Timer -= diff;
174 }
175
176 npc_escortAI::UpdateAI(diff);
177 }
178 };
179};

30enum Spells
31{
32 SPELL_CURSEOFBLOOD = 8282,
33 SPELL_DISPELMAGIC = 15090,
34 SPELL_FEAR = 12096,
35 SPELL_HEAL = 12039,
36 SPELL_POWERWORDSHIELD = 11647,
37 SPELL_SLEEP = 8399

39
40class boss_high_inquisitor_fairbanks : public CreatureScript
41{
42public:
43 boss_high_inquisitor_fairbanks() : CreatureScript("boss_high_inquisitor_fairbanks") { }
44
45 CreatureAI* GetAI(Creature* creature) const override
46 {
47 return GetInstanceAI<boss_high_inquisitor_fairbanksAI>(creature);
48 }
49
50 struct boss_high_inquisitor_fairbanksAI : public ScriptedAI
51 {
52 boss_high_inquisitor_fairbanksAI(Creature* creature) : ScriptedAI(creature)
53 {
54 Initialize();
55 instance = creature->GetInstanceScript();
56 }
57
58 void Initialize()
59 {
60 CurseOfBlood_Timer = 10000;
61 DispelMagic_Timer = 30000;
62 Fear_Timer = 40000;
63 Heal_Timer = 30000;
64 Sleep_Timer = 30000;
65 Dispel_Timer = 20000;
66 PowerWordShield = false;
67 }
68
69 uint32 CurseOfBlood_Timer;
70 uint32 DispelMagic_Timer;
71 uint32 Fear_Timer;
72 uint32 Heal_Timer;
73 uint32 Sleep_Timer;
74 uint32 Dispel_Timer;
75 bool PowerWordShield;
76 InstanceScript* instance;
77
78 void Reset() override
79 {
80 Initialize();
81 me->SetStandState(UNIT_STAND_STATE_DEAD);
82 me->SetUInt32Value(UNIT_FIELD_BYTES_1, 7);
83 instance->SetBossState(DATA_HIGH_INQUISITOR_FAIRBANKS, NOT_STARTED);
84 }
85
86 void EnterCombat(Unit* /*who*/) override
87 {
88 me->SetStandState(UNIT_STAND_STATE_STAND);
89 me->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
90 instance->SetBossState(DATA_HIGH_INQUISITOR_FAIRBANKS, IN_PROGRESS);
91 }
92
93 void JustDied(Unit* /*killer*/) override
94 {
95 instance->SetBossState(DATA_HIGH_INQUISITOR_FAIRBANKS, DONE);
96 }
97
98 void UpdateAI(uint32 diff) override
99 {
100 if (!UpdateVictim())
101 return;
102
103 //If we are <25% hp cast Heal
104 if (!HealthAbovePct(25) && !me->IsNonMeleeSpellCast(false) && Heal_Timer <= diff)
105 {
106 DoCast(me, SPELL_HEAL);
107 Heal_Timer = 30000;
108 }
109 else Heal_Timer -= diff;
110
111 //Fear_Timer
112 if (Fear_Timer <= diff)
113 {
114 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
115 DoCast(target, SPELL_FEAR);
116
117 Fear_Timer = 40000;
118 }
119 else Fear_Timer -= diff;
120
121 //Sleep_Timer
122 if (Sleep_Timer <= diff)
123 {
124 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0))
125 DoCast(target, SPELL_SLEEP);
126
127 Sleep_Timer = 30000;
128 }
129 else Sleep_Timer -= diff;
130
131 //PowerWordShield_Timer
132 if (!PowerWordShield && !HealthAbovePct(25))
133 {
134 DoCast(me, SPELL_POWERWORDSHIELD);
135 PowerWordShield = true;
136 }
137
138 //Dispel_Timer
139 if (Dispel_Timer <= diff)
140 {
141 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
142 DoCast(target, SPELL_DISPELMAGIC);
143
144 DispelMagic_Timer = 30000;
145 }
146 else DispelMagic_Timer -= diff;
147
148 //CurseOfBlood_Timer
149 if (CurseOfBlood_Timer <= diff)
150 {
151 DoCastVictim(SPELL_CURSEOFBLOOD);
152 CurseOfBlood_Timer = 25000;
153 }
154 else CurseOfBlood_Timer -= diff;
155
156 DoMeleeAttackIfReady();
157 }
158 };
159};

164}

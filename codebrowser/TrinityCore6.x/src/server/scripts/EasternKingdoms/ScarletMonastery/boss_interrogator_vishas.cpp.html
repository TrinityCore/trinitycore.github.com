18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "scarlet_monastery.h"
21
22enum Says
23{
24 SAY_AGGRO = 0,
25 SAY_HEALTH1 = 1,
26 SAY_HEALTH2 = 2,
27 SAY_KILL = 3,
28 SAY_TRIGGER_VORREL = 0
29};
30
31enum Spells
32{
33 SPELL_SHADOW_WORD_PAIN = 2767
34};
35
36enum Events
37{
38 EVENT_SHADOW_WORD_PAIN = 1
39};
40
41class boss_interrogator_vishas : public CreatureScript
42{
43 public:
44 boss_interrogator_vishas() : CreatureScript("boss_interrogator_vishas") { }
45
46 struct boss_interrogator_vishasAI : public BossAI
47 {
48 boss_interrogator_vishasAI(Creature* creature) : BossAI(creature, DATA_INTERROGATOR_VISHAS)
49 {
50 Initialize();
51 }
52
53 void Initialize()
54 {
55 _yellCount = 0;
56 }
57
58 void Reset() override
59 {
60 Initialize();
61 _Reset();
62 }
63
64 void EnterCombat(Unit* /*who*/) override
65 {
66 Talk(SAY_AGGRO);
67 _EnterCombat();
68 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, 5000);
69 }
70
71 void KilledUnit(Unit* victim) override
72 {
73 if (victim->GetTypeId() == TYPEID_PLAYER)
74 Talk(SAY_KILL);
75 }
76
77 void JustDied(Unit* /*killer*/) override
78 {
79 _JustDied();
80 if (Creature* vorrel = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_VORREL)))
81 vorrel->AI()->Talk(SAY_TRIGGER_VORREL);
82 }
83
84 void DamageTaken(Unit* /*attacker*/, uint32 &damage) override
85 {
86 if (me->HealthBelowPctDamaged(60, damage) && _yellCount < 1)
87 {
88 Talk(SAY_HEALTH1);
89 ++_yellCount;
90 }
91
92 if (me->HealthBelowPctDamaged(30, damage) && _yellCount < 2)
93 {
94 Talk(SAY_HEALTH2);
95 ++_yellCount;
96 }
97 }
98
99 void ExecuteEvent(uint32 eventId) override
100 {
101 switch (eventId)
102 {
103 case EVENT_SHADOW_WORD_PAIN:
104 DoCastVictim(SPELL_SHADOW_WORD_PAIN);
105 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(5000, 15000));
106 break;
107 default:
108 break;
109 }
110 }
111
112 private:
113 uint8 _yellCount;
114 };
115
116 CreatureAI* GetAI(Creature* creature) const override
117 {
118 return GetInstanceAI<boss_interrogator_vishasAI>(creature);
119 }
120};
121
122void AddSC_boss_interrogator_vishas()
123{
124 new boss_interrogator_vishas();
125}

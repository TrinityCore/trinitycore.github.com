39enum eOOX
40{
41 SAY_OOX_START = 0,
42 SAY_OOX_AGGRO = 1,
43 SAY_OOX_AMBUSH = 2,
44 SAY_OOX_AMBUSH_REPLY = 3,
45 SAY_OOX_END = 4,
46 QUEST_RESQUE_OOX_09 = 836,
47 NPC_MARAUDING_OWL = 7808,
48 NPC_VILE_AMBUSHER = 7809,
49 FACTION_ESCORTEE_A = 774,
50 FACTION_ESCORTEE_H = 775
51};
52
53class npc_oox09hl : public CreatureScript
54{
55public:
56 npc_oox09hl() : CreatureScript("npc_oox09hl") { }
57
58 struct npc_oox09hlAI : public npc_escortAI
59 {
60 npc_oox09hlAI(Creature* creature) : npc_escortAI(creature) { }
61
62 void Reset() override { }
63
64 void EnterCombat(Unit* who) override
65 {
66 if (who->GetEntry() == NPC_MARAUDING_OWL || who->GetEntry() == NPC_VILE_AMBUSHER)
67 return;
68
69 Talk(SAY_OOX_AGGRO);
70 }
71
72 void JustSummoned(Creature* summoned) override
73 {
74 summoned->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
75 }
76
77 void sQuestAccept(Player* player, Quest const* quest) override
78 {
79 if (quest->GetQuestId() == QUEST_RESQUE_OOX_09)
80 {
81 me->SetStandState(UNIT_STAND_STATE_STAND);
82 me->setFaction(player->GetTeam() == ALLIANCE ? FACTION_ESCORTEE_A : FACTION_ESCORTEE_H);
83 Talk(SAY_OOX_START, player);
84 npc_escortAI::Start(false, false, player->GetGUID(), quest);
85 }
86 }
87
88 void WaypointReached(uint32 waypointId) override
89 {
90 switch (waypointId)
91 {
92 case 26:
93 Talk(SAY_OOX_AMBUSH);
94 break;
95 case 43:
96 Talk(SAY_OOX_AMBUSH);
97 break;
98 case 64:
99 Talk(SAY_OOX_END);
100 if (Player* player = GetPlayerForEscort())
101 player->GroupEventHappens(QUEST_RESQUE_OOX_09, me);
102 break;
103 }
104 }
105
106 void WaypointStart(uint32 pointId) override
107 {
108 switch (pointId)
109 {
110 case 27:
111 for (uint8 i = 0; i < 3; ++i)
112 {
113 const Position src = {147.927444f, -3851.513428f, 130.893f, 0};
114 Position dst = me->GetRandomPoint(src, 7.0f);
115 DoSummon(NPC_MARAUDING_OWL, dst, 25000, TEMPSUMMON_CORPSE_TIMED_DESPAWN);
116 }
117 break;
118 case 44:
119 for (uint8 i = 0; i < 3; ++i)
120 {
121 const Position src = {-141.151581f, -4291.213867f, 120.130f, 0};
122 Position dst = me->GetRandomPoint(src, 7.0f);
123 me->SummonCreature(NPC_VILE_AMBUSHER, dst, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 25000);
124 }
125 break;
126 }
127 }
128 };
129
130 CreatureAI* GetAI(Creature* creature) const override
131 {
132 return new npc_oox09hlAI(creature);
133 }
134};
135
136void AddSC_hinterlands()
137{
138 new npc_oox09hl();
139}

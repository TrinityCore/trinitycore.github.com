145 void StartIntro()
146 {
147 if (!Intro || IsIntro)
148 return;
149
150 if (Creature* Madrigosa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MADRIGOSA)))
151 {
152 Madrigosa->Respawn();
153 Madrigosa->setActive(true);
154 IsIntro = true;
155 Madrigosa->SetMaxHealth(me->GetMaxHealth());
156 Madrigosa->SetHealth(me->GetMaxHealth());
157 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
158 me->Attack(Madrigosa, true);
159 Madrigosa->Attack(me, true);
160 }
161 else
162 {
163 // Madrigosa not found, end intro
164 TC_LOG_ERROR("scripts", "Madrigosa was not found");
165 EndIntro();
166 }
167 }
168
169 void EndIntro()
170 {
171 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
172 Intro = false;
173 IsIntro = false;
174 }
175
176 void AttackStart(Unit* who) override
177 {
178 if (!who || Intro || IsIntro)
179 return;
180 ScriptedAI::AttackStart(who);
181 }
182
183 void DoIntro()
184 {
185 Creature* Madrigosa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MADRIGOSA));
186 if (!Madrigosa)
187 return;
188
189 switch (IntroPhase)
190 {
191 case 0:
192 Madrigosa->AI()->Talk(YELL_MADR_ICE_BARRIER);
193 IntroPhaseTimer = 7000;
194 ++IntroPhase;
195 break;
196 case 1:
197 me->SetInFront(Madrigosa);
198 Madrigosa->SetInFront(me);
199 Madrigosa->AI()->Talk(YELL_MADR_INTRO, me);
200 IntroPhaseTimer = 9000;
201 ++IntroPhase;
202 break;
203 case 2:
204 Talk(YELL_INTRO, Madrigosa);
205 IntroPhaseTimer = 13000;
206 ++IntroPhase;
207 break;
208 case 3:
209 DoCast(me, SPELL_INTRO_FROST_BLAST);
210 Madrigosa->SetDisableGravity(true);
211 me->AttackStop();
212 Madrigosa->AttackStop();
213 IntroFrostBoltTimer = 3000;
214 IntroPhaseTimer = 28000;
215 ++IntroPhase;
216 break;
217 case 4:
218 Talk(YELL_INTRO_BREAK_ICE);
219 IntroPhaseTimer = 6000;
220 ++IntroPhase;
221 break;
222 case 5:
223 Madrigosa->CastSpell(me, SPELL_INTRO_ENCAPSULATE_CHANELLING, false);
224 Madrigosa->AI()->Talk(YELL_MADR_TRAP);
225 DoCast(me, SPELL_INTRO_ENCAPSULATE);
226 IntroPhaseTimer = 11000;
227 ++IntroPhase;
228 break;
229 case 6:
230 Talk(YELL_INTRO_CHARGE);
231 IntroPhaseTimer = 5000;
232 ++IntroPhase;
233 break;
234 case 7:
235 me->Kill(Madrigosa);
236 Madrigosa->AI()->Talk(YELL_MADR_DEATH);
237 me->SetFullHealth();
238 me->AttackStop();
239 IntroPhaseTimer = 4000;
240 ++IntroPhase;
241 break;
242 case 8:
243 Talk(YELL_INTRO_KILL_MADRIGOSA);
244 me->SetOrientation(0.14f);
245 me->StopMoving();
246 Madrigosa->setDeathState(CORPSE);
247 IntroPhaseTimer = 8000;
248 ++IntroPhase;
249 break;
250 case 9:
251 Talk(YELL_INTRO_TAUNT);
252 IntroPhaseTimer = 5000;
253 ++IntroPhase;
254 break;
255 case 10:
256 EndIntro();
257 break;
258 }
259 }

264 return;

276 void UpdateAI(uint32 diff) override
277 {
278 if (IsIntro)
279 {
280 if (IntroPhaseTimer <= diff)
281 DoIntro();
282 else IntroPhaseTimer -= diff;
283
284 if (IntroPhase == 3 + 1)
285 {
286 if (IntroFrostBoltTimer <= diff)
287 {
288 if (Creature* Madrigosa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MADRIGOSA)))
289 {
290 Madrigosa->CastSpell(me, SPELL_INTRO_FROSTBOLT, true);
291 IntroFrostBoltTimer = 2000;
292 }
293 }
294 else
295 IntroFrostBoltTimer -= diff;
296 }
297
298 if (!UpdateVictim())
299 return;
300
301 DoMeleeAttackIfReady();
302 }
303
304 if (!UpdateVictim() || IsIntro)
305 return;
306
307 if (SlashTimer <= diff)
308 {
309 DoCastVictim(SPELL_METEOR_SLASH);
310 SlashTimer = 11000;
311 } else SlashTimer -= diff;
312
313 if (StompTimer <= diff)
314 {
315 Talk(YELL_LOVE);
316 DoCastVictim(SPELL_STOMP);
317 StompTimer = 30000;
318 } else StompTimer -= diff;
319
320 if (BurnTimer <= diff)
321 {
322 std::list<Unit*> targets;
323 SelectTargetList(targets, 10, SELECT_TARGET_RANDOM, 100, true);
324 for (std::list<Unit*>::const_iterator i = targets.begin(); i != targets.end(); ++i)
325 if (!(*i)->HasAura(SPELL_BURN))
326 {
327 (*i)->CastSpell((*i), SPELL_BURN, true);
328 break;
329 }
330 BurnTimer = urand(60000, 180000);
331 } else BurnTimer -= diff;
332
333 if (BerserkTimer < diff && !Enraged)
334 {
335 Talk(YELL_BERSERK);
336 DoCast(me, SPELL_BERSERK);
337 Enraged = true;
338 } else BerserkTimer -= diff;
339
340 DoMeleeAttackIfReady();
341 }
342 };

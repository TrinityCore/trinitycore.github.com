242 struct boss_kalecgos_kjAI : public ScriptedAI
243 {
244 boss_kalecgos_kjAI(Creature* creature) : ScriptedAI(creature)
245 {
246 Initialize();
247 instance = creature->GetInstanceScript();
248 }
249
250 void Initialize()
251 {
252 OrbsEmpowered = 0;
253 EmpowerCount = 0;
254 }
255
256 InstanceScript* instance;
257 uint8 OrbsEmpowered;
258 uint8 EmpowerCount;
259
260 void Reset() override
261 {
262 Initialize();
263 me->SetDisableGravity(true);
264 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
265 me->setActive(true);
266
267 for (uint8 i = 0; i < 4; ++i)
268 if (GameObject* pOrb = GetOrb(i))
269 pOrb->SetGoType(GAMEOBJECT_TYPE_BUTTON);
270 }
271
272 GameObject* GetOrb(int32 index)
273 {
274 switch (index)
275 {
276 case 0:
277 return ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_ORB_OF_THE_BLUE_DRAGONFLIGHT_1));
278 case 1:
279 return ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_ORB_OF_THE_BLUE_DRAGONFLIGHT_2));
280 case 2:
281 return ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_ORB_OF_THE_BLUE_DRAGONFLIGHT_3));
282 case 3:
283 return ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_ORB_OF_THE_BLUE_DRAGONFLIGHT_4));
284 }
285
286 return NULL;
287 }
288
289 void ResetOrbs()
290 {
291 me->RemoveDynObject(SPELL_RING_OF_BLUE_FLAMES);
292 for (uint8 i = 0; i < 4; ++i)
293 if (GameObject* pOrb = GetOrb(i))
294 pOrb->SetUInt32Value(GAMEOBJECT_FACTION, 0);
295 }
296
297 void EmpowerOrb(bool all)
298 {
299 GameObject* pOrbEmpowered = GetOrb(OrbsEmpowered);
300 if (!pOrbEmpowered)
301 return;
302
303 if (all)
304 {
305 me->RemoveDynObject(SPELL_RING_OF_BLUE_FLAMES);
306 for (uint8 i = 0; i < 4; ++i)
307 {
308 if (GameObject* pOrb = GetOrb(i))
309 {
310 pOrb->CastSpell(me, SPELL_RING_OF_BLUE_FLAMES);
311 pOrb->SetUInt32Value(GAMEOBJECT_FACTION, 35);
312 pOrb->setActive(true);
313 pOrb->Refresh();
314 }
315 }
316 Talk(SAY_KALECGOS_ENCOURAGE);
317 }
318 else
319 {
320 if (GameObject* pOrb = GetOrb(urand(0, 3)))
321 {
322 pOrb->CastSpell(me, SPELL_RING_OF_BLUE_FLAMES);
323 pOrb->SetUInt32Value(GAMEOBJECT_FACTION, 35);
324 pOrb->setActive(true);
325 pOrb->Refresh();
326
327 OrbsEmpowered = (OrbsEmpowered+1)%4;
328
329 ++EmpowerCount;
330 switch (EmpowerCount)
331 {
332 case 1: Talk(SAY_KALECGOS_READY1); break;
333 case 2: Talk(SAY_KALECGOS_READY2); break;
334 case 3: Talk(SAY_KALECGOS_READY3); break;
335 case 4: Talk(SAY_KALECGOS_READY4); break;
336 }
337 }
338 }
339 }
340
341 void UpdateAI(uint32 /*diff*/) override
342 {
343 }
344
345 void SetRingOfBlueFlames()
346 {
347 me->RemoveDynObject(SPELL_RING_OF_BLUE_FLAMES);
348 for (uint8 i = 0; i < 4; ++i)
349 {
350 if (GameObject* pOrb = GetOrb(i))
351 {
352 if (pOrb->GetUInt32Value(GAMEOBJECT_FACTION) == 35)
353 {
354 pOrb->CastSpell(me, SPELL_RING_OF_BLUE_FLAMES);
355 pOrb->setActive(true);
356 pOrb->Refresh();
357 }
358 }
359 }
360 }
361 };
362};

445 void JustSummoned(Creature* summoned) override
446 {
447 switch (summoned->GetEntry())
448 {
449 case NPC_HAND_OF_THE_DECEIVER:
450 summoned->CastSpell(summoned, SPELL_SHADOW_CHANNELING, false);
451 break;
452 case NPC_ANVEENA:
453 summoned->SetDisableGravity(true);
454 summoned->CastSpell(summoned, SPELL_ANVEENA_PRISON, true);
455 summoned->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
456 break;
457 case NPC_KILJAEDEN:
458 summoned->CastSpell(summoned, SPELL_REBIRTH, false);
459 summoned->AddThreat(me->GetVictim(), 1.0f);
460 break;
461 }
462 summons.Summon(summoned);
463 }
464
465 void UpdateAI(uint32 diff) override
466 {
467 if (uiRandomSayTimer < diff)
468 {
469 if (instance->GetBossState(DATA_MURU) != DONE && instance->GetBossState(DATA_KILJAEDEN) == NOT_STARTED)
470 Talk(SAY_KJ_OFFCOMBAT);
471 uiRandomSayTimer = 30000;
472 } else uiRandomSayTimer -= diff;
473
474 if (!bSummonedDeceivers)
475 {
476 for (uint8 i = 0; i < 3; ++i)
477 me->SummonCreature(NPC_HAND_OF_THE_DECEIVER, DeceiverLocations[i], TEMPSUMMON_DEAD_DESPAWN, 0);
478
479 DoSpawnCreature(NPC_ANVEENA, 0, 0, 40, 0, TEMPSUMMON_DEAD_DESPAWN, 0);
480 DoCast(me, SPELL_ANVEENA_ENERGY_DRAIN);
481 bSummonedDeceivers = true;
482 }
483
484 if (deceiverDeathCount > 2 && phase == PHASE_DECEIVERS)
485 {
486 me->RemoveAurasDueToSpell(SPELL_ANVEENA_ENERGY_DRAIN);
487 phase = PHASE_NORMAL;
488 DoSpawnCreature(NPC_KILJAEDEN, 0, 0, 0, 0, TEMPSUMMON_MANUAL_DESPAWN, 0);
489 }
490 }
491 };

723 TimerIsDeactivated[TIMER_SPEECH]=true;
724 speechCount++;
725 }
726 SpeechTimer += diff;
727 break;
728 case TIMER_SOUL_FLAY:

731 DoCastVictim(SPELL_SOUL_FLAY_SLOW, false);

735 break;

791 break;
792 case TIMER_DARKNESS: //Phase 3

806 }

847 break;

852 //Time runs over!
853 for (uint8 i = 0; i < ActiveTimers; ++i)
854 if (!TimerIsDeactivated[i])
855 {
856 Timer[i] -= diff;
857 if (((int32)Timer[i]) < 0) Timer[i] = 0;

877 Phase = PHASE_ARMAGEDDON;
878 ActiveTimers = 10;
879 EnterNextPhase();
880 }
881 else return;
882 }
883
884 //Phase 5 specific spells all we can
885 if (Phase <= PHASE_ARMAGEDDON && !IsInDarkness)
886 {
887 if (Phase == PHASE_ARMAGEDDON && HealthBelowPct(25))
888 {
889 Phase = PHASE_SACRIFICE;
890 EnterNextPhase();
891 }
892 else return;
893 }

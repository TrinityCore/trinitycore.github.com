113 void Reset() override
114 {
115 Enraged = false;
116
117 if (Creature* temp = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ALYTHESS)))
118 {

121 else if (temp->GetVictim())

163 void SpellHitTarget(Unit* target, const SpellInfo* spell) override
164 {
165 switch (spell->Id)
166 {
167 case SPELL_SHADOW_BLADES:
168 case SPELL_SHADOW_NOVA:
169 case SPELL_CONFOUNDING_BLOW:
170 case SPELL_SHADOW_FURY:
171 HandleTouchedSpells(target, SPELL_DARK_TOUCHED);
172 break;
173 case SPELL_CONFLAGRATION:
174 HandleTouchedSpells(target, SPELL_FLAME_TOUCHED);
175 break;
176 }
177 }
178
179 void HandleTouchedSpells(Unit* target, uint32 TouchedType)
180 {
181 switch (TouchedType)
182 {
183 case SPELL_FLAME_TOUCHED:
184 if (!target->HasAura(SPELL_DARK_FLAME))
185 {
186 if (target->HasAura(SPELL_DARK_TOUCHED))
187 {
188 target->RemoveAurasDueToSpell(SPELL_DARK_TOUCHED);
189 target->CastSpell(target, SPELL_DARK_FLAME, true);
190 } else target->CastSpell(target, SPELL_FLAME_TOUCHED, true);
191 }
192 break;
193 case SPELL_DARK_TOUCHED:
194 if (!target->HasAura(SPELL_DARK_FLAME))
195 {
196 if (target->HasAura(SPELL_FLAME_TOUCHED))
197 {
198 target->RemoveAurasDueToSpell(SPELL_FLAME_TOUCHED);
199 target->CastSpell(target, SPELL_DARK_FLAME, true);
200 } else target->CastSpell(target, SPELL_DARK_TOUCHED, true);
201 }
202 break;
203 }
204 }
205
206 void UpdateAI(uint32 diff) override
207 {
208 if (!SisterDeath)
209 {
210 Unit* Temp = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_ALYTHESS));
211 if (Temp && Temp->isDead())
212 {
213 Talk(YELL_SISTER_ALYTHESS_DEAD);
214 DoCast(me, SPELL_EMPOWER);
215 me->InterruptSpell(CURRENT_GENERIC_SPELL);
216 SisterDeath = true;
217 }
218 }
219
220 if (!UpdateVictim())
221 return;
222
223 if (SisterDeath)
224 {
225 if (ConflagrationTimer <= diff)
226 {
227 if (!me->IsNonMeleeSpellCast(false))
228 {
229 me->InterruptSpell(CURRENT_GENERIC_SPELL);
230 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
231 DoCast(target, SPELL_CONFLAGRATION);
232 ConflagrationTimer = 30000 + (rand32() % 5000);
233 }
234 } else ConflagrationTimer -= diff;
235 }
236 else
237 {
238 if (ShadownovaTimer <= diff)
239 {
240 if (!me->IsNonMeleeSpellCast(false))
241 {
242 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0);
243 if (target)
244 DoCast(target, SPELL_SHADOW_NOVA);
245
246 if (!SisterDeath)
247 {
248 if (target)
249 Talk(EMOTE_SHADOW_NOVA, target);
250 Talk(YELL_SHADOW_NOVA);
251 }
252 ShadownovaTimer = 30000 + (rand32() % 5000);
253 }
254 } else ShadownovaTimer -=diff;
255 }
256
257 if (ConfoundingblowTimer <= diff)
258 {
259 if (!me->IsNonMeleeSpellCast(false))
260 {
261 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
262 DoCast(target, SPELL_CONFOUNDING_BLOW);
263 ConfoundingblowTimer = 20000 + (rand32() % 5000);
264 }
265 } else ConfoundingblowTimer -=diff;
266
267 if (ShadowimageTimer <= diff)
268 {
269 Unit* target = NULL;
270 Creature* temp = NULL;
271 for (uint8 i = 0; i<3; ++i)
272 {
273 target = SelectTarget(SELECT_TARGET_RANDOM, 0);
274 temp = DoSpawnCreature(NPC_SHADOW_IMAGE, 0, 0, 0, 0, TEMPSUMMON_CORPSE_DESPAWN, 10000);
275 if (temp && target)
276 {
277 temp->AddThreat(target, 1000000); //don't change target(healers)
278 temp->AI()->AttackStart(target);
279 }
280 }
281 ShadowimageTimer = 20000;
282 } else ShadowimageTimer -=diff;
283
284 if (ShadowbladesTimer <= diff)
285 {
286 if (!me->IsNonMeleeSpellCast(false))
287 {
288 DoCast(me, SPELL_SHADOW_BLADES);
289 ShadowbladesTimer = 10000;
290 }
291 } else ShadowbladesTimer -=diff;
292
293 if (EnrageTimer < diff && !Enraged)
294 {
295 me->InterruptSpell(CURRENT_GENERIC_SPELL);
296 Talk(YELL_ENRAGE);
297 DoCast(me, SPELL_ENRAGE);
298 Enraged = true;
299 } else EnrageTimer -= diff;
300
301 if (me->isAttackReady() && !me->IsNonMeleeSpellCast(false))
302 {
303 //If we are within range melee the target
304 if (me->IsWithinMeleeRange(me->GetVictim()))
305 {
306 HandleTouchedSpells(me->GetVictim(), SPELL_DARK_TOUCHED);
307 me->AttackerStateUpdate(me->GetVictim());
308 me->resetAttackTimer();
309 }
310 }
311 }
312 };

314 CreatureAI* GetAI(Creature* creature) const override
315 {
316 return GetSunwellPlateauAI<boss_sacrolashAI>(creature);
317 };

488 {

492 case 0:
493 return 0;

496 Sacrolash->AI()->Talk(YELL_INTRO_SAC_1);

498 case 2:
499 Talk(YELL_INTRO_ALY_2);
500 return 1000;
501 case 3:
502 if (Sacrolash)
503 Sacrolash->AI()->Talk(YELL_INTRO_SAC_3);
504 return 2000;
505 case 4:
506 Talk(YELL_INTRO_ALY_4);
507 return 1000;
508 case 5:
509 if (Sacrolash)
510 Sacrolash->AI()->Talk(YELL_INTRO_SAC_5);
511 return 2000;
512 case 6:
513 Talk(YELL_INTRO_ALY_6);
514 return 1000;
515 case 7:

526 void UpdateAI(uint32 diff) override
527 {
528 if (IntroStepCounter < 9)
529 {
530 if (IntroYellTimer <= diff)
531 {
532 IntroYellTimer = IntroStep(++IntroStepCounter);
533 } else IntroYellTimer -= diff;
534 }
535
536 if (!SisterDeath)
537 {
538 Unit* Temp = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_SACROLASH));
539 if (Temp && Temp->isDead())
540 {
541 Talk(YELL_SISTER_SACROLASH_DEAD);
542 DoCast(me, SPELL_EMPOWER);
543 me->InterruptSpell(CURRENT_GENERIC_SPELL);
544 SisterDeath = true;
545 }
546 }
547 if (!me->GetVictim())
548 {
549 Creature* sisiter = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SACROLASH));
550 if (sisiter && !sisiter->isDead() && sisiter->GetVictim())
551 {
552 me->AddThreat(sisiter->GetVictim(), 0.0f);
553 DoStartNoMovement(sisiter->GetVictim());
554 me->Attack(sisiter->GetVictim(), false);
555 }
556 }
557
558 if (!UpdateVictim())
559 return;
560
561 if (SisterDeath)
562 {
563 if (ShadownovaTimer <= diff)
564 {
565 if (!me->IsNonMeleeSpellCast(false))
566 {
567 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
568 DoCast(target, SPELL_SHADOW_NOVA);
569 ShadownovaTimer = 30000 + (rand32() % 5000);
570 }
571 } else ShadownovaTimer -=diff;
572 }
573 else
574 {
575 if (ConflagrationTimer <= diff)
576 {
577 if (!me->IsNonMeleeSpellCast(false))
578 {
579 me->InterruptSpell(CURRENT_GENERIC_SPELL);
580 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0);
581 if (target)
582 DoCast(target, SPELL_CONFLAGRATION);
583 ConflagrationTimer = 30000 + (rand32() % 5000);
584
585 if (!SisterDeath)
586 {
587 if (target)
588 Talk(EMOTE_CONFLAGRATION, target);

591

593 }
594 } else ConflagrationTimer -= diff;
595 }
596
597 if (FlamesearTimer <= diff)
598 {
599 if (!me->IsNonMeleeSpellCast(false))
600 {
601 DoCast(me, SPELL_FLAME_SEAR);
602 FlamesearTimer = 15000;
603 }
604 } else FlamesearTimer -=diff;
605
606 if (PyrogenicsTimer <= diff)
607 {
608 if (!me->IsNonMeleeSpellCast(false))
609 {
610 DoCast(me, SPELL_PYROGENICS, true);
611 PyrogenicsTimer = 15000;
612 }
613 } else PyrogenicsTimer -= diff;
614
615 if (BlazeTimer <= diff)
616 {
617 if (!me->IsNonMeleeSpellCast(false))
618 {
619 DoCastVictim(SPELL_BLAZE);
620 BlazeTimer = 3800;
621 }
622 } else BlazeTimer -= diff;
623

627 Talk(YELL_BERSERK);

630 } else EnrageTimer -= diff;
631 }

703 } else KillTimer -= diff;
704

711 ShadowfuryTimer = 10000;
712 } else ShadowfuryTimer -=diff;

722 DarkstrikeTimer = 3000;
723 } else DarkstrikeTimer -= diff;

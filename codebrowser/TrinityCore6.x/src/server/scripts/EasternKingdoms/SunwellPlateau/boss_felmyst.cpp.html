244 void EnterPhase(PhaseFelmyst NextPhase)
245 {
246 switch (NextPhase)
247 {
248 case PHASE_GROUND:
249 me->CastStop(SPELL_FOG_BREATH);
250 me->RemoveAurasDueToSpell(SPELL_FOG_BREATH);
251 me->StopMoving();
252 me->SetSpeed(MOVE_RUN, 2.0f);
253
254 events.ScheduleEvent(EVENT_CLEAVE, urand(5000, 10000));
255 events.ScheduleEvent(EVENT_CORROSION, urand(10000, 20000));
256 events.ScheduleEvent(EVENT_GAS_NOVA, urand(15000, 20000));
257 events.ScheduleEvent(EVENT_ENCAPSULATE, urand(20000, 25000));
258 events.ScheduleEvent(EVENT_FLIGHT, 60000);
259 break;
260 case PHASE_FLIGHT:
261 me->SetDisableGravity(true);
262 events.ScheduleEvent(EVENT_FLIGHT_SEQUENCE, 1000);
263 uiFlightCount = 0;
264 uiBreathCount = 0;
265 break;
266 default:
267 break;
268 }
269 phase = NextPhase;
270 }

272 void HandleFlightSequence()
273 {
274 switch (uiFlightCount)
275 {
276 case 0:
277 //me->AttackStop();
278 me->GetMotionMaster()->Clear(false);
279 me->HandleEmoteCommand(EMOTE_ONESHOT_LIFTOFF);
280 me->StopMoving();
281 Talk(YELL_TAKEOFF);
282 events.ScheduleEvent(EVENT_FLIGHT_SEQUENCE, 2000);
283 break;
284 case 1:
285 me->GetMotionMaster()->MovePoint(0, me->GetPositionX()+1, me->GetPositionY(), me->GetPositionZ()+10);
286 break;
287 case 2:
288 {
289 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 150, true);
290 if (!target)
291 target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_PLAYER_GUID));
292
293 if (!target)
294 {
295 EnterEvadeMode();
296 return;
297 }
298
299 if (Creature* Vapor = me->SummonCreature(NPC_VAPOR, target->GetPositionX() - 5 + rand32() % 10, target->GetPositionY() - 5 + rand32() % 10, target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 9000))
300 {
301 Vapor->AI()->AttackStart(target);
302 me->InterruptNonMeleeSpells(false);
303 DoCast(Vapor, SPELL_VAPOR_CHANNEL, false); // core bug
304 Vapor->CastSpell(Vapor, SPELL_VAPOR_TRIGGER, true);
305 }
306
307 events.ScheduleEvent(EVENT_FLIGHT_SEQUENCE, 10000);
308 break;
309 }
310 case 3:
311 {
312 DespawnSummons(NPC_VAPOR_TRAIL);
313 //DoCast(me, SPELL_VAPOR_SELECT); need core support
314
315 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 150, true);
316 if (!target)
317 target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_PLAYER_GUID));
318
319 if (!target)
320 {
321 EnterEvadeMode();
322 return;
323 }
324
325 //target->CastSpell(target, SPELL_VAPOR_SUMMON, true); need core support
326 if (Creature* pVapor = me->SummonCreature(NPC_VAPOR, target->GetPositionX() - 5 + rand32() % 10, target->GetPositionY() - 5 + rand32() % 10, target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 9000))
327 {
328 if (pVapor->AI())
329 pVapor->AI()->AttackStart(target);
330 me->InterruptNonMeleeSpells(false);
331 DoCast(pVapor, SPELL_VAPOR_CHANNEL, false); // core bug
332 pVapor->CastSpell(pVapor, SPELL_VAPOR_TRIGGER, true);
333 }
334
335 events.ScheduleEvent(EVENT_FLIGHT_SEQUENCE, 10000);
336 break;
337 }
338 case 4:
339 DespawnSummons(NPC_VAPOR_TRAIL);
340 events.ScheduleEvent(EVENT_FLIGHT_SEQUENCE, 1);
341 break;
342 case 5:
343 {
344 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 150, true);
345 if (!target)
346 target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_PLAYER_GUID));
347
348 if (!target)
349 {
350 EnterEvadeMode();
351 return;
352 }
353
354 breathX = target->GetPositionX();
355 breathY = target->GetPositionY();
356 float x, y, z;
357 target->GetContactPoint(me, x, y, z, 70);
358 me->GetMotionMaster()->MovePoint(0, x, y, z+10);
359 break;
360 }
361 case 6:
362 me->SetFacingTo(me->GetAngle(breathX, breathY));
363 //DoTextEmote("takes a deep breath.", NULL);
364 events.ScheduleEvent(EVENT_FLIGHT_SEQUENCE, 10000);
365 break;
366 case 7:
367 {
368 DoCast(me, SPELL_FOG_BREATH, true);
369 float x, y, z;
370 me->GetPosition(x, y, z);
371 x = 2 * breathX - x;
372 y = 2 * breathY - y;
373 me->GetMotionMaster()->MovePoint(0, x, y, z);
374 events.ScheduleEvent(EVENT_SUMMON_FOG, 1);
375 break;
376 }
377 case 8:
378 me->CastStop(SPELL_FOG_BREATH);
379 me->RemoveAurasDueToSpell(SPELL_FOG_BREATH);
380 ++uiBreathCount;
381 events.ScheduleEvent(EVENT_FLIGHT_SEQUENCE, 1);
382 if (uiBreathCount < 3)
383 uiFlightCount = 4;
384 break;
385 case 9:
386 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO))
387 DoStartMovement(target);
388 else
389 {
390 EnterEvadeMode();
391 return;
392 }
393 break;
394 case 10:
395 me->SetDisableGravity(false);
396 me->HandleEmoteCommand(EMOTE_ONESHOT_LAND);
397 EnterPhase(PHASE_GROUND);
398 AttackStart(SelectTarget(SELECT_TARGET_TOPAGGRO));
399 break;
400 }
401 ++uiFlightCount;
402 }

404 void UpdateAI(uint32 diff) override
405 {
406 if (!UpdateVictim())
407 {
408 if (phase == PHASE_FLIGHT && !me->IsInEvadeMode())
409 EnterEvadeMode();
410 return;
411 }
412
413 events.Update(diff);
414
415 if (me->IsNonMeleeSpellCast(false))
416 return;
417
418 if (phase == PHASE_GROUND)
419 {
420 switch (events.ExecuteEvent())
421 {
422 case EVENT_BERSERK:
423 Talk(YELL_BERSERK);
424 DoCast(me, SPELL_BERSERK, true);
425 events.ScheduleEvent(EVENT_BERSERK, 10000);
426 break;
427 case EVENT_CLEAVE:
428 DoCastVictim(SPELL_CLEAVE, false);
429 events.ScheduleEvent(EVENT_CLEAVE, urand(5000, 10000));
430 break;
431 case EVENT_CORROSION:
432 DoCastVictim(SPELL_CORROSION, false);
433 events.ScheduleEvent(EVENT_CORROSION, urand(20000, 30000));
434 break;
435 case EVENT_GAS_NOVA:
436 DoCast(me, SPELL_GAS_NOVA, false);
437 events.ScheduleEvent(EVENT_GAS_NOVA, urand(20000, 25000));
438 break;
439 case EVENT_ENCAPSULATE:
440 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 150, true))
441 DoCast(target, SPELL_ENCAPSULATE_CHANNEL, false);
442 events.ScheduleEvent(EVENT_ENCAPSULATE, urand(25000, 30000));
443 break;
444 case EVENT_FLIGHT:
445 EnterPhase(PHASE_FLIGHT);
446 break;
447 default:
448 DoMeleeAttackIfReady();
449 break;
450 }
451 }
452
453 if (phase == PHASE_FLIGHT)
454 {
455 switch (events.ExecuteEvent())
456 {
457 case EVENT_BERSERK:
458 Talk(YELL_BERSERK);
459 DoCast(me, SPELL_BERSERK, true);
460 break;
461 case EVENT_FLIGHT_SEQUENCE:
462 HandleFlightSequence();
463 break;
464 case EVENT_SUMMON_FOG:
465 {
466 float x, y, z;
467 me->GetPosition(x, y, z);
468 me->UpdateGroundPositionZ(x, y, z);
469 if (Creature* Fog = me->SummonCreature(NPC_VAPOR_TRAIL, x, y, z, 0, TEMPSUMMON_TIMED_DESPAWN, 10000))
470 {
471 Fog->RemoveAurasDueToSpell(SPELL_TRAIL_TRIGGER);
472 Fog->CastSpell(Fog, SPELL_FOG_TRIGGER, true);
473 me->CastSpell(Fog, SPELL_FOG_FORCE, true);
474 }
475 }
476 events.ScheduleEvent(EVENT_SUMMON_FOG, 1000);
477 break;
478 }
479 }

482 void DespawnSummons(uint32 entry)
483 {
484 std::list<Creature*> templist;
485 float x, y, z;
486 me->GetPosition(x, y, z);
487
488 CellCoord pair(Trinity::ComputeCellCoord(x, y));
489 Cell cell(pair);
490 cell.SetNoCreate();
491
492 Trinity::AllCreaturesOfEntryInRange check(me, entry, 100);
493 Trinity::CreatureListSearcher<Trinity::AllCreaturesOfEntryInRange> searcher(me, templist, check);
494 TypeContainerVisitor<Trinity::CreatureListSearcher<Trinity::AllCreaturesOfEntryInRange>, GridTypeMapContainer> cSearcher(searcher);
495 cell.Visit(pair, cSearcher, *(me->GetMap()), *me, me->GetGridActivationRange());
496
497 for (std::list<Creature*>::const_iterator i = templist.begin(); i != templist.end(); ++i)
498 {
499 if (entry == NPC_VAPOR_TRAIL && phase == PHASE_FLIGHT)
500 {
501 (*i)->GetPosition(x, y, z);
502 me->SummonCreature(NPC_DEAD, x, y, z, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 5000);
503 }
504 (*i)->SetVisible(false);
505 (*i)->setDeathState(JUST_DIED);
506 if ((*i)->getDeathState() == CORPSE)
507 (*i)->RemoveCorpse();
508 }
509 }

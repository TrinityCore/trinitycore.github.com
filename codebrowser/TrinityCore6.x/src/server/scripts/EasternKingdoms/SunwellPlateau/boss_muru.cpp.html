102class boss_entropius : public CreatureScript
103{
104public:
105 boss_entropius() : CreatureScript("boss_entropius") { }
106
107 struct boss_entropiusAI : public BossAI
108 {
109 boss_entropiusAI(Creature* creature) : BossAI(creature, DATA_MURU) { }
110
111 void Reset() override
112 {
113 DoCastAOE(SPELL_NEGATIVE_ENERGY_E);
114 }
115
116 void EnterCombat(Unit* /*who*/) override
117 {
118 _EnterCombat();
119 DoCastAOE(SPELL_NEGATIVE_ENERGY_E, true);
120 DoCast(me, SPELL_ENTROPIUS_SPAWN);
121 events.ScheduleEvent(EVENT_SUMMON_BLACK_HOLE, 15000);
122 }
123
124 void JustSummoned(Creature* summoned) override
125 {
126 switch (summoned->GetEntry())
127 {
128 case NPC_DARK_FIENDS:
129 summoned->CastSpell(summoned, SPELL_DARKFIEND_VISUAL);
130 break;
131 case NPC_DARKNESS:
132 summoned->AddUnitState(UNIT_STATE_STUNNED);
133 float x, y, z, o;
134 summoned->GetHomePosition(x, y, z, o);
135 me->SummonCreature(NPC_DARK_FIENDS, x, y, z, o, TEMPSUMMON_CORPSE_DESPAWN, 0);
136 break;
137 }
138 summoned->AI()->AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0, 50, true));
139 summons.Summon(summoned);
140 }
141
142 void ExecuteEvent(uint32 eventId) override
143 {
144 if (eventId == EVENT_SUMMON_BLACK_HOLE)
145 {
146 if (Unit* random = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
147 DoCast(random, SPELL_DARKNESS_E);
148 if (Unit* random = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
149 random->CastSpell(random, SPELL_BLACKHOLE);
150 events.ScheduleEvent(EVENT_SUMMON_BLACK_HOLE, 15000);
151 }
152 }
153
154 void EnterEvadeMode() override
155 {
156 if (Creature* muru = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MURU)))
157 muru->AI()->Reset(); // Reset encounter.
158 me->DisappearAndDie();
159 summons.DespawnAll();
160 }
161 };
162
163 CreatureAI* GetAI(Creature* creature) const override
164 {
165 return GetSunwellPlateauAI<boss_entropiusAI>(creature);
166 }

169class boss_muru : public CreatureScript
170{
171public:
172 boss_muru() : CreatureScript("boss_muru") { }
173
174 struct boss_muruAI : public BossAI
175 {
176 boss_muruAI(Creature* creature) : BossAI(creature, DATA_MURU)
177 {
178 Initialize();
179 SetCombatMovement(false);
180 }
181
182 void Initialize()
183 {
184 DarkFiend = false;
185 HasEnraged = false;
186 EntropiusGUID.Clear();
187 }
188
189 void Reset() override
190 {
191 Initialize();
192 _Reset();
193 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
194 me->SetVisible(true);
195 }
196
197 void EnterCombat(Unit* /*who*/) override
198 {
199 _EnterCombat();
200 events.SetPhase(PHASE_ONE);
201 events.ScheduleEvent(EVENT_ENRAGE, 600000);
202 events.ScheduleEvent(EVENT_DARKNESS, 45000, 0, PHASE_ONE);
203 events.ScheduleEvent(EVENT_SUMMON_HUMANOIDS, 10000, 0, PHASE_ONE);
204 events.ScheduleEvent(EVENT_SUMMON_SENTINEL, 31500, 0, PHASE_ONE);
205 DoCastAOE(SPELL_NEGATIVE_ENERGY);
206 }
207
208 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override
209 {
210 if (damage >= me->GetHealth() && events.IsInPhase(PHASE_ONE))
211 {
212 damage = 0;
213 me->RemoveAllAuras();
214 DoCast(me, SPELL_OPEN_ALL_PORTALS);
215 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
216 events.SetPhase(PHASE_TWO);
217 events.ScheduleEvent(EVENT_PHASE_TRANSITION, 2000);
218 }
219 }
220
221 void JustSummoned(Creature* summoned) override
222 {
223 switch (summoned->GetEntry())
224 {
225 case NPC_ENTROPIUS:
226 me->SetVisible(false);
227 EntropiusGUID = summoned->GetGUID();
228 if (HasEnraged) // If we hit phase transition while enraged, enrage Entropius as well.
229 summoned->CastSpell(summoned, SPELL_ENRAGE);
230 break;
231 case NPC_DARK_FIENDS:
232 summoned->CastSpell(summoned, SPELL_DARKFIEND_VISUAL);
233 break;
234 }
235 summoned->AI()->AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0, 50, true));
236 summons.Summon(summoned);
237 }
238
239 void ExecuteEvent(uint32 eventId) override
240 {
241 switch (eventId)
242 {
243 case EVENT_DARKNESS:
244 if (!DarkFiend)
245 {
246 DarkFiend = true;
247 DoCastAOE(SPELL_DARKNESS);
248 }
249 else
250 me->SummonCreatureGroup(CREATURE_GROUP_DARKFIENDS);
251 events.ScheduleEvent(EVENT_DARKNESS, DarkFiend ? 3000 : 42000, 0, PHASE_ONE);
252 break;
253 case EVENT_SUMMON_HUMANOIDS:
254 me->SummonCreatureGroup(CREATURE_GROUP_HUMANOIDS);
255 events.ScheduleEvent(EVENT_SUMMON_HUMANOIDS, 60000, 0, PHASE_ONE);
256 break;
257 case EVENT_SUMMON_SENTINEL:
258 DoCastAOE(SPELL_OPEN_PORTAL_2);
259 events.ScheduleEvent(EVENT_SUMMON_SENTINEL, 30000, 0, PHASE_ONE);
260 break;
261 case EVENT_PHASE_TRANSITION:
262 DoCast(me, SPELL_SUMMON_ENTROPIUS);
263 break;
264 case EVENT_ENRAGE:
265 if (Creature* entropius = ObjectAccessor::GetCreature(*me, EntropiusGUID))
266 entropius->CastSpell(entropius, SPELL_ENRAGE);
267 DoCast(me, SPELL_ENRAGE);
268 HasEnraged = true;
269 break;
270 default:
271 break;
272 }
273 }
274
275 private:
276 bool DarkFiend;
277 bool HasEnraged;
278 ObjectGuid EntropiusGUID;
279 };
280
281 CreatureAI* GetAI(Creature* creature) const override
282 {
283 return GetSunwellPlateauAI<boss_muruAI>(creature);
284 }

299 npc_muru_portalAI(Creature* creature) : ScriptedAI(creature), Summons(creature)
300 {
301 Initialize();

303 instance = creature->GetInstanceScript();
304 }
305
306 void Initialize()
307 {
308 SummonTimer = 5000;
309
310 InAction = false;
311 SummonSentinel = false;
312 }
313
314 InstanceScript* instance;
315
316 SummonList Summons;
317
318 bool SummonSentinel;
319 bool InAction;
320
321 uint32 SummonTimer;
322
323 void Reset() override
324 {
325 Initialize();
326
327 me->AddUnitState(UNIT_STATE_STUNNED);
328
329 Summons.DespawnAll();
330 }
331
332 void JustSummoned(Creature* summoned) override
333 {
334 if (Player* target = ObjectAccessor::GetPlayer(*me, instance->GetGuidData(DATA_PLAYER_GUID)))
335 summoned->AI()->AttackStart(target);
336
337 Summons.Summon(summoned);
338 }
339
340 void SpellHit(Unit* /*caster*/, const SpellInfo* Spell) override
341 {
342 float x, y, z, o;
343 me->GetHomePosition(x, y, z, o);
344 me->NearTeleportTo(x, y, z, o);
345 InAction = true;
346 switch (Spell->Id)
347 {
348 case SPELL_OPEN_ALL_PORTALS:
349 DoCastAOE(SPELL_OPEN_PORTAL);
350 break;
351 case SPELL_OPEN_PORTAL_2:
352 DoCastAOE(SPELL_OPEN_PORTAL);
353 SummonSentinel = true;
354 break;
355 }
356 }
357
358 void UpdateAI(uint32 diff) override
359 {
360 if (!SummonSentinel)
361 {
362 if (InAction && instance->GetBossState(DATA_MURU) == NOT_STARTED)
363 Reset();
364 return;
365 }
366
367 if (SummonTimer <= diff)
368 {
369 DoCastAOE(SPELL_SUMMON_VOID_SENTINEL, false);
370 SummonTimer = 5000;
371 SummonSentinel = false;
372 } else SummonTimer -= diff;
373 }
374 };

393
394 void Initialize()
395 {
396 WaitTimer = 2000;
397 InAction = false;
398 }
399
400 uint32 WaitTimer;
401 bool InAction;
402
403 void Reset() override
404 {
405 Initialize();
406
407 me->AddUnitState(UNIT_STATE_STUNNED);
408 }
409
410 void SpellHit(Unit* /*caster*/, const SpellInfo* Spell) override
411 {
412 for (SpellEffectInfo const* effect : Spell->GetEffectsForDifficulty(DIFFICULTY_NONE))
413 {
414 if (effect && effect->Effect == 38)
415 {
416 me->DisappearAndDie();
417 return;
418 }
419 }
420 }
421
422 void UpdateAI(uint32 diff) override
423 {
424 if (!UpdateVictim())
425 return;
426
427 if (WaitTimer <= diff)
428 {
429 if (!InAction)
430 {
431 me->ClearUnitState(UNIT_STATE_STUNNED);
432 DoCastAOE(SPELL_DARKFIEND_SKIN, false);
433 AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true));
434 InAction = true;
435 WaitTimer = 500;
436 }
437 else
438 {
439 if (me->IsWithinDist(me->GetVictim(), 5))
440 {
441 DoCastAOE(SPELL_DARKFIEND_AOE, false);
442 me->DisappearAndDie();
443 }
444 WaitTimer = 500;
445 }
446 } else WaitTimer -= diff;
447 }
448 };

525 struct npc_blackholeAI : public ScriptedAI
526 {
527 npc_blackholeAI(Creature* creature) : ScriptedAI(creature)
528 {
529 Initialize();
530 instance = creature->GetInstanceScript();
531 }
532
533 void Initialize()
534 {
535 DespawnTimer = 15000;
536 SpellTimer = 5000;
537 Phase = 0;
538 NeedForAHack = 0;
539 }
540
541 InstanceScript* instance;
542
543 uint32 DespawnTimer;
544 uint32 SpellTimer;
545 uint8 Phase;
546 uint8 NeedForAHack;
547
548 void Reset() override
549 {
550 Initialize();
551
552 me->AddUnitState(UNIT_STATE_STUNNED);
553 DoCastAOE(SPELL_BLACKHOLE_SPAWN, true);
554 }
555
556 void UpdateAI(uint32 diff) override
557 {
558 if (SpellTimer <= diff)
559 {
560 Unit* Victim = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_PLAYER_GUID));
561 switch (NeedForAHack)
562 {
563 case 0:
564 me->ClearUnitState(UNIT_STATE_STUNNED);
565 DoCastAOE(SPELL_BLACKHOLE_GROW, false);
566 if (Victim)
567 AttackStart(Victim);
568 SpellTimer = 700;
569 NeedForAHack = 2;
570 break;
571 case 1:
572 me->AddAura(SPELL_BLACKHOLE_GROW, me);
573 NeedForAHack = 2;
574 SpellTimer = 600;
575 break;
576 case 2:
577 SpellTimer = 400;
578 NeedForAHack = 3;
579 me->RemoveAura(SPELL_BLACKHOLE_GROW);
580 break;
581 case 3:
582 SpellTimer = urand(400, 900);
583 NeedForAHack = 1;
584 if (Unit* Temp = me->GetVictim())
585 {
586 if (Temp->GetPositionZ() > 73 && Victim)
587 AttackStart(Victim);
588 } else
589 return;
590 }
591 } else SpellTimer -= diff;
592
593 if (DespawnTimer <= diff)
594 me->DisappearAndDie();
595 else DespawnTimer -= diff;
596 }
597 };

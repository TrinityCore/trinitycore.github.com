65class boss_selin_fireheart : public CreatureScript
66{
67 public:
68 boss_selin_fireheart() : CreatureScript("boss_selin_fireheart") { }
69
70 struct boss_selin_fireheartAI : public BossAI
71 {
72 boss_selin_fireheartAI(Creature* creature) : BossAI(creature, DATA_SELIN)
73 {
74 _scheduledEvents = false;
75 }
76
77 void Reset() override
78 {
79 Crystals.clear();
80 me->GetCreatureListWithEntryInGrid(Crystals, NPC_FEL_CRYSTAL, 250.0f);
81
82 for (Creature* creature : Crystals)
83 {
84 if (!creature->IsAlive())
85 creature->Respawn();
86
87 creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
88 }
89
90 _Reset();
91 CrystalGUID.Clear();
92 _scheduledEvents = false;
93 }
94
95 void DoAction(int32 action) override
96 {
97 switch (action)
98 {
99 case ACTION_SWITCH_PHASE:
100 events.SetPhase(PHASE_NORMAL);
101 events.ScheduleEvent(EVENT_FEL_EXPLOSION, 2000, 0, PHASE_NORMAL);
102 AttackStart(me->GetVictim());
103 me->GetMotionMaster()->MoveChase(me->GetVictim());
104 break;
105 default:
106 break;
107 }
108 }
109
110 void SelectNearestCrystal()
111 {
112 if (Crystals.empty())
113 return;
114
115 Crystals.sort(Trinity::ObjectDistanceOrderPred(me));
116 if (Creature* CrystalChosen = Crystals.front())
117 {
118 Talk(SAY_ENERGY);
119 Talk(EMOTE_CRYSTAL);
120
121 DoCast(CrystalChosen, SPELL_FEL_CRYSTAL_DUMMY);
122 CrystalGUID = CrystalChosen->GetGUID();
123 Crystals.remove(CrystalChosen);
124
125 float x, y, z;
126 CrystalChosen->GetClosePoint(x, y, z, me->GetObjectSize(), CONTACT_DISTANCE);
127
128 events.SetPhase(PHASE_DRAIN);
129 me->SetWalk(false);
130 me->GetMotionMaster()->MovePoint(1, x, y, z);
131 }
132 }
133
134 void ShatterRemainingCrystals()
135 {
136 if (Crystals.empty())
137 return;
138
139 for (Creature* crystal : Crystals)
140 {
141 if (crystal && crystal->IsAlive())
142 crystal->KillSelf();
143 }
144 }
145
146 void EnterCombat(Unit* /*who*/) override
147 {
148 Talk(SAY_AGGRO);
149 _EnterCombat();
150
151 events.SetPhase(PHASE_NORMAL);
152 events.ScheduleEvent(EVENT_FEL_EXPLOSION, 2100, 0, PHASE_NORMAL);
153 }
154
155 void KilledUnit(Unit* victim) override
156 {
157 if (victim->GetTypeId() == TYPEID_PLAYER)
158 Talk(SAY_KILL);
159 }
160
161 void MovementInform(uint32 type, uint32 id) override
162 {
163 if (type == POINT_MOTION_TYPE && id == 1)
164 {
165 Unit* CrystalChosen = ObjectAccessor::GetUnit(*me, CrystalGUID);
166 if (CrystalChosen && CrystalChosen->IsAlive())
167 {
168 CrystalChosen->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
169 CrystalChosen->CastSpell(me, SPELL_MANA_RAGE, true);
170 events.ScheduleEvent(EVENT_EMPOWER, 10000, PHASE_DRAIN);
171 }
172 }
173 }
174
175 void JustDied(Unit* /*killer*/) override
176 {
177 Talk(SAY_DEATH);
178 _JustDied();
179
180 ShatterRemainingCrystals();
181 }
182
183 void UpdateAI(uint32 diff) override
184 {
185 if (!UpdateVictim())
186 return;
187
188 events.Update(diff);
189
190 if (me->HasUnitState(UNIT_STATE_CASTING))
191 return;
192
193 while (uint32 eventId = events.ExecuteEvent())
194 {
195 switch (eventId)
196 {
197 case EVENT_FEL_EXPLOSION:
198 DoCastAOE(SPELL_FEL_EXPLOSION);
199 events.ScheduleEvent(EVENT_FEL_EXPLOSION, 2000, 0, PHASE_NORMAL);
200 break;
201 case EVENT_DRAIN_CRYSTAL:
202 SelectNearestCrystal();
203 _scheduledEvents = false;
204 break;
205 case EVENT_DRAIN_MANA:
206 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
207 DoCast(target, SPELL_DRAIN_MANA);
208 events.ScheduleEvent(EVENT_DRAIN_MANA, 10000, 0, PHASE_NORMAL);
209 break;
210 case EVENT_DRAIN_LIFE:
211 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 20.0f, true))
212 DoCast(target, SPELL_DRAIN_LIFE);
213 events.ScheduleEvent(EVENT_DRAIN_LIFE, 10000, 0, PHASE_NORMAL);
214 break;
215 case EVENT_EMPOWER:
216 {
217 Talk(SAY_EMPOWERED);
218
219 Creature* CrystalChosen = ObjectAccessor::GetCreature(*me, CrystalGUID);
220 if (CrystalChosen && CrystalChosen->IsAlive())
221 CrystalChosen->KillSelf();
222
223 CrystalGUID.Clear();
224
225 me->GetMotionMaster()->Clear();
226 me->GetMotionMaster()->MoveChase(me->GetVictim());
227 break;
228 }
229 default:
230 break;
231 }
232 }
233
234 if (me->GetPowerPct(POWER_MANA) < 10.f)
235 {
236 if (events.IsInPhase(PHASE_NORMAL) && !_scheduledEvents)
237 {
238 _scheduledEvents = true;
239 uint32 timer = urand(3000, 7000);
240 events.ScheduleEvent(EVENT_DRAIN_LIFE, timer, 0, PHASE_NORMAL);
241
242 if (IsHeroic())
243 {
244 events.ScheduleEvent(EVENT_DRAIN_CRYSTAL, urand(10000, 15000), 0, PHASE_NORMAL);
245 events.ScheduleEvent(EVENT_DRAIN_MANA, timer + 5000, 0, PHASE_NORMAL);
246 }
247 else
248 events.ScheduleEvent(EVENT_DRAIN_CRYSTAL, urand(20000, 25000), 0, PHASE_NORMAL);
249 }
250 }
251
252 DoMeleeAttackIfReady();
253 }
254
255 private:
256 std::list<Creature*> Crystals;
257 ObjectGuid CrystalGUID;
258 bool _scheduledEvents;
259 };
260
261 CreatureAI* GetAI(Creature* creature) const override
262 {
263 return GetInstanceAI<boss_selin_fireheartAI>(creature);
264 };

273 {

276 void JustDied(Unit* /*killer*/) override
277 {
278 if (InstanceScript* instance = me->GetInstanceScript())
279 {
280 Creature* Selin = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SELIN));

282 Selin->AI()->DoAction(ACTION_SWITCH_PHASE);
283 }
284 }
285 };

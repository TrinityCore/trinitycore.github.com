60class boss_vexallus : public CreatureScript
61{
62 public:
63 boss_vexallus() : CreatureScript("boss_vexallus") { }
64
65 struct boss_vexallusAI : public BossAI
66 {
67 boss_vexallusAI(Creature* creature) : BossAI(creature, DATA_VEXALLUS)
68 {
69 _intervalHealthAmount = 1;
70 _enraged = false;
71 }
72
73 void Reset() override
74 {
75 _Reset();
76 _intervalHealthAmount = 1;
77 _enraged = false;
78 }
79
80 void KilledUnit(Unit* /*victim*/) override
81 {
82 Talk(SAY_KILL);
83 }
84
85 void JustDied(Unit* /*killer*/) override
86 {
87 _JustDied();
88 }
89
90 void EnterCombat(Unit* /*who*/) override
91 {
92 Talk(SAY_AGGRO);
93 _EnterCombat();
94
95 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 8000);
96 events.ScheduleEvent(EVENT_ARCANE_SHOCK, 5000);
97 }
98
99 void JustSummoned(Creature* summoned) override
100 {
101 if (Unit* temp = SelectTarget(SELECT_TARGET_RANDOM, 0))
102 summoned->GetMotionMaster()->MoveFollow(temp, 0, 0);
103
104 summons.Summon(summoned);
105 }
106
107 void DamageTaken(Unit* /*who*/, uint32& /*damage*/) override
108 {
109 if (_enraged)
110 return;
111
112 // 85%, 70%, 55%, 40%, 25%
113 if (!HealthAbovePct(100 - INTERVAL_MODIFIER * _intervalHealthAmount))
114 {
115 // increase amount, unless we're at 10%, then we switch and return
116 if (_intervalHealthAmount == INTERVAL_SWITCH)
117 {
118 _enraged = true;
119 events.Reset();
120 events.ScheduleEvent(EVENT_OVERLOAD, 1200);
121 return;
122 }
123 else
124 ++_intervalHealthAmount;
125
126 Talk(SAY_ENERGY);
127 Talk(EMOTE_DISCHARGE_ENERGY);
128
129 if (IsHeroic())
130 {
131 DoCast(me, H_SPELL_SUMMON_PURE_ENERGY1);
132 DoCast(me, H_SPELL_SUMMON_PURE_ENERGY2);
133 }
134 else
135 DoCast(me, SPELL_SUMMON_PURE_ENERGY);
136 }
137 }
138
139 void UpdateAI(uint32 diff) override
140 {
141 if (!UpdateVictim())
142 return;
143
144 events.Update(diff);
145
146 if (me->HasUnitState(UNIT_STATE_CASTING))
147 return;
148
149 while (uint32 eventId = events.ExecuteEvent())
150 {
151 switch (eventId)
152 {
153 case EVENT_CHAIN_LIGHTNING:
154 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
155 DoCast(target, SPELL_CHAIN_LIGHTNING);
156 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 8000);
157 break;
158 case EVENT_ARCANE_SHOCK:
159 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 20.0f, true))
160 DoCast(target, SPELL_ARCANE_SHOCK);
161 events.ScheduleEvent(EVENT_ARCANE_SHOCK, 8000);
162 break;
163 case EVENT_OVERLOAD:
164 DoCastVictim(SPELL_OVERLOAD);
165 events.ScheduleEvent(EVENT_OVERLOAD, 2000);
166 break;
167 default:
168 break;
169 }
170 }
171
172 DoMeleeAttackIfReady();
173 }
174
175 private:
176 uint32 _intervalHealthAmount;
177 bool _enraged;
178 };
179
180 CreatureAI* GetAI(Creature* creature) const override
181 {
182 return GetInstanceAI<boss_vexallusAI>(creature);
183 };
184};

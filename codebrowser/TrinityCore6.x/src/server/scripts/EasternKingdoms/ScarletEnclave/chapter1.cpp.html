96 struct npc_unworthy_initiateAI : public ScriptedAI
97 {
98 npc_unworthy_initiateAI(Creature* creature) : ScriptedAI(creature)
99 {
100 Initialize();
101 me->SetReactState(REACT_PASSIVE);
102 if (!me->GetCurrentEquipmentId())
103 me->SetCurrentEquipmentId(me->GetOriginalEquipmentId());
104
105 wait_timer = 0;
106 anchorX = 0.f;
107 anchorY = 0.f;
108 }
109
110 void Initialize()
111 {
112 anchorGUID.Clear();
113 phase = PHASE_CHAINED;
114 }
115
116 ObjectGuid playerGUID;
117 UnworthyInitiatePhase phase;
118 uint32 wait_timer;
119 float anchorX, anchorY;
120 ObjectGuid anchorGUID;
121
122 EventMap events;
123
124 void Reset() override
125 {
126 Initialize();
127 events.Reset();
128 me->setFaction(7);
129 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
130 me->SetUInt32Value(UNIT_FIELD_BYTES_1, 8);
131 me->LoadEquipment(0, true);
132 }
133
134 void EnterCombat(Unit* /*who*/) override
135 {
136 events.ScheduleEvent(EVENT_ICY_TOUCH, 1000, GCD_CAST);
137 events.ScheduleEvent(EVENT_PLAGUE_STRIKE, 3000, GCD_CAST);
138 events.ScheduleEvent(EVENT_BLOOD_STRIKE, 2000, GCD_CAST);
139 events.ScheduleEvent(EVENT_DEATH_COIL, 5000, GCD_CAST);
140 }
141
142 void MovementInform(uint32 type, uint32 id) override
143 {
144 if (type != POINT_MOTION_TYPE)
145 return;
146
147 if (id == 1)
148 {
149 wait_timer = 5000;
150 me->LoadEquipment(1);
151 me->CastSpell(me, SPELL_DK_INITIATE_VISUAL, true);
152
153 if (Player* starter = ObjectAccessor::GetPlayer(*me, playerGUID))
154 Talk(SAY_EVENT_ATTACK, starter);
155
156 phase = PHASE_TO_ATTACK;
157 }
158 }
159
160 void EventStart(Creature* anchor, Player* target)
161 {
162 wait_timer = 5000;
163 phase = PHASE_TO_EQUIP;
164
165 me->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
166 me->RemoveAurasDueToSpell(SPELL_SOUL_PRISON_CHAIN_SELF);
167 me->RemoveAurasDueToSpell(SPELL_SOUL_PRISON_CHAIN);
168
169 float z;
170 anchor->GetContactPoint(me, anchorX, anchorY, z, 1.0f);
171
172 playerGUID = target->GetGUID();
173 Talk(SAY_EVENT_START, target);
174 }
175
176 void UpdateAI(uint32 diff) override
177 {
178 switch (phase)
179 {
180 case PHASE_CHAINED:
181 if (!anchorGUID)
182 {
183 if (Creature* anchor = me->FindNearestCreature(29521, 30))
184 {
185 anchor->AI()->SetGUID(me->GetGUID());
186 anchor->CastSpell(me, SPELL_SOUL_PRISON_CHAIN, true);
187 anchorGUID = anchor->GetGUID();
188 }
189 else
190 TC_LOG_ERROR("scripts", "npc_unworthy_initiateAI: unable to find anchor!");
191
192 float dist = 99.0f;
193 GameObject* prison = NULL;
194
195 for (uint8 i = 0; i < 12; ++i)
196 {
197 if (GameObject* temp_prison = me->FindNearestGameObject(acherus_soul_prison[i], 30))
198 {
199 if (me->IsWithinDist(temp_prison, dist, false))
200 {
201 dist = me->GetDistance2d(temp_prison);
202 prison = temp_prison;
203 }
204 }
205 }
206
207 if (prison)
208 prison->ResetDoorOrButton();
209 else
210 TC_LOG_ERROR("scripts", "npc_unworthy_initiateAI: unable to find prison!");
211 }
212 break;
213 case PHASE_TO_EQUIP:
214 if (wait_timer)
215 {
216 if (wait_timer > diff)
217 wait_timer -= diff;
218 else
219 {
220 me->GetMotionMaster()->MovePoint(1, anchorX, anchorY, me->GetPositionZ());
221 //TC_LOG_DEBUG("scripts", "npc_unworthy_initiateAI: move to %f %f %f", anchorX, anchorY, me->GetPositionZ());
222 phase = PHASE_EQUIPING;
223 wait_timer = 0;
224 }
225 }
226 break;
227 case PHASE_TO_ATTACK:
228 if (wait_timer)
229 {
230 if (wait_timer > diff)
231 wait_timer -= diff;
232 else
233 {
234 me->setFaction(14);
235 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
236 phase = PHASE_ATTACKING;
237
238 if (Player* target = ObjectAccessor::GetPlayer(*me, playerGUID))
239 AttackStart(target);
240 wait_timer = 0;
241 }
242 }
243 break;
244 case PHASE_ATTACKING:
245 if (!UpdateVictim())
246 return;
247
248 events.Update(diff);
249
250 while (uint32 eventId = events.ExecuteEvent())
251 {
252 switch (eventId)
253 {
254 case EVENT_ICY_TOUCH:
255 DoCastVictim(SPELL_ICY_TOUCH);
256 events.DelayEvents(1000, GCD_CAST);
257 events.ScheduleEvent(EVENT_ICY_TOUCH, 5000, GCD_CAST);
258 break;
259 case EVENT_PLAGUE_STRIKE:
260 DoCastVictim(SPELL_PLAGUE_STRIKE);
261 events.DelayEvents(1000, GCD_CAST);
262 events.ScheduleEvent(EVENT_PLAGUE_STRIKE, 5000, GCD_CAST);
263 break;
264 case EVENT_BLOOD_STRIKE:
265 DoCastVictim(SPELL_BLOOD_STRIKE);
266 events.DelayEvents(1000, GCD_CAST);
267 events.ScheduleEvent(EVENT_BLOOD_STRIKE, 5000, GCD_CAST);
268 break;
269 case EVENT_DEATH_COIL:
270 DoCastVictim(SPELL_DEATH_COIL);
271 events.DelayEvents(1000, GCD_CAST);
272 events.ScheduleEvent(EVENT_DEATH_COIL, 5000, GCD_CAST);
273 break;
274 }
275 }
276
277 DoMeleeAttackIfReady();
278 break;
279 default:
280 break;
281 }
282 }
283 };

285 CreatureAI* GetAI(Creature* creature) const override

530 void Initialize()
531 {
532 m_uiDuelerGUID.Clear();

537
538 bool lose;

541 bool m_bIsDuelInProgress;

545 Initialize();
546

551
552 void SpellHit(Unit* pCaster, const SpellInfo* pSpell) override
553 {
554 if (!m_bIsDuelInProgress && pSpell->Id == SPELL_DUEL)

557 Talk(SAY_DUEL, pCaster);
558 m_bIsDuelInProgress = true;
559 }
560 }
561
562 void DamageTaken(Unit* pDoneBy, uint32 &uiDamage) override
563 {
564 if (m_bIsDuelInProgress && pDoneBy->IsControlledByPlayer())
565 {
566 if (pDoneBy->GetGUID() != m_uiDuelerGUID && pDoneBy->GetOwnerGUID() != m_uiDuelerGUID) // other players cannot help
567 uiDamage = 0;
568 else if (uiDamage >= me->GetHealth())
569 {
570 uiDamage = 0;
571
572 if (!lose)
573 {
574 pDoneBy->RemoveGameObject(SPELL_DUEL_FLAG, true);
575 pDoneBy->AttackStop();
576 me->CastSpell(pDoneBy, SPELL_DUEL_VICTORY, true);
577 lose = true;
578 me->CastSpell(me, SPELL_GROVEL, true);
579 me->RestoreFaction();
580 }
581 }
582 }
583 }
584
585 void UpdateAI(uint32 uiDiff) override
586 {
587 if (!UpdateVictim())
588 {
589 if (m_bIsDuelInProgress)
590 {
591 if (m_uiDuelTimer <= uiDiff)
592 {
593 me->setFaction(FACTION_HOSTILE);
594
595 if (Unit* unit = ObjectAccessor::GetUnit(*me, m_uiDuelerGUID))
596 AttackStart(unit);
597 }
598 else
599 m_uiDuelTimer -= uiDiff;
600 }
601 return;
602 }
603
604 if (m_bIsDuelInProgress)
605 {
606 if (lose)
607 {
608 if (!me->HasAura(SPELL_GROVEL))
609 EnterEvadeMode();
610 return;
611 }
612 else if (me->GetVictim() && me->EnsureVictim()->GetTypeId() == TYPEID_PLAYER && me->EnsureVictim()->HealthBelowPct(10))
613 {
614 me->EnsureVictim()->CastSpell(me->GetVictim(), SPELL_GROVEL, true); // beg
615 me->EnsureVictim()->RemoveGameObject(SPELL_DUEL_FLAG, true);
616 EnterEvadeMode();
617 return;
618 }
619 }
620
621 /// @todo spells
622
623 CombatAI::UpdateAI(uiDiff);
624 }
625 };

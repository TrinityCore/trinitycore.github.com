27enum Spells_VBM

29 SPELL_REVIVE = 51918
30};
31
32enum Says_VBM

34 WHISPER_REVIVE = 0

37class npc_valkyr_battle_maiden : public CreatureScript
38{
39public:
40 npc_valkyr_battle_maiden() : CreatureScript("npc_valkyr_battle_maiden") { }
41
42 CreatureAI* GetAI(Creature* creature) const override
43 {
44 return new npc_valkyr_battle_maidenAI(creature);
45 }
46
47 struct npc_valkyr_battle_maidenAI : public PassiveAI
48 {
49 npc_valkyr_battle_maidenAI(Creature* creature) : PassiveAI(creature)
50 {
51 Initialize();
52 }
53
54 void Initialize()
55 {
56 FlyBackTimer = 500;
57 phase = 0;
58 x = 0.f;
59 y = 0.f;
60 z = 0.f;
61 }
62
63 uint32 FlyBackTimer;
64 float x, y, z;
65 uint32 phase;
66
67 void Reset() override
68 {
69 me->setActive(true);
70 me->SetVisible(false);
71 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
72 me->SetCanFly(true);
73
74 me->GetPosition(x, y, z);
75 z += 4.0f;
76 x -= 3.5f;
77 y -= 5.0f;
78 me->GetMotionMaster()->Clear(false);
79 me->SetPosition(x, y, z, 0.0f);
80 }
81
82 void UpdateAI(uint32 diff) override
83 {
84 if (FlyBackTimer <= diff)
85 {
86 Player* player = NULL;
87 if (me->IsSummon())
88 if (Unit* summoner = me->ToTempSummon()->GetSummoner())
89 player = summoner->ToPlayer();
90
91 if (!player)
92 phase = 3;
93
94 switch (phase)
95 {
96 case 0:
97 me->SetWalk(false);
98 me->HandleEmoteCommand(EMOTE_STATE_FLYGRABCLOSED);
99 FlyBackTimer = 500;
100 break;
101 case 1:
102 player->GetClosePoint(x, y, z, me->GetObjectSize());
103 z += 2.5f;
104 x -= 2.0f;
105 y -= 1.5f;
106 me->GetMotionMaster()->MovePoint(0, x, y, z);
107 me->SetTarget(player->GetGUID());
108 me->SetVisible(true);
109 FlyBackTimer = 4500;
110 break;
111 case 2:
112 if (!player->IsResurrectRequested())
113 {
114 me->HandleEmoteCommand(EMOTE_ONESHOT_CUSTOM_SPELL_01);
115 DoCast(player, SPELL_REVIVE, true);
116 Talk(WHISPER_REVIVE, player);
117 }
118 FlyBackTimer = 5000;
119 break;
120 case 3:
121 me->SetVisible(false);
122 FlyBackTimer = 3000;
123 break;
124 case 4:
125 me->DisappearAndDie();
126 break;
127 default:
128 //Nothing To DO
129 break;
130 }
131 ++phase;
132 } else FlyBackTimer-=diff;
133 }
134 };
135
136};
137
138void AddSC_the_scarlet_enclave()
139{
140 new npc_valkyr_battle_maiden();
141}

42enum RangerLilatha
43{
44 SAY_START = 0,
45 SAY_PROGRESS1 = 1,
46 SAY_PROGRESS2 = 2,
47 SAY_PROGRESS3 = 3,
48 SAY_END1 = 4,
49 SAY_END2 = 5,
50 SAY_CAPTAIN_ANSWER = 0,
51 QUEST_ESCAPE_FROM_THE_CATACOMBS = 9212,
52 GO_CAGE = 181152,
53 NPC_CAPTAIN_HELIOS = 16220,
54 NPC_MUMMIFIED_HEADHUNTER = 16342,
55 NPC_SHADOWPINE_ORACLE = 16343,
56 FACTION_QUEST_ESCAPE = 113
57};
58
59class npc_ranger_lilatha : public CreatureScript
60{
61public:
62 npc_ranger_lilatha() : CreatureScript("npc_ranger_lilatha") { }
63
64 struct npc_ranger_lilathaAI : public npc_escortAI
65 {
66 npc_ranger_lilathaAI(Creature* creature) : npc_escortAI(creature) { }
67
68 void WaypointReached(uint32 waypointId) override
69 {
70 Player* player = GetPlayerForEscort();
71 if (!player)
72 return;
73
74 switch (waypointId)
75 {
76 case 0:
77 me->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
78 if (GameObject* Cage = me->FindNearestGameObject(GO_CAGE, 20))
79 Cage->SetGoState(GO_STATE_ACTIVE);
80 Talk(SAY_START, player);
81 break;
82 case 5:
83 Talk(SAY_PROGRESS1, player);
84 break;
85 case 11:
86 Talk(SAY_PROGRESS2, player);
87 me->SetFacingTo(4.762841f);
88 break;
89 case 18:
90 {
91 Talk(SAY_PROGRESS3, player);
92 Creature* Summ1 = me->SummonCreature(NPC_MUMMIFIED_HEADHUNTER, 7627.083984f, -7532.538086f, 152.128616f, 1.082733f, TEMPSUMMON_DEAD_DESPAWN, 0);
93 Creature* Summ2 = me->SummonCreature(NPC_SHADOWPINE_ORACLE, 7620.432129f, -7532.550293f, 152.454865f, 0.827478f, TEMPSUMMON_DEAD_DESPAWN, 0);
94 if (Summ1 && Summ2)
95 {
96 Summ1->Attack(me, true);
97 Summ2->Attack(player, true);
98 }
99 AttackStart(Summ1);
100 }
101 break;
102 case 19:
103 me->SetWalk(false);
104 break;
105 case 25:
106 me->SetWalk(true);
107 break;
108 case 30:
109 player->GroupEventHappens(QUEST_ESCAPE_FROM_THE_CATACOMBS, me);
110 break;
111 case 32:
112 me->SetFacingTo(2.978281f);
113 Talk(SAY_END1, player);
114 break;
115 case 33:
116 me->SetFacingTo(5.858011f);
117 Talk(SAY_END2, player);
118 Creature* CaptainHelios = me->FindNearestCreature(NPC_CAPTAIN_HELIOS, 50);
119 if (CaptainHelios)
120 CaptainHelios->AI()->Talk(SAY_CAPTAIN_ANSWER, player);
121 break;
122 }
123 }
124
125 void Reset() override
126 {
127 if (GameObject* Cage = me->FindNearestGameObject(GO_CAGE, 20))
128 Cage->SetGoState(GO_STATE_READY);
129 }
130 };
131
132 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
133 {
134 if (quest->GetQuestId() == QUEST_ESCAPE_FROM_THE_CATACOMBS)
135 {
136 creature->setFaction(FACTION_QUEST_ESCAPE);
137
138 if (npc_escortAI* pEscortAI = CAST_AI(npc_ranger_lilatha::npc_ranger_lilathaAI, creature->AI()))
139 pEscortAI->Start(true, false, player->GetGUID());
140 }
141 return true;
142 }
143
144 CreatureAI* GetAI(Creature* creature) const override
145 {
146 return new npc_ranger_lilathaAI(creature);
147 }
148
149};

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20
21enum Spells
22{
23 SPELL_CURSE_OF_BLOOD = 24673,
24 SPELL_ILLUSION = 17773
25};
26
27enum Events
28{
29 EVENT_CURSE_OF_BLOOD = 1,
30 EVENT_ILLUSION,
31 EVENT_CLEAVE,
32 EVENT_SET_VISIBILITY
33};
34
35class boss_jandice_barov : public CreatureScript
36{
37public:
38 boss_jandice_barov() : CreatureScript("boss_jandice_barov") { }
39
40 struct boss_jandicebarovAI : public ScriptedAI
41 {
42 boss_jandicebarovAI(Creature* creature) : ScriptedAI(creature), Summons(me) { }
43
44 void Reset() override
45 {
46 events.Reset();
47 Summons.DespawnAll();
48 }
49
50 void JustSummoned(Creature* summoned) override
51 {
52 // Illusions should attack a random target.
53 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
54 summoned->AI()->AttackStart(target);
55
56 summoned->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_MAGIC, true); // Not sure if this is correct.
57 Summons.Summon(summoned);
58 }
59
60 void EnterCombat(Unit* /*who*/) override
61 {
62 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, 15000);
63 events.ScheduleEvent(EVENT_ILLUSION, 30000);
64 }
65
66 void JustDied(Unit* /*killer*/) override
67 {
68 Summons.DespawnAll();
69 }
70
71 void UpdateAI(uint32 diff) override
72 {
73 if (!UpdateVictim())
74 return;
75
76 events.Update(diff);
77
78 if (me->HasUnitState(UNIT_STATE_CASTING))
79 return;
80
81 while (uint32 eventId = events.ExecuteEvent())
82 {
83 switch (eventId)
84 {
85 case EVENT_CURSE_OF_BLOOD:
86 DoCastVictim(SPELL_CURSE_OF_BLOOD);
87 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, 30000);
88 break;
89 case EVENT_ILLUSION:
90 DoCast(SPELL_ILLUSION);
91 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
92 me->SetDisplayId(11686); // Invisible Model
93 DoModifyThreatPercent(me->GetVictim(), -99);
94 events.ScheduleEvent(EVENT_SET_VISIBILITY, 3000);
95 events.ScheduleEvent(EVENT_ILLUSION, 25000);
96 break;
97 case EVENT_SET_VISIBILITY:
98 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
99 me->SetDisplayId(11073); //Jandice Model
100 break;
101 default:
102 break;
103 }
104 }
105
106 DoMeleeAttackIfReady();
107 }
108
109 private:
110 EventMap events;
111 SummonList Summons;
112 };
113
114 CreatureAI* GetAI(Creature* creature) const override
115 {
116 return GetInstanceAI<boss_jandicebarovAI>(creature);
117 }
118};
119
120void AddSC_boss_jandicebarov()
121{
122 new boss_jandice_barov();
123}

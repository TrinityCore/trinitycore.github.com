45class boss_kormok : public CreatureScript
46{
47public:
48 boss_kormok() : CreatureScript("boss_kormok") { }
49
50 struct boss_kormokAI : public ScriptedAI
51 {
52 boss_kormokAI(Creature* creature) : ScriptedAI(creature)
53 {
54 Initialize();
55 }
56
57 void Initialize()
58 {
59 Mages = false;
60 }
61
62 void Reset() override
63 {
64 Initialize();
65 events.Reset();
66 }
67
68 void EnterCombat(Unit* /*who*/) override
69 {
70 events.ScheduleEvent(EVENT_SHADOWBOLT_VOLLEY, 10000);
71 events.ScheduleEvent(EVENT_BONE_SHIELD, 2000);
72 events.ScheduleEvent(EVENT_SUMMON_MINIONS, 15000);
73 }
74
75 void JustSummoned(Creature* summoned) override
76 {
77 summoned->AI()->AttackStart(me->GetVictim());
78 }
79
80 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
81 {
82 if (me->HealthBelowPctDamaged(25, damage) && !Mages)
83 {
84 DoCast(SPELL_SUMMON_BONE_MAGES);
85 Mages = true;
86 }
87 }
88
89 void UpdateAI(uint32 diff) override
90 {
91 if (!UpdateVictim())
92 return;
93
94 events.Update(diff);
95
96 if (me->HasUnitState(UNIT_STATE_CASTING))
97 return;
98
99 while (uint32 eventId = events.ExecuteEvent())
100 {
101 switch (eventId)
102 {
103 case EVENT_SHADOWBOLT_VOLLEY:
104 DoCastVictim(SPELL_SHADOWBOLT_VOLLEY);
105 events.ScheduleEvent(EVENT_SHADOWBOLT_VOLLEY, 15000);
106 break;
107 case EVENT_BONE_SHIELD:
108 DoCastVictim(SPELL_BONE_SHIELD);
109 events.ScheduleEvent(EVENT_BONE_SHIELD, 45000);
110 break;
111 case EVENT_SUMMON_MINIONS:
112 DoCast(SPELL_SUMMON_BONE_MINIONS);
113 events.ScheduleEvent(EVENT_SUMMON_MINIONS, 12000);
114 break;
115 default:
116 break;
117 }
118 }
119
120 DoMeleeAttackIfReady();
121 }
122
123 private:
124 EventMap events;
125 bool Mages;
126 };
127
128 CreatureAI* GetAI(Creature* creature) const override
129 {
130 return new boss_kormokAI(creature);
131 }
132};

142// 27695 - Summon Bone Mages
143class spell_kormok_summon_bone_mages : SpellScriptLoader
144{
145 public:
146 spell_kormok_summon_bone_mages() : SpellScriptLoader("spell_kormok_summon_bone_mages") { }
147
148 class spell_kormok_summon_bone_magesSpellScript : public SpellScript
149 {
150 PrepareSpellScript(spell_kormok_summon_bone_magesSpellScript);
151
152 bool Validate(SpellInfo const* /*spell*/) override
153 {
154 for (uint32 i = 0; i < 4; ++i)
155 if (!sSpellMgr->GetSpellInfo(SummonMageSpells[i]))
156 return false;
157 return true;
158 }
159
160 void HandleScript(SpellEffIndex effIndex)
161 {
162 PreventHitDefaultEffect(effIndex);
163 for (uint32 i = 0; i < 2; ++i)
164 GetCaster()->CastSpell(GetCaster(), SummonMageSpells[urand(0, 3)], true);
165 }
166
167 void Register() override
168 {
169 OnEffectHitTarget += SpellEffectFn(spell_kormok_summon_bone_magesSpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
170 }
171 };
172
173 SpellScript* GetSpellScript() const override
174 {
175 return new spell_kormok_summon_bone_magesSpellScript();
176 }
177};

188

190 {
191 if (!sSpellMgr->GetSpellInfo(SPELL_SUMMON_BONE_MINIONS))
192 return false;
193 return true;
194 }
195
196 void HandleScript(SpellEffIndex effIndex)
197 {
198 PreventHitDefaultEffect(effIndex);
199
200 // Possible spells to handle this not found.
201 for (uint32 i = 0; i < 4; ++i)
202 GetCaster()->SummonCreature(NPC_BONE_MINION, GetCaster()->GetPositionX() + float(irand(-7, 7)), GetCaster()->GetPositionY() + float(irand(-7, 7)), GetCaster()->GetPositionZ(), 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 120000);
203 }
204
205 void Register() override
206 {
207 OnEffectHitTarget += SpellEffectFn(spell_kormok_summon_bone_minionsSpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
208 }
209 };
210
211 SpellScript* GetSpellScript() const override
212 {
213 return new spell_kormok_summon_bone_minionsSpellScript();
214 }

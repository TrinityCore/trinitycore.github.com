18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "Player.h"
21#include "scholomance.h"
22
23Position const GandlingLoc = { 180.7712f, -5.428603f, 75.57024f, 1.291544f };
24
25class instance_scholomance : public InstanceMapScript
26{
27 public:
28 instance_scholomance() : InstanceMapScript("instance_scholomance", 289) { }
29
30 InstanceScript* GetInstanceScript(InstanceMap* map) const override
31 {
32 return new instance_scholomance_InstanceMapScript(map);
33 }
34
35 struct instance_scholomance_InstanceMapScript : public InstanceScript
36 {
37 instance_scholomance_InstanceMapScript(Map* map) : InstanceScript(map)
38 {
39 SetHeaders(DataHeader);
40 SetBossNumber(EncounterCount);
41 }
42
43 void OnGameObjectCreate(GameObject* go) override
44 {
45 switch (go->GetEntry())
46 {
47 case GO_GATE_KIRTONOS:
48 GateKirtonosGUID = go->GetGUID();
49 break;
50 case GO_GATE_GANDLING:
51 GateGandlingGUID = go->GetGUID();
52 break;
53 case GO_GATE_MALICIA:
54 GateMiliciaGUID = go->GetGUID();
55 break;
56 case GO_GATE_THEOLEN:
57 GateTheolenGUID = go->GetGUID();
58 break;
59 case GO_GATE_POLKELT:
60 GatePolkeltGUID = go->GetGUID();
61 break;
62 case GO_GATE_RAVENIAN:
63 GateRavenianGUID = go->GetGUID();
64 break;
65 case GO_GATE_BAROV:
66 GateBarovGUID = go->GetGUID();
67 break;
68 case GO_GATE_ILLUCIA:
69 GateIlluciaGUID = go->GetGUID();
70 break;
71 case GO_BRAZIER_OF_THE_HERALD:
72 BrazierOfTheHeraldGUID = go->GetGUID();
73 break;
74 default:
75 break;
76 }
77 }
78
79 bool SetBossState(uint32 type, EncounterState state) override
80 {
81 if (!InstanceScript::SetBossState(type, state))
82 return false;
83
84 switch (type)
85 {
86 case DATA_LORDALEXEIBAROV:
87 case DATA_DOCTORTHEOLENKRASTINOV:
88 case DATA_THERAVENIAN:
89 case DATA_LOREKEEPERPOLKELT:
90 case DATA_INSTRUCTORMALICIA:
91 case DATA_LADYILLUCIABAROV:
92 CheckToSpawnGandling();
93 break;
94 default:
95 break;
96 }
97
98 return true;
99 }
100
101 ObjectGuid GetGuidData(uint32 type) const override
102 {
103 switch (type)
104 {
105 case GO_GATE_KIRTONOS:
106 return GateKirtonosGUID;
107 case GO_GATE_GANDLING:
108 return GateGandlingGUID;
109 case GO_GATE_MALICIA:
110 return GateMiliciaGUID;
111 case GO_GATE_THEOLEN:
112 return GateTheolenGUID;
113 case GO_GATE_POLKELT:
114 return GatePolkeltGUID;
115 case GO_GATE_RAVENIAN:
116 return GateRavenianGUID;
117 case GO_GATE_BAROV:
118 return GateBarovGUID;
119 case GO_GATE_ILLUCIA:
120 return GateIlluciaGUID;
121 case GO_BRAZIER_OF_THE_HERALD:
122 return BrazierOfTheHeraldGUID;
123 default:
124 break;
125 }
126
127 return ObjectGuid::Empty;
128 }
129
130 bool CheckPreBosses(uint32 bossId) const
131 {
132 switch (bossId)
133 {
134 case DATA_DARKMASTERGANDLING:
135 if (GetBossState(DATA_LORDALEXEIBAROV) != DONE)
136 return false;
137 if (GetBossState(DATA_DOCTORTHEOLENKRASTINOV) != DONE)
138 return false;
139 if (GetBossState(DATA_THERAVENIAN) != DONE)
140 return false;
141 if (GetBossState(DATA_LOREKEEPERPOLKELT) != DONE)
142 return false;
143 if (GetBossState(DATA_INSTRUCTORMALICIA) != DONE)
144 return false;
145 if (GetBossState(DATA_LADYILLUCIABAROV) != DONE)
146 return false;
147 if (GetBossState(DATA_DARKMASTERGANDLING) == DONE)
148 return false;
149 break;
150 default:
151 break;
152 }
153
154 return true;
155 }
156
157 void CheckToSpawnGandling()
158 {
159 if (CheckPreBosses(DATA_DARKMASTERGANDLING))
160 instance->SummonCreature(NPC_DARKMASTER_GANDLING, GandlingLoc);
161 }
162
163 void ReadSaveDataMore(std::istringstream& /*data*/) override
164 {
165 CheckToSpawnGandling();
166 }
167
168 protected:
169 ObjectGuid GateKirtonosGUID;
170 ObjectGuid GateGandlingGUID;
171 ObjectGuid GateMiliciaGUID;
172 ObjectGuid GateTheolenGUID;
173 ObjectGuid GatePolkeltGUID;
174 ObjectGuid GateRavenianGUID;
175 ObjectGuid GateBarovGUID;
176 ObjectGuid GateIlluciaGUID;
177 ObjectGuid BrazierOfTheHeraldGUID;
178 };
179};
180
181void AddSC_instance_scholomance()
182{
183 new instance_scholomance();
184}

55 struct boss_darkmaster_gandlingAI : public BossAI
56 {
57 boss_darkmaster_gandlingAI(Creature* creature) : BossAI(creature, DATA_DARKMASTERGANDLING) { }
58
59 void Reset() override
60 {
61 _Reset();
62 if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GATE_GANDLING)))
63 gate->SetGoState(GO_STATE_ACTIVE);
64 }
65
66 void JustDied(Unit* /*killer*/) override
67 {
68 _JustDied();
69 if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GATE_GANDLING)))
70 gate->SetGoState(GO_STATE_ACTIVE);
71 }
72
73 void EnterCombat(Unit* /*who*/) override
74 {
75 _EnterCombat();
76 events.ScheduleEvent(EVENT_ARCANEMISSILES, 4500);
77 events.ScheduleEvent(EVENT_SHADOWSHIELD, 12000);
78 events.ScheduleEvent(EVENT_CURSE, 2000);
79 events.ScheduleEvent(EVENT_SHADOW_PORTAL, 16000);
80
81 if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GATE_GANDLING)))
82 gate->SetGoState(GO_STATE_READY);
83 }
84
85 void IsSummonedBy(Unit* /*summoner*/) override
86 {
87 Talk(YELL_SUMMONED);
88 me->GetMotionMaster()->MoveRandom(5);
89 }
90
91 void UpdateAI(uint32 diff) override
92 {
93 if (!UpdateVictim())
94 return;
95
96 events.Update(diff);
97
98 if (me->HasUnitState(UNIT_STATE_CASTING))
99 return;
100
101 while (uint32 eventId = events.ExecuteEvent())
102 {
103 switch (eventId)
104 {
105 case EVENT_ARCANEMISSILES:
106 DoCastVictim(SPELL_ARCANEMISSILES, true);
107 events.ScheduleEvent(EVENT_ARCANEMISSILES, 8000);
108 break;
109 case EVENT_SHADOWSHIELD:
110 DoCast(me, SPELL_SHADOWSHIELD);
111 events.ScheduleEvent(EVENT_SHADOWSHIELD, urand(14000, 28000));
112 break;
113 case EVENT_CURSE:
114 DoCastVictim(SPELL_CURSE, true);
115 events.ScheduleEvent(EVENT_CURSE, urand(15000, 27000));
116 break;
117 case EVENT_SHADOW_PORTAL:
118 if (HealthAbovePct(3))
119 {
120 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_SHADOW_PORTAL, true);
121 events.ScheduleEvent(EVENT_SHADOW_PORTAL, urand(17000, 27000));
122 }
123 }
124 }
125 DoMeleeAttackIfReady();
126 }
127 };
128
129 CreatureAI* GetAI(Creature* creature) const override
130 {
131 return GetInstanceAI<boss_darkmaster_gandlingAI>(creature);
132 }

178 void HandleCast(SpellEffIndex /*effIndex*/)
179 {
180 Unit* caster = GetCaster();
181 uint8 attempts = 0;
182 uint32 spellId = 0;
183
184 while (!spellId)
185 {
186 if (attempts++ >= 6) break;
187
188 switch (urand(0, 5))
189 {
190 case ROOM_HALL_OF_SECRETS:
191 if (GameObject* go = ObjectAccessor::GetGameObject(*caster, _instance->GetGuidData(GO_GATE_RAVENIAN)))
192 if (go->GetGoState() == GO_STATE_ACTIVE)
193 spellId = SPELL_SHADOW_PORTAL_HALLOFSECRETS;
194 break;
195 case ROOM_HALL_OF_THE_DAMNED:
196 if (GameObject* go = ObjectAccessor::GetGameObject(*caster, _instance->GetGuidData(GO_GATE_THEOLEN)))
197 if (go->GetGoState() == GO_STATE_ACTIVE)
198 spellId = SPELL_SHADOW_PORTAL_HALLOFTHEDAMNED;
199 break;
200 case ROOM_THE_COVEN:
201 if (GameObject* go = ObjectAccessor::GetGameObject(*caster, _instance->GetGuidData(GO_GATE_MALICIA)))
202 if (go->GetGoState() == GO_STATE_ACTIVE)
203 spellId = SPELL_SHADOW_PORTAL_THECOVEN;
204 break;
205 case ROOM_THE_SHADOW_VAULT:
206 if (GameObject* go = ObjectAccessor::GetGameObject(*caster, _instance->GetGuidData(GO_GATE_ILLUCIA)))
207 if (go->GetGoState() == GO_STATE_ACTIVE)
208 spellId = SPELL_SHADOW_PORTAL_THESHADOWVAULT;
209 break;
210 case ROOM_BAROV_FAMILY_VAULT:
211 if (GameObject* go = ObjectAccessor::GetGameObject(*caster, _instance->GetGuidData(GO_GATE_BAROV)))
212 if (go->GetGoState() == GO_STATE_ACTIVE)
213 spellId = SPELL_SHADOW_PORTAL_BAROVFAMILYVAULT;
214 break;
215 case ROOM_VAULT_OF_THE_RAVENIAN:
216 if (GameObject* go = ObjectAccessor::GetGameObject(*caster, _instance->GetGuidData(GO_GATE_POLKELT)))
217 if (go->GetGoState() == GO_STATE_ACTIVE)
218 spellId = SPELL_SHADOW_PORTAL_VAULTOFTHERAVENIAN;
219 break;
220 }
221
222 if (spellId)
223 GetHitUnit()->CastSpell(GetHitUnit(), spellId);
224 }
225 }
226
227 void Register() override
228 {
229 OnEffectHitTarget += SpellEffectFn(spell_shadow_portal_SpellScript::HandleCast, EFFECT_0, SPELL_EFFECT_DUMMY);
230 }
231
232 InstanceScript* _instance;
233 };

235 SpellScript* GetSpellScript() const override

307 void HandleSendEvent(SpellEffIndex effIndex)
308 {
309 // If only one player in threat list fail spell
310
311 Unit* caster = GetCaster();
312
313 int8 pos_to_summon = 0;
314 int8 phase_to_set = 0;
315 int32 gate_to_close = 0;
316
317 switch (GetSpellInfo()->GetEffect(effIndex)->MiscValue)
318 {
319 case SPELL_EVENT_HALLOFSECRETS:
320 pos_to_summon = 0; // Not yet spawned
321 phase_to_set = 1;
322 gate_to_close = GO_GATE_RAVENIAN;
323 break;
324 case SPELL_EVENT_HALLOFTHEDAMNED:
325 pos_to_summon = 0;
326 phase_to_set = 2;
327 gate_to_close = GO_GATE_THEOLEN;
328 break;
329 case SPELL_EVENT_THECOVEN:
330 pos_to_summon = 3;
331 phase_to_set = 3;
332 gate_to_close = GO_GATE_MALICIA;
333 break;
334 case SPELL_EVENT_THESHADOWVAULT:
335 pos_to_summon = 6;
336 phase_to_set = 4;
337 gate_to_close = GO_GATE_ILLUCIA;
338 break;
339 case SPELL_EVENT_BAROVFAMILYVAULT:
340 pos_to_summon = 9;
341 phase_to_set = 5;
342 gate_to_close = GO_GATE_BAROV;
343 break;
344 case SPELL_EVENT_VAULTOFTHERAVENIAN:
345 pos_to_summon = 0; // Not yet spawned
346 phase_to_set = 6;
347 gate_to_close = GO_GATE_POLKELT;
348 break;
349 default:
350 break;
351 }
352
353 if (gate_to_close && (caster->GetMap()->GetId() == 289))
354 {
355 for (uint8 i = 0; i < 3; ++i)
356 {
357 if (Creature* Summoned = caster->SummonCreature(NPC_RISEN_GUARDIAN, SummonPos[pos_to_summon++], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 120000))
358 {
359 Summoned->GetMotionMaster()->MoveRandom(5);
360 Summoned->AI()->SetData(0, phase_to_set);
361 }
362 }
363
364 if (GameObject* gate = ObjectAccessor::GetGameObject(*caster, _instance->GetGuidData(gate_to_close)))
365 gate->SetGoState(GO_STATE_READY);
366 }
367 }
368
369 void Register() override
370 {
371 OnEffectHit += SpellEffectFn(spell_shadow_portal_rooms_SpellScript::HandleSendEvent, EFFECT_1, SPELL_EFFECT_SEND_EVENT);
372 }

374 InstanceScript* _instance;

102class boss_mandokir : public CreatureScript
103{
104 public:
105
106 boss_mandokir() : CreatureScript("boss_mandokir") { }
107
108 struct boss_mandokirAI : public BossAI
109 {
110 boss_mandokirAI(Creature* creature) : BossAI(creature, DATA_MANDOKIR)
111 {
112 Initialize();
113 }
114
115 void Initialize()
116 {
117 _ohganotSoFast = true;
118 _reanimateOhganCooldown = false;
119 }
120
121 void Reset() override
122 {
123 DoCastAOE(SPELL_SPIRIT_VENGEANCE_CANCEL);
124
125 _Reset();
126
127 me->SummonCreatureGroup(SUMMON_GROUP_CHAINED_SPIRIT);
128 Initialize();
129 _reviveGUID.Clear();
130 }
131
132 void EnterCombat(Unit* /*who*/) override
133 {
134 _EnterCombat();
135 Talk(SAY_AGGRO);
136
137 DoCastAOE(SPELL_BLOODLORD_AURA);
138
139 if (!summons.empty())
140 {
141 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
142 {
143 if (Creature* chainedSpirit = ObjectAccessor::GetCreature(*me, *itr))
144 if (chainedSpirit->GetEntry() == NPC_CHAINED_SPIRIT && chainedSpirit->AI())
145 chainedSpirit->setFaction(FACTION_NONE);
146 }
147 }
148
149 events.ScheduleEvent(EVENT_DECAPITATE, 10000);
150 events.ScheduleEvent(EVENT_BLOODLETTING, 15000);
151 events.ScheduleEvent(EVENT_SUMMON_OHGAN, 20000);
152 events.ScheduleEvent(EVENT_DEVASTATING_SLAM, 25000);
153 }
154
155 void JustDied(Unit* /*killer*/) override
156 {
157 DoCastAOE(SPELL_SPIRIT_VENGEANCE_CANCEL);
158 _JustDied();
159 Talk(SAY_DEATH);
160 }
161
162 void KilledUnit(Unit* victim) override
163 {
164 if (victim->GetTypeId() == TYPEID_PLAYER)
165 {
166 Talk(SAY_PLAYER_KILL);
167 DoCast(SPELL_LEVEL_UP);
168 _reviveGUID = victim->GetGUID();
169 DoAction(ACTION_START_REVIVE);
170 }
171 }
172
173 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
174 {
175 if (me->HealthBelowPctDamaged(20, damage) && !me->HasAura(SPELL_FRENZY))
176 {
177 DoCast(me, SPELL_FRENZY, true);
178 Talk(SAY_FRENZY);
179 Talk(EMOTE_FRENZY);
180 }
181 }
182
183 void DoAction(int32 action) override
184 {
185 switch (action)
186 {
187 case ACTION_OHGAN_IS_DEATH:
188 events.ScheduleEvent(EVENT_REANIMATE_OHGAN, 4000);
189 _ohganotSoFast = false;
190 break;
191 case ACTION_START_REVIVE:
192 {
193 std::list<Creature*> creatures;
194 GetCreatureListWithEntryInGrid(creatures, me, NPC_CHAINED_SPIRIT, 200.0f);
195 creatures.remove_if(Trinity::AnyDeadUnitCheck());
196 creatures.remove_if(Trinity::UnitAuraCheck(true, SPELL_OHGAN_ORDERS_TRIGGER));
197 Trinity::Containers::RandomResizeList(creatures, 1);
198 if (creatures.empty())
199 return;
200
201 for (std::list<Creature*>::iterator itr = creatures.begin(); itr != creatures.end(); ++itr)
202 {
203 if (Creature* chainedSpirit = ObjectAccessor::GetCreature(*me, (*itr)->GetGUID()))
204 {
205 chainedSpirit->AI()->SetGUID(_reviveGUID);
206 chainedSpirit->AI()->DoAction(ACTION_REVIVE);
207 _reviveGUID.Clear();
208 }
209 }
210 break;
211 }
212 default:
213 break;
214
215 }
216 }
217
218 uint32 GetData(uint32 type) const override
219 {
220 if (type == DATA_OHGANOT_SO_FAST)
221 return _ohganotSoFast;
222
223 return 0;
224 }
225
226 void SetGUID(ObjectGuid guid, int32 /*type = 0 */) override
227 {
228 _reviveGUID = guid;
229 }
230
231 void UpdateAI(uint32 diff) override
232 {
233 if (!UpdateVictim())
234 return;
235
236 events.Update(diff);
237
238 if (me->HasUnitState(UNIT_STATE_CASTING))
239 return;
240
241 while (uint32 eventId = events.ExecuteEvent())
242 {
243 switch (eventId)
244 {
245 case EVENT_SUMMON_OHGAN:
246 me->SetUInt32Value(UNIT_FIELD_MOUNTDISPLAYID, 0);
247 DoCast(me, SPELL_SUMMON_OHGAN, true);
248 break;
249 case EVENT_DECAPITATE:
250 DoCastAOE(SPELL_DECAPITATE);
251 events.ScheduleEvent(EVENT_DECAPITATE, me->HasAura(SPELL_FRENZY) ? 17500 : 35000);
252 break;
253 case EVENT_BLOODLETTING:
254 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
255 {
256 DoCast(target, SPELL_BLOODLETTING, true);
257 me->ClearUnitState(UNIT_STATE_CASTING);
258 }
259 events.ScheduleEvent(EVENT_BLOODLETTING, 25000);
260 break;
261 case EVENT_REANIMATE_OHGAN:
262 if (_reanimateOhganCooldown)
263 events.ScheduleEvent(EVENT_REANIMATE_OHGAN, 1000);
264 else
265 {
266 DoCastAOE(SPELL_REANIMATE_OHGAN);
267 Talk(SAY_REANIMATE_OHGAN);
268 // Cooldown
269 _reanimateOhganCooldown = true;
270 events.ScheduleEvent(EVENT_REANIMATE_OHGAN_COOLDOWN, 20000);
271 }
272 break;
273 case EVENT_REANIMATE_OHGAN_COOLDOWN:
274 _reanimateOhganCooldown = false;
275 break;
276 case EVENT_DEVASTATING_SLAM:
277 DoCastAOE(SPELL_DEVASTATING_SLAM_TRIGGER);
278 events.ScheduleEvent(EVENT_DEVASTATING_SLAM, 30000);
279 break;
280 default:
281 break;
282 }
283 }
284
285 DoMeleeAttackIfReady();
286 }
287
288 private:
289 bool _ohganotSoFast;
290 bool _reanimateOhganCooldown;
291 ObjectGuid _reviveGUID;
292 };
293
294 CreatureAI* GetAI(Creature* creature) const override
295 {
296 return GetZulGurubAI<boss_mandokirAI>(creature);
297 }

362 public:

364
365 struct npc_chained_spiritAI : public ScriptedAI
366 {
367 npc_chained_spiritAI(Creature* creature) : ScriptedAI(creature)
368 {
369 _instance = me->GetInstanceScript();
370 me->AddUnitMovementFlag(MOVEMENTFLAG_HOVER);
371 me->SetReactState(REACT_PASSIVE); // correct?
372 }
373
374 void Reset() override
375 {
376 _revivePlayerGUID.Clear();
377 }
378
379 void SetGUID(ObjectGuid guid, int32 /*type = 0 */) override
380 {
381 _revivePlayerGUID = guid;
382 }
383
384 void DoAction(int32 action) override
385 {
386 if (action == ACTION_REVIVE)
387 {
388 Position pos;
389 if (Player* target = ObjectAccessor::GetPlayer(*me, _revivePlayerGUID))
390 {
391 target->GetNearPoint(me, pos.m_positionX, pos.m_positionY, pos.m_positionZ, 0.0f, 5.0f, target->GetAngle(me));
392 me->GetMotionMaster()->MovePoint(POINT_START_REVIVE, pos);
393 }
394 }
395 }
396
397 void MovementInform(uint32 type, uint32 pointId) override
398 {
399 if (type != POINT_MOTION_TYPE || !_revivePlayerGUID)
400 return;
401
402 if (pointId == POINT_START_REVIVE)
403 {
404 if (Player* target = ObjectAccessor::GetPlayer(*me, _revivePlayerGUID))
405 DoCast(target, SPELL_REVIVE);
406
407 me->DespawnOrUnsummon(2000);
408 }
409 }
410
411 void JustDied(Unit* /*killer*/) override
412 {
413 Player* target = ObjectAccessor::GetPlayer(*me, _revivePlayerGUID);
414 if (!target || target->IsAlive())
415 return;
416
417 if (Creature* mandokir = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_MANDOKIR)))
418 {
419 mandokir->GetAI()->SetGUID(target->GetGUID());
420 mandokir->GetAI()->DoAction(ACTION_START_REVIVE);
421 }
422
423 me->DespawnOrUnsummon();
424 }
425
426 void UpdateAI(uint32 /*diff*/) override { }
427
428 private:
429 InstanceScript* _instance;
430 ObjectGuid _revivePlayerGUID;
431 };
432
433 CreatureAI* GetAI(Creature* creature) const override
434 {
435 return GetZulGurubAI<npc_chained_spiritAI>(creature);
436 }

549class DevastatingSlamTargetSelector : public std::unary_function<Unit *, bool>
550{
551 public:
552 DevastatingSlamTargetSelector(Creature* me, const Unit* victim) : _me(me), _victim(victim) {}
553
554 bool operator() (WorldObject* target)
555 {
556 if (target == _victim && _me->getThreatManager().getThreatList().size() > 1)
557 return true;
558
559 if (target->GetTypeId() != TYPEID_PLAYER)
560 return true;
561
562 return false;
563 }
564
565 Creature* _me;
566 Unit const* _victim;

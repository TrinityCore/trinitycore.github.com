39enum GreengillSlave
40{
41 NPC_DARKSPINE_MYRIDON = 25060,
42 QUEST_GREENGILL_COAST = 11541,
43 SPELL_ENRAGE = 45111,
44 SPELL_ORB_MURLOC_CONTROL = 45109,
45 SPELL_GREENGILL_SLAVE_FREED = 45110
46};
47
48class npc_greengill_slave : public CreatureScript
49{
50public:
51 npc_greengill_slave() : CreatureScript("npc_greengill_slave") { }
52
53 struct npc_greengill_slaveAI : public ScriptedAI
54 {
55 npc_greengill_slaveAI(Creature* creature) : ScriptedAI(creature) { }
56
57 void SpellHit(Unit* caster, SpellInfo const* spellInfo) override
58 {
59 Player* player = caster->ToPlayer();
60
61 if (!player)
62 return;
63
64 if (spellInfo->Id == SPELL_ORB_MURLOC_CONTROL && !me->HasAura(SPELL_ENRAGE))
65 {
66 if (player->GetQuestStatus(QUEST_GREENGILL_COAST) == QUEST_STATUS_INCOMPLETE)
67 DoCast(player, SPELL_GREENGILL_SLAVE_FREED, true);
68
69 DoCast(me, SPELL_ENRAGE);
70
71 if (Creature* Myrmidon = me->FindNearestCreature(NPC_DARKSPINE_MYRIDON, 70))
72 {
73 me->AddThreat(Myrmidon, 100000.0f);
74 AttackStart(Myrmidon);
75 }
76 }
77 }
78
79 void UpdateAI(uint32 /*diff*/) override
80 {
81 DoMeleeAttackIfReady();
82 }
83 };
84
85 CreatureAI* GetAI(Creature* creature) const override
86 {
87 return new npc_greengill_slaveAI(creature);
88 }
89};
90
91void AddSC_isle_of_queldanas()
92{
93 new npc_greengill_slave();
94}

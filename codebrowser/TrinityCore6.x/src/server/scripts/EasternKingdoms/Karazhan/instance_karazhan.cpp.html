57 struct instance_karazhan_InstanceMapScript : public InstanceScript
58 {
59 instance_karazhan_InstanceMapScript(Map* map) : InstanceScript(map)
60 {
61 SetHeaders(DataHeader);
62 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
63
64 // 1 - OZ, 2 - HOOD, 3 - RAJ, this never gets altered.
65 m_uiOperaEvent = urand(1, 3);
66 m_uiOzDeathCount = 0;
67 }
68
69 uint32 m_auiEncounter[MAX_ENCOUNTER];
70 std::string strSaveData;
71
72 uint32 m_uiOperaEvent;
73 uint32 m_uiOzDeathCount;
74
75 ObjectGuid m_uiCurtainGUID;
76 ObjectGuid m_uiStageDoorLeftGUID;
77 ObjectGuid m_uiStageDoorRightGUID;
78 ObjectGuid m_uiKilrekGUID;
79 ObjectGuid m_uiTerestianGUID;
80 ObjectGuid m_uiMoroesGUID;
81 ObjectGuid m_uiLibraryDoor; // Door at Shade of Aran
82 ObjectGuid m_uiMassiveDoor; // Door at Netherspite
83 ObjectGuid m_uiSideEntranceDoor; // Side Entrance
84 ObjectGuid m_uiGamesmansDoor; // Door before Chess
85 ObjectGuid m_uiGamesmansExitDoor; // Door after Chess
86 ObjectGuid m_uiNetherspaceDoor; // Door at Malchezaar
87 ObjectGuid MastersTerraceDoor[2];
88 ObjectGuid ImageGUID;
89 ObjectGuid DustCoveredChest;
90
91 bool IsEncounterInProgress() const override
92 {
93 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
94 if (m_auiEncounter[i] == IN_PROGRESS)
95 return true;
96
97 return false;
98 }
99
100 void OnCreatureCreate(Creature* creature) override
101 {
102 switch (creature->GetEntry())
103 {
104 case 17229: m_uiKilrekGUID = creature->GetGUID(); break;
105 case 15688: m_uiTerestianGUID = creature->GetGUID(); break;
106 case 15687: m_uiMoroesGUID = creature->GetGUID(); break;
107 }
108 }
109
110 void SetData(uint32 type, uint32 uiData) override
111 {
112 switch (type)
113 {
114 case TYPE_ATTUMEN: m_auiEncounter[0] = uiData; break;
115 case TYPE_MOROES:
116 if (m_auiEncounter[1] == DONE)
117 break;
118 m_auiEncounter[1] = uiData;
119 break;
120 case TYPE_MAIDEN: m_auiEncounter[2] = uiData; break;
121 case TYPE_OPTIONAL_BOSS: m_auiEncounter[3] = uiData; break;
122 case TYPE_OPERA:
123 m_auiEncounter[4] = uiData;
124 if (uiData == DONE)
125 UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, 16812, NULL);
126 break;
127 case TYPE_CURATOR: m_auiEncounter[5] = uiData; break;
128 case TYPE_ARAN: m_auiEncounter[6] = uiData; break;
129 case TYPE_TERESTIAN: m_auiEncounter[7] = uiData; break;
130 case TYPE_NETHERSPITE: m_auiEncounter[8] = uiData; break;
131 case TYPE_CHESS:
132 if (uiData == DONE)
133 DoRespawnGameObject(DustCoveredChest, DAY);
134 m_auiEncounter[9] = uiData;
135 break;
136 case TYPE_MALCHEZZAR: m_auiEncounter[10] = uiData; break;
137 case TYPE_NIGHTBANE:
138 if (m_auiEncounter[11] != DONE)
139 m_auiEncounter[11] = uiData;
140 break;
141 case DATA_OPERA_OZ_DEATHCOUNT:
142 if (uiData == SPECIAL)
143 ++m_uiOzDeathCount;
144 else if (uiData == IN_PROGRESS)
145 m_uiOzDeathCount = 0;
146 break;
147 }
148
149 if (uiData == DONE)
150 {
151 OUT_SAVE_INST_DATA;
152
153 std::ostringstream saveStream;
154 saveStream << m_auiEncounter[0] << ' ' << m_auiEncounter[1] << ' ' << m_auiEncounter[2] << ' '
155 << m_auiEncounter[3] << ' ' << m_auiEncounter[4] << ' ' << m_auiEncounter[5] << ' ' << m_auiEncounter[6] << ' '
156 << m_auiEncounter[7] << ' ' << m_auiEncounter[8] << ' ' << m_auiEncounter[9] << ' ' << m_auiEncounter[10] << ' ' << m_auiEncounter[11];
157
158 strSaveData = saveStream.str();
159
160 SaveToDB();
161 OUT_SAVE_INST_DATA_COMPLETE;
162 }
163 }
164
165 void SetGuidData(uint32 identifier, ObjectGuid data) override
166 {
167 switch (identifier)
168 {
169 case DATA_IMAGE_OF_MEDIVH: ImageGUID = data;
170 }
171 }
172
173 void OnGameObjectCreate(GameObject* go) override
174 {
175 switch (go->GetEntry())
176 {
177 case 183932: m_uiCurtainGUID = go->GetGUID(); break;
178 case 184278:
179 m_uiStageDoorLeftGUID = go->GetGUID();
180 if (m_auiEncounter[4] == DONE)
181 go->SetGoState(GO_STATE_ACTIVE);
182 break;
183 case 184279:
184 m_uiStageDoorRightGUID = go->GetGUID();
185 if (m_auiEncounter[4] == DONE)
186 go->SetGoState(GO_STATE_ACTIVE);
187 break;
188 case 184517: m_uiLibraryDoor = go->GetGUID(); break;
189 case 185521: m_uiMassiveDoor = go->GetGUID(); break;
190 case 184276: m_uiGamesmansDoor = go->GetGUID(); break;
191 case 184277: m_uiGamesmansExitDoor = go->GetGUID(); break;
192 case 185134: m_uiNetherspaceDoor = go->GetGUID(); break;
193 case 184274: MastersTerraceDoor[0] = go->GetGUID(); break;
194 case 184280: MastersTerraceDoor[1] = go->GetGUID(); break;
195 case 184275:
196 m_uiSideEntranceDoor = go->GetGUID();
197 if (m_auiEncounter[4] == DONE)
198 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
199 else
200 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
201 break;
202 case 185119: DustCoveredChest = go->GetGUID(); break;
203 }
204
205 switch (m_uiOperaEvent)
206 {
207 /// @todo Set Object visibilities for Opera based on performance
208 case EVENT_OZ:
209 break;
210
211 case EVENT_HOOD:
212 break;
213
214 case EVENT_RAJ:
215 break;
216 }
217 }

261 case DATA_GO_GAME_EXIT_DOOR: return m_uiGamesmansExitDoor;
262 case DATA_GO_NETHER_DOOR: return m_uiNetherspaceDoor;
263 case DATA_MASTERS_TERRACE_DOOR_1: return MastersTerraceDoor[0];

265 case DATA_IMAGE_OF_MEDIVH: return ImageGUID;
266 }
267

269 }
270
271 void Load(char const* chrIn) override
272 {
273 if (!chrIn)
274 {
275 OUT_LOAD_INST_DATA_FAIL;
276 return;
277 }
278
279 OUT_LOAD_INST_DATA(chrIn);
280 std::istringstream loadStream(chrIn);
281
282 loadStream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3]
283 >> m_auiEncounter[4] >> m_auiEncounter[5] >> m_auiEncounter[6] >> m_auiEncounter[7]
284 >> m_auiEncounter[8] >> m_auiEncounter[9] >> m_auiEncounter[10] >> m_auiEncounter[11];
285 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
286 if (m_auiEncounter[i] == IN_PROGRESS) // Do not load an encounter as "In Progress" - reset it instead.
287 m_auiEncounter[i] = NOT_STARTED;
288 OUT_LOAD_INST_DATA_COMPLETE;
289 }
290 };
291};

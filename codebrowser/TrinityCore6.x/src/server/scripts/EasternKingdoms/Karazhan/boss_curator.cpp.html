51class boss_curator : public CreatureScript

53public:
54 boss_curator() : CreatureScript("boss_curator") { }

61 struct boss_curatorAI : public ScriptedAI
62 {
63 boss_curatorAI(Creature* creature) : ScriptedAI(creature)
64 {
65 Initialize();
66 }

68 void Initialize()
69 {
70 AddTimer = 10000;
71 HatefulBoltTimer = 15000; //This time may be wrong
72 BerserkTimer = 720000; //12 minutes
73 Enraged = false;
74 Evocating = false;
75 }
76
77 uint32 AddTimer;
78 uint32 HatefulBoltTimer;
79 uint32 BerserkTimer;
80
81 bool Enraged;
82 bool Evocating;
83
84 void Reset() override
85 {
86 Initialize();
87
88 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_ARCANE, true);
89 }

93 Talk(SAY_KILL);

98 Talk(SAY_DEATH);

101 void EnterCombat(Unit* /*who*/) override
102 {
103 Talk(SAY_AGGRO);
104 }
105
106 void UpdateAI(uint32 diff) override
107 {
108 if (!UpdateVictim())
109 return;
110
111 //always decrease BerserkTimer
112 if (BerserkTimer <= diff)
113 {
114 //if evocate, then break evocate
115 if (Evocating)
116 {
117 if (me->HasAura(SPELL_EVOCATION))
118 me->RemoveAurasDueToSpell(SPELL_EVOCATION);
119
120 Evocating = false;
121 }
122
123 //may not be correct SAY (generic hard enrage)
124 Talk(SAY_ENRAGE);
125
126 me->InterruptNonMeleeSpells(true);
127 DoCast(me, SPELL_BERSERK);
128
129 //don't know if he's supposed to do summon/evocate after hard enrage (probably not)
130 Enraged = true;
131 } else BerserkTimer -= diff;
132
133 if (Evocating)
134 {
135 //not supposed to do anything while evocate
136 if (me->HasAura(SPELL_EVOCATION))
137 return;
138 else
139 Evocating = false;
140 }
141
142 if (!Enraged)
143 {
144 if (AddTimer <= diff)
145 {
146 //Summon Astral Flare
147 Creature* AstralFlare = DoSpawnCreature(17096, float(rand32() % 37), float(rand32() % 37), 0, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 5000);
148 Unit* target = NULL;
149 target = SelectTarget(SELECT_TARGET_RANDOM, 0);
150
151 if (AstralFlare && target)
152 {
153 AstralFlare->CastSpell(AstralFlare, SPELL_ASTRAL_FLARE_PASSIVE, false);
154 AstralFlare->AI()->AttackStart(target);
155 }
156
157 //Reduce Mana by 10% of max health
158 if (int32 mana = me->GetMaxPower(POWER_MANA))
159 {
160 mana /= 10;
161 me->ModifyPower(POWER_MANA, -mana);
162
163 //if this get's us below 10%, then we evocate (the 10th should be summoned now)
164 if (me->GetPower(POWER_MANA)*100 / me->GetMaxPower(POWER_MANA) < 10)
165 {
166 Talk(SAY_EVOCATE);
167 me->InterruptNonMeleeSpells(false);
168 DoCast(me, SPELL_EVOCATION);
169 Evocating = true;
170 //no AddTimer cooldown, this will make first flare appear instantly after evocate end, like expected
171 return;
172 }
173 else
174 {
175 if (urand(0, 1) == 0)
176 {
177 Talk(SAY_SUMMON);
178 }
179 }
180 }
181
182 AddTimer = 10000;
183 } else AddTimer -= diff;
184
185 if (!HealthAbovePct(15))
186 {
187 Enraged = true;
188 DoCast(me, SPELL_ENRAGE);
189 Talk(SAY_ENRAGE);
190 }
191 }
192
193 if (HatefulBoltTimer <= diff)
194 {
195 if (Enraged)
196 HatefulBoltTimer = 7000;
197 else
198 HatefulBoltTimer = 15000;
199
200 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 1))
201 DoCast(target, SPELL_HATEFUL_BOLT);
202 } else HatefulBoltTimer -= diff;
203
204 DoMeleeAttackIfReady();
205 }
206 };
207};

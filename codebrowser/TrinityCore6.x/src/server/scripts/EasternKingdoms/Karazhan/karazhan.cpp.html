182 void WaypointReached(uint32 waypointId) override
183 {
184 switch (waypointId)
185 {
186 case 0:
187 DoCast(me, SPELL_TUXEDO, false);
188 instance->DoUseDoorOrButton(instance->GetGuidData(DATA_GO_STAGEDOORLEFT));
189 break;
190 case 4:
191 TalkCount = 0;
192 SetEscortPaused(true);
193
194 if (Creature* spotlight = me->SummonCreature(NPC_SPOTLIGHT,
195 me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0.0f,
196 TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 60000))
197 {
198 spotlight->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
199 spotlight->CastSpell(spotlight, SPELL_SPOTLIGHT, false);
200 m_uiSpotlightGUID = spotlight->GetGUID();
201 }
202 break;
203 case 8:
204 instance->DoUseDoorOrButton(instance->GetGuidData(DATA_GO_STAGEDOORLEFT));
205 PerformanceReady = true;
206 break;
207 case 9:
208 PrepareEncounter();
209 instance->DoUseDoorOrButton(instance->GetGuidData(DATA_GO_CURTAINS));
210 break;

214 void Talk(uint32 count)
215 {
216 int32 text = 0;
217
218 switch (m_uiEventId)
219 {
220 case EVENT_OZ:
221 if (OzDialogue[count].textid)
222 text = OzDialogue[count].textid;
223 if (OzDialogue[count].timer)
224 TalkTimer = OzDialogue[count].timer;
225 break;
226
227 case EVENT_HOOD:
228 if (HoodDialogue[count].textid)
229 text = HoodDialogue[count].textid;
230 if (HoodDialogue[count].timer)
231 TalkTimer = HoodDialogue[count].timer;
232 break;
233
234 case EVENT_RAJ:
235 if (RAJDialogue[count].textid)
236 text = RAJDialogue[count].textid;
237 if (RAJDialogue[count].timer)
238 TalkTimer = RAJDialogue[count].timer;
239 break;
240 }
241
242 if (text)
243 CreatureAI::Talk(text);
244 }

246 void PrepareEncounter()
247 {
248 TC_LOG_DEBUG("scripts", "Barnes Opera Event - Introduction complete - preparing encounter %d", m_uiEventId);
249 uint8 index = 0;
250 uint8 count = 0;
251
252 switch (m_uiEventId)
253 {
254 case EVENT_OZ:
255 index = 0;
256 count = 4;
257 break;
258 case EVENT_HOOD:
259 index = 4;
260 count = index+1;
261 break;
262 case EVENT_RAJ:
263 index = 5;
264 count = index+1;
265 break;
266 }
267
268 for (; index < count; ++index)
269 {
270 uint32 entry = ((uint32)Spawns[index][0]);
271 float PosX = Spawns[index][1];
272
273 if (Creature* creature = me->SummonCreature(entry, PosX, SPAWN_Y, SPAWN_Z, SPAWN_O, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, HOUR*2*IN_MILLISECONDS))
274 {
275 // In case database has bad flags
276 creature->SetUInt32Value(UNIT_FIELD_FLAGS, 0);
277 creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
278 }
279 }
280
281 RaidWiped = false;
282 }

284 void UpdateAI(uint32 diff) override
285 {
286 npc_escortAI::UpdateAI(diff);
287
288 if (HasEscortState(STATE_ESCORT_PAUSED))
289 {
290 if (TalkTimer <= diff)
291 {
292 if (TalkCount > 3)
293 {
294 if (Creature* pSpotlight = ObjectAccessor::GetCreature(*me, m_uiSpotlightGUID))
295 pSpotlight->DespawnOrUnsummon();
296
297 SetEscortPaused(false);
298 return;
299 }
300
301 Talk(TalkCount);
302 ++TalkCount;
303 } else TalkTimer -= diff;
304 }
305
306 if (PerformanceReady)
307 {
308 if (!RaidWiped)
309 {
310 if (WipeTimer <= diff)
311 {
312 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
313 if (PlayerList.isEmpty())
314 return;
315
316 RaidWiped = true;
317 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
318 {
319 if (i->GetSource()->IsAlive() && !i->GetSource()->IsGameMaster())
320 {
321 RaidWiped = false;
322 break;
323 }
324 }
325
326 if (RaidWiped)
327 {
328 RaidWiped = true;
329 EnterEvadeMode();
330 return;
331 }
332
333 WipeTimer = 15000;
334 } else WipeTimer -= diff;
335 }
336 }
337 }

340 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
341 {
342 player->PlayerTalkClass->ClearMenus();
343 npc_barnesAI* pBarnesAI = ENSURE_AI(npc_barnes::npc_barnesAI, creature->AI());
344
345 switch (action)
346 {
347 case GOSSIP_ACTION_INFO_DEF+1:
348 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, OZ_GOSSIP2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+2);
349 player->SEND_GOSSIP_MENU(8971, creature->GetGUID());
350 break;
351 case GOSSIP_ACTION_INFO_DEF+2:
352 player->CLOSE_GOSSIP_MENU();
353 pBarnesAI->StartEvent();
354 break;
355 case GOSSIP_ACTION_INFO_DEF+3:
356 player->CLOSE_GOSSIP_MENU();
357 pBarnesAI->m_uiEventId = EVENT_OZ;
358 TC_LOG_DEBUG("scripts", "player (%s) manually set Opera event to EVENT_OZ", player->GetGUID().ToString().c_str());
359 break;
360 case GOSSIP_ACTION_INFO_DEF+4:
361 player->CLOSE_GOSSIP_MENU();
362 pBarnesAI->m_uiEventId = EVENT_HOOD;
363 TC_LOG_DEBUG("scripts", "player (%s) manually set Opera event to EVENT_HOOD", player->GetGUID().ToString().c_str());
364 break;
365 case GOSSIP_ACTION_INFO_DEF+5:
366 player->CLOSE_GOSSIP_MENU();
367 pBarnesAI->m_uiEventId = EVENT_RAJ;
368 TC_LOG_DEBUG("scripts", "player (%s) manually set Opera event to EVENT_RAJ", player->GetGUID().ToString().c_str());
369 break;
370 }
371
372 return true;
373 }

375 bool OnGossipHello(Player* player, Creature* creature) override
376 {
377 if (InstanceScript* instance = creature->GetInstanceScript())
378 {
379 // Check for death of Moroes and if opera event is not done already
380 if (instance->GetData(TYPE_MOROES) == DONE && instance->GetData(TYPE_OPERA) != DONE)
381 {
382 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, OZ_GOSSIP1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
383
384 if (player->IsGameMaster())
385 {
386 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, OZ_GM_GOSSIP1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
387 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, OZ_GM_GOSSIP2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 4);
388 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, OZ_GM_GOSSIP3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 5);
389 }
390
391 if (npc_barnesAI* pBarnesAI = CAST_AI(npc_barnes::npc_barnesAI, creature->AI()))
392 {
393 if (!pBarnesAI->RaidWiped)
394 player->SEND_GOSSIP_MENU(8970, creature->GetGUID());
395 else
396 player->SEND_GOSSIP_MENU(8975, creature->GetGUID());
397
398 return true;
399 }
400 }
401 }
402
403 player->SEND_GOSSIP_MENU(8978, creature->GetGUID());
404 return true;
405 }

500 void StartEvent()
501 {
502 Step = 1;
503 EventStarted = true;
504 Creature* Arcanagos = me->SummonCreature(NPC_ARCANAGOS, ArcanagosPos[0], ArcanagosPos[1], ArcanagosPos[2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000);

510 Arcanagos->SetOrientation(ArcanagosPos[3]);
511 me->SetOrientation(MedivPos[3]);
512 YellTimer = 10000;

596 void UpdateAI(uint32 diff) override
597 {
598 if (YellTimer <= diff)
599 {
600 if (EventStarted)
601 YellTimer = NextStep(Step++);
602 } else YellTimer -= diff;
603
604 if (Step >= 7 && Step <= 12)
605 {
606 Unit* arca = ObjectAccessor::GetUnit(*me, ArcanagosGUID);
607
608 if (FireArcanagosTimer <= diff)
609 {
610 if (arca)
611 arca->CastSpell(me, SPELL_FIRE_BALL, false);
612 FireArcanagosTimer = 6000;
613 } else FireArcanagosTimer -= diff;
614
615 if (FireMedivhTimer <= diff)
616 {
617 if (arca)
618 DoCast(arca, SPELL_FIRE_BALL);
619 FireMedivhTimer = 5000;
620 } else FireMedivhTimer -= diff;
621 }
622 }

22enum Spells
23{
24 SPELL_REPENTANCE = 29511,
25 SPELL_HOLYFIRE = 29522,
26 SPELL_HOLYWRATH = 32445,
27 SPELL_HOLYGROUND = 29512,
28 SPELL_BERSERK = 26662
29};
30
31enum Yells
32{
33 SAY_AGGRO = 0,
34 SAY_SLAY = 1,
35 SAY_REPENTANCE = 2,
36 SAY_DEATH = 3
37};
38
39enum Events
40{
41 EVENT_REPENTANCE = 1,
42 EVENT_HOLYFIRE = 2,
43 EVENT_HOLYWRATH = 3,
44 EVENT_HOLYGROUND = 4,
45 EVENT_ENRAGE = 5
46};
47
48class boss_maiden_of_virtue : public CreatureScript
49{
50public:
51 boss_maiden_of_virtue() : CreatureScript("boss_maiden_of_virtue") { }
52
53 CreatureAI* GetAI(Creature* creature) const override
54 {
55 return new boss_maiden_of_virtueAI(creature);
56 }
57
58 struct boss_maiden_of_virtueAI : public BossAI
59 {
60 boss_maiden_of_virtueAI(Creature* creature) : BossAI(creature, TYPE_MAIDEN) { }
61
62 void Reset() override
63 {
64 _Reset();
65 }
66
67 void KilledUnit(Unit* /*Victim*/) override
68 {
69 if (urand(0, 1) == 0)
70 Talk(SAY_SLAY);
71 }
72
73 void JustDied(Unit* /*killer*/) override
74 {
75 Talk(SAY_DEATH);
76 _JustDied();
77 }
78
79 void EnterCombat(Unit* /*who*/) override
80 {
81 _EnterCombat();
82 Talk(SAY_AGGRO);
83
84 events.ScheduleEvent(EVENT_REPENTANCE, urand(33, 45) * IN_MILLISECONDS);
85 events.ScheduleEvent(EVENT_HOLYFIRE, 12 * IN_MILLISECONDS);
86 events.ScheduleEvent(EVENT_HOLYWRATH, urand(15, 25) * IN_MILLISECONDS);
87 events.ScheduleEvent(EVENT_HOLYGROUND, 3 * IN_MILLISECONDS);
88 events.ScheduleEvent(EVENT_ENRAGE, 600 * IN_MILLISECONDS);
89 }
90
91 void UpdateAI(uint32 diff) override
92 {
93 if (!UpdateVictim())
94 return;
95
96 events.Update(diff);
97
98 if (me->HasUnitState(UNIT_STATE_CASTING))
99 return;
100
101 while (uint32 eventId = events.ExecuteEvent())
102 {
103 switch (eventId)
104 {
105 case EVENT_REPENTANCE:
106 DoCastVictim(SPELL_REPENTANCE);
107 Talk(SAY_REPENTANCE);
108 events.ScheduleEvent(EVENT_REPENTANCE, urand(33, 45) * IN_MILLISECONDS);
109 break;
110 case EVENT_HOLYFIRE:
111 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50, true))
112 DoCast(target, SPELL_HOLYFIRE);
113 events.ScheduleEvent(EVENT_HOLYFIRE, 12 * IN_MILLISECONDS);
114 break;
115 case EVENT_HOLYWRATH:
116 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 80, true))
117 DoCast(target, SPELL_HOLYWRATH);
118 events.ScheduleEvent(EVENT_HOLYWRATH, urand(15, 25) * IN_MILLISECONDS);
119 break;
120 case EVENT_HOLYGROUND:
121 DoCast(me, SPELL_HOLYGROUND, true);
122 events.ScheduleEvent(EVENT_HOLYGROUND, 3 * IN_MILLISECONDS);
123 break;
124 case EVENT_ENRAGE:
125 DoCast(me, SPELL_BERSERK, true);
126 break;
127 default:
128 break;
129 }
130 }
131
132 DoMeleeAttackIfReady();
133 }
134 };
135};
136
137void AddSC_boss_maiden_of_virtue()
138{
139 new boss_maiden_of_virtue();
140}

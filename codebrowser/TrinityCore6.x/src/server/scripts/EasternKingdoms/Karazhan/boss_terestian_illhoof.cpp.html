88 }

90 InstanceScript* instance;
91
92 uint32 AmplifyTimer;
93
94 void Reset() override
95 {
96 Initialize();
97 }
98

100
101 void JustDied(Unit* /*killer*/) override
102 {
103 Creature* Terestian = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_TERESTIAN));
104 if (Terestian && Terestian->IsAlive())
105 DoCast(Terestian, SPELL_BROKEN_PACT, true);
106 }
107
108 void UpdateAI(uint32 diff) override
109 {
110 //Return since we have no target
111 if (!UpdateVictim())
112 return;
113
114 if (AmplifyTimer <= diff)
115 {
116 me->InterruptNonMeleeSpells(false);
117 DoCastVictim(SPELL_AMPLIFY_FLAMES);
118
119 AmplifyTimer = urand(10000, 20000);
120 } else AmplifyTimer -= diff;
121
122 DoMeleeAttackIfReady();
123 }
124 };

257class boss_terestian_illhoof : public CreatureScript
258{
259public:
260 boss_terestian_illhoof() : CreatureScript("boss_terestian_illhoof") { }
261
262 CreatureAI* GetAI(Creature* creature) const override
263 {
264 return GetInstanceAI<boss_terestianAI>(creature);
265 }
266
267 struct boss_terestianAI : public ScriptedAI
268 {
269 boss_terestianAI(Creature* creature) : ScriptedAI(creature)
270 {
271 instance = creature->GetInstanceScript();
272 Initialize();
273 }
274
275 void Initialize()
276 {
277 PortalsCount = 0;
278 SacrificeTimer = 30000;
279 ShadowboltTimer = 5000;
280 SummonTimer = 10000;
281 BerserkTimer = 600000;
282
283 SummonedPortals = false;
284 Berserk = false;
285 }
286
287 InstanceScript* instance;
288
289 ObjectGuid PortalGUID[2];
290 uint8 PortalsCount;
291
292 uint32 SacrificeTimer;
293 uint32 ShadowboltTimer;
294 uint32 SummonTimer;
295 uint32 BerserkTimer;
296
297 bool SummonedPortals;
298 bool Berserk;
299
300 void Reset() override
301 {
302 for (uint8 i = 0; i < 2; ++i)
303 {
304 if (!PortalGUID[i].IsEmpty())
305 {
306 if (Creature* pPortal = ObjectAccessor::GetCreature(*me, PortalGUID[i]))
307 {
308 ENSURE_AI(npc_fiendish_portal::npc_fiendish_portalAI, pPortal->AI())->DespawnAllImp();
309 pPortal->DespawnOrUnsummon();
310 }
311
312 PortalGUID[i].Clear();
313 }
314 }
315
316 Initialize();
317
318 instance->SetData(TYPE_TERESTIAN, NOT_STARTED);
319
320 me->RemoveAurasDueToSpell(SPELL_BROKEN_PACT);
321
322 if (Minion* Kilrek = me->GetFirstMinion())
323 {
324 if (!Kilrek->IsAlive())
325 {
326 Kilrek->UnSummon();
327 DoCast(me, SPELL_SUMMON_IMP, true);
328 }
329 }
330 else DoCast(me, SPELL_SUMMON_IMP, true);
331 }
332
333 void EnterCombat(Unit* /*who*/) override
334 {
335 Talk(SAY_AGGRO);
336 }
337
338 void JustSummoned(Creature* summoned) override
339 {
340 if (summoned->GetEntry() == NPC_PORTAL)
341 {
342 PortalGUID[PortalsCount] = summoned->GetGUID();
343 ++PortalsCount;
344
345 if (summoned->GetUInt32Value(UNIT_CREATED_BY_SPELL) == SPELL_FIENDISH_PORTAL_1)
346 {
347 Talk(SAY_SUMMON);
348 SummonedPortals = true;
349 }
350 }
351 }
352
353 void KilledUnit(Unit* /*victim*/) override
354 {
355 Talk(SAY_SLAY);
356 }
357
358 void JustDied(Unit* /*killer*/) override
359 {
360 for (uint8 i = 0; i < 2; ++i)
361 {
362 if (!PortalGUID[i].IsEmpty())
363 {
364 if (Creature* pPortal = ObjectAccessor::GetCreature((*me), PortalGUID[i]))
365 pPortal->DespawnOrUnsummon();
366
367 PortalGUID[i].Clear();
368 }
369 }
370
371 Talk(SAY_DEATH);
372
373 instance->SetData(TYPE_TERESTIAN, DONE);
374 }
375
376 void UpdateAI(uint32 diff) override
377 {
378 if (!UpdateVictim())
379 return;
380
381 if (SacrificeTimer <= diff)
382 {
383 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true);
384 if (target && target->IsAlive())
385 {
386 DoCast(target, SPELL_SACRIFICE, true);
387 DoCast(target, SPELL_SUMMON_DEMONCHAINS, true);
388
389 if (Creature* Chains = me->FindNearestCreature(NPC_DEMONCHAINS, 5000))
390 {
391 ENSURE_AI(npc_demon_chain::npc_demon_chainAI, Chains->AI())->SacrificeGUID = target->GetGUID();
392 Chains->CastSpell(Chains, SPELL_DEMON_CHAINS, true);
393 Talk(SAY_SACRIFICE);
394 SacrificeTimer = 30000;
395 }
396 }
397 } else SacrificeTimer -= diff;
398
399 if (ShadowboltTimer <= diff)
400 {
401 DoCast(SelectTarget(SELECT_TARGET_TOPAGGRO, 0), SPELL_SHADOW_BOLT);
402 ShadowboltTimer = 10000;
403 } else ShadowboltTimer -= diff;
404
405 if (SummonTimer <= diff)
406 {
407 if (!PortalGUID[0])
408 DoCastVictim(SPELL_FIENDISH_PORTAL, false);
409
410 if (!PortalGUID[1])
411 DoCastVictim(SPELL_FIENDISH_PORTAL_1, false);
412
413 if (!PortalGUID[0].IsEmpty() && !PortalGUID[1].IsEmpty())
414 {
415 if (Creature* pPortal = ObjectAccessor::GetCreature(*me, PortalGUID[urand(0, 1)]))
416 pPortal->CastSpell(me->GetVictim(), SPELL_SUMMON_FIENDISIMP, false);
417 SummonTimer = 5000;
418 }
419 } else SummonTimer -= diff;
420
421 if (!Berserk)
422 {
423 if (BerserkTimer <= diff)
424 {
425 DoCast(me, SPELL_BERSERK);
426 Berserk = true;
427 } else BerserkTimer -= diff;
428 }
429
430 DoMeleeAttackIfReady();
431 }
432 };
433};

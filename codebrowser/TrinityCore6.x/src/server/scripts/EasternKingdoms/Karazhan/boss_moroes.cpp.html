104 struct boss_moroesAI : public ScriptedAI
105 {
106 boss_moroesAI(Creature* creature) : ScriptedAI(creature)
107 {
108 Initialize();
109 memset(AddId, 0, sizeof(AddId));
110
111 instance = creature->GetInstanceScript();
112 }
113
114 void Initialize()
115 {
116 Vanish_Timer = 30000;
117 Blind_Timer = 35000;
118 Gouge_Timer = 23000;
119 Wait_Timer = 0;
120 CheckAdds_Timer = 5000;
121
122 Enrage = false;
123 InVanish = false;
124 }
125
126 InstanceScript* instance;
127
128 ObjectGuid AddGUID[4];
129
130 uint32 Vanish_Timer;
131 uint32 Blind_Timer;
132 uint32 Gouge_Timer;
133 uint32 Wait_Timer;
134 uint32 CheckAdds_Timer;
135 uint32 AddId[4];
136
137 bool InVanish;
138 bool Enrage;
139
140 void Reset() override
141 {
142 Initialize();
143 if (me->IsAlive())
144 SpawnAdds();
145
146 instance->SetData(TYPE_MOROES, NOT_STARTED);
147 }
148
149 void StartEvent()
150 {
151 instance->SetData(TYPE_MOROES, IN_PROGRESS);
152
153 DoZoneInCombat();
154 }
155
156 void EnterCombat(Unit* /*who*/) override
157 {
158 StartEvent();
159
160 Talk(SAY_AGGRO);
161 AddsAttack();
162 DoZoneInCombat();
163 }
164
165 void KilledUnit(Unit* /*victim*/) override
166 {
167 Talk(SAY_KILL);
168 }
169
170 void JustDied(Unit* /*killer*/) override
171 {
172 Talk(SAY_DEATH);
173
174 instance->SetData(TYPE_MOROES, DONE);
175
176 DeSpawnAdds();
177
178 //remove aura from spell Garrote when Moroes dies
179 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_GARROTE);
180 }
181
182 void SpawnAdds()
183 {
184 DeSpawnAdds();
185
186 if (isAddlistEmpty())
187 {
188 std::list<uint32> AddList;
189
190 for (uint8 i = 0; i < 6; ++i)
191 AddList.push_back(Adds[i]);
192
193 Trinity::Containers::RandomResizeList(AddList, 4);
194
195 uint8 i = 0;
196 for (std::list<uint32>::const_iterator itr = AddList.begin(); itr != AddList.end() && i < 4; ++itr, ++i)
197 {
198 uint32 entry = *itr;
199
200 if (Creature* creature = me->SummonCreature(entry, Locations[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000))
201 {
202 AddGUID[i] = creature->GetGUID();
203 AddId[i] = entry;
204 }
205 }
206 }
207 else
208 {
209 for (uint8 i = 0; i < 4; ++i)
210 {
211 if (Creature* creature = me->SummonCreature(AddId[i], Locations[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000))
212 AddGUID[i] = creature->GetGUID();
213 }
214 }
215 }
216
217 bool isAddlistEmpty()
218 {
219 for (uint8 i = 0; i < 4; ++i)
220 if (AddId[i] == 0)
221 return true;
222
223 return false;
224 }
225
226 void DeSpawnAdds()
227 {
228 for (uint8 i = 0; i < 4; ++i)
229 {
230 if (!AddGUID[i].IsEmpty())
231 {
232 if (Creature* temp = ObjectAccessor::GetCreature(*me, AddGUID[i]))
233 temp->DespawnOrUnsummon();
234 }
235 }
236 }
237
238 void AddsAttack()
239 {
240 for (uint8 i = 0; i < 4; ++i)
241 {
242 if (!AddGUID[i].IsEmpty())
243 {
244 Creature* temp = ObjectAccessor::GetCreature((*me), AddGUID[i]);
245 if (temp && temp->IsAlive())
246 {
247 temp->AI()->AttackStart(me->GetVictim());
248 DoZoneInCombat(temp);
249 } else
250 EnterEvadeMode();
251 }
252 }
253 }
254
255 void UpdateAI(uint32 diff) override
256 {
257 if (!UpdateVictim())
258 return;
259
260 if (!instance->GetData(TYPE_MOROES))
261 {
262 EnterEvadeMode();
263 return;
264 }
265
266 if (!Enrage && HealthBelowPct(30))
267 {
268 DoCast(me, SPELL_FRENZY);
269 Enrage = true;
270 }
271
272 if (CheckAdds_Timer <= diff)
273 {
274 for (uint8 i = 0; i < 4; ++i)
275 {
276 if (!AddGUID[i].IsEmpty())
277 {
278 Creature* temp = ObjectAccessor::GetCreature((*me), AddGUID[i]);
279 if (temp && temp->IsAlive())
280 if (!temp->GetVictim())
281 temp->AI()->AttackStart(me->GetVictim());
282 }
283 }
284 CheckAdds_Timer = 5000;
285 } else CheckAdds_Timer -= diff;
286
287 if (!Enrage)
288 {
289 //Cast Vanish, then Garrote random victim
290 if (Vanish_Timer <= diff)
291 {
292 DoCast(me, SPELL_VANISH);
293 InVanish = true;
294 Vanish_Timer = 30000;
295 Wait_Timer = 5000;
296 } else Vanish_Timer -= diff;
297
298 if (Gouge_Timer <= diff)
299 {
300 DoCastVictim(SPELL_GOUGE);
301 Gouge_Timer = 40000;
302 } else Gouge_Timer -= diff;
303
304 if (Blind_Timer <= diff)
305 {
306 std::list<Unit*> targets;
307 SelectTargetList(targets, 5, SELECT_TARGET_RANDOM, me->GetMeleeReach()*5, true);
308 for (std::list<Unit*>::const_iterator i = targets.begin(); i != targets.end(); ++i)
309 if (!me->IsWithinMeleeRange(*i))
310 {
311 DoCast(*i, SPELL_BLIND);
312 break;
313 }
314 Blind_Timer = 40000;
315 } else Blind_Timer -= diff;
316 }
317
318 if (InVanish)
319 {
320 if (Wait_Timer <= diff)
321 {
322 Talk(SAY_SPECIAL);
323
324 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
325 target->CastSpell(target, SPELL_GARROTE, true);
326
327 InVanish = false;
328 } else Wait_Timer -= diff;
329 }
330
331 if (!InVanish)
332 DoMeleeAttackIfReady();
333 }
334 };

337struct boss_moroes_guestAI : public ScriptedAI
338{
339 InstanceScript* instance;
340
341 ObjectGuid GuestGUID[4];
342
343 boss_moroes_guestAI(Creature* creature) : ScriptedAI(creature)
344 {
345 instance = creature->GetInstanceScript();
346 }
347
348 void Reset() override
349 {
350 instance->SetData(TYPE_MOROES, NOT_STARTED);
351 }
352
353 void AcquireGUID()
354 {
355 if (Creature* Moroes = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MOROES)))
356 {
357 for (uint8 i = 0; i < 4; ++i)
358 {
359 ObjectGuid GUID = ENSURE_AI(boss_moroes::boss_moroesAI, Moroes->AI())->AddGUID[i];
360 if (!GUID.IsEmpty())
361 GuestGUID[i] = GUID;
362 }
363 }
364 }
365
366 Unit* SelectGuestTarget()
367 {
368 ObjectGuid TempGUID = GuestGUID[rand32() % 4];
369 if (!TempGUID.IsEmpty())
370 {
371 Unit* unit = ObjectAccessor::GetUnit(*me, TempGUID);
372 if (unit && unit->IsAlive())
373 return unit;
374 }
375
376 return me;
377 }
378
379 void UpdateAI(uint32 /*diff*/) override
380 {
381 if (!instance->GetData(TYPE_MOROES))
382 EnterEvadeMode();
383
384 DoMeleeAttackIfReady();
385 }

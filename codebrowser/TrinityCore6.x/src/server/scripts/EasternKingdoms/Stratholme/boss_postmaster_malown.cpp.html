33enum Says
34{
35 SAY_KILL = 0
36};

38enum Spells
39{
40 SPELL_WAILINGDEAD = 7713,
41 SPELL_BACKHAND = 6253,
42 SPELL_CURSEOFWEAKNESS = 8552,
43 SPELL_CURSEOFTONGUES = 12889,
44 SPELL_CALLOFTHEGRAVE = 17831
45};

47enum Events
48{
49 EVENT_WAILINGDEAD = 1,
50 EVENT_BACKHAND = 2,
51 EVENT_CURSEOFWEAKNESS = 3,
52 EVENT_CURSEOFTONGUES = 4,
53 EVENT_CALLOFTHEGRAVE = 5
54};
55
56class boss_postmaster_malown : public CreatureScript
57{
58 public:
59 boss_postmaster_malown() : CreatureScript("boss_postmaster_malown") { }
60
61 struct boss_postmaster_malownAI : public BossAI
62 {
63 boss_postmaster_malownAI(Creature* creature) : BossAI(creature, TYPE_MALOWN) { }
64
65 void Reset() override { }
66
67 void EnterCombat(Unit* /*who*/) override
68 {
69 events.ScheduleEvent(EVENT_WAILINGDEAD, 19000); // lasts 6 sec
70 events.ScheduleEvent(EVENT_BACKHAND, 8000); // 2 sec stun
71 events.ScheduleEvent(EVENT_CURSEOFWEAKNESS, 20000); // lasts 2 mins
72 events.ScheduleEvent(EVENT_CURSEOFTONGUES, 22000);
73 events.ScheduleEvent(EVENT_CALLOFTHEGRAVE, 25000);
74 }
75
76 void KilledUnit(Unit* /*victim*/) override
77 {
78 Talk(SAY_KILL);
79 }
80
81 void UpdateAI(uint32 diff) override
82 {
83 if (!UpdateVictim())
84 return;
85
86 events.Update(diff);
87
88 if (me->HasUnitState(UNIT_STATE_CASTING))
89 return;
90
91 while (uint32 eventId = events.ExecuteEvent())
92 {
93 switch (eventId)
94 {
95 case EVENT_WAILINGDEAD:
96 if (rand32() % 100 < 65) //65% chance to cast
97 DoCastVictim(SPELL_WAILINGDEAD, true);
98 events.ScheduleEvent(EVENT_WAILINGDEAD, 19000);
99 break;
100 case EVENT_BACKHAND:
101 if (rand32() % 100 < 45) //45% chance to cast
102 DoCastVictim(SPELL_BACKHAND, true);
103 events.ScheduleEvent(EVENT_WAILINGDEAD, 8000);
104 break;
105 case EVENT_CURSEOFWEAKNESS:
106 if (rand32() % 100 < 3) //3% chance to cast
107 DoCastVictim(SPELL_CURSEOFWEAKNESS, true);
108 events.ScheduleEvent(EVENT_WAILINGDEAD, 20000);
109 break;
110 case EVENT_CURSEOFTONGUES:
111 if (rand32() % 100 < 3) //3% chance to cast
112 DoCastVictim(SPELL_CURSEOFTONGUES, true);
113 events.ScheduleEvent(EVENT_WAILINGDEAD, 22000);
114 break;
115 case EVENT_CALLOFTHEGRAVE:
116 if (rand32() % 100 < 5) //5% chance to cast
117 DoCastVictim(SPELL_CALLOFTHEGRAVE, true);
118 events.ScheduleEvent(EVENT_WAILINGDEAD, 25000);
119 break;
120 default:
121 break;
122 }
123 }
124
125 DoMeleeAttackIfReady();
126 }
127 };
128
129 CreatureAI* GetAI(Creature* creature) const override
130 {
131 return GetInstanceAI<boss_postmaster_malownAI>(creature);
132 }
133};

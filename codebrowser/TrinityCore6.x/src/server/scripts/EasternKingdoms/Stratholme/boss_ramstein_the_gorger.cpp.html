30enum Spells

32 SPELL_TRAMPLE = 5568,
33 SPELL_KNOCKOUT = 17307

36enum CreatureId

38 NPC_MINDLESS_UNDEAD = 11030

40
41class boss_ramstein_the_gorger : public CreatureScript
42{
43public:
44 boss_ramstein_the_gorger() : CreatureScript("boss_ramstein_the_gorger") { }
45
46 CreatureAI* GetAI(Creature* creature) const override
47 {
48 return GetInstanceAI<boss_ramstein_the_gorgerAI>(creature);
49 }
50
51 struct boss_ramstein_the_gorgerAI : public ScriptedAI
52 {
53 boss_ramstein_the_gorgerAI(Creature* creature) : ScriptedAI(creature)
54 {
55 Initialize();
56 instance = me->GetInstanceScript();
57 }
58
59 void Initialize()
60 {
61 Trample_Timer = 3000;
62 Knockout_Timer = 12000;
63 }
64
65 InstanceScript* instance;
66
67 uint32 Trample_Timer;
68 uint32 Knockout_Timer;
69
70 void Reset() override
71 {
72 Initialize();
73 }
74
75 void EnterCombat(Unit* /*who*/) override
76 {
77 }
78
79 void JustDied(Unit* /*killer*/) override
80 {
81 for (uint8 i = 0; i < 30; ++i)
82 {
83 if (Creature* mob = me->SummonCreature(NPC_MINDLESS_UNDEAD, 3969.35f+irand(-10, 10), -3391.87f+irand(-10, 10), 119.11f, 5.91f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 1800000))
84 mob->AI()->AttackStart(me->SelectNearestTarget(100.0f));
85 }
86
87 instance->SetData(TYPE_RAMSTEIN, DONE);
88 }
89
90 void UpdateAI(uint32 diff) override
91 {
92 //Return since we have no target
93 if (!UpdateVictim())
94 return;
95
96 //Trample
97 if (Trample_Timer <= diff)
98 {
99 DoCast(me, SPELL_TRAMPLE);
100 Trample_Timer = 7000;
101 } else Trample_Timer -= diff;
102
103 //Knockout
104 if (Knockout_Timer <= diff)
105 {
106 DoCastVictim(SPELL_KNOCKOUT);
107 Knockout_Timer = 10000;
108 } else Knockout_Timer -= diff;
109
110 DoMeleeAttackIfReady();
111 }
112 };
113
114};
115
116void AddSC_boss_ramstein_the_gorger()

118 new boss_ramstein_the_gorger();
119}

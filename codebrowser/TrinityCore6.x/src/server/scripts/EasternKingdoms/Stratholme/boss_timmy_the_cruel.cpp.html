39class boss_timmy_the_cruel : public CreatureScript
40{
41public:
42 boss_timmy_the_cruel() : CreatureScript("boss_timmy_the_cruel") { }
43
44 CreatureAI* GetAI(Creature* creature) const override
45 {
46 return new boss_timmy_the_cruelAI(creature);
47 }
48
49 struct boss_timmy_the_cruelAI : public ScriptedAI
50 {
51 boss_timmy_the_cruelAI(Creature* creature) : ScriptedAI(creature)
52 {
53 Initialize();
54 }
55
56 void Initialize()
57 {
58 RavenousClaw_Timer = 10000;
59 HasYelled = false;
60 }
61
62 uint32 RavenousClaw_Timer;
63 bool HasYelled;
64
65 void Reset() override
66 {
67 Initialize();
68 }
69
70 void EnterCombat(Unit* /*who*/) override
71 {
72 if (!HasYelled)
73 {
74 Talk(SAY_SPAWN);
75 HasYelled = true;
76 }
77 }
78
79 void UpdateAI(uint32 diff) override
80 {
81 //Return since we have no target
82 if (!UpdateVictim())
83 return;
84
85 //RavenousClaw
86 if (RavenousClaw_Timer <= diff)
87 {
88 //Cast
89 DoCastVictim(SPELL_RAVENOUSCLAW);
90 //15 seconds until we should cast this again
91 RavenousClaw_Timer = 15000;
92 } else RavenousClaw_Timer -= diff;
93
94 DoMeleeAttackIfReady();
95 }
96 };
97
98};
99
100void AddSC_boss_timmy_the_cruel()
101{
102 new boss_timmy_the_cruel();
103}

44class go_gauntlet_gate : public GameObjectScript
45{
46public:
47 go_gauntlet_gate() : GameObjectScript("go_gauntlet_gate") { }
48
49 bool OnGossipHello(Player* player, GameObject* go) override
50 {
51 InstanceScript* instance = go->GetInstanceScript();
52
53 if (!instance)
54 return false;
55
56 if (instance->GetData(TYPE_BARON_RUN) != NOT_STARTED)
57 return false;
58
59 if (Group* group = player->GetGroup())
60 {
61 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
62 {
63 Player* pGroupie = itr->GetSource();
64 if (!pGroupie)
65 continue;
66
67 if (pGroupie->GetQuestStatus(QUEST_DEAD_MAN_PLEA) == QUEST_STATUS_INCOMPLETE &&
68 !pGroupie->HasAura(SPELL_BARON_ULTIMATUM) &&
69 pGroupie->GetMap() == go->GetMap())
70 pGroupie->CastSpell(pGroupie, SPELL_BARON_ULTIMATUM, true);
71 }
72 } else if (player->GetQuestStatus(QUEST_DEAD_MAN_PLEA) == QUEST_STATUS_INCOMPLETE &&
73 !player->HasAura(SPELL_BARON_ULTIMATUM) &&
74 player->GetMap() == go->GetMap())
75 player->CastSpell(player, SPELL_BARON_ULTIMATUM, true);
76
77 instance->SetData(TYPE_BARON_RUN, IN_PROGRESS);
78 return false;
79 }
80
81};

101class npc_restless_soul : public CreatureScript
102{
103public:
104 npc_restless_soul() : CreatureScript("npc_restless_soul") { }
105
106 CreatureAI* GetAI(Creature* creature) const override
107 {
108 return new npc_restless_soulAI(creature);
109 }
110
111 struct npc_restless_soulAI : public ScriptedAI
112 {
113 npc_restless_soulAI(Creature* creature) : ScriptedAI(creature)
114 {
115 Initialize();
116 }
117
118 void Initialize()
119 {
120 Tagger.Clear();
121 Die_Timer = 5000;
122 Tagged = false;
123 }
124
125 ObjectGuid Tagger;
126 uint32 Die_Timer;
127 bool Tagged;
128
129 void Reset() override
130 {
131 Initialize();
132 }
133
134 void EnterCombat(Unit* /*who*/) override { }
135
136 void SpellHit(Unit* caster, const SpellInfo* spell) override
137 {
138 if (Tagged || spell->Id != SPELL_EGAN_BLASTER)
139 return;
140
141 Player* player = caster->ToPlayer();
142 if (!player || player->GetQuestStatus(QUEST_RESTLESS_SOUL) != QUEST_STATUS_INCOMPLETE)
143 return;
144
145 Tagged = true;
146 Tagger = caster->GetGUID();
147 }
148
149 void JustSummoned(Creature* summoned) override
150 {
151 summoned->CastSpell(summoned, SPELL_SOUL_FREED, false);
152
153 if (Player* player = ObjectAccessor::GetPlayer(*me, Tagger))
154 summoned->GetMotionMaster()->MoveFollow(player, 0.0f, 0.0f);
155 }
156
157 void JustDied(Unit* /*killer*/) override
158 {
159 if (Tagged)
160 me->SummonCreature(NPC_FREED, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN, 300000);
161 }
162
163 void UpdateAI(uint32 diff) override
164 {
165 if (Tagged)
166 {
167 if (Die_Timer <= diff)
168 {
169 if (Unit* temp = ObjectAccessor::GetUnit(*me, Tagger))
170 {
171 if (Player* player = temp->ToPlayer())
172 player->KilledMonsterCredit(NPC_RESTLESS, me->GetGUID());
173 me->KillSelf();
174 }
175 }
176 else
177 Die_Timer -= diff;
178 }
179 }
180 };
181
182};

194class npc_spectral_ghostly_citizen : public CreatureScript
195{
196public:
197 npc_spectral_ghostly_citizen() : CreatureScript("npc_spectral_ghostly_citizen") { }
198
199 CreatureAI* GetAI(Creature* creature) const override
200 {
201 return new npc_spectral_ghostly_citizenAI(creature);
202 }
203
204 struct npc_spectral_ghostly_citizenAI : public ScriptedAI
205 {
206 npc_spectral_ghostly_citizenAI(Creature* creature) : ScriptedAI(creature)
207 {
208 Initialize();
209 }
210
211 void Initialize()
212 {
213 Die_Timer = 5000;
214 Tagged = false;
215 }
216
217 uint32 Die_Timer;
218 bool Tagged;
219
220 void Reset() override
221 {
222 Initialize();
223 }
224
225 void EnterCombat(Unit* /*who*/) override { }
226
227 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
228 {
229 if (!Tagged && spell->Id == SPELL_EGAN_BLASTER)
230 Tagged = true;
231 }
232
233 void JustDied(Unit* /*killer*/) override
234 {
235 if (Tagged)
236 {
237 for (uint32 i = 1; i <= 4; ++i)
238 {
239 //100%, 50%, 33%, 25% chance to spawn
240 if (urand(1, i) == 1)
241 DoSummon(NPC_RESTLESS, me, 20.0f, 600000);
242 }
243 }
244 }
245
246 void UpdateAI(uint32 diff) override
247 {
248 if (Tagged)
249 {
250 if (Die_Timer <= diff)
251 me->KillSelf();
252 else Die_Timer -= diff;
253 }
254
255 if (!UpdateVictim())
256 return;
257
258 DoMeleeAttackIfReady();
259 }
260
261 void ReceiveEmote(Player* player, uint32 emote) override
262 {
263 switch (emote)
264 {
265 case TEXT_EMOTE_DANCE:
266 EnterEvadeMode();
267 break;
268 case TEXT_EMOTE_RUDE:
269 if (me->IsWithinDistInMap(player, 5))
270 DoCast(player, SPELL_SLAP, false);
271 else
272 me->HandleEmoteCommand(EMOTE_ONESHOT_RUDE);
273 break;
274 case TEXT_EMOTE_WAVE:
275 me->HandleEmoteCommand(EMOTE_ONESHOT_WAVE);
276 break;
277 case TEXT_EMOTE_BOW:
278 me->HandleEmoteCommand(EMOTE_ONESHOT_BOW);
279 break;
280 case TEXT_EMOTE_KISS:
281 me->HandleEmoteCommand(EMOTE_ONESHOT_FLEX);
282 break;
283 }
284 }
285 };

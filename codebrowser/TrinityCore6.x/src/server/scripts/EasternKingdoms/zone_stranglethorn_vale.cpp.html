36## npc_yenniku

38
39class npc_yenniku : public CreatureScript
40{
41public:
42 npc_yenniku() : CreatureScript("npc_yenniku") { }
43
44 CreatureAI* GetAI(Creature* creature) const override
45 {
46 return new npc_yennikuAI(creature);
47 }
48
49 struct npc_yennikuAI : public ScriptedAI
50 {
51 npc_yennikuAI(Creature* creature) : ScriptedAI(creature)
52 {
53 Initialize();
54 bReset = false;
55 }
56
57 void Initialize()
58 {
59 Reset_Timer = 0;
60 }
61
62 uint32 Reset_Timer;
63 bool bReset;
64
65 void Reset() override
66 {
67 Initialize();
68 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_NONE);
69 }
70
71 void SpellHit(Unit* caster, const SpellInfo* spell) override
72 {
73 if (bReset || spell->Id != 3607)
74 return;
75
76 if (Player* player = caster->ToPlayer())
77 {
78 if (player->GetQuestStatus(592) == QUEST_STATUS_INCOMPLETE) //Yenniku's Release
79 {
80 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_STUN);
81 me->CombatStop(); //stop combat
82 me->DeleteThreatList(); //unsure of this
83 me->setFaction(83); //horde generic
84
85 bReset = true;
86 Reset_Timer = 60000;
87 }
88 }
89 }
90
91 void EnterCombat(Unit* /*who*/) override { }
92
93 void UpdateAI(uint32 diff) override
94 {
95 if (bReset)
96 {
97 if (Reset_Timer <= diff)
98 {
99 EnterEvadeMode();
100 bReset = false;
101 me->setFaction(28); //troll, bloodscalp
102 return;
103 }
104
105 Reset_Timer -= diff;
106
107 if (me->IsInCombat() && me->GetVictim())
108 {
109 if (Player* player = me->EnsureVictim()->ToPlayer())
110 {
111 if (player->GetTeam() == HORDE)
112 {
113 me->CombatStop();
114 me->DeleteThreatList();
115 }
116 }
117 }
118 }
119
120 //Return since we have no target
121 if (!UpdateVictim())
122 return;
123
124 DoMeleeAttackIfReady();
125 }
126 };
127};

135 new npc_yenniku();

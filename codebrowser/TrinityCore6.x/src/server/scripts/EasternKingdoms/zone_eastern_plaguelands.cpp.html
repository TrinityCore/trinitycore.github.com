69class npc_augustus_the_touched : public CreatureScript
70{
71public:
72 npc_augustus_the_touched() : CreatureScript("npc_augustus_the_touched") { }
73
74 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
75 {
76 player->PlayerTalkClass->ClearMenus();
77 if (action == GOSSIP_ACTION_TRADE)
78 player->GetSession()->SendListInventory(creature->GetGUID());
79 return true;
80 }
81
82 bool OnGossipHello(Player* player, Creature* creature) override
83 {
84 if (creature->IsQuestGiver())
85 player->PrepareQuestMenu(creature->GetGUID());
86
87 if (creature->IsVendor() && player->GetQuestRewardStatus(6164))
88 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_VENDOR, GOSSIP_TEXT_BROWSE_GOODS, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRADE);
89
90 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
91 return true;
92 }
93};

140#define GOSSIP_HELLO "I am ready to hear your tale, Tirion."
141#define GOSSIP_SELECT1 "Thank you, Tirion. What of your identity?"
142#define GOSSIP_SELECT2 "That is terrible."
143#define GOSSIP_SELECT3 "I will, Tirion."
144
145class npc_tirion_fordring : public CreatureScript
146{
147public:
148 npc_tirion_fordring() : CreatureScript("npc_tirion_fordring") { }
149
150 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
151 {
152 player->PlayerTalkClass->ClearMenus();
153 switch (action)
154 {
155 case GOSSIP_ACTION_INFO_DEF+1:
156 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_SELECT1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
157 player->SEND_GOSSIP_MENU(4493, creature->GetGUID());
158 break;
159 case GOSSIP_ACTION_INFO_DEF+2:
160 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_SELECT2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
161 player->SEND_GOSSIP_MENU(4494, creature->GetGUID());
162 break;
163 case GOSSIP_ACTION_INFO_DEF+3:
164 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_SELECT3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 4);
165 player->SEND_GOSSIP_MENU(4495, creature->GetGUID());
166 break;
167 case GOSSIP_ACTION_INFO_DEF+4:
168 player->CLOSE_GOSSIP_MENU();
169 player->AreaExploredOrEventHappens(5742);
170 break;
171 }
172 return true;
173 }
174
175 bool OnGossipHello(Player* player, Creature* creature) override
176 {
177 if (creature->IsQuestGiver())
178 player->PrepareQuestMenu(creature->GetGUID());
179
180 if (player->GetQuestStatus(5742) == QUEST_STATUS_INCOMPLETE && player->GetStandState() == UNIT_STAND_STATE_SIT)
181 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HELLO, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
182
183 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
184
185 return true;
186 }
187};

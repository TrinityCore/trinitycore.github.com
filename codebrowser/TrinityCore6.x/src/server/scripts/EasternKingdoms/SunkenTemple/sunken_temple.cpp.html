39class at_malfurion_stormrage : public AreaTriggerScript
40{
41public:
42 at_malfurion_stormrage() : AreaTriggerScript("at_malfurion_stormrage") { }
43
44 bool OnTrigger(Player* player, const AreaTriggerEntry* /*areaTrigger*/, bool /*entered*/) override
45 {
46 if (player->GetInstanceScript() && !player->FindNearestCreature(15362, 15))
47 player->SummonCreature(15362, player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), -1.52f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 100000);
48 return false;
49 }
50
51};

56class go_atalai_statue : public GameObjectScript
57{
58public:
59 go_atalai_statue() : GameObjectScript("go_atalai_statue") { }
60
61 bool OnGossipHello(Player* player, GameObject* go) override
62 {
63 if (InstanceScript* instance = player->GetInstanceScript())
64 instance->SetData(EVENT_STATE, go->GetEntry());
65 return false;
66 }
67
68};

70void AddSC_sunken_temple()

39enum eErland
40{
41 SAY_QUESTACCEPT = 0,
42 SAY_START = 1,
43 SAY_AGGRO = 2,
44 SAY_PROGRESS = 3,
45 SAY_LAST = 4,
46
47 SAY_RANE = 0,
48 SAY_RANE_ANSWER = 5,
49 SAY_MOVE_QUINN = 6,
50
51 SAY_QUINN = 7,
52 SAY_QUINN_ANSWER = 0,
53 SAY_BYE = 8,
54
55 QUEST_ESCORTING = 435,
56 NPC_RANE = 1950,
57 NPC_QUINN = 1951
58};
59
60class npc_deathstalker_erland : public CreatureScript
61{
62public:
63 npc_deathstalker_erland() : CreatureScript("npc_deathstalker_erland") { }
64
65 struct npc_deathstalker_erlandAI : public npc_escortAI
66 {
67 npc_deathstalker_erlandAI(Creature* creature) : npc_escortAI(creature) { }
68
69 void WaypointReached(uint32 waypointId) override
70 {
71 Player* player = GetPlayerForEscort();
72 if (!player)
73 return;
74
75 switch (waypointId)
76 {
77 case 1:
78 Talk(SAY_START, player);
79 break;
80 case 10:
81 Talk(SAY_PROGRESS);
82 break;
83 case 13:
84 Talk(SAY_LAST, player);
85 player->GroupEventHappens(QUEST_ESCORTING, me);
86 break;
87 case 15:
88 if (Creature* rane = me->FindNearestCreature(NPC_RANE, 20.0f))
89 rane->AI()->Talk(SAY_RANE);
90 break;
91 case 16:
92 Talk(SAY_RANE_ANSWER);
93 break;
94 case 17:
95 Talk(SAY_MOVE_QUINN);
96 break;
97 case 24:
98 Talk(SAY_QUINN);
99 break;
100 case 25:
101 if (Creature* quinn = me->FindNearestCreature(NPC_QUINN, 20.0f))
102 quinn->AI()->Talk(SAY_QUINN_ANSWER);
103 break;
104 case 26:
105 Talk(SAY_BYE);
106 break;
107 }
108 }
109
110 void Reset() override { }
111
112 void EnterCombat(Unit* who) override
113 {
114 Talk(SAY_AGGRO, who);
115 }
116 };
117
118 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
119 {
120 if (quest->GetQuestId() == QUEST_ESCORTING)
121 {
122 creature->AI()->Talk(SAY_QUESTACCEPT, player);
123
124 if (npc_escortAI* pEscortAI = CAST_AI(npc_deathstalker_erland::npc_deathstalker_erlandAI, creature->AI()))
125 pEscortAI->Start(true, false, player->GetGUID());
126 }
127
128 return true;
129 }
130
131 CreatureAI* GetAI(Creature* creature) const override
132 {
133 return new npc_deathstalker_erlandAI(creature);
134 }
135};

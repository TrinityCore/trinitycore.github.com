19/* ScriptData
20SDName: boss_archaedas
21SD%Complete: 100
22SDComment: Archaedas is activated when 1 person (was 3, changed in 3.0.8) clicks on his altar.
23Every 10 seconds he will awaken one of his minions along the wall.
24At 66%, he will awaken the 6 Guardians.
25At 33%, he will awaken the Vault Walkers
26On his death the vault door opens.
27EndScriptData */

66 struct boss_archaedasAI : public ScriptedAI

68 boss_archaedasAI(Creature* creature) : ScriptedAI(creature)
69 {
70 Initialize();
71 instance = me->GetInstanceScript();
72 }

74 void Initialize()
75 {
76 uiTremorTimer = 60000;
77 iAwakenTimer = 0;
78 uiWallMinionTimer = 10000;
79
80 bWakingUp = false;
81 bGuardiansAwake = false;
82 bVaultWalkersAwake = false;
83 }
84
85 uint32 uiTremorTimer;
86 int32 iAwakenTimer;
87 uint32 uiWallMinionTimer;
88 bool bWakingUp;
89
90 bool bGuardiansAwake;
91 bool bVaultWalkersAwake;
92 InstanceScript* instance;
93
94 void Reset() override
95 {
96 Initialize();
97
98 instance->SetData(0, 5); // respawn any dead minions
99 me->setFaction(35);
100 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
101 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
102 me->AddAura(SPELL_FREEZE_ANIM, me);
103 }
104
105 void ActivateMinion(ObjectGuid uiGuid, bool flag)
106 {
107 Unit* minion = ObjectAccessor::GetUnit(*me, uiGuid);
108
109 if (minion && minion->IsAlive())
110 {
111 DoCast(minion, SPELL_AWAKEN_VAULT_WALKER, flag);
112 minion->CastSpell(minion, SPELL_ARCHAEDAS_AWAKEN, true);
113 minion->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
114 minion->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
115 minion->setFaction(14);
116 minion->RemoveAura(SPELL_MINION_FREEZE_ANIM);
117 }
118 }
119
120 void EnterCombat(Unit* /*who*/) override
121 {
122 me->setFaction(14);
123 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
124 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
125 }
126
127 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
128 {
129 // Being woken up from the altar, start the awaken sequence
130 if (spell == sSpellMgr->GetSpellInfo(SPELL_ARCHAEDAS_AWAKEN))
131 {
132 Talk(SAY_AGGRO);
133 iAwakenTimer = 4000;
134 bWakingUp = true;
135 }
136 }
137
138 void KilledUnit(Unit* /*victim*/) override
139 {
140 Talk(SAY_KILL);
141 }
142
143 void UpdateAI(uint32 uiDiff) override
144 {
145 // we're still doing awaken animation
146 if (bWakingUp && iAwakenTimer >= 0)
147 {
148 iAwakenTimer -= uiDiff;
149 return; // dont do anything until we are done
150 } else if (bWakingUp && iAwakenTimer <= 0)
151 {
152 bWakingUp = false;
153 AttackStart(ObjectAccessor::GetUnit(*me, instance->GetGuidData(0)));
154 return; // dont want to continue until we finish the AttackStart method
155 }
156
157 //Return since we have no target
158 if (!UpdateVictim())
159 return;
160
161 // wake a wall minion
162 if (uiWallMinionTimer <= uiDiff)
163 {
164 instance->SetData(DATA_MINIONS, IN_PROGRESS);
165
166 uiWallMinionTimer = 10000;
167 } else uiWallMinionTimer -= uiDiff;
168
169 //If we are <66 summon the guardians
170 if (!bGuardiansAwake && !HealthAbovePct(66))
171 {
172 ActivateMinion(instance->GetGuidData(5), true); // EarthenGuardian1
173 ActivateMinion(instance->GetGuidData(6), true); // EarthenGuardian2
174 ActivateMinion(instance->GetGuidData(7), true); // EarthenGuardian3
175 ActivateMinion(instance->GetGuidData(8), true); // EarthenGuardian4
176 ActivateMinion(instance->GetGuidData(9), true); // EarthenGuardian5
177 ActivateMinion(instance->GetGuidData(10), false); // EarthenGuardian6
178 Talk(SAY_SUMMON_GUARDIANS);
179 bGuardiansAwake = true;
180 }
181
182 //If we are <33 summon the vault walkers
183 if (!bVaultWalkersAwake && !HealthAbovePct(33))
184 {
185 ActivateMinion(instance->GetGuidData(1), true); // VaultWalker1
186 ActivateMinion(instance->GetGuidData(2), true); // VaultWalker2
187 ActivateMinion(instance->GetGuidData(3), true); // VaultWalker3
188 ActivateMinion(instance->GetGuidData(4), false); // VaultWalker4
189 Talk(SAY_SUMMON_VAULT_WALKERS);
190 bVaultWalkersAwake = true;
191 }
192
193 if (uiTremorTimer <= uiDiff)
194 {
195 //Cast
196 DoCastVictim(SPELL_GROUND_TREMOR);
197
198 //45 seconds until we should cast this agian
199 uiTremorTimer = 45000;
200 } else uiTremorTimer -= uiDiff;
201
202 DoMeleeAttackIfReady();
203 }
204
205 void JustDied (Unit* /*killer*/) override
206 {
207 instance->SetData(DATA_ANCIENT_DOOR, DONE); // open the vault door
208 instance->SetData(DATA_MINIONS, SPECIAL); // deactivate his minions
209 }
210 };
211
212 CreatureAI* GetAI(Creature* creature) const override
213 {
214 return GetInstanceAI<boss_archaedasAI>(creature);
215 }
216};

404 {

406 if (!instance)
407 return false;
408
409 player->CastSpell (player, SPELL_BOSS_OBJECT_VISUAL, false);

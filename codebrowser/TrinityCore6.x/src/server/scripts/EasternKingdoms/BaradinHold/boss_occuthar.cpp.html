53class boss_occuthar : public CreatureScript
54{
55 public:
56 boss_occuthar() : CreatureScript("boss_occuthar") { }
57
58 struct boss_occutharAI : public BossAI
59 {
60 boss_occutharAI(Creature* creature) : BossAI(creature, DATA_OCCUTHAR),
61 _vehicle(me->GetVehicleKit())
62 {
63 ASSERT(_vehicle);
64 }
65
66 void EnterCombat(Unit* /*who*/) override
67 {
68 _EnterCombat();
69 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, me);
70 events.ScheduleEvent(EVENT_SEARING_SHADOWS, 8 * IN_MILLISECONDS);
71 events.ScheduleEvent(EVENT_FOCUSED_FIRE, 15 * IN_MILLISECONDS);
72 events.ScheduleEvent(EVENT_EYES_OF_OCCUTHAR, 30 * IN_MILLISECONDS);
73 events.ScheduleEvent(EVENT_BERSERK, 5 * MINUTE * IN_MILLISECONDS);
74 }
75
76 void EnterEvadeMode() override
77 {
78 BossAI::EnterEvadeMode();
79 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
80 _DespawnAtEvade();
81 }
82
83 void JustDied(Unit* /*killer*/) override
84 {
85 _JustDied();
86 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
87 }
88
89 void JustSummoned(Creature* summon) override
90 {
91 summons.Summon(summon);
92
93 if (summon->GetEntry() == NPC_FOCUS_FIRE_DUMMY)
94 {
95 DoCast(summon, SPELL_FOCUSED_FIRE);
96
97 for (uint8 i = 0; i < MAX_OCCUTHAR_VEHICLE_SEATS; ++i)
98 {
99 if (Unit* vehicle = _vehicle->GetPassenger(i))
100 vehicle->CastSpell(summon, SPELL_FOCUSED_FIRE_VISUAL);
101 }
102 }
103 }
104
105 void UpdateAI(uint32 diff) override
106 {
107 if (!UpdateVictim())
108 return;
109
110 events.Update(diff);
111
112 if (me->HasUnitState(UNIT_STATE_CASTING))
113 return;
114
115 while (uint32 eventId = events.ExecuteEvent())
116 {
117 switch (eventId)
118 {
119 case EVENT_SEARING_SHADOWS:
120 DoCastAOE(SPELL_SEARING_SHADOWS);
121 events.ScheduleEvent(EVENT_SEARING_SHADOWS, 25 * IN_MILLISECONDS);
122 break;
123 case EVENT_FOCUSED_FIRE:
124 DoCastAOE(SPELL_FOCUSED_FIRE_TRIGGER, true);
125 events.ScheduleEvent(EVENT_FOCUSED_FIRE, 15 * IN_MILLISECONDS);
126 break;
127 case EVENT_EYES_OF_OCCUTHAR:
128 DoCastAOE(SPELL_EYES_OF_OCCUTHAR);
129 events.RescheduleEvent(EVENT_FOCUSED_FIRE, 15 * IN_MILLISECONDS);
130 events.ScheduleEvent(EVENT_EYES_OF_OCCUTHAR, 60 * IN_MILLISECONDS);
131 break;
132 case EVENT_BERSERK:
133 DoCast(me, SPELL_BERSERK, true);
134 break;
135 default:
136 break;
137 }
138 }
139
140 DoMeleeAttackIfReady();
141 }
142
143 private:
144 Vehicle* _vehicle;
145 };
146
147 CreatureAI* GetAI(Creature* creature) const override
148 {
149 return GetBaradinHoldAI<boss_occutharAI>(creature);
150 }
151};

153class npc_eyestalk : public CreatureScript
154{
155 public:
156 npc_eyestalk() : CreatureScript("npc_eyestalk") { }
157
158 struct npc_eyestalkAI : public ScriptedAI
159 {
160 npc_eyestalkAI(Creature* creature) : ScriptedAI(creature),
161 _instance(creature->GetInstanceScript())
162 {
163 _damageCount = 0;
164 }
165
166 void IsSummonedBy(Unit* /*summoner*/) override
167 {
168 // player is the spellcaster so register summon manually
169 if (Creature* occuthar = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_OCCUTHAR)))
170 occuthar->AI()->JustSummoned(me);
171 }
172
173 void Reset() override
174 {
175 _events.Reset();
176 _events.ScheduleEvent(EVENT_FOCUSED_FIRE_FIRST_DAMAGE, 0);
177 }
178
179 void UpdateAI(uint32 diff) override
180 {
181 _events.Update(diff);
182
183 if (_events.ExecuteEvent() == EVENT_FOCUSED_FIRE_FIRST_DAMAGE)
184 {
185 DoCastAOE(SPELL_FOCUSED_FIRE_FIRST_DAMAGE);
186 if (++_damageCount < 2)
187 _events.ScheduleEvent(EVENT_FOCUSED_FIRE_FIRST_DAMAGE, 1 * IN_MILLISECONDS);
188 }
189 }
190
191 void EnterEvadeMode() override { } // Never evade
192
193 private:
194 InstanceScript* _instance;
195 EventMap _events;
196 uint8 _damageCount;
197 };
198
199 CreatureAI* GetAI(Creature* creature) const override
200 {
201 return GetBaradinHoldAI<npc_eyestalkAI>(creature);
202 }

205class FocusedFireTargetSelector : public std::unary_function<Unit *, bool>
206{
207 public:
208 FocusedFireTargetSelector(Creature* me, const Unit* victim) : _me(me), _victim(victim) { }
209
210 bool operator() (WorldObject* target)
211 {
212 if (target == _victim && _me->getThreatManager().getThreatList().size() > 1)
213 return true;
214
215 if (target->GetTypeId() != TYPEID_PLAYER)
216 return true;
217
218 return false;
219 }
220
221 Creature* _me;
222 Unit const* _victim;
223};

236 {

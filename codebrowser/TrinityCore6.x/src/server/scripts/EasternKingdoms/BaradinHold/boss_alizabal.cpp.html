63class at_alizabal_intro : public AreaTriggerScript
64{
65 public:
66 at_alizabal_intro() : AreaTriggerScript("at_alizabal_intro") { }
67
68 bool OnTrigger(Player* player, AreaTriggerEntry const* /*areaTrigger*/, bool /*entered*/) override
69 {
70 if (InstanceScript* instance = player->GetInstanceScript())
71 if (Creature* alizabal = ObjectAccessor::GetCreature(*player, instance->GetGuidData(DATA_ALIZABAL)))
72 alizabal->AI()->DoAction(ACTION_INTRO);
73 return true;
74 }
75};

77class boss_alizabal : public CreatureScript
78{
79 public:
80 boss_alizabal() : CreatureScript("boss_alizabal") { }
81
82 struct boss_alizabalAI : public BossAI
83 {
84 boss_alizabalAI(Creature* creature) : BossAI(creature, DATA_ALIZABAL) { }
85
86 void Reset() override
87 {
88 _Reset();
89 _hate = false;
90 _skewer = false;
91 }
92
93 void EnterCombat(Unit* /*who*/) override
94 {
95 _EnterCombat();
96 Talk(SAY_AGGRO);
97 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, me);
98 events.ScheduleEvent(EVENT_RANDOM_CAST, 10000);
99 }
100
101 void JustDied(Unit* /*killer*/) override
102 {
103 _JustDied();
104 Talk(SAY_DEATH);
105 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
106 }
107
108 void KilledUnit(Unit* who) override
109 {
110 if (who->GetTypeId() == TYPEID_PLAYER)
111 Talk(SAY_SLAY);
112 }
113
114 void EnterEvadeMode() override
115 {
116 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
117 me->GetMotionMaster()->MoveTargetedHome();
118 _DespawnAtEvade();
119 }
120
121 void DoAction(int32 action) override
122 {
123 switch (action)
124 {
125 case ACTION_INTRO:
126 if (!_intro)
127 {
128 Talk(SAY_INTRO);
129 _intro = true;
130 }
131 break;
132 }
133 }
134
135 void MovementInform(uint32 /*type*/, uint32 pointId) override
136 {
137 switch (pointId)
138 {
139 case POINT_STORM:
140 events.ScheduleEvent(EVENT_CAST_STORM, 1);
141 break;
142 }
143 }
144
145 void UpdateAI(uint32 diff) override
146 {
147 if (!UpdateVictim())
148 return;
149
150 events.Update(diff);
151
152 while (uint32 eventId = events.ExecuteEvent())
153 {
154 switch (eventId)
155 {
156 case EVENT_RANDOM_CAST:
157 switch (urand(0, 1))
158 {
159 case 0:
160 if (!_skewer)
161 {
162 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0))
163 {
164 DoCast(target, SPELL_SKEWER, true);
165 Talk(SAY_SKEWER);
166 Talk(SAY_SKEWER_ANNOUNCE, target);
167 }
168 _skewer = true;
169 events.ScheduleEvent(EVENT_RANDOM_CAST, urand(7000, 10000));
170 }
171 else if (!_hate)
172 {
173 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me)))
174 {
175 DoCast(target, SPELL_SEETHING_HATE, true);
176 Talk(SAY_HATE);
177 }
178 _hate = true;
179 events.ScheduleEvent(EVENT_RANDOM_CAST, urand(7000, 10000));
180 }
181 else if (_hate && _skewer)
182 {
183 Talk(SAY_BLADE_STORM);
184 DoCastAOE(SPELL_BLADE_DANCE_DUMMY);
185 DoCastAOE(SPELL_BLADE_DANCE);
186 events.ScheduleEvent(EVENT_RANDOM_CAST, 21000);
187 events.ScheduleEvent(EVENT_MOVE_STORM, 4050);
188 events.ScheduleEvent(EVENT_STOP_STORM, 13000);
189 }
190 break;
191 case 1:
192 if (!_hate)
193 {
194 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me)))
195 {
196 DoCast(target, SPELL_SEETHING_HATE, true);
197 Talk(SAY_HATE);
198 }
199 _hate = true;
200 events.ScheduleEvent(EVENT_RANDOM_CAST, urand(7000, 10000));
201 }
202 else if (!_skewer)
203 {
204 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0))
205 {
206 DoCast(target, SPELL_SKEWER, true);
207 Talk(SAY_SKEWER);
208 Talk(SAY_SKEWER_ANNOUNCE, target);
209 }
210 _skewer = true;
211 events.ScheduleEvent(EVENT_RANDOM_CAST, urand(7000, 10000));
212 }
213 else if (_hate && _skewer)
214 {
215 Talk(SAY_BLADE_STORM);
216 DoCastAOE(SPELL_BLADE_DANCE_DUMMY);
217 DoCastAOE(SPELL_BLADE_DANCE);
218 events.ScheduleEvent(EVENT_RANDOM_CAST, 21000);
219 events.ScheduleEvent(EVENT_MOVE_STORM, 4050);
220 events.ScheduleEvent(EVENT_STOP_STORM, 13000);
221 }
222 break;
223 }
224 break;
225 case EVENT_MOVE_STORM:
226 me->SetSpeed(MOVE_RUN, 4.0f);
227 me->SetSpeed(MOVE_WALK, 4.0f);
228 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me)))
229 me->GetMotionMaster()->MovePoint(POINT_STORM, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ());
230 events.ScheduleEvent(EVENT_MOVE_STORM, 4050);
231 break;
232 case EVENT_STOP_STORM:
233 me->RemoveAura(SPELL_BLADE_DANCE);
234 me->RemoveAura(SPELL_BLADE_DANCE_DUMMY);
235 me->SetSpeed(MOVE_WALK, 1.0f);
236 me->SetSpeed(MOVE_RUN, 1.14f);
237 me->GetMotionMaster()->MoveChase(me->GetVictim());
238 _hate = false;
239 _skewer = false;
240 break;
241 case EVENT_CAST_STORM:
242 DoCastAOE(SPELL_BLADE_DANCE);
243 break;
244 }
245 }
246
247 DoMeleeAttackIfReady();
248 }
249
250 private:
251 bool _intro = false;
252 bool _hate = false;
253 bool _skewer = false;
254 };
255
256 CreatureAI* GetAI(Creature* creature) const override
257 {
258 return GetBaradinHoldAI<boss_alizabalAI>(creature);
259 }
260};

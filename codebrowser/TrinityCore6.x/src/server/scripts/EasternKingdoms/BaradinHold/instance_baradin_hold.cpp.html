18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "baradin_hold.h"
21
22DoorData const doorData[] =
23{
24 { GO_ARGALOTH_DOOR, DATA_ARGALOTH, DOOR_TYPE_ROOM, BOUNDARY_NONE },
25 { GO_OCCUTHAR_DOOR, DATA_OCCUTHAR, DOOR_TYPE_ROOM, BOUNDARY_NONE },
26 { GO_ALIZABAL_DOOR, DATA_ALIZABAL, DOOR_TYPE_ROOM, BOUNDARY_NONE },
27 { 0, 0, DOOR_TYPE_ROOM, BOUNDARY_NONE } // END
28};
29
30class instance_baradin_hold: public InstanceMapScript
31{
32 public:
33 instance_baradin_hold() : InstanceMapScript(BHScriptName, 757) { }
34
35 struct instance_baradin_hold_InstanceMapScript: public InstanceScript
36 {
37 instance_baradin_hold_InstanceMapScript(InstanceMap* map) : InstanceScript(map)
38 {
39 SetHeaders(DataHeader);
40 SetBossNumber(EncounterCount);
41 LoadDoorData(doorData);
42 }
43
44 void OnCreatureCreate(Creature* creature) override
45 {
46 switch(creature->GetEntry())
47 {
48 case BOSS_ARGALOTH:
49 ArgalothGUID = creature->GetGUID();
50 break;
51 case BOSS_OCCUTHAR:
52 OccutharGUID = creature->GetGUID();
53 break;
54 case BOSS_ALIZABAL:
55 AlizabalGUID = creature->GetGUID();
56 break;
57 }
58 }
59
60 void OnGameObjectCreate(GameObject* go) override
61 {
62 switch(go->GetEntry())
63 {
64 case GO_ARGALOTH_DOOR:
65 case GO_OCCUTHAR_DOOR:
66 case GO_ALIZABAL_DOOR:
67 AddDoor(go, true);
68 break;
69 }
70 }
71
72 ObjectGuid GetGuidData(uint32 data) const override
73 {
74 switch (data)
75 {
76 case DATA_ARGALOTH:
77 return ArgalothGUID;
78 case DATA_OCCUTHAR:
79 return OccutharGUID;
80 case DATA_ALIZABAL:
81 return AlizabalGUID;
82 default:
83 break;
84 }
85
86 return ObjectGuid::Empty;
87 }
88
89 void OnGameObjectRemove(GameObject* go) override
90 {
91 switch(go->GetEntry())
92 {
93 case GO_ARGALOTH_DOOR:
94 case GO_OCCUTHAR_DOOR:
95 case GO_ALIZABAL_DOOR:
96 AddDoor(go, false);
97 break;
98 }
99 }
100
101 protected:
102 ObjectGuid ArgalothGUID;
103 ObjectGuid OccutharGUID;
104 ObjectGuid AlizabalGUID;
105 };
106
107 InstanceScript* GetInstanceScript(InstanceMap* map) const override
108 {
109 return new instance_baradin_hold_InstanceMapScript(map);
110 }
111};
112
113void AddSC_instance_baradin_hold()
114{
115 new instance_baradin_hold();
116}

39 EVENT_CONSUMING_DARKNESS = 2,
40 EVENT_BERSERK = 3

43class boss_pit_lord_argaloth : public CreatureScript
44{
45 public:
46 boss_pit_lord_argaloth() : CreatureScript("boss_pit_lord_argaloth") { }
47
48 struct boss_pit_lord_argalothAI : public BossAI
49 {
50 boss_pit_lord_argalothAI(Creature* creature) : BossAI(creature, DATA_ARGALOTH) { }
51
52 void EnterCombat(Unit* /*who*/) override
53 {
54 _EnterCombat();
55 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, me);
56 events.ScheduleEvent(EVENT_METEOR_SLASH, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
57 events.ScheduleEvent(EVENT_CONSUMING_DARKNESS, urand(20 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
58 events.ScheduleEvent(EVENT_BERSERK, 5 * MINUTE * IN_MILLISECONDS);
59 }
60
61 void EnterEvadeMode() override
62 {
63 me->GetMotionMaster()->MoveTargetedHome();
64 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
65 _DespawnAtEvade();
66 }
67
68 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
69 {
70 if (me->HealthBelowPctDamaged(33, damage) ||
71 me->HealthBelowPctDamaged(66, damage))
72 {
73 DoCastAOE(SPELL_FEL_FIRESTORM);
74 }
75 }
76
77 void JustDied(Unit* /*killer*/) override
78 {
79 _JustDied();
80 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
81 }
82
83 void UpdateAI(uint32 diff) override
84 {
85 if (!UpdateVictim())
86 return;
87
88 events.Update(diff);
89
90 if (me->HasUnitState(UNIT_STATE_CASTING))
91 return;
92
93 while (uint32 eventId = events.ExecuteEvent())
94 {
95 switch (eventId)
96 {
97 case EVENT_METEOR_SLASH:
98 DoCastAOE(SPELL_METEOR_SLASH);
99 events.ScheduleEvent(EVENT_METEOR_SLASH, urand(15 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
100 break;
101 case EVENT_CONSUMING_DARKNESS:
102 DoCastAOE(SPELL_CONSUMING_DARKNESS, true);
103 events.ScheduleEvent(EVENT_CONSUMING_DARKNESS, urand(20 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
104 break;
105 case EVENT_BERSERK:
106 DoCast(me, SPELL_BERSERK, true);
107 break;
108 default:
109 break;
110 }
111 }
112
113 DoMeleeAttackIfReady();
114 }
115 };
116
117 CreatureAI* GetAI(Creature* creature) const override
118 {
119 return GetBaradinHoldAI<boss_pit_lord_argalothAI>(creature);
120 }
121};

127 spell_argaloth_consuming_darkness() : SpellScriptLoader("spell_argaloth_consuming_darkness") { }

129 class spell_argaloth_consuming_darkness_SpellScript : public SpellScript

142 };

150// 88942 / 95172 - Meteor Slash
151class spell_argaloth_meteor_slash : public SpellScriptLoader
152{
153 public:
154 spell_argaloth_meteor_slash() : SpellScriptLoader("spell_argaloth_meteor_slash") { }
155
156 class spell_argaloth_meteor_slash_SpellScript : public SpellScript
157 {
158 PrepareSpellScript(spell_argaloth_meteor_slash_SpellScript);
159
160 void CountTargets(std::list<WorldObject*>& targets)
161 {
162 _targetCount = targets.size();
163 }
164
165 void SplitDamage()
166 {
167 if (!_targetCount)
168 return;
169
170 SetHitDamage(GetHitDamage() / _targetCount);
171 }
172
173 void Register() override
174 {
175 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_argaloth_meteor_slash_SpellScript::CountTargets, EFFECT_0, TARGET_UNIT_CONE_ENEMY_104);
176 OnHit += SpellHitFn(spell_argaloth_meteor_slash_SpellScript::SplitDamage);
177 }
178
179 private:
180 uint32 _targetCount = 0;
181 };
182
183 SpellScript* GetSpellScript() const override
184 {
185 return new spell_argaloth_meteor_slash_SpellScript();
186 }
187};

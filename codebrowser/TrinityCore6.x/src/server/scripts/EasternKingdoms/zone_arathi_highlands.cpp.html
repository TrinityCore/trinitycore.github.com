39enum ProfessorPhizzlethorpe
40{
41 // Yells
42 SAY_PROGRESS_1 = 0,
43 SAY_PROGRESS_2 = 1,
44 SAY_PROGRESS_3 = 2,
45 EMOTE_PROGRESS_4 = 3,
46 SAY_AGGRO = 4,
47 SAY_PROGRESS_5 = 5,
48 SAY_PROGRESS_6 = 6,
49 SAY_PROGRESS_7 = 7,
50 EMOTE_PROGRESS_8 = 8,
51 SAY_PROGRESS_9 = 9,
52 EVENT_SAY_3 = 1,
53 EVENT_SAY_6 = 2,
54 EVENT_SAY_8 = 3,
55
56 // Quests
57 QUEST_SUNKEN_TREASURE = 665,
58 QUEST_GOGGLE_BOGGLE = 26050,
59 // Creatures
60 NPC_VENGEFUL_SURGE = 2776,
61 FACTION_SUNKEN_TREASURE = 113
62};
63
64class npc_professor_phizzlethorpe : public CreatureScript
65{
66 public:
67 npc_professor_phizzlethorpe() : CreatureScript("npc_professor_phizzlethorpe") { }
68
69 struct npc_professor_phizzlethorpeAI : public npc_escortAI
70 {
71 npc_professor_phizzlethorpeAI(Creature* creature) : npc_escortAI(creature) { }
72
73 void WaypointReached(uint32 waypointId) override
74 {
75 Player* player = GetPlayerForEscort();
76 if (!player)
77 return;
78
79 switch (waypointId)
80 {
81 case 6:
82 Talk(SAY_PROGRESS_2, player);
83 events.ScheduleEvent(EVENT_SAY_3, 3000);
84 break;
85 case 8:
86 Talk(EMOTE_PROGRESS_4);
87 me->SummonCreature(NPC_VENGEFUL_SURGE, -2065.505f, -2136.88f, 22.20362f, 1.0f, TEMPSUMMON_CORPSE_DESPAWN, 0);
88 me->SummonCreature(NPC_VENGEFUL_SURGE, -2059.249f, -2134.88f, 21.51582f, 1.0f, TEMPSUMMON_CORPSE_DESPAWN, 0);
89 break;
90 case 11:
91 Talk(SAY_PROGRESS_5, player);
92 events.ScheduleEvent(EVENT_SAY_6, 11000);
93 break;
94 case 17:
95 Talk(SAY_PROGRESS_7, player);
96 events.ScheduleEvent(EVENT_SAY_8, 6000);
97 break;
98 }
99 }
100
101 void JustSummoned(Creature* summoned) override
102 {
103 summoned->AI()->AttackStart(me);
104 }
105
106 void EnterCombat(Unit* /*who*/) override
107 {
108 Talk(SAY_AGGRO);
109 }
110
111 void sQuestAccept(Player* player, Quest const* quest) override
112 {
113 if (quest->GetQuestId() == QUEST_SUNKEN_TREASURE)
114 {
115 Talk(SAY_PROGRESS_1, player);
116 npc_escortAI::Start(false, false, player->GetGUID(), quest);
117 me->setFaction(FACTION_SUNKEN_TREASURE);
118 }
119 }
120
121 void UpdateAI(uint32 diff) override
122 {
123 Player* player = GetPlayerForEscort();
124 if (!player)
125 return;
126
127 events.Update(diff);
128
129 while (uint32 event = events.ExecuteEvent())
130 {
131 switch (event)
132 {
133 case EVENT_SAY_3:
134 Talk(SAY_PROGRESS_3, player);
135 break;
136 case EVENT_SAY_6:
137 Talk(SAY_PROGRESS_6, player);
138 SetRun();
139 break;
140 case EVENT_SAY_8:
141 Talk(EMOTE_PROGRESS_8);
142 Talk(SAY_PROGRESS_9, player);
143 player->GroupEventHappens(QUEST_GOGGLE_BOGGLE, me);
144 break;
145 }
146 }
147 npc_escortAI::UpdateAI(diff);
148 }
149
150 EventMap events;
151 };
152
153 CreatureAI* GetAI(Creature* creature) const override
154 {
155 return new npc_professor_phizzlethorpeAI(creature);
156 }
157};

162}

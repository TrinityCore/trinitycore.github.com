56void OutdoorPvPTF::FillInitialWorldStates(WorldPackets::WorldState::InitWorldStates& packet)
57{
58 packet.Worldstates.emplace_back(uint32(TF_UI_TOWER_COUNT_H), int32(m_HordeTowersControlled));
59 packet.Worldstates.emplace_back(uint32(TF_UI_TOWER_COUNT_A), int32(m_AllianceTowersControlled));
60 packet.Worldstates.emplace_back(uint32(TF_UI_TOWERS_CONTROLLED_DISPLAY), int32(!m_IsLocked));
61 packet.Worldstates.emplace_back(uint32(TF_UI_LOCKED_TIME_MINUTES_FIRST_DIGIT), int32(first_digit));
62 packet.Worldstates.emplace_back(uint32(TF_UI_LOCKED_TIME_MINUTES_SECOND_DIGIT), int32(second_digit));
63 packet.Worldstates.emplace_back(uint32(TF_UI_LOCKED_TIME_HOURS), int32(hours_left));
64
65 packet.Worldstates.emplace_back(uint32(TF_UI_LOCKED_DISPLAY_NEUTRAL), int32(m_IsLocked && !m_HordeTowersControlled && !m_AllianceTowersControlled));
66 packet.Worldstates.emplace_back(uint32(TF_UI_LOCKED_DISPLAY_HORDE), int32(m_IsLocked && (m_HordeTowersControlled > m_AllianceTowersControlled)));
67 packet.Worldstates.emplace_back(uint32(TF_UI_LOCKED_DISPLAY_ALLIANCE), int32(m_IsLocked && (m_HordeTowersControlled < m_AllianceTowersControlled)));
68
69 for (OPvPCapturePointMap::iterator itr = m_capturePoints.begin(); itr != m_capturePoints.end(); ++itr)
70 itr->second->FillInitialWorldStates(packet);
71}

73void OutdoorPvPTF::SendRemoveWorldStates(Player* player)
74{
75 player->SendUpdateWorldState(TF_UI_TOWER_COUNT_H, uint32(0));
76 player->SendUpdateWorldState(TF_UI_TOWER_COUNT_A, uint32(0));
77 player->SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(0));
78
79 player->SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_FIRST_DIGIT, uint32(0));
80 player->SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_SECOND_DIGIT, uint32(0));
81 player->SendUpdateWorldState(TF_UI_LOCKED_TIME_HOURS, uint32(0));
82
83 player->SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
84 player->SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(0));
85 player->SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(0));
86
87 for (int i = 0; i < TF_TOWER_NUM; ++i)
88 {
89 player->SendUpdateWorldState(uint32(TFTowerWorldStates[i].n), uint32(0));
90 player->SendUpdateWorldState(uint32(TFTowerWorldStates[i].h), uint32(0));
91 player->SendUpdateWorldState(uint32(TFTowerWorldStates[i].a), uint32(0));
92 }

102bool OutdoorPvPTF::Update(uint32 diff)
103{
104 bool changed = OutdoorPvP::Update(diff);
105
106 if (changed)
107 {
108 if (m_AllianceTowersControlled == TF_TOWER_NUM)
109 {
110 TeamApplyBuff(TEAM_ALLIANCE, TF_CAPTURE_BUFF);
111 m_IsLocked = true;
112 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
113 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(0));
114 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(1));
115 SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(0));
116 }
117 else if (m_HordeTowersControlled == TF_TOWER_NUM)
118 {
119 TeamApplyBuff(TEAM_HORDE, TF_CAPTURE_BUFF);
120 m_IsLocked = true;
121 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
122 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(1));
123 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(0));
124 SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(0));
125 }
126 else
127 {
128 TeamCastSpell(TEAM_ALLIANCE, -TF_CAPTURE_BUFF);
129 TeamCastSpell(TEAM_HORDE, -TF_CAPTURE_BUFF);
130 }
131 SendUpdateWorldState(TF_UI_TOWER_COUNT_A, m_AllianceTowersControlled);
132 SendUpdateWorldState(TF_UI_TOWER_COUNT_H, m_HordeTowersControlled);
133 }
134 if (m_IsLocked)
135 {
136 // lock timer is down, release lock
137 if (m_LockTimer < diff)
138 {
139 m_LockTimer = TF_LOCK_TIME;
140 m_LockTimerUpdate = 0;
141 m_IsLocked = false;
142 SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(1));
143 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
144 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(0));
145 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(0));
146 }
147 else
148 {
149 // worldstateui update timer is down, update ui with new time data
150 if (m_LockTimerUpdate < diff)
151 {
152 m_LockTimerUpdate = TF_LOCK_TIME_UPDATE;
153 uint32 minutes_left = m_LockTimer / 60000;
154 hours_left = minutes_left / 60;
155 minutes_left -= hours_left * 60;
156 second_digit = minutes_left % 10;
157 first_digit = minutes_left / 10;
158
159 SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_FIRST_DIGIT, first_digit);
160 SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_SECOND_DIGIT, second_digit);
161 SendUpdateWorldState(TF_UI_LOCKED_TIME_HOURS, hours_left);
162 } else m_LockTimerUpdate -= diff;
163 m_LockTimer -= diff;
164 }
165 }
166 return changed;
167}

169void OutdoorPvPTF::HandlePlayerEnterZone(Player* player, uint32 zone)
170{
171 if (player->GetTeam() == ALLIANCE)
172 {
173 if (m_AllianceTowersControlled >= TF_TOWER_NUM)
174 player->CastSpell(player, TF_CAPTURE_BUFF, true);
175 }
176 else
177 {
178 if (m_HordeTowersControlled >= TF_TOWER_NUM)
179 player->CastSpell(player, TF_CAPTURE_BUFF, true);
180 }
181 OutdoorPvP::HandlePlayerEnterZone(player, zone);
182}

216bool OutdoorPvPTF::SetupOutdoorPvP()
217{
218 m_AllianceTowersControlled = 0;
219 m_HordeTowersControlled = 0;
220
221 m_IsLocked = false;
222 m_LockTimer = TF_LOCK_TIME;
223 m_LockTimerUpdate = 0;
224 hours_left = 6;
225 second_digit = 0;
226 first_digit = 0;
227
228 SetMapFromZone(OutdoorPvPTFBuffZones[0]);
229
230 // add the zones affected by the pvp buff
231 for (uint8 i = 0; i < OutdoorPvPTFBuffZonesNum; ++i)
232 RegisterZone(OutdoorPvPTFBuffZones[i]);
233
234 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_NW));
235 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_N));
236 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_NE));
237 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_SE));
238 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_S));
239
240 return true;
241}

243bool OPvPCapturePointTF::Update(uint32 diff)
244{
245 // can update even in locked state if gathers the controlling faction
246 bool canupdate = ((((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled() > 0) && m_activePlayers[0].size() > m_activePlayers[1].size()) ||
247 ((((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled() > 0) && m_activePlayers[0].size() < m_activePlayers[1].size());
248 // if gathers the other faction, then only update if the pvp is unlocked
249 canupdate = canupdate || !((OutdoorPvPTF*)m_PvP)->IsLocked();
250 return canupdate && OPvPCapturePoint::Update(diff);
251}

253void OPvPCapturePointTF::ChangeState()
254{
255 // if changing from controlling alliance to horde
256 if (m_OldState == OBJECTIVESTATE_ALLIANCE)
257 {
258 if (uint32 alliance_towers = ((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled())
259 ((OutdoorPvPTF*)m_PvP)->SetAllianceTowersControlled(--alliance_towers);
260 m_PvP->SendDefenseMessage(OutdoorPvPTFBuffZones[0], TEXT_SPIRIT_TOWER_LOSE_ALLIANCE);
261 }
262 // if changing from controlling horde to alliance
263 else if (m_OldState == OBJECTIVESTATE_HORDE)
264 {
265 if (uint32 horde_towers = ((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled())
266 ((OutdoorPvPTF*)m_PvP)->SetHordeTowersControlled(--horde_towers);
267 m_PvP->SendDefenseMessage(OutdoorPvPTFBuffZones[0], TEXT_SPIRIT_TOWER_LOSE_HORDE);
268 }
269
270 uint32 artkit = 21;
271
272 switch (m_State)
273 {
274 case OBJECTIVESTATE_ALLIANCE:
275 {
276 m_TowerState = TF_TOWERSTATE_A;
277 artkit = 2;
278 uint32 alliance_towers = ((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled();
279 if (alliance_towers < TF_TOWER_NUM)
280 ((OutdoorPvPTF*)m_PvP)->SetAllianceTowersControlled(++alliance_towers);
281
282 m_PvP->SendDefenseMessage(OutdoorPvPTFBuffZones[0], TEXT_SPIRIT_TOWER_TAKEN_ALLIANCE);
283
284 for (GuidSet::iterator itr = m_activePlayers[0].begin(); itr != m_activePlayers[0].end(); ++itr)
285 if (Player* player = ObjectAccessor::FindPlayer(*itr))
286 player->AreaExploredOrEventHappens(TF_ALLY_QUEST);
287 break;
288 }
289 case OBJECTIVESTATE_HORDE:
290 {
291 m_TowerState = TF_TOWERSTATE_H;
292 artkit = 1;
293 uint32 horde_towers = ((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled();
294 if (horde_towers < TF_TOWER_NUM)
295 ((OutdoorPvPTF*)m_PvP)->SetHordeTowersControlled(++horde_towers);
296
297 m_PvP->SendDefenseMessage(OutdoorPvPTFBuffZones[0], TEXT_SPIRIT_TOWER_TAKEN_HORDE);
298
299 for (GuidSet::iterator itr = m_activePlayers[1].begin(); itr != m_activePlayers[1].end(); ++itr)
300 if (Player* player = ObjectAccessor::FindPlayer(*itr))
301 player->AreaExploredOrEventHappens(TF_HORDE_QUEST);
302 break;
303 }
304 case OBJECTIVESTATE_NEUTRAL:
305 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
306 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
307 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
308 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
309 m_TowerState = TF_TOWERSTATE_N;
310 break;
311 }
312
313 auto bounds = sMapMgr->FindMap(530, 0)->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
314 for (auto itr = bounds.first; itr != bounds.second; ++itr)
315 itr->second->SetGoArtKit(artkit);
316
317 UpdateTowerState();
318}

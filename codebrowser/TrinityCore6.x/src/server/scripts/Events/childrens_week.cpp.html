134ObjectGuid getOrphanGUID(Player* player, uint32 orphan)
135{
136 if (Aura* orphanOut = player->GetAura(SPELL_ORPHAN_OUT))
137 if (orphanOut->GetCaster() && orphanOut->GetCaster()->GetEntry() == orphan)
138 return orphanOut->GetCaster()->GetGUID();
139
140 return ObjectGuid::Empty;
141}

143/*######
144## npc_winterfin_playmate
145######*/
146class npc_winterfin_playmate : public CreatureScript
147{
148 public:
149 npc_winterfin_playmate() : CreatureScript("npc_winterfin_playmate") { }
150
151 struct npc_winterfin_playmateAI : public ScriptedAI
152 {
153 npc_winterfin_playmateAI(Creature* creature) : ScriptedAI(creature)
154 {
155 Initialize();
156 }
157
158 void Initialize()
159 {
160 timer = 0;
161 phase = 0;
162 playerGUID.Clear();
163 orphanGUID.Clear();
164 }
165
166 void Reset() override
167 {
168 Initialize();
169 }
170
171 void MoveInLineOfSight(Unit* who) override
172 {
173 if (!phase && who && who->GetDistance2d(me) < 10.0f)
174 if (Player* player = who->ToPlayer())
175 if (player->GetQuestStatus(QUEST_PLAYMATE_ORACLE) == QUEST_STATUS_INCOMPLETE)
176 {
177 playerGUID = player->GetGUID();
178 orphanGUID = getOrphanGUID(player, ORPHAN_ORACLE);
179 if (!orphanGUID.IsEmpty())
180 phase = 1;
181 }
182 }
183
184 void UpdateAI(uint32 diff) override
185 {
186 if (!phase)
187 return;
188
189 if (timer <= diff)
190 {
191 Player* player = ObjectAccessor::GetPlayer(*me, playerGUID);
192 Creature* orphan = ObjectAccessor::GetCreature(*me, orphanGUID);
193
194 if (!orphan || !player)
195 {
196 Reset();
197 return;
198 }
199
200 switch (phase)
201 {
202 case 1:
203 orphan->GetMotionMaster()->MovePoint(0, me->GetPositionX() + std::cos(me->GetOrientation()) * 5, me->GetPositionY() + std::sin(me->GetOrientation()) * 5, me->GetPositionZ());
204 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_1);
205 timer = 3000;
206 break;
207 case 2:
208 orphan->SetFacingToObject(me);
209 Talk(TEXT_WINTERFIN_PLAYMATE_1);
210 me->HandleEmoteCommand(EMOTE_STATE_DANCE);
211 timer = 3000;
212 break;
213 case 3:
214 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_2);
215 timer = 3000;
216 break;
217 case 4:
218 Talk(TEXT_WINTERFIN_PLAYMATE_2);
219 timer = 5000;
220 break;
221 case 5:
222 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_3);
223 me->HandleEmoteCommand(EMOTE_STATE_NONE);
224 player->GroupEventHappens(QUEST_PLAYMATE_ORACLE, me);
225 orphan->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
226 Reset();
227 return;
228 }
229 ++phase;
230 }
231 else
232 timer -= diff;
233 }
234
235 private:
236 uint32 timer;
237 int8 phase;
238 ObjectGuid playerGUID;
239 ObjectGuid orphanGUID;
240
241 };
242
243 CreatureAI* GetAI(Creature* creature) const override
244 {
245 return new npc_winterfin_playmateAI(creature);
246 }
247};

800 if (!orphanGUID.IsEmpty())

805 playerGUID = player->GetGUID();

807 if (!orphanGUID.IsEmpty())
808 phase = 7;
809 }
810 }

813 void UpdateAI(uint32 diff) override
814 {
815 if (!phase)
816 return;
817
818 if (timer <= diff)
819 {
820 Player* player = ObjectAccessor::GetPlayer(*me, playerGUID);
821 Creature* orphan = ObjectAccessor::GetCreature(*me, orphanGUID);
822
823 if (!orphan || !player)
824 {
825 Reset();
826 return;
827 }
828
829 switch (phase)
830 {
831 case 1:
832 orphan->GetMotionMaster()->MovePoint(0, me->GetPositionX() + std::cos(me->GetOrientation()) * 5, me->GetPositionY() + std::sin(me->GetOrientation()) * 5, me->GetPositionZ());
833 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_11);
834 timer = 5000;
835 break;
836 case 2:
837 orphan->SetFacingToObject(me);
838 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_12);
839 timer = 5000;
840 break;
841 case 3:
842 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_13);
843 timer = 5000;
844 break;
845 case 4:
846 Talk(TEXT_ALEXSTRASZA_2);
847 me->SetStandState(UNIT_STAND_STATE_KNEEL);
848 me->SetFacingToObject(orphan);
849 timer = 5000;
850 break;
851 case 5:
852 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_14);
853 timer = 5000;
854 break;
855 case 6:
856 me->SetStandState(UNIT_STAND_STATE_STAND);
857 me->SetOrientation(me->GetHomePosition().GetOrientation());
858 player->GroupEventHappens(QUEST_THE_DRAGON_QUEEN_ORACLE, me);
859 orphan->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
860 Reset();
861 return;
862 case 7:
863 orphan->GetMotionMaster()->MovePoint(0, me->GetPositionX() + std::cos(me->GetOrientation()) * 5, me->GetPositionY() + std::sin(me->GetOrientation()) * 5, me->GetPositionZ());
864 orphan->AI()->Talk(TEXT_WOLVAR_ORPHAN_11);
865 timer = 5000;
866 break;
867 case 8:
868 if (Creature* krasus = me->FindNearestCreature(NPC_KRASUS, 10.0f))
869 {
870 orphan->SetFacingToObject(krasus);
871 krasus->AI()->Talk(TEXT_KRASUS_8);
872 }
873 timer = 5000;
874 break;
875 case 9:
876 orphan->AI()->Talk(TEXT_WOLVAR_ORPHAN_12);
877 timer = 5000;
878 break;
879 case 10:
880 orphan->SetFacingToObject(me);
881 Talk(TEXT_ALEXSTRASZA_2);
882 timer = 5000;
883 break;
884 case 11:
885 orphan->AI()->Talk(TEXT_WOLVAR_ORPHAN_13);
886 timer = 5000;
887 break;
888 case 12:
889 player->GroupEventHappens(QUEST_THE_DRAGON_QUEEN_WOLVAR, me);
890 orphan->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
891 Reset();
892 return;
893 }
894 ++phase;
895 }
896 else
897 timer -= diff;
898 }

901 int8 phase;
902 uint32 timer;
903 ObjectGuid playerGUID;
904 ObjectGuid orphanGUID;
905

32enum WhisperwindLasher
33{
34 EVENT_CHECK_OOC = 1,
35 SPELL_INFECTED_WOULD = 52225,
36 SPELL_STAND = 37752,
37 NPC_WHISPERWIND_LASHER = 47747,
38 NPC_CORRUPTED_LASHER = 48387,
39 FACTION_HOSTILE = 14,
40 CHANCE_HOSTILE = 30
41};
42
43class npc_whisperwind_lasher : public CreatureScript
44{
45public:
46 npc_whisperwind_lasher() : CreatureScript("npc_whisperwind_lasher") { }
47
48 struct npc_whisperwind_lasherAI : public ScriptedAI
49 {
50 npc_whisperwind_lasherAI(Creature* creature) : ScriptedAI(creature) { }
51
52 void Reset() override
53 {
54 lasherClicked = false;
55 }
56
57 void OnSpellClick(Unit* clicker, bool& result) override
58 {
59 if (!result)
60 return;
61
62 if (roll_chance_i(CHANCE_HOSTILE))
63 {
64 me->CastSpell(me, SPELL_INFECTED_WOULD);
65 me->SetEntry(NPC_CORRUPTED_LASHER);
66 me->setFaction(FACTION_HOSTILE);
67 }
68 else
69 {
70 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
71 }
72
73 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_SPELLCLICK);
74 me->CastSpell(me, SPELL_STAND);
75 me->GetMotionMaster()->MoveRandom(8.0f);
76 events.ScheduleEvent(EVENT_CHECK_OOC, 20000);
77 lasherClicked = true;
78
79 if (Player* player = clicker->ToPlayer())
80 player->KilledMonsterCredit(NPC_WHISPERWIND_LASHER);
81 }
82
83 void UpdateAI(uint32 diff) override
84 {
85 if (!lasherClicked)
86 return;
87
88 events.Update(diff);
89
90 while (uint32 eventId = events.ExecuteEvent())
91 {
92 switch (eventId)
93 {
94 case EVENT_CHECK_OOC:
95 if (!me->IsInCombat())
96 me->DespawnOrUnsummon();
97 else
98 events.ScheduleEvent(EVENT_CHECK_OOC, 5000);
99 break;
100 }
101 }
102 DoMeleeAttackIfReady();
103 }
104
105 private:
106 EventMap events;
107 bool lasherClicked = false;
108 };
109
110 CreatureAI* GetAI(Creature* creature) const override
111 {
112 return new npc_whisperwind_lasherAI(creature);
113 }
114};

120enum SwipeHoney
121{
122 NPC_HONEY_BUNNY = 47308,
123 ITEM_HONEY_GLOB = 62820,
124 SPELL_BEES = 94064
125};
126
127class spell_swipe_honey : public SpellScriptLoader
128{
129public:
130 spell_swipe_honey() : SpellScriptLoader("spell_swipe_honey") { }
131
132 class spell_swipe_honey_SpellScript : public SpellScript
133 {
134 PrepareSpellScript(spell_swipe_honey_SpellScript);
135
136 SpellCastResult CheckTarget()
137 {
138 if (GetCaster()->FindNearestCreature(NPC_HONEY_BUNNY, 5.0f, true))
139 return SPELL_CAST_OK;
140 return SPELL_FAILED_OUT_OF_RANGE;
141 }
142
143 void HandleDummy(SpellEffIndex /* effIndex */)
144 {
145 if (Creature* honey = GetCaster()->FindNearestCreature(NPC_HONEY_BUNNY, 5.0f, true))
146 {
147 if (Player* player = GetCaster()->ToPlayer())
148 {
149 player->AddItem(ITEM_HONEY_GLOB, 1);
150
151 if (urand(1, 100) < 30)
152 honey->CastSpell(player, SPELL_BEES);
153 else
154 honey->DespawnOrUnsummon();
155 }
156 }
157 }
158
159 void Register() override
160 {
161 OnCheckCast += SpellCheckCastFn(spell_swipe_honey_SpellScript::CheckTarget);
162 OnEffectHitTarget += SpellEffectFn(spell_swipe_honey_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
163 }
164 };
165
166 SpellScript* GetSpellScript() const override
167 {
168 return new spell_swipe_honey_SpellScript();
169 }
170};

226Position const DrizzleSpawnPos = { 3852.52f, -1321.92f, 213.3353f, 5.72468f };
227Position const FerliSpawnPos = { 3850.44f, -1323.34f, 213.2113f, 5.637414f };
228
229class spell_ruumbos_silly_dance : public SpellScriptLoader
230{
231public: spell_ruumbos_silly_dance() : SpellScriptLoader("spell_ruumbos_silly_dance") { }
232
233 class spell_ruumbos_silly_dance_SpellScript : public SpellScript
234 {
235 PrepareSpellScript(spell_ruumbos_silly_dance_SpellScript);
236
237 void HandleScriptEffect(SpellEffIndex /* effIndex */)
238 {
239 if (Player* player = GetHitPlayer())
240 {
241 player->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_DANCE);
242
243 if (player->GetMap()->GetId() == MAP_KALIMDOR)
244 {
245 player->SummonCreature(NPC_DRIZZLE, DrizzleSpawnPos, TEMPSUMMON_TIMED_DESPAWN, 20000);
246 player->SummonCreature(NPC_FERLI, FerliSpawnPos, TEMPSUMMON_TIMED_DESPAWN, 20000);
247 }
248 }
249 }
250
251 void Register() override
252 {
253 OnEffectHitTarget += SpellEffectFn(spell_ruumbos_silly_dance_SpellScript::HandleScriptEffect, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
254 }
255 };
256
257 SpellScript* GetSpellScript() const override
258 {
259 return new spell_ruumbos_silly_dance_SpellScript();
260 }

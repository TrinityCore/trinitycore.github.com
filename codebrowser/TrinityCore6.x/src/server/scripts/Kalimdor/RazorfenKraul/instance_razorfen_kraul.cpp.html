26#include "ScriptMgr.h"
27#include "InstanceScript.h"
28#include "razorfen_kraul.h"
29#include "Player.h"
30
31#define WARD_KEEPERS_NR 2
32
33class instance_razorfen_kraul : public InstanceMapScript
34{
35public:
36 instance_razorfen_kraul() : InstanceMapScript("instance_razorfen_kraul", 47) { }
37
38 InstanceScript* GetInstanceScript(InstanceMap* map) const override
39 {
40 return new instance_razorfen_kraul_InstanceMapScript(map);
41 }
42
43 struct instance_razorfen_kraul_InstanceMapScript : public InstanceScript
44 {
45 instance_razorfen_kraul_InstanceMapScript(Map* map) : InstanceScript(map)
46 {
47 SetHeaders(DataHeader);
48 WardKeeperDeath = 0;
49 }
50
51 ObjectGuid DoorWardGUID;
52 int WardKeeperDeath;
53
54 Player* GetPlayerInMap()
55 {
56 Map::PlayerList const& players = instance->GetPlayers();
57 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
58 {
59 if (Player* player = itr->GetSource())
60 return player;
61 }
62 TC_LOG_DEBUG("scripts", "Instance Razorfen Kraul: GetPlayerInMap, but PlayerList is empty!");
63 return NULL;
64 }
65
66 void OnGameObjectCreate(GameObject* go) override
67 {
68 switch (go->GetEntry())
69 {
70 case 21099: DoorWardGUID = go->GetGUID(); break;
71 case 20920: go->SetUInt32Value(GAMEOBJECT_FACTION, 0); break; // big fat fugly hack
72 }
73 }
74
75 void Update(uint32 /*diff*/) override
76 {
77 if (WardKeeperDeath == WARD_KEEPERS_NR)
78 if (GameObject* go = instance->GetGameObject(DoorWardGUID))
79 {
80 go->SetUInt32Value(GAMEOBJECT_FLAGS, 33);
81 go->SetGoState(GO_STATE_ACTIVE);
82 }
83 }
84
85 void SetData(uint32 type, uint32 /*data*/) override
86 {
87 switch (type)
88 {
89 case EVENT_WARD_KEEPER: WardKeeperDeath++; break;
90 }
91 }
92
93 };
94
95};
96
97void AddSC_instance_razorfen_kraul()
98{
99 new instance_razorfen_kraul();
100}

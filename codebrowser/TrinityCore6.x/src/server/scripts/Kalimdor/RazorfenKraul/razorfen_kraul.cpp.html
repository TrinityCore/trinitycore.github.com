54 void sQuestAccept(Player* player, Quest const* quest) override
55 {
56 if (quest->GetQuestId() == QUEST_WILLIX_THE_IMPORTER)
57 {
58 Start(true, false, player->GetGUID());

64 void WaypointReached(uint32 waypointId) override
65 {
66 Player* player = GetPlayerForEscort();
67 if (!player)
68 return;
69
70 switch (waypointId)
71 {
72 case 3:
73 me->HandleEmoteCommand(EMOTE_STATE_POINT);
74 Talk(SAY_POINT, player);
75 break;
76 case 4:
77 me->SummonCreature(ENTRY_BOAR, 2137.66f, 1843.98f, 48.08f, 1.54f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
78 break;
79 case 8:
80 Talk(SAY_BLUELEAF, player);
81 break;
82 case 9:
83 Talk(SAY_DANGER, player);
84 break;
85 case 13:
86 Talk(SAY_BAD, player);
87 break;
88 case 14:
89 me->SummonCreature(ENTRY_BOAR, 2078.91f, 1704.54f, 56.77f, 1.54f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
90 break;
91 case 25:
92 Talk(SAY_THINK, player);
93 break;
94 case 31:
95 Talk(SAY_SOON, player);
96 break;
97 case 42:
98 Talk(SAY_FINALY, player);
99 break;
100 case 43:
101 me->SummonCreature(ENTRY_BOAR, 1956.43f, 1596.97f, 81.75f, 1.54f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
102 break;
103 case 45:
104 Talk(SAY_WIN, player);
105 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
106 player->GroupEventHappens(QUEST_WILLIX_THE_IMPORTER, me);
107 break;
108 case 46:
109 Talk(SAY_END, player);
110 break;
111 }
112 }

157 struct npc_snufflenose_gopherAI : public PetAI
158 {
159 npc_snufflenose_gopherAI(Creature* creature) : PetAI(creature)
160 {
161 IsMovementActive = false;
162 }
163
164 void Reset() override
165 {
166 IsMovementActive = false;
167 TargetTubberGUID.Clear();
168 }
169
170 void MovementInform(uint32 type, uint32 id) override
171 {
172 if (type == POINT_MOTION_TYPE && id == POINT_TUBBER)
173 {
174 if (GameObject* go = me->GetMap()->GetGameObject(TargetTubberGUID))
175 {
176 go->SetRespawnTime(5 * MINUTE);
177 go->Refresh();
178 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_INTERACT_COND);
179 }
180
181 IsMovementActive = false;
182 }
183 else
184 PetAI::MovementInform(type, id);
185 }
186
187 void DoFindNewTubber()
188 {
189 std::list<GameObject*> tubbersInRange;
190 GetGameObjectListWithEntryInGrid(tubbersInRange, me, GO_BLUELEAF_TUBBER, 40.0f);
191
192 if (tubbersInRange.empty())
193 return;
194
195 tubbersInRange.remove_if([](GameObject* go)
196 {
197 return go->isSpawned() || !go->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_INTERACT_COND);
198 });
199
200 tubbersInRange.sort(Trinity::ObjectDistanceOrderPred(me));
201
202 GameObject* nearestTubber = tubbersInRange.front();
203 if (!nearestTubber)
204 return;
205
206 TargetTubberGUID = nearestTubber->GetGUID();
207
208 // XFurry was wrong...
209 me->GetMotionMaster()->MovePoint(POINT_TUBBER, nearestTubber->GetPositionX(),
210 nearestTubber->GetPositionY(),
211 nearestTubber->GetPositionZ());
212 IsMovementActive = true;
213 }
214
215 void UpdateAI(uint32 diff) override
216 {
217 if (!IsMovementActive)
218 PetAI::UpdateAI(diff);
219 }
220
221 void DoAction(int32 action) override
222 {
223 if (action == ACTION_FIND_NEW_TUBBER)
224 DoFindNewTubber();
225 }
226
227
228 bool IsMovementActive;
229 ObjectGuid TargetTubberGUID;
230 };
231};

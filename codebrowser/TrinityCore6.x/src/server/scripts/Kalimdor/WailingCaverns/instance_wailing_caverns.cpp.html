26#include "ScriptMgr.h"
27#include "InstanceScript.h"
28#include "wailing_caverns.h"
29
30#define MAX_ENCOUNTER 9
31
32class instance_wailing_caverns : public InstanceMapScript
33{
34public:
35 instance_wailing_caverns() : InstanceMapScript("instance_wailing_caverns", 43) { }
36
37 InstanceScript* GetInstanceScript(InstanceMap* map) const override
38 {
39 return new instance_wailing_caverns_InstanceMapScript(map);
40 }
41
42 struct instance_wailing_caverns_InstanceMapScript : public InstanceScript
43 {
44 instance_wailing_caverns_InstanceMapScript(Map* map) : InstanceScript(map)
45 {
46 SetHeaders(DataHeader);
47 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
48
49 yelled = false;
50 }
51
52 uint32 m_auiEncounter[MAX_ENCOUNTER];
53
54 bool yelled;
55 ObjectGuid NaralexGUID;
56
57 void OnCreatureCreate(Creature* creature) override
58 {
59 if (creature->GetEntry() == DATA_NARALEX)
60 NaralexGUID = creature->GetGUID();
61 }
62
63 void SetData(uint32 type, uint32 data) override
64 {
65 switch (type)
66 {
67 case TYPE_LORD_COBRAHN: m_auiEncounter[0] = data;break;
68 case TYPE_LORD_PYTHAS: m_auiEncounter[1] = data;break;
69 case TYPE_LADY_ANACONDRA: m_auiEncounter[2] = data;break;
70 case TYPE_LORD_SERPENTIS: m_auiEncounter[3] = data;break;
71 case TYPE_NARALEX_EVENT: m_auiEncounter[4] = data;break;
72 case TYPE_NARALEX_PART1: m_auiEncounter[5] = data;break;
73 case TYPE_NARALEX_PART2: m_auiEncounter[6] = data;break;
74 case TYPE_NARALEX_PART3: m_auiEncounter[7] = data;break;
75 case TYPE_MUTANUS_THE_DEVOURER: m_auiEncounter[8] = data;break;
76 case TYPE_NARALEX_YELLED: yelled = true; break;
77 }
78 if (data == DONE)SaveToDB();
79 }
80
81 uint32 GetData(uint32 type) const override
82 {
83 switch (type)
84 {
85 case TYPE_LORD_COBRAHN: return m_auiEncounter[0];
86 case TYPE_LORD_PYTHAS: return m_auiEncounter[1];
87 case TYPE_LADY_ANACONDRA: return m_auiEncounter[2];
88 case TYPE_LORD_SERPENTIS: return m_auiEncounter[3];
89 case TYPE_NARALEX_EVENT: return m_auiEncounter[4];
90 case TYPE_NARALEX_PART1: return m_auiEncounter[5];
91 case TYPE_NARALEX_PART2: return m_auiEncounter[6];
92 case TYPE_NARALEX_PART3: return m_auiEncounter[7];
93 case TYPE_MUTANUS_THE_DEVOURER: return m_auiEncounter[8];
94 case TYPE_NARALEX_YELLED: return yelled;
95 }
96 return 0;
97 }
98
99 ObjectGuid GetGuidData(uint32 data) const override
100 {
101 if (data == DATA_NARALEX)return NaralexGUID;
102 return ObjectGuid::Empty;
103 }
104
105 std::string GetSaveData() override
106 {
107 OUT_SAVE_INST_DATA;
108
109 std::ostringstream saveStream;
110 saveStream << m_auiEncounter[0] << ' ' << m_auiEncounter[1] << ' ' << m_auiEncounter[2] << ' '
111 << m_auiEncounter[3] << ' ' << m_auiEncounter[4] << ' ' << m_auiEncounter[5] << ' '
112 << m_auiEncounter[6] << ' ' << m_auiEncounter[7] << ' ' << m_auiEncounter[8];
113
114 OUT_SAVE_INST_DATA_COMPLETE;
115 return saveStream.str();
116 }
117
118 void Load(const char* in) override
119 {
120 if (!in)
121 {
122 OUT_LOAD_INST_DATA_FAIL;
123 return;
124 }
125
126 OUT_LOAD_INST_DATA(in);
127
128 std::istringstream loadStream(in);
129 loadStream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3]
130 >> m_auiEncounter[4] >> m_auiEncounter[5] >> m_auiEncounter[6] >> m_auiEncounter[7] >> m_auiEncounter[8];
131
132 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
133 if (m_auiEncounter[i] != DONE)
134 m_auiEncounter[i] = NOT_STARTED;
135
136 OUT_LOAD_INST_DATA_COMPLETE;
137 }
138
139 };
140
141};
142
143void AddSC_instance_wailing_caverns()
144{
145 new instance_wailing_caverns();
146}

59class boss_rajaxx : public CreatureScript
60{
61 public:
62 boss_rajaxx() : CreatureScript("boss_rajaxx") { }
63
64 struct boss_rajaxxAI : public BossAI
65 {
66 boss_rajaxxAI(Creature* creature) : BossAI(creature, DATA_RAJAXX)
67 {
68 Initialize();
69 }
70
71 void Initialize()
72 {
73 enraged = false;
74 }
75
76 void Reset() override
77 {
78 _Reset();
79 Initialize();
80 events.ScheduleEvent(EVENT_DISARM, 10000);
81 events.ScheduleEvent(EVENT_THUNDERCRASH, 12000);
82 }
83
84 void JustDied(Unit* /*killer*/) override
85 {
86 //SAY_DEATH
87 _JustDied();
88 }
89
90 void EnterCombat(Unit* /*victim*/) override
91 {
92 _EnterCombat();
93 }
94
95 void UpdateAI(uint32 diff) override
96 {
97 if (!UpdateVictim())
98 return;
99
100 events.Update(diff);
101
102 if (me->HasUnitState(UNIT_STATE_CASTING))
103 return;
104
105 while (uint32 eventId = events.ExecuteEvent())
106 {
107 switch (eventId)
108 {
109 case EVENT_DISARM:
110 DoCastVictim(SPELL_DISARM);
111 events.ScheduleEvent(EVENT_DISARM, 22000);
112 break;
113 case EVENT_THUNDERCRASH:
114 DoCast(me, SPELL_THUNDERCRASH);
115 events.ScheduleEvent(EVENT_THUNDERCRASH, 21000);
116 break;
117 default:
118 break;
119 }
120 }
121
122 DoMeleeAttackIfReady();
123 }
124 private:
125 bool enraged;
126 };
127
128 CreatureAI* GetAI(Creature* creature) const override
129 {
130 return new boss_rajaxxAI(creature);
131 }
132};

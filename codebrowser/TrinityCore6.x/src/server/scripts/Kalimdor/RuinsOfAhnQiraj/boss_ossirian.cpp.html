77float RoomRadius = 165.0f;
78uint8 const NUM_TORNADOS = 5; /// @todo This number is completly random!
79uint8 const NUM_WEAKNESS = 5;
80uint32 const SpellWeakness[NUM_WEAKNESS] = { 25177, 25178, 25180, 25181, 25183 };
81Position const RoomCenter = { -9343.041992f, 1923.278198f, 85.555984f, 0.0 };

83class boss_ossirian : public CreatureScript
84{
85 public:
86 boss_ossirian() : CreatureScript("boss_ossirian") { }
87
88 struct boss_ossirianAI : public BossAI
89 {
90 boss_ossirianAI(Creature* creature) : BossAI(creature, DATA_OSSIRIAN)
91 {
92 Initialize();
93 SaidIntro = false;
94 }
95
96 void Initialize()
97 {
98 CrystalIterator = 0;
99 TriggerGUID.Clear();
100 CrystalGUID.Clear();
101 }
102
103 ObjectGuid TriggerGUID;
104 ObjectGuid CrystalGUID;
105 uint8 CrystalIterator;
106 bool SaidIntro;
107
108 void Reset() override
109 {
110 _Reset();
111 Initialize();
112 }
113
114 void SpellHit(Unit* caster, SpellInfo const* spell) override
115 {
116 for (uint8 i = 0; i < NUM_WEAKNESS; ++i)
117 {
118 if (spell->Id == SpellWeakness[i])
119 {
120 me->RemoveAurasDueToSpell(SPELL_SUPREME);
121 ((TempSummon*)caster)->UnSummon();
122 SpawnNextCrystal();
123 }
124 }
125 }
126
127 void DoAction(int32 action) override
128 {
129 if (action == ACTION_TRIGGER_WEAKNESS)
130 if (Creature* Trigger = me->GetMap()->GetCreature(TriggerGUID))
131 if (!Trigger->HasUnitState(UNIT_STATE_CASTING))
132 Trigger->CastSpell(Trigger, SpellWeakness[urand(0, 4)], false);
133 }
134
135 void EnterCombat(Unit* /*who*/) override
136 {
137 _EnterCombat();
138 events.Reset();
139 events.ScheduleEvent(EVENT_SILENCE, 30000);
140 events.ScheduleEvent(EVENT_CYCLONE, 20000);
141 events.ScheduleEvent(EVENT_STOMP, 30000);
142
143 DoCast(me, SPELL_SUPREME);
144 Talk(SAY_AGGRO);
145
146 Map* map = me->GetMap();
147
148 WorldPackets::Misc::Weather weather(WEATHER_STATE_HEAVY_SANDSTORM, 1.0f);
149 map->SendToPlayers(weather.Write());
150
151 for (uint8 i = 0; i < NUM_TORNADOS; ++i)
152 {
153 Position Point = me->GetRandomPoint(RoomCenter, RoomRadius);
154 if (Creature* Tornado = map->SummonCreature(NPC_SAND_VORTEX, Point))
155 Tornado->CastSpell(Tornado, SPELL_SAND_STORM, true);
156 }
157
158 SpawnNextCrystal();
159 }
160
161 void KilledUnit(Unit* /*victim*/) override
162 {
163 Talk(SAY_SLAY);
164 }
165
166 void EnterEvadeMode() override
167 {
168 Cleanup();
169 summons.DespawnAll();
170 BossAI::EnterEvadeMode();
171 }
172
173 void JustDied(Unit* /*killer*/) override
174 {
175 Cleanup();
176 _JustDied();
177 }
178
179 void Cleanup()
180 {
181 if (GameObject* Crystal = me->GetMap()->GetGameObject(CrystalGUID))
182 Crystal->Use(me);
183 }
184
185 void SpawnNextCrystal()
186 {
187 if (CrystalIterator == NUM_CRYSTALS)
188 CrystalIterator = 0;
189
190 if (Creature* Trigger = me->GetMap()->SummonCreature(NPC_OSSIRIAN_TRIGGER, CrystalCoordinates[CrystalIterator]))
191 {
192 TriggerGUID = Trigger->GetGUID();
193 if (GameObject* Crystal = Trigger->SummonGameObject(GO_OSSIRIAN_CRYSTAL,
194 CrystalCoordinates[CrystalIterator].GetPositionX(),
195 CrystalCoordinates[CrystalIterator].GetPositionY(),
196 CrystalCoordinates[CrystalIterator].GetPositionZ(),
197 0, 0, 0, 0, 0, uint32(-1)))
198 {
199 CrystalGUID = Crystal->GetGUID();
200 ++CrystalIterator;
201 }
202 }
203 }
204
205 void MoveInLineOfSight(Unit* who) override
206
207 {
208 if (!SaidIntro)
209 {
210 Talk(SAY_INTRO);
211 SaidIntro = true;
212 }
213 BossAI::MoveInLineOfSight(who);
214 }
215
216 void UpdateAI(uint32 diff) override
217 {
218 if (!UpdateVictim())
219 return;
220
221 events.Update(diff);
222
223 // No kiting!
224 if (me->GetDistance(me->GetVictim()) > 60.00f && me->GetDistance(me->GetVictim()) < 120.00f)
225 DoCastVictim(SPELL_SUMMON);
226
227 bool ApplySupreme = true;
228
229 if (me->HasAura(SPELL_SUPREME))
230 ApplySupreme = false;
231 else
232 {
233 for (uint8 i = 0; i < NUM_WEAKNESS; ++i)
234 {
235 if (me->HasAura(SpellWeakness[i]))
236 {
237 ApplySupreme = false;
238 break;
239 }
240 }
241 }
242
243 if (ApplySupreme)
244 {
245 DoCast(me, SPELL_SUPREME);
246 Talk(SAY_SUPREME);
247 }
248
249 while (uint32 eventId = events.ExecuteEvent())
250 {
251 switch (eventId)
252 {
253 case EVENT_SILENCE:
254 DoCast(me, SPELL_SILENCE);
255 events.ScheduleEvent(EVENT_SILENCE, urand(20000, 30000));
256 break;
257 case EVENT_CYCLONE:
258 DoCastVictim(SPELL_CYCLONE);
259 events.ScheduleEvent(EVENT_CYCLONE, 20000);
260 break;
261 case EVENT_STOMP:
262 DoCast(me, SPELL_STOMP);
263 events.ScheduleEvent(EVENT_STOMP, 30000);
264 break;
265 default:
266 break;
267 }
268 }
269
270 DoMeleeAttackIfReady();
271 }
272 };
273
274 CreatureAI* GetAI(Creature* creature) const override
275 {
276 return GetInstanceAI<boss_ossirianAI>(creature);
277 }
278};

280class go_ossirian_crystal : public GameObjectScript
281{
282 public:
283 go_ossirian_crystal() : GameObjectScript("go_ossirian_crystal") { }
284
285 bool OnGossipHello(Player* player, GameObject* /*go*/) override
286 {
287 InstanceScript* Instance = player->GetInstanceScript();
288 if (!Instance)
289 return false;
290
291 Creature* Ossirian = player->FindNearestCreature(NPC_OSSIRIAN, 30.0f);
292 if (!Ossirian || Instance->GetBossState(DATA_OSSIRIAN) != IN_PROGRESS)
293 return false;
294
295 Ossirian->AI()->DoAction(ACTION_TRIGGER_WEAKNESS);
296 return true;
297 }
298};

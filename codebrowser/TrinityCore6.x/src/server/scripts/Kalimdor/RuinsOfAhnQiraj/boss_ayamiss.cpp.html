74class boss_ayamiss : public CreatureScript
75{
76 public:
77 boss_ayamiss() : CreatureScript("boss_ayamiss") { }
78
79 struct boss_ayamissAI : public BossAI
80 {
81 boss_ayamissAI(Creature* creature) : BossAI(creature, DATA_AYAMISS)
82 {
83 Initialize();
84 }
85
86 void Initialize()
87 {
88 _phase = PHASE_AIR;
89 _enraged = false;
90 }
91
92 void Reset() override
93 {
94 _Reset();
95 Initialize();
96 SetCombatMovement(false);
97 }
98
99 void JustSummoned(Creature* who) override
100 {
101 switch (who->GetEntry())
102 {
103 case NPC_SWARMER:
104 _swarmers.push_back(who->GetGUID());
105 break;
106 case NPC_LARVA:
107 who->GetMotionMaster()->MovePoint(POINT_PARALYZE, AltarPos);
108 break;
109 case NPC_HORNET:
110 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
111 who->AI()->AttackStart(target);
112 break;
113 }
114 }
115
116 void MovementInform(uint32 type, uint32 id) override
117 {
118 if (type == POINT_MOTION_TYPE)
119 {
120 switch (id)
121 {
122 case POINT_AIR:
123 me->AddUnitState(UNIT_STATE_ROOT);
124 break;
125 case POINT_GROUND:
126 me->ClearUnitState(UNIT_STATE_ROOT);
127 break;
128 }
129 }
130 }
131
132 void EnterEvadeMode() override
133 {
134 me->ClearUnitState(UNIT_STATE_ROOT);
135 BossAI::EnterEvadeMode();
136 }
137
138 void EnterCombat(Unit* attacker) override
139 {
140 BossAI::EnterCombat(attacker);
141
142 events.ScheduleEvent(EVENT_STINGER_SPRAY, urand(20000, 30000));
143 events.ScheduleEvent(EVENT_POISON_STINGER, 5000);
144 events.ScheduleEvent(EVENT_SUMMON_SWARMER, 5000);
145 events.ScheduleEvent(EVENT_SWARMER_ATTACK, 60000);
146 events.ScheduleEvent(EVENT_PARALYZE, 15000);
147
148 me->SetCanFly(true);
149 me->SetDisableGravity(true);
150 me->GetMotionMaster()->MovePoint(POINT_AIR, AyamissAirPos);
151 }
152
153 void UpdateAI(uint32 diff) override
154 {
155 if (!UpdateVictim())
156 return;
157
158 events.Update(diff);
159
160 if (_phase == PHASE_AIR && me->GetHealthPct() < 70.0f)
161 {
162 _phase = PHASE_GROUND;
163 SetCombatMovement(true);
164 me->SetCanFly(false);
165 if (me->GetVictim())
166 {
167 Position VictimPos = me->EnsureVictim()->GetPosition();
168 me->GetMotionMaster()->MovePoint(POINT_GROUND, VictimPos);
169 }
170 DoResetThreat();
171 events.ScheduleEvent(EVENT_LASH, urand(5000, 8000));
172 events.ScheduleEvent(EVENT_TRASH, urand(3000, 6000));
173 events.CancelEvent(EVENT_POISON_STINGER);
174 }
175 else
176 {
177 DoMeleeAttackIfReady();
178 }
179
180 if (!_enraged && me->GetHealthPct() < 20.0f)
181 {
182 DoCast(me, SPELL_FRENZY);
183 Talk(EMOTE_FRENZY);
184 _enraged = true;
185 }
186
187 while (uint32 eventId = events.ExecuteEvent())
188 {
189 switch (eventId)
190 {
191 case EVENT_STINGER_SPRAY:
192 DoCast(me, SPELL_STINGER_SPRAY);
193 events.ScheduleEvent(EVENT_STINGER_SPRAY, urand(15000, 20000));
194 break;
195 case EVENT_POISON_STINGER:
196 DoCastVictim(SPELL_POISON_STINGER);
197 events.ScheduleEvent(EVENT_POISON_STINGER, urand(2000, 3000));
198 break;
199 case EVENT_PARALYZE:
200 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0, true))
201 {
202 DoCast(target, SPELL_PARALYZE);
203 instance->SetGuidData(DATA_PARALYZED, target->GetGUID());
204 uint8 Index = urand(0, 1);
205 me->SummonCreature(NPC_LARVA, LarvaPos[Index], TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 30000);
206 }
207 events.ScheduleEvent(EVENT_PARALYZE, 15000);
208 break;
209 case EVENT_SWARMER_ATTACK:
210 for (GuidList::iterator i = _swarmers.begin(); i != _swarmers.end(); ++i)
211 if (Creature* swarmer = me->GetMap()->GetCreature(*i))
212 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
213 swarmer->AI()->AttackStart(target);
214
215 _swarmers.clear();
216 events.ScheduleEvent(EVENT_SWARMER_ATTACK, 60000);
217 break;
218 case EVENT_SUMMON_SWARMER:
219 {
220 Position Pos = me->GetRandomPoint(SwarmerPos, 80.0f);
221 me->SummonCreature(NPC_SWARMER, Pos);
222 events.ScheduleEvent(EVENT_SUMMON_SWARMER, 5000);
223 break;
224 }
225 case EVENT_TRASH:
226 DoCastVictim(SPELL_TRASH);
227 events.ScheduleEvent(EVENT_TRASH, urand(5000, 7000));
228 break;
229 case EVENT_LASH:
230 DoCastVictim(SPELL_LASH);
231 events.ScheduleEvent(EVENT_LASH, urand(8000, 15000));
232 break;
233 }
234 }
235 }
236 private:
237 GuidList _swarmers;
238 uint8 _phase;
239 bool _enraged;
240 };
241
242 CreatureAI* GetAI(Creature* creature) const override
243 {
244 return GetInstanceAI<boss_ayamissAI>(creature);
245 }

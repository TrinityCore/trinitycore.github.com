44#define GOSSIP_HCB "I know this is rather silly but a young ward who is a bit shy would like your hoofprint."
45/// @todo verify abilities/timers
46class npc_cairne_bloodhoof : public CreatureScript
47{
48public:
49 npc_cairne_bloodhoof() : CreatureScript("npc_cairne_bloodhoof") { }
50
51 CreatureAI* GetAI(Creature* creature) const override
52 {
53 return new npc_cairne_bloodhoofAI (creature);
54 }
55
56 struct npc_cairne_bloodhoofAI : public ScriptedAI
57 {
58 npc_cairne_bloodhoofAI(Creature* creature) : ScriptedAI(creature)
59 {
60 Initialize();
61 }
62
63 void Initialize()
64 {
65 BerserkerChargeTimer = 30000;
66 CleaveTimer = 5000;
67 MortalStrikeTimer = 10000;
68 ThunderclapTimer = 15000;
69 UppercutTimer = 10000;
70 }
71
72 uint32 BerserkerChargeTimer;
73 uint32 CleaveTimer;
74 uint32 MortalStrikeTimer;
75 uint32 ThunderclapTimer;
76 uint32 UppercutTimer;
77
78 void Reset() override
79 {
80 Initialize();
81 }
82
83 void EnterCombat(Unit* /*who*/) override { }
84
85 void UpdateAI(uint32 diff) override
86 {
87 if (!UpdateVictim())
88 return;
89
90 if (BerserkerChargeTimer <= diff)
91 {
92 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
93 DoCast(target, SPELL_BERSERKER_CHARGE);
94 BerserkerChargeTimer = 25000;
95 } else BerserkerChargeTimer -= diff;
96
97 if (UppercutTimer <= diff)
98 {
99 DoCastVictim(SPELL_UPPERCUT);
100 UppercutTimer = 20000;
101 } else UppercutTimer -= diff;
102
103 if (ThunderclapTimer <= diff)
104 {
105 DoCastVictim(SPELL_THUNDERCLAP);
106 ThunderclapTimer = 15000;
107 } else ThunderclapTimer -= diff;
108
109 if (MortalStrikeTimer <= diff)
110 {
111 DoCastVictim(SPELL_MORTAL_STRIKE);
112 MortalStrikeTimer = 15000;
113 } else MortalStrikeTimer -= diff;
114
115 if (CleaveTimer <= diff)
116 {
117 DoCastVictim(SPELL_CLEAVE);
118 CleaveTimer = 7000;
119 } else CleaveTimer -= diff;
120
121 DoMeleeAttackIfReady();
122 }
123 };
124
125};
126
127void AddSC_thunder_bluff()
128{
129 new npc_cairne_bloodhoof();
130}

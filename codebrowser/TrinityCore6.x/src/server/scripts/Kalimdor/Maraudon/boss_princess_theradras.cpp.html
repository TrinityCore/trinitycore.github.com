30enum Spells
31{
32 SPELL_DUSTFIELD = 21909,
33 SPELL_BOULDER = 21832,
34 SPELL_THRASH = 3391,
35 SPELL_REPULSIVEGAZE = 21869
36};
37
38class boss_princess_theradras : public CreatureScript
39{
40public:
41 boss_princess_theradras() : CreatureScript("boss_princess_theradras") { }
42
43 CreatureAI* GetAI(Creature* creature) const override
44 {
45 return new boss_ptheradrasAI(creature);
46 }
47
48 struct boss_ptheradrasAI : public ScriptedAI
49 {
50 boss_ptheradrasAI(Creature* creature) : ScriptedAI(creature)
51 {
52 Initialize();
53 }
54
55 void Initialize()
56 {
57 DustfieldTimer = 8000;
58 BoulderTimer = 2000;
59 ThrashTimer = 5000;
60 RepulsiveGazeTimer = 23000;
61 }
62
63 uint32 DustfieldTimer;
64 uint32 BoulderTimer;
65 uint32 ThrashTimer;
66 uint32 RepulsiveGazeTimer;
67
68 void Reset() override
69 {
70 Initialize();
71 }
72
73 void EnterCombat(Unit* /*who*/) override { }
74
75 void JustDied(Unit* /*killer*/) override
76 {
77 me->SummonCreature(12238, 28.067f, 61.875f, -123.405f, 4.67f, TEMPSUMMON_TIMED_DESPAWN, 600000);
78 }
79
80 void UpdateAI(uint32 diff) override
81 {
82 if (!UpdateVictim())
83 return;
84
85 //DustfieldTimer
86 if (DustfieldTimer <= diff)
87 {
88 DoCast(me, SPELL_DUSTFIELD);
89 DustfieldTimer = 14000;
90 }
91 else DustfieldTimer -= diff;
92
93 //BoulderTimer
94 if (BoulderTimer <= diff)
95 {
96 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
97 DoCast(target, SPELL_BOULDER);
98 BoulderTimer = 10000;
99 }
100 else BoulderTimer -= diff;
101
102 //RepulsiveGazeTimer
103 if (RepulsiveGazeTimer <= diff)
104 {
105 DoCastVictim(SPELL_REPULSIVEGAZE);
106 RepulsiveGazeTimer = 20000;
107 }
108 else RepulsiveGazeTimer -= diff;
109
110 //ThrashTimer
111 if (ThrashTimer <= diff)
112 {
113 DoCast(me, SPELL_THRASH);
114 ThrashTimer = 18000;
115 }
116 else ThrashTimer -= diff;
117
118 DoMeleeAttackIfReady();
119 }
120 };
121};
122
123void AddSC_boss_ptheradras()
124{
125 new boss_princess_theradras();
126}

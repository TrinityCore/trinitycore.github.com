29enum Spells
30{
31 SPELL_TOXICVOLLEY = 21687,
32 SPELL_UPPERCUT = 22916
33};
34
35class boss_noxxion : public CreatureScript
36{
37public:
38 boss_noxxion() : CreatureScript("boss_noxxion") { }
39
40 CreatureAI* GetAI(Creature* creature) const override
41 {
42 return new boss_noxxionAI(creature);
43 }
44
45 struct boss_noxxionAI : public ScriptedAI
46 {
47 boss_noxxionAI(Creature* creature) : ScriptedAI(creature)
48 {
49 Initialize();
50 }
51
52 void Initialize()
53 {
54 ToxicVolleyTimer = 7000;
55 UppercutTimer = 16000;
56 AddsTimer = 19000;
57 InvisibleTimer = 15000; //Too much too low?
58 Invisible = false;
59 }
60
61 uint32 ToxicVolleyTimer;
62 uint32 UppercutTimer;
63 uint32 AddsTimer;
64 uint32 InvisibleTimer;
65 bool Invisible;
66
67 void Reset() override
68 {
69 Initialize();
70 }
71
72 void EnterCombat(Unit* /*who*/) override { }
73
74 void SummonAdds(Unit* victim)
75 {
76 if (Creature* Add = DoSpawnCreature(13456, float(irand(-7, 7)), float(irand(-7, 7)), 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 90000))
77 Add->AI()->AttackStart(victim);
78 }
79
80 void UpdateAI(uint32 diff) override
81 {
82 if (Invisible && InvisibleTimer <= diff)
83 {
84 //Become visible again
85 me->setFaction(14);
86 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
87 //Noxxion model
88 me->SetDisplayId(11172);
89 Invisible = false;
90 //me->m_canMove = true;
91 }
92 else if (Invisible)
93 {
94 InvisibleTimer -= diff;
95 //Do nothing while invisible
96 return;
97 }
98
99 //Return since we have no target
100 if (!UpdateVictim())
101 return;
102
103 //ToxicVolleyTimer
104 if (ToxicVolleyTimer <= diff)
105 {
106 DoCastVictim(SPELL_TOXICVOLLEY);
107 ToxicVolleyTimer = 9000;
108 }
109 else ToxicVolleyTimer -= diff;
110
111 //UppercutTimer
112 if (UppercutTimer <= diff)
113 {
114 DoCastVictim(SPELL_UPPERCUT);
115 UppercutTimer = 12000;
116 }
117 else UppercutTimer -= diff;
118
119 //AddsTimer
120 if (!Invisible && AddsTimer <= diff)
121 {
122 //Interrupt any spell casting
123 //me->m_canMove = true;
124 me->InterruptNonMeleeSpells(false);
125 me->setFaction(35);
126 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
127 // Invisible Model
128 me->SetDisplayId(11686);
129 SummonAdds(me->GetVictim());
130 SummonAdds(me->GetVictim());
131 SummonAdds(me->GetVictim());
132 SummonAdds(me->GetVictim());
133 SummonAdds(me->GetVictim());
134 Invisible = true;
135 InvisibleTimer = 15000;
136
137 AddsTimer = 40000;
138 }
139 else AddsTimer -= diff;
140
141 DoMeleeAttackIfReady();
142 }
143 };
144};
145

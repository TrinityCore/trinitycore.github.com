18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "blackfathom_deeps.h"
21
22enum Spells
23{
24 SPELL_NET = 6533
25};
26
27enum Events
28{
29 EVENT_THROW_NET = 1
30};
31
32class boss_gelihast : public CreatureScript
33{
34public:
35 boss_gelihast() : CreatureScript("boss_gelihast") { }
36
37 struct boss_gelihastAI : public BossAI
38 {
39 boss_gelihastAI(Creature* creature) : BossAI(creature, DATA_GELIHAST) { }
40
41 void EnterCombat(Unit* /*who*/) override
42 {
43 _EnterCombat();
44 events.ScheduleEvent(EVENT_THROW_NET, urand(2000, 4000));
45 }
46
47 void UpdateAI(uint32 diff) override
48 {
49 if (!UpdateVictim())
50 return;
51
52 events.Update(diff);
53
54 while (uint32 eventId = events.ExecuteEvent())
55 {
56 if (eventId == EVENT_THROW_NET)
57 {
58 DoCastVictim(SPELL_NET);
59 events.ScheduleEvent(EVENT_THROW_NET, urand(4000, 7000));
60 }
61 }
62
63 DoMeleeAttackIfReady();
64 }
65 };
66
67 CreatureAI* GetAI(Creature* creature) const override
68 {
69 return GetInstanceAI<boss_gelihastAI>(creature);
70 }
71};
72
73void AddSC_boss_gelihast()
74{
75 new boss_gelihast();
76}

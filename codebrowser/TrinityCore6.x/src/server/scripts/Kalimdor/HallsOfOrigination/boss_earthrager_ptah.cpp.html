74class SummonScarab : public BasicEvent
75{
76public:
77 SummonScarab(Unit* owner, InstanceScript* instance) : _owner(owner), _instance(instance) { }
78
79 bool Execute(uint64 /*execTime*/, uint32 /*diff*/) override
80 {
81 if (!_instance || _instance->GetBossState(DATA_EARTHRAGER_PTAH) != IN_PROGRESS)
82 return true; // delete event
83
84 _owner->CastSpell(_owner, SPELL_SUMMON_JEWELED_SCARAB);
85 _owner->RemoveAurasDueToSpell(SPELL_BEETLE_BURROW);
86 return true;
87 }
88protected:
89 Unit* _owner;
90 InstanceScript* _instance;
91};

93class boss_earthrager_ptah : public CreatureScript
94{
95public:
96 boss_earthrager_ptah() : CreatureScript("boss_earthrager_ptah") { }
97
98 struct boss_earthrager_ptahAI : public BossAI
99 {
100 boss_earthrager_ptahAI(Creature* creature) : BossAI(creature, DATA_EARTHRAGER_PTAH), _summonDeaths(0), _hasDispersed(false) { }
101
102 void Cleanup()
103 {
104 std::list<Creature*> units;
105
106 GetCreatureListWithEntryInGrid(units, me, NPC_DUSTBONE_HORROR, 100.0f);
107 for (std::list<Creature*>::iterator itr = units.begin(); itr != units.end(); ++itr)
108 (*itr)->DespawnOrUnsummon();
109
110 GetCreatureListWithEntryInGrid(units, me, NPC_JEWELED_SCARAB, 100.0f);
111 for (std::list<Creature*>::iterator itr = units.begin(); itr != units.end(); ++itr)
112 (*itr)->DespawnOrUnsummon();
113 }
114
115 void Reset() override
116 {
117 _summonDeaths = 0;
118 _hasDispersed = false;
119 Cleanup();
120 _Reset();
121 events.SetPhase(PHASE_NORMAL);
122 events.ScheduleEvent(EVENT_RAGING_SMASH, urand(7000, 12000), 0, PHASE_NORMAL);
123 events.ScheduleEvent(EVENT_FLAME_BOLT, 15000, 0, PHASE_NORMAL);
124 events.ScheduleEvent(EVENT_EARTH_SPIKE, urand(16000, 21000), 0, PHASE_NORMAL);
125 }
126
127 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
128 {
129 if (me->HealthBelowPctDamaged(50, damage) && (events.GetPhaseMask() & PHASE_MASK_NORMAL) && !_hasDispersed)
130 {
131 events.SetPhase(PHASE_DISPERSE);
132 _hasDispersed = true;
133
134 me->AttackStop();
135 DoCast(me, SPELL_SANDSTORM);
136 me->GetMap()->SetZoneWeather(AREA_TOMB_OF_THE_EARTHRAGER, WEATHER_STATE_LIGHT_SANDSTORM, 1.0f);
137 events.ScheduleEvent(EVENT_PTAH_EXPLODE, 6000, 0, PHASE_DISPERSE);
138 events.ScheduleEvent(EVENT_QUICKSAND, 10000, 0, PHASE_DISPERSE);
139
140 std::list<Creature*> stalkers;
141 GetCreatureListWithEntryInGrid(stalkers, me, NPC_BEETLE_STALKER, 100.0f);
142 std::list<Creature*> beetlers = stalkers;
143
144 Trinity::Containers::RandomResizeList(beetlers, 9); // Holds the summoners of Jeweled Scarab
145
146 for (std::list<Creature*>::iterator itr = beetlers.begin(); itr != beetlers.end(); ++itr)
147 {
148 stalkers.remove((*itr)); // Remove it to prevent a single trigger from spawning multiple npcs.
149 (*itr)->CastSpell((*itr), SPELL_BEETLE_BURROW); // Cast visual
150 // Summon after 5 seconds.
151 (*itr)->m_Events.AddEvent(new SummonScarab((*itr), instance), (*itr)->m_Events.CalculateTime(5000));
152 }
153
154 Trinity::Containers::RandomResizeList(stalkers, 2); // Holds the summoners of Dustbone Horror
155
156 for (std::list<Creature*>::iterator itr = stalkers.begin(); itr != stalkers.end(); ++itr)
157 (*itr)->CastSpell((*itr), SPELL_SUMMON_DUSTBONE_HORROR);
158 }
159 }
160
161 void SetData(uint32 index, uint32 /*value*/) override
162 {
163 if (index == DATA_SUMMON_DEATHS)
164 {
165 ++_summonDeaths;
166 if (_summonDeaths == 11) // All summons died
167 {
168 me->GetMap()->SetZoneWeather(AREA_TOMB_OF_THE_EARTHRAGER, WEATHER_STATE_FOG, 0.0f);
169 me->RemoveAurasDueToSpell(SPELL_PTAH_EXPLOSION);
170 events.SetPhase(PHASE_NORMAL);
171 events.ScheduleEvent(EVENT_RAGING_SMASH, urand(7000, 12000), 0, PHASE_NORMAL);
172 events.ScheduleEvent(EVENT_FLAME_BOLT, 15000, 0, PHASE_NORMAL);
173 events.ScheduleEvent(EVENT_EARTH_SPIKE, urand(16000, 21000), 0, PHASE_NORMAL);
174 }
175 }
176 }

200 void UpdateAI(uint32 diff) override
201 {
202 if (!UpdateVictim() || !CheckInRoom())
203 return;
204
205 events.Update(diff);
206
207 if (me->HasUnitState(UNIT_STATE_CASTING))
208 return;
209
210 while (uint32 eventId = events.ExecuteEvent())
211 {
212 switch (eventId)
213 {
214 case EVENT_RAGING_SMASH:
215 DoCastVictim(SPELL_RAGING_SMASH);
216 events.ScheduleEvent(EVENT_RAGING_SMASH, urand(7000, 12000), 0, PHASE_NORMAL);
217 break;
218 case EVENT_FLAME_BOLT:
219 DoCast(me, SPELL_FLAME_BOLT);
220 events.ScheduleEvent(EVENT_FLAME_BOLT, 15000, 0, PHASE_NORMAL);
221 break;
222 case EVENT_EARTH_SPIKE:
223 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
224 DoCast(target, SPELL_EARTH_SPIKE_WARN);
225 events.ScheduleEvent(EVENT_EARTH_SPIKE, urand(16000, 21000), 0, PHASE_NORMAL);
226 break;
227 case EVENT_PTAH_EXPLODE:
228 DoCast(me, SPELL_PTAH_EXPLOSION);
229 break;
230 case EVENT_QUICKSAND:
231 // Spell not in DBC, it is not cast either, according to sniffs
232 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
233 if (Creature* quicksand = me->SummonCreature(NPC_QUICKSAND, *target))
234 quicksand->SetUInt32Value(UNIT_CREATED_BY_SPELL, SPELL_SUMMON_QUICKSAND);
235 events.ScheduleEvent(EVENT_QUICKSAND, 10000, 0, PHASE_DISPERSE);
236 break;
237 }
238 }
239
240 if (events.GetPhaseMask() & PHASE_MASK_NORMAL) // Do not melee in the disperse phase
241 DoMeleeAttackIfReady();
242 }
243
244 protected:
245 uint8 _summonDeaths;
246 bool _hasDispersed;
247 };

251 return GetHallsOfOriginationAI<boss_earthrager_ptahAI>(creature);

286 class spell_earthrager_ptah_explosion_AuraScript : public AuraScript
287 {
288 PrepareAuraScript(spell_earthrager_ptah_explosion_AuraScript);
289
290 void SetFlags(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
291 {
292 if (Unit* ptah = GetCaster())
293 {
294 ptah->SetFlag(UNIT_FIELD_FLAGS, uint32(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_UNK_29 | UNIT_FLAG_UNK_31));
295 ptah->SetFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
296 }
297 }
298
299 void RemoveFlags(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
300 {
301 if (Unit* ptah = GetCaster())
302 {
303 ptah->RemoveFlag(UNIT_FIELD_FLAGS, uint32(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_UNK_29 | UNIT_FLAG_UNK_31));
304 ptah->RemoveFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
305 }
306 }
307
308 void Register() override
309 {
310 OnEffectApply += AuraEffectApplyFn(spell_earthrager_ptah_explosion_AuraScript::SetFlags, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
311 OnEffectRemove += AuraEffectRemoveFn(spell_earthrager_ptah_explosion_AuraScript::RemoveFlags, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
312 }
313 };

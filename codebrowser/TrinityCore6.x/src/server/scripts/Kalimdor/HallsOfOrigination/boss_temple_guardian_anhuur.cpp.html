76class boss_temple_guardian_anhuur : public CreatureScript

78public:
79 boss_temple_guardian_anhuur() : CreatureScript("boss_temple_guardian_anhuur") { }

81 struct boss_temple_guardian_anhuurAI : public BossAI
82 {
83 boss_temple_guardian_anhuurAI(Creature* creature) : BossAI(creature, DATA_TEMPLE_GUARDIAN_ANHUUR)
84 {
85 Initialize();
86 }
87
88 void Initialize()
89 {
90 _phase = PHASE_FIRST_SHIELD;
91 _oldPhase = PHASE_FIRST_SHIELD;
92 _beacons = 0;
93 }
94
95 void CleanStalkers()
96 {
97 std::list<Creature*> stalkers;
98 GetCreatureListWithEntryInGrid(stalkers, me, NPC_CAVE_IN_STALKER, 100.0f);
99 for (std::list<Creature*>::iterator itr = stalkers.begin(); itr != stalkers.end(); ++itr)
100 {
101 (*itr)->RemoveAurasDueToSpell(SPELL_BEAM_OF_LIGHT_RIGHT);
102 (*itr)->RemoveAurasDueToSpell(SPELL_BEAM_OF_LIGHT_LEFT);
103 }
104 }
105
106 void Reset() override
107 {
108 Initialize();
109 _Reset();
110 CleanStalkers();
111 me->RemoveAurasDueToSpell(SPELL_SHIELD_OF_LIGHT);
112 events.ScheduleEvent(EVENT_DIVINE_RECKONING, urand(10000, 12000));
113 events.ScheduleEvent(EVENT_BURNING_LIGHT, 12000);
114 }
115
116 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
117 {
118 if ((me->HealthBelowPctDamaged(66, damage) && _phase == PHASE_FIRST_SHIELD) ||
119 (me->HealthBelowPctDamaged(33, damage) && _phase == PHASE_SECOND_SHIELD))
120 {
121 _beacons = 2;
122 _phase++; // Increase the phase
123 _oldPhase = _phase;
124
125 _phase = PHASE_SHIELDED;
126
127 me->InterruptNonMeleeSpells(true);
128 me->AttackStop();
129 DoCast(me, SPELL_TELEPORT);
130
131 DoCast(me, SPELL_SHIELD_OF_LIGHT);
132 me->SetFlag(UNIT_FIELD_FLAGS, uint32(UNIT_FLAG_UNK_31));
133
134 DoCastAOE(SPELL_ACTIVATE_BEACONS);
135
136 GameObject* door = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_ANHUUR_DOOR));
137 if (door)
138 {
139 std::list<Creature*> stalkers;
140 GetCreatureListWithEntryInGrid(stalkers, me, NPC_CAVE_IN_STALKER, 100.0f);
141
142 stalkers.remove_if(Trinity::HeightDifferenceCheck(door, 0.0f, false)); // Target only the bottom ones
143 for (std::list<Creature*>::iterator itr = stalkers.begin(); itr != stalkers.end(); ++itr)
144 {
145 if ((*itr)->GetPositionX() > door->GetPositionX())
146 {
147 (*itr)->CastSpell((*itr), SPELL_SHIELD_VISUAL_LEFT, true);
148 (*itr)->CastSpell((*itr), SPELL_BEAM_OF_LIGHT_LEFT, true);
149 }
150 else
151 {
152 (*itr)->CastSpell((*itr), SPELL_SHIELD_VISUAL_RIGHT, true);
153 (*itr)->CastSpell((*itr), SPELL_BEAM_OF_LIGHT_RIGHT, true);
154 }
155 }
156 }
157
158 DoCast(me, SPELL_REVERBERATING_HYMN);
159
160 Talk(EMOTE_SHIELD);
161 Talk(SAY_SHIELD);
162 }
163 }
164
165 void DoAction(int32 action) override
166 {
167 if (action == ACTION_DISABLE_BEACON)
168 {
169 --_beacons;
170 if (!_beacons)
171 {
172 me->RemoveAurasDueToSpell(SPELL_SHIELD_OF_LIGHT);
173 Talk(EMOTE_UNSHIELD);
174 _phase = _oldPhase;
175 }
176 }
177 }
178
179 void EnterCombat(Unit* /*who*/) override
180 {
181 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, me, 1);
182 Talk(SAY_AGGRO);
183 _EnterCombat();
184 }
185
186 void JustDied(Unit* /*killer*/) override
187 {
188 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
189 Talk(SAY_DEATH);
190 _JustDied();
191 }
192
193 void KilledUnit(Unit* victim) override
194 {
195 if (victim->GetTypeId() == TYPEID_PLAYER)
196 Talk(SAY_KILL);
197 }
198
199 void JustReachedHome() override
200 {
201 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
202 _JustReachedHome();
203 instance->SetBossState(DATA_TEMPLE_GUARDIAN_ANHUUR, FAIL);
204 }
205
206 void UpdateAI(uint32 diff) override
207 {
208 if (!UpdateVictim() || !CheckInRoom() || me->GetCurrentSpell(CURRENT_CHANNELED_SPELL) || _phase == PHASE_SHIELDED)
209 return;
210
211 events.Update(diff);
212
213 if (me->HasUnitState(UNIT_STATE_CASTING))
214 return;
215
216 while (uint32 eventId = events.ExecuteEvent())
217 {
218 switch (eventId)
219 {
220 case EVENT_DIVINE_RECKONING:
221 DoCastVictim(SPELL_DIVINE_RECKONING);
222 events.ScheduleEvent(EVENT_DIVINE_RECKONING, urand(10000, 12000));
223 break;
224 case EVENT_BURNING_LIGHT:
225 {
226 Unit* unit = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me));
227 if (!unit)
228 unit = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true);
229 DoCast(unit, SPELL_BURNING_LIGHT);
230 events.ScheduleEvent(EVENT_SEAR, 2000);
231 events.ScheduleEvent(EVENT_BURNING_LIGHT, 12000);
232 break;
233 }
234 case EVENT_SEAR:
235 {
236 Unit* target = me->FindNearestCreature(NPC_SEARING_LIGHT, 100.0f);
237 if (!target)
238 break;
239
240 std::list<Creature*> stalkers;
241 GetCreatureListWithEntryInGrid(stalkers, me, NPC_CAVE_IN_STALKER, 100.0f);
242 stalkers.remove_if(Trinity::HeightDifferenceCheck(ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_ANHUUR_DOOR)), 5.0f, true));
243
244 if (stalkers.empty())
245 break;
246
247 stalkers.sort(Trinity::ObjectDistanceOrderPred(target));
248
249 // Get the closest statue face (any of its eyes)
250 Creature* eye1 = stalkers.front();
251 stalkers.remove(eye1); // Remove the eye.
252 stalkers.sort(Trinity::ObjectDistanceOrderPred(eye1)); // Find the second eye.
253 Creature* eye2 = stalkers.front();
254
255 eye1->CastSpell(eye1, SPELL_SEARING_LIGHT, true);
256 eye2->CastSpell(eye2, SPELL_SEARING_LIGHT, true);
257 break;
258 }
259 }
260 }
261
262 DoMeleeAttackIfReady();
263 }
264
265 private:
266 uint8 _phase;
267 uint8 _oldPhase;
268 uint8 _beacons;
269 };
270
271 CreatureAI* GetAI(Creature* creature) const override
272 {
273 return GetHallsOfOriginationAI<boss_temple_guardian_anhuurAI>(creature);
274 }

277class spell_anhuur_shield_of_light : public SpellScriptLoader
278{
279 public:
280 spell_anhuur_shield_of_light() : SpellScriptLoader("spell_anhuur_shield_of_light") { }
281
282 class spell_anhuur_shield_of_light_SpellScript : public SpellScript
283 {
284 PrepareSpellScript(spell_anhuur_shield_of_light_SpellScript);
285
286 void FilterTargets(std::list<WorldObject*>& targets)
287 {
288 if (InstanceMap* instance = GetCaster()->GetMap()->ToInstanceMap())
289 {
290 if (InstanceScript* const script = instance->GetInstanceScript())
291 {
292 if (GameObject* go = ObjectAccessor::GetGameObject(*GetCaster(), script->GetGuidData(DATA_ANHUUR_DOOR)))
293 {
294 targets.remove_if(Trinity::HeightDifferenceCheck(go, 5.0f, false));
295 targets.remove(GetCaster());
296 targets.sort(Trinity::ObjectDistanceOrderPred(GetCaster()));
297 targets.resize(2);
298 }
299 }
300 }
301 }
302
303 void Register() override
304 {
305 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_anhuur_shield_of_light_SpellScript::FilterTargets, EFFECT_1, TARGET_UNIT_SRC_AREA_ENTRY);
306 }
307 };
308
309 SpellScript* GetSpellScript() const override
310 {
311 return new spell_anhuur_shield_of_light_SpellScript();
312 }

319
320 class spell_anhuur_disable_beacon_beams_SpellScript : public SpellScript
321 {
322 PrepareSpellScript(spell_anhuur_disable_beacon_beams_SpellScript);
323
324 void HandleScript(SpellEffIndex /*effIndex*/)
325 {
326 GetHitUnit()->RemoveAurasDueToSpell(GetEffectValue());
327 }
328
329 void Notify(SpellEffIndex /*index*/)
330 {
331 if (InstanceMap* instance = GetCaster()->GetMap()->ToInstanceMap())
332 if (InstanceScript* const script = instance->GetInstanceScript())
333 if (Creature* anhuur = instance->GetCreature(script->GetGuidData(DATA_ANHUUR_GUID)))
334 anhuur->AI()->DoAction(ACTION_DISABLE_BEACON);
335 }
336
337 void Register() override
338 {
339 OnEffectHitTarget += SpellEffectFn(spell_anhuur_disable_beacon_beams_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
340 OnEffectHit += SpellEffectFn(spell_anhuur_disable_beacon_beams_SpellScript::Notify, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
341 }
342 };
343
344 SpellScript* GetSpellScript() const override
345 {
346 return new spell_anhuur_disable_beacon_beams_SpellScript();

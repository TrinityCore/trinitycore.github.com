128class boss_anraphet : public CreatureScript
129{
130public:
131 boss_anraphet() : CreatureScript("boss_anraphet") { }
132
133 struct boss_anraphetAI : public BossAI
134 {
135 boss_anraphetAI(Creature* creature) : BossAI(creature, DATA_ANRAPHET) { }
136
137 void ScheduleCombatEvents()
138 {
139 events.ScheduleEvent(EVENT_ANRAPHET_NEMESIS_STRIKE, 8000, 0, PHASE_COMBAT);
140 events.ScheduleEvent(EVENT_ANRAPHET_ALPHA_BEAMS, 10000, 0, PHASE_COMBAT);
141 events.ScheduleEvent(EVENT_ANRAPHET_OMEGA_STANCE, 35000, 0, PHASE_COMBAT);
142 }
143
144 void Reset() override
145 {
146 _Reset();
147 me->SetWalk(false);
148 events.SetPhase(PHASE_INTRO);
149 if (instance->GetData(DATA_DEAD_ELEMENTALS) == 4)
150 {
151 // Set to combat automatically, Brann's event won't repeat
152 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
153 events.SetPhase(PHASE_COMBAT);
154 ScheduleCombatEvents();
155 me->SetHomePosition(AnraphetActivatePos);
156 }
157 }
158
159 void EnterCombat(Unit* /*who*/) override
160 {
161 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, me, 1);
162 Talk(ANRAPHET_SAY_AGGRO);
163 _EnterCombat();
164 }
165
166 void JustDied(Unit* /*killer*/) override
167 {
168 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
169 Talk(ANRAPHET_SAY_DEATH);
170
171 if (Creature* brann = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BRANN_0_GUID)))
172 brann->AI()->DoAction(ACTION_ANRAPHET_DIED);
173
174 _JustDied();
175 }
176
177 void KilledUnit(Unit* victim) override
178 {
179 if (victim->GetTypeId() == TYPEID_PLAYER)
180 Talk(ANRAPHET_SAY_KILL);
181 }
182
183 void JustReachedHome() override
184 {
185 instance->SendEncounterUnit(ENCOUNTER_FRAME_DISENGAGE, me);
186 _JustReachedHome();
187 instance->SetBossState(DATA_ANRAPHET, FAIL);
188 }
189
190 void DoAction(int32 action) override
191 {
192 if (action == ACTION_ANRAPHET_INTRO)
193 events.ScheduleEvent(EVENT_ANRAPHET_APPEAR, 6000, 0, PHASE_INTRO);
194 }
195
196 void MovementInform(uint32 type, uint32 point) override
197 {
198 if (type != POINT_MOTION_TYPE)
199 return;
200
201 if (point == POINT_ANRAPHET_ACTIVATE)
202 {
203 events.ScheduleEvent(EVENT_ANRAPHET_ACTIVATE, 1500, 0, PHASE_INTRO);
204 me->SetHomePosition(AnraphetActivatePos);
205 }
206 }
207
208 void UpdateAI(uint32 diff) override
209 {
210 if ((events.GetPhaseMask() & PHASE_MASK_COMBAT) && (!UpdateVictim() || !CheckInRoom()))
211 return;
212
213 events.Update(diff);
214
215 if (me->HasUnitState(UNIT_STATE_CASTING))
216 return;
217
218 while (uint32 eventId = events.ExecuteEvent())
219 {
220 switch (eventId)
221 {
222 case EVENT_ANRAPHET_APPEAR:
223 me->SetWalk(true);
224 me->GetMotionMaster()->MovePoint(POINT_ANRAPHET_ACTIVATE, AnraphetActivatePos);
225 break;
226 case EVENT_ANRAPHET_ACTIVATE:
227 me->SetWalk(false);
228 Talk(ANRAPHET_SAY_INTRO);
229 events.ScheduleEvent(EVENT_ANRAPHET_DESTROY, 17500, 0, PHASE_INTRO);
230 return;
231 case EVENT_ANRAPHET_DESTROY:
232 DoCastAOE(SPELL_DESTRUCTION_PROTOCOL);
233 events.ScheduleEvent(EVENT_ANRAPHET_READY, 6000, 0, PHASE_INTRO);
234 break;
235 case EVENT_ANRAPHET_READY:
236 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
237 events.SetPhase(PHASE_COMBAT);
238 ScheduleCombatEvents();
239 break;
240 case EVENT_ANRAPHET_NEMESIS_STRIKE:
241 DoCastVictim(SPELL_NEMESIS_STRIKE);
242 events.ScheduleEvent(EVENT_ANRAPHET_NEMESIS_STRIKE, 21500, 0, PHASE_COMBAT);
243 break;
244 case EVENT_ANRAPHET_ALPHA_BEAMS:
245 DoCast(me, SPELL_ALPHA_BEAMS);
246 events.ScheduleEvent(EVENT_ANRAPHET_CRUMBLING_RUIN, 12500, 0, PHASE_COMBAT);
247 events.ScheduleEvent(EVENT_ANRAPHET_ALPHA_BEAMS, urand(40000, 45000), 0, PHASE_COMBAT);
248 break;
249 case EVENT_ANRAPHET_OMEGA_STANCE:
250 DoCast(me, SPELL_OMEGA_STANCE_SUMMON);
251 DoCast(me, SPELL_OMEGA_STANCE);
252 Talk(ANRAPHET_SAY_OMEGA_STANCE);
253 events.ScheduleEvent(EVENT_ANRAPHET_OMEGA_STANCE, urand(45000, 50000), 0, PHASE_COMBAT);
254 events.ScheduleEvent(EVENT_ANRAPHET_CRUMBLING_RUIN, 13000, 0, PHASE_COMBAT);
255 break;
256 case EVENT_ANRAPHET_CRUMBLING_RUIN:
257 DoCast(me, SPELL_CRUMBLING_RUIN);
258 break;
259 }
260 }
261
262 if (events.GetPhaseMask() & PHASE_MASK_COMBAT)
263 DoMeleeAttackIfReady();
264 }
265 };
266
267 CreatureAI* GetAI(Creature* creature) const override
268 {
269 return GetHallsOfOriginationAI<boss_anraphetAI>(creature);
270 }

323class npc_brann_bronzebeard_anraphet : public CreatureScript
324{
325 public:
326 npc_brann_bronzebeard_anraphet() : CreatureScript("npc_brann_bronzebeard_anraphet") { }
327
328 struct npc_brann_bronzebeard_anraphetAI : public CreatureAI
329 {
330 npc_brann_bronzebeard_anraphetAI(Creature* creature) : CreatureAI(creature), _currentPoint(0), _instance(creature->GetInstanceScript()) { }
331
332 void sGossipSelect(Player* /*player*/, uint32 sender, uint32 action) override
333 {
334 if (_instance->GetBossState(DATA_VAULT_OF_LIGHTS) == DONE)
335 return;
336
337 if (me->GetCreatureTemplate()->GossipMenuId == sender && !action)
338 {
339 _instance->SetBossState(DATA_VAULT_OF_LIGHTS, IN_PROGRESS);
340 _currentPoint = 0;
341 events.Reset();
342 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
343 me->SetWalk(true);
344 Talk(BRANN_SAY_DOOR_INTRO);
345 events.ScheduleEvent(EVENT_BRANN_UNLOCK_DOOR, 7500);
346 }
347 }
348
349 void DoAction(int32 action) override
350 {
351 switch (action)
352 {
353 case ACTION_ELEMENTAL_DIED:
354 {
355 uint32 dead = _instance->GetData(DATA_DEAD_ELEMENTALS);
356 Talk(BRANN_1_ELEMENTAL_DEAD + dead - 1);
357 if (dead == 4)
358 {
359 _instance->DoCastSpellOnPlayers(SPELL_VAULT_OF_LIGHTS_CREDIT);
360 if (Creature* anraphet = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_ANRAPHET_GUID)))
361 anraphet->AI()->DoAction(ACTION_ANRAPHET_INTRO);
362 }
363 break;
364 }
365 case ACTION_ANRAPHET_DIED:
366 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
367 events.ScheduleEvent(EVENT_BRANN_MOVE_INTRO, 1000);
368 break;
369 }
370 }
371
372 void UpdateAI(uint32 diff) override
373 {
374 events.Update(diff);
375
376 while (uint32 eventId = events.ExecuteEvent())
377 {
378 switch (eventId)
379 {
380 case EVENT_BRANN_MOVE_INTRO:
381 if (_currentPoint < MAX_BRANN_WAYPOINTS_INTRO)
382 me->GetMotionMaster()->MovePoint(_currentPoint, BrannIntroWaypoint[_currentPoint]);
383 break;
384 case EVENT_BRANN_UNLOCK_DOOR:
385 Talk(BRANN_SAY_UNLOCK_DOOR);
386 _instance->SetBossState(DATA_VAULT_OF_LIGHTS, DONE);
387 _instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_VAULT_OF_LIGHTS_EVENT);
388 events.ScheduleEvent(EVENT_BRANN_MOVE_INTRO, 3500);
389 break;
390 case EVENT_BRANN_THINK:
391 Talk(BRANN_SAY_THINK);
392 events.ScheduleEvent(EVENT_BRANN_SET_ORIENTATION_1, 6000);
393 break;
394 case EVENT_BRANN_SET_ORIENTATION_1:
395 me->SetFacingTo(5.445427f);
396 Talk(BRANN_SAY_MIRRORS);
397 events.ScheduleEvent(EVENT_BRANN_SET_ORIENTATION_2, 1000);
398 break;
399 case EVENT_BRANN_SET_ORIENTATION_2:
400 me->SetFacingTo(0.6283185f);
401 events.ScheduleEvent(EVENT_BRANN_SET_ORIENTATION_3, 2500);
402 break;
403 case EVENT_BRANN_SET_ORIENTATION_3:
404 me->SetFacingTo(0.01745329f);
405 events.ScheduleEvent(EVENT_BRANN_SAY_ELEMENTALS, 200);
406 break;
407 case EVENT_BRANN_SAY_ELEMENTALS:
408 Talk(BRANN_SAY_ELEMENTALS);
409 events.ScheduleEvent(EVENT_BRANN_SAY_GET_IT, 3500);
410 break;
411 case EVENT_BRANN_SAY_GET_IT:
412 Talk(BRANN_SAY_GET_IT);
413 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
414 break;
415 case EVENT_BRANN_SET_ORIENTATION_4:
416 me->SetFacingTo(3.141593f);
417 break;
418 }
419 }
420 }
421
422 void MovementInform(uint32 movementType, uint32 pointId) override
423 {
424 if (movementType != POINT_MOTION_TYPE)
425 return;
426
427 _currentPoint = pointId + 1;
428 uint32 delay = 1;
429
430 switch (pointId)
431 {
432 case 0:
433 Talk(BRANN_SAY_TROGGS);
434 events.ScheduleEvent(EVENT_BRANN_THINK, 15000);
435 return;
436 case 1:
437 Talk(BRANN_SAY_ANRAPHET_DIED);
438 delay = 1000;
439 break;
440 case 14:
441 Talk(BRANN_SAY_MOMENT);
442 delay = 2200;
443 break;
444 case 16:
445 events.ScheduleEvent(EVENT_BRANN_SET_ORIENTATION_4, 6000);
446 return;
447 default:
448 break;
449 }
450
451 events.ScheduleEvent(EVENT_BRANN_MOVE_INTRO, delay);
452 }
453
454 protected:
455 EventMap events;
456 uint32 _currentPoint;
457 InstanceScript* _instance;
458 };

462 return GetHallsOfOriginationAI<npc_brann_bronzebeard_anraphetAI>(creature);

531 {
532 PrepareSpellScript(spell_omega_stance_spider_effect_SpellScript);
533
534 void SetDestPosition(SpellEffIndex effIndex)
535 {
536 // Do our own calculations for the destination position.
537 /// TODO: Remove this once we find a general rule for WorldObject::MovePosition (this spell shouldn't take the Z change into consideration)
538 Unit* caster = GetCaster();
539 float angle = float(rand_norm()) * static_cast<float>(2 * M_PI);
540 uint32 dist = caster->GetObjectSize() + GetSpellInfo()->GetEffect(effIndex)->CalcRadius(GetCaster()) * (float)rand_norm();
541
542 float x = caster->GetPositionX() + dist * std::cos(angle);
543 float y = caster->GetPositionY() + dist * std::sin(angle);
544 float z = caster->GetMap()->GetHeight(x, y, caster->GetPositionZ());
545
546 const_cast<WorldLocation*>(GetExplTargetDest())->Relocate(x, y, z);
547 GetHitDest()->Relocate(x, y, z);
548 }
549
550 void Register() override
551 {
552 OnEffectLaunch += SpellEffectFn(spell_omega_stance_spider_effect_SpellScript::SetDestPosition, EFFECT_0, SPELL_EFFECT_DUMMY);
553 }
554 };

557 {
558 return new spell_omega_stance_spider_effect_SpellScript();

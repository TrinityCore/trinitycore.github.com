47class instance_halls_of_origination : public InstanceMapScript
48{
49 public:
50 instance_halls_of_origination() : InstanceMapScript(HoOScriptName, 644) { }
51
52 struct instance_halls_of_origination_InstanceMapScript : public InstanceScript
53 {
54 instance_halls_of_origination_InstanceMapScript(InstanceMap* map) : InstanceScript(map)
55 {
56 SetHeaders(DataHeader);
57 SetBossNumber(EncounterCount);
58 LoadDoorData(doorData);
59 _deadElementals = 0;
60 }
61
62 void OnGameObjectCreate(GameObject* go) override
63 {
64 switch (go->GetEntry())
65 {
66 case GO_ANHUURS_BRIDGE:
67 AnhuursBridgeGUID = go->GetGUID();
68 // no break
69 case GO_DOODAD_ULDUM_ELEVATOR_COL01:
70 case GO_VAULT_OF_LIGHTS_DOOR:
71 case GO_DOODAD_ULDUM_LIGHTMACHINE_01:
72 case GO_DOODAD_ULDUM_LIGHTMACHINE_02:
73 case GO_DOODAD_ULDUM_LIGHTMACHINE_03:
74 case GO_DOODAD_ULDUM_LIGHTMACHINE_04:
75 case GO_DOODAD_ULDUM_LASERBEAMS01:
76 case GO_DOODAD_ULDUM_LASERBEAMS_01:
77 case GO_DOODAD_ULDUM_LASERBEAMS_02:
78 case GO_DOODAD_ULDUM_LASERBEAMS_03:
79 AddDoor(go, true);
80 break;
81 case GO_ANHUURS_DOOR:
82 AnhuursDoorGUID = go->GetGUID();
83 AddDoor(go, true);
84 break;
85 case GO_ANHUURS_RIGHT_BEACON:
86 AnhuurRightBeaconGUID = go->GetGUID();
87 break;
88 case GO_ANHUURS_LEFT_BEACON:
89 AnhuurLeftBeaconGUID = go->GetGUID();
90 break;
91 case GO_SUN_MIRROR:
92 SunMirrorGUID = go->GetGUID();
93 break;
94 case GO_ANRAPHET_DOOR:
95 AnraphetDoorGUID = go->GetGUID();
96 break;
97 }
98 }
99
100 void OnGameObjectRemove(GameObject* go) override
101 {
102 switch (go->GetEntry())
103 {
104 case GO_ANHUURS_BRIDGE:
105 case GO_DOODAD_ULDUM_ELEVATOR_COL01:
106 case GO_ANHUURS_DOOR:
107 case GO_VAULT_OF_LIGHTS_DOOR:
108 case GO_DOODAD_ULDUM_LIGHTMACHINE_01:
109 case GO_DOODAD_ULDUM_LIGHTMACHINE_02:
110 case GO_DOODAD_ULDUM_LIGHTMACHINE_03:
111 case GO_DOODAD_ULDUM_LIGHTMACHINE_04:
112 case GO_DOODAD_ULDUM_LASERBEAMS01:
113 case GO_DOODAD_ULDUM_LASERBEAMS_01:
114 case GO_DOODAD_ULDUM_LASERBEAMS_02:
115 case GO_DOODAD_ULDUM_LASERBEAMS_03:
116 AddDoor(go, false);
117 break;
118 }
119 }
120
121 void OnCreatureCreate(Creature* creature) override
122 {
123 switch (creature->GetEntry())
124 {
125 case BOSS_TEMPLE_GUARDIAN_ANHUUR:
126 TempleGuardianAnhuurGUID = creature->GetGUID();
127 break;
128 case NPC_BRANN_BRONZEBEARD_0:
129 BrannBronzebeardGUID = creature->GetGUID();
130 break;
131 case BOSS_ANRAPHET:
132 AnraphetGUID = creature->GetGUID();
133 break;
134 }
135 }
136
137 uint32 GetData(uint32 data) const override
138 {
139 switch (data)
140 {
141 case DATA_DEAD_ELEMENTALS:
142 return _deadElementals;
143 default:
144 break;
145 }
146
147 return 0;
148 }
149
150 ObjectGuid GetGuidData(uint32 index) const override
151 {
152 switch (index)
153 {
154 case DATA_ANHUUR_BRIDGE:
155 return AnhuursBridgeGUID;
156 case DATA_ANHUUR_DOOR:
157 return AnhuursDoorGUID;
158 case DATA_ANHUUR_LEFT_BEACON:
159 return AnhuurLeftBeaconGUID;
160 case DATA_ANHUUR_RIGHT_BEACON:
161 return AnhuurRightBeaconGUID;
162 case DATA_ANHUUR_GUID:
163 return TempleGuardianAnhuurGUID;
164 case DATA_BRANN_0_GUID:
165 return BrannBronzebeardGUID;
166 case DATA_ANRAPHET_GUID:
167 return AnraphetGUID;
168 }
169
170 return ObjectGuid::Empty;
171 }
172
173 void IncreaseDeadElementals(uint32 inc)
174 {
175 _deadElementals += inc;
176 if (_deadElementals == 4)
177 {
178 if (GameObject* mirror = instance->GetGameObject(SunMirrorGUID))
179 mirror->SetGoState(GO_STATE_ACTIVE);
180 if (GameObject* door = instance->GetGameObject(AnraphetDoorGUID))
181 door->SetGoState(GO_STATE_ACTIVE);
182 }
183 }
184
185 void OnUnitDeath(Unit* unit) override
186 {
187 Creature* creature = unit->ToCreature();
188 if (!creature)
189 return;
190
191 switch (creature->GetEntry())
192 {
193 case NPC_FIRE_WARDEN:
194 case NPC_EARTH_WARDEN:
195 case NPC_WATER_WARDEN:
196 case NPC_AIR_WARDEN:
197 uint32 data = creature->GetEntry() - WARDEN_ENTRY_DATA_DELTA;
198 SetBossState(data, IN_PROGRESS); // Needs to be set to IN_PROGRESS or else the gameobjects state won't be updated
199 SetBossState(data, DONE);
200 IncreaseDeadElementals(1);
201 if (Creature* brann = instance->GetCreature(BrannBronzebeardGUID))
202 brann->AI()->DoAction(ACTION_ELEMENTAL_DIED);
203 break;
204 }
205 }
206
207 void WriteSaveDataMore(std::ostringstream& data) override
208 {
209 data << _deadElementals;
210 }
211
212 void ReadSaveDataMore(std::istringstream& data) override
213 {
214 uint32 deadElementals;
215 data >> deadElementals;
216 IncreaseDeadElementals(deadElementals);
217 }

228 ObjectGuid SunMirrorGUID;

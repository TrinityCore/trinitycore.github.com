39enum Mist
40{
41 SAY_AT_HOME = 0,
42 EMOTE_AT_HOME = 1,
43 QUEST_MIST = 938,
44 NPC_ARYNIA = 3519,
45 FACTION_DARNASSUS = 79
46};
47
48class npc_mist : public CreatureScript
49{
50public:
51 npc_mist() : CreatureScript("npc_mist") { }
52
53 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
54 {
55 if (quest->GetQuestId() == QUEST_MIST)
56 if (npc_mistAI* pMistAI = CAST_AI(npc_mist::npc_mistAI, creature->AI()))
57 pMistAI->StartFollow(player, FACTION_DARNASSUS, quest);
58
59 return true;
60 }
61
62 CreatureAI* GetAI(Creature* creature) const override
63 {
64 return new npc_mistAI(creature);
65 }
66
67 struct npc_mistAI : public FollowerAI
68 {
69 npc_mistAI(Creature* creature) : FollowerAI(creature) { }
70
71 void Reset() override { }
72
73 void MoveInLineOfSight(Unit* who) override
74
75 {
76 FollowerAI::MoveInLineOfSight(who);
77
78 if (!me->GetVictim() && !HasFollowState(STATE_FOLLOW_COMPLETE) && who->GetEntry() == NPC_ARYNIA)
79 {
80 if (me->IsWithinDistInMap(who, 10.0f))
81 {
82 Talk(SAY_AT_HOME, who);
83 DoComplete();
84 }
85 }
86 }
87
88 void DoComplete()
89 {
90 Talk(EMOTE_AT_HOME);
91
92 Player* player = GetLeaderForFollower();
93 if (player && player->GetQuestStatus(QUEST_MIST) == QUEST_STATUS_INCOMPLETE)
94 player->GroupEventHappens(QUEST_MIST, me);
95
96 //The follow is over (and for later development, run off to the woods before really end)
97 SetFollowComplete();
98 }
99

104 return;
105
106 DoMeleeAttackIfReady();
107 }*/
108 };
109
110};

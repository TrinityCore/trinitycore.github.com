137 void Initialize()
138 {
139 Phase = PHASE_START;
140 MovePoint = urand(0, 5);
141 PointData = GetMoveData();
142 SummonWhelpCount = 0;
143 triggerGUID.Clear();
144 tankGUID.Clear();
145 IsMoving = false;
146 }
147
148 void Reset() override
149 {
150 Initialize();
151
152 if (!IsCombatMovementAllowed())
153 SetCombatMovement(true);
154
155 _Reset();
156 me->SetReactState(REACT_AGGRESSIVE);
157 instance->SetData(DATA_ONYXIA_PHASE, Phase);
158 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMED_START_EVENT);
159 }
160
161 void EnterCombat(Unit* /*who*/) override
162 {
163 _EnterCombat();
164 Talk(SAY_AGGRO);
165 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(10000, 20000));
166 events.ScheduleEvent(EVENT_TAIL_SWEEP, urand(15000, 20000));
167 events.ScheduleEvent(EVENT_CLEAVE, urand(2000, 5000));
168 events.ScheduleEvent(EVENT_WING_BUFFET, urand(10000, 20000));
169 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMED_START_EVENT);
170 }
171
172 void JustSummoned(Creature* summoned) override
173 {
174 summoned->SetInCombatWithZone();
175 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
176 summoned->AI()->AttackStart(target);
177
178 switch (summoned->GetEntry())
179 {
180 case NPC_WHELP:
181 ++SummonWhelpCount;
182 break;
183 case NPC_LAIRGUARD:
184 summoned->setActive(true);
185 break;
186 }
187 summons.Summon(summoned);
188 }
189
190
191 void KilledUnit(Unit* /*victim*/) override
192 {
193 Talk(SAY_KILL);
194 }
195
196 void SpellHit(Unit* /*pCaster*/, const SpellInfo* Spell) override
197 {
198 if (Spell->Id == SPELL_BREATH_EAST_TO_WEST ||
199 Spell->Id == SPELL_BREATH_WEST_TO_EAST ||
200 Spell->Id == SPELL_BREATH_SE_TO_NW ||
201 Spell->Id == SPELL_BREATH_NW_TO_SE ||
202 Spell->Id == SPELL_BREATH_SW_TO_NE ||
203 Spell->Id == SPELL_BREATH_NE_TO_SW)
204 {
205 PointData = GetMoveData();
206 MovePoint = PointData->LocIdEnd;
207
208 me->SetSpeed(MOVE_FLIGHT, 1.5f);
209 me->GetMotionMaster()->MovePoint(8, MiddleRoomLocation);
210 }
211 }
212
213 void MovementInform(uint32 type, uint32 id) override
214 {
215 if (type == POINT_MOTION_TYPE)
216 {
217 switch (id)
218 {
219 case 8:
220 PointData = GetMoveData();
221 if (PointData)
222 {
223 me->SetSpeed(MOVE_FLIGHT, 1.0f);
224 me->GetMotionMaster()->MovePoint(PointData->LocId, PointData->fX, PointData->fY, PointData->fZ);
225 }
226 break;
227 case 9:
228 me->SetCanFly(false);
229 me->SetDisableGravity(false);
230 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
231 if (Creature* trigger = ObjectAccessor::GetCreature(*me, triggerGUID))
232 me->Kill(trigger);
233 me->SetReactState(REACT_AGGRESSIVE);
234 // tank selection based on phase one. If tank is not there i take nearest one
235 if (Unit* tank = ObjectAccessor::GetUnit(*me, tankGUID))
236 me->GetMotionMaster()->MoveChase(tank);
237 else if (Unit* newtarget = SelectTarget(SELECT_TARGET_NEAREST, 0))
238 me->GetMotionMaster()->MoveChase(newtarget);
239 events.ScheduleEvent(EVENT_BELLOWING_ROAR, 5000);
240 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(10000, 20000));
241 events.ScheduleEvent(EVENT_TAIL_SWEEP, urand(15000, 20000));
242 events.ScheduleEvent(EVENT_CLEAVE, urand(2000, 5000));
243 events.ScheduleEvent(EVENT_WING_BUFFET, urand(15000, 30000));
244 break;
245 case 10:
246 me->SetCanFly(true);
247 me->SetDisableGravity(true);
248 me->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
249 me->SetFacingTo(me->GetOrientation() + float(M_PI));
250 if (Creature * trigger = me->SummonCreature(NPC_TRIGGER, MiddleRoomLocation, TEMPSUMMON_CORPSE_DESPAWN))
251 triggerGUID = trigger->GetGUID();
252 me->GetMotionMaster()->MoveTakeoff(11, Phase2Floating);
253 me->SetSpeed(MOVE_FLIGHT, 1.0f);
254 Talk(SAY_PHASE_2_TRANS);
255 instance->SetData(DATA_ONYXIA_PHASE, Phase);
256 events.ScheduleEvent(EVENT_WHELP_SPAWN, 5000);
257 events.ScheduleEvent(EVENT_LAIR_GUARD, 15000);
258 events.ScheduleEvent(EVENT_DEEP_BREATH, 75000);
259 events.ScheduleEvent(EVENT_MOVEMENT, 10000);
260 events.ScheduleEvent(EVENT_FIREBALL, 18000);
261 break;
262 case 11:
263 if (PointData)
264 me->GetMotionMaster()->MovePoint(PointData->LocId, PointData->fX, PointData->fY, PointData->fZ);
265 me->GetMotionMaster()->MoveIdle();
266 break;
267 default:
268 IsMoving = false;
269 break;
270 }
271 }

274 void SpellHitTarget(Unit* target, const SpellInfo* Spell) override
275 {
276 //Workaround - Couldn't find a way to group this spells (All Eruption)
277 if (((Spell->Id >= 17086 && Spell->Id <= 17095) ||
278 (Spell->Id == 17097) ||
279 (Spell->Id >= 18351 && Spell->Id <= 18361) ||
280 (Spell->Id >= 18564 && Spell->Id <= 18576) ||
281 (Spell->Id >= 18578 && Spell->Id <= 18607) ||
282 (Spell->Id == 18609) ||
283 (Spell->Id >= 18611 && Spell->Id <= 18628) ||
284 (Spell->Id >= 21132 && Spell->Id <= 21133) ||
285 (Spell->Id >= 21135 && Spell->Id <= 21139) ||
286 (Spell->Id >= 22191 && Spell->Id <= 22202) ||
287 (Spell->Id >= 22267 && Spell->Id <= 22268)) &&
288 (target->GetTypeId() == TYPEID_PLAYER))
289 {
290 instance->SetData(DATA_SHE_DEEP_BREATH_MORE, FAIL);
291 }

294 OnyxMove* GetMoveData()
295 {
296 uint8 MaxCount = sizeof(MoveData) / sizeof(OnyxMove);
297
298 for (uint8 i = 0; i < MaxCount; ++i)
299 {
300 if (MoveData[i].LocId == MovePoint)
301 return &MoveData[i];
302 }
303
304 return NULL;
305 }
306
307 void SetNextRandomPoint()
308 {
309 uint8 MaxCount = sizeof(MoveData) / sizeof(OnyxMove);
310
311 uint8 iTemp = urand(0, MaxCount - 1);
312
313 if (iTemp >= MovePoint)
314 ++iTemp;
315
316 MovePoint = iTemp;
317 }

319 bool CheckInRoom() override
320 {
321 if (me->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 95.0f)
322 {
323 EnterEvadeMode();
324 return false;
325 }
326
327 return true;
328 }

330 void UpdateAI(uint32 diff) override
331 {
332 if (!UpdateVictim() || !CheckInRoom())
333 return;
334
335 //Common to PHASE_START && PHASE_END
336 if (Phase == PHASE_START || Phase == PHASE_END)
337 {
338 //Specific to PHASE_START || PHASE_END
339 if (Phase == PHASE_START)
340 {
341 if (HealthBelowPct(65))
342 {
343 if (Unit* target = me->GetVictim())
344 tankGUID = target->GetGUID();
345 SetCombatMovement(false);
346 Phase = PHASE_BREATH;
347 me->SetReactState(REACT_PASSIVE);
348 me->AttackStop();
349 me->GetMotionMaster()->MovePoint(10, Phase2Location);
350 return;
351 }
352 }
353
354 events.Update(diff);
355
356 while (uint32 eventId = events.ExecuteEvent())
357 {
358 switch (eventId)
359 {
360 case EVENT_BELLOWING_ROAR: // Phase PHASE_END
361 {
362 DoCastVictim(SPELL_BELLOWING_ROAR);
363 // Eruption
364 GameObject* Floor = NULL;
365 Trinity::GameObjectInRangeCheck check(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 15);
366 Trinity::GameObjectLastSearcher<Trinity::GameObjectInRangeCheck> searcher(me, Floor, check);
367 me->VisitNearbyGridObject(30, searcher);
368 if (Floor)
369 instance->SetGuidData(DATA_FLOOR_ERUPTION_GUID, Floor->GetGUID());
370 events.ScheduleEvent(EVENT_BELLOWING_ROAR, 30000);
371 break;
372 }
373 case EVENT_FLAME_BREATH: // Phase PHASE_START and PHASE_END
374 DoCastVictim(SPELL_FLAME_BREATH);
375 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(10000, 20000));
376 break;
377 case EVENT_TAIL_SWEEP: // Phase PHASE_START and PHASE_END
378 DoCastAOE(SPELL_TAIL_SWEEP);
379 events.ScheduleEvent(EVENT_TAIL_SWEEP, urand(15000, 20000));
380 break;
381 case EVENT_CLEAVE: // Phase PHASE_START and PHASE_END
382 DoCastVictim(SPELL_CLEAVE);
383 events.ScheduleEvent(EVENT_CLEAVE, urand(2000, 5000));
384 break;
385 case EVENT_WING_BUFFET: // Phase PHASE_START and PHASE_END
386 DoCastVictim(SPELL_WING_BUFFET);
387 events.ScheduleEvent(EVENT_WING_BUFFET, urand(15000, 30000));
388 break;
389 default:
390 break;
391 }
392 }
393 DoMeleeAttackIfReady();
394 }
395 else
396 {
397 if (HealthBelowPct(40))
398 {
399 Phase = PHASE_END;
400 instance->SetData(DATA_ONYXIA_PHASE, PHASE_END);
401 Talk(SAY_PHASE_3_TRANS);
402 SetCombatMovement(true);
403 IsMoving = false;
404 Position const pos = me->GetHomePosition();
405 me->GetMotionMaster()->MovePoint(9, pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ() + 12.0f);
406 events.ScheduleEvent(EVENT_BELLOWING_ROAR, 30000);
407 return;
408 }
409
410 if (!me->isMoving())
411 if (Creature* trigger = ObjectAccessor::GetCreature(*me, triggerGUID))
412 me->SetFacingToObject(trigger);
413
414 events.Update(diff);
415
416 while (uint32 eventId = events.ExecuteEvent())
417 {
418 switch (eventId)
419 {
420 case EVENT_DEEP_BREATH: // Phase PHASE_BREATH
421 if (!IsMoving)
422 {
423 if (me->IsNonMeleeSpellCast(false))
424 me->InterruptNonMeleeSpells(false);
425
426 Talk(EMOTE_BREATH);
427 if (PointData) /// @todo: In what cases is this null? What should we do?
428 DoCast(me, PointData->SpellId);
429 events.ScheduleEvent(EVENT_DEEP_BREATH, 75000);
430 }
431 else
432 events.ScheduleEvent(EVENT_DEEP_BREATH, 1000);
433 break;
434 case EVENT_MOVEMENT: // Phase PHASE_BREATH
435 if (!IsMoving && !(me->HasUnitState(UNIT_STATE_CASTING)))
436 {
437 SetNextRandomPoint();
438 PointData = GetMoveData();
439
440 if (!PointData)
441 return;
442
443 me->GetMotionMaster()->MovePoint(PointData->LocId, PointData->fX, PointData->fY, PointData->fZ);
444 IsMoving = true;
445 events.ScheduleEvent(EVENT_MOVEMENT, 25000);
446 }
447 else
448 events.ScheduleEvent(EVENT_MOVEMENT, 500);
449 break;
450 case EVENT_FIREBALL: // Phase PHASE_BREATH
451 if (!IsMoving)
452 {
453 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
454 DoCast(target, SPELL_FIREBALL);
455 events.ScheduleEvent(EVENT_FIREBALL, 8000);
456 }
457 else
458 events.ScheduleEvent(EVENT_FIREBALL, 1000);
459 break;
460 case EVENT_LAIR_GUARD: // Phase PHASE_BREATH
461 me->SummonCreature(NPC_LAIRGUARD, SpawnLocations[2], TEMPSUMMON_CORPSE_DESPAWN);
462 events.ScheduleEvent(EVENT_LAIR_GUARD, 30000);
463 break;
464 case EVENT_WHELP_SPAWN: // Phase PHASE_BREATH
465 me->SummonCreature(NPC_WHELP, SpawnLocations[0], TEMPSUMMON_CORPSE_DESPAWN);
466 me->SummonCreature(NPC_WHELP, SpawnLocations[1], TEMPSUMMON_CORPSE_DESPAWN);
467 if (SummonWhelpCount >= RAID_MODE(20, 40))
468 {
469 SummonWhelpCount = 0;
470 events.ScheduleEvent(EVENT_WHELP_SPAWN, 90000);
471 }
472 else
473 events.ScheduleEvent(EVENT_WHELP_SPAWN, 500);
474 break;
475 default:
476 break;
477 }
478 }
479 }
480 }

117class MoltenEggCheck
118{
119 public:
120 explicit MoltenEggCheck(Creature* pile) : _eggPile(pile) { }
121
122 bool operator()(Unit* object) const
123 {
124 if (object->GetEntry() != NPC_MOLTEN_EGG_TRASH)
125 return false;
126
127 if (object->GetDisplayId() != object->GetNativeDisplayId())
128 return false;
129
130 if (_eggPile->GetDistance2d(object) > 20.0f)
131 return false;
132
133 return true;
134 }
135
136 private:
137 Creature* _eggPile;
138};

140class TrashRespawnWorker
141{
142 public:
143 void operator()(Creature* creature) const
144 {
145 switch (creature->GetEntry())
146 {
147 case NPC_BLAZING_MONSTROSITY_LEFT:
148 case NPC_BLAZING_MONSTROSITY_RIGHT:
149 case NPC_EGG_PILE:
150 case NPC_HARBINGER_OF_FLAME:
151 case NPC_MOLTEN_EGG_TRASH:
152 if (!creature->IsAlive())
153 creature->Respawn(true);
154 break;
155 case NPC_SMOULDERING_HATCHLING:
156 creature->DespawnOrUnsummon();
157 break;
158 }
159 }
160};
161
162static void AlysrazorTrashEvaded(Creature* creature)
163{
164 TrashRespawnWorker check;
165 Trinity::CreatureWorker<TrashRespawnWorker> worker(creature, check);
166 creature->VisitNearbyGridObject(SIZE_OF_GRIDS, worker);
167}

169class npc_harbinger_of_flame : public CreatureScript
170{
171 public:
172 npc_harbinger_of_flame() : CreatureScript("npc_harbinger_of_flame") { }
173
174 struct npc_harbinger_of_flameAI : public ScriptedAI
175 {
176 npc_harbinger_of_flameAI(Creature* creature) : ScriptedAI(creature)
177 {
178 }
179
180 void EnterCombat(Unit* /*target*/) override
181 {
182 if (Creature* bird = ObjectAccessor::GetCreature(*me, me->GetChannelObjectGuid()))
183 DoZoneInCombat(bird, 200.0f);
184
185 me->InterruptSpell(CURRENT_CHANNELED_SPELL);
186 _events.Reset();
187 _events.ScheduleEvent(EVENT_FIEROBLAST, 1);
188 _events.ScheduleEvent(EVENT_FIEROCLAST_BARRAGE, 6000);
189 }
190
191 void JustReachedHome() override
192 {
193 AlysrazorTrashEvaded(me);
194 }
195
196 void MoveInLineOfSight(Unit* unit) override
197 {
198 if (me->IsInCombat())
199 return;
200
201 if (!unit->IsCharmedOwnedByPlayerOrPlayer())
202 return;
203
204 ScriptedAI::MoveInLineOfSight(unit);
205 }
206
207 void UpdateAI(uint32 diff) override
208 {
209 if (!me->IsInCombat())
210 if (!me->GetCurrentSpell(CURRENT_CHANNELED_SPELL))
211 if (Creature* fireBird = me->FindNearestCreature((me->GetHomePosition().GetPositionY() > -275.0f ? NPC_BLAZING_MONSTROSITY_LEFT : NPC_BLAZING_MONSTROSITY_RIGHT), 100.0f))
212 DoCast(fireBird, SPELL_FIRE_CHANNELING);
213
214 if (!UpdateVictim())
215 return;
216
217 _events.Update(diff);
218
219 if (me->HasUnitState(UNIT_STATE_CASTING))
220 return;
221
222 while (uint32 eventId = _events.ExecuteEvent())
223 {
224 switch (eventId)
225 {
226 case EVENT_FIEROBLAST:
227 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, false, -SPELL_RIDE_MONSTROSITY))
228 DoCast(target, SPELL_FIEROBLAST_TRASH);
229 _events.RescheduleEvent(EVENT_FIEROBLAST, 500); // cast time is longer, but thanks to UNIT_STATE_CASTING check it won't trigger more often (need this because this creature gets a stacking haste aura)
230 break;
231 case EVENT_FIEROCLAST_BARRAGE:
232 DoCastAOE(SPELL_FIEROCLAST_BARRAGE);
233 _events.ScheduleEvent(EVENT_FIEROCLAST_BARRAGE, urand(9000, 12000));
234 break;
235 }
236 }
237
238 DoMeleeAttackIfReady();
239 }
240
241 private:
242 EventMap _events;
243 };
244
245 CreatureAI* GetAI(Creature* creature) const override
246 {
247 return new npc_harbinger_of_flameAI(creature);
248 }

291 {
292 if (!apply)
293 return;
294
295 // Our passenger is another vehicle (boardable by players)
296 DoCast(passenger, SPELL_SHARE_HEALTH, true);
297 passenger->setFaction(35);
298 passenger->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
299
300 // Hack to relocate vehicle on vehicle so exiting players are not moved under map
301 Movement::MoveSplineInit init(passenger);
302 init.DisableTransportPathTransformations();
303 init.MoveTo(0.6654003f, 0.0f, 1.9815f);
304 init.SetFacing(0.0f);
305 init.Launch();
306 }
307
308 void JustSummoned(Creature* summon) override
309 {
310 _summons.Summon(summon);
311 }
312
313 void SummonedCreatureDespawn(Creature* summon) override
314 {
315 _summons.Despawn(summon);
316 }
317
318 void UpdateAI(uint32 diff) override
319 {
320 if (!UpdateVictim())
321 return;
322
323 _events.Update(diff);
324
325 while (uint32 eventId = _events.ExecuteEvent())
326 {
327 switch (eventId)
328 {
329 case EVENT_START_SPITTING:
330 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, false, -SPELL_RIDE_MONSTROSITY))
331 DoCast(target, SPELL_MOLTEN_BARRAGE);
332 break;
333 case EVENT_CONTINUE_SPITTING:
334 DoCastAOE(SPELL_MOLTEN_BARRAGE_EFFECT);
335 if (Creature* egg = me->FindNearestCreature(NPC_EGG_PILE, 100.0f))
336 egg->AI()->DoAction(me->GetEntry());
337 break;
338 }
339 }
340 }
341
342 private:
343 SummonList _summons;
344 EventMap _events;
345 };
346
347 CreatureAI* GetAI(Creature* creature) const override
348 {
349 return new npc_blazing_monstrosityAI(creature);

393class npc_egg_pile : public CreatureScript

395 public:
396 npc_egg_pile() : CreatureScript("npc_egg_pile") { }
397
398 struct npc_egg_pileAI : public CreatureAI
399 {
400 npc_egg_pileAI(Creature* creature) : CreatureAI(creature)
401 {
402 }
403
404 void AttackStart(Unit* /*target*/) override { }
405
406 void Reset() override
407 {
408 me->SetReactState(REACT_PASSIVE);
409 _events.Reset();
410 _callHatchlingSpell = 0;
411 }
412
413 void JustDied(Unit* /*killer*/) override
414 {
415 _events.Reset();
416 std::list<Creature*> eggs;
417 GetCreatureListWithEntryInGrid(eggs, me, NPC_MOLTEN_EGG_TRASH, 20.0f);
418 for (std::list<Creature*>::const_iterator itr = eggs.begin(); itr != eggs.end(); ++itr)
419 (*itr)->CastSpell(*itr, SPELL_ALYSRAZOR_COSMETIC_EGG_XPLOSION, TRIGGERED_FULL_MASK);
420
421 DoCast(me, SPELL_ALYSRAZOR_COSMETIC_EGG_XPLOSION, true);
422 }
423
424 void JustReachedHome() override
425 {
426 AlysrazorTrashEvaded(me);
427 }
428
429 void DoAction(int32 action) override
430 {
431 if (action != NPC_BLAZING_MONSTROSITY_LEFT &&
432 action != NPC_BLAZING_MONSTROSITY_RIGHT)
433 return;
434
435 if (action == NPC_BLAZING_MONSTROSITY_LEFT)
436 Talk(EMOTE_CRACKING_EGGS);
437
438 _callHatchlingSpell = (action == NPC_BLAZING_MONSTROSITY_LEFT) ? SPELL_MOLTEN_EGG_TRASH_CALL_L : SPELL_MOLTEN_EGG_TRASH_CALL_R;
439 DoZoneInCombat();
440 _events.Reset();
441 _events.ScheduleEvent(EVENT_SUMMON_SMOULDERING_HATCHLING, 1);
442 }
443
444 void UpdateAI(uint32 diff) override
445 {
446 if (!UpdateVictim())
447 return;
448
449 _events.Update(diff);
450
451 if (me->HasUnitState(UNIT_STATE_CASTING))
452 return;
453
454 while (uint32 eventId = _events.ExecuteEvent())
455 {
456 switch (eventId)
457 {
458 case EVENT_SUMMON_SMOULDERING_HATCHLING:
459 {
460 std::list<Creature*> eggs;
461 MoltenEggCheck check(me);
462 Trinity::CreatureListSearcher<MoltenEggCheck> searcher(me, eggs, check);
463 me->VisitNearbyGridObject(20.0f, searcher);
464 if (!eggs.empty())
465 {
466 Creature* egg = Trinity::Containers::SelectRandomContainerElement(eggs);
467 egg->CastSpell(egg, SPELL_SUMMON_SMOULDERING_HATCHLING, TRIGGERED_FULL_MASK);
468 egg->SetDisplayId(MODEL_INVISIBLE_STALKER);
469 egg->m_Events.AddEvent(new RespawnEggEvent(egg), egg->m_Events.CalculateTime(5000));
470 }
471
472 if (_callHatchlingSpell)
473 DoCastAOE(_callHatchlingSpell, true);
474 _events.ScheduleEvent(EVENT_SUMMON_SMOULDERING_HATCHLING, urand(6000, 10000));
475 break;
476 }
477 default:
478 break;
479 }
480 }
481
482 DoMeleeAttackIfReady();
483 }
484
485 private:
486 EventMap _events;
487 uint32 _callHatchlingSpell = 0;
488 };
489
490 CreatureAI* GetAI(Creature* creature) const override
491 {
492 return new npc_egg_pileAI(creature);
493 }

534 public:
535 spell_alysrazor_turn_monstrosity() : SpellScriptLoader("spell_alysrazor_turn_monstrosity") { }

537 class spell_alysrazor_turn_monstrosity_SpellScript : public SpellScript
538 {
539 PrepareSpellScript(spell_alysrazor_turn_monstrosity_SpellScript);
540
541 bool Validate(SpellInfo const* /*spellInfo*/) override
542 {
543 if (!sSpellMgr->GetSpellInfo(SPELL_GENERIC_DUMMY_CAST))
544 return false;
545 if (!sSpellMgr->GetSpellInfo(SPELL_KNOCKBACK_RIGHT))
546 return false;
547 if (!sSpellMgr->GetSpellInfo(SPELL_KNOCKBACK_LEFT))
548 return false;
549 if (!sSpellMgr->GetSpellInfo(SPELL_KNOCKBACK_FORWARD))
550 return false;
551 if (!sSpellMgr->GetSpellInfo(SPELL_KNOCKBACK_BACK))
552 return false;
553 return true;
554 }
555
556 void KnockBarrage(SpellEffIndex effIndex)
557 {
558 PreventHitDefaultEffect(effIndex);
559 GetHitUnit()->GetMotionMaster()->MoveIdle();
560 if (TempSummon* summ = GetHitUnit()->ToTempSummon())
561 if (Unit* summoner = summ->GetSummoner())
562 GetHitUnit()->CastSpell(summoner, SPELL_GENERIC_DUMMY_CAST, TRIGGERED_FULL_MASK);
563
564 float angle = 0.0f;
565 if (Unit* bird = GetCaster()->GetVehicleBase())
566 {
567 bird->SetInFront(GetHitUnit());
568 angle = bird->GetOrientation();
569 }
570
571 uint32 spellId = 0;
572 switch (GetSpellInfo()->Id)
573 {
574 case SPELL_RIGHT_SIDE_SMACK_R:
575 case SPELL_RIGHT_SIDE_SMACK_L:
576 spellId = SPELL_KNOCKBACK_RIGHT;
577 angle -= float(M_PI) * 0.5f;
578 break;
579 case SPELL_LEFT_SIDE_SMACK_R:
580 case SPELL_LEFT_SIDE_SMACK_L:
581 spellId = SPELL_KNOCKBACK_LEFT;
582 angle += float(M_PI) * 0.5f;
583 break;
584 case SPELL_HEAD_BONK_R:
585 case SPELL_HEAD_BONK_L:
586 spellId = SPELL_KNOCKBACK_FORWARD;
587 break;
588 case SPELL_TICKLE_R:
589 case SPELL_TICKLE_L:
590 spellId = SPELL_KNOCKBACK_BACK;
591 angle -= float(M_PI);
592 break;
593 }
594
595 // Cannot wait for object update to process facing spline, it's needed in next spell cast
596 GetHitUnit()->SetOrientation(angle);
597 GetHitUnit()->SetFacingTo(angle);
598 GetHitUnit()->AddUnitState(UNIT_STATE_CANNOT_TURN);
599 GetHitUnit()->CastSpell(GetHitUnit(), spellId, TRIGGERED_FULL_MASK);
600 }
601
602 void TurnBird(SpellEffIndex effIndex)
603 {
604 PreventHitDefaultEffect(effIndex);
605 GetHitUnit()->PlayOneShotAnimKitId(ANIM_KIT_BIRD_TURN);
606 }
607
608 void Register() override
609 {
610 OnEffectHitTarget += SpellEffectFn(spell_alysrazor_turn_monstrosity_SpellScript::KnockBarrage, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
611 OnEffectHitTarget += SpellEffectFn(spell_alysrazor_turn_monstrosity_SpellScript::TurnBird, EFFECT_1, SPELL_EFFECT_SCRIPT_EFFECT);
612 }
613 };
614
615 SpellScript* GetSpellScript() const override
616 {
617 return new spell_alysrazor_turn_monstrosity_SpellScript();
618 }

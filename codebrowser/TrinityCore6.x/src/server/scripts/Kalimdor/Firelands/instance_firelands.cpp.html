18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "firelands.h"
21
22class instance_firelands : public InstanceMapScript
23{
24 public:
25 instance_firelands() : InstanceMapScript(FirelandsScriptName, 720) { }
26
27 struct instance_firelands_InstanceScript : public InstanceScript
28 {
29 instance_firelands_InstanceScript(InstanceMap* map) : InstanceScript(map)
30 {
31 SetHeaders(DataHeader);
32 SetBossNumber(EncounterCount);
33 }
34
35 void OnCreatureCreate(Creature* creature) override
36 {
37 switch (creature->GetEntry())
38 {
39 case NPC_SMOULDERING_HATCHLING:
40 // Cannot directly start attacking here as the creature is not yet on map
41 creature->m_Events.AddEvent(new DelayedAttackStartEvent(creature), creature->m_Events.CalculateTime(500));
42 break;
43 }
44 }
45 };
46
47 InstanceScript* GetInstanceScript(InstanceMap* map) const override
48 {
49 return new instance_firelands_InstanceScript(map);
50 }
51};
52
53void AddSC_instance_firelands()
54{
55 new instance_firelands();
56}

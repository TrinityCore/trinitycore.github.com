62class npc_sergeant_bly : public CreatureScript
63{
64public:
65 npc_sergeant_bly() : CreatureScript("npc_sergeant_bly") { }
66
67 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
68 {
69 player->PlayerTalkClass->ClearMenus();
70 if (action == GOSSIP_ACTION_INFO_DEF+1)
71 {
72 player->CLOSE_GOSSIP_MENU();
73 ENSURE_AI(npc_sergeant_bly::npc_sergeant_blyAI, creature->AI())->PlayerGUID = player->GetGUID();
74 creature->AI()->DoAction(0);
75 }
76 return true;
77 }
78
79 bool OnGossipHello(Player* player, Creature* creature) override
80 {
81 if (InstanceScript* instance = creature->GetInstanceScript())
82 {
83 if (instance->GetData(EVENT_PYRAMID) == PYRAMID_KILLED_ALL_TROLLS)
84 {
85 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_BLY, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
86 player->SEND_GOSSIP_MENU(1517, creature->GetGUID());
87 }
88 else
89 if (instance->GetData(EVENT_PYRAMID) == PYRAMID_NOT_STARTED)
90 player->SEND_GOSSIP_MENU(1515, creature->GetGUID());
91 else
92 player->SEND_GOSSIP_MENU(1516, creature->GetGUID());
93 return true;
94 }
95 return false;
96 }

99 {
100 return GetInstanceAI<npc_sergeant_blyAI>(creature);

103 struct npc_sergeant_blyAI : public ScriptedAI
104 {
105 npc_sergeant_blyAI(Creature* creature) : ScriptedAI(creature)
106 {
107 Initialize();
108 instance = creature->GetInstanceScript();
109 postGossipStep = 0;
110 Text_Timer = 0;
111 }
112
113 void Initialize()
114 {
115 ShieldBash_Timer = 5000;
116 Revenge_Timer = 8000;
117 }
118
119 InstanceScript* instance;
120
121 uint32 postGossipStep;
122 uint32 Text_Timer;
123 uint32 ShieldBash_Timer;
124 uint32 Revenge_Timer; //this is wrong, spell should never be used unless me->GetVictim() dodge, parry or block attack. Trinity support required.
125 ObjectGuid PlayerGUID;
126
127 void Reset() override
128 {
129 Initialize();
130
131 me->setFaction(FACTION_FRIENDLY);
132 }
133
134 void UpdateAI(uint32 diff) override
135 {
136 if (postGossipStep>0 && postGossipStep<4)
137 {
138 if (Text_Timer<diff)
139 {
140 switch (postGossipStep)
141 {
142 case 1:
143 //weegli doesn't fight - he goes & blows up the door
144 if (Creature* pWeegli = instance->instance->GetCreature(instance->GetGuidData(ENTRY_WEEGLI)))
145 pWeegli->AI()->DoAction(0);
146 Talk(SAY_1);
147 Text_Timer = 5000;
148 break;
149 case 2:
150 Talk(SAY_2);
151 Text_Timer = 5000;
152 break;
153 case 3:
154 me->setFaction(FACTION_HOSTILE);
155 if (Player* target = ObjectAccessor::GetPlayer(*me, PlayerGUID))
156 AttackStart(target);
157
158 switchFactionIfAlive(ENTRY_RAVEN);
159 switchFactionIfAlive(ENTRY_ORO);
160 switchFactionIfAlive(ENTRY_MURTA);
161 }
162 postGossipStep++;
163 }
164 else Text_Timer -= diff;
165 }
166
167 if (!UpdateVictim())
168 return;
169
170 if (ShieldBash_Timer <= diff)
171 {
172 DoCastVictim(SPELL_SHIELD_BASH);
173 ShieldBash_Timer = 15000;
174 }
175 else
176 ShieldBash_Timer -= diff;
177
178 if (Revenge_Timer <= diff)
179 {
180 DoCastVictim(SPELL_REVENGE);
181 Revenge_Timer = 10000;
182 }
183 else
184 Revenge_Timer -= diff;
185
186 DoMeleeAttackIfReady();
187 }
188
189 void DoAction(int32 /*param*/) override
190 {
191 postGossipStep=1;
192 Text_Timer = 0;
193 }
194
195 void switchFactionIfAlive(uint32 entry)
196 {
197 if (Creature* crew = ObjectAccessor::GetCreature(*me, instance->GetGuidData(entry)))
198 if (crew->IsAlive())
199 crew->setFaction(FACTION_HOSTILE);
200 }
201 };
202
203};

210{

213
214 bool OnGossipHello(Player* /*player*/, GameObject* go) override
215 {
216 if (InstanceScript* instance = go->GetInstanceScript())
217 {
218 instance->SetData(EVENT_PYRAMID, PYRAMID_CAGES_OPEN);
219 //set bly & co to aggressive & start moving to top of stairs
220 initBlyCrewMember(instance, ENTRY_BLY, 1884.99f, 1263, 41.52f);
221 initBlyCrewMember(instance, ENTRY_RAVEN, 1882.5f, 1263, 41.52f);
222 initBlyCrewMember(instance, ENTRY_ORO, 1886.47f, 1270.68f, 41.68f);
223 initBlyCrewMember(instance, ENTRY_WEEGLI, 1890, 1263, 41.52f);
224 initBlyCrewMember(instance, ENTRY_MURTA, 1891.19f, 1272.03f, 41.60f);
225 }
226 return false;
227 }
228
229private:
230 void initBlyCrewMember(InstanceScript* instance, uint32 entry, float x, float y, float z)
231 {
232 if (Creature* crew = instance->instance->GetCreature(instance->GetGuidData(entry)))
233 {
234 crew->SetReactState(REACT_AGGRESSIVE);
235 crew->SetWalk(true);
236 crew->SetHomePosition(x, y, z, 0);
237 crew->GetMotionMaster()->MovePoint(1, x, y, z);
238 crew->setFaction(FACTION_FREED);
239 }
240 }

336 void UpdateAI(uint32 diff) override
337 {
338 if (!UpdateVictim())
339 return;
340
341 if (Bomb_Timer < diff)
342 {
343 DoCastVictim(SPELL_BOMB);
344 Bomb_Timer = 10000;
345 }
346 else
347 Bomb_Timer -= diff;
348
349 if (me->isAttackReady() && !me->IsWithinMeleeRange(me->GetVictim()))
350 {
351 DoCastVictim(SPELL_SHOOT);
352 me->SetSheath(SHEATH_STATE_RANGED);

354 else
355 {
356 me->SetSheath(SHEATH_STATE_MELEE);
357 DoMeleeAttackIfReady();
358 }

362 {

366 Talk(SAY_WEEGLI_OHNO);

410class go_shallow_grave : public GameObjectScript
411{
412public:
413 go_shallow_grave() : GameObjectScript("go_shallow_grave") { }
414
415 bool OnGossipHello(Player* /*player*/, GameObject* go) override
416 {
417 // randomly summon a zombie or dead hero the first time a grave is used
418 if (go->GetUseCount() == 0)
419 {
420 uint32 randomchance = urand(0, 100);
421 if (randomchance < CHANCE_ZOMBIE)
422 go->SummonCreature(NPC_ZOMBIE, go->GetPositionX(), go->GetPositionY(), go->GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 30000);
423 else
424 if ((randomchance - CHANCE_ZOMBIE) < CHANCE_DEAD_HERO)
425 go->SummonCreature(NPC_DEAD_HERO, go->GetPositionX(), go->GetPositionY(), go->GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 30000);
426 }
427 go->AddUse();
428 return false;
429 }

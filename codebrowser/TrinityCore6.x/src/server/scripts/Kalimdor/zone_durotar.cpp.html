33enum LazyPeon
34{
35 QUEST_LAZY_PEONS = 25134,

37 SPELL_BUFF_SLEEP = 17743,
38 SPELL_AWAKEN_PEON = 19938

41class npc_lazy_peon : public CreatureScript
42{
43public:
44 npc_lazy_peon() : CreatureScript("npc_lazy_peon") { }
45
46 CreatureAI* GetAI(Creature* creature) const override
47 {
48 return new npc_lazy_peonAI(creature);
49 }
50
51 struct npc_lazy_peonAI : public ScriptedAI
52 {
53 npc_lazy_peonAI(Creature* creature) : ScriptedAI(creature)
54 {
55 Initialize();
56 }
57
58 void Initialize()
59 {
60 RebuffTimer = 0;
61 work = false;
62 }
63
64 uint32 RebuffTimer;
65 bool work;
66
67 void Reset() override
68 {
69 Initialize();
70 }
71
72 void MovementInform(uint32 /*type*/, uint32 id) override
73 {
74 if (id == 1)
75 work = true;
76 }
77
78 void SpellHit(Unit* caster, const SpellInfo* spell) override
79 {
80 if (spell->Id != SPELL_AWAKEN_PEON)
81 return;
82
83 Player* player = caster->ToPlayer();
84 if (player && player->GetQuestStatus(QUEST_LAZY_PEONS) == QUEST_STATUS_INCOMPLETE)
85 {
86 player->KilledMonsterCredit(me->GetEntry(), me->GetGUID());
87 Talk(SAY_SPELL_HIT, caster);
88 me->RemoveAllAuras();
89 if (GameObject* Lumberpile = me->FindNearestGameObject(GO_LUMBERPILE, 20))
90 me->GetMotionMaster()->MovePoint(1, Lumberpile->GetPositionX()-1, Lumberpile->GetPositionY(), Lumberpile->GetPositionZ());
91 }
92 }
93
94 void UpdateAI(uint32 diff) override
95 {
96 if (work == true)
97 me->HandleEmoteCommand(EMOTE_ONESHOT_WORK_CHOPWOOD);
98 if (RebuffTimer <= diff)
99 {
100 DoCast(me, SPELL_BUFF_SLEEP);
101 RebuffTimer = 300000; //Rebuff agian in 5 minutes
102 }
103 else
104 RebuffTimer -= diff;
105 if (!UpdateVictim())
106 return;
107 DoMeleeAttackIfReady();
108 }
109 };
110};

123// 17009
124class spell_voodoo : public SpellScriptLoader
125{
126 public:
127 spell_voodoo() : SpellScriptLoader("spell_voodoo") { }
128
129 class spell_voodoo_SpellScript : public SpellScript
130 {
131 PrepareSpellScript(spell_voodoo_SpellScript);
132
133 bool Validate(SpellInfo const* /*spellInfo*/) override
134 {
135 if (!sSpellMgr->GetSpellInfo(SPELL_BREW) || !sSpellMgr->GetSpellInfo(SPELL_GHOSTLY) ||
136 !sSpellMgr->GetSpellInfo(SPELL_HEX1) || !sSpellMgr->GetSpellInfo(SPELL_HEX2) ||
137 !sSpellMgr->GetSpellInfo(SPELL_HEX3) || !sSpellMgr->GetSpellInfo(SPELL_GROW) ||
138 !sSpellMgr->GetSpellInfo(SPELL_LAUNCH))
139 return false;
140 return true;
141 }
142
143 void HandleDummy(SpellEffIndex /*effIndex*/)
144 {
145 uint32 spellid = RAND(SPELL_BREW, SPELL_GHOSTLY, RAND(SPELL_HEX1, SPELL_HEX2, SPELL_HEX3), SPELL_GROW, SPELL_LAUNCH);
146 if (Unit* target = GetHitUnit())
147 GetCaster()->CastSpell(target, spellid, false);
148 }
149
150 void Register() override
151 {
152 OnEffectHitTarget += SpellEffectFn(spell_voodoo_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
153 }
154 };
155
156 SpellScript* GetSpellScript() const override
157 {
158 return new spell_voodoo_SpellScript();
159 }
160};

38enum DyingKodo
39{
40 SAY_SMEED_HOME = 0,
41
42 QUEST_KODO = 5561,
43
44 NPC_SMEED = 11596,
45 NPC_AGED_KODO = 4700,
46 NPC_DYING_KODO = 4701,
47 NPC_ANCIENT_KODO = 4702,
48 NPC_TAMED_KODO = 11627,
49
50 SPELL_KODO_KOMBO_ITEM = 18153,
51 SPELL_KODO_KOMBO_PLAYER_BUFF = 18172,
52 SPELL_KODO_KOMBO_DESPAWN_BUFF = 18377,
53 SPELL_KODO_KOMBO_GOSSIP = 18362
54

57class npc_aged_dying_ancient_kodo : public CreatureScript
58{
59public:
60 npc_aged_dying_ancient_kodo() : CreatureScript("npc_aged_dying_ancient_kodo") { }
61
62 struct npc_aged_dying_ancient_kodoAI : public ScriptedAI
63 {
64 npc_aged_dying_ancient_kodoAI(Creature* creature) : ScriptedAI(creature) { }
65
66 void MoveInLineOfSight(Unit* who) override
67 {
68 if (who->GetEntry() == NPC_SMEED && me->IsWithinDistInMap(who, 10.0f) && !me->HasAura(SPELL_KODO_KOMBO_GOSSIP))
69 {
70 me->GetMotionMaster()->Clear();
71 DoCast(me, SPELL_KODO_KOMBO_GOSSIP, true);
72 if (Creature* smeed = who->ToCreature())
73 smeed->AI()->Talk(SAY_SMEED_HOME);
74 }
75 }
76
77 void SpellHit(Unit* caster, SpellInfo const* spell) override
78 {
79 if (spell->Id == SPELL_KODO_KOMBO_ITEM)
80 {
81 if (!(caster->HasAura(SPELL_KODO_KOMBO_PLAYER_BUFF) || me->HasAura(SPELL_KODO_KOMBO_DESPAWN_BUFF))
82 && (me->GetEntry() == NPC_AGED_KODO || me->GetEntry() == NPC_DYING_KODO || me->GetEntry() == NPC_ANCIENT_KODO))
83 {
84 caster->CastSpell(caster, SPELL_KODO_KOMBO_PLAYER_BUFF, true);
85 DoCast(me, SPELL_KODO_KOMBO_DESPAWN_BUFF, true);
86
87 me->UpdateEntry(NPC_TAMED_KODO);
88 me->CombatStop();
89 me->DeleteThreatList();
90 me->SetSpeed(MOVE_RUN, 0.6f, true);
91 me->GetMotionMaster()->MoveFollow(caster, PET_FOLLOW_DIST, me->GetFollowAngle());
92 me->setActive(true);
93 }
94 }
95 else if (spell->Id == SPELL_KODO_KOMBO_GOSSIP)
96 {
97 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
98 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
99 me->GetMotionMaster()->Clear();
100 me->GetMotionMaster()->MoveIdle();
101 me->setActive(false);
102 me->DespawnOrUnsummon(60000);
103 }
104 }
105 };
106
107 bool OnGossipHello(Player* player, Creature* creature) override
108 {
109 if (player->HasAura(SPELL_KODO_KOMBO_PLAYER_BUFF) && creature->HasAura(SPELL_KODO_KOMBO_DESPAWN_BUFF))
110 {
111 player->TalkedToCreature(creature->GetEntry(), ObjectGuid::Empty);
112 player->RemoveAurasDueToSpell(SPELL_KODO_KOMBO_PLAYER_BUFF);
113 }
114
115 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
116 return true;
117 }
118
119 CreatureAI* GetAI(Creature* creature) const override
120 {
121 return new npc_aged_dying_ancient_kodoAI(creature);
122 }
123
124};

136class go_demon_portal : public GameObjectScript
137{
138 public:
139 go_demon_portal() : GameObjectScript("go_demon_portal") { }
140
141 bool OnGossipHello(Player* player, GameObject* go) override
142 {
143 if (player->GetQuestStatus(QUEST_PORTAL_OF_THE_LEGION) == QUEST_STATUS_INCOMPLETE && !go->FindNearestCreature(NPC_DEMON_GUARDIAN, 5.0f, true))
144 {
145 if (Creature* guardian = player->SummonCreature(NPC_DEMON_GUARDIAN, go->GetPositionX(), go->GetPositionY(), go->GetPositionZ(), 0.0f, TEMPSUMMON_DEAD_DESPAWN, 0))
146 guardian->AI()->AttackStart(player);
147 }
148
149 return true;
150 }
151};

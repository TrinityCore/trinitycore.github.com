45class boss_kazrogal : public CreatureScript
46{
47public:
48 boss_kazrogal() : CreatureScript("boss_kazrogal") { }
49
50 CreatureAI* GetAI(Creature* creature) const override
51 {
52 return GetInstanceAI<boss_kazrogalAI>(creature);
53 }
54
55 struct boss_kazrogalAI : public hyjal_trashAI
56 {
57 boss_kazrogalAI(Creature* creature) : hyjal_trashAI(creature)
58 {
59 Initialize();
60 instance = creature->GetInstanceScript();
61 go = false;
62 }
63
64 void Initialize()
65 {
66 damageTaken = 0;
67 CleaveTimer = 5000;
68 WarStompTimer = 15000;
69 MarkTimer = 45000;
70 MarkTimerBase = 45000;
71 }
72
73 uint32 CleaveTimer;
74 uint32 WarStompTimer;
75 uint32 MarkTimer;
76 uint32 MarkTimerBase;
77 bool go;
78
79 void Reset() override
80 {
81 Initialize();
82
83 if (IsEvent)
84 instance->SetData(DATA_KAZROGALEVENT, NOT_STARTED);
85 }
86
87 void EnterCombat(Unit* /*who*/) override
88 {
89 if (IsEvent)
90 instance->SetData(DATA_KAZROGALEVENT, IN_PROGRESS);
91 Talk(SAY_ONAGGRO);
92 }
93
94 void KilledUnit(Unit* /*victim*/) override
95 {
96 Talk(SAY_ONSLAY);
97 }
98
99 void WaypointReached(uint32 waypointId) override
100 {
101 if (waypointId == 7 && instance)
102 {
103 Unit* target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_THRALL));
104 if (target && target->IsAlive())
105 me->AddThreat(target, 0.0f);
106 }
107 }
108
109 void JustDied(Unit* killer) override
110 {
111 hyjal_trashAI::JustDied(killer);
112 if (IsEvent)
113 instance->SetData(DATA_KAZROGALEVENT, DONE);
114 DoPlaySoundToSet(me, SOUND_ONDEATH);
115 }
116
117 void UpdateAI(uint32 diff) override
118 {
119 if (IsEvent)
120 {
121 //Must update npc_escortAI
122 npc_escortAI::UpdateAI(diff);
123 if (!go)
124 {
125 go = true;
126 AddWaypoint(0, 5492.91f, -2404.61f, 1462.63f);
127 AddWaypoint(1, 5531.76f, -2460.87f, 1469.55f);
128 AddWaypoint(2, 5554.58f, -2514.66f, 1476.12f);
129 AddWaypoint(3, 5554.16f, -2567.23f, 1479.90f);
130 AddWaypoint(4, 5540.67f, -2625.99f, 1480.89f);
131 AddWaypoint(5, 5508.16f, -2659.2f, 1480.15f);
132 AddWaypoint(6, 5489.62f, -2704.05f, 1482.18f);
133 AddWaypoint(7, 5457.04f, -2726.26f, 1485.10f);
134 Start(false, true);
135 SetDespawnAtEnd(false);
136 }
137 }
138
139 //Return since we have no target
140 if (!UpdateVictim())
141 return;
142
143 if (CleaveTimer <= diff)
144 {
145 DoCast(me, SPELL_CLEAVE);
146 CleaveTimer = 6000 + rand32() % 15000;
147 } else CleaveTimer -= diff;
148
149 if (WarStompTimer <= diff)
150 {
151 DoCast(me, SPELL_WARSTOMP);
152 WarStompTimer = 60000;
153 } else WarStompTimer -= diff;
154
155 if (MarkTimer <= diff)
156 {
157 DoCastAOE(SPELL_MARK);
158
159 MarkTimerBase -= 5000;
160 if (MarkTimerBase < 5500)
161 MarkTimerBase = 5500;
162 MarkTimer = MarkTimerBase;
163 Talk(SAY_MARK);
164 } else MarkTimer -= diff;
165
166 DoMeleeAttackIfReady();
167 }
168 };
169
170};

172class MarkTargetFilter
173{
174 public:
175 bool operator()(WorldObject* target) const
176 {
177 if (Unit* unit = target->ToUnit())
178 return unit->getPowerType() != POWER_MANA;
179 return false;
180 }
181};

183class spell_mark_of_kazrogal : public SpellScriptLoader
184{
185 public:
186 spell_mark_of_kazrogal() : SpellScriptLoader("spell_mark_of_kazrogal") { }
187
188 class spell_mark_of_kazrogal_SpellScript : public SpellScript
189 {
190 PrepareSpellScript(spell_mark_of_kazrogal_SpellScript);
191
192 void FilterTargets(std::list<WorldObject*>& targets)
193 {
194 targets.remove_if(MarkTargetFilter());
195 }
196
197 void Register() override
198 {
199 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_mark_of_kazrogal_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
200 }
201 };
202
203 class spell_mark_of_kazrogal_AuraScript : public AuraScript
204 {
205 PrepareAuraScript(spell_mark_of_kazrogal_AuraScript);
206
207 bool Validate(SpellInfo const* /*spell*/) override
208 {
209 if (!sSpellMgr->GetSpellInfo(SPELL_MARK_DAMAGE))
210 return false;
211 return true;
212 }
213
214 void OnPeriodic(AuraEffect const* aurEff)
215 {
216 Unit* target = GetTarget();
217
218 if (target->GetPower(POWER_MANA) == 0)
219 {
220 target->CastSpell(target, SPELL_MARK_DAMAGE, true, NULL, aurEff);
221 // Remove aura
222 SetDuration(0);
223 }
224 }
225
226 void Register() override
227 {
228 OnEffectPeriodic += AuraEffectPeriodicFn(spell_mark_of_kazrogal_AuraScript::OnPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_MANA_LEECH);
229 }
230 };
231
232 SpellScript* GetSpellScript() const override
233 {
234 return new spell_mark_of_kazrogal_SpellScript();
235 }
236
237 AuraScript* GetAuraScript() const override
238 {
239 return new spell_mark_of_kazrogal_AuraScript();
240 }
241};

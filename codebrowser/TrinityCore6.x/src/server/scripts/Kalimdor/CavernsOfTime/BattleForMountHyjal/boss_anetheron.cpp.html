35 SAY_ONDEATH = 0,
36 SAY_ONSLAY = 1,
37 SAY_SWARM = 2,
38 SAY_SLEEP = 3,
39 SAY_INFERNO = 4,
40 SAY_ONAGGRO = 5,
41};

43class boss_anetheron : public CreatureScript
44{
45public:
46 boss_anetheron() : CreatureScript("boss_anetheron") { }
47
48 CreatureAI* GetAI(Creature* creature) const override
49 {
50 return GetInstanceAI<boss_anetheronAI>(creature);
51 }
52
53 struct boss_anetheronAI : public hyjal_trashAI
54 {
55 boss_anetheronAI(Creature* creature) : hyjal_trashAI(creature)
56 {
57 Initialize();
58 instance = creature->GetInstanceScript();
59 go = false;
60 }
61
62 void Initialize()
63 {
64 SwarmTimer = 45000;
65 SleepTimer = 60000;
66 AuraTimer = 5000;
67 InfernoTimer = 45000;
68 damageTaken = 0;
69 }
70
71 uint32 SwarmTimer;
72 uint32 SleepTimer;
73 uint32 AuraTimer;
74 uint32 InfernoTimer;
75 bool go;
76
77 void Reset() override
78 {
79 Initialize();
80
81 if (IsEvent)
82 instance->SetData(DATA_ANETHERONEVENT, NOT_STARTED);
83 }
84
85 void EnterCombat(Unit* /*who*/) override
86 {
87 if (IsEvent)
88 instance->SetData(DATA_ANETHERONEVENT, IN_PROGRESS);
89
90 Talk(SAY_ONAGGRO);
91 }
92
93 void KilledUnit(Unit* who) override
94 {
95 if (who->GetTypeId() == TYPEID_PLAYER)
96 Talk(SAY_ONSLAY);
97 }
98
99 void WaypointReached(uint32 waypointId) override
100 {
101 if (waypointId == 7)
102 {
103 Unit* target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_JAINAPROUDMOORE));
104 if (target && target->IsAlive())
105 me->AddThreat(target, 0.0f);
106 }
107 }
108
109 void JustDied(Unit* killer) override
110 {
111 hyjal_trashAI::JustDied(killer);
112 if (IsEvent)
113 instance->SetData(DATA_ANETHERONEVENT, DONE);
114 Talk(SAY_ONDEATH);
115 }
116
117 void UpdateAI(uint32 diff) override
118 {
119 if (IsEvent)
120 {
121 //Must update npc_escortAI
122 npc_escortAI::UpdateAI(diff);
123 if (!go)
124 {
125 go = true;
126 AddWaypoint(0, 4896.08f, -1576.35f, 1333.65f);
127 AddWaypoint(1, 4898.68f, -1615.02f, 1329.48f);
128 AddWaypoint(2, 4907.12f, -1667.08f, 1321.00f);
129 AddWaypoint(3, 4963.18f, -1699.35f, 1340.51f);
130 AddWaypoint(4, 4989.16f, -1716.67f, 1335.74f);
131 AddWaypoint(5, 5026.27f, -1736.89f, 1323.02f);
132 AddWaypoint(6, 5037.77f, -1770.56f, 1324.36f);
133 AddWaypoint(7, 5067.23f, -1789.95f, 1321.17f);
134 Start(false, true);
135 SetDespawnAtEnd(false);
136 }
137 }
138
139 //Return since we have no target
140 if (!UpdateVictim())
141 return;
142
143 if (SwarmTimer <= diff)
144 {
145 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
146 DoCast(target, SPELL_CARRION_SWARM);
147
148 SwarmTimer = urand(45000, 60000);
149 Talk(SAY_SWARM);
150 } else SwarmTimer -= diff;
151
152 if (SleepTimer <= diff)
153 {
154 for (uint8 i = 0; i < 3; ++i)
155 {
156 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
157 target->CastSpell(target, SPELL_SLEEP, true);
158 }
159 SleepTimer = 60000;
160 Talk(SAY_SLEEP);
161 } else SleepTimer -= diff;
162 if (AuraTimer <= diff)
163 {
164 DoCast(me, SPELL_VAMPIRIC_AURA, true);
165 AuraTimer = urand(10000, 20000);
166 } else AuraTimer -= diff;
167 if (InfernoTimer <= diff)
168 {
169 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_INFERNO);
170 InfernoTimer = 45000;
171 Talk(SAY_INFERNO);
172 } else InfernoTimer -= diff;
173
174 DoMeleeAttackIfReady();
175 }
176 };
177
178};

180class npc_towering_infernal : public CreatureScript
181{
182public:
183 npc_towering_infernal() : CreatureScript("npc_towering_infernal") { }
184
185 CreatureAI* GetAI(Creature* creature) const override
186 {
187 return GetInstanceAI<npc_towering_infernalAI>(creature);
188 }
189
190 struct npc_towering_infernalAI : public ScriptedAI
191 {
192 npc_towering_infernalAI(Creature* creature) : ScriptedAI(creature)
193 {
194 ImmolationTimer = 5000;
195 CheckTimer = 5000;
196 instance = creature->GetInstanceScript();
197 AnetheronGUID = instance->GetGuidData(DATA_ANETHERON);
198 }
199
200 uint32 ImmolationTimer;
201 uint32 CheckTimer;
202 ObjectGuid AnetheronGUID;
203 InstanceScript* instance;
204
205 void Reset() override
206 {
207 DoCast(me, SPELL_INFERNO_EFFECT);
208 ImmolationTimer = 5000;
209 CheckTimer = 5000;
210 }
211
212 void EnterCombat(Unit* /*who*/) override
213 {
214 }
215
216 void KilledUnit(Unit* /*victim*/) override
217 {
218 }
219
220 void JustDied(Unit* /*killer*/) override
221 {
222 }
223
224 void MoveInLineOfSight(Unit* who) override
225
226 {
227 if (me->IsWithinDist(who, 50) && !me->IsInCombat() && me->IsValidAttackTarget(who))
228 AttackStart(who);
229 }
230
231 void UpdateAI(uint32 diff) override
232 {
233 if (CheckTimer <= diff)
234 {
235 if (!AnetheronGUID.IsEmpty())
236 {
237 Creature* boss = ObjectAccessor::GetCreature(*me, AnetheronGUID);
238 if (!boss || boss->isDead())
239 {
240 me->setDeathState(JUST_DIED);
241 me->RemoveCorpse();
242 return;
243 }
244 }
245 CheckTimer = 5000;
246 } else CheckTimer -= diff;
247
248 //Return since we have no target
249 if (!UpdateVictim())
250 return;
251
252 if (ImmolationTimer <= diff)
253 {
254 DoCast(me, SPELL_IMMOLATION);
255 ImmolationTimer = 5000;
256 } else ImmolationTimer -= diff;
257
258 DoMeleeAttackIfReady();
259 }
260 };
261
262};
263
264void AddSC_boss_anetheron()
265{
266 new boss_anetheron();
267 new npc_towering_infernal();
268}

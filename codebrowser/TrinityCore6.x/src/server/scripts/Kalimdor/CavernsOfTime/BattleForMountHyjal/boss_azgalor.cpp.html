44class boss_azgalor : public CreatureScript
45{
46public:
47 boss_azgalor() : CreatureScript("boss_azgalor") { }
48
49 CreatureAI* GetAI(Creature* creature) const override
50 {
51 return GetInstanceAI<boss_azgalorAI>(creature);
52 }
53
54 struct boss_azgalorAI : public hyjal_trashAI
55 {
56 boss_azgalorAI(Creature* creature) : hyjal_trashAI(creature)
57 {
58 Initialize();
59 instance = creature->GetInstanceScript();
60 go = false;
61 }
62
63 void Initialize()
64 {
65 damageTaken = 0;
66 RainTimer = 20000;
67 DoomTimer = 50000;
68 HowlTimer = 30000;
69 CleaveTimer = 10000;
70 EnrageTimer = 600000;
71 enraged = false;
72 }
73
74 uint32 RainTimer;
75 uint32 DoomTimer;
76 uint32 HowlTimer;
77 uint32 CleaveTimer;
78 uint32 EnrageTimer;
79 bool enraged;
80
81 bool go;
82
83 void Reset() override
84 {
85 Initialize();
86
87 if (IsEvent)
88 instance->SetData(DATA_AZGALOREVENT, NOT_STARTED);
89 }
90
91 void EnterCombat(Unit* /*who*/) override
92 {
93 if (IsEvent)
94 instance->SetData(DATA_AZGALOREVENT, IN_PROGRESS);
95
96 Talk(SAY_ONAGGRO);
97 }
98
99 void KilledUnit(Unit* /*victim*/) override
100 {
101 Talk(SAY_ONSLAY);
102 }
103
104 void WaypointReached(uint32 waypointId) override
105 {
106 if (waypointId == 7 && instance)
107 {
108 Unit* target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_THRALL));
109 if (target && target->IsAlive())
110 me->AddThreat(target, 0.0f);
111 }
112 }
113
114 void JustDied(Unit* killer) override
115 {
116 hyjal_trashAI::JustDied(killer);
117 if (IsEvent)
118 instance->SetData(DATA_AZGALOREVENT, DONE);
119 Talk(SAY_ONDEATH);
120 }
121
122 void UpdateAI(uint32 diff) override
123 {
124 if (IsEvent)
125 {
126 //Must update npc_escortAI
127 npc_escortAI::UpdateAI(diff);
128 if (!go)
129 {
130 go = true;
131 AddWaypoint(0, 5492.91f, -2404.61f, 1462.63f);
132 AddWaypoint(1, 5531.76f, -2460.87f, 1469.55f);
133 AddWaypoint(2, 5554.58f, -2514.66f, 1476.12f);
134 AddWaypoint(3, 5554.16f, -2567.23f, 1479.90f);
135 AddWaypoint(4, 5540.67f, -2625.99f, 1480.89f);
136 AddWaypoint(5, 5508.16f, -2659.2f, 1480.15f);
137 AddWaypoint(6, 5489.62f, -2704.05f, 1482.18f);
138 AddWaypoint(7, 5457.04f, -2726.26f, 1485.10f);
139 Start(false, true);
140 SetDespawnAtEnd(false);
141 }
142 }
143
144 //Return since we have no target
145 if (!UpdateVictim())
146 return;
147
148 if (RainTimer <= diff)
149 {
150 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 30, true), SPELL_RAIN_OF_FIRE);
151 RainTimer = 20000 + rand32() % 15000;
152 } else RainTimer -= diff;
153
154 if (DoomTimer <= diff)
155 {
156 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true), SPELL_DOOM);//never on tank
157 DoomTimer = 45000 + rand32() % 5000;
158 } else DoomTimer -= diff;
159
160 if (HowlTimer <= diff)
161 {
162 DoCast(me, SPELL_HOWL_OF_AZGALOR);
163 HowlTimer = 30000;
164 } else HowlTimer -= diff;
165
166 if (CleaveTimer <= diff)
167 {
168 DoCastVictim(SPELL_CLEAVE);
169 CleaveTimer = 10000 + rand32() % 5000;
170 } else CleaveTimer -= diff;
171
172 if (EnrageTimer < diff && !enraged)
173 {
174 me->InterruptNonMeleeSpells(false);
175 DoCast(me, SPELL_BERSERK, true);
176 enraged = true;
177 EnrageTimer = 600000;
178 } else EnrageTimer -= diff;
179
180 DoMeleeAttackIfReady();
181 }
182 };
183
184};

186class npc_lesser_doomguard : public CreatureScript
187{
188public:
189 npc_lesser_doomguard() : CreatureScript("npc_lesser_doomguard") { }
190
191 CreatureAI* GetAI(Creature* creature) const override
192 {
193 return GetInstanceAI<npc_lesser_doomguardAI>(creature);
194 }
195
196 struct npc_lesser_doomguardAI : public hyjal_trashAI
197 {
198 npc_lesser_doomguardAI(Creature* creature) : hyjal_trashAI(creature)
199 {
200 CrippleTimer = 50000;
201 WarstompTimer = 10000;
202 CheckTimer = 5000;
203 AzgalorGUID = instance->GetGuidData(DATA_AZGALOR);
204 }
205
206 uint32 CrippleTimer;
207 uint32 WarstompTimer;
208 uint32 CheckTimer;
209 ObjectGuid AzgalorGUID;
210
211 void Reset() override
212 {
213 CrippleTimer = 50000;
214 WarstompTimer = 10000;
215 DoCast(me, SPELL_THRASH);
216 CheckTimer = 5000;
217 }
218
219 void EnterCombat(Unit* /*who*/) override
220 {
221 }
222
223 void KilledUnit(Unit* /*victim*/) override
224 {
225 }
226
227 void WaypointReached(uint32 /*waypointId*/) override
228 {
229 }
230
231 void MoveInLineOfSight(Unit* who) override
232
233 {
234 if (me->IsWithinDist(who, 50) && !me->IsInCombat() && me->IsValidAttackTarget(who))
235 AttackStart(who);
236 }
237
238 void JustDied(Unit* /*killer*/) override
239 {
240 }
241
242 void UpdateAI(uint32 diff) override
243 {
244 if (CheckTimer <= diff)
245 {
246 if (!AzgalorGUID.IsEmpty())
247 {
248 Creature* boss = ObjectAccessor::GetCreature(*me, AzgalorGUID);
249 if (!boss || boss->isDead())
250 {
251 me->setDeathState(JUST_DIED);
252 me->RemoveCorpse();
253 return;
254 }
255 }
256 CheckTimer = 5000;
257 } else CheckTimer -= diff;
258
259 //Return since we have no target
260 if (!UpdateVictim())
261 return;
262
263 if (WarstompTimer <= diff)
264 {
265 DoCast(me, SPELL_WARSTOMP);
266 WarstompTimer = 10000 + rand32() % 5000;
267 } else WarstompTimer -= diff;
268
269 if (CrippleTimer <= diff)
270 {
271 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_CRIPPLE);
272 CrippleTimer = 25000 + rand32() % 5000;
273 } else CrippleTimer -= diff;
274
275 DoMeleeAttackIfReady();
276 }
277 };
278
279};
280

283 new boss_azgalor();
284 new npc_lesser_doomguard();
285}

76 struct npc_medivh_bmAI : public ScriptedAI
77 {
78 npc_medivh_bmAI(Creature* creature) : ScriptedAI(creature)

84 void Initialize()
85 {
86 SpellCorrupt_Timer = 0;
87 Check_Timer = 0;
88 Life75 = true;
89 Life50 = true;

91 }

93 InstanceScript* instance;
94
95 uint32 SpellCorrupt_Timer;
96 uint32 Check_Timer;
97
98 bool Life75;
99 bool Life50;
100 bool Life25;
101
102 void Reset() override
103 {
104 Initialize();
105
106 if (instance->GetData(TYPE_MEDIVH) == IN_PROGRESS)
107 DoCast(me, SPELL_CHANNEL, true);
108 else if (me->HasAura(SPELL_CHANNEL))
109 me->RemoveAura(SPELL_CHANNEL);
110
111 DoCast(me, SPELL_PORTAL_RUNE, true);
112 }
113
114 void MoveInLineOfSight(Unit* who) override
115 {
116 if (who->GetTypeId() == TYPEID_PLAYER && me->IsWithinDistInMap(who, 10.0f))
117 {
118 if (instance->GetData(TYPE_MEDIVH) == IN_PROGRESS || instance->GetData(TYPE_MEDIVH) == DONE)
119 return;
120
121 Talk(SAY_INTRO);
122 instance->SetData(TYPE_MEDIVH, IN_PROGRESS);
123 DoCast(me, SPELL_CHANNEL, false);
124 Check_Timer = 5000;
125 }
126 else if (who->GetTypeId() == TYPEID_UNIT && me->IsWithinDistInMap(who, 15.0f))
127 {
128 if (instance->GetData(TYPE_MEDIVH) != IN_PROGRESS)
129 return;
130
131 uint32 entry = who->GetEntry();
132 if (entry == NPC_INFINITE_ASSASIN || entry == NPC_INFINITE_WHELP || entry == NPC_INFINITE_CRONOMANCER || entry == NPC_INFINITE_EXECUTIONER || entry == NPC_INFINITE_VANQUISHER)
133 {
134 who->StopMoving();
135 who->CastSpell(me, SPELL_CORRUPT, false);
136 }
137 else if (entry == NPC_AEONUS)
138 {
139 who->StopMoving();
140 who->CastSpell(me, SPELL_CORRUPT_AEONUS, false);
141 }
142 }
143 }

146 {
147 //if (instance->GetData(TYPE_MEDIVH) == IN_PROGRESS)
148 // return;
149
150 //ScriptedAI::AttackStart(who);

153 void EnterCombat(Unit* /*who*/) override { }

155 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
156 {
157 if (SpellCorrupt_Timer)
158 return;
159
160 if (spell->Id == SPELL_CORRUPT_AEONUS)
161 SpellCorrupt_Timer = 1000;
162
163 if (spell->Id == SPELL_CORRUPT)
164 SpellCorrupt_Timer = 3000;
165 }
166
167 void JustDied(Unit* killer) override
168 {
169 if (killer->GetEntry() == me->GetEntry())
170 return;
171
172 Talk(SAY_DEATH);
173 }
174
175 void UpdateAI(uint32 diff) override
176 {
177 if (SpellCorrupt_Timer)
178 {
179 if (SpellCorrupt_Timer <= diff)
180 {
181 instance->SetData(TYPE_MEDIVH, SPECIAL);
182
183 if (me->HasAura(SPELL_CORRUPT_AEONUS))
184 SpellCorrupt_Timer = 1000;
185 else if (me->HasAura(SPELL_CORRUPT))
186 SpellCorrupt_Timer = 3000;
187 else
188 SpellCorrupt_Timer = 0;
189 } else SpellCorrupt_Timer -= diff;
190 }
191
192 if (Check_Timer)
193 {
194 if (Check_Timer <= diff)
195 {
196 uint32 pct = instance->GetData(DATA_SHIELD);
197
198 Check_Timer = 5000;
199
200 if (Life25 && pct <= 25)
201 {
202 Talk(SAY_WEAK25);
203 Life25 = false;
204 }
205 else if (Life50 && pct <= 50)
206 {
207 Talk(SAY_WEAK50);
208 Life50 = false;
209 }
210 else if (Life75 && pct <= 75)
211 {
212 Talk(SAY_WEAK75);
213 Life75 = false;
214 }
215
216 //if we reach this it means event was running but at some point reset.
217 if (instance->GetData(TYPE_MEDIVH) == NOT_STARTED)
218 {
219 me->DealDamage(me, me->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
220 me->RemoveCorpse();
221 me->Respawn();
222 return;
223 }
224
225 if (instance->GetData(TYPE_RIFT) == DONE)
226 {
227 Talk(SAY_WIN);
228 Check_Timer = 0;
229
230 if (me->HasAura(SPELL_CHANNEL))
231 me->RemoveAura(SPELL_CHANNEL);
232
233 /// @todo start the post-event here
234 instance->SetData(TYPE_MEDIVH, DONE);
235 }
236 } else Check_Timer -= diff;
237 }
238
239 //if (!UpdateVictim())
240 //return;
241
242 //DoMeleeAttackIfReady();
243 }

261{
262public:

270 struct npc_time_riftAI : public ScriptedAI
271 {
272 npc_time_riftAI(Creature* creature) : ScriptedAI(creature)
273 {
274 Initialize();
275 instance = creature->GetInstanceScript();
276 }
277
278 void Initialize()
279 {
280 TimeRiftWave_Timer = 15000;
281 mRiftWaveCount = 0;
282 mWaveId = 0;
283 }
284
285 InstanceScript* instance;
286
287 uint32 TimeRiftWave_Timer;
288 uint8 mRiftWaveCount;
289 uint8 mWaveId;
290
291 void Reset() override
292 {
293 Initialize();
294
295 uint8 mPortalCount = instance->GetData(DATA_PORTAL_COUNT);
296
297 if (mPortalCount < 6)
298 mWaveId = 0;
299 else if (mPortalCount > 12)
300 mWaveId = 2;
301 else mWaveId = 1;
302
303 }
304 void EnterCombat(Unit* /*who*/) override { }
305
306 void DoSummonAtRift(uint32 creature_entry)
307 {
308 if (!creature_entry)
309 return;
310
311 if (instance->GetData(TYPE_MEDIVH) != IN_PROGRESS)
312 {
313 me->InterruptNonMeleeSpells(true);
314 me->RemoveAllAuras();
315 return;
316 }
317
318 Position pos = me->GetRandomNearPosition(10.0f);
319
320 //normalize Z-level if we can, if rift is not at ground level.
321 pos.m_positionZ = std::max(me->GetMap()->GetHeight(pos.m_positionX, pos.m_positionY, MAX_HEIGHT), me->GetMap()->GetWaterLevel(pos.m_positionX, pos.m_positionY));
322
323 if (Unit* Summon = DoSummon(creature_entry, pos, 30000, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT))
324 if (Unit* temp = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_MEDIVH)))
325 Summon->AddThreat(temp, 0.0f);
326 }
327
328 void DoSelectSummon()
329 {
330 if ((mRiftWaveCount > 2 && mWaveId < 1) || mRiftWaveCount > 3)
331 mRiftWaveCount = 0;
332
333 uint32 entry = 0;
334
335 entry = PortalWaves[mWaveId].PortalMob[mRiftWaveCount];
336 TC_LOG_DEBUG("scripts", "npc_time_rift: summoning wave Creature (Wave %u, Entry %u).", mRiftWaveCount, entry);
337
338 ++mRiftWaveCount;
339
340 if (entry == NPC_INFINITE_WHELP)
341 {
342 for (uint8 i = 0; i < 3; ++i)
343 DoSummonAtRift(entry);
344 } else DoSummonAtRift(entry);
345 }
346
347 void UpdateAI(uint32 diff) override
348 {
349 if (TimeRiftWave_Timer <= diff)
350 {
351 DoSelectSummon();
352 TimeRiftWave_Timer = 15000;
353 } else TimeRiftWave_Timer -= diff;
354
355 if (me->IsNonMeleeSpellCast(false))
356 return;
357
358 TC_LOG_DEBUG("scripts", "npc_time_rift: not casting anylonger, i need to die.");
359 me->setDeathState(JUST_DIED);
360
361 if (instance->GetData(TYPE_RIFT) == IN_PROGRESS)
362 instance->SetData(TYPE_RIFT, SPECIAL);
363 }
364 };

381 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
382 {
383 player->PlayerTalkClass->ClearMenus();
384 if (action == GOSSIP_ACTION_INFO_DEF+1)
385 {
386 player->CLOSE_GOSSIP_MENU();
387 creature->CastSpell(player, SPELL_CHRONO_BEACON, false);
388 }
389 return true;
390 }
391
392 bool OnGossipHello(Player* player, Creature* creature) override
393 {
394 if (creature->IsQuestGiver())
395 player->PrepareQuestMenu(creature->GetGUID());
396
397 if (player->GetQuestStatus(QUEST_OPENING_PORTAL) == QUEST_STATUS_INCOMPLETE && !player->HasItemCount(ITEM_CHRONO_BEACON))
398 {
399 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_OBTAIN, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
400 player->SEND_GOSSIP_MENU(10000, creature->GetGUID());
401 return true;
402 }
403 else if (player->GetQuestRewardStatus(QUEST_OPENING_PORTAL) && !player->HasItemCount(ITEM_CHRONO_BEACON))
404 {
405 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_OBTAIN, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
406 player->SEND_GOSSIP_MENU(10001, creature->GetGUID());
407 return true;
408 }
409
410 player->SEND_GOSSIP_MENU(10002, creature->GetGUID());
411 return true;
412 }

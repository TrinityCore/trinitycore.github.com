45enum Events
46{
47 EVENT_HASTE = 1,
48 EVENT_MORTAL_WOUND = 2,
49 EVENT_WING_BUFFET = 3,

53class boss_temporus : public CreatureScript
54{
55public:
56 boss_temporus() : CreatureScript("boss_temporus") { }
57
58 struct boss_temporusAI : public BossAI
59 {
60 boss_temporusAI(Creature* creature) : BossAI(creature, TYPE_TEMPORUS) { }
61
62 void Reset() override { }
63
64 void EnterCombat(Unit* /*who*/) override
65 {
66 events.ScheduleEvent(EVENT_HASTE, urand(15000, 23000));
67 events.ScheduleEvent(EVENT_MORTAL_WOUND, 8000);
68 events.ScheduleEvent(EVENT_WING_BUFFET, urand(25000, 35000));
69 if (IsHeroic())
70 events.ScheduleEvent(EVENT_SPELL_REFLECTION, 30000);
71
72 Talk(SAY_AGGRO);
73 }
74
75 void KilledUnit(Unit* /*victim*/) override
76 {
77 Talk(SAY_SLAY);
78 }
79
80 void JustDied(Unit* /*killer*/) override
81 {
82 Talk(SAY_DEATH);
83
84 instance->SetData(TYPE_RIFT, SPECIAL);
85 }
86
87 void MoveInLineOfSight(Unit* who) override
88
89 {
90 //Despawn Time Keeper
91 if (who->GetTypeId() == TYPEID_UNIT && who->GetEntry() == NPC_TIME_KEEPER)
92 {
93 if (me->IsWithinDistInMap(who, 20.0f))
94 {
95 Talk(SAY_BANISH);
96
97 me->DealDamage(who, who->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
98 }
99 }
100
101 ScriptedAI::MoveInLineOfSight(who);
102 }
103
104 void UpdateAI(uint32 diff) override
105 {
106 //Return since we have no target
107 if (!UpdateVictim())
108 return;
109
110 events.Update(diff);
111
112 if (me->HasUnitState(UNIT_STATE_CASTING))
113 return;
114
115 while (uint32 eventId = events.ExecuteEvent())
116 {
117 switch (eventId)
118 {
119 case EVENT_HASTE:
120 DoCast(me, SPELL_HASTE);
121 events.ScheduleEvent(EVENT_HASTE, urand(20000, 25000));
122 break;
123 case EVENT_MORTAL_WOUND:
124 DoCast(me, SPELL_MORTAL_WOUND);
125 events.ScheduleEvent(EVENT_MORTAL_WOUND, urand(10000, 20000));
126 break;
127 case EVENT_WING_BUFFET:
128 DoCast(me, SPELL_WING_BUFFET);
129 events.ScheduleEvent(EVENT_WING_BUFFET, urand(20000, 30000));
130 break;
131 case EVENT_SPELL_REFLECTION: // Only in Heroic
132 DoCast(me, SPELL_REFLECT);
133 events.ScheduleEvent(EVENT_SPELL_REFLECTION, urand(25000, 35000));
134 break;
135 default:
136 break;
137 }
138 }
139 DoMeleeAttackIfReady();
140 }
141 };
142
143 CreatureAI* GetAI(Creature* creature) const override
144 {
145 return GetInstanceAI<boss_temporusAI>(creature);
146 }
147};

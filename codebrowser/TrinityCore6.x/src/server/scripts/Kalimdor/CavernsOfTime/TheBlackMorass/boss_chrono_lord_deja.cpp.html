53
54class boss_chrono_lord_deja : public CreatureScript
55{
56public:
57 boss_chrono_lord_deja() : CreatureScript("boss_chrono_lord_deja") { }
58
59 struct boss_chrono_lord_dejaAI : public BossAI
60 {
61 boss_chrono_lord_dejaAI(Creature* creature) : BossAI(creature, TYPE_CRONO_LORD_DEJA) { }
62
63 void Reset() override { }
64
65 void EnterCombat(Unit* /*who*/) override
66 {
67 events.ScheduleEvent(EVENT_ARCANE_BLAST, urand(18000, 23000));
68 events.ScheduleEvent(EVENT_TIME_LAPSE, urand(10000, 15000));
69 events.ScheduleEvent(EVENT_ARCANE_DISCHARGE, urand(20000, 30000));
70 if (IsHeroic())
71 events.ScheduleEvent(EVENT_ATTRACTION, urand(25000, 35000));
72
73 Talk(SAY_AGGRO);
74 }
75
76 void MoveInLineOfSight(Unit* who) override
77
78 {
79 //Despawn Time Keeper
80 if (who->GetTypeId() == TYPEID_UNIT && who->GetEntry() == NPC_TIME_KEEPER)
81 {
82 if (me->IsWithinDistInMap(who, 20.0f))
83 {
84 Talk(SAY_BANISH);
85 me->DealDamage(who, who->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
86 }
87 }
88
89 ScriptedAI::MoveInLineOfSight(who);
90 }
91
92 void KilledUnit(Unit* /*victim*/) override
93 {
94 Talk(SAY_SLAY);
95 }
96
97 void JustDied(Unit* /*killer*/) override
98 {
99 Talk(SAY_DEATH);
100
101 instance->SetData(TYPE_RIFT, SPECIAL);
102 }
103
104 void UpdateAI(uint32 diff) override
105 {
106 if (!UpdateVictim())
107 return;
108
109 if (me->HasUnitState(UNIT_STATE_CASTING))
110 return;
111
112 events.Update(diff);
113
114 while (uint32 eventId = events.ExecuteEvent())
115 {
116 switch (eventId)
117 {
118 case EVENT_ARCANE_BLAST:
119 DoCastVictim(SPELL_ARCANE_BLAST);
120 events.ScheduleEvent(EVENT_ARCANE_BLAST, urand(15000, 25000));
121 break;
122 case EVENT_TIME_LAPSE:
123 Talk(SAY_BANISH);
124 DoCast(me, SPELL_TIME_LAPSE);
125 events.ScheduleEvent(EVENT_TIME_LAPSE, urand(15000, 25000));
126 break;
127 case EVENT_ARCANE_DISCHARGE:
128 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
129 DoCast(target, SPELL_ARCANE_DISCHARGE);
130 events.ScheduleEvent(EVENT_ARCANE_DISCHARGE, urand(20000, 30000));
131 break;
132 case EVENT_ATTRACTION: // Only in Heroic
133 DoCast(me, SPELL_ATTRACTION);
134 events.ScheduleEvent(EVENT_ATTRACTION, urand(25000, 35000));
135 break;
136 default:
137 break;
138 }
139 }
140
141 DoMeleeAttackIfReady();
142 }
143 };
144
145 CreatureAI* GetAI(Creature* creature) const override
146 {
147 return GetInstanceAI<boss_chrono_lord_dejaAI>(creature);
148 }
149};

153 new boss_chrono_lord_deja();

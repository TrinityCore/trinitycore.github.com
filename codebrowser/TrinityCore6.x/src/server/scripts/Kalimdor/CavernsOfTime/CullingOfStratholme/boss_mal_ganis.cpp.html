70 struct boss_mal_ganisAI : public ScriptedAI
71 {
72 boss_mal_ganisAI(Creature* creature) : ScriptedAI(creature)
73 {
74 Initialize();
75 instance = creature->GetInstanceScript();
76 uiOutroStep = 0;

79 void Initialize()
80 {
81 bYelled = false;
82 bYelled2 = false;
83 Phase = COMBAT;
84 uiCarrionSwarmTimer = 6000;
85 uiMindBlastTimer = 11000;
86 uiVampiricTouchTimer = urand(10000, 15000);
87 uiSleepTimer = urand(15000, 20000);
88 uiOutroTimer = 1000;
89 }
90
91 uint32 uiCarrionSwarmTimer;
92 uint32 uiMindBlastTimer;
93 uint32 uiVampiricTouchTimer;
94 uint32 uiSleepTimer;
95
96 uint8 uiOutroStep;
97 uint32 uiOutroTimer;
98
99 bool bYelled;
100 bool bYelled2;
101
102 CombatPhases Phase;
103
104 InstanceScript* instance;
105
106 void Reset() override
107 {
108 Initialize();
109 instance->SetBossState(DATA_MAL_GANIS, NOT_STARTED);
110 }
111
112 void EnterCombat(Unit* /*who*/) override
113 {
114 Talk(SAY_AGGRO);
115 instance->SetBossState(DATA_MAL_GANIS, IN_PROGRESS);
116 }
117
118 void DamageTaken(Unit* done_by, uint32 &damage) override
119 {
120 if (damage >= me->GetHealth() && done_by != me)
121 damage = me->GetHealth()-1;
122 }
123
124 void UpdateAI(uint32 diff) override
125 {
126 switch (Phase)
127 {
128 case COMBAT:
129 //Return since we have no target
130 if (!UpdateVictim())
131 return;
132
133 if (!bYelled && HealthBelowPct(30))
134 {
135 Talk(SAY_30HEALTH);
136 bYelled = true;
137 }
138
139 if (!bYelled2 && HealthBelowPct(15))
140 {
141 Talk(SAY_15HEALTH);
142 bYelled2 = true;
143 }
144
145 if (HealthBelowPct(1))
146 {
147 //Handle Escape Event: Don't forget to add Player::RewardPlayerAndGroupAtEvent
148 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
149 uiOutroStep = 1;
150 Phase = OUTRO;
151 return;
152 }
153
154 if (Creature* pArthas = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ARTHAS)))
155 if (pArthas->isDead())
156 {
157 EnterEvadeMode();
158 me->DisappearAndDie();
159 instance->SetBossState(DATA_MAL_GANIS, FAIL);
160 }
161
162 if (uiCarrionSwarmTimer < diff)
163 {
164 DoCastVictim(SPELL_CARRION_SWARM);
165 uiCarrionSwarmTimer = 7000;
166 } else uiCarrionSwarmTimer -= diff;
167
168 if (uiMindBlastTimer < diff)
169 {
170 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
171 DoCast(target, SPELL_MIND_BLAST);
172 uiMindBlastTimer = 6000;
173 } else uiMindBlastTimer -= diff;
174
175 if (uiVampiricTouchTimer < diff)
176 {
177 DoCast(me, SPELL_VAMPIRIC_TOUCH);
178 uiVampiricTouchTimer = 32000;
179 } else uiVampiricTouchTimer -= diff;
180
181 if (uiSleepTimer < diff)
182 {
183 Talk(SAY_SLEEP);
184 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
185 DoCast(target, SPELL_SLEEP);
186 uiSleepTimer = urand(15000, 20000);
187 } else uiSleepTimer -= diff;
188
189 DoMeleeAttackIfReady();
190 break;
191 case OUTRO:
192 if (uiOutroTimer < diff)
193 {
194 switch (uiOutroStep)
195 {
196 case 1:
197 Talk(SAY_OUTRO);
198 me->GetMotionMaster()->MoveTargetedHome();
199 ++uiOutroStep;
200 uiOutroTimer = 8000;
201 break;
202 case 2:
203 me->SetTarget(instance->GetGuidData(DATA_ARTHAS));
204 me->HandleEmoteCommand(29);
205 Talk(SAY_ESCAPE_SPEECH_2);
206 ++uiOutroStep;
207 uiOutroTimer = 9000;
208 break;
209 case 3:
210 Talk(SAY_OUTRO);
211 ++uiOutroStep;
212 //uiOutroTimer = 16000;
213 break;
214 case 4:
215 me->HandleEmoteCommand(33);
216 ++uiOutroStep;
217 uiOutroTimer = 500;
218 break;
219 case 5:
220 me->SetVisible(false);
221 me->KillSelf();
222 break;
223
224 }
225 } else uiOutroTimer -= diff;
226 break;
227 }
228 }
229
230 void JustDied(Unit* /*killer*/) override
231 {
232 instance->SetBossState(DATA_MAL_GANIS, DONE);
233 DoCastAOE(SPELL_MAL_GANIS_KILL_CREDIT, true);

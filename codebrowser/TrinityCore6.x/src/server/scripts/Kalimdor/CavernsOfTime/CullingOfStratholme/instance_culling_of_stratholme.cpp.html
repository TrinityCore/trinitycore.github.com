69 void FillInitialWorldStates(WorldPackets::WorldState::InitWorldStates& packet) override
70 {
71 packet.Worldstates.emplace_back(uint32(WORLDSTATE_SHOW_CRATES), 1);
72 packet.Worldstates.emplace_back(uint32(WORLDSTATE_CRATES_REVEALED), int32(_crateCount));
73 packet.Worldstates.emplace_back(uint32(WORLDSTATE_WAVE_COUNT), 0);
74 packet.Worldstates.emplace_back(uint32(WORLDSTATE_TIME_GUARDIAN), 25);
75 packet.Worldstates.emplace_back(uint32(WORLDSTATE_TIME_GUARDIAN_SHOW), 0);
76 }

78 void OnCreatureCreate(Creature* creature) override
79 {
80 switch (creature->GetEntry())
81 {
82 case NPC_CHROMIE:
83 _chromieGUID = creature->GetGUID();
84 break;
85 case NPC_ARTHAS:
86 _arthasGUID = creature->GetGUID();
87 break;
88 case NPC_MEATHOOK:
89 _meathookGUID = creature->GetGUID();
90 break;
91 case NPC_SALRAMM:
92 _salrammGUID = creature->GetGUID();
93 break;
94 case NPC_EPOCH:
95 _epochGUID = creature->GetGUID();
96 break;
97 case NPC_MAL_GANIS:
98 _malGanisGUID = creature->GetGUID();
99 break;
100 case NPC_INFINITE:
101 _infiniteGUID = creature->GetGUID();
102 DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN_SHOW, 1);
103 break;
104 case NPC_GENERIC_BUNNY:

152 void SetData(uint32 type, uint32 data) override
153 {
154 switch (type)
155 {
156 case DATA_CRATE_COUNT:
157 _crateCount = data;
158 if (_crateCount == 5)
159 {
160 if (Creature* bunny = instance->GetCreature(_genericBunnyGUID))
161 bunny->CastSpell(bunny, SPELL_CRATES_CREDIT, true);
162
163 // Summon Chromie and global whisper
164 if (Creature* chromie = instance->SummonCreature(NPC_CHROMIE_2, ChromieSummonPos[0]))
165 if (!instance->GetPlayers().isEmpty())
166 chromie->AI()->Talk(SAY_CRATES_COMPLETED);
167 }
168 DoUpdateWorldState(WORLDSTATE_CRATES_REVEALED, _crateCount);
169 break;
170 case DATA_INFINITE_COUNTER:
171 _infiniteCouterState = data;
172 if (data == IN_PROGRESS)
173 {
174 if (!_infiniteGUID)
175 {
176 _eventTimer = 25;
177 instance->SummonCreature(NPC_INFINITE, InfiniteCorruptorPos);
178 instance->SummonCreature(NPC_TIME_RIFT, TimeRiftPos);
179 instance->SummonCreature(NPC_GUARDIAN_OF_TIME, GuardianOfTimePos);
180 events.ScheduleEvent(EVENT_INFINITE_TIMER, 1);
181 }
182 }
183 break;
184 default:
185 break;
186 }
187 }
188
189 bool SetBossState(uint32 type, EncounterState state) override
190 {
191 if (!InstanceScript::SetBossState(type, state))
192 return false;
193
194 switch (type)
195 {
196 case DATA_INFINITE:
197 if (state == DONE)
198 {
199 DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN_SHOW, 0);
200 DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN, 0);
201 }
202 break;
203 case DATA_MAL_GANIS:
204 if (state == DONE)
205 {
206 if (GameObject* go = instance->GetGameObject(_malGanisChestGUID))
207 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
208 instance->SummonCreature(NPC_CHROMIE_3, ChromieSummonPos[1]);
209 }
210 break;
211 default:
212 break;
213 }
214
215 return true;
216 }

264 void Update(uint32 diff) override
265 {
266 events.Update(diff);
267
268 while (uint32 eventId = events.ExecuteEvent())
269 {
270 switch (eventId)
271 {
272 case EVENT_INFINITE_TIMER:
273 DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN, _eventTimer);
274
275 switch (_eventTimer)
276 {
277 case 25:
278 if (instance->HavePlayers())
279 if (Creature* chromie = instance->GetCreature(_chromieGUID))
280 chromie->AI()->Talk(SAY_INFINITE_START);
281 break;
282 case 5:
283 if (instance->HavePlayers())
284 if (Creature* chromie = instance->GetCreature(_chromieGUID))
285 chromie->AI()->Talk(SAY_INFINITE);
286 break;
287 case 0:
288 if (instance->HavePlayers())
289 if (Creature* chromie = instance->GetCreature(_chromieGUID))
290 chromie->AI()->Talk(SAY_INFINITE_FAIL);
291
292 if (Creature* infinite = instance->GetCreature(_infiniteGUID))
293 {
294 if (Creature* guardian = infinite->FindNearestCreature(NPC_GUARDIAN_OF_TIME, 100.0f))
295 infinite->Kill(guardian);
296
297 if (Creature* rift = infinite->FindNearestCreature(NPC_TIME_RIFT, 100.0f))
298 {
299 infinite->GetMotionMaster()->MovePoint(0, rift->GetPositionX(), rift->GetPositionY(), rift->GetPositionZ());
300 rift->DespawnOrUnsummon(3000);
301 }
302
303 infinite->DespawnOrUnsummon(3000);
304 infinite->AI()->Talk(SAY_FAIL_EVENT);
305 }
306 DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN_SHOW, 0);
307 return;
308 default:
309 break;
310 }
311 events.ScheduleEvent(EVENT_INFINITE_TIMER, 60000);
312 --_eventTimer;
313 break;
314 default:
315 break;
316 }
317 }
318 }

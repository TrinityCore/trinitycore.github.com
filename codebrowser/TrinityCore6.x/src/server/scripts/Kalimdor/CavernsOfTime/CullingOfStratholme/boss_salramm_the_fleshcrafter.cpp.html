18#include "ScriptMgr.h"

22enum Spells
23{
24 SPELL_CURSE_OF_TWISTED_FLESH = 58845,
25 SPELL_EXPLODE_GHOUL = 52480,
26 SPELL_SHADOW_BOLT = 57725,
27 SPELL_STEAL_FLESH = 52708,
28 SPELL_SUMMON_GHOULS = 52451
29};
30
31enum Yells
32{
33 SAY_AGGRO = 0,
34 SAY_SPAWN = 1,
35 SAY_SLAY = 2,
36 SAY_DEATH = 3,
37 SAY_EXPLODE_GHOUL = 4,
38 SAY_STEAL_FLESH = 5,
39 SAY_SUMMON_GHOULS = 6
40};
41
42enum Events
43{
44 EVENT_CURSE_FLESH = 1,
45 EVENT_EXPLODE_GHOUL,
46 EVENT_SHADOW_BOLT,
47 EVENT_STEAL_FLESH,
48 EVENT_SUMMON_GHOULS
49};
50
51class boss_salramm : public CreatureScript
52{
53 public:
54 boss_salramm() : CreatureScript("boss_salramm") { }
55
56 struct boss_salrammAI : public BossAI
57 {
58 boss_salrammAI(Creature* creature) : BossAI(creature, DATA_SALRAMM)
59 {
60 Talk(SAY_SPAWN);
61 }
62
63 void EnterCombat(Unit* /*who*/) override
64 {
65 Talk(SAY_AGGRO);
66 _EnterCombat();
67
68 events.ScheduleEvent(EVENT_CURSE_FLESH, 30000);
69 events.ScheduleEvent(EVENT_SUMMON_GHOULS, urand(19000, 24000));
70 events.ScheduleEvent(EVENT_SHADOW_BOLT, urand(8000, 12000));
71 events.ScheduleEvent(EVENT_STEAL_FLESH, 12345); /// @todo: adjust timer
72 }
73
74 void ExecuteEvent(uint32 eventId) override
75 {
76 switch (eventId)
77 {
78 case EVENT_CURSE_FLESH:
79 DoCastVictim(SPELL_CURSE_OF_TWISTED_FLESH);
80 events.ScheduleEvent(EVENT_CURSE_FLESH, 37000);
81 break;
82 case EVENT_SUMMON_GHOULS:
83 Talk(SAY_SUMMON_GHOULS);
84 DoCast(me, SPELL_SUMMON_GHOULS);
85 events.ScheduleEvent(EVENT_SUMMON_GHOULS, 10000);
86 events.ScheduleEvent(EVENT_EXPLODE_GHOUL, 6000);
87 break;
88 case EVENT_SHADOW_BOLT:
89 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40.0f, true))
90 DoCast(target, SPELL_SHADOW_BOLT);
91 events.ScheduleEvent(EVENT_SHADOW_BOLT, urand(8000, 12000));
92 break;
93 case EVENT_STEAL_FLESH:
94 Talk(SAY_STEAL_FLESH);
95 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true))
96 DoCast(target, SPELL_STEAL_FLESH);
97 events.ScheduleEvent(EVENT_STEAL_FLESH, 12345);
98 break;
99 case EVENT_EXPLODE_GHOUL:
100 Talk(SAY_EXPLODE_GHOUL);
101 DoCast(me, SPELL_EXPLODE_GHOUL, true);
102 break;
103 default:
104 break;
105 }
106 }
107
108 void JustDied(Unit* /*killer*/) override
109 {
110 Talk(SAY_DEATH);
111 _JustDied();
112 }
113
114 void KilledUnit(Unit* victim) override
115 {
116 if (victim->GetTypeId() == TYPEID_PLAYER)
117 Talk(SAY_SLAY);
118 }
119 };
120
121 CreatureAI* GetAI(Creature* creature) const override
122 {
123 return GetInstanceAI<boss_salrammAI>(creature);
124 }
125};
126
127void AddSC_boss_salramm()
128{
129 new boss_salramm();
130}

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "culling_of_stratholme.h"

22enum Spells
23{
24 SPELL_CORRUPTING_BLIGHT = 60588,
25 SPELL_VOID_STRIKE = 60590,
26 SPELL_CORRUPTION_OF_TIME_CHANNEL = 60422,
27 SPELL_CORRUPTION_OF_TIME_TARGET = 60451
28};
29
30enum Yells
31{
32 SAY_AGGRO = 0,
33 SAY_DEATH = 1,
34 SAY_FAIL = 2
35};
36
37enum Events
38{
39 EVENT_CORRUPTING_BLIGHT = 1,
40 EVENT_VOID_STRIKE
41};
42
43class boss_infinite_corruptor : public CreatureScript
44{
45 public:
46 boss_infinite_corruptor() : CreatureScript("boss_infinite_corruptor") { }
47
48 struct boss_infinite_corruptorAI : public BossAI
49 {
50 boss_infinite_corruptorAI(Creature* creature) : BossAI(creature, DATA_INFINITE) { }
51
52 void Reset() override
53 {
54 _Reset();
55
56 if (Creature* guardian = me->FindNearestCreature(NPC_GUARDIAN_OF_TIME, 100.0f))
57 {
58 DoCast((Unit*)NULL, SPELL_CORRUPTION_OF_TIME_CHANNEL, false);
59 guardian->CastSpell(guardian, SPELL_CORRUPTION_OF_TIME_TARGET, false);
60 }
61 }
62
63 void EnterCombat(Unit* /*who*/) override
64 {
65 Talk(SAY_AGGRO);
66 _EnterCombat();
67 events.ScheduleEvent(EVENT_CORRUPTING_BLIGHT, 7000);
68 events.ScheduleEvent(EVENT_VOID_STRIKE, 5000);
69 }
70
71 void JustDied(Unit* /*killer*/) override
72 {
73 Talk(SAY_DEATH);
74 _JustDied();
75
76 if (Creature* guardian = me->FindNearestCreature(NPC_GUARDIAN_OF_TIME, 100.0f))
77 {
78 guardian->RemoveAurasDueToSpell(SPELL_CORRUPTION_OF_TIME_TARGET);
79 guardian->DespawnOrUnsummon(5000);
80 }
81
82 if (Creature* rift = me->FindNearestCreature(NPC_TIME_RIFT, 100.0f))
83 rift->DespawnOrUnsummon();
84 }
85
86 void ExecuteEvent(uint32 eventId) override
87 {
88 switch (eventId)
89 {
90 case EVENT_CORRUPTING_BLIGHT:
91 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 60.0f, true))
92 DoCast(target, SPELL_CORRUPTING_BLIGHT);
93 events.ScheduleEvent(EVENT_CORRUPTING_BLIGHT, 17000);
94 break;
95 case EVENT_VOID_STRIKE:
96 DoCastVictim(SPELL_VOID_STRIKE);
97 events.ScheduleEvent(EVENT_VOID_STRIKE, 5000);
98 break;
99 default:
100 break;
101 }
102 }
103 };
104
105 CreatureAI* GetAI(Creature* creature) const override
106 {
107 return GetInstanceAI<boss_infinite_corruptorAI>(creature);
108 }
109};
110
111void AddSC_boss_infinite_corruptor()
112{
113 new boss_infinite_corruptor();
114}

67 struct npc_aquementasAI : public ScriptedAI
68 {
69 npc_aquementasAI(Creature* creature) : ScriptedAI(creature)
70 {
71 Initialize();
72 }
73
74 void Initialize()
75 {
76 SendItemTimer = 0;
77 SwitchFactionTimer = 10000;
78
79 isFriendly = true;
80
81 AquaJetTimer = 5000;
82 FrostShockTimer = 1000;
83 }
84
85 uint32 SendItemTimer;
86 uint32 SwitchFactionTimer;
87 bool isFriendly;
88
89 uint32 FrostShockTimer;
90 uint32 AquaJetTimer;
91
92 void Reset() override
93 {
94 Initialize();
95 me->setFaction(35);
96 }
97
98 void SendItem(Unit* receiver)
99 {
100 Player* player = receiver->ToPlayer();
101
102 if (player && player->HasItemCount(ITEM_BOOK_OF_AQUOR, 1, false) &&
103 player->HasItemCount(ITEM_SILVERY_CLAWS, 11, false) &&
104 player->HasItemCount(ITEM_IRONTREE_HEART, 1, false) &&
105 !player->HasItemCount(ITEM_SILVER_TOTEM, 1, true))
106 {
107 ItemPosCountVec dest;
108 uint8 msg = player->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, 11522, 1, NULL);
109 if (msg == EQUIP_ERR_OK)
110 player->StoreNewItem(dest, ITEM_SILVER_TOTEM, true);
111 }
112 }
113
114 void EnterCombat(Unit* who) override
115 {
116 Talk(AGGRO_YELL_AQUE, who);
117 }
118
119 void UpdateAI(uint32 diff) override
120 {
121 if (isFriendly)
122 {
123 if (SwitchFactionTimer <= diff)
124 {
125 me->setFaction(91);
126 isFriendly = false;
127 } else SwitchFactionTimer -= diff;
128 }
129
130 if (!UpdateVictim())
131 return;
132
133 if (!isFriendly)
134 {
135 if (SendItemTimer <= diff)
136 {
137 if (me->GetVictim() && me->EnsureVictim()->GetTypeId() == TYPEID_PLAYER)
138 SendItem(me->GetVictim());
139 SendItemTimer = 5000;
140 } else SendItemTimer -= diff;
141 }
142
143 if (FrostShockTimer <= diff)
144 {
145 DoCastVictim(SPELL_FROST_SHOCK);
146 FrostShockTimer = 15000;
147 } else FrostShockTimer -= diff;
148
149 if (AquaJetTimer <= diff)
150 {
151 DoCast(me, SPELL_AQUA_JET);
152 AquaJetTimer = 15000;
153 } else AquaJetTimer -= diff;
154
155 DoMeleeAttackIfReady();
156 }
157 };

476 struct npc_toogaAI : public FollowerAI

478 npc_toogaAI(Creature* creature) : FollowerAI(creature)
479 {
480 Initialize();
481 }
482
483 void Initialize()
484 {
485 CheckSpeechTimer = 2500;
486 PostEventTimer = 1000;
487 PhasePostEvent = 0;
488
489 TortaGUID.Clear();
490 }
491
492 uint32 CheckSpeechTimer;
493 uint32 PostEventTimer;
494 uint32 PhasePostEvent;
495
496 ObjectGuid TortaGUID;
497
498 void Reset() override
499 {
500 Initialize();
501 }
502
503 void MoveInLineOfSight(Unit* who) override
504 {
505 FollowerAI::MoveInLineOfSight(who);
506
507 if (!me->GetVictim() && !HasFollowState(STATE_FOLLOW_COMPLETE | STATE_FOLLOW_POSTEVENT) && who->GetEntry() == NPC_TORTA)
508 {
509 if (me->IsWithinDistInMap(who, INTERACTION_DISTANCE))
510 {
511 Player* player = GetLeaderForFollower();
512 if (player && player->GetQuestStatus(QUEST_TOOGA) == QUEST_STATUS_INCOMPLETE)
513 player->GroupEventHappens(QUEST_TOOGA, me);
514
515 TortaGUID = who->GetGUID();
516 SetFollowComplete(true);
517 }
518 }
519 }
520
521 void MovementInform(uint32 MotionType, uint32 PointId) override
522 {
523 FollowerAI::MovementInform(MotionType, PointId);
524
525 if (MotionType != POINT_MOTION_TYPE)
526 return;
527
528 if (PointId == POINT_ID_TO_WATER)
529 SetFollowComplete();
530 }
531
532 void UpdateFollowerAI(uint32 Diff) override
533 {
534 if (!UpdateVictim())
535 {
536 //we are doing the post-event, or...
537 if (HasFollowState(STATE_FOLLOW_POSTEVENT))
538 {
539 if (PostEventTimer <= Diff)
540 {
541 PostEventTimer = 5000;
542
543 Creature* torta = ObjectAccessor::GetCreature(*me, TortaGUID);
544 if (!torta || !torta->IsAlive())
545 {
546 //something happened, so just complete
547 SetFollowComplete();
548 return;
549 }
550
551 switch (PhasePostEvent)
552 {
553 case 1:
554 Talk(SAY_TOOG_POST_1);
555 break;
556 case 2:
557 torta->AI()->Talk(SAY_TORT_POST_2);
558 break;
559 case 3:
560 Talk(SAY_TOOG_POST_3);
561 break;
562 case 4:
563 torta->AI()->Talk(SAY_TORT_POST_4);
564 break;
565 case 5:
566 Talk(SAY_TOOG_POST_5);
567 break;
568 case 6:
569 torta->AI()->Talk(SAY_TORT_POST_6);
570 me->GetMotionMaster()->MovePoint(POINT_ID_TO_WATER, ToWaterLoc);
571 break;
572 }
573
574 ++PhasePostEvent;
575 }
576 else
577 PostEventTimer -= Diff;
578 }
579 //...we are doing regular speech check
580 else if (HasFollowState(STATE_FOLLOW_INPROGRESS))
581 {
582 if (CheckSpeechTimer <= Diff)
583 {
584 CheckSpeechTimer = 5000;
585
586 if (urand(0, 9) > 8)
587 Talk(SAY_TOOG_WORRIED);
588 }
589 else
590 CheckSpeechTimer -= Diff;
591 }
592
593 return;
594 }
595
596 DoMeleeAttackIfReady();
597 }
598 };

33enum Omen
34{
35 NPC_OMEN = 15467,
36
37 SPELL_OMEN_CLEAVE = 15284,
38 SPELL_OMEN_STARFALL = 26540,
39 SPELL_OMEN_SUMMON_SPOTLIGHT = 26392,
40 SPELL_ELUNE_CANDLE = 26374,
41
42 GO_ELUNE_TRAP_1 = 180876,
43 GO_ELUNE_TRAP_2 = 180877,
44
45 EVENT_CAST_CLEAVE = 1,
46 EVENT_CAST_STARFALL = 2,
47 EVENT_DESPAWN = 3,
48};
49
50class npc_omen : public CreatureScript
51{
52public:
53 npc_omen() : CreatureScript("npc_omen") { }
54
55 struct npc_omenAI : public ScriptedAI
56 {
57 npc_omenAI(Creature* creature) : ScriptedAI(creature)
58 {
59 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
60 me->GetMotionMaster()->MovePoint(1, 7549.977f, -2855.137f, 456.9678f);
61 }
62
63 EventMap events;
64
65 void MovementInform(uint32 type, uint32 pointId) override
66 {
67 if (type != POINT_MOTION_TYPE)
68 return;
69
70 if (pointId == 1)
71 {
72 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
73 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
74 if (Player* player = me->SelectNearestPlayer(40.0f))
75 AttackStart(player);
76 }
77 }
78
79 void EnterCombat(Unit* /*attacker*/) override
80 {
81 events.Reset();
82 events.ScheduleEvent(EVENT_CAST_CLEAVE, urand(3000, 5000));
83 events.ScheduleEvent(EVENT_CAST_STARFALL, urand(8000, 10000));
84 }
85
86 void JustDied(Unit* /*killer*/) override
87 {
88 DoCast(SPELL_OMEN_SUMMON_SPOTLIGHT);
89 }
90
91 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
92 {
93 if (spell->Id == SPELL_ELUNE_CANDLE)
94 {
95 if (me->HasAura(SPELL_OMEN_STARFALL))
96 me->RemoveAurasDueToSpell(SPELL_OMEN_STARFALL);
97
98 events.RescheduleEvent(EVENT_CAST_STARFALL, urand(14000, 16000));
99 }
100 }
101
102 void UpdateAI(uint32 diff) override
103 {
104 if (!UpdateVictim())
105 return;
106
107 events.Update(diff);
108
109 switch (events.ExecuteEvent())
110 {
111 case EVENT_CAST_CLEAVE:
112 DoCastVictim(SPELL_OMEN_CLEAVE);
113 events.ScheduleEvent(EVENT_CAST_CLEAVE, urand(8000, 10000));
114 break;
115 case EVENT_CAST_STARFALL:
116 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
117 DoCast(target, SPELL_OMEN_STARFALL);
118 events.ScheduleEvent(EVENT_CAST_STARFALL, urand(14000, 16000));
119 break;
120 }
121
122 DoMeleeAttackIfReady();
123 }
124 };
125
126 CreatureAI* GetAI(Creature* creature) const override
127 {
128 return new npc_omenAI(creature);
129 }
130};

132class npc_giant_spotlight : public CreatureScript
133{
134public:
135 npc_giant_spotlight() : CreatureScript("npc_giant_spotlight") { }
136
137 struct npc_giant_spotlightAI : public ScriptedAI
138 {
139 npc_giant_spotlightAI(Creature* creature) : ScriptedAI(creature) { }
140
141 EventMap events;
142
143 void Reset() override
144 {
145 events.Reset();
146 events.ScheduleEvent(EVENT_DESPAWN, 5*MINUTE*IN_MILLISECONDS);
147 }
148
149 void UpdateAI(uint32 diff) override
150 {
151 events.Update(diff);
152
153 if (events.ExecuteEvent() == EVENT_DESPAWN)
154 {
155 if (GameObject* trap = me->FindNearestGameObject(GO_ELUNE_TRAP_1, 5.0f))
156 trap->RemoveFromWorld();
157
158 if (GameObject* trap = me->FindNearestGameObject(GO_ELUNE_TRAP_2, 5.0f))
159 trap->RemoveFromWorld();
160
161 if (Creature* omen = me->FindNearestCreature(NPC_OMEN, 5.0f, false))
162 omen->DespawnOrUnsummon();
163
164 me->DespawnOrUnsummon();
165 }
166 }
167 };
168
169 CreatureAI* GetAI(Creature* creature) const override
170 {
171 return new npc_giant_spotlightAI(creature);
172 }
173};
174
175void AddSC_moonglade()
176{
177 new npc_omen();
178 new npc_giant_spotlight();
179}

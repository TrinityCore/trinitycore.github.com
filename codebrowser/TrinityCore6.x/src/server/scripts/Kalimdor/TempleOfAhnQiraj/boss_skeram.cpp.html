56 struct boss_skeramAI : public BossAI
57 {
58 boss_skeramAI(Creature* creature) : BossAI(creature, DATA_SKERAM)

60 Initialize();

62
63 void Initialize()
64 {
65 _flag = 0;
66 _hpct = 75.0f;
67 }
68
69 void Reset() override
70 {
71 Initialize();
72 me->SetVisible(true);
73 }
74
75 void KilledUnit(Unit* /*victim*/) override
76 {
77 Talk(SAY_SLAY);
78 }
79
80 void EnterEvadeMode() override
81 {
82 ScriptedAI::EnterEvadeMode();
83 if (me->IsSummon())
84 ((TempSummon*)me)->UnSummon();
85 }
86
87 void JustSummoned(Creature* creature) override
88 {
89 // Shift the boss and images (Get it? *Shift*?)
90 uint8 rand = 0;
91 if (_flag != 0)
92 {
93 while (_flag & (1 << rand))
94 rand = urand(0, 2);
95 DoCast(me, BlinkSpells[rand]);
96 _flag |= (1 << rand);
97 _flag |= (1 << 7);
98 }
99
100 while (_flag & (1 << rand))
101 rand = urand(0, 2);
102 creature->CastSpell(creature, BlinkSpells[rand]);
103 _flag |= (1 << rand);
104
105 if (_flag & (1 << 7))
106 _flag = 0;
107
108 if (Unit* Target = SelectTarget(SELECT_TARGET_RANDOM))
109 creature->AI()->AttackStart(Target);
110
111 float ImageHealthPct;
112
113 if (me->GetHealthPct() < 25.0f)
114 ImageHealthPct = 0.50f;
115 else if (me->GetHealthPct() < 50.0f)
116 ImageHealthPct = 0.20f;
117 else
118 ImageHealthPct = 0.10f;
119
120 creature->SetMaxHealth(me->GetMaxHealth() * ImageHealthPct);
121 creature->SetHealth(creature->GetMaxHealth() * (me->GetHealthPct() / 100.0f));
122 }
123
124 void JustDied(Unit* /*killer*/) override
125 {
126 if (!me->IsSummon())
127 Talk(SAY_DEATH);
128 else
129 me->RemoveCorpse();
130 }
131
132 void EnterCombat(Unit* /*who*/) override
133 {
134 _EnterCombat();
135 events.Reset();
136
137 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(6000, 12000));
138 events.ScheduleEvent(EVENT_FULLFILMENT, 15000);
139 events.ScheduleEvent(EVENT_BLINK, urand(30000, 45000));
140 events.ScheduleEvent(EVENT_EARTH_SHOCK, 2000);
141
142 Talk(SAY_AGGRO);
143 }
144
145 void UpdateAI(uint32 diff) override
146 {
147 if (!UpdateVictim())
148 return;
149
150 events.Update(diff);
151
152 while (uint32 eventId = events.ExecuteEvent())
153 {
154 switch (eventId)
155 {
156 case EVENT_ARCANE_EXPLOSION:
157 DoCastAOE(SPELL_ARCANE_EXPLOSION, true);
158 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(8000, 18000));
159 break;
160 case EVENT_FULLFILMENT:
161 /// @todo For some weird reason boss does not cast this
162 // Spell actually works, tested in duel
163 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true), SPELL_TRUE_FULFILLMENT, true);
164 events.ScheduleEvent(EVENT_FULLFILMENT, urand(20000, 30000));
165 break;
166 case EVENT_BLINK:
167 DoCast(me, BlinkSpells[urand(0, 2)]);
168 DoResetThreat();
169 me->SetVisible(true);
170 events.ScheduleEvent(EVENT_BLINK, urand(10000, 30000));
171 break;
172 case EVENT_EARTH_SHOCK:
173 DoCastVictim(SPELL_EARTH_SHOCK);
174 events.ScheduleEvent(EVENT_EARTH_SHOCK, 2000);
175 break;
176 }
177 }
178
179 if (!me->IsSummon() && me->GetHealthPct() < _hpct)
180 {
181 DoCast(me, SPELL_SUMMON_IMAGES);
182 Talk(SAY_SPLIT);
183 _hpct -= 25.0f;
184 me->SetVisible(false);
185 events.RescheduleEvent(EVENT_BLINK, 2000);
186 }
187
188 if (me->IsWithinMeleeRange(me->GetVictim()))
189 {
190 events.RescheduleEvent(EVENT_EARTH_SHOCK, 2000);
191 DoMeleeAttackIfReady();
192 }
193 }
194
195 private:
196 float _hpct;
197 uint8 _flag;
198 };
199
200 CreatureAI* GetAI(Creature* creature) const override
201 {
202 return new boss_skeramAI(creature);
203 }

206class PlayerOrPetCheck
207{
208 public:
209 bool operator()(WorldObject* object) const
210 {
211 if (object->GetTypeId() == TYPEID_PLAYER)
212 return false;
213
214 if (Creature* creature = object->ToCreature())
215 return !creature->IsPet();
216
217 return true;
218 }

26#include "ScriptMgr.h"
27#include "InstanceScript.h"
28#include "temple_of_ahnqiraj.h"
29
30class instance_temple_of_ahnqiraj : public InstanceMapScript
31{
32 public:
33 instance_temple_of_ahnqiraj() : InstanceMapScript("instance_temple_of_ahnqiraj", 531) { }
34
35 InstanceScript* GetInstanceScript(InstanceMap* map) const override
36 {
37 return new instance_temple_of_ahnqiraj_InstanceMapScript(map);
38 }
39
40 struct instance_temple_of_ahnqiraj_InstanceMapScript : public InstanceScript
41 {
42 instance_temple_of_ahnqiraj_InstanceMapScript(Map* map) : InstanceScript(map)
43 {
44 SetHeaders(DataHeader);
45 IsBossDied[0] = false;
46 IsBossDied[1] = false;
47 IsBossDied[2] = false;
48
49 BugTrioDeathCount = 0;
50
51 CthunPhase = 0;
52 }
53
54 //If Vem is dead...
55 bool IsBossDied[3];
56
57 //Storing Skeram, Vem and Kri.
58 ObjectGuid SkeramGUID;
59 ObjectGuid VemGUID;
60 ObjectGuid KriGUID;
61 ObjectGuid VeklorGUID;
62 ObjectGuid VeknilashGUID;
63 ObjectGuid ViscidusGUID;
64
65 uint32 BugTrioDeathCount;
66
67 uint32 CthunPhase;
68
69 void OnCreatureCreate(Creature* creature) override
70 {
71 switch (creature->GetEntry())
72 {
73 case NPC_SKERAM:
74 SkeramGUID = creature->GetGUID();
75 break;
76 case NPC_VEM:
77 VemGUID = creature->GetGUID();
78 break;
79 case NPC_KRI:
80 KriGUID = creature->GetGUID();
81 break;
82 case NPC_VEKLOR:
83 VeklorGUID = creature->GetGUID();
84 break;
85 case NPC_VEKNILASH:
86 VeknilashGUID = creature->GetGUID();
87 break;
88 case NPC_VISCIDUS:
89 ViscidusGUID = creature->GetGUID();
90 break;
91 }
92 }
93
94 bool IsEncounterInProgress() const override
95 {
96 //not active in AQ40
97 return false;
98 }
99
100 uint32 GetData(uint32 type) const override
101 {
102 switch (type)
103 {
104 case DATA_VEMISDEAD:
105 if (IsBossDied[0])
106 return 1;
107 break;
108
109 case DATA_VEKLORISDEAD:
110 if (IsBossDied[1])
111 return 1;
112 break;
113
114 case DATA_VEKNILASHISDEAD:
115 if (IsBossDied[2])
116 return 1;
117 break;
118
119 case DATA_BUG_TRIO_DEATH:
120 return BugTrioDeathCount;
121
122 case DATA_CTHUN_PHASE:
123 return CthunPhase;
124 }
125 return 0;
126 }
127
128 ObjectGuid GetGuidData(uint32 identifier) const override
129 {
130 switch (identifier)
131 {
132 case DATA_SKERAM:
133 return SkeramGUID;
134 case DATA_VEM:
135 return VemGUID;
136 case DATA_KRI:
137 return KriGUID;
138 case DATA_VEKLOR:
139 return VeklorGUID;
140 case DATA_VEKNILASH:
141 return VeknilashGUID;
142 case DATA_VISCIDUS:
143 return ViscidusGUID;
144 }
145 return ObjectGuid::Empty;
146 } // end GetGuidData
147
148 void SetData(uint32 type, uint32 data) override
149 {
150 switch (type)
151 {
152 case DATA_VEM_DEATH:
153 IsBossDied[0] = true;
154 break;
155
156 case DATA_BUG_TRIO_DEATH:
157 ++BugTrioDeathCount;
158 break;
159
160 case DATA_VEKLOR_DEATH:
161 IsBossDied[1] = true;
162 break;
163
164 case DATA_VEKNILASH_DEATH:
165 IsBossDied[2] = true;
166 break;
167
168 case DATA_CTHUN_PHASE:
169 CthunPhase = data;
170 break;
171 }
172 }
173 };
174
175};
176
177void AddSC_instance_temple_of_ahnqiraj()
178{
179 new instance_temple_of_ahnqiraj();
180}

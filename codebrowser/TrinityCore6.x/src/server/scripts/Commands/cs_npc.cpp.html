234 //add spawn of creature
235 static bool HandleNpcAddCommand(ChatHandler* handler, char const* args)
236 {
237 if (!*args)
238 return false;
239
240 char* charID = handler->extractKeyFromLink((char*)args, "Hcreature_entry");
241 if (!charID)
242 return false;
243
244 uint32 id = atoi(charID);
245 if (!sObjectMgr->GetCreatureTemplate(id))
246 return false;
247
248 Player* chr = handler->GetSession()->GetPlayer();
249 float x = chr->GetPositionX();
250 float y = chr->GetPositionY();
251 float z = chr->GetPositionZ();
252 float o = chr->GetOrientation();
253 Map* map = chr->GetMap();
254
255 if (Transport* trans = chr->GetTransport())
256 {
257 ObjectGuid::LowType guid = map->GenerateLowGuid<HighGuid::Creature>();
258 CreatureData& data = sObjectMgr->NewOrExistCreatureData(guid);
259 data.id = id;
260 data.phaseMask = chr->GetPhaseMask();
261 data.posX = chr->GetTransOffsetX();
262 data.posY = chr->GetTransOffsetY();
263 data.posZ = chr->GetTransOffsetZ();
264 data.orientation = chr->GetTransOffsetO();
265
266 Creature* creature = trans->CreateNPCPassenger(guid, &data);
267
268 creature->SaveToDB(trans->GetGOInfo()->moTransport.mapID, 1 << map->GetSpawnMode(), chr->GetPhaseMask());
269
270 sObjectMgr->AddCreatureToGrid(guid, &data);
271 return true;
272 }
273
274 Creature* creature = new Creature();
275 if (!creature->Create(map->GenerateLowGuid<HighGuid::Creature>(), map, chr->GetPhaseMask(), id, x, y, z, o))
276 {
277 delete creature;
278 return false;
279 }
280
281 creature->SaveToDB(map->GetId(), (1 << map->GetSpawnMode()), chr->GetPhaseMask());
282
283 ObjectGuid::LowType db_guid = creature->GetSpawnId();
284
285 // To call _LoadGoods(); _LoadQuests(); CreateTrainerSpells()
286 // current "creature" variable is deleted and created fresh new, otherwise old values might trigger asserts or cause undefined behavior
287 creature->CleanupsBeforeDelete();
288 delete creature;
289 creature = new Creature();
290 if (!creature->LoadCreatureFromDB(db_guid, map))
291 {
292 delete creature;
293 return false;
294 }
295
296 sObjectMgr->AddCreatureToGrid(db_guid, sObjectMgr->GetCreatureData(db_guid));
297 return true;
298 }

300 //add item in vendorlist
301 static bool HandleNpcAddVendorItemCommand(ChatHandler* handler, char const* args)
302 {
303 if (!*args)
304 return false;
305
306 const uint8 type = 1; // FIXME: make type (1 item, 2 currency) an argument
307
308 char* pitem = handler->extractKeyFromLink((char*)args, "Hitem");
309 if (!pitem)
310 {
311 handler->SendSysMessage(LANG_COMMAND_NEEDITEMSEND);
312 handler->SetSentErrorMessage(true);
313 return false;
314 }
315
316 int32 item_int = atol(pitem);
317 if (item_int <= 0)
318 return false;
319
320 uint32 itemId = item_int;
321
322 char* fmaxcount = strtok(NULL, " "); //add maxcount, default: 0
323 uint32 maxcount = 0;
324 if (fmaxcount)
325 maxcount = atoul(fmaxcount);
326
327 char* fincrtime = strtok(NULL, " "); //add incrtime, default: 0
328 uint32 incrtime = 0;
329 if (fincrtime)
330 incrtime = atoul(fincrtime);
331
332 char* fextendedcost = strtok(NULL, " "); //add ExtendedCost, default: 0
333 uint32 extendedcost = fextendedcost ? atoul(fextendedcost) : 0;
334 Creature* vendor = handler->getSelectedCreature();
335 if (!vendor)
336 {
337 handler->SendSysMessage(LANG_SELECT_CREATURE);
338 handler->SetSentErrorMessage(true);
339 return false;
340 }
341
342 uint32 vendor_entry = vendor->GetEntry();
343
344 if (!sObjectMgr->IsVendorItemValid(vendor_entry, itemId, maxcount, incrtime, extendedcost, type, handler->GetSession()->GetPlayer()))
345 {
346 handler->SetSentErrorMessage(true);
347 return false;
348 }
349
350 sObjectMgr->AddVendorItem(vendor_entry, itemId, maxcount, incrtime, extendedcost, type);
351
352 ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(itemId);
353
354 handler->PSendSysMessage(LANG_ITEM_ADDED_TO_LIST, itemId, itemTemplate->GetDefaultLocaleName(), maxcount, incrtime, extendedcost);
355 return true;
356 }

358 //add move for creature
359 static bool HandleNpcAddMoveCommand(ChatHandler* handler, char const* args)
360 {
361 if (!*args)
362 return false;
363
364 char* guidStr = strtok((char*)args, " ");
365 char* waitStr = strtok((char*)NULL, " ");
366
367 ObjectGuid::LowType lowGuid = strtoull(guidStr, nullptr, 10);
368
369 // attempt check creature existence by DB data
370 CreatureData const* data = sObjectMgr->GetCreatureData(lowGuid);
371 if (!data)
372 {
373 handler->PSendSysMessage(LANG_COMMAND_CREATGUIDNOTFOUND, lowGuid);
374 handler->SetSentErrorMessage(true);
375 return false;
376 }
377
378 int wait = waitStr ? atoi(waitStr) : 0;
379
380 if (wait < 0)
381 wait = 0;
382
383 // Update movement type
384 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_UPD_CREATURE_MOVEMENT_TYPE);
385
386 stmt->setUInt8(0, uint8(WAYPOINT_MOTION_TYPE));
387 stmt->setUInt64(1, lowGuid);
388
389 WorldDatabase.Execute(stmt);
390
391 handler->SendSysMessage(LANG_WAYPOINT_ADDED);
392
393 return true;
394 }

435 //change level of creature or pet
436 static bool HandleNpcSetLevelCommand(ChatHandler* handler, char const* args)
437 {
438 if (!*args)
439 return false;
440
441 uint8 lvl = (uint8) atoi((char*)args);
442 if (lvl < 1 || lvl > sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL) + 3)
443 {
444 handler->SendSysMessage(LANG_BAD_VALUE);
445 handler->SetSentErrorMessage(true);
446 return false;
447 }
448
449 Creature* creature = handler->getSelectedCreature();
450 if (!creature)
451 {
452 handler->SendSysMessage(LANG_SELECT_CREATURE);
453 handler->SetSentErrorMessage(true);
454 return false;
455 }
456
457 if (creature->IsPet())
458 {
459 if (((Pet*)creature)->getPetType() == HUNTER_PET)
460 {
461 creature->SetUInt32Value(UNIT_FIELD_PETNEXTLEVELEXP, sObjectMgr->GetXPForLevel(lvl)/4);
462 creature->SetUInt32Value(UNIT_FIELD_PETEXPERIENCE, 0);
463 }
464 ((Pet*)creature)->GivePetLevel(lvl);
465 }
466 else
467 {
468 creature->SetMaxHealth(100 + 30*lvl);
469 creature->SetHealth(100 + 30*lvl);
470 creature->SetLevel(lvl);
471 creature->SaveToDB();
472 }
473
474 return true;
475 }

477 static bool HandleNpcDeleteCommand(ChatHandler* handler, char const* args)
478 {
479 Creature* unit = NULL;
480
481 if (*args)
482 {
483 // number or [name] Shift-click form |color|Hcreature:creature_guid|h[name]|h|r
484 char* cId = handler->extractKeyFromLink((char*)args, "Hcreature");
485 if (!cId)
486 return false;
487
488 ObjectGuid::LowType lowguid = strtoull(cId, nullptr, 10);
489 if (!lowguid)
490 return false;
491
492 if (CreatureData const* cr_data = sObjectMgr->GetCreatureData(lowguid))
493 unit = handler->GetSession()->GetPlayer()->GetMap()->GetCreature(ObjectGuid::Create<HighGuid::Creature>(cr_data->mapid, cr_data->id, lowguid));
494 }
495 else
496 unit = handler->getSelectedCreature();
497
498 if (!unit || unit->IsPet() || unit->IsTotem())
499 {
500 handler->SendSysMessage(LANG_SELECT_CREATURE);
501 handler->SetSentErrorMessage(true);
502 return false;
503 }
504
505 // Delete the creature
506 unit->CombatStop();
507 unit->DeleteFromDB();
508 unit->AddObjectToRemoveList();
509
510 handler->SendSysMessage(LANG_COMMAND_DELCREATMESSAGE);
511
512 return true;
513 }

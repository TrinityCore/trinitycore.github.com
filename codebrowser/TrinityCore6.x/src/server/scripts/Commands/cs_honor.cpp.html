25#include "Chat.h"

31class honor_commandscript : public CommandScript
32{
33public:
34 honor_commandscript() : CommandScript("honor_commandscript") { }
35
36 std::vector<ChatCommand> GetCommands() const override
37 {
38 static std::vector<ChatCommand> honorAddCommandTable =
39 {
40 { "kill", rbac::RBAC_PERM_COMMAND_HONOR_ADD_KILL, false, &HandleHonorAddKillCommand, "" },
41 { "", rbac::RBAC_PERM_COMMAND_HONOR_ADD, false, &HandleHonorAddCommand, "" },
42 };
43
44 static std::vector<ChatCommand> honorCommandTable =
45 {
46 { "add", rbac::RBAC_PERM_COMMAND_HONOR_ADD, false, NULL, "", honorAddCommandTable },
47 { "update", rbac::RBAC_PERM_COMMAND_HONOR_UPDATE, false, &HandleHonorUpdateCommand, "" },
48 };
49
50 static std::vector<ChatCommand> commandTable =
51 {
52 { "honor", rbac::RBAC_PERM_COMMAND_HONOR, false, NULL, "", honorCommandTable },
53 };
54 return commandTable;
55 }
56
57 static bool HandleHonorAddCommand(ChatHandler* handler, char const* args)
58 {
59 if (!*args)
60 return false;
61
62 Player* target = handler->getSelectedPlayer();
63 if (!target)
64 {
65 handler->SendSysMessage(LANG_PLAYER_NOT_FOUND);
66 handler->SetSentErrorMessage(true);
67 return false;
68 }
69
70 // check online security
71 if (handler->HasLowerSecurity(target, ObjectGuid::Empty))
72 return false;
73
74 uint32 amount = (uint32)atoi(args);
75 target->RewardHonor(NULL, 1, amount);
76 return true;
77 }
78
79 static bool HandleHonorAddKillCommand(ChatHandler* handler, char const* /*args*/)
80 {
81 Unit* target = handler->getSelectedUnit();
82 if (!target)
83 {
84 handler->SendSysMessage(LANG_PLAYER_NOT_FOUND);
85 handler->SetSentErrorMessage(true);
86 return false;
87 }
88
89 // check online security
90 if (Player* player = target->ToPlayer())
91 if (handler->HasLowerSecurity(player, ObjectGuid::Empty))
92 return false;
93
94 handler->GetSession()->GetPlayer()->RewardHonor(target, 1);
95 return true;
96 }
97
98 static bool HandleHonorUpdateCommand(ChatHandler* handler, char const* /*args*/)
99 {
100 Player* target = handler->getSelectedPlayer();
101 if (!target)
102 {
103 handler->SendSysMessage(LANG_PLAYER_NOT_FOUND);
104 handler->SetSentErrorMessage(true);
105 return false;
106 }
107
108 // check online security
109 if (handler->HasLowerSecurity(target, ObjectGuid::Empty))
110 return false;
111
112 target->UpdateHonorFields();
113 return true;
114 }
115};
116
117void AddSC_honor_commandscript()
118{
119 new honor_commandscript();
120}

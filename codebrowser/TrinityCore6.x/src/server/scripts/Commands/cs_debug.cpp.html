110 static bool HandleDebugPlayCinematicCommand(ChatHandler* handler, char const* args)
111 {
112 // USAGE: .debug play cinematic #cinematicid
113 // #cinematicid - ID decimal number from CinemaicSequences.dbc (1st column)
114 if (!*args)
115 {
116 handler->SendSysMessage(LANG_BAD_VALUE);
117 handler->SetSentErrorMessage(true);
118 return false;
119 }
120
121 uint32 id = atoi((char*)args);
122
123 if (!sCinematicSequencesStore.LookupEntry(id))
124 {
125 handler->PSendSysMessage(LANG_CINEMATIC_NOT_EXIST, id);
126 handler->SetSentErrorMessage(true);
127 return false;
128 }
129
130 handler->GetSession()->GetPlayer()->SendCinematicStart(id);
131 return true;
132 }

158 //Play sound
159 static bool HandleDebugPlaySoundCommand(ChatHandler* handler, char const* args)
160 {
161 // USAGE: .debug playsound #soundid
162 // #soundid - ID decimal number from SoundEntries.dbc (1st column)
163 if (!*args)
164 {
165 handler->SendSysMessage(LANG_BAD_VALUE);
166 handler->SetSentErrorMessage(true);
167 return false;
168 }
169
170 uint32 soundId = atoi((char*)args);
171
172 if (!sSoundEntriesStore.LookupEntry(soundId))
173 {
174 handler->PSendSysMessage(LANG_SOUND_NOT_EXIST, soundId);
175 handler->SetSentErrorMessage(true);
176 return false;
177 }
178
179 Unit* unit = handler->getSelectedUnit();
180 if (!unit)
181 {
182 handler->SendSysMessage(LANG_SELECT_CHAR_OR_CREATURE);
183 handler->SetSentErrorMessage(true);
184 return false;
185 }
186
187 if (!handler->GetSession()->GetPlayer()->GetTarget().IsEmpty())
188 unit->PlayDistanceSound(soundId, handler->GetSession()->GetPlayer());
189 else
190 unit->PlayDirectSound(soundId, handler->GetSession()->GetPlayer());
191
192 handler->PSendSysMessage(LANG_YOU_HEAR_SOUND, soundId);
193 return true;
194 }

197 {
198 if (!*args)

200
201 char* result = strtok((char*)args, " ");
202 if (!result)
203 return false;
204
205 uint8 failNum = (uint8)atoi(result);
206 if (failNum == 0 && *result != '0')
207 return false;

219 castFailed.FailedArg1 = failArg1;
220 castFailed.FailedArg2 = failArg2;
221 handler->GetSession()->SendPacket(castFailed.Write());
222
223 return true;

256 static bool HandleDebugSendOpcodeCommand(ChatHandler* handler, char const* /*args*/)
257 {
258 Unit* unit = handler->getSelectedUnit();
259 Player* player = NULL;
260 if (!unit || (unit->GetTypeId() != TYPEID_PLAYER))
261 player = handler->GetSession()->GetPlayer();
262 else
263 player = unit->ToPlayer();
264
265 if (!unit)
266 unit = player;
267
268 std::ifstream ifs("opcode.txt");
269 if (ifs.fail())
270 return false;
271
272 // remove comments from file
273 std::stringstream parsedStream;
274 while (!ifs.eof())
275 {
276 char commentToken[2];
277 ifs.get(commentToken[0]);
278 if (commentToken[0] == '/' && !ifs.eof())
279 {
280 ifs.get(commentToken[1]);
281 // /* comment
282 if (commentToken[1] == '*')
283 {
284 while (!ifs.eof())
285 {
286 ifs.get(commentToken[0]);
287 if (commentToken[0] == '*' && !ifs.eof())
288 {
289 ifs.get(commentToken[1]);
290 if (commentToken[1] == '/')
291 break;
292 else
293 ifs.putback(commentToken[1]);
294 }
295 }
296 continue;
297 }
298 // line comment
299 else if (commentToken[1] == '/')
300 {
301 std::string str;
302 std::getline(ifs, str);
303 continue;
304 }
305 // regular data
306 else
307 ifs.putback(commentToken[1]);
308 }
309 parsedStream.put(commentToken[0]);
310 }
311 ifs.close();
312
313 uint32 opcode;
314 parsedStream >> opcode;
315
316 WorldPacket data(OpcodeServer(opcode), 0);
317
318 while (!parsedStream.eof())
319 {
320 std::string type;
321 parsedStream >> type;
322
323 if (type.empty())
324 break;
325
326 if (type == "uint8")
327 {
328 uint16 val1;
329 parsedStream >> val1;
330 data << uint8(val1);
331 }
332 else if (type == "uint16")
333 {
334 uint16 val2;
335 parsedStream >> val2;
336 data << val2;
337 }
338 else if (type == "uint32")
339 {
340 uint32 val3;
341 parsedStream >> val3;
342 data << val3;
343 }
344 else if (type == "uint64")
345 {
346 uint64 val4;
347 parsedStream >> val4;
348 data << val4;
349 }
350 else if (type == "float")
351 {
352 float val5;
353 parsedStream >> val5;
354 data << val5;
355 }
356 else if (type == "string")
357 {
358 std::string val6;
359 parsedStream >> val6;
360 data << val6;
361 }
362 else if (type == "appitsguid")
363 {
364 data << unit->GetPackGUID();
365 }
366 else if (type == "appmyguid")
367 {
368 data << player->GetPackGUID();
369 }
370 else if (type == "appgoguid")
371 {
372 GameObject* obj = handler->GetNearbyGameObject();
373 if (!obj)
374 {
375 handler->PSendSysMessage(LANG_COMMAND_OBJNOTFOUND, UI64LIT(0));
376 handler->SetSentErrorMessage(true);
377 ifs.close();
378 return false;
379 }
380 data << obj->GetPackGUID();
381 }
382 else if (type == "goguid")
383 {
384 GameObject* obj = handler->GetNearbyGameObject();
385 if (!obj)
386 {
387 handler->PSendSysMessage(LANG_COMMAND_OBJNOTFOUND, UI64LIT(0));
388 handler->SetSentErrorMessage(true);
389 ifs.close();
390 return false;
391 }
392 data << obj->GetGUID();
393 }
394 else if (type == "myguid")
395 {
396 data << player->GetGUID();
397 }
398 else if (type == "itsguid")
399 {
400 data << unit->GetGUID();
401 }
402 else if (type == "itspos")
403 {
404 data << unit->GetPositionX();
405 data << unit->GetPositionY();
406 data << unit->GetPositionZ();
407 }
408 else if (type == "mypos")
409 {
410 data << player->GetPositionX();
411 data << player->GetPositionY();
412 data << player->GetPositionZ();
413 }
414 else
415 {
416 TC_LOG_ERROR("misc", "Sending opcode that has unknown type '%s'", type.c_str());
417 break;
418 }
419 }
420 TC_LOG_DEBUG("network", "Sending opcode %u", data.GetOpcode());
421 data.hexlike();
422 player->GetSession()->SendPacket(&data, true);
423 handler->PSendSysMessage(LANG_COMMAND_OPCODESENT, data.GetOpcode(), unit->GetName().c_str());
424 return true;
425 }

515 static bool HandleDebugGetItemStateCommand(ChatHandler* handler, char const* args)
516 {
517 if (!*args)
518 return false;
519
520 std::string itemState = args;
521
522 ItemUpdateState state = ITEM_UNCHANGED;
523 bool listQueue = false;
524 bool checkAll = false;
525
526 if (itemState == "unchanged")
527 state = ITEM_UNCHANGED;
528 else if (itemState == "changed")
529 state = ITEM_CHANGED;
530 else if (itemState == "new")
531 state = ITEM_NEW;
532 else if (itemState == "removed")
533 state = ITEM_REMOVED;
534 else if (itemState == "queue")
535 listQueue = true;
536 else if (itemState == "check_all")
537 checkAll = true;
538 else
539 return false;
540
541 Player* player = handler->getSelectedPlayer();
542 if (!player)
543 player = handler->GetSession()->GetPlayer();
544
545 if (!listQueue && !checkAll)
546 {
547 itemState = "The player has the following " + itemState + " items: ";
548 handler->SendSysMessage(itemState.c_str());
549 for (uint8 i = PLAYER_SLOT_START; i < PLAYER_SLOT_END; ++i)
550 {
551 if (i >= BUYBACK_SLOT_START && i < BUYBACK_SLOT_END)
552 continue;
553
554 if (Item* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, i))
555 {
556 if (Bag* bag = item->ToBag())
557 {
558 for (uint8 j = 0; j < bag->GetBagSize(); ++j)
559 if (Item* item2 = bag->GetItemByPos(j))
560 if (item2->GetState() == state)
561 handler->PSendSysMessage("bag: 255 slot: %d %s owner: %s", item2->GetSlot(), item2->GetGUID().ToString().c_str(), item2->GetOwnerGUID().ToString().c_str());
562 }
563 else if (item->GetState() == state)
564 handler->PSendSysMessage("bag: 255 slot: %d %s owner: %s", item->GetSlot(), item->GetGUID().ToString().c_str(), item->GetOwnerGUID().ToString().c_str());
565 }
566 }
567 }
568
569 if (listQueue)
570 {
571 std::vector<Item*>& updateQueue = player->GetItemUpdateQueue();
572 for (size_t i = 0; i < updateQueue.size(); ++i)
573 {
574 Item* item = updateQueue[i];
575 if (!item)
576 continue;
577
578 Bag* container = item->GetContainer();
579 uint8 bagSlot = container ? container->GetSlot() : uint8(INVENTORY_SLOT_BAG_0);
580
581 std::string st;
582 switch (item->GetState())
583 {
584 case ITEM_UNCHANGED:
585 st = "unchanged";
586 break;
587 case ITEM_CHANGED:
588 st = "changed";
589 break;
590 case ITEM_NEW:
591 st = "new";
592 break;
593 case ITEM_REMOVED:
594 st = "removed";
595 break;
596 }
597
598 handler->PSendSysMessage("bag: %d slot: %d %s - state: %s", bagSlot, item->GetSlot(), item->GetGUID().ToString().c_str(), st.c_str());
599 }
600 if (updateQueue.empty())
601 handler->PSendSysMessage("The player's updatequeue is empty");
602 }
603
604 if (checkAll)
605 {
606 bool error = false;
607 std::vector<Item*>& updateQueue = player->GetItemUpdateQueue();
608 for (uint8 i = PLAYER_SLOT_START; i < PLAYER_SLOT_END; ++i)
609 {
610 if (i >= BUYBACK_SLOT_START && i < BUYBACK_SLOT_END)
611 continue;
612
613 Item* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, i);
614 if (!item)
615 continue;
616
617 if (item->GetSlot() != i)
618 {
619 handler->PSendSysMessage("Item with slot %d and %s has an incorrect slot value: %d", i, item->GetGUID().ToString().c_str(), item->GetSlot());
620 error = true;
621 continue;
622 }
623
624 if (item->GetOwnerGUID() != player->GetGUID())
625 {
626 handler->PSendSysMessage("The item with slot %d %s does have non-matching owner guid %s and %s!", item->GetSlot(), item->GetGUID().ToString().c_str(), item->GetOwnerGUID().ToString().c_str(), player->GetGUID().ToString().c_str());
627 error = true;
628 continue;
629 }
630
631 if (Bag* container = item->GetContainer())
632 {
633 handler->PSendSysMessage("The item with slot %d %s has a container (slot: %d, %s) but shouldn't!", item->GetSlot(), item->GetGUID().ToString().c_str(), container->GetSlot(), container->GetGUID().ToString().c_str());
634 error = true;
635 continue;
636 }
637
638 if (item->IsInUpdateQueue())
639 {
640 uint16 qp = item->GetQueuePos();
641 if (qp > updateQueue.size())
642 {
643 handler->PSendSysMessage("The item with slot %d and %s has its queuepos (%d) larger than the update queue size! ", item->GetSlot(), item->GetGUID().ToString().c_str(), qp);
644 error = true;
645 continue;
646 }
647
648 if (updateQueue[qp] == NULL)
649 {
650 handler->PSendSysMessage("The item with slot %d and %s has its queuepos (%d) pointing to NULL in the queue!", item->GetSlot(), item->GetGUID().ToString().c_str(), qp);
651 error = true;
652 continue;
653 }
654
655 if (updateQueue[qp] != item)
656 {
657 handler->PSendSysMessage("The item with slot %d and %s has a queuepos (%d) that points to another item in the queue (bag: %d, slot: %d, %s)", item->GetSlot(), item->GetGUID().ToString().c_str(), qp, updateQueue[qp]->GetBagSlot(), updateQueue[qp]->GetSlot(), updateQueue[qp]->GetGUID().ToString().c_str());
658 error = true;
659 continue;
660 }
661 }
662 else if (item->GetState() != ITEM_UNCHANGED)
663 {

681 continue;
682 }

684 if (item2->GetOwnerGUID() != player->GetGUID())

686 handler->PSendSysMessage("The item in bag %d at slot %d and %s, the owner (%s) and the player (%s) don't match!", bag->GetSlot(), item2->GetSlot(), item2->GetGUID().ToString().c_str(), item2->GetOwnerGUID().ToString().c_str(), player->GetGUID().ToString().c_str());
687 error = true;
688 continue;
689 }
690
691 Bag* container = item2->GetContainer();
692 if (!container)
693 {
694 handler->PSendSysMessage("The item in bag %d at slot %d %s has no container!", bag->GetSlot(), item2->GetSlot(), item2->GetGUID().ToString().c_str());
695 error = true;
696 continue;
697 }
698
699 if (container != bag)

704 }

733 error = true;

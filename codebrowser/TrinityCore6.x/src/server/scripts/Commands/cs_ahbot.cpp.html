75 static bool HandleAHBotItemsAmountCommand(ChatHandler* handler, const char* args)
76 {
77 uint32 qVals[MAX_AUCTION_QUALITY];
78 char* arg = strtok((char*)args, " ");
79 for (int i = 0; i < MAX_AUCTION_QUALITY; ++i)
80 {
81 if (!arg)
82 return false;
83 qVals[i] = atoi(arg);
84 arg = strtok(NULL, " ");
85 }
86
87 sAuctionBot->SetItemsAmount(qVals);
88
89 for (int i = 0; i < MAX_AUCTION_QUALITY; ++i)
90 handler->PSendSysMessage(LANG_AHBOT_ITEMS_AMOUNT, handler->GetTrinityString(ahbotQualityIds[i]), sAuctionBotConfig->GetConfigItemQualityAmount(AuctionQuality(i)));
91
92 return true;
93 }

95 template <AuctionQuality Q>
96 static bool HandleAHBotItemsAmountQualityCommand(ChatHandler* handler, const char* args)
97 {
98 char* arg = strtok((char*)args, " ");
99 if (!arg)
100 return false;
101 uint32 qualityVal = atoi(arg);
102
103 sAuctionBot->SetItemsAmountForQuality(Q, qualityVal);
104 handler->PSendSysMessage(LANG_AHBOT_ITEMS_AMOUNT, handler->GetTrinityString(ahbotQualityIds[Q]),
105 sAuctionBotConfig->GetConfigItemQualityAmount(Q));
106
107 return true;
108 }

110 static bool HandleAHBotItemsRatioCommand(ChatHandler* handler, const char* args)
111 {
112 uint32 rVal[MAX_AUCTION_QUALITY];
113 char* arg = strtok((char*)args, " ");
114 for (int i = 0; i < MAX_AUCTION_QUALITY; ++i)
115 {
116 if (!arg)
117 return false;
118 rVal[i] = atoi(arg);
119 arg = strtok(NULL, " ");
120 }
121
122 sAuctionBot->SetItemsRatio(rVal[0], rVal[1], rVal[2]);
123
124 for (int i = 0; i < MAX_AUCTION_HOUSE_TYPE; ++i)
125 handler->PSendSysMessage(LANG_AHBOT_ITEMS_RATIO, AuctionBotConfig::GetHouseTypeName(AuctionHouseType(i)), sAuctionBotConfig->GetConfigItemAmountRatio(AuctionHouseType(i)));
126 return true;
127 }

161 static bool HandleAHBotStatusCommand(ChatHandler* handler, const char* args)
162 {
163 char* arg = strtok((char*)args, " ");
164 if (!arg)
165 return false;
166
167 bool all = false;
168 if (strcmp(arg, "all") == 0)
169 all = true;
170
171 AuctionHouseBotStatusInfo statusInfo;
172 sAuctionBot->PrepareStatusInfos(statusInfo);
173
174 WorldSession* session = handler->GetSession();
175
176 if (!session)
177 {
178 handler->SendSysMessage(LANG_AHBOT_STATUS_BAR_CONSOLE);
179 handler->SendSysMessage(LANG_AHBOT_STATUS_TITLE1_CONSOLE);
180 handler->SendSysMessage(LANG_AHBOT_STATUS_MIDBAR_CONSOLE);
181 }
182 else
183 handler->SendSysMessage(LANG_AHBOT_STATUS_TITLE1_CHAT);
184
185 uint32 fmtId = session ? LANG_AHBOT_STATUS_FORMAT_CHAT : LANG_AHBOT_STATUS_FORMAT_CONSOLE;
186
187 handler->PSendSysMessage(fmtId, handler->GetTrinityString(LANG_AHBOT_STATUS_ITEM_COUNT),
188 statusInfo[AUCTION_HOUSE_ALLIANCE].ItemsCount,
189 statusInfo[AUCTION_HOUSE_HORDE].ItemsCount,
190 statusInfo[AUCTION_HOUSE_NEUTRAL].ItemsCount,
191 statusInfo[AUCTION_HOUSE_ALLIANCE].ItemsCount +
192 statusInfo[AUCTION_HOUSE_HORDE].ItemsCount +
193 statusInfo[AUCTION_HOUSE_NEUTRAL].ItemsCount);
194
195 if (all)
196 {
197 handler->PSendSysMessage(fmtId, handler->GetTrinityString(LANG_AHBOT_STATUS_ITEM_RATIO),
198 sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ALLIANCE_ITEM_AMOUNT_RATIO),
199 sAuctionBotConfig->GetConfig(CONFIG_AHBOT_HORDE_ITEM_AMOUNT_RATIO),
200 sAuctionBotConfig->GetConfig(CONFIG_AHBOT_NEUTRAL_ITEM_AMOUNT_RATIO),
201 sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ALLIANCE_ITEM_AMOUNT_RATIO) +
202 sAuctionBotConfig->GetConfig(CONFIG_AHBOT_HORDE_ITEM_AMOUNT_RATIO) +
203 sAuctionBotConfig->GetConfig(CONFIG_AHBOT_NEUTRAL_ITEM_AMOUNT_RATIO));
204
205 if (!session)
206 {
207 handler->SendSysMessage(LANG_AHBOT_STATUS_BAR_CONSOLE);
208 handler->SendSysMessage(LANG_AHBOT_STATUS_TITLE2_CONSOLE);
209 handler->SendSysMessage(LANG_AHBOT_STATUS_MIDBAR_CONSOLE);
210 }
211 else
212 handler->SendSysMessage(LANG_AHBOT_STATUS_TITLE2_CHAT);
213
214 for (int i = 0; i < MAX_AUCTION_QUALITY; ++i)
215 handler->PSendSysMessage(fmtId, handler->GetTrinityString(ahbotQualityIds[i]),
216 statusInfo[AUCTION_HOUSE_ALLIANCE].QualityInfo[i],
217 statusInfo[AUCTION_HOUSE_HORDE].QualityInfo[i],
218 statusInfo[AUCTION_HOUSE_NEUTRAL].QualityInfo[i],
219 sAuctionBotConfig->GetConfigItemQualityAmount(AuctionQuality(i)));
220 }
221
222 if (!session)
223 handler->SendSysMessage(LANG_AHBOT_STATUS_BAR_CONSOLE);
224
225 return true;
226 }

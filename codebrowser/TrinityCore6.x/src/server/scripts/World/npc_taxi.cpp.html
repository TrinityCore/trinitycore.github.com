66 bool OnGossipHello(Player* player, Creature* creature) override
67 {
68 if (creature->IsQuestGiver())
69 player->PrepareQuestMenu(creature->GetGUID());
70
71 switch (creature->GetEntry())
72 {
73 case 17435: // Azuremyst Isle - Susurrus
74 if (player->HasItemCount(23843, 1, true))
75 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_SUSURRUS, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
76 break;
77 case 20903: // Netherstorm - Protectorate Nether Drake
78 if (player->GetQuestStatus(10438) == QUEST_STATUS_INCOMPLETE && player->HasItemCount(29778))
79 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_NETHER_DRAKE, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
80 break;
81 case 18725: // Old Hillsbrad Foothills - Brazen
82 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_BRAZEN, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
83 break;
84 case 29154: // Stormwind City - Thargold Ironwing
85 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_IRONWING, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
86 break;
87 case 19409: // Hellfire Peninsula - Wing Commander Dabir'ee
88 //Mission: The Murketh and Shaadraz Gateways
89 if (player->GetQuestStatus(10146) == QUEST_STATUS_INCOMPLETE)
90 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_DABIREE1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 4);
91
92 //Shatter Point
93 if (!player->GetQuestRewardStatus(10340))
94 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_DABIREE2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 5);
95 break;
96 case 20235: // Hellfire Peninsula - Gryphoneer Windbellow
97 //Mission: The Abyssal Shelf || Return to the Abyssal Shelf
98 if (player->GetQuestStatus(10163) == QUEST_STATUS_INCOMPLETE || player->GetQuestStatus(10346) == QUEST_STATUS_INCOMPLETE)
99 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_WINDBELLOW1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 6);
100
101 //Go to the Front
102 if (player->GetQuestStatus(10382) != QUEST_STATUS_NONE && !player->GetQuestRewardStatus(10382))
103 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_WINDBELLOW2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 7);
104 break;
105 case 19401: // Hellfire Peninsula - Wing Commander Brack

108 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_BRACK1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 8);
109
110 //Mission: The Abyssal Shelf || Return to the Abyssal Shelf

117 break;
118 case 23413: // Blade's Edge Mountains - Skyguard Handler Irena
119 if (player->GetReputationRank(1031) >= REP_HONORED)
120 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_IRENA, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 11);
121 break;
122 case 25059: // Isle of Quel'Danas - Ayren Cloudbreaker
123 if (player->GetQuestStatus(11532) == QUEST_STATUS_INCOMPLETE || player->GetQuestStatus(11533) == QUEST_STATUS_INCOMPLETE)

125

127 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_CLOUDBREAKER2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 13);
128 break;
129 case 25236: // Isle of Quel'Danas - Unrestrained Dragonhawk
130 if (player->GetQuestStatus(11542) == QUEST_STATUS_COMPLETE || player->GetQuestStatus(11543) == QUEST_STATUS_COMPLETE)

167 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_WILLIAMKEILAR3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 28);
168 break;
169 }

171 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
172 return true;

175 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
176 {
177 player->PlayerTalkClass->ClearMenus();
178 switch (action)
179 {
180 case GOSSIP_ACTION_INFO_DEF:
181 //spellId is correct, however it gives flight a somewhat funny effect //TaxiPath 506.
182 player->CLOSE_GOSSIP_MENU();
183 player->CastSpell(player, 32474, true);
184 break;
185 case GOSSIP_ACTION_INFO_DEF + 1:
186 player->CLOSE_GOSSIP_MENU();
187 player->ActivateTaxiPathTo(627); //TaxiPath 627 (possibly 627+628(152->153->154->155))
188 break;
189 case GOSSIP_ACTION_INFO_DEF + 2:
190 if (!player->HasItemCount(25853))

193 {
194 player->CLOSE_GOSSIP_MENU();

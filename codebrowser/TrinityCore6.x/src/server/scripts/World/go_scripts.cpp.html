202enum EthereumPrison
203{
204 SPELL_REP_LC = 39456,
205 SPELL_REP_SHAT = 39457,
206 SPELL_REP_CE = 39460,
207 SPELL_REP_CON = 39474,
208 SPELL_REP_KT = 39475,
209 SPELL_REP_SPOR = 39476
210};
211
212const uint32 NpcPrisonEntry[] =
213{
214 22810, 22811, 22812, 22813, 22814, 22815, //good guys
215 20783, 20784, 20785, 20786, 20788, 20789, 20790 //bad guys
216};
217
218class go_ethereum_prison : public GameObjectScript
219{
220public:
221 go_ethereum_prison() : GameObjectScript("go_ethereum_prison") { }
222
223 bool OnGossipHello(Player* player, GameObject* go) override
224 {
225 go->UseDoorOrButton();
226 int Random = rand32() % (sizeof(NpcPrisonEntry) / sizeof(uint32));
227
228 if (Creature* creature = player->SummonCreature(NpcPrisonEntry[Random], go->GetPositionX(), go->GetPositionY(), go->GetPositionZ(), go->GetAngle(player),
229 TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000))
230 {
231 if (!creature->IsHostileTo(player))
232 {
233 if (FactionTemplateEntry const* pFaction = creature->GetFactionTemplateEntry())
234 {
235 uint32 Spell = 0;
236
237 switch (pFaction->Faction)
238 {
239 case 1011: Spell = SPELL_REP_LC; break;
240 case 935: Spell = SPELL_REP_SHAT; break;
241 case 942: Spell = SPELL_REP_CE; break;
242 case 933: Spell = SPELL_REP_CON; break;
243 case 989: Spell = SPELL_REP_KT; break;
244 case 970: Spell = SPELL_REP_SPOR; break;
245 }
246
247 if (Spell)
248 creature->CastSpell(player, Spell, false);
249 else
250 TC_LOG_ERROR("scripts", "go_ethereum_prison summoned Creature (entry %u) but faction (%u) are not expected by script.", creature->GetEntry(), creature->getFaction());
251 }
252 }
253 }
254
255 return false;
256 }
257};

336{
337 NPC_HAWK_GUARD = 22992,
338 NPC_EAGLE_GUARD = 22993,
339 NPC_FALCON_GUARD = 22994,
340 GO_SHRINE_HAWK = 185551,
341 GO_SHRINE_EAGLE = 185547,
342 GO_SHRINE_FALCON = 185553
343};
344
345class go_shrine_of_the_birds : public GameObjectScript
346{
347public:
348 go_shrine_of_the_birds() : GameObjectScript("go_shrine_of_the_birds") { }
349
350 bool OnGossipHello(Player* player, GameObject* go) override
351 {
352 uint32 BirdEntry = 0;
353
354 float fX, fY, fZ;
355 go->GetClosePoint(fX, fY, fZ, go->GetObjectSize(), INTERACTION_DISTANCE);
356
357 switch (go->GetEntry())
358 {
359 case GO_SHRINE_HAWK:
360 BirdEntry = NPC_HAWK_GUARD;
361 break;
362 case GO_SHRINE_EAGLE:
363 BirdEntry = NPC_EAGLE_GUARD;
364 break;
365 case GO_SHRINE_FALCON:
366 BirdEntry = NPC_FALCON_GUARD;
367 break;
368 }
369
370 if (BirdEntry)
371 player->SummonCreature(BirdEntry, fX, fY, fZ, go->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
372
373 return false;
374 }

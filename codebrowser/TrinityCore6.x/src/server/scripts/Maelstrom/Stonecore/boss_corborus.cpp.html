81class boss_corborus : public CreatureScript
82{
83 public:
84 boss_corborus() : CreatureScript("boss_corborus") { }
85
86 struct boss_corborusAI : public BossAI
87 {
88 boss_corborusAI(Creature* creature) : BossAI(creature, DATA_CORBORUS)
89 {
90 stateIntro = NOT_STARTED;
91 }
92
93 void Reset() override
94 {
95 _Reset();
96
97 countTrashingCharge = 0;
98
99 events.ScheduleEvent(EVENT_DAMPENING_WAVE, 10000);
100 events.ScheduleEvent(EVENT_CRYSTAL_BARRAGE, 15000);
101 events.ScheduleEvent(EVENT_SUBMERGE, 36000);
102 }
103
104 void DoAction(int32 action) override
105 {
106 switch (action)
107 {
108 case ACTION_CORBORUS_INTRO: // Executes Corborus intro event
109 {
110 if (stateIntro != NOT_STARTED)
111 return;
112
113 stateIntro = IN_PROGRESS;
114
115 if (Creature* Millhouse = instance->GetCreature(DATA_MILLHOUSE_MANASTORM))
116 {
117 Millhouse->InterruptNonMeleeSpells(true);
118 Millhouse->RemoveAllAuras();
119 Millhouse->HandleEmoteCommand(EMOTE_ONESHOT_KNOCKDOWN);
120 }
121
122 events.ScheduleEvent(EVENT_CORBORUS_CHARGE, 1000);
123 break;
124 }
125 default:
126 break;
127 }
128 }
129
130 void UpdateAI(uint32 diff) override
131 {
132 if (!UpdateVictim() && stateIntro != IN_PROGRESS)
133 return;
134
135 events.Update(diff);
136
137 if (me->HasUnitState(UNIT_STATE_CASTING))
138 return;
139
140 while (uint32 eventId = events.ExecuteEvent())
141 {
142 switch (eventId)
143 {
144 case EVENT_CORBORUS_CHARGE:
145 // Face Millhouse and other mobs
146 instance->SetData(DATA_MILLHOUSE_EVENT_FACE, 0);
147
148 // Open rock gate and cast visual from nearby worldtrigger
149 instance->SetData(DATA_HANDLE_CORBORUS_ROCKDOOR, 0);
150 if (Creature* worldtrigger = me->FindNearestCreature(NPC_WORLDTRIGGER, 60.0f))
151 worldtrigger->CastSpell(worldtrigger, SPELL_DOOR_BREAK, true);
152
153 // Make Corborus charge
154 DoCast(me, SPELL_RING_WYRM_CHARGE, true);
155
156 events.ScheduleEvent(EVENT_CORBORUS_KNOCKBACK, 1000);
157 break;
158 case EVENT_CORBORUS_KNOCKBACK:
159 // Spawn Twilight Documents (quest gameobject)
160 if (Creature* Millhouse = instance->GetCreature(DATA_MILLHOUSE_MANASTORM))
161 Millhouse->CastSpell(Millhouse, SPELL_TWILIGHT_DOCUMENTS, true);
162
163 // Knockback Millhouse and other mobs
164 instance->SetData(DATA_MILLHOUSE_EVENT_KNOCKBACK, 0);
165
166 events.ScheduleEvent(EVENT_CORBORUS_FACEPLAYERS, 2000);
167 break;
168 case EVENT_CORBORUS_FACEPLAYERS:
169 // Face Corborus to players and set new home position
170 me->SetFacingTo(3.176499f);
171 me->SetHomePosition(1154.55f, 878.843f, 284.963f, 3.176499f);
172 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
173
174 // Despawn Millhouse and all trash
175 instance->SetData(DATA_MILLHOUSE_EVENT_DESPAWN, 0);
176
177 stateIntro = DONE;
178 break;
179 case EVENT_DAMPENING_WAVE:
180 DoCastVictim(SPELL_DAMPENING_WAVE);
181 events.ScheduleEvent(EVENT_DAMPENING_WAVE, 15000);
182 break;
183 case EVENT_CRYSTAL_BARRAGE:
184 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
185 DoCast(target, SPELL_CRYSTAL_BARRAGE);
186 events.ScheduleEvent(EVENT_CRYSTAL_BARRAGE, 10000);
187 break;
188 case EVENT_SUBMERGE:
189 events.RescheduleEvent(EVENT_DAMPENING_WAVE, 35000);
190 events.RescheduleEvent(EVENT_CRYSTAL_BARRAGE, 30000);
191 events.RescheduleEvent(EVENT_SUBMERGE, 100000);
192
193 me->SetReactState(REACT_PASSIVE);
194 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
195 DoCast(me, SPELL_CLEAR_ALL_DEBUFFS);
196 me->AttackStop();
197
198 DoCast(me, SPELL_SUBMERGE);
199
200 countTrashingCharge = 0;
201 events.ScheduleEvent(EVENT_TELEPORT, 500);
202 break;
203 case EVENT_TELEPORT:
204 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
205 DoCast(target, SPELL_TRASHING_CHARGE_TELEPORT);
206 countTrashingCharge += 1;
207 if (countTrashingCharge <= 4)
208 events.ScheduleEvent(EVENT_TRASHING_CHARGE, 1000);
209 else
210 events.ScheduleEvent(EVENT_EMERGE, 2500);
211 break;
212 case EVENT_TRASHING_CHARGE:
213 DoCast(me, SPELL_SUMMON_TRASHING_CHARGE);
214 DoCast(me, SPELL_TRASHING_CHARGE_VISUAL);
215 events.ScheduleEvent(EVENT_TELEPORT, 5000);
216 break;
217 case EVENT_EMERGE:
218 me->RemoveAllAuras();
219 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
220 DoCast(me, SPELL_EMERGE);
221 events.ScheduleEvent(EVENT_ATTACK, 2500);
222 break;
223 case EVENT_ATTACK:
224 me->SetReactState(REACT_AGGRESSIVE);
225 break;
226 default:
227 break;
228 }
229 }
230
231 DoMeleeAttackIfReady();
232 }
233
234 void JustSummoned(Creature* summon) override
235 {
236 if (summon->GetEntry() == NPC_TRASHING_CHARGE)
237 {
238 summon->SetReactState(REACT_PASSIVE);
239 summon->CastSpell(summon, SPELL_TRASHING_CHARGE_EFFECT);
240 summon->DespawnOrUnsummon(6000);
241 }
242
243 BossAI::JustSummoned(summon);
244 }
245
246 private:
247 EncounterState stateIntro;
248 uint32 countTrashingCharge = 0;
249 };
250
251 CreatureAI* GetAI(Creature* creature) const override
252 {
253 return GetInstanceAI<boss_corborusAI>(creature);
254 }

257// 43391 - Rock Borer
258class npc_rock_borer : public CreatureScript
259{
260 public:
261 npc_rock_borer() : CreatureScript("npc_rock_borer") { }
262
263 struct npc_rock_borerAI : public ScriptedAI
264 {
265 npc_rock_borerAI(Creature* creature) : ScriptedAI(creature)
266 {
267 me->SetDisableGravity(true);
268 me->SetReactState(REACT_PASSIVE);
269 events.ScheduleEvent(EVENT_EMERGED, 1200);
270 events.ScheduleEvent(EVENT_ROCK_BORE, urand(15000, 20000)); // Need sniffs for this timer
271 }
272
273 void IsSummonedBy(Unit* summoner) override
274 {
275 me->SetInCombatState(false, summoner);
276 DoCast(me, SPELL_ROCK_BORER_EMERGE);
277 }
278
279 void UpdateAI(uint32 diff) override
280 {
281 if (!UpdateVictim() && me->GetReactState() != REACT_PASSIVE)
282 return;
283
284 events.Update(diff);
285
286 if (me->HasUnitState(UNIT_STATE_CASTING))
287 return;
288
289 while (uint32 eventId = events.ExecuteEvent())
290 {
291 switch (eventId)
292 {
293 case EVENT_EMERGED:
294 me->RemoveAurasDueToSpell(SPELL_ROCK_BORER_EMERGE);
295 me->SetReactState(REACT_AGGRESSIVE);
296 break;
297 case EVENT_ROCK_BORE:
298 DoCast(me, SPELL_ROCK_BORE);
299 events.ScheduleEvent(EVENT_ROCK_BORE, urand(15000, 20000)); // Need sniffs for this timer
300 break;
301 default:
302 break;
303 }
304 }
305
306 DoMeleeAttackIfReady();
307 }
308
309 private:
310 EventMap events;
311 };
312
313 CreatureAI* GetAI(Creature* creature) const override
314 {
315 return GetInstanceAI<npc_rock_borerAI>(creature);
316 }
317};

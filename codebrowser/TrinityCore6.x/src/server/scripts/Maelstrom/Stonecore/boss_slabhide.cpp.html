104class boss_slabhide : public CreatureScript
105{
106 public:
107 boss_slabhide() : CreatureScript("boss_slabhide") { }
108
109 struct boss_slabhideAI : public BossAI
110 {
111 boss_slabhideAI(Creature* creature) : BossAI(creature, DATA_SLABHIDE)
112 {
113 me->setActive(true);
114 me->SetCanFly(true);
115 me->SetDisableGravity(true);
116 me->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
117 me->SetReactState(REACT_PASSIVE);
118 instance->SetData(DATA_SLABHIDE_INTRO, NOT_STARTED);
119 _isFlying = false;
120 }
121
122 void Reset() override
123 {
124 if (instance->GetData(DATA_SLABHIDE_INTRO) == NOT_STARTED)
125 return;
126
127 _Reset();
128 DespawnAll();
129
130 me->SetCanFly(false);
131 me->SetDisableGravity(false);
132 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
133 me->SetReactState(REACT_AGGRESSIVE);
134 _isFlying = false;
135 }
136
137 void EnterCombat(Unit* /*victim*/) override
138 {
139 _EnterCombat();
140
141 events.ScheduleEvent(EVENT_HANDLE_ROCK_WALLS, 4000);
142 events.ScheduleEvent(EVENT_LAVA_FISSURE, urand(6000, 8000));
143 events.ScheduleEvent(EVENT_SAND_BLAST, urand(8000, 10000));
144 events.ScheduleEvent(EVENT_AIR_PHASE, 10000);
145 }
146
147 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
148 {
149 if (_isFlying && damage >= me->GetHealth())
150 damage = me->GetHealth() - 1; // Let creature health fall to 1 hp but prevent it from dying during air phase.
151 }
152
153 void JustDied(Unit* /*killer*/) override
154 {
155 _JustDied();
156
157 // Despawn related npcs and gameobjects
158 DespawnAll();
159 }
160
161 void DoAction(int32 action) override
162 {
163 switch (action)
164 {
165 case ACTION_SLABHIDE_INTRO:
166 {
167 if (instance->GetData(DATA_SLABHIDE_INTRO) != NOT_STARTED)
168 return;
169
170 instance->SetData(DATA_SLABHIDE_INTRO, IN_PROGRESS);
171
172 // Execute Slabhide intro event
173 me->GetMotionMaster()->MovePoint(POINT_SLABHIDE_INTRO, SlabhideIntroPos);
174 break;
175 }
176 default:
177 break;
178 }
179 }
180
181 void MovementInform(uint32 type, uint32 id) override
182 {
183 if (type != POINT_MOTION_TYPE && type != EFFECT_MOTION_TYPE)
184 return;
185
186 switch (id)
187 {
188 case POINT_SLABHIDE_INTRO:
189 me->SetFacingTo(SlabhideIntroLandPos.GetOrientation());
190 me->GetMotionMaster()->MoveLand(POINT_SLABHIDE_INTRO_LAND, SlabhideIntroLandPos);
191 break;
192 case POINT_SLABHIDE_INTRO_LAND:
193 me->SetCanFly(false);
194 me->SetDisableGravity(false);
195 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
196 me->SetHover(false);
197 me->SetHomePosition(SlabhideIntroLandPos);
198 me->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
199 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
200 me->SetReactState(REACT_AGGRESSIVE);
201 instance->SetData(DATA_SLABHIDE_INTRO, DONE);
202 break;
203 case POINT_SLABHIDE_MIDDLE:
204 _isFlying = true;
205 events.ScheduleEvent(EVENT_TAKEOFF, 100);
206 break;
207 case POINT_SLABHIDE_IN_AIR:
208 events.ScheduleEvent(EVENT_STALACTITE, 400);
209 break;
210 case POINT_SLABHIDE_LAND:
211 _isFlying = false;
212 //DoCast(me, SPELL_COOLDOWN_5S); // unknown purpose
213 events.ScheduleEvent(EVENT_ATTACK, 1200);
214 break;
215 default:
216 break;
217 }
218 }
219
220 void UpdateAI(uint32 diff) override
221 {
222 if (!UpdateVictim())
223 return;
224
225 events.Update(diff);
226
227 if (me->HasUnitState(UNIT_STATE_CASTING))
228 return;
229
230 while (uint32 eventId = events.ExecuteEvent())
231 {
232 switch (eventId)
233 {
234 case EVENT_HANDLE_ROCK_WALLS: // Close rock walls
235 instance->SetData(DATA_SLABHIDE_ROCK_WALL, false);
236 break;
237 case EVENT_LAVA_FISSURE:
238 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
239 DoCast(target, SPELL_LAVA_FISSURE);
240 events.ScheduleEvent(EVENT_LAVA_FISSURE, urand(6000, 8000));
241 break;
242 case EVENT_SAND_BLAST:
243 DoCast(me, SPELL_SAND_BLAST);
244 events.ScheduleEvent(EVENT_SAND_BLAST, urand(8000, 11000));
245 break;
246 case EVENT_AIR_PHASE:
247 events.Reset();
248 me->SetReactState(REACT_PASSIVE);
249 me->AttackStop();
250 me->GetMotionMaster()->MovePoint(POINT_SLABHIDE_MIDDLE, SlabhideMiddlePos);
251 events.ScheduleEvent(EVENT_AIR_PHASE, 60000);
252 break;
253 case EVENT_TAKEOFF:
254 me->GetMotionMaster()->MoveTakeoff(POINT_SLABHIDE_IN_AIR, SlabhideInAirPos);
255 break;
256 case EVENT_STALACTITE:
257 me->SetCanFly(true);
258 me->SetDisableGravity(true);
259 me->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
260 me->SetHover(true);
261
262 DoCast(me, SPELL_STALACTITE_SUMMON);
263
264 events.ScheduleEvent(EVENT_LAND, 8000);
265 break;
266 case EVENT_LAND:
267 {
268 Position pos = me->GetPosition();
269 pos.m_positionZ = me->GetMap()->GetHeight(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ());
270 me->GetMotionMaster()->MoveLand(POINT_SLABHIDE_LAND, pos);
271 break;
272 }
273 case EVENT_ATTACK:
274 me->SetCanFly(false);
275 me->SetDisableGravity(false);
276 me->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_ALWAYS_STAND | UNIT_BYTE1_FLAG_HOVER);
277 me->SetHover(false);
278
279 events.ScheduleEvent(EVENT_LAVA_FISSURE, urand(6000, 8000));
280 events.ScheduleEvent(EVENT_SAND_BLAST, urand(8000, 10000));
281 DoCast(me, SPELL_CRYSTAL_STORM);
282 me->SetReactState(REACT_AGGRESSIVE);
283 break;
284 default:
285 break;
286 }
287 }
288
289 DoMeleeAttackIfReady();
290 }
291
292 private:
293 void DespawnAll()
294 {
295 // Despawn stalactite triggers npcs
296 std::list<Creature*> listStalactiteTrigger;
297 me->GetCreatureListWithEntryInGrid(listStalactiteTrigger, NPC_STALACTITE_TRIGGER, 200.0f);
298 if (!listStalactiteTrigger.empty())
299 for (std::list<Creature*>::const_iterator itr = listStalactiteTrigger.begin(); itr != listStalactiteTrigger.end(); ++itr)
300 (*itr)->DespawnOrUnsummon();
301
302 // Despawn stalactite objects
303 std::list<GameObject*> listStalactite;
304 me->GetGameObjectListWithEntryInGrid(listStalactite, GO_STALACTITE, 200.0f);
305 if (!listStalactite.empty())
306 for (std::list<GameObject*>::const_iterator itr = listStalactite.begin(); itr != listStalactite.end(); ++itr)
307 (*itr)->Delete();
308 }
309
310 bool _isFlying;
311 };

315 return GetInstanceAI<boss_slabhideAI>(creature);

534// 92300 - Crystal Storm (damage)
535class BehindObjectCheck
536{
537 public:
538 BehindObjectCheck(Unit* caster, std::list<GameObject*> objectList) : caster(caster), objectList(objectList) { }
539
540 bool operator()(WorldObject* unit)
541 {
542 for (std::list<GameObject*>::const_iterator itr = objectList.begin(); itr != objectList.end(); ++itr)
543 if (!(*itr)->IsInvisibleDueToDespawn() && (*itr)->IsInBetween(caster, unit, 1.5f))
544 return true;
545 return false;
546 }
547
548 private:
549 Unit* caster;
550 std::list<GameObject*> objectList;
551};

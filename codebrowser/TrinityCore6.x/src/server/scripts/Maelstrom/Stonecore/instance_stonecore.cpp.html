36ObjectData const creatureData[] =
37{
38 { NPC_MILLHOUSE_MANASTORM, DATA_MILLHOUSE_MANASTORM },
39 { NPC_CORBORUS, DATA_CORBORUS },
40 { NPC_SLABHIDE, DATA_SLABHIDE },
41 { NPC_HIGH_PRIESTESS_AZIL, DATA_HIGH_PRIESTESS_AZIL },
42 { NPC_STONECORE_TELEPORTER, DATA_STONECORE_TELEPORTER },
43 { NPC_STONECORE_TELEPORTER_2, DATA_STONECORE_TELEPORTER_2 },
44 { 0, 0 } // END
45};
46
47class instance_stonecore : public InstanceMapScript
48{
49 public:
50 instance_stonecore() : InstanceMapScript(SCScriptName, 725) { }
51
52 struct instance_stonecore_InstanceScript : public InstanceScript
53 {
54 instance_stonecore_InstanceScript(Map* map) : InstanceScript(map)
55 {
56 SetHeaders(DataHeader);
57 SetBossNumber(MAX_ENCOUNTER);
58 LoadObjectData(creatureData, nullptr);
59 }
60
61 void OnGameObjectCreate(GameObject* go) override
62 {
63 switch (go->GetEntry())
64 {
65 case GAMEOBJECT_CORBORUS_ROCKDOOR:
66 corborusRockDoorGUID = go->GetGUID();
67 go->SetGoState(GetBossState(DATA_CORBORUS) != DONE ? GO_STATE_READY : GO_STATE_ACTIVE);
68 break;
69 case GAMEOBJECT_SLABHIDE_ROCK_WALL:
70 slabhideRockWallGUIDs.push_back(go->GetGUID());
71 break;
72 default:
73 break;
74 }
75 }
76
77 void OnCreatureCreate(Creature* creature) override
78 {
79 switch (creature->GetEntry())
80 {
81 case NPC_STONECORE_TELEPORTER:
82 case NPC_STONECORE_TELEPORTER_2:
83 if (GetBossState(DATA_SLABHIDE) == DONE)
84 ActivateTeleporter(creature);
85 break;
86 default:
87 break;
88 }
89
90 // Check if creature is part of Millhouse event
91 creature->SearchFormation();
92 if (CreatureGroup* group = creature->GetFormation())
93 {
94 switch (group->GetId())
95 {
96 case CREATURE_FORMATION_MILLHOUSE_EVENT_TRASH:
97 millhouseTrashGUIDs.push_back(creature->GetGUID());
98 break;
99 case CREATURE_FORMATION_MILLHOUSE_EVENT_LAST_GROUP:
100 millhouseLastGroupGUIDs.push_back(creature->GetGUID());
101 creature->SetReactState(REACT_PASSIVE);
102 creature->SetMeleeAnimKitId(ANIM_READY2H);
103 break;
104 }
105 }
106
107 InstanceScript::OnCreatureCreate(creature);
108 }
109
110 bool SetBossState(uint32 type, EncounterState state) override
111 {
112 switch (type)
113 {
114 case DATA_SLABHIDE:
115 // Open rock walls (Slabhide AI handles closing because it must be delayed)
116 if (state != IN_PROGRESS)
117 SetData(DATA_SLABHIDE_ROCK_WALL, true);
118
119 // Activate teleporters
120 if (state == DONE)
121 {
122 ActivateTeleporter(GetCreature(DATA_STONECORE_TELEPORTER));
123 ActivateTeleporter(GetCreature(DATA_STONECORE_TELEPORTER_2));
124 }
125 break;
126 default:
127 break;
128 }
129
130 return InstanceScript::SetBossState(type, state);
131 }
132
133 uint32 GetData(uint32 type) const override
134 {
135 switch (type)
136 {
137 case DATA_SLABHIDE_INTRO:
138 return slabhideIntro;
139 default:
140 break;
141 }
142
143 return 0;
144 }
145
146 void SetData(uint32 type, uint32 data) override
147 {
148 switch (type)
149 {
150 case DATA_HANDLE_CORBORUS_ROCKDOOR:
151 HandleGameObject(corborusRockDoorGUID, true);
152 break;
153 case DATA_MILLHOUSE_EVENT_FACE:
154 MillhouseEvent_Face();
155 break;
156 case DATA_MILLHOUSE_EVENT_KNOCKBACK:
157 MillhouseEvent_Knockback();
158 break;
159 case DATA_MILLHOUSE_EVENT_DESPAWN:
160 MillhouseEvent_Despawn();
161 break;
162 case DATA_SLABHIDE_INTRO:
163 slabhideIntro = EncounterState(data);
164 break;
165 case DATA_SLABHIDE_ROCK_WALL: // Handles rock walls
166 for (std::vector<ObjectGuid>::iterator itr = slabhideRockWallGUIDs.begin(); itr != slabhideRockWallGUIDs.end(); ++itr)
167 HandleGameObject((*itr), data ? true : false);
168 break;
169 default:
170 break;
171 }
172 }
173
174 private:
175 // Face Millhouse and other nearby mobs to Corborus
176 void MillhouseEvent_Face()
177 {
178 if (Creature* Millhouse = GetCreature(DATA_MILLHOUSE_MANASTORM))
179 Millhouse->SetFacingTo(1.570796f);
180 for (GuidVector::const_iterator i = millhouseLastGroupGUIDs.begin(); i != millhouseLastGroupGUIDs.end(); ++i)
181 if (Creature* creature = instance->GetCreature(*i))
182 creature->SetFacingTo(1.570796f);
183 }
184
185 // Knock back Millhouse and other mobs
186 void MillhouseEvent_Knockback()
187 {
188 if (Creature* Millhouse = GetCreature(DATA_MILLHOUSE_MANASTORM))
189 Millhouse->CastSpell(Millhouse, SPELL_RING_WYRM_KNOCKBACK, true);
190 for (GuidVector::const_iterator itr = millhouseLastGroupGUIDs.begin(); itr != millhouseLastGroupGUIDs.end(); ++itr)
191 if (Creature* creature = instance->GetCreature(*itr))
192 creature->CastSpell(creature, SPELL_RING_WYRM_KNOCKBACK, true);
193 }
194
195 // Despawn all mobs
196 void MillhouseEvent_Despawn()
197 {
198 if (Creature* Millhouse = GetCreature(DATA_MILLHOUSE_MANASTORM))
199 Millhouse->DespawnOrUnsummon(3000);
200 for (GuidVector::const_iterator itr = millhouseTrashGUIDs.begin(); itr != millhouseTrashGUIDs.end(); ++itr)
201 if (Creature* creature = instance->GetCreature(*itr))
202 creature->DespawnOrUnsummon(3000);
203 for (GuidVector::const_iterator itr = millhouseLastGroupGUIDs.begin(); itr != millhouseLastGroupGUIDs.end(); ++itr)
204 if (Creature* creature = instance->GetCreature(*itr))
205 creature->DespawnOrUnsummon(3000);
206 }
207
208 void ActivateTeleporter(Creature* teleporter)
209 {
210 if (!teleporter)
211 return;
212
213 teleporter->CastSpell(teleporter, SPELL_TELEPORTER_ACTIVE_VISUAL, true);
214 teleporter->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_SPELLCLICK);
215 }
216
217 GuidVector millhouseTrashGUIDs;
218 GuidVector millhouseLastGroupGUIDs;
219 ObjectGuid corborusRockDoorGUID;
220 GuidVector slabhideRockWallGUIDs;
221
222 EncounterState slabhideIntro = NOT_STARTED;
223 };
224
225 InstanceScript* GetInstanceScript(InstanceMap* map) const override
226 {
227 return new instance_stonecore_InstanceScript(map);
228 }
229};
230
231void AddSC_instance_stonecore()
232{
233 new instance_stonecore();
234}

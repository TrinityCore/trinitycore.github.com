97// 43391 - Millhouse Manastorm
98class npc_sc_millhouse_manastorm : public CreatureScript
99{
100 public:
101 npc_sc_millhouse_manastorm() : CreatureScript("npc_sc_millhouse_manastorm") { }
102
103 struct npc_sc_millhouse_manastormAI : public ScriptedAI
104 {
105 npc_sc_millhouse_manastormAI(Creature* creature) : ScriptedAI(creature),
106 _instance(creature->GetInstanceScript())
107 {
108 events.SetPhase(PHASE_MILLHOUSE_GROUP_1);
109 }
110
111 void ScheduleEvents()
112 {
113 events.ScheduleEvent(EVENT_SHADOWFURY, 3000);
114 events.ScheduleEvent(EVENT_FROSTBOLT_VOLLEY, 5000);
115 events.ScheduleEvent(EVENT_FEAR, 8000);
116 }
117
118 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
119 {
120 if (damage >= me->GetHealth())
121 damage = me->GetHealth() - 1;
122
123 if (!HealthBelowPct(50) || me->HasAura(SPELL_BLUR))
124 return;
125
126 switch (events.GetPhaseMask())
127 {
128 case PHASE_MASK_MILLHOUSE_GROUP_1:
129 events.Reset();
130 events.SetPhase(PHASE_MILLHOUSE_GROUP_2);
131 me->SetReactState(REACT_PASSIVE);
132 me->InterruptNonMeleeSpells(true);
133 DoCast(me, SPELL_CLEAR_ALL_DEBUFFS);
134 DoCast(me, SPELL_BLUR);
135 Talk(SAY_MILLHOUSE_EVENT_1);
136 me->GetMotionMaster()->MovePoint(POINT_MILLHOUSE_GROUP_2, MillhousePointGroup2);
137 break;
138 case PHASE_MASK_MILLHOUSE_GROUP_2:
139 events.Reset();
140 events.SetPhase(PHASE_MILLHOUSE_GROUP_3);
141 me->SetReactState(REACT_PASSIVE);
142 me->InterruptNonMeleeSpells(true);
143 DoCast(me, SPELL_CLEAR_ALL_DEBUFFS);
144 DoCast(me, SPELL_BLUR);
145 Talk(SAY_MILLHOUSE_EVENT_1);
146 me->GetMotionMaster()->MovePoint(POINT_MILLHOUSE_GROUP_3, MillhousePointGroup3);
147 break;
148 case PHASE_MASK_MILLHOUSE_GROUP_3:
149 events.Reset();
150 events.SetPhase(PHASE_MILLHOUSE_GROUP_4);
151 me->SetReactState(REACT_PASSIVE);
152 me->InterruptNonMeleeSpells(true);
153 DoCast(me, SPELL_CLEAR_ALL_DEBUFFS);
154 DoCast(me, SPELL_BLUR);
155 me->GetMotionMaster()->MovePoint(POINT_MILLHOUSE_GROUP_4, MillhousePointGroup4);
156 break;
157 default:
158 break;
159 }
160
161 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IN_COMBAT);
162 }
163
164 void MovementInform(uint32 type, uint32 pointId) override
165 {
166 if (type != POINT_MOTION_TYPE)
167 return;
168
169 if (pointId < POINT_MILLHOUSE_GROUP_2 || pointId > POINT_MILLHOUSE_GROUP_4)
170 return;
171
172 me->RemoveAllAuras();
173 me->CombatStop(true);
174 me->DeleteThreatList();
175
176 switch (pointId)
177 {
178 case POINT_MILLHOUSE_GROUP_2:
179 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IN_COMBAT);
180 me->SetReactState(REACT_AGGRESSIVE);
181 if (Creature* worldtrigger = me->FindNearestCreature(NPC_WORLDTRIGGER, 200.0f))
182 me->SetFacingToObject(worldtrigger);
183 DoCast(me, SPELL_ANCHOR_HERE);
184 DoCast(me, SPELL_TIGULE_AND_FORORS_SPECIAL_BLEND);
185 events.ScheduleEvent(EVENT_READY_FOR_COMBAT, 10000);
186 break;
187 case POINT_MILLHOUSE_GROUP_3:
188 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IN_COMBAT);
189 me->SetReactState(REACT_AGGRESSIVE);
190 me->SetFacingTo(5.931499f);
191 DoCast(me, SPELL_ANCHOR_HERE);
192 DoCast(me, SPELL_TIGULE_AND_FORORS_SPECIAL_BLEND);
193 events.ScheduleEvent(EVENT_READY_FOR_COMBAT, 10000);
194 break;
195 case POINT_MILLHOUSE_GROUP_4:
196 me->SetFacingTo(3.455752f);
197 DoCast(me, SPELL_ANCHOR_HERE);
198 Talk(SAY_MILLHOUSE_EVENT_2);
199 events.ScheduleEvent(EVENT_CAST_IMPENDING_DOOM, 1000);
200 break;
201 default:
202 break;
203 }
204 }
205
206 void UpdateAI(uint32 diff) override
207 {
208 // Do not update events if Millhouse is aggressive and has no combat.
209 if (!UpdateVictim() && me->GetReactState() == REACT_AGGRESSIVE)
210 return;
211
212 events.Update(diff);
213
214 // Impending Doom is exception because it needs to be interrupted.
215 if (me->HasUnitState(UNIT_STATE_CASTING) && !me->FindCurrentSpellBySpellId(SPELL_IMPENDING_DOOM))
216 return;
217
218 while (uint32 eventId = events.ExecuteEvent())
219 {
220 switch (eventId)
221 {
222 case EVENT_FROSTBOLT_VOLLEY:
223 DoCastAOE(SPELL_FROSTBOLT_VOLLEY);
224 events.ScheduleEvent(EVENT_FROSTBOLT_VOLLEY, 7000);
225 break;
226 case EVENT_SHADOWFURY:
227 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
228 DoCast(target, SPELL_SHADOWFURY);
229 events.ScheduleEvent(EVENT_SHADOWFURY, 7000);
230 break;
231 case EVENT_FEAR:
232 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
233 DoCast(target, SPELL_FEAR);
234 events.ScheduleEvent(EVENT_FEAR, 18000);
235 break;
236 case EVENT_READY_FOR_COMBAT:
237 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IN_COMBAT);
238 me->SetReactState(REACT_AGGRESSIVE);
239 ScheduleEvents();
240 break;
241 case EVENT_CAST_IMPENDING_DOOM:
242 DoCast(me, SPELL_IMPENDING_DOOM);
243 DoCast(me, SPELL_IMPENDING_DOOM_CHANNEL);
244 events.ScheduleEvent(EVENT_INTERRUPT_IMPENDING_DOOM, urand(15000,20000));
245 break;
246 case EVENT_INTERRUPT_IMPENDING_DOOM:
247 me->InterruptNonMeleeSpells(true);
248 me->RemoveAllAuras();
249 me->HandleEmoteCommand(EMOTE_ONESHOT_KNOCKDOWN);
250 events.ScheduleEvent(EVENT_CAST_IMPENDING_DOOM, 3000);
251 break;
252 default:
253 break;
254 }
255 }
256
257 DoSpellAttackIfReady(SPELL_SHADOW_BOLT);
258 }
259
260 private:
261 InstanceScript* _instance;
262 EventMap events;
263 };
264
265 CreatureAI* GetAI(Creature* creature) const override
266 {
267 return GetInstanceAI<npc_sc_millhouse_manastormAI>(creature);
268 }

362class JumpCheck
363{
364 public:
365 bool operator()(WorldObject* object) const
366 {
367 Player* player = object->ToPlayer();
368 return (player && (player->IsFalling() || player->HasUnitState(UNIT_STATE_JUMPING)));
369 }

74class boss_ozruk : public CreatureScript
75{
76 public:
77 boss_ozruk() : CreatureScript("boss_ozruk") { }
78
79 struct boss_ozrukAI : public BossAI
80 {
81 boss_ozrukAI(Creature* creature) : BossAI(creature, DATA_OZRUK) { }
82
83 void Reset() override
84 {
85 _Reset();
86
87 me->SetReactState(REACT_AGGRESSIVE);
88
89 events.ScheduleEvent(EVENT_ELEMENTIUM_BULWARK, 5000);
90 events.ScheduleEvent(EVENT_GROUND_SLAM, 10000);
91 events.ScheduleEvent(EVENT_ELEMENTIUM_SPIKE_SHIELD, 13000);
92 }
93
94 void EnterCombat(Unit* /*victim*/) override
95 {
96 _EnterCombat();
97
98 Talk(SAY_AGGRO);
99 }
100
101 void JustSummoned(Creature* summon) override
102 {
103 if (summon->GetEntry() == NPC_RUPTURE_CONTROLLER)
104 {
105 summon->CastSpell(summon, SPELL_RUPTURE, true);
106 summon->DespawnOrUnsummon(10000);
107 }
108
109 BossAI::JustSummoned(summon);
110 }
111
112 void DamageTaken(Unit* /*attacker*/, uint32 &damage) override
113 {
114 if (!me->HealthBelowPctDamaged(25, damage) || me->HasAura(SPELL_ENRAGE))
115 return;
116
117 DoCast(me, SPELL_ENRAGE);
118 Talk(SAY_ENRAGE);
119 }
120
121 void JustDied(Unit* /*killer*/) override
122 {
123 _JustDied();
124
125 Talk(SAY_DEATH);
126 }
127
128 void UpdateAI(uint32 diff) override
129 {
130 if (!UpdateVictim())
131 return;
132
133 events.Update(diff);
134
135 if (me->HasUnitState(UNIT_STATE_CASTING))
136 return;
137
138 while (uint32 eventId = events.ExecuteEvent())
139 {
140 switch (eventId)
141 {
142 case EVENT_ELEMENTIUM_BULWARK:
143 DoCast(me, SPELL_ELEMENTIUM_BULWARK);
144 Talk(SAY_ELEMENTIUM_BULWARK);
145 break;
146 case EVENT_GROUND_SLAM:
147 me->SetReactState(REACT_PASSIVE);
148 me->AttackStop();
149 DoCast(me, SPELL_GROUND_SLAM);
150 events.ScheduleEvent(EVENT_START_ATTACK, 4600);
151 break;
152 case EVENT_ELEMENTIUM_SPIKE_SHIELD:
153 DoCast(me, SPELL_ELEMENTIUM_SPIKE_SHIELD);
154 Talk(SAY_ELEMENTIUM_SPIKE_SHIELD);
155 events.ScheduleEvent(EVENT_SHATTER, 10000);
156 break;
157 case EVENT_SHATTER:
158 summons.DespawnEntry(NPC_BOUNCER_SPIKE);
159 me->SetReactState(REACT_PASSIVE);
160 me->AttackStop();
161 DoCast(me, SPELL_SHATTER);
162 events.ScheduleEvent(EVENT_START_ATTACK, 4600);
163 // Spells are cast in same order everytime after Shatter, so we schedule them here
164 events.ScheduleEvent(EVENT_ELEMENTIUM_BULWARK, urand(3000,4000));
165 events.ScheduleEvent(EVENT_GROUND_SLAM, urand(7000,9000));
166 events.ScheduleEvent(EVENT_ELEMENTIUM_SPIKE_SHIELD, urand(10000,12000));
167 break;
168 case EVENT_START_ATTACK:
169 me->SetReactState(REACT_AGGRESSIVE);
170 DoStartMovement(me->GetVictim());
171 break;
172 default:
173 break;
174 }
175 }
176
177 DoMeleeAttackIfReady();
178 }
179 };
180
181 CreatureAI* GetAI(Creature* creature) const override
182 {
183 return GetInstanceAI<boss_ozrukAI>(creature);
184 }
185};

187// 92393 - Rupture
188class spell_rupture : public SpellScriptLoader
189{
190public:
191 spell_rupture() : SpellScriptLoader("spell_rupture") { }
192
193 class spell_rupture_AuraScript : public AuraScript
194 {
195 PrepareAuraScript(spell_rupture_AuraScript);
196
197 void HandleEffectPeriodic(AuraEffect const* aurEff)
198 {
199 Unit* caster = GetCaster();
200
201 float dist = aurEff->GetTickNumber() * 8.0f;
202
203 // probably hack, should use spells (see Spells enum above)
204 Position pos = caster->GetNearPosition(dist, 0.0f);
205 SummonRupture(caster, pos);
206
207 pos = caster->GetNearPosition(dist, 0.2f);
208 SummonRupture(caster, pos);
209
210 pos = caster->GetNearPosition(dist, -0.2f);
211 SummonRupture(caster, pos);
212 }
213
214 void SummonRupture(Unit* caster, Position pos)
215 {
216 Creature* rupture = caster->SummonCreature(NPC_RUPTURE, pos, TEMPSUMMON_TIMED_DESPAWN, 2500);
217 if (!rupture)
218 return;
219
220 rupture->CastSpell(rupture, SPELL_RUPTURE_DAMAGE, true);
221 }
222
223 void Register() override
224 {
225 OnEffectPeriodic += AuraEffectPeriodicFn(spell_rupture_AuraScript::HandleEffectPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_TRIGGER_SPELL);
226 }
227 };
228
229 AuraScript* GetAuraScript() const override
230 {
231 return new spell_rupture_AuraScript();
232 }
233};

235// 78835 - Elementium Spike Shield
236class spell_elementium_spike_shield : public SpellScriptLoader
237{
238public:
239 spell_elementium_spike_shield() : SpellScriptLoader("spell_elementium_spike_shield") { }
240
241 class spell_elementium_spike_shield_SpellScript : public SpellScript
242 {
243 PrepareSpellScript(spell_elementium_spike_shield_SpellScript);
244
245 void HandleBouncerSpikes()
246 {
247 Unit* caster = GetCaster();
248 Vehicle* vehicle = caster->GetVehicleKit();
249 if (!vehicle)
250 return;
251
252 for (uint8 i = 0; i < vehicle->GetAvailableSeatCount(); i++)
253 if (Creature* summon = caster->SummonCreature(NPC_BOUNCER_SPIKE, caster->GetPosition(), TEMPSUMMON_TIMED_DESPAWN, 10000))
254 summon->EnterVehicle(caster, i);
255 }
256
257 void Register() override
258 {
259 OnCast += SpellCastFn(spell_elementium_spike_shield_SpellScript::HandleBouncerSpikes);
260 }
261 };
262
263 SpellScript* GetSpellScript() const override
264 {
265 return new spell_elementium_spike_shield_SpellScript();
266 }
267};

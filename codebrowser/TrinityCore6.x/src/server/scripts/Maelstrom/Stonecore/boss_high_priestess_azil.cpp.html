121class boss_high_priestess_azil : public CreatureScript
122{
123 public:
124 boss_high_priestess_azil() : CreatureScript("boss_high_priestess_azil") { }
125
126 struct boss_high_priestess_azilAI : public BossAI
127 {
128 boss_high_priestess_azilAI(Creature* creature) : BossAI(creature, DATA_HIGH_PRIESTESS_AZIL) { }
129 void Reset() override
130 {
131 _Reset();
132
133 me->SetCanFly(false);
134 me->SetDisableGravity(false);
135 me->SetReactState(REACT_PASSIVE);
136
137 events.ScheduleEvent(EVENT_INTRO_MOVE, 2000);
138 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, 6000);
139 events.ScheduleEvent(EVENT_FORCE_GRIP, urand(8000,10000));
140 events.ScheduleEvent(EVENT_SUMMON_GRAVITY_WELL, 16000);
141 events.ScheduleEvent(EVENT_ENERGY_SHIELD, urand(35000,36000));
142 events.ScheduleEvent(EVENT_SUMMON_WAVE_SOUTH, 0);
143 events.ScheduleEvent(EVENT_SUMMON_WAVE_WEST, 40000);
144 }
145
146 void EnterCombat(Unit* /*victim*/) override
147 {
148 _EnterCombat();
149
150 DoCast(me, SPELL_ENERGY_SHIELD);
151 Talk(SAY_AGGRO);
152 }
153
154 void JustDied(Unit* /*killer*/) override
155 {
156 _JustDied();
157
158 Talk(SAY_DEATH);
159 }
160
161 void MovementInform(uint32 type, uint32 id) override
162 {
163 if (type != POINT_MOTION_TYPE && id != POINT_INTRO_MOVE)
164 return;
165
166 switch (id)
167 {
168 case POINT_INTRO_MOVE:
169 me->RemoveAurasDueToSpell(SPELL_ENERGY_SHIELD);
170 me->SetReactState(REACT_AGGRESSIVE);
171 DoStartMovement(me->GetVictim());
172 break;
173 case POINT_FLY_UP:
174 me->SetCanFly(true);
175 me->SetDisableGravity(true);
176 events.ScheduleEvent(EVENT_EARTH_FURY_FLY_ABOVE_PLATFORM, 1000);
177 break;
178 case POINT_ABOVE_PLATFORM:
179 me->SetFacingTo(5.218534f);
180 DoCast(me, SPELL_EARTH_FURY_CASTING_VISUAL);
181 DoCast(me, SPELL_SEISMIC_SHARD_SUMMON_1);
182 DoCast(me, SPELL_SEISMIC_SHARD_SUMMON_2);
183 DoCast(me, SPELL_SEISMIC_SHARD_SUMMON_3);
184 events.ScheduleEvent(EVENT_EARTH_FURY_PREPARE_SHARD, 6700);
185 break;
186 case POINT_GROUND:
187 DoCast(me, SPELL_EJECT_ALL_PASSENGERS);
188 me->SetCanFly(false);
189 me->SetDisableGravity(false);
190 me->SetReactState(REACT_AGGRESSIVE);
191 DoStartMovement(me->GetVictim());
192 // Find more sniffs to correct these timers, this was copied from Reset() void.
193 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, 6000);
194 events.ScheduleEvent(EVENT_FORCE_GRIP, urand(8000, 10000));
195 events.ScheduleEvent(EVENT_SUMMON_GRAVITY_WELL, 16000);
196 break;
197 default:
198 break;
199 }
200 }
201
202 void UpdateAI(uint32 diff) override
203 {
204 if (!UpdateVictim())
205 return;
206
207 events.Update(diff);
208
209 if (me->HasUnitState(UNIT_STATE_CASTING))
210 return;
211
212 while (uint32 eventId = events.ExecuteEvent())
213 {
214 switch (eventId)
215 {
216 case EVENT_INTRO_MOVE:
217 me->GetMotionMaster()->MoveJump(GroundPos, me->GetSpeed(MOVE_FLIGHT), 1.918408f, POINT_INTRO_MOVE);
218 break;
219 case EVENT_CURSE_OF_BLOOD:
220 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
221 DoCast(target, SPELL_CURSE_OF_BLOOD);
222 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, urand(13000, 15000));
223 break;
224 case EVENT_FORCE_GRIP:
225 DoCastVictim(SPELL_FORCE_GRIP);
226 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, urand(13000, 15000));
227 break;
228 case EVENT_SUMMON_GRAVITY_WELL:
229 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
230 DoCast(target, SPELL_SUMMON_GRAVITY_WELL);
231 events.ScheduleEvent(EVENT_SUMMON_GRAVITY_WELL, urand(13000, 15000));
232 break;
233 case EVENT_ENERGY_SHIELD:
234 events.Reset();
235 DoCast(me, SPELL_EARTH_FURY_ENERGY_SHIELD);
236 events.ScheduleEvent(EVENT_EARTH_FURY, 0);
237 break;
238 case EVENT_EARTH_FURY:
239 countSeismicShard = 3;
240 me->SetReactState(REACT_PASSIVE);
241 me->SetFacingTo(5.862942f);
242 events.ScheduleEvent(EVENT_EARTH_FURY_FLY_UP, 1600);
243 break;
244 case EVENT_EARTH_FURY_FLY_UP:
245 Talk(SAY_PHASE_TWO);
246 me->GetMotionMaster()->MovePoint(POINT_FLY_UP, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ() + 5);
247 break;
248 case EVENT_EARTH_FURY_FLY_ABOVE_PLATFORM:
249 me->GetMotionMaster()->MovePoint(POINT_ABOVE_PLATFORM, AbovePlatformPos);
250 break;
251 case EVENT_EARTH_FURY_PREPARE_SHARD:
252 DoCast(me, SPELL_SEISMIC_SHARD_PREPARE);
253 events.ScheduleEvent(EVENT_EARTH_FURY_LAUNCH_SHARD, 1800);
254 break;
255 case EVENT_EARTH_FURY_LAUNCH_SHARD:
256 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
257 {
258 me->SetFacingToObject(target);
259 DoCast(target, SPELL_SEISMIC_SHARD_TARGETING);
260 DoCast(me, SPELL_SEISMIC_SHARD_LAUNCH);
261 countSeismicShard -= 1;
262 }
263 events.ScheduleEvent(countSeismicShard > 0 ? EVENT_EARTH_FURY_PREPARE_SHARD : EVENT_EARTH_FURY_FLY_DOWN, 4800);
264 break;
265 case EVENT_EARTH_FURY_FLY_DOWN:
266 {
267 me->RemoveAurasDueToSpell(SPELL_EARTH_FURY_CASTING_VISUAL);
268 me->RemoveAurasDueToSpell(SPELL_EARTH_FURY_ENERGY_SHIELD);
269 Position pos = me->GetPosition();
270 pos.m_positionZ = me->GetMap()->GetHeight(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ());
271 me->GetMotionMaster()->MovePoint(POINT_GROUND, pos);
272 break;
273 }
274 case EVENT_SUMMON_WAVE_SOUTH:
275 if (Creature* worldtrigger = me->FindNearestCreature(NPC_WORLDTRIGGER, 150.0f))
276 worldtrigger->CastSpell(worldtrigger, SPELL_SUMMON_WAVE_SOUTH);
277 events.ScheduleEvent(EVENT_SUMMON_WAVE_SOUTH, 12000);
278 break;
279 case EVENT_SUMMON_WAVE_WEST:
280 if (Creature* worldtrigger = me->FindNearestCreature(NPC_WORLDTRIGGER, 150.0f))
281 worldtrigger->CastSpell(worldtrigger, SPELL_SUMMON_WAVE_WEST);
282 events.ScheduleEvent(EVENT_SUMMON_WAVE_WEST, 20000);
283 break;
284 default:
285 break;
286 }
287 }
288
289 DoMeleeAttackIfReady();
290 }
291
292 private:
293 uint8 countSeismicShard = 3;
294 };
295
296 CreatureAI* GetAI(Creature* creature) const override
297 {
298 return GetInstanceAI<boss_high_priestess_azilAI>(creature);
299 }

400 struct npc_seismic_shardAI : public ScriptedAI

402 npc_seismic_shardAI(Creature* creature) : ScriptedAI(creature)
403 {
404 instance = creature->GetInstanceScript();
405 me->SetDisableGravity(true);
406 DoCast(me, SPELL_SEISMIC_SHARD_VISUAL);
407
408 Movement::MoveSplineInit init(me);
409 FillPath(me->GetPosition(), init.Path());
410 init.SetFly();
411 init.Launch();
412
413 events.ScheduleEvent(EVENT_SEISMIC_SHARD_MOUNT, 2400);
414 }
415
416 void UpdateAI(uint32 diff) override
417 {
418 events.Update(diff);
419
420 while (uint32 eventId = events.ExecuteEvent())
421 {
422 switch (eventId)
423 {
424 case EVENT_SEISMIC_SHARD_MOUNT:
425 if (Creature* highPriestAzil = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_HIGH_PRIESTESS_AZIL)))
426 if (Vehicle* vehicle = highPriestAzil->GetVehicleKit())
427 me->EnterVehicle(highPriestAzil, vehicle->GetNextEmptySeat(0, false)->first);
428 break;
429 default:
430 break;
431 }
432 }
433
434 }
435
436 private:
437 void FillPath(Position const& pos, Movement::PointsArray& path)
438 {
439 G3D::Vector3 point;
440
441 point.x = pos.GetPositionX();
442 point.y = pos.GetPositionY();
443 point.z = pos.GetPositionZ();
444
445 point.x -= 1.0f;
446 path.push_back(point);
447
448 point.x += 1.0f;
449 path.push_back(point);
450
451 point.z += 25.0f;
452 path.push_back(point);
453
454 path.push_back(point);
455 }
456
457 InstanceScript* instance;
458 EventMap events;
459 };
460

462 {
463 return GetInstanceAI<npc_seismic_shardAI>(creature);

86int main(int argc, char** argv)
87{
88 signal(SIGABRT, &Trinity::AbortHandler);
89
90 std::string configFile = _TRINITY_BNET_CONFIG;
91 std::string configService;
92 auto vm = GetConsoleArguments(argc, argv, configFile, configService);
93 // exit if help or version is enabled
94 if (vm.count("help") || vm.count("version"))
95 return 0;
96
97#if PLATFORM == PLATFORM_WINDOWS
98 if (configService.compare("install") == 0)
99 return WinServiceInstall() ? 0 : 1;
100 else if (configService.compare("uninstall") == 0)
101 return WinServiceUninstall() ? 0 : 1;
102 else if (configService.compare("run") == 0)
103 return WinServiceRun() ? 0 : 1;
104#endif
105
106 std::string configError;
107 if (!sConfigMgr->LoadInitial(configFile, configError))
108 {
109 printf("Error in config file: %s\n", configError.c_str());
110 return 1;
111 }
112
113 sLog->RegisterAppender<AppenderDB>();
114 sLog->Initialize(nullptr);
115
116 TC_LOG_INFO("server.bnetserver", "%s (bnetserver)", GitRevision::GetFullVersion());
117 TC_LOG_INFO("server.bnetserver", "<Ctrl-C> to stop.\n");
118 TC_LOG_INFO("server.bnetserver", "Using configuration file %s.", configFile.c_str());
119 TC_LOG_INFO("server.bnetserver", "Using SSL version: %s (library: %s)", OPENSSL_VERSION_TEXT, SSLeay_version(SSLEAY_VERSION));
120 TC_LOG_INFO("server.bnetserver", "Using Boost version: %i.%i.%i", BOOST_VERSION / 100000, BOOST_VERSION / 100 % 1000, BOOST_VERSION % 100);
121
122 // Seed the OpenSSL's PRNG here.
123 // That way it won't auto-seed when calling BigNumber::SetRand and slow down the first world login
124 BigNumber seed;
125 seed.SetRand(16 * 8);
126
127 // bnetserver PID file creation
128 std::string pidFile = sConfigMgr->GetStringDefault("PidFile", "");
129 if (!pidFile.empty())
130 {
131 if (uint32 pid = CreatePIDFile(pidFile))
132 TC_LOG_INFO("server.bnetserver", "Daemon PID: %u\n", pid);
133 else
134 {
135 TC_LOG_ERROR("server.bnetserver", "Cannot create PID file %s.\n", pidFile.c_str());
136 return 1;
137 }
138 }
139
140 int32 worldListenPort = sConfigMgr->GetIntDefault("WorldserverListenPort", 1118);
141 if (worldListenPort < 0 || worldListenPort > 0xFFFF)
142 {
143 TC_LOG_ERROR("server.bnetserver", "Specified worldserver listen port (%d) out of allowed range (1-65535)", worldListenPort);
144 return 1;
145 }
146
147 // Initialize the database connection
148 if (!StartDB())
149 return 1;
150
151 sIpcContext->Initialize();
152
153 _ioService = new boost::asio::io_service();
154
155 // Get the list of realms for the server
156 sRealmList->Initialize(*_ioService, sConfigMgr->GetIntDefault("RealmsStateUpdateDelay", 10), worldListenPort);
157
158 // Start the listening port (acceptor) for auth connections
159 int32 bnport = sConfigMgr->GetIntDefault("BattlenetPort", 1119);
160 if (bnport < 0 || bnport > 0xFFFF)
161 {
162 TC_LOG_ERROR("server.bnetserver", "Specified battle.net port (%d) out of allowed range (1-65535)", bnport);
163 StopDB();
164 delete _ioService;
165 return 1;
166 }
167
168 std::string bindIp = sConfigMgr->GetStringDefault("BindIP", "0.0.0.0");
169
170 sSessionMgr.StartNetwork(*_ioService, bindIp, bnport);
171
172 // Set signal handlers
173 boost::asio::signal_set signals(*_ioService, SIGINT, SIGTERM);
174#if PLATFORM == PLATFORM_WINDOWS
175 signals.add(SIGBREAK);
176#endif
177 signals.async_wait(SignalHandler);
178
179 // Set process priority according to configuration settings
180 SetProcessPriority("server.bnetserver");
181
182 // Enabled a timed callback for handling the database keep alive ping
183 _dbPingInterval = sConfigMgr->GetIntDefault("MaxPingTime", 30);
184 _dbPingTimer = new boost::asio::deadline_timer(*_ioService);
185 _dbPingTimer->expires_from_now(boost::posix_time::minutes(_dbPingInterval));
186 _dbPingTimer->async_wait(KeepDatabaseAliveHandler);
187
188 _banExpiryCheckInterval = sConfigMgr->GetIntDefault("BanExpiryCheckInterval", 60);
189 _banExpiryCheckTimer = new boost::asio::deadline_timer(*_ioService);
190 _banExpiryCheckTimer->expires_from_now(boost::posix_time::seconds(_banExpiryCheckInterval));
191 _banExpiryCheckTimer->async_wait(BanExpiryHandler);
192
193 sComponentMgr->Load();
194 sModuleMgr->Load();
195
196#if PLATFORM == PLATFORM_WINDOWS
197 if (m_ServiceStatus != -1)
198 {
199 _serviceStatusWatchTimer = new boost::asio::deadline_timer(*_ioService);
200 _serviceStatusWatchTimer->expires_from_now(boost::posix_time::seconds(1));
201 _serviceStatusWatchTimer->async_wait(ServiceStatusWatcher);
202 }
203#endif
204
205 // Start the io service worker loop
206 _ioService->run();
207
208 _banExpiryCheckTimer->cancel();
209 _dbPingTimer->cancel();
210
211 sSessionMgr.StopNetwork();
212
213 sIpcContext->Close();
214
215 sRealmList->Close();
216
217 // Close the Database Pool and library
218 StopDB();
219
220 TC_LOG_INFO("server.bnetserver", "Halting process...");
221
222 signals.cancel();
223
224 delete _banExpiryCheckTimer;
225 delete _dbPingTimer;
226 delete _ioService;
227 return 0;
228}

230/// Initialize connection to the database
231bool StartDB()
232{
233 MySQL::Library_Init();
234
235 // Load databases
236 DatabaseLoader loader("server.bnetserver", DatabaseLoader::DATABASE_NONE);
237 loader
238 .AddDatabase(LoginDatabase, "Login");
239
240 if (!loader.Load())
241 return false;
242
243 TC_LOG_INFO("server.bnetserver", "Started auth database connection pool.");
244 sLog->SetRealmId(0); // Enables DB appenders when realm is set.
245 return true;
246}
247
248/// Close the connection to the database
249void StopDB()
250{
251 LoginDatabase.Close();
252 MySQL::Library_End();
253}
254
255void SignalHandler(boost::system::error_code const& error, int /*signalNumber*/)
256{
257 if (!error)
258 _ioService->stop();
259}
260
261void KeepDatabaseAliveHandler(boost::system::error_code const& error)
262{
263 if (!error)
264 {
265 TC_LOG_INFO("server.bnetserver", "Ping MySQL to keep connection alive");
266 LoginDatabase.KeepAlive();
267
268 _dbPingTimer->expires_from_now(boost::posix_time::minutes(_dbPingInterval));
269 _dbPingTimer->async_wait(KeepDatabaseAliveHandler);
270 }
271}
272
273void BanExpiryHandler(boost::system::error_code const& error)
274{
275 if (!error)
276 {
277 LoginDatabase.Execute(LoginDatabase.GetPreparedStatement(LOGIN_DEL_EXPIRED_IP_BANS));
278 LoginDatabase.Execute(LoginDatabase.GetPreparedStatement(LOGIN_UPD_EXPIRED_ACCOUNT_BANS));
279
280 _banExpiryCheckTimer->expires_from_now(boost::posix_time::seconds(_banExpiryCheckInterval));
281 _banExpiryCheckTimer->async_wait(BanExpiryHandler);
282 }
283}

286void ServiceStatusWatcher(boost::system::error_code const& error)
287{
288 if (!error)
289 {
290 if (m_ServiceStatus == 0)
291 {
292 _ioService->stop();
293 delete _serviceStatusWatchTimer;
294 }
295 else
296 {
297 _serviceStatusWatchTimer->expires_from_now(boost::posix_time::seconds(1));
298 _serviceStatusWatchTimer->async_wait(ServiceStatusWatcher);
299 }
300 }
301}
302#endif

304variables_map GetConsoleArguments(int argc, char** argv, std::string& configFile, std::string& configService)
305{
306 (void)configService;
307
308 options_description all("Allowed options");
309 all.add_options()
310 ("help,h", "print usage message")
311 ("version,v", "print version build info")
312 ("config,c", value<std::string>(&configFile)->default_value(_TRINITY_BNET_CONFIG), "use <arg> as configuration file")
313 ;
314#if PLATFORM == PLATFORM_WINDOWS
315 options_description win("Windows platform specific options");
316 win.add_options()
317 ("service,s", value<std::string>(&configService)->default_value(""), "Windows service options: [install | uninstall]")
318 ;
319
320 all.add(win);
321#endif
322 variables_map variablesMap;
323 try
324 {
325 store(command_line_parser(argc, argv).options(all).allow_unregistered().run(), variablesMap);
326 notify(variablesMap);
327 }
328 catch (std::exception& e)
329 {
330 std::cerr << e.what() << "\n";
331 }
332
333 if (variablesMap.count("help"))
334 {
335 std::cout << all << "\n";
336 }
337 else if (variablesMap.count("version"))
338 {
339 std::cout << GitRevision::GetFullVersion() << "\n";
340 }
341
342 return variablesMap;
343}

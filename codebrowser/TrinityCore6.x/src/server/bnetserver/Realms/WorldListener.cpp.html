35WorldListener::WorldListener(uint16 worldListenPort) : _worldListenPort(worldListenPort)
36{
37 _worldSocket = sIpcContext->CreateNewSocket(zmqpp::socket_type::pull);
38}
39
40WorldListener::~WorldListener()
41{
42 delete _worldSocket;
43}
44
45void WorldListener::Run()
46{
47 while (!ProcessExit())
48 {
49 _poller->poll();
50 if (_poller->events(*_worldSocket) & zmqpp::poller::poll_in)
51 {
52 int32 op1;
53 do
54 {
55 zmqpp::message msg;
56 _worldSocket->receive(msg);
57 Dispatch(msg);
58 _worldSocket->get(zmqpp::socket_option::events, op1);
59 } while (op1 & zmqpp::poller::poll_in);
60 }
61 }
62}
63
64void WorldListener::HandleOpen()
65{
66 try
67 {
68 _worldSocket->bind(std::string("tcp://*:") + std::to_string(_worldListenPort));
69 }
70 catch (zmqpp::zmq_internal_exception& ex)
71 {
72 TC_LOG_FATAL("server.ipc", "Could not bind to WorldserverListenPort %u. Exception: %s. Shutting down bnetserver.", _worldListenPort, ex.what());
73 abort();
74 }
75
76 _poller->add(*_worldSocket);
77 TC_LOG_INFO("server.ipc", "Listening on connections from worldservers on port %u...", _worldListenPort);
78}
79
80void WorldListener::HandleClose()
81{
82 _worldSocket->close();
83 TC_LOG_INFO("server.ipc", "Shutting down connections from worldservers...");
84}

86void WorldListener::Dispatch(zmqpp::message& msg) const
87{
88 IPC::BattlenetComm::Header ipcHeader;
89 msg >> ipcHeader;
90
91 if (ipcHeader.Ipc.Channel != IPC_CHANNEL_BNET)
92 return;
93
94 if (ipcHeader.Ipc.Command < IPC_BNET_MAX_COMMAND)
95 (this->*_handlers[ipcHeader.Ipc.Command].Handler)(ipcHeader.Realm, msg);
96}

98void WorldListener::HandleToonOnlineStatusChange(Battlenet::RealmHandle const& realm, zmqpp::message& msg) const
99{
100 IPC::BattlenetComm::ToonHandle toonHandle;
101 bool online;
102 msg >> toonHandle;
103 msg >> online;
104
105 if (Battlenet::Session* session = sSessionMgr.GetSession(toonHandle.AccountId, toonHandle.GameAccountId))
106 {
107 if (online)
108 {
109 if (!session->IsToonOnline())
110 {
111 Battlenet::WoWRealm::ToonReady* toonReady = new Battlenet::WoWRealm::ToonReady();
112 toonReady->Name.Region = realm.Region;
113 toonReady->Name.ProgramId = "WoW";
114 toonReady->Name.Realm = realm.GetAddress();
115 toonReady->Name.Name = toonHandle.Name;
116
117 toonReady->Handle.Region = realm.Region;
118 toonReady->Handle.ProgramId = "WoW";
119 toonReady->Handle.Realm = realm.GetAddress();
120 toonReady->Handle.Id = toonHandle.Guid;
121
122 session->SetToonOnline(true);
123 session->AsyncWrite(toonReady);
124 }
125 }
126 else if (session->IsToonOnline())
127 {
128 session->AsyncWrite(new Battlenet::WoWRealm::ToonLoggedOut());
129 session->SetToonOnline(false);
130 }
131 }

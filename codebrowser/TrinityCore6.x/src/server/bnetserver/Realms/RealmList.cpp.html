38// Load the realm list from the database
39void RealmList::Initialize(boost::asio::io_service& ioService, uint32 updateInterval, uint16 worldListenPort)
40{
41 _updateInterval = updateInterval;
42 _updateTimer = new boost::asio::deadline_timer(ioService);
43 _resolver = new boost::asio::ip::tcp::resolver(ioService);
44
45 // Get the content of the realmlist table in the database
46 UpdateRealms(boost::system::error_code());
47
48 _worldListener = new WorldListener(worldListenPort);
49 _worldListener->Start();
50}
51
52void RealmList::Close()
53{
54 _worldListener->End();
55 _updateTimer->cancel();
56}

58template<typename FieldType>
59inline void UpdateField(FieldType& out, FieldType const& in, bool& changed)
60{
61 if (out != in)
62 {
63 out = in;
64 changed = true;
65 }
66}

68void RealmList::UpdateRealm(Battlenet::RealmHandle const& id, uint32 build, const std::string& name, ip::address const& address, ip::address const& localAddr,
69 ip::address const& localSubmask, uint16 port, uint8 icon, RealmFlags flag, uint8 timezone, AccountTypes allowedSecurityLevel,
70 float population)
71{
72 // Create new if not exist or update existed
73 Realm& realm = _realms[id];
74
75 realm.Keep = true;
76
77 realm.Id = id;
78 UpdateField(realm.Build, build, realm.Updated);
79 UpdateField(realm.Name, name, realm.Updated);
80 UpdateField(realm.Type, icon, realm.Updated);
81 UpdateField(realm.Flags, flag, realm.Updated);
82 UpdateField(realm.Timezone, timezone, realm.Updated);
83 UpdateField(realm.AllowedSecurityLevel, allowedSecurityLevel, realm.Updated);
84 UpdateField(realm.PopulationLevel, population, realm.Updated);
85 UpdateField(realm.ExternalAddress, address, realm.Updated);
86 UpdateField(realm.LocalAddress, localAddr, realm.Updated);
87 UpdateField(realm.LocalSubnetMask, localSubmask, realm.Updated);
88 UpdateField(realm.Port, port, realm.Updated);
89}

91void RealmList::UpdateRealms(boost::system::error_code const& error)
92{
93 if (error)
94 return;
95
96 TC_LOG_DEBUG("realmlist", "Updating Realm List...");
97
98 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_REALMLIST);
99 PreparedQueryResult result = LoginDatabase.Query(stmt);
100
101 // Circle through results and add them to the realm map
102 if (result)
103 {
104 do
105 {
106 try
107 {
108 boost::asio::ip::tcp::resolver::iterator end;
109
110 Field* fields = result->Fetch();
111 std::string name = fields[1].GetString();
112 boost::asio::ip::tcp::resolver::query externalAddressQuery(ip::tcp::v4(), fields[2].GetString(), "");
113
114 boost::system::error_code ec;
115 boost::asio::ip::tcp::resolver::iterator endPoint = _resolver->resolve(externalAddressQuery, ec);
116 if (endPoint == end || ec)
117 {
118 TC_LOG_ERROR("realmlist", "Could not resolve address %s", fields[2].GetString().c_str());
119 continue;
120 }
121
122 ip::address externalAddress = (*endPoint).endpoint().address();
123
124 boost::asio::ip::tcp::resolver::query localAddressQuery(ip::tcp::v4(), fields[3].GetString(), "");
125 endPoint = _resolver->resolve(localAddressQuery, ec);
126 if (endPoint == end || ec)
127 {
128 TC_LOG_ERROR("realmlist", "Could not resolve address %s", fields[3].GetString().c_str());
129 continue;
130 }
131
132 ip::address localAddress = (*endPoint).endpoint().address();
133
134 boost::asio::ip::tcp::resolver::query localSubmaskQuery(ip::tcp::v4(), fields[4].GetString(), "");
135 endPoint = _resolver->resolve(localSubmaskQuery, ec);
136 if (endPoint == end || ec)
137 {
138 TC_LOG_ERROR("realmlist", "Could not resolve address %s", fields[4].GetString().c_str());
139 continue;
140 }
141
142 ip::address localSubmask = (*endPoint).endpoint().address();
143
144 uint16 port = fields[5].GetUInt16();
145 uint8 icon = fields[6].GetUInt8();
146 if (icon == REALM_TYPE_FFA_PVP)
147 icon = REALM_TYPE_PVP;
148 if (icon >= MAX_CLIENT_REALM_TYPE)
149 icon = REALM_TYPE_NORMAL;
150 RealmFlags flag = RealmFlags(fields[7].GetUInt8());
151 uint8 timezone = fields[8].GetUInt8();
152 uint8 allowedSecurityLevel = fields[9].GetUInt8();
153 float pop = fields[10].GetFloat();
154 uint32 realmId = fields[0].GetUInt32();
155 uint32 build = fields[11].GetUInt32();
156 uint8 region = fields[12].GetUInt8();
157 uint8 battlegroup = fields[13].GetUInt8();
158
159 Battlenet::RealmHandle id{ region, battlegroup, realmId };
160
161 UpdateRealm(id, build, name, externalAddress, localAddress, localSubmask, port, icon, flag,
162 timezone, (allowedSecurityLevel <= SEC_ADMINISTRATOR ? AccountTypes(allowedSecurityLevel) : SEC_ADMINISTRATOR), pop);
163
164 TC_LOG_TRACE("realmlist", "Realm \"%s\" at %s:%u.", name.c_str(), externalAddress.to_string().c_str(), port);
165 }
166 catch (std::exception& ex)
167 {
168 TC_LOG_ERROR("realmlist", "Realmlist::UpdateRealms has thrown an exception: %s", ex.what());
169 ABORT();
170 }
171 }
172 while (result->NextRow());
173 }
174
175 std::vector<Realm const*> updatedRealms;
176 std::vector<Battlenet::RealmHandle> deletedRealms;
177
178 for (RealmMap::value_type& pair : _realms)
179 {
180 if (pair.second.Updated)
181 updatedRealms.push_back(&pair.second);
182 else if (!pair.second.Keep)
183 deletedRealms.push_back(pair.first);
184
185 pair.second.Updated = false;
186 pair.second.Keep = false;
187 }
188
189 for (Battlenet::RealmHandle const& deleted : deletedRealms)
190 _realms.erase(deleted);
191
192 if (!updatedRealms.empty() || !deletedRealms.empty())
193 {
194 sSessionMgr.LockedForEach([&updatedRealms, &deletedRealms](Battlenet::Session* session)
195 {
196 if (session->IsSubscribedToRealmListUpdates())
197 session->UpdateRealms(updatedRealms, deletedRealms);
198 });
199 }
200
201 if (_updateInterval)
202 {
203 _updateTimer->expires_from_now(boost::posix_time::seconds(_updateInterval));
204 _updateTimer->async_wait(std::bind(&RealmList::UpdateRealms, this, std::placeholders::_1));
205 }
206}

208Realm const* RealmList::GetRealm(Battlenet::RealmHandle const& id) const
209{
210 auto itr = _realms.find(id);
211 if (itr != _realms.end())
212 return &itr->second;
213
214 return NULL;
215}

20template<>
21bool Battlenet::BitStream::Read<bool>(uint32 /*bitCount*/)
22{
23 return Read<uint8>(1) != 0;
24}
25
26template<>
27void Battlenet::BitStream::Write<bool>(bool value, uint32 /*bitCount*/)
28{
29 Write<uint8>(value ? 1 : 0, 1);
30}

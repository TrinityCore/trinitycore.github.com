37 class BitStreamPositionException : public std::exception
38 {
39 static uint32 const MessageSize = 128;
40
41 public:
42 BitStreamPositionException(bool read, uint32 operationSize, uint32 position, uint32 streamSize)
43 {
44 memset(_message, 0, MessageSize);
45 snprintf(_message, MessageSize, "Attempted to %s more bits (%u) %s stream than %s (%u)\n",
46 (read ? "read" : "write"),
47 operationSize + position,
48 (read ? "from" : "to"),
49 (read ? "exist" : "allowed"),
50 streamSize);
51 }
52
53 char const* what() const throw()
54 {
55 return _message;
56 }
57
58 private:
59 char _message[MessageSize];
60 };

62 class BitStream
63 {
64 public:
65 static uint32 const MaxSize = 0x4000;
66
67 // length : The maximum number of bytes to read
68 BitStream(uint32 length) : _writePos(length * 8), _readPos(0)
69 {
70 _buffer.resize(length, 0);
71 }
72
73 BitStream(MessageBuffer&& buffer) : _writePos(uint32(buffer.GetActiveSize()) << 3), _readPos(0), _buffer(buffer.Move())
74 {
75 }
76
77 BitStream() : _writePos(0), _readPos(0)
78 {
79 _buffer.reserve(0x1000);
80 }
81
82 void AlignToNextByte()
83 {
84 _readPos = (_readPos + 7) & ~7;
85 _writePos = (_writePos + 7) & ~7;
86 }
87
88 std::string ReadString(uint32 bitCount, int32 baseLength = 0)
89 {
90 uint32 len = Read<uint32>(bitCount) + baseLength;
91 AlignToNextByte();
92 std::string str(reinterpret_cast<char*>(&_buffer[_readPos >> 3]), len);
93 _readPos += len * 8;
94 return str;
95 }
96
97 std::unique_ptr<uint8[]> ReadBytes(uint32 count)
98 {
99 AlignToNextByte();
100 if (_readPos + count * 8 > _writePos)
101 throw BitStreamPositionException(true, count * 8, _readPos, _writePos);
102
103 std::unique_ptr<uint8[]> buf(new uint8[count]);
104 memcpy(buf.get(), &_buffer[_readPos >> 3], count);
105 _readPos += count * 8;
106 return buf;
107 }

121 std::string ReadFourCC()
122 {
123 uint32 fcc = Read<uint32>(32);
124 EndianConvertReverse(fcc);
125 size_t len = 4;
126 while (!(fcc & 0xFF) && len)
127 {
128 fcc >>= 8;
129 --len;
130 }
131
132 return std::string(reinterpret_cast<char*>(&fcc), len);
133 }

135 template<typename T>
136 T Read(uint32 bitCount)
137 {
138 static_assert(std::is_integral<T>::value || std::is_enum<T>::value, "T must be an integer type");
139
140 if (_readPos + bitCount > _writePos)
141 throw BitStreamPositionException(true, bitCount, _readPos, _writePos);
142
143 uint64 ret = 0;
144 while (bitCount != 0)
145 {
146 uint32 bitPos = (_readPos & 7);
147 uint32 bitsLeftInByte = 8 - bitPos;
148 if (bitsLeftInByte >= bitCount)
149 bitsLeftInByte = bitCount;
150
151 bitCount -= bitsLeftInByte;
152 ret |= (uint64)(_buffer[_readPos >> 3] >> bitPos & (uint32)((uint8)(1 << bitsLeftInByte) - 1)) << bitCount;
153 _readPos += bitsLeftInByte;
154 }
155
156 return static_cast<T>(ret);
157 }

159 void WriteString(std::string const& str, uint32 bitCount, int32 baseLength = 0)
160 {
161 Write(str.length() + baseLength, bitCount);
162 WriteBytes(str.c_str(), uint32(str.length()));
163 }

165 template<typename T>
166 void WriteBytes(T* data, uint32 count)
167 {
168 AlignToNextByte();
169 if (!count || !data)
170 return;
171
172 if ((_writePos >> 3) + count > MaxSize)
173 throw BitStreamPositionException(false, count * 8, _writePos, MaxSize * 8);
174
175 _buffer.resize(_buffer.size() + count);
176 memcpy(&_buffer[_writePos >> 3], data, count);
177 _writePos += count * 8;
178 }

180 void WriteFloat(float value)
181 {
182 union
183 {
184 float AsFloat;
185 uint32 AsInt;
186 } convert;

188 convert.AsFloat = value;
189 Write(convert.AsInt, 32);

192 void WriteFourCC(std::string const& fcc)
193 {
194 uint32 intVal = *(uint32*)fcc.c_str();
195 size_t len = fcc.length();
196 EndianConvertReverse(intVal);
197 // Add padding
198 while (len++ < 4)
199 intVal >>= 8;
200
201 Write(intVal, 32);
202 }

204 template<typename T>
205 void Write(T value, uint32 bitCount)
206 {
207 static_assert(std::is_integral<T>::value || std::is_enum<T>::value, "T must be an integer type");
208
209 if (_writePos + bitCount > 8 * MaxSize)
210 throw BitStreamPositionException(false, bitCount, _writePos, MaxSize * 8);
211
212 while (bitCount != 0)
213 {
214 uint32 bitPos = (_writePos & 7);
215 uint32 bitsLeftInByte = 8 - bitPos;
216 if (bitsLeftInByte >= bitCount)
217 bitsLeftInByte = bitCount;
218
219 bitCount -= bitsLeftInByte;
220
221 uint8 firstHalf = (uint8)(~(((uint8)(1 << bitsLeftInByte) - 1) << bitPos));
222 uint8 secondHalf = (uint8)((((uint8)(1 << bitsLeftInByte) - 1) & (uint8)(value >> bitCount)) << bitPos);
223
224 if (_buffer.size() > (_writePos >> 3))
225 _buffer[_writePos >> 3] = (uint8)((_buffer[_writePos >> 3] & firstHalf) | secondHalf);
226 else
227 _buffer.push_back(secondHalf);
228
229 _writePos += bitsLeftInByte;
230 }
231 }
232
233 bool IsRead() const { return _readPos >= _writePos; }
234
235 uint8* GetBuffer() { return _buffer.data(); }
236 uint8 const* GetBuffer() const { return _buffer.data(); }
237
238 uint32 GetReadPos() const { return _readPos; }
239 size_t GetSize() const { return ((_writePos + 7) & ~7) / 8; }
240
241 // These methods are meant to only be used when their corresponding actions in the client ignore the value completely
242 void ReadSkip(uint32 bitCount) { _readPos += bitCount; }
243 void WriteSkip(uint32 bitCount) { Write(0, bitCount); }
244
245 private:
246 uint32 _writePos;
247 uint32 _readPos;
248 std::vector<uint8> _buffer;
249 };

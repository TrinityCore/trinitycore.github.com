34 namespace Format
35 {
36 template<typename T>
37 struct is_printable : std::is_base_of<PrintableComponent, typename std::remove_pointer<T>::type>::type
38 {
39 };
40
41 template<typename T>
42 typename std::enable_if<!std::is_pointer<T>::value, T>::type const& Dereference(T const& t) { return t; };
43
44 template<typename T>
45 typename std::enable_if<std::is_pointer<T>::value, typename std::remove_pointer<T>::type>::type const& Dereference(T const& t) { return *t; };
46
47 template<typename T>
48 std::ostream& FieldToString(std::ostream& stream, char const* fieldName, T const& u, std::false_type)
49 {
50 if (fieldName[0])
51 stream << fieldName << ": ";
52 return stream << u << std::endl;
53 }
54
55 std::ostream& FieldToString(std::ostream& stream, char const* fieldName, PrintableComponent const& u, std::true_type);
56 std::ostream& FieldToString(std::ostream& stream, char const* fieldName, uint8 const& u, std::false_type);
57
58 template<typename T>
59 std::ostream& FieldToString(std::ostream& stream, char const* fieldName, std::vector<T> const& u, std::false_type)
60 {
61 stream << fieldName << ":" << std::endl;
62 for (T const& t : u)
63 FieldToString(stream, "", Dereference(t), is_printable<T>());
64 return stream;
65 }
66
67 template<typename T>
68 std::ostream& FieldToString(std::ostream& stream, char const* fieldName, Optional<T> const& u)
69 {
70 if (u)
71 FieldToString(stream, fieldName, Dereference(*u), is_printable<T>());
72 return stream;
73 }
74
75 template<typename T>
76 std::ostream& FieldToString(std::ostream& stream, char const* fieldName, T const& u)
77 {
78 return FieldToString(stream, fieldName, Dereference(u), is_printable<T>());
79 }
80 }

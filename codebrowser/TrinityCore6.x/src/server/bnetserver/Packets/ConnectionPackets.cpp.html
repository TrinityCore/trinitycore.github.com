66void Battlenet::Connection::ConnectionClosing::Read()
67{
68 Packets.resize(_stream.Read<uint8>(6));
69 for (size_t i = 0; i < Packets.size(); ++i)
70 {
71 PacketInfo& info = Packets[i];
72 info.Command = _stream.ReadFourCC();
73 info.Time = _stream.Read<uint32>(32);
74 info.Size = _stream.Read<uint32>(16);
75 info.Layer = _stream.ReadFourCC();
76 info.Offset = _stream.Read<uint32>(16);
77 }
78
79 Reason = _stream.Read<ClosingReason>(4);
80 _stream.ReadBytes(_stream.Read<uint8>(8)); // BadData
81
82 if (_stream.Read<bool>(1)) // HasHeader
83 {
84 Header.Command = _stream.Read<uint32>(6);
85 if (_stream.Read<bool>(1))
86 Header.Channel = _stream.Read<int32>(4);
87 }
88
89 Now = _stream.Read<time_t>(32);
90}

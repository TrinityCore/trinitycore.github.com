20void Battlenet::Friends::SendInvitationRequest::Read()
21{
22 Token = _stream.Read<uint32>(32);
23
24 if (_stream.Read<bool>(1))
25 PresenceId = _stream.Read<uint32>(32);
26
27 if (_stream.Read<bool>(1))
28 AccountMail = _stream.ReadString(9, 3);
29
30 if (_stream.Read<bool>(1))
31 AccountId = _stream.Read<uint32>(32);
32
33 if (_stream.Read<bool>(1))
34 {
35 GameAccount = boost::in_place();
36 GameAccount->Region = _stream.Read<uint8>(8);
37 GameAccount->ProgramId = _stream.ReadFourCC();
38 GameAccount->Id = _stream.Read<uint32>(32);
39 }
40
41 _stream.ReadSkip(7);
42
43 if (_stream.Read<bool>(1))
44 Nickname = _stream.ReadString(7);
45
46 Source = _stream.ReadFourCC();
47 Role = _stream.Read<uint32>(32);
48
49 if (_stream.Read<bool>(1))
50 InvitationMsg = _stream.ReadString(9);
51}

53std::string Battlenet::Friends::SendInvitationRequest::ToString() const
54{
55 std::ostringstream stream;
56 stream << "Battlenet::Friends::SendInvitationRequest" << std::endl;
57 APPEND_FIELD(stream, Token);
58 APPEND_FIELD(stream, PresenceId);
59 APPEND_FIELD(stream, GameAccount);
60 APPEND_FIELD(stream, AccountId);
61 APPEND_FIELD(stream, AccountMail);
62 APPEND_FIELD(stream, Nickname);
63 APPEND_FIELD(stream, InvitationMsg);
64 APPEND_FIELD(stream, Source);
65 APPEND_FIELD(stream, Role);
66 return stream.str();
67}

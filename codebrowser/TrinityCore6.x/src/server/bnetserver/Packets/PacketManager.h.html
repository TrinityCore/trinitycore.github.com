25template<typename T>
26struct has_call_handler
27{
28 template<typename U, void(U::*)(Battlenet::Session*)> struct test_has_call_handler { };
29 template<typename U> static char Test(test_has_call_handler<U, &U::CallHandler>*);
30 template<typename U> static int Test(...);
31 static const bool value = sizeof(Test<T>(nullptr)) == sizeof(char);
32};
33
34namespace Battlenet
35{
36 class PacketManager
37 {
38 typedef ClientPacket*(*PacketCreateFn)(PacketHeader const& header, BitStream& stream);
39
40 struct PacketInfo
41 {
42 PacketCreateFn Constructor;
43 char const* Name;
44 bool HasHandler;
45 };
46
47 PacketManager();
48
49 void RegisterAuthenticationPackets();
50 void RegisterConnectionPackets();
51 void RegisterWoWRealmPackets();
52 void RegisterFriendsPackets();
53 void RegisterPresencePackets();
54 void RegisterChatPackets();
55 void RegisterSupportPackets();
56 void RegisterAchievementPackets();
57 void RegisterCachePackets();
58 void RegisterProfilePackets();
59
60 template<class PacketType>
61 static ClientPacket* New(PacketHeader const& header, BitStream& stream)
62 {
63 return new PacketType(header, stream);
64 }
65
66 void RegisterPacketName(std::map<PacketHeader, PacketInfo>& packetTable, PacketHeader const& header, char const* name)
67 {
68 PacketInfo& info = packetTable[header];
69 info.Constructor = nullptr;
70 info.Name = name;
71 info.HasHandler = false;
72 }
73
74 template<class PacketType>
75 void RegisterClientPacket(PacketHeader const& header, char const* name)
76 {
77 PacketInfo& info = _clientPacketTable[header];
78 info.Constructor = &New<PacketType>;
79 info.Name = name;
80 info.HasHandler = has_call_handler<PacketType>::value;
81 }
82
83 public:
84 ClientPacket* CreateClientPacket(PacketHeader const& header, BitStream& stream);
85
86 char const* GetClientPacketName(PacketHeader const& header);
87 char const* GetServerPacketName(PacketHeader const& header);
88
89 bool IsHandled(PacketHeader const& header);
90
91 static PacketManager& Instance()
92 {
93 static PacketManager instance;
94 return instance;
95 }
96
97 private:
98 std::map<PacketHeader, PacketInfo> _clientPacketTable;
99 std::map<PacketHeader, PacketInfo> _serverPacketTable;
100 };
101}

22std::string Battlenet::PacketHeader::ToString() const
23{
24 std::ostringstream stream;
25 stream << "Battlenet::PacketHeader" << std::endl;
26 APPEND_FIELD(stream, Command);
27 APPEND_FIELD(stream, Channel);
28 return stream.str();
29}
30
31Battlenet::ServerPacket::ServerPacket(PacketHeader const& header) : Packet(header, *new BitStream())
32{
33 _stream.Write(header.Command, 6);
34 _stream.Write(1, 1);
35 _stream.Write(header.Channel, 4);
36}
37
38Battlenet::ServerPacket::~ServerPacket()
39{
40 delete &_stream;
41}

43void Battlenet::ClientPacket::CallHandler(Session* session)

33void Battlenet::Authentication::ResumeRequest::Read()
34{
35 Common.Program = _stream.ReadFourCC();
36 Common.Platform = _stream.ReadFourCC();
37 Common.Locale = _stream.ReadFourCC();
38
39 Common.Versions.resize(_stream.Read<uint32>(6));
40 for (size_t i = 0; i < Common.Versions.size(); ++i)
41 {
42 Version::Record& component = Common.Versions[i];
43 component.ProgramId = _stream.ReadFourCC();
44 component.Component = _stream.ReadFourCC();
45 component.Version = _stream.Read<uint32>(32);
46 }
47
48 Account = _stream.ReadString(9, 3);
49 GameAccountRegion = _stream.Read<uint8>(8);
50 GameAccountName = _stream.ReadString(5, 1);
51}

75void Battlenet::Authentication::ProofResponse::Read()
76{
77 Response.resize(_stream.Read<uint32>(3));
78 for (size_t i = 0; i < Response.size(); ++i)
79 {
80 BitStream*& dataStream = Response[i];
81 dataStream = new BitStream(_stream.Read<uint32>(10));
82 memcpy(dataStream->GetBuffer(), _stream.ReadBytes(dataStream->GetSize()).get(), dataStream->GetSize());
83 }
84}

141void Battlenet::Authentication::LogonResponse::Write()
142{
143 _stream.Write(Result.Type, 1);
144 if (Result.Type == ResultType::SUCCESS)
145 {
146 _stream.Write(Result.Success.FinalRequest.size(), 3);
147 for (size_t i = 0; i < Result.Success.FinalRequest.size(); ++i)
148 {
149 ModuleInfo* info = Result.Success.FinalRequest[i];
150 _stream.WriteBytes(info->Handle.Type.c_str(), 4);
151 _stream.WriteFourCC(info->Handle.Region);
152 _stream.WriteBytes(info->Handle.ModuleId, 32);
153 _stream.Write(info->DataSize, 10);
154 _stream.WriteBytes(info->Data, info->DataSize);
155 }
156
157 _stream.Write(Result.Success.PingTimeout + std::numeric_limits<int32>::min(), 32);
158 _stream.Write(Result.Success.RegulatorRules.is_initialized(), 1);
159 if (Result.Success.RegulatorRules.is_initialized())
160 {
161 _stream.Write(Result.Success.RegulatorRules->Type == Regulator::LEAKY_BUCKET, 1);
162 if (Result.Success.RegulatorRules->Type == Regulator::LEAKY_BUCKET)
163 {
164 _stream.Write(Result.Success.RegulatorRules->LeakyBucket.Threshold, 32);
165 _stream.Write(Result.Success.RegulatorRules->LeakyBucket.Rate, 32);
166 }
167 }
168
169 _stream.WriteString(Result.Success.FullName.GivenName, 8);
170 _stream.WriteString(Result.Success.FullName.Surname, 8);
171 _stream.Write(Result.Success.AccountId, 32);
172 _stream.Write(Result.Success.Region, 8);
173 _stream.Write(Result.Success.Flags, 64);
174 _stream.Write(Result.Success.GameAccountRegion, 8);
175 _stream.WriteString(Result.Success.GameAccountName, 5, -1);
176 _stream.Write(Result.Success.GameAccountFlags, 64);
177 _stream.Write(Result.Success.LogonFailures, 32);
178 }
179 else
180 {
181 _stream.Write(Result.Failure.Strings.is_initialized(), 1);
182 if (Result.Failure.Strings.is_initialized())
183 {
184 _stream.WriteBytes(Result.Failure.Strings->Type.c_str(), 4);
185 _stream.WriteFourCC(Result.Failure.Strings->Region);
186 _stream.WriteBytes(Result.Failure.Strings->ModuleId, 32);
187 }
188
189 _stream.Write(Result.Failure.Result.Type, 2);
190 if (Result.Failure.Result.Type == FailureType::FAILURE)
191 {
192 _stream.Write(Result.Failure.Result.Failure.Error, 16);
193 _stream.Write(Result.Failure.Result.Failure.Wait + std::numeric_limits<int32>::min(), 32);
194 }
195 }
196
197 _stream.Write(Raf.is_initialized(), 1);
198 if (Raf.is_initialized())
199 {
200 _stream.Write(Raf->size(), 10);
201 _stream.WriteBytes(Raf->data(), Raf->size());
202 }
203}

22void Battlenet::Cache::GetStreamItemsRequest::Read()
23{
24 _stream.ReadSkip(31);
25 Token = _stream.Read<uint32>(32);
26 ReferenceTime = _stream.Read<int32>(32) - std::numeric_limits<int32>::min();
27 Direction = _stream.Read<uint8>(1);
28 MaxItems = _stream.Read<uint8>(6);
29 Locale = _stream.ReadFourCC();
30 Stream.Type = _stream.Read<uint8>(1);
31 if (Stream.Type == StreamId::DESCRIPTION)
32 {
33 Stream.Description.ItemName = _stream.ReadFourCC();
34 Stream.Description.Channel = _stream.ReadFourCC();
35 }
36 else
37 Stream.Index = _stream.Read<uint16>(16);
38}

80Battlenet::Cache::GetStreamItemsResponse::~GetStreamItemsResponse()
81{
82 for (size_t i = 0; i < Items.size(); ++i)
83 delete Items[i];
84}
85
86void Battlenet::Cache::GetStreamItemsResponse::Write()
87{
88 _stream.Write(Offset, 16);
89 _stream.Write(TotalNumItems, 16);
90 _stream.Write(Token, 32);
91 _stream.Write(Items.size(), 6);
92 for (ModuleInfo const* info : Items)
93 {
94 _stream.WriteBytes(info->Handle.Type.c_str(), 4);
95 _stream.WriteFourCC(info->Handle.Region);
96 _stream.WriteBytes(info->Handle.ModuleId, 32);
97 _stream.WriteSkip(27);
98 _stream.WriteBytes(info->Data, 4);
99 }
100}

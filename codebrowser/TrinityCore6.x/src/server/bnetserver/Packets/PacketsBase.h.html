36 enum Channel
37 {
38 AUTHENTICATION = 0,
39 CONNECTION = 1,
40 WOWREALM = 2,
41 FRIENDS = 3,
42 PRESENCE = 4,
43 CHAT = 5,
44 SUPPORT = 7,
45 ACHIEVEMENT = 8,
46 CACHE = 11,
47 PROFILE = 14
48 };
49
50 struct PacketHeader : public PrintableComponent
51 {
52 PacketHeader(uint32 opcode, uint32 channel) : Command(opcode), Channel(channel) { }
53 PacketHeader() : Command(0), Channel(AUTHENTICATION) { }
54
55 uint32 Command;
56 int32 Channel;
57
58 bool operator<(PacketHeader const& right) const
59 {
60 if (Command < right.Command)
61 return true;
62 if (Command > right.Command)
63 return false;
64
65 return Channel < right.Channel;
66 }
67
68 bool operator==(PacketHeader const& right) const
69 {
70 return Command == right.Command && Channel == right.Channel;
71 }
72
73 std::string ToString() const override;
74 };
75
76 class Packet : public PrintableComponent
77 {
78 public:
79 Packet(PacketHeader const& header, BitStream& stream) : _header(header), _stream(stream) { }
80 virtual ~Packet() { }
81
82 PacketHeader const& GetHeader() const { return _header; }
83
84 virtual void Write() = 0;
85 virtual void Read() = 0;
86
87 protected:
88 PacketHeader _header;
89 BitStream& _stream;
90
91 private:
92 Packet(Packet const& right) = delete;
93 Packet& operator=(Packet const& right) = delete;
94 };
95
96 class ClientPacket : public Packet
97 {
98 public:
99 ClientPacket(PacketHeader const& header, BitStream& stream) : Packet(header, stream) { }
100
101 void Write() override final { ASSERT(!"Write not implemented for client packets."); }
102 virtual void CallHandler(Session* session);
103 };
104
105 class ServerPacket : public Packet
106 {
107 public:
108 ServerPacket(PacketHeader const& header);
109 ~ServerPacket();
110
111 void Read() override final { ASSERT(!"Read not implemented for server packets."); }
112
113 uint8* GetData() { return _stream.GetBuffer(); }
114 uint8 const* GetData() const { return _stream.GetBuffer(); }
115 size_t GetSize() const { return _stream.GetSize(); }
116 };

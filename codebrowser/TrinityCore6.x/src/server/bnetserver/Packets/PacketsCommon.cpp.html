21std::string Battlenet::Version::Record::ToString() const
22{
23 std::ostringstream stream;
24 stream << "Battlenet::Version::Record" << std::endl;
25 APPEND_FIELD(stream, ProgramId);
26 APPEND_FIELD(stream, Component);
27 APPEND_FIELD(stream, Version);
28 return stream.str();
29}

101std::ostream& Battlenet::Format::FieldToString(std::ostream& stream, char const* fieldName, PrintableComponent const& u, std::true_type)
102{
103 if (fieldName[0])
104 stream << fieldName << ": ";
105 return stream << u.ToString();
106}

108std::ostream& Battlenet::Format::FieldToString(std::ostream& stream, char const* fieldName, uint8 const& u, std::false_type)
109{
110 return FieldToString(stream, fieldName, uint32(u), std::false_type());
111}

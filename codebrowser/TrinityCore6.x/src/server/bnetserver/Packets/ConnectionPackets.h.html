27 enum Opcode
28 {
29 CMSG_PING = 0x0,
30 CMSG_ENABLE_ENCRYPTION = 0x5,
31 CMSG_LOGOUT_REQUEST = 0x6,
32 CMSG_DISCONNECT_REQUEST = 0x7, // Not handled
33 CMSG_CONNECTION_CLOSING = 0x9,
34
35 SMSG_PONG = 0x0,
36 SMSG_BOOM = 0x1, // Not implemented
37 SMSG_REGULATOR_UPDATE = 0x2, // Not implemented
38 SMSG_SERVER_VERSION = 0x3, // Not implemented
39 SMSG_STUN_SERVERS = 0x4 // Not implemented
40 };

42 class Ping final : public ClientPacket
43 {
44 public:
45 Ping(PacketHeader const& header, BitStream& stream) : ClientPacket(header, stream)
46 {
47 ASSERT(header == PacketHeader(CMSG_PING, CONNECTION) && "Invalid packet header for Ping");
48 }
49
50 void Read() override { }
51 std::string ToString() const override;
52 void CallHandler(Session* session) override;
53 };
54
55 class EnableEncryption final : public ClientPacket
56 {
57 public:
58 EnableEncryption(PacketHeader const& header, BitStream& stream) : ClientPacket(header, stream)
59 {
60 ASSERT(header == PacketHeader(CMSG_ENABLE_ENCRYPTION, CONNECTION) && "Invalid packet header for EnableEncryption");
61 }
62
63 void Read() override { }
64 std::string ToString() const override;
65 void CallHandler(Session* session) override;
66 uint8* GetRemainingData() { return _stream.GetBuffer() + (((_stream.GetReadPos() + 7) & ~7) / 8); }
67 size_t GetRemainingSize() const { return _stream.GetSize() - (((_stream.GetReadPos() + 7) & ~7) / 8); }
68 };

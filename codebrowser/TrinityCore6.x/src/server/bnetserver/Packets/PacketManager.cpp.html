20Battlenet::ClientPacket* Battlenet::PacketManager::CreateClientPacket(PacketHeader const& header, BitStream& stream)
21{
22 auto packetInfo = _clientPacketTable.find(header);
23 if (packetInfo == _clientPacketTable.end())
24 return nullptr;
25
26 if (!packetInfo->second.Constructor)
27 return nullptr;
28
29 ClientPacket* packet = packetInfo->second.Constructor(header, stream);
30 packet->Read();
31 return packet;
32}

52bool Battlenet::PacketManager::IsHandled(PacketHeader const& header)
53{
54 auto packetInfo = _clientPacketTable.find(header);
55 if (packetInfo == _clientPacketTable.end())
56 return false;
57
58 return packetInfo->second.HasHandler;
59}

57void Battlenet::WoWRealm::ListSubscribeResponse::Write()
58{
59 _stream.Write(Type, 1);
60 if (Type == SUCCESS)
61 {
62 _stream.Write(ToonCounts.size(), 7);
63 for (ToonCountEntry const& entry : ToonCounts)
64 {
65 _stream.Write(entry.Realm.Region, 8);
66 _stream.WriteSkip(12);
67 _stream.Write(entry.Realm.Site, 8);
68 _stream.Write(entry.Realm.Realm, 32);
69 _stream.Write(entry.Count, 16);
70 }
71 }
72 else
73 _stream.Write(Failure, 8);
74}

98void Battlenet::WoWRealm::ListUpdate::Write()
99{
100 _stream.Write(State.Type, 1);
101 if (State.Type == StateType::UPDATE)
102 {
103 _stream.Write(State.Update.Category, 32);
104 _stream.WriteFloat(State.Update.Population);
105 _stream.Write(State.Update.StateFlags, 8);
106 _stream.WriteSkip(19);
107 _stream.Write(State.Update.Type + -std::numeric_limits<int32>::min(), 32);
108 _stream.WriteString(State.Update.Name, 10);
109 _stream.Write(State.Update.PrivilegedData.is_initialized(), 1);
110 if (State.Update.PrivilegedData.is_initialized())
111 {
112 _stream.WriteString(State.Update.PrivilegedData->Version, 5);
113 _stream.Write(State.Update.PrivilegedData->ConfigId, 32);
114
115 boost::asio::ip::address_v4::bytes_type ip = State.Update.PrivilegedData->Address.address().to_v4().to_bytes();
116 uint16 port = State.Update.PrivilegedData->Address.port();
117
118 EndianConvertReverse(ip);
119 EndianConvertReverse(port);
120
121 _stream.WriteBytes(ip.data(), 4);
122 _stream.WriteBytes(&port, 2);
123 }
124
125 _stream.Write(State.Update.InfoFlags, 8);
126 }
127
128 _stream.Write(Id.Region, 8);
129 _stream.WriteSkip(12);
130 _stream.Write(Id.Site, 8);
131 _stream.Write(Id.Realm, 32);
132}

180void Battlenet::WoWRealm::ToonReady::Write()
181{
182 _stream.Write(Name.Region, 8);
183 _stream.WriteFourCC(Name.ProgramId);
184 _stream.Write(Name.Realm, 32);
185 _stream.WriteString(Name.Name, 7, -2);
186 _stream.WriteSkip(21);
187 _stream.Write(ProfileAddress.Id, 64);
188 _stream.Write(ProfileAddress.Label, 32);
189 _stream.Write(Handle.Id, 64);
190 _stream.Write(Handle.Realm, 32);
191 _stream.Write(Handle.Region, 8);
192 _stream.WriteFourCC(Handle.ProgramId);
193}

205void Battlenet::WoWRealm::JoinResponseV2::Write()
206{
207 _stream.Write(Type, 1);
208 if (Type == SUCCESS)
209 {
210 _stream.Write(Success.ServerSalt, 32);
211 _stream.Write(Success.IPv4.size(), 5);
212 for (tcp::endpoint const& addr : Success.IPv4)
213 {
214 boost::asio::ip::address_v4::bytes_type ip = addr.address().to_v4().to_bytes();
215 uint16 port = addr.port();
216
217 EndianConvertReverse(port);
218
219 _stream.WriteBytes(ip.data(), 4);
220 _stream.WriteBytes(&port, 2);
221 }
222
223 _stream.Write(Success.IPv6.size(), 5);
224 for (tcp::endpoint const& addr : Success.IPv6)
225 {
226 boost::asio::ip::address_v6::bytes_type ip = addr.address().to_v6().to_bytes();
227 uint16 port = addr.port();
228
229 EndianConvertReverse(port);
230
231 _stream.WriteBytes(ip.data(), 16);
232 _stream.WriteBytes(&port, 2);
233 }
234 }
235 else
236 _stream.Write(Failure, 8);
237}

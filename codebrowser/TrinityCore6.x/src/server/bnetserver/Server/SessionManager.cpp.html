20bool Battlenet::SessionManager::StartNetwork(boost::asio::io_service& service, std::string const& bindIp, uint16 port)
21{
22 if (!BaseSocketMgr::StartNetwork(service, bindIp, port))
23 return false;
24
25 _acceptor->SetSocketFactory(std::bind(&BaseSocketMgr::GetSocketForAccept, this));
26 _acceptor->AsyncAcceptWithCallback<&OnSocketAccept>();
27 return true;
28}

40void Battlenet::SessionManager::AddSession(Session* session)
41{
42 std::unique_lock<boost::shared_mutex> lock(_sessionMutex);
43 _sessions[{ session->GetAccountId(), session->GetGameAccountId() }] = session;
44 _sessionsByAccountId[session->GetAccountId()].push_back(session);
45}
46
47void Battlenet::SessionManager::RemoveSession(Session* session)
48{
49 std::unique_lock<boost::shared_mutex> lock(_sessionMutex);
50 auto itr = _sessions.find({ session->GetAccountId(), session->GetGameAccountId() });
51 // Remove old session only if it was not overwritten by reconnecting
52 if (itr != _sessions.end() && itr->second == session)
53 _sessions.erase(itr);
54
55 _sessionsByAccountId[session->GetAccountId()].remove(session);
56}
57
58Battlenet::Session* Battlenet::SessionManager::GetSession(uint32 accountId, uint32 gameAccountId) const
59{
60 boost::shared_lock<boost::shared_mutex> lock(_sessionMutex);
61 auto itr = _sessions.find({ accountId, gameAccountId });
62 if (itr != _sessions.end())
63 return itr->second;
64
65 return nullptr;
66}
67
68std::list<Battlenet::Session*> Battlenet::SessionManager::GetSessions(uint32 accountId) const
69{
70 boost::shared_lock<boost::shared_mutex> lock(_sessionMutex);
71 std::list<Session*> sessions;
72 auto itr = _sessionsByAccountId.find(accountId);
73 if (itr != _sessionsByAccountId.end())
74 sessions = itr->second;
75
76 return sessions;
77}

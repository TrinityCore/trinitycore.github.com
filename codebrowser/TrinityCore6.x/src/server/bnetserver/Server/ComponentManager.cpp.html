21Battlenet::ComponentMgr::~ComponentMgr()
22{
23 for (Version::Record* component : _components)
24 delete component;
25}
26
27void Battlenet::ComponentMgr::Load()
28{
29 QueryResult result = LoginDatabase.Query("SELECT Program, Platform, Build FROM battlenet_components");
30 if (result)
31 {
32 do
33 {
34 Field* fields = result->Fetch();
35 Version::Record* component = new Version::Record();
36 component->ProgramId = fields[0].GetString();
37 component->Component = fields[1].GetString();
38 component->Version = fields[2].GetUInt32();
39
40 _components.insert(component);
41 _programs.insert(component->ProgramId);
42 _platforms.insert(component->Component);
43
44 } while (result->NextRow());
45 }
46}
47
48bool Battlenet::ComponentMgr::HasComponent(Battlenet::Version::Record const* component) const
49{
50 for (Version::Record const* c : _components)
51 if (component->ProgramId == c->ProgramId && component->Component == c->Component && component->Version == c->Version)
52 return true;
53
54 return false;
55}

27namespace Battlenet
28{
29 class ComponentMgr
30 {
31 ComponentMgr() { }
32 ~ComponentMgr();
33
34 public:
35 void Load();
36 bool HasComponent(Version::Record const* component) const;
37 bool HasProgram(std::string const& program) const { return _programs.count(program) != 0; }
38 bool HasPlatform(std::string const& platform) const { return _platforms.count(platform) != 0; }
39
40 static ComponentMgr* instance()

42 static ComponentMgr instance;
43 return &instance;
44 }
45
46 private:
47 std::set<Version::Record*> _components;
48 std::set<std::string> _programs;
49 std::set<std::string> _platforms;
50 };
51}

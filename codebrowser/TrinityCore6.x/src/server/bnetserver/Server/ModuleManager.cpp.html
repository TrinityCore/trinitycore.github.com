21Battlenet::ModuleManager::~ModuleManager()
22{
23 for (auto const& m : _modules)
24 delete m.second;
25}

27void Battlenet::ModuleManager::Load()
28{
29 QueryResult result = LoginDatabase.Query("SELECT `Hash`, `Name`, `Type`, `System`, `Data` FROM battlenet_modules");
30 if (result)
31 {
32 do
33 {
34 Field* fields = result->Fetch();
35 ModuleInfo* module = new ModuleInfo();
36 module->Handle.Type = fields[2].GetString();
37 HexStrToByteArray(fields[0].GetString(), module->Handle.ModuleId);
38 std::string data = fields[4].GetString();
39 module->DataSize = data.length() / 2;
40 if (module->DataSize)
41 {
42 module->Data = new uint8[data.length() / 2];
43 HexStrToByteArray(data, module->Data);
44 }
45
46 _modules[{ fields[3].GetString(), fields[1].GetString() }] = module;
47 } while (result->NextRow());
48 }
49}

51Battlenet::ModuleInfo* Battlenet::ModuleManager::CreateModule(std::string const& os, std::string const& name) const
52{
53 ModuleKey key { os, name };
54 if (!_modules.count(key))
55 return nullptr;
56
57 return new ModuleInfo(*_modules.at(key));
58}

60std::string Battlenet::ModuleInfo::ToString() const
61{
62 std::ostringstream stream;
63 stream << "Battlenet::ModuleInput" << std::endl;
64 APPEND_FIELD(stream, Handle);
65 return stream.str();
66}

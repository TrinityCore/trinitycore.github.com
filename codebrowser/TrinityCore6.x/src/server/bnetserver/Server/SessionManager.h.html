21#include "Session.h"
22#include "SocketMgr.h"
23#include <boost/thread/locks.hpp>
24#include <boost/thread/shared_mutex.hpp>
25
26namespace Battlenet
27{
28#pragma pack(push, 1)
29
30 struct SessionInfo
31 {
32 uint32 AccountId;
33 uint32 GameAccountId;
34
35 bool operator<(SessionInfo const& right) const
36 {
37 return memcmp(this, &right, sizeof(SessionInfo)) < 0;
38 }
39 };
40
41#pragma pack(pop)
42
43 class SessionManager : public SocketMgr<Session>
44 {
45 typedef SocketMgr<Session> BaseSocketMgr;
46 typedef std::map<SessionInfo, Session*> SessionMap;
47 typedef std::map<uint32, std::list<Session*>> SessionByAccountMap;
48
49 public:
50 static SessionManager& Instance()
51 {
52 static SessionManager instance;
53 return instance;
54 }
55
56 bool StartNetwork(boost::asio::io_service& service, std::string const& bindIp, uint16 port) override;
57
58 // noop for now, will be needed later to broadcast realmlist updates for example
59 void AddSession(Session* /*session*/);
60
61 void RemoveSession(Session* /*session*/);
62
63 Session* GetSession(uint32 accountId, uint32 gameAccountId) const;
64 std::list<Session*> GetSessions(uint32 accountId) const;
65
66 template<typename Iterator>
67 void LockedForEach(Iterator iterator) const
68 {
69 boost::shared_lock<boost::shared_mutex> lock(_sessionMutex);
70 for (SessionMap::value_type const& pair : _sessions)
71 iterator(pair.second);
72 }
73
74 protected:
75 NetworkThread<Session>* CreateThreads() const override;
76
77 private:
78 static void OnSocketAccept(tcp::socket&& sock, uint32 threadIndex);
79
80 SessionMap _sessions;
81 SessionByAccountMap _sessionsByAccountId;
82 mutable boost::shared_mutex _sessionMutex;
83 };
84}

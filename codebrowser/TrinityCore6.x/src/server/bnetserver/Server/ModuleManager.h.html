23namespace Battlenet
24{
25 struct ModuleKey
26 {
27 std::string Platform;
28 std::string Name;
29
30 bool operator<(ModuleKey const& right) const
31 {
32 int32 res = Platform.compare(right.Platform);
33 if (res < 0)
34 return true;
35 else if (res > 0)
36 return false;
37
38 return Name < right.Name;
39 }
40 };
41
42 struct ModuleInfo : public PrintableComponent
43 {
44 ModuleInfo() : DataSize(0), Data(nullptr) { Handle.Region = "EU"; }
45 ModuleInfo(ModuleInfo const& right) : DataSize(right.DataSize), Data(nullptr)
46 {
47 Handle.Type = right.Handle.Type;
48 Handle.Region = right.Handle.Region;
49 memcpy(Handle.ModuleId, right.Handle.ModuleId, 32);
50 if (DataSize)
51 {
52 Data = new uint8[DataSize];
53 memcpy(Data, right.Data, DataSize);
54 }
55 }
56
57 virtual ~ModuleInfo()
58 {
59 delete[] Data;
60 }
61
62 Cache::Handle Handle;
63 uint32 DataSize;
64 uint8* Data;
65
66 std::string ToString() const override;
67 };

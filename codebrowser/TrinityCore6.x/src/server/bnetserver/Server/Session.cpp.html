39void Battlenet::AccountInfo::LoadResult(Field* fields)
40{
41 // ba.id, ba.email, ba.locked, ba.lock_country, ba.last_ip, ba.failed_logins, bab.unbandate > UNIX_TIMESTAMP() OR bab.unbandate = bab.bandate, bab.unbandate = bab.bandate FROM battlenet_accounts ba LEFT JOIN battlenet_account_bans bab WHERE email = ?
42 Id = fields[0].GetUInt32();
43 Login = fields[1].GetString();
44 IsLockedToIP = fields[2].GetBool();
45 LockCountry = fields[3].GetString();
46 LastIP = fields[4].GetString();
47 FailedLogins = fields[5].GetUInt32();
48 IsBanned = fields[6].GetUInt64() != 0;
49 IsPermanenetlyBanned = fields[7].GetUInt64() != 0;
50}

52void Battlenet::GameAccountInfo::LoadResult(Field* fields)
53{
54 // a.id, a.username, ab.unbandate > UNIX_TIMESTAMP() OR ab.unbandate = ab.bandate, ab.unbandate = ab.bandate, aa.gmlevel
55 Id = fields[0].GetUInt32();
56 Name = fields[1].GetString();
57 IsBanned = fields[2].GetUInt64() != 0;
58 IsPermanenetlyBanned = fields[3].GetUInt64() != 0;
59 SecurityLevel = AccountTypes(fields[4].GetUInt8());
60
61 std::size_t hashPos = Name.find('#');
62 if (hashPos != std::string::npos)
63 DisplayName = std::string("WoW") + Name.substr(hashPos + 1);
64 else
65 DisplayName = Name;
66}

101void Battlenet::Session::_SetVSFields(std::string const& pstr)
102{
103 s.SetRand(uint32(BufferSizes::SRP_6_S) * 8);
104
105 BigNumber p;
106 p.SetHexStr(pstr.c_str());
107
108 SHA256Hash sha;
109 sha.UpdateBigNumbers(&s, &p, NULL);
110 sha.Finalize();
111 BigNumber x;
112 x.SetBinary(sha.GetDigest(), sha.GetLength());
113 v = g.ModExp(x, N);
114
115 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_BNET_VS_FIELDS);
116 stmt->setString(0, v.AsHexStr());
117 stmt->setString(1, s.AsHexStr());
118 stmt->setString(2, _accountInfo->Login);
119
120 LoginDatabase.Execute(stmt);
121}

128void Battlenet::Session::HandleLogonRequest(Authentication::LogonRequest3 const& logonRequest)
129{
130 if (_queryCallback)
131 {
132 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
133 logonResponse->SetAuthResult(AUTH_LOGON_TOO_FAST);
134 AsyncWrite(logonResponse);
135 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s attempted to log too quick after previous attempt!", GetClientInfo().c_str());
136 return;
137 }
138
139 if (logonRequest.Common.Program != "WoW")
140 {
141 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
142 logonResponse->SetAuthResult(AUTH_INVALID_PROGRAM);
143 AsyncWrite(logonResponse);
144 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s attempted to log in with game other than WoW (using %s)!", GetClientInfo().c_str(), logonRequest.Common.Program.c_str());
145 return;
146 }
147
148 if (!sComponentMgr->HasPlatform(logonRequest.Common.Platform))
149 {
150 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
151 logonResponse->SetAuthResult(AUTH_INVALID_OS);
152 AsyncWrite(logonResponse);
153 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s attempted to log in from an unsupported platform (using %s)!", GetClientInfo().c_str(), logonRequest.Common.Platform.c_str());
154 return;
155 }
156
157 if (!sComponentMgr->HasPlatform(logonRequest.Common.Locale))
158 {
159 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
160 logonResponse->SetAuthResult(AUTH_UNSUPPORTED_LANGUAGE);
161 AsyncWrite(logonResponse);
162 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s attempted to log in with unsupported locale (using %s)!", GetClientInfo().c_str(), logonRequest.Common.Locale.c_str());
163 return;
164 }
165
166 for (Version::Record const& component : logonRequest.Common.Versions)
167 {
168 if (!sComponentMgr->HasComponent(&component))
169 {
170 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
171 if (!sComponentMgr->HasProgram(component.ProgramId))
172 {
173 logonResponse->SetAuthResult(AUTH_INVALID_PROGRAM);
174 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s is using unsupported component program %s!", GetClientInfo().c_str(), component.ProgramId.c_str());
175 }
176 else if (!sComponentMgr->HasPlatform(component.Component))
177 {
178 logonResponse->SetAuthResult(AUTH_INVALID_OS);
179 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s is using unsupported component platform %s!", GetClientInfo().c_str(), component.Component.c_str());
180 }
181 else
182 {
183 if (component.ProgramId != "WoW" || AuthHelper::IsBuildSupportingBattlenet(component.Version))
184 logonResponse->SetAuthResult(AUTH_REGION_BAD_VERSION);
185 else
186 logonResponse->SetAuthResult(AUTH_USE_GRUNT_LOGON);
187
188 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s is using unsupported component version %u!", GetClientInfo().c_str(), component.Version);
189 }
190
191 AsyncWrite(logonResponse);
192 return;
193 }
194
195 if (component.Component == "base")
196 _build = component.Version;
197 }
198
199 std::string login = logonRequest.Account;
200 _locale = logonRequest.Common.Locale;
201 _os = logonRequest.Common.Platform;
202
203 Utf8ToUpperOnlyLatin(login);
204 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_BNET_ACCOUNT_INFO);
205 stmt->setString(0, login);
206
207 _queryCallback = std::bind(&Battlenet::Session::HandleLogonRequestCallback, this, std::placeholders::_1);
208 _queryFuture = LoginDatabase.AsyncQuery(stmt);
209}

211void Battlenet::Session::HandleLogonRequestCallback(PreparedQueryResult result)
212{
213 if (!result)
214 {
215 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
216 logonResponse->SetAuthResult(AUTH_UNKNOWN_ACCOUNT);
217 AsyncWrite(logonResponse);
218 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] %s is trying to log in from unknown account!", GetClientInfo().c_str());
219 return;
220 }
221
222 Field* fields = result->Fetch();
223 _accountInfo->LoadResult(fields);
224
225 std::string pStr = fields[8].GetString();
226 std::string databaseV = fields[9].GetString();
227 std::string databaseS = fields[10].GetString();
228
229 _gameAccounts.resize(result->GetRowCount());
230 uint32 i = 0;
231 do
232 {
233 _gameAccounts[i++].LoadResult(result->Fetch() + 11);
234
235 } while (result->NextRow());
236
237 std::string ip_address = GetRemoteIpAddress().to_string();
238 // If the IP is 'locked', check that the player comes indeed from the correct IP address
239 if (_accountInfo->IsLockedToIP)
240 {
241 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] Account '%s' is locked to IP - '%s' is logging in from '%s'", _accountInfo->Login.c_str(), _accountInfo->LastIP.c_str(), ip_address.c_str());
242
243 if (_accountInfo->LastIP != ip_address)
244 {
245 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
246 logonResponse->SetAuthResult(AUTH_ACCOUNT_LOCKED);
247 AsyncWrite(logonResponse);
248 return;
249 }
250 }
251 else
252 {
253 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] Account '%s' is not locked to ip", _accountInfo->Login.c_str());
254 if (_accountInfo->LockCountry.empty() || _accountInfo->LockCountry == "00")
255 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] Account '%s' is not locked to country", _accountInfo->Login.c_str());
256 else if (!_accountInfo->LockCountry.empty() && !_ipCountry.empty())
257 {
258 TC_LOG_DEBUG("session", "[Battlenet::LogonRequest] Account '%s' is locked to country: '%s' Player country is '%s'", _accountInfo->Login.c_str(), _accountInfo->LockCountry.c_str(), _ipCountry.c_str());
259 if (_ipCountry != _accountInfo->LockCountry)
260 {
261 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
262 logonResponse->SetAuthResult(AUTH_ACCOUNT_LOCKED);
263 AsyncWrite(logonResponse);
264 return;
265 }
266 }
267 }
268
269 // If the account is banned, reject the logon attempt
270 if (_accountInfo->IsBanned)
271 {
272 if (_accountInfo->IsPermanenetlyBanned)
273 {
274 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
275 logonResponse->SetAuthResult(LOGIN_BANNED);
276 AsyncWrite(logonResponse);
277 TC_LOG_DEBUG("session", "'%s:%d' [Battlenet::LogonRequest] Banned account %s tried to login!", ip_address.c_str(), GetRemotePort(), _accountInfo->Login.c_str());
278 return;
279 }
280 else
281 {
282 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
283 logonResponse->SetAuthResult(LOGIN_SUSPENDED);
284 AsyncWrite(logonResponse);
285 TC_LOG_DEBUG("session", "'%s:%d' [Battlenet::LogonRequest] Temporarily banned account %s tried to login!", ip_address.c_str(), GetRemotePort(), _accountInfo->Login.c_str());
286 return;
287 }
288 }
289
290 SHA256Hash sha;
291 sha.UpdateData(_accountInfo->Login);
292 sha.Finalize();
293
294 I.SetBinary(sha.GetDigest(), sha.GetLength());
295
296 ModuleInfo* password = sModuleMgr->CreateModule(_os, "Password");
297 ModuleInfo* thumbprint = sModuleMgr->CreateModule(_os, "Thumbprint");
298
299 if (databaseV.size() != size_t(BufferSizes::SRP_6_V) * 2 || databaseS.size() != size_t(BufferSizes::SRP_6_S) * 2)
300 _SetVSFields(pStr);
301 else
302 {
303 s.SetHexStr(databaseS.c_str());
304 v.SetHexStr(databaseV.c_str());
305 }
306
307 b.SetRand(128 * 8);
308 B = ((v * k) + g.ModExp(b, N)) % N;
309 BigNumber unk;
310 unk.SetRand(128 * 8);
311
312 BitStream passwordData;
313 uint8 state = 0;
314 passwordData.WriteBytes(&state, 1);
315 passwordData.WriteBytes(I.AsByteArray(32).get(), 32);
316 passwordData.WriteBytes(s.AsByteArray(32).get(), 32);
317 passwordData.WriteBytes(B.AsByteArray(128).get(), 128);
318 passwordData.WriteBytes(unk.AsByteArray(128).get(), 128);
319
320 password->DataSize = passwordData.GetSize();
321 password->Data = new uint8[password->DataSize];
322 memcpy(password->Data, passwordData.GetBuffer(), password->DataSize);
323
324 _modulesWaitingForData.push(MODULE_PASSWORD);
325
326 Authentication::ProofRequest* proofRequest = new Authentication::ProofRequest();
327 proofRequest->Modules.push_back(password);
328 // if has authenticator, send Token module
329 proofRequest->Modules.push_back(thumbprint);
330 AsyncWrite(proofRequest);
331}

360void Battlenet::Session::HandleResumeRequestCallback(PreparedQueryResult result)
361{
362 if (!result)
363 {
364 Authentication::ResumeResponse* resumeResponse = new Authentication::ResumeResponse();
365 resumeResponse->SetAuthResult(AUTH_UNKNOWN_ACCOUNT);
366 AsyncWrite(resumeResponse);
367 return;
368 }
369
370 Field* fields = result->Fetch();
371 _accountInfo->LoadResult(fields);
372 K.SetHexStr(fields[8].GetString().c_str());
373
374 _gameAccounts.resize(1);
375 _gameAccountInfo = &_gameAccounts[0];
376 _gameAccountInfo->LoadResult(fields + 9);
377
378 ModuleInfo* thumbprint = sModuleMgr->CreateModule(_os, "Thumbprint");
379 ModuleInfo* resume = sModuleMgr->CreateModule(_os, "Resume");
380 BitStream resumeData;
381 uint8 state = 0;
382 _reconnectProof.SetRand(16 * 8);
383
384 resumeData.WriteBytes(&state, 1);
385 resumeData.WriteBytes(_reconnectProof.AsByteArray().get(), 16);
386
387 resume->DataSize = resumeData.GetSize();
388 resume->Data = new uint8[resume->DataSize];
389 memcpy(resume->Data, resumeData.GetBuffer(), resume->DataSize);
390
391 _modulesWaitingForData.push(MODULE_RESUME);
392
393 Authentication::ProofRequest* proofRequest = new Authentication::ProofRequest();
394 proofRequest->Modules.push_back(thumbprint);
395 proofRequest->Modules.push_back(resume);
396 AsyncWrite(proofRequest);
397}

399void Battlenet::Session::HandleProofResponse(Authentication::ProofResponse const& proofResponse)
400{
401 if (_modulesWaitingForData.size() < proofResponse.Response.size())
402 {
403 Authentication::LogonResponse* complete = new Authentication::LogonResponse();
404 complete->SetAuthResult(AUTH_CORRUPTED_MODULE);
405 AsyncWrite(complete);
406 return;
407 }
408
409 ServerPacket* response = nullptr;
410 for (size_t i = 0; i < proofResponse.Response.size(); ++i)
411 {
412 if (!(this->*(ModuleHandlers[_modulesWaitingForData.front()]))(proofResponse.Response[i], &response))
413 break;
414
415 _modulesWaitingForData.pop();
416 }
417
418 if (!response)
419 {
420 response = new Authentication::LogonResponse();
421 static_cast<Authentication::LogonResponse*>(response)->SetAuthResult(AUTH_INTERNAL_ERROR);
422 }
423
424 AsyncWrite(response);
425}

491void Battlenet::Session::HandleJoinRequestV2(WoWRealm::JoinRequestV2 const& joinRequest)
492{
493 WoWRealm::JoinResponseV2* joinResponse = new WoWRealm::JoinResponseV2();
494 Realm const* realm = sRealmList->GetRealm(joinRequest.Id);
495 if (!realm || realm->Flags & (REALM_FLAG_VERSION_MISMATCH | REALM_FLAG_OFFLINE) || realm->Build != _build)
496 {
497 joinResponse->Type = WoWRealm::JoinResponseV2::FAILURE;
498 AsyncWrite(joinResponse);
499 return;
500 }
501
502 joinResponse->Success.ServerSalt = rand32();
503
504 uint8 sessionKey[40];
505 HmacSha1 hmac(K.GetNumBytes(), K.AsByteArray().get());
506 hmac.UpdateData((uint8*)"WoW\0", 4);
507 hmac.UpdateData((uint8*)&joinRequest.ClientSalt, 4);
508 hmac.UpdateData((uint8*)&joinResponse->Success.ServerSalt, 4);
509 hmac.Finalize();
510
511 memcpy(sessionKey, hmac.GetDigest(), hmac.GetLength());
512
513 HmacSha1 hmac2(K.GetNumBytes(), K.AsByteArray().get());
514 hmac2.UpdateData((uint8*)"WoW\0", 4);
515 hmac2.UpdateData((uint8*)&joinResponse->Success.ServerSalt, 4);
516 hmac2.UpdateData((uint8*)&joinRequest.ClientSalt, 4);
517 hmac2.Finalize();
518
519 memcpy(sessionKey + hmac.GetLength(), hmac2.GetDigest(), hmac2.GetLength());
520
521 LoginDatabase.DirectPExecute("UPDATE account SET sessionkey = '%s', last_ip = '%s', last_login = NOW(), locale = %u, failed_logins = 0, os = '%s' WHERE id = %u",
522 ByteArrayToHexStr(sessionKey, 40, true).c_str(), GetRemoteIpAddress().to_string().c_str(), GetLocaleByName(_locale), _os.c_str(), _gameAccountInfo->Id);
523
524 joinResponse->Success.IPv4.push_back(realm->GetAddressForClient(GetRemoteIpAddress()));
525
526 AsyncWrite(joinResponse);
527}

559void Battlenet::Session::ReadHandler()
560{
561 BitStream stream(std::move(GetReadBuffer()));
562 _crypt.DecryptRecv(stream.GetBuffer(), stream.GetSize());
563
564 while (!stream.IsRead())
565 {
566 try
567 {
568 PacketHeader header;
569 header.Command = stream.Read<uint32>(6);
570 if (stream.Read<bool>(1))
571 header.Channel = stream.Read<int32>(4);
572
573 if (header.Channel != AUTHENTICATION && (header.Channel != CONNECTION || header.Command != Connection::CMSG_PING) && !_authed)
574 {
575 TC_LOG_DEBUG("session.packets", "%s Received not allowed %s. Client has not authed yet.", GetClientInfo().c_str(), header.ToString().c_str());
576 CloseSocket();
577 return;
578 }
579
580 if (ClientPacket* packet = sPacketManager.CreateClientPacket(header, stream))
581 {
582 if (sPacketManager.IsHandled(header))
583 TC_LOG_DEBUG("session.packets", "%s Received %s", GetClientInfo().c_str(), PacketToStringHelper(packet).c_str());
584
585 packet->CallHandler(this);
586 delete packet;
587 }
588 else if (sPacketManager.GetClientPacketName(header))
589 {
590 LogUnhandledPacket(header);
591 break;
592 }
593 else
594 {
595 TC_LOG_DEBUG("session.packets", "%s Received unknown %s", GetClientInfo().c_str(), header.ToString().c_str());
596 break;
597 }
598
599 stream.AlignToNextByte();
600 }
601 catch (BitStreamPositionException const& e)
602 {
603 TC_LOG_ERROR("session.packets", "%s Exception thrown during packet processing %s", GetClientInfo().c_str(), e.what());
604 CloseSocket();
605 return;
606 }
607 }
608
609 GetReadBuffer().Resize(size_t(BufferSizes::Read));
610 AsyncRead();
611}

655bool Battlenet::Session::Update()
656{
657 EncryptableBuffer* queued;
658 MessageBuffer buffer((std::size_t(BufferSizes::Read)));
659 while (_bufferQueue.Dequeue(queued))
660 {
661 std::size_t packetSize = queued->Buffer.GetActiveSize();
662 if (queued->Encrypt)
663 _crypt.EncryptSend(queued->Buffer.GetReadPointer(), packetSize);
664
665 if (buffer.GetRemainingSpace() < packetSize)
666 {
667 QueuePacket(std::move(buffer));
668 buffer.Resize(std::size_t(BufferSizes::Read));
669 }
670
671 if (buffer.GetRemainingSpace() >= packetSize)
672 buffer.Write(queued->Buffer.GetReadPointer(), packetSize);
673 else // single packet larger than 16384 bytes - client will reject.
674 QueuePacket(std::move(queued->Buffer));
675
676 delete queued;
677 }
678
679 if (buffer.GetActiveSize() > 0)
680 QueuePacket(std::move(buffer));
681
682 if (!BattlenetSocket::Update())
683 return false;
684
685 if (_queryFuture.valid() && _queryFuture.wait_for(std::chrono::seconds(0)) == std::future_status::ready)
686 {
687 auto callback = _queryCallback;
688 _queryCallback = nullptr;
689 callback(_queryFuture.get());
690 }
691
692 return true;
693}

723bool Battlenet::Session::HandlePasswordModule(BitStream* dataStream, ServerPacket** response)
724{
725 if (dataStream->GetSize() != 1 + 128 + 32 + 128)
726 {
727 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
728 logonResponse->SetAuthResult(AUTH_CORRUPTED_MODULE);
729 ReplaceResponse(response, logonResponse);
730 return false;
731 }
732
733 if (dataStream->Read<uint8>(8) != 2) // State
734 {
735 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
736 logonResponse->SetAuthResult(AUTH_CORRUPTED_MODULE);
737 ReplaceResponse(response, logonResponse);
738 return false;
739 }
740
741 BigNumber A, clientM1, clientChallenge;
742 A.SetBinary(dataStream->ReadBytes(128).get(), 128);
743 clientM1.SetBinary(dataStream->ReadBytes(32).get(), 32);
744 clientChallenge.SetBinary(dataStream->ReadBytes(128).get(), 128);
745
746 if (A.IsZero())
747 {
748 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
749 logonResponse->SetAuthResult(AUTH_CORRUPTED_MODULE);
750 ReplaceResponse(response, logonResponse);
751 return false;
752 }
753
754 SHA256Hash sha;
755 sha.UpdateBigNumbers(&A, &B, NULL);
756 sha.Finalize();
757
758 BigNumber u;
759 u.SetBinary(sha.GetDigest(), sha.GetLength());
760
761 BigNumber S = ((A * v.ModExp(u, N)) % N).ModExp(b, N);
762
763 uint8 S_bytes[128];
764 memcpy(S_bytes, S.AsByteArray(128).get(), 128);
765
766 uint8 part1[64];
767 uint8 part2[64];
768
769 for (int i = 0; i < 64; ++i)
770 {
771 part1[i] = S_bytes[i * 2];
772 part2[i] = S_bytes[i * 2 + 1];
773 }
774
775 SHA256Hash part1sha, part2sha;
776 part1sha.UpdateData(part1, 64);
777 part1sha.Finalize();
778 part2sha.UpdateData(part2, 64);
779 part2sha.Finalize();
780
781 uint8 sessionKey[SHA256_DIGEST_LENGTH * 2];
782 for (int i = 0; i < SHA256_DIGEST_LENGTH; ++i)
783 {
784 sessionKey[i * 2] = part1sha.GetDigest()[i];
785 sessionKey[i * 2 + 1] = part2sha.GetDigest()[i];
786 }
787
788 K.SetBinary(sessionKey, SHA256_DIGEST_LENGTH * 2);
789
790 BigNumber M1;
791
792 uint8 hash[SHA256_DIGEST_LENGTH];
793 sha.Initialize();
794 sha.UpdateBigNumbers(&N, NULL);
795 sha.Finalize();
796 memcpy(hash, sha.GetDigest(), sha.GetLength());
797
798 sha.Initialize();
799 sha.UpdateBigNumbers(&g, NULL);
800 sha.Finalize();
801
802 for (int i = 0; i < sha.GetLength(); ++i)
803 hash[i] ^= sha.GetDigest()[i];
804
805 SHA256Hash shaI;
806 shaI.UpdateData(ByteArrayToHexStr(I.AsByteArray().get(), 32));
807 shaI.Finalize();
808
809 // Concat all variables for M1 hash
810 sha.Initialize();
811 sha.UpdateData(hash, SHA256_DIGEST_LENGTH);
812 sha.UpdateData(shaI.GetDigest(), shaI.GetLength());
813 sha.UpdateBigNumbers(&s, &A, &B, &K, NULL);
814 sha.Finalize();
815
816 M1.SetBinary(sha.GetDigest(), sha.GetLength());
817
818 if (memcmp(M1.AsByteArray().get(), clientM1.AsByteArray().get(), 32))
819 {
820 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_BNET_FAILED_LOGINS);
821 stmt->setString(0, _accountInfo->Login);
822 LoginDatabase.Execute(stmt);
823
824 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
825 logonResponse->SetAuthResult(AUTH_UNKNOWN_ACCOUNT);
826 ReplaceResponse(response, logonResponse);
827 TC_LOG_DEBUG("session", "[Battlenet::Password] %s attempted to log in with invalid password!", GetClientInfo().c_str());
828 return false;
829 }
830
831 if (_gameAccounts.empty())
832 {
833 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
834 logonResponse->SetAuthResult(LOGIN_NO_GAME_ACCOUNT);
835 ReplaceResponse(response, logonResponse);
836 TC_LOG_DEBUG("session", "[Battlenet::Password] %s does not have any linked game accounts!", GetClientInfo().c_str());
837 return false;
838 }
839
840 BigNumber M;
841 sha.Initialize();
842 sha.UpdateBigNumbers(&A, &M1, &K, NULL);
843 sha.Finalize();
844 M.SetBinary(sha.GetDigest(), sha.GetLength());
845
846 BigNumber serverProof;
847 serverProof.SetRand(128 * 8); // just send garbage, server signature check is patched out in client
848
849 BitStream stream;
850 ModuleInfo* password = sModuleMgr->CreateModule(_os, "Password");
851 uint8 state = 3;
852
853 stream.WriteBytes(&state, 1);
854 stream.WriteBytes(M.AsByteArray(32).get(), 32);
855 stream.WriteBytes(serverProof.AsByteArray(128).get(), 128);
856
857 password->DataSize = stream.GetSize();
858 password->Data = new uint8[password->DataSize];
859 memcpy(password->Data, stream.GetBuffer(), password->DataSize);
860
861 Authentication::ProofRequest* proofRequest = new Authentication::ProofRequest();
862 proofRequest->Modules.push_back(password);
863 if (_gameAccounts.size() > 1)
864 {
865 BitStream accounts;
866 state = 0;
867 accounts.WriteBytes(&state, 1);
868 accounts.Write(_gameAccounts.size(), 8);
869 for (GameAccountInfo const& gameAccount : _gameAccounts)
870 {
871 accounts.Write(2, 8);
872 accounts.WriteString(gameAccount.DisplayName, 8);
873 }
874
875 ModuleInfo* selectGameAccount = sModuleMgr->CreateModule(_os, "SelectGameAccount");
876 selectGameAccount->DataSize = accounts.GetSize();
877 selectGameAccount->Data = new uint8[selectGameAccount->DataSize];
878 memcpy(selectGameAccount->Data, accounts.GetBuffer(), selectGameAccount->DataSize);
879 proofRequest->Modules.push_back(selectGameAccount);
880 _modulesWaitingForData.push(MODULE_SELECT_GAME_ACCOUNT);
881 }
882 else
883 {
884 _gameAccountInfo = &_gameAccounts[0];
885
886 if (_gameAccountInfo->IsBanned)
887 {
888 delete proofRequest;
889
890 Authentication::LogonResponse* logonResponse = new Authentication::LogonResponse();
891 if (_gameAccountInfo->IsPermanenetlyBanned)
892 {
893 logonResponse->SetAuthResult(LOGIN_BANNED);
894 TC_LOG_DEBUG("session", "'%s:%d' [Battlenet::Password] Banned account %s tried to login!", GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), _accountInfo->Login.c_str());
895 }
896 else
897 {
898 logonResponse->SetAuthResult(LOGIN_SUSPENDED);
899 TC_LOG_DEBUG("session", "'%s:%d' [Battlenet::Password] Temporarily banned account %s tried to login!", GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), _accountInfo->Login.c_str());
900 }
901
902 ReplaceResponse(response, logonResponse);
903 return false;
904 }
905
906 proofRequest->Modules.push_back(sModuleMgr->CreateModule(_os, "RiskFingerprint"));
907 _modulesWaitingForData.push(MODULE_RISK_FINGERPRINT);
908 }
909
910 ReplaceResponse(response, proofRequest);
911 return true;
912}

1015bool Battlenet::Session::HandleResumeModule(BitStream* dataStream, ServerPacket** response)
1016{
1017 if (dataStream->Read<uint8>(8) != 1)
1018 {
1019 Authentication::ResumeResponse* resumeResponse = new Authentication::ResumeResponse();
1020 resumeResponse->SetAuthResult(AUTH_CORRUPTED_MODULE);
1021 ReplaceResponse(response, resumeResponse);
1022 return false;
1023 }
1024
1025 static uint8 const ResumeClient = 0;
1026 static uint8 const ResumeServer = 1;
1027
1028 std::unique_ptr<uint8[]> clientChallenge = dataStream->ReadBytes(16);
1029 std::unique_ptr<uint8[]> clientProof = dataStream->ReadBytes(32);
1030 std::unique_ptr<uint8[]> serverChallenge = _reconnectProof.AsByteArray(16);
1031 std::unique_ptr<uint8[]> sessionKey = K.AsByteArray(64);
1032
1033 HmacSha256 clientPart(64, sessionKey.get());
1034 clientPart.UpdateData(&ResumeClient, 1);
1035 clientPart.UpdateData(clientChallenge.get(), 16);
1036 clientPart.UpdateData(serverChallenge.get(), 16);
1037 clientPart.Finalize();
1038
1039 HmacSha256 serverPart(64, sessionKey.get());
1040 serverPart.UpdateData(&ResumeServer, 1);
1041 serverPart.UpdateData(serverChallenge.get(), 16);
1042 serverPart.UpdateData(clientChallenge.get(), 16);
1043 serverPart.Finalize();
1044
1045 uint8 newSessionKey[64];
1046 memcpy(&newSessionKey[0], clientPart.GetDigest(), clientPart.GetLength());
1047 memcpy(&newSessionKey[32], serverPart.GetDigest(), serverPart.GetLength());
1048
1049 K.SetBinary(newSessionKey, 64);
1050
1051 HmacSha256 proof(64, newSessionKey);
1052 proof.UpdateData(&ResumeClient, 1);
1053 proof.UpdateData(clientChallenge.get(), 16);
1054 proof.UpdateData(serverChallenge.get(), 16);
1055 proof.Finalize();
1056
1057 if (memcmp(proof.GetDigest(), clientProof.get(), serverPart.GetLength()))
1058 {
1059 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_BNET_FAILED_LOGINS);
1060 stmt->setString(0, _accountInfo->Login);
1061 LoginDatabase.Execute(stmt);
1062
1063 TC_LOG_DEBUG("session", "[Battlenet::Resume] %s attempted to reconnect with invalid password!", GetClientInfo().c_str());
1064 Authentication::ResumeResponse* resumeResponse = new Authentication::ResumeResponse();
1065 resumeResponse->SetAuthResult(AUTH_UNKNOWN_ACCOUNT);
1066 ReplaceResponse(response, resumeResponse);
1067 return false;
1068 }
1069
1070 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_BNET_SESSION_KEY);
1071 stmt->setString(0, K.AsHexStr());
1072 stmt->setBool(1, true);
1073 stmt->setUInt32(2, _accountInfo->Id);
1074 LoginDatabase.Execute(stmt);
1075
1076 HmacSha256 serverProof(64, newSessionKey);
1077 serverProof.UpdateData(&ResumeServer, 1);
1078 serverProof.UpdateData(serverChallenge.get(), 16);
1079 serverProof.UpdateData(clientChallenge.get(), 16);
1080 serverProof.Finalize();
1081
1082 ModuleInfo* resume = sModuleMgr->CreateModule(_os, "Resume");
1083
1084 BitStream resumeData;
1085 uint8 state = 2;
1086 resumeData.WriteBytes(&state, 1);
1087 resumeData.WriteBytes(serverProof.GetDigest(), serverProof.GetLength());
1088
1089 resume->DataSize = resumeData.GetSize();
1090 resume->Data = new uint8[resume->DataSize];
1091 memcpy(resume->Data, resumeData.GetBuffer(), resume->DataSize);
1092
1093 Authentication::ResumeResponse* resumeResponse = new Authentication::ResumeResponse();
1094 resumeResponse->Result.Success.FinalRequest.push_back(resume);
1095 ReplaceResponse(response, resumeResponse);
1096 _authed = true;
1097 sSessionMgr.AddSession(this);
1098 return true;
1099}

24}

26void Battlenet::PacketCrypt::Init(BigNumber* K)
27{
28 uint8 ServerEncryptionKey[SEED_KEY_SIZE] = { 0x68, 0xE0, 0xC7, 0x2E, 0xDD, 0xD6, 0xD2, 0xF3, 0x1E, 0x5A, 0xB1, 0x55, 0xB1, 0x8B, 0x63, 0x1E };
29 uint8 ClientDecryptionKey[SEED_KEY_SIZE] = { 0xDE, 0xA9, 0x65, 0xAE, 0x54, 0x3A, 0x1E, 0x93, 0x9E, 0x69, 0x0C, 0xAA, 0x68, 0xDE, 0x78, 0x39 };
30
31 HmacSha256 serverEncryptHmac(K->GetNumBytes(), K->AsByteArray().get());
32 serverEncryptHmac.UpdateData(ServerEncryptionKey, SEED_KEY_SIZE);
33 serverEncryptHmac.Finalize();
34
35 HmacSha256 clientDecryptHmac(K->GetNumBytes(), K->AsByteArray().get());
36 clientDecryptHmac.UpdateData(ClientDecryptionKey, SEED_KEY_SIZE);
37 clientDecryptHmac.Finalize();
38
39 _clientDecrypt.Init(clientDecryptHmac.GetDigest());
40 _serverEncrypt.Init(serverEncryptHmac.GetDigest());
41 _initialized = true;
42}

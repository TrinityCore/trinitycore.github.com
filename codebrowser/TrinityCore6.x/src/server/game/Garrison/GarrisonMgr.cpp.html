27void GarrisonMgr::Initialize()
28{
29 for (GarrSiteLevelPlotInstEntry const* plotInstance : sGarrSiteLevelPlotInstStore)
30 _garrisonPlotInstBySiteLevel[plotInstance->GarrSiteLevelID].push_back(plotInstance);
31
32 for (GameObjectsEntry const* gameObject : sGameObjectsStore)
33 if (gameObject->Type == GAMEOBJECT_TYPE_GARRISON_PLOT)
34 _garrisonPlots[gameObject->MapID][gameObject->Data[0]] = gameObject;
35
36 for (GarrPlotBuildingEntry const* plotBuilding : sGarrPlotBuildingStore)
37 _garrisonBuildingsByPlot[plotBuilding->GarrPlotID].insert(plotBuilding->GarrBuildingID);
38
39 for (GarrBuildingPlotInstEntry const* buildingPlotInst : sGarrBuildingPlotInstStore)
40 _garrisonBuildingPlotInstances[MAKE_PAIR64(buildingPlotInst->GarrBuildingID, buildingPlotInst->GarrSiteLevelPlotInstID)] = buildingPlotInst->ID;
41
42 for (GarrBuildingEntry const* building : sGarrBuildingStore)
43 _garrisonBuildingsByType[building->Type].push_back(building);
44
45 for (GarrFollowerXAbilityEntry const* followerAbility : sGarrFollowerXAbilityStore)
46 {
47 if (GarrAbilityEntry const* ability = sGarrAbilityStore.LookupEntry(followerAbility->GarrAbilityID))
48 {
49 if (ability->FollowerTypeID != FOLLOWER_TYPE_GARRISON)
50 continue;
51
52 if (!(ability->Flags & GARRISON_ABILITY_CANNOT_ROLL) && ability->Flags & GARRISON_ABILITY_FLAG_TRAIT)
53 _garrisonFollowerRandomTraits.insert(ability);
54
55 if (followerAbility->FactionIndex < 2)
56 {
57 if (ability->Flags & GARRISON_ABILITY_FLAG_TRAIT)
58 _garrisonFollowerAbilities[followerAbility->FactionIndex][followerAbility->GarrFollowerID].Traits.insert(ability);
59 else
60 _garrisonFollowerAbilities[followerAbility->FactionIndex][followerAbility->GarrFollowerID].Counters.insert(ability);
61 }
62 }
63 }
64
65 InitializeDbIdSequences();
66 LoadPlotFinalizeGOInfo();
67 LoadFollowerClassSpecAbilities();
68}

161std::list<GarrAbilityEntry const*> GarrisonMgr::RollFollowerAbilities(GarrFollowerEntry const* follower, uint32 quality, uint32 faction, bool initial) const
162{
163 ASSERT(faction < 2);
164
165 bool hasForcedExclusiveTrait = false;
166 std::list<GarrAbilityEntry const*> result;
167 uint32 slots[2] = { AbilitiesForQuality[quality][0], AbilitiesForQuality[quality][1] };
168
169 GarrAbilities const* abilities = nullptr;
170 auto itr = _garrisonFollowerAbilities[faction].find(follower->ID);
171 if (itr != _garrisonFollowerAbilities[faction].end())
172 abilities = &itr->second;
173
174 std::list<GarrAbilityEntry const*> abilityList, forcedAbilities, traitList, forcedTraits;
175 if (abilities)
176 {
177 for (GarrAbilityEntry const* ability : abilities->Counters)
178 {
179 if (ability->Flags & GARRISON_ABILITY_HORDE_ONLY && faction != GARRISON_FACTION_INDEX_HORDE)
180 continue;
181 else if (ability->Flags & GARRISON_ABILITY_ALLIANCE_ONLY && faction != GARRISON_FACTION_INDEX_ALLIANCE)
182 continue;
183
184 if (ability->Flags & GARRISON_ABILITY_FLAG_CANNOT_REMOVE)
185 forcedAbilities.push_back(ability);
186 else
187 abilityList.push_back(ability);
188 }
189
190 for (GarrAbilityEntry const* ability : abilities->Traits)
191 {
192 if (ability->Flags & GARRISON_ABILITY_HORDE_ONLY && faction != GARRISON_FACTION_INDEX_HORDE)
193 continue;
194 else if (ability->Flags & GARRISON_ABILITY_ALLIANCE_ONLY && faction != GARRISON_FACTION_INDEX_ALLIANCE)
195 continue;
196
197 if (ability->Flags & GARRISON_ABILITY_FLAG_CANNOT_REMOVE)
198 forcedTraits.push_back(ability);
199 else
200 traitList.push_back(ability);
201 }
202 }
203
204 Trinity::Containers::RandomResizeList(abilityList, std::max<int32>(0, slots[0] - forcedAbilities.size()));
205 Trinity::Containers::RandomResizeList(traitList, std::max<int32>(0, slots[1] - forcedTraits.size()));
206
207 // Add abilities specified in GarrFollowerXAbility.db2 before generic classspec ones on follower creation
208 if (initial)
209 {
210 forcedAbilities.splice(forcedAbilities.end(), abilityList);
211 forcedTraits.splice(forcedTraits.end(), traitList);
212 }
213
214 forcedAbilities.sort();
215 abilityList.sort();
216 forcedTraits.sort();
217 traitList.sort();
218
219 // check if we have a trait from exclusive category
220 for (GarrAbilityEntry const* ability : forcedTraits)
221 {
222 if (ability->Flags & GARRISON_ABILITY_FLAG_EXCLUSIVE)
223 {
224 hasForcedExclusiveTrait = true;
225 break;
226 }
227 }
228
229 if (slots[0] > forcedAbilities.size() + abilityList.size())
230 {
231 std::list<GarrAbilityEntry const*> classSpecAbilities = GetClassSpecAbilities(follower, faction);
232 std::list<GarrAbilityEntry const*> classSpecAbilitiesTemp, classSpecAbilitiesTemp2;
233 classSpecAbilitiesTemp2.swap(abilityList);
234 std::set_difference(classSpecAbilities.begin(), classSpecAbilities.end(), forcedAbilities.begin(), forcedAbilities.end(), std::back_inserter(classSpecAbilitiesTemp));
235 std::set_union(classSpecAbilitiesTemp.begin(), classSpecAbilitiesTemp.end(), classSpecAbilitiesTemp2.begin(), classSpecAbilitiesTemp2.end(), std::back_inserter(abilityList));
236
237 Trinity::Containers::RandomResizeList(abilityList, std::max<int32>(0, slots[0] - forcedAbilities.size()));
238 }
239
240 if (slots[1] > forcedTraits.size() + traitList.size())
241 {
242 std::list<GarrAbilityEntry const*> genericTraits, genericTraitsTemp;
243 for (GarrAbilityEntry const* ability : _garrisonFollowerRandomTraits)
244 {
245 if (ability->Flags & GARRISON_ABILITY_HORDE_ONLY && faction != GARRISON_FACTION_INDEX_HORDE)
246 continue;
247 else if (ability->Flags & GARRISON_ABILITY_ALLIANCE_ONLY && faction != GARRISON_FACTION_INDEX_ALLIANCE)
248 continue;
249
250 // forced exclusive trait exists, skip other ones entirely
251 if (hasForcedExclusiveTrait && ability->Flags & GARRISON_ABILITY_FLAG_EXCLUSIVE)
252 continue;
253
254 genericTraitsTemp.push_back(ability);
255 }
256
257 std::set_difference(genericTraitsTemp.begin(), genericTraitsTemp.end(), forcedTraits.begin(), forcedTraits.end(), std::back_inserter(genericTraits));
258 genericTraits.splice(genericTraits.begin(), traitList);
259 // "split" the list into two parts [nonexclusive, exclusive] to make selection later easier
260 genericTraits.sort([](GarrAbilityEntry const* a1, GarrAbilityEntry const* a2)
261 {
262 uint32 e1 = a1->Flags & GARRISON_ABILITY_FLAG_EXCLUSIVE;
263 uint32 e2 = a2->Flags & GARRISON_ABILITY_FLAG_EXCLUSIVE;
264 if (e1 != e2)
265 return e1 < e2;
266
267 return a1->ID < a2->ID;
268 });
269 genericTraits.unique();
270
271 std::size_t firstExclusive = 0, total = genericTraits.size();
272 for (auto genericTraitItr = genericTraits.begin(); genericTraitItr != genericTraits.end(); ++genericTraitItr, ++firstExclusive)
273 if ((*genericTraitItr)->Flags & GARRISON_ABILITY_FLAG_EXCLUSIVE)
274 break;
275
276 while (traitList.size() < size_t(std::max<int32>(0, slots[1] - forcedTraits.size())) && total)
277 {
278 auto genericTraitItr = genericTraits.begin();
279 std::advance(genericTraitItr, urand(0, total-- - 1));
280 if ((*genericTraitItr)->Flags & GARRISON_ABILITY_FLAG_EXCLUSIVE)
281 total = firstExclusive; // selected exclusive trait - no other can be selected now
282 else
283 --firstExclusive;
284
285 traitList.push_back(*genericTraitItr);
286 genericTraits.erase(genericTraitItr);
287 }
288 }
289
290 result.splice(result.end(), forcedAbilities);
291 result.splice(result.end(), abilityList);
292 result.splice(result.end(), forcedTraits);
293 result.splice(result.end(), traitList);
294
295 return result;
296}

298std::list<GarrAbilityEntry const*> GarrisonMgr::GetClassSpecAbilities(GarrFollowerEntry const* follower, uint32 faction) const
299{
300 std::list<GarrAbilityEntry const*> abilities;
301 uint32 classSpecId;
302 switch (faction)
303 {
304 case GARRISON_FACTION_INDEX_HORDE:
305 classSpecId = follower->HordeGarrClassSpecID;
306 break;
307 case GARRISON_FACTION_INDEX_ALLIANCE:
308 classSpecId = follower->AllianceGarrClassSpecID;
309 break;
310 default:
311 return abilities;
312 }
313
314 if (!sGarrClassSpecStore.LookupEntry(classSpecId))
315 return abilities;
316
317 auto itr = _garrisonFollowerClassSpecAbilities.find(classSpecId);
318 if (itr != _garrisonFollowerClassSpecAbilities.end())
319 abilities = itr->second;
320
321 return abilities;
322}

330void GarrisonMgr::LoadPlotFinalizeGOInfo()
331{
332 // 0 1 2 3 4 5 6
333 QueryResult result = WorldDatabase.Query("SELECT garrPlotInstanceId, hordeGameObjectId, hordeX, hordeY, hordeZ, hordeO, hordeAnimKitId, "
334 // 7 8 9 10 11 12
335 "allianceGameObjectId, allianceX, allianceY, allianceZ, allianceO, allianceAnimKitId FROM garrison_plot_finalize_info");
336 if (!result)
337 {
338 TC_LOG_INFO("server.loading", ">> Loaded 0 garrison follower class spec abilities. DB table `garrison_plot_finalize_info` is empty.");
339 return;
340 }
341
342 uint32 msTime = getMSTime();
343 do
344 {
345 Field* fields = result->Fetch();
346 uint32 garrPlotInstanceId = fields[0].GetUInt32();
347 uint32 hordeGameObjectId = fields[1].GetUInt32();
348 uint32 allianceGameObjectId = fields[7].GetUInt32();
349 uint16 hordeAnimKitId = fields[6].GetUInt16();
350 uint16 allianceAnimKitId = fields[12].GetUInt16();
351
352 if (!sGarrPlotInstanceStore.LookupEntry(garrPlotInstanceId))
353 {
354 TC_LOG_ERROR("sql.sql", "Non-existing GarrPlotInstance.db2 entry %u was referenced in `garrison_plot_finalize_info`.", garrPlotInstanceId);
355 continue;
356 }
357
358 GameObjectTemplate const* goTemplate = sObjectMgr->GetGameObjectTemplate(hordeGameObjectId);
359 if (!goTemplate)
360 {
361 TC_LOG_ERROR("sql.sql", "Non-existing gameobject_template entry %u was referenced in `garrison_plot_finalize_info`.`hordeGameObjectId` for garrPlotInstanceId %u.",
362 hordeGameObjectId, garrPlotInstanceId);
363 continue;
364 }
365
366 if (goTemplate->type != GAMEOBJECT_TYPE_GOOBER)
367 {
368 TC_LOG_ERROR("sql.sql", "Invalid gameobject type %u (entry %u) was referenced in `garrison_plot_finalize_info`.`hordeGameObjectId` for garrPlotInstanceId %u.",
369 goTemplate->type, hordeGameObjectId, garrPlotInstanceId);
370 continue;
371 }
372
373 goTemplate = sObjectMgr->GetGameObjectTemplate(allianceGameObjectId);
374 if (!goTemplate)
375 {
376 TC_LOG_ERROR("sql.sql", "Non-existing gameobject_template entry %u was referenced in `garrison_plot_finalize_info`.`allianceGameObjectId` for garrPlotInstanceId %u.",
377 allianceGameObjectId, garrPlotInstanceId);
378 continue;
379 }
380
381 if (goTemplate->type != GAMEOBJECT_TYPE_GOOBER)
382 {
383 TC_LOG_ERROR("sql.sql", "Invalid gameobject type %u (entry %u) was referenced in `garrison_plot_finalize_info`.`allianceGameObjectId` for garrPlotInstanceId %u.",
384 goTemplate->type, allianceGameObjectId, garrPlotInstanceId);
385 continue;
386 }
387
388 if (hordeAnimKitId && !sAnimKitStore.LookupEntry(hordeAnimKitId))
389 {
390 TC_LOG_ERROR("sql.sql", "Non-existing AnimKit.dbc entry %u was referenced in `garrison_plot_finalize_info`.`hordeAnimKitId` for garrPlotInstanceId %u.",
391 hordeAnimKitId, garrPlotInstanceId);
392 continue;
393 }
394
395 if (allianceAnimKitId && !sAnimKitStore.LookupEntry(allianceAnimKitId))
396 {
397 TC_LOG_ERROR("sql.sql", "Non-existing AnimKit.dbc entry %u was referenced in `garrison_plot_finalize_info`.`allianceAnimKitId` for garrPlotInstanceId %u.",
398 allianceAnimKitId, garrPlotInstanceId);
399 continue;
400 }
401
402 FinalizeGarrisonPlotGOInfo& info = _finalizePlotGOInfo[garrPlotInstanceId];
403 info.FactionInfo[GARRISON_FACTION_INDEX_HORDE].GameObjectId = hordeGameObjectId;
404 info.FactionInfo[GARRISON_FACTION_INDEX_HORDE].Pos.Relocate(fields[2].GetFloat(), fields[3].GetFloat(), fields[4].GetFloat(), fields[5].GetFloat());
405 info.FactionInfo[GARRISON_FACTION_INDEX_HORDE].AnimKitId = hordeAnimKitId;
406
407 info.FactionInfo[GARRISON_FACTION_INDEX_ALLIANCE].GameObjectId = allianceGameObjectId;
408 info.FactionInfo[GARRISON_FACTION_INDEX_ALLIANCE].Pos.Relocate(fields[8].GetFloat(), fields[9].GetFloat(), fields[10].GetFloat(), fields[11].GetFloat());
409 info.FactionInfo[GARRISON_FACTION_INDEX_ALLIANCE].AnimKitId = allianceAnimKitId;
410
411 } while (result->NextRow());
412
413 TC_LOG_INFO("server.loading", ">> Loaded %u garrison plot finalize entries in %u.", uint32(_finalizePlotGOInfo.size()), GetMSTimeDiffToNow(msTime));
414}

416void GarrisonMgr::LoadFollowerClassSpecAbilities()
417{
418 QueryResult result = WorldDatabase.Query("SELECT classSpecId, abilityId FROM garrison_follower_class_spec_abilities");
419 if (!result)
420 {
421 TC_LOG_INFO("server.loading", ">> Loaded 0 garrison follower class spec abilities. DB table `garrison_follower_class_spec_abilities` is empty.");
422 return;
423 }
424
425 uint32 msTime = getMSTime();
426 uint32 count = 0;
427 do
428 {
429 Field* fields = result->Fetch();
430 uint32 classSpecId = fields[0].GetUInt32();
431 uint32 abilityId = fields[1].GetUInt32();
432
433 if (!sGarrClassSpecStore.LookupEntry(classSpecId))
434 {
435 TC_LOG_ERROR("sql.sql", "Non-existing GarrClassSpec.db2 entry %u was referenced in `garrison_follower_class_spec_abilities` by row (%u, %u).", classSpecId, classSpecId, abilityId);
436 continue;
437 }
438
439 GarrAbilityEntry const* ability = sGarrAbilityStore.LookupEntry(abilityId);
440 if (!ability)
441 {
442 TC_LOG_ERROR("sql.sql", "Non-existing GarrAbility.db2 entry %u was referenced in `garrison_follower_class_spec_abilities` by row (%u, %u).", abilityId, classSpecId, abilityId);
443 continue;
444 }
445
446 _garrisonFollowerClassSpecAbilities[classSpecId].push_back(ability);
447 ++count;
448
449 } while (result->NextRow());
450
451 for (auto& pair : _garrisonFollowerClassSpecAbilities)
452 pair.second.sort();
453
454 TC_LOG_INFO("server.loading", ">> Loaded %u garrison follower class spec abilities in %u.", count, GetMSTimeDiffToNow(msTime));
455}

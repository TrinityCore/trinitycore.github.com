48void GarrisonGridLoader::LoadN()
49{
50 if (i_garrison)
51 {
52 i_cell.data.Part.cell_y = 0;
53 for (uint32 x = 0; x < MAX_NUMBER_OF_CELLS; ++x)
54 {
55 i_cell.data.Part.cell_x = x;
56 for (uint32 y = 0; y < MAX_NUMBER_OF_CELLS; ++y)
57 {
58 i_cell.data.Part.cell_y = y;
59
60 //Load creatures and game objects
61 TypeContainerVisitor<GarrisonGridLoader, GridTypeMapContainer> visitor(*this);
62 i_grid->VisitGrid(x, y, visitor);
63 }
64 }
65 }
66
67 TC_LOG_DEBUG("maps", "%u GameObjects and %u Creatures loaded for grid %u on map %u", i_gameObjects, i_creatures, i_grid->GetGridId(), i_map->GetId());
68}

70void GarrisonGridLoader::Visit(GameObjectMapType& m)
71{
72 std::vector<Garrison::Plot*> plots = i_garrison->GetPlots();
73 if (!plots.empty())
74 {
75 CellCoord cellCoord = i_cell.GetCellCoord();
76 for (Garrison::Plot* plot : plots)
77 {
78 Position const& spawn = plot->PacketInfo.PlotPos;
79 if (cellCoord != Trinity::ComputeCellCoord(spawn.GetPositionX(), spawn.GetPositionY()))
80 continue;
81
82 GameObject* go = plot->CreateGameObject(i_map, i_garrison->GetFaction());
83 if (!go)
84 continue;
85
86 go->AddToGrid(m);
87 ObjectGridLoader::SetObjectCell(go, cellCoord);
88 go->AddToWorld();
89 ++i_gameObjects;
90 }
91 }
92}

105void GarrisonMap::LoadGridObjects(NGridType* grid, Cell const& cell)
106{
107 Map::LoadGridObjects(grid, cell);
108
109 GarrisonGridLoader loader(grid, this, cell);
110 loader.LoadN();
111}

113Garrison* GarrisonMap::GetGarrison()
114{
115 if (_loadingPlayer)
116 return _loadingPlayer->GetGarrison();
117
118 if (Player* owner = ObjectAccessor::FindConnectedPlayer(_owner))
119 return owner->GetGarrison();
120
121 return nullptr;
122}

131bool GarrisonMap::AddPlayerToMap(Player* player, bool initPlayer /*= true*/)
132{
133 if (player->GetGUID() == _owner)
134 _loadingPlayer = player;
135
136 bool result = Map::AddPlayerToMap(player, initPlayer);
137
138 if (player->GetGUID() == _owner)
139 _loadingPlayer = nullptr;
140
141 return result;
142}

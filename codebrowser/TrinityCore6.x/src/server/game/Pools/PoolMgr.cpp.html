130// Method to add a gameobject/creature guid to the proper list depending on pool type and chance value
131template <class T>
132void PoolGroup<T>::AddEntry(PoolObject& poolitem, uint32 maxentries)
133{
134 if (poolitem.chance != 0 && maxentries == 1)
135 ExplicitlyChanced.push_back(poolitem);
136 else
137 EqualChanced.push_back(poolitem);
138}

140// Method to check the chances are proper in this object pool
141template <class T>
142bool PoolGroup<T>::CheckPool() const
143{
144 if (EqualChanced.empty())
145 {
146 float chance = 0;
147 for (uint32 i = 0; i < ExplicitlyChanced.size(); ++i)
148 chance += ExplicitlyChanced[i].chance;
149 if (chance != 100 && chance != 0)
150 return false;
151 }
152 return true;
153}

155template <class T>
156PoolObject* PoolGroup<T>::RollOne(ActivePoolData& spawns, uint64 triggerFrom)
157{
158 if (!ExplicitlyChanced.empty())
159 {
160 float roll = (float)rand_chance();
161
162 for (uint32 i = 0; i < ExplicitlyChanced.size(); ++i)
163 {
164 roll -= ExplicitlyChanced[i].chance;
165 // Triggering object is marked as spawned at this time and can be also rolled (respawn case)
166 // so this need explicit check for this case
167 if (roll < 0 && (ExplicitlyChanced[i].guid == triggerFrom || !spawns.IsActiveObject<T>(ExplicitlyChanced[i].guid)))
168 return &ExplicitlyChanced[i];
169 }
170 }
171 if (!EqualChanced.empty())
172 {
173 uint32 index = urand(0, EqualChanced.size()-1);
174 // Triggering object is marked as spawned at this time and can be also rolled (respawn case)
175 // so this need explicit check for this case
176 if (EqualChanced[index].guid == triggerFrom || !spawns.IsActiveObject<T>(EqualChanced[index].guid))
177 return &EqualChanced[index];
178 }
179
180 return NULL;
181}

183// Main method to despawn a creature or gameobject in a pool
184// If no guid is passed, the pool is just removed (event end case)
185// If guid is filled, cache will be used and no removal will occur, it just fill the cache
186template<class T>
187void PoolGroup<T>::DespawnObject(ActivePoolData& spawns, uint64 guid)
188{
189 for (size_t i=0; i < EqualChanced.size(); ++i)
190 {
191 // if spawned
192 if (spawns.IsActiveObject<T>(EqualChanced[i].guid))
193 {
194 if (!guid || EqualChanced[i].guid == guid)
195 {
196 Despawn1Object(EqualChanced[i].guid);
197 spawns.RemoveObject<T>(EqualChanced[i].guid, poolId);
198 }
199 }
200 }
201
202 for (size_t i = 0; i < ExplicitlyChanced.size(); ++i)
203 {
204 // spawned
205 if (spawns.IsActiveObject<T>(ExplicitlyChanced[i].guid))
206 {
207 if (!guid || ExplicitlyChanced[i].guid == guid)
208 {
209 Despawn1Object(ExplicitlyChanced[i].guid);
210 spawns.RemoveObject<T>(ExplicitlyChanced[i].guid, poolId);
211 }
212 }
213 }
214}

217template<>
218void PoolGroup<Creature>::Despawn1Object(uint64 guid)
219{
220 if (CreatureData const* data = sObjectMgr->GetCreatureData(guid))
221 {
222 sObjectMgr->RemoveCreatureFromGrid(guid, data);
223
224 Map* map = sMapMgr->CreateBaseMap(data->mapid);
225 if (!map->Instanceable())
226 {
227 auto creatureBounds = map->GetCreatureBySpawnIdStore().equal_range(guid);
228 for (auto itr = creatureBounds.first; itr != creatureBounds.second;)
229 {
230 Creature* creature = itr->second;
231 ++itr;
232 creature->AddObjectToRemoveList();
233 }
234 }
235 }
236}

267// Same on one quest
268template<>
269void PoolGroup<Quest>::Despawn1Object(uint64 quest_id)
270{
271 // Creatures
272 QuestRelations* questMap = sObjectMgr->GetCreatureQuestRelationMap();
273 PooledQuestRelationBoundsNC qr = sPoolMgr->mQuestCreatureRelation.equal_range(quest_id);
274 for (PooledQuestRelation::iterator itr = qr.first; itr != qr.second; ++itr)
275 {
276 QuestRelations::iterator qitr = questMap->find(itr->second);
277 if (qitr == questMap->end())
278 continue;
279 QuestRelations::iterator lastElement = questMap->upper_bound(itr->second);
280 for (; qitr != lastElement; ++qitr)
281 {
282 if (qitr->first == itr->second && qitr->second == itr->first)
283 {
284 questMap->erase(qitr); // iterator is now no more valid
285 break; // but we can exit loop since the element is found
286 }
287 }
288 }
289
290 // Gameobjects
291 questMap = sObjectMgr->GetGOQuestRelationMap();
292 qr = sPoolMgr->mQuestGORelation.equal_range(quest_id);
293 for (PooledQuestRelation::iterator itr = qr.first; itr != qr.second; ++itr)
294 {
295 QuestRelations::iterator qitr = questMap->find(itr->second);
296 if (qitr == questMap->end())
297 continue;
298 QuestRelations::iterator lastElement = questMap->upper_bound(itr->second);
299 for (; qitr != lastElement; ++qitr)
300 {
301 if (qitr->first == itr->second && qitr->second == itr->first)
302 {
303 questMap->erase(qitr); // iterator is now no more valid
304 break; // but we can exit loop since the element is found
305 }
306 }
307 }
308}

310// Method for a pool only to remove any found record causing a circular dependency loop
311template<>
312void PoolGroup<Pool>::RemoveOneRelation(uint32 child_pool_id)
313{
314 for (PoolObjectList::iterator itr = ExplicitlyChanced.begin(); itr != ExplicitlyChanced.end(); ++itr)
315 {
316 if (itr->guid == child_pool_id)
317 {
318 ExplicitlyChanced.erase(itr);
319 break;
320 }
321 }
322 for (PoolObjectList::iterator itr = EqualChanced.begin(); itr != EqualChanced.end(); ++itr)
323 {
324 if (itr->guid == child_pool_id)
325 {
326 EqualChanced.erase(itr);
327 break;
328 }
329 }

332template <class T>
333void PoolGroup<T>::SpawnObject(ActivePoolData& spawns, uint32 limit, uint64 triggerFrom)
334{
335 uint32 lastDespawned = 0;
336 int count = limit - spawns.GetActiveObjectCount(poolId);
337
338 // If triggered from some object respawn this object is still marked as spawned
339 // and also counted into m_SpawnedPoolAmount so we need increase count to be
340 // spawned by 1
341 if (triggerFrom)
342 ++count;
343
344 // This will try to spawn the rest of pool, not guaranteed
345 for (int i = 0; i < count; ++i)
346 {
347 PoolObject* obj = RollOne(spawns, triggerFrom);
348 if (!obj)
349 continue;
350 if (obj->guid == lastDespawned)
351 continue;
352
353 if (obj->guid == triggerFrom)
354 {
355 ReSpawn1Object(obj);
356 triggerFrom = 0;
357 continue;
358 }
359 spawns.ActivateObject<T>(obj->guid, poolId);
360 Spawn1Object(obj);
361
362 if (triggerFrom)
363 {
364 // One spawn one despawn no count increase
365 DespawnObject(spawns, triggerFrom);
366 lastDespawned = triggerFrom;
367 triggerFrom = 0;
368 }
369 }

372// Method that is actualy doing the spawn job on 1 creature
373template <>
374void PoolGroup<Creature>::Spawn1Object(PoolObject* obj)
375{
376 if (CreatureData const* data = sObjectMgr->GetCreatureData(obj->guid))
377 {
378 sObjectMgr->AddCreatureToGrid(obj->guid, data);
379
380 // Spawn if necessary (loaded grids only)
381 Map* map = sMapMgr->CreateBaseMap(data->mapid);
382 // We use spawn coords to spawn
383 if (!map->Instanceable() && map->IsGridLoaded(data->posX, data->posY))
384 {
385 Creature* creature = new Creature();
386 //TC_LOG_DEBUG("pool", "Spawning creature %u", guid);
387 if (!creature->LoadCreatureFromDB(obj->guid, map))
388 {
389 delete creature;
390 return;
391 }
392 }
393 }
394}

396// Same for 1 gameobject
397template <>
398void PoolGroup<GameObject>::Spawn1Object(PoolObject* obj)
399{
400 if (GameObjectData const* data = sObjectMgr->GetGOData(obj->guid))
401 {
402 sObjectMgr->AddGameobjectToGrid(obj->guid, data);
403 // Spawn if necessary (loaded grids only)
404 // this base map checked as non-instanced and then only existed
405 Map* map = sMapMgr->CreateBaseMap(data->mapid);
406 // We use current coords to unspawn, not spawn coords since creature can have changed grid
407 if (!map->Instanceable() && map->IsGridLoaded(data->posX, data->posY))
408 {
409 GameObject* pGameobject = new GameObject;
410 //TC_LOG_DEBUG("pool", "Spawning gameobject %u", guid);
411 if (!pGameobject->LoadGameObjectFromDB(obj->guid, map, false))
412 {
413 delete pGameobject;
414 return;
415 }
416 else
417 {
418 if (pGameobject->isSpawnedByDefault())
419 map->AddToMap(pGameobject);
420 }
421 }
422 }

455template <>
456void PoolGroup<Quest>::SpawnObject(ActivePoolData& spawns, uint32 limit, uint64 triggerFrom)
457{
458 TC_LOG_DEBUG("pool", "PoolGroup<Quest>: Spawning pool %u", poolId);
459 // load state from db
460 if (!triggerFrom)
461 {
462 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_POOL_QUEST_SAVE);
463
464 stmt->setUInt32(0, poolId);
465
466 PreparedQueryResult result = CharacterDatabase.Query(stmt);
467
468 if (result)
469 {
470 do
471 {
472 uint32 questId = result->Fetch()[0].GetUInt32();
473 spawns.ActivateObject<Quest>(questId, poolId);
474 PoolObject tempObj(questId, 0.0f);
475 Spawn1Object(&tempObj);
476 --limit;
477 } while (result->NextRow() && limit);
478 return;
479 }
480 }
481
482 ActivePoolObjects currentQuests = spawns.GetActiveQuests();
483 ActivePoolObjects newQuests;
484
485 // always try to select different quests
486 for (PoolObjectList::iterator itr = EqualChanced.begin(); itr != EqualChanced.end(); ++itr)
487 {
488 if (spawns.IsActiveObject<Quest>(itr->guid))
489 continue;
490 newQuests.insert(itr->guid);
491 }
492
493 // clear the pool
494 DespawnObject(spawns);
495
496 // recycle minimal amount of quests if possible count is lower than limit
497 if (limit > newQuests.size() && !currentQuests.empty())
498 {
499 do
500 {
501 uint32 questId = Trinity::Containers::SelectRandomContainerElement(currentQuests);
502 newQuests.insert(questId);
503 currentQuests.erase(questId);
504 } while (newQuests.size() < limit && !currentQuests.empty()); // failsafe
505 }
506
507 if (newQuests.empty())
508 return;
509
510 // activate <limit> random quests
511 do
512 {
513 uint32 questId = Trinity::Containers::SelectRandomContainerElement(newQuests);
514 spawns.ActivateObject<Quest>(questId, poolId);
515 PoolObject tempObj(questId, 0.0f);
516 Spawn1Object(&tempObj);
517 newQuests.erase(questId);
518 --limit;
519 } while (limit && !newQuests.empty());
520
521 // if we are here it means the pool is initialized at startup and did not have previous saved state
522 if (!triggerFrom)
523 sPoolMgr->SaveQuestsToDB();
524}

553void PoolMgr::Initialize()
554{
555 QueryResult result = WorldDatabase.Query("SELECT MAX(entry) FROM pool_template");
556 if (result)
557 {
558 Field* fields = result->Fetch();
559 max_pool_id = fields[0].GetUInt32();
560 }
561
562 mPoolTemplate.resize(max_pool_id + 1);
563 mPoolCreatureGroups.resize(max_pool_id + 1);
564 mPoolGameobjectGroups.resize(max_pool_id + 1);
565 mPoolPoolGroups.resize(max_pool_id + 1);
566 mPoolQuestGroups.resize(max_pool_id + 1);
567
568 mQuestSearchMap.clear();
569 mGameobjectSearchMap.clear();
570 mCreatureSearchMap.clear();
571}

573void PoolMgr::LoadFromDB()
574{
575 // Pool templates
576 {
577 uint32 oldMSTime = getMSTime();
578
579 QueryResult result = WorldDatabase.Query("SELECT entry, max_limit FROM pool_template");
580 if (!result)
581 {
582 mPoolTemplate.clear();
583 TC_LOG_INFO("server.loading", ">> Loaded 0 object pools. DB table `pool_template` is empty.");
584 return;
585 }
586
587 uint32 count = 0;
588 do
589 {
590 Field* fields = result->Fetch();
591
592 uint32 pool_id = fields[0].GetUInt32();
593
594 PoolTemplateData& pPoolTemplate = mPoolTemplate[pool_id];
595 pPoolTemplate.MaxLimit = fields[1].GetUInt32();
596
597 ++count;
598 }
599 while (result->NextRow());
600
601 TC_LOG_INFO("server.loading", ">> Loaded %u objects pools in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
602 }
603
604 // Creatures
605
606 TC_LOG_INFO("server.loading", "Loading Creatures Pooling Data...");
607 {
608 uint32 oldMSTime = getMSTime();
609
610 // 1 2 3
611 QueryResult result = WorldDatabase.Query("SELECT guid, pool_entry, chance FROM pool_creature");
612
613 if (!result)
614 {
615 TC_LOG_INFO("server.loading", ">> Loaded 0 creatures in pools. DB table `pool_creature` is empty.");
616 }
617 else
618 {
619 uint32 count = 0;
620 do
621 {
622 Field* fields = result->Fetch();
623
624 uint64 guid = fields[0].GetUInt64();
625 uint32 pool_id = fields[1].GetUInt32();
626 float chance = fields[2].GetFloat();
627
628 CreatureData const* data = sObjectMgr->GetCreatureData(guid);
629 if (!data)
630 {
631 TC_LOG_ERROR("sql.sql", "`pool_creature` has a non existing creature spawn (GUID: " UI64FMTD ") defined for pool id (%u), skipped.", guid, pool_id);
632 continue;
633 }
634 if (pool_id > max_pool_id)
635 {
636 TC_LOG_ERROR("sql.sql", "`pool_creature` pool id (%u) is out of range compared to max pool id in `pool_template`, skipped.", pool_id);
637 continue;
638 }
639 if (chance < 0 || chance > 100)
640 {
641 TC_LOG_ERROR("sql.sql", "`pool_creature` has an invalid chance (%f) for creature guid (" UI64FMTD ") in pool id (%u), skipped.", chance, guid, pool_id);
642 continue;
643 }
644 PoolTemplateData* pPoolTemplate = &mPoolTemplate[pool_id];
645 PoolObject plObject = PoolObject(guid, chance);
646 PoolGroup<Creature>& cregroup = mPoolCreatureGroups[pool_id];
647 cregroup.SetPoolId(pool_id);
648 cregroup.AddEntry(plObject, pPoolTemplate->MaxLimit);
649 SearchPair p(guid, pool_id);
650 mCreatureSearchMap.insert(p);
651
652 ++count;
653 }
654 while (result->NextRow());
655
656 TC_LOG_INFO("server.loading", ">> Loaded %u creatures in pools in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
657 }
658 }
659
660 // Gameobjects
661
662 TC_LOG_INFO("server.loading", "Loading Gameobject Pooling Data...");
663 {
664 uint32 oldMSTime = getMSTime();
665
666 // 1 2 3
667 QueryResult result = WorldDatabase.Query("SELECT guid, pool_entry, chance FROM pool_gameobject");
668
669 if (!result)
670 {
671 TC_LOG_INFO("server.loading", ">> Loaded 0 gameobjects in pools. DB table `pool_gameobject` is empty.");
672 }
673 else
674 {
675 uint32 count = 0;
676 do
677 {
678 Field* fields = result->Fetch();
679
680 uint64 guid = fields[0].GetUInt64();
681 uint32 pool_id = fields[1].GetUInt32();
682 float chance = fields[2].GetFloat();
683
684 GameObjectData const* data = sObjectMgr->GetGOData(guid);
685 if (!data)
686 {
687 TC_LOG_ERROR("sql.sql", "`pool_gameobject` has a non existing gameobject spawn (GUID: " UI64FMTD ") defined for pool id (%u), skipped.", guid, pool_id);
688 continue;
689 }
690
691 GameObjectTemplate const* goinfo = sObjectMgr->GetGameObjectTemplate(data->id);
692 if (goinfo->type != GAMEOBJECT_TYPE_CHEST &&
693 goinfo->type != GAMEOBJECT_TYPE_GOOBER &&
694 goinfo->type != GAMEOBJECT_TYPE_FISHINGHOLE)
695 {
696 TC_LOG_ERROR("sql.sql", "`pool_gameobject` has a not lootable gameobject spawn (GUID: " UI64FMTD ", type: %u) defined for pool id (%u), skipped.", guid, goinfo->type, pool_id);
697 continue;
698 }
699
700 if (pool_id > max_pool_id)
701 {
702 TC_LOG_ERROR("sql.sql", "`pool_gameobject` pool id (%u) is out of range compared to max pool id in `pool_template`, skipped.", pool_id);
703 continue;
704 }
705
706 if (chance < 0 || chance > 100)
707 {
708 TC_LOG_ERROR("sql.sql", "`pool_gameobject` has an invalid chance (%f) for gameobject guid (" UI64FMTD ") in pool id (%u), skipped.", chance, guid, pool_id);
709 continue;
710 }
711
712 PoolTemplateData* pPoolTemplate = &mPoolTemplate[pool_id];
713 PoolObject plObject = PoolObject(guid, chance);
714 PoolGroup<GameObject>& gogroup = mPoolGameobjectGroups[pool_id];
715 gogroup.SetPoolId(pool_id);
716 gogroup.AddEntry(plObject, pPoolTemplate->MaxLimit);
717 SearchPair p(guid, pool_id);
718 mGameobjectSearchMap.insert(p);
719
720 ++count;
721 }
722 while (result->NextRow());
723
724 TC_LOG_INFO("server.loading", ">> Loaded %u gameobject in pools in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
725 }
726 }
727
728 // Pool of pools
729
730 TC_LOG_INFO("server.loading", "Loading Mother Pooling Data...");
731 {
732 uint32 oldMSTime = getMSTime();
733
734 // 1 2 3
735 QueryResult result = WorldDatabase.Query("SELECT pool_id, mother_pool, chance FROM pool_pool");
736
737 if (!result)
738 {
739 TC_LOG_INFO("server.loading", ">> Loaded 0 pools in pools");
740 }
741 else
742 {
743 uint32 count = 0;
744 do
745 {
746 Field* fields = result->Fetch();
747
748 uint32 child_pool_id = fields[0].GetUInt32();
749 uint32 mother_pool_id = fields[1].GetUInt32();
750 float chance = fields[2].GetFloat();
751
752 if (mother_pool_id > max_pool_id)
753 {
754 TC_LOG_ERROR("sql.sql", "`pool_pool` mother_pool id (%u) is out of range compared to max pool id in `pool_template`, skipped.", mother_pool_id);
755 continue;
756 }
757 if (child_pool_id > max_pool_id)
758 {
759 TC_LOG_ERROR("sql.sql", "`pool_pool` included pool_id (%u) is out of range compared to max pool id in `pool_template`, skipped.", child_pool_id);
760 continue;
761 }
762 if (mother_pool_id == child_pool_id)
763 {
764 TC_LOG_ERROR("sql.sql", "`pool_pool` pool_id (%u) includes itself, dead-lock detected, skipped.", child_pool_id);
765 continue;
766 }
767 if (chance < 0 || chance > 100)
768 {
769 TC_LOG_ERROR("sql.sql", "`pool_pool` has an invalid chance (%f) for pool id (%u) in mother pool id (%u), skipped.", chance, child_pool_id, mother_pool_id);
770 continue;
771 }
772 PoolTemplateData* pPoolTemplateMother = &mPoolTemplate[mother_pool_id];
773 PoolObject plObject = PoolObject(child_pool_id, chance);
774 PoolGroup<Pool>& plgroup = mPoolPoolGroups[mother_pool_id];
775 plgroup.SetPoolId(mother_pool_id);
776 plgroup.AddEntry(plObject, pPoolTemplateMother->MaxLimit);
777 SearchPair p(child_pool_id, mother_pool_id);
778 mPoolSearchMap.insert(p);
779
780 ++count;
781 }
782 while (result->NextRow());
783
784 // Now check for circular reference
785 for (uint32 i=0; i < max_pool_id; ++i)
786 {
787 std::set<uint32> checkedPools;
788 for (SearchMap::iterator poolItr = mPoolSearchMap.find(i); poolItr != mPoolSearchMap.end(); poolItr = mPoolSearchMap.find(poolItr->second))
789 {
790 checkedPools.insert(poolItr->first);
791 if (checkedPools.find(poolItr->second) != checkedPools.end())
792 {
793 std::ostringstream ss;
794 ss<< "The pool(s) ";
795 for (std::set<uint32>::const_iterator itr=checkedPools.begin(); itr != checkedPools.end(); ++itr)
796 ss << *itr << ' ';
797 ss << "create(s) a circular reference, which can cause the server to freeze.\nRemoving the last link between mother pool "
798 << poolItr->first << " and child pool " << poolItr->second;
799 TC_LOG_ERROR("sql.sql", "%s", ss.str().c_str());
800 mPoolPoolGroups[poolItr->second].RemoveOneRelation(poolItr->first);
801 mPoolSearchMap.erase(poolItr);
802 --count;
803 break;
804 }
805 }
806 }
807
808 TC_LOG_INFO("server.loading", ">> Loaded %u pools in mother pools in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
809 }
810 }
811
812 TC_LOG_INFO("server.loading", "Loading Quest Pooling Data...");
813 {
814 uint32 oldMSTime = getMSTime();
815
816 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_QUEST_POOLS);
817 PreparedQueryResult result = WorldDatabase.Query(stmt);
818
819 if (!result)
820 {
821 TC_LOG_INFO("server.loading", ">> Loaded 0 quests in pools");
822 }
823 else
824 {
825 PooledQuestRelationBounds creBounds;
826 PooledQuestRelationBounds goBounds;
827
828 enum eQuestTypes
829 {
830 QUEST_NONE = 0,
831 QUEST_DAILY = 1,
832 QUEST_WEEKLY = 2
833 };
834
835 std::map<uint32, int32> poolTypeMap;
836 uint32 count = 0;
837 do
838 {
839 Field* fields = result->Fetch();
840
841 uint32 entry = fields[0].GetUInt32();
842 uint32 pool_id = fields[1].GetUInt32();
843
844 Quest const* quest = sObjectMgr->GetQuestTemplate(entry);
845 if (!quest)
846 {
847 TC_LOG_ERROR("sql.sql", "`pool_quest` has a non existing quest template (Entry: %u) defined for pool id (%u), skipped.", entry, pool_id);
848 continue;
849 }
850
851 if (pool_id > max_pool_id)
852 {
853 TC_LOG_ERROR("sql.sql", "`pool_quest` pool id (%u) is out of range compared to max pool id in `pool_template`, skipped.", pool_id);
854 continue;
855 }
856
857 if (!quest->IsDailyOrWeekly())
858 {
859 TC_LOG_ERROR("sql.sql", "`pool_quest` has an quest (%u) which is not daily or weekly in pool id (%u), use ExclusiveGroup instead, skipped.", entry, pool_id);
860 continue;
861 }
862
863 if (poolTypeMap[pool_id] == QUEST_NONE)
864 poolTypeMap[pool_id] = quest->IsDaily() ? QUEST_DAILY : QUEST_WEEKLY;
865
866 int32 currType = quest->IsDaily() ? QUEST_DAILY : QUEST_WEEKLY;
867
868 if (poolTypeMap[pool_id] != currType)
869 {
870 TC_LOG_ERROR("sql.sql", "`pool_quest` quest %u is %s but pool (%u) is specified for %s, mixing not allowed, skipped.",
871 entry, currType == QUEST_DAILY ? "QUEST_DAILY" : "QUEST_WEEKLY", pool_id, poolTypeMap[pool_id] == QUEST_DAILY ? "QUEST_DAILY" : "QUEST_WEEKLY");
872 continue;
873 }
874
875 creBounds = mQuestCreatureRelation.equal_range(entry);
876 goBounds = mQuestGORelation.equal_range(entry);
877
878 if (creBounds.first == creBounds.second && goBounds.first == goBounds.second)
879 {
880 TC_LOG_ERROR("sql.sql", "`pool_quest` lists entry (%u) as member of pool (%u) but is not started anywhere, skipped.", entry, pool_id);
881 continue;
882 }
883
884 PoolTemplateData* pPoolTemplate = &mPoolTemplate[pool_id];
885 PoolObject plObject = PoolObject(entry, 0.0f);
886 PoolGroup<Quest>& questgroup = mPoolQuestGroups[pool_id];
887 questgroup.SetPoolId(pool_id);
888 questgroup.AddEntry(plObject, pPoolTemplate->MaxLimit);
889 SearchPair p(entry, pool_id);
890 mQuestSearchMap.insert(p);
891
892 ++count;
893 }
894 while (result->NextRow());
895
896 TC_LOG_INFO("server.loading", ">> Loaded %u quests in pools in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
897 }
898 }
899
900 // The initialize method will spawn all pools not in an event and not in another pool, this is why there is 2 left joins with 2 null checks
901 TC_LOG_INFO("server.loading", "Starting objects pooling system...");
902 {
903 uint32 oldMSTime = getMSTime();
904
905 QueryResult result = WorldDatabase.Query("SELECT DISTINCT pool_template.entry, pool_pool.pool_id, pool_pool.mother_pool FROM pool_template"
906 " LEFT JOIN game_event_pool ON pool_template.entry=game_event_pool.pool_entry"
907 " LEFT JOIN pool_pool ON pool_template.entry=pool_pool.pool_id WHERE game_event_pool.pool_entry IS NULL");
908
909 if (!result)
910 {
911 TC_LOG_INFO("server.loading", ">> Pool handling system initialized, 0 pools spawned.");
912 }
913 else
914 {
915 uint32 count = 0;
916 do
917 {
918 Field* fields = result->Fetch();
919 uint32 pool_entry = fields[0].GetUInt32();
920 uint32 pool_pool_id = fields[1].GetUInt32();
921
922 if (!CheckPool(pool_entry))
923 {
924 if (pool_pool_id)
925 // The pool is a child pool in pool_pool table. Ideally we should remove it from the pool handler to ensure it never gets spawned,
926 // however that could recursively invalidate entire chain of mother pools. It can be done in the future but for now we'll do nothing.
927 TC_LOG_ERROR("sql.sql", "Pool Id %u has no equal chance pooled entites defined and explicit chance sum is not 100. This broken pool is a child pool of Id %u and cannot be safely removed.", pool_entry, fields[2].GetUInt32());
928 else
929 TC_LOG_ERROR("sql.sql", "Pool Id %u has no equal chance pooled entites defined and explicit chance sum is not 100. The pool will not be spawned.", pool_entry);
930 continue;
931 }
932
933 // Don't spawn child pools, they are spawned recursively by their parent pools
934 if (!pool_pool_id)
935 {
936 SpawnPool(pool_entry);
937 count++;
938 }
939 }
940 while (result->NextRow());
941
942 TC_LOG_DEBUG("pool", "Pool handling system initialized, %u pools spawned in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
943
944 }
945 }

953void PoolMgr::SaveQuestsToDB()
954{
955 SQLTransaction trans = CharacterDatabase.BeginTransaction();
956
957 for (PoolGroupQuestMap::iterator itr = mPoolQuestGroups.begin(); itr != mPoolQuestGroups.end(); ++itr)
958 {
959 if (itr->isEmpty())
960 continue;
961 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_QUEST_POOL_SAVE);
962 stmt->setUInt32(0, itr->GetPoolId());
963 trans->Append(stmt);
964 }
965
966 for (SearchMap::iterator itr = mQuestSearchMap.begin(); itr != mQuestSearchMap.end(); ++itr)
967 {
968 if (IsSpawnedObject<Quest>(itr->first))
969 {
970 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_QUEST_POOL_SAVE);
971 stmt->setUInt32(0, itr->second);
972 stmt->setUInt32(1, itr->first);
973 trans->Append(stmt);
974 }
975 }
976
977 CharacterDatabase.CommitTransaction(trans);
978}

1081// Call to update the pool when a gameobject/creature part of pool [pool_id] is ready to respawn
1082// Here we cache only the creature/gameobject whose guid is passed as parameter
1083// Then the spawn pool call will use this cache to decide
1084template<typename T>
1085void PoolMgr::UpdatePool(uint32 pool_id, uint64 db_guid_or_pool_id)
1086{

1088 SpawnPool<Pool>(motherpoolid, pool_id);

1090 SpawnPool<T>(pool_id, db_guid_or_pool_id);

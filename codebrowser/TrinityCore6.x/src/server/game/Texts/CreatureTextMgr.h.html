119template<class Builder>
120class CreatureTextLocalizer
121{
122 public:
123 CreatureTextLocalizer(Builder const& builder, ChatMsg msgType) : _builder(builder), _msgType(msgType)
124 {
125 _packetCache.resize(TOTAL_LOCALES, NULL);
126 }
127
128 ~CreatureTextLocalizer()
129 {
130 for (size_t i = 0; i < _packetCache.size(); ++i)
131 delete _packetCache[i];
132 }
133
134 void operator()(Player* player)
135 {
136 LocaleConstant loc_idx = player->GetSession()->GetSessionDbLocaleIndex();
137 WorldPackets::Chat::Chat* messageTemplate;
138
139 // create if not cached yet
140 if (!_packetCache[loc_idx])
141 {
142 messageTemplate = _builder(loc_idx);
143 _packetCache[loc_idx] = messageTemplate;
144 }
145 else
146 messageTemplate = _packetCache[loc_idx];
147
148 WorldPackets::Chat::Chat message(*messageTemplate);
149
150 switch (_msgType)
151 {
152 case CHAT_MSG_MONSTER_WHISPER:
153 case CHAT_MSG_RAID_BOSS_WHISPER:
154 message.SetReceiver(player, loc_idx);
155 break;
156 default:
157 break;
158 }
159
160 player->SendDirectMessage(message.Write());
161 }
162
163 private:
164 std::vector<WorldPackets::Chat::Chat*> _packetCache;
165 Builder const& _builder;
166 ChatMsg _msgType;
167};

169template<class Builder>
170void CreatureTextMgr::SendChatPacket(WorldObject* source, Builder const& builder, ChatMsg msgType, WorldObject const* whisperTarget /*= nullptr*/, CreatureTextRange range /*= TEXT_RANGE_NORMAL*/, Team team /*= TEAM_OTHER*/, bool gmOnly /*= false*/)
171{
172 if (!source)
173 return;
174
175 CreatureTextLocalizer<Builder> localizer(builder, msgType);
176
177 switch (msgType)
178 {
179 case CHAT_MSG_MONSTER_PARTY:
180 {
181 if (!whisperTarget)
182 return;
183
184 if (Player* whisperPlayer = const_cast<Player*>(whisperTarget->ToPlayer()))
185 {
186 if (Group* group = whisperPlayer->GetGroup())
187 group->BroadcastWorker(localizer);
188 }
189 return;
190 }
191 case CHAT_MSG_MONSTER_WHISPER:
192 case CHAT_MSG_RAID_BOSS_WHISPER:
193 {
194 if (range == TEXT_RANGE_NORMAL) // ignores team and gmOnly
195 {
196 if (!whisperTarget || whisperTarget->GetTypeId() != TYPEID_PLAYER)
197 return;
198
199 localizer(const_cast<Player*>(whisperTarget->ToPlayer()));
200 return;
201 }
202 break;
203 }
204 default:
205 break;
206 }
207
208 switch (range)
209 {
210 case TEXT_RANGE_AREA:
211 {
212 uint32 areaId = source->GetAreaId();
213 Map::PlayerList const& players = source->GetMap()->GetPlayers();
214 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
215 if (itr->GetSource()->GetAreaId() == areaId && (!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
216 localizer(itr->GetSource());
217 return;
218 }
219 case TEXT_RANGE_ZONE:
220 {
221 uint32 zoneId = source->GetZoneId();
222 Map::PlayerList const& players = source->GetMap()->GetPlayers();
223 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
224 if (itr->GetSource()->GetZoneId() == zoneId && (!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
225 localizer(itr->GetSource());
226 return;
227 }
228 case TEXT_RANGE_MAP:
229 {
230 Map::PlayerList const& players = source->GetMap()->GetPlayers();
231 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
232 if ((!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
233 localizer(itr->GetSource());
234 return;
235 }
236 case TEXT_RANGE_WORLD:
237 {
238 SessionMap const& smap = sWorld->GetAllSessions();
239 for (SessionMap::const_iterator iter = smap.begin(); iter != smap.end(); ++iter)
240 if (Player* player = iter->second->GetPlayer())
241 if ((!team || Team(player->GetTeam()) == team) && (!gmOnly || player->IsGameMaster()))
242 localizer(player);
243 return;
244 }
245 case TEXT_RANGE_NORMAL:
246 default:
247 break;
248 }
249
250 float dist = GetRangeForChatType(msgType);
251 Trinity::PlayerDistWorker<CreatureTextLocalizer<Builder> > worker(source, dist, localizer);
252 source->VisitNearbyWorldObject(dist, worker);
253}

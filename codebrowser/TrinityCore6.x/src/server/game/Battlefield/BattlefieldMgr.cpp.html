35void BattlefieldMgr::InitBattlefield()
36{
37 Battlefield* wg = new BattlefieldWG();
38 // respawn, init variables
39 if (!wg->SetupBattlefield())
40 {
41 TC_LOG_INFO("bg.battlefield", "Battlefield: Wintergrasp init failed.");
42 delete wg;
43 }

72void BattlefieldMgr::HandlePlayerEnterZone(Player* player, uint32 zoneId)
73{
74 BattlefieldMap::iterator itr = _battlefieldMap.find(zoneId);
75 if (itr == _battlefieldMap.end())
76 return;
77
78 Battlefield* bf = itr->second;
79 if (!bf->IsEnabled() || bf->HasPlayer(player))
80 return;
81
82 bf->HandlePlayerEnterZone(player, zoneId);
83 TC_LOG_DEBUG("bg.battlefield", "%s entered battlefield id %u", player->GetGUID().ToString().c_str(), bf->GetTypeId());
84}
85
86void BattlefieldMgr::HandlePlayerLeaveZone(Player* player, uint32 zoneId)
87{
88 BattlefieldMap::iterator itr = _battlefieldMap.find(zoneId);
89 if (itr == _battlefieldMap.end())
90 return;
91
92 // teleport: remove once in removefromworld, once in updatezone
93 if (!itr->second->HasPlayer(player))
94 return;
95
96 itr->second->HandlePlayerLeaveZone(player, zoneId);
97 TC_LOG_DEBUG("bg.battlefield", "Player %s left battlefield id %u", player->GetGUID().ToString().c_str(), itr->second->GetTypeId());
98}

100Battlefield* BattlefieldMgr::GetBattlefieldToZoneId(uint32 zoneId)
101{
102 BattlefieldMap::iterator itr = _battlefieldMap.find(zoneId);
103 if (itr == _battlefieldMap.end())
104 {
105 // no handle for this zone, return
106 return NULL;
107 }
108
109 if (!itr->second->IsEnabled())
110 return NULL;
111
112 return itr->second;
113}

115Battlefield* BattlefieldMgr::GetBattlefieldByBattleId(uint32 battleId)
116{
117 for (BattlefieldSet::iterator itr = _battlefieldSet.begin(); itr != _battlefieldSet.end(); ++itr)
118 {
119 if ((*itr)->GetBattleId() == battleId)
120 return *itr;
121 }
122 return NULL;
123}
124
125Battlefield* BattlefieldMgr::GetBattlefieldByQueueId(uint64 queueId)
126{
127 for (Battlefield* bf : _battlefieldSet)
128 if (bf->GetQueueId() == queueId)
129 return bf;
130
131 return NULL;
132}

134ZoneScript* BattlefieldMgr::GetZoneScript(uint32 zoneId)
135{
136 BattlefieldMap::iterator itr = _battlefieldMap.find(zoneId);
137 if (itr != _battlefieldMap.end())
138 return itr->second;
139
140 return NULL;
141}

143void BattlefieldMgr::Update(uint32 diff)
144{
145 _updateTimer += diff;
146 if (_updateTimer > BATTLEFIELD_OBJECTIVE_UPDATE_INTERVAL)
147 {
148 for (BattlefieldSet::iterator itr = _battlefieldSet.begin(); itr != _battlefieldSet.end(); ++itr)
149 if ((*itr)->IsEnabled())
150 (*itr)->Update(_updateTimer);
151 _updateTimer = 0;
152 }
153}

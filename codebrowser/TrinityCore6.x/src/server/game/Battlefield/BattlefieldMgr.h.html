18#ifndef BATTLEFIELD_MGR_H_
19#define BATTLEFIELD_MGR_H_
20
21#include "Battlefield.h"
22
23class Player;
24class ZoneScript;
25
26// class to handle player enter / leave / areatrigger / GO use events
27class BattlefieldMgr
28{
29 public:
30 static BattlefieldMgr* instance()
31 {
32 static BattlefieldMgr instance;
33 return &instance;
34 }
35
36 // create battlefield events
37 void InitBattlefield();
38
39 // called when a player enters an battlefield area
40 void HandlePlayerEnterZone(Player* player, uint32 zoneId);
41 // called when player leaves an battlefield area
42 void HandlePlayerLeaveZone(Player* player, uint32 zoneId);
43
44 // return assigned battlefield
45 Battlefield* GetBattlefieldToZoneId(uint32 zoneId);
46 Battlefield* GetBattlefieldByBattleId(uint32 battleId);
47 Battlefield* GetBattlefieldByQueueId(uint64 queueId);
48
49 ZoneScript* GetZoneScript(uint32 zoneId);
50
51 void AddZone(uint32 zoneId, Battlefield* bf);
52
53 void Update(uint32 diff);
54
55 private:
56 BattlefieldMgr();
57 ~BattlefieldMgr();
58
59 typedef std::vector<Battlefield*> BattlefieldSet;
60 typedef std::map<uint32 /*zoneId*/, Battlefield*> BattlefieldMap;
61 // contains all initiated battlefield events
62 // used when initing / cleaning up
63 BattlefieldSet _battlefieldSet;
64 // maps the zone ids to an battlefield event
65 // used in player event handling
66 BattlefieldMap _battlefieldMap;
67 // update interval
68 uint32 _updateTimer;
69};
70
71#define sBattlefieldMgr BattlefieldMgr::instance()
72
73#endif // BATTLEFIELD_MGR_H_

22#include "MoveSplineFlag.h"
23#include "ObjectGuid.h"
24#include <G3D/Vector3.h>
25
26class Unit;
27
28namespace Movement
29{
30 typedef std::vector<Vector3> PointsArray;
31
32 struct FacingInfo
33 {
34 G3D::Vector3 f;
35 ObjectGuid target;
36 float angle;
37
38 FacingInfo() : angle(0.0f) { }
39 };
40
41 struct MoveSplineInitArgs
42 {
43 MoveSplineInitArgs(size_t path_capacity = 16) : path_Idx_offset(0), velocity(0.f),
44 parabolic_amplitude(0.f), time_perc(0.f), splineId(0), initialOrientation(0.f),
45 HasVelocity(false), TransformForTransport(true)
46 {
47 path.reserve(path_capacity);
48 }
49
50 PointsArray path;
51 FacingInfo facing;
52 MoveSplineFlag flags;
53 int32 path_Idx_offset;
54 float velocity;
55 float parabolic_amplitude;
56 float time_perc;
57 uint32 splineId;
58 float initialOrientation;
59 bool HasVelocity;
60 bool TransformForTransport;
61
62 /** Returns true to show that the arguments were configured correctly and MoveSpline initialization will succeed. */
63 bool Validate(Unit* unit) const;
64
65 private:
66 bool _checkPathBounds() const;
67 };
68}

64template<typename length_type> void Spline<length_type>::computeIndex(float t, index_type& index, float& u) const
65{
66 ASSERT(t >= 0.f && t <= 1.f);
67 length_type length_ = t * length();
68 index = computeIndexInBounds(length_);
69 ASSERT(index < index_hi);
70 u = (length_ - length(index)) / (float)length(index, index+1);
71}
72
73template<typename length_type> SplineBase::index_type Spline<length_type>::computeIndexInBounds( float t ) const
74{
75 ASSERT(t >= 0.f && t <= 1.f);
76 return computeIndexInBounds(t * length());
77}

79template<typename length_type> void Spline<length_type>::initLengths()
80{
81 index_type i = index_lo;
82 length_type length = 0;
83 lengths.resize(index_hi+1);
84 while (i < index_hi)
85 {
86 length += SegLength(i);
87 lengths[++i] = length;
88 }
89}

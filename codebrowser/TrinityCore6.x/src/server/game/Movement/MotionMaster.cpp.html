34inline bool isStatic(MovementGenerator *mv)
35{
36 return (mv == &si_idleMovement);
37}

39void MotionMaster::Initialize()
40{
41 // clear ALL movement generators (including default)
42 while (!empty())
43 {
44 MovementGenerator *curr = top();
45 pop();
46 if (curr)
47 DirectDelete(curr);
48 }
49
50 InitDefault();
51}

53// set new default movement generator
54void MotionMaster::InitDefault()
55{
56 if (_owner->GetTypeId() == TYPEID_UNIT)
57 {
58 MovementGenerator* movement = FactorySelector::selectMovementGenerator(_owner->ToCreature());
59 Mutate(movement == NULL ? &si_idleMovement : movement, MOTION_SLOT_IDLE);
60 }
61 else
62 {
63 Mutate(&si_idleMovement, MOTION_SLOT_IDLE);
64 }
65}

67MotionMaster::~MotionMaster()
68{
69 // clear ALL movement generators (including default)
70 while (!empty())
71 {
72 MovementGenerator *curr = top();
73 pop();
74 if (curr && !isStatic(curr))
75 delete curr; // Skip finalizing on delete, it might launch new movement
76 }
77}

79void MotionMaster::UpdateMotion(uint32 diff)
80{
81 if (!_owner)
82 return;
83
84 if (_owner->HasUnitState(UNIT_STATE_ROOT | UNIT_STATE_STUNNED)) // what about UNIT_STATE_DISTRACTED? Why is this not included?
85 return;
86
87 ASSERT(!empty());
88
89 _cleanFlag |= MMCF_UPDATE;
90 if (!top()->Update(_owner, diff))
91 {
92 _cleanFlag &= ~MMCF_UPDATE;
93 MovementExpired();
94 }
95 else
96 _cleanFlag &= ~MMCF_UPDATE;
97
98 if (_expList)
99 {
100 for (size_t i = 0; i < _expList->size(); ++i)
101 {
102 MovementGenerator* mg = (*_expList)[i];
103 DirectDelete(mg);
104 }
105
106 delete _expList;
107 _expList = NULL;
108
109 if (empty())
110 Initialize();
111 else if (needInitTop())
112 InitTop();
113 else if (_cleanFlag & MMCF_RESET)
114 top()->Reset(_owner);
115
116 _cleanFlag &= ~MMCF_RESET;
117 }
118
119 // probably not the best place to pu this but im not really sure where else to put it.
120 _owner->UpdateUnderwaterState(_owner->GetMap(), _owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ());
121}

123void MotionMaster::DirectClean(bool reset)
124{
125 while (size() > 1)
126 {
127 MovementGenerator *curr = top();
128 pop();
129 if (curr) DirectDelete(curr);
130 }
131
132 if (empty())
133 return;
134
135 if (needInitTop())
136 InitTop();
137 else if (reset)
138 top()->Reset(_owner);
139}

141void MotionMaster::DelayedClean()

148 DelayedDelete(curr);

152void MotionMaster::DirectExpire(bool reset)
153{
154 if (size() > 1)
155 {
156 MovementGenerator *curr = top();
157 pop();
158 DirectDelete(curr);
159 }
160
161 while (!empty() && !top())
162 --_top;
163
164 if (empty())
165 Initialize();
166 else if (needInitTop())
167 InitTop();
168 else if (reset)
169 top()->Reset(_owner);
170}
171
172void MotionMaster::DelayedExpire()
173{
174 if (size() > 1)
175 {
176 MovementGenerator *curr = top();
177 pop();
178 DelayedDelete(curr);
179 }
180
181 while (!empty() && !top())
182 --_top;
183}

327void MotionMaster::MoveKnockbackFrom(float srcX, float srcY, float speedXY, float speedZ)
328{
329 //this function may make players fall below map
330 if (_owner->GetTypeId() == TYPEID_PLAYER)
331 return;
332
333 if (speedXY <= 0.1f)
334 return;
335
336 float x, y, z;
337 float moveTimeHalf = speedZ / Movement::gravity;
338 float dist = 2 * moveTimeHalf * speedXY;
339 float max_height = -Movement::computeFallElevation(moveTimeHalf, false, -speedZ);
340
341 _owner->GetNearPoint(_owner, x, y, z, _owner->GetObjectSize(), dist, _owner->GetAngle(srcX, srcY) + float(M_PI));
342
343 Movement::MoveSplineInit init(_owner);
344 init.MoveTo(x, y, z);
345 init.SetParabolic(max_height, 0);
346 init.SetOrientationFixed(true);
347 init.SetVelocity(speedXY);
348 init.Launch();
349 Mutate(new EffectMovementGenerator(0), MOTION_SLOT_CONTROLLED);
350}

383void MotionMaster::MoveCirclePath(float x, float y, float z, float radius, bool clockwise, uint8 stepCount)
384{
385 float step = 2 * float(M_PI) / stepCount * (clockwise ? -1.0f : 1.0f);
386 Position const& pos = { x, y, z, 0.0f };
387 float angle = pos.GetAngle(_owner->GetPositionX(), _owner->GetPositionY());
388
389 Movement::MoveSplineInit init(_owner);
390
391 for (uint8 i = 0; i < stepCount; angle += step, ++i)
392 {
393 G3D::Vector3 point;
394 point.x = x + radius * cosf(angle);
395 point.y = y + radius * sinf(angle);
396
397 if (_owner->IsFlying())
398 point.z = z;
399 else
400 point.z = _owner->GetMap()->GetHeight(_owner->GetPhaseMask(), point.x, point.y, z);
401
402 init.Path().push_back(point);
403 }
404
405 if (_owner->IsFlying())
406 {
407 init.SetFly();
408 init.SetCyclic();
409 init.SetAnimation(Movement::ToFly);
410 }
411 else
412 {
413 init.SetWalk(true);
414 init.SetCyclic();
415 }
416
417 init.Launch();
418}

438void MotionMaster::MoveFall(uint32 id /*=0*/)
439{
440 // use larger distance for vmap height search than in most other cases
441 float tz = _owner->GetMap()->GetHeight(_owner->GetPhaseMask(), _owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ(), true, MAX_FALL_DISTANCE);
442 if (tz <= INVALID_HEIGHT)
443 {
444 TC_LOG_DEBUG("misc", "MotionMaster::MoveFall: unable retrive a proper height at map %u (x: %f, y: %f, z: %f).",
445 _owner->GetMap()->GetId(), _owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ());
446 return;
447 }
448
449 // Abort too if the ground is very near
450 if (std::fabs(_owner->GetPositionZ() - tz) < 0.1f)
451 return;
452
453 if (_owner->GetTypeId() == TYPEID_PLAYER)
454 _owner->SetFall(true);
455
456 Movement::MoveSplineInit init(_owner);
457 init.MoveTo(_owner->GetPositionX(), _owner->GetPositionY(), tz, false);
458 init.SetFall();
459 init.Launch();
460 Mutate(new EffectMovementGenerator(id), MOTION_SLOT_CONTROLLED);
461}

591void MotionMaster::Mutate(MovementGenerator *m, MovementSlot slot)
592{
593 if (MovementGenerator *curr = Impl[slot])
594 {
595 Impl[slot] = NULL; // in case a new one is generated in this slot during directdelete
596 if (_top == slot && (_cleanFlag & MMCF_UPDATE))
597 DelayedDelete(curr);
598 else
599 DirectDelete(curr);
600 }
601 else if (_top < slot)
602 {
603 _top = slot;
604 }
605
606 Impl[slot] = m;
607 if (_top > slot)
608 _needInit[slot] = true;
609 else
610 {
611 _needInit[slot] = false;
612 m->Initialize(_owner);
613 }
614}

30template<class T, typename D>
31void TargetedMovementGeneratorMedium<T, D>::_setTargetLocation(T* owner, bool updateDestination)
32{
33 if (!i_target.isValid() || !i_target->IsInWorld())
34 return;
35
36 if (owner->HasUnitState(UNIT_STATE_NOT_MOVE))
37 return;
38
39 if (owner->GetTypeId() == TYPEID_UNIT && !i_target->isInAccessiblePlaceFor(owner->ToCreature()))
40 return;
41
42 float x, y, z;
43
44 if (updateDestination || !i_path)
45 {
46 if (!i_offset)
47 {
48 // to nearest contact position
49 i_target->GetContactPoint(owner, x, y, z);
50 }
51 else
52 {
53 float dist;
54 float size;
55
56 // Pets need special handling.
57 // We need to subtract GetObjectSize() because it gets added back further down the chain
58 // and that makes pets too far away. Subtracting it allows pets to properly
59 // be (GetCombatReach() + i_offset) away.
60 // Only applies when i_target is pet's owner otherwise pets and mobs end up
61 // doing a "dance" while fighting
62 if (owner->IsPet() && i_target->GetTypeId() == TYPEID_PLAYER)
63 {
64 dist = i_target->GetCombatReach();
65 size = i_target->GetCombatReach() - i_target->GetObjectSize();
66 }
67 else
68 {
69 dist = i_offset + 1.0f;
70 size = owner->GetObjectSize();
71 }
72
73 if (i_target->IsWithinDistInMap(owner, dist))
74 return;
75
76 // to at i_offset distance from target and i_angle from target facing
77 i_target->GetClosePoint(x, y, z, size, i_offset, i_angle);
78 }
79 }
80 else
81 {
82 // the destination has not changed, we just need to refresh the path (usually speed change)
83 G3D::Vector3 end = i_path->GetEndPosition();
84 x = end.x;
85 y = end.y;
86 z = end.z;
87 }
88
89 if (!i_path)
90 i_path = new PathGenerator(owner);
91
92 // allow pets to use shortcut if no path found when following their master
93 bool forceDest = (owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature()->IsPet()
94 && owner->HasUnitState(UNIT_STATE_FOLLOW));
95
96 bool result = i_path->CalculatePath(x, y, z, forceDest);
97 if (!result || (i_path->GetPathType() & PATHFIND_NOPATH))
98 {
99 // Cant reach target
100 i_recalculateTravel = true;
101 return;
102 }
103
104 D::_addUnitStateMove(owner);
105 i_targetReached = false;
106 i_recalculateTravel = false;
107 owner->AddUnitState(UNIT_STATE_CHASE);
108
109 Movement::MoveSplineInit init(owner);
110 init.MovebyPath(i_path->GetPath());
111 init.SetWalk(((D*)this)->EnableWalking());
112 // Using the same condition for facing target as the one that is used for SetInFront on movement end
113 // - applies to ChaseMovementGenerator mostly
114 if (i_angle == 0.f)
115 init.SetFacing(i_target.getTarget());
116
117 init.Launch();
118}

120template<class T, typename D>
121bool TargetedMovementGeneratorMedium<T, D>::DoUpdate(T* owner, uint32 time_diff)
122{
123 if (!i_target.isValid() || !i_target->IsInWorld())
124 return false;
125
126 if (!owner || !owner->IsAlive())
127 return false;
128
129 if (owner->HasUnitState(UNIT_STATE_NOT_MOVE))
130 {
131 D::_clearUnitStateMove(owner);
132 return true;
133 }
134
135 // prevent movement while casting spells with cast time or channel time
136 if (owner->HasUnitState(UNIT_STATE_CASTING))
137 {
138 if (!owner->IsStopped())
139 owner->StopMoving();
140 return true;
141 }
142
143 // prevent crash after creature killed pet
144 if (static_cast<D*>(this)->_lostTarget(owner))
145 {
146 D::_clearUnitStateMove(owner);
147 return true;
148 }
149
150 bool targetMoved = false;
151 i_recheckDistance.Update(time_diff);
152 if (i_recheckDistance.Passed())
153 {
154 i_recheckDistance.Reset(100);
155 //More distance let have better performance, less distance let have more sensitive reaction at target move.
156 float allowed_dist = owner->GetCombatReach() + sWorld->getRate(RATE_TARGET_POS_RECALCULATION_RANGE);
157 G3D::Vector3 dest = owner->movespline->FinalDestination();
158 if (owner->movespline->onTransport)
159 if (TransportBase* transport = owner->GetDirectTransport())
160 transport->CalculatePassengerPosition(dest.x, dest.y, dest.z);
161
162 // First check distance
163 if (owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature()->CanFly())
164 targetMoved = !i_target->IsWithinDist3d(dest.x, dest.y, dest.z, allowed_dist);
165 else
166 targetMoved = !i_target->IsWithinDist2d(dest.x, dest.y, allowed_dist);
167
168 // then, if the target is in range, check also Line of Sight.
169 if (!targetMoved)
170 targetMoved = !i_target->IsWithinLOSInMap(owner);
171 }
172
173 if (i_recalculateTravel || targetMoved)
174 _setTargetLocation(owner, targetMoved);
175
176 if (owner->movespline->Finalized())
177 {
178 static_cast<D*>(this)->MovementInform(owner);
179 if (i_angle == 0.f && !owner->HasInArc(0.01f, i_target.getTarget()))
180 owner->SetInFront(i_target.getTarget());
181
182 if (!i_targetReached)
183 {
184 i_targetReached = true;
185 static_cast<D*>(this)->_reachTarget(owner);
186 }
187 }
188
189 return true;
190}

193template<class T>
194void ChaseMovementGenerator<T>::_reachTarget(T* owner)
195{
196 if (owner->IsWithinMeleeRange(this->i_target.getTarget()))
197 owner->Attack(this->i_target.getTarget(), true);
198}

200template<>
201void ChaseMovementGenerator<Player>::DoInitialize(Player* owner)
202{
203 owner->AddUnitState(UNIT_STATE_CHASE | UNIT_STATE_CHASE_MOVE);
204 _setTargetLocation(owner, true);
205}

207template<>
208void ChaseMovementGenerator<Creature>::DoInitialize(Creature* owner)
209{
210 owner->SetWalk(false);
211 owner->AddUnitState(UNIT_STATE_CHASE | UNIT_STATE_CHASE_MOVE);
212 _setTargetLocation(owner, true);
213}

257template<>
258void FollowMovementGenerator<Creature>::_updateSpeed(Creature* owner)
259{
260 // pet only sync speed with owner
261 /// Make sure we are not in the process of a map change (IsInWorld)
262 if (!owner->IsPet() || !owner->IsInWorld() || !i_target.isValid() || i_target->GetGUID() != owner->GetOwnerGUID())
263 return;
264
265 owner->UpdateSpeed(MOVE_RUN, true);
266 owner->UpdateSpeed(MOVE_WALK, true);
267 owner->UpdateSpeed(MOVE_SWIM, true);
268}

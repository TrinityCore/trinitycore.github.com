85enum AchievementCriteriaDataType
86{ // value1 value2 comment
87 ACHIEVEMENT_CRITERIA_DATA_TYPE_NONE = 0, // 0 0
88 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_CREATURE = 1, // creature_id 0
89 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_CLASS_RACE = 2, // class_id race_id
90 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_LESS_HEALTH= 3, // health_percent 0
91 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AURA = 5, // spell_id effect_idx
92 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_AURA = 7, // spell_id effect_idx
93 ACHIEVEMENT_CRITERIA_DATA_TYPE_VALUE = 8, // minvalue value provided with achievement update must be not less that limit
94 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_LEVEL = 9, // minlevel minlevel of target
95 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_GENDER = 10, // gender 0=male; 1=female
96 ACHIEVEMENT_CRITERIA_DATA_TYPE_SCRIPT = 11, // scripted requirement
97 // REUSE
98 ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_PLAYER_COUNT = 13, // count "with less than %u people in the zone"
99 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_TEAM = 14, // team HORDE(67), ALLIANCE(469)
100 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_DRUNK = 15, // drunken_state 0 (enum DrunkenState) of player
101 ACHIEVEMENT_CRITERIA_DATA_TYPE_HOLIDAY = 16, // holiday_id 0 event in holiday time
102 ACHIEVEMENT_CRITERIA_DATA_TYPE_BG_LOSS_TEAM_SCORE = 17, // min_score max_score player's team win bg and opposition team have team score in range
103 ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT = 18, // 0 0 maker instance script call for check current criteria requirements fit
104 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_EQUIPED_ITEM = 19, // item_level item_quality for equipped item in slot to check item level and quality
105 ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_ID = 20, // map_id 0 player must be on map with id in map_id
106 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_PLAYER_CLASS_RACE = 21, // class_id race_id
107 // REUSE
108 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_KNOWN_TITLE = 23, // title_id known (pvp) title, values from dbc
109 ACHIEVEMENT_CRITERIA_DATA_TYPE_GAME_EVENT = 24, // game_event_id 0
110
111 MAX_ACHIEVEMENT_CRITERIA_DATA_TYPE
112};

114struct AchievementCriteriaData
115{
116 AchievementCriteriaDataType dataType;
117 union
118 {
119 // ACHIEVEMENT_CRITERIA_DATA_TYPE_NONE = 0 (no data)
120 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_CREATURE = 1
121 struct
122 {
123 uint32 id;
124 } creature;
125 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_CLASS_RACE = 2
126 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_PLAYER_CLASS_RACE = 21
127 struct
128 {
129 uint32 class_id;
130 uint32 race_id;
131 } classRace;
132 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_LESS_HEALTH = 3
133 struct
134 {
135 uint32 percent;
136 } health;
137 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AURA = 5
138 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_AURA = 7
139 struct
140 {
141 uint32 spell_id;
142 uint32 effect_idx;
143 } aura;
144 // ACHIEVEMENT_CRITERIA_DATA_TYPE_VALUE = 8
145 struct
146 {
147 uint32 value;
148 uint32 compType;
149 } value;
150 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_LEVEL = 9
151 struct
152 {
153 uint32 minlevel;
154 } level;
155 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_GENDER = 10
156 struct
157 {
158 uint32 gender;
159 } gender;
160 // ACHIEVEMENT_CRITERIA_DATA_TYPE_SCRIPT = 11 (no data)
161 // ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_PLAYER_COUNT = 13
162 struct
163 {
164 uint32 maxcount;
165 } map_players;
166 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_TEAM = 14
167 struct
168 {
169 uint32 team;
170 } team;
171 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_DRUNK = 15
172 struct
173 {
174 uint32 state;
175 } drunk;
176 // ACHIEVEMENT_CRITERIA_DATA_TYPE_HOLIDAY = 16
177 struct
178 {
179 uint32 id;
180 } holiday;
181 // ACHIEVEMENT_CRITERIA_DATA_TYPE_BG_LOSS_TEAM_SCORE= 17
182 struct
183 {
184 uint32 min_score;
185 uint32 max_score;
186 } bg_loss_team_score;
187 // ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT = 18 (no data)
188 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_EQUIPED_ITEM = 19
189 struct
190 {
191 uint32 item_level;
192 uint32 item_quality;
193 } equipped_item;
194 // ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_ID = 20
195 struct
196 {
197 uint32 mapId;
198 } map_id;
199 // ACHIEVEMENT_CRITERIA_DATA_TYPE_KNOWN_TITLE = 22
200 struct
201 {
202 uint32 title_id;
203 } known_title;
204 // ACHIEVEMENT_CRITERIA_DATA_TYPE_GAME_EVENT = 24
205 struct
206 {
207 uint32 id;
208 } game_event;
209 // raw
210 struct
211 {
212 uint32 value1;
213 uint32 value2;
214 } raw;
215 };
216 uint32 ScriptId;
217
218 AchievementCriteriaData() : dataType(ACHIEVEMENT_CRITERIA_DATA_TYPE_NONE)
219 {
220 raw.value1 = 0;
221 raw.value2 = 0;
222 ScriptId = 0;
223 }
224
225 AchievementCriteriaData(uint32 _dataType, uint32 _value1, uint32 _value2, uint32 _scriptId) : dataType(AchievementCriteriaDataType(_dataType))
226 {
227 raw.value1 = _value1;
228 raw.value2 = _value2;
229 ScriptId = _scriptId;
230 }
231
232 bool IsValid(AchievementCriteria const* criteria);
233 bool Meets(uint32 criteria_id, Player const* source, Unit const* target, uint32 miscValue1 = 0) const;
234};

392
393 bool IsRealmCompleted(AchievementEntry const* achievement, uint32 instanceId) const
394 {
395 AllCompletedAchievements::const_iterator itr = _allCompletedAchievements.find(achievement->ID);
396 if (itr == _allCompletedAchievements.end())
397 return false;
398
399 if (achievement->Flags & ACHIEVEMENT_FLAG_REALM_FIRST_KILL)
400 return itr->second != instanceId;
401
402 return true;
403 }
404
405 void SetRealmCompleted(AchievementEntry const* achievement, uint32 instanceId)
406 {
407 if (IsRealmCompleted(achievement, instanceId))
408 return;
409
410 _allCompletedAchievements[achievement->ID] = instanceId;
411 }
412
413 bool IsGroupCriteriaType(AchievementCriteriaTypes type) const
414 {
415 switch (type)
416 {
417 case ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE:
418 case ACHIEVEMENT_CRITERIA_TYPE_WIN_BG:
419 case ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET: // NYI
420 case ACHIEVEMENT_CRITERIA_TYPE_WIN_RATED_ARENA:
421 case ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET2: // NYI
422 case ACHIEVEMENT_CRITERIA_TYPE_WIN_RATED_BATTLEGROUND: // NYI
423 return true;
424 default:
425 break;
426 }
427
428 return false;
429 }

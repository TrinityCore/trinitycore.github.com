183// All basic config data used by other AHBot classes for self-configure.
184class AuctionBotConfig
185{
186private:
187 AuctionBotConfig(): _itemsPerCycleBoost(1000), _itemsPerCycleNormal(20) {}
188 ~AuctionBotConfig() {}
189 AuctionBotConfig(const AuctionBotConfig&);
190 AuctionBotConfig& operator=(const AuctionBotConfig&);
191
192public:
193 static AuctionBotConfig* instance()
194 {
195 static AuctionBotConfig instance;
196 return &instance;
197 }
198
199 bool Initialize();
200 const std::string& GetAHBotIncludes() const { return _AHBotIncludes; }
201 const std::string& GetAHBotExcludes() const { return _AHBotExcludes; }
202
203 uint32 GetConfig(AuctionBotConfigUInt32Values index) const { return _configUint32Values[index]; }
204 bool GetConfig(AuctionBotConfigBoolValues index) const { return _configBoolValues[index]; }
205 float GetConfig(AuctionBotConfigFloatValues index) const { return _configFloatValues[index]; }
206 void SetConfig(AuctionBotConfigBoolValues index, bool value) { _configBoolValues[index] = value; }
207 void SetConfig(AuctionBotConfigUInt32Values index, uint32 value) { _configUint32Values[index] = value; }
208 void SetConfig(AuctionBotConfigFloatValues index, float value) { _configFloatValues[index] = value; }
209
210 uint32 GetConfigItemAmountRatio(AuctionHouseType houseType) const;
211 bool GetConfigBuyerEnabled(AuctionHouseType houseType) const;
212 uint32 GetConfigItemQualityAmount(AuctionQuality quality) const;
213
214 uint32 GetItemPerCycleBoost() const { return _itemsPerCycleBoost; }
215 uint32 GetItemPerCycleNormal() const { return _itemsPerCycleNormal; }
216 void Reload() { GetConfigFromFile(); }
217
218 static char const* GetHouseTypeName(AuctionHouseType houseType);
219
220private:
221 std::string _AHBotIncludes;
222 std::string _AHBotExcludes;
223 uint32 _itemsPerCycleBoost;
224 uint32 _itemsPerCycleNormal;
225
226 uint32 _configUint32Values[CONFIG_UINT32_AHBOT_UINT32_COUNT];
227 bool _configBoolValues[CONFIG_UINT32_AHBOT_BOOL_COUNT];
228 float _configFloatValues[CONFIG_AHBOT_FLOAT_COUNT];
229
230 void SetAHBotIncludes(const std::string& AHBotIncludes) { _AHBotIncludes = AHBotIncludes; }
231 void SetAHBotExcludes(const std::string& AHBotExcludes) { _AHBotExcludes = AHBotExcludes; }
232
233 void SetConfig(AuctionBotConfigUInt32Values index, char const* fieldname, uint32 defvalue);
234 void SetConfigMax(AuctionBotConfigUInt32Values index, char const* fieldname, uint32 defvalue, uint32 maxvalue);
235 void SetConfigMinMax(AuctionBotConfigUInt32Values index, char const* fieldname, uint32 defvalue, uint32 minvalue, uint32 maxvalue);
236 void SetConfig(AuctionBotConfigBoolValues index, char const* fieldname, bool defvalue);
237 void SetConfig(AuctionBotConfigFloatValues index, char const* fieldname, float defvalue);
238 void GetConfigFromFile();
239};

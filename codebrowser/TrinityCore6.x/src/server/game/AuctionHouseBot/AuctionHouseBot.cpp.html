27bool AuctionBotConfig::Initialize()
28{
29 GetConfigFromFile();
30
31 if (!GetConfig(CONFIG_AHBOT_BUYER_ENABLED) && !GetConfig(CONFIG_AHBOT_SELLER_ENABLED))
32 {
33 TC_LOG_INFO("ahbot", "AHBOT is Disabled.");
34 return false;
35 }
36
37 if (GetConfig(CONFIG_AHBOT_ALLIANCE_ITEM_AMOUNT_RATIO) == 0 && GetConfig(CONFIG_AHBOT_HORDE_ITEM_AMOUNT_RATIO) == 0 && GetConfig(CONFIG_AHBOT_NEUTRAL_ITEM_AMOUNT_RATIO) == 0 &&
38 !GetConfig(CONFIG_AHBOT_BUYER_ALLIANCE_ENABLED) && !GetConfig(CONFIG_AHBOT_BUYER_HORDE_ENABLED) && !GetConfig(CONFIG_AHBOT_BUYER_NEUTRAL_ENABLED))
39 {
40 TC_LOG_INFO("ahbot", "All feature of AuctionHouseBot are disabled!");
41 return false;
42 }
43
44 if (GetConfig(CONFIG_AHBOT_ALLIANCE_ITEM_AMOUNT_RATIO) == 0 && GetConfig(CONFIG_AHBOT_HORDE_ITEM_AMOUNT_RATIO) == 0 && GetConfig(CONFIG_AHBOT_NEUTRAL_ITEM_AMOUNT_RATIO) == 0)
45 TC_LOG_INFO("ahbot", "AuctionHouseBot SELLER is disabled!");
46
47 if (!GetConfig(CONFIG_AHBOT_BUYER_ALLIANCE_ENABLED) && !GetConfig(CONFIG_AHBOT_BUYER_HORDE_ENABLED) && !GetConfig(CONFIG_AHBOT_BUYER_NEUTRAL_ENABLED))
48 TC_LOG_INFO("ahbot", "AuctionHouseBot BUYER is disabled!");
49
50 _itemsPerCycleBoost = GetConfig(CONFIG_AHBOT_ITEMS_PER_CYCLE_BOOST);
51 _itemsPerCycleNormal = GetConfig(CONFIG_AHBOT_ITEMS_PER_CYCLE_NORMAL);
52
53 return true;
54}

56void AuctionBotConfig::SetConfig(AuctionBotConfigUInt32Values index, char const* fieldname, uint32 defvalue)
57{
58 SetConfig(index, sConfigMgr->GetIntDefault(fieldname, defvalue));
59
60 if (int32(GetConfig(index)) < 0)
61 {
62 TC_LOG_ERROR("ahbot", "AHBot: %s (%i) can't be negative. Using %u instead.", fieldname, int32(GetConfig(index)), defvalue);
63 SetConfig(index, defvalue);
64 }

67void AuctionBotConfig::SetConfigMax(AuctionBotConfigUInt32Values index, char const* fieldname, uint32 defvalue, uint32 maxvalue)
68{
69 SetConfig(index, sConfigMgr->GetIntDefault(fieldname, defvalue));
70
71 if (GetConfig(index) > maxvalue)
72 {
73 TC_LOG_ERROR("ahbot", "AHBot: %s (%u) must be in range 0...%u. Using %u instead.", fieldname, GetConfig(index), maxvalue, maxvalue);
74 SetConfig(index, maxvalue);
75 }
76}

78void AuctionBotConfig::SetConfigMinMax(AuctionBotConfigUInt32Values index, char const* fieldname, uint32 defvalue, uint32 minvalue, uint32 maxvalue)
79{
80 SetConfig(index, sConfigMgr->GetIntDefault(fieldname, defvalue));
81
82 if (GetConfig(index) > maxvalue)
83 {
84 TC_LOG_ERROR("ahbot", "AHBot: %s (%u) must be in range %u...%u. Using %u instead.", fieldname, GetConfig(index), minvalue, maxvalue, maxvalue);
85 SetConfig(index, maxvalue);
86 }
87
88 if (GetConfig(index) < minvalue)
89 {
90 TC_LOG_ERROR("ahbot", "AHBot: %s (%u) must be in range %u...%u. Using %u instead.", fieldname, GetConfig(index), minvalue, maxvalue, minvalue);
91 SetConfig(index, minvalue);
92 }

310void AuctionHouseBot::InitializeAgents()
311{

313 {

315

319 delete _seller;

321 }
322 }
323
324 if (sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BUYER_ENABLED))
325 {
326 delete _buyer;
327
328 _buyer = new AuctionBotBuyer();
329 if (!_buyer->Initialize())

331 delete _buyer;
332 _buyer = nullptr;
333 }
334 }
335}

373void AuctionHouseBot::PrepareStatusInfos(AuctionHouseBotStatusInfo& statusInfo)
374{
375 for (uint32 i = 0; i < MAX_AUCTION_HOUSE_TYPE; ++i)
376 {
377 statusInfo[i].ItemsCount = 0;
378
379 for (int j = 0; j < MAX_AUCTION_QUALITY; ++j)
380 statusInfo[i].QualityInfo[j] = 0;
381
382 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(AuctionHouseType(i));
383 for (AuctionHouseObject::AuctionEntryMap::const_iterator itr = auctionHouse->GetAuctionsBegin(); itr != auctionHouse->GetAuctionsEnd(); ++itr)
384 {
385 AuctionEntry* auctionEntry = itr->second;
386 if (Item* item = sAuctionMgr->GetAItem(auctionEntry->itemGUIDLow))
387 {
388 ItemTemplate const* prototype = item->GetTemplate();
389 if (!auctionEntry->owner) // Add only ahbot items
390 {
391 if (prototype->GetQuality() < MAX_AUCTION_QUALITY)
392 ++statusInfo[i].QualityInfo[prototype->GetQuality()];
393
394 ++statusInfo[i].ItemsCount;
395 }
396 }
397 }
398 }

401void AuctionHouseBot::Rebuild(bool all)
402{
403 for (uint32 i = 0; i < MAX_AUCTION_HOUSE_TYPE; ++i)
404 {
405 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(AuctionHouseType(i));
406 for (AuctionHouseObject::AuctionEntryMap::const_iterator itr = auctionHouse->GetAuctionsBegin(); itr != auctionHouse->GetAuctionsEnd(); ++itr)
407 if (!itr->second->owner) // ahbot auction
408 if (all || itr->second->bid == 0) // expire now auction if no bid or forced
409 itr->second->expire_time = sWorld->GetGameTime();
410 }

413void AuctionHouseBot::Update()
414{
415 // nothing do...
416 if (!_buyer && !_seller)
417 return;
418
419 // scan all possible update cases until first success
420 for (uint32 count = 0; count < 2 * MAX_AUCTION_HOUSE_TYPE; ++count)
421 {
422 bool successStep = false;
423
424 if (_operationSelector < MAX_AUCTION_HOUSE_TYPE)
425 {
426 if (_seller)
427 successStep = _seller->Update(AuctionHouseType(_operationSelector));
428 }
429 else
430 {
431 if (_buyer)
432 successStep = _buyer->Update(AuctionHouseType(_operationSelector - MAX_AUCTION_HOUSE_TYPE));
433 }
434
435 ++_operationSelector;
436 if (_operationSelector >= 2 * MAX_AUCTION_HOUSE_TYPE)
437 _operationSelector = 0;
438
439 // one success update per call
440 if (successStep)
441 break;
442 }
443}

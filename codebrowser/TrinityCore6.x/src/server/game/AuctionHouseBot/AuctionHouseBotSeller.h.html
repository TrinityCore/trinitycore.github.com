55class SellerConfiguration
56{
57public:
58 SellerConfiguration(): LastMissedItem(0), _houseType(AUCTION_HOUSE_NEUTRAL), _minTime(1), _maxTime(72)
59 {
60 }
61
62 ~SellerConfiguration() {}
63
64 void Initialize(AuctionHouseType houseType)
65 {
66 _houseType = houseType;
67 }
68
69 AuctionHouseType GetHouseType() const { return _houseType; }
70
71 uint32 LastMissedItem;
72
73 void SetMinTime(uint32 value)
74 {
75 _minTime = value;
76 }
77 uint32 GetMinTime() const
78 {
79 return std::min(1u, std::min(_minTime, _maxTime));
80 }
81
82 void SetMaxTime(uint32 value) { _maxTime = value; }
83 uint32 GetMaxTime() const { return _maxTime; }
84 // Data access classified by item class and item quality
85 void SetItemsAmountPerClass(AuctionQuality quality, ItemClass itemclass, uint32 amount) { _ItemInfo[quality].ItemClassInfos[itemclass].AmountOfItems = amount * _ItemInfo[quality].ItemClassInfos[itemclass].Quantity; }
86 uint32 GetItemsAmountPerClass(AuctionQuality quality, ItemClass itemclass) const { return _ItemInfo[quality].ItemClassInfos[itemclass].AmountOfItems; }
87 void SetItemsQuantityPerClass(AuctionQuality quality, ItemClass itemclass, uint32 qty) { _ItemInfo[quality].ItemClassInfos[itemclass].Quantity = qty; }
88 uint32 GetItemsQuantityPerClass(AuctionQuality quality, ItemClass itemclass) const { return _ItemInfo[quality].ItemClassInfos[itemclass].Quantity; }
89 void SetMissedItemsPerClass(AuctionQuality quality, ItemClass itemclass, uint32 found)
90 {
91 if (_ItemInfo[quality].ItemClassInfos[itemclass].AmountOfItems > found)
92 _ItemInfo[quality].ItemClassInfos[itemclass].MissItems = _ItemInfo[quality].ItemClassInfos[itemclass].AmountOfItems - found;
93 else
94 _ItemInfo[quality].ItemClassInfos[itemclass].MissItems = 0;
95 }
96 uint32 GetMissedItemsPerClass(AuctionQuality quality, ItemClass itemclass) const { return _ItemInfo[quality].ItemClassInfos[itemclass].MissItems; }
97
98 // Data for every quality of item
99 void SetItemsAmountPerQuality(AuctionQuality quality, uint32 cnt) { _ItemInfo[quality].AmountOfItems = cnt; }
100 uint32 GetItemsAmountPerQuality(AuctionQuality quality) const { return _ItemInfo[quality].AmountOfItems; }
101 void SetPriceRatioPerQuality(AuctionQuality quality, uint32 value) { _ItemInfo[quality].PriceRatio = value; }
102 uint32 GetPriceRatioPerQuality(AuctionQuality quality) const { return _ItemInfo[quality].PriceRatio; }
103 void SetPriceRatioPerClass(ItemClass item, uint32 value) { _ItemInfo[0].ItemClassInfos[item].PriceRatio = value; }
104 uint32 GetPriceRatioPerClass(ItemClass item) const { return _ItemInfo[0].ItemClassInfos[item].PriceRatio; }
105
106private:
107 AuctionHouseType _houseType;
108 uint32 _minTime;
109 uint32 _maxTime;
110 SellerItemInfo _ItemInfo[MAX_AUCTION_QUALITY];
111};

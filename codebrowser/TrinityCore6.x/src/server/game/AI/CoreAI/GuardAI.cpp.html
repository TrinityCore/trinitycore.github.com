33bool GuardAI::CanSeeAlways(WorldObject const* obj)
34{
35 if (!obj->isType(TYPEMASK_UNIT))
36 return false;
37
38 ThreatContainer::StorageType threatList = me->getThreatManager().getThreatList();
39 for (ThreatContainer::StorageType::const_iterator itr = threatList.begin(); itr != threatList.end(); ++itr)
40 if ((*itr)->getUnitGuid() == obj->GetGUID())
41 return true;
42
43 return false;
44}

46void GuardAI::EnterEvadeMode()
47{
48 if (!me->IsAlive())
49 {
50 me->GetMotionMaster()->MoveIdle();
51 me->CombatStop(true);
52 me->DeleteThreatList();
53 return;
54 }
55
56 TC_LOG_DEBUG("entities.unit", "Guard entry: %u enters evade mode.", me->GetEntry());
57
58 me->RemoveAllAuras();
59 me->DeleteThreatList();
60 me->CombatStop(true);
61
62 // Remove ChaseMovementGenerator from MotionMaster stack list, and add HomeMovementGenerator instead
63 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
64 me->GetMotionMaster()->MoveTargetedHome();
65}

67void GuardAI::JustDied(Unit* killer)
68{
69 if (Player* player = killer->GetCharmerOrOwnerPlayerOrPlayerItself())
70 me->SendZoneUnderAttackMessage(player);
71}

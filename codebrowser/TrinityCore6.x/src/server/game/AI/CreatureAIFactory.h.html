26
27struct SelectableAI : public FactoryHolder<CreatureAI>, public Permissible<Creature>
28{
29 SelectableAI(const char* id) : FactoryHolder<CreatureAI>(id) { }
30};
31
32template<class REAL_AI>
33struct CreatureAIFactory : public SelectableAI
34{
35 CreatureAIFactory(const char* name) : SelectableAI(name) { }
36
37 CreatureAI* Create(void*) const override;
38
39 int Permit(const Creature* c) const override { return REAL_AI::Permissible(c); }
40};
41
42template<class REAL_AI>
43inline CreatureAI*
44CreatureAIFactory<REAL_AI>::Create(void* data) const
45{
46 Creature* creature = reinterpret_cast<Creature*>(data);
47 return (new REAL_AI(creature));
48}
49
50typedef FactoryHolder<CreatureAI> CreatureAICreator;
51typedef FactoryHolder<CreatureAI>::FactoryHolderRegistry CreatureAIRegistry;
52
53#define sCreatureAIRegistry CreatureAIRegistry::instance()

56struct SelectableGameObjectAI : public FactoryHolder<GameObjectAI>, public Permissible<GameObject>
57{
58 SelectableGameObjectAI(const char* id) : FactoryHolder<GameObjectAI>(id) { }
59};
60
61template<class REAL_GO_AI>
62struct GameObjectAIFactory : public SelectableGameObjectAI
63{
64 GameObjectAIFactory(const char* name) : SelectableGameObjectAI(name) { }
65
66 GameObjectAI* Create(void*) const override;
67
68 int Permit(const GameObject* g) const override { return REAL_GO_AI::Permissible(g); }
69};
70
71template<class REAL_GO_AI>
72inline GameObjectAI*
73GameObjectAIFactory<REAL_GO_AI>::Create(void* data) const
74{
75 GameObject* go = reinterpret_cast<GameObject*>(data);
76 return (new REAL_GO_AI(go));
77}
78
79typedef FactoryHolder<GameObjectAI> GameObjectAICreator;
80typedef FactoryHolder<GameObjectAI>::FactoryHolderRegistry GameObjectAIRegistry;
81
82#define sGameObjectAIRegistry GameObjectAIRegistry::instance()

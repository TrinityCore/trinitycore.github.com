46void CreatureAI::DoZoneInCombat(Creature* creature /*= NULL*/, float maxRangeToNearestTarget /* = 50.0f*/)
47{
48 if (!creature)
49 creature = me;
50
51 if (!creature->CanHaveThreatList())
52 return;
53
54 Map* map = creature->GetMap();
55 if (!map->IsDungeon()) //use IsDungeon instead of Instanceable, in case battlegrounds will be instantiated
56 {
57 TC_LOG_ERROR("misc", "DoZoneInCombat call for map that isn't an instance (creature entry = %d)", creature->GetTypeId() == TYPEID_UNIT ? creature->ToCreature()->GetEntry() : 0);
58 return;
59 }
60
61 if (!creature->HasReactState(REACT_PASSIVE) && !creature->GetVictim())
62 {
63 if (Unit* nearTarget = creature->SelectNearestTarget(maxRangeToNearestTarget))
64 creature->AI()->AttackStart(nearTarget);
65 else if (creature->IsSummon())
66 {
67 if (Unit* summoner = creature->ToTempSummon()->GetSummoner())
68 {
69 Unit* target = summoner->getAttackerForHelper();
70 if (!target && summoner->CanHaveThreatList() && !summoner->getThreatManager().isThreatListEmpty())
71 target = summoner->getThreatManager().getHostilTarget();
72 if (target && (creature->IsFriendlyTo(summoner) || creature->IsHostileTo(target)))
73 creature->AI()->AttackStart(target);
74 }
75 }
76 }
77
78 // Intended duplicated check, the code above this should select a victim
79 // If it can't find a suitable attack target then we should error out.
80 if (!creature->HasReactState(REACT_PASSIVE) && !creature->GetVictim())
81 {
82 TC_LOG_ERROR("misc", "DoZoneInCombat called for creature that has empty threat list (creature entry = %u)", creature->GetEntry());
83 return;
84 }
85
86 Map::PlayerList const& playerList = map->GetPlayers();
87
88 if (playerList.isEmpty())
89 return;
90
91 for (Map::PlayerList::const_iterator itr = playerList.begin(); itr != playerList.end(); ++itr)
92 {
93 if (Player* player = itr->GetSource())
94 {
95 if (player->IsGameMaster())
96 continue;
97
98 if (player->IsAlive())
99 {
100 creature->SetInCombatWith(player);
101 player->SetInCombatWith(creature);
102 creature->AddThreat(player, 0.0f);
103 }
104
105 /* Causes certain things to never leave the threat list (Priest Lightwell, etc):
106 for (Unit::ControlList::const_iterator itr = player->m_Controlled.begin(); itr != player->m_Controlled.end(); ++itr)
107 {
108 creature->SetInCombatWith(*itr);
109 (*itr)->SetInCombatWith(creature);
110 creature->AddThreat(*itr, 0.0f);
111 }*/
112 }
113 }
114}

143// Distract creature, if player gets too close while stealthed/prowling
144void CreatureAI::TriggerAlert(Unit const* who) const
145{
146 // If there's no target, or target isn't a player do nothing
147 if (!who || who->GetTypeId() != TYPEID_PLAYER)
148 return;
149
150 // If this unit isn't an NPC, is already distracted, is in combat, is confused, stunned or fleeing, do nothing
151 if (me->GetTypeId() != TYPEID_UNIT || me->IsInCombat() || me->HasUnitState(UNIT_STATE_CONFUSED | UNIT_STATE_STUNNED | UNIT_STATE_FLEEING | UNIT_STATE_DISTRACTED))
152 return;
153
154 // Only alert for hostiles!
155 if (me->IsCivilian() || me->HasReactState(REACT_PASSIVE) || !me->IsHostileTo(who) || !me->_IsTargetAcceptable(who))
156 return;
157
158 // Send alert sound (if any) for this creature
159 me->SendAIReaction(AI_REACTION_ALERT);
160
161 // Face the unit (stealthed player) and set distracted state for 5 seconds
162 me->SetFacingTo(me->GetAngle(who->GetPositionX(), who->GetPositionY()));
163 me->StopMoving();
164 me->GetMotionMaster()->MoveDistract(5 * IN_MILLISECONDS);
165}

167void CreatureAI::EnterEvadeMode()
168{
169 if (!_EnterEvadeMode())
170 return;
171
172 TC_LOG_DEBUG("entities.unit", "Creature %u enters evade mode.", me->GetEntry());
173
174 if (!me->GetVehicle()) // otherwise me will be in evade mode forever
175 {
176 if (Unit* owner = me->GetCharmerOrOwner())
177 {
178 me->GetMotionMaster()->Clear(false);
179 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, me->GetFollowAngle(), MOTION_SLOT_ACTIVE);
180 }
181 else
182 {
183 // Required to prevent attacking creatures that are evading and cause them to reenter combat
184 // Does not apply to MoveFollow
185 me->AddUnitState(UNIT_STATE_EVADE);
186 me->GetMotionMaster()->MoveTargetedHome();
187 }
188 }
189
190 Reset();
191
192 if (me->IsVehicle()) // use the same sequence of addtoworld, aireset may remove all summons!
193 me->GetVehicleKit()->Reset(true);
194}

211bool CreatureAI::UpdateVictimWithGaze()
212{
213 if (!me->IsInCombat())
214 return false;
215
216 if (me->HasReactState(REACT_PASSIVE))
217 {
218 if (me->GetVictim())
219 return true;
220 else
221 me->SetReactState(REACT_AGGRESSIVE);
222 }
223
224 if (Unit* victim = me->SelectVictim())
225 AttackStart(victim);
226
227 return me->GetVictim() != nullptr;
228}

230bool CreatureAI::UpdateVictim()
231{
232 if (!me->IsInCombat())
233 return false;
234
235 if (!me->HasReactState(REACT_PASSIVE))
236 {
237 if (Unit* victim = me->SelectVictim())
238 AttackStart(victim);
239
240 return me->GetVictim() != nullptr;
241 }
242 else if (me->getThreatManager().isThreatListEmpty())
243 {
244 EnterEvadeMode();
245 return false;
246 }
247
248 return true;
249}

251bool CreatureAI::_EnterEvadeMode()
252{
253 if (!me->IsAlive())
254 return false;
255
256 // don't remove vehicle auras, passengers aren't supposed to drop off the vehicle
257 // don't remove clone caster on evade (to be verified)
258 me->RemoveAllAurasExceptType(SPELL_AURA_CONTROL_VEHICLE, SPELL_AURA_CLONE_CASTER);
259
260 // sometimes bosses stuck in combat?
261 me->DeleteThreatList();
262 me->CombatStop(true);
263 me->LoadCreaturesAddon();
264 me->SetLootRecipient(NULL);
265 me->ResetPlayerDamageReq();
266 me->SetLastDamagedTime(0);
267
268 if (me->IsInEvadeMode())
269 return false;
270
271 return true;
272}

29template<typename First, typename Second, typename... Rest>
30static inline First const& RAND(First const& first, Second const& second, Rest const&... rest)
31{
32 std::reference_wrapper<typename std::add_const<First>::type> const pack[] = { first, second, rest... };
33 return pack[urand(0, sizeof...(rest) + 1)].get();
34}

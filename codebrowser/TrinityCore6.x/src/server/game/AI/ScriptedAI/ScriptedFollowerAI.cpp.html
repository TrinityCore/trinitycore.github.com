44void FollowerAI::AttackStart(Unit* who)
45{
46 if (!who)
47 return;
48
49 if (me->Attack(who, true))
50 {
51 me->AddThreat(who, 0.0f);
52 me->SetInCombatWith(who);
53 who->SetInCombatWith(me);
54
55 if (me->HasUnitState(UNIT_STATE_FOLLOW))
56 me->ClearUnitState(UNIT_STATE_FOLLOW);
57
58 if (IsCombatMovementAllowed())
59 me->GetMotionMaster()->MoveChase(who);
60 }
61}
62
63//This part provides assistance to a player that are attacked by who, even if out of normal aggro range
64//It will cause me to attack who that are attacking _any_ player (which has been confirmed may happen also on offi)
65//The flag (type_flag) is unconfirmed, but used here for further research and is a good candidate.
66bool FollowerAI::AssistPlayerInCombat(Unit* who)
67{
68 if (!who || !who->GetVictim())
69 return false;
70
71 //experimental (unknown) flag not present
72 if (!(me->GetCreatureTemplate()->type_flags & CREATURE_TYPEFLAGS_CAN_ASSIST))
73 return false;
74
75 //not a player
76 if (!who->EnsureVictim()->GetCharmerOrOwnerPlayerOrPlayerItself())
77 return false;
78
79 //never attack friendly
80 if (me->IsFriendlyTo(who))
81 return false;
82
83 //too far away and no free sight?
84 if (me->IsWithinDistInMap(who, MAX_PLAYER_DISTANCE) && me->IsWithinLOSInMap(who))
85 {
86 //already fighting someone?
87 if (!me->GetVictim())
88 {
89 AttackStart(who);
90 return true;
91 }
92 else
93 {
94 who->SetInCombatWith(me);
95 me->AddThreat(who, 0.0f);
96 return true;
97 }
98 }
99
100 return false;
101}

103void FollowerAI::MoveInLineOfSight(Unit* who)
104{
105 if (!me->HasUnitState(UNIT_STATE_STUNNED) && who->isTargetableForAttack() && who->isInAccessiblePlaceFor(me))
106 {
107 if (HasFollowState(STATE_FOLLOW_INPROGRESS) && AssistPlayerInCombat(who))
108 return;
109
110 if (!me->CanFly() && me->GetDistanceZ(who) > CREATURE_Z_ATTACK_RANGE)
111 return;
112
113 if (me->IsHostileTo(who))
114 {
115 float fAttackRadius = me->GetAttackDistance(who);
116 if (me->IsWithinDistInMap(who, fAttackRadius) && me->IsWithinLOSInMap(who))
117 {
118 if (!me->GetVictim())
119 {
120 // Clear distracted state on combat
121 if (me->HasUnitState(UNIT_STATE_DISTRACTED))
122 {
123 me->ClearUnitState(UNIT_STATE_DISTRACTED);
124 me->GetMotionMaster()->Clear();
125 }
126
127 AttackStart(who);
128 }
129 else if (me->GetMap()->IsDungeon())
130 {
131 who->SetInCombatWith(me);
132 me->AddThreat(who, 0.0f);
133 }
134 }
135 }
136 }

139void FollowerAI::JustDied(Unit* /*killer*/)
140{
141 if (!HasFollowState(STATE_FOLLOW_INPROGRESS) || !m_uiLeaderGUID || !m_pQuestForFollow)
142 return;
143
144 /// @todo need a better check for quests with time limit.
145 if (Player* player = GetLeaderForFollower())
146 {
147 if (Group* group = player->GetGroup())
148 {
149 for (GroupReference* groupRef = group->GetFirstMember(); groupRef != NULL; groupRef = groupRef->next())
150 {
151 if (Player* member = groupRef->GetSource())
152 {
153 if (member->GetQuestStatus(m_pQuestForFollow->GetQuestId()) == QUEST_STATUS_INCOMPLETE)
154 member->FailQuest(m_pQuestForFollow->GetQuestId());
155 }
156 }
157 }
158 else
159 {
160 if (player->GetQuestStatus(m_pQuestForFollow->GetQuestId()) == QUEST_STATUS_INCOMPLETE)
161 player->FailQuest(m_pQuestForFollow->GetQuestId());
162 }
163 }

166void FollowerAI::JustRespawned()
167{
168 m_uiFollowState = STATE_FOLLOW_NONE;
169
170 if (!IsCombatMovementAllowed())
171 SetCombatMovement(true);
172
173 if (me->getFaction() != me->GetCreatureTemplate()->faction)
174 me->setFaction(me->GetCreatureTemplate()->faction);
175

179void FollowerAI::EnterEvadeMode()
180{
181 me->RemoveAllAuras();
182 me->DeleteThreatList();
183 me->CombatStop(true);
184 me->SetLootRecipient(NULL);
185
186 if (HasFollowState(STATE_FOLLOW_INPROGRESS))
187 {
188 TC_LOG_DEBUG("scripts", "FollowerAI left combat, returning to CombatStartPosition.");
189
190 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
191 {
192 float fPosX, fPosY, fPosZ;
193 me->GetPosition(fPosX, fPosY, fPosZ);
194 me->GetMotionMaster()->MovePoint(POINT_COMBAT_START, fPosX, fPosY, fPosZ);
195 }
196 }
197 else
198 {
199 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
200 me->GetMotionMaster()->MoveTargetedHome();
201 }
202
203 Reset();
204}

206void FollowerAI::UpdateAI(uint32 uiDiff)
207{
208 if (HasFollowState(STATE_FOLLOW_INPROGRESS) && !me->GetVictim())
209 {
210 if (m_uiUpdateFollowTimer <= uiDiff)
211 {
212 if (HasFollowState(STATE_FOLLOW_COMPLETE) && !HasFollowState(STATE_FOLLOW_POSTEVENT))
213 {
214 TC_LOG_DEBUG("scripts", "FollowerAI is set completed, despawns.");
215 me->DespawnOrUnsummon();
216 return;
217 }
218
219 bool bIsMaxRangeExceeded = true;
220
221 if (Player* player = GetLeaderForFollower())
222 {
223 if (HasFollowState(STATE_FOLLOW_RETURNING))
224 {
225 TC_LOG_DEBUG("scripts", "FollowerAI is returning to leader.");
226
227 RemoveFollowState(STATE_FOLLOW_RETURNING);
228 me->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
229 return;
230 }
231
232 if (Group* group = player->GetGroup())
233 {
234 for (GroupReference* groupRef = group->GetFirstMember(); groupRef != NULL; groupRef = groupRef->next())
235 {
236 Player* member = groupRef->GetSource();
237
238 if (member && me->IsWithinDistInMap(member, MAX_PLAYER_DISTANCE))
239 {
240 bIsMaxRangeExceeded = false;
241 break;
242 }
243 }
244 }
245 else
246 {
247 if (me->IsWithinDistInMap(player, MAX_PLAYER_DISTANCE))
248 bIsMaxRangeExceeded = false;
249 }
250 }
251
252 if (bIsMaxRangeExceeded)
253 {
254 TC_LOG_DEBUG("scripts", "FollowerAI failed because player/group was to far away or not found");
255 me->DespawnOrUnsummon();
256 return;
257 }
258
259 m_uiUpdateFollowTimer = 1000;
260 }
261 else
262 m_uiUpdateFollowTimer -= uiDiff;
263 }
264
265 UpdateFollowerAI(uiDiff);
266}

276void FollowerAI::MovementInform(uint32 motionType, uint32 pointId)
277{
278 if (motionType != POINT_MOTION_TYPE || !HasFollowState(STATE_FOLLOW_INPROGRESS))
279 return;
280
281 if (pointId == POINT_COMBAT_START)
282 {
283 if (GetLeaderForFollower())
284 {
285 if (!HasFollowState(STATE_FOLLOW_PAUSED))
286 AddFollowState(STATE_FOLLOW_RETURNING);
287 }
288 else
289 me->DespawnOrUnsummon();
290 }

293void FollowerAI::StartFollow(Player* player, uint32 factionForFollower, const Quest* quest)
294{
295 if (me->GetVictim())
296 {
297 TC_LOG_DEBUG("scripts", "FollowerAI attempt to StartFollow while in combat.");
298 return;
299 }
300
301 if (HasFollowState(STATE_FOLLOW_INPROGRESS))
302 {
303 TC_LOG_ERROR("scripts", "FollowerAI attempt to StartFollow while already following.");
304 return;
305 }
306
307 //set variables
308 m_uiLeaderGUID = player->GetGUID();
309
310 if (factionForFollower)
311 me->setFaction(factionForFollower);
312
313 m_pQuestForFollow = quest;
314
315 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
316 {
317 me->GetMotionMaster()->Clear();
318 me->GetMotionMaster()->MoveIdle();
319 TC_LOG_DEBUG("scripts", "FollowerAI start with WAYPOINT_MOTION_TYPE, set to MoveIdle.");
320 }
321
322 me->SetUInt64Value(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_NONE);
323
324 AddFollowState(STATE_FOLLOW_INPROGRESS);
325
326 me->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
327
328 TC_LOG_DEBUG("scripts", "FollowerAI start follow %s (%s)", player->GetName().c_str(), m_uiLeaderGUID.ToString().c_str());
329}

331Player* FollowerAI::GetLeaderForFollower()
332{
333 if (Player* player = ObjectAccessor::GetPlayer(*me, m_uiLeaderGUID))
334 {
335 if (player->IsAlive())
336 return player;
337 else
338 {
339 if (Group* group = player->GetGroup())
340 {
341 for (GroupReference* groupRef = group->GetFirstMember(); groupRef != NULL; groupRef = groupRef->next())
342 {
343 Player* member = groupRef->GetSource();
344
345 if (member && member->IsAlive() && me->IsWithinDistInMap(member, MAX_PLAYER_DISTANCE))
346 {
347 TC_LOG_DEBUG("scripts", "FollowerAI GetLeader changed and returned new leader.");
348 m_uiLeaderGUID = member->GetGUID();
349 return member;
350 }
351 }
352 }
353 }
354 }
355
356 TC_LOG_DEBUG("scripts", "FollowerAI GetLeader can not find suitable leader.");
357 return NULL;
358}

360void FollowerAI::SetFollowComplete(bool bWithEndEvent)
361{
362 if (me->HasUnitState(UNIT_STATE_FOLLOW))
363 {
364 me->ClearUnitState(UNIT_STATE_FOLLOW);
365
366 me->StopMoving();
367 me->GetMotionMaster()->Clear();
368 me->GetMotionMaster()->MoveIdle();
369 }
370
371 if (bWithEndEvent)
372 AddFollowState(STATE_FOLLOW_POSTEVENT);
373 else
374 {
375 if (HasFollowState(STATE_FOLLOW_POSTEVENT))
376 RemoveFollowState(STATE_FOLLOW_POSTEVENT);
377 }
378
379 AddFollowState(STATE_FOLLOW_COMPLETE);
380}

382void FollowerAI::SetFollowPaused(bool paused)
383{
384 if (!HasFollowState(STATE_FOLLOW_INPROGRESS) || HasFollowState(STATE_FOLLOW_COMPLETE))
385 return;
386
387 if (paused)
388 {
389 AddFollowState(STATE_FOLLOW_PAUSED);
390
391 if (me->HasUnitState(UNIT_STATE_FOLLOW))
392 {
393 me->ClearUnitState(UNIT_STATE_FOLLOW);
394
395 me->StopMoving();
396 me->GetMotionMaster()->Clear();
397 me->GetMotionMaster()->MoveIdle();
398 }
399 }
400 else
401 {
402 RemoveFollowState(STATE_FOLLOW_PAUSED);
403
404 if (Player* leader = GetLeaderForFollower())
405 me->GetMotionMaster()->MoveFollow(leader, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
406 }

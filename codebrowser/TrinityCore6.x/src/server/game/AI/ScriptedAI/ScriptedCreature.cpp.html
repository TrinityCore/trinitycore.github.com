34void SummonList::DoZoneInCombat(uint32 entry)
35{
36 for (StorageType::iterator i = storage_.begin(); i != storage_.end();)
37 {
38 Creature* summon = ObjectAccessor::GetCreature(*me, *i);
39 ++i;
40 if (summon && summon->IsAIEnabled
41 && (!entry || summon->GetEntry() == entry))
42 {
43 summon->AI()->DoZoneInCombat();
44 }
45 }

48void SummonList::DespawnEntry(uint32 entry)
49{
50 for (StorageType::iterator i = storage_.begin(); i != storage_.end();)
51 {
52 Creature* summon = ObjectAccessor::GetCreature(*me, *i);
53 if (!summon)
54 i = storage_.erase(i);
55 else if (summon->GetEntry() == entry)
56 {
57 i = storage_.erase(i);
58 summon->DespawnOrUnsummon();
59 }
60 else
61 ++i;
62 }
63}

65void SummonList::DespawnAll()
66{
67 while (!storage_.empty())
68 {
69 Creature* summon = ObjectAccessor::GetCreature(*me, storage_.front());
70 storage_.pop_front();
71 if (summon)
72 summon->DespawnOrUnsummon();
73 }

182SpellInfo const* ScriptedAI::SelectSpell(Unit* target, uint32 school, uint32 mechanic, SelectTargetType targets, float rangeMin, float rangeMax, SelectEffect effect)
183{
184 //No target so we can't cast
185 if (!target)
186 return NULL;
187
188 //Silenced so we can't cast
189 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SILENCED))
190 return NULL;
191
192 //Using the extended script system we first create a list of viable spells
193 SpellInfo const* apSpell[CREATURE_MAX_SPELLS];
194 memset(apSpell, 0, CREATURE_MAX_SPELLS * sizeof(SpellInfo*));
195
196 uint32 spellCount = 0;
197
198 SpellInfo const* tempSpell = NULL;
199
200 //Check if each spell is viable(set it to null if not)
201 for (uint32 i = 0; i < CREATURE_MAX_SPELLS; i++)
202 {
203 tempSpell = sSpellMgr->GetSpellInfo(me->m_spells[i]);
204
205 //This spell doesn't exist
206 if (!tempSpell)
207 continue;
208
209 // Targets and Effects checked first as most used restrictions
210 //Check the spell targets if specified
211 if (targets && !(SpellSummary[me->m_spells[i]].Targets & (1 << (targets-1))))
212 continue;
213
214 //Check the type of spell if we are looking for a specific spell type
215 if (effect && !(SpellSummary[me->m_spells[i]].Effects & (1 << (effect-1))))
216 continue;
217
218 //Check for school if specified
219 if (school && (tempSpell->SchoolMask & school) == 0)
220 continue;
221
222 //Check for spell mechanic if specified
223 if (mechanic && tempSpell->Mechanic != mechanic)
224 continue;
225
226 //Check if the spell meets our range requirements
227 if (rangeMin && me->GetSpellMinRangeForTarget(target, tempSpell) < rangeMin)
228 continue;
229 if (rangeMax && me->GetSpellMaxRangeForTarget(target, tempSpell) > rangeMax)
230 continue;
231
232 //Check if our target is in range
233 if (me->IsWithinDistInMap(target, float(me->GetSpellMinRangeForTarget(target, tempSpell))) || !me->IsWithinDistInMap(target, float(me->GetSpellMaxRangeForTarget(target, tempSpell))))
234 continue;
235
236 //All good so lets add it to the spell list
237 apSpell[spellCount] = tempSpell;
238 ++spellCount;
239 }
240
241 //We got our usable spells so now lets randomly pick one
242 if (!spellCount)
243 return NULL;
244
245 return apSpell[urand(0, spellCount - 1)];
246}

248void ScriptedAI::DoResetThreat()
249{
250 if (!me->CanHaveThreatList() || me->getThreatManager().isThreatListEmpty())
251 {
252 TC_LOG_ERROR("scripts", "DoResetThreat called for creature that either cannot have threat list or has empty threat list (me entry = %d)", me->GetEntry());
253 return;
254 }
255
256 ThreatContainer::StorageType threatlist = me->getThreatManager().getThreatList();
257
258 for (ThreatContainer::StorageType::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
259 {
260 Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid());
261 if (unit && DoGetThreat(unit))
262 DoModifyThreatPercent(unit, -100);
263 }
264}

347Player* ScriptedAI::GetPlayerAtMinimumRange(float minimumRange)
348{
349 Player* player = NULL;
350
351 CellCoord pair(Trinity::ComputeCellCoord(me->GetPositionX(), me->GetPositionY()));
352 Cell cell(pair);
353 cell.SetNoCreate();
354
355 Trinity::PlayerAtMinimumRangeAway check(me, minimumRange);
356 Trinity::PlayerSearcher<Trinity::PlayerAtMinimumRangeAway> searcher(me, player, check);
357 TypeContainerVisitor<Trinity::PlayerSearcher<Trinity::PlayerAtMinimumRangeAway>, GridTypeMapContainer> visitor(searcher);
358
359 cell.Visit(pair, visitor, *me->GetMap(), *me, minimumRange);
360
361 return player;
362}

395// Hacklike storage used for misc creatures that are expected to evade of outside of a certain area.
396// It is assumed the information is found elswehere and can be handled by the core. So far no luck finding such information/way to extract it.
397bool ScriptedAI::EnterEvadeIfOutOfCombatArea(uint32 const diff)
398{
399 if (_evadeCheckCooldown <= diff)
400 _evadeCheckCooldown = 2500;
401 else
402 {
403 _evadeCheckCooldown -= diff;
404 return false;
405 }
406
407 if (me->IsInEvadeMode() || !me->GetVictim())
408 return false;
409
410 float x = me->GetPositionX();
411 float y = me->GetPositionY();
412 float z = me->GetPositionZ();
413
414 switch (me->GetEntry())
415 {
416 case NPC_BROODLORD: // broodlord (not move down stairs)
417 if (z > 448.60f)
418 return false;
419 break;
420 case NPC_VOID_REAVER: // void reaver (calculate from center of room)
421 if (me->GetDistance2d(432.59f, 371.93f) < 105.0f)
422 return false;
423 break;
424 case NPC_JAN_ALAI: // jan'alai (calculate by Z)
425 if (z > 12.0f)
426 return false;
427 break;
428 case NPC_SARTHARION: // sartharion (calculate box)
429 if (x > 3218.86f && x < 3275.69f && y < 572.40f && y > 484.68f)
430 return false;
431 break;
432 default: // For most of creatures that certain area is their home area.
433 TC_LOG_INFO("misc", "TSCR: EnterEvadeIfOutOfCombatArea used for creature entry %u, but does not have any definition. Using the default one.", me->GetEntry());
434 uint32 homeAreaId = me->GetMap()->GetAreaId(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY(), me->GetHomePosition().GetPositionZ());
435 if (me->GetAreaId() == homeAreaId)
436 return false;
437 }
438
439 EnterEvadeMode();
440 return true;
441}

499{
500 float x, y, z;
501 me->GetPosition(x, y, z);
502
503 ThreatContainer::StorageType threatList = me->getThreatManager().getThreatList();
504 for (ThreatContainer::StorageType::const_iterator itr = threatList.begin(); itr != threatList.end(); ++itr)

510bool BossAI::CheckBoundary(Unit* who)
511{
512 if (!GetBoundary() || !who)
513 return true;
514
515 for (BossBoundaryMap::const_iterator itr = GetBoundary()->begin(); itr != GetBoundary()->end(); ++itr)
516 {
517 switch (itr->first)
518 {
519 case BOUNDARY_N:
520 if (who->GetPositionX() > itr->second)
521 return false;
522 break;
523 case BOUNDARY_S:
524 if (who->GetPositionX() < itr->second)
525 return false;
526 break;
527 case BOUNDARY_E:
528 if (who->GetPositionY() < itr->second)
529 return false;
530 break;
531 case BOUNDARY_W:
532 if (who->GetPositionY() > itr->second)
533 return false;
534 break;
535 case BOUNDARY_NW:
536 if (who->GetPositionX() + who->GetPositionY() > itr->second)
537 return false;
538 break;
539 case BOUNDARY_SE:
540 if (who->GetPositionX() + who->GetPositionY() < itr->second)
541 return false;
542 break;
543 case BOUNDARY_NE:
544 if (who->GetPositionX() - who->GetPositionY() > itr->second)
545 return false;
546 break;
547 case BOUNDARY_SW:
548 if (who->GetPositionX() - who->GetPositionY() < itr->second)
549 return false;
550 break;
551 default:
552 break;
553 }
554 }
555
556 return true;
557}

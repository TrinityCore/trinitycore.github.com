31void SmartWaypointMgr::LoadFromDB()
32{
33 uint32 oldMSTime = getMSTime();
34
35 for (std::unordered_map<uint32, WPPath*>::iterator itr = waypoint_map.begin(); itr != waypoint_map.end(); ++itr)
36 {
37 for (WPPath::iterator pathItr = itr->second->begin(); pathItr != itr->second->end(); ++pathItr)
38 delete pathItr->second;
39
40 delete itr->second;
41 }
42
43 waypoint_map.clear();
44
45 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_SMARTAI_WP);
46 PreparedQueryResult result = WorldDatabase.Query(stmt);
47
48 if (!result)
49 {
50 TC_LOG_INFO("server.loading", ">> Loaded 0 SmartAI Waypoint Paths. DB table `waypoints` is empty.");
51
52 return;
53 }
54
55 uint32 count = 0;
56 uint32 total = 0;
57 uint32 last_entry = 0;
58 uint32 last_id = 1;
59
60 do
61 {
62 Field* fields = result->Fetch();
63 uint32 entry = fields[0].GetUInt32();
64 uint32 id = fields[1].GetUInt32();
65 float x, y, z;
66 x = fields[2].GetFloat();
67 y = fields[3].GetFloat();
68 z = fields[4].GetFloat();
69
70 if (last_entry != entry)
71 {
72 waypoint_map[entry] = new WPPath();
73 last_id = 1;
74 count++;
75 }
76
77 if (last_id != id)
78 TC_LOG_ERROR("sql.sql", "SmartWaypointMgr::LoadFromDB: Path entry %u, unexpected point id %u, expected %u.", entry, id, last_id);
79
80 last_id++;
81 (*waypoint_map[entry])[id] = new WayPoint(id, x, y, z);
82
83 last_entry = entry;
84 total++;
85 }
86 while (result->NextRow());
87
88 TC_LOG_INFO("server.loading", ">> Loaded %u SmartAI waypoint paths (total %u waypoints) in %u ms", count, total, GetMSTimeDiffToNow(oldMSTime));
89}

102void SmartAIMgr::LoadSmartAIFromDB()
103{
104 LoadHelperStores();
105
106 uint32 oldMSTime = getMSTime();
107
108 for (uint8 i = 0; i < SMART_SCRIPT_TYPE_MAX; i++)
109 mEventMap[i].clear(); //Drop Existing SmartAI List
110
111 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_SMART_SCRIPTS);
112 PreparedQueryResult result = WorldDatabase.Query(stmt);
113
114 if (!result)
115 {
116 TC_LOG_INFO("server.loading", ">> Loaded 0 SmartAI scripts. DB table `smartai_scripts` is empty.");
117 return;
118 }
119
120 uint32 count = 0;
121
122 do
123 {
124 Field* fields = result->Fetch();
125
126 SmartScriptHolder temp;
127
128 temp.entryOrGuid = fields[0].GetInt64();
129 SmartScriptType source_type = (SmartScriptType)fields[1].GetUInt8();
130 if (source_type >= SMART_SCRIPT_TYPE_MAX)
131 {
132 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: invalid source_type (%u), skipped loading.", uint32(source_type));
133 continue;
134 }
135 if (temp.entryOrGuid >= 0)
136 {
137 switch (source_type)
138 {
139 case SMART_SCRIPT_TYPE_CREATURE:
140 {
141 CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate((uint32)temp.entryOrGuid);
142 if (!creatureInfo)
143 {
144 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) does not exist, skipped loading.", uint32(temp.entryOrGuid));
145 continue;
146 }
147
148 if (creatureInfo->AIName != "SmartAI")
149 {
150 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) is not using SmartAI, skipped loading.", uint32(temp.entryOrGuid));
151 continue;
152 }
153 break;
154 }
155 case SMART_SCRIPT_TYPE_GAMEOBJECT:
156 {
157 GameObjectTemplate const* gameObjectInfo = sObjectMgr->GetGameObjectTemplate((uint32)temp.entryOrGuid);
158 if (!gameObjectInfo)
159 {
160 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: GameObject entry (%u) does not exist, skipped loading.", uint32(temp.entryOrGuid));
161 continue;
162 }
163
164 if (gameObjectInfo->AIName != "SmartGameObjectAI")
165 {
166 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: GameObject entry (%u) is not using SmartGameObjectAI, skipped loading.", uint32(temp.entryOrGuid));
167 continue;
168 }
169 break;
170 }
171 case SMART_SCRIPT_TYPE_AREATRIGGER:
172 {
173 if (!sAreaTriggerStore.LookupEntry((uint32)temp.entryOrGuid))
174 {
175 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: AreaTrigger entry (%u) does not exist, skipped loading.", uint32(temp.entryOrGuid));
176 continue;
177 }
178 break;
179 }
180 case SMART_SCRIPT_TYPE_TIMED_ACTIONLIST:
181 break;//nothing to check, really
182 default:
183 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: not yet implemented source_type %u", (uint32)source_type);
184 continue;
185 }
186 }
187 else
188 {
189 CreatureData const* creature = sObjectMgr->GetCreatureData(uint64(-temp.entryOrGuid));
190 if (!creature)
191 {
192 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature guid (" SI64FMTD ") does not exist, skipped loading.", -temp.entryOrGuid);
193 continue;
194 }
195
196 CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate(creature->id);
197 if (!creatureInfo)
198 {
199 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) guid (" SI64FMTD ") does not exist, skipped loading.", creature->id, -temp.entryOrGuid);
200 continue;
201 }
202
203 if (creatureInfo->AIName != "SmartAI")
204 {
205 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) guid (" SI64FMTD ") is not using SmartAI, skipped loading.", creature->id, -temp.entryOrGuid);
206 continue;
207 }
208 }
209
210 temp.source_type = source_type;
211 temp.event_id = fields[2].GetUInt16();
212 temp.link = fields[3].GetUInt16();
213 temp.event.type = (SMART_EVENT)fields[4].GetUInt8();
214 temp.event.event_phase_mask = fields[5].GetUInt8();
215 temp.event.event_chance = fields[6].GetUInt8();
216 temp.event.event_flags = fields[7].GetUInt8();
217
218 temp.event.raw.param1 = fields[8].GetUInt32();
219 temp.event.raw.param2 = fields[9].GetUInt32();
220 temp.event.raw.param3 = fields[10].GetUInt32();
221 temp.event.raw.param4 = fields[11].GetUInt32();
222
223 temp.action.type = (SMART_ACTION)fields[12].GetUInt8();
224 temp.action.raw.param1 = fields[13].GetUInt32();
225 temp.action.raw.param2 = fields[14].GetUInt32();
226 temp.action.raw.param3 = fields[15].GetUInt32();
227 temp.action.raw.param4 = fields[16].GetUInt32();
228 temp.action.raw.param5 = fields[17].GetUInt32();
229 temp.action.raw.param6 = fields[18].GetUInt32();
230
231 temp.target.type = (SMARTAI_TARGETS)fields[19].GetUInt8();
232 temp.target.raw.param1 = fields[20].GetUInt32();
233 temp.target.raw.param2 = fields[21].GetUInt32();
234 temp.target.raw.param3 = fields[22].GetUInt32();
235 temp.target.x = fields[23].GetFloat();
236 temp.target.y = fields[24].GetFloat();
237 temp.target.z = fields[25].GetFloat();
238 temp.target.o = fields[26].GetFloat();
239
240 //check target
241 if (!IsTargetValid(temp))
242 continue;
243
244 // check all event and action params
245 if (!IsEventValid(temp))
246 continue;
247
248 // creature entry / guid not found in storage, create empty event list for it and increase counters
249 if (mEventMap[source_type].find(temp.entryOrGuid) == mEventMap[source_type].end())
250 {
251 ++count;
252 SmartAIEventList eventList;
253 mEventMap[source_type][temp.entryOrGuid] = eventList;
254 }
255 // store the new event
256 mEventMap[source_type][temp.entryOrGuid].push_back(temp);
257 }
258 while (result->NextRow());
259
260 // Post Loading Validation
261 for (uint8 i = 0; i < SMART_SCRIPT_TYPE_MAX; ++i)
262 {
263 for (SmartAIEventMap::iterator itr = mEventMap[i].begin(); itr != mEventMap[i].end(); ++itr)
264 {
265 for (SmartScriptHolder const& e : itr->second)
266 {
267 if (e.link)
268 {
269 if (!FindLinkedEvent(itr->second, e.link))
270 {
271 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Entry " SI64FMTD " SourceType %u, Event %u, Link Event %u not found or invalid.",
272 e.entryOrGuid, e.GetScriptType(), e.event_id, e.link);
273 }
274 }
275
276 if (e.GetEventType() == SMART_EVENT_LINK)
277 {
278 if (!FindLinkedSourceEvent(itr->second, e.event_id))
279 {
280 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Entry " SI64FMTD " SourceType %u, Event %u, Link Source Event not found or invalid. Event will never trigger.",
281 e.entryOrGuid, e.GetScriptType(), e.event_id);
282 }
283 }
284 }
285 }
286 }
287
288 TC_LOG_INFO("server.loading", ">> Loaded %u SmartAI scripts in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
289
290 UnLoadHelperStores();
291}

293bool SmartAIMgr::IsTargetValid(SmartScriptHolder const& e)
294{
295 if (std::abs(e.target.o) > 2 * float(M_PI))
296 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u has abs(`target.o` = %f) > 2*PI (orientation is expressed in radians)",
297 e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.o);
298
299 if (e.GetActionType() == SMART_ACTION_INSTALL_AI_TEMPLATE)
300 return true; // AI template has special handling
301
302 switch (e.GetTargetType())
303 {
304 case SMART_TARGET_CREATURE_DISTANCE:
305 case SMART_TARGET_CREATURE_RANGE:
306 {
307 if (e.target.unitDistance.creature && !sObjectMgr->GetCreatureTemplate(e.target.unitDistance.creature))
308 {
309 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses non-existent Creature entry %u as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.unitDistance.creature);
310 return false;
311 }
312 break;
313 }
314 case SMART_TARGET_GAMEOBJECT_DISTANCE:
315 case SMART_TARGET_GAMEOBJECT_RANGE:
316 {
317 if (e.target.goDistance.entry && !sObjectMgr->GetGameObjectTemplate(e.target.goDistance.entry))
318 {
319 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses non-existent GameObject entry %u as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.goDistance.entry);
320 return false;
321 }
322 break;
323 }
324 case SMART_TARGET_CREATURE_GUID:
325 {
326 if (e.target.unitGUID.entry && !IsCreatureValid(e, e.target.unitGUID.entry))
327 return false;
328 break;
329 }
330 case SMART_TARGET_GAMEOBJECT_GUID:
331 {
332 if (e.target.goGUID.entry && !IsGameObjectValid(e, e.target.goGUID.entry))
333 return false;
334 break;
335 }
336 case SMART_TARGET_PLAYER_DISTANCE:
337 case SMART_TARGET_CLOSEST_PLAYER:
338 {
339 if (e.target.playerDistance.dist == 0)
340 {
341 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u has maxDist 0 as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType());
342 return false;
343 }
344 break;
345 }
346 case SMART_TARGET_PLAYER_RANGE:
347 case SMART_TARGET_SELF:
348 case SMART_TARGET_VICTIM:
349 case SMART_TARGET_HOSTILE_SECOND_AGGRO:
350 case SMART_TARGET_HOSTILE_LAST_AGGRO:
351 case SMART_TARGET_HOSTILE_RANDOM:
352 case SMART_TARGET_HOSTILE_RANDOM_NOT_TOP:
353 case SMART_TARGET_ACTION_INVOKER:
354 case SMART_TARGET_INVOKER_PARTY:
355 case SMART_TARGET_POSITION:
356 case SMART_TARGET_NONE:
357 case SMART_TARGET_ACTION_INVOKER_VEHICLE:
358 case SMART_TARGET_OWNER_OR_SUMMONER:
359 case SMART_TARGET_THREAT_LIST:
360 case SMART_TARGET_CLOSEST_GAMEOBJECT:
361 case SMART_TARGET_CLOSEST_CREATURE:
362 case SMART_TARGET_CLOSEST_ENEMY:
363 case SMART_TARGET_CLOSEST_FRIENDLY:
364 case SMART_TARGET_STORED:
365 break;
366 default:
367 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Not handled target_type(%u), Entry " SI64FMTD " SourceType %u Event %u Action %u, skipped.", e.GetTargetType(), e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType());
368 return false;
369 }
370 return true;
371}

373bool SmartAIMgr::IsEventValid(SmartScriptHolder& e)
374{
375 if (e.event.type >= SMART_EVENT_END)
376 {
377 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid " SI64FMTD " using event(%u) has invalid event type (%u), skipped.", e.entryOrGuid, e.event_id, e.GetEventType());
378 return false;
379 }
380
381 // in SMART_SCRIPT_TYPE_TIMED_ACTIONLIST all event types are overriden by core
382 if (e.GetScriptType() != SMART_SCRIPT_TYPE_TIMED_ACTIONLIST && !(SmartAIEventMask[e.event.type][1] & SmartAITypeMask[e.GetScriptType()][1]))
383 {
384 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid " SI64FMTD ", event type %u can not be used for Script type %u", e.entryOrGuid, e.GetEventType(), e.GetScriptType());
385 return false;
386 }
387
388 if (e.action.type <= 0 || e.action.type >= SMART_ACTION_END)
389 {
390 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid " SI64FMTD " using event(%u) has invalid action type (%u), skipped.", e.entryOrGuid, e.event_id, e.GetActionType());
391 return false;
392 }
393
394 if (e.event.event_phase_mask > SMART_EVENT_PHASE_ALL)
395 {
396 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid " SI64FMTD " using event(%u) has invalid phase mask (%u), skipped.", e.entryOrGuid, e.event_id, e.event.event_phase_mask);
397 return false;
398 }
399
400 if (e.event.event_flags > SMART_EVENT_FLAGS_ALL)
401 {
402 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid " SI64FMTD " using event(%u) has invalid event flags (%u), skipped.", e.entryOrGuid, e.event_id, e.event.event_flags);
403 return false;
404 }
405
406 if (e.link && e.link == e.event_id)
407 {
408 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid " SI64FMTD " SourceType %u, Event %u, Event is linking self (infinite loop), skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id);
409 return false;
410 }
411
412 if (e.GetScriptType() == SMART_SCRIPT_TYPE_TIMED_ACTIONLIST)
413 {
414 e.event.type = SMART_EVENT_UPDATE_OOC;//force default OOC, can change when calling the script!
415 if (!IsMinMaxValid(e, e.event.minMaxRepeat.min, e.event.minMaxRepeat.max))
416 return false;
417
418 if (!IsMinMaxValid(e, e.event.minMaxRepeat.repeatMin, e.event.minMaxRepeat.repeatMax))
419 return false;
420 }
421 else
422 {
423 switch (e.GetEventType())
424 {
425 case SMART_EVENT_UPDATE:
426 case SMART_EVENT_UPDATE_IC:
427 case SMART_EVENT_UPDATE_OOC:
428 case SMART_EVENT_HEALT_PCT:
429 case SMART_EVENT_MANA_PCT:
430 case SMART_EVENT_TARGET_HEALTH_PCT:
431 case SMART_EVENT_TARGET_MANA_PCT:
432 case SMART_EVENT_RANGE:
433 case SMART_EVENT_DAMAGED:
434 case SMART_EVENT_DAMAGED_TARGET:
435 case SMART_EVENT_RECEIVE_HEAL:
436 if (!IsMinMaxValid(e, e.event.minMaxRepeat.min, e.event.minMaxRepeat.max))
437 return false;
438
439 if (!IsMinMaxValid(e, e.event.minMaxRepeat.repeatMin, e.event.minMaxRepeat.repeatMax))
440 return false;
441 break;
442 case SMART_EVENT_SPELLHIT:
443 case SMART_EVENT_SPELLHIT_TARGET:
444 if (e.event.spellHit.spell)
445 {
446 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(e.event.spellHit.spell);
447 if (!spellInfo)
448 {
449 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses non-existent Spell entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.spellHit.spell);
450 return false;
451 }
452 if (e.event.spellHit.school && (e.event.spellHit.school & spellInfo->SchoolMask) != spellInfo->SchoolMask)
453 {
454 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses Spell entry %u with invalid school mask, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.spellHit.spell);
455 return false;
456 }
457 }
458 if (!IsMinMaxValid(e, e.event.spellHit.cooldownMin, e.event.spellHit.cooldownMax))
459 return false;
460 break;
461 case SMART_EVENT_OOC_LOS:
462 case SMART_EVENT_IC_LOS:
463 if (!IsMinMaxValid(e, e.event.los.cooldownMin, e.event.los.cooldownMax))
464 return false;
465 break;
466 case SMART_EVENT_RESPAWN:
467 if (e.event.respawn.type == SMART_SCRIPT_RESPAWN_CONDITION_MAP && !sMapStore.LookupEntry(e.event.respawn.map))
468 {
469 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses non-existent Map entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.respawn.map);
470 return false;
471 }
472 if (e.event.respawn.type == SMART_SCRIPT_RESPAWN_CONDITION_AREA && !sAreaTableStore.LookupEntry(e.event.respawn.area))
473 {
474 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses non-existent Area entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.respawn.area);
475 return false;
476 }
477 break;
478 case SMART_EVENT_FRIENDLY_HEALTH:
479 if (!NotNULL(e, e.event.friendlyHealth.radius))
480 return false;
481
482 if (!IsMinMaxValid(e, e.event.friendlyHealth.repeatMin, e.event.friendlyHealth.repeatMax))
483 return false;
484 break;
485 case SMART_EVENT_FRIENDLY_IS_CC:
486 if (!IsMinMaxValid(e, e.event.friendlyCC.repeatMin, e.event.friendlyCC.repeatMax))
487 return false;
488 break;
489 case SMART_EVENT_FRIENDLY_MISSING_BUFF:
490 {
491 if (!IsSpellValid(e, e.event.missingBuff.spell))
492 return false;
493
494 if (!NotNULL(e, e.event.missingBuff.radius))
495 return false;
496
497 if (!IsMinMaxValid(e, e.event.missingBuff.repeatMin, e.event.missingBuff.repeatMax))
498 return false;
499 break;
500 }
501 case SMART_EVENT_KILL:
502 if (!IsMinMaxValid(e, e.event.kill.cooldownMin, e.event.kill.cooldownMax))
503 return false;
504
505 if (e.event.kill.creature && !IsCreatureValid(e, e.event.kill.creature))
506 return false;
507 break;
508 case SMART_EVENT_VICTIM_CASTING:
509 if (e.event.targetCasting.spellId > 0 && !sSpellMgr->GetSpellInfo(e.event.targetCasting.spellId))
510 {
511 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses non-existent Spell entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.spellHit.spell);
512 return false;
513 }
514
515 if (!IsMinMaxValid(e, e.event.targetCasting.repeatMin, e.event.targetCasting.repeatMax))
516 return false;
517 break;
518 case SMART_EVENT_PASSENGER_BOARDED:
519 case SMART_EVENT_PASSENGER_REMOVED:
520 if (!IsMinMaxValid(e, e.event.minMax.repeatMin, e.event.minMax.repeatMax))
521 return false;
522 break;
523 case SMART_EVENT_SUMMON_DESPAWNED:
524 case SMART_EVENT_SUMMONED_UNIT:
525 if (e.event.summoned.creature && !IsCreatureValid(e, e.event.summoned.creature))
526 return false;
527
528 if (!IsMinMaxValid(e, e.event.summoned.cooldownMin, e.event.summoned.cooldownMax))
529 return false;
530 break;
531 case SMART_EVENT_ACCEPTED_QUEST:
532 case SMART_EVENT_REWARD_QUEST:
533 if (e.event.quest.quest && !IsQuestValid(e, e.event.quest.quest))
534 return false;
535 break;
536 case SMART_EVENT_RECEIVE_EMOTE:
537 {
538 if (e.event.emote.emote && !IsTextEmoteValid(e, e.event.emote.emote))
539 return false;
540
541 if (!IsMinMaxValid(e, e.event.emote.cooldownMin, e.event.emote.cooldownMax))
542 return false;
543 break;
544 }
545 case SMART_EVENT_HAS_AURA:
546 case SMART_EVENT_TARGET_BUFFED:
547 {
548 if (!IsSpellValid(e, e.event.aura.spell))
549 return false;
550
551 if (!IsMinMaxValid(e, e.event.aura.repeatMin, e.event.aura.repeatMax))
552 return false;
553 break;
554 }
555 case SMART_EVENT_TRANSPORT_ADDCREATURE:
556 {
557 if (e.event.transportAddCreature.creature && !IsCreatureValid(e, e.event.transportAddCreature.creature))
558 return false;
559 break;
560 }
561 case SMART_EVENT_MOVEMENTINFORM:
562 {
563 if (e.event.movementInform.type > NULL_MOTION_TYPE)
564 {
565 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry " SI64FMTD " SourceType %u Event %u Action %u uses invalid Motion type %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.movementInform.type);
566 return false;
567 }
568 break;
569 }
570 case SMART_EVENT_DATA_SET:
571 {
572 if (!IsMinMaxValid(e, e.event.dataSet.cooldownMin, e.event.dataSet.cooldownMax))
573 return false;
574 break;
575 }

1315 SummonGameObjectSpellStore.insert(std::make_pair(uint32(effect->MiscValue), std::make_pair(i, SpellEffIndex(effect->EffectIndex))));

1318 KillCreditSpellStore.insert(std::make_pair(uint32(effect->MiscValue), std::make_pair(i, SpellEffIndex(effect->EffectIndex))));

1321 CreateItemSpellStore.insert(std::make_pair(uint32(effect->ItemType), std::make_pair(i, SpellEffIndex(effect->EffectIndex))));

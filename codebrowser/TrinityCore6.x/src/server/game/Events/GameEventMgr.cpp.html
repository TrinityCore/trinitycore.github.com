32bool GameEventMgr::CheckOneGameEvent(uint16 entry) const
33{
34 switch (mGameEvent[entry].state)
35 {
36 default:
37 case GAMEEVENT_NORMAL:
38 {
39 time_t currenttime = time(NULL);
40 // Get the event information
41 return mGameEvent[entry].start < currenttime
42 && currenttime < mGameEvent[entry].end
43 && (currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * MINUTE) < mGameEvent[entry].length * MINUTE;
44 }
45 // if the state is conditions or nextphase, then the event should be active
46 case GAMEEVENT_WORLD_CONDITIONS:
47 case GAMEEVENT_WORLD_NEXTPHASE:
48 return true;
49 // finished world events are inactive
50 case GAMEEVENT_WORLD_FINISHED:
51 case GAMEEVENT_INTERNAL:
52 return false;
53 // if inactive world event, check the prerequisite events
54 case GAMEEVENT_WORLD_INACTIVE:
55 {
56 time_t currenttime = time(NULL);
57 for (std::set<uint16>::const_iterator itr = mGameEvent[entry].prerequisite_events.begin(); itr != mGameEvent[entry].prerequisite_events.end(); ++itr)
58 {
59 if ((mGameEvent[*itr].state != GAMEEVENT_WORLD_NEXTPHASE && mGameEvent[*itr].state != GAMEEVENT_WORLD_FINISHED) || // if prereq not in nextphase or finished state, then can't start this one
60 mGameEvent[*itr].nextstart > currenttime) // if not in nextphase state for long enough, can't start this one
61 return false;
62 }
63 // all prerequisite events are met
64 // but if there are no prerequisites, this can be only activated through gm command
65 return !(mGameEvent[entry].prerequisite_events.empty());
66 }
67 }

70uint32 GameEventMgr::NextCheck(uint16 entry) const
71{
72 time_t currenttime = time(NULL);
73
74 // for NEXTPHASE state world events, return the delay to start the next event, so the followup event will be checked correctly
75 if ((mGameEvent[entry].state == GAMEEVENT_WORLD_NEXTPHASE || mGameEvent[entry].state == GAMEEVENT_WORLD_FINISHED) && mGameEvent[entry].nextstart >= currenttime)
76 return uint32(mGameEvent[entry].nextstart - currenttime);
77
78 // for CONDITIONS state world events, return the length of the wait period, so if the conditions are met, this check will be called again to set the timer as NEXTPHASE event
79 if (mGameEvent[entry].state == GAMEEVENT_WORLD_CONDITIONS)
80 {
81 if (mGameEvent[entry].length)
82 return mGameEvent[entry].length * 60;
83 else
84 return max_ge_check_delay;
85 }
86
87 // outdated event: we return max
88 if (currenttime > mGameEvent[entry].end)
89 return max_ge_check_delay;
90
91 // never started event, we return delay before start
92 if (mGameEvent[entry].start > currenttime)
93 return uint32(mGameEvent[entry].start - currenttime);
94
95 uint32 delay;
96 // in event, we return the end of it
97 if ((((currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * 60)) < (mGameEvent[entry].length * 60)))
98 // we return the delay before it ends
99 delay = (mGameEvent[entry].length * MINUTE) - ((currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * MINUTE));
100 else // not in window, we return the delay before next start
101 delay = (mGameEvent[entry].occurence * MINUTE) - ((currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * MINUTE));
102 // In case the end is before next check
103 if (mGameEvent[entry].end < time_t(currenttime + delay))
104 return uint32(mGameEvent[entry].end - currenttime);
105 else
106 return delay;
107}

109void GameEventMgr::StartInternalEvent(uint16 event_id)
110{
111 if (event_id < 1 || event_id >= mGameEvent.size())
112 return;
113
114 if (!mGameEvent[event_id].isValid())
115 return;
116
117 if (m_ActiveEvents.find(event_id) != m_ActiveEvents.end())
118 return;
119
120 StartEvent(event_id);
121}

123bool GameEventMgr::StartEvent(uint16 event_id, bool overwrite)
124{
125 GameEventData &data = mGameEvent[event_id];
126 if (data.state == GAMEEVENT_NORMAL || data.state == GAMEEVENT_INTERNAL)
127 {
128 AddActiveEvent(event_id);
129 ApplyNewEvent(event_id);
130 if (overwrite)
131 {
132 mGameEvent[event_id].start = time(NULL);
133 if (data.end <= data.start)
134 data.end = data.start + data.length;
135 }
136 return false;
137 }
138 else
139 {
140 if (data.state == GAMEEVENT_WORLD_INACTIVE)
141 // set to conditions phase
142 data.state = GAMEEVENT_WORLD_CONDITIONS;
143
144 // add to active events
145 AddActiveEvent(event_id);
146 // add spawns
147 ApplyNewEvent(event_id);
148
149 // check if can go to next state
150 bool conditions_met = CheckOneGameEventConditions(event_id);
151 // save to db
152 SaveWorldEventStateToDB(event_id);
153 // force game event update to set the update timer if conditions were met from a command
154 // this update is needed to possibly start events dependent on the started one
155 // or to scedule another update where the next event will be started
156 if (overwrite && conditions_met)
157 sWorld->ForceGameEventUpdate();
158
159 return conditions_met;
160 }

163void GameEventMgr::StopEvent(uint16 event_id, bool overwrite)
164{
165 GameEventData &data = mGameEvent[event_id];
166 bool serverwide_evt = data.state != GAMEEVENT_NORMAL && data.state != GAMEEVENT_INTERNAL;
167
168 RemoveActiveEvent(event_id);
169 UnApplyEvent(event_id);
170
171 if (overwrite && !serverwide_evt)
172 {
173 data.start = time(NULL) - data.length * MINUTE;
174 if (data.end <= data.start)
175 data.end = data.start + data.length;
176 }
177 else if (serverwide_evt)
178 {
179 // if finished world event, then only gm command can stop it
180 if (overwrite || data.state != GAMEEVENT_WORLD_FINISHED)
181 {
182 // reset conditions
183 data.nextstart = 0;
184 data.state = GAMEEVENT_WORLD_INACTIVE;
185 GameEventConditionMap::iterator itr;
186 for (itr = data.conditions.begin(); itr != data.conditions.end(); ++itr)
187 itr->second.done = 0;
188
189 SQLTransaction trans = CharacterDatabase.BeginTransaction();
190 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_ALL_GAME_EVENT_CONDITION_SAVE);
191 stmt->setUInt8(0, uint8(event_id));
192 trans->Append(stmt);
193
194 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_GAME_EVENT_SAVE);
195 stmt->setUInt8(0, uint8(event_id));
196 trans->Append(stmt);
197
198 CharacterDatabase.CommitTransaction(trans);
199 }
200 }

993uint32 GameEventMgr::Update() // return the next event delay in ms
994{
995 time_t currenttime = time(NULL);
996 uint32 nextEventDelay = max_ge_check_delay; // 1 day
997 uint32 calcDelay;
998 std::set<uint16> activate, deactivate;
999 for (uint16 itr = 1; itr < mGameEvent.size(); ++itr)
1000 {
1001 // must do the activating first, and after that the deactivating
1002 // so first queue it
1003 //TC_LOG_ERROR("sql.sql", "Checking event %u", itr);
1004 if (CheckOneGameEvent(itr))
1005 {
1006 // if the world event is in NEXTPHASE state, and the time has passed to finish this event, then do so
1007 if (mGameEvent[itr].state == GAMEEVENT_WORLD_NEXTPHASE && mGameEvent[itr].nextstart <= currenttime)
1008 {
1009 // set this event to finished, null the nextstart time
1010 mGameEvent[itr].state = GAMEEVENT_WORLD_FINISHED;
1011 mGameEvent[itr].nextstart = 0;
1012 // save the state of this gameevent
1013 SaveWorldEventStateToDB(itr);
1014 // queue for deactivation
1015 if (IsActiveEvent(itr))
1016 deactivate.insert(itr);
1017 // go to next event, this no longer needs an event update timer
1018 continue;
1019 }
1020 else if (mGameEvent[itr].state == GAMEEVENT_WORLD_CONDITIONS && CheckOneGameEventConditions(itr))
1021 // changed, save to DB the gameevent state, will be updated in next update cycle
1022 SaveWorldEventStateToDB(itr);
1023
1024 //TC_LOG_DEBUG("misc", "GameEvent %u is active", itr->first);
1025 // queue for activation
1026 if (!IsActiveEvent(itr))
1027 activate.insert(itr);
1028 }
1029 else
1030 {
1031 //TC_LOG_DEBUG("misc", "GameEvent %u is not active", itr->first);
1032 if (IsActiveEvent(itr))
1033 deactivate.insert(itr);
1034 else
1035 {
1036 if (!isSystemInit)
1037 {
1038 int16 event_nid = (-1) * (itr);
1039 // spawn all negative ones for this event
1040 GameEventSpawn(event_nid);
1041 }
1042 }
1043 }
1044 calcDelay = NextCheck(itr);
1045 if (calcDelay < nextEventDelay)
1046 nextEventDelay = calcDelay;
1047 }
1048 // now activate the queue
1049 // a now activated event can contain a spawn of a to-be-deactivated one
1050 // following the activate - deactivate order, deactivating the first event later will leave the spawn in (wont disappear then reappear clientside)
1051 for (std::set<uint16>::iterator itr = activate.begin(); itr != activate.end(); ++itr)
1052 // start the event
1053 // returns true the started event completed
1054 // in that case, initiate next update in 1 second
1055 if (StartEvent(*itr))
1056 nextEventDelay = 0;
1057 for (std::set<uint16>::iterator itr = deactivate.begin(); itr != deactivate.end(); ++itr)
1058 StopEvent(*itr);
1059 TC_LOG_INFO("gameevent", "Next game event check in %u seconds.", nextEventDelay + 1);
1060 return (nextEventDelay + 1) * IN_MILLISECONDS; // Add 1 second to be sure event has started/stopped at next call
1061}

1190 if (!map->Instanceable() && map->IsGridLoaded(data->posX, data->posY))
1191 {

1193 //TC_LOG_DEBUG("misc", "Spawning creature %u", *itr);

1195 delete creature;

1203 internal_event_id, mGameEventGameobjectGuids.size());
1204 return;
1205 }
1206
1207 for (GuidList::iterator itr = mGameEventGameobjectGuids[internal_event_id].begin(); itr != mGameEventGameobjectGuids[internal_event_id].end(); ++itr)

1218 {

1220 //TC_LOG_DEBUG("misc", "Spawning gameobject %u", *itr);

1227 map->AddToMap(pGameobject);

1229 }
1230 }
1231 }

1516void GameEventMgr::HandleQuestComplete(uint32 quest_id)
1517{
1518 // translate the quest to event and condition
1519 QuestIdToEventConditionMap::iterator itr = mQuestToEventConditions.find(quest_id);
1520 // quest is registered
1521 if (itr != mQuestToEventConditions.end())
1522 {
1523 uint16 event_id = itr->second.event_id;
1524 uint32 condition = itr->second.condition;
1525 float num = itr->second.num;
1526
1527 // the event is not active, so return, don't increase condition finishes
1528 if (!IsActiveEvent(event_id))
1529 return;
1530 // not in correct phase, return
1531 if (mGameEvent[event_id].state != GAMEEVENT_WORLD_CONDITIONS)
1532 return;
1533 GameEventConditionMap::iterator citr = mGameEvent[event_id].conditions.find(condition);
1534 // condition is registered
1535 if (citr != mGameEvent[event_id].conditions.end())
1536 {
1537 // increase the done count, only if less then the req
1538 if (citr->second.done < citr->second.reqNum)
1539 {
1540 citr->second.done += num;
1541 // check max limit
1542 if (citr->second.done > citr->second.reqNum)
1543 citr->second.done = citr->second.reqNum;
1544 // save the change to db
1545 SQLTransaction trans = CharacterDatabase.BeginTransaction();
1546
1547 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_GAME_EVENT_CONDITION_SAVE);
1548 stmt->setUInt8(0, uint8(event_id));
1549 stmt->setUInt32(1, condition);
1550 trans->Append(stmt);
1551
1552 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_GAME_EVENT_CONDITION_SAVE);
1553 stmt->setUInt8(0, uint8(event_id));
1554 stmt->setUInt32(1, condition);
1555 stmt->setFloat(2, citr->second.done);
1556 trans->Append(stmt);
1557 CharacterDatabase.CommitTransaction(trans);
1558 // check if all conditions are met, if so, update the event state
1559 if (CheckOneGameEventConditions(event_id))
1560 {
1561 // changed, save to DB the gameevent state
1562 SaveWorldEventStateToDB(event_id);
1563 // force update events to set timer
1564 sWorld->ForceGameEventUpdate();
1565 }
1566 }
1567 }
1568 }

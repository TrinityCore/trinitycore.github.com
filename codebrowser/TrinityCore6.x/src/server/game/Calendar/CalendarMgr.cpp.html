49void CalendarMgr::LoadFromDB()
50{
51 uint32 count = 0;
52 _maxEventId = 0;
53 _maxInviteId = 0;
54
55 // 0 1 2 3 4 5 6 7 8
56 if (QueryResult result = CharacterDatabase.Query("SELECT EventID, Owner, Title, Description, EventType, TextureID, Date, Flags, LockDate FROM calendar_events"))
57 do
58 {
59 Field* fields = result->Fetch();
60
61 uint64 eventID = fields[0].GetUInt64();
62 ObjectGuid ownerGUID = ObjectGuid::Create<HighGuid::Player>(fields[1].GetUInt64());
63 std::string title = fields[2].GetString();
64 std::string description = fields[3].GetString();
65 CalendarEventType type = CalendarEventType(fields[4].GetUInt8());
66 int32 textureID = fields[5].GetInt32();
67 uint32 date = fields[6].GetUInt32();
68 uint32 flags = fields[7].GetUInt32();
69 uint32 lockDate = fields[8].GetUInt32();
70 ObjectGuid::LowType guildID = UI64LIT(0);
71
72 if (flags & CALENDAR_FLAG_GUILD_EVENT || flags & CALENDAR_FLAG_WITHOUT_INVITES)
73 guildID = Player::GetGuildIdFromDB(ownerGUID);
74
75 CalendarEvent* calendarEvent = new CalendarEvent(eventID, ownerGUID, guildID, type, textureID, time_t(date), flags, title, description, time_t(lockDate));
76 _events.insert(calendarEvent);
77
78 _maxEventId = std::max(_maxEventId, eventID);
79
80 ++count;
81 }
82 while (result->NextRow());
83
84 TC_LOG_INFO("server.loading", ">> Loaded %u calendar events", count);
85 count = 0;
86
87 // 0 1 2 3 4 5 6 7
88 if (QueryResult result = CharacterDatabase.Query("SELECT InviteID, EventID, Invitee, Sender, Status, ResponseTime, ModerationRank, Note FROM calendar_invites"))
89 do
90 {
91 Field* fields = result->Fetch();
92
93 uint64 inviteId = fields[0].GetUInt64();
94 uint64 eventId = fields[1].GetUInt64();
95 ObjectGuid invitee = ObjectGuid::Create<HighGuid::Player>(fields[2].GetUInt64());
96 ObjectGuid senderGUID = ObjectGuid::Create<HighGuid::Player>(fields[3].GetUInt64());
97 CalendarInviteStatus status = CalendarInviteStatus(fields[4].GetUInt8());
98 uint32 responseTime = fields[5].GetUInt32();
99 CalendarModerationRank rank = CalendarModerationRank(fields[6].GetUInt8());
100 std::string note = fields[7].GetString();
101
102 CalendarInvite* invite = new CalendarInvite(inviteId, eventId, invitee, senderGUID, time_t(responseTime), status, rank, note);
103 _invites[eventId].push_back(invite);
104
105 _maxInviteId = std::max(_maxInviteId, inviteId);
106
107 ++count;
108 }
109 while (result->NextRow());
110
111 TC_LOG_INFO("server.loading", ">> Loaded %u calendar invites", count);
112
113 for (uint64 i = 1; i < _maxEventId; ++i)
114 if (!GetEvent(i))
115 _freeEventIds.push_back(i);
116
117 for (uint64 i = 1; i < _maxInviteId; ++i)
118 if (!GetInvite(i))
119 _freeInviteIds.push_back(i);
120}

135void CalendarMgr::AddInvite(CalendarEvent* calendarEvent, CalendarInvite* invite, SQLTransaction& trans)
136{
137 if (!calendarEvent->IsGuildAnnouncement() && calendarEvent->GetOwnerGUID() != invite->GetInviteeGUID())

150void CalendarMgr::RemoveEvent(uint64 eventId, ObjectGuid remover)
151{
152 CalendarEvent* calendarEvent = GetEvent(eventId);
153
154 if (!calendarEvent)
155 {
156 SendCalendarCommandResult(remover, CALENDAR_ERROR_EVENT_INVALID);
157 return;
158 }
159
160 SendCalendarEventRemovedAlert(*calendarEvent);
161
162 SQLTransaction trans = CharacterDatabase.BeginTransaction();
163 PreparedStatement* stmt;
164 MailDraft mail(calendarEvent->BuildCalendarMailSubject(remover), calendarEvent->BuildCalendarMailBody());
165
166 CalendarInviteStore& eventInvites = _invites[eventId];
167 for (size_t i = 0; i < eventInvites.size(); ++i)
168 {
169 CalendarInvite* invite = eventInvites[i];
170 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_INVITE);
171 stmt->setUInt64(0, invite->GetInviteId());
172 trans->Append(stmt);
173
174 // guild events only? check invite status here?
175 // When an event is deleted, all invited (accepted/declined? - verify) guildies are notified via in-game mail. (wowwiki)
176 if (!remover.IsEmpty() && invite->GetInviteeGUID() != remover)
177 mail.SendMailTo(trans, MailReceiver(invite->GetInviteeGUID().GetCounter()), calendarEvent, MAIL_CHECK_MASK_COPIED);
178
179 delete invite;
180 }
181
182 _invites.erase(eventId);
183
184 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_EVENT);
185 stmt->setUInt64(0, eventId);
186 trans->Append(stmt);
187 CharacterDatabase.CommitTransaction(trans);
188
189 delete calendarEvent;
190 _events.erase(calendarEvent);
191}

193void CalendarMgr::RemoveInvite(uint64 inviteId, uint64 eventId, ObjectGuid /*remover*/)
194{
195 CalendarEvent* calendarEvent = GetEvent(eventId);
196
197 if (!calendarEvent)
198 return;
199
200 CalendarInviteStore::iterator itr = _invites[eventId].begin();
201 for (; itr != _invites[eventId].end(); ++itr)
202 if ((*itr)->GetInviteId() == inviteId)
203 break;
204
205 if (itr == _invites[eventId].end())
206 return;
207
208 SQLTransaction trans = CharacterDatabase.BeginTransaction();
209 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_INVITE);
210 stmt->setUInt64(0, (*itr)->GetInviteId());
211 trans->Append(stmt);
212 CharacterDatabase.CommitTransaction(trans);
213
214 if (!calendarEvent->IsGuildEvent())
215 SendCalendarEventInviteRemoveAlert((*itr)->GetInviteeGUID(), *calendarEvent, CALENDAR_STATUS_REMOVED);
216
217 SendCalendarEventInviteRemove(*calendarEvent, **itr, calendarEvent->GetFlags());
218
219 // we need to find out how to use CALENDAR_INVITE_REMOVED_MAIL_SUBJECT to force client to display different mail
220 //if ((*itr)->GetInviteeGUID() != remover)
221 // MailDraft(calendarEvent->BuildCalendarMailSubject(remover), calendarEvent->BuildCalendarMailBody())
222 // .SendMailTo(trans, MailReceiver((*itr)->GetInvitee()), calendarEvent, MAIL_CHECK_MASK_COPIED);
223
224 delete *itr;
225 _invites[eventId].erase(itr);
226}

228void CalendarMgr::UpdateEvent(CalendarEvent* calendarEvent)
229{
230 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_REP_CALENDAR_EVENT);
231 stmt->setUInt64(0, calendarEvent->GetEventId());
232 stmt->setUInt64(1, calendarEvent->GetOwnerGUID().GetCounter());
233 stmt->setString(2, calendarEvent->GetTitle());
234 stmt->setString(3, calendarEvent->GetDescription());
235 stmt->setUInt8(4, calendarEvent->GetType());
236 stmt->setInt32(5, calendarEvent->GetTextureId());
237 stmt->setUInt32(6, uint32(calendarEvent->GetDate()));
238 stmt->setUInt32(7, calendarEvent->GetFlags());
239 stmt->setUInt32(8, uint32(calendarEvent->GetLockDate()));
240 CharacterDatabase.Execute(stmt);
241}

263void CalendarMgr::RemoveAllPlayerEventsAndInvites(ObjectGuid guid)
264{
265 for (CalendarEventStore::const_iterator itr = _events.begin(); itr != _events.end(); ++itr)
266 if ((*itr)->GetOwnerGUID() == guid)
267 RemoveEvent((*itr)->GetEventId(), ObjectGuid::Empty); // don't send mail if removing a character
268
269 CalendarInviteStore playerInvites = GetPlayerInvites(guid);
270 for (CalendarInviteStore::const_iterator itr = playerInvites.begin(); itr != playerInvites.end(); ++itr)
271 RemoveInvite((*itr)->GetInviteId(), (*itr)->GetEventId(), guid);
272}

274void CalendarMgr::RemovePlayerGuildEventsAndSignups(ObjectGuid guid, ObjectGuid::LowType guildId)
275{
276 for (CalendarEventStore::const_iterator itr = _events.begin(); itr != _events.end(); ++itr)
277 if ((*itr)->GetOwnerGUID() == guid && ((*itr)->IsGuildEvent() || (*itr)->IsGuildAnnouncement()))
278 RemoveEvent((*itr)->GetEventId(), guid);
279
280 CalendarInviteStore playerInvites = GetPlayerInvites(guid);
281 for (CalendarInviteStore::const_iterator itr = playerInvites.begin(); itr != playerInvites.end(); ++itr)
282 if (CalendarEvent* calendarEvent = GetEvent((*itr)->GetEventId()))
283 if (calendarEvent->IsGuildEvent() && calendarEvent->GetGuildId() == guildId)
284 RemoveInvite((*itr)->GetInviteId(), (*itr)->GetEventId(), guid);
285}

344CalendarEventStore CalendarMgr::GetPlayerEvents(ObjectGuid guid)
345{
346 CalendarEventStore events;
347
348 for (CalendarEventInviteStore::const_iterator itr = _invites.begin(); itr != _invites.end(); ++itr)
349 for (CalendarInviteStore::const_iterator itr2 = itr->second.begin(); itr2 != itr->second.end(); ++itr2)
350 if ((*itr2)->GetInviteeGUID() == guid)
351 if (CalendarEvent* event = GetEvent(itr->first)) // NULL check added as attempt to fix #11512
352 events.insert(event);
353
354 if (Player* player = ObjectAccessor::FindConnectedPlayer(guid))
355 for (CalendarEventStore::const_iterator itr = _events.begin(); itr != _events.end(); ++itr)
356 if ((*itr)->GetGuildId() == player->GetGuildId())
357 events.insert(*itr);
358
359 return events;

421void CalendarMgr::SendCalendarEventInvite(CalendarInvite const& invite)
422{
423 CalendarEvent* calendarEvent = GetEvent(invite.GetEventId());
424
425 ObjectGuid invitee = invite.GetInviteeGUID();
426 Player* player = ObjectAccessor::FindConnectedPlayer(invitee);
427
428 uint8 level = player ? player->getLevel() : Player::GetLevelFromDB(invitee);
429
430 WorldPackets::Calendar::SCalendarEventInvite packet;
431 packet.EventID = calendarEvent ? calendarEvent->GetEventId() : 0;
432 packet.InviteGuid = invitee;
433 packet.InviteID = calendarEvent ? invite.GetInviteId() : 0;
434 packet.Level = level;
435 packet.ResponseTime = invite.GetResponseTime();
436 packet.Status = invite.GetStatus();
437 packet.Type = calendarEvent ? calendarEvent->IsGuildEvent() : 0; // Correct ?
438 packet.ClearPending = calendarEvent ? !calendarEvent->IsGuildEvent() : true; // Correct ?
439
440 if (!calendarEvent) // Pre-invite
441 {
442 if (Player* playerSender = ObjectAccessor::FindConnectedPlayer(invite.GetSenderGUID()))
443 playerSender->SendDirectMessage(packet.Write());
444 }
445 else
446 {
447 if (calendarEvent->GetOwnerGUID() != invite.GetInviteeGUID()) // correct?
448 SendPacketToAllEventRelatives(packet.Write(), *calendarEvent);
449 }
450}

543void CalendarMgr::SendCalendarEvent(ObjectGuid guid, CalendarEvent const& calendarEvent, CalendarSendEventType sendType)
544{
545 Player* player = ObjectAccessor::FindConnectedPlayer(guid);
546 if (!player)
547 return;
548
549 CalendarInviteStore const& eventInviteeList = _invites[calendarEvent.GetEventId()];
550
551 WorldPackets::Calendar::CalendarSendEvent packet;
552 packet.Date = calendarEvent.GetDate();
553 packet.Description = calendarEvent.GetDescription();
554 packet.EventID = calendarEvent.GetEventId();
555 packet.EventName = calendarEvent.GetTitle();
556 packet.EventType = sendType;
557 packet.Flags = calendarEvent.GetFlags();
558 packet.GetEventType = calendarEvent.GetType();
559 packet.LockDate = calendarEvent.GetLockDate(); // Always 0 ?
560 packet.OwnerGuid = calendarEvent.GetOwnerGUID();
561 packet.TextureID = calendarEvent.GetTextureId();
562
563 Guild* guild = sGuildMgr->GetGuildById(calendarEvent.GetGuildId());
564 packet.EventGuildID = (guild ? guild->GetGUID() : ObjectGuid::Empty);
565
566 for (auto const& calendarInvite : eventInviteeList)
567 {
568 ObjectGuid inviteeGuid = calendarInvite->GetInviteeGUID();
569 Player* invitee = ObjectAccessor::FindPlayer(inviteeGuid);
570
571 uint8 inviteeLevel = invitee ? invitee->getLevel() : Player::GetLevelFromDB(inviteeGuid);
572 ObjectGuid::LowType inviteeGuildId = invitee ? invitee->GetGuildId() : Player::GetGuildIdFromDB(inviteeGuid);
573
574 WorldPackets::Calendar::CalendarEventInviteInfo inviteInfo;
575 inviteInfo.Guid = inviteeGuid;
576 inviteInfo.Level = inviteeLevel;
577 inviteInfo.Status = calendarInvite->GetStatus();
578 inviteInfo.Moderator = calendarInvite->GetRank();
579 inviteInfo.InviteType = calendarEvent.IsGuildEvent() && calendarEvent.GetGuildId() == inviteeGuildId;
580 inviteInfo.InviteID = calendarInvite->GetInviteId();
581 inviteInfo.ResponseTime = calendarInvite->GetResponseTime();
582 inviteInfo.Notes = calendarInvite->GetNote();
583
584 packet.Invites.push_back(inviteInfo);
585 }
586
587 player->SendDirectMessage(packet.Write());
588}

633void CalendarMgr::SendPacketToAllEventRelatives(WorldPacket const* packet, CalendarEvent const& calendarEvent)
634{
635 // Send packet to all guild members
636 if (calendarEvent.IsGuildEvent() || calendarEvent.IsGuildAnnouncement())
637 if (Guild* guild = sGuildMgr->GetGuildById(calendarEvent.GetGuildId()))
638 guild->BroadcastPacket(packet);
639
640 // Send packet to all invitees if event is non-guild, in other case only to non-guild invitees (packet was broadcasted for them)
641 CalendarInviteStore invites = _invites[calendarEvent.GetEventId()];
642 for (CalendarInviteStore::iterator itr = invites.begin(); itr != invites.end(); ++itr)

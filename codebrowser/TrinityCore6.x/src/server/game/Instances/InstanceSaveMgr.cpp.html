41void InstanceSaveManager::Unload()
42{
43 lock_instLists = true;
44 for (InstanceSaveHashMap::iterator itr = m_instanceSaveById.begin(); itr != m_instanceSaveById.end(); ++itr)
45 {
46 InstanceSave* save = itr->second;
47
48 for (InstanceSave::PlayerListType::iterator itr2 = save->m_playerList.begin(), next = itr2; itr2 != save->m_playerList.end(); itr2 = next)
49 {
50 ++next;
51 (*itr2)->UnbindInstance(save->GetMapId(), save->GetDifficultyID(), true);
52 }
53
54 for (InstanceSave::GroupListType::iterator itr2 = save->m_groupList.begin(), next = itr2; itr2 != save->m_groupList.end(); itr2 = next)
55 {
56 ++next;
57 (*itr2)->UnbindInstance(save->GetMapId(), save->GetDifficultyID(), true);
58 }
59
60 delete save;
61 }
62}

68InstanceSave* InstanceSaveManager::AddInstanceSave(uint32 mapId, uint32 instanceId, Difficulty difficulty, time_t resetTime, bool canReset, bool load)
69{
70 if (InstanceSave* old_save = GetInstanceSave(instanceId))
71 return old_save;
72
73 const MapEntry* entry = sMapStore.LookupEntry(mapId);
74 if (!entry)
75 {
76 TC_LOG_ERROR("misc", "InstanceSaveManager::AddInstanceSave: wrong mapid = %d, instanceid = %d!", mapId, instanceId);
77 return NULL;
78 }
79
80 if (instanceId == 0)
81 {
82 TC_LOG_ERROR("misc", "InstanceSaveManager::AddInstanceSave: mapid = %d, wrong instanceid = %d!", mapId, instanceId);
83 return NULL;
84 }
85
86 DifficultyEntry const* difficultyEntry = sDifficultyStore.LookupEntry(difficulty);
87 if (!difficultyEntry || difficultyEntry->InstanceType != entry->InstanceType)
88 {
89 TC_LOG_ERROR("misc", "InstanceSaveManager::AddInstanceSave: mapid = %d, instanceid = %d, wrong dificalty %u!", mapId, instanceId, difficulty);
90 return NULL;
91 }
92
93 if (!resetTime)
94 {
95 // initialize reset time
96 // for normal instances if no creatures are killed the instance will reset in two hours
97 if (entry->IsRaid() || difficulty > DIFFICULTY_NORMAL)
98 resetTime = GetResetTimeFor(mapId, difficulty);
99 else
100 {
101 resetTime = time(NULL) + 2 * HOUR;
102 // normally this will be removed soon after in InstanceMap::Add, prevent error
103 ScheduleReset(true, resetTime, InstResetEvent(0, mapId, difficulty, instanceId));
104 }
105 }
106
107 TC_LOG_DEBUG("maps", "InstanceSaveManager::AddInstanceSave: mapid = %d, instanceid = %d", mapId, instanceId);
108
109 InstanceSave* save = new InstanceSave(mapId, instanceId, difficulty, resetTime, canReset);
110 if (!load)
111 save->SaveToDB();
112
113 m_instanceSaveById[instanceId] = save;
114 return save;
115}

143void InstanceSaveManager::RemoveInstanceSave(uint32 InstanceId)
144{
145 InstanceSaveHashMap::iterator itr = m_instanceSaveById.find(InstanceId);
146 if (itr != m_instanceSaveById.end())
147 {
148 // save the resettime for normal instances only when they get unloaded
149 if (time_t resettime = itr->second->GetResetTimeForDB())
150 {
151 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_INSTANCE_RESETTIME);
152
153 stmt->setUInt32(0, uint32(resettime));
154 stmt->setUInt32(1, InstanceId);
155
156 CharacterDatabase.Execute(stmt);
157 }
158
159 itr->second->SetToDelete(true);
160 m_instanceSaveById.erase(itr);
161 }

183void InstanceSave::SaveToDB()
184{
185 // save instance data too
186 std::string data;
187 uint32 completedEncounters = 0;
188
189 Map* map = sMapMgr->FindMap(GetMapId(), m_instanceid);
190 if (map)
191 {
192 ASSERT(map->IsDungeon());
193 if (InstanceScript* instanceScript = ((InstanceMap*)map)->GetInstanceScript())
194 {
195 data = instanceScript->GetSaveData();
196 completedEncounters = instanceScript->GetCompletedEncounterMask();
197 }
198 }
199
200 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_INSTANCE_SAVE);
201 stmt->setUInt32(0, m_instanceid);
202 stmt->setUInt16(1, GetMapId());
203 stmt->setUInt32(2, uint32(GetResetTimeForDB()));
204 stmt->setUInt8(3, uint8(GetDifficultyID()));
205 stmt->setUInt32(4, completedEncounters);
206 stmt->setString(5, data);
207 CharacterDatabase.Execute(stmt);
208}

236/* true if the instance save is still valid */
237bool InstanceSave::UnloadIfEmpty()
238{
239 if (m_playerList.empty() && m_groupList.empty())
240 {
241 // don't remove the save if there are still players inside the map
242 if (Map* map = sMapMgr->FindMap(GetMapId(), GetInstanceId()))
243 if (map->HavePlayers())
244 return true;
245
246 if (!sInstanceSaveMgr->lock_instLists)
247 sInstanceSaveMgr->RemoveInstanceSave(GetInstanceId());
248
249 return false;
250 }
251 else
252 return true;
253}

255void InstanceSaveManager::LoadInstances()
256{
257 uint32 oldMSTime = getMSTime();
258
259 // Delete expired instances (Instance related spawns are removed in the following cleanup queries)
260 CharacterDatabase.DirectExecute("DELETE i FROM instance i LEFT JOIN instance_reset ir ON mapid = map AND i.difficulty = ir.difficulty "
261 "WHERE (i.resettime > 0 AND i.resettime < UNIX_TIMESTAMP()) OR (ir.resettime IS NOT NULL AND ir.resettime < UNIX_TIMESTAMP())");
262
263 // Delete invalid character_instance and group_instance references
264 CharacterDatabase.DirectExecute("DELETE ci.* FROM character_instance AS ci LEFT JOIN characters AS c ON ci.guid = c.guid WHERE c.guid IS NULL");
265 CharacterDatabase.DirectExecute("DELETE gi.* FROM group_instance AS gi LEFT JOIN groups AS g ON gi.guid = g.guid WHERE g.guid IS NULL");
266
267 // Delete invalid instance references
268 CharacterDatabase.DirectExecute("DELETE i.* FROM instance AS i LEFT JOIN character_instance AS ci ON i.id = ci.instance LEFT JOIN group_instance AS gi ON i.id = gi.instance WHERE ci.guid IS NULL AND gi.guid IS NULL");
269
270 // Delete invalid references to instance
271 CharacterDatabase.DirectExecute("DELETE FROM creature_respawn WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");
272 CharacterDatabase.DirectExecute("DELETE FROM gameobject_respawn WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");
273 CharacterDatabase.DirectExecute("DELETE tmp.* FROM character_instance AS tmp LEFT JOIN instance ON tmp.instance = instance.id WHERE tmp.instance > 0 AND instance.id IS NULL");
274 CharacterDatabase.DirectExecute("DELETE tmp.* FROM group_instance AS tmp LEFT JOIN instance ON tmp.instance = instance.id WHERE tmp.instance > 0 AND instance.id IS NULL");
275
276 // Clean invalid references to instance
277 CharacterDatabase.DirectExecute("UPDATE corpse SET instanceId = 0 WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");
278 CharacterDatabase.DirectExecute("UPDATE characters AS tmp LEFT JOIN instance ON tmp.instance_id = instance.id SET tmp.instance_id = 0 WHERE tmp.instance_id > 0 AND instance.id IS NULL");
279
280 // Initialize instance id storage (Needs to be done after the trash has been clean out)
281 sMapMgr->InitInstanceIds();
282
283 // Load reset times and clean expired instances
284 sInstanceSaveMgr->LoadResetTimes();
285
286 TC_LOG_INFO("server.loading", ">> Loaded instances in %u ms", GetMSTimeDiffToNow(oldMSTime));
287
288}

290void InstanceSaveManager::LoadResetTimes()
291{
292 time_t now = time(NULL);
293 time_t today = (now / DAY) * DAY;
294
295 // NOTE: Use DirectPExecute for tables that will be queried later
296
297 // get the current reset times for normal instances (these may need to be updated)
298 // these are only kept in memory for InstanceSaves that are loaded later
299 // resettime = 0 in the DB for raid/heroic instances so those are skipped
300 typedef std::pair<uint32 /*PAIR32(map, difficulty)*/, time_t> ResetTimeMapDiffType;
301 typedef std::map<uint32, ResetTimeMapDiffType> InstResetTimeMapDiffType;
302 InstResetTimeMapDiffType instResetTime;
303
304 // index instance ids by map/difficulty pairs for fast reset warning send
305 typedef std::multimap<uint32 /*PAIR32(map, difficulty)*/, uint32 /*instanceid*/ > ResetTimeMapDiffInstances;
306 typedef std::pair<ResetTimeMapDiffInstances::const_iterator, ResetTimeMapDiffInstances::const_iterator> ResetTimeMapDiffInstancesBounds;
307 ResetTimeMapDiffInstances mapDiffResetInstances;
308
309 QueryResult result = CharacterDatabase.Query("SELECT id, map, difficulty, resettime FROM instance ORDER BY id ASC");
310 if (result)
311 {
312 do
313 {
314 Field* fields = result->Fetch();
315
316 uint32 instanceId = fields[0].GetUInt32();
317
318 // Instances are pulled in ascending order from db and nextInstanceId is initialized with 1,
319 // so if the instance id is used, increment until we find the first unused one for a potential new instance
320 if (sMapMgr->GetNextInstanceId() == instanceId)
321 sMapMgr->SetNextInstanceId(instanceId + 1);
322
323 // Mark instance id as being used
324 sMapMgr->RegisterInstanceId(instanceId);
325
326 if (time_t resettime = time_t(fields[3].GetUInt32()))
327 {
328 uint32 mapid = fields[1].GetUInt16();
329 uint32 difficulty = fields[2].GetUInt8();
330
331 instResetTime[instanceId] = ResetTimeMapDiffType(MAKE_PAIR32(mapid, difficulty), resettime);
332 mapDiffResetInstances.insert(ResetTimeMapDiffInstances::value_type(MAKE_PAIR32(mapid, difficulty), instanceId));
333 }
334 }
335 while (result->NextRow());
336
337 // update reset time for normal instances with the max creature respawn time + X hours
338 if (PreparedQueryResult result2 = CharacterDatabase.Query(CharacterDatabase.GetPreparedStatement(CHAR_SEL_MAX_CREATURE_RESPAWNS)))
339 {
340 do
341 {
342 Field* fields = result2->Fetch();
343 uint32 instance = fields[1].GetUInt32();
344 time_t resettime = time_t(fields[0].GetUInt32() + 2 * HOUR);
345 InstResetTimeMapDiffType::iterator itr = instResetTime.find(instance);
346 if (itr != instResetTime.end() && itr->second.second != resettime)
347 {
348 CharacterDatabase.DirectPExecute("UPDATE instance SET resettime = '" UI64FMTD "' WHERE id = '%u'", uint64(resettime), instance);
349 itr->second.second = resettime;
350 }
351 }
352 while (result->NextRow());
353 }
354
355 // schedule the reset times
356 for (InstResetTimeMapDiffType::iterator itr = instResetTime.begin(); itr != instResetTime.end(); ++itr)
357 if (itr->second.second > now)
358 ScheduleReset(true, itr->second.second, InstResetEvent(0, PAIR32_LOPART(itr->second.first), Difficulty(PAIR32_HIPART(itr->second.first)), itr->first));
359 }
360
361 // load the global respawn times for raid/heroic instances
362 uint32 diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
363 result = CharacterDatabase.Query("SELECT mapid, difficulty, resettime FROM instance_reset");
364 if (result)
365 {
366 do
367 {
368 Field* fields = result->Fetch();
369 uint32 mapid = fields[0].GetUInt16();
370 Difficulty difficulty = Difficulty(fields[1].GetUInt8());
371 uint64 oldresettime = fields[2].GetUInt32();
372
373 MapDifficultyEntry const* mapDiff = GetMapDifficultyData(mapid, difficulty);
374 if (!mapDiff)
375 {
376 TC_LOG_ERROR("misc", "InstanceSaveManager::LoadResetTimes: invalid mapid(%u)/difficulty(%u) pair in instance_reset!", mapid, difficulty);
377 CharacterDatabase.DirectPExecute("DELETE FROM instance_reset WHERE mapid = '%u' AND difficulty = '%u'", mapid, difficulty);
378 continue;
379 }
380
381 // update the reset time if the hour in the configs changes
382 uint64 newresettime = (oldresettime / DAY) * DAY + diff;
383 if (oldresettime != newresettime)
384 CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '%u' WHERE mapid = '%u' AND difficulty = '%u'", uint32(newresettime), mapid, difficulty);
385
386 InitializeResetTimeFor(mapid, difficulty, newresettime);
387 } while (result->NextRow());
388 }
389
390 // calculate new global reset times for expired instances and those that have never been reset yet
391 // add the global reset times to the priority queue
392 for (auto& mapDifficultyPair : sMapDifficultyMap)
393 {
394 uint32 mapid = mapDifficultyPair.first;
395
396 for (auto& difficultyPair : mapDifficultyPair.second)
397 {
398 Difficulty difficulty = Difficulty(difficultyPair.first);
399 MapDifficultyEntry const* mapDiff = difficultyPair.second;
400 if (!mapDiff->RaidDuration)
401 continue;
402
403 // the reset_delay must be at least one day
404 uint32 period = uint32(((mapDiff->RaidDuration * sWorld->getRate(RATE_INSTANCE_RESET_TIME)) / DAY) * DAY);
405 if (period < DAY)
406 period = DAY;
407
408 time_t t = GetResetTimeFor(mapid, difficulty);
409 if (!t)
410 {
411 // initialize the reset time
412 t = today + period + diff;
413 CharacterDatabase.DirectPExecute("INSERT INTO instance_reset VALUES ('%u', '%u', '%u')", mapid, difficulty, (uint32)t);
414 }
415
416 if (t < now)
417 {
418 // assume that expired instances have already been cleaned
419 // calculate the next reset time
420 t = (t / DAY) * DAY;
421 t += ((today - t) / period + 1) * period + diff;
422 CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '" UI64FMTD "' WHERE mapid = '%u' AND difficulty= '%u'", (uint64)t, mapid, difficulty);
423 }
424
425 InitializeResetTimeFor(mapid, difficulty, t);
426
427 // schedule the global reset/warning
428 uint8 type;
429 for (type = 1; type < 4; ++type)
430 if (t - ResetTimeDelay[type - 1] > now)
431 break;
432
433 ScheduleReset(true, t - ResetTimeDelay[type - 1], InstResetEvent(type, mapid, difficulty, 0));
434
435 ResetTimeMapDiffInstancesBounds range = mapDiffResetInstances.equal_range(MAKE_PAIR32(mapid, difficulty));
436 for (; range.first != range.second; ++range.first)
437 ScheduleReset(true, t - ResetTimeDelay[type - 1], InstResetEvent(type, mapid, difficulty, range.first->second));
438 }
439 }
440}

442void InstanceSaveManager::ScheduleReset(bool add, time_t time, InstResetEvent event)
443{
444 if (!add)
445 {
446 // find the event in the queue and remove it
447 ResetTimeQueue::iterator itr;
448 std::pair<ResetTimeQueue::iterator, ResetTimeQueue::iterator> range;
449 range = m_resetTimeQueue.equal_range(time);
450 for (itr = range.first; itr != range.second; ++itr)
451 {
452 if (itr->second == event)
453 {
454 m_resetTimeQueue.erase(itr);
455 return;
456 }
457 }
458
459 // in case the reset time changed (should happen very rarely), we search the whole queue
460 if (itr == range.second)
461 {
462 for (itr = m_resetTimeQueue.begin(); itr != m_resetTimeQueue.end(); ++itr)
463 {
464 if (itr->second == event)
465 {
466 m_resetTimeQueue.erase(itr);
467 return;
468 }
469 }
470
471 if (itr == m_resetTimeQueue.end())
472 TC_LOG_ERROR("misc", "InstanceSaveManager::ScheduleReset: cannot cancel the reset, the event(%d, %d, %d) was not found!", event.type, event.mapid, event.instanceId);
473 }
474 }
475 else
476 m_resetTimeQueue.insert(std::pair<time_t, InstResetEvent>(time, event));
477}
478
479void InstanceSaveManager::Update()
480{
481 time_t now = time(NULL);
482 time_t t;
483
484 while (!m_resetTimeQueue.empty())
485 {
486 t = m_resetTimeQueue.begin()->first;
487 if (t >= now)
488 break;
489
490 InstResetEvent &event = m_resetTimeQueue.begin()->second;
491 if (event.type == 0)
492 {
493 // for individual normal instances, max creature respawn + X hours
494 _ResetInstance(event.mapid, event.instanceId);
495 m_resetTimeQueue.erase(m_resetTimeQueue.begin());
496 }
497 else
498 {
499 // global reset/warning for a certain map
500 time_t resetTime = GetResetTimeFor(event.mapid, event.difficulty);
501 _ResetOrWarnAll(event.mapid, event.difficulty, event.type != 4, resetTime);
502 if (event.type != 4)
503 {
504 // schedule the next warning/reset
505 ++event.type;
506 ScheduleReset(true, resetTime - ResetTimeDelay[event.type-1], event);
507 }
508 m_resetTimeQueue.erase(m_resetTimeQueue.begin());
509 }
510 }

513void InstanceSaveManager::_ResetSave(InstanceSaveHashMap::iterator &itr)
514{
515 // unbind all players bound to the instance
516 // do not allow UnbindInstance to automatically unload the InstanceSaves
517 lock_instLists = true;
518
519 InstanceSave::PlayerListType &pList = itr->second->m_playerList;
520 while (!pList.empty())
521 {
522 Player* player = *(pList.begin());
523 player->UnbindInstance(itr->second->GetMapId(), itr->second->GetDifficultyID(), true);
524 }
525
526 InstanceSave::GroupListType &gList = itr->second->m_groupList;
527 while (!gList.empty())
528 {
529 Group* group = *(gList.begin());
530 group->UnbindInstance(itr->second->GetMapId(), itr->second->GetDifficultyID(), true);
531 }
532
533 delete itr->second;
534 m_instanceSaveById.erase(itr++);
535
536 lock_instLists = false;
537}

539void InstanceSaveManager::_ResetInstance(uint32 mapid, uint32 instanceId)
540{
541 TC_LOG_DEBUG("maps", "InstanceSaveMgr::_ResetInstance %u, %u", mapid, instanceId);
542 Map const* map = sMapMgr->CreateBaseMap(mapid);
543 if (!map->Instanceable())
544 return;
545
546 InstanceSaveHashMap::iterator itr = m_instanceSaveById.find(instanceId);
547 if (itr != m_instanceSaveById.end())
548 _ResetSave(itr);
549
550 DeleteInstanceFromDB(instanceId); // even if save not loaded
551
552 Map* iMap = ((MapInstanced*)map)->FindInstanceMap(instanceId);
553
554 if (iMap && iMap->IsDungeon())
555 ((InstanceMap*)iMap)->Reset(INSTANCE_RESET_RESPAWN_DELAY);
556
557 if (iMap)
558 {
559 iMap->DeleteRespawnTimes();
560 iMap->DeleteCorpseData();
561 }
562 else
563 Map::DeleteRespawnTimesInDB(mapid, instanceId);
564
565 // Free up the instance id and allow it to be reused
566 sMapMgr->FreeInstanceId(instanceId);
567}

569void InstanceSaveManager::_ResetOrWarnAll(uint32 mapid, Difficulty difficulty, bool warn, time_t resetTime)
570{
571 // global reset for all instances of the given map
572 MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
573 if (!mapEntry->Instanceable())
574 return;
575
576 time_t now = time(NULL);
577
578 if (!warn)
579 {
580 MapDifficultyEntry const* mapDiff = GetMapDifficultyData(mapid, difficulty);
581 if (!mapDiff || !mapDiff->RaidDuration)
582 {
583 TC_LOG_ERROR("misc", "InstanceSaveManager::ResetOrWarnAll: not valid difficulty or no reset delay for map %d", mapid);
584 return;
585 }
586
587 // remove all binds to instances of the given map
588 for (InstanceSaveHashMap::iterator itr = m_instanceSaveById.begin(); itr != m_instanceSaveById.end();)
589 {
590 if (itr->second->GetMapId() == mapid && itr->second->GetDifficultyID() == difficulty)
591 _ResetSave(itr);
592 else
593 ++itr;
594 }
595
596 // delete them from the DB, even if not loaded
597 SQLTransaction trans = CharacterDatabase.BeginTransaction();
598
599 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHAR_INSTANCE_BY_MAP_DIFF);
600 stmt->setUInt16(0, uint16(mapid));
601 stmt->setUInt8(1, uint8(difficulty));
602 trans->Append(stmt);
603
604 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_GROUP_INSTANCE_BY_MAP_DIFF);
605 stmt->setUInt16(0, uint16(mapid));
606 stmt->setUInt8(1, uint8(difficulty));
607 trans->Append(stmt);
608
609 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_INSTANCE_BY_MAP_DIFF);
610 stmt->setUInt16(0, uint16(mapid));
611 stmt->setUInt8(1, uint8(difficulty));
612 trans->Append(stmt);
613
614 CharacterDatabase.CommitTransaction(trans);
615
616 // calculate the next reset time
617 uint32 diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
618
619 uint32 period = uint32(((mapDiff->RaidDuration * sWorld->getRate(RATE_INSTANCE_RESET_TIME))/DAY) * DAY);
620 if (period < DAY)
621 period = DAY;
622
623 uint32 next_reset = uint32(((resetTime + MINUTE) / DAY * DAY) + period + diff);
624
625 SetResetTimeFor(mapid, difficulty, next_reset);
626 ScheduleReset(true, time_t(next_reset-3600), InstResetEvent(1, mapid, difficulty, 0));
627
628 // Update it in the DB
629 stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_GLOBAL_INSTANCE_RESETTIME);
630
631 stmt->setUInt32(0, next_reset);
632 stmt->setUInt16(1, uint16(mapid));
633 stmt->setUInt8(2, uint8(difficulty));
634
635 CharacterDatabase.Execute(stmt);
636 }
637
638 // note: this isn't fast but it's meant to be executed very rarely
639 Map const* map = sMapMgr->CreateBaseMap(mapid); // _not_ include difficulty
640 MapInstanced::InstancedMaps &instMaps = ((MapInstanced*)map)->GetInstancedMaps();
641 MapInstanced::InstancedMaps::iterator mitr;
642 uint32 timeLeft;
643
644 for (mitr = instMaps.begin(); mitr != instMaps.end(); ++mitr)
645 {
646 Map* map2 = mitr->second;
647 if (!map2->IsDungeon())
648 continue;
649
650 if (warn)
651 {
652 if (now >= resetTime)

658 }
659 else

661 }

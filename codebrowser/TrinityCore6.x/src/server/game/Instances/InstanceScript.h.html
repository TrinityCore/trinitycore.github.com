299template<class AI, class T>
300AI* GetInstanceAI(T* obj, char const* scriptName)
301{
302 if (InstanceMap* instance = obj->GetMap()->ToInstanceMap())
303 if (instance->GetInstanceScript())
304 if (instance->GetScriptId() == sObjectMgr->GetScriptId(scriptName))
305 return new AI(obj);
306
307 return NULL;
308}
309
310template<class AI, class T>
311AI* GetInstanceAI(T* obj)
312{
313 if (InstanceMap* instance = obj->GetMap()->ToInstanceMap())
314 if (instance->GetInstanceScript())
315 return new AI(obj);
316
317 return NULL;
318}

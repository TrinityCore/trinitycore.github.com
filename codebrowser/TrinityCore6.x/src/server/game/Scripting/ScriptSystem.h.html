80 ScriptPointVector const& GetPointMoveList(uint32 creatureEntry) const
81 {
82 PointMoveMap::const_iterator itr = m_mPointMoveMap.find(creatureEntry);
83
84 if (itr == m_mPointMoveMap.end())
85 return _empty;
86
87 return itr->second;
88 }

90 protected:

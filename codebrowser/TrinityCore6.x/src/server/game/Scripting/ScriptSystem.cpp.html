26void SystemMgr::LoadScriptWaypoints()
27{
28 uint32 oldMSTime = getMSTime();
29
30 // Drop Existing Waypoint list
31 m_mPointMoveMap.clear();
32
33 uint64 uiCreatureCount = 0;
34
35 // Load Waypoints
36 QueryResult result = WorldDatabase.Query("SELECT COUNT(entry) FROM script_waypoint GROUP BY entry");
37 if (result)
38 uiCreatureCount = result->GetRowCount();
39
40 TC_LOG_INFO("server.loading", "Loading Script Waypoints for " UI64FMTD " creature(s)...", uiCreatureCount);
41
42 // 0 1 2 3 4 5
43 result = WorldDatabase.Query("SELECT entry, pointid, location_x, location_y, location_z, waittime FROM script_waypoint ORDER BY pointid");
44
45 if (!result)
46 {
47 TC_LOG_INFO("server.loading", ">> Loaded 0 Script Waypoints. DB table `script_waypoint` is empty.");
48 return;
49 }
50
51 uint32 count = 0;
52
53 do
54 {
55 Field* pFields = result->Fetch();
56 ScriptPointMove temp;
57
58 temp.uiCreatureEntry = pFields[0].GetUInt32();
59 uint32 uiEntry = temp.uiCreatureEntry;
60 temp.uiPointId = pFields[1].GetUInt32();
61 temp.fX = pFields[2].GetFloat();
62 temp.fY = pFields[3].GetFloat();
63 temp.fZ = pFields[4].GetFloat();
64 temp.uiWaitTime = pFields[5].GetUInt32();
65
66 CreatureTemplate const* pCInfo = sObjectMgr->GetCreatureTemplate(temp.uiCreatureEntry);
67
68 if (!pCInfo)
69 {
70 TC_LOG_ERROR("sql.sql", "TSCR: DB table script_waypoint has waypoint for non-existant creature entry %u", temp.uiCreatureEntry);
71 continue;
72 }
73
74 if (!pCInfo->ScriptID)
75 TC_LOG_ERROR("sql.sql", "TSCR: DB table script_waypoint has waypoint for creature entry %u, but creature does not have ScriptName defined and then useless.", temp.uiCreatureEntry);
76
77 m_mPointMoveMap[uiEntry].push_back(temp);
78 ++count;
79 }
80 while (result->NextRow());
81
82 TC_LOG_INFO("server.loading", ">> Loaded %u Script Waypoint nodes in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
83}

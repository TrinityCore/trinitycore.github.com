48void MapInstanced::Update(const uint32 t)
49{
50 // take care of loaded GridMaps (when unused, unload it!)
51 Map::Update(t);
52
53 // update the instanced maps
54 InstancedMaps::iterator i = m_InstancedMaps.begin();
55
56 while (i != m_InstancedMaps.end())
57 {
58 if (i->second->CanUnload(t))
59 {
60 if (!DestroyInstance(i)) // iterator incremented
61 {
62 //m_unloadTimer
63 }
64 }
65 else
66 {
67 // update only here, because it may schedule some bad things before delete
68 if (sMapMgr->GetMapUpdater()->activated())
69 sMapMgr->GetMapUpdater()->schedule_update(*i->second, t);
70 else
71 i->second->Update(t);
72 ++i;
73 }
74 }
75}

93void MapInstanced::UnloadAll()
94{
95 // Unload instanced maps
96 for (InstancedMaps::iterator i = m_InstancedMaps.begin(); i != m_InstancedMaps.end(); ++i)
97 i->second->UnloadAll();
98
99 // Delete the maps only after everything is unloaded to prevent crashes
100 for (InstancedMaps::iterator i = m_InstancedMaps.begin(); i != m_InstancedMaps.end(); ++i)
101 delete i->second;
102
103 m_InstancedMaps.clear();
104
105 // Unload own grids (just dummy(placeholder) grids, neccesary to unload GridMaps!)
106 Map::UnloadAll();
107}

109/*
110- return the right instance for the object, based on its InstanceId
111- create the instance if it's not created already
112- the player is not actually added to the instance (only in InstanceMap::Add)
113*/
114Map* MapInstanced::CreateInstanceForPlayer(const uint32 mapId, Player* player)
115{
116 if (GetId() != mapId || !player)
117 return NULL;
118
119 Map* map = NULL;
120 uint32 newInstanceId = 0; // instanceId of the resulting map
121
122 if (IsBattlegroundOrArena())
123 {
124 // instantiate or find existing bg map for player
125 // the instance id is set in battlegroundid
126 newInstanceId = player->GetBattlegroundId();
127 if (!newInstanceId)
128 return NULL;
129
130 map = sMapMgr->FindMap(mapId, newInstanceId);
131 if (!map)
132 {
133 if (Battleground* bg = player->GetBattleground())
134 map = CreateBattleground(newInstanceId, bg);
135 else
136 {
137 player->TeleportToBGEntryPoint();
138 return NULL;
139 }
140 }
141 }
142 else if (!IsGarrison())
143 {
144 InstancePlayerBind* pBind = player->GetBoundInstance(GetId(), player->GetDifficultyID(GetEntry()));
145 InstanceSave* pSave = pBind ? pBind->save : NULL;
146
147 // the player's permanent player bind is taken into consideration first
148 // then the player's group bind and finally the solo bind.
149 if (!pBind || !pBind->perm)
150 {
151 InstanceGroupBind* groupBind = NULL;
152 Group* group = player->GetGroup();
153 // use the player's difficulty setting (it may not be the same as the group's)
154 if (group)
155 {
156 groupBind = group->GetBoundInstance(this);
157 if (groupBind)
158 pSave = groupBind->save;
159 }
160 }
161 if (pSave)
162 {
163 // solo/perm/group
164 newInstanceId = pSave->GetInstanceId();
165 map = FindInstanceMap(newInstanceId);
166 // it is possible that the save exists but the map doesn't
167 if (!map)
168 map = CreateInstance(newInstanceId, pSave, pSave->GetDifficultyID());
169 }
170 else
171 {
172 // if no instanceId via group members or instance saves is found
173 // the instance will be created for the first time
174 newInstanceId = sMapMgr->GenerateInstanceId();
175
176 Difficulty diff = player->GetGroup() ? player->GetGroup()->GetDifficultyID(GetEntry()) : player->GetDifficultyID(GetEntry());
177 //Seems it is now possible, but I do not know if it should be allowed
178 //ASSERT(!FindInstanceMap(NewInstanceId));
179 map = FindInstanceMap(newInstanceId);
180 if (!map)
181 map = CreateInstance(newInstanceId, NULL, diff);
182 }
183 }
184 else
185 {
186 newInstanceId = player->GetGUID().GetCounter();
187 map = FindInstanceMap(newInstanceId);
188 if (!map)
189 map = CreateGarrison(newInstanceId, player);
190 }
191
192 return map;
193}

195InstanceMap* MapInstanced::CreateInstance(uint32 InstanceId, InstanceSave* save, Difficulty difficulty)
196{
197 // load/create a map
198 std::lock_guard<std::mutex> lock(_mapLock);
199
200 // make sure we have a valid map id
201 const MapEntry* entry = sMapStore.LookupEntry(GetId());
202 if (!entry)
203 {
204 TC_LOG_ERROR("maps", "CreateInstance: no entry for map %d", GetId());
205 ABORT();
206 }
207 const InstanceTemplate* iTemplate = sObjectMgr->GetInstanceTemplate(GetId());
208 if (!iTemplate)
209 {
210 TC_LOG_ERROR("maps", "CreateInstance: no instance template for map %d", GetId());
211 ABORT();
212 }
213
214 // some instances only have one difficulty
215 GetDownscaledMapDifficultyData(GetId(), difficulty);
216
217 TC_LOG_DEBUG("maps", "MapInstanced::CreateInstance: %s map instance %d for %d created with difficulty %s", save ? "" : "new ", InstanceId, GetId(), difficulty ? "heroic" : "normal");
218
219 InstanceMap* map = new InstanceMap(GetId(), GetGridExpiry(), InstanceId, difficulty, this);
220 ASSERT(map->IsDungeon());
221
222 map->LoadRespawnTimes();
223 map->LoadCorpseData();
224
225 bool load_data = save != NULL;
226 map->CreateInstanceData(load_data);
227
228 m_InstancedMaps[InstanceId] = map;
229 return map;
230}

232BattlegroundMap* MapInstanced::CreateBattleground(uint32 InstanceId, Battleground* bg)
233{
234 // load/create a map
235 std::lock_guard<std::mutex> lock(_mapLock);
236
237 TC_LOG_DEBUG("maps", "MapInstanced::CreateBattleground: map bg %d for %d created.", InstanceId, GetId());
238
239 BattlegroundMap* map = new BattlegroundMap(GetId(), GetGridExpiry(), InstanceId, this, DIFFICULTY_NONE);
240 ASSERT(map->IsBattlegroundOrArena());
241 map->SetBG(bg);
242 bg->SetBgMap(map);
243
244 m_InstancedMaps[InstanceId] = map;
245 return map;
246}

259// increments the iterator after erase
260bool MapInstanced::DestroyInstance(InstancedMaps::iterator &itr)
261{
262 itr->second->RemoveAllPlayers();
263 if (itr->second->HavePlayers())
264 {
265 ++itr;
266 return false;
267 }
268
269 itr->second->UnloadAll();
270 // should only unload VMaps if this is the last instance and grid unloading is enabled
271 if (m_InstancedMaps.size() <= 1 && sWorld->getBoolConfig(CONFIG_GRID_UNLOAD))
272 {
273 VMAP::VMapFactory::createOrGetVMapManager()->unloadMap(itr->second->GetId());
274 MMAP::MMapFactory::createOrGetMMapManager()->unloadMap(itr->second->GetId());
275 // in that case, unload grids of the base map, too
276 // so in the next map creation, (EnsureGridCreated actually) VMaps will be reloaded
277 Map::UnloadAll();
278 }
279
280 // Free up the instance id and allow it to be reused for bgs and arenas (other instances are handled in the InstanceSaveMgr)
281 if (itr->second->IsBattlegroundOrArena())
282 sMapMgr->FreeInstanceId(itr->second->GetInstanceId());
283
284 // erase map
285 delete itr->second;
286 m_InstancedMaps.erase(itr++);
287
288 return true;
289}

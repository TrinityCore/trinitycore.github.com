789template<class T, class CONTAINER>
790inline void Map::Visit(Cell const& cell, TypeContainerVisitor<T, CONTAINER>& visitor)
791{
792 const uint32 x = cell.GridX();
793 const uint32 y = cell.GridY();
794 const uint32 cell_x = cell.CellX();
795 const uint32 cell_y = cell.CellY();
796
797 if (!cell.NoCreate() || IsGridLoaded(GridCoord(x, y)))
798 {
799 EnsureGridLoaded(cell);
800 getNGrid(x, y)->VisitGrid(cell_x, cell_y, visitor);
801 }
802}

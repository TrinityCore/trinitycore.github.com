65Map* MapManager::CreateBaseMap(uint32 id)
66{
67 Map* map = FindBaseMap(id);
68
69 if (map == NULL)
70 {
71 std::lock_guard<std::mutex> lock(_mapsLock);
72
73 MapEntry const* entry = sMapStore.LookupEntry(id);
74 ASSERT(entry);
75
76 if (entry->Instanceable())
77 map = new MapInstanced(id, i_gridCleanUpDelay);
78 else
79 {
80 map = new Map(id, i_gridCleanUpDelay, 0, DIFFICULTY_NONE);
81 map->LoadRespawnTimes();
82 map->LoadCorpseData();
83 }
84
85 i_maps[id] = map;
86 }
87
88 ASSERT(map);
89 return map;
90}

110Map* MapManager::FindMap(uint32 mapid, uint32 instanceId) const
111{
112 Map* map = FindBaseMap(mapid);

122bool MapManager::CanPlayerEnter(uint32 mapid, Player* player, bool loginCheck)
123{
124 MapEntry const* entry = sMapStore.LookupEntry(mapid);
125 if (!entry)
126 return false;
127
128 if (!entry->IsDungeon())
129 return true;
130
131 InstanceTemplate const* instance = sObjectMgr->GetInstanceTemplate(mapid);
132 if (!instance)
133 return false;
134
135 Difficulty targetDifficulty, requestedDifficulty;
136 targetDifficulty = requestedDifficulty = player->GetDifficultyID(entry);
137 // Get the highest available difficulty if current setting is higher than the instance allows
138 MapDifficultyEntry const* mapDiff = GetDownscaledMapDifficultyData(entry->ID, targetDifficulty);
139 if (!mapDiff)
140 {
141 player->SendTransferAborted(mapid, TRANSFER_ABORT_DIFFICULTY, requestedDifficulty);
142 return false;
143 }
144
145 //Bypass checks for GMs
146 if (player->IsGameMaster())
147 return true;
148
149 char const* mapName = entry->MapName_lang;
150
151 Group* group = player->GetGroup();
152 if (entry->IsRaid())
153 {
154 // can only enter in a raid group
155 if ((!group || !group->isRaidGroup()) && !sWorld->getBoolConfig(CONFIG_INSTANCE_IGNORE_RAID))
156 {
157 /// @todo this is not a good place to send the message
158 player->SendRaidGroupOnlyMessage(RAID_GROUP_ERR_ONLY, 0);
159 TC_LOG_DEBUG("maps", "MAP: Player '%s' must be in a raid group to enter instance '%s'", player->GetName().c_str(), mapName);
160 return false;
161 }
162 }
163
164 if (!player->IsAlive())
165 {
166 if (player->HasCorpse())
167 {
168 // let enter in ghost mode in instance that connected to inner instance with corpse
169 uint32 corpseMap = player->GetCorpseLocation().GetMapId();
170 do
171 {
172 if (corpseMap == mapid)
173 break;
174
175 InstanceTemplate const* corpseInstance = sObjectMgr->GetInstanceTemplate(corpseMap);
176 corpseMap = corpseInstance ? corpseInstance->Parent : 0;
177 } while (corpseMap);
178
179 if (!corpseMap)
180 {
181 WorldPackets::Misc::AreaTriggerNoCorpse packet;
182 player->GetSession()->SendPacket(packet.Write());
183 TC_LOG_DEBUG("maps", "MAP: Player '%s' does not have a corpse in instance '%s' and cannot enter.", player->GetName().c_str(), mapName);
184 return false;
185 }
186
187 TC_LOG_DEBUG("maps", "MAP: Player '%s' has corpse in instance '%s' and can enter.", player->GetName().c_str(), mapName);
188 }
189 else
190 TC_LOG_DEBUG("maps", "Map::CanPlayerEnter - player '%s' is dead but does not have a corpse!", player->GetName().c_str());
191 }
192
193 //Get instance where player's group is bound & its map
194 if (group)
195 {
196 InstanceGroupBind* boundInstance = group->GetBoundInstance(entry);
197 if (boundInstance && boundInstance->save)
198 if (Map* boundMap = sMapMgr->FindMap(mapid, boundInstance->save->GetInstanceId()))
199 if (!loginCheck && !boundMap->CanEnter(player))
200 return false;
201 /*
202 This check has to be moved to InstanceMap::CanEnter()
203 // Player permanently bounded to different instance than groups one
204 InstancePlayerBind* playerBoundedInstance = player->GetBoundInstance(mapid, player->GetDifficultyID(entry));
205 if (playerBoundedInstance && playerBoundedInstance->perm && playerBoundedInstance->save &&
206 boundedInstance->save->GetInstanceId() != playerBoundedInstance->save->GetInstanceId())
207 {
208 /// @todo send some kind of error message to the player
209 return false;
210 }*/
211 }
212
213 // players are only allowed to enter 5 instances per hour
214 if (entry->IsDungeon() && (!player->GetGroup() || (player->GetGroup() && !player->GetGroup()->isLFGGroup())))
215 {
216 uint32 instanceIdToCheck = 0;
217 if (InstanceSave* save = player->GetInstanceSave(mapid))
218 instanceIdToCheck = save->GetInstanceId();
219
220 // instanceId can never be 0 - will not be found
221 if (!player->CheckInstanceCount(instanceIdToCheck) && !player->isDead())
222 {
223 player->SendTransferAborted(mapid, TRANSFER_ABORT_TOO_MANY_INSTANCES);
224 return false;
225 }
226 }
227
228 //Other requirements
229 return player->Satisfy(sObjectMgr->GetAccessRequirement(mapid, targetDifficulty), mapid, true);
230}

232void MapManager::Update(uint32 diff)
233{
234 i_timer.Update(diff);
235 if (!i_timer.Passed())
236 return;
237
238 MapMapType::iterator iter = i_maps.begin();
239 for (; iter != i_maps.end(); ++iter)
240 {
241 if (m_updater.activated())
242 m_updater.schedule_update(*iter->second, uint32(i_timer.GetCurrent()));
243 else
244 iter->second->Update(uint32(i_timer.GetCurrent()));
245 }
246 if (m_updater.activated())
247 m_updater.wait();
248
249 for (iter = i_maps.begin(); iter != i_maps.end(); ++iter)
250 iter->second->DelayedUpdate(uint32(i_timer.GetCurrent()));
251
252 i_timer.SetCurrent(0);
253}

257bool MapManager::ExistMapAndVMap(uint32 mapid, float x, float y)
258{
259 GridCoord p = Trinity::ComputeGridCoord(x, y);
260
261 int gx = (MAX_NUMBER_OF_GRIDS - 1) - p.x_coord;
262 int gy = (MAX_NUMBER_OF_GRIDS - 1) - p.y_coord;
263
264 return Map::ExistMap(mapid, gx, gy) && Map::ExistVMap(mapid, gx, gy);
265}

294uint32 MapManager::GetNumInstances()
295{
296 std::lock_guard<std::mutex> lock(_mapsLock);
297
298 uint32 ret = 0;
299 for (MapMapType::iterator itr = i_maps.begin(); itr != i_maps.end(); ++itr)
300 {
301 Map* map = itr->second;
302 if (!map->Instanceable())
303 continue;
304 MapInstanced::InstancedMaps &maps = ((MapInstanced*)map)->GetInstancedMaps();
305 for (MapInstanced::InstancedMaps::iterator mitr = maps.begin(); mitr != maps.end(); ++mitr)
306 if (mitr->second->IsDungeon()) ret++;
307 }
308 return ret;
309}

329void MapManager::InitInstanceIds()
330{
331 _nextInstanceId = 1;
332
333 QueryResult result = CharacterDatabase.Query("SELECT MAX(id) FROM instance");
334 if (result)
335 {
336 uint32 maxId = (*result)[0].GetUInt32();
337
338 // Resize to multiples of 32 (vector<bool> allocates memory the same way)
339 _instanceIds.resize((maxId / 32) * 32 + (maxId % 32 > 0 ? 32 : 0));
340 }
341}

349uint32 MapManager::GenerateInstanceId()
350{
351 uint32 newInstanceId = _nextInstanceId;
352
353 // Find the lowest available id starting from the current NextInstanceId (which should be the lowest according to the logic in FreeInstanceId()
354 for (uint32 i = ++_nextInstanceId; i < 0xFFFFFFFF; ++i)
355 {
356 if ((i < _instanceIds.size() && !_instanceIds[i]) || i >= _instanceIds.size())
357 {
358 _nextInstanceId = i;
359 break;
360 }
361 }
362
363 if (newInstanceId == _nextInstanceId)
364 {
365 TC_LOG_ERROR("maps", "Instance ID overflow!! Can't continue, shutting down server. ");
366 World::StopNow(ERROR_EXIT_CODE);
367 }
368
369 // Allocate space if necessary
370 if (newInstanceId >= uint32(_instanceIds.size()))
371 {
372 // Due to the odd memory allocation behavior of vector<bool> we match size to capacity before triggering a new allocation
373 if (_instanceIds.size() < _instanceIds.capacity())
374 {
375 _instanceIds.resize(_instanceIds.capacity());
376 }
377 else
378 _instanceIds.resize((newInstanceId / 32) * 32 + (newInstanceId % 32 > 0 ? 32 : 0));
379 }
380
381 _instanceIds[newInstanceId] = true;
382
383 return newInstanceId;
384}

386void MapManager::FreeInstanceId(uint32 instanceId)
387{
388 // If freed instance id is lower than the next id available for new instances, use the freed one instead
389 if (instanceId < _nextInstanceId)
390 SetNextInstanceId(instanceId);
391
392 _instanceIds[instanceId] = false;
393 sAchievementMgr->OnInstanceDestroyed(instanceId);
394}

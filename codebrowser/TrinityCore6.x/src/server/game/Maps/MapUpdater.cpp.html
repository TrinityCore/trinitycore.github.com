19#include "MapUpdater.h"
20#include "Map.h"
21
22#include <mutex>
23
24
25class MapUpdateRequest
26{
27 private:
28
29 Map& m_map;
30 MapUpdater& m_updater;
31 uint32 m_diff;
32
33 public:
34
35 MapUpdateRequest(Map& m, MapUpdater& u, uint32 d)
36 : m_map(m), m_updater(u), m_diff(d)
37 {
38 }
39
40 void call()
41 {
42 m_map.Update (m_diff);
43 m_updater.update_finished();
44 }
45};
46
47void MapUpdater::activate(size_t num_threads)
48{
49 for (size_t i = 0; i < num_threads; ++i)
50 {
51 _workerThreads.push_back(std::thread(&MapUpdater::WorkerThread, this));
52 }
53}
54
55void MapUpdater::deactivate()
56{
57 _cancelationToken = true;
58
59 wait();
60
61 _queue.Cancel();
62
63 for (auto& thread : _workerThreads)
64 {
65 thread.join();
66 }
67}
68
69void MapUpdater::wait()
70{
71 std::unique_lock<std::mutex> lock(_lock);
72
73 while (pending_requests > 0)
74 _condition.wait(lock);
75
76 lock.unlock();
77}
78
79void MapUpdater::schedule_update(Map& map, uint32 diff)
80{
81 std::lock_guard<std::mutex> lock(_lock);
82
83 ++pending_requests;
84
85 _queue.Push(new MapUpdateRequest(map, *this, diff));
86}
87
88bool MapUpdater::activated()
89{
90 return _workerThreads.size() > 0;
91}
92
93void MapUpdater::update_finished()
94{
95 std::lock_guard<std::mutex> lock(_lock);
96
97 --pending_requests;
98
99 _condition.notify_all();
100}
101
102void MapUpdater::WorkerThread()
103{
104 while (1)
105 {
106 MapUpdateRequest* request = nullptr;
107
108 _queue.WaitAndPop(request);
109
110 if (_cancelationToken)
111 return;
112
113 request->call();
114
115 delete request;
116 }
117}

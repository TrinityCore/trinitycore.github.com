159template<typename Worker>
160void MapManager::DoForAllMaps(Worker&& worker)
161{
162 std::lock_guard<std::mutex> lock(_mapsLock);
163
164 for (auto& mapPair : i_maps)
165 {
166 Map* map = mapPair.second;
167 if (MapInstanced* mapInstanced = map->ToMapInstanced())
168 {
169 MapInstanced::InstancedMaps& instances = mapInstanced->GetInstancedMaps();
170 for (auto& instancePair : instances)
171 worker(instancePair.second);
172 }
173 else
174 worker(map);
175 }
176}
177
178template<typename Worker>
179inline void MapManager::DoForAllMapsWithMapId(uint32 mapId, Worker&& worker)
180{
181 std::lock_guard<std::mutex> lock(_mapsLock);
182
183 auto itr = i_maps.find(mapId);
184 if (itr != i_maps.end())
185 {
186 Map* map = itr->second;
187 if (MapInstanced* mapInstanced = map->ToMapInstanced())
188 {
189 MapInstanced::InstancedMaps& instances = mapInstanced->GetInstancedMaps();
190 for (auto& p : instances)
191 worker(p.second);
192 }
193 else
194 worker(map);
195 }
196}

40struct KeyFrame
41{
42 explicit KeyFrame(TaxiPathNodeEntry const* node) : Index(0), Node(node), InitialOrientation(0.0f),
43 DistSinceStop(-1.0f), DistUntilStop(-1.0f), DistFromPrev(-1.0f), TimeFrom(0.0f), TimeTo(0.0f),
44 Teleport(false), ArriveTime(0), DepartureTime(0), Spline(NULL), NextDistFromPrev(0.0f), NextArriveTime(0)
45 {
46 }
47
48 uint32 Index;
49 TaxiPathNodeEntry const* Node;
50 float InitialOrientation;
51 float DistSinceStop;
52 float DistUntilStop;
53 float DistFromPrev;
54 float TimeFrom;
55 float TimeTo;
56 bool Teleport;
57 uint32 ArriveTime;
58 uint32 DepartureTime;
59 TransportSpline* Spline;
60
61 // Data needed for next frame
62 float NextDistFromPrev;
63 uint32 NextArriveTime;
64
65 bool IsTeleportFrame() const { return Teleport; }
66 bool IsStopFrame() const { return (Node->Flags & TAXI_PATH_NODE_FLAG_STOP) != 0; }
67};

100class TransportMgr
101{
102 friend void DB2Manager::LoadStores(std::string const&, uint32);
103
104 public:
105 static TransportMgr* instance()
106 {
107 static TransportMgr instance;
108 return &instance;
109 }
110
111 void Unload();
112
113 void LoadTransportTemplates();
114
115 // Creates a transport using given GameObject template entry
116 Transport* CreateTransport(uint32 entry, ObjectGuid::LowType guid = UI64LIT(0), Map* map = nullptr, uint32 phaseid = 0, uint32 phasegroup = 0);
117
118 // Spawns all continent transports, used at core startup
119 void SpawnContinentTransports();
120
121 // creates all transports for instance
122 void CreateInstanceTransports(Map* map);
123
124 TransportTemplate const* GetTransportTemplate(uint32 entry) const
125 {
126 TransportTemplates::const_iterator itr = _transportTemplates.find(entry);
127 if (itr != _transportTemplates.end())
128 return &itr->second;
129 return NULL;
130 }
131
132 TransportAnimation const* GetTransportAnimInfo(uint32 entry) const
133 {
134 TransportAnimationContainer::const_iterator itr = _transportAnimations.find(entry);
135 if (itr != _transportAnimations.end())
136 return &itr->second;
137
138 return NULL;
139 }
140
141 private:
142 TransportMgr();
143 ~TransportMgr();
144 TransportMgr(TransportMgr const&);
145 TransportMgr& operator=(TransportMgr const&);
146
147 // Generates and precaches a path for transport to avoid generation each time transport instance is created
148 void GeneratePath(GameObjectTemplate const* goInfo, TransportTemplate* transport);
149
150 void AddPathNodeToTransport(uint32 transportEntry, uint32 timeSeg, TransportAnimationEntry const* node);
151
152 void AddPathRotationToTransport(uint32 transportEntry, uint32 timeSeg, TransportRotationEntry const* node)
153 {
154 _transportAnimations[transportEntry].Rotations[timeSeg] = node;
155 }
156
157 // Container storing transport templates
158 TransportTemplates _transportTemplates;
159
160 // Container storing transport entries to create for instanced maps
161 TransportInstanceMap _instanceTransports;
162
163 TransportAnimationContainer _transportAnimations;
164};

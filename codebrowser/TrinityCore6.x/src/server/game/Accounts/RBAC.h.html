18/**
19* @file RBAC.h
20* @brief Role Based Access Control related classes definition
21*
22* This file contains all the classes and enums used to implement
23* Role Based Access Control
24*
25* RBAC Rules:
26* - Pemission: Defines an autorization to perform certain operation.
27* - Role: Set of permissions.
28* - Group: Set of roles.
29* - An Account can have multiple groups, roles and permissions.
30* - Account Groups can only be granted or revoked
31* - Account Roles and Permissions can be granted, denied or revoked
32* - Grant: Assignment of the object (role/permission) and allow it
33* - Deny: Assignment of the object (role/permission) and deny it
34* - Revoke: Removal of the object (role/permission) no matter if it was granted or denied
35* - Global Permissions are computed as:
36* Group Grants + Role Grants + User Grans - Role Grants - User Grants
37* - Groups, Roles and Permissions can be assigned by realm
38*/
39

771/**
772 * @name RBACData
773 * @brief Contains all needed information about the acccount
774 *
775 * This class contains all the data needed to calculate the account permissions.
776 * RBACDAta is formed by granted and denied permissions and all the inherited permissions
777 *
778 * Calculation of current Permissions: Granted permissions - Denied permissions
779 * - Granted permissions: through linked permissions and directly assigned
780 * - Denied permissions: through linked permissions and directly assigned
781 */
782class RBACData
783{
784 public:
785 RBACData(uint32 id, std::string const& name, int32 realmId, uint8 secLevel = 255):
786 _id(id), _name(name), _realmId(realmId), _secLevel(secLevel),
787 _grantedPerms(), _deniedPerms(), _globalPerms() { }
788
789 /// Gets the Name of the Object
790 std::string const& GetName() const { return _name; }
791 /// Gets the Id of the Object
792 uint32 GetId() const { return _id; }
793
794 /**
795 * @name HasPermission
796 * @brief Checks if certain action is allowed
797 *
798 * Checks if certain action can be performed.
799 *
800 * @return grant or deny action
801 *
802 * Example Usage:
803 * @code
804 * bool Player::CanJoinArena(Battleground* bg)
805 * {
806 * return bg->isArena() && HasPermission(RBAC_PERM_JOIN_ARENA);
807 * }
808 * @endcode
809 */
810 bool HasPermission(uint32 permission) const
811 {
812 return _globalPerms.find(permission) != _globalPerms.end();
813 }

826 *

831 *
832 * @param permissionId permission to be granted
833 * @param realmId realm affected
834 *
835 * @return Success or failure (with reason) to grant the permission
836 *
837 * Example Usage:

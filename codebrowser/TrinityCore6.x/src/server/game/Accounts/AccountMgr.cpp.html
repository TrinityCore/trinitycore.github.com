36AccountOpResult AccountMgr::CreateAccount(std::string username, std::string password, std::string email /*= ""*/, uint32 bnetAccountId /*= 0*/, uint8 bnetIndex /*= 0*/)
37{
38 if (utf8length(username) > MAX_ACCOUNT_STR)
39 return AccountOpResult::AOR_NAME_TOO_LONG; // username's too long
40
41 Utf8ToUpperOnlyLatin(username);
42 Utf8ToUpperOnlyLatin(password);
43 Utf8ToUpperOnlyLatin(email);
44
45 if (GetId(username))
46 return AccountOpResult::AOR_NAME_ALREADY_EXIST; // username does already exist
47
48 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_ACCOUNT);
49
50 stmt->setString(0, username);
51 stmt->setString(1, CalculateShaPassHash(username, password));
52 stmt->setString(2, email);
53 stmt->setString(3, email);
54 if (bnetAccountId && bnetIndex)
55 {
56 stmt->setUInt32(4, bnetAccountId);
57 stmt->setUInt8(5, bnetIndex);
58 }
59 else
60 {
61 stmt->setNull(4);
62 stmt->setNull(5);
63 }
64
65 LoginDatabase.DirectExecute(stmt); // Enforce saving, otherwise AddGroup can fail
66
67 stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_REALM_CHARACTERS_INIT);
68 LoginDatabase.Execute(stmt);
69
70 return AccountOpResult::AOR_OK; // everything's fine
71}

73AccountOpResult AccountMgr::DeleteAccount(uint32 accountId)
74{
75 // Check if accounts exists
76 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_ACCOUNT_BY_ID);
77 stmt->setUInt32(0, accountId);
78 PreparedQueryResult result = LoginDatabase.Query(stmt);
79
80 if (!result)
81 return AccountOpResult::AOR_NAME_NOT_EXIST;
82
83 // Obtain accounts characters
84 stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHARS_BY_ACCOUNT_ID);
85
86 stmt->setUInt32(0, accountId);
87
88 result = CharacterDatabase.Query(stmt);
89
90 if (result)
91 {
92 do
93 {
94 ObjectGuid guid = ObjectGuid::Create<HighGuid::Player>((*result)[0].GetUInt64());
95
96 // Kick if player is online
97 if (Player* p = ObjectAccessor::FindConnectedPlayer(guid))
98 {
99 WorldSession* s = p->GetSession();
100 s->KickPlayer(); // mark session to remove at next session list update
101 s->LogoutPlayer(false); // logout player without waiting next session list update
102 }
103
104 Player::DeleteFromDB(guid, accountId, false); // no need to update realm characters
105 } while (result->NextRow());
106 }
107
108 // table realm specific but common for all characters of account for realm
109 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_TUTORIALS);
110 stmt->setUInt32(0, accountId);
111 CharacterDatabase.Execute(stmt);
112
113 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_ACCOUNT_DATA);
114 stmt->setUInt32(0, accountId);
115 CharacterDatabase.Execute(stmt);
116
117 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHARACTER_BAN);
118 stmt->setUInt32(0, accountId);
119 CharacterDatabase.Execute(stmt);
120
121 SQLTransaction trans = LoginDatabase.BeginTransaction();
122
123 stmt = LoginDatabase.GetPreparedStatement(LOGIN_DEL_ACCOUNT);
124 stmt->setUInt32(0, accountId);
125 trans->Append(stmt);
126
127 stmt = LoginDatabase.GetPreparedStatement(LOGIN_DEL_ACCOUNT_ACCESS);
128 stmt->setUInt32(0, accountId);
129 trans->Append(stmt);
130
131 stmt = LoginDatabase.GetPreparedStatement(LOGIN_DEL_REALM_CHARACTERS);
132 stmt->setUInt32(0, accountId);
133 trans->Append(stmt);
134
135 stmt = LoginDatabase.GetPreparedStatement(LOGIN_DEL_ACCOUNT_BANNED);
136 stmt->setUInt32(0, accountId);
137 trans->Append(stmt);
138
139 LoginDatabase.CommitTransaction(trans);
140
141 return AccountOpResult::AOR_OK;
142}

174AccountOpResult AccountMgr::ChangePassword(uint32 accountId, std::string newPassword)
175{
176 std::string username;
177
178 if (!GetName(accountId, username))
179 {
180 sScriptMgr->OnFailedPasswordChange(accountId);
181 return AccountOpResult::AOR_NAME_NOT_EXIST; // account doesn't exist
182 }
183
184 if (utf8length(newPassword) > MAX_PASS_STR)
185 {
186 sScriptMgr->OnFailedPasswordChange(accountId);
187 return AccountOpResult::AOR_PASS_TOO_LONG;
188 }
189
190 Utf8ToUpperOnlyLatin(username);
191 Utf8ToUpperOnlyLatin(newPassword);
192
193 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_PASSWORD);
194
195 stmt->setString(0, CalculateShaPassHash(username, newPassword));
196 stmt->setUInt32(1, accountId);
197
198 LoginDatabase.Execute(stmt);
199
200 stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_VS);
201
202 stmt->setString(0, "");
203 stmt->setString(1, "");
204 stmt->setString(2, username);
205
206 LoginDatabase.Execute(stmt);
207
208 sScriptMgr->OnPasswordChange(accountId);
209 return AccountOpResult::AOR_OK;
210}

327bool AccountMgr::CheckPassword(uint32 accountId, std::string password)
328{
329 std::string username;
330
331 if (!GetName(accountId, username))
332 return false;
333
334 Utf8ToUpperOnlyLatin(username);
335 Utf8ToUpperOnlyLatin(password);
336
337 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_CHECK_PASSWORD);
338 stmt->setUInt32(0, accountId);
339 stmt->setString(1, CalculateShaPassHash(username, password));
340 PreparedQueryResult result = LoginDatabase.Query(stmt);
341
342 return (result) ? true : false;
343}

345bool AccountMgr::CheckEmail(uint32 accountId, std::string newEmail)
346{
347 std::string oldEmail;
348
349 // We simply return false for a non-existing email
350 if (!GetEmail(accountId, oldEmail))
351 return false;
352
353 Utf8ToUpperOnlyLatin(oldEmail);
354 Utf8ToUpperOnlyLatin(newEmail);
355
356 if (strcmp(oldEmail.c_str(), newEmail.c_str()) == 0)
357 return true;
358
359 return false;
360}

372std::string AccountMgr::CalculateShaPassHash(std::string const& name, std::string const& password)
373{
374 SHA1Hash sha;
375 sha.Initialize();
376 sha.UpdateData(name);
377 sha.UpdateData(":");
378 sha.UpdateData(password);
379 sha.Finalize();
380
381 return ByteArrayToHexStr(sha.GetDigest(), sha.GetLength());
382}

399void AccountMgr::LoadRBAC()
400{
401 ClearRBAC();
402
403 TC_LOG_DEBUG("rbac", "AccountMgr::LoadRBAC");
404 uint32 oldMSTime = getMSTime();
405 uint32 count1 = 0;
406 uint32 count2 = 0;
407 uint32 count3 = 0;
408
409 TC_LOG_DEBUG("rbac", "AccountMgr::LoadRBAC: Loading permissions");
410 QueryResult result = LoginDatabase.Query("SELECT id, name FROM rbac_permissions");
411 if (!result)
412 {
413 TC_LOG_INFO("server.loading", ">> Loaded 0 account permission definitions. DB table `rbac_permissions` is empty.");
414 return;
415 }
416
417 do
418 {
419 Field* field = result->Fetch();
420 uint32 id = field[0].GetUInt32();
421 _permissions[id] = new rbac::RBACPermission(id, field[1].GetString());
422 ++count1;
423 }
424 while (result->NextRow());
425
426 TC_LOG_DEBUG("rbac", "AccountMgr::LoadRBAC: Loading linked permissions");
427 result = LoginDatabase.Query("SELECT id, linkedId FROM rbac_linked_permissions ORDER BY id ASC");
428 if (!result)
429 {
430 TC_LOG_INFO("server.loading", ">> Loaded 0 linked permissions. DB table `rbac_linked_permissions` is empty.");
431 return;
432 }
433
434 uint32 permissionId = 0;
435 rbac::RBACPermission* permission = NULL;
436
437 do
438 {
439 Field* field = result->Fetch();
440 uint32 newId = field[0].GetUInt32();
441 if (permissionId != newId)
442 {
443 permissionId = newId;
444 permission = _permissions[newId];
445 }
446
447 uint32 linkedPermissionId = field[1].GetUInt32();
448 if (linkedPermissionId == permissionId)
449 {
450 TC_LOG_ERROR("sql.sql", "RBAC Permission %u has itself as linked permission. Ignored", permissionId);
451 continue;
452 }
453 permission->AddLinkedPermission(linkedPermissionId);
454 ++count2;
455 }
456 while (result->NextRow());
457
458 TC_LOG_DEBUG("rbac", "AccountMgr::LoadRBAC: Loading default permissions");
459 result = LoginDatabase.PQuery("SELECT secId, permissionId FROM rbac_default_permissions WHERE (realmId = %u OR realmId = -1) ORDER BY secId ASC", realm.Id.Realm);
460 if (!result)
461 {
462 TC_LOG_INFO("server.loading", ">> Loaded 0 default permission definitions. DB table `rbac_default_permissions` is empty.");
463 return;
464 }
465
466 uint8 secId = 255;
467 rbac::RBACPermissionContainer* permissions = NULL;
468 do
469 {
470 Field* field = result->Fetch();
471 uint32 newId = field[0].GetUInt32();
472 if (secId != newId || permissions == NULL)
473 {
474 secId = newId;
475 permissions = &_defaultPermissions[secId];
476 }
477
478 permissions->insert(field[1].GetUInt32());
479 ++count3;
480 }
481 while (result->NextRow());
482
483 TC_LOG_INFO("server.loading", ">> Loaded %u permission definitions, %u linked permissions and %u default permissions in %u ms", count1, count2, count3, GetMSTimeDiffToNow(oldMSTime));
484}

486void AccountMgr::UpdateAccountAccess(rbac::RBACData* rbac, uint32 accountId, uint8 securityLevel, int32 realmId)
487{
488 if (rbac && securityLevel == rbac->GetSecurityLevel())
489 rbac->SetSecurityLevel(securityLevel);
490
491 // Delete old security level from DB
492 if (realmId == -1)
493 {
494 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_DEL_ACCOUNT_ACCESS);
495 stmt->setUInt32(0, accountId);
496 LoginDatabase.Execute(stmt);
497 }
498 else
499 {
500 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_DEL_ACCOUNT_ACCESS_BY_REALM);
501 stmt->setUInt32(0, accountId);
502 stmt->setUInt32(1, realmId);
503 LoginDatabase.Execute(stmt);
504 }
505
506 // Add new security level
507 if (securityLevel)
508 {
509 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_ACCOUNT_ACCESS);
510 stmt->setUInt32(0, accountId);
511 stmt->setUInt8(1, securityLevel);
512 stmt->setInt32(2, realmId);
513 LoginDatabase.Execute(stmt);
514 }
515}

517rbac::RBACPermission const* AccountMgr::GetRBACPermission(uint32 permissionId) const
518{
519 TC_LOG_TRACE("rbac", "AccountMgr::GetRBACPermission: %u", permissionId);
520 rbac::RBACPermissionsContainer::const_iterator it = _permissions.find(permissionId);
521 if (it != _permissions.end())
522 return it->second;
523
524 return NULL;
525}

527bool AccountMgr::HasPermission(uint32 accountId, uint32 permissionId, uint32 realmId)
528{
529 if (!accountId)
530 {
531 TC_LOG_ERROR("rbac", "AccountMgr::HasPermission: Wrong accountId 0");
532 return false;
533 }
534
535 rbac::RBACData rbac(accountId, "", realmId, GetSecurity(accountId));
536 rbac.LoadFromDB();
537 bool hasPermission = rbac.HasPermission(permissionId);
538
539 TC_LOG_DEBUG("rbac", "AccountMgr::HasPermission [AccountId: %u, PermissionId: %u, realmId: %d]: %u",
540 accountId, permissionId, realmId, hasPermission);
541 return hasPermission;
542}

544void AccountMgr::ClearRBAC()
545{
546 for (rbac::RBACPermissionsContainer::iterator itr = _permissions.begin(); itr != _permissions.end(); ++itr)
547 delete itr->second;
548
549 _permissions.clear();
550 _defaultPermissions.clear();
551}

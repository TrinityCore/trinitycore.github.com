26AccountOpResult Battlenet::AccountMgr::CreateBattlenetAccount(std::string email, std::string password, bool withGameAccount /*= true*/)
27{
28 if (utf8length(email) > MAX_BNET_EMAIL_STR)
29 return AccountOpResult::AOR_NAME_TOO_LONG;
30
31 if (utf8length(password) > MAX_PASS_STR)
32 return AccountOpResult::AOR_PASS_TOO_LONG;
33
34 Utf8ToUpperOnlyLatin(email);
35 Utf8ToUpperOnlyLatin(password);
36
37 if (GetId(email))
38 return AccountOpResult::AOR_NAME_ALREADY_EXIST;
39
40 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_BNET_ACCOUNT);
41 stmt->setString(0, email);
42 stmt->setString(1, CalculateShaPassHash(email, password));
43 LoginDatabase.DirectExecute(stmt);
44
45 uint32 newAccountId = GetId(email);
46 ASSERT(newAccountId);
47
48 if (withGameAccount)
49 GameAccountMgr::instance()->CreateAccount(std::to_string(newAccountId) + "#1", password, email, newAccountId, 1);
50
51 return AccountOpResult::AOR_OK;
52}

54AccountOpResult Battlenet::AccountMgr::ChangePassword(uint32 accountId, std::string newPassword)
55{
56 std::string username;
57 if (!GetName(accountId, username))
58 return AccountOpResult::AOR_NAME_NOT_EXIST; // account doesn't exist
59
60 Utf8ToUpperOnlyLatin(username);
61 Utf8ToUpperOnlyLatin(newPassword);
62 if (utf8length(newPassword) > MAX_PASS_STR)
63 return AccountOpResult::AOR_PASS_TOO_LONG;
64
65 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_BNET_PASSWORD);
66 stmt->setString(0, CalculateShaPassHash(username, newPassword));
67 stmt->setUInt32(1, accountId);
68 LoginDatabase.Execute(stmt);
69
70 return AccountOpResult::AOR_OK;
71}

73bool Battlenet::AccountMgr::CheckPassword(uint32 accountId, std::string password)
74{
75 std::string username;
76
77 if (!GetName(accountId, username))
78 return false;
79
80 Utf8ToUpperOnlyLatin(username);
81 Utf8ToUpperOnlyLatin(password);
82
83 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_BNET_CHECK_PASSWORD);
84 stmt->setUInt32(0, accountId);
85 stmt->setString(1, CalculateShaPassHash(username, password));
86
87 return LoginDatabase.Query(stmt) != nullptr;
88}

90AccountOpResult Battlenet::AccountMgr::LinkWithGameAccount(std::string const& email, std::string const& gameAccountName)
91{
92 uint32 bnetAccountId = GetId(email);
93 if (!bnetAccountId)
94 return AccountOpResult::AOR_NAME_NOT_EXIST;
95
96 uint32 gameAccountId = GameAccountMgr::GetId(gameAccountName);
97 if (!gameAccountId)
98 return AccountOpResult::AOR_NAME_NOT_EXIST;
99
100 if (GetIdByGameAccount(gameAccountId))
101 return AccountOpResult::AOR_ACCOUNT_BAD_LINK;
102
103 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_BNET_GAME_ACCOUNT_LINK);
104 stmt->setUInt32(0, bnetAccountId);
105 stmt->setUInt8(1, GetMaxIndex(bnetAccountId) + 1);
106 stmt->setUInt32(2, gameAccountId);
107 LoginDatabase.Execute(stmt);
108 return AccountOpResult::AOR_OK;
109}

111AccountOpResult Battlenet::AccountMgr::UnlinkGameAccount(std::string const& gameAccountName)
112{
113 uint32 gameAccountId = GameAccountMgr::GetId(gameAccountName);
114 if (!gameAccountId)
115 return AccountOpResult::AOR_NAME_NOT_EXIST;
116
117 if (!GetIdByGameAccount(gameAccountId))
118 return AccountOpResult::AOR_ACCOUNT_BAD_LINK;
119
120 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_BNET_GAME_ACCOUNT_LINK);
121 stmt->setNull(0);
122 stmt->setNull(1);
123 stmt->setUInt32(2, gameAccountId);
124 LoginDatabase.Execute(stmt);
125 return AccountOpResult::AOR_OK;
126}

172std::string Battlenet::AccountMgr::CalculateShaPassHash(std::string const& name, std::string const& password)
173{
174 SHA256Hash email;
175 email.UpdateData(name);
176 email.Finalize();
177
178 SHA256Hash sha;
179 sha.UpdateData(ByteArrayToHexStr(email.GetDigest(), email.GetLength()));
180 sha.UpdateData(":");
181 sha.UpdateData(password);
182 sha.Finalize();
183
184 return ByteArrayToHexStr(sha.GetDigest(), sha.GetLength(), true);
185}

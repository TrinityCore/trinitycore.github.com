28inline bool _ModifyUInt32(bool apply, uint32& baseValue, int32& amount)
29{
30 // If amount is negative, change sign and value of apply.
31 if (amount < 0)
32 {
33 apply = !apply;
34 amount = -amount;
35 }
36 if (apply)
37 baseValue += amount;
38 else
39 {
40 // Make sure we do not get uint32 overflow.
41 if (amount > int32(baseValue))
42 amount = baseValue;
43 baseValue -= amount;
44 }
45 return apply;
46}

60void Unit::UpdateDamagePhysical(WeaponAttackType attType)
61{
62 float minDamage = 0.0f;
63 float maxDamage = 0.0f;
64
65 CalculateMinMaxDamage(attType, false, true, minDamage, maxDamage);
66
67 switch (attType)
68 {
69 case BASE_ATTACK:
70 default:
71 SetStatFloatValue(UNIT_FIELD_MINDAMAGE, minDamage);
72 SetStatFloatValue(UNIT_FIELD_MAXDAMAGE, maxDamage);
73 break;
74 case OFF_ATTACK:
75 SetStatFloatValue(UNIT_FIELD_MINOFFHANDDAMAGE, minDamage);
76 SetStatFloatValue(UNIT_FIELD_MAXOFFHANDDAMAGE, maxDamage);
77 break;
78 case RANGED_ATTACK:
79 SetStatFloatValue(UNIT_FIELD_MINRANGEDDAMAGE, minDamage);
80 SetStatFloatValue(UNIT_FIELD_MAXRANGEDDAMAGE, maxDamage);
81 break;
82 }
83}

91bool Player::UpdateStats(Stats stat)
92{
93 if (stat > STAT_SPIRIT)
94 return false;
95
96 // value = ((base_value * base_pct) + total_value) * total_pct
97 float value = GetTotalStatValue(stat);
98
99 SetStat(stat, int32(value));
100
101 if (stat == STAT_STAMINA || stat == STAT_INTELLECT || stat == STAT_STRENGTH)
102 {
103 Pet* pet = GetPet();
104 if (pet)
105 pet->UpdateStats(stat);
106 }
107
108 switch (stat)
109 {
110 case STAT_AGILITY:
111 UpdateArmor();
112 UpdateAllCritPercentages();
113 UpdateDodgePercentage();
114 break;
115 case STAT_STAMINA:
116 UpdateMaxHealth();
117 break;
118 case STAT_INTELLECT:
119 UpdateAllSpellCritChances();
120 UpdateArmor(); //SPELL_AURA_MOD_RESISTANCE_OF_INTELLECT_PERCENT, only armor currently
121 break;
122 case STAT_SPIRIT:
123 break;
124 default:
125 break;
126 }
127
128 if (stat == STAT_STRENGTH)
129 UpdateAttackPowerAndDamage(false);
130 else if (stat == STAT_AGILITY)
131 {
132 UpdateAttackPowerAndDamage(false);
133 UpdateAttackPowerAndDamage(true);
134 }
135
136 UpdateSpellDamageAndHealingBonus();
137 UpdateManaRegen();
138
139 // Update ratings in exist SPELL_AURA_MOD_RATING_FROM_STAT and only depends from stat
140 uint32 mask = 0;
141 AuraEffectList const& modRatingFromStat = GetAuraEffectsByType(SPELL_AURA_MOD_RATING_FROM_STAT);
142 for (AuraEffectList::const_iterator i = modRatingFromStat.begin(); i != modRatingFromStat.end(); ++i)
143 if (Stats((*i)->GetMiscValueB()) == stat)
144 mask |= (*i)->GetMiscValue();
145 if (mask)
146 {
147 for (uint32 rating = 0; rating < MAX_COMBAT_RATING; ++rating)
148 if (mask & (1 << rating))
149 ApplyRatingMod(CombatRating(rating), 0, true);
150 }
151 return true;
152}

154void Player::ApplySpellPowerBonus(int32 amount, bool apply)
155{
156 if (HasAuraType(SPELL_AURA_OVERRIDE_SPELL_POWER_BY_AP_PCT))
157 return;
158
159 apply = _ModifyUInt32(apply, m_baseSpellPower, amount);
160
161 // For speed just update for client
162 ApplyModUInt32Value(PLAYER_FIELD_MOD_HEALING_DONE_POS, amount, apply);
163 for (int i = SPELL_SCHOOL_HOLY; i < MAX_SPELL_SCHOOL; ++i)
164 ApplyModUInt32Value(PLAYER_FIELD_MOD_DAMAGE_DONE_POS + i, amount, apply);
165
166 if (HasAuraType(SPELL_AURA_OVERRIDE_ATTACK_POWER_BY_SP_PCT))
167 UpdateAttackPowerAndDamage();
168}

184bool Player::UpdateAllStats()
185{
186 for (int8 i = STAT_STRENGTH; i < MAX_STATS; ++i)
187 {
188 float value = GetTotalStatValue(Stats(i));
189 SetStat(Stats(i), int32(value));
190 }
191
192 UpdateArmor();
193 // calls UpdateAttackPowerAndDamage() in UpdateArmor for SPELL_AURA_MOD_ATTACK_POWER_OF_ARMOR
194 UpdateAttackPowerAndDamage(true);
195 UpdateMaxHealth();
196
197 for (uint8 i = POWER_MANA; i < MAX_POWERS; ++i)
198 UpdateMaxPower(Powers(i));
199
200 UpdateAllRatings();
201 UpdateAllCritPercentages();
202 UpdateAllSpellCritChances();
203 UpdateBlockPercentage();
204 UpdateParryPercentage();
205 UpdateDodgePercentage();
206 UpdateSpellDamageAndHealingBonus();
207 UpdateManaRegen();
208 UpdateExpertise(BASE_ATTACK);
209 UpdateExpertise(OFF_ATTACK);
210 RecalculateRating(CR_ARMOR_PENETRATION);
211 UpdateAllResistances();
212
213 return true;
214}

237void Player::UpdateArmor()
238{
239 UnitMods unitMod = UNIT_MOD_ARMOR;
240
241 float value = GetModifierValue(unitMod, BASE_VALUE); // base armor (from items)
242 value *= GetModifierValue(unitMod, BASE_PCT); // armor percent from items
243 value += GetModifierValue(unitMod, TOTAL_VALUE);
244
245 //add dynamic flat mods
246 AuraEffectList const& mResbyIntellect = GetAuraEffectsByType(SPELL_AURA_MOD_RESISTANCE_OF_STAT_PERCENT);
247 for (AuraEffectList::const_iterator i = mResbyIntellect.begin(); i != mResbyIntellect.end(); ++i)
248 {
249 if ((*i)->GetMiscValue() & SPELL_SCHOOL_MASK_NORMAL)
250 value += CalculatePct(GetStat(Stats((*i)->GetMiscValueB())), (*i)->GetAmount());
251 }
252
253 value *= GetModifierValue(unitMod, TOTAL_PCT);
254
255 SetArmor(int32(value));
256
257 Pet* pet = GetPet();
258 if (pet)
259 pet->UpdateArmor();
260
261 UpdateAttackPowerAndDamage(); // armor dependent auras update for SPELL_AURA_MOD_ATTACK_POWER_OF_ARMOR
262}

265{

267 float ratio = 10.0f;
268 if (GtOCTHpPerStaminaEntry const* hpBase = sGtOCTHpPerStaminaStore.EvaluateTable(getLevel() - 1, 0))

270

272

300void Player::UpdateAttackPowerAndDamage(bool ranged)
301{
302 float val2 = 0.0f;
303 float level = float(getLevel());
304
305 ChrClassesEntry const* entry = sChrClassesStore.LookupEntry(getClass());
306 UnitMods unitMod = ranged ? UNIT_MOD_ATTACK_POWER_RANGED : UNIT_MOD_ATTACK_POWER;
307
308 uint16 index = UNIT_FIELD_ATTACK_POWER;
309 uint16 index_mod = UNIT_FIELD_ATTACK_POWER_MOD_POS;
310 uint16 index_mult = UNIT_FIELD_ATTACK_POWER_MULTIPLIER;
311
312 if (ranged)
313 {
314 index = UNIT_FIELD_RANGED_ATTACK_POWER;
315 index_mod = UNIT_FIELD_RANGED_ATTACK_POWER_MOD_POS;
316 index_mult = UNIT_FIELD_RANGED_ATTACK_POWER_MULTIPLIER;
317 val2 = (level + std::max(GetStat(STAT_AGILITY), 0.0f)) * entry->RangedAttackPowerPerAgility;
318 }
319 else if (!HasAuraType(SPELL_AURA_OVERRIDE_ATTACK_POWER_BY_SP_PCT))
320 {
321 float strengthValue = std::max(GetStat(STAT_STRENGTH) * entry->AttackPowerPerStrength, 0.0f);
322 float agilityValue = std::max(GetStat(STAT_AGILITY) * entry->AttackPowerPerAgility, 0.0f);
323
324 SpellShapeshiftFormEntry const* form = sSpellShapeshiftFormStore.LookupEntry(GetShapeshiftForm());
325 // Directly taken from client, SHAPESHIFT_FLAG_AP_FROM_STRENGTH ?
326 if (form && form->Flags & 0x20)
327 agilityValue += std::max(GetStat(STAT_AGILITY) * entry->AttackPowerPerStrength, 0.0f);
328
329 val2 = strengthValue + agilityValue;
330 }
331 else
332 {
333 int32 minSpellPower = GetInt32Value(PLAYER_FIELD_MOD_HEALING_DONE_POS);
334 for (int i = SPELL_SCHOOL_HOLY; i < MAX_SPELL_SCHOOL; ++i)
335 minSpellPower = std::min(minSpellPower, GetInt32Value(PLAYER_FIELD_MOD_DAMAGE_DONE_POS + i));
336
337 val2 = CalculatePct(float(minSpellPower), GetFloatValue(PLAYER_FIELD_OVERRIDE_AP_BY_SPELL_POWER_PERCENT));
338 }
339
340 SetModifierValue(unitMod, BASE_VALUE, val2);
341
342 float base_attPower = GetModifierValue(unitMod, BASE_VALUE) * GetModifierValue(unitMod, BASE_PCT);
343 float attPowerMod = GetModifierValue(unitMod, TOTAL_VALUE);
344 float attPowerMultiplier = GetModifierValue(unitMod, TOTAL_PCT) - 1.0f;
345
346 //add dynamic flat mods
347 if (!ranged)
348 {
349 AuraEffectList const& mAPbyArmor = GetAuraEffectsByType(SPELL_AURA_MOD_ATTACK_POWER_OF_ARMOR);
350 for (AuraEffectList::const_iterator iter = mAPbyArmor.begin(); iter != mAPbyArmor.end(); ++iter)
351 // always: ((*i)->GetModifier()->m_miscvalue == 1 == SPELL_SCHOOL_MASK_NORMAL)
352 attPowerMod += int32(GetArmor() / (*iter)->GetAmount());
353 }
354
355 SetInt32Value(index, (uint32)base_attPower); //UNIT_FIELD_(RANGED)_ATTACK_POWER field
356 SetInt32Value(index_mod, (uint32)attPowerMod); //UNIT_FIELD_(RANGED)_ATTACK_POWER_MOD_POS field
357 SetFloatValue(index_mult, attPowerMultiplier); //UNIT_FIELD_(RANGED)_ATTACK_POWER_MULTIPLIER field
358
359 Pet* pet = GetPet(); //update pet's AP
360 Guardian* guardian = GetGuardianPet();
361 //automatically update weapon damage after attack power modification
362 if (ranged)
363 {
364 UpdateDamagePhysical(RANGED_ATTACK);
365 if (pet && pet->IsHunterPet()) // At ranged attack change for hunter pet
366 pet->UpdateAttackPowerAndDamage();
367 }
368 else
369 {
370 UpdateDamagePhysical(BASE_ATTACK);
371 if (CanDualWield() && haveOffhandWeapon()) //allow update offhand damage only if player knows DualWield Spec and has equipped offhand weapon
372 UpdateDamagePhysical(OFF_ATTACK);
373 if (HasAuraType(SPELL_AURA_MOD_SPELL_DAMAGE_OF_ATTACK_POWER) ||
374 HasAuraType(SPELL_AURA_MOD_SPELL_HEALING_OF_ATTACK_POWER) ||
375 HasAuraType(SPELL_AURA_OVERRIDE_SPELL_POWER_BY_AP_PCT))
376 UpdateSpellDamageAndHealingBonus();
377
378 if (pet && pet->IsPetGhoul()) // At melee attack power change for DK pet
379 pet->UpdateAttackPowerAndDamage();
380
381 if (guardian && guardian->IsSpiritWolf()) // At melee attack power change for Shaman feral spirit
382 guardian->UpdateAttackPowerAndDamage();
383 }
384}

386void Player::CalculateMinMaxDamage(WeaponAttackType attType, bool normalized, bool addTotalPct, float& minDamage, float& maxDamage)
387{
388 UnitMods unitMod;
389
390 switch (attType)
391 {
392 case BASE_ATTACK:
393 default:
394 unitMod = UNIT_MOD_DAMAGE_MAINHAND;
395 break;
396 case OFF_ATTACK:
397 unitMod = UNIT_MOD_DAMAGE_OFFHAND;
398 break;
399 case RANGED_ATTACK:
400 unitMod = UNIT_MOD_DAMAGE_RANGED;
401 break;
402 }
403
404 float attackSpeedMod = GetAPMultiplier(attType, normalized);
405
406 float baseValue = GetModifierValue(unitMod, BASE_VALUE) + GetTotalAttackPowerValue(attType) / 3.5f * attackSpeedMod;
407 float basePct = GetModifierValue(unitMod, BASE_PCT);
408 float totalValue = GetModifierValue(unitMod, TOTAL_VALUE);
409 float totalPct = addTotalPct ? GetModifierValue(unitMod, TOTAL_PCT) : 1.0f;
410
411 float weaponMinDamage = GetWeaponDamageRange(attType, MINDAMAGE);
412 float weaponMaxDamage = GetWeaponDamageRange(attType, MAXDAMAGE);
413
414 if (IsInFeralForm()) // check if player is druid and in cat or bear forms
415 {
416 float weaponSpeed = BASE_ATTACK_TIME / 1000.f;
417 if (Item* weapon = GetWeaponForAttack(BASE_ATTACK, true))
418 weaponSpeed = weapon->GetTemplate()->GetDelay() / 1000;
419
420 if (GetShapeshiftForm() == FORM_CAT)
421 {
422 weaponMinDamage = weaponMinDamage / weaponSpeed;
423 weaponMaxDamage = weaponMaxDamage / weaponSpeed;
424 }
425 else if (GetShapeshiftForm() == FORM_BEAR)
426 {
427 weaponMinDamage = weaponMinDamage / weaponSpeed + weaponMinDamage / 2.5;
428 weaponMaxDamage = weaponMinDamage / weaponSpeed + weaponMaxDamage / 2.5;
429 }
430 }
431 else if (!CanUseAttackType(attType)) // check if player not in form but still can't use (disarm case)
432 {
433 // cannot use ranged/off attack, set values to 0
434 if (attType != BASE_ATTACK)
435 {
436 minDamage = 0;
437 maxDamage = 0;
438 return;
439 }
440 weaponMinDamage = BASE_MINDAMAGE;
441 weaponMaxDamage = BASE_MAXDAMAGE;
442 }
443 /*
444 TODO: Is this still needed after ammo has been removed?
445 else if (attType == RANGED_ATTACK) // add ammo DPS to ranged damage
446 {
447 weaponMinDamage += GetAmmoDPS() * attackSpeedMod;
448 weaponMaxDamage += GetAmmoDPS() * attackSpeedMod;
449 }*/
450
451 minDamage = ((weaponMinDamage + baseValue) * basePct + totalValue) * totalPct;
452 maxDamage = ((weaponMaxDamage + baseValue) * basePct + totalValue) * totalPct;
453}

455void Player::UpdateBlockPercentage()
456{
457 // No block
458 float value = 0.0f;
459 if (CanBlock())
460 {
461 // Base value
462 value = 5.0f;
463 // Increase from SPELL_AURA_MOD_BLOCK_PERCENT aura
464 value += GetTotalAuraModifier(SPELL_AURA_MOD_BLOCK_PERCENT);
465 // Increase from rating
466 value += GetRatingBonusValue(CR_BLOCK);
467
468 if (sWorld->getBoolConfig(CONFIG_STATS_LIMITS_ENABLE))
469 value = value > sWorld->getFloatConfig(CONFIG_STATS_LIMITS_BLOCK) ? sWorld->getFloatConfig(CONFIG_STATS_LIMITS_BLOCK) : value;
470
471 value = value < 0.0f ? 0.0f : value;
472 }
473 SetStatFloatValue(PLAYER_BLOCK_PERCENTAGE, value);
474}

476void Player::UpdateCritPercentage(WeaponAttackType attType)
477{
478 BaseModGroup modGroup;
479 uint16 index;
480 CombatRating cr;
481
482 switch (attType)
483 {
484 case OFF_ATTACK:
485 modGroup = OFFHAND_CRIT_PERCENTAGE;
486 index = PLAYER_OFFHAND_CRIT_PERCENTAGE;
487 cr = CR_CRIT_MELEE;
488 break;
489 case RANGED_ATTACK:
490 modGroup = RANGED_CRIT_PERCENTAGE;
491 index = PLAYER_RANGED_CRIT_PERCENTAGE;
492 cr = CR_CRIT_RANGED;
493 break;
494 case BASE_ATTACK:
495 default:
496 modGroup = CRIT_PERCENTAGE;
497 index = PLAYER_CRIT_PERCENTAGE;
498 cr = CR_CRIT_MELEE;
499 break;
500 }
501
502 float value = GetTotalPercentageModValue(modGroup) + GetRatingBonusValue(cr);
503 // Modify crit from weapon skill and maximized defense skill of same level victim difference
504 value += (int32(GetMaxSkillValueForLevel()) - int32(GetMaxSkillValueForLevel())) * 0.04f;
505
506 if (sWorld->getBoolConfig(CONFIG_STATS_LIMITS_ENABLE))
507 value = value > sWorld->getFloatConfig(CONFIG_STATS_LIMITS_CRIT) ? sWorld->getFloatConfig(CONFIG_STATS_LIMITS_CRIT) : value;
508
509 value = value < 0.0f ? 0.0f : value;
510 SetStatFloatValue(index, value);
511}

526void Player::UpdateMastery()
527{
528 if (!CanUseMastery())
529 {
530 SetFloatValue(PLAYER_MASTERY, 0.0f);
531 return;
532 }
533
534 float value = GetTotalAuraModifier(SPELL_AURA_MASTERY);
535 value += GetRatingBonusValue(CR_MASTERY);
536 SetFloatValue(PLAYER_MASTERY, value);
537
538 ChrSpecializationEntry const* chrSpec = sChrSpecializationStore.LookupEntry(GetSpecId(GetActiveTalentGroup()));
539 if (!chrSpec)
540 return;
541
542 for (uint32 i = 0; i < MAX_MASTERY_SPELLS; ++i)
543 {
544 if (!chrSpec->MasterySpellID[i])
545 continue;
546
547 if (Aura* aura = GetAura(chrSpec->MasterySpellID[i]))
548 {
549 for (SpellEffectInfo const* effect : aura->GetSpellEffectInfos())
550 {
551 if (!effect)
552 continue;
553
554 float mult = effect->BonusCoefficient;
555 if (G3D::fuzzyEq(mult, 0.0f))
556 continue;
557
558 aura->GetEffect(effect->EffectIndex)->ChangeAmount(int32(value * effect->BonusCoefficient));
559 }
560 }
561 }

579void Player::UpdateParryPercentage()
580{
581 const float parry_cap[MAX_CLASSES] =
582 {
583 65.631440f, // Warrior
584 65.631440f, // Paladin
585 145.560408f, // Hunter
586 145.560408f, // Rogue
587 0.0f, // Priest
588 65.631440f, // DK
589 145.560408f, // Shaman
590 0.0f, // Mage
591 0.0f, // Warlock
592 0.0f, // ??
593 0.0f // Druid
594 };
595
596 // No parry
597 float value = 0.0f;
598 uint32 pclass = getClass()-1;
599 if (CanParry() && parry_cap[pclass] > 0.0f)
600 {
601 float nondiminishing = 5.0f;
602 // Parry from rating
603 float diminishing = GetRatingBonusValue(CR_PARRY);
604 // Parry from SPELL_AURA_MOD_PARRY_PERCENT aura
605 nondiminishing += GetTotalAuraModifier(SPELL_AURA_MOD_PARRY_PERCENT);
606 // apply diminishing formula to diminishing parry chance
607 value = nondiminishing + diminishing * parry_cap[pclass] / (diminishing + parry_cap[pclass] * m_diminishing_k[pclass]);
608
609 if (sWorld->getBoolConfig(CONFIG_STATS_LIMITS_ENABLE))
610 value = value > sWorld->getFloatConfig(CONFIG_STATS_LIMITS_PARRY) ? sWorld->getFloatConfig(CONFIG_STATS_LIMITS_PARRY) : value;
611
612 value = value < 0.0f ? 0.0f : value;
613 }
614 SetStatFloatValue(PLAYER_PARRY_PERCENTAGE, value);
615}

617void Player::UpdateDodgePercentage()
618{
619 const float dodge_cap[MAX_CLASSES] =
620 {
621 65.631440f, // Warrior
622 65.631440f, // Paladin
623 145.560408f, // Hunter
624 145.560408f, // Rogue
625 150.375940f, // Priest
626 65.631440f, // DK
627 145.560408f, // Shaman
628 150.375940f, // Mage
629 150.375940f, // Warlock
630 0.0f, // ??
631 116.890707f // Druid
632 };
633
634 float diminishing = 0.0f, nondiminishing = 0.0f;
635 GetDodgeFromAgility(diminishing, nondiminishing);
636 // Dodge from SPELL_AURA_MOD_DODGE_PERCENT aura
637 nondiminishing += GetTotalAuraModifier(SPELL_AURA_MOD_DODGE_PERCENT);
638 // Dodge from rating
639 diminishing += GetRatingBonusValue(CR_DODGE);
640 // apply diminishing formula to diminishing dodge chance
641 uint32 pclass = getClass()-1;
642 float value = nondiminishing + (diminishing * dodge_cap[pclass] / (diminishing + dodge_cap[pclass] * m_diminishing_k[pclass]));
643
644 if (sWorld->getBoolConfig(CONFIG_STATS_LIMITS_ENABLE))

907void Creature::CalculateMinMaxDamage(WeaponAttackType attType, bool normalized, bool addTotalPct, float& minDamage, float& maxDamage)
908{
909 float variance = 1.0f;
910 UnitMods unitMod;
911 switch (attType)
912 {
913 case BASE_ATTACK:
914 default:
915 variance = GetCreatureTemplate()->BaseVariance;
916 unitMod = UNIT_MOD_DAMAGE_MAINHAND;
917 break;
918 case OFF_ATTACK:
919 variance = GetCreatureTemplate()->BaseVariance;
920 unitMod = UNIT_MOD_DAMAGE_OFFHAND;
921 break;
922 case RANGED_ATTACK:
923 variance = GetCreatureTemplate()->RangeVariance;
924 unitMod = UNIT_MOD_DAMAGE_RANGED;
925 break;
926 }
927
928 if (attType == OFF_ATTACK && !haveOffhandWeapon())
929 {
930 minDamage = 0.0f;
931 maxDamage = 0.0f;
932 return;
933 }
934
935 float weaponMinDamage = GetWeaponDamageRange(attType, MINDAMAGE);
936 float weaponMaxDamage = GetWeaponDamageRange(attType, MAXDAMAGE);
937
938 if (!CanUseAttackType(attType)) // disarm case
939 {
940 weaponMinDamage = 0.0f;
941 weaponMaxDamage = 0.0f;
942 }
943
944 float attackPower = GetTotalAttackPowerValue(attType);
945 float attackSpeedMulti = GetAPMultiplier(attType, normalized);
946 float baseValue = GetModifierValue(unitMod, BASE_VALUE) + (attackPower / 3.5f) * variance;
947 float basePct = GetModifierValue(unitMod, BASE_PCT) * attackSpeedMulti;
948 float totalValue = GetModifierValue(unitMod, TOTAL_VALUE);
949 float totalPct = addTotalPct ? GetModifierValue(unitMod, TOTAL_PCT) : 1.0f;
950 float dmgMultiplier = GetCreatureTemplate()->ModDamage; // = ModDamage * _GetDamageMod(rank);
951
952 minDamage = ((weaponMinDamage + baseValue) * dmgMultiplier * basePct + totalValue) * totalPct;
953 maxDamage = ((weaponMaxDamage + baseValue) * dmgMultiplier * basePct + totalValue) * totalPct;
954}

1232 bonusDamage = spellDmg * 0.09f;

1234 //greater fire elemental

1236 {

24AreaTrigger::AreaTrigger() : WorldObject(false), _duration(0)
25{
26 m_objectType |= TYPEMASK_AREATRIGGER;
27 m_objectTypeId = TYPEID_AREATRIGGER;
28
29 m_updateFlag = UPDATEFLAG_STATIONARY_POSITION;
30
31 m_valuesCount = AREATRIGGER_END;
32 _dynamicValuesCount = AREATRIGGER_DYNAMIC_END;
33}

37}

39void AreaTrigger::AddToWorld()
40{
41 ///- Register the AreaTrigger for guid lookup and for caster
42 if (!IsInWorld())
43 {
44 GetMap()->GetObjectsStore().Insert<AreaTrigger>(GetGUID(), this);
45 WorldObject::AddToWorld();
46 }
47}
48
49void AreaTrigger::RemoveFromWorld()
50{
51 ///- Remove the AreaTrigger from the accessor and from all lists of objects in world
52 if (IsInWorld())
53 {
54 WorldObject::RemoveFromWorld();
55 GetMap()->GetObjectsStore().Remove<AreaTrigger>(GetGUID());
56 }
57}

59bool AreaTrigger::CreateAreaTrigger(ObjectGuid::LowType guidlow, uint32 triggerEntry, Unit* caster, SpellInfo const* spell, Position const& pos)
60{
61 SetMap(caster->GetMap());
62 Relocate(pos);
63 if (!IsPositionValid())
64 {
65 TC_LOG_ERROR("misc", "AreaTrigger (spell %u) not created. Invalid coordinates (X: %f Y: %f)", spell->Id, GetPositionX(), GetPositionY());
66 return false;
67 }
68
69 Object::_Create(ObjectGuid::Create<HighGuid::AreaTrigger>(GetMapId(), triggerEntry, guidlow));
70 SetPhaseMask(caster->GetPhaseMask(), false);
71
72 SetEntry(triggerEntry);
73 SetDuration(spell->GetDuration());
74 SetObjectScale(1);
75
76 SetGuidValue(AREATRIGGER_CASTER, caster->GetGUID());
77 SetUInt32Value(AREATRIGGER_SPELLID, spell->Id);
78 SetUInt32Value(AREATRIGGER_SPELLVISUALID, spell->GetSpellVisual(GetMap()->GetDifficultyID()));
79 SetUInt32Value(AREATRIGGER_DURATION, spell->GetDuration());
80
81 CopyPhaseFrom(caster);
82
83 if (!GetMap()->AddToMap(this))
84 return false;
85
86 return true;
87}
88
89void AreaTrigger::Update(uint32 p_time)
90{
91 if (GetDuration() > int32(p_time))
92 _duration -= p_time;
93 else
94 Remove(); // expired
95
96 WorldObject::Update(p_time);
97}
98
99void AreaTrigger::Remove()
100{
101 if (IsInWorld())
102 {
103 SendObjectDeSpawnAnim(GetGUID());
104 RemoveFromWorld();
105 AddObjectToRemoveList();
106 }
107}

18#ifndef TAXIPATHGRAPH_HPP
19#define TAXIPATHGRAPH_HPP
20
21#include "Position.h"
22#include "Define.h"
23#include <boost/graph/adjacency_list.hpp>
24
25class Player;
26struct TaxiNodesEntry;
27
28class TaxiPathGraph
29{
30public:
31 static TaxiPathGraph& Instance()
32 {
33 static TaxiPathGraph instance;
34 return instance;
35 }
36
37 void Initialize();
38 std::size_t GetCompleteNodeRoute(TaxiNodesEntry const* from, TaxiNodesEntry const* to, Player const* player, std::vector<uint32>& shortestPath);
39
40private:
41 struct EdgeCost
42 {
43 TaxiNodesEntry const* To;
44 uint32 Distance;
45 uint32 EvaluateDistance(Player const* player) const;
46 };
47 typedef boost::adjacency_list<boost::listS, boost::vecS, boost::directedS, boost::property<boost::vertex_index_t, uint32>, boost::property<boost::edge_weight_t, EdgeCost>> Graph;
48 typedef boost::property_map<Graph, boost::edge_weight_t>::type WeightMap;
49 typedef Graph::vertex_descriptor vertex_descriptor;
50 typedef Graph::edge_descriptor edge_descriptor;
51 typedef std::pair<uint32, uint32> edge;
52
53 TaxiPathGraph() { }
54 ~TaxiPathGraph() { }
55
56 void AddVerticeAndEdgeFromNodeInfo(TaxiNodesEntry const* from, TaxiNodesEntry const* to, uint32 pathId, std::vector<std::pair<edge, EdgeCost>>& edges);
57 vertex_descriptor GetVertexIDFromNodeID(TaxiNodesEntry const* node);
58 uint32 GetNodeIDFromVertexID(vertex_descriptor vertexID);
59 vertex_descriptor CreateVertexFromFromNodeInfoIfNeeded(TaxiNodesEntry const* node);
60 std::size_t GetVertexCount();
61
62 Graph m_graph;
63 std::vector<TaxiNodesEntry const*> m_vertices;
64
65 TaxiPathGraph(TaxiPathGraph const&) = delete;
66 TaxiPathGraph& operator=(TaxiPathGraph const&) = delete;
67};
68
69#define sTaxiPathGraph TaxiPathGraph::Instance()
70
71#endif /* TAXIPATHGRAPH_HPP */

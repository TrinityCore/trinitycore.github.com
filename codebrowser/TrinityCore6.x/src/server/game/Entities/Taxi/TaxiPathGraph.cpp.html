28void TaxiPathGraph::Initialize()
29{
30 if (GetVertexCount() > 0)
31 return;
32
33 std::vector<std::pair<edge, EdgeCost>> edges;
34
35 // Initialize here
36 for (TaxiPathEntry const* path : sTaxiPathStore)
37 {
38 TaxiNodesEntry const* from = sTaxiNodesStore.LookupEntry(path->From);
39 TaxiNodesEntry const* to = sTaxiNodesStore.LookupEntry(path->To);
40 if (from && to && from->Flags & (TAXI_NODE_FLAG_ALLIANCE | TAXI_NODE_FLAG_HORDE) && to->Flags & (TAXI_NODE_FLAG_ALLIANCE | TAXI_NODE_FLAG_HORDE))
41 AddVerticeAndEdgeFromNodeInfo(from, to, path->ID, edges);
42 }
43
44 // create graph
45 m_graph = Graph(GetVertexCount());
46 WeightMap weightmap = boost::get(boost::edge_weight, m_graph);
47
48 for (std::size_t j = 0; j < edges.size(); ++j)
49 {
50 edge_descriptor e = boost::add_edge(edges[j].first.first, edges[j].first.second, m_graph).first;
51 weightmap[e] = edges[j].second;
52 }
53}

74void TaxiPathGraph::AddVerticeAndEdgeFromNodeInfo(TaxiNodesEntry const* from, TaxiNodesEntry const* to, uint32 pathId, std::vector<std::pair<edge, EdgeCost>>& edges)
75{
76 if (from != to)
77 {
78 vertex_descriptor fromVertexID = CreateVertexFromFromNodeInfoIfNeeded(from);
79 vertex_descriptor toVertexID = CreateVertexFromFromNodeInfoIfNeeded(to);
80
81 float totalDist = 0.0f;
82 TaxiPathNodeList const& nodes = sTaxiPathNodesByPath[pathId];
83 if (nodes.size() < 2)
84 {
85 edges.push_back(std::make_pair(edge(fromVertexID, toVertexID), EdgeCost{ to, 0xFFFF }));
86 return;
87 }
88
89 std::size_t last = nodes.size();
90 std::size_t first = 0;
91 if (nodes.size() > 2)
92 {
93 --last;
94 ++first;
95 }
96
97 for (std::size_t i = first + 1; i < last; ++i)
98 {
99 if (nodes[i - 1]->Flags & TAXI_PATH_NODE_FLAG_TELEPORT)
100 continue;
101
102 uint32 map1, map2;
103 DBCPosition2D pos1, pos2;
104
105 DeterminaAlternateMapPosition(nodes[i - 1]->MapID, nodes[i - 1]->Loc.X, nodes[i - 1]->Loc.Y, nodes[i - 1]->Loc.Z, &map1, &pos1);
106 DeterminaAlternateMapPosition(nodes[i]->MapID, nodes[i]->Loc.X, nodes[i]->Loc.Y, nodes[i]->Loc.Z, &map2, &pos2);
107
108 if (map1 != map2)
109 continue;
110
111 totalDist += std::sqrt(
112 std::pow(pos2.X - pos1.X, 2) +
113 std::pow(pos2.Y - pos1.Y, 2) +
114 std::pow(nodes[i]->Loc.Z - nodes[i - 1]->Loc.Z, 2));
115 }
116
117 uint32 dist = uint32(totalDist);
118 if (dist > 0xFFFF)
119 dist = 0xFFFF;
120
121 edges.push_back(std::make_pair(edge(fromVertexID, toVertexID), EdgeCost{ to, dist }));
122 }
123}

125std::size_t TaxiPathGraph::GetCompleteNodeRoute(TaxiNodesEntry const* from, TaxiNodesEntry const* to, Player const* player, std::vector<uint32>& shortestPath)
126{
127 /*
128 Information about node algorithm from client
129 Since client does not give information about *ALL* nodes you have to pass by when going from sourceNodeID to destinationNodeID, we need to use Dijkstra algorithm.
130 Examining several paths I discovered the following algorithm:
131 * If destinationNodeID has is the next destination, connected directly to sourceNodeID, then, client just pick up this route regardless of distance
132 * else we use dijkstra to find the shortest path.
133 * When early landing is requested, according to behavior on retail, you can never end in a node you did not discovered before
134 */
135
136 // Find if we have a direct path
137 uint32 pathId, goldCost;
138 sObjectMgr->GetTaxiPath(from->ID, to->ID, pathId, goldCost);
139 if (pathId)
140 shortestPath = { from->ID, to->ID };
141 else
142 {
143 shortestPath.clear();
144 std::vector<vertex_descriptor> p(boost::num_vertices(m_graph));
145
146 boost::dijkstra_shortest_paths(m_graph, GetVertexIDFromNodeID(from),
147 boost::predecessor_map(boost::make_iterator_property_map(p.begin(), boost::get(boost::vertex_index, m_graph)))
148 .weight_map(boost::make_transform_value_property_map(
149 [player](EdgeCost const& edgeCost) { return edgeCost.EvaluateDistance(player); },
150 boost::get(boost::edge_weight, m_graph))));
151
152 // found a path to the goal
153 for (vertex_descriptor v = GetVertexIDFromNodeID(to); ; v = p[v])
154 {
155 shortestPath.push_back(GetNodeIDFromVertexID(v));
156 if (v == p[v])
157 break;
158 }
159
160 std::reverse(shortestPath.begin(), shortestPath.end());
161 }
162
163 return shortestPath.size();
164}

166TaxiPathGraph::vertex_descriptor TaxiPathGraph::CreateVertexFromFromNodeInfoIfNeeded(TaxiNodesEntry const* node)
167{
168 //Check if we need a new one or if it may be already created
169 if (m_vertices.size() <= node->LearnableIndex)
170 m_vertices.resize(node->LearnableIndex + 1);
171
172 m_vertices[node->LearnableIndex] = node;
173 return node->LearnableIndex;
174}

176uint32 TaxiPathGraph::EdgeCost::EvaluateDistance(Player const* player) const
177{
178 uint32 requireFlag = (player->GetTeam() == ALLIANCE) ? TAXI_NODE_FLAG_ALLIANCE : TAXI_NODE_FLAG_HORDE;
179 if (!(To->Flags & requireFlag))
180 return std::numeric_limits<uint16>::max();
181
182 //if (To->ConditionID && !player->MeetsCondition(To->ConditionID))
183 // return std::numeric_limits<uint16>::max();
184
185 return Distance;
186}

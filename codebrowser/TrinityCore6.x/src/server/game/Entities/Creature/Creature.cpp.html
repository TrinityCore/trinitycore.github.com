150bool AssistDelayEvent::Execute(uint64 /*e_time*/, uint32 /*p_time*/)
151{
152 if (Unit* victim = ObjectAccessor::GetUnit(m_owner, m_victim))
153 {
154 while (!m_assistants.empty())
155 {
156 Creature* assistant = ObjectAccessor::GetCreature(m_owner, *m_assistants.begin());
157 m_assistants.pop_front();
158
159 if (assistant && assistant->CanAssistTo(&m_owner, victim))
160 {
161 assistant->SetNoCallAssistance(true);
162 assistant->CombatStart(victim);
163 if (assistant->IsAIEnabled)
164 assistant->AI()->AttackStart(victim);
165 }
166 }
167 }
168 return true;

306bool Creature::InitEntry(uint32 entry, CreatureData const* data /*= nullptr*/)
307{
308 CreatureTemplate const* normalInfo = sObjectMgr->GetCreatureTemplate(entry);
309 if (!normalInfo)
310 {
311 TC_LOG_ERROR("sql.sql", "Creature::InitEntry creature entry %u does not exist.", entry);
312 return false;
313 }
314
315 // get difficulty 1 mode entry
316 CreatureTemplate const* cinfo = nullptr;
317 DifficultyEntry const* difficultyEntry = sDifficultyStore.LookupEntry(GetMap()->GetSpawnMode());
318 while (!cinfo && difficultyEntry)
319 {
320 int32 idx = CreatureTemplate::DifficultyIDToDifficultyEntryIndex(difficultyEntry->ID);
321 if (idx == -1)
322 break;
323
324 if (normalInfo->DifficultyEntry[idx])
325 {
326 cinfo = sObjectMgr->GetCreatureTemplate(normalInfo->DifficultyEntry[idx]);
327 break;
328 }
329
330 if (!difficultyEntry->FallbackDifficultyID)
331 break;
332
333 difficultyEntry = sDifficultyStore.LookupEntry(difficultyEntry->FallbackDifficultyID);
334 }
335
336 if (!cinfo)
337 cinfo = normalInfo;
338
339 // Initialize loot duplicate count depending on raid difficulty
340 if (GetMap()->Is25ManRaid())
341 loot.maxDuplicates = 3;
342
343 SetEntry(entry); // normal entry always
344 m_creatureInfo = cinfo; // map mode related always
345
346 // equal to player Race field, but creature does not have race
347 SetByteValue(UNIT_FIELD_BYTES_0, UNIT_BYTES_0_OFFSET_RACE, 0);
348
349 // known valid are: CLASS_WARRIOR, CLASS_PALADIN, CLASS_ROGUE, CLASS_MAGE
350 SetByteValue(UNIT_FIELD_BYTES_0, UNIT_BYTES_0_OFFSET_CLASS, uint8(cinfo->unit_class));
351
352 // Cancel load if no model defined
353 if (!(cinfo->GetFirstValidModelId()))
354 {
355 TC_LOG_ERROR("sql.sql", "Creature (Entry: %u) has no model defined in table `creature_template`, can't load. ", entry);
356 return false;
357 }
358
359 uint32 displayID = ObjectMgr::ChooseDisplayId(GetCreatureTemplate(), data);
360 CreatureModelInfo const* minfo = sObjectMgr->GetCreatureModelRandomGender(&displayID);
361 if (!minfo) // Cancel load if no model defined
362 {
363 TC_LOG_ERROR("sql.sql", "Creature (Entry: %u) has invalid model %u defined in table `creature_template`, can't load.", entry, displayID);
364 return false;
365 }
366
367 SetDisplayId(displayID);
368 SetNativeDisplayId(displayID);
369 SetByteValue(UNIT_FIELD_BYTES_0, UNIT_BYTES_0_OFFSET_GENDER, minfo->gender);
370
371 // Load creature equipment
372 if (!data || data->equipmentId == 0)
373 LoadEquipment(); // use default equipment (if available)
374 else if (data && data->equipmentId != 0) // override, 0 means no equipment
375 {
376 m_originalEquipmentId = data->equipmentId;
377 LoadEquipment(data->equipmentId);
378 }
379
380 SetName(normalInfo->Name); // at normal entry always
381
382 SetFloatValue(UNIT_MOD_CAST_SPEED, 1.0f);
383 SetFloatValue(UNIT_MOD_CAST_HASTE, 1.0f);
384
385 SetSpeed(MOVE_WALK, cinfo->speed_walk);
386 SetSpeed(MOVE_RUN, cinfo->speed_run);
387 SetSpeed(MOVE_SWIM, 1.0f); // using 1.0 rate
388 SetSpeed(MOVE_FLIGHT, 1.0f); // using 1.0 rate
389
390 // Will set UNIT_FIELD_BOUNDINGRADIUS and UNIT_FIELD_COMBATREACH
391 SetObjectScale(cinfo->scale);
392
393 SetFloatValue(UNIT_FIELD_HOVERHEIGHT, cinfo->HoverHeight);
394
395 // checked at loading
396 m_defaultMovementType = MovementGeneratorType(cinfo->MovementType);
397 if (!m_respawnradius && m_defaultMovementType == RANDOM_MOTION_TYPE)
398 m_defaultMovementType = IDLE_MOTION_TYPE;
399
400 for (uint8 i=0; i < CREATURE_MAX_SPELLS; ++i)
401 m_spells[i] = GetCreatureTemplate()->spells[i];
402
403 return true;
404}

406bool Creature::UpdateEntry(uint32 entry, CreatureData const* data /*= nullptr*/)
407{
408 if (!InitEntry(entry, data))
409 return false;
410
411 CreatureTemplate const* cInfo = GetCreatureTemplate();
412
413 m_regenHealth = cInfo->RegenHealth;
414
415 // creatures always have melee weapon ready if any unless specified otherwise
416 if (!GetCreatureAddon())
417 SetSheath(SHEATH_STATE_MELEE);
418
419 setFaction(cInfo->faction);
420
421 uint32 unit_flags, dynamicflags;
422 uint64 npcflag;
423 ObjectMgr::ChooseCreatureFlags(cInfo, npcflag, unit_flags, dynamicflags, data);
424
425 if (cInfo->flags_extra & CREATURE_FLAG_EXTRA_WORLDEVENT)
426 SetUInt64Value(UNIT_NPC_FLAGS, npcflag | sGameEventMgr->GetNPCFlag(this));
427 else
428 SetUInt64Value(UNIT_NPC_FLAGS, npcflag);
429
430 SetUInt32Value(UNIT_FIELD_FLAGS, unit_flags);
431 SetUInt32Value(UNIT_FIELD_FLAGS_2, cInfo->unit_flags2);
432
433 SetUInt32Value(OBJECT_DYNAMIC_FLAGS, dynamicflags);
434
435 RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IN_COMBAT);
436
437 SetAttackTime(BASE_ATTACK, cInfo->BaseAttackTime);
438 SetAttackTime(OFF_ATTACK, cInfo->BaseAttackTime);
439 SetAttackTime(RANGED_ATTACK, cInfo->RangeAttackTime);
440
441 SelectLevel();
442
443 SetMeleeDamageSchool(SpellSchools(cInfo->dmgschool));
444 CreatureBaseStats const* stats = sObjectMgr->GetCreatureBaseStats(getLevel(), cInfo->unit_class);
445 float armor = (float)stats->GenerateArmor(cInfo); /// @todo Why is this treated as uint32 when it's a float?
446 SetModifierValue(UNIT_MOD_ARMOR, BASE_VALUE, armor);
447 SetModifierValue(UNIT_MOD_RESISTANCE_HOLY, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_HOLY]));
448 SetModifierValue(UNIT_MOD_RESISTANCE_FIRE, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_FIRE]));
449 SetModifierValue(UNIT_MOD_RESISTANCE_NATURE, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_NATURE]));
450 SetModifierValue(UNIT_MOD_RESISTANCE_FROST, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_FROST]));
451 SetModifierValue(UNIT_MOD_RESISTANCE_SHADOW, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_SHADOW]));
452 SetModifierValue(UNIT_MOD_RESISTANCE_ARCANE, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_ARCANE]));
453
454 SetCanModifyStats(true);
455 UpdateAllStats();
456
457 // checked and error show at loading templates
458 if (FactionTemplateEntry const* factionTemplate = sFactionTemplateStore.LookupEntry(cInfo->faction))
459 {
460 if (factionTemplate->Flags & FACTION_TEMPLATE_FLAG_PVP)
461 SetPvP(true);
462 else
463 SetPvP(false);
464 }
465
466 // updates spell bars for vehicles and set player's faction - should be called here, to overwrite faction that is set from the new template
467 if (IsVehicle())
468 {
469 if (Player* owner = Creature::GetCharmerOrOwnerPlayerOrPlayerItself()) // this check comes in case we don't have a player
470 {
471 setFaction(owner->getFaction()); // vehicles should have same as owner faction
472 owner->VehicleSpellInitialize();
473 }
474 }
475
476 // trigger creature is always not selectable and can not be attacked
477 if (IsTrigger())
478 SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
479
480 InitializeReactState();
481
482 if (cInfo->flags_extra & CREATURE_FLAG_EXTRA_NO_TAUNT)
483 {
484 ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, true);
485 ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, true);
486 }
487
488 UpdateMovementFlags();
489 LoadCreaturesAddon();
490 return true;
491}

493void Creature::Update(uint32 diff)
494{
495 if (IsAIEnabled && TriggerJustRespawned)
496 {
497 TriggerJustRespawned = false;
498 AI()->JustRespawned();
499 if (m_vehicleKit)
500 m_vehicleKit->Reset();
501 }
502
503 UpdateMovementFlags();
504
505 switch (m_deathState)
506 {
507 case JUST_RESPAWNED:
508 // Must not be called, see Creature::setDeathState JUST_RESPAWNED -> ALIVE promoting.
509 TC_LOG_ERROR("entities.unit", "Creature (%s) in wrong state: JUST_RESPAWNED (4)", GetGUID().ToString().c_str());
510 break;
511 case JUST_DIED:
512 // Must not be called, see Creature::setDeathState JUST_DIED -> CORPSE promoting.
513 TC_LOG_ERROR("entities.unit", "Creature (%s) in wrong state: JUST_DEAD (1)", GetGUID().ToString().c_str());
514 break;
515 case DEAD:
516 {
517 time_t now = time(NULL);
518 if (m_respawnTime <= now)
519 {
520 bool allowed = IsAIEnabled ? AI()->CanRespawn() : true; // First check if there are any scripts that object to us respawning
521 if (!allowed) // Will be rechecked on next Update call
522 break;
523
524 ObjectGuid dbtableHighGuid = ObjectGuid::Create<HighGuid::Creature>(GetMapId(), GetEntry(), m_spawnId);
525 time_t linkedRespawntime = GetMap()->GetLinkedRespawnTime(dbtableHighGuid);
526 if (!linkedRespawntime) // Can respawn
527 Respawn();
528 else // the master is dead
529 {
530 ObjectGuid targetGuid = sObjectMgr->GetLinkedRespawnGuid(dbtableHighGuid);
531 if (targetGuid == dbtableHighGuid) // if linking self, never respawn (check delayed to next day)
532 SetRespawnTime(DAY);
533 else
534 m_respawnTime = (now > linkedRespawntime ? now : linkedRespawntime) + urand(5, MINUTE); // else copy time from master and add a little
535 SaveRespawnTime(); // also save to DB immediately
536 }
537 }
538 break;
539 }
540 case CORPSE:
541 {
542 Unit::Update(diff);
543 // deathstate changed on spells update, prevent problems
544 if (m_deathState != CORPSE)
545 break;
546
547 if (m_groupLootTimer && !lootingGroupLowGUID.IsEmpty())
548 {
549 if (m_groupLootTimer <= diff)
550 {
551 if (Group* group = sGroupMgr->GetGroupByGUID(lootingGroupLowGUID))
552 group->EndRoll(&loot);
553
554 m_groupLootTimer = 0;
555 lootingGroupLowGUID.Clear();
556 }
557 else
558 m_groupLootTimer -= diff;
559 }
560 else if (m_corpseRemoveTime <= time(NULL))
561 {
562 RemoveCorpse(false);
563 TC_LOG_DEBUG("entities.unit", "Removing corpse... %u ", GetUInt32Value(OBJECT_FIELD_ENTRY));
564 }
565 break;
566 }
567 case ALIVE:
568 {
569 Unit::Update(diff);
570
571 // creature can be dead after Unit::Update call
572 // CORPSE/DEAD state will processed at next tick (in other case death timer will be updated unexpectedly)
573 if (!IsAlive())
574 break;
575
576 // if creature is charmed, switch to charmed AI (and back)
577 if (NeedChangeAI)
578 {
579 UpdateCharmAI();
580 NeedChangeAI = false;
581 IsAIEnabled = true;
582 if (!IsInEvadeMode() && !LastCharmerGUID.IsEmpty())
583 if (Unit* charmer = ObjectAccessor::GetUnit(*this, LastCharmerGUID))
584 i_AI->AttackStart(charmer);
585
586 LastCharmerGUID.Clear();
587 }
588
589 // if periodic combat pulse is enabled and we are both in combat and in a dungeon, do this now
590 if (m_combatPulseDelay > 0 && IsInCombat() && GetMap()->IsDungeon())
591 {
592 if (diff > m_combatPulseTime)
593 m_combatPulseTime = 0;
594 else
595 m_combatPulseTime -= diff;
596
597 if (m_combatPulseTime == 0)
598 {
599 Map::PlayerList const &players = GetMap()->GetPlayers();
600 if (!players.isEmpty())
601 for (Map::PlayerList::const_iterator it = players.begin(); it != players.end(); ++it)
602 {
603 if (Player* player = it->GetSource())
604 {
605 if (player->IsGameMaster())
606 continue;
607
608 if (player->IsAlive() && this->IsHostileTo(player))
609 {
610 if (CanHaveThreatList())
611 AddThreat(player, 0.0f);
612 this->SetInCombatWith(player);
613 player->SetInCombatWith(this);
614 }
615 }
616 }
617
618 m_combatPulseTime = m_combatPulseDelay * IN_MILLISECONDS;
619 }
620 }
621
622 if (!IsInEvadeMode() && IsAIEnabled)
623 {
624 // do not allow the AI to be changed during update
625 m_AI_locked = true;
626
627 i_AI->UpdateAI(diff);
628 m_AI_locked = false;
629 }
630
631 // creature can be dead after UpdateAI call
632 // CORPSE/DEAD state will processed at next tick (in other case death timer will be updated unexpectedly)
633 if (!IsAlive())
634 break;
635
636 if (m_regenTimer > 0)
637 {
638 if (diff >= m_regenTimer)
639 m_regenTimer = 0;
640 else
641 m_regenTimer -= diff;
642 }
643
644 if (m_regenTimer != 0)
645 break;
646
647 bool bInCombat = IsInCombat() && (!GetVictim() || // if IsInCombat() is true and this has no victim
648 !EnsureVictim()->GetCharmerOrOwnerPlayerOrPlayerItself() || // or the victim/owner/charmer is not a player
649 !EnsureVictim()->GetCharmerOrOwnerPlayerOrPlayerItself()->IsGameMaster()); // or the victim/owner/charmer is not a GameMaster
650
651 /*if (m_regenTimer <= diff)
652 {*/
653 if (!IsInEvadeMode() && (!bInCombat || IsPolymorphed())) // regenerate health if not in combat or if polymorphed
654 RegenerateHealth();
655
656 if (HasFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_REGENERATE_POWER))
657 {
658 if (getPowerType() == POWER_ENERGY)
659 Regenerate(POWER_ENERGY);
660 else
661 RegenerateMana();
662 }
663
664 /*if (!bIsPolymorphed) // only increase the timer if not polymorphed
665 m_regenTimer += CREATURE_REGEN_INTERVAL - diff;
666 }
667 else
668 if (!bIsPolymorphed) // if polymorphed, skip the timer
669 m_regenTimer -= diff;*/
670 m_regenTimer = CREATURE_REGEN_INTERVAL;
671 break;
672 }
673 default:
674 break;
675 }
676
677 sScriptMgr->OnCreatureUpdate(this, diff);
678}

683 uint32 maxValue = GetMaxPower(POWER_MANA);
684

694 float Spirit = GetStat(STAT_SPIRIT);

700
701 // Apply modifiers (if any).
702 AuraEffectList const& ModPowerRegenPCTAuras = GetAuraEffectsByType(SPELL_AURA_MOD_POWER_REGEN_PERCENT);

704 if ((*i)->GetMiscValue() == POWER_MANA)
705 AddPct(addvalue, (*i)->GetAmount());
706
707 addvalue += GetTotalAuraModifierByMiscValue(SPELL_AURA_MOD_POWER_REGEN, POWER_MANA) * CREATURE_REGEN_INTERVAL / (5 * IN_MILLISECONDS);
708
709 ModifyPower(POWER_MANA, addvalue);

1477{

1537float Creature::GetAttackDistance(Unit const* player) const
1538{
1539 float aggroRate = sWorld->getRate(RATE_CREATURE_AGGRO);
1540 if (aggroRate == 0)
1541 return 0.0f;
1542
1543 uint32 playerlevel = player->getLevelForTarget(this);
1544 uint32 creaturelevel = getLevelForTarget(player);
1545
1546 int32 leveldif = int32(playerlevel) - int32(creaturelevel);
1547
1548 // "The maximum Aggro Radius has a cap of 25 levels under. Example: A level 30 char has the same Aggro Radius of a level 5 char on a level 60 mob."
1549 if (leveldif < - 25)
1550 leveldif = -25;
1551
1552 // "The aggro radius of a mob having the same level as the player is roughly 20 yards"
1553 float RetDistance = 20;
1554
1555 // "Aggro Radius varies with level difference at a rate of roughly 1 yard/level"
1556 // radius grow if playlevel < creaturelevel
1557 RetDistance -= (float)leveldif;
1558
1559 if (creaturelevel+5 <= sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
1560 {
1561 // detect range auras
1562 RetDistance += GetTotalAuraModifier(SPELL_AURA_MOD_DETECT_RANGE);
1563
1564 // detected range auras
1565 RetDistance += player->GetTotalAuraModifier(SPELL_AURA_MOD_DETECTED_RANGE);
1566 }
1567
1568 // "Minimum Aggro Radius for a mob seems to be combat range (5 yards)"
1569 if (RetDistance < 5)
1570 RetDistance = 5;
1571
1572 return (RetDistance*aggroRate);
1573}

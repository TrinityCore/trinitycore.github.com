47void TempSummon::Update(uint32 diff)
48{
49 Creature::Update(diff);
50
51 if (m_deathState == DEAD)
52 {
53 UnSummon();
54 return;
55 }
56 switch (m_type)
57 {
58 case TEMPSUMMON_MANUAL_DESPAWN:
59 break;
60 case TEMPSUMMON_TIMED_DESPAWN:
61 {
62 if (m_timer <= diff)
63 {
64 UnSummon();
65 return;
66 }
67
68 m_timer -= diff;
69 break;
70 }
71 case TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT:
72 {
73 if (!IsInCombat())
74 {
75 if (m_timer <= diff)
76 {
77 UnSummon();
78 return;
79 }
80
81 m_timer -= diff;
82 }
83 else if (m_timer != m_lifetime)
84 m_timer = m_lifetime;
85
86 break;
87 }
88
89 case TEMPSUMMON_CORPSE_TIMED_DESPAWN:
90 {
91 if (m_deathState == CORPSE)
92 {
93 if (m_timer <= diff)
94 {
95 UnSummon();
96 return;
97 }
98
99 m_timer -= diff;
100 }
101 break;
102 }
103 case TEMPSUMMON_CORPSE_DESPAWN:
104 {
105 // if m_deathState is DEAD, CORPSE was skipped
106 if (m_deathState == CORPSE || m_deathState == DEAD)
107 {
108 UnSummon();
109 return;
110 }
111
112 break;
113 }
114 case TEMPSUMMON_DEAD_DESPAWN:
115 {
116 if (m_deathState == DEAD)
117 {
118 UnSummon();
119 return;
120 }
121 break;
122 }
123 case TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN:
124 {
125 // if m_deathState is DEAD, CORPSE was skipped
126 if (m_deathState == CORPSE || m_deathState == DEAD)
127 {
128 UnSummon();
129 return;
130 }
131
132 if (!IsInCombat())
133 {
134 if (m_timer <= diff)
135 {
136 UnSummon();
137 return;
138 }
139 else
140 m_timer -= diff;
141 }
142 else if (m_timer != m_lifetime)
143 m_timer = m_lifetime;
144 break;
145 }
146 case TEMPSUMMON_TIMED_OR_DEAD_DESPAWN:
147 {
148 // if m_deathState is DEAD, CORPSE was skipped
149 if (m_deathState == DEAD)
150 {
151 UnSummon();
152 return;
153 }
154
155 if (!IsInCombat() && IsAlive())
156 {
157 if (m_timer <= diff)
158 {
159 UnSummon();
160 return;
161 }
162 else
163 m_timer -= diff;
164 }
165 else if (m_timer != m_lifetime)
166 m_timer = m_lifetime;
167 break;
168 }
169 default:
170 UnSummon();
171 TC_LOG_ERROR("entities.unit", "Temporary summoned creature (entry: %u) have unknown type %u of ", GetEntry(), m_type);
172 break;
173 }
174}

176void TempSummon::InitStats(uint32 duration)
177{
178 ASSERT(!IsPet());
179
180 m_timer = duration;
181 m_lifetime = duration;
182
183 if (m_type == TEMPSUMMON_MANUAL_DESPAWN)
184 m_type = (duration == 0) ? TEMPSUMMON_DEAD_DESPAWN : TEMPSUMMON_TIMED_DESPAWN;
185
186 Unit* owner = GetSummoner();
187
188 if (owner && IsTrigger() && m_spells[0])
189 {
190 setFaction(owner->getFaction());
191 SetLevel(owner->getLevel());
192 if (owner->GetTypeId() == TYPEID_PLAYER)
193 m_ControlledByPlayer = true;
194 }
195
196 if (!m_Properties)
197 return;
198
199 if (owner)
200 {
201 int32 slot = m_Properties->Slot;
202 if (slot > 0)
203 {
204 if (!owner->m_SummonSlot[slot].IsEmpty() && owner->m_SummonSlot[slot] != GetGUID())
205 {
206 Creature* oldSummon = GetMap()->GetCreature(owner->m_SummonSlot[slot]);
207 if (oldSummon && oldSummon->IsSummon())
208 oldSummon->ToTempSummon()->UnSummon();
209 }
210 owner->m_SummonSlot[slot] = GetGUID();
211 }
212 }
213
214 if (m_Properties->Faction)
215 setFaction(m_Properties->Faction);
216 else if (IsVehicle() && owner) // properties should be vehicle
217 setFaction(owner->getFaction());
218}

220void TempSummon::InitSummon()
221{
222 Unit* owner = GetSummoner();
223 if (owner)
224 {
225 if (owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature()->IsAIEnabled)
226 owner->ToCreature()->AI()->JustSummoned(this);
227 if (IsAIEnabled)
228 AI()->IsSummonedBy(owner);
229 }
230}

237void TempSummon::UnSummon(uint32 msTime)
238{
239 if (msTime)
240 {
241 ForcedUnsummonDelayEvent* pEvent = new ForcedUnsummonDelayEvent(*this);
242
243 m_Events.AddEvent(pEvent, m_Events.CalculateTime(msTime));
244 return;
245 }
246
247 //ASSERT(!IsPet());
248 if (IsPet())
249 {
250 ((Pet*)this)->Remove(PET_SAVE_NOT_IN_SLOT);
251 ASSERT(!IsInWorld());
252 return;
253 }
254
255 Unit* owner = GetSummoner();
256 if (owner && owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature()->IsAIEnabled)
257 owner->ToCreature()->AI()->SummonedCreatureDespawn(this);
258
259 AddObjectToRemoveList();
260}

268void TempSummon::RemoveFromWorld()
269{
270 if (!IsInWorld())
271 return;
272
273 if (m_Properties)
274 {
275 int32 slot = m_Properties->Slot;
276 if (slot > 0)
277 if (Unit* owner = GetSummoner())
278 if (owner->m_SummonSlot[slot] == GetGUID())
279 owner->m_SummonSlot[slot].Clear();
280 }
281
282 //if (GetOwnerGUID())
283 // TC_LOG_ERROR("entities.unit", "Unit %u has owner guid when removed from world", GetEntry());
284
285 Creature::RemoveFromWorld();
286}

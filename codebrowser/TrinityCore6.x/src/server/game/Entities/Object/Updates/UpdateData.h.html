59class UpdateData
60{
61 public:
62 UpdateData(uint32 map);
63 UpdateData(UpdateData&& right) : m_map(right.m_map), m_blockCount(right.m_blockCount),
64 m_outOfRangeGUIDs(std::move(right.m_outOfRangeGUIDs)),
65 m_data(std::move(right.m_data))
66 {
67 }
68
69 void AddOutOfRangeGUID(GuidSet& guids);
70 void AddOutOfRangeGUID(ObjectGuid guid);
71 void AddUpdateBlock(const ByteBuffer &block);
72 bool BuildPacket(WorldPacket* packet);
73 bool HasData() const { return m_blockCount > 0 || !m_outOfRangeGUIDs.empty(); }
74 void Clear();
75
76 GuidSet const& GetOutOfRangeGUIDs() const { return m_outOfRangeGUIDs; }
77
78 protected:
79 uint32 m_map;
80 uint32 m_blockCount;
81 GuidSet m_outOfRangeGUIDs;
82 ByteBuffer m_data;
83
84 UpdateData(UpdateData const& right) = delete;
85 UpdateData& operator=(UpdateData const& right) = delete;
86};

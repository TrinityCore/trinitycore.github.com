109{
110 m_uint32Values = new uint32[m_valuesCount];

113 _changesMask.SetCount(m_valuesCount);

118 _dynamicValues = new std::vector<uint32>[_dynamicValuesCount];
119 _dynamicChangesArrayMask = new UpdateMask[_dynamicValuesCount];

121
122 m_objectUpdated = false;
123}

167void Object::BuildCreateUpdateBlockForPlayer(UpdateData* data, Player* target) const
168{
169 if (!target)
170 return;
171
172 uint8 updateType = UPDATETYPE_CREATE_OBJECT;
173 uint32 flags = m_updateFlag;
174
175 /** lower flag1 **/
176 if (target == this) // building packet for yourself
177 flags |= UPDATEFLAG_SELF;
178
179 switch (GetGUID().GetHigh())
180 {
181 case HighGuid::Player:
182 case HighGuid::Pet:
183 case HighGuid::Corpse:
184 case HighGuid::DynamicObject:
185 case HighGuid::AreaTrigger:
186 updateType = UPDATETYPE_CREATE_OBJECT2;
187 break;
188 case HighGuid::Creature:
189 case HighGuid::Vehicle:
190 {
191 if (TempSummon const* summon = ToUnit()->ToTempSummon())
192 if (summon->GetSummonerGUID().IsPlayer())
193 updateType = UPDATETYPE_CREATE_OBJECT2;
194
195 break;
196 }
197 case HighGuid::GameObject:
198 {
199 if (ToGameObject()->GetOwnerGUID().IsPlayer())
200 updateType = UPDATETYPE_CREATE_OBJECT2;
201 break;
202 }
203 default:
204 break;
205 }
206
207 if (WorldObject const* worldObject = dynamic_cast<WorldObject const*>(this))
208 {
209 if (!(flags & UPDATEFLAG_LIVING))
210 if (!worldObject->m_movementInfo.transport.guid.IsEmpty())
211 flags |= UPDATEFLAG_TRANSPORT_POSITION;
212
213 if (worldObject->GetAIAnimKitId() || worldObject->GetMovementAnimKitId() || worldObject->GetMeleeAnimKitId())
214 flags |= UPDATEFLAG_ANIMKITS;
215 }
216
217 if (flags & UPDATEFLAG_STATIONARY_POSITION)
218 {
219 // UPDATETYPE_CREATE_OBJECT2 for some gameobject types...
220 if (isType(TYPEMASK_GAMEOBJECT))
221 {
222 switch (ToGameObject()->GetGoType())
223 {
224 case GAMEOBJECT_TYPE_TRAP:
225 case GAMEOBJECT_TYPE_DUEL_ARBITER:
226 case GAMEOBJECT_TYPE_FLAGSTAND:
227 case GAMEOBJECT_TYPE_FLAGDROP:
228 updateType = UPDATETYPE_CREATE_OBJECT2;
229 break;
230 default:
231 break;
232 }
233 }
234 }
235
236 if (Unit const* unit = ToUnit())
237 if (unit->GetVictim())
238 flags |= UPDATEFLAG_HAS_TARGET;
239
240 ByteBuffer buf(0x400);
241 buf << uint8(updateType);
242 buf << GetPackGUID();
243 buf << uint8(m_objectTypeId);
244
245 BuildMovementUpdate(&buf, flags);
246 BuildValuesUpdate(updateType, &buf, target);
247 BuildDynamicValuesUpdate(updateType, &buf, target);
248 data->AddUpdateBlock(buf);
249}

251void Object::SendUpdateToPlayer(Player* player)
252{
253 // send create update to player
254 UpdateData upd(player->GetMapId());
255 WorldPacket packet;
256
257 BuildCreateUpdateBlockForPlayer(&upd, player);
258 upd.BuildPacket(&packet);
259 player->GetSession()->SendPacket(&packet);
260}

746void Object::BuildValuesUpdate(uint8 updateType, ByteBuffer* data, Player* target) const
747{
748 if (!target)
749 return;
750
751 ByteBuffer fieldBuffer;
752 UpdateMask updateMask;
753 updateMask.SetCount(m_valuesCount);
754
755 uint32* flags = NULL;
756 uint32 visibleFlag = GetUpdateFieldData(target, flags);
757 ASSERT(flags);
758
759 for (uint16 index = 0; index < m_valuesCount; ++index)
760 {
761 if (_fieldNotifyFlags & flags[index] ||
762 ((updateType == UPDATETYPE_VALUES ? _changesMask.GetBit(index) : m_uint32Values[index]) && (flags[index] & visibleFlag)))
763 {
764 updateMask.SetBit(index);
765 fieldBuffer << m_uint32Values[index];
766 }
767 }
768
769 *data << uint8(updateMask.GetBlockCount());
770 updateMask.AppendToPacket(data);
771 data->append(fieldBuffer);
772}

774void Object::BuildDynamicValuesUpdate(uint8 updateType, ByteBuffer* data, Player* target) const
775{
776 if (!target)
777 return;
778
779 ByteBuffer fieldBuffer;
780 UpdateMask updateMask;
781 updateMask.SetCount(_dynamicValuesCount);
782
783 uint32* flags = nullptr;
784 uint32 visibleFlag = GetDynamicUpdateFieldData(target, flags);
785
786 for (uint16 index = 0; index < _dynamicValuesCount; ++index)
787 {
788 ByteBuffer buffer;
789 std::vector<uint32> const& values = _dynamicValues[index];
790 if (_fieldNotifyFlags & flags[index] ||
791 ((updateType == UPDATETYPE_VALUES ? _dynamicChangesMask.GetBit(index) : !values.empty()) && (flags[index] & visibleFlag)))
792 {
793 updateMask.SetBit(index);
794
795 UpdateMask arrayMask;
796 arrayMask.SetCount(values.size());
797 for (std::size_t v = 0; v < values.size(); ++v)
798 {
799 if (updateType != UPDATETYPE_VALUES || _dynamicChangesArrayMask[index].GetBit(v))
800 {
801 arrayMask.SetBit(v);
802 buffer << uint32(values[v]);
803 }
804 }
805
806 fieldBuffer << uint8(arrayMask.GetBlockCount());
807 arrayMask.AppendToPacket(&fieldBuffer);
808 fieldBuffer.append(buffer);
809 }
810 }
811
812 *data << uint8(updateMask.GetBlockCount());
813 updateMask.AppendToPacket(data);
814 data->append(fieldBuffer);
815}

855uint32 Object::GetUpdateFieldData(Player const* target, uint32*& flags) const
856{
857 uint32 visibleFlag = UF_FLAG_PUBLIC;
858
859 if (target == this)
860 visibleFlag |= UF_FLAG_PRIVATE;
861
862 switch (GetTypeId())
863 {
864 case TYPEID_ITEM:
865 case TYPEID_CONTAINER:
866 flags = ItemUpdateFieldFlags;
867 if (((Item const*)this)->GetOwnerGUID() == target->GetGUID())
868 visibleFlag |= UF_FLAG_OWNER | UF_FLAG_ITEM_OWNER;
869 break;
870 case TYPEID_UNIT:
871 case TYPEID_PLAYER:
872 {
873 Player* plr = ToUnit()->GetCharmerOrOwnerPlayerOrPlayerItself();
874 flags = UnitUpdateFieldFlags;
875 if (ToUnit()->GetOwnerGUID() == target->GetGUID())
876 visibleFlag |= UF_FLAG_OWNER;
877
878 if (HasFlag(OBJECT_DYNAMIC_FLAGS, UNIT_DYNFLAG_SPECIALINFO))
879 if (ToUnit()->HasAuraTypeWithCaster(SPELL_AURA_EMPATHY, target->GetGUID()))
880 visibleFlag |= UF_FLAG_SPECIAL_INFO;
881
882 if (plr && plr->IsInSameRaidWith(target))
883 visibleFlag |= UF_FLAG_PARTY_MEMBER;
884 break;
885 }
886 case TYPEID_GAMEOBJECT:
887 flags = GameObjectUpdateFieldFlags;
888 if (ToGameObject()->GetOwnerGUID() == target->GetGUID())
889 visibleFlag |= UF_FLAG_OWNER;
890 break;
891 case TYPEID_DYNAMICOBJECT:
892 flags = DynamicObjectUpdateFieldFlags;
893 if (ToDynObject()->GetCasterGUID() == target->GetGUID())
894 visibleFlag |= UF_FLAG_OWNER;
895 break;
896 case TYPEID_CORPSE:
897 flags = CorpseUpdateFieldFlags;
898 if (ToCorpse()->GetOwnerGUID() == target->GetGUID())
899 visibleFlag |= UF_FLAG_OWNER;
900 break;
901 case TYPEID_AREATRIGGER:
902 flags = AreaTriggerUpdateFieldFlags;
903 break;
904 case TYPEID_SCENEOBJECT:
905 flags = SceneObjectUpdateFieldFlags;
906 break;
907 case TYPEID_CONVERSATION:
908 flags = ConversationUpdateFieldFlags;
909 break;
910 case TYPEID_OBJECT:
911 ABORT();
912 break;
913 }
914
915 return visibleFlag;
916}

1937bool WorldObject::CanSeeOrDetect(WorldObject const* obj, bool ignoreStealth, bool distanceCheck, bool checkAlert) const
1938{
1939 if (this == obj)
1940 return true;
1941
1942 if (obj->IsNeverVisible() || CanNeverSee(obj))
1943 return false;
1944
1945 if (obj->IsAlwaysVisibleFor(this) || CanAlwaysSee(obj))
1946 return true;
1947
1948 bool corpseVisibility = false;
1949 if (distanceCheck)
1950 {
1951 bool corpseCheck = false;
1952 if (Player const* thisPlayer = ToPlayer())
1953 {
1954 if (thisPlayer->isDead() && thisPlayer->GetHealth() > 0 && // Cheap way to check for ghost state
1955 !(obj->m_serverSideVisibility.GetValue(SERVERSIDE_VISIBILITY_GHOST) & m_serverSideVisibility.GetValue(SERVERSIDE_VISIBILITY_GHOST) & GHOST_VISIBILITY_GHOST))
1956 {
1957 if (Corpse* corpse = thisPlayer->GetCorpse())
1958 {
1959 corpseCheck = true;
1960 if (corpse->IsWithinDist(thisPlayer, GetSightRange(obj), false))
1961 if (corpse->IsWithinDist(obj, GetSightRange(obj), false))
1962 corpseVisibility = true;
1963 }
1964 }
1965 }
1966
1967 WorldObject const* viewpoint = this;
1968 if (Player const* player = this->ToPlayer())
1969 viewpoint = player->GetViewpoint();
1970
1971 if (!viewpoint)
1972 viewpoint = this;
1973
1974 if (!corpseCheck && !viewpoint->IsWithinDist(obj, GetSightRange(obj), false))
1975 return false;
1976 }
1977
1978 // GM visibility off or hidden NPC
1979 if (!obj->m_serverSideVisibility.GetValue(SERVERSIDE_VISIBILITY_GM))
1980 {
1981 // Stop checking other things for GMs
1982 if (m_serverSideVisibilityDetect.GetValue(SERVERSIDE_VISIBILITY_GM))
1983 return true;
1984 }
1985 else
1986 return m_serverSideVisibilityDetect.GetValue(SERVERSIDE_VISIBILITY_GM) >= obj->m_serverSideVisibility.GetValue(SERVERSIDE_VISIBILITY_GM);
1987
1988 // Ghost players, Spirit Healers, and some other NPCs
1989 if (!corpseVisibility && !(obj->m_serverSideVisibility.GetValue(SERVERSIDE_VISIBILITY_GHOST) & m_serverSideVisibilityDetect.GetValue(SERVERSIDE_VISIBILITY_GHOST)))
1990 {
1991 // Alive players can see dead players in some cases, but other objects can't do that
1992 if (Player const* thisPlayer = ToPlayer())
1993 {
1994 if (Player const* objPlayer = obj->ToPlayer())
1995 {
1996 if (thisPlayer->GetTeam() != objPlayer->GetTeam() || !thisPlayer->IsGroupVisibleFor(objPlayer))
1997 return false;
1998 }
1999 else
2000 return false;
2001 }
2002 else
2003 return false;
2004 }
2005
2006 if (obj->IsInvisibleDueToDespawn())
2007 return false;
2008
2009 if (!CanDetect(obj, ignoreStealth, checkAlert))
2010 return false;
2011
2012 return true;
2013}

2065bool WorldObject::CanDetectStealthOf(WorldObject const* obj, bool checkAlert) const
2066{
2067 // Combat reach is the minimal distance (both in front and behind),
2068 // and it is also used in the range calculation.
2069 // One stealth point increases the visibility range by 0.3 yard.
2070
2071 if (!obj->m_stealth.GetFlags())
2072 return true;
2073
2074 float distance = GetExactDist(obj);
2075 float combatReach = 0.0f;
2076
2077 Unit const* unit = ToUnit();
2078 if (unit)
2079 combatReach = unit->GetCombatReach();
2080
2081 if (distance < combatReach)
2082 return true;
2083
2084 if (!HasInArc(float(M_PI), obj))
2085 return false;
2086
2087 GameObject const* go = ToGameObject();
2088 for (uint32 i = 0; i < TOTAL_STEALTH_TYPES; ++i)
2089 {
2090 if (!(obj->m_stealth.GetFlags() & (1 << i)))
2091 continue;
2092
2093 if (unit && unit->HasAuraTypeWithMiscvalue(SPELL_AURA_DETECT_STEALTH, i))
2094 return true;
2095
2096 // Starting points
2097 int32 detectionValue = 30;
2098
2099 // Level difference: 5 point / level, starting from level 1.
2100 // There may be spells for this and the starting points too, but
2101 // not in the DBCs of the client.
2102 detectionValue += int32(getLevelForTarget(obj) - 1) * 5;
2103
2104 // Apply modifiers
2105 detectionValue += m_stealthDetect.GetValue(StealthType(i));
2106 if (go)
2107 if (Unit* owner = go->GetOwner())
2108 detectionValue -= int32(owner->getLevelForTarget(this) - 1) * 5;
2109
2110 detectionValue -= obj->m_stealth.GetValue(StealthType(i));
2111
2112 // Calculate max distance
2113 float visibilityRange = float(detectionValue) * 0.3f + combatReach;

2737 // Prevent invalid coordinates here, position is unchanged
2738 if (!Trinity::IsValidMapCoord(destx, desty))
2739 {
2740 TC_LOG_FATAL("misc", "WorldObject::MovePositionToFirstCollision invalid coordinates X: %f and Y: %f were passed!", destx, desty);
2741 return;
2742 }
2743
2744 destz = NormalizeZforCollision(this, destx, desty, pos.GetPositionZ());
2745 bool col = VMAP::VMapFactory::createOrGetVMapManager()->getObjectHitPos(GetMapId(), pos.m_positionX, pos.m_positionY, pos.m_positionZ + 0.5f, destx, desty, destz + 0.5f, destx, desty, destz, -0.5f);
2746
2747 // collision occured
2748 if (col)
2749 {
2750 // move back a bit
2751 destx -= CONTACT_DISTANCE * std::cos(angle);
2752 desty -= CONTACT_DISTANCE * std::sin(angle);
2753 dist = std::sqrt((pos.m_positionX - destx)*(pos.m_positionX - destx) + (pos.m_positionY - desty)*(pos.m_positionY - desty));
2754 }
2755
2756 // check dynamic collision
2757 col = GetMap()->getObjectHitPos(GetPhaseMask(), pos.m_positionX, pos.m_positionY, pos.m_positionZ + 0.5f, destx, desty, destz + 0.5f, destx, desty, destz, -0.5f);
2758
2759 // Collided with a gameobject
2760 if (col)
2761 {
2762 destx -= CONTACT_DISTANCE * std::cos(angle);
2763 desty -= CONTACT_DISTANCE * std::sin(angle);
2764 dist = std::sqrt((pos.m_positionX - destx)*(pos.m_positionX - destx) + (pos.m_positionY - desty)*(pos.m_positionY - desty));
2765 }
2766
2767 float step = dist / 10.0f;
2768
2769 for (uint8 j = 0; j < 10; ++j)
2770 {
2771 // do not allow too big z changes
2772 if (std::fabs(pos.m_positionZ - destz) > 6.0f)
2773 {
2774 destx -= step * std::cos(angle);
2775 desty -= step * std::sin(angle);
2776 destz = NormalizeZforCollision(this, destx, desty, pos.GetPositionZ());
2777 }
2778 // we have correct destz now
2779 else
2780 {
2781 pos.Relocate(destx, desty, destz);
2782 break;
2783 }
2784 }
2785
2786 Trinity::NormalizeMapCoord(pos.m_positionX);
2787 Trinity::NormalizeMapCoord(pos.m_positionY);
2788 pos.m_positionZ = NormalizeZforCollision(this, destx, desty, pos.GetPositionZ());
2789 pos.SetOrientation(GetOrientation());
2790}

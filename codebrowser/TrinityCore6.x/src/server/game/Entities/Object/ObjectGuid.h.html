183class ObjectGuid
184{
185 friend std::ostream& operator<<(std::ostream& stream, ObjectGuid const& guid);
186 friend ByteBuffer& operator<<(ByteBuffer& buf, ObjectGuid const& guid);
187 friend ByteBuffer& operator>>(ByteBuffer& buf, ObjectGuid& guid);
188
189 public:
190 static ObjectGuid const Empty;
191 static ObjectGuid const TradeItem;
192
193 typedef uint64 LowType;
194
195 template<HighGuid type>
196 static typename std::enable_if<ObjectGuidTraits<type>::Global, ObjectGuid>::type Create(LowType counter) { return Global(type, counter); }
197
198 template<HighGuid type>
199 static typename std::enable_if<ObjectGuidTraits<type>::RealmSpecific, ObjectGuid>::type Create(LowType counter) { return RealmSpecific(type, counter); }
200
201 template<HighGuid type>
202 static typename std::enable_if<ObjectGuidTraits<type>::MapSpecific, ObjectGuid>::type Create(uint16 mapId, uint32 entry, LowType counter) { return MapSpecific(type, 0, mapId, 0, entry, counter); }
203
204 ObjectGuid() : _low(0), _high(0) { }
205
206 std::vector<uint8> GetRawValue() const;
207 void SetRawValue(std::vector<uint8> const& guid);
208 void SetRawValue(uint64 high, uint64 low) { _high = high; _low = low; }
209 void Clear() { _high = 0; _low = 0; }
210
211 HighGuid GetHigh() const { return HighGuid((_high >> 58) & 0x3F); }
212 uint32 GetRealmId() const { return uint32((_high >> 42) & 0x1FFF); }
213 uint32 GetMapId() const { return uint32((_high >> 29) & 0x1FFF); }
214 uint32 GetEntry() const { return uint32((_high >> 6) & 0x7FFFFF); }
215 LowType GetCounter() const { return _low & UI64LIT(0x000000FFFFFFFFFF); }
216
217 static LowType GetMaxCounter(HighGuid /*high*/)
218 {
219 return UI64LIT(0xFFFFFFFFFF);
220 }
221
222 LowType GetMaxCounter() const { return GetMaxCounter(GetHigh()); }
223
224 uint8& operator[](uint32 index)
225 {
226 ASSERT(index < sizeof(uint64) * 2);
227 return ((uint8*)&_low)[index];
228 }
229
230 uint8 const& operator[](uint32 index) const
231 {
232 ASSERT(index < sizeof(uint64) * 2);
233 return ((uint8 const*)&_low)[index];
234 }
235
236 bool IsEmpty() const { return _low == 0 && _high == 0; }
237 bool IsCreature() const { return GetHigh() == HighGuid::Creature; }
238 bool IsPet() const { return GetHigh() == HighGuid::Pet; }
239 bool IsVehicle() const { return GetHigh() == HighGuid::Vehicle; }
240 bool IsCreatureOrPet() const { return IsCreature() || IsPet(); }
241 bool IsCreatureOrVehicle() const { return IsCreature() || IsVehicle(); }
242 bool IsAnyTypeCreature() const { return IsCreature() || IsPet() || IsVehicle(); }
243 bool IsPlayer() const { return !IsEmpty() && GetHigh() == HighGuid::Player; }
244 bool IsUnit() const { return IsAnyTypeCreature() || IsPlayer(); }
245 bool IsItem() const { return GetHigh() == HighGuid::Item; }
246 bool IsGameObject() const { return GetHigh() == HighGuid::GameObject; }
247 bool IsDynamicObject() const { return GetHigh() == HighGuid::DynamicObject; }
248 bool IsCorpse() const { return GetHigh() == HighGuid::Corpse; }
249 bool IsAreaTrigger() const { return GetHigh() == HighGuid::AreaTrigger; }
250 bool IsMOTransport() const { return GetHigh() == HighGuid::Transport; }
251 bool IsAnyTypeGameObject() const { return IsGameObject() || IsMOTransport(); }
252 bool IsParty() const { return GetHigh() == HighGuid::Party; }
253 bool IsGuild() const { return GetHigh() == HighGuid::Guild; }
254 bool IsSceneObject() const { return GetHigh() == HighGuid::SceneObject; }
255 bool IsConversation() const { return GetHigh() == HighGuid::Conversation; }
256
257 static TypeID GetTypeId(HighGuid high)
258 {
259 switch (high)
260 {
261 case HighGuid::Item: return TYPEID_ITEM;
262 case HighGuid::Creature:
263 case HighGuid::Pet:
264 case HighGuid::Vehicle: return TYPEID_UNIT;
265 case HighGuid::Player: return TYPEID_PLAYER;
266 case HighGuid::GameObject:
267 case HighGuid::Transport: return TYPEID_GAMEOBJECT;
268 case HighGuid::DynamicObject: return TYPEID_DYNAMICOBJECT;
269 case HighGuid::Corpse: return TYPEID_CORPSE;
270 case HighGuid::AreaTrigger: return TYPEID_AREATRIGGER;
271 case HighGuid::SceneObject: return TYPEID_SCENEOBJECT;
272 case HighGuid::Conversation: return TYPEID_CONVERSATION;
273 default: return TYPEID_OBJECT;
274 }
275 }
276
277 TypeID GetTypeId() const { return GetTypeId(GetHigh()); }
278
279 bool operator!() const { return IsEmpty(); }
280 bool operator== (ObjectGuid const& guid) const { return _low == guid._low && _high == guid._high; }
281 bool operator!= (ObjectGuid const& guid) const { return !(*this == guid); }
282 bool operator< (ObjectGuid const& guid) const
283 {
284 if (_high < guid._high)
285 return true;
286 else if (_high > guid._high)
287 return false;
288
289 return _low < guid._low;
290 }
291
292 static char const* GetTypeName(HighGuid high);
293 char const* GetTypeName() const { return !IsEmpty() ? GetTypeName(GetHigh()) : "None"; }
294 std::string ToString() const;
295
296 private:
297 static bool HasEntry(HighGuid high)
298 {
299 switch (high)
300 {
301 case HighGuid::GameObject:
302 case HighGuid::Creature:
303 case HighGuid::Pet:
304 case HighGuid::Vehicle:
305 default:
306 return true;
307 }
308 }
309
310 bool HasEntry() const { return HasEntry(GetHigh()); }
311
312 static ObjectGuid Global(HighGuid type, LowType counter);
313 static ObjectGuid RealmSpecific(HighGuid type, LowType counter);
314 static ObjectGuid MapSpecific(HighGuid type, uint8 subType, uint16 mapId, uint32 serverId, uint32 entry, LowType counter);
315
316 ObjectGuid(uint64 high, uint64 low) : _low(low), _high(high) { }
317 explicit ObjectGuid(uint32 const&) = delete; // no implementation, used to catch wrong type assignment
318
319 uint64 _low;
320 uint64 _high;
321};

351class ObjectGuidGeneratorBase
352{
353public:
354 ObjectGuidGeneratorBase(ObjectGuid::LowType start = UI64LIT(1)) : _nextGuid(start) { }
355
356 virtual void Set(uint64 val) { _nextGuid = val; }
357 virtual ObjectGuid::LowType Generate() = 0;
358 ObjectGuid::LowType GetNextAfterMaxUsed() const { return _nextGuid; }
359
360protected:
361 static void HandleCounterOverflow(HighGuid high);
362 uint64 _nextGuid;
363};
364
365template<HighGuid high>
366class ObjectGuidGenerator : public ObjectGuidGeneratorBase
367{
368public:
369 explicit ObjectGuidGenerator(ObjectGuid::LowType start = UI64LIT(1)) : ObjectGuidGeneratorBase(start) { }
370
371 ObjectGuid::LowType Generate() override
372 {
373 if (_nextGuid >= ObjectGuid::GetMaxCounter(high) - 1)
374 HandleCounterOverflow(high);
375 return _nextGuid++;
376 }
377};

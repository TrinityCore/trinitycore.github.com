73char const* ObjectGuid::GetTypeName(HighGuid high)
74{
75 if (high >= HighGuid::Count)
76 return "<unknown>";
77
78 return TypeNames[uint32(high)];
79}

81std::string ObjectGuid::ToString() const
82{
83 std::ostringstream str;
84 str << "GUID Full: 0x" << std::hex << std::uppercase << std::setw(16) << std::setfill('0') << _high << std::setw(16) << _low << std::dec << std::nouppercase;
85 str << " Type: " << GetTypeName();
86 if (HasEntry())
87 str << (IsPet() ? " Pet number: " : " Entry: ") << GetEntry();
88
89 str << " Low: " << GetCounter();
90 return str.str();
91}

98ObjectGuid ObjectGuid::RealmSpecific(HighGuid type, LowType counter)
99{

103ObjectGuid ObjectGuid::MapSpecific(HighGuid type, uint8 subType, uint16 mapId, uint32 serverId, uint32 entry, LowType counter)
104{
105 return ObjectGuid(uint64((uint64(type) << 58) | (uint64(realm.Id.Realm & 0x1FFF) << 42) | (uint64(mapId & 0x1FFF) << 29) | (uint64(entry & 0x7FFFFF) << 6) | (uint64(subType) & 0x3F)),
106 uint64((uint64(serverId & 0xFFFFFF) << 40) | (counter & UI64LIT(0xFFFFFFFFFF))));
107}

128ByteBuffer& operator<<(ByteBuffer& buf, ObjectGuid const& guid)
129{
130 uint8 lowMask = 0;
131 uint8 highMask = 0;
132 buf.FlushBits(); // flush any unwritten bits to make wpos return a meaningful value
133 std::size_t pos = buf.wpos();
134 buf << uint8(lowMask);
135 buf << uint8(highMask);
136
137 uint8 packed[8];
138 if (size_t packedSize = ByteBuffer::PackUInt64(guid._low, &lowMask, packed))
139 buf.append(packed, packedSize);
140 if (size_t packedSize = ByteBuffer::PackUInt64(guid._high, &highMask, packed))
141 buf.append(packed, packedSize);
142
143 buf.put(pos, lowMask);
144 buf.put(pos + 1, highMask);
145
146 return buf;
147}
148
149ByteBuffer& operator>>(ByteBuffer& buf, ObjectGuid& guid)
150{
151 uint8 lowMask, highMask;
152 buf >> lowMask >> highMask;
153 buf.ReadPackedUInt64(lowMask, guid._low);
154 buf.ReadPackedUInt64(highMask, guid._high);
155 return buf;
156}
157
158ByteBuffer& operator<<(ByteBuffer& buf, PackedGuid const& guid)
159{
160 buf.append(guid._packedGuid);
161 return buf;
162}

164std::ostream& operator<<(std::ostream& stream, ObjectGuid const& guid)
165{
166 std::ostringstream tmp;
167 tmp << std::hex << std::setw(16) << std::setfill('0') << guid._high << std::setw(16) << std::setfill('0') << guid._low;
168 stream << tmp.str();
169 return stream;
170}

175void ObjectGuidGeneratorBase::HandleCounterOverflow(HighGuid high)
176{
177 TC_LOG_ERROR("misc", "%s guid overflow!! Can't continue, shutting down server. ", ObjectGuid::GetTypeName(high));
178 World::StopNow(ERROR_EXIT_CODE);
179}

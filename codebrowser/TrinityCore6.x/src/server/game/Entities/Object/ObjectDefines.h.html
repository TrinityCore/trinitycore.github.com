25// used for creating values for respawn for example
26inline uint64 MAKE_PAIR64(uint32 l, uint32 h);
27inline uint32 PAIR64_HIPART(uint64 x);
28inline uint32 PAIR64_LOPART(uint64 x);
29inline uint16 MAKE_PAIR16(uint8 l, uint8 h);
30inline uint32 MAKE_PAIR32(uint16 l, uint16 h);
31inline uint16 PAIR32_HIPART(uint32 x);
32inline uint16 PAIR32_LOPART(uint32 x);
33
34uint64 MAKE_PAIR64(uint32 l, uint32 h)
35{
36 return uint64(l | (uint64(h) << 32));
37}
38
39uint32 PAIR64_HIPART(uint64 x)
40{
41 return (uint32)((x >> 32) & UI64LIT(0x00000000FFFFFFFF));
42}
43
44uint32 PAIR64_LOPART(uint64 x)
45{
46 return (uint32)(x & UI64LIT(0x00000000FFFFFFFF));
47}
48
49uint16 MAKE_PAIR16(uint8 l, uint8 h)
50{
51 return uint16(l | (uint16(h) << 8));
52}
53
54uint32 MAKE_PAIR32(uint16 l, uint16 h)
55{
56 return uint32(l | (uint32(h) << 16));
57}
58
59uint16 PAIR32_HIPART(uint32 x)
60{
61 return (uint16)((x >> 16) & 0x0000FFFF);
62}
63
64uint16 PAIR32_LOPART(uint32 x)
65{
66 return (uint16)(x & 0x0000FFFF);
67}

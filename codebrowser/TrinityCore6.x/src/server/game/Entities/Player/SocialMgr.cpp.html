27uint32 PlayerSocial::GetNumberOfSocialsWithFlag(SocialFlag flag)
28{
29 uint32 counter = 0;
30 for (PlayerSocialMap::const_iterator itr = _playerSocialMap.begin(); itr != _playerSocialMap.end(); ++itr)
31 if ((itr->second.Flags & flag) != 0)
32 ++counter;
33
34 return counter;
35}

37bool PlayerSocial::AddToSocialList(ObjectGuid const& friendGuid, SocialFlag flag)
38{
39 // check client limits
40 if (GetNumberOfSocialsWithFlag(flag) >= (((flag & SOCIAL_FLAG_FRIEND) != 0) ? SOCIALMGR_FRIEND_LIMIT : SOCIALMGR_IGNORE_LIMIT))
41 return false;
42
43 PlayerSocialMap::iterator itr = _playerSocialMap.find(friendGuid);
44 if (itr != _playerSocialMap.end())
45 {
46 itr->second.Flags |= flag;
47
48 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_CHARACTER_SOCIAL_FLAGS);
49
50 stmt->setUInt8(0, itr->second.Flags);
51 stmt->setUInt64(1, GetPlayerGUID().GetCounter());
52 stmt->setUInt64(2, friendGuid.GetCounter());
53
54 CharacterDatabase.Execute(stmt);
55 }
56 else
57 {
58 _playerSocialMap[friendGuid].Flags |= flag;
59
60 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_CHARACTER_SOCIAL);
61
62 stmt->setUInt64(0, GetPlayerGUID().GetCounter());
63 stmt->setUInt64(1, friendGuid.GetCounter());
64 stmt->setUInt8(2, flag);
65
66 CharacterDatabase.Execute(stmt);
67 }
68
69 return true;
70}

72void PlayerSocial::RemoveFromSocialList(ObjectGuid const& friendGuid, SocialFlag flag)
73{
74 PlayerSocialMap::iterator itr = _playerSocialMap.find(friendGuid);
75 if (itr == _playerSocialMap.end())
76 return;
77
78 itr->second.Flags &= ~flag;
79
80 if (!itr->second.Flags)
81 {
82 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHARACTER_SOCIAL);
83
84 stmt->setUInt64(0, GetPlayerGUID().GetCounter());
85 stmt->setUInt64(1, friendGuid.GetCounter());
86
87 CharacterDatabase.Execute(stmt);
88
89 _playerSocialMap.erase(itr);
90 }
91 else
92 {
93 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_CHARACTER_SOCIAL_FLAGS);
94
95 stmt->setUInt8(0, itr->second.Flags);
96 stmt->setUInt64(1, GetPlayerGUID().GetCounter());
97 stmt->setUInt64(2, friendGuid.GetCounter());
98
99 CharacterDatabase.Execute(stmt);
100 }
101}

103void PlayerSocial::SetFriendNote(ObjectGuid const& friendGuid, std::string const& note)
104{
105 PlayerSocialMap::iterator itr = _playerSocialMap.find(friendGuid);
106 if (itr == _playerSocialMap.end()) // not exist
107 return;
108
109 itr->second.Note = note;
110 utf8truncate(itr->second.Note, 48); // DB and client size limitation
111
112 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_CHARACTER_SOCIAL_NOTE);
113
114 stmt->setString(0, itr->second.Note);
115 stmt->setUInt64(1, GetPlayerGUID().GetCounter());
116 stmt->setUInt64(2, friendGuid.GetCounter());
117
118 CharacterDatabase.Execute(stmt);
119}

121void PlayerSocial::SendSocialList(Player* player, uint32 flags)
122{
123 ASSERT(player);
124
125 WorldPackets::Social::ContactList contactList;
126 contactList.Flags = flags;
127
128 for (PlayerSocialMap::value_type& v : _playerSocialMap)
129 {
130 if (!(v.second.Flags & flags))
131 continue;
132
133 sSocialMgr->GetFriendInfo(player, v.first, v.second);
134
135 contactList.Contacts.emplace_back(v.first, v.second);
136
137 // client's friends list and ignore list limit
138 if (contactList.Contacts.size() >= (((flags & SOCIAL_FLAG_FRIEND) != 0) ? SOCIALMGR_FRIEND_LIMIT : SOCIALMGR_IGNORE_LIMIT))
139 break;
140 }
141
142 player->SendDirectMessage(contactList.Write());
143}

164void SocialMgr::GetFriendInfo(Player* player, ObjectGuid const& friendGUID, FriendInfo& friendInfo)
165{
166 if (!player)
167 return;
168
169 friendInfo.Status = FRIEND_STATUS_OFFLINE;
170 friendInfo.Area = 0;
171 friendInfo.Level = 0;
172 friendInfo.Class = 0;
173
174 Player* target = ObjectAccessor::FindPlayer(friendGUID);
175 if (!target)
176 return;
177
178 PlayerSocial::PlayerSocialMap::iterator itr = player->GetSocial()->_playerSocialMap.find(friendGUID);
179 if (itr != player->GetSocial()->_playerSocialMap.end())
180 friendInfo.Note = itr->second.Note;
181
182 // PLAYER see his team only and PLAYER can't see MODERATOR, GAME MASTER, ADMINISTRATOR characters
183 // MODERATOR, GAME MASTER, ADMINISTRATOR can see all
184
185 if (!player->GetSession()->HasPermission(rbac::RBAC_PERM_WHO_SEE_ALL_SEC_LEVELS) &&
186 target->GetSession()->GetSecurity() > AccountTypes(sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_WHO_LIST)))
187 return;
188
189 // player can see member of other team only if CONFIG_ALLOW_TWO_SIDE_WHO_LIST
190 if (target->GetTeam() != player->GetTeam() && !player->GetSession()->HasPermission(rbac::RBAC_PERM_TWO_SIDE_WHO_LIST))
191 return;
192
193 if (target->IsVisibleGloballyFor(player))
194 {
195 if (target->isDND())
196 friendInfo.Status = FRIEND_STATUS_DND;
197 else if (target->isAFK())
198 friendInfo.Status = FRIEND_STATUS_AFK;
199 else
200 friendInfo.Status = FRIEND_STATUS_ONLINE;
201
202 friendInfo.Area = target->GetZoneId();
203 friendInfo.Level = target->getLevel();
204 friendInfo.Class = target->getClass();
205 }
206}

208void SocialMgr::SendFriendStatus(Player* player, FriendsResult result, ObjectGuid const& friendGuid, bool broadcast /*= false*/)
209{
210 FriendInfo fi;
211 GetFriendInfo(player, friendGuid, fi);
212
213 WorldPackets::Social::FriendStatus friendStatus;
214 friendStatus.Initialize(friendGuid, result, fi);
215
216 if (broadcast)
217 BroadcastToFriendListers(player, friendStatus.Write());
218 else
219 player->SendDirectMessage(friendStatus.Write());
220}

222void SocialMgr::BroadcastToFriendListers(Player* player, WorldPacket const* packet)
223{
224 ASSERT(player);
225
226 AccountTypes gmSecLevel = AccountTypes(sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_WHO_LIST));
227 for (SocialMap::const_iterator itr = _socialMap.begin(); itr != _socialMap.end(); ++itr)
228 {
229 PlayerSocial::PlayerSocialMap::const_iterator itr2 = itr->second._playerSocialMap.find(player->GetGUID());
230 if (itr2 != itr->second._playerSocialMap.end() && (itr2->second.Flags & SOCIAL_FLAG_FRIEND) != 0)
231 {
232 Player* target = ObjectAccessor::FindPlayer(itr->first);
233 if (!target)
234 continue;
235
236 WorldSession* session = target->GetSession();
237 if (!session->HasPermission(rbac::RBAC_PERM_WHO_SEE_ALL_SEC_LEVELS) && player->GetSession()->GetSecurity() > gmSecLevel)
238 continue;
239
240 if (target->GetTeam() != player->GetTeam() && !session->HasPermission(rbac::RBAC_PERM_TWO_SIDE_WHO_LIST))
241 continue;
242
243 if (player->IsVisibleGloballyFor(target))
244 session->SendPacket(packet);
245 }
246 }

249PlayerSocial* SocialMgr::LoadFromDB(PreparedQueryResult result, ObjectGuid const& guid)
250{
251 PlayerSocial* social = &_socialMap[guid];
252 social->SetPlayerGUID(guid);
253
254 if (result)
255 {
256 do
257 {
258 Field* fields = result->Fetch();
259
260 ObjectGuid friendGuid = ObjectGuid::Create<HighGuid::Player>(fields[0].GetUInt64());
261 ObjectGuid friendAccountGuid = ObjectGuid::Create<HighGuid::WowAccount>(uint64(fields[1].GetUInt32()));
262
263 uint8 flag = fields[2].GetUInt8();
264 social->_playerSocialMap[friendGuid] = FriendInfo(friendAccountGuid, flag, fields[3].GetString());
265 }
266 while (result->NextRow());
267 }
268
269 return social;
270}

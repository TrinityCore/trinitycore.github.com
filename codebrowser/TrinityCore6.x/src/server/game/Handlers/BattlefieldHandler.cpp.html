30/**
31 * @fn void WorldSession::SendBfInvitePlayerToWar(uint64 queueId, uint32 zoneId, uint32 acceptTime)
32 *
33 * @brief This send to player windows for invite player to join the war.
34 *
35 * @param queueId The queue id of Bf
36 * @param zoneId The zone where the battle is (4197 for wg)
37 * @param acceptTime Time in second that the player have for accept
38 */
39void WorldSession::SendBfInvitePlayerToWar(uint64 queueId, uint32 zoneId, uint32 acceptTime)
40{
41 WorldPackets::Battlefield::BFMgrEntryInvite bfMgrEntryInvite;
42 bfMgrEntryInvite.QueueID = queueId;
43 bfMgrEntryInvite.AreaID = zoneId;
44 bfMgrEntryInvite.ExpireTime = time(nullptr) + acceptTime;
45 SendPacket(bfMgrEntryInvite.Write());
46}

63/**
64 * @fn void WorldSession::SendBfQueueInviteResponse(uint64 queueId, uint32 zoneId, int8 battleStatus, bool canQueue, bool loggingIn)
65 *
66 * @brief This send packet for inform player that he join queue.
67 *
68 * @param queueId The queue id of Bf
69 * @param zoneId The zone where the battle is (4197 for wg)
70 * @param battleStatus Battlefield status
71 * @param canQueue if able to queue
72 * @param loggingIn on log in send queue status
73 */
74void WorldSession::SendBfQueueInviteResponse(uint64 queueId, uint32 zoneId, int8 battleStatus, bool canQueue /*= true*/, bool loggingIn /*= false*/)
75{
76 WorldPackets::Battlefield::BFMgrQueueRequestResponse bfMgrQueueRequestResponse;
77 bfMgrQueueRequestResponse.QueueID = queueId;
78 bfMgrQueueRequestResponse.AreaID = zoneId;
79 bfMgrQueueRequestResponse.Result = canQueue ? 1 : 0;
80 bfMgrQueueRequestResponse.BattleState = battleStatus;
81 bfMgrQueueRequestResponse.LoggingIn = loggingIn;
82 SendPacket(bfMgrQueueRequestResponse.Write());
83}

85/**
86 * @fn void WorldSession::SendBfEntered(uint64 queueId, bool relocated, bool onOffense)
87 *
88 * @brief This is call when player accept to join war.
89 *
90 * @param queueId The queue id of Bf
91 * @param relocated Whether player is added to Bf on the spot or teleported from queue
92 * @param onOffense Whether player belongs to attacking team or not
93 */
94void WorldSession::SendBfEntered(uint64 queueId, bool relocated, bool onOffense)
95{
96 WorldPackets::Battlefield::BFMgrEntering bfMgrEntering;
97 bfMgrEntering.ClearedAFK = _player->isAFK();
98 bfMgrEntering.Relocated = relocated;
99 bfMgrEntering.OnOffense = onOffense;
100 bfMgrEntering.QueueID = queueId;
101 SendPacket(bfMgrEntering.Write());
102}

124/**
125 * @fn void WorldSession::HandleBfEntryInviteResponse(WorldPackets::Battlefield::BFMgrEntryInviteResponse& bfMgrEntryInviteResponse)
126 *
127 * @brief Send by client on clicking in accept or refuse of invitation windows for join game.
128 */
129void WorldSession::HandleBfEntryInviteResponse(WorldPackets::Battlefield::BFMgrEntryInviteResponse& bfMgrEntryInviteResponse)
130{
131 Battlefield* bf = sBattlefieldMgr->GetBattlefieldByQueueId(bfMgrEntryInviteResponse.QueueID);
132 if (!bf)
133 return;
134
135 // If player accept invitation
136 if (bfMgrEntryInviteResponse.AcceptedInvite)
137 {
138 bf->PlayerAcceptInviteToWar(_player);
139 }
140 else
141 {
142 if (_player->GetZoneId() == bf->GetZoneId())
143 bf->KickPlayerFromBattlefield(_player->GetGUID());
144 }
145}

155 if (!bf)
156 return;
157
158 if (bfMgrQueueInviteResponse.AcceptedInvite)
159 bf->PlayerAcceptInviteToQueue(_player);
160}

163 * @fn void WorldSession::HandleBfExitRequest(WorldPackets::Battlefield::BFMgrQueueExitRequest& bfMgrQueueExitRequest)

169 Battlefield* bf = sBattlefieldMgr->GetBattlefieldByQueueId(bfMgrQueueExitRequest.QueueID);

171 return;
172
173 bf->AskToLeaveQueue(_player);
174}

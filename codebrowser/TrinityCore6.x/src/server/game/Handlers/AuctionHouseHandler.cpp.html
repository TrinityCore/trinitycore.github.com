49//this void causes that auction window is opened
50void WorldSession::SendAuctionHello(ObjectGuid guid, Creature* unit)
51{
52 if (GetPlayer()->getLevel() < sWorld->getIntConfig(CONFIG_AUCTION_LEVEL_REQ))
53 {
54 SendNotification(GetTrinityString(LANG_AUCTION_REQ), sWorld->getIntConfig(CONFIG_AUCTION_LEVEL_REQ));
55 return;
56 }
57
58 AuctionHouseEntry const* ahEntry = AuctionHouseMgr::GetAuctionHouseEntry(unit->getFaction());
59 if (!ahEntry)
60 return;
61
62 WorldPackets::AuctionHouse::AuctionHelloResponse auctionHelloResponse;
63 auctionHelloResponse.Guid = guid;
64 auctionHelloResponse.OpenForBusiness = true; // 3.3.3: 1 - AH enabled, 0 - AH disabled
65 SendPacket(auctionHelloResponse.Write());
66}

111//this void creates new auction and adds auction to some auctionhouse
112void WorldSession::HandleAuctionSellItem(WorldPackets::AuctionHouse::AuctionSellItem& packet)
113{
114 if (packet.Items.size() > MAX_AUCTION_ITEMS)
115 {
116 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
117 return;
118 }
119
120 for (auto const& item : packet.Items)
121 if (!item.Guid || !item.UseCount || item.UseCount > 1000)
122 return;
123
124 if (!packet.MinBid || !packet.RunTime)
125 return;
126
127 if (packet.MinBid > MAX_MONEY_AMOUNT || packet.BuyoutPrice > MAX_MONEY_AMOUNT)
128 {
129 TC_LOG_DEBUG("network", "WORLD: HandleAuctionSellItem - Player %s (%s) attempted to sell item with higher price than max gold amount.", _player->GetName().c_str(), _player->GetGUID().ToString().c_str());
130 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
131 return;
132 }
133
134
135 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(packet.Auctioneer, UNIT_NPC_FLAG_AUCTIONEER);
136 if (!creature)
137 {
138 TC_LOG_DEBUG("network", "WORLD: HandleAuctionSellItem - Unit (%s) not found or you can't interact with him.", packet.Auctioneer.ToString().c_str());
139 return;
140 }
141
142 AuctionHouseEntry const* auctionHouseEntry = AuctionHouseMgr::GetAuctionHouseEntry(creature->getFaction());
143 if (!auctionHouseEntry)
144 {
145 TC_LOG_DEBUG("network", "WORLD: HandleAuctionSellItem - Unit (%s) has wrong faction.", packet.Auctioneer.ToString().c_str());
146 return;
147 }
148
149 packet.RunTime *= MINUTE;
150
151 switch (packet.RunTime)
152 {
153 case 1 * MIN_AUCTION_TIME:
154 case 2 * MIN_AUCTION_TIME:
155 case 4 * MIN_AUCTION_TIME:
156 break;
157 default:
158 return;
159 }
160
161 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
162 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
163
164 uint32 finalCount = 0;
165 for (auto const& packetItem : packet.Items)
166 {
167 Item* item = _player->GetItemByGuid(packetItem.Guid);
168
169 if (!item)
170 {
171 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_ITEM_NOT_FOUND);
172 return;
173 }
174
175 if (sAuctionMgr->GetAItem(item->GetGUID().GetCounter()) || !item->CanBeTraded() || item->IsNotEmptyBag() ||
176 item->GetTemplate()->GetFlags() & ITEM_FLAG_CONJURED || item->GetUInt32Value(ITEM_FIELD_DURATION) ||
177 item->GetCount() < packetItem.UseCount)
178 {
179 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
180 return;
181 }
182
183 finalCount += packetItem.UseCount;
184 }
185
186 if (packet.Items.empty())
187 {
188 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
189 return;
190 }
191
192 if (!finalCount)
193 {
194 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
195 return;
196 }
197
198 // check if there are 2 identical guids, in this case user is most likely cheating
199 for (uint32 i = 0; i < packet.Items.size() - 1; ++i)
200 {
201 for (uint32 j = i + 1; j < packet.Items.size(); ++j)
202 {
203 if (packet.Items[i].Guid == packet.Items[j].Guid)
204 {
205 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
206 return;
207 }
208 }
209 }
210
211 for (auto const& packetItem : packet.Items)
212 {
213 Item* item = _player->GetItemByGuid(packetItem.Guid);
214
215 if (item->GetMaxStackCount() < finalCount)
216 {
217 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
218 return;
219 }
220 }
221
222 Item* item = _player->GetItemByGuid(packet.Items[0].Guid);
223
224 uint32 auctionTime = uint32(packet.RunTime * sWorld->getRate(RATE_AUCTION_TIME));
225 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());
226
227 uint32 deposit = sAuctionMgr->GetAuctionDeposit(auctionHouseEntry, packet.RunTime, item, finalCount);
228 if (!_player->HasEnoughMoney((uint64)deposit))
229 {
230 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_NOT_ENOUGHT_MONEY);
231 return;
232 }
233
234 AuctionEntry* AH = new AuctionEntry();
235
236 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_AUCTION))
237 AH->auctioneer = UI64LIT(23442); ///@TODO - HARDCODED DB GUID, BAD BAD BAD
238 else
239 AH->auctioneer = packet.Auctioneer.GetCounter();
240
241 // Required stack size of auction matches to current item stack size, just move item to auctionhouse
242 if (packet.Items.size() == 1 && item->GetCount() == packet.Items[0].UseCount)
243 {
244 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
245 {
246 sLog->outCommand(GetAccountId(), "GM %s (Account: %u) create auction: %s (Entry: %u Count: %u)",
247 GetPlayerName().c_str(), GetAccountId(), item->GetTemplate()->GetDefaultLocaleName(), item->GetEntry(), item->GetCount());
248 }
249
250 AH->Id = sObjectMgr->GenerateAuctionID();
251 AH->itemGUIDLow = item->GetGUID().GetCounter();
252 AH->itemEntry = item->GetEntry();
253 AH->itemCount = item->GetCount();
254 AH->owner = _player->GetGUID().GetCounter();
255 AH->startbid = packet.MinBid;
256 AH->bidder = UI64LIT(0);
257 AH->bid = 0;
258 AH->buyout = packet.BuyoutPrice;
259 AH->expire_time = time(NULL) + auctionTime;
260 AH->deposit = deposit;
261 AH->auctionHouseEntry = auctionHouseEntry;
262
263 TC_LOG_INFO("network", "CMSG_AUCTION_SELL_ITEM: %s %s is selling item %s %s to auctioneer " UI64FMTD " with count %u with initial bid " UI64FMTD " with buyout " UI64FMTD " and with time %u (in sec) in auctionhouse %u",
264 _player->GetGUID().ToString().c_str(), _player->GetName().c_str(), item->GetGUID().ToString().c_str(), item->GetTemplate()->GetDefaultLocaleName(), AH->auctioneer, item->GetCount(), packet.MinBid, packet.BuyoutPrice, auctionTime, AH->GetHouseId());
265 sAuctionMgr->AddAItem(item);
266 auctionHouse->AddAuction(AH);
267
268 _player->MoveItemFromInventory(item->GetBagSlot(), item->GetSlot(), true);
269
270 SQLTransaction trans = CharacterDatabase.BeginTransaction();
271 item->DeleteFromInventoryDB(trans);
272 item->SaveToDB(trans);
273 AH->SaveToDB(trans);
274 _player->SaveInventoryAndGoldToDB(trans);
275 CharacterDatabase.CommitTransaction(trans);
276
277 SendAuctionCommandResult(AH, AUCTION_SELL_ITEM, ERR_AUCTION_OK);
278
279 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_CREATE_AUCTION, 1);
280 }
281 else // Required stack size of auction does not match to current item stack size, clone item and set correct stack size
282 {
283 Item* newItem = item->CloneItem(finalCount, _player);
284 if (!newItem)
285 {
286 TC_LOG_ERROR("network", "CMSG_AUCTION_SELL_ITEM: Could not create clone of item %u", item->GetEntry());
287 SendAuctionCommandResult(NULL, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
288 delete AH;
289 return;
290 }
291
292 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
293 {
294 sLog->outCommand(GetAccountId(), "GM %s (Account: %u) create auction: %s (Entry: %u Count: %u)",
295 GetPlayerName().c_str(), GetAccountId(), newItem->GetTemplate()->GetDefaultLocaleName(), newItem->GetEntry(), newItem->GetCount());
296 }
297
298 AH->Id = sObjectMgr->GenerateAuctionID();
299 AH->itemGUIDLow = newItem->GetGUID().GetCounter();
300 AH->itemEntry = newItem->GetEntry();
301 AH->itemCount = newItem->GetCount();
302 AH->owner = _player->GetGUID().GetCounter();
303 AH->startbid = packet.MinBid;
304 AH->bidder = UI64LIT(0);
305 AH->bid = 0;
306 AH->buyout = packet.BuyoutPrice;
307 AH->expire_time = time(NULL) + auctionTime;
308 AH->deposit = deposit;
309 AH->auctionHouseEntry = auctionHouseEntry;
310
311 TC_LOG_INFO("network", "CMSG_AUCTION_SELL_ITEM: %s %s is selling %s %s to auctioneer " UI64FMTD " with count %u with initial bid " UI64FMTD " with buyout " UI64FMTD " and with time %u (in sec) in auctionhouse %u",
312 _player->GetGUID().ToString().c_str(), _player->GetName().c_str(), newItem->GetGUID().ToString().c_str(), newItem->GetTemplate()->GetDefaultLocaleName(), AH->auctioneer, newItem->GetCount(), packet.MinBid, packet.BuyoutPrice, auctionTime, AH->GetHouseId());
313 sAuctionMgr->AddAItem(newItem);
314 auctionHouse->AddAuction(AH);
315
316 for (auto const& packetItem : packet.Items)
317 {
318 Item* item2 = _player->GetItemByGuid(packetItem.Guid);
319
320 // Item stack count equals required count, ready to delete item - cloned item will be used for auction
321 if (item2->GetCount() == packetItem.UseCount)
322 {
323 _player->MoveItemFromInventory(item2->GetBagSlot(), item2->GetSlot(), true);
324
325 SQLTransaction trans = CharacterDatabase.BeginTransaction();
326 item2->DeleteFromInventoryDB(trans);
327 item2->DeleteFromDB(trans);
328 CharacterDatabase.CommitTransaction(trans);
329 delete item2;
330 }
331 else // Item stack count is bigger than required count, update item stack count and save to database - cloned item will be used for auction
332 {
333 item2->SetCount(item2->GetCount() - packetItem.UseCount);
334 item2->SetState(ITEM_CHANGED, _player);
335 _player->ItemRemovedQuestCheck(item2->GetEntry(), packetItem.UseCount);
336 item2->SendUpdateToPlayer(_player);
337
338 SQLTransaction trans = CharacterDatabase.BeginTransaction();
339 item2->SaveToDB(trans);
340 CharacterDatabase.CommitTransaction(trans);
341 }
342 }
343
344 SQLTransaction trans = CharacterDatabase.BeginTransaction();
345 newItem->SaveToDB(trans);
346 AH->SaveToDB(trans);
347 _player->SaveInventoryAndGoldToDB(trans);
348 CharacterDatabase.CommitTransaction(trans);
349
350 SendAuctionCommandResult(AH, AUCTION_SELL_ITEM, ERR_AUCTION_OK);
351
352 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_CREATE_AUCTION, 1);
353 }
354
355 _player->ModifyMoney(-int32(deposit));
356}

358// this function is called when client bids or buys out auction
359void WorldSession::HandleAuctionPlaceBid(WorldPackets::AuctionHouse::AuctionPlaceBid& packet)
360{
361 if (!packet.AuctionItemID || !packet.BidAmount)
362 return; // check for cheaters
363
364 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(packet.Auctioneer, UNIT_NPC_FLAG_AUCTIONEER);
365 if (!creature)
366 {
367 TC_LOG_DEBUG("network", "WORLD: HandleAuctionPlaceBid - %s not found or you can't interact with him.", packet.Auctioneer.ToString().c_str());
368 return;
369 }
370
371 // remove fake death
372 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
373 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
374
375 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());
376
377 AuctionEntry* auction = auctionHouse->GetAuction(packet.AuctionItemID);
378 Player* player = GetPlayer();
379
380 if (!auction || auction->owner == player->GetGUID().GetCounter())
381 {
382 //you cannot bid your own auction:
383 SendAuctionCommandResult(NULL, AUCTION_PLACE_BID, ERR_AUCTION_BID_OWN);
384 return;
385 }
386
387 // impossible have online own another character (use this for speedup check in case online owner)
388 ObjectGuid ownerGuid = ObjectGuid::Create<HighGuid::Player>(auction->owner);
389 Player* auction_owner = ObjectAccessor::FindPlayer(ownerGuid);
390 if (!auction_owner && ObjectMgr::GetPlayerAccountIdByGUID(ownerGuid) == player->GetSession()->GetAccountId())
391 {
392 //you cannot bid your another character auction:
393 SendAuctionCommandResult(NULL, AUCTION_PLACE_BID, ERR_AUCTION_BID_OWN);
394 return;
395 }
396
397 // cheating
398 if (packet.BidAmount <= auction->bid || packet.BidAmount < auction->startbid)
399 return;
400
401 // price too low for next bid if not buyout
402 if ((packet.BidAmount < auction->buyout || auction->buyout == 0) &&
403 packet.BidAmount < auction->bid + auction->GetAuctionOutBid())
404 {
405 // client already test it but just in case ...
406 SendAuctionCommandResult(auction, AUCTION_PLACE_BID, ERR_AUCTION_HIGHER_BID);
407 return;
408 }
409
410 if (!player->HasEnoughMoney(packet.BidAmount))
411 {
412 // client already test it but just in case ...
413 SendAuctionCommandResult(auction, AUCTION_PLACE_BID, ERR_AUCTION_NOT_ENOUGHT_MONEY);
414 return;
415 }
416
417 SQLTransaction trans = CharacterDatabase.BeginTransaction();
418
419 if (packet.BidAmount < auction->buyout || auction->buyout == 0)
420 {
421 if (auction->bidder)
422 {
423 if (auction->bidder == player->GetGUID().GetCounter())
424 player->ModifyMoney(-int64(packet.BidAmount - auction->bid));
425 else
426 {
427 // mail to last bidder and return money
428 sAuctionMgr->SendAuctionOutbiddedMail(auction, packet.BidAmount, GetPlayer(), trans);
429 player->ModifyMoney(-int64(packet.BidAmount));
430 }
431 }
432 else
433 player->ModifyMoney(-int64(packet.BidAmount));
434
435 auction->bidder = player->GetGUID().GetCounter();
436 auction->bid = packet.BidAmount;
437 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_BID, packet.BidAmount);
438
439 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_AUCTION_BID);
440 stmt->setUInt64(0, auction->bidder);
441 stmt->setUInt32(1, auction->bid);
442 stmt->setUInt32(2, auction->Id);
443 trans->Append(stmt);
444
445 SendAuctionCommandResult(auction, AUCTION_PLACE_BID, ERR_AUCTION_OK);
446
447 // Not sure if we must send this now.
448 Player* owner = ObjectAccessor::FindConnectedPlayer(ObjectGuid::Create<HighGuid::Player>(auction->owner));
449 Item* item = sAuctionMgr->GetAItem(auction->itemGUIDLow);
450 if (owner && item)
451 owner->GetSession()->SendAuctionOwnerBidNotification(auction, item);
452 }
453 else
454 {
455 //buyout:
456 if (player->GetGUID().GetCounter() == auction->bidder)
457 player->ModifyMoney(-int64(auction->buyout - auction->bid));
458 else
459 {
460 player->ModifyMoney(-int64(auction->buyout));
461 if (auction->bidder) //buyout for bidded auction ..
462 sAuctionMgr->SendAuctionOutbiddedMail(auction, auction->buyout, GetPlayer(), trans);
463 }
464 auction->bidder = player->GetGUID().GetCounter();
465 auction->bid = auction->buyout;
466 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_BID, auction->buyout);
467
468 SendAuctionCommandResult(auction, AUCTION_PLACE_BID, ERR_AUCTION_OK);
469
470 //- Mails must be under transaction control too to prevent data loss
471 sAuctionMgr->SendAuctionSalePendingMail(auction, trans);
472 sAuctionMgr->SendAuctionSuccessfulMail(auction, trans);
473 sAuctionMgr->SendAuctionWonMail(auction, trans);
474
475 auction->DeleteFromDB(trans);
476
477 sAuctionMgr->RemoveAItem(auction->itemGUIDLow);
478 auctionHouse->RemoveAuction(auction);
479 }
480
481 player->SaveInventoryAndGoldToDB(trans);
482 CharacterDatabase.CommitTransaction(trans);
483}

485//this void is called when auction_owner cancels his auction
486void WorldSession::HandleAuctionRemoveItem(WorldPackets::AuctionHouse::AuctionRemoveItem& packet)
487{
488 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(packet.Auctioneer, UNIT_NPC_FLAG_AUCTIONEER);
489 if (!creature)
490 {
491 TC_LOG_DEBUG("network", "WORLD: HandleAuctionRemoveItem - %s not found or you can't interact with him.", packet.Auctioneer.ToString().c_str());
492 return;
493 }
494
495 // remove fake death
496 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
497 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
498
499 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());
500
501 AuctionEntry* auction = auctionHouse->GetAuction(packet.AuctionItemID);
502 Player* player = GetPlayer();
503
504 SQLTransaction trans = CharacterDatabase.BeginTransaction();
505 if (auction && auction->owner == player->GetGUID().GetCounter())
506 {
507 Item* item = sAuctionMgr->GetAItem(auction->itemGUIDLow);
508 if (item)
509 {
510 if (auction->bidder) // If we have a bidder, we have to send him the money he paid
511 {
512 uint32 auctionCut = auction->GetAuctionCut();
513 if (!player->HasEnoughMoney((uint64)auctionCut)) //player doesn't have enough money, maybe message needed
514 return;
515 sAuctionMgr->SendAuctionCancelledToBidderMail(auction, trans);
516 player->ModifyMoney(-int64(auctionCut));
517 }
518
519 // item will deleted or added to received mail list
520 MailDraft(auction->BuildAuctionMailSubject(AUCTION_CANCELED), AuctionEntry::BuildAuctionMailBody(0, 0, auction->buyout, auction->deposit, 0))
521 .AddItem(item)
522 .SendMailTo(trans, player, auction, MAIL_CHECK_MASK_COPIED);
523 }
524 else
525 {
526 TC_LOG_ERROR("network", "Auction id: %u got non existing item (item guid : " UI64FMTD ")!", auction->Id, auction->itemGUIDLow);
527 SendAuctionCommandResult(NULL, AUCTION_CANCEL, ERR_AUCTION_DATABASE_ERROR);
528 return;
529 }
530 }
531 else
532 {
533 SendAuctionCommandResult(NULL, AUCTION_CANCEL, ERR_AUCTION_DATABASE_ERROR);
534 //this code isn't possible ... maybe there should be assert
535 TC_LOG_ERROR("network", "CHEATER: %s tried to cancel auction (id: %u) of another player or auction is NULL", player->GetGUID().ToString().c_str(), packet.AuctionItemID);
536 return;
537 }
538
539 //inform player, that auction is removed
540 SendAuctionCommandResult(auction, AUCTION_CANCEL, ERR_AUCTION_OK);
541
542 // Now remove the auction
543
544 player->SaveInventoryAndGoldToDB(trans);
545 auction->DeleteFromDB(trans);
546 CharacterDatabase.CommitTransaction(trans);
547
548 sAuctionMgr->RemoveAItem(auction->itemGUIDLow);
549 auctionHouse->RemoveAuction(auction);
550}

600void WorldSession::HandleAuctionListItems(WorldPackets::AuctionHouse::AuctionListItems& packet)
601{
602 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(packet.Auctioneer, UNIT_NPC_FLAG_AUCTIONEER);
603 if (!creature)
604 {
605 TC_LOG_DEBUG("network", "WORLD: HandleAuctionListItems - %s not found or you can't interact with him.", packet.Auctioneer.ToString().c_str());
606 return;
607 }
608
609 // remove fake death
610 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
611 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
612
613 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());
614
615 TC_LOG_DEBUG("auctionHouse", "Auctionhouse search (%s), searchedname: %s, levelmin: %u, levelmax: %u, auctionSlotID: %u, auctionMainCategory: %u, auctionSubCategory: %u, quality: %u, usable: %u",
616 packet.Auctioneer.ToString().c_str(), packet.Name.c_str(), packet.MinLevel, packet.MaxLevel , packet.InvType, packet.ItemClass, packet.ItemSubclass, packet.Quality, packet.OnlyUsable);
617
618 WorldPackets::AuctionHouse::AuctionListItemsResult result;
619
620 // converting string that we try to find to lower case
621 std::wstring wsearchedname;
622 if (!Utf8toWStr(packet.Name, wsearchedname))
623 return;
624
625 wstrToLower(wsearchedname);
626
627 auctionHouse->BuildListAuctionItems(result, _player,
628 wsearchedname, packet.Offset, packet.MinLevel, packet.MaxLevel, packet.OnlyUsable,
629 packet.InvType, packet.ItemClass, packet.ItemSubclass, packet.Quality, result.TotalCount);
630
631 result.DesiredDelay = 300;
632 result.OnlyUsable = packet.OnlyUsable;
633 SendPacket(result.Write());
634}

29void WorldSession::HandleAttackSwingOpcode(WorldPackets::Combat::AttackSwing& packet)
30{
31 Unit* enemy = ObjectAccessor::GetUnit(*_player, packet.Victim);
32
33 if (!enemy)
34 {
35 // stop attack state at client
36 SendAttackStop(nullptr);
37 return;
38 }
39
40 if (!_player->IsValidAttackTarget(enemy))
41 {
42 // stop attack state at client
43 SendAttackStop(enemy);
44 return;
45 }
46
47 //! Client explicitly checks the following before sending CMSG_ATTACKSWING packet,
48 //! so we'll place the same check here. Note that it might be possible to reuse this snippet
49 //! in other places as well.
50 if (Vehicle* vehicle = _player->GetVehicle())
51 {
52 VehicleSeatEntry const* seat = vehicle->GetSeatForPassenger(_player);
53 ASSERT(seat);
54 if (!(seat->Flags & VEHICLE_SEAT_FLAG_CAN_ATTACK))
55 {
56 SendAttackStop(enemy);
57 return;
58 }
59 }
60
61 _player->Attack(enemy, true);
62}

64void WorldSession::HandleAttackStopOpcode(WorldPackets::Combat::AttackStop& /*recvData*/)
65{
66 GetPlayer()->AttackStop();
67}

69void WorldSession::HandleSetSheathedOpcode(WorldPackets::Combat::SetSheathed& packet)
70{
71 if (packet.CurrentSheathState >= MAX_SHEATH_STATE)
72 {
73 TC_LOG_ERROR("network", "Unknown sheath state %u ??", packet.CurrentSheathState);
74 return;
75 }
76
77 GetPlayer()->SetSheath(SheathState(packet.CurrentSheathState));
78}

80void WorldSession::SendAttackStop(Unit const* enemy)
81{
82 SendPacket(WorldPackets::Combat::SAttackStop(GetPlayer(), enemy).Write());
83}

35void WorldSession::HandleQuestgiverStatusQueryOpcode(WorldPackets::Quest::QuestGiverStatusQuery& packet)
36{
37 uint32 questStatus = DIALOG_STATUS_NONE;
38
39 Object* questGiver = ObjectAccessor::GetObjectByTypeMask(*_player, packet.QuestGiverGUID, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
40 if (!questGiver)
41 {
42 TC_LOG_INFO("network", "Error in CMSG_QUESTGIVER_STATUS_QUERY, called for non-existing questgiver (%s)", packet.QuestGiverGUID.ToString().c_str());
43 return;
44 }
45
46 switch (questGiver->GetTypeId())
47 {
48 case TYPEID_UNIT:
49 {
50 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_STATUS_QUERY for npc, %s", questGiver->GetGUID().ToString().c_str());
51 if (!questGiver->ToCreature()->IsHostileTo(_player)) // do not show quest status to enemies
52 questStatus = _player->GetQuestDialogStatus(questGiver);
53 break;
54 }
55 case TYPEID_GAMEOBJECT:
56 {
57 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_STATUS_QUERY for GameObject %s", questGiver->GetGUID().ToString().c_str());
58 questStatus = _player->GetQuestDialogStatus(questGiver);
59 break;
60 }
61 default:
62 TC_LOG_ERROR("network", "QuestGiver called for unexpected type %u", questGiver->GetTypeId());
63 break;
64 }
65
66 //inform client about status of quest
67 _player->PlayerTalkClass->SendQuestGiverStatus(questStatus, packet.QuestGiverGUID);
68}

70void WorldSession::HandleQuestgiverHelloOpcode(WorldPackets::Quest::QuestGiverHello& packet)
71{
72 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_HELLO %s", packet.QuestGiverGUID.ToString().c_str());
73
74 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(packet.QuestGiverGUID, UNIT_NPC_FLAG_QUESTGIVER);
75 if (!creature)
76 {
77 TC_LOG_DEBUG("network", "WORLD: HandleQuestgiverHelloOpcode - %s not found or you can't interact with him.",
78 packet.QuestGiverGUID.ToString().c_str());
79 return;
80 }
81
82 // remove fake death
83 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
84 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
85 // Stop the npc if moving
86 creature->StopMoving();
87
88 if (sScriptMgr->OnGossipHello(_player, creature))
89 return;
90
91 _player->PrepareGossipMenu(creature, creature->GetCreatureTemplate()->GossipMenuId, true);
92 _player->SendPreparedGossip(creature);
93
94 creature->AI()->sGossipHello(_player);
95}

97void WorldSession::HandleQuestgiverAcceptQuestOpcode(WorldPackets::Quest::QuestGiverAcceptQuest& packet)
98{
99 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_ACCEPT_QUEST %s, quest = %u, startcheat = %u", packet.QuestGiverGUID.ToString().c_str(), packet.QuestID, packet.StartCheat);
100
101 Object* object = ObjectAccessor::GetObjectByTypeMask(*_player, packet.QuestGiverGUID, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT | TYPEMASK_ITEM | TYPEMASK_PLAYER);
102
103#define CLOSE_GOSSIP_CLEAR_DIVIDER() \
104 do { \
105 _player->PlayerTalkClass->SendCloseGossip(); \
106 _player->SetDivider(ObjectGuid::Empty); \
107 } while (0)
108
109 // no or incorrect quest giver
110 if (!object)
111 {
112 CLOSE_GOSSIP_CLEAR_DIVIDER();
113 return;
114 }
115
116 if (Player* playerQuestObject = object->ToPlayer())
117 {
118 if ((_player->GetDivider().IsEmpty() && _player->GetDivider() != packet.QuestGiverGUID) || !playerQuestObject->CanShareQuest(packet.QuestID))
119 {
120 CLOSE_GOSSIP_CLEAR_DIVIDER();
121 return;
122 }
123 }
124 else
125 {
126 if (!object->hasQuest(packet.QuestID))
127 {
128 CLOSE_GOSSIP_CLEAR_DIVIDER();
129 return;
130 }
131 }
132
133 // some kind of WPE protection
134 if (!_player->CanInteractWithQuestGiver(object))
135 return;
136
137 if (Quest const* quest = sObjectMgr->GetQuestTemplate(packet.QuestID))
138 {
139 // prevent cheating
140 if (!GetPlayer()->CanTakeQuest(quest, true))
141 {
142 CLOSE_GOSSIP_CLEAR_DIVIDER();
143 return;
144 }
145
146 if (!_player->GetDivider().IsEmpty())
147 {
148 Player* player = ObjectAccessor::FindPlayer(_player->GetDivider());
149 if (player)
150 {
151 player->SendPushToPartyResponse(_player, QUEST_PUSH_ACCEPTED);
152 _player->SetDivider(ObjectGuid::Empty);
153 }
154 }
155
156 if (_player->CanAddQuest(quest, true))
157 {
158 _player->AddQuestAndCheckCompletion(quest, object);
159
160 if (quest->HasFlag(QUEST_FLAGS_PARTY_ACCEPT))
161 {
162 if (Group* group = _player->GetGroup())
163 {
164 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
165 {
166 Player* player = itr->GetSource();
167
168 if (!player || player == _player) // not self
169 continue;
170
171 if (player->CanTakeQuest(quest, true))
172 {
173 player->SetDivider(_player->GetGUID());
174
175 // need confirmation that any gossip window will close
176 player->PlayerTalkClass->SendCloseGossip();
177
178 _player->SendQuestConfirmAccept(quest, player);
179 }
180 }
181 }
182 }
183
184 _player->PlayerTalkClass->SendCloseGossip();
185
186 if (quest->GetSrcSpell() > 0)
187 _player->CastSpell(_player, quest->GetSrcSpell(), true);
188
189 return;
190 }
191 }
192
193 _player->PlayerTalkClass->SendCloseGossip();
194
195#undef CLOSE_GOSSIP_CLEAR_DIVIDER
196}

198void WorldSession::HandleQuestgiverQueryQuestOpcode(WorldPackets::Quest::QuestGiverQueryQuest& packet)
199{
200 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_QUERY_QUEST QuestGiverGUID = %s, QuestID = %u, RespondToGiver = %u", packet.QuestGiverGUID.ToString().c_str(), packet.QuestID, packet.RespondToGiver);
201
202 // Verify that the guid is valid and is a questgiver or involved in the requested quest
203 Object* object = ObjectAccessor::GetObjectByTypeMask(*_player, packet.QuestGiverGUID, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT | TYPEMASK_ITEM);
204 if (!object || (!object->hasQuest(packet.QuestID) && !object->hasInvolvedQuest(packet.QuestID)))
205 {
206 _player->PlayerTalkClass->SendCloseGossip();
207 return;
208 }
209
210 if (Quest const* quest = sObjectMgr->GetQuestTemplate(packet.QuestID))
211 {
212 if (!_player->CanTakeQuest(quest, true))
213 return;
214
215 if (quest->IsAutoAccept() && _player->CanAddQuest(quest, true))
216 _player->AddQuestAndCheckCompletion(quest, object);
217
218 if (quest->IsAutoComplete())
219 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, object->GetGUID(), _player->CanCompleteQuest(quest->GetQuestId()), true);
220 else
221 _player->PlayerTalkClass->SendQuestGiverQuestDetails(quest, object->GetGUID(), true);
222 }

236void WorldSession::HandleQuestgiverChooseRewardOpcode(WorldPackets::Quest::QuestGiverChooseReward& packet)
237{
238 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_CHOOSE_REWARD npc = %s, quest = %u, reward = %u", packet.QuestGiverGUID.ToString().c_str(), packet.QuestID, packet.ItemChoiceID);
239
240 Quest const* quest = sObjectMgr->GetQuestTemplate(packet.QuestID);
241 if (!quest)
242 return;
243
244 // This is Real Item Entry, not slot id as pre 5.x
245 if (packet.ItemChoiceID)
246 {
247 ItemTemplate const* rewardProto = sObjectMgr->GetItemTemplate(packet.ItemChoiceID);
248 if (!rewardProto)
249 {
250 TC_LOG_ERROR("network", "Error in CMSG_QUESTGIVER_CHOOSE_REWARD: player %s (%s) tried to get invalid reward item (Item Entry: %u) for quest %u (possible packet-hacking detected)", _player->GetName().c_str(), _player->GetGUID().ToString().c_str(), packet.ItemChoiceID, packet.QuestID);
251 return;
252 }
253
254 bool itemValid = false;
255 for (uint32 i = 0; i < quest->GetRewChoiceItemsCount(); ++i)
256 {
257 if (quest->RewardChoiceItemId[i] && quest->RewardChoiceItemId[i] == uint32(packet.ItemChoiceID))
258 {
259 itemValid = true;
260 break;
261 }
262 }
263
264 if (!itemValid && quest->GetQuestPackageID())
265 {
266 if (std::vector<QuestPackageItemEntry const*> const* questPackageItems = sDB2Manager.GetQuestPackageItems(quest->GetQuestPackageID()))
267 {
268 for (QuestPackageItemEntry const* questPackageItem : *questPackageItems)
269 {
270 if (questPackageItem->ItemID != uint32(packet.ItemChoiceID))
271 continue;
272
273 rewardProto = sObjectMgr->GetItemTemplate(questPackageItem->ItemID);
274 if (rewardProto)
275 {
276 if (rewardProto->CanWinForPlayer(_player))
277 {
278 itemValid = true;
279 break;
280 }
281 }
282 }
283 }
284 }
285
286 if (!itemValid)
287 {
288 TC_LOG_ERROR("network", "Error in CMSG_QUESTGIVER_CHOOSE_REWARD: player %s (%s) tried to get reward item (Item Entry: %u) wich is not a reward for quest %u (possible packet-hacking detected)", _player->GetName().c_str(), _player->GetGUID().ToString().c_str(), packet.ItemChoiceID, packet.QuestID);
289 return;
290 }
291 }
292
293 Object* object = _player;
294
295 if (!quest->HasFlag(QUEST_FLAGS_AUTOCOMPLETE))
296 {
297 object = ObjectAccessor::GetObjectByTypeMask(*_player, packet.QuestGiverGUID, TYPEMASK_UNIT|TYPEMASK_GAMEOBJECT);
298 if (!object || !object->hasInvolvedQuest(packet.QuestID))
299 return;
300
301 // some kind of WPE protection
302 if (!_player->CanInteractWithQuestGiver(object))
303 return;
304 }
305
306 if ((!_player->CanSeeStartQuest(quest) && _player->GetQuestStatus(packet.QuestID) == QUEST_STATUS_NONE) ||
307 (_player->GetQuestStatus(packet.QuestID) != QUEST_STATUS_COMPLETE && !quest->IsAutoComplete()))
308 {
309 TC_LOG_ERROR("network", "Error in QUEST_STATUS_COMPLETE: player %s (%s) tried to complete quest %u, but is not allowed to do so (possible packet-hacking or high latency)",
310 _player->GetName().c_str(), _player->GetGUID().ToString().c_str(), packet.QuestID);
311 return;
312 }
313
314 if (_player->CanRewardQuest(quest, packet.ItemChoiceID, true))
315 {
316 _player->RewardQuest(quest, packet.ItemChoiceID, object);
317
318 switch (object->GetTypeId())
319 {
320 case TYPEID_UNIT:
321 case TYPEID_PLAYER:
322 {
323 //For AutoSubmition was added plr case there as it almost same exclute AI script cases.
324 Creature* creatureQGiver = object->ToCreature();
325 if (!creatureQGiver || !sScriptMgr->OnQuestReward(_player, creatureQGiver, quest, packet.ItemChoiceID))
326 {
327 // Send next quest
328 if (Quest const* nextQuest = _player->GetNextQuest(packet.QuestGiverGUID, quest))
329 {
330 // Only send the quest to the player if the conditions are met
331 if (_player->CanTakeQuest(nextQuest, false))
332 {
333 if (nextQuest->IsAutoAccept() && _player->CanAddQuest(nextQuest, true))
334 _player->AddQuestAndCheckCompletion(nextQuest, object);
335
336 _player->PlayerTalkClass->SendQuestGiverQuestDetails(nextQuest, packet.QuestGiverGUID, true);
337 }
338 }
339
340 if (creatureQGiver)
341 creatureQGiver->AI()->sQuestReward(_player, quest, packet.ItemChoiceID);
342 }
343 break;
344 }
345 case TYPEID_GAMEOBJECT:
346 {
347 GameObject* questGiver = object->ToGameObject();
348 if (!sScriptMgr->OnQuestReward(_player, questGiver, quest, packet.ItemChoiceID))
349 {
350 // Send next quest
351 if (Quest const* nextQuest = _player->GetNextQuest(packet.QuestGiverGUID, quest))
352 {
353 // Only send the quest to the player if the conditions are met
354 if (_player->CanTakeQuest(nextQuest, false))
355 {
356 if (nextQuest->IsAutoAccept() && _player->CanAddQuest(nextQuest, true))
357 _player->AddQuestAndCheckCompletion(nextQuest, object);
358
359 _player->PlayerTalkClass->SendQuestGiverQuestDetails(nextQuest, packet.QuestGiverGUID, true);
360 }
361 }
362
363 questGiver->AI()->QuestReward(_player, quest, packet.ItemChoiceID);
364 }
365 break;
366 }
367 default:
368 break;
369 }
370 }
371 else
372 _player->PlayerTalkClass->SendQuestGiverOfferReward(quest, packet.QuestGiverGUID, true);
373}

402void WorldSession::HandleQuestLogRemoveQuest(WorldPackets::Quest::QuestLogRemoveQuest& packet)
403{
404 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTLOG_REMOVE_QUEST slot = %u", packet.Entry);
405
406 if (packet.Entry < MAX_QUEST_LOG_SIZE)
407 {
408 if (uint32 questId = _player->GetQuestSlotQuestId(packet.Entry))
409 {
410 if (!_player->TakeQuestSourceItem(questId, true))
411 return; // can't un-equip some items, reject quest cancel
412
413 if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
414 {
415 if (quest->HasSpecialFlag(QUEST_SPECIAL_FLAGS_TIMED))
416 _player->RemoveTimedQuest(questId);
417
418 if (quest->HasFlag(QUEST_FLAGS_FLAGS_PVP))
419 {
420 _player->pvpInfo.IsHostile = _player->pvpInfo.IsInHostileArea || _player->HasPvPForcingQuest();
421 _player->UpdatePvPState();
422 }
423 }
424
425 _player->TakeQuestSourceItem(questId, true); // remove quest src item from player
426 _player->RemoveActiveQuest(questId);
427 _player->RemoveTimedAchievement(ACHIEVEMENT_TIMED_TYPE_QUEST, questId);
428
429 TC_LOG_INFO("network", "%s abandoned quest %u", _player->GetGUID().ToString().c_str(), questId);
430
431 if (sWorld->getBoolConfig(CONFIG_QUEST_ENABLE_QUEST_TRACKER)) // check if Quest Tracker is enabled
432 {
433 // prepare Quest Tracker datas
434 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_QUEST_TRACK_ABANDON_TIME);
435 stmt->setUInt32(0, questId);
436 stmt->setUInt64(1, _player->GetGUID().GetCounter());
437
438 // add to Quest Tracker
439 CharacterDatabase.Execute(stmt);
440 }
441 }
442
443 _player->SetQuestSlot(packet.Entry, 0);
444
445 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_QUEST_ABANDONED, 1);
446 }
447}

449void WorldSession::HandleQuestConfirmAccept(WorldPackets::Quest::QuestConfirmAccept& packet)
450{
451 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUEST_CONFIRM_ACCEPT questId = %u", packet.QuestID);
452
453 if (Quest const* quest = sObjectMgr->GetQuestTemplate(packet.QuestID))
454 {
455 if (!quest->HasFlag(QUEST_FLAGS_PARTY_ACCEPT))
456 return;
457
458 Player* originalPlayer = ObjectAccessor::FindPlayer(_player->GetDivider());
459 if (!originalPlayer)
460 return;
461
462 if (!_player->IsInSameRaidWith(originalPlayer))
463 return;
464
465 if (!originalPlayer->CanShareQuest(packet.QuestID))
466 return;
467
468 if (!_player->CanTakeQuest(quest, true))
469 return;
470
471 if (_player->CanAddQuest(quest, true))
472 _player->AddQuestAndCheckCompletion(quest, NULL); // NULL, this prevent DB script from duplicate running
473
474 _player->SetDivider(ObjectGuid::Empty);
475 }

478void WorldSession::HandleQuestgiverCompleteQuest(WorldPackets::Quest::QuestGiverCompleteQuest& packet)
479{
480 bool autoCompleteMode = packet.FromScript; // 0 - standart complete quest mode with npc, 1 - auto-complete mode
481
482 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_COMPLETE_QUEST npc = %s, questId = %u self-complete: %u", packet.QuestGiverGUID.ToString().c_str(), packet.QuestID, autoCompleteMode ? 1 : 0);
483
484 Quest const* quest = sObjectMgr->GetQuestTemplate(packet.QuestID);
485 if (!quest)
486 return;
487
488 if (autoCompleteMode && !quest->HasFlag(QUEST_FLAGS_AUTOCOMPLETE))
489 return;
490
491 Object* object = nullptr;
492 if (autoCompleteMode)
493 object = _player;
494 else
495 object = ObjectAccessor::GetObjectByTypeMask(*_player, packet.QuestGiverGUID, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
496
497 if (!object)
498 return;
499
500 if (autoCompleteMode == 0)
501 {
502 if (!object->hasInvolvedQuest(packet.QuestID))
503 return;
504
505 // some kind of WPE protection
506 if (!_player->CanInteractWithQuestGiver(object))
507 return;
508 }
509 else
510 {
511 // Do not allow completing quests on other players.
512 if (packet.QuestGiverGUID != _player->GetGUID())
513 return;
514 }
515
516 if (!_player->CanSeeStartQuest(quest) && _player->GetQuestStatus(packet.QuestID) == QUEST_STATUS_NONE)
517 {
518 TC_LOG_ERROR("network", "Possible hacking attempt: Player %s [%s] tried to complete quest [entry: %u] without being in possession of the quest!",
519 _player->GetName().c_str(), _player->GetGUID().ToString().c_str(), packet.QuestID);
520 return;
521 }
522
523 if (Battleground* bg = _player->GetBattleground())
524 bg->HandleQuestComplete(packet.QuestID, _player);
525
526 if (_player->GetQuestStatus(packet.QuestID) != QUEST_STATUS_COMPLETE)
527 {
528 if (quest->IsRepeatable())
529 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, packet.QuestGiverGUID, _player->CanCompleteRepeatableQuest(quest), false);
530 else
531 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, packet.QuestGiverGUID, _player->CanRewardQuest(quest, false), false);
532 }
533 else
534 {
535 if (quest->HasSpecialFlag(QUEST_SPECIAL_FLAGS_DELIVER)) // some items required
536 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, packet.QuestGiverGUID, _player->CanRewardQuest(quest, false), false);
537 else // no items required
538 _player->PlayerTalkClass->SendQuestGiverOfferReward(quest, packet.QuestGiverGUID, true);
539 }

546void WorldSession::HandlePushQuestToParty(WorldPackets::Quest::PushQuestToParty& packet)
547{
548 if (!_player->CanShareQuest(packet.QuestID))
549 return;
550
551 Quest const* quest = sObjectMgr->GetQuestTemplate(packet.QuestID);
552 if (!quest)
553 return;
554
555 Player * const sender = GetPlayer();
556
557 Group* group = sender->GetGroup();
558 if (!group)
559 return;
560
561 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
562 {
563 Player* receiver = itr->GetSource();
564
565 if (!receiver || receiver == sender)
566 continue;
567
568 if (!receiver->SatisfyQuestStatus(quest, false))
569 {
570 sender->SendPushToPartyResponse(receiver, QUEST_PUSH_ONQUEST);
571 continue;
572 }
573
574 if (receiver->GetQuestStatus(packet.QuestID) == QUEST_STATUS_COMPLETE)
575 {
576 sender->SendPushToPartyResponse(receiver, QUEST_PUSH_ALREADY_DONE);
577 continue;
578 }
579
580 if (!receiver->CanTakeQuest(quest, false))
581 {
582 sender->SendPushToPartyResponse(receiver, QUEST_PUSH_INVALID);
583 continue;
584 }
585
586 if (!receiver->SatisfyQuestLog(false))
587 {
588 sender->SendPushToPartyResponse(receiver, QUEST_PUSH_LOG_FULL);
589 continue;
590 }
591
592 if (!receiver->GetDivider().IsEmpty())
593 {
594 sender->SendPushToPartyResponse(receiver, QUEST_PUSH_BUSY);
595 continue;
596 }
597
598 sender->SendPushToPartyResponse(receiver, QUEST_PUSH_SUCCESS);
599
600 if (quest->IsAutoAccept() && receiver->CanAddQuest(quest, true) && receiver->CanTakeQuest(quest, true))
601 receiver->AddQuestAndCheckCompletion(quest, sender);
602
603 if (quest->IsAutoComplete() && quest->IsRepeatable() && !quest->IsDailyOrWeekly())
604 receiver->PlayerTalkClass->SendQuestGiverRequestItems(quest, sender->GetGUID(), receiver->CanCompleteRepeatableQuest(quest), true);
605 else
606 {
607 receiver->SetDivider(sender->GetGUID());
608 receiver->PlayerTalkClass->SendQuestGiverQuestDetails(quest, sender->GetGUID(), true);
609 }
610 }

625void WorldSession::HandleQuestgiverStatusMultipleQuery(WorldPackets::Quest::QuestGiverStatusMultipleQuery& /*packet*/)
626{
627 WorldPackets::Quest::QuestGiverStatusMultiple response;
628
629 for (auto itr = _player->m_clientGUIDs.begin(); itr != _player->m_clientGUIDs.end(); ++itr)
630 {
631 if (itr->IsAnyTypeCreature())
632 {
633 // need also pet quests case support
634 Creature* questgiver = ObjectAccessor::GetCreatureOrPetOrVehicle(*GetPlayer(), *itr);
635 if (!questgiver || questgiver->IsHostileTo(_player))
636 continue;
637 if (!questgiver->HasFlag64(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER))
638 continue;
639
640 response.QuestGiver.emplace_back(questgiver->GetGUID(), _player->GetQuestDialogStatus(questgiver));
641 }
642 else if (itr->IsGameObject())
643 {
644 GameObject* questgiver = GetPlayer()->GetMap()->GetGameObject(*itr);
645 if (!questgiver || questgiver->GetGoType() != GAMEOBJECT_TYPE_QUESTGIVER)
646 continue;
647
648 response.QuestGiver.emplace_back(questgiver->GetGUID(), _player->GetQuestDialogStatus(questgiver));
649 }
650 }
651
652 SendPacket(response.Write());
653}

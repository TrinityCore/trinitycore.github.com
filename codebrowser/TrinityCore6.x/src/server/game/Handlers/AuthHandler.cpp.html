24void WorldSession::SendAuthResponse(uint8 code, bool queued, uint32 queuePos)
25{
26 WorldPackets::Auth::AuthResponse response;
27 response.Result = code;
28
29 if (queued)
30 {
31 response.WaitInfo = boost::in_place();
32 response.WaitInfo->WaitCount = queuePos;
33 }
34
35 if (code == AUTH_OK)
36 {
37 response.SuccessInfo = boost::in_place();
38
39 response.SuccessInfo->AccountExpansionLevel = GetExpansion();
40 response.SuccessInfo->ActiveExpansionLevel = GetExpansion();
41 response.SuccessInfo->VirtualRealmAddress = GetVirtualRealmAddress();
42
43 std::string realmName = sObjectMgr->GetRealmName(realm.Id.Realm);
44
45 // Send current home realm. Also there is no need to send it later in realm queries.
46 response.SuccessInfo->VirtualRealms.emplace_back(GetVirtualRealmAddress(), true, false, realmName, realmName);
47
48 if (HasPermission(rbac::RBAC_PERM_USE_CHARACTER_TEMPLATES))
49 for (auto& templ : sObjectMgr->GetCharacterTemplates())
50 response.SuccessInfo->Templates.emplace_back(templ.second);
51
52 response.SuccessInfo->AvailableClasses = &sObjectMgr->GetClassExpansionRequirements();
53 response.SuccessInfo->AvailableRaces = &sObjectMgr->GetRaceExpansionRequirements();
54 }
55
56 SendPacket(response.Write());
57}

59void WorldSession::SendAuthWaitQue(uint32 position)
60{
61 WorldPackets::Auth::AuthResponse response;
62
63 if (position == 0)
64 response.Result = AUTH_OK;
65 else
66 {
67 response.WaitInfo = boost::in_place();
68 response.WaitInfo->WaitCount = position;
69 response.Result = AUTH_WAIT_QUEUE;
70 }
71
72 SendPacket(response.Write());
73}

93void WorldSession::SendFeatureSystemStatusGlueScreen()
94{
95 WorldPackets::System::FeatureSystemStatusGlueScreen features;
96 features.BpayStoreAvailable = false;
97 features.BpayStoreDisabledByParentalControls = false;
98 features.CharUndeleteEnabled = sWorld->getBoolConfig(CONFIG_FEATURE_SYSTEM_CHARACTER_UNDELETE_ENABLED);
99 features.BpayStoreEnabled = sWorld->getBoolConfig(CONFIG_FEATURE_SYSTEM_BPAY_STORE_ENABLED);
100
101 SendPacket(features.Write());
102}

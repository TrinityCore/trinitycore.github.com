38void WardenMac::Init(WorldSession* pClient, BigNumber* K)
39{
40 _session = pClient;
41 // Generate Warden Key
42 SHA1Randx WK(K->AsByteArray().get(), K->GetNumBytes());
43 WK.Generate(_inputKey, 16);
44 WK.Generate(_outputKey, 16);
45 /*
46 Seed: 4D808D2C77D905C41A6380EC08586AFE (0x05 packet)
47 Hash: <?> (0x04 packet)
48 Module MD5: 0DBBF209A27B1E279A9FEC5C168A15F7
49 New Client Key: <?>
50 New Cerver Key: <?>
51 */
52 uint8 mod_seed[16] = { 0x4D, 0x80, 0x8D, 0x2C, 0x77, 0xD9, 0x05, 0xC4, 0x1A, 0x63, 0x80, 0xEC, 0x08, 0x58, 0x6A, 0xFE };
53
54 memcpy(_seed, mod_seed, 16);
55
56 _inputCrypto.Init(_inputKey);
57 _outputCrypto.Init(_outputKey);
58 TC_LOG_DEBUG("warden", "Server side warden for client %u initializing...", pClient->GetAccountId());
59 TC_LOG_DEBUG("warden", "C->S Key: %s", ByteArrayToHexStr(_inputKey, 16).c_str());
60 TC_LOG_DEBUG("warden", "S->C Key: %s", ByteArrayToHexStr(_outputKey, 16).c_str());
61 TC_LOG_DEBUG("warden", " Seed: %s", ByteArrayToHexStr(_seed, 16).c_str());
62 TC_LOG_DEBUG("warden", "Loading Module...");
63
64 _module = GetModuleForClient();
65
66 TC_LOG_DEBUG("warden", "Module Key: %s", ByteArrayToHexStr(_module->Key, 16).c_str());
67 TC_LOG_DEBUG("warden", "Module ID: %s", ByteArrayToHexStr(_module->Id, 16).c_str());
68 RequestModule();
69}

71ClientWardenModule* WardenMac::GetModuleForClient()
72{
73 ClientWardenModule *mod = new ClientWardenModule;
74
75 uint32 len = sizeof(Module_0DBBF209A27B1E279A9FEC5C168A15F7_Data);
76
77 // data assign
78 mod->CompressedSize = len;
79 mod->CompressedData = new uint8[len];
80 memcpy(mod->CompressedData, Module_0DBBF209A27B1E279A9FEC5C168A15F7_Data, len);
81 memcpy(mod->Key, Module_0DBBF209A27B1E279A9FEC5C168A15F7_Key, 16);
82
83 // md5 hash
84 MD5_CTX ctx;
85 MD5_Init(&ctx);
86 MD5_Update(&ctx, mod->CompressedData, len);
87 MD5_Final((uint8*)&mod->Id, &ctx);
88
89 return mod;
90}

97void WardenMac::RequestHash()
98{
99 TC_LOG_DEBUG("warden", "Request hash");
100
101 // Create packet structure
102 WardenHashRequest Request;
103 Request.Command = WARDEN_SMSG_HASH_REQUEST;
104 memcpy(Request.Seed, _seed, 16);
105
106 // Encrypt with warden RC4 key.
107 EncryptData((uint8*)&Request, sizeof(WardenHashRequest));
108
109 WorldPacket pkt(SMSG_WARDEN_DATA, sizeof(WardenHashRequest));
110 pkt.append((uint8*)&Request, sizeof(WardenHashRequest));
111 _session->SendPacket(&pkt);
112}

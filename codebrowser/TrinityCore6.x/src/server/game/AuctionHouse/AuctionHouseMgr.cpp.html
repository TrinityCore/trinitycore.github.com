48AuctionHouseObject* AuctionHouseMgr::GetAuctionsMap(uint32 factionTemplateId)
49{
50 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_AUCTION))
51 return &mNeutralAuctions;
52
53 // teams have linked auction houses
54 FactionTemplateEntry const* uEntry = sFactionTemplateStore.LookupEntry(factionTemplateId);
55 if (!uEntry)
56 return &mNeutralAuctions;
57 else if (uEntry->Mask & FACTION_MASK_ALLIANCE)
58 return &mAllianceAuctions;
59 else if (uEntry->Mask & FACTION_MASK_HORDE)
60 return &mHordeAuctions;
61 else
62 return &mNeutralAuctions;
63}

65uint32 AuctionHouseMgr::GetAuctionDeposit(AuctionHouseEntry const* entry, uint32 time, Item* pItem, uint32 count)
66{
67 uint32 MSV = pItem->GetTemplate()->GetSellPrice();
68
69 if (MSV <= 0)
70 return AH_MINIMUM_DEPOSIT * sWorld->getRate(RATE_AUCTION_DEPOSIT);
71
72 float multiplier = CalculatePct(float(entry->DepositRate), 3);
73 uint32 timeHr = (((time / 60) / 60) / 12);
74 uint32 deposit = uint32(((multiplier * MSV * count / 3) * timeHr * 3) * sWorld->getRate(RATE_AUCTION_DEPOSIT));
75
76 TC_LOG_DEBUG("auctionHouse", "MSV: %u", MSV);
77 TC_LOG_DEBUG("auctionHouse", "Items: %u", count);
78 TC_LOG_DEBUG("auctionHouse", "Multiplier: %f", multiplier);
79 TC_LOG_DEBUG("auctionHouse", "Deposit: %u", deposit);
80
81 if (deposit < AH_MINIMUM_DEPOSIT * sWorld->getRate(RATE_AUCTION_DEPOSIT))
82 return AH_MINIMUM_DEPOSIT * sWorld->getRate(RATE_AUCTION_DEPOSIT);
83 else
84 return deposit;
85}

87//does not clear ram
88void AuctionHouseMgr::SendAuctionWonMail(AuctionEntry* auction, SQLTransaction& trans)
89{
90 Item* item = GetAItem(auction->itemGUIDLow);
91 if (!item)
92 return;
93
94 uint32 bidderAccId = 0;
95 ObjectGuid bidderGuid = ObjectGuid::Create<HighGuid::Player>(auction->bidder);
96 Player* bidder = ObjectAccessor::FindConnectedPlayer(bidderGuid);
97 // data for gm.log
98 std::string bidderName;
99 bool logGmTrade = false;
100
101 if (bidder)
102 {
103 bidderAccId = bidder->GetSession()->GetAccountId();
104 bidderName = bidder->GetName();
105 logGmTrade = bidder->GetSession()->HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE);
106 }
107 else
108 {
109 bidderAccId = ObjectMgr::GetPlayerAccountIdByGUID(bidderGuid);
110 logGmTrade = AccountMgr::HasPermission(bidderAccId, rbac::RBAC_PERM_LOG_GM_TRADE, realm.Id.Realm);
111
112 if (logGmTrade && !ObjectMgr::GetPlayerNameByGUID(bidderGuid, bidderName))
113 bidderName = sObjectMgr->GetTrinityStringForDBCLocale(LANG_UNKNOWN);
114 }
115
116 if (logGmTrade)
117 {
118 ObjectGuid ownerGuid = ObjectGuid::Create<HighGuid::Player>(auction->owner);
119 std::string ownerName;
120 if (!ObjectMgr::GetPlayerNameByGUID(ownerGuid, ownerName))
121 ownerName = sObjectMgr->GetTrinityStringForDBCLocale(LANG_UNKNOWN);
122
123 uint32 ownerAccId = ObjectMgr::GetPlayerAccountIdByGUID(ownerGuid);
124
125 sLog->outCommand(bidderAccId, "GM %s (Account: %u) won item in auction: %s (Entry: %u Count: %u) and pay money: %u. Original owner %s (Account: %u)",
126 bidderName.c_str(), bidderAccId, item->GetTemplate()->GetDefaultLocaleName(), item->GetEntry(), item->GetCount(), auction->bid, ownerName.c_str(), ownerAccId);
127 }
128
129 // receiver exist
130 if (bidder || bidderAccId)
131 {
132 // set owner to bidder (to prevent delete item with sender char deleting)
133 // owner in `data` will set at mail receive and item extracting
134 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ITEM_OWNER);
135 stmt->setUInt64(0, auction->bidder);
136 stmt->setUInt64(1, item->GetGUID().GetCounter());
137 trans->Append(stmt);
138
139 if (bidder)
140 {
141 bidder->GetSession()->SendAuctionWonNotification(auction, item);
142 // FIXME: for offline player need also
143 bidder->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_WON_AUCTIONS, 1);
144 }
145
146 MailDraft(auction->BuildAuctionMailSubject(AUCTION_WON), AuctionEntry::BuildAuctionMailBody(auction->owner, auction->bid, auction->buyout, 0, 0))
147 .AddItem(item)
148 .SendMailTo(trans, MailReceiver(bidder, auction->bidder), auction, MAIL_CHECK_MASK_COPIED);
149 }
150 else
151 {
152 // bidder doesn't exist, delete the item
153 sAuctionMgr->RemoveAItem(auction->itemGUIDLow, true);
154 }
155}

168//call this method to send mail to auction owner, when auction is successful, it does not clear ram
169void AuctionHouseMgr::SendAuctionSuccessfulMail(AuctionEntry* auction, SQLTransaction& trans)
170{
171 ObjectGuid owner_guid = ObjectGuid::Create<HighGuid::Player>(auction->owner);
172 Player* owner = ObjectAccessor::FindConnectedPlayer(owner_guid);
173 uint32 owner_accId = ObjectMgr::GetPlayerAccountIdByGUID(owner_guid);
174 Item* item = GetAItem(auction->itemGUIDLow);
175
176 // owner exist
177 if (owner || owner_accId)
178 {
179 uint32 profit = auction->bid + auction->deposit - auction->GetAuctionCut();
180
181 //FIXME: what do if owner offline
182 if (owner && item)
183 {
184 owner->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GOLD_EARNED_BY_AUCTIONS, profit);
185 owner->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_SOLD, auction->bid);
186 //send auction owner notification, bidder must be current!
187 owner->GetSession()->SendAuctionClosedNotification(auction, (float)sWorld->getIntConfig(CONFIG_MAIL_DELIVERY_DELAY), true, item);
188 }
189
190 MailDraft(auction->BuildAuctionMailSubject(AUCTION_SUCCESSFUL), AuctionEntry::BuildAuctionMailBody(auction->bidder, auction->bid, auction->buyout, auction->deposit, auction->GetAuctionCut()))
191 .AddMoney(profit)
192 .SendMailTo(trans, MailReceiver(owner, auction->owner), auction, MAIL_CHECK_MASK_COPIED, sWorld->getIntConfig(CONFIG_MAIL_DELIVERY_DELAY));
193 }
194}

216 }
217 else

266void AuctionHouseMgr::LoadAuctionItems()
267{
268 uint32 oldMSTime = getMSTime();
269
270 // need to clear in case we are reloading
271 if (!mAitems.empty())
272 {
273 for (ItemMap::iterator itr = mAitems.begin(); itr != mAitems.end(); ++itr)
274 delete itr->second;
275
276 mAitems.clear();
277 }
278
279 // data needs to be at first place for Item::LoadFromDB
280 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_AUCTION_ITEMS);
281 PreparedQueryResult result = CharacterDatabase.Query(stmt);
282
283 if (!result)
284 {
285 TC_LOG_INFO("server.loading", ">> Loaded 0 auction items. DB table `auctionhouse` or `item_instance` is empty!");
286 return;
287 }
288
289 uint32 count = 0;
290
291 do
292 {
293 Field* fields = result->Fetch();
294
295 ObjectGuid::LowType itemGuid = fields[0].GetUInt64();
296 uint32 itemEntry = fields[1].GetUInt32();
297
298 ItemTemplate const* proto = sObjectMgr->GetItemTemplate(itemEntry);
299 if (!proto)
300 {
301 TC_LOG_ERROR("misc", "AuctionHouseMgr::LoadAuctionItems: Unknown item (GUID: " UI64FMTD " item entry: #%u) in auction, skipped.", itemGuid, itemEntry);
302 continue;
303 }
304
305 Item* item = NewItemOrBag(proto);
306 if (!item->LoadFromDB(itemGuid, ObjectGuid::Empty, fields, itemEntry))
307 {
308 delete item;
309 continue;
310 }
311 AddAItem(item);
312
313 ++count;
314 }
315 while (result->NextRow());
316
317 TC_LOG_INFO("server.loading", ">> Loaded %u auction items in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
318}

320void AuctionHouseMgr::LoadAuctions()
321{
322 uint32 oldMSTime = getMSTime();
323
324 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_AUCTIONS);
325 PreparedQueryResult result = CharacterDatabase.Query(stmt);
326
327 if (!result)
328 {
329 TC_LOG_INFO("server.loading", ">> Loaded 0 auctions. DB table `auctionhouse` is empty.");
330 return;
331 }
332
333 uint32 count = 0;
334
335 SQLTransaction trans = CharacterDatabase.BeginTransaction();
336 do
337 {
338 Field* fields = result->Fetch();
339
340 AuctionEntry* aItem = new AuctionEntry();
341 if (!aItem->LoadFromDB(fields))
342 {
343 aItem->DeleteFromDB(trans);
344 delete aItem;
345 continue;
346 }
347
348 GetAuctionsMap(aItem->factionTemplateId)->AddAuction(aItem);
349 ++count;
350 } while (result->NextRow());
351
352 CharacterDatabase.CommitTransaction(trans);
353
354 TC_LOG_INFO("server.loading", ">> Loaded %u auctions in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
355
356}

365bool AuctionHouseMgr::RemoveAItem(ObjectGuid::LowType id, bool deleteItem)
366{
367 ItemMap::iterator i = mAitems.find(id);
368 if (i == mAitems.end())
369 return false;
370
371 if (deleteItem)
372 {
373 SQLTransaction trans = SQLTransaction(nullptr);
374 i->second->FSetState(ITEM_REMOVED);
375 i->second->SaveToDB(trans);
376 }
377
378 mAitems.erase(i);
379 return true;

389AuctionHouseEntry const* AuctionHouseMgr::GetAuctionHouseEntry(uint32 factionTemplateId)
390{
391 uint32 houseid = 7; // goblin auction house
392
393 if (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_AUCTION))
394 {
395 // FIXME: found way for proper auctionhouse selection by another way
396 // AuctionHouse.dbc have faction field with _player_ factions associated with auction house races.
397 // but no easy way convert creature faction to player race faction for specific city
398 switch (factionTemplateId)
399 {
400 case 12: houseid = 1; break; // human
401 case 29: houseid = 6; break; // orc, and generic for horde
402 case 55: houseid = 2; break; // dwarf, and generic for alliance
403 case 68: houseid = 4; break; // undead
404 case 80: houseid = 3; break; // n-elf
405 case 104: houseid = 5; break; // trolls
406 case 120: houseid = 7; break; // booty bay, neutral
407 case 474: houseid = 7; break; // gadgetzan, neutral
408 case 855: houseid = 7; break; // everlook, neutral
409 case 1604: houseid = 6; break; // b-elfs,
410 default: // for unknown case
411 {
412 FactionTemplateEntry const* u_entry = sFactionTemplateStore.LookupEntry(factionTemplateId);
413 if (!u_entry)
414 houseid = 7; // goblin auction house
415 else if (u_entry->Mask & FACTION_MASK_ALLIANCE)
416 houseid = 1; // human auction house
417 else if (u_entry->Mask & FACTION_MASK_HORDE)
418 houseid = 6; // orc auction house
419 else
420 houseid = 7; // goblin auction house
421 break;
422 }
423 }
424 }
425
426 return sAuctionHouseStore.LookupEntry(houseid);
427}

448void AuctionHouseObject::Update()
449{
450 time_t curTime = sWorld->GetGameTime();
451 ///- Handle expired auctions
452
453 // If storage is empty, no need to update. next == NULL in this case.
454 if (AuctionsMap.empty())
455 return;
456
457 SQLTransaction trans = CharacterDatabase.BeginTransaction();
458
459 for (AuctionEntryMap::iterator it = AuctionsMap.begin(); it != AuctionsMap.end();)
460 {
461 // from auctionhousehandler.cpp, creates auction pointer & player pointer
462 AuctionEntry* auction = it->second;
463 // Increment iterator due to AuctionEntry deletion
464 ++it;
465
466 ///- filter auctions expired on next update
467 if (auction->expire_time > curTime + 60)
468 continue;
469
470 ///- Either cancel the auction if there was no bidder
471 if (auction->bidder == 0 && auction->bid == 0)
472 {
473 sAuctionMgr->SendAuctionExpiredMail(auction, trans);
474 sScriptMgr->OnAuctionExpire(this, auction);
475 }
476 ///- Or perform the transaction
477 else
478 {
479 //we should send an "item sold" message if the seller is online
480 //we send the item to the winner
481 //we send the money to the seller
482 sAuctionMgr->SendAuctionSuccessfulMail(auction, trans);
483 sAuctionMgr->SendAuctionWonMail(auction, trans);
484 sScriptMgr->OnAuctionSuccessful(this, auction);
485 }
486
487 ///- In any case clear the auction
488 auction->DeleteFromDB(trans);
489
490 sAuctionMgr->RemoveAItem(auction->itemGUIDLow);
491 RemoveAuction(auction);
492 }
493
494 // Run DB changes
495 CharacterDatabase.CommitTransaction(trans);
496}

524void AuctionHouseObject::BuildListAuctionItems(WorldPackets::AuctionHouse::AuctionListItemsResult& packet, Player* player,
525 std::wstring const& wsearchedname, uint32 listfrom, uint8 levelmin, uint8 levelmax, uint8 usable,
526 uint32 inventoryType, uint32 itemClass, uint32 itemSubClass, uint32 quality, uint32& totalcount)
527{
528 time_t curTime = sWorld->GetGameTime();
529
530 for (AuctionEntryMap::const_iterator itr = AuctionsMap.begin(); itr != AuctionsMap.end(); ++itr)
531 {
532 AuctionEntry* Aentry = itr->second;
533 // Skip expired auctions
534 if (Aentry->expire_time < curTime)
535 continue;
536
537 Item* item = sAuctionMgr->GetAItem(Aentry->itemGUIDLow);
538 if (!item)
539 continue;
540
541 ItemTemplate const* proto = item->GetTemplate();
542
543 if (itemClass != 0xffffffff && proto->GetClass() != itemClass)
544 continue;
545
546 if (itemSubClass != 0xffffffff && proto->GetSubClass() != itemSubClass)
547 continue;
548
549 if (inventoryType != 0xffffffff && proto->GetInventoryType() != InventoryType(inventoryType))
550 continue;
551
552 if (quality != 0xffffffff && proto->GetQuality() != quality)
553 continue;
554
555 if (levelmin != 0 && (proto->GetBaseRequiredLevel() < levelmin || (levelmax != 0 && proto->GetBaseRequiredLevel() > levelmax)))
556 continue;
557
558 if (usable != 0 && player->CanUseItem(item) != EQUIP_ERR_OK)
559 continue;
560
561 // Allow search by suffix (ie: of the Monkey) or partial name (ie: Monkey)
562 // No need to do any of this if no search term was entered
563 if (!wsearchedname.empty())
564 {
565 std::string name = proto->GetName(player->GetSession()->GetSessionDbcLocale());
566 if (name.empty())
567 continue;
568
569 // DO NOT use GetItemEnchantMod(proto->RandomProperty) as it may return a result
570 // that matches the search but it may not equal item->GetItemRandomPropertyId()
571 // used in BuildAuctionInfo() which then causes wrong items to be listed
572 int32 propRefID = item->GetItemRandomPropertyId();
573
574 if (propRefID)
575 {
576 // Append the suffix to the name (ie: of the Monkey) if one exists
577 // These are found in ItemRandomSuffix.dbc and ItemRandomProperties.dbc
578 // even though the DBC names seem misleading
579
580 const char* suffix = nullptr;
581
582 if (propRefID < 0)
583 {
584 const ItemRandomSuffixEntry* itemRandSuffix = sItemRandomSuffixStore.LookupEntry(-propRefID);
585 if (itemRandSuffix)
586 suffix = itemRandSuffix->Name->Str[player->GetSession()->GetSessionDbcLocale()];
587 }
588 else
589 {
590 const ItemRandomPropertiesEntry* itemRandProp = sItemRandomPropertiesStore.LookupEntry(propRefID);
591 if (itemRandProp)
592 suffix = itemRandProp->Name->Str[player->GetSession()->GetSessionDbcLocale()];
593 }
594
595 // dbc local name
596 if (suffix)
597 {
598 // Append the suffix (ie: of the Monkey) to the name using localization
599 // or default enUS if localization is invalid
600 name += ' ';
601 name += suffix;
602 }
603 }
604
605 // Perform the search (with or without suffix)
606 if (!Utf8FitTo(name, wsearchedname))
607 continue;
608 }
609
610 // Add the item if no search term or if entered search term was found
611 if (packet.Items.size() < 50 && totalcount >= listfrom)
612 Aentry->BuildAuctionInfo(packet.Items, true);
613
614 ++totalcount;
615 }
616}

618//this function inserts to WorldPacket auction's data
619void AuctionEntry::BuildAuctionInfo(std::vector<WorldPackets::AuctionHouse::AuctionItem>& items, bool listAuctionItems) const
620{
621 Item* item = sAuctionMgr->GetAItem(itemGUIDLow);
622 if (!item)
623 {
624 TC_LOG_ERROR("misc", "AuctionEntry::BuildAuctionInfo: Auction %u has a non-existent item: " UI64FMTD, Id, itemGUIDLow);
625 return;
626 }
627
628 WorldPackets::AuctionHouse::AuctionItem auctionItem;
629
630 auctionItem.AuctionItemID = Id;
631 auctionItem.Item.Initialize(item);
632 auctionItem.BuyoutPrice = buyout;
633 auctionItem.CensorBidInfo = false;
634 auctionItem.CensorServerSideInfo = listAuctionItems;
635 auctionItem.Charges = item->GetSpellCharges();
636 auctionItem.Count = item->GetCount();
637 auctionItem.DeleteReason = 0; // Always 0 ?
638 auctionItem.DurationLeft = (expire_time - time(NULL)) * IN_MILLISECONDS;
639 auctionItem.EndTime = expire_time;
640 auctionItem.Flags = 0; // todo
641 auctionItem.ItemGuid = item->GetGUID();
642 auctionItem.MinBid = startbid;
643 auctionItem.Owner = ObjectGuid::Create<HighGuid::Player>(owner);
644 auctionItem.OwnerAccountID = ObjectGuid::Create<HighGuid::WowAccount>(ObjectMgr::GetPlayerAccountIdByGUID(auctionItem.Owner));
645 auctionItem.MinIncrement = bidder ? GetAuctionOutBid() : 0;
646 auctionItem.Bidder = bidder ? ObjectGuid::Create<HighGuid::Player>(bidder) : ObjectGuid::Empty;
647 auctionItem.BidAmount = bidder ? bid : 0;
648
649 for (uint8 i = 0; i < MAX_INSPECTED_ENCHANTMENT_SLOT; i++)
650 {
651 if (!item->GetEnchantmentId((EnchantmentSlot) i))
652 continue;
653
654 auctionItem.Enchantments.emplace_back(item->GetEnchantmentId((EnchantmentSlot) i), item->GetEnchantmentDuration((EnchantmentSlot) i), item->GetEnchantmentCharges((EnchantmentSlot) i), i);
655 }
656
657 items.emplace_back(auctionItem);
658}

660uint32 AuctionEntry::GetAuctionCut() const
661{
662 int32 cut = int32(CalculatePct(bid, auctionHouseEntry->ConsignmentRate) * sWorld->getRate(RATE_AUCTION_CUT));
663 return std::max(cut, 0);
664}

668{
669 uint32 outbid = CalculatePct(bid, 5);
670 return outbid ? outbid : 1;
671}

696bool AuctionEntry::LoadFromDB(Field* fields)
697{
698 Id = fields[0].GetUInt32();
699 auctioneer = fields[1].GetUInt64();
700 itemGUIDLow = fields[2].GetUInt64();
701 itemEntry = fields[3].GetUInt32();
702 itemCount = fields[4].GetUInt32();
703 owner = fields[5].GetUInt64();
704 buyout = fields[6].GetUInt32();
705 expire_time = fields[7].GetUInt32();
706 bidder = fields[8].GetUInt64();
707 bid = fields[9].GetUInt32();
708 startbid = fields[10].GetUInt32();
709 deposit = fields[11].GetUInt32();
710
711 CreatureData const* auctioneerData = sObjectMgr->GetCreatureData(auctioneer);
712 if (!auctioneerData)
713 {
714 TC_LOG_ERROR("misc", "Auction %u has not a existing auctioneer (GUID : " UI64FMTD ")", Id, auctioneer);
715 return false;
716 }
717
718 CreatureTemplate const* auctioneerInfo = sObjectMgr->GetCreatureTemplate(auctioneerData->id);
719 if (!auctioneerInfo)
720 {
721 TC_LOG_ERROR("misc", "Auction %u has not a existing auctioneer (GUID : " UI64FMTD " Entry: %u)", Id, auctioneer, auctioneerData->id);
722 return false;
723 }
724
725 factionTemplateId = auctioneerInfo->faction;
726 auctionHouseEntry = AuctionHouseMgr::GetAuctionHouseEntry(factionTemplateId);
727 if (!auctionHouseEntry)
728 {
729 TC_LOG_ERROR("misc", "Auction %u has auctioneer (GUID : " UI64FMTD " Entry: %u) with wrong faction %u", Id, auctioneer, auctioneerData->id, factionTemplateId);
730 return false;
731 }
732
733 // check if sold item exists for guid
734 // and itemEntry in fact (GetAItem will fail if problematic in result check in AuctionHouseMgr::LoadAuctionItems)
735 if (!sAuctionMgr->GetAItem(itemGUIDLow))
736 {
737 TC_LOG_ERROR("misc", "Auction %u has not a existing item : " UI64FMTD, Id, itemGUIDLow);
738 return false;
739 }
740 return true;
741}

98class AuctionHouseObject
99{
100 public:
101 ~AuctionHouseObject()
102 {
103 for (AuctionEntryMap::iterator itr = AuctionsMap.begin(); itr != AuctionsMap.end(); ++itr)
104 delete itr->second;
105 }
106
107 typedef std::map<uint32, AuctionEntry*> AuctionEntryMap;
108
109 uint32 Getcount() const { return uint32(AuctionsMap.size()); }
110
111 AuctionEntryMap::iterator GetAuctionsBegin() {return AuctionsMap.begin();}
112 AuctionEntryMap::iterator GetAuctionsEnd() {return AuctionsMap.end();}
113
114 AuctionEntry* GetAuction(uint32 id) const
115 {
116 AuctionEntryMap::const_iterator itr = AuctionsMap.find(id);
117 return itr != AuctionsMap.end() ? itr->second : NULL;
118 }

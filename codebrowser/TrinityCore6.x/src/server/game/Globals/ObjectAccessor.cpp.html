37WorldObject* ObjectAccessor::GetWorldObject(WorldObject const& p, ObjectGuid const& guid)
38{
39 switch (guid.GetHigh())
40 {
41 case HighGuid::Player: return GetPlayer(p, guid);
42 case HighGuid::Transport:
43 case HighGuid::GameObject: return GetGameObject(p, guid);
44 case HighGuid::Vehicle:
45 case HighGuid::Creature: return GetCreature(p, guid);
46 case HighGuid::Pet: return GetPet(p, guid);
47 case HighGuid::DynamicObject: return GetDynamicObject(p, guid);
48 case HighGuid::AreaTrigger: return GetAreaTrigger(p, guid);
49 case HighGuid::Corpse: return GetCorpse(p, guid);
50 default: return NULL;
51 }
52}

54Object* ObjectAccessor::GetObjectByTypeMask(WorldObject const& p, ObjectGuid const& guid, uint32 typemask)
55{
56 switch (guid.GetHigh())
57 {
58 case HighGuid::Item:
59 if (typemask & TYPEMASK_ITEM && p.GetTypeId() == TYPEID_PLAYER)
60 return ((Player const&)p).GetItemByGuid(guid);
61 break;
62 case HighGuid::Player:
63 if (typemask & TYPEMASK_PLAYER)
64 return GetPlayer(p, guid);
65 break;
66 case HighGuid::Transport:
67 case HighGuid::GameObject:
68 if (typemask & TYPEMASK_GAMEOBJECT)
69 return GetGameObject(p, guid);
70 break;
71 case HighGuid::Creature:
72 case HighGuid::Vehicle:
73 if (typemask & TYPEMASK_UNIT)
74 return GetCreature(p, guid);
75 break;
76 case HighGuid::Pet:
77 if (typemask & TYPEMASK_UNIT)
78 return GetPet(p, guid);
79 break;
80 case HighGuid::DynamicObject:
81 if (typemask & TYPEMASK_DYNAMICOBJECT)
82 return GetDynamicObject(p, guid);
83 break;
84 case HighGuid::AreaTrigger:
85 if (typemask & TYPEMASK_AREATRIGGER)
86 return GetAreaTrigger(p, guid);
87 case HighGuid::Corpse:
88 break;
89 default:
90 break;
91 }
92
93 return NULL;
94}

142Player* ObjectAccessor::GetPlayer(Map const* m, ObjectGuid const& guid)
143{
144 if (Player* player = HashMapHolder<Player>::Find(guid))
145 if (player->IsInWorld() && player->GetMap() == m)
146 return player;
147
148 return nullptr;

178Player* ObjectAccessor::FindPlayerByName(std::string const& name)
179{
180 boost::shared_lock<boost::shared_mutex> lock(*HashMapHolder<Player>::GetLock());
181
182 std::string nameStr = name;
183 std::transform(nameStr.begin(), nameStr.end(), nameStr.begin(), ::tolower);
184 HashMapHolder<Player>::MapType const& m = GetPlayers();
185 for (HashMapHolder<Player>::MapType::const_iterator iter = m.begin(); iter != m.end(); ++iter)
186 {
187 if (!iter->second->IsInWorld())
188 continue;
189 std::string currentName = iter->second->GetName();
190 std::transform(currentName.begin(), currentName.end(), currentName.begin(), ::tolower);
191 if (nameStr.compare(currentName) == 0)
192 return iter->second;
193 }
194
195 return NULL;
196}

221void ObjectAccessor::SaveAllPlayers()
222{
223 boost::shared_lock<boost::shared_mutex> lock(*HashMapHolder<Player>::GetLock());
224
225 HashMapHolder<Player>::MapType const& m = GetPlayers();
226 for (HashMapHolder<Player>::MapType::const_iterator itr = m.begin(); itr != m.end(); ++itr)
227 itr->second->SaveToDB();
228}

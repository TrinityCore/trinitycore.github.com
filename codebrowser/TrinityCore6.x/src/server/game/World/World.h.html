520/// Storage class for commands issued for delayed execution
521struct CliCommandHolder
522{
523 typedef void Print(void*, const char*);
524 typedef void CommandFinished(void*, bool success);
525
526 void* m_callbackArg;
527 char *m_command;
528 Print* m_print;
529
530 CommandFinished* m_commandFinished;
531
532 CliCommandHolder(void* callbackArg, const char *command, Print* zprint, CommandFinished* commandFinished)
533 : m_callbackArg(callbackArg), m_command(strdup(command)), m_print(zprint), m_commandFinished(commandFinished)
534 {
535 }
536
537 ~CliCommandHolder() { free(m_command); }
538
539private:
540 CliCommandHolder(CliCommandHolder const& right) = delete;
541 CliCommandHolder& operator=(CliCommandHolder const& right) = delete;
542};

40void OutdoorPvPMgr::InitOutdoorPvP()
41{
42 uint32 oldMSTime = getMSTime();
43
44 // 0 1
45 QueryResult result = WorldDatabase.Query("SELECT TypeId, ScriptName FROM outdoorpvp_template");
46
47 if (!result)
48 {
49 TC_LOG_ERROR("server.loading", ">> Loaded 0 outdoor PvP definitions. DB table `outdoorpvp_template` is empty.");
50 return;
51 }
52
53 uint32 count = 0;
54 uint32 typeId = 0;
55
56 do
57 {
58 Field* fields = result->Fetch();
59
60 typeId = fields[0].GetUInt8();
61
62 if (DisableMgr::IsDisabledFor(DISABLE_TYPE_OUTDOORPVP, typeId, NULL))
63 continue;
64
65 if (typeId >= MAX_OUTDOORPVP_TYPES)
66 {
67 TC_LOG_ERROR("sql.sql", "Invalid OutdoorPvPTypes value %u in outdoorpvp_template; skipped.", typeId);
68 continue;
69 }
70
71 OutdoorPvPData* data = new OutdoorPvPData();
72 OutdoorPvPTypes realTypeId = OutdoorPvPTypes(typeId);
73 data->TypeId = realTypeId;
74 data->ScriptId = sObjectMgr->GetScriptId(fields[1].GetString());
75 m_OutdoorPvPDatas[realTypeId] = data;
76
77 ++count;
78 }
79 while (result->NextRow());
80
81 OutdoorPvP* pvp;
82 for (uint8 i = 1; i < MAX_OUTDOORPVP_TYPES; ++i)
83 {
84 OutdoorPvPDataMap::iterator iter = m_OutdoorPvPDatas.find(OutdoorPvPTypes(i));
85 if (iter == m_OutdoorPvPDatas.end())
86 {
87 TC_LOG_ERROR("sql.sql", "Could not initialize OutdoorPvP object for type ID %u; no entry in database.", uint32(i));
88 continue;
89 }
90
91 pvp = sScriptMgr->CreateOutdoorPvP(iter->second);
92 if (!pvp)
93 {
94 TC_LOG_ERROR("outdoorpvp", "Could not initialize OutdoorPvP object for type ID %u; got NULL pointer from script.", uint32(i));
95 continue;
96 }
97
98 if (!pvp->SetupOutdoorPvP())
99 {
100 TC_LOG_ERROR("outdoorpvp", "Could not initialize OutdoorPvP object for type ID %u; SetupOutdoorPvP failed.", uint32(i));
101 delete pvp;
102 continue;
103 }
104
105 m_OutdoorPvPSet.push_back(pvp);
106 }
107
108 TC_LOG_INFO("server.loading", ">> Loaded %u outdoor PvP definitions in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
109}

116void OutdoorPvPMgr::HandlePlayerEnterZone(Player* player, uint32 zoneid)
117{
118 OutdoorPvPMap::iterator itr = m_OutdoorPvPMap.find(zoneid);
119 if (itr == m_OutdoorPvPMap.end())
120 return;
121
122 if (itr->second->HasPlayer(player))
123 return;
124
125 itr->second->HandlePlayerEnterZone(player, zoneid);
126 TC_LOG_DEBUG("outdoorpvp", "%s entered outdoorpvp id %u", player->GetGUID().ToString().c_str(), itr->second->GetTypeId());
127}
128
129void OutdoorPvPMgr::HandlePlayerLeaveZone(Player* player, uint32 zoneid)
130{
131 OutdoorPvPMap::iterator itr = m_OutdoorPvPMap.find(zoneid);
132 if (itr == m_OutdoorPvPMap.end())
133 return;
134
135 // teleport: remove once in removefromworld, once in updatezone
136 if (!itr->second->HasPlayer(player))
137 return;
138
139 itr->second->HandlePlayerLeaveZone(player, zoneid);
140 TC_LOG_DEBUG("outdoorpvp", "%s left outdoorpvp id %u", player->GetGUID().ToString().c_str(), itr->second->GetTypeId());
141}

154void OutdoorPvPMgr::Update(uint32 diff)
155{
156 m_UpdateTimer += diff;
157 if (m_UpdateTimer > OUTDOORPVP_OBJECTIVE_UPDATE_INTERVAL)
158 {
159 for (OutdoorPvPSet::iterator itr = m_OutdoorPvPSet.begin(); itr != m_OutdoorPvPSet.end(); ++itr)
160 (*itr)->Update(m_UpdateTimer);
161 m_UpdateTimer = 0;
162 }
163}

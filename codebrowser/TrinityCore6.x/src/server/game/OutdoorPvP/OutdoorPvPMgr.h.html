37// class to handle player enter / leave / areatrigger / GO use events
38class OutdoorPvPMgr
39{
40 private:
41 OutdoorPvPMgr();
42 ~OutdoorPvPMgr() { };
43
44 public:
45 static OutdoorPvPMgr* instance()
46 {
47 static OutdoorPvPMgr instance;
48 return &instance;
49 }
50
51 // create outdoor pvp events
52 void InitOutdoorPvP();
53
54 // cleanup
55 void Die();
56
57 // called when a player enters an outdoor pvp area
58 void HandlePlayerEnterZone(Player* player, uint32 areaflag);
59
60 // called when player leaves an outdoor pvp area
61 void HandlePlayerLeaveZone(Player* player, uint32 areaflag);
62
63 // called when player resurrects
64 void HandlePlayerResurrects(Player* player, uint32 areaflag);
65
66 // return assigned outdoor pvp
67 OutdoorPvP* GetOutdoorPvPToZoneId(uint32 zoneid);
68
69 // handle custom (non-exist in dbc) spell if registered
70 bool HandleCustomSpell(Player* player, uint32 spellId, GameObject* go);
71
72 // handle custom go if registered
73 bool HandleOpenGo(Player* player, GameObject* go);
74
75 ZoneScript* GetZoneScript(uint32 zoneId);
76
77 void AddZone(uint32 zoneid, OutdoorPvP* handle);
78
79 void Update(uint32 diff);
80
81 void HandleGossipOption(Player* player, Creature* creature, uint32 gossipid);
82
83 bool CanTalkTo(Player* player, Creature* creature, GossipMenuItems const& gso);
84
85 void HandleDropFlag(Player* player, uint32 spellId);
86
87 std::string GetDefenseMessage(uint32 zoneId, uint32 id, LocaleConstant locale) const;
88
89 private:
90 typedef std::vector<OutdoorPvP*> OutdoorPvPSet;
91 typedef std::map<uint32 /* zoneid */, OutdoorPvP*> OutdoorPvPMap;
92 typedef std::map<OutdoorPvPTypes, OutdoorPvPData*> OutdoorPvPDataMap;
93
94 // contains all initiated outdoor pvp events
95 // used when initing / cleaning up
96 OutdoorPvPSet m_OutdoorPvPSet;
97
98 // maps the zone ids to an outdoor pvp event
99 // used in player event handling
100 OutdoorPvPMap m_OutdoorPvPMap;
101
102 // Holds the outdoor PvP templates
103 OutdoorPvPDataMap m_OutdoorPvPDatas;
104
105 // update interval
106 uint32 m_UpdateTimer;
107};

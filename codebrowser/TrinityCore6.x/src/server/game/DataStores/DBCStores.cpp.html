184template<class T>
185inline void LoadDBC(uint32& availableDbcLocales, StoreProblemList& errors, DBCStorage<T>& storage, std::string const& dbcPath, std::string const& filename, uint32 defaultLocale, std::string const* customFormat = NULL, std::string const* customIndexName = NULL)
186{
187 // compatibility format and C++ structure sizes
188 ASSERT(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()) == sizeof(T),
189 "Size of '%s' set by format string (%u) not equal size of C++ structure (%u).",
190 filename.c_str(), DBCFileLoader::GetFormatRecordSize(storage.GetFormat()), uint32(sizeof(T)));
191
192 ++DBCFileCount;
193 std::string dbcFilename = dbcPath + localeNames[defaultLocale] + '/' + filename;
194 SqlDbc * sql = NULL;
195 if (customFormat)
196 sql = new SqlDbc(&filename, customFormat, customIndexName, storage.GetFormat());
197
198 if (storage.Load(dbcFilename.c_str(), sql))
199 {
200 for (uint8 i = 0; i < TOTAL_LOCALES; ++i)
201 {
202 if (i == LOCALE_none || !(availableDbcLocales & (1 << i)))
203 continue;
204
205 std::string localizedName(dbcPath);
206 localizedName.append(localeNames[i]);
207 localizedName.push_back('/');
208 localizedName.append(filename);
209
210 if (!storage.LoadStringsFrom(localizedName.c_str()))
211 availableDbcLocales &= ~(1<<i); // mark as not available for speedup next checks
212 }
213 }
214 else
215 {
216 // sort problematic dbc to (1) non compatible and (2) non-existed
217 if (FILE* f = fopen(dbcFilename.c_str(), "rb"))
218 {
219 std::ostringstream stream;
220 stream << dbcFilename << " exists, and has " << storage.GetFieldCount() << " field(s) (expected " << strlen(storage.GetFormat()) << "). Extracted file might be from wrong client version or a database-update has been forgotten.";
221 std::string buf = stream.str();
222 errors.push_back(buf);
223 fclose(f);
224 }
225 else
226 errors.push_back(dbcFilename);
227 }
228
229 delete sql;
230}

232template<class T>
233inline void LoadGameTable(StoreProblemList& errors, std::string const& tableName, GameTable<T>& storage, std::string const& dbcPath, std::string const& filename)
234{
235 // compatibility format and C++ structure sizes
236 ASSERT(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()) == sizeof(T),
237 "Size of '%s' set by format string (%u) not equal size of C++ structure (%u).",
238 filename.c_str(), DBCFileLoader::GetFormatRecordSize(storage.GetFormat()), uint32(sizeof(T)));
239
240 ++GameTableCount;
241 std::string dbcFilename = dbcPath + filename;
242
243 if (storage.Load(dbcFilename.c_str()))
244 {
245 bool found = false;
246 // Find table definition in GameTables.db2
247 for (uint32 i = 0; i < sGameTablesStore.GetNumRows(); ++i)
248 {
249 GameTablesEntry const* gt = sGameTablesStore.LookupEntry(i);
250 if (!gt)
251 continue;
252
253 for (uint32 l = 0; l < TOTAL_LOCALES; ++l)
254 {
255 if (l != LOCALE_none && tableName == gt->Name->Str[l])
256 {
257 found = true;
258 storage.SetGameTableEntry(gt);
259 break;
260 }
261 }
262
263 if (found)
264 break;
265 }
266
267 ASSERT(found, "Game table %s definition not found in GameTables.db2", tableName.c_str());
268 }
269 else
270 {
271 // sort problematic dbc to (1) non compatible and (2) non-existed
272 if (FILE* f = fopen(dbcFilename.c_str(), "rb"))
273 {
274 std::ostringstream stream;
275 stream << dbcFilename << " exists, and has " << storage.GetFieldCount() << " field(s) (expected " << strlen(storage.GetFormat()) << "). Extracted file might be from wrong client version or a database-update has been forgotten.";
276 std::string buf = stream.str();
277 errors.push_back(buf);
278 fclose(f);
279 }
280 else
281 errors.push_back(dbcFilename);
282 }
283}

455 }

475 exit(1);
476 }

479 if (!sAreaTableStore.LookupEntry(7941) || // last area added in 6.2.2 (20444)
480 !sCharTitlesStore.LookupEntry(457) || // last char title added in 6.2.2 (20444)
481 !sGemPropertiesStore.LookupEntry(2544) || // last gem property added in 6.2.2 (20444)
482 !sMapStore.LookupEntry(1497) || // last map added in 6.2.2 (20444)

484 {

486 exit(1);
487 }

489 TC_LOG_INFO("server.loading", ">> Initialized %d DBC data stores in %u ms", DBCFileCount, GetMSTimeDiffToNow(oldMSTime));

724{

726 if (!diffEntry)
727 return GetDefaultMapDifficulty(mapId, &difficulty);
728

731 while (!mapDiff)

734 diffEntry = sDifficultyStore.LookupEntry(tmpDiff);

736 return GetDefaultMapDifficulty(mapId, &difficulty);
737
738 // pull new data
739 mapDiff = GetMapDifficultyData(mapId, Difficulty(tmpDiff)); // we are 10 normal or 25 normal
740 }
741
742 difficulty = Difficulty(tmpDiff);
743 return mapDiff;

847void DeterminaAlternateMapPosition(uint32 mapId, float x, float y, float z, uint32* newMapId /*= nullptr*/, DBCPosition2D* newPos /*= nullptr*/)
848{
849 ASSERT(newMapId || newPos);
850 WorldMapTransformsEntry const* transformation = nullptr;
851 for (WorldMapTransformsEntry const* transform : sWorldMapTransformsStore)
852 {
853 if (transform->MapID != mapId)
854 continue;
855
856 if (transform->RegionMin.X > x || transform->RegionMax.X < x)
857 continue;
858 if (transform->RegionMin.Y > y || transform->RegionMax.Y < y)
859 continue;
860 if (transform->RegionMin.Z > z || transform->RegionMax.Z < z)
861 continue;
862
863 transformation = transform;
864 break;
865 }
866
867 if (!transformation)
868 {
869 if (newMapId)
870 *newMapId = mapId;
871
872 if (newPos)
873 {
874 newPos->X = x;
875 newPos->Y = y;
876 }
877 return;
878 }
879
880 if (newMapId)
881 *newMapId = transformation->NewMapID;
882
883 if (!newPos)
884 return;
885
886 if (transformation->RegionScale > 0.0f && transformation->RegionScale < 1.0f)
887 {
888 x = (x - transformation->RegionMin.X) * transformation->RegionScale + transformation->RegionMin.X;
889 y = (y - transformation->RegionMin.Y) * transformation->RegionScale + transformation->RegionMin.Y;
890 }
891
892 newPos->X = x + transformation->RegionOffset.X;
893 newPos->Y = y + transformation->RegionOffset.Y;
894}

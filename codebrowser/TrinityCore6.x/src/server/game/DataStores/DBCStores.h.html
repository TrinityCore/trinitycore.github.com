96template<class T>
97class GameTable
98{
99public:
100 GameTable(char const* format) : _storage(format), _gtEntry(nullptr) { }
101
102 void SetGameTableEntry(GameTablesEntry const* gtEntry) { _gtEntry = gtEntry; }
103
104 T const* EvaluateTable(uint32 row, uint32 column) const
105 {
106 ASSERT(row < _gtEntry->NumRows, "Requested row %u from GameTable %s but there are only %u rows!", row, _gtEntry->Name->Str[0], _gtEntry->NumRows);
107 ASSERT(column < _gtEntry->NumColumns, "Requested column %u from GameTable %s but there are only %u columns!", column, _gtEntry->Name->Str[0], _gtEntry->NumColumns);
108
109 return _storage.LookupEntry(_gtEntry->NumRows * column + row);
110 }
111
112 char const* GetFormat() const { return _storage.GetFormat(); }
113 uint32 GetFieldCount() const { return _storage.GetFieldCount(); }
114 bool Load(char const* fileName) { return _storage.Load(fileName, nullptr); }
115
116 uint32 GetTableRowCount() const { return _gtEntry->NumRows; }
117 uint32 GetTableColumnCount() const { return _gtEntry->NumColumns; }
118
119private:
120 DBCStorage<T> _storage;
121 GameTablesEntry const* _gtEntry;
122};

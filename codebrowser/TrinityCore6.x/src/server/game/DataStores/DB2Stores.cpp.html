141typedef std::list<std::string> DB2StoreProblemList;
142
143uint32 DB2FilesCount = 0;
144
145template<class T>
146inline void LoadDB2(uint32& availableDb2Locales, DB2StoreProblemList& errlist, DB2Manager::StorageMap& stores, DB2Storage<T>* storage, std::string const& db2Path, uint32 defaultLocale)
147{
148 // compatibility format and C++ structure sizes
149 ASSERT(DB2FileLoader::GetFormatRecordSize(storage->GetFormat()) == sizeof(T),
150 "Size of '%s' set by format string (%u) not equal size of C++ structure (" SZFMTD ").",
151 storage->GetFileName().c_str(), DB2FileLoader::GetFormatRecordSize(storage->GetFormat()), sizeof(T));
152
153 ++DB2FilesCount;
154
155 if (storage->Load(db2Path + localeNames[defaultLocale] + '/', defaultLocale))
156 {
157 storage->LoadFromDB();
158
159 for (uint32 i = 0; i < TOTAL_LOCALES; ++i)
160 {
161 if (defaultLocale == i || i == LOCALE_none)
162 continue;
163
164 if (availableDb2Locales & (1 << i))
165 if (!storage->LoadStringsFrom((db2Path + localeNames[i] + '/'), i))
166 availableDb2Locales &= ~(1 << i); // mark as not available for speedup next checks
167
168 storage->LoadStringsFromDB(i);
169 }
170 }
171 else
172 {
173 // sort problematic db2 to (1) non compatible and (2) nonexistent
174 if (FILE* f = fopen((db2Path + storage->GetFileName()).c_str(), "rb"))
175 {
176 std::ostringstream stream;
177 stream << storage->GetFileName() << " exists, and has " << storage->GetFieldCount() << " field(s) (expected " << strlen(storage->GetFormat())
178 << "). Extracted file might be from wrong client version.";
179 std::string buf = stream.str();
180 errlist.push_back(buf);
181 fclose(f);
182 }
183 else
184 errlist.push_back(storage->GetFileName());
185 }
186
187 stores[storage->GetHash()] = storage;
188}

360 for (ScalingStatDistributionEntry const* ssd : sScalingStatDistributionStore)

362
363 for (CurvePointEntry const* curvePoint : sCurvePointStore)
364 if (scalingCurves.count(curvePoint->CurveID))
365 _heirloomCurvePoints[curvePoint->CurveID][curvePoint->Index] = curvePoint;
366 }
367
368 for (MountEntry const* mount : sMountStore)
369 _mountsBySpellId[mount->SpellId] = mount;

371 for (MountTypeXCapabilityEntry const* mount : sMountTypeXCapabilityStore)
372 _mountCapabilitiesByType[mount->MountTypeID].insert(mount);
373
374 for (NameGenEntry const* entry : sNameGenStore)
375 _nameGenData[entry->Race][entry->Sex].push_back(entry);
376
377 for (NamesProfanityEntry const* namesProfanity : sNamesProfanityStore)
378 {
379 ASSERT(namesProfanity->Language < TOTAL_LOCALES || namesProfanity->Language == -1);
380 if (namesProfanity->Language != -1)
381 _nameValidators[namesProfanity->Language].emplace_back(namesProfanity->Name, boost::regex::perl | boost::regex::icase | boost::regex::optimize);
382 else
383 {
384 for (uint32 i = 0; i < TOTAL_LOCALES; ++i)
385 {
386 if (i == LOCALE_none)
387 continue;
388
389 _nameValidators[i].emplace_back(namesProfanity->Name, boost::regex::perl | boost::regex::icase | boost::regex::optimize);
390 }
391 }
392 }
393
394 for (NamesReservedEntry const* namesReserved : sNamesReservedStore)
395 _nameValidators[TOTAL_LOCALES].emplace_back(namesReserved->Name, boost::regex::perl | boost::regex::icase | boost::regex::optimize);
396
397 for (NamesReservedLocaleEntry const* namesReserved : sNamesReservedLocaleStore)
398 {
399 ASSERT(!(namesReserved->LocaleMask & ~((1 << TOTAL_LOCALES) - 1)));
400 for (uint32 i = 0; i < TOTAL_LOCALES; ++i)
401 {
402 if (i == LOCALE_none)
403 continue;
404
405 if (namesReserved->LocaleMask & (1 << i))
406 _nameValidators[i].emplace_back(namesReserved->Name, boost::regex::perl | boost::regex::icase | boost::regex::optimize);
407 }
408 }

411 if (PhaseEntry const* phase = sPhaseStore.LookupEntry(group->PhaseID))
412 _phasesByGroup[group->PhaseGroupID].insert(phase->ID);

423 {

425 if (powers.size() <= powerDifficulty->PowerIndex)
426 powers.resize(powerDifficulty->PowerIndex + 1);

429 }

432 std::vector<SpellPowerEntry const*>& powers = _spellPowers[power->SpellID];
433 if (powers.size() <= power->PowerIndex)
434 powers.resize(power->PowerIndex + 1);
435
436 powers[power->PowerIndex] = power;
437 }

440 for (TaxiPathEntry const* entry : sTaxiPathStore)
441 sTaxiPathSetBySource[entry->From][entry->To] = TaxiPathBySourceAndDestination(entry->ID, entry->Cost);
442
443 uint32 pathCount = sTaxiPathStore.GetNumRows();
444
445 // Calculate path nodes count
446 std::vector<uint32> pathLength;
447 pathLength.resize(pathCount); // 0 and some other indexes not used
448 for (TaxiPathNodeEntry const* entry : sTaxiPathNodeStore)
449 if (pathLength[entry->PathID] < entry->NodeIndex + 1)
450 pathLength[entry->PathID] = entry->NodeIndex + 1;
451
452 // Set path length
453 sTaxiPathNodesByPath.resize(pathCount); // 0 and some other indexes not used
454 for (uint32 i = 0; i < sTaxiPathNodesByPath.size(); ++i)
455 sTaxiPathNodesByPath[i].resize(pathLength[i]);
456
457 // fill data
458 for (TaxiPathNodeEntry const* entry : sTaxiPathNodeStore)
459 sTaxiPathNodesByPath[entry->PathID][entry->NodeIndex] = entry;
460
461 // Initialize global taxinodes mask
462 // include existed nodes that have at least single not spell base (scripted) path
463 {
464 if (sTaxiNodesStore.GetNumRows())
465 {
466 ASSERT(TaxiMaskSize >= ((sTaxiNodesStore.GetNumRows() - 1) / 8) + 1,
467 "TaxiMaskSize is not large enough to contain all taxi nodes! (current value %d, required %d)",
468 TaxiMaskSize, (((sTaxiNodesStore.GetNumRows() - 1) / 8) + 1));
469 }
470
471 sTaxiNodesMask.fill(0);
472 sOldContinentsNodesMask.fill(0);
473 sHordeTaxiNodesMask.fill(0);
474 sAllianceTaxiNodesMask.fill(0);
475 for (TaxiNodesEntry const* node : sTaxiNodesStore)
476 {
477 if (!(node->Flags & (TAXI_NODE_FLAG_ALLIANCE | TAXI_NODE_FLAG_HORDE)))
478 continue;
479
480 // valid taxi network node
481 uint8 field = (uint8)((node->ID - 1) / 8);
482 uint32 submask = 1 << ((node->ID - 1) % 8);
483
484 sTaxiNodesMask[field] |= submask;
485 if (node->Flags & TAXI_NODE_FLAG_HORDE)
486 sHordeTaxiNodesMask[field] |= submask;
487 if (node->Flags & TAXI_NODE_FLAG_ALLIANCE)
488 sAllianceTaxiNodesMask[field] |= submask;
489
490 uint32 nodeMap;
491 DeterminaAlternateMapPosition(node->MapID, node->Pos.X, node->Pos.Y, node->Pos.Z, &nodeMap);
492 if (nodeMap < 2)
493 sOldContinentsNodesMask[field] |= submask;
494 }
495 }
496
497 for (TransportAnimationEntry const* anim : sTransportAnimationStore)
498 sTransportMgr->AddPathNodeToTransport(anim->TransportID, anim->TimeIndex, anim);
499
500 for (TransportRotationEntry const* rot : sTransportRotationStore)
501 sTransportMgr->AddPathRotationToTransport(rot->TransportID, rot->TimeIndex, rot);
502
503 for (ToyEntry const* toy : sToyStore)
504 _toys.insert(toy->ItemID);
505
506 for (HeirloomEntry const* heirloom : sHeirloomStore)
507 _heirlooms[heirloom->ItemID] = heirloom;
508
509 // error checks
510 if (bad_db2_files.size() >= DB2FilesCount)
511 {
512 TC_LOG_ERROR("misc", "\nIncorrect DataDir value in worldserver.conf or ALL required *.db2 files (%d) not found by path: %sdbc/%s/", DB2FilesCount, dataPath.c_str(), localeNames[defaultLocale]);
513 exit(1);
514 }
515 else if (!bad_db2_files.empty())
516 {
517 std::string str;
518 for (std::list<std::string>::iterator i = bad_db2_files.begin(); i != bad_db2_files.end(); ++i)
519 str += *i + "\n";
520
521 TC_LOG_ERROR("misc", "\nSome required *.db2 files (%u from %d) not found or not compatible:\n%s", (uint32)bad_db2_files.size(), DB2FilesCount, str.c_str());
522 exit(1);
523 }
524
525 // Check loaded DB2 files proper version
526 if (!sItemStore.LookupEntry(128706) || // last item added in 6.2.0 (20216)
527 !sItemExtendedCostStore.LookupEntry(5923) ) // last item extended cost added in 6.2.0 (20216)
528 {
529 TC_LOG_ERROR("misc", "You have _outdated_ DB2 files. Please extract correct versions from current using client.");
530 exit(1);
531 }
532
533 TC_LOG_INFO("server.loading", ">> Initialized %d DB2 data stores in %u ms", DB2FilesCount, GetMSTimeDiffToNow(oldMSTime));
534}

545void DB2Manager::LoadHotfixData()
546{
547 uint32 oldMSTime = getMSTime();
548
549 QueryResult result = HotfixDatabase.Query("SELECT TableHash, RecordID, `Timestamp`, Deleted FROM hotfix_data");
550
551 if (!result)
552 {
553 TC_LOG_INFO("misc", ">> Loaded 0 hotfix info entries.");
554 return;
555 }
556
557 uint32 count = 0;
558
559 _hotfixData.reserve(result->GetRowCount());
560
561 do
562 {
563 Field* fields = result->Fetch();
564
565 HotfixNotify info;
566 info.TableHash = fields[0].GetUInt32();
567 info.Entry = fields[1].GetUInt32();
568 info.Timestamp = fields[2].GetUInt32();
569 _hotfixData.push_back(info);
570
571 if (fields[3].GetBool())
572 {
573 auto itr = _stores.find(info.TableHash);
574 if (itr != _stores.end())
575 itr->second->EraseRecord(info.Entry);
576 }
577
578 ++count;
579 } while (result->NextRow());
580
581 TC_LOG_INFO("misc", ">> Loaded %u hotfix info entries in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
582}

634uint32 DB2Manager::GetHeirloomItemLevel(uint32 curveId, uint32 level) const
635{
636 // Assuming linear item level scaling for heirlooms
637 auto itr = _heirloomCurvePoints.find(curveId);
638 if (itr == _heirloomCurvePoints.end())
639 return 0;
640
641 auto it2 = itr->second.begin(); // Highest scaling point
642 if (level >= it2->second->X)
643 return it2->second->Y;
644
645 auto previousItr = it2++;
646 for (; it2 != itr->second.end(); ++it2, ++previousItr)
647 if (level >= it2->second->X)
648 return uint32((previousItr->second->Y - it2->second->Y) / (previousItr->second->X - it2->second->X) * (float(level) - it2->second->X) + it2->second->Y);
649
650 return uint32(previousItr->second->Y); // Lowest scaling point
651}

808std::vector<SpellPowerEntry const*> DB2Manager::GetSpellPowers(uint32 spellId, Difficulty difficulty /*= DIFFICULTY_NONE*/, bool* hasDifficultyPowers /*= nullptr*/) const
809{
810 std::vector<SpellPowerEntry const*> powers;
811
812 auto difficultyItr = _spellPowerDifficulties.find(spellId);
813 if (difficultyItr != _spellPowerDifficulties.end())
814 {
815 if (hasDifficultyPowers)
816 *hasDifficultyPowers = true;
817
818 DifficultyEntry const* difficultyEntry = sDifficultyStore.LookupEntry(difficulty);
819 while (difficultyEntry)
820 {
821 auto powerDifficultyItr = difficultyItr->second.find(difficultyEntry->ID);
822 if (powerDifficultyItr != difficultyItr->second.end())
823 {
824 if (powerDifficultyItr->second.size() > powers.size())
825 powers.resize(powerDifficultyItr->second.size());
826
827 for (SpellPowerEntry const* difficultyPower : powerDifficultyItr->second)
828 if (!powers[difficultyPower->PowerIndex])
829 powers[difficultyPower->PowerIndex] = difficultyPower;
830 }
831
832 difficultyEntry = sDifficultyStore.LookupEntry(difficultyEntry->FallbackDifficultyID);
833 }
834 }
835
836 auto itr = _spellPowers.find(spellId);
837 if (itr != _spellPowers.end())
838 {
839 if (itr->second.size() > powers.size())
840 powers.resize(itr->second.size());
841
842 for (SpellPowerEntry const* power : itr->second)
843 if (!powers[power->PowerIndex])
844 powers[power->PowerIndex] = power;
845 }
846
847 return powers;
848}

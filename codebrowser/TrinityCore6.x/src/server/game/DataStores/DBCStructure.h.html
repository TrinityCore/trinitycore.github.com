424struct FactionTemplateEntry
425{
426 uint32 ID; // 0
427 uint32 Faction; // 1
428 uint32 Flags; // 2
429 uint32 Mask; // 3 m_factionGroup
430 uint32 FriendMask; // 4 m_friendGroup
431 uint32 EnemyMask; // 5 m_enemyGroup
432 uint32 Enemies[MAX_FACTION_RELATIONS]; // 6
433 uint32 Friends[MAX_FACTION_RELATIONS]; // 10
434 //------------------------------------------------------- end structure
435
436 // helpers
437 bool IsFriendlyTo(FactionTemplateEntry const& entry) const
438 {
439 if (ID == entry.ID)
440 return true;
441 if (entry.Faction)
442 {
443 for (int i = 0; i < MAX_FACTION_RELATIONS; ++i)
444 if (Enemies[i] == entry.Faction)
445 return false;
446 for (int i = 0; i < MAX_FACTION_RELATIONS; ++i)
447 if (Friends[i] == entry.Faction)
448 return true;
449 }
450 return (FriendMask & entry.Mask) || (Mask & entry.FriendMask);
451 }
452 bool IsHostileTo(FactionTemplateEntry const& entry) const
453 {
454 if (ID == entry.ID)
455 return false;
456 if (entry.Faction)
457 {
458 for (int i = 0; i < MAX_FACTION_RELATIONS; ++i)
459 if (Enemies[i] == entry.Faction)
460 return true;
461 for (int i = 0; i < MAX_FACTION_RELATIONS; ++i)
462 if (Friends[i] == entry.Faction)
463 return false;
464 }
465 return (EnemyMask & entry.Mask) != 0;
466 }
467 bool IsHostileToPlayers() const { return (EnemyMask & FACTION_MASK_PLAYER) !=0; }
468 bool IsNeutralToAll() const
469 {
470 for (int i = 0; i < MAX_FACTION_RELATIONS; ++i)
471 if (Enemies[i] != 0)
472 return false;
473 return EnemyMask == 0 && FriendMask == 0;
474 }
475 bool IsContestedGuardFaction() const { return (Flags & FACTION_TEMPLATE_FLAG_CONTESTED_GUARD) != 0; }
476};

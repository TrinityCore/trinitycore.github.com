28void BattlegroundBE::StartingEventCloseDoors()
29{
30 for (uint32 i = BG_BE_OBJECT_DOOR_1; i <= BG_BE_OBJECT_DOOR_4; ++i)
31 SpawnBGObject(i, RESPAWN_IMMEDIATELY);
32
33 for (uint32 i = BG_BE_OBJECT_BUFF_1; i <= BG_BE_OBJECT_BUFF_2; ++i)
34 SpawnBGObject(i, RESPAWN_ONE_DAY);
35}
36
37void BattlegroundBE::StartingEventOpenDoors()
38{
39 for (uint32 i = BG_BE_OBJECT_DOOR_1; i <= BG_BE_OBJECT_DOOR_2; ++i)
40 DoorOpen(i);
41
42 for (uint32 i = BG_BE_OBJECT_BUFF_1; i <= BG_BE_OBJECT_BUFF_2; ++i)
43 SpawnBGObject(i, 60);
44}

46void BattlegroundBE::HandleAreaTrigger(Player* player, uint32 trigger, bool entered)
47{
48 if (GetStatus() != STATUS_IN_PROGRESS)
49 return;
50
51 switch (trigger)
52 {
53 case 4538: // buff trigger?
54 case 4539: // buff trigger?
55 break;
56 default:
57 Battleground::HandleAreaTrigger(player, trigger, entered);
58 break;
59 }
60}

68bool BattlegroundBE::SetupBattleground()
69{
70 // gates
71 if (!AddObject(BG_BE_OBJECT_DOOR_1, BG_BE_OBJECT_TYPE_DOOR_1, 6287.277f, 282.1877f, 3.810925f, -2.260201f, 0, 0, 0.9044551f, -0.4265689f, RESPAWN_IMMEDIATELY)
72 || !AddObject(BG_BE_OBJECT_DOOR_2, BG_BE_OBJECT_TYPE_DOOR_2, 6189.546f, 241.7099f, 3.101481f, 0.8813917f, 0, 0, 0.4265689f, 0.9044551f, RESPAWN_IMMEDIATELY)
73 || !AddObject(BG_BE_OBJECT_DOOR_3, BG_BE_OBJECT_TYPE_DOOR_3, 6299.116f, 296.5494f, 3.308032f, 0.8813917f, 0, 0, 0.4265689f, 0.9044551f, RESPAWN_IMMEDIATELY)
74 || !AddObject(BG_BE_OBJECT_DOOR_4, BG_BE_OBJECT_TYPE_DOOR_4, 6177.708f, 227.3481f, 3.604374f, -2.260201f, 0, 0, 0.9044551f, -0.4265689f, RESPAWN_IMMEDIATELY)
75 // buffs
76 || !AddObject(BG_BE_OBJECT_BUFF_1, BG_BE_OBJECT_TYPE_BUFF_1, 6249.042f, 275.3239f, 11.22033f, -1.448624f, 0, 0, 0.6626201f, -0.7489557f, 120)
77 || !AddObject(BG_BE_OBJECT_BUFF_2, BG_BE_OBJECT_TYPE_BUFF_2, 6228.26f, 249.566f, 11.21812f, -0.06981307f, 0, 0, 0.03489945f, -0.9993908f, 120))
78 {
79 TC_LOG_ERROR("sql.sql", "BatteGroundBE: Failed to spawn some object!");
80 return false;
81 }
82
83 return true;
84}

34void BattlegroundRV::PostUpdateImpl(uint32 diff)
35{
36 if (GetStatus() != STATUS_IN_PROGRESS)
37 return;
38
39 if (_timer < diff)
40 {
41 switch (_state)
42 {
43 case BG_RV_STATE_OPEN_FENCES:
44 // Open fire (only at game start)
45 for (uint8 i = BG_RV_OBJECT_FIRE_1; i <= BG_RV_OBJECT_FIREDOOR_2; ++i)
46 DoorOpen(i);
47 _timer = BG_RV_CLOSE_FIRE_TIMER;
48 _state = BG_RV_STATE_CLOSE_FIRE;
49 break;
50 case BG_RV_STATE_CLOSE_FIRE:
51 for (uint8 i = BG_RV_OBJECT_FIRE_1; i <= BG_RV_OBJECT_FIREDOOR_2; ++i)
52 DoorClose(i);
53 // Fire got closed after five seconds, leaves twenty seconds before toggling pillars
54 _timer = BG_RV_FIRE_TO_PILLAR_TIMER;
55 _state = BG_RV_STATE_SWITCH_PILLARS;
56 break;
57 case BG_RV_STATE_SWITCH_PILLARS:
58 TogglePillarCollision();
59 _timer = BG_RV_PILLAR_SWITCH_TIMER;
60 break;
61 }
62 }
63 else
64 _timer -= diff;
65}

67void BattlegroundRV::StartingEventOpenDoors()
68{
69 // Buff respawn
70 SpawnBGObject(BG_RV_OBJECT_BUFF_1, 90);
71 SpawnBGObject(BG_RV_OBJECT_BUFF_2, 90);
72 // Elevators
73 DoorOpen(BG_RV_OBJECT_ELEVATOR_1);
74 DoorOpen(BG_RV_OBJECT_ELEVATOR_2);
75
76 _state = BG_RV_STATE_OPEN_FENCES;
77 _timer = BG_RV_FIRST_TIMER;
78
79 // Should be false at first, TogglePillarCollision will do it.
80 _pillarCollision = true;
81 TogglePillarCollision();
82}

85{
86 if (GetStatus() != STATUS_IN_PROGRESS)
87 return;
88
89 switch (trigger)
90 {
91 case 5224:
92 case 5226:

94 case 5473:
95 case 5474:
96 break;
97 default:
98 Battleground::HandleAreaTrigger(player, trigger, entered);
99 break;
100 }

146void BattlegroundRV::TogglePillarCollision()
147{
148 // Toggle visual pillars, pulley, gear, and collision based on previous state
149 for (uint8 i = BG_RV_OBJECT_PILAR_1; i <= BG_RV_OBJECT_GEAR_2; ++i)
150 _pillarCollision ? DoorOpen(i) : DoorClose(i);
151
152 for (uint8 i = BG_RV_OBJECT_PILAR_2; i <= BG_RV_OBJECT_PULLEY_2; ++i)
153 _pillarCollision ? DoorClose(i) : DoorOpen(i);
154
155 for (uint8 i = BG_RV_OBJECT_PILAR_1; i <= BG_RV_OBJECT_PILAR_COLLISION_4; ++i)
156 {
157 if (GameObject* go = GetBGObject(i))
158 {
159 if (i >= BG_RV_OBJECT_PILAR_COLLISION_1)
160 {
161 GOState state = ((go->GetGOInfo()->door.startOpen != 0) == _pillarCollision) ? GO_STATE_ACTIVE : GO_STATE_READY;
162 go->SetGoState(state);
163 }
164
165 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
166 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
167 go->SendUpdateToPlayer(player);
168 }
169 }
170
171 _pillarCollision = !_pillarCollision;
172}

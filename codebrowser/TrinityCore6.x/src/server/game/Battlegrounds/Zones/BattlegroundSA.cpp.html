27BattlegroundSA::BattlegroundSA()
28{
29 StartMessageIds[BG_STARTING_EVENT_FIRST] = LANG_BG_SA_START_TWO_MINUTES;
30 StartMessageIds[BG_STARTING_EVENT_SECOND] = LANG_BG_SA_START_ONE_MINUTE;
31 StartMessageIds[BG_STARTING_EVENT_THIRD] = LANG_BG_SA_START_HALF_MINUTE;
32 StartMessageIds[BG_STARTING_EVENT_FOURTH] = 0;
33
34 BgObjects.resize(BG_SA_MAXOBJ);
35 BgCreatures.resize(BG_SA_MAXNPC + BG_SA_MAX_GY);
36 TimerEnabled = false;
37 UpdateWaitTimer = 0;
38 SignaledRoundTwo = false;
39 SignaledRoundTwoHalfMin = false;
40 InitSecondRound = false;
41 _gateDestroyed = false;
42 Attackers = TEAM_ALLIANCE;
43 TotalTime = 0;
44 EndRoundTimer = 0;
45 ShipsStarted = false;
46 Status = BG_SA_NOT_STARTED;
47
48 for (uint8 i = 0; i < MAX_GATES; ++i)
49 GateStatus[i] = BG_SA_GATE_OK;
50
51 for (uint8 i = 0; i < 2; i++)
52 {
53 RoundScores[i].winner = TEAM_ALLIANCE;
54 RoundScores[i].time = 0;
55 _allVehiclesAlive[i] = true;
56 }
57
58 //! This is here to prevent an uninitialised variable warning
59 //! The warning only occurs when SetUpBattleGround fails though.
60 //! In the future this function should be called BEFORE sending initial worldstates.
61 memset(&GraveyardStatus, 0, sizeof(GraveyardStatus));
62}

66void BattlegroundSA::Reset()
67{
68 TotalTime = 0;
69 Attackers = ((urand(0, 1)) ? TEAM_ALLIANCE : TEAM_HORDE);
70 for (uint8 i = 0; i <= 5; i++)
71 GateStatus[i] = BG_SA_GATE_OK;
72 ShipsStarted = false;
73 _gateDestroyed = false;
74 _allVehiclesAlive[TEAM_ALLIANCE] = true;
75 _allVehiclesAlive[TEAM_HORDE] = true;
76 Status = BG_SA_WARMUP;
77}

84bool BattlegroundSA::ResetObjs()
85{
86 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
87 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
88 SendTransportsRemove(player);
89
90 uint32 atF = BG_SA_Factions[Attackers];
91 uint32 defF = BG_SA_Factions[Attackers ? TEAM_ALLIANCE : TEAM_HORDE];
92
93 for (uint8 i = 0; i < BG_SA_MAXOBJ; i++)
94 DelObject(i);
95
96 for (uint8 i = 0; i < BG_SA_MAXNPC; i++)
97 DelCreature(i);
98
99 for (uint8 i = BG_SA_MAXNPC; i < BG_SA_MAXNPC + BG_SA_MAX_GY; i++)
100 DelCreature(i);
101
102 for (uint8 i = 0; i < MAX_GATES; ++i)
103 GateStatus[i] = BG_SA_GATE_OK;
104
105 if (!AddCreature(BG_SA_NpcEntries[BG_SA_NPC_KANRETHAD], BG_SA_NPC_KANRETHAD, BG_SA_NpcSpawnlocs[BG_SA_NPC_KANRETHAD]))
106 {
107 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn Kanrethad, aborted. Entry: %u", BG_SA_NpcEntries[BG_SA_NPC_KANRETHAD]);
108 return false;
109 }
110
111 for (uint8 i = 0; i <= BG_SA_PORTAL_DEFFENDER_RED; i++)
112 {
113 if (!AddObject(i, BG_SA_ObjEntries[i], BG_SA_ObjSpawnlocs[i], 0, 0, 0, 0, RESPAWN_ONE_DAY))
114 {
115 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn BG_SA_PORTAL_DEFFENDER_RED, Entry: %u", BG_SA_ObjEntries[i]);
116 continue;
117 }
118 }
119
120 for (uint8 i = BG_SA_BOAT_ONE; i <= BG_SA_BOAT_TWO; i++)
121 {
122 uint32 boatid = 0;
123 switch (i)
124 {
125 case BG_SA_BOAT_ONE:
126 boatid = Attackers ? BG_SA_BOAT_ONE_H : BG_SA_BOAT_ONE_A;
127 break;
128 case BG_SA_BOAT_TWO:
129 boatid = Attackers ? BG_SA_BOAT_TWO_H : BG_SA_BOAT_TWO_A;
130 break;
131 default:
132 break;
133 }
134 if (!AddObject(i, boatid, BG_SA_ObjSpawnlocs[i].GetPositionX(),
135 BG_SA_ObjSpawnlocs[i].GetPositionY(),
136 BG_SA_ObjSpawnlocs[i].GetPositionZ() + (Attackers ? -3.750f: 0),
137 BG_SA_ObjSpawnlocs[i].GetOrientation(), 0, 0, 0, 0, RESPAWN_ONE_DAY))
138 {
139 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn one of the BG_SA_BOAT, Entry: %u", boatid);
140 continue;
141 }
142 }
143
144 for (uint8 i = BG_SA_SIGIL_1; i <= BG_SA_LEFT_FLAGPOLE; i++)
145 {
146 if (!AddObject(i, BG_SA_ObjEntries[i], BG_SA_ObjSpawnlocs[i], 0, 0, 0, 0, RESPAWN_ONE_DAY))
147 {
148 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn Sigil, Entry: %u", BG_SA_ObjEntries[i]);
149 continue;
150 }
151 }
152
153 // MAD props for Kiper for discovering those values - 4 hours of his work.
154 GetBGObject(BG_SA_BOAT_ONE)->UpdateRotationFields(1.0f, 0.0002f);
155 GetBGObject(BG_SA_BOAT_TWO)->UpdateRotationFields(1.0f, 0.00001f);
156 SpawnBGObject(BG_SA_BOAT_ONE, RESPAWN_IMMEDIATELY);
157 SpawnBGObject(BG_SA_BOAT_TWO, RESPAWN_IMMEDIATELY);
158
159 //Cannons and demolishers - NPCs are spawned
160 //By capturing GYs.
161 for (uint8 i = 0; i < BG_SA_DEMOLISHER_5; i++)
162 {
163 if (!AddCreature(BG_SA_NpcEntries[i], i, BG_SA_NpcSpawnlocs[i], Attackers == TEAM_ALLIANCE ? TEAM_HORDE : TEAM_ALLIANCE, 600))
164 {
165 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn Cannon or demolisher, Entry: %u, Attackers: %s", BG_SA_NpcEntries[i], Attackers == TEAM_ALLIANCE ? "Horde(1)" : "Alliance(0)");
166 continue;
167 }
168 }
169
170 OverrideGunFaction();
171 DemolisherStartState(true);
172
173 for (uint8 i = 0; i <= BG_SA_PORTAL_DEFFENDER_RED; i++)
174 {
175 SpawnBGObject(i, RESPAWN_IMMEDIATELY);
176 GetBGObject(i)->SetFaction(defF);
177 }
178
179 GetBGObject(BG_SA_TITAN_RELIC)->SetFaction(atF);
180 GetBGObject(BG_SA_TITAN_RELIC)->Refresh();
181
182 TotalTime = 0;
183 ShipsStarted = false;
184
185 //Graveyards
186 for (uint8 i = 0; i < BG_SA_MAX_GY; i++)
187 {
188 WorldSafeLocsEntry const* sg = NULL;
189 sg = sWorldSafeLocsStore.LookupEntry(BG_SA_GYEntries[i]);
190
191 if (!sg)
192 {
193 TC_LOG_ERROR("bg.battleground", "SOTA: Can't find GY entry %u", BG_SA_GYEntries[i]);
194 return false;
195 }
196
197 if (i == BG_SA_BEACH_GY)
198 {
199 GraveyardStatus[i] = Attackers;
200 AddSpiritGuide(i + BG_SA_MAXNPC, sg->Loc.X, sg->Loc.Y, sg->Loc.Z, BG_SA_GYOrientation[i], Attackers);
201 }
202 else
203 {
204 GraveyardStatus[i] = ((Attackers == TEAM_HORDE)? TEAM_ALLIANCE : TEAM_HORDE);
205 if (!AddSpiritGuide(i + BG_SA_MAXNPC, sg->Loc.X, sg->Loc.Y, sg->Loc.Z, BG_SA_GYOrientation[i], Attackers == TEAM_HORDE ? TEAM_ALLIANCE : TEAM_HORDE))
206 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn GY: %u", i);
207 }
208 }
209
210 //GY capture points
211 for (uint8 i = BG_SA_CENTRAL_FLAG; i <= BG_SA_LEFT_FLAG; i++)
212 {
213 if (!AddObject(i, (BG_SA_ObjEntries[i] - (Attackers == TEAM_ALLIANCE ? 1 : 0)), BG_SA_ObjSpawnlocs[i], 0, 0, 0, 0, RESPAWN_ONE_DAY))
214 {
215 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn Central Flag Entry: %u", BG_SA_ObjEntries[i] - (Attackers == TEAM_ALLIANCE ? 1 : 0));
216 continue;
217 }
218 GetBGObject(i)->SetFaction(atF);
219 }
220
221 UpdateObjectInteractionFlags();
222
223 for (uint8 i = BG_SA_BOMB; i < BG_SA_MAXOBJ; i++)
224 {
225 if (!AddObject(i, BG_SA_ObjEntries[BG_SA_BOMB], BG_SA_ObjSpawnlocs[i], 0, 0, 0, 0, RESPAWN_ONE_DAY))
226 {
227 TC_LOG_ERROR("bg.battleground", "SOTA: couldn't spawn SA Bomb Entry: %u", BG_SA_ObjEntries[BG_SA_BOMB] + i);
228 continue;
229 }
230 GetBGObject(i)->SetFaction(atF);
231 }
232
233 //Player may enter BEFORE we set up BG - lets update his worldstates anyway...
234 UpdateWorldState(BG_SA_RIGHT_GY_HORDE, GraveyardStatus[BG_SA_RIGHT_CAPTURABLE_GY] == TEAM_HORDE ? 1 : 0);
235 UpdateWorldState(BG_SA_LEFT_GY_HORDE, GraveyardStatus[BG_SA_LEFT_CAPTURABLE_GY] == TEAM_HORDE ? 1 : 0);
236 UpdateWorldState(BG_SA_CENTER_GY_HORDE, GraveyardStatus[BG_SA_CENTRAL_CAPTURABLE_GY] == TEAM_HORDE ? 1 : 0);
237
238 UpdateWorldState(BG_SA_RIGHT_GY_ALLIANCE, GraveyardStatus[BG_SA_RIGHT_CAPTURABLE_GY] == TEAM_ALLIANCE ? 1 : 0);
239 UpdateWorldState(BG_SA_LEFT_GY_ALLIANCE, GraveyardStatus[BG_SA_LEFT_CAPTURABLE_GY] == TEAM_ALLIANCE ? 1 : 0);
240 UpdateWorldState(BG_SA_CENTER_GY_ALLIANCE, GraveyardStatus[BG_SA_CENTRAL_CAPTURABLE_GY] == TEAM_ALLIANCE ? 1 : 0);
241
242 if (Attackers == TEAM_ALLIANCE)
243 {
244 UpdateWorldState(BG_SA_ALLY_ATTACKS, 1);
245 UpdateWorldState(BG_SA_HORDE_ATTACKS, 0);
246
247 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_ALL, 1);
248 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_ALL, 1);
249 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_HRD, 0);
250 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_HRD, 0);
251
252 UpdateWorldState(BG_SA_HORDE_DEFENCE_TOKEN, 1);
253 UpdateWorldState(BG_SA_ALLIANCE_DEFENCE_TOKEN, 0);
254 }
255 else
256 {
257 UpdateWorldState(BG_SA_HORDE_ATTACKS, 1);
258 UpdateWorldState(BG_SA_ALLY_ATTACKS, 0);
259
260 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_ALL, 0);
261 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_ALL, 0);
262 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_HRD, 1);
263 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_HRD, 1);
264
265 UpdateWorldState(BG_SA_HORDE_DEFENCE_TOKEN, 0);
266 UpdateWorldState(BG_SA_ALLIANCE_DEFENCE_TOKEN, 1);
267 }
268
269 UpdateWorldState(BG_SA_PURPLE_GATEWS, 1);
270 UpdateWorldState(BG_SA_RED_GATEWS, 1);
271 UpdateWorldState(BG_SA_BLUE_GATEWS, 1);
272 UpdateWorldState(BG_SA_GREEN_GATEWS, 1);
273 UpdateWorldState(BG_SA_YELLOW_GATEWS, 1);
274 UpdateWorldState(BG_SA_ANCIENT_GATEWS, 1);
275
276 for (int i = BG_SA_BOAT_ONE; i <= BG_SA_BOAT_TWO; i++)
277 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
278 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
279 SendTransportInit(player);
280
281 // set status manually so preparation is cast correctly in 2nd round too
282 SetStatus(STATUS_WAIT_JOIN);
283
284 TeleportPlayers();
285 return true;
286}

288void BattlegroundSA::StartShips()
289{
290 if (ShipsStarted)
291 return;
292
293 DoorOpen(BG_SA_BOAT_ONE);
294 DoorOpen(BG_SA_BOAT_TWO);
295
296 for (int i = BG_SA_BOAT_ONE; i <= BG_SA_BOAT_TWO; i++)
297 {
298 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
299 {
300 if (Player* p = ObjectAccessor::FindPlayer(itr->first))
301 {
302 UpdateData data(p->GetMapId());
303 WorldPacket pkt;
304 GetBGObject(i)->BuildValuesUpdateBlockForPlayer(&data, p);
305 data.BuildPacket(&pkt);
306 p->SendDirectMessage(&pkt);
307 }
308 }
309 }
310 ShipsStarted = true;
311}

313void BattlegroundSA::PostUpdateImpl(uint32 diff)
314{
315 if (InitSecondRound)
316 {
317 if (UpdateWaitTimer < diff)
318 {
319 if (!SignaledRoundTwo)
320 {
321 SignaledRoundTwo = true;
322 InitSecondRound = false;
323 SendMessageToAll(LANG_BG_SA_ROUND_TWO_ONE_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
324 }
325 }
326 else
327 {
328 UpdateWaitTimer -= diff;
329 return;
330 }
331 }
332 TotalTime += diff;
333
334 if (Status == BG_SA_WARMUP)
335 {
336 EndRoundTimer = BG_SA_ROUNDLENGTH;
337 if (TotalTime >= BG_SA_WARMUPLENGTH)
338 {
339 if (Creature* c = GetBGCreature(BG_SA_NPC_KANRETHAD))
340 SendChatMessage(c, TEXT_ROUND_STARTED);
341
342 TotalTime = 0;
343 ToggleTimer();
344 DemolisherStartState(false);
345 Status = BG_SA_ROUND_ONE;
346 StartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, (Attackers == TEAM_ALLIANCE) ? 23748 : 21702);
347 }
348 if (TotalTime >= BG_SA_BOAT_START)
349 StartShips();
350 return;
351 }
352 else if (Status == BG_SA_SECOND_WARMUP)
353 {
354 if (RoundScores[0].time<BG_SA_ROUNDLENGTH)
355 EndRoundTimer = RoundScores[0].time;
356 else
357 EndRoundTimer = BG_SA_ROUNDLENGTH;
358
359 if (TotalTime >= 60000)
360 {
361 if (Creature* c = GetBGCreature(BG_SA_NPC_KANRETHAD))
362 SendChatMessage(c, TEXT_ROUND_STARTED);
363
364 TotalTime = 0;
365 ToggleTimer();
366 DemolisherStartState(false);
367 Status = BG_SA_ROUND_TWO;
368 StartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, (Attackers == TEAM_ALLIANCE) ? 23748 : 21702);
369 // status was set to STATUS_WAIT_JOIN manually for Preparation, set it back now
370 SetStatus(STATUS_IN_PROGRESS);
371 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
372 if (Player* p = ObjectAccessor::FindPlayer(itr->first))
373 p->RemoveAurasDueToSpell(SPELL_PREPARATION);
374 }
375 if (TotalTime >= 30000)
376 {
377 if (!SignaledRoundTwoHalfMin)
378 {
379 SignaledRoundTwoHalfMin = true;
380 SendMessageToAll(LANG_BG_SA_ROUND_TWO_START_HALF_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
381 }
382 }
383 StartShips();
384 return;
385 }
386 else if (GetStatus() == STATUS_IN_PROGRESS)
387 {
388 if (Status == BG_SA_ROUND_ONE)
389 {
390 if (TotalTime >= BG_SA_ROUNDLENGTH)
391 {
392 RoundScores[0].winner = Attackers;
393 RoundScores[0].time = BG_SA_ROUNDLENGTH;
394 TotalTime = 0;
395 Status = BG_SA_SECOND_WARMUP;
396 Attackers = (Attackers == TEAM_ALLIANCE) ? TEAM_HORDE : TEAM_ALLIANCE;
397 UpdateWaitTimer = 5000;
398 SignaledRoundTwo = false;
399 SignaledRoundTwoHalfMin = false;
400 InitSecondRound = true;
401 ToggleTimer();
402 ResetObjs();
403 GetBgMap()->UpdateAreaDependentAuras();
404 CastSpellOnTeam(SPELL_END_OF_ROUND, ALLIANCE);
405 CastSpellOnTeam(SPELL_END_OF_ROUND, HORDE);
406 return;
407 }
408 }
409 else if (Status == BG_SA_ROUND_TWO)
410 {
411 if (TotalTime >= EndRoundTimer)
412 {
413 RoundScores[1].time = BG_SA_ROUNDLENGTH;
414 RoundScores[1].winner = (Attackers == TEAM_ALLIANCE) ? TEAM_HORDE : TEAM_ALLIANCE;
415 if (RoundScores[0].time == RoundScores[1].time)
416 EndBattleground(0);
417 else if (RoundScores[0].time < RoundScores[1].time)
418 EndBattleground(RoundScores[0].winner == TEAM_ALLIANCE ? ALLIANCE : HORDE);
419 else
420 EndBattleground(RoundScores[1].winner == TEAM_ALLIANCE ? ALLIANCE : HORDE);
421 return;
422 }
423 }
424 if (Status == BG_SA_ROUND_ONE || Status == BG_SA_ROUND_TWO)
425 {
426 SendTime();
427 UpdateDemolisherSpawns();
428 }
429 }
430}

495void BattlegroundSA::TeleportPlayers()
496{
497 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
498 {
499 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
500 {
501 // should remove spirit of redemption
502 if (player->HasAuraType(SPELL_AURA_SPIRIT_OF_REDEMPTION))
503 player->RemoveAurasByType(SPELL_AURA_MOD_SHAPESHIFT);
504
505 if (!player->IsAlive())
506 {
507 player->ResurrectPlayer(1.0f);
508 player->SpawnCorpseBones();
509 }
510
511 player->ResetAllPowers();
512 player->CombatStopWithPets(true);
513
514 player->CastSpell(player, SPELL_PREPARATION, true);
515
516 TeleportToEntrancePosition(player);
517 }
518 }
519}

537 else
538 player->TeleportTo(607, 1209.7f, -65.16f, 70.1f, 0.0f, 0);
539}

541void BattlegroundSA::ProcessEvent(WorldObject* obj, uint32 eventId, WorldObject* invoker /*= NULL*/)
542{
543 if (GameObject* go = obj->ToGameObject())
544 {
545 switch (go->GetGoType())
546 {
547 case GAMEOBJECT_TYPE_GOOBER:
548 if (invoker)
549 if (eventId == BG_SA_EVENT_TITAN_RELIC_ACTIVATED)
550 TitanRelicActivated(invoker->ToPlayer());
551 break;
552 case GAMEOBJECT_TYPE_DESTRUCTIBLE_BUILDING:
553 {
554 if (GateInfo const* gate = GetGate(obj->GetEntry()))
555 {
556 uint8 gateId = gate->GateId;
557
558 // damaged
559 if (eventId == go->GetGOInfo()->destructibleBuilding.DamagedEvent)
560 {
561 GateStatus[gateId] = BG_SA_GATE_DAMAGED;
562
563 if (Creature* c = obj->FindNearestCreature(NPC_WORLD_TRIGGER, 500.0f))
564 SendChatMessage(c, gate->DamagedText, invoker);
565
566 PlaySoundToAll(Attackers == TEAM_ALLIANCE ? SOUND_WALL_ATTACKED_ALLIANCE : SOUND_WALL_ATTACKED_HORDE);
567 }
568 // destroyed
569 else if (eventId == go->GetGOInfo()->destructibleBuilding.DestroyedEvent)
570 {
571 GateStatus[gate->GateId] = BG_SA_GATE_DESTROYED;
572 _gateDestroyed = true;
573
574 if (gateId < 5)
575 DelObject(gateId + 14);
576
577 if (Creature* c = obj->FindNearestCreature(NPC_WORLD_TRIGGER, 500.0f))
578 SendChatMessage(c, gate->DestroyedText, invoker);
579
580 PlaySoundToAll(Attackers == TEAM_ALLIANCE ? SOUND_WALL_DESTROYED_ALLIANCE : SOUND_WALL_DESTROYED_HORDE);
581
582 bool rewardHonor = true;
583 switch (gateId)
584 {
585 case BG_SA_GREEN_GATE:
586 if (GateStatus[BG_SA_BLUE_GATE] == BG_SA_GATE_DESTROYED)
587 rewardHonor = false;
588 break;
589 case BG_SA_BLUE_GATE:
590 if (GateStatus[BG_SA_GREEN_GATE] == BG_SA_GATE_DESTROYED)
591 rewardHonor = false;
592 break;
593 case BG_SA_RED_GATE:
594 if (GateStatus[BG_SA_PURPLE_GATE] == BG_SA_GATE_DESTROYED)
595 rewardHonor = false;
596 break;
597 case BG_SA_PURPLE_GATE:
598 if (GateStatus[BG_SA_RED_GATE] == BG_SA_GATE_DESTROYED)
599 rewardHonor = false;
600 break;
601 default:
602 break;
603 }
604
605 if (invoker)
606 {
607 if (Unit* unit = invoker->ToUnit())
608 {
609 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
610 {
611 UpdatePlayerScore(player, SCORE_DESTROYED_WALL, 1);
612 if (rewardHonor)
613 UpdatePlayerScore(player, SCORE_BONUS_HONOR, GetBonusHonorFromKill(1));
614 }
615 }
616 }
617
618 UpdateObjectInteractionFlags();
619 }
620 else
621 break;
622
623 UpdateWorldState(gate->WorldState, GateStatus[gateId]);
624 }
625 break;
626 }
627 default:
628 break;
629 }
630 }

686WorldSafeLocsEntry const* BattlegroundSA::GetClosestGraveYard(Player* player)
687{
688 uint32 safeloc = 0;
689 WorldSafeLocsEntry const* ret;
690 WorldSafeLocsEntry const* closest;
691 float dist, nearest;
692 float x, y, z;
693
694 player->GetPosition(x, y, z);
695
696 if (player->GetTeamId() == Attackers)
697 safeloc = BG_SA_GYEntries[BG_SA_BEACH_GY];
698 else
699 safeloc = BG_SA_GYEntries[BG_SA_DEFENDER_LAST_GY];
700
701 closest = sWorldSafeLocsStore.LookupEntry(safeloc);
702 nearest = player->GetExactDistSq(closest->Loc.X, closest->Loc.Y, closest->Loc.Z);
703
704 for (uint8 i = BG_SA_RIGHT_CAPTURABLE_GY; i < BG_SA_MAX_GY; i++)
705 {
706 if (GraveyardStatus[i] != player->GetTeamId())
707 continue;
708
709 ret = sWorldSafeLocsStore.LookupEntry(BG_SA_GYEntries[i]);
710 dist = player->GetExactDistSq(ret->Loc.X, ret->Loc.Y, ret->Loc.Z);
711 if (dist < nearest)
712 {
713 closest = ret;
714 nearest = dist;
715 }
716 }
717
718 return closest;

796void BattlegroundSA::CaptureGraveyard(BG_SA_Graveyards i, Player* Source)
797{
798 if (GraveyardStatus[i] == Attackers)
799 return;
800
801 DelCreature(BG_SA_MAXNPC + i);
802 GraveyardStatus[i] = Source->GetTeamId();
803 WorldSafeLocsEntry const* sg = sWorldSafeLocsStore.LookupEntry(BG_SA_GYEntries[i]);
804 if (!sg)
805 {
806 TC_LOG_ERROR("bg.battleground", "BattlegroundSA::CaptureGraveyard: non-existant GY entry: %u", BG_SA_GYEntries[i]);
807 return;
808 }
809
810 AddSpiritGuide(i + BG_SA_MAXNPC, sg->Loc.X, sg->Loc.Y, sg->Loc.Z, BG_SA_GYOrientation[i], GraveyardStatus[i]);
811 uint32 npc = 0;
812 uint32 flag = 0;
813
814 switch (i)
815 {
816 case BG_SA_LEFT_CAPTURABLE_GY:
817 flag = BG_SA_LEFT_FLAG;
818 DelObject(flag);
819 AddObject(flag, (BG_SA_ObjEntries[flag] - (Source->GetTeamId() == TEAM_ALLIANCE ? 0 : 1)),
820 BG_SA_ObjSpawnlocs[flag], 0, 0, 0, 0, RESPAWN_ONE_DAY);
821
822 npc = BG_SA_NPC_RIGSPARK;
823 if (Creature* rigspark = AddCreature(BG_SA_NpcEntries[npc], npc, BG_SA_NpcSpawnlocs[npc], Attackers))

830 if (Creature* dem = GetBGCreature(j))
831 dem->setFaction(BG_SA_Factions[Attackers]);
832 }
833
834 UpdateWorldState(BG_SA_LEFT_GY_ALLIANCE, GraveyardStatus[i] == TEAM_ALLIANCE);
835 UpdateWorldState(BG_SA_LEFT_GY_HORDE, GraveyardStatus[i] == TEAM_HORDE);

839
840 break;
841 case BG_SA_RIGHT_CAPTURABLE_GY:
842 flag = BG_SA_RIGHT_FLAG;
843 DelObject(flag);
844 AddObject(flag, (BG_SA_ObjEntries[flag] - (Source->GetTeamId() == TEAM_ALLIANCE ? 0 : 1)),
845 BG_SA_ObjSpawnlocs[flag], 0, 0, 0, 0, RESPAWN_ONE_DAY);
846
847 npc = BG_SA_NPC_SPARKLIGHT;
848 if (Creature* sparklight = AddCreature(BG_SA_NpcEntries[npc], npc, BG_SA_NpcSpawnlocs[npc], Attackers))

851 for (uint8 j = BG_SA_DEMOLISHER_5; j <= BG_SA_DEMOLISHER_6; j++)

853 AddCreature(BG_SA_NpcEntries[j], j, BG_SA_NpcSpawnlocs[j], Attackers == TEAM_ALLIANCE ? TEAM_HORDE : TEAM_ALLIANCE, 600);

885void BattlegroundSA::TitanRelicActivated(Player* clicker)
886{
887 if (!clicker)
888 return;
889
890 if (CanInteractWithObject(BG_SA_TITAN_RELIC))
891 {
892 if (clicker->GetTeamId() == Attackers)
893 {
894 if (clicker->GetTeamId() == TEAM_ALLIANCE)
895 SendMessageToAll(LANG_BG_SA_ALLIANCE_CAPTURED_RELIC, CHAT_MSG_BG_SYSTEM_NEUTRAL);
896 else
897 SendMessageToAll(LANG_BG_SA_HORDE_CAPTURED_RELIC, CHAT_MSG_BG_SYSTEM_NEUTRAL);
898
899 if (Status == BG_SA_ROUND_ONE)
900 {
901 RoundScores[0].winner = Attackers;
902 RoundScores[0].time = TotalTime;
903 // Achievement Storm the Beach (1310)
904 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
905 {
906 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
907 if (player->GetTeamId() == Attackers)
908 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, 65246);
909 }
910
911 Attackers = (Attackers == TEAM_ALLIANCE) ? TEAM_HORDE : TEAM_ALLIANCE;
912 Status = BG_SA_SECOND_WARMUP;
913 TotalTime = 0;
914 ToggleTimer();
915
916 if (Creature* c = GetBGCreature(BG_SA_NPC_KANRETHAD))
917 SendChatMessage(c, TEXT_ROUND_1_FINISHED);
918
919 UpdateWaitTimer = 5000;
920 SignaledRoundTwo = false;
921 SignaledRoundTwoHalfMin = false;
922 InitSecondRound = true;
923 ResetObjs();
924 GetBgMap()->UpdateAreaDependentAuras();
925 CastSpellOnTeam(SPELL_END_OF_ROUND, ALLIANCE);
926 CastSpellOnTeam(SPELL_END_OF_ROUND, HORDE);
927 }
928 else if (Status == BG_SA_ROUND_TWO)
929 {
930 RoundScores[1].winner = Attackers;
931 RoundScores[1].time = TotalTime;
932 ToggleTimer();
933 // Achievement Storm the Beach (1310)
934 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
935 {
936 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
937 if (player->GetTeamId() == Attackers && RoundScores[1].winner == Attackers)
938 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, 65246);
939 }
940
941 if (RoundScores[0].time == RoundScores[1].time)
942 EndBattleground(0);
943 else if (RoundScores[0].time < RoundScores[1].time)
944 EndBattleground(RoundScores[0].winner == TEAM_ALLIANCE ? ALLIANCE : HORDE);
945 else
946 EndBattleground(RoundScores[1].winner == TEAM_ALLIANCE ? ALLIANCE : HORDE);
947 }
948 }
949 }

22#include "LinkedReference/Reference.h"

24class Group;
25class Player;
26
27class GroupReference : public Reference<Group, Player>
28{
29 protected:
30 uint8 iSubGroup;
31 void targetObjectBuildLink() override;
32 void targetObjectDestroyLink() override;
33 void sourceObjectDestroyLink() override;
34 public:
35 GroupReference() : Reference<Group, Player>(), iSubGroup(0) { }
36 ~GroupReference() { unlink(); }
37 GroupReference* next() { return (GroupReference*)Reference<Group, Player>::next(); }
38 GroupReference const* next() const { return (GroupReference const*)Reference<Group, Player>::next(); }
39 uint8 getSubGroup() const { return iSubGroup; }
40 void setSubGroup(uint8 pSubGroup) { iSubGroup = pSubGroup; }
41};

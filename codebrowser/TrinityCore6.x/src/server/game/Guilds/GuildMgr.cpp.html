46ObjectGuid::LowType GuildMgr::GenerateGuildId()
47{
48 if (NextGuildId >= 0xFFFFFFFE)
49 {
50 TC_LOG_ERROR("guild", "Guild ids overflow!! Can't continue, shutting down server. ");
51 World::StopNow(ERROR_EXIT_CODE);
52 }
53 return NextGuildId++;
54}

57Guild* GuildMgr::GetGuildById(ObjectGuid::LowType guildId) const
58{
59 GuildContainer::const_iterator itr = GuildStore.find(guildId);
60 if (itr != GuildStore.end())
61 return itr->second;
62
63 return NULL;
64}

66Guild* GuildMgr::GetGuildByGuid(ObjectGuid guid) const
67{
68 // Full guids are only used when receiving/sending data to client
69 // everywhere else guild id is used
70 if (guid.IsGuild())
71 if (ObjectGuid::LowType guildId = guid.GetCounter())
72 return GetGuildById(guildId);
73
74 return NULL;
75}

77Guild* GuildMgr::GetGuildByName(const std::string& guildName) const
78{
79 std::string search = guildName;
80 std::transform(search.begin(), search.end(), search.begin(), ::toupper);
81 for (GuildContainer::const_iterator itr = GuildStore.begin(); itr != GuildStore.end(); ++itr)
82 {
83 std::string gname = itr->second->GetName();
84 std::transform(gname.begin(), gname.end(), gname.begin(), ::toupper);
85 if (search == gname)
86 return itr->second;
87 }
88 return NULL;
89}

110 // 1. Load all guilds
111 TC_LOG_INFO("server.loading", "Loading guilds definitions...");
112 {
113 uint32 oldMSTime = getMSTime();
114
115 // 0 1 2 3 4 5 6
116 QueryResult result = CharacterDatabase.Query("SELECT g.guildid, g.name, g.leaderguid, g.EmblemStyle, g.EmblemColor, g.BorderStyle, g.BorderColor, "
117 // 7 8 9 10 11 12 13
118 "g.BackgroundColor, g.info, g.motd, g.createdate, g.BankMoney, g.level, COUNT(gbt.guildid) "
119 "FROM guild g LEFT JOIN guild_bank_tab gbt ON g.guildid = gbt.guildid GROUP BY g.guildid ORDER BY g.guildid ASC");
120
121 if (!result)
122 {
123 TC_LOG_INFO("server.loading", ">> Loaded 0 guild definitions. DB table `guild` is empty.");
124 return;
125 }
126 else
127 {
128 uint32 count = 0;
129 do
130 {
131 Field* fields = result->Fetch();

133

138 }
139
140 AddGuild(guild);
141
142 ++count;
143 }
144 while (result->NextRow());

461void GuildMgr::LoadGuildRewards()
462{
463 uint32 oldMSTime = getMSTime();
464
465 // 0 1 2 3
466 QueryResult result = WorldDatabase.Query("SELECT ItemID, MinGuildRep, RaceMask, Cost FROM guild_rewards");
467
468 if (!result)
469 {
470 TC_LOG_ERROR("server.loading", ">> Loaded 0 guild reward definitions. DB table `guild_rewards` is empty.");
471 return;
472 }
473
474 uint32 count = 0;
475
476 do
477 {
478 GuildReward reward;
479 Field* fields = result->Fetch();
480 reward.ItemID = fields[0].GetUInt32();
481 reward.MinGuildRep = fields[1].GetUInt8();
482 reward.RaceMask = fields[2].GetInt32();
483 reward.Cost = fields[3].GetUInt64();
484
485 if (!sObjectMgr->GetItemTemplate(reward.ItemID))
486 {
487 TC_LOG_ERROR("server.loading", "Guild rewards constains not existing item entry %u", reward.ItemID);
488 continue;
489 }
490
491 if (reward.MinGuildRep >= MAX_REPUTATION_RANK)
492 {
493 TC_LOG_ERROR("server.loading", "Guild rewards contains wrong reputation standing %u, max is %u", uint32(reward.MinGuildRep), MAX_REPUTATION_RANK - 1);
494 continue;
495 }
496
497 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_GUILD_REWARDS_REQ_ACHIEVEMENTS);
498 stmt->setUInt32(0, reward.ItemID);
499 PreparedQueryResult reqAchievementResult = WorldDatabase.Query(stmt);
500 if (reqAchievementResult)
501 {
502 do
503 {
504 fields = reqAchievementResult->Fetch();
505
506 uint32 requiredAchievementId = fields[0].GetUInt32();
507
508 if (!sAchievementMgr->GetAchievement(requiredAchievementId))
509 {
510 TC_LOG_ERROR("server.loading", "Guild rewards constains not existing achievement entry %u", requiredAchievementId);
511 continue;
512 }
513
514 reward.AchievementsRequired.push_back(requiredAchievementId);
515 } while (reqAchievementResult->NextRow());
516 }
517
518 GuildRewards.push_back(reward);
519 ++count;
520 } while (result->NextRow());
521
522 TC_LOG_INFO("server.loading", ">> Loaded %u guild reward definitions in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
523}

525void GuildMgr::ResetTimes(bool week)
526{

528
529 for (GuildContainer::const_iterator itr = GuildStore.begin(); itr != GuildStore.end(); ++itr)

531 guild->ResetTimes(week);
532}

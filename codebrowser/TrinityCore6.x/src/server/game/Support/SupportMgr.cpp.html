63void BugTicket::LoadFromDB(Field* fields)
64{
65 uint8 idx = 0;
66 _id = fields[ idx].GetUInt32();
67 _playerGuid = ObjectGuid::Create<HighGuid::Player>(fields[++idx].GetUInt64());
68 _note = fields[++idx].GetString();
69 _createTime = fields[++idx].GetUInt32();
70 _mapId = fields[++idx].GetUInt16();
71 _pos.x = fields[++idx].GetFloat();
72 _pos.y = fields[++idx].GetFloat();
73 _pos.z = fields[++idx].GetFloat();
74 _facing = fields[++idx].GetFloat();
75
76 int64 closedBy = fields[++idx].GetInt64();
77 if (closedBy == 0)
78 _closedBy = ObjectGuid::Empty;
79 else if (closedBy < 0)
80 _closedBy.SetRawValue(0, uint64(closedBy));
81 else
82 _closedBy = ObjectGuid::Create<HighGuid::Player>(uint64(closedBy));
83
84 uint64 assignedTo = fields[++idx].GetUInt64();
85 if (assignedTo == 0)
86 _assignedTo = ObjectGuid::Empty;
87 else
88 _assignedTo = ObjectGuid::Create<HighGuid::Player>(assignedTo);
89
90 _comment = fields[++idx].GetString();
91}

189void ComplaintTicket::SaveToDB() const
190{
191 SQLTransaction trans = CharacterDatabase.BeginTransaction();
192
193 uint8 idx = 0;
194 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_REP_GM_COMPLAINT);
195 stmt->setUInt32(idx, _id);
196 stmt->setUInt64(++idx, _playerGuid.GetCounter());
197 stmt->setString(++idx, _note);
198 stmt->setUInt16(++idx, _mapId);
199 stmt->setFloat(++idx, _pos.x);
200 stmt->setFloat(++idx, _pos.y);
201 stmt->setFloat(++idx, _pos.z);
202 stmt->setFloat(++idx, _facing);
203 stmt->setUInt64(++idx, _targetCharacterGuid.GetCounter());
204 stmt->setUInt8(++idx, _complaintType);
205 if (_chatLog.ReportLineIndex)
206 stmt->setInt32(++idx, *_chatLog.ReportLineIndex);
207 else
208 stmt->setInt32(++idx, -1); // empty ReportLineIndex
209 stmt->setInt64(++idx, _closedBy.GetCounter());
210 stmt->setUInt64(++idx, _assignedTo.GetCounter());
211 stmt->setString(++idx, _comment);
212 trans->Append(stmt);
213
214 uint32 lineIndex = 0;
215 for (auto const& c : _chatLog.Lines)
216 {
217 idx = 0;
218 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_GM_COMPLAINT_CHATLINE);
219 stmt->setUInt32(idx, _id);
220 stmt->setUInt32(++idx, lineIndex);
221 stmt->setUInt32(++idx, c.Timestamp);
222 stmt->setString(++idx, c.Text);
223
224 trans->Append(stmt);
225 ++lineIndex;
226 }
227
228 CharacterDatabase.CommitTransaction(trans);
229}

415void SupportMgr::LoadBugTickets()
416{
417 uint32 oldMSTime = getMSTime();
418
419 for (auto const& c : _bugTicketList)
420 delete c.second;
421 _bugTicketList.clear();
422
423 _lastBugId = 0;
424 _openBugTicketCount = 0;
425
426 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_GM_BUGS);
427 PreparedQueryResult result = CharacterDatabase.Query(stmt);
428 if (!result)
429 {
430 TC_LOG_INFO("server.loading", ">> Loaded 0 GM bugs. DB table `gm_bug` is empty!");
431 return;
432 }
433
434 uint32 count = 0;
435 do
436 {
437 Field* fields = result->Fetch();
438 BugTicket* bug = new BugTicket();
439 bug->LoadFromDB(fields);
440
441 if (!bug->IsClosed())
442 ++_openBugTicketCount;
443
444 uint32 id = bug->GetId();
445 if (_lastBugId < id)
446 _lastBugId = id;
447
448 _bugTicketList[id] = bug;
449 ++count;
450 } while (result->NextRow());
451
452 TC_LOG_INFO("server.loading", ">> Loaded %u GM bugs in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
453}

455void SupportMgr::LoadComplaintTickets()
456{
457 uint32 oldMSTime = getMSTime();
458
459 for (auto const& c : _complaintTicketList)
460 delete c.second;
461 _complaintTicketList.clear();
462
463 _lastComplaintId = 0;
464 _openComplaintTicketCount = 0;
465
466 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_GM_COMPLAINTS);
467 PreparedQueryResult result = CharacterDatabase.Query(stmt);
468 if (!result)
469 {
470 TC_LOG_INFO("server.loading", ">> Loaded 0 GM complaints. DB table `gm_complaint` is empty!");
471 return;
472 }
473
474 uint32 count = 0;
475 PreparedStatement* chatLogStmt;
476 PreparedQueryResult chatLogResult;
477 do
478 {
479 Field* fields = result->Fetch();
480 ComplaintTicket* complaint = new ComplaintTicket();
481 complaint->LoadFromDB(fields);
482
483 if (!complaint->IsClosed())
484 ++_openComplaintTicketCount;
485
486 uint32 id = complaint->GetId();
487 if (_lastComplaintId < id)
488 _lastComplaintId = id;
489
490 chatLogStmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_GM_COMPLAINT_CHATLINES);
491 chatLogStmt->setUInt32(0, id);
492 chatLogResult = CharacterDatabase.Query(chatLogStmt);
493
494 if (chatLogResult)
495 {
496 do
497 {
498 Field* chatLineFields = chatLogResult->Fetch();
499 complaint->LoadChatLineFromDB(chatLineFields);
500 } while (chatLogResult->NextRow());
501 }
502
503 _complaintTicketList[id] = complaint;
504 ++count;
505 } while (result->NextRow());
506
507 TC_LOG_INFO("server.loading", ">> Loaded %u GM complaints in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
508}

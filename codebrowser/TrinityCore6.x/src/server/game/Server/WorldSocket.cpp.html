94void WorldSocket::CheckIpCallback(PreparedQueryResult result)
95{
96 if (result)
97 {
98 bool banned = false;
99 do
100 {
101 Field* fields = result->Fetch();
102 if (fields[0].GetUInt64() != 0)
103 banned = true;
104
105 if (!fields[1].GetString().empty())
106 _ipCountry = fields[1].GetString();
107
108 } while (result->NextRow());
109
110 if (banned)
111 {
112 SendAuthResponseError(AUTH_REJECT);
113 TC_LOG_ERROR("network", "WorldSocket::CheckIpCallback: Sent Auth Response (IP %s banned).", GetRemoteIpAddress().to_string().c_str());
114 DelayedCloseSocket();
115 return;
116 }
117 }
118
119 AsyncRead();
120
121 MessageBuffer initializer;
122 ServerPktHeader header;
123 header.Setup.Size = ServerConnectionInitialize.size();
124 initializer.Write(&header, sizeof(header.Setup.Size));
125 initializer.Write(ServerConnectionInitialize.c_str(), ServerConnectionInitialize.length());
126
127 // - io_service.run thread, safe.
128 QueuePacket(std::move(initializer));
129}

131bool WorldSocket::Update()
132{
133 EncryptablePacket* queued;
134 MessageBuffer buffer;
135 while (_bufferQueue.Dequeue(queued))
136 {
137 uint32 sizeOfHeader = SizeOfServerHeader[queued->NeedsEncryption()];
138 uint32 packetSize = queued->size();
139 if (packetSize > MinSizeForCompression && queued->NeedsEncryption())
140 packetSize = compressBound(packetSize) + sizeof(CompressedWorldPacket);
141
142 if (buffer.GetRemainingSpace() < packetSize + sizeOfHeader)
143 {
144 QueuePacket(std::move(buffer));
145 buffer.Resize(4096);
146 }
147
148 if (buffer.GetRemainingSpace() >= packetSize + sizeOfHeader)
149 WritePacketToBuffer(*queued, buffer);
150 else // single packet larger than 4096 bytes
151 {
152 MessageBuffer packetBuffer(packetSize + sizeOfHeader);
153 WritePacketToBuffer(*queued, packetBuffer);
154 QueuePacket(std::move(packetBuffer));
155 }
156
157 delete queued;
158 }
159
160 if (buffer.GetActiveSize() > 0)
161 QueuePacket(std::move(buffer));
162
163 if (!BaseSocket::Update())
164 return false;
165
166 if (_queryFuture.valid() && _queryFuture.wait_for(std::chrono::seconds(0)) == std::future_status::ready)
167 {
168 auto callback = _queryCallback;
169 _queryCallback = nullptr;
170 callback(_queryFuture.get());
171 }
172
173 return true;
174}

176void WorldSocket::HandleSendAuthSession()
177{
178 _encryptSeed.SetRand(16 * 8);
179 _decryptSeed.SetRand(16 * 8);
180
181 WorldPackets::Auth::AuthChallenge challenge;
182 challenge.Challenge = _authSeed;
183 memcpy(&challenge.DosChallenge[0], _encryptSeed.AsByteArray(16).get(), 16);
184 memcpy(&challenge.DosChallenge[4], _decryptSeed.AsByteArray(16).get(), 16);

186
187 SendPacketAndLogOpcode(*challenge.Write());

198void WorldSocket::ReadHandler()
199{
200 if (!IsOpen())
201 return;
202
203 MessageBuffer& packet = GetReadBuffer();
204 while (packet.GetActiveSize() > 0)
205 {
206 if (_headerBuffer.GetRemainingSpace() > 0)
207 {
208 // need to receive the header
209 std::size_t readHeaderSize = std::min(packet.GetActiveSize(), _headerBuffer.GetRemainingSpace());
210 _headerBuffer.Write(packet.GetReadPointer(), readHeaderSize);
211 packet.ReadCompleted(readHeaderSize);
212
213 if (_headerBuffer.GetRemainingSpace() > 0)
214 {
215 // Couldn't receive the whole header this time.
216 ASSERT(packet.GetActiveSize() == 0);
217 break;
218 }
219
220 // We just received nice new header
221 if (!ReadHeaderHandler())
222 {
223 CloseSocket();
224 return;
225 }
226 }
227
228 // We have full read header, now check the data payload
229 if (_packetBuffer.GetRemainingSpace() > 0)
230 {
231 // need more data in the payload
232 std::size_t readDataSize = std::min(packet.GetActiveSize(), _packetBuffer.GetRemainingSpace());
233 _packetBuffer.Write(packet.GetReadPointer(), readDataSize);
234 packet.ReadCompleted(readDataSize);
235
236 if (_packetBuffer.GetRemainingSpace() > 0)
237 {
238 // Couldn't receive the whole data this time.
239 ASSERT(packet.GetActiveSize() == 0);
240 break;
241 }
242 }
243
244 // just received fresh new payload
245 ReadDataHandlerResult result = ReadDataHandler();
246 _headerBuffer.Reset();
247 if (result != ReadDataHandlerResult::Ok)
248 {
249 if (result != ReadDataHandlerResult::WaitingForQuery)
250 CloseSocket();
251
252 return;
253 }
254 }
255
256 AsyncRead();
257}

282bool WorldSocket::ReadHeaderHandler()
283{
284 ASSERT(_headerBuffer.GetActiveSize() == SizeOfClientHeader[_initialized][_authCrypt.IsInitialized()], "Header size " SZFMTD " different than expected %u", _headerBuffer.GetActiveSize(), SizeOfClientHeader[_initialized][_authCrypt.IsInitialized()]);
285
286 _authCrypt.DecryptRecv(_headerBuffer.GetReadPointer(), _headerBuffer.GetActiveSize());
287
288 ClientPktHeader* header = reinterpret_cast<ClientPktHeader*>(_headerBuffer.GetReadPointer());
289 uint32 opcode;
290 uint32 size;
291
292 ExtractOpcodeAndSize(header, opcode, size);
293
294 if (!ClientPktHeader::IsValidSize(size) || (_initialized && !ClientPktHeader::IsValidOpcode(opcode)))
295 {
296 TC_LOG_ERROR("network", "WorldSocket::ReadHeaderHandler(): client %s sent malformed packet (size: %u, cmd: %u)",
297 GetRemoteIpAddress().to_string().c_str(), size, opcode);
298 return false;
299 }
300
301 _packetBuffer.Resize(size);
302 return true;
303}

305WorldSocket::ReadDataHandlerResult WorldSocket::ReadDataHandler()
306{
307 if (_initialized)
308 {
309 ClientPktHeader* header = reinterpret_cast<ClientPktHeader*>(_headerBuffer.GetReadPointer());
310 uint32 cmd;
311 uint32 size;
312
313 ExtractOpcodeAndSize(header, cmd, size);
314
315 OpcodeClient opcode = static_cast<OpcodeClient>(cmd);
316
317 WorldPacket packet(opcode, std::move(_packetBuffer), GetConnectionType());
318
319 if (sPacketLog->CanLogPacket())
320 sPacketLog->LogPacket(packet, CLIENT_TO_SERVER, GetRemoteIpAddress(), GetRemotePort(), GetConnectionType());
321
322 std::unique_lock<std::mutex> sessionGuard(_worldSessionLock, std::defer_lock);
323
324 switch (opcode)
325 {
326 case CMSG_PING:
327 LogOpcodeText(opcode, sessionGuard);
328 return HandlePing(packet) ? ReadDataHandlerResult::Ok : ReadDataHandlerResult::Error;
329 case CMSG_AUTH_SESSION:
330 {
331 LogOpcodeText(opcode, sessionGuard);
332 if (_authed)
333 {
334 // locking just to safely log offending user is probably overkill but we are disconnecting him anyway
335 if (sessionGuard.try_lock())
336 TC_LOG_ERROR("network", "WorldSocket::ProcessIncoming: received duplicate CMSG_AUTH_SESSION from %s", _worldSession->GetPlayerInfo().c_str());
337 return ReadDataHandlerResult::Error;
338 }
339
340 std::shared_ptr<WorldPackets::Auth::AuthSession> authSession = std::make_shared<WorldPackets::Auth::AuthSession>(std::move(packet));
341 authSession->Read();
342 HandleAuthSession(authSession);
343 return ReadDataHandlerResult::WaitingForQuery;
344 }
345 case CMSG_AUTH_CONTINUED_SESSION:
346 {
347 LogOpcodeText(opcode, sessionGuard);
348 if (_authed)
349 {
350 // locking just to safely log offending user is probably overkill but we are disconnecting him anyway
351 if (sessionGuard.try_lock())
352 TC_LOG_ERROR("network", "WorldSocket::ProcessIncoming: received duplicate CMSG_AUTH_CONTINUED_SESSION from %s", _worldSession->GetPlayerInfo().c_str());
353 return ReadDataHandlerResult::Error;
354 }
355
356 std::shared_ptr<WorldPackets::Auth::AuthContinuedSession> authSession = std::make_shared<WorldPackets::Auth::AuthContinuedSession>(std::move(packet));
357 authSession->Read();
358 HandleAuthContinuedSession(authSession);
359 return ReadDataHandlerResult::WaitingForQuery;
360 }
361 case CMSG_KEEP_ALIVE:
362 LogOpcodeText(opcode, sessionGuard);
363 break;
364 case CMSG_LOG_DISCONNECT:
365 LogOpcodeText(opcode, sessionGuard);
366 packet.rfinish(); // contains uint32 disconnectReason;
367 break;
368 case CMSG_ENABLE_NAGLE:
369 LogOpcodeText(opcode, sessionGuard);
370 SetNoDelay(false);
371 break;
372 case CMSG_CONNECT_TO_FAILED:
373 {
374 sessionGuard.lock();
375
376 LogOpcodeText(opcode, sessionGuard);
377 WorldPackets::Auth::ConnectToFailed connectToFailed(std::move(packet));
378 connectToFailed.Read();
379 HandleConnectToFailed(connectToFailed);
380 break;
381 }
382 default:
383 {
384 sessionGuard.lock();
385
386 LogOpcodeText(opcode, sessionGuard);
387
388 if (!_worldSession)
389 {
390 TC_LOG_ERROR("network.opcode", "ProcessIncoming: Client not authed opcode = %u", uint32(opcode));
391 return ReadDataHandlerResult::Error;
392 }
393
394 OpcodeHandler const* handler = opcodeTable[opcode];
395 if (!handler)
396 {
397 TC_LOG_ERROR("network.opcode", "No defined handler for opcode %s sent by %s", GetOpcodeNameForLogging(static_cast<OpcodeClient>(packet.GetOpcode())).c_str(), _worldSession->GetPlayerInfo().c_str());
398 break;
399 }
400
401 // Our Idle timer will reset on any non PING opcodes.
402 // Catches people idling on the login screen and any lingering ingame connections.
403 _worldSession->ResetTimeOutTime();
404
405 // Copy the packet to the heap before enqueuing
406 _worldSession->QueuePacket(new WorldPacket(std::move(packet)));
407 break;
408 }
409 }
410 }
411 else
412 {
413 std::string initializer(reinterpret_cast<char const*>(_packetBuffer.GetReadPointer()), std::min(_packetBuffer.GetActiveSize(), ClientConnectionInitialize.length()));
414 if (initializer != ClientConnectionInitialize)
415 return ReadDataHandlerResult::Error;
416
417 _compressionStream = new z_stream();
418 _compressionStream->zalloc = (alloc_func)NULL;
419 _compressionStream->zfree = (free_func)NULL;
420 _compressionStream->opaque = (voidpf)NULL;
421 _compressionStream->avail_in = 0;
422 _compressionStream->next_in = NULL;
423 int32 z_res = deflateInit2(_compressionStream, sWorld->getIntConfig(CONFIG_COMPRESSION), Z_DEFLATED, -15, 8, Z_DEFAULT_STRATEGY);
424 if (z_res != Z_OK)
425 {
426 TC_LOG_ERROR("network", "Can't initialize packet compression (zlib: deflateInit) Error code: %i (%s)", z_res, zError(z_res));
427 return ReadDataHandlerResult::Error;
428 }
429
430 _initialized = true;
431 _headerBuffer.Resize(SizeOfClientHeader[1][0]);
432 _packetBuffer.Reset();
433 HandleSendAuthSession();
434 }
435
436 return ReadDataHandlerResult::Ok;
437}

469void WorldSocket::WritePacketToBuffer(EncryptablePacket const& packet, MessageBuffer& buffer)
470{
471 ServerPktHeader header;
472 uint32 sizeOfHeader = SizeOfServerHeader[packet.NeedsEncryption()];
473 uint32 opcode = packet.GetOpcode();
474 uint32 packetSize = packet.size();
475
476 // Reserve space for buffer
477 uint8* headerPos = buffer.GetWritePointer();
478 buffer.WriteCompleted(sizeOfHeader);
479
480 if (packetSize > MinSizeForCompression && packet.NeedsEncryption())
481 {
482 CompressedWorldPacket cmp;
483 cmp.UncompressedSize = packetSize + 4;
484 cmp.UncompressedAdler = adler32(adler32(0x9827D8F1, (Bytef*)&opcode, 4), packet.contents(), packetSize);
485
486 // Reserve space for compression info - uncompressed size and checksums
487 uint8* compressionInfo = buffer.GetWritePointer();
488 buffer.WriteCompleted(sizeof(CompressedWorldPacket));
489
490 uint32 compressedSize = CompressPacket(buffer.GetWritePointer(), packet);
491
492 cmp.CompressedAdler = adler32(0x9827D8F1, buffer.GetWritePointer(), compressedSize);
493
494 memcpy(compressionInfo, &cmp, sizeof(CompressedWorldPacket));
495 buffer.WriteCompleted(compressedSize);
496 packetSize = compressedSize + sizeof(CompressedWorldPacket);
497
498 opcode = SMSG_COMPRESSED_PACKET;
499 }
500 else if (!packet.empty())
501 buffer.Write(packet.contents(), packet.size());
502
503 if (packet.NeedsEncryption())
504 {
505 header.Normal.Size = packetSize;
506 header.Normal.Command = opcode;
507 _authCrypt.EncryptSend((uint8*)&header, sizeOfHeader);
508 }
509 else
510 {
511 header.Setup.Size = packetSize + 4;
512 header.Setup.Command = opcode;
513 }
514
515 memcpy(headerPos, &header, sizeOfHeader);
516}

518uint32 WorldSocket::CompressPacket(uint8* buffer, WorldPacket const& packet)
519{
520 uint32 opcode = packet.GetOpcode();
521 uint32 bufferSize = deflateBound(_compressionStream, packet.size() + sizeof(opcode));
522
523 _compressionStream->next_out = buffer;
524 _compressionStream->avail_out = bufferSize;
525 _compressionStream->next_in = (Bytef*)&opcode;
526 _compressionStream->avail_in = sizeof(uint32);
527
528 int32 z_res = deflate(_compressionStream, Z_NO_FLUSH);
529 if (z_res != Z_OK)
530 {
531 TC_LOG_ERROR("network", "Can't compress packet opcode (zlib: deflate) Error code: %i (%s, msg: %s)", z_res, zError(z_res), _compressionStream->msg);
532 return 0;
533 }
534
535 _compressionStream->next_in = (Bytef*)packet.contents();
536 _compressionStream->avail_in = packet.size();
537
538 z_res = deflate(_compressionStream, Z_SYNC_FLUSH);
539 if (z_res != Z_OK)
540 {
541 TC_LOG_ERROR("network", "Can't compress packet data (zlib: deflate) Error code: %i (%s, msg: %s)", z_res, zError(z_res), _compressionStream->msg);
542 return 0;
543 }
544
545 return bufferSize - _compressionStream->avail_out;
546}

613 _headerBuffer.Resize(SizeOfClientHeader[1][1]);
614

616 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_ACCOUNT_INFO_BY_NAME);
617 stmt->setInt32(0, int32(realm.Id.Realm));
618 stmt->setString(1, authSession->Account);

621 _queryFuture = LoginDatabase.AsyncQuery(stmt);

623
624void WorldSocket::HandleAuthSessionCallback(std::shared_ptr<WorldPackets::Auth::AuthSession> authSession, PreparedQueryResult result)
625{
626 // Stop if the account is not found
627 if (!result)
628 {
629 // We can not log here, as we do not know the account. Thus, no accountId.
630 SendAuthResponseError(AUTH_UNKNOWN_ACCOUNT);
631 TC_LOG_ERROR("network", "WorldSocket::HandleAuthSession: Sent Auth Response (unknown account).");
632 DelayedCloseSocket();
633 return;
634 }
635
636 AccountInfo account(result->Fetch());
637
638 // For hook purposes, we get Remoteaddress at this point.
639 std::string address = GetRemoteIpAddress().to_string();
640
641 // As we don't know if attempted login process by ip works, we update last_attempt_ip right away
642 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_LAST_ATTEMPT_IP);
643 stmt->setString(0, address);
644 stmt->setString(1, authSession->Account);
645 LoginDatabase.Execute(stmt);
646 // This also allows to check for possible "hack" attempts on account
647
648 // even if auth credentials are bad, try using the session key we have - client cannot read auth response error without it
649 _authCrypt.Init(&account.Game.SessionKey);
650
651 // First reject the connection if packet contains invalid data or realm state doesn't allow logging in
652 if (sWorld->IsClosed())
653 {
654 SendAuthResponseError(AUTH_REJECT);
655 TC_LOG_ERROR("network", "WorldSocket::HandleAuthSession: World closed, denying client (%s).", GetRemoteIpAddress().to_string().c_str());
656 DelayedCloseSocket();
657 return;
658 }
659
660 if (authSession->RealmID != realm.Id.Realm)
661 {
662 SendAuthResponseError(REALM_LIST_REALM_NOT_FOUND);
663 TC_LOG_ERROR("network", "WorldSocket::HandleAuthSession: Client %s requested connecting with realm id %u but this realm has id %u set in config.",
664 GetRemoteIpAddress().to_string().c_str(), authSession->RealmID, realm.Id.Realm);
665 DelayedCloseSocket();
666 return;
667 }
668
669 // Must be done before WorldSession is created
670 bool wardenActive = sWorld->getBoolConfig(CONFIG_WARDEN_ENABLED);
671 if (wardenActive && account.BattleNet.OS != "Win" && account.BattleNet.OS != "Wn64" && account.BattleNet.OS != "Mc64")
672 {
673 SendAuthResponseError(AUTH_REJECT);
674 TC_LOG_ERROR("network", "WorldSocket::HandleAuthSession: Client %s attempted to log in using invalid client OS (%s).", address.c_str(), account.BattleNet.OS.c_str());
675 DelayedCloseSocket();
676 return;
677 }
678
679 if (!account.BattleNet.Id || authSession->LoginServerType != 1)
680 {
681 SendAuthResponseError(AUTH_REJECT);
682 TC_LOG_ERROR("network", "WorldSocket::HandleAuthSession: Client %s (%s) attempted to log in using deprecated login method (GRUNT).", authSession->Account.c_str(), address.c_str());
683 DelayedCloseSocket();
684 return;
685 }
686
687 // Check that Key and account name are the same on client and server
688 uint32 t = 0;
689
690 SHA1Hash sha;
691 sha.UpdateData(authSession->Account);
692 sha.UpdateData((uint8*)&t, 4);
693 sha.UpdateData((uint8*)&authSession->LocalChallenge, 4);
694 sha.UpdateData((uint8*)&_authSeed, 4);
695 sha.UpdateBigNumbers(&account.Game.SessionKey, NULL);
696 sha.Finalize();
697
698 if (memcmp(sha.GetDigest(), authSession->Digest, SHA_DIGEST_LENGTH) != 0)
699 {
700 SendAuthResponseError(AUTH_FAILED);
701 TC_LOG_ERROR("network", "WorldSocket::HandleAuthSession: Authentication failed for account: %u ('%s') address: %s", account.Game.Id, authSession->Account.c_str(), address.c_str());
702 DelayedCloseSocket();
703 return;
704 }
705
706 ///- Re-check ip locking (same check as in auth).
707 if (account.BattleNet.IsLockedToIP)
708 {
709 if (account.BattleNet.LastIP != address)
710 {
711 SendAuthResponseError(AUTH_FAILED);
712 TC_LOG_DEBUG("network", "WorldSocket::HandleAuthSession: Sent Auth Response (Account IP differs. Original IP: %s, new IP: %s).", account.BattleNet.LastIP.c_str(), address.c_str());
713 // We could log on hook only instead of an additional db log, however action logger is config based. Better keep DB logging as well
714 sScriptMgr->OnFailedAccountLogin(account.Game.Id);
715 DelayedCloseSocket();
716 return;
717 }
718 }
719 else if (!account.BattleNet.LockCountry.empty() && account.BattleNet.LockCountry != "00" && !_ipCountry.empty())
720 {
721 if (account.BattleNet.LockCountry != _ipCountry)
722 {
723 SendAuthResponseError(AUTH_FAILED);
724 TC_LOG_DEBUG("network", "WorldSocket::HandleAuthSession: Sent Auth Response (Account country differs. Original country: %s, new country: %s).", account.BattleNet.LockCountry.c_str(), _ipCountry.c_str());
725 // We could log on hook only instead of an additional db log, however action logger is config based. Better keep DB logging as well
726 sScriptMgr->OnFailedAccountLogin(account.Game.Id);
727 DelayedCloseSocket();
728 return;
729 }
730 }
731
732 int64 mutetime = account.Game.MuteTime;
733 //! Negative mutetime indicates amount of seconds to be muted effective on next login - which is now.
734 if (mutetime < 0)
735 {
736 mutetime = time(NULL) + llabs(mutetime);
737
738 stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_MUTE_TIME_LOGIN);
739 stmt->setInt64(0, mutetime);
740 stmt->setUInt32(1, account.Game.Id);
741 LoginDatabase.Execute(stmt);
742 }
743
744 if (account.IsBanned())
745 {
746 SendAuthResponseError(AUTH_BANNED);
747 TC_LOG_ERROR("network", "WorldSocket::HandleAuthSession: Sent Auth Response (Account banned).");
748 sScriptMgr->OnFailedAccountLogin(account.Game.Id);
749 DelayedCloseSocket();
750 return;
751 }
752
753 // Check locked state for server
754 AccountTypes allowedAccountType = sWorld->GetPlayerSecurityLimit();
755 TC_LOG_DEBUG("network", "Allowed Level: %u Player Level %u", allowedAccountType, account.Game.Security);
756 if (allowedAccountType > SEC_PLAYER && account.Game.Security < allowedAccountType)
757 {
758 SendAuthResponseError(AUTH_UNAVAILABLE);
759 TC_LOG_DEBUG("network", "WorldSocket::HandleAuthSession: User tries to login but his security level is not enough");
760 sScriptMgr->OnFailedAccountLogin(account.Game.Id);
761 DelayedCloseSocket();
762 return;
763 }
764
765 TC_LOG_DEBUG("network", "WorldSocket::HandleAuthSession: Client '%s' authenticated successfully from %s.", authSession->Account.c_str(), address.c_str());
766
767 // Update the last_ip in the database as it was successful for login
768 stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_LAST_IP);
769
770 stmt->setString(0, address);
771 stmt->setString(1, authSession->Account);
772
773 LoginDatabase.Execute(stmt);
774
775 // At this point, we can safely hook a successful login
776 sScriptMgr->OnAccountLogin(account.Game.Id);
777
778 _authed = true;
779 _worldSession = new WorldSession(account.Game.Id, std::move(authSession->Account), account.BattleNet.Id, shared_from_this(), account.Game.Security,
780 account.Game.Expansion, mutetime, account.BattleNet.Locale, account.Game.Recruiter, account.Game.IsRectuiter);
781 _worldSession->ReadAddonsInfo(authSession->AddonInfo);
782
783 // Initialize Warden system only if it is enabled by config
784 if (wardenActive)
785 _worldSession->InitWarden(&account.Game.SessionKey, account.BattleNet.OS);
786
787 _queryCallback = std::bind(&WorldSocket::LoadSessionPermissionsCallback, this, std::placeholders::_1);
788 _queryFuture = _worldSession->LoadPermissionsAsync();
789 AsyncRead();
790}

800void WorldSocket::HandleAuthContinuedSession(std::shared_ptr<WorldPackets::Auth::AuthContinuedSession> authSession)
801{
802 WorldSession::ConnectToKey key;
803 key.Raw = authSession->Key;
804
805 _type = ConnectionType(key.Fields.ConnectionType);
806 if (_type != CONNECTION_TYPE_INSTANCE)
807 {
808 SendAuthResponseError(AUTH_UNKNOWN_ACCOUNT);
809 DelayedCloseSocket();
810 return;
811 }
812
813 // Client switches packet headers after sending CMSG_AUTH_CONTINUED_SESSION
814 _headerBuffer.Resize(SizeOfClientHeader[1][1]);
815
816 uint32 accountId = uint32(key.Fields.AccountId);
817 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_ACCOUNT_INFO_CONTINUED_SESSION);
818 stmt->setUInt32(0, accountId);
819
820 _queryCallback = std::bind(&WorldSocket::HandleAuthContinuedSessionCallback, this, authSession, std::placeholders::_1);
821 _queryFuture = LoginDatabase.AsyncQuery(stmt);
822}

824void WorldSocket::HandleAuthContinuedSessionCallback(std::shared_ptr<WorldPackets::Auth::AuthContinuedSession> authSession, PreparedQueryResult result)
825{
826 if (!result)
827 {
828 SendAuthResponseError(AUTH_UNKNOWN_ACCOUNT);
829 DelayedCloseSocket();
830 return;
831 }
832
833 WorldSession::ConnectToKey key;
834 key.Raw = authSession->Key;
835
836 uint32 accountId = uint32(key.Fields.AccountId);
837 Field* fields = result->Fetch();
838 std::string login = fields[0].GetString();
839 BigNumber k;
840 k.SetHexStr(fields[1].GetCString());
841
842 _authCrypt.Init(&k, _encryptSeed.AsByteArray().get(), _decryptSeed.AsByteArray().get());
843
844 SHA1Hash sha;
845 sha.UpdateData(login);
846 sha.UpdateBigNumbers(&k, NULL);
847 sha.UpdateData((uint8*)&_authSeed, 4);
848 sha.Finalize();
849
850 if (memcmp(sha.GetDigest(), authSession->Digest, sha.GetLength()))
851 {
852 SendAuthResponseError(AUTH_UNKNOWN_ACCOUNT);
853 TC_LOG_ERROR("network", "WorldSocket::HandleAuthContinuedSession: Authentication failed for account: %u ('%s') address: %s", accountId, login.c_str(), GetRemoteIpAddress().to_string().c_str());
854 DelayedCloseSocket();
855 return;
856 }
857
858 sWorld->AddInstanceSocket(shared_from_this(), authSession->Key);
859 AsyncRead();
860}

907bool WorldSocket::HandlePing(WorldPacket& recvPacket)
908{
909 using namespace std::chrono;
910
911 uint32 ping;
912 uint32 latency;
913
914 // Get the ping packet content
915 recvPacket >> ping;
916 recvPacket >> latency;
917
918 if (_LastPingTime == steady_clock::time_point())
919 {
920 _LastPingTime = steady_clock::now();
921 }
922 else
923 {
924 steady_clock::time_point now = steady_clock::now();
925
926 steady_clock::duration diff = now - _LastPingTime;
927
928 _LastPingTime = now;
929
930 if (diff < seconds(27))
931 {
932 ++_OverSpeedPings;
933
934 uint32 maxAllowed = sWorld->getIntConfig(CONFIG_MAX_OVERSPEED_PINGS);
935
936 if (maxAllowed && _OverSpeedPings > maxAllowed)
937 {
938 std::unique_lock<std::mutex> sessionGuard(_worldSessionLock);
939
940 if (_worldSession && !_worldSession->HasPermission(rbac::RBAC_PERM_SKIP_CHECK_OVERSPEED_PING))
941 {
942 TC_LOG_ERROR("network", "WorldSocket::HandlePing: %s kicked for over-speed pings (address: %s)",
943 _worldSession->GetPlayerInfo().c_str(), GetRemoteIpAddress().to_string().c_str());
944
945 return false;
946 }
947 }
948 }
949 else
950 _OverSpeedPings = 0;
951 }
952
953 {
954 std::lock_guard<std::mutex> sessionGuard(_worldSessionLock);
955
956 if (_worldSession)
957 {
958 _worldSession->SetLatency(latency);
959 _worldSession->ResetClientTimeDelay();
960 }
961 else
962 {
963 TC_LOG_ERROR("network", "WorldSocket::HandlePing: peer sent CMSG_PING, but is not authenticated or got recently kicked, address = %s", GetRemoteIpAddress().to_string().c_str());
964 return false;
965 }
966 }
967
968 WorldPacket packet(SMSG_PONG, 4);
969 packet << ping;
970 SendPacketAndLogOpcode(packet);
971 return true;
972}

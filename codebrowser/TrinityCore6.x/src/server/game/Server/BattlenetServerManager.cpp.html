23void IPC::BattlenetComm::ServerManager::InitializeConnection()
24{
25 std::string bnetserverAddress = sConfigMgr->GetStringDefault("BnetServer.Address", "127.0.0.1");
26 int32 bnetserverPort = sConfigMgr->GetIntDefault("BnetServer.Port", 1118);
27 _socket = new ZmqMux("inproc://bnetmgr", "tcp://" + bnetserverAddress + ":" + std::to_string(bnetserverPort));
28 _socket->Start();
29}
30
31void IPC::BattlenetComm::ServerManager::CloseConnection()
32{
33 _socket->End();
34 delete _socket;
35 _socket = nullptr;
36}

38IPC::BattlenetComm::Header IPC::BattlenetComm::ServerManager::CreateHeader(BnetCommands command)
39{
40 Header header;
41 header.Ipc.Channel = IPC_CHANNEL_BNET;
42 header.Ipc.Command = command;
43 header.Realm = realm.Id;
44 return header;
45}

47void IPC::BattlenetComm::ServerManager::SendChangeToonOnlineState(uint32 battlenetAccountId, uint32 gameAccountId, ObjectGuid guid, std::string const& name, bool online)
48{
49 // Do nothing for Grunt login
50 if (!battlenetAccountId)
51 return;
52
53 Header header = CreateHeader(BNET_CHANGE_TOON_ONLINE_STATE);
54 ToonHandle toon;
55 toon.AccountId = battlenetAccountId;
56 toon.GameAccountId = gameAccountId;
57 toon.Guid = guid.GetCounter();
58 toon.Name = name;
59
60 zmqpp::message msg;
61 msg << header;
62 msg << toon;
63 msg << online;
64
65 Send(&msg);
66}
67
68void IPC::BattlenetComm::ServerManager::Send(zmqpp::message* msg)
69{
70 if (!_socket)
71 return;
72
73 _socket->Send(msg);
74}

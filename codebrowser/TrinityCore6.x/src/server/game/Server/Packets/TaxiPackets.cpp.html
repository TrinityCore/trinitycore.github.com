34WorldPacket const* WorldPackets::Taxi::ShowTaxiNodes::Write()
35{
36 _worldPacket.WriteBit(WindowInfo.is_initialized());
37 _worldPacket.FlushBits();
38
39 _worldPacket << uint32(Nodes->size());
40
41 if (WindowInfo.is_initialized())
42 {
43 _worldPacket << WindowInfo->UnitGUID;
44 _worldPacket << uint32(WindowInfo->CurrentNode);
45 }
46
47 _worldPacket.append(Nodes->data(), Nodes->size());
48
49 return &_worldPacket;
50}

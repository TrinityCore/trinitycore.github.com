71 class AuthResponse final : public ServerPacket
72 {
73 public:
74 struct RealmInfo
75 {
76 RealmInfo(uint32 realmAddress, bool isHomeRealm, bool isInternalRealm, std::string const& realmNameActual, std::string const& realmNameNormalized) :
77 RealmAddress(realmAddress), IsLocal(isHomeRealm), IsInternalRealm(isInternalRealm), RealmNameActual(realmNameActual), RealmNameNormalized(realmNameNormalized) { }
78
79 uint32 RealmAddress; ///< the virtual address of this realm, constructed as RealmHandle::Region << 24 | RealmHandle::Battlegroup << 16 | RealmHandle::Index
80 bool IsLocal; ///< true if the realm is the same as the account's home realm
81 bool IsInternalRealm; ///< @todo research
82 std::string RealmNameActual; ///< the name of the realm
83 std::string RealmNameNormalized; ///< the name of the realm without spaces
84 };
85
86 struct AuthSuccessInfo
87 {
88 uint32 TimeRemain = 0; ///< the remaining game time that the account has in seconds. It is not currently implemented and probably won't ever be.
89 uint8 AccountExpansionLevel = 0; ///< the current expansion of this account, the possible values are in @ref Expansions
90 uint8 ActiveExpansionLevel = 0; ///< the current server expansion, the possible values are in @ref Expansions
91 uint32 TimeRested = 0; ///< affects the return value of the GetBillingTimeRested() client API call, it is the number of seconds you have left until the experience points and loot you receive from creatures and quests is reduced. It is only used in the Asia region in retail, it's not implemented in TC and will probably never be.
92 uint8 TimeOptions = 0; ///< controls the behavior of the client regarding billing, used in Asia realms, as they don't have monthly subscriptions, possible values are in @ref BillingPlanFlags. It is not currently implemented and will probably never be.
93
94 uint32 VirtualRealmAddress = 0; ///< a special identifier made from the Index, BattleGroup and Region.
95 uint32 RealmNamesCount = 0; ///< the number of realms connected to this one (inclusive). @todo implement
96 uint32 TimeSecondsUntilPCKick = 0; ///< @todo research
97 uint32 CurrencyID = 0; ///< this is probably used for the ingame shop. @todo implement
98
99 std::vector<RealmInfo> VirtualRealms; ///< list of realms connected to this one (inclusive) @todo implement
100 std::vector<CharacterTemplate> Templates; ///< list of pre-made character templates.
101
102 ExpansionRequirementContainer const* AvailableClasses = nullptr; ///< the minimum AccountExpansion required to select the classes
103 ExpansionRequirementContainer const* AvailableRaces = nullptr; ///< the minimum AccountExpansion required to select the races
104
105 bool IsExpansionTrial = false;
106 bool ForceCharacterTemplate = false; ///< forces the client to always use a character template when creating a new character. @see Templates. @todo implement
107 Optional<uint16> NumPlayersHorde; ///< number of horde players in this realm. @todo implement
108 Optional<uint16> NumPlayersAlliance; ///< number of alliance players in this realm. @todo implement
109 bool IsVeteranTrial = false; ///< @todo research
110 };
111
112 struct AuthWaitInfo
113 {
114 uint32 WaitCount = 0; ///< position of the account in the login queue
115 bool HasFCM = false; ///< true if the account has a forced character migration pending. @todo implement
116 };
117
118 AuthResponse();
119
120 WorldPacket const* Write() override;
121
122 Optional<AuthSuccessInfo> SuccessInfo; ///< contains the packet data in case that it has account information (It is never set when WaitInfo is set), otherwise its contents are undefined.
123 Optional<AuthWaitInfo> WaitInfo; ///< contains the queue wait information in case the account is in the login queue.
124 uint8 Result = 0; ///< the result of the authentication process, it is AUTH_OK if it succeeded and the account is ready to log in. It can also be AUTH_WAIT_QUEUE if the account entered the login queue (Queued, QueuePos), possible values are @ref ResponseCodes
125 };

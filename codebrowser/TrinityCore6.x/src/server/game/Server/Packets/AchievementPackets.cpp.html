20ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::Achievement::EarnedAchievement const& earned)
21{
22 data << uint32(earned.Id);
23 data.AppendPackedTime(earned.Date);
24 data << earned.Owner;
25 data << uint32(earned.VirtualRealmAddress);
26 data << uint32(earned.NativeRealmAddress);
27 return data;
28}
29
30ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::Achievement::AllAchievements const& allAchievements)
31{
32 data << uint32(allAchievements.Earned.size());
33 data << uint32(allAchievements.Progress.size());
34
35 for (WorldPackets::Achievement::EarnedAchievement const& earned : allAchievements.Earned)
36 data << earned;
37
38 for (WorldPackets::Achievement::CriteriaProgress const& progress : allAchievements.Progress)
39 {
40 data << uint32(progress.Id);
41 data << uint64(progress.Quantity);
42 data << progress.Player;
43 data.AppendPackedTime(progress.Date);
44 data << uint32(progress.TimeFromStart);
45 data << uint32(progress.TimeFromCreate);
46 data.WriteBits(progress.Flags, 4);
47 data.FlushBits();
48 }
49
50 return data;
51}

96WorldPacket const* WorldPackets::Achievement::AchievementEarned::Write()
97{
98 _worldPacket << Sender;
99 _worldPacket << Earner;
100 _worldPacket << uint32(AchievementID);
101 _worldPacket.AppendPackedTime(Time);
102 _worldPacket << uint32(EarnerNativeRealm);
103 _worldPacket << uint32(EarnerVirtualRealm);
104 _worldPacket.WriteBit(Initial);
105 _worldPacket.FlushBits();
106
107 return &_worldPacket;

110WorldPacket const* WorldPackets::Achievement::ServerFirstAchievement::Write()
111{
112 _worldPacket.WriteBits(Name.length(), 7);
113 _worldPacket.WriteBit(GuildAchievement);
114 _worldPacket << PlayerGUID;
115 _worldPacket << AchievementID;
116 _worldPacket.WriteString(Name);

121WorldPacket const* WorldPackets::Achievement::GuildCriteriaUpdate::Write()
122{
123 _worldPacket << uint32(Progress.size());
124
125 for (GuildCriteriaProgress const& progress : Progress)
126 {
127 _worldPacket << int32(progress.CriteriaID);
128 _worldPacket << uint32(progress.DateCreated);
129 _worldPacket << uint32(progress.DateStarted);
130 _worldPacket.AppendPackedTime(progress.DateUpdated);
131 _worldPacket << uint64(progress.Quantity);
132 _worldPacket << progress.PlayerGUID;
133 _worldPacket << int32(progress.Flags);
134 }
135
136 return &_worldPacket;

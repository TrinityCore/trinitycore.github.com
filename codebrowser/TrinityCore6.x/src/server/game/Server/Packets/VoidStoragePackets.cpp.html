52WorldPacket const* WorldPackets::VoidStorage::VoidStorageContents::Write()
53{
54 _worldPacket.reserve(1 + Items.size() * sizeof(VoidItem));
55
56 _worldPacket.WriteBits(Items.size(), 8);
57 _worldPacket.FlushBits();
58
59 for (VoidItem const& voidItem : Items)
60 _worldPacket << voidItem;
61
62 return &_worldPacket;
63}

65void WorldPackets::VoidStorage::VoidStorageTransfer::Read()
66{
67 _worldPacket >> Npc;
68 Deposits.resize(_worldPacket.read<uint32>());
69 Withdrawals.resize(_worldPacket.read<uint32>());
70
71 for (ObjectGuid& deposit : Deposits)
72 _worldPacket >> deposit;
73
74 for (ObjectGuid& withdrawal : Withdrawals)
75 _worldPacket >> withdrawal;
76}

78WorldPacket const* WorldPackets::VoidStorage::VoidStorageTransferChanges::Write()
79{
80 _worldPacket.reserve(1 + AddedItems.size() * sizeof(VoidItem) + RemovedItems.size() * 16);
81
82 _worldPacket.WriteBits(AddedItems.size(), 4);
83 _worldPacket.WriteBits(RemovedItems.size(), 4);
84 _worldPacket.FlushBits();
85
86 for (VoidItem const& addedItem : AddedItems)
87 _worldPacket << addedItem;
88
89 for (ObjectGuid const& removedItem : RemovedItems)
90 _worldPacket << removedItem;
91
92 return &_worldPacket;
93}

59WorldPacket const* WorldPackets::Party::PartyInvite::Write()
60{
61 // Order guessed
62 _worldPacket.WriteBit(CanAccept);
63 _worldPacket.WriteBit(MightCRZYou);
64 _worldPacket.WriteBit(MustBeBNetFriend);
65 _worldPacket.WriteBit(AllowMultipleRoles);
66 _worldPacket.WriteBit(IsXRealm);
67
68 _worldPacket.WriteBits(InviterName.size(), 6);
69
70 _worldPacket << InviterGUID;
71 _worldPacket << InviterBNetAccountId;
72
73 _worldPacket << InviterVirtualRealmAddress;
74 _worldPacket << Unk1;
75
76 _worldPacket.WriteBit(IsLocal);
77 _worldPacket.WriteBit(Unk2);
78
79 _worldPacket.WriteBits(InviterRealmNameActual.size(), 8);
80 _worldPacket.WriteBits(InviterRealmNameNormalized.size(), 8);
81 _worldPacket.WriteString(InviterRealmNameActual);
82 _worldPacket.WriteString(InviterRealmNameNormalized);
83
84 _worldPacket << ProposedRoles;
85 _worldPacket << int32(LfgSlots.size());
86 _worldPacket << LfgCompletedMask;
87
88 _worldPacket.WriteString(InviterName);
89
90 for (int32 LfgSlot : LfgSlots)
91 _worldPacket << LfgSlot;
92
93 return &_worldPacket;
94}

414void WorldPackets::Party::PartyMemberStats::Initialize(Player const* player)
415{
416 ForEnemy = false;
417
418 MemberStats.GUID = player->GetGUID();
419
420 // Status
421 MemberStats.Status = MEMBER_STATUS_ONLINE;
422
423 if (player->IsPvP())
424 MemberStats.Status |= MEMBER_STATUS_PVP;
425
426 if (!player->IsAlive())
427 {
428 if (player->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_GHOST))
429 MemberStats.Status |= MEMBER_STATUS_GHOST;
430 else
431 MemberStats.Status |= MEMBER_STATUS_DEAD;
432 }
433
434 if (player->IsFFAPvP())
435 MemberStats.Status |= MEMBER_STATUS_PVP_FFA;
436
437 if (player->isAFK())
438 MemberStats.Status |= MEMBER_STATUS_AFK;
439
440 if (player->isDND())
441 MemberStats.Status |= MEMBER_STATUS_DND;
442
443 // Level
444 MemberStats.Level = player->getLevel();
445
446 // Health
447 MemberStats.CurrentHealth = player->GetHealth();
448 MemberStats.MaxHealth = player->GetMaxHealth();
449
450 // Power
451 MemberStats.PowerType = player->getPowerType();
452 MemberStats.CurrentPower = player->GetPower(player->getPowerType());
453 MemberStats.MaxPower = player->GetMaxPower(player->getPowerType());
454
455 // Position
456 MemberStats.ZoneID = player->GetZoneId();
457 MemberStats.PositionX = int16(player->GetPositionX());
458 MemberStats.PositionY = int16(player->GetPositionY());
459 MemberStats.PositionZ = int16(player->GetPositionZ());
460
461 // Unk
462 MemberStats.Unk322 = 0; // Always 0
463 MemberStats.Unk704[0] = 1; // Always 1
464 MemberStats.Unk704[1] = 0; // Always 0
465 MemberStats.Unk200000 = 0; // Always 0
466
467 MemberStats.Unk2000000 = 0;
468 MemberStats.Unk4000000 = 0;
469
470 // Vehicle
471 if (player->GetVehicle() && player->GetVehicle()->GetVehicleInfo())
472 MemberStats.VehicleSeat = player->GetVehicle()->GetVehicleInfo()->SeatID[player->m_movementInfo.transport.seat];
473
474 // Auras
475 for (uint8 i = 0; i < MAX_AURAS; ++i)
476 {
477 if (AuraApplication const* aurApp = player->GetVisibleAura(i))
478 {
479 WorldPackets::Party::GroupAura aura;
480
481 aura.SpellId = aurApp->GetBase()->GetId();
482 aura.EffectMask = aurApp->GetEffectMask();
483 aura.Scalings = aurApp->GetFlags(); // ??
484
485 if (aurApp->GetFlags() & AFLAG_SCALABLE)
486 {
487 for (uint32 e = 0; e < MAX_SPELL_EFFECTS; ++e)
488 {
489 float scale = 0.0f;
490 if (AuraEffect const* eff = aurApp->GetBase()->GetEffect(e))
491 scale = float(eff->GetAmount());
492 aura.EffectScales.push_back(scale);
493 }
494 }
495
496 MemberStats.AuraList.push_back(aura);
497 }
498 }
499
500 // Phases
501 std::set<uint32> const& phases = player->GetPhases();
502 MemberStats.Phases.PhaseShiftFlags = 0x08 | (phases.size() ? 0x10 : 0);
503 MemberStats.Phases.PersonalGUID = ObjectGuid::Empty;
504 for (uint32 phaseId : phases)
505 {
506 WorldPackets::Party::GroupPhase phase;
507 phase.Id = phaseId;
508 phase.Flags = 1;
509 MemberStats.Phases.List.push_back(phase);
510 }
511
512 // Pet
513 if (player->GetPet())
514 {
515 ::Pet* pet = player->GetPet();
516
517 MemberStats.PetStats = boost::in_place();
518
519 MemberStats.PetStats->GUID = pet->GetGUID();
520 MemberStats.PetStats->Name = pet->GetName();
521 MemberStats.PetStats->ModelId = pet->GetDisplayId();
522
523 MemberStats.PetStats->CurrentHealth = pet->GetHealth();
524 MemberStats.PetStats->MaxHealth = pet->GetMaxHealth();
525
526 for (uint8 i = 0; i < MAX_AURAS; ++i)
527 {
528 if (AuraApplication const* aurApp = pet->GetVisibleAura(i))
529 {
530 WorldPackets::Party::GroupAura aura;
531
532 aura.SpellId = aurApp->GetBase()->GetId();
533 aura.EffectMask = aurApp->GetEffectMask();
534 aura.Scalings = aurApp->GetFlags(); // ??
535
536 if (aurApp->GetFlags() & AFLAG_SCALABLE)
537 {
538 for (uint32 e = 0; e < MAX_SPELL_EFFECTS; ++e)
539 {
540 float scale = 0.0f;
541 if (AuraEffect const* eff = aurApp->GetBase()->GetEffect(e))
542 scale = float(eff->GetAmount());
543 aura.EffectScales.push_back(scale);
544 }
545 }
546
547 MemberStats.PetStats->AuraList.push_back(aura);
548 }
549 }
550 }
551}

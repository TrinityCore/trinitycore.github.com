23}

25WorldPacket const* WorldPackets::Token::UpdateListedAuctionableTokensResponse::Write()
26{
27 _worldPacket << UnkInt;
28 _worldPacket << Result;
29 _worldPacket << uint32(AuctionableTokenAuctionableList.size());
30 for (AuctionableTokenAuctionable const& auctionableTokenAuctionable : AuctionableTokenAuctionableList)
31 {
32 _worldPacket << auctionableTokenAuctionable.UnkInt1;
33 _worldPacket << auctionableTokenAuctionable.UnkInt2;
34 _worldPacket << auctionableTokenAuctionable.Owner;
35 _worldPacket << auctionableTokenAuctionable.BuyoutPrice;
36 _worldPacket << auctionableTokenAuctionable.EndTime;
37 }
38
39 return &_worldPacket;
40}

28WorldPacket const* WorldPackets::EquipmentSet::LoadEquipmentSet::Write()
29{
30 _worldPacket << uint32(SetData.size());
31
32 for (EquipmentSetInfo::EquipmentSetData const* equipSet : SetData)
33 {
34 _worldPacket << uint64(equipSet->Guid);
35 _worldPacket << uint32(equipSet->SetID);
36 _worldPacket << uint32(equipSet->IgnoreMask);
37
38 for (ObjectGuid const& guid : equipSet->Pieces)
39 _worldPacket << guid;
40
41 _worldPacket.WriteBits(equipSet->SetName.length(), 8);
42 _worldPacket.WriteBits(equipSet->SetIcon.length(), 9);
43 _worldPacket.FlushBits();
44
45 _worldPacket.WriteString(equipSet->SetName);
46 _worldPacket.WriteString(equipSet->SetIcon);
47 }
48
49 return &_worldPacket;
50}

52void WorldPackets::EquipmentSet::SaveEquipmentSet::Read()
53{
54 _worldPacket >> Set.Guid;
55 _worldPacket >> Set.SetID;
56 _worldPacket >> Set.IgnoreMask;
57
58 for (uint8 i = 0; i < EQUIPMENT_SLOT_END; ++i)
59 _worldPacket >> Set.Pieces[i];
60
61 uint32 setNameLength = _worldPacket.ReadBits(8);
62 uint32 setIconLength = _worldPacket.ReadBits(9);
63
64 Set.SetName = _worldPacket.ReadString(setNameLength);
65 Set.SetIcon = _worldPacket.ReadString(setIconLength);
66}

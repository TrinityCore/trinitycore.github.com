29WorldPacket const* WorldPackets::Guild::QueryGuildInfoResponse::Write()
30{
31 _worldPacket << GuildGuid;
32 _worldPacket.WriteBit(Info.is_initialized());
33 _worldPacket.FlushBits();
34
35 if (Info)
36 {
37 _worldPacket << Info->GuildGUID;
38 _worldPacket << uint32(Info->VirtualRealmAddress);
39 _worldPacket << uint32(Info->Ranks.size());
40 _worldPacket << uint32(Info->EmblemStyle);
41 _worldPacket << uint32(Info->EmblemColor);
42 _worldPacket << uint32(Info->BorderStyle);
43 _worldPacket << uint32(Info->BorderColor);
44 _worldPacket << uint32(Info->BackgroundColor);
45
46 for (GuildInfo::GuildInfoRank const& rank : Info->Ranks)
47 {
48 _worldPacket << uint32(rank.RankID);
49 _worldPacket << uint32(rank.RankOrder);
50
51 _worldPacket.WriteBits(rank.RankName.size(), 7);
52 _worldPacket.FlushBits();
53
54 _worldPacket.WriteString(rank.RankName);
55 }
56
57 _worldPacket.WriteBits(Info->GuildName.size(), 7);
58 _worldPacket.FlushBits();
59
60 _worldPacket.WriteString(Info->GuildName);
61 }
62
63 return &_worldPacket;
64}

596WorldPacket const* WorldPackets::Guild::GuildBankQueryResults::Write()
597{
598 _worldPacket << Money;
599 _worldPacket << Tab;
600 _worldPacket << WithdrawalsRemaining;
601 _worldPacket << uint32(TabInfo.size());
602 _worldPacket << uint32(ItemInfo.size());
603
604 for (GuildBankTabInfo const& tab : TabInfo)
605 {
606 _worldPacket << tab.TabIndex;
607 _worldPacket.WriteBits(tab.Name.length(), 7);
608 _worldPacket.WriteBits(tab.Icon.length(), 9);
609 _worldPacket.FlushBits();
610
611 _worldPacket.WriteString(tab.Name);
612 _worldPacket.WriteString(tab.Icon);
613 }
614
615 for (GuildBankItemInfo const& item : ItemInfo)
616 {
617 _worldPacket << item.Slot;
618 _worldPacket << item.Item;
619 _worldPacket << item.Count;
620 _worldPacket << item.EnchantmentID;
621 _worldPacket << item.Charges;
622 _worldPacket << item.OnUseEnchantmentID;
623 _worldPacket << uint32(item.SocketEnchant.size());
624 _worldPacket << item.Flags;
625
626 for (GuildBankItemInfo::GuildBankSocketEnchant const& socketEnchant : item.SocketEnchant)
627 {
628 _worldPacket << socketEnchant.SocketIndex;
629 _worldPacket << socketEnchant.SocketEnchantID;
630 }
631
632 _worldPacket.WriteBit(item.Locked);
633 _worldPacket.FlushBits();
634 }
635
636 _worldPacket.WriteBit(FullUpdate);
637 _worldPacket.FlushBits();
638
639 return &_worldPacket;
640}

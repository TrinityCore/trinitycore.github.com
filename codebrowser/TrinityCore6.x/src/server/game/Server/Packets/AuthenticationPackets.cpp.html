29void WorldPackets::Auth::AuthSession::Read()
30{
31 uint32 addonDataSize;
32
33 _worldPacket >> LoginServerID;
34 _worldPacket >> Build;
35 _worldPacket >> RegionID;
36 _worldPacket >> BattlegroupID;
37 _worldPacket >> RealmID;
38 _worldPacket >> LoginServerType;
39 _worldPacket >> BuildType;
40 _worldPacket >> LocalChallenge;
41 _worldPacket >> DosResponse;
42 _worldPacket.read(Digest, SHA_DIGEST_LENGTH);
43 Account = _worldPacket.ReadString(_worldPacket.ReadBits(11));
44 UseIPv6 = _worldPacket.ReadBit(); // UseIPv6
45 _worldPacket >> addonDataSize;
46 if (addonDataSize)
47 {
48 AddonInfo.resize(addonDataSize);
49 _worldPacket.read(AddonInfo.contents(), addonDataSize);
50 }

58WorldPacket const* WorldPackets::Auth::AuthResponse::Write()
59{
60 _worldPacket << uint8(Result);
61 _worldPacket.WriteBit(SuccessInfo.is_initialized());
62 _worldPacket.WriteBit(WaitInfo.is_initialized());
63 _worldPacket.FlushBits();
64
65 if (SuccessInfo)
66 {
67 _worldPacket << uint32(SuccessInfo->VirtualRealmAddress);
68 _worldPacket << uint32(SuccessInfo->VirtualRealms.size());
69 _worldPacket << uint32(SuccessInfo->TimeRemain);
70 _worldPacket << uint32(SuccessInfo->TimeOptions);
71 _worldPacket << uint32(SuccessInfo->TimeRested);
72 _worldPacket << uint8(SuccessInfo->ActiveExpansionLevel);
73 _worldPacket << uint8(SuccessInfo->AccountExpansionLevel);
74 _worldPacket << uint32(SuccessInfo->TimeSecondsUntilPCKick);
75 _worldPacket << uint32(SuccessInfo->AvailableRaces->size());
76 _worldPacket << uint32(SuccessInfo->AvailableClasses->size());
77 _worldPacket << uint32(SuccessInfo->Templates.size());
78 _worldPacket << uint32(SuccessInfo->CurrencyID);
79
80 for (auto& virtualRealm : SuccessInfo->VirtualRealms)
81 {
82 _worldPacket << uint32(virtualRealm.RealmAddress);
83 _worldPacket.WriteBit(virtualRealm.IsLocal);
84 _worldPacket.WriteBit(virtualRealm.IsInternalRealm);
85 _worldPacket.WriteBits(virtualRealm.RealmNameActual.length(), 8);
86 _worldPacket.WriteBits(virtualRealm.RealmNameNormalized.length(), 8);
87 _worldPacket.FlushBits();
88
89 _worldPacket.WriteString(virtualRealm.RealmNameActual);
90 _worldPacket.WriteString(virtualRealm.RealmNameNormalized);
91 }
92
93 for (auto& race : *SuccessInfo->AvailableRaces)
94 {
95 _worldPacket << uint8(race.first); /// the current race
96 _worldPacket << uint8(race.second); /// the required Expansion
97 }
98
99 for (auto& klass : *SuccessInfo->AvailableClasses)
100 {
101 _worldPacket << uint8(klass.first); /// the current class
102 _worldPacket << uint8(klass.second); /// the required Expansion
103 }
104
105 for (auto& templat : SuccessInfo->Templates)
106 {
107 _worldPacket << uint32(templat.TemplateSetId);
108 _worldPacket << uint32(templat.Classes.size());
109 for (auto& templateClass : templat.Classes)
110 {
111 _worldPacket << uint8(templateClass.ClassID);
112 _worldPacket << uint8(templateClass.FactionGroup);
113 }
114
115 _worldPacket.WriteBits(templat.Name.length(), 7);
116 _worldPacket.WriteBits(templat.Description.length(), 10);
117 _worldPacket.FlushBits();
118
119 _worldPacket.WriteString(templat.Name);
120 _worldPacket.WriteString(templat.Description);
121 }
122
123 _worldPacket.WriteBit(SuccessInfo->IsExpansionTrial);
124 _worldPacket.WriteBit(SuccessInfo->ForceCharacterTemplate);
125 _worldPacket.WriteBit(SuccessInfo->NumPlayersHorde.is_initialized());
126 _worldPacket.WriteBit(SuccessInfo->NumPlayersAlliance.is_initialized());
127 _worldPacket.WriteBit(SuccessInfo->IsVeteranTrial);
128 _worldPacket.FlushBits();
129
130 if (SuccessInfo->NumPlayersHorde)
131 _worldPacket << uint16(*SuccessInfo->NumPlayersHorde);
132
133 if (SuccessInfo->NumPlayersAlliance)
134 _worldPacket << uint16(*SuccessInfo->NumPlayersAlliance);
135 }
136
137 if (WaitInfo)
138 {
139 _worldPacket << uint32(WaitInfo->WaitCount);
140 _worldPacket.WriteBit(WaitInfo->HasFCM);
141 _worldPacket.FlushBits();
142 }
143
144 return &_worldPacket;
145}

34WorldPackets::Combat::SAttackStop::SAttackStop(Unit const* attacker, Unit const* victim) : ServerPacket(SMSG_ATTACK_STOP, 16 + 16 + 1)
35{
36 Attacker = attacker->GetGUID();
37 if (victim)
38 {
39 Victim = victim->GetGUID();
40 NowDead = victim->isDead();
41 }
42}
43
44WorldPacket const* WorldPackets::Combat::SAttackStop::Write()
45{
46 _worldPacket << Attacker;
47 _worldPacket << Victim;
48 _worldPacket.WriteBit(NowDead);
49 _worldPacket.FlushBits();
50
51 return &_worldPacket;
52}

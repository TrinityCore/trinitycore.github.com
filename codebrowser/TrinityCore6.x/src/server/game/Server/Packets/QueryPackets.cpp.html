29WorldPacket const* WorldPackets::Query::QueryCreatureResponse::Write()
30{
31 _worldPacket << CreatureID;
32 _worldPacket.WriteBit(Allow);
33
34 _worldPacket.FlushBits();
35
36 if (Allow)
37 {
38 _worldPacket.WriteBits(Stats.Title.length() + 1, 11);
39 _worldPacket.WriteBits(Stats.TitleAlt.length() + 1, 11);
40 _worldPacket.WriteBits(Stats.CursorName.length() + 1, 6);
41 _worldPacket.WriteBit(Stats.Leader);
42
43 for (uint32 i = 0; i < MAX_CREATURE_NAMES; ++i)
44 {
45 _worldPacket.WriteBits(Stats.Name[i].length() + 1, 11);
46 _worldPacket.WriteBits(Stats.NameAlt[i].length() + 1, 11);
47 }
48
49 for (uint32 i = 0; i < MAX_CREATURE_NAMES; ++i)
50 {
51 if (!Stats.Name[i].empty())
52 _worldPacket << Stats.Name[i];
53
54 if (!Stats.NameAlt[i].empty())
55 _worldPacket << Stats.NameAlt[i];
56 }
57
58 for (uint8 i = 0; i < 2; ++i)
59 _worldPacket << Stats.Flags[i];
60
61 _worldPacket << Stats.CreatureType;
62 _worldPacket << Stats.CreatureFamily;
63 _worldPacket << Stats.Classification;
64
65 for (uint32 i = 0; i < MAX_KILL_CREDIT; ++i)
66 _worldPacket << Stats.ProxyCreatureID[i];
67
68 for (uint32 i = 0; i < MAX_CREATURE_MODELS; ++i)
69 _worldPacket << Stats.CreatureDisplayID[i];
70
71 _worldPacket << Stats.HpMulti;
72 _worldPacket << Stats.EnergyMulti;
73
74 _worldPacket << int32(Stats.QuestItems.size());
75 _worldPacket << Stats.CreatureMovementInfoID;
76 _worldPacket << Stats.RequiredExpansion;
77
78 _worldPacket << int32(0); // FlagQuest
79
80 if (!Stats.Title.empty())
81 _worldPacket << Stats.Title;
82
83 if (!Stats.TitleAlt.empty())
84 _worldPacket << Stats.TitleAlt;
85
86 if (!Stats.CursorName.empty())
87 _worldPacket << Stats.CursorName;
88
89 for (int32 questItem : Stats.QuestItems)
90 _worldPacket << questItem;
91 }
92
93 return &_worldPacket;
94}

101ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::Query::PlayerGuidLookupHint const& lookupHint)
102{
103 data.WriteBit(lookupHint.VirtualRealmAddress.is_initialized());
104 data.WriteBit(lookupHint.NativeRealmAddress.is_initialized());
105 data.FlushBits();
106
107 if (lookupHint.VirtualRealmAddress)
108 data << uint32(*lookupHint.VirtualRealmAddress);
109
110 if (lookupHint.NativeRealmAddress)
111 data << uint32(*lookupHint.NativeRealmAddress);
112
113 return data;

116bool WorldPackets::Query::PlayerGuidLookupData::Initialize(ObjectGuid const& guid, Player const* player /*= nullptr*/)
117{
118 CharacterInfo const* characterInfo = sWorld->GetCharacterInfo(guid);
119 if (!characterInfo)
120 return false;
121
122 if (player)
123 {
124 ASSERT(player->GetGUID() == guid);
125
126 AccountID = player->GetSession()->GetAccountGUID();
127 BnetAccountID = player->GetSession()->GetBattlenetAccountGUID();
128 Name = player->GetName();
129 Race = player->getRace();
130 Sex = player->GetByteValue(PLAYER_BYTES_3, PLAYER_BYTES_3_OFFSET_GENDER);
131 ClassID = player->getClass();
132 Level = player->getLevel();
133
134 if (DeclinedName const* names = player->GetDeclinedNames())
135 DeclinedNames = *names;
136 }
137 else
138 {
139 uint32 accountId = ObjectMgr::GetPlayerAccountIdByGUID(guid);
140 uint32 bnetAccountId = Battlenet::AccountMgr::GetIdByGameAccount(accountId);
141
142 AccountID = ObjectGuid::Create<HighGuid::WowAccount>(accountId);
143 BnetAccountID = ObjectGuid::Create<HighGuid::BNetAccount>(bnetAccountId);
144 Name = characterInfo->Name;
145 Race = characterInfo->Race;
146 Sex = characterInfo->Sex;
147 ClassID = characterInfo->Class;
148 Level = characterInfo->Level;
149 }
150
151 IsDeleted = characterInfo->IsDeleted;
152 GuidActual = guid;
153 VirtualRealmAddress = GetVirtualRealmAddress();
154
155 return true;
156}

158ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::Query::PlayerGuidLookupData const& lookupData)
159{
160 data.WriteBit(lookupData.IsDeleted);
161 data.WriteBits(lookupData.Name.length(), 6);
162
163 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
164 data.WriteBits(lookupData.DeclinedNames.name[i].length(), 7);
165
166 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
167 data.WriteString(lookupData.DeclinedNames.name[i]);
168
169 data << lookupData.AccountID;
170 data << lookupData.BnetAccountID;
171 data << lookupData.GuidActual;
172 data << uint32(lookupData.VirtualRealmAddress);
173 data << uint8(lookupData.Race);
174 data << uint8(lookupData.Sex);
175 data << uint8(lookupData.ClassID);
176 data << uint8(lookupData.Level);
177 data.WriteString(lookupData.Name);
178
179 return data;
180}

290WorldPacket const* WorldPackets::Query::QueryGameObjectResponse::Write()
291{
292 _worldPacket << GameObjectID;
293 _worldPacket.WriteBit(Allow);
294 _worldPacket.FlushBits();
295
296 if (Allow)
297 {
298 uint32 dataSize = Stats.GetDataSize();
299
300 _worldPacket << uint32(dataSize);
301 if (dataSize)
302 {
303 _worldPacket << Stats.Type;
304 _worldPacket << Stats.DisplayID;
305 for (int8 i = 0; i < 4; i++)
306 _worldPacket << Stats.Name[i];
307
308 _worldPacket << Stats.IconName;
309 _worldPacket << Stats.CastBarCaption;
310 _worldPacket << Stats.UnkString;
311
312 for (uint32 i = 0; i < MAX_GAMEOBJECT_DATA; i++)
313 _worldPacket << Stats.Data[i];
314
315 _worldPacket << Stats.Size;
316
317 _worldPacket << uint8(Stats.QuestItems.size());
318 for (int32 questItem : Stats.QuestItems)
319 _worldPacket << questItem;
320
321 _worldPacket << Stats.Expansion;
322 }
323 }
324
325 return &_worldPacket;
326}

374WorldPacket const* WorldPackets::Query::QuestPOIQueryResponse::Write()
375{
376 _worldPacket << int32(QuestPOIDataStats.size());
377 _worldPacket << int32(QuestPOIDataStats.size());
378
379 for (QuestPOIData const& questPOIData : QuestPOIDataStats)
380 {
381 _worldPacket << int32(questPOIData.QuestID);
382
383 _worldPacket << int32(questPOIData.QuestPOIBlobDataStats.size());
384
385 for (QuestPOIBlobData const& questPOIBlobData : questPOIData.QuestPOIBlobDataStats)
386 {
387 _worldPacket << int32(questPOIBlobData.BlobIndex);
388 _worldPacket << int32(questPOIBlobData.ObjectiveIndex);
389 _worldPacket << int32(questPOIBlobData.QuestObjectiveID);
390 _worldPacket << int32(questPOIBlobData.QuestObjectID);
391 _worldPacket << int32(questPOIBlobData.MapID);
392 _worldPacket << int32(questPOIBlobData.WorldMapAreaID);
393 _worldPacket << int32(questPOIBlobData.Floor);
394 _worldPacket << int32(questPOIBlobData.Priority);
395 _worldPacket << int32(questPOIBlobData.Flags);
396 _worldPacket << int32(questPOIBlobData.WorldEffectID);
397 _worldPacket << int32(questPOIBlobData.PlayerConditionID);
398 _worldPacket << int32(questPOIBlobData.UnkWoD1);
399 _worldPacket << int32(questPOIBlobData.QuestPOIBlobPointStats.size());
400
401 for (QuestPOIBlobPoint const& questPOIBlobPoint : questPOIBlobData.QuestPOIBlobPointStats)
402 {
403 _worldPacket << int32(questPOIBlobPoint.X);
404 _worldPacket << int32(questPOIBlobPoint.Y);
405 }
406 }
407 }
408
409 return &_worldPacket;
410}

443WorldPacket const* WorldPackets::Query::QueryPetNameResponse::Write()
444{
445 _worldPacket << UnitGUID;
446 _worldPacket.WriteBit(Allow);
447
448 if (Allow)
449 {
450 _worldPacket.WriteBits(Name.length(), 8);
451 _worldPacket.WriteBit(HasDeclined);
452
453 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
454 _worldPacket.WriteBits(DeclinedNames.name[i].length(), 7);
455
456 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
457 _worldPacket.WriteString(DeclinedNames.name[i]);
458
459 _worldPacket << Timestamp;
460 _worldPacket.WriteString(Name);
461 }
462
463 _worldPacket.FlushBits();
464
465 return &_worldPacket;

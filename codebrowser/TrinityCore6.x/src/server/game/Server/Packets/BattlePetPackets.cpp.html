21ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::BattlePet::BattlePetSlot const& slot)
22{
23 data << (slot.Pet.Guid.IsEmpty() ? ObjectGuid::Create<HighGuid::BattlePet>(0) : slot.Pet.Guid);
24 data << uint32(slot.CollarID);
25 data << uint8(slot.Index);
26 data.WriteBit(slot.Locked);
27 data.FlushBits();
28
29 return data;
30}

32ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::BattlePet::BattlePet const& pet)
33{
34 data << pet.Guid;
35 data << uint32(pet.Species);
36 data << uint32(pet.CreatureID);
37 data << uint32(pet.CollarID);
38 data << uint16(pet.Breed);
39 data << uint16(pet.Level);
40 data << uint16(pet.Exp);
41 data << uint16(pet.Flags);
42 data << uint32(pet.Power);
43 data << uint32(pet.Health);
44 data << uint32(pet.MaxHealth);
45 data << uint32(pet.Speed);
46 data << uint8(pet.Quality);
47 data.WriteBits(pet.Name.size(), 7);
48 data.WriteBit(!pet.Owner.IsEmpty()); // HasOwnerInfo
49 data.WriteBit(pet.Name.empty()); // NoRename
50 data.FlushBits();
51
52 if (!pet.Owner.IsEmpty())
53 {
54 data << pet.Owner;
55 data << uint32(GetVirtualRealmAddress()); // Virtual
56 data << uint32(GetVirtualRealmAddress()); // Native
57 }
58
59 data.WriteString(pet.Name);
60
61 return data;
62}

64WorldPacket const* WorldPackets::BattlePet::BattlePetJournal::Write()
65{
66 _worldPacket << uint16(Trap);
67 _worldPacket << uint32(Slots.size());
68 _worldPacket << uint32(Pets.size());
69
70 for (auto const& slot : Slots)
71 _worldPacket << slot;
72
73 for (auto const& pet : Pets)
74 _worldPacket << pet;
75
76 _worldPacket.WriteBit(HasJournalLock);
77 _worldPacket.FlushBits();
78
79 return &_worldPacket;
80}

116void WorldPackets::BattlePet::BattlePetModifyName::Read()
117{
118 _worldPacket >> PetGuid;
119 uint32 nameLength = _worldPacket.ReadBits(7);
120 bool hasDeclinedNames = _worldPacket.ReadBit();
121 Name = _worldPacket.ReadString(nameLength);
122
123 if (hasDeclinedNames)
124 {
125 uint8 declinedNameLengths[MAX_DECLINED_NAME_CASES];
126
127 for (uint8 i = 0; i < 5; ++i)
128 declinedNameLengths[i] = _worldPacket.ReadBits(7);
129
130 for (uint8 i = 0; i < 5; ++i)
131 Declined.name[i] = _worldPacket.ReadString(declinedNameLengths[i]);
132 }
133}

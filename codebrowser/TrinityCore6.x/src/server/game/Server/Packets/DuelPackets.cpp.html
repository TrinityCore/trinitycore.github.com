64WorldPacket const* WorldPackets::Duel::DuelWinner::Write()
65{
66 _worldPacket.WriteBits(BeatenName.size(), 6);
67 _worldPacket.WriteBits(WinnerName.size(), 6);
68 _worldPacket.WriteBit(Fled);
69 _worldPacket << BeatenVirtualRealmAddress;
70 _worldPacket << WinnerVirtualRealmAddress;
71 _worldPacket.WriteString(BeatenName);
72 _worldPacket.WriteString(WinnerName);
73
74 return &_worldPacket;
75}

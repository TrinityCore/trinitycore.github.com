23WorldPackets::Character::EnumCharactersResult::CharacterInfo::CharacterInfo(Field* fields)
24{
25 // 0 1 2 3 4 5 6 7
26 // "SELECT characters.guid, characters.name, characters.race, characters.class, characters.gender, characters.skin, characters.face, characters.hairStyle, "
27 // 8 9 10 11 12 13 14 15
28 // "characters.hairColor, characters.facialStyle, characters.level, characters.zone, characters.map, characters.position_x, characters.position_y, characters.position_z, "
29 // 16 17 18 19 20 21 22
30 // "guild_member.guildid, characters.playerFlags, characters.at_login, character_pet.entry, character_pet.modelid, character_pet.level, characters.equipmentCache, "
31 // 23 24 25
32 // "character_banned.guid, characters.slot, character_declinedname.genitive"
33
34 Guid = ObjectGuid::Create<HighGuid::Player>(fields[0].GetUInt64());
35 Name = fields[1].GetString();
36 Race = fields[2].GetUInt8();
37 Class = fields[3].GetUInt8();
38 Sex = fields[4].GetUInt8();
39 Skin = fields[5].GetUInt8();
40 Face = fields[6].GetUInt8();
41 HairStyle = fields[7].GetUInt8();
42 HairColor = fields[8].GetUInt8();
43 FacialHair = fields[9].GetUInt8();
44 Level = fields[10].GetUInt8();
45 ZoneId = int32(fields[11].GetUInt16());
46 MapId = int32(fields[12].GetUInt16());
47 PreLoadPosition.x = fields[13].GetFloat();
48 PreLoadPosition.y = fields[14].GetFloat();
49 PreLoadPosition.z = fields[15].GetFloat();
50
51 if (ObjectGuid::LowType guildId = fields[16].GetUInt64())
52 GuildGuid = ObjectGuid::Create<HighGuid::Guild>(guildId);
53
54 uint32 playerFlags = fields[17].GetUInt32();
55 uint32 atLoginFlags = fields[18].GetUInt16();
56
57 if (atLoginFlags & AT_LOGIN_RESURRECT)
58 playerFlags &= ~PLAYER_FLAGS_GHOST;
59
60 if (playerFlags & PLAYER_FLAGS_HIDE_HELM)
61 Flags |= CHARACTER_FLAG_HIDE_HELM;
62
63 if (playerFlags & PLAYER_FLAGS_HIDE_CLOAK)
64 Flags |= CHARACTER_FLAG_HIDE_CLOAK;
65
66 if (playerFlags & PLAYER_FLAGS_GHOST)
67 Flags |= CHARACTER_FLAG_GHOST;
68
69 if (atLoginFlags & AT_LOGIN_RENAME)
70 Flags |= CHARACTER_FLAG_RENAME;
71
72 if (fields[23].GetUInt64())
73 Flags |= CHARACTER_FLAG_LOCKED_BY_BILLING;
74
75 if (sWorld->getBoolConfig(CONFIG_DECLINED_NAMES_USED) && !fields[25].GetString().empty())
76 Flags |= CHARACTER_FLAG_DECLINED;
77
78 if (atLoginFlags & AT_LOGIN_CUSTOMIZE)
79 CustomizationFlag = CHAR_CUSTOMIZE_FLAG_CUSTOMIZE;
80 else if (atLoginFlags & AT_LOGIN_CHANGE_FACTION)
81 CustomizationFlag = CHAR_CUSTOMIZE_FLAG_FACTION;
82 else if (atLoginFlags & AT_LOGIN_CHANGE_RACE)
83 CustomizationFlag = CHAR_CUSTOMIZE_FLAG_RACE;
84
85 Flags3 = 0;
86 FirstLogin = (atLoginFlags & AT_LOGIN_FIRST) != 0;
87
88 // show pet at selection character in character list only for non-ghost character
89 if (!(playerFlags & PLAYER_FLAGS_GHOST) && (Class == CLASS_WARLOCK || Class == CLASS_HUNTER || Class == CLASS_DEATH_KNIGHT))
90 {
91 if (CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate(fields[19].GetUInt32()))
92 {
93 Pet.CreatureDisplayId = fields[20].GetUInt32();
94 Pet.Level = fields[21].GetUInt16();
95 Pet.CreatureFamily = creatureInfo->family;
96 }
97 }
98
99 BoostInProgress = false;
100 ProfessionIds[0] = 0;
101 ProfessionIds[1] = 0;
102
103 Tokenizer equipment(fields[22].GetString(), ' ');
104 ListPosition = fields[24].GetUInt8();
105
106 for (uint8 slot = 0; slot < INVENTORY_SLOT_BAG_END; ++slot)
107 {
108 uint32 visualBase = slot * 3;
109 VisualItems[slot].InventoryType = Player::GetUInt32ValueFromArray(equipment, visualBase);
110 VisualItems[slot].DisplayId = Player::GetUInt32ValueFromArray(equipment, visualBase + 1);
111 VisualItems[slot].DisplayEnchantId = Player::GetUInt32ValueFromArray(equipment, visualBase + 2);
112 }
113}

115WorldPacket const* WorldPackets::Character::EnumCharactersResult::Write()
116{
117 _worldPacket.reserve(9 + Characters.size() * sizeof(CharacterInfo) + FactionChangeRestrictions.size() * sizeof(RestrictedFactionChangeRuleInfo));
118
119 _worldPacket.WriteBit(Success);
120 _worldPacket.WriteBit(IsDeletedCharacters);
121 _worldPacket << uint32(Characters.size());
122 _worldPacket << uint32(FactionChangeRestrictions.size());
123
124 for (CharacterInfo const& charInfo : Characters)
125 {
126 _worldPacket << charInfo.Guid;
127 _worldPacket << uint8(charInfo.ListPosition);
128 _worldPacket << uint8(charInfo.Race);
129 _worldPacket << uint8(charInfo.Class);
130 _worldPacket << uint8(charInfo.Sex);
131 _worldPacket << uint8(charInfo.Skin);
132 _worldPacket << uint8(charInfo.Face);
133 _worldPacket << uint8(charInfo.HairStyle);
134 _worldPacket << uint8(charInfo.HairColor);
135 _worldPacket << uint8(charInfo.FacialHair);
136 _worldPacket << uint8(charInfo.Level);
137 _worldPacket << int32(charInfo.ZoneId);
138 _worldPacket << int32(charInfo.MapId);
139 _worldPacket << charInfo.PreLoadPosition;
140 _worldPacket << charInfo.GuildGuid;
141 _worldPacket << uint32(charInfo.Flags);
142 _worldPacket << uint32(charInfo.CustomizationFlag);
143 _worldPacket << uint32(charInfo.Flags3);
144 _worldPacket << uint32(charInfo.Pet.CreatureDisplayId);
145 _worldPacket << uint32(charInfo.Pet.Level);
146 _worldPacket << uint32(charInfo.Pet.CreatureFamily);
147
148 _worldPacket << uint32(charInfo.ProfessionIds[0]);
149 _worldPacket << uint32(charInfo.ProfessionIds[1]);
150
151 for (uint8 slot = 0; slot < INVENTORY_SLOT_BAG_END; ++slot)
152 {
153 _worldPacket << uint32(charInfo.VisualItems[slot].DisplayId);
154 _worldPacket << uint32(charInfo.VisualItems[slot].DisplayEnchantId);
155 _worldPacket << uint8(charInfo.VisualItems[slot].InventoryType);
156 }
157
158 _worldPacket.WriteBits(charInfo.Name.length(), 6);
159 _worldPacket.WriteBit(charInfo.FirstLogin);
160 _worldPacket.WriteBit(charInfo.BoostInProgress);
161 _worldPacket.WriteBits(charInfo.unkWod61x, 5);
162 _worldPacket.FlushBits();
163
164 _worldPacket.WriteString(charInfo.Name);
165 }
166
167 for (RestrictedFactionChangeRuleInfo const& rule : FactionChangeRestrictions)
168 {
169 _worldPacket << int32(rule.Mask);
170 _worldPacket << uint8(rule.Race);
171 }
172
173 return &_worldPacket;
174}

176void WorldPackets::Character::CreateCharacter::Read()
177{
178 CreateInfo.reset(new CharacterCreateInfo());
179 uint32 nameLength = _worldPacket.ReadBits(6);
180 bool const hasTemplateSet = _worldPacket.ReadBit();
181
182 _worldPacket >> CreateInfo->Race;
183 _worldPacket >> CreateInfo->Class;
184 _worldPacket >> CreateInfo->Sex;
185 _worldPacket >> CreateInfo->Skin;
186 _worldPacket >> CreateInfo->Face;
187 _worldPacket >> CreateInfo->HairStyle;
188 _worldPacket >> CreateInfo->HairColor;
189 _worldPacket >> CreateInfo->FacialHairStyle;
190 _worldPacket >> CreateInfo->OutfitId;
191 CreateInfo->Name = _worldPacket.ReadString(nameLength);
192 if (hasTemplateSet)
193 CreateInfo->TemplateSet = _worldPacket.read<int32>();
194}

248void WorldPackets::Character::CharRaceOrFactionChange::Read()
249{
250 RaceOrFactionChangeInfo.reset(new CharRaceOrFactionChangeInfo());
251
252 RaceOrFactionChangeInfo->FactionChange = _worldPacket.ReadBit();
253
254 uint32 nameLength = _worldPacket.ReadBits(6);
255
256 bool const hasSkinID = _worldPacket.ReadBit();
257 bool const hasHairColorID = _worldPacket.ReadBit();
258 bool const hasHairStyleID = _worldPacket.ReadBit();
259 bool const hasFacialHairStyleID = _worldPacket.ReadBit();
260 bool const hasFaceID = _worldPacket.ReadBit();
261
262 _worldPacket >> RaceOrFactionChangeInfo->Guid;
263 _worldPacket >> RaceOrFactionChangeInfo->SexID;
264 _worldPacket >> RaceOrFactionChangeInfo->RaceID;
265
266 RaceOrFactionChangeInfo->Name = _worldPacket.ReadString(nameLength);
267
268 if (hasSkinID)
269 RaceOrFactionChangeInfo->SkinID = _worldPacket.read<uint8>();
270
271 if (hasHairColorID)
272 RaceOrFactionChangeInfo->HairColorID = _worldPacket.read<uint8>();
273
274 if (hasHairStyleID)
275 RaceOrFactionChangeInfo->HairStyleID = _worldPacket.read<uint8>();
276
277 if (hasFacialHairStyleID)
278 RaceOrFactionChangeInfo->FacialHairStyleID = _worldPacket.read<uint8>();
279
280 if (hasFaceID)
281 RaceOrFactionChangeInfo->FaceID = _worldPacket.read<uint8>();
282}

284WorldPacket const* WorldPackets::Character::CharFactionChangeResult::Write()
285{
286 _worldPacket << uint8(Result);
287 _worldPacket << Guid;
288 _worldPacket.WriteBit(Display.is_initialized());
289 _worldPacket.FlushBits();
290
291 if (Display)
292 {
293 _worldPacket.WriteBits(Display->Name.length(), 6);
294 _worldPacket << uint8(Display->SexID);
295 _worldPacket << uint8(Display->SkinID);
296 _worldPacket << uint8(Display->HairColorID);
297 _worldPacket << uint8(Display->HairStyleID);
298 _worldPacket << uint8(Display->FacialHairStyleID);
299 _worldPacket << uint8(Display->FaceID);
300 _worldPacket << uint8(Display->RaceID);
301 _worldPacket.WriteString(Display->Name);
302 }
303
304 return &_worldPacket;
305}

549void WorldPackets::Character::SetPlayerDeclinedNames::Read()
550{
551 _worldPacket >> Player;
552
553 uint8 stringLengths[MAX_DECLINED_NAME_CASES];
554
555 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
556 stringLengths[i] = _worldPacket.ReadBits(7);
557
558 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
559 DeclinedNames.name[i] = _worldPacket.ReadString(stringLengths[i]);
560}

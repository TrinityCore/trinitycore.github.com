20WorldPacket const* WorldPackets::Scenes::PlayScene::Write()
21{
22 _worldPacket << SceneID;
23 _worldPacket << PlaybackFlags;
24 _worldPacket << SceneInstanceID;
25 _worldPacket << SceneScriptPackageID;
26 _worldPacket << TransportGUID;
27 _worldPacket << Location.PositionXYZOStream();
28
29 return &_worldPacket;
30}

32void WorldPackets::Scenes::SceneTriggerEvent::Read()

41WorldPacket const* WorldPackets::Ticket::GMTicketCaseStatus::Write()
42{
43 _worldPacket << int32(Cases.size());
44
45 for (auto const& c : Cases)
46 {
47 _worldPacket << int32(c.CaseID);
48 _worldPacket << int32(c.CaseOpened);
49 _worldPacket << int32(c.CaseStatus);
50 _worldPacket << int16(c.CfgRealmID);
51 _worldPacket << int64(c.CharacterID);
52 _worldPacket << int32(c.WaitTimeOverrideMinutes);
53
54 _worldPacket.WriteBits(c.Url.size(), 11);
55 _worldPacket.WriteBits(c.WaitTimeOverrideMessage.size(), 10);
56 }
57
58 _worldPacket.FlushBits();
59 return &_worldPacket;
60}

97ByteBuffer& operator>>(ByteBuffer& data, WorldPackets::Ticket::SupportTicketSubmitComplaint::SupportTicketChatLog& chatlog)
98{
99 uint32 linesCount = data.read<uint32>();
100
101 for (uint32 i = 0; i < linesCount; i++)
102 chatlog.Lines.emplace_back(data);
103
104 bool hasReportLineIndex = data.ReadBit();
105 if (hasReportLineIndex)
106 chatlog.ReportLineIndex = data.read<uint32>();
107
108 data.ResetBitPos();
109
110 return data;
111}

189void WorldPackets::Ticket::SupportTicketSubmitComplaint::Read()
190{
191 _worldPacket >> Header;
192 _worldPacket >> ChatLog;
193 _worldPacket >> TargetCharacterGUID;
194 ComplaintType = _worldPacket.ReadBits(5);
195
196 uint32 noteLength = _worldPacket.ReadBits(10);
197 bool hasMailInfo = _worldPacket.ReadBit();
198 bool hasCalendarInfo = _worldPacket.ReadBit();
199 bool hasPetInfo = _worldPacket.ReadBit();
200 bool hasGuildInfo = _worldPacket.ReadBit();
201 bool hasLFGListSearchResult = _worldPacket.ReadBit();
202 bool hasLFGListApplicant = _worldPacket.ReadBit();
203
204 _worldPacket.ResetBitPos();
205
206 Note = _worldPacket.ReadString(noteLength);
207
208 if (hasMailInfo)
209 _worldPacket >> MailInfo;
210
211 if (hasCalendarInfo)
212 _worldPacket >> CalenderInfo;
213
214 if (hasPetInfo)
215 _worldPacket >> PetInfo;
216
217 if (hasGuildInfo)
218 _worldPacket >> GuildInfo;
219
220 if (hasLFGListSearchResult)
221 _worldPacket >> LFGListSearchResult;
222
223 if (hasLFGListApplicant)
224 _worldPacket >> LFGListApplicant;
225}

245void WorldPackets::Ticket::Complaint::Read()
246{
247 _worldPacket >> ComplaintType;
248 _worldPacket >> Offender;
249
250 switch (ComplaintType)
251 {
252 case SUPPORT_SPAM_TYPE_MAIL:
253 _worldPacket >> MailID;
254 break;
255 case SUPPORT_SPAM_TYPE_CHAT:
256 _worldPacket >> Chat;
257 break;
258 case SUPPORT_SPAM_TYPE_CALENDAR:
259 _worldPacket >> EventGuid;
260 _worldPacket >> InviteGuid;
261 break;
262 default:
263 break;
264 }
265}

40WorldPacket const* WorldPackets::ClientConfig::AddonInfo::Write()
41{
42 _worldPacket << uint32(Addons ? Addons->size() : 0);
43 _worldPacket << uint32(BannedAddons ? BannedAddons->size() : 0);
44
45 if (Addons)
46 {
47 for (::AddonInfo const& addon : *Addons)
48 {
49 bool KeyProvided = addon.UsePublicKeyOrCRC && addon.CRC != STANDARD_ADDON_CRC;
50 _worldPacket << uint8(addon.Status);
51
52 _worldPacket.WriteBit(addon.Enabled); // InfoProvided
53 _worldPacket.WriteBit(KeyProvided); // KeyProvided
54 _worldPacket.WriteBit(0); // UrlProvided
55
56 if (addon.Enabled)
57 {
58 _worldPacket << uint8(1); // KeyVersion
59 _worldPacket << uint32(0); // Revision
60 }
61
62 if (KeyProvided)
63 _worldPacket.append(PublicKey, sizeof(PublicKey)); // KeyData
64 }
65 }
66
67 if (BannedAddons)
68 {
69 for (BannedAddon const& addon : *BannedAddons)
70 {
71 _worldPacket << uint32(addon.Id); // Id
72 _worldPacket.append(addon.NameMD5, MD5_DIGEST_LENGTH); // NameMD5 - MD5 of addon .toc filename without extension
73 _worldPacket.append(addon.VersionMD5, MD5_DIGEST_LENGTH); // VersionMD5 - MD5 of version string declared in addon .toc file
74 _worldPacket << uint32(addon.Timestamp); // LastModified
75 _worldPacket << uint32(1); // Flags
76 }
77 }
78
79 return &_worldPacket;
80}

105{

116void WorldPackets::ClientConfig::UserClientUpdateAccountData::Read()
117{
118 _worldPacket >> PlayerGuid;
119 _worldPacket >> Time;
120 _worldPacket >> Size;
121 DataType = _worldPacket.ReadBits(3);
122
123 uint32 compressedSize = _worldPacket.read<uint32>();
124 if (compressedSize)
125 {
126 CompressedData.resize(compressedSize);
127 _worldPacket.read(CompressedData.contents(), compressedSize);
128 }
129}

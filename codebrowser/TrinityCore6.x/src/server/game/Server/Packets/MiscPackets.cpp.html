73WorldPacket const* WorldPackets::Misc::SetupCurrency::Write()
74{
75 _worldPacket << uint32(Data.size());
76
77 for (Record const& data : Data)
78 {
79 _worldPacket << uint32(data.Type);
80 _worldPacket << uint32(data.Quantity);
81
82 _worldPacket.WriteBit(data.WeeklyQuantity.is_initialized());
83 _worldPacket.WriteBit(data.MaxWeeklyQuantity.is_initialized());
84 _worldPacket.WriteBit(data.TrackedQuantity.is_initialized());
85
86 _worldPacket.WriteBits(data.Flags, 5);
87
88 if (data.WeeklyQuantity)
89 _worldPacket << uint32(*data.WeeklyQuantity);
90 if (data.MaxWeeklyQuantity)
91 _worldPacket << uint32(*data.MaxWeeklyQuantity);
92 if (data.TrackedQuantity)
93 _worldPacket << uint32(*data.TrackedQuantity);
94 }
95
96 _worldPacket.FlushBits();
97
98 return &_worldPacket;
99}

387WorldPacket const* WorldPackets::Misc::PhaseShift::Write()
388{
389 _worldPacket << ClientGUID; // CLientGUID
390 _worldPacket << uint32(PhaseShifts.size() ? 0 : 8); // PhaseShiftFlags
391 _worldPacket << uint32(PhaseShifts.size()); // PhaseShiftCount
392 _worldPacket << PersonalGUID; // PersonalGUID
393 for (uint32 phase : PhaseShifts)
394 {
395 _worldPacket << uint16(1); // PhaseFlags
396 _worldPacket << uint16(phase); // PhaseID
397 }
398
399 _worldPacket << uint32(VisibleMapIDs.size() * 2); // Active terrain swaps size
400 for (uint32 map : VisibleMapIDs)
401 _worldPacket << uint16(map); // Active terrain swap map id
402
403 _worldPacket << uint32(PreloadMapIDs.size() * 2); // Inactive terrain swaps size
404 for (uint32 map : PreloadMapIDs)
405 _worldPacket << uint16(map); // Inactive terrain swap map id
406
407 _worldPacket << uint32(UiWorldMapAreaIDSwaps.size() * 2); // UI map swaps size
408 for (uint32 map : UiWorldMapAreaIDSwaps)
409 _worldPacket << uint16(map); // UI map id, WorldMapArea.dbc, controls map display
410
411 return &_worldPacket;
412}

458void WorldPackets::Misc::SaveCUFProfiles::Read()
459{
460 uint32 count;
461 _worldPacket >> count;
462
463 for (uint8 i = 0; i < count && i < MAX_CUF_PROFILES; i++)
464 {
465 std::unique_ptr<CUFProfile> cufProfile = Trinity::make_unique<CUFProfile>();
466
467 uint8 strLen = _worldPacket.ReadBits(7);
468
469 // Bool Options
470 for (uint8 option = 0; option < CUF_BOOL_OPTIONS_COUNT; option++)
471 cufProfile->BoolOptions.set(option, _worldPacket.ReadBit());
472
473 // Other Options
474 _worldPacket >> cufProfile->FrameHeight;
475 _worldPacket >> cufProfile->FrameWidth;
476
477 _worldPacket >> cufProfile->SortBy;
478 _worldPacket >> cufProfile->HealthText;
479
480 _worldPacket >> cufProfile->TopPoint;
481 _worldPacket >> cufProfile->BottomPoint;
482 _worldPacket >> cufProfile->LeftPoint;
483
484 _worldPacket >> cufProfile->TopOffset;
485 _worldPacket >> cufProfile->BottomOffset;
486 _worldPacket >> cufProfile->LeftOffset;
487
488 cufProfile->ProfileName = _worldPacket.ReadString(strLen);
489
490 CUFProfiles.push_back(std::move(cufProfile));
491 }
492}

494WorldPacket const* WorldPackets::Misc::LoadCUFProfiles::Write()
495{
496 _worldPacket << uint32(CUFProfiles.size());
497
498 for (CUFProfile const* cufProfile : CUFProfiles)
499 {
500 _worldPacket.WriteBits(cufProfile->ProfileName.size(), 7);
501
502 // Bool Options
503 for (uint8 option = 0; option < CUF_BOOL_OPTIONS_COUNT; option++)
504 _worldPacket.WriteBit(cufProfile->BoolOptions[option]);
505
506 // Other Options
507 _worldPacket << cufProfile->FrameHeight;
508 _worldPacket << cufProfile->FrameWidth;
509
510 _worldPacket << cufProfile->SortBy;
511 _worldPacket << cufProfile->HealthText;
512
513 _worldPacket << cufProfile->TopPoint;
514 _worldPacket << cufProfile->BottomPoint;
515 _worldPacket << cufProfile->LeftPoint;
516
517 _worldPacket << cufProfile->TopOffset;
518 _worldPacket << cufProfile->BottomOffset;
519 _worldPacket << cufProfile->LeftOffset;
520
521 _worldPacket.WriteString(cufProfile->ProfileName);
522 }
523
524 return &_worldPacket;
525}

581WorldPacket const* WorldPackets::Misc::AccountHeirloomUpdate::Write()
582{
583 _worldPacket.WriteBit(IsFullUpdate);
584 _worldPacket.FlushBits();
585
586 _worldPacket << int32(Unk);
587
588 // both lists have to have the same size
589 _worldPacket << int32(Heirlooms->size());
590 _worldPacket << int32(Heirlooms->size());
591
592 for (auto const& item : *Heirlooms)
593 _worldPacket << uint32(item.first);
594
595 for (auto const& flags : *Heirlooms)
596 _worldPacket << uint32(flags.second.flags);
597
598 return &_worldPacket;
599}

20WorldPacket const* WorldPackets::System::FeatureSystemStatus::Write()
21{
22 _worldPacket << uint8(ComplaintStatus);
23
24 _worldPacket << uint32(ScrollOfResurrectionRequestsRemaining);
25 _worldPacket << uint32(ScrollOfResurrectionMaxRequestsPerDay);
26 _worldPacket << uint32(CfgRealmID);
27 _worldPacket << int32(CfgRealmRecID);
28 _worldPacket << uint32(TwitterPostThrottleLimit);
29 _worldPacket << uint32(TwitterPostThrottleCooldown);
30 _worldPacket << uint32(TokenPollTimeSeconds);
31 _worldPacket << uint32(TokenRedeemIndex);
32
33 _worldPacket.WriteBit(VoiceEnabled);
34 _worldPacket.WriteBit(EuropaTicketSystemStatus.is_initialized());
35 _worldPacket.WriteBit(ScrollOfResurrectionEnabled);
36 _worldPacket.WriteBit(BpayStoreEnabled);
37 _worldPacket.WriteBit(BpayStoreAvailable);
38 _worldPacket.WriteBit(BpayStoreDisabledByParentalControls);
39 _worldPacket.WriteBit(ItemRestorationButtonEnabled);
40 _worldPacket.WriteBit(BrowserEnabled);
41 _worldPacket.WriteBit(SessionAlert.is_initialized());
42 _worldPacket.WriteBit(RecruitAFriendSendingEnabled);
43 _worldPacket.WriteBit(CharUndeleteEnabled);
44 _worldPacket.WriteBit(RestrictedAccount);
45 _worldPacket.WriteBit(TutorialsEnabled);
46 _worldPacket.WriteBit(NPETutorialsEnabled);
47 _worldPacket.WriteBit(TwitterEnabled);
48 _worldPacket.WriteBit(CommerceSystemEnabled);
49 _worldPacket.WriteBit(Unk67);
50 _worldPacket.WriteBit(WillKickFromWorld);
51 _worldPacket.WriteBit(UnkBit61);
52
53 _worldPacket.FlushBits();
54
55 if (EuropaTicketSystemStatus)
56 {
57 _worldPacket.WriteBit(EuropaTicketSystemStatus->TicketsEnabled);
58 _worldPacket.WriteBit(EuropaTicketSystemStatus->BugsEnabled);
59 _worldPacket.WriteBit(EuropaTicketSystemStatus->ComplaintsEnabled);
60 _worldPacket.WriteBit(EuropaTicketSystemStatus->SuggestionsEnabled);
61
62 _worldPacket << uint32(EuropaTicketSystemStatus->ThrottleState.MaxTries);
63 _worldPacket << uint32(EuropaTicketSystemStatus->ThrottleState.PerMilliseconds);
64 _worldPacket << uint32(EuropaTicketSystemStatus->ThrottleState.TryCount);
65 _worldPacket << uint32(EuropaTicketSystemStatus->ThrottleState.LastResetTimeBeforeNow);
66 }
67
68 if (SessionAlert)
69 {
70 _worldPacket << int32(SessionAlert->Delay);
71 _worldPacket << int32(SessionAlert->Period);
72 _worldPacket << int32(SessionAlert->DisplayTime);
73 }
74
75 /*if (bit61)
76 {
77 var int88 = packet.ReadInt32("int88");
78 for (int i = 0; i < int88; i++)
79 packet.ReadByte("byte23", i);
80 }*/
81
82 _worldPacket.FlushBits();
83
84 return &_worldPacket;
85}

105WorldPacket const* WorldPackets::System::MOTD::Write()
106{
107 ASSERT(Text);
108 _worldPacket.WriteBits(Text->size(), 4);
109 _worldPacket.FlushBits();
110
111 for (std::string const& line : *Text)
112 {
113 _worldPacket.WriteBits(line.length(), 7);
114 _worldPacket.FlushBits();
115 _worldPacket.WriteString(line);
116 }
117
118 return &_worldPacket;
119}

121WorldPacket const* WorldPackets::System::SetTimeZoneInformation::Write()
122{

124 _worldPacket.WriteBits(GameTimeTZ.length(), 7);
125 _worldPacket.FlushBits();
126
127 _worldPacket.WriteString(ServerTimeTZ);
128 _worldPacket.WriteString(GameTimeTZ);
129
130 return &_worldPacket;

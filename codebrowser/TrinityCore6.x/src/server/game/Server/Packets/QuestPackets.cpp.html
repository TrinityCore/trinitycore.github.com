56WorldPacket const* WorldPackets::Quest::QueryQuestInfoResponse::Write()
57{
58 _worldPacket << QuestID;
59
60 _worldPacket.WriteBit(Allow);
61 _worldPacket.FlushBits();
62
63 if (Allow)
64 {
65 _worldPacket << Info.QuestID;
66 _worldPacket << Info.QuestType;
67 _worldPacket << Info.QuestLevel;
68 _worldPacket << Info.QuestPackageID;
69 _worldPacket << Info.QuestMinLevel;
70 _worldPacket << Info.QuestSortID;
71 _worldPacket << Info.QuestInfoID;
72 _worldPacket << Info.SuggestedGroupNum;
73 _worldPacket << Info.RewardNextQuest;
74 _worldPacket << Info.RewardXPDifficulty;
75 _worldPacket << Info.RewardXPMultiplier;
76 _worldPacket << Info.RewardMoney;
77 _worldPacket << Info.RewardMoneyDifficulty;
78 _worldPacket << Info.RewardMoneyMultiplier;
79 _worldPacket << Info.RewardBonusMoney;
80 _worldPacket << Info.RewardDisplaySpell;
81 _worldPacket << Info.RewardSpell;
82 _worldPacket << Info.RewardHonor;
83 _worldPacket << Info.RewardKillHonor;
84 _worldPacket << Info.StartItem;
85 _worldPacket << Info.Flags;
86 _worldPacket << Info.FlagsEx;
87
88 for (uint32 i = 0; i < QUEST_REWARD_ITEM_COUNT; ++i)
89 {
90 _worldPacket << Info.RewardItems[i];
91 _worldPacket << Info.RewardAmount[i];
92 _worldPacket << Info.ItemDrop[i];
93 _worldPacket << Info.ItemDropQuantity[i];
94 }
95
96 for (uint32 i = 0; i < QUEST_REWARD_CHOICES_COUNT; ++i)
97 {
98 _worldPacket << Info.UnfilteredChoiceItems[i].ItemID;
99 _worldPacket << Info.UnfilteredChoiceItems[i].Quantity;
100 _worldPacket << Info.UnfilteredChoiceItems[i].DisplayID;
101 }
102
103 _worldPacket << Info.POIContinent;
104 _worldPacket << Info.POIx;
105 _worldPacket << Info.POIy;
106 _worldPacket << Info.POIPriority;
107
108 _worldPacket << Info.RewardTitle;
109 _worldPacket << Info.RewardTalents;
110 _worldPacket << Info.RewardArenaPoints;
111 _worldPacket << Info.RewardSkillLineID;
112 _worldPacket << Info.RewardNumSkillUps;
113
114 _worldPacket << Info.PortraitGiver;
115 _worldPacket << Info.PortraitTurnIn;
116
117 for (uint32 i = 0; i < QUEST_REWARD_REPUTATIONS_COUNT; ++i)
118 {
119 _worldPacket << Info.RewardFactionID[i];
120 _worldPacket << Info.RewardFactionValue[i];
121 _worldPacket << Info.RewardFactionOverride[i];
122 }
123
124 _worldPacket << Info.RewardFactionFlags;
125
126 for (uint32 i = 0; i < QUEST_REWARD_CURRENCY_COUNT; ++i)
127 {
128 _worldPacket << Info.RewardCurrencyID[i];
129 _worldPacket << Info.RewardCurrencyQty[i];
130 }
131
132 _worldPacket << Info.AcceptedSoundKitID;
133 _worldPacket << Info.CompleteSoundKitID;
134
135 _worldPacket << Info.AreaGroupID;
136 _worldPacket << Info.TimeAllowed;
137
138 _worldPacket << int32(Info.Objectives.size());
139 _worldPacket << int32(Info.AllowableRaces);
140
141 for (QuestObjective const& questObjective : Info.Objectives)
142 {
143 _worldPacket << questObjective.ID;
144 _worldPacket << questObjective.Type;
145 _worldPacket << questObjective.StorageIndex;
146 _worldPacket << questObjective.ObjectID;
147 _worldPacket << questObjective.Amount;
148 _worldPacket << questObjective.Flags;
149 _worldPacket << questObjective.UnkFloat;
150
151 _worldPacket << int32(questObjective.VisualEffects.size());
152 for (int32 visualEffect : questObjective.VisualEffects)
153 _worldPacket << visualEffect;
154
155 _worldPacket.WriteBits(questObjective.Description.size(), 8);
156 _worldPacket.FlushBits();
157
158 _worldPacket.WriteString(questObjective.Description);
159 }
160
161 _worldPacket.WriteBits(Info.LogTitle.size(), 9);
162 _worldPacket.WriteBits(Info.LogDescription.size(), 12);
163 _worldPacket.WriteBits(Info.QuestDescription.size(), 12);
164 _worldPacket.WriteBits(Info.AreaDescription.size(), 9);
165 _worldPacket.WriteBits(Info.PortraitGiverText.size(), 10);
166 _worldPacket.WriteBits(Info.PortraitGiverName.size(), 8);
167 _worldPacket.WriteBits(Info.PortraitTurnInText.size(), 10);
168 _worldPacket.WriteBits(Info.PortraitTurnInName.size(), 8);
169 _worldPacket.WriteBits(Info.QuestCompletionLog.size(), 11);
170 _worldPacket.FlushBits();
171
172 _worldPacket.WriteString(Info.LogTitle);
173 _worldPacket.WriteString(Info.LogDescription);
174 _worldPacket.WriteString(Info.QuestDescription);
175 _worldPacket.WriteString(Info.AreaDescription);
176 _worldPacket.WriteString(Info.PortraitGiverText);
177 _worldPacket.WriteString(Info.PortraitGiverName);
178 _worldPacket.WriteString(Info.PortraitTurnInText);
179 _worldPacket.WriteString(Info.PortraitTurnInName);
180 _worldPacket.WriteString(Info.QuestCompletionLog);
181 }
182
183 return &_worldPacket;
184}

198ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::Quest::QuestRewards const& questRewards)
199{
200 data << questRewards.ChoiceItemCount;
201
202 for (uint32 i = 0; i < QUEST_REWARD_CHOICES_COUNT; ++i)
203 {
204 data << questRewards.ChoiceItems[i].ItemID;
205 data << questRewards.ChoiceItems[i].Quantity;
206 }
207
208 data << questRewards.ItemCount;
209
210 for (uint32 i = 0; i < QUEST_REWARD_ITEM_COUNT; ++i)
211 {
212 data << questRewards.ItemID[i];
213 data << questRewards.ItemQty[i];
214 }
215
216 data << questRewards.Money;
217 data << questRewards.XP;
218 data << questRewards.Title;
219 data << questRewards.Talents;
220 data << questRewards.FactionFlags;
221
222 for (uint32 i = 0; i < QUEST_REWARD_REPUTATIONS_COUNT; ++i)
223 {
224 data << questRewards.FactionID[i];
225 data << questRewards.FactionValue[i];
226 data << questRewards.FactionOverride[i];
227 }
228
229 data << questRewards.SpellCompletionDisplayID;
230 data << questRewards.SpellCompletionID;
231
232 for (uint32 i = 0; i < QUEST_REWARD_CURRENCY_COUNT; ++i)
233 {
234 data << questRewards.CurrencyID[i];
235 data << questRewards.CurrencyQty[i];
236 }
237
238 data << questRewards.SkillLineID;
239 data << questRewards.NumSkillUps;
240
241 data.WriteBit(false); // Unk
242 data.FlushBits();
243
244 return data;
245}

326WorldPacket const* WorldPackets::Quest::QuestGiverQuestDetails::Write()
327{
328 _worldPacket << QuestGiverGUID;
329 _worldPacket << InformUnit;
330 _worldPacket << QuestID;
331 _worldPacket << QuestPackageID;
332 _worldPacket << PortraitGiver;
333 _worldPacket << SuggestedPartyMembers;
334 _worldPacket << QuestFlags[0]; // Flags
335 _worldPacket << QuestFlags[1]; // FlagsEx
336 _worldPacket << PortraitTurnIn;
337 _worldPacket << int32(LearnSpells.size());
338 _worldPacket << Rewards; // WorldPackets::Quest::QuestRewards
339 _worldPacket << int32(DescEmotes.size());
340 _worldPacket << int32(Objectives.size());
341
342 for (int32 spell : LearnSpells)
343 _worldPacket << spell;
344
345 for (WorldPackets::Quest::QuestDescEmote const& emote : DescEmotes)
346 {
347 _worldPacket << emote.Type;
348 _worldPacket << emote.Delay;
349 }
350
351 for (WorldPackets::Quest::QuestObjectiveSimple const& obj : Objectives)
352 {
353 _worldPacket << obj.ID;
354 _worldPacket << obj.ObjectID;
355 _worldPacket << obj.Amount;
356 _worldPacket << obj.Type;
357 }
358
359 _worldPacket.WriteBits(QuestTitle.size(), 9);
360 _worldPacket.WriteBits(DescriptionText.size(), 12);
361 _worldPacket.WriteBits(LogDescription.size(), 12);
362 _worldPacket.WriteBits(PortraitGiverText.size(), 10);
363 _worldPacket.WriteBits(PortraitGiverName.size(), 8);
364 _worldPacket.WriteBits(PortraitTurnInText.size(), 10);
365 _worldPacket.WriteBits(PortraitTurnInName.size(), 8);
366 _worldPacket.WriteBit(DisplayPopup);
367 _worldPacket.WriteBit(StartCheat);
368 _worldPacket.WriteBit(AutoLaunched);
369 _worldPacket.FlushBits();
370
371 _worldPacket.WriteString(QuestTitle);
372 _worldPacket.WriteString(DescriptionText);
373 _worldPacket.WriteString(LogDescription);
374 _worldPacket.WriteString(PortraitGiverText);
375 _worldPacket.WriteString(PortraitGiverName);

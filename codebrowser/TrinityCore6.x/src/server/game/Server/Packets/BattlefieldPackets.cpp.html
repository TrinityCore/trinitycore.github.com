34WorldPacket const* WorldPackets::Battlefield::BFMgrQueueInvite::Write()
35{
36 _worldPacket << uint64(QueueID);
37 _worldPacket << int8(BattleState);
38 _worldPacket << uint32(Timeout);
39 _worldPacket << int32(MinLevel);
40 _worldPacket << int32(MaxLevel);

20WorldPacket const* WorldPackets::Channel::ChannelListResponse::Write()
21{
22 _worldPacket.WriteBit(_Display);
23 _worldPacket.WriteBits(_Channel.length(), 7);
24 _worldPacket << uint32(_ChannelFlags);
25 _worldPacket << uint32(_Members.size());
26 _worldPacket.WriteString(_Channel);
27
28 for (ChannelPlayer const& player : _Members)
29 {
30 _worldPacket << player.Guid;
31 _worldPacket << uint32(player.VirtualRealmAddress);
32 _worldPacket << uint8(player.Flags);
33 }
34
35 return &_worldPacket;
36}

38WorldPacket const* WorldPackets::Channel::ChannelNotify::Write()
39{
40 _worldPacket.WriteBits(Type, 6);
41 _worldPacket.WriteBits(_Channel.length(), 7);
42 _worldPacket.WriteBits(Sender.length(), 6);
43
44 _worldPacket << SenderGuid;
45 _worldPacket << SenderAccountID;
46 _worldPacket << uint32(SenderVirtualRealm);
47 _worldPacket << TargetGuid;
48 _worldPacket << uint32(TargetVirtualRealm);
49 _worldPacket << int32(ChatChannelID);
50
51 if (Type == CHAT_MODE_CHANGE_NOTICE)
52 {
53 _worldPacket << uint8(OldFlags);
54 _worldPacket << uint8(NewFlags);
55 }
56
57 _worldPacket.WriteString(_Channel);
58 _worldPacket.WriteString(Sender);
59
60 return &_worldPacket;
61}

121void WorldPackets::Channel::ChannelPlayerCommand::Read()
122{
123 switch (GetOpcode())
124 {
125 case CMSG_CHAT_CHANNEL_BAN:
126 case CMSG_CHAT_CHANNEL_INVITE:
127 case CMSG_CHAT_CHANNEL_KICK:
128 case CMSG_CHAT_CHANNEL_MODERATOR:
129 case CMSG_CHAT_CHANNEL_MUTE:
130 case CMSG_CHAT_CHANNEL_SET_OWNER:
131 case CMSG_CHAT_CHANNEL_SILENCE_ALL:
132 case CMSG_CHAT_CHANNEL_SILENCE_VOICE:
133 case CMSG_CHAT_CHANNEL_UNBAN:
134 case CMSG_CHAT_CHANNEL_UNMODERATOR:
135 case CMSG_CHAT_CHANNEL_UNMUTE:
136 case CMSG_CHAT_CHANNEL_UNSILENCE_ALL:
137 case CMSG_CHAT_CHANNEL_UNSILENCE_VOICE:
138 {
139 uint32 channelNameLength = _worldPacket.ReadBits(7);
140 uint32 nameLength = _worldPacket.ReadBits(9);
141 ChannelName = _worldPacket.ReadString(channelNameLength);
142 Name = _worldPacket.ReadString(nameLength);
143 break;
144 }
145 case CMSG_CHAT_CHANNEL_ANNOUNCEMENTS:
146 case CMSG_CHAT_CHANNEL_DECLINE_INVITE:
147 case CMSG_CHAT_CHANNEL_DISPLAY_LIST:
148 case CMSG_CHAT_CHANNEL_LIST:
149 case CMSG_CHAT_CHANNEL_MODERATE:
150 case CMSG_CHAT_CHANNEL_OWNER:
151 case CMSG_CHAT_CHANNEL_VOICE_OFF:
152 case CMSG_CHAT_CHANNEL_VOICE_ON:
153 {
154 ChannelName = _worldPacket.ReadString(_worldPacket.ReadBits(7));
155 break;
156 }
157 case CMSG_CHAT_CHANNEL_PASSWORD:
158 {
159 uint32 channelNameLength = _worldPacket.ReadBits(7);
160 uint32 nameLength = _worldPacket.ReadBits(7);
161 ChannelName = _worldPacket.ReadString(channelNameLength);
162 Name = _worldPacket.ReadString(nameLength);
163 break;
164 }
165 default:
166 break;
167 }
168}

170void WorldPackets::Channel::JoinChannel::Read()
171{
172 _worldPacket >> ChatChannelId;
173 CreateVoiceSession = _worldPacket.ReadBit();
174 Internal = _worldPacket.ReadBit();
175 uint32 channelLength = _worldPacket.ReadBits(7);
176 uint32 passwordLength = _worldPacket.ReadBits(7);
177 ChannelName = _worldPacket.ReadString(channelLength);
178 Password = _worldPacket.ReadString(passwordLength);
179}

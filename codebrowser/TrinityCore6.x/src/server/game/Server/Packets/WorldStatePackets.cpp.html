20WorldPackets::WorldState::InitWorldStates::InitWorldStates()
21 : ServerPacket(SMSG_INIT_WORLD_STATES, 16) { }
22
23WorldPacket const* WorldPackets::WorldState::InitWorldStates::Write()
24{
25 _worldPacket.reserve(16 + Worldstates.size() * 8);
26
27 _worldPacket << uint32(MapID);
28 _worldPacket << uint32(AreaID);
29 _worldPacket << uint32(SubareaID);
30
31 _worldPacket << uint32(Worldstates.size());
32 for (WorldStateInfo const& wsi : Worldstates)
33 {
34 _worldPacket << uint32(wsi.VariableID);
35 _worldPacket << int32(wsi.Value);
36 }
37
38 return &_worldPacket;
39}
40
41WorldPackets::WorldState::UpdateWorldState::UpdateWorldState()
42 : ServerPacket(SMSG_UPDATE_WORLD_STATE, 9) { }
43
44WorldPacket const* WorldPackets::WorldState::UpdateWorldState::Write()
45{
46 _worldPacket << uint32(VariableID);
47 _worldPacket << int32(Value);
48 _worldPacket.WriteBit(Hidden);
49 _worldPacket.FlushBits();
50
51 return &_worldPacket;
52}

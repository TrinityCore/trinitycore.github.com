20void WorldPackets::Totem::TotemDestroyed::Read()

26WorldPacket const* WorldPackets::Totem::TotemCreated::Write()
27{
28 _worldPacket << Slot;
29 _worldPacket << Totem;
30 _worldPacket << int32(Duration);
31 _worldPacket << int32(SpellID);
32
33 return &_worldPacket;
34}

36WorldPacket const* WorldPackets::Totem::TotemMoved::Write()
37{
38 _worldPacket << uint8(Slot);
39 _worldPacket << uint8(NewSlot);
40 _worldPacket << Totem;
41
42 return &_worldPacket;
43}

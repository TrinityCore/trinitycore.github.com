20WorldPacket const* WorldPackets::Talent::UpdateTalentData::Write()
21{
22 _worldPacket << Info.ActiveGroup;
23 _worldPacket << uint32(Info.TalentGroups.size());
24
25 for (auto& talentGroupInfo : Info.TalentGroups)
26 {
27 _worldPacket << talentGroupInfo.SpecID;
28 _worldPacket << uint32(talentGroupInfo.TalentIDs.size());
29
30 for (uint32 i = 0; i < MAX_GLYPH_SLOT_INDEX; ++i)
31 _worldPacket << talentGroupInfo.GlyphIDs[i];
32
33 for (uint16 talentID : talentGroupInfo.TalentIDs)
34 _worldPacket << talentID;
35 }
36
37 return &_worldPacket;
38}

46void WorldPackets::Talent::LearnTalents::Read()
47{
48 uint32 count;
49 _worldPacket >> count;
50
51 for (uint32 i = 0; i < count; ++i)
52 {
53 uint16 talent;
54 _worldPacket >> talent;
55 Talents.push_back(talent);
56 }
57}

27 static uint16 const FactionCount = 256;
28
29 class InitializeFactions final : public ServerPacket
30 {
31 public:
32 InitializeFactions() : ServerPacket(SMSG_INITIALIZE_FACTIONS, 1312)
33 {
34 for (uint16 i = 0; i < FactionCount; ++i)
35 {
36 FactionStandings[i] = 0;
37 FactionHasBonus[i] = false;
38 FactionFlags[i] = 0;
39 }
40 }
41
42 WorldPacket const* Write() override;
43
44 int32 FactionStandings[FactionCount];
45 bool FactionHasBonus[FactionCount]; ///< @todo: implement faction bonus
46 uint8 FactionFlags[FactionCount]; ///< @see enum FactionFlags
47 };
48
49 class RequestForcedReactions final : public ClientPacket
50 {
51 public:
52 RequestForcedReactions(WorldPacket&& packet) : ClientPacket(CMSG_REQUEST_FORCED_REACTIONS, std::move(packet)) { }
53
54 void Read() override { }
55 };

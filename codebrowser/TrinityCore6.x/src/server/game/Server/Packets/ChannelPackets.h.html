139 class ChannelPlayerCommand final : public ClientPacket
140 {
141 public:
142 ChannelPlayerCommand(WorldPacket&& packet) : ClientPacket(std::move(packet))
143 {
144 switch (GetOpcode())
145 {
146 default:
147 ABORT();
148 case CMSG_CHAT_CHANNEL_ANNOUNCEMENTS:
149 case CMSG_CHAT_CHANNEL_BAN:
150 case CMSG_CHAT_CHANNEL_DECLINE_INVITE:
151 case CMSG_CHAT_CHANNEL_DISPLAY_LIST:
152 case CMSG_CHAT_CHANNEL_INVITE:
153 case CMSG_CHAT_CHANNEL_KICK:
154 case CMSG_CHAT_CHANNEL_LIST:
155 case CMSG_CHAT_CHANNEL_MODERATE:
156 case CMSG_CHAT_CHANNEL_MODERATOR:
157 case CMSG_CHAT_CHANNEL_MUTE:
158 case CMSG_CHAT_CHANNEL_OWNER:
159 case CMSG_CHAT_CHANNEL_PASSWORD:
160 case CMSG_CHAT_CHANNEL_SET_OWNER:
161 case CMSG_CHAT_CHANNEL_SILENCE_ALL:
162 case CMSG_CHAT_CHANNEL_SILENCE_VOICE:
163 case CMSG_CHAT_CHANNEL_UNBAN:
164 case CMSG_CHAT_CHANNEL_UNMODERATOR:
165 case CMSG_CHAT_CHANNEL_UNMUTE:
166 case CMSG_CHAT_CHANNEL_UNSILENCE_ALL:
167 case CMSG_CHAT_CHANNEL_UNSILENCE_VOICE:
168 case CMSG_CHAT_CHANNEL_VOICE_OFF:
169 case CMSG_CHAT_CHANNEL_VOICE_ON:
170 break;
171 }
172 }
173
174 void Read() override;
175
176 std::string ChannelName;
177 std::string Name;
178 };

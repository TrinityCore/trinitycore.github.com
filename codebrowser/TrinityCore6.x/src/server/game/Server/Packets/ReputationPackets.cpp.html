20WorldPacket const* WorldPackets::Reputation::InitializeFactions::Write()
21{
22 for (uint16 i = 0; i < FactionCount; ++i)
23 {
24 _worldPacket << uint8(FactionFlags[i]);
25 _worldPacket << int32(FactionStandings[i]);
26 }
27
28 for (uint16 i = 0; i < FactionCount; ++i)
29 _worldPacket.WriteBit(FactionHasBonus[i]);
30
31 _worldPacket.FlushBits();
32
33 return &_worldPacket;
34}

43WorldPacket const* WorldPackets::Reputation::SetForcedReactions::Write()
44{
45 _worldPacket.WriteBits(Reactions.size(), 6);
46 for (ForcedReaction const& reaction : Reactions)
47 _worldPacket << reaction;
48
49 _worldPacket.FlushBits();
50
51 return &_worldPacket;
52}

61WorldPacket const* WorldPackets::Reputation::SetFactionStanding::Write()
62{
63 _worldPacket << float(ReferAFriendBonus);
64 _worldPacket << float(BonusFromAchievementSystem);
65 _worldPacket << uint32(Faction.size());
66 for (FactionStandingData const& factionStanding : Faction)
67 _worldPacket << factionStanding;
68
69 _worldPacket.WriteBit(ShowVisual);
70 _worldPacket.FlushBits();
71
72 return &_worldPacket;
73}

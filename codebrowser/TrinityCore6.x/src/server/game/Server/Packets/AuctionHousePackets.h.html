102 class AuctionCommandResult final : public ServerPacket
103 {
104 public:
105 AuctionCommandResult();
106
107 /**
108 * @fn void WorldPackets::AuctionHousePackets::AuctionCommandResult::InitializeAuction(AuctionEntry* auction);
109 *
110 * @brief Initialize the following fields: AuctionId, Bid, AuctionOutBid, Bidder
111 *
112 * @param auction The relevant auction object
113 */
114 void InitializeAuction(::AuctionEntry* auction);
115
116 WorldPacket const* Write() override;
117
118 uint32 AuctionItemID = 0; ///< the id of the auction that triggered this notification
119 uint32 Command = 0; ///< the type of action that triggered this notification. Possible values are @ref AuctionAction
120 int32 ErrorCode = 0; ///< the error code that was generated when trying to perform the action. Possible values are @ref AuctionError
121 uint64 Money = 0; ///< the amount of money that the player bid in copper
122 int32 BagResult = 0; ///< the bid error. Possible values are @ref AuctionError
123 ObjectGuid Guid; ///< the GUID of the bidder for this auction.
124 uint64 MinIncrement = 0; ///< the sum of outbid is (1% of current bid) * 5, if the bid is too small, then this value is 1 copper.
125 };

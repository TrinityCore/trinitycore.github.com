25WorldPacket const* WorldPackets::NPC::GossipMessage::Write()
26{
27 _worldPacket << GossipGUID;
28 _worldPacket << GossipID;
29 _worldPacket << FriendshipFactionID;
30 _worldPacket << TextID;
31
32 _worldPacket << int32(GossipOptions.size());
33 _worldPacket << int32(GossipText.size());
34
35 for (ClientGossipOptions const& options : GossipOptions)
36 {
37 _worldPacket << options.ClientOption;
38 _worldPacket << options.OptionNPC;
39 _worldPacket << options.OptionFlags;
40 _worldPacket << options.OptionCost;
41
42 _worldPacket.WriteBits(options.Text.size(), 12);
43 _worldPacket.WriteBits(options.Confirm.size(), 12);
44 _worldPacket.FlushBits();
45
46 _worldPacket.WriteString(options.Text);
47 _worldPacket.WriteString(options.Confirm);
48 }
49
50 for (ClientGossipText const& text : GossipText)
51 {
52 _worldPacket << text.QuestID;
53 _worldPacket << text.QuestType;
54 _worldPacket << text.QuestLevel;
55 _worldPacket << text.QuestFlags[0];
56 _worldPacket << text.QuestFlags[1];
57
58 _worldPacket.WriteBit(text.Repeatable);
59 _worldPacket.WriteBits(text.QuestTitle.size(), 9);
60 _worldPacket.FlushBits();
61
62 _worldPacket.WriteString(text.QuestTitle);
63 }
64
65 return &_worldPacket;
66}

68WorldPacket const* WorldPackets::NPC::VendorInventory::Write()
69{
70 _worldPacket << Vendor;
71 _worldPacket << Reason;
72
73 _worldPacket << int32(Items.size());
74 for (VendorItem const& item : Items)
75 {
76 _worldPacket << item.MuID;
77 _worldPacket << item.Type;
78 _worldPacket << item.Item;
79 _worldPacket << item.Quantity;
80 _worldPacket << item.Price;
81 _worldPacket << item.Durability;
82 _worldPacket << item.StackCount;
83 _worldPacket << item.ExtendedCostID;
84 _worldPacket << item.PlayerConditionFailed;
85
86 _worldPacket.WriteBit(item.DoNotFilterOnVendor);
87 _worldPacket.FlushBits();
88 }
89
90 return &_worldPacket;
91}

93WorldPacket const* WorldPackets::NPC::TrainerList::Write()
94{
95 _worldPacket << TrainerGUID;
96 _worldPacket << TrainerType;
97 _worldPacket << TrainerID;
98
99 _worldPacket << int32(Spells.size());
100 for (TrainerListSpell const& spell : Spells)
101 {
102 _worldPacket << spell.SpellID;
103 _worldPacket << spell.MoneyCost;
104 _worldPacket << spell.ReqSkillLine;
105 _worldPacket << spell.ReqSkillRank;
106
107 for (uint32 i = 0; i < MAX_TRAINERSPELL_ABILITY_REQS; ++i)
108 _worldPacket << spell.ReqAbility[i];
109
110 _worldPacket << spell.Usable;
111 _worldPacket << spell.ReqLevel;
112 }
113
114 _worldPacket.WriteBits(Greeting.length(), 11);
115 _worldPacket.FlushBits();
116 _worldPacket.WriteString(Greeting);
117
118 return &_worldPacket;
119}

25WorldPacket const* WorldPackets::Loot::LootResponse::Write()
26{
27 _worldPacket << LootObj;
28 _worldPacket << Owner;
29 _worldPacket << FailureReason;
30 _worldPacket << AcquireReason;
31 _worldPacket << LootMethod;
32 _worldPacket << Threshold;
33 _worldPacket << Coins;
34 _worldPacket << uint32(Items.size());
35 _worldPacket << uint32(Currencies.size());
36
37 for (LootItemData const& item : Items)
38 {
39 _worldPacket.WriteBits(item.Type, 2);
40 _worldPacket.WriteBits(item.UIType, 3);
41 _worldPacket.WriteBit(item.CanTradeToTapList);
42 _worldPacket.FlushBits();
43
44 _worldPacket << item.Quantity;
45 _worldPacket << item.LootItemType;
46 _worldPacket << item.LootListID;
47 _worldPacket << item.Loot; // WorldPackets::Item::ItemInstance
48 }
49
50 for (LootCurrency const& currency : Currencies)
51 {
52 _worldPacket << currency.CurrencyID;
53 _worldPacket << currency.Quantity;
54 _worldPacket << currency.LootListID;
55 _worldPacket.WriteBits(currency.UIType, 3);
56 _worldPacket.FlushBits();
57 }
58
59 _worldPacket.WriteBit(Acquired);
60 _worldPacket.WriteBit(AELooting);
61 _worldPacket.WriteBit(PersonalLooting);
62 _worldPacket.FlushBits();
63
64 return &_worldPacket;
65}

67void WorldPackets::Loot::LootItem::Read()
68{
69 uint32 Count;
70 _worldPacket >> Count;
71
72 Loot.resize(Count);
73 for (uint32 i = 0; i < Count; ++i)
74 {
75 _worldPacket >> Loot[i].Object;
76 _worldPacket >> Loot[i].LootListID;
77 }
78}

125WorldPacket const* WorldPackets::Loot::LootList::Write()
126{
127 _worldPacket << Owner;
128
129 _worldPacket.WriteBit(Master.is_initialized());
130 _worldPacket.WriteBit(RoundRobinWinner.is_initialized());
131
132 _worldPacket.FlushBits();
133
134 if (Master)
135 _worldPacket << *Master;
136
137 if (RoundRobinWinner)
138 _worldPacket << *RoundRobinWinner;
139
140 return &_worldPacket;
141}

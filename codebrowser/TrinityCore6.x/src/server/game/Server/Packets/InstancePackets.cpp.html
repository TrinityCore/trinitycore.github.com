34WorldPacket const* WorldPackets::Instance::InstanceInfo::Write()
35{
36 _worldPacket << int32(LockList.size());
37
38 for (InstanceLockInfos const& lockInfos : LockList)
39 _worldPacket << lockInfos;
40
41 return &_worldPacket;
42}
43
44ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::Instance::InstanceLockInfos const& lockInfos)
45{
46 data << lockInfos.MapID;
47 data << lockInfos.DifficultyID;
48 data << lockInfos.InstanceID;
49 data << lockInfos.TimeRemaining;
50 data << lockInfos.CompletedMask;
51
52 data.WriteBit(lockInfos.Locked);
53 data.WriteBit(lockInfos.Extended);
54
55 data.FlushBits();
56
57 return data;
58}

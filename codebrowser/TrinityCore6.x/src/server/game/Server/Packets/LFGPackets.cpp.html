20ByteBuffer& operator>>(ByteBuffer& data, WorldPackets::LFG::RideTicket& ticket)
21{
22 data >> ticket.RequesterGuid;
23 data >> ticket.Id;
24 data >> ticket.Type;
25 data >> ticket.Time;
26
27 return data;
28}

30ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::LFG::RideTicket const& ticket)
31{
32 data << ticket.RequesterGuid;
33 data << int32(ticket.Id);
34 data << int32(ticket.Type);
35 data << uint32(ticket.Time);
36
37 return data;
38}

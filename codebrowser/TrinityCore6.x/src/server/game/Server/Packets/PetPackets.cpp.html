20WorldPacket const* WorldPackets::Pet::PetSpells::Write()
21{
22 _worldPacket << PetGUID;
23 _worldPacket << int16(CreatureFamily);
24 _worldPacket << int16(Specialization);
25 _worldPacket << int32(TimeLimit);
26
27 uint32 petModeAndOrders = ReactState + (CommandState << 8) + (Flag << 16);
28 _worldPacket << uint32(petModeAndOrders);
29
30 for (uint32 actionButton : ActionButtons)
31 _worldPacket << int32(actionButton);
32
33 _worldPacket << int32(Actions.size());
34 _worldPacket << int32(Cooldowns.size());
35 _worldPacket << int32(SpellHistory.size());
36
37 for (uint32 action : Actions)
38 _worldPacket << action;
39
40 for (PetSpellCooldown const& cooldown : Cooldowns)
41 {
42 _worldPacket << int32(cooldown.SpellID);
43 _worldPacket << int32(cooldown.Duration);
44 _worldPacket << int32(cooldown.CategoryDuration);
45 _worldPacket << int16(cooldown.Category);
46 }
47
48 for (PetSpellHistory const& history : SpellHistory)
49 {
50 _worldPacket << int32(history.CategoryID);
51 _worldPacket << int32(history.RecoveryTime);
52 _worldPacket << int8(history.ConsumedCharges);
53 }
54
55 return &_worldPacket;
56}

58WorldPacket const* WorldPackets::Pet::PetStableList::Write()
59{
60 _worldPacket << StableMaster;
61
62 _worldPacket << uint32(Pets.size());
63 for (PetStableInfo const& pet : Pets)
64 {
65 _worldPacket << int32(pet.PetSlot);
66 _worldPacket << int32(pet.PetNumber);
67 _worldPacket << int32(pet.CreatureID);
68 _worldPacket << int32(pet.DisplayID);
69 _worldPacket << int32(pet.ExperienceLevel);
70 _worldPacket << int32(pet.PetFlags);
71
72 _worldPacket << int8(pet.PetName.length());
73 _worldPacket.WriteString(pet.PetName);
74 }
75
76 return &_worldPacket;
77}

95WorldPacket const* WorldPackets::Pet::PetNameInvalid::Write()
96{
97 _worldPacket << RenameData.PetGUID;
98 _worldPacket << int32(RenameData.PetNumber);
99
100 _worldPacket << uint8(RenameData.NewName.length());
101
102 _worldPacket.WriteBit(RenameData.HasDeclinedNames);
103 _worldPacket.FlushBits();
104
105 if (RenameData.HasDeclinedNames)
106 {
107 for (int32 i = 0; i < MAX_DECLINED_NAME_CASES; i++)
108 {
109 _worldPacket.WriteBits(RenameData.DeclinedNames.name[i].length(), 7);
110 _worldPacket.FlushBits();
111 }
112
113 for (int32 i = 0; i < MAX_DECLINED_NAME_CASES; i++)
114 _worldPacket << RenameData.DeclinedNames.name[i];
115 }
116
117 _worldPacket.WriteString(RenameData.NewName);
118 return &_worldPacket;
119}

121void WorldPackets::Pet::PetRename::Read()
122{
123 _worldPacket >> RenameData.PetGUID;
124 _worldPacket >> RenameData.PetNumber;
125
126 int8 nameLen = 0;
127 _worldPacket >> nameLen;
128
129 RenameData.HasDeclinedNames = _worldPacket.ReadBit();
130 if (RenameData.HasDeclinedNames)
131 {
132 int32 count[MAX_DECLINED_NAME_CASES];
133 for (int32 i = 0; i < MAX_DECLINED_NAME_CASES; i++)
134 count[i] = _worldPacket.ReadBits(7);
135
136 for (int32 i = 0; i < MAX_DECLINED_NAME_CASES; i++)
137 RenameData.DeclinedNames.name[i] = _worldPacket.ReadString(count[i]);
138 }
139
140 RenameData.NewName = _worldPacket.ReadString(nameLen);
141}

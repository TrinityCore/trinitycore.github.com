58ByteBuffer& operator<<(ByteBuffer& data, WorldPackets::Garrison::GarrisonFollower const& follower)
59{
60 data << uint64(follower.DbID);
61 data << uint32(follower.GarrFollowerID);
62 data << uint32(follower.Quality);
63 data << uint32(follower.FollowerLevel);
64 data << uint32(follower.ItemLevelWeapon);
65 data << uint32(follower.ItemLevelArmor);
66 data << uint32(follower.Xp);
67 data << uint32(follower.CurrentBuildingID);
68 data << uint32(follower.CurrentMissionID);
69 data << uint32(follower.AbilityID.size());
70 data << uint32(follower.FollowerStatus);
71 for (GarrAbilityEntry const* ability : follower.AbilityID)
72 data << uint32(ability->ID);
73
74 data.WriteBits(follower.CustomName.length(), 7);
75 data.FlushBits();
76 data.WriteString(follower.CustomName);
77
78 return data;
79}

103WorldPacket const* WorldPackets::Garrison::GetGarrisonInfoResult::Write()
104{
105 _worldPacket.reserve(4 + 4 + 4 + 4 + 4 + 4 + 4 + 4 + 4 +
106 Buildings.size() * sizeof(GarrisonBuildingInfo) +
107 Plots.size() * sizeof(GarrisonPlotInfo) +
108 Followers.size() * (sizeof(GarrisonFollower) + 5 * 4) +
109 Missions.size() * sizeof(GarrisonMission) +
110 MissionAreaBonuses.size() * sizeof(GarrisonMissionAreaBonus) +
111 ArchivedMissions.size() * 4 +
112 CanStartMission.size());
113
114 _worldPacket << int32(GarrSiteID);
115 _worldPacket << int32(GarrSiteLevelID);
116 _worldPacket << int32(FactionIndex);
117 _worldPacket << uint32(Buildings.size());
118 _worldPacket << uint32(Plots.size());
119 _worldPacket << uint32(Followers.size());
120 _worldPacket << uint32(Missions.size());
121 _worldPacket << uint32(CanStartMission.size());
122 _worldPacket << uint32(MissionAreaBonuses.size());
123 _worldPacket << uint32(ArchivedMissions.size());
124 _worldPacket << int32(NumFollowerActivationsRemaining);
125
126 for (GarrisonBuildingInfo const* building : Buildings)
127 _worldPacket << *building;
128
129 for (GarrisonPlotInfo* plot : Plots)
130 _worldPacket << *plot;
131
132 for (GarrisonFollower const* follower : Followers)
133 _worldPacket << *follower;
134
135 for (GarrisonMission const* mission : Missions)
136 _worldPacket << *mission;
137
138 for (GarrisonMissionAreaBonus const* areaBonus : MissionAreaBonuses)
139 _worldPacket << *areaBonus;
140
141 if (!ArchivedMissions.empty())
142 _worldPacket.append(ArchivedMissions.data(), ArchivedMissions.size());
143
144 for (bool canStartMission : CanStartMission)
145 _worldPacket.WriteBit(canStartMission);
146
147 _worldPacket.FlushBits();
148
149 return &_worldPacket;
150}

227WorldPacket const* WorldPackets::Garrison::GarrisonRequestBlueprintAndSpecializationDataResult::Write()
228{
229 _worldPacket << uint32(BlueprintsKnown ? BlueprintsKnown->size() : 0);
230 _worldPacket << uint32(SpecializationsKnown ? SpecializationsKnown->size() : 0);
231 if (BlueprintsKnown)
232 for (uint32 blueprint : *BlueprintsKnown)
233 _worldPacket << uint32(blueprint);
234
235 if (SpecializationsKnown)
236 for (uint32 specialization : *SpecializationsKnown)
237 _worldPacket << uint32(specialization);
238
239 return &_worldPacket;
240}

30#define DEFAULT_VISIBILITY_NOTIFY_PERIOD 1000
31
32class GridInfo
33{
34public:
35 GridInfo()
36 : i_timer(0), vis_Update(0, irand(0, DEFAULT_VISIBILITY_NOTIFY_PERIOD)),
37 i_unloadActiveLockCount(0), i_unloadExplicitLock(false), i_unloadReferenceLock(false) { }
38 GridInfo(time_t expiry, bool unload = true )
39 : i_timer(expiry), vis_Update(0, irand(0, DEFAULT_VISIBILITY_NOTIFY_PERIOD)),
40 i_unloadActiveLockCount(0), i_unloadExplicitLock(!unload), i_unloadReferenceLock(false) { }
41 const TimeTracker& getTimeTracker() const { return i_timer; }
42 bool getUnloadLock() const { return i_unloadActiveLockCount || i_unloadExplicitLock || i_unloadReferenceLock; }
43 void setUnloadExplicitLock(bool on) { i_unloadExplicitLock = on; }
44 void setUnloadReferenceLock(bool on) { i_unloadReferenceLock = on; }
45 void incUnloadActiveLock() { ++i_unloadActiveLockCount; }
46 void decUnloadActiveLock() { if (i_unloadActiveLockCount) --i_unloadActiveLockCount; }
47
48 void setTimer(const TimeTracker& pTimer) { i_timer = pTimer; }
49 void ResetTimeTracker(time_t interval) { i_timer.Reset(interval); }
50 void UpdateTimeTracker(time_t diff) { i_timer.Update(diff); }
51 PeriodicTimer& getRelocationTimer() { return vis_Update; }
52private:
53 TimeTracker i_timer;
54 PeriodicTimer vis_Update;
55
56 uint16 i_unloadActiveLockCount : 16; // lock from active object spawn points (prevent clone loading)
57 bool i_unloadExplicitLock : 1; // explicit manual lock or config setting
58 bool i_unloadReferenceLock : 1; // lock from instance map copy
59};

61typedef enum

63 GRID_STATE_INVALID = 0,
64 GRID_STATE_ACTIVE = 1,
65 GRID_STATE_IDLE = 2,

144 // Visit all Grids (cells) in NGrid (grid)
145 template<class T, class TT>
146 void VisitAllGrids(TypeContainerVisitor<T, TypeMapContainer<TT> > &visitor)
147 {
148 for (uint32 x = 0; x < N; ++x)
149 for (uint32 y = 0; y < N; ++y)
150 GetGridType(x, y).Visit(visitor);
151 }
152
153 // Visit a single Grid (cell) in NGrid (grid)
154 template<class T, class TT>
155 void VisitGrid(const uint32 x, const uint32 y, TypeContainerVisitor<T, TypeMapContainer<TT> > &visitor)
156 {
157 GetGridType(x, y).Visit(visitor);
158 }

26void ActiveState::Update(Map& map, NGridType& grid, GridInfo& info, uint32 diff) const
27{
28 // Only check grid activity every (grid_expiry/10) ms, because it's really useless to do it every cycle
29 info.UpdateTimeTracker(diff);
30 if (info.getTimeTracker().Passed())
31 {
32 if (!grid.GetWorldObjectCountInNGrid<Player>() && !map.ActiveObjectsNearGrid(grid))
33 {
34 ObjectGridStoper worker;
35 TypeContainerVisitor<ObjectGridStoper, GridTypeMapContainer> visitor(worker);
36 grid.VisitAllGrids(visitor);
37 grid.SetGridState(GRID_STATE_IDLE);
38 TC_LOG_DEBUG("maps", "Grid[%u, %u] on map %u moved to IDLE state", grid.getX(), grid.getY(), map.GetId());
39 }
40 else
41 map.ResetGridExpiry(grid, 0.1f);
42 }
43}
44
45void IdleState::Update(Map& map, NGridType& grid, GridInfo&, uint32) const
46{
47 map.ResetGridExpiry(grid);
48 grid.SetGridState(GRID_STATE_REMOVAL);
49 TC_LOG_DEBUG("maps", "Grid[%u, %u] on map %u moved to REMOVAL state", grid.getX(), grid.getY(), map.GetId());
50}
51
52void RemovalState::Update(Map& map, NGridType& grid, GridInfo& info, uint32 diff) const
53{
54 if (!info.getUnloadLock())
55 {
56 info.UpdateTimeTracker(diff);
57 if (info.getTimeTracker().Passed() && !map.UnloadGrid(grid, false))
58 {
59 TC_LOG_DEBUG("maps", "Grid[%u, %u] for map %u differed unloading due to players or active objects nearby", grid.getX(), grid.getY(), map.GetId());
60 map.ResetGridExpiry(grid);
61 }
62 }

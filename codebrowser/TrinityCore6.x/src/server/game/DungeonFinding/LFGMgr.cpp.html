53void LFGMgr::_LoadFromDB(Field* fields, ObjectGuid guid)
54{
55 if (!fields)
56 return;
57
58 if (!guid.IsParty())
59 return;
60
61 SetLeader(guid, ObjectGuid::Create<HighGuid::Player>(fields[0].GetUInt64()));
62
63 uint32 dungeon = fields[18].GetUInt32();
64 uint8 state = fields[19].GetUInt8();
65
66 if (!dungeon || !state)
67 return;
68
69 SetDungeon(guid, dungeon);
70
71 switch (state)
72 {
73 case LFG_STATE_DUNGEON:
74 case LFG_STATE_FINISHED_DUNGEON:
75 SetState(guid, (LfgState)state);
76 break;
77 default:
78 break;
79 }
80}

82void LFGMgr::_SaveToDB(ObjectGuid guid, uint32 db_guid)
83{
84 if (!guid.IsParty())
85 return;
86
87 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_LFG_DATA);
88
89 stmt->setUInt32(0, db_guid);
90
91 CharacterDatabase.Execute(stmt);
92
93 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_LFG_DATA);
94 stmt->setUInt32(0, db_guid);
95
96 stmt->setUInt32(1, GetDungeon(guid));
97 stmt->setUInt32(2, GetState(guid));
98
99 CharacterDatabase.Execute(stmt);
100}

102/// Load rewards for completing dungeons
103void LFGMgr::LoadRewards()
104{
105 uint32 oldMSTime = getMSTime();
106
107 for (LfgRewardContainer::iterator itr = RewardMapStore.begin(); itr != RewardMapStore.end(); ++itr)
108 delete itr->second;
109 RewardMapStore.clear();
110
111 // ORDER BY is very important for GetRandomDungeonReward!
112 QueryResult result = WorldDatabase.Query("SELECT dungeonId, maxLevel, firstQuestId, otherQuestId FROM lfg_dungeon_rewards ORDER BY dungeonId, maxLevel ASC");
113
114 if (!result)
115 {
116 TC_LOG_ERROR("server.loading", ">> Loaded 0 lfg dungeon rewards. DB table `lfg_dungeon_rewards` is empty!");
117 return;
118 }
119
120 uint32 count = 0;
121
122 Field* fields = NULL;
123 do
124 {
125 fields = result->Fetch();
126 uint32 dungeonId = fields[0].GetUInt32();
127 uint32 maxLevel = fields[1].GetUInt8();
128 uint32 firstQuestId = fields[2].GetUInt32();
129 uint32 otherQuestId = fields[3].GetUInt32();
130
131 if (!GetLFGDungeonEntry(dungeonId))
132 {
133 TC_LOG_ERROR("sql.sql", "Dungeon %u specified in table `lfg_dungeon_rewards` does not exist!", dungeonId);
134 continue;
135 }
136
137 if (!maxLevel || maxLevel > sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
138 {
139 TC_LOG_ERROR("sql.sql", "Level %u specified for dungeon %u in table `lfg_dungeon_rewards` can never be reached!", maxLevel, dungeonId);
140 maxLevel = sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL);
141 }
142
143 if (!firstQuestId || !sObjectMgr->GetQuestTemplate(firstQuestId))
144 {
145 TC_LOG_ERROR("sql.sql", "First quest %u specified for dungeon %u in table `lfg_dungeon_rewards` does not exist!", firstQuestId, dungeonId);
146 continue;
147 }
148
149 if (otherQuestId && !sObjectMgr->GetQuestTemplate(otherQuestId))
150 {
151 TC_LOG_ERROR("sql.sql", "Other quest %u specified for dungeon %u in table `lfg_dungeon_rewards` does not exist!", otherQuestId, dungeonId);
152 otherQuestId = 0;
153 }
154
155 RewardMapStore.insert(LfgRewardContainer::value_type(dungeonId, new LfgReward(maxLevel, firstQuestId, otherQuestId)));
156 ++count;
157 }
158 while (result->NextRow());
159
160 TC_LOG_INFO("server.loading", ">> Loaded %u lfg dungeon rewards in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
161}

172void LFGMgr::LoadLFGDungeons(bool reload /* = false */)
173{
174 uint32 oldMSTime = getMSTime();
175
176 LfgDungeonStore.clear();
177
178 // Initialize Dungeon map with data from dbcs
179 for (uint32 i = 0; i < sLFGDungeonStore.GetNumRows(); ++i)
180 {
181 LFGDungeonEntry const* dungeon = sLFGDungeonStore.LookupEntry(i);
182 if (!dungeon)
183 continue;
184
185 switch (dungeon->Type)
186 {
187 case LFG_TYPE_DUNGEON:
188 case LFG_TYPE_HEROIC:
189 case LFG_TYPE_RAID:
190 case LFG_TYPE_RANDOM:
191 LfgDungeonStore[dungeon->ID] = LFGDungeonData(dungeon);
192 break;
193 }
194 }
195
196 // Fill teleport locations from DB
197 QueryResult result = WorldDatabase.Query("SELECT dungeonId, position_x, position_y, position_z, orientation, requiredItemLevel FROM lfg_dungeon_template");
198 if (!result)
199 {
200 TC_LOG_ERROR("server.loading", ">> Loaded 0 lfg dungeon templates. DB table `lfg_dungeon_template` is empty!");
201 return;
202 }
203
204 uint32 count = 0;
205
206 do
207 {
208 Field* fields = result->Fetch();
209 uint32 dungeonId = fields[0].GetUInt32();
210 LFGDungeonContainer::iterator dungeonItr = LfgDungeonStore.find(dungeonId);
211 if (dungeonItr == LfgDungeonStore.end())
212 {
213 TC_LOG_ERROR("sql.sql", "table `lfg_entrances` contains coordinates for wrong dungeon %u", dungeonId);
214 continue;
215 }
216
217 LFGDungeonData& data = dungeonItr->second;
218 data.x = fields[1].GetFloat();
219 data.y = fields[2].GetFloat();
220 data.z = fields[3].GetFloat();
221 data.o = fields[4].GetFloat();
222 data.requiredItemLevel = fields[5].GetUInt16();
223
224 ++count;
225 }
226 while (result->NextRow());
227
228 TC_LOG_INFO("server.loading", ">> Loaded %u lfg dungeon templates in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
229
230 // Fill all other teleport coords from areatriggers
231 for (LFGDungeonContainer::iterator itr = LfgDungeonStore.begin(); itr != LfgDungeonStore.end(); ++itr)
232 {
233 LFGDungeonData& dungeon = itr->second;
234
235 // No teleport coords in database, load from areatriggers
236 if (dungeon.type != LFG_TYPE_RANDOM && dungeon.x == 0.0f && dungeon.y == 0.0f && dungeon.z == 0.0f)
237 {
238 AreaTriggerStruct const* at = sObjectMgr->GetMapEntranceTrigger(dungeon.map);
239 if (!at)
240 {
241 TC_LOG_ERROR("sql.sql", "Failed to load dungeon %s (Id: %u), cant find areatrigger for map %u", dungeon.name.c_str(), dungeon.id, dungeon.map);
242 continue;
243 }
244
245 dungeon.map = at->target_mapId;
246 dungeon.x = at->target_X;
247 dungeon.y = at->target_Y;
248 dungeon.z = at->target_Z;
249 dungeon.o = at->target_Orientation;
250 }
251
252 if (dungeon.type != LFG_TYPE_RANDOM)
253 CachedDungeonMapStore[dungeon.group].insert(dungeon.id);
254 CachedDungeonMapStore[0].insert(dungeon.id);
255 }
256
257 if (reload)
258 CachedDungeonMapStore.clear();
259}

261void LFGMgr::Update(uint32 diff)
262{
263 if (!isOptionEnabled(LFG_OPTION_ENABLE_DUNGEON_FINDER | LFG_OPTION_ENABLE_RAID_BROWSER))
264 return;
265
266 time_t currTime = time(NULL);
267
268 // Remove obsolete role checks
269 for (LfgRoleCheckContainer::iterator it = RoleChecksStore.begin(); it != RoleChecksStore.end();)
270 {
271 LfgRoleCheckContainer::iterator itRoleCheck = it++;
272 LfgRoleCheck& roleCheck = itRoleCheck->second;
273 if (currTime < roleCheck.cancelTime)
274 continue;
275 roleCheck.state = LFG_ROLECHECK_MISSING_ROLE;
276
277 for (LfgRolesMap::const_iterator itRoles = roleCheck.roles.begin(); itRoles != roleCheck.roles.end(); ++itRoles)
278 {
279 ObjectGuid guid = itRoles->first;
280 RestoreState(guid, "Remove Obsolete RoleCheck");
281 SendLfgRoleCheckUpdate(guid, roleCheck);
282 if (guid == roleCheck.leader)
283 SendLfgJoinResult(guid, LfgJoinResultData(LFG_JOIN_ROLE_CHECK_FAILED, LFG_ROLECHECK_MISSING_ROLE));
284 }
285
286 RestoreState(itRoleCheck->first, "Remove Obsolete RoleCheck");
287 RoleChecksStore.erase(itRoleCheck);
288 }
289
290 // Remove obsolete proposals
291 for (LfgProposalContainer::iterator it = ProposalsStore.begin(); it != ProposalsStore.end();)
292 {
293 LfgProposalContainer::iterator itRemove = it++;
294 if (itRemove->second.cancelTime < currTime)
295 RemoveProposal(itRemove, LFG_UPDATETYPE_PROPOSAL_FAILED);
296 }
297
298 // Remove obsolete kicks
299 for (LfgPlayerBootContainer::iterator it = BootsStore.begin(); it != BootsStore.end();)
300 {
301 LfgPlayerBootContainer::iterator itBoot = it++;
302 LfgPlayerBoot& boot = itBoot->second;
303 if (boot.cancelTime < currTime)
304 {
305 boot.inProgress = false;
306 for (LfgAnswerContainer::const_iterator itVotes = boot.votes.begin(); itVotes != boot.votes.end(); ++itVotes)
307 {
308 ObjectGuid pguid = itVotes->first;
309 if (pguid != boot.victim)
310 SendLfgBootProposalUpdate(pguid, boot);
311 }
312 SetVoteKick(itBoot->first, false);
313 BootsStore.erase(itBoot);
314 }
315 }
316
317 uint32 lastProposalId = m_lfgProposalId;
318 // Check if a proposal can be formed with the new groups being added
319 for (LfgQueueContainer::iterator it = QueuesStore.begin(); it != QueuesStore.end(); ++it)
320 if (uint8 newProposals = it->second.FindGroups())
321 TC_LOG_DEBUG("lfg.update", "Found %u new groups in queue %u", newProposals, it->first);
322
323 if (lastProposalId != m_lfgProposalId)
324 {
325 // FIXME lastProposalId ? lastProposalId +1 ?
326 for (LfgProposalContainer::const_iterator itProposal = ProposalsStore.find(m_lfgProposalId); itProposal != ProposalsStore.end(); ++itProposal)
327 {
328 uint32 proposalId = itProposal->first;
329 LfgProposal& proposal = ProposalsStore[proposalId];
330
331 ObjectGuid guid;
332 for (LfgProposalPlayerContainer::const_iterator itPlayers = proposal.players.begin(); itPlayers != proposal.players.end(); ++itPlayers)
333 {
334 guid = itPlayers->first;
335 SetState(guid, LFG_STATE_PROPOSAL);
336 ObjectGuid gguid = GetGroup(guid);
337 if (!gguid.IsEmpty())
338 {
339 SetState(gguid, LFG_STATE_PROPOSAL);
340 SendLfgUpdateStatus(guid, LfgUpdateData(LFG_UPDATETYPE_PROPOSAL_BEGIN, GetSelectedDungeons(guid), GetComment(guid)), true);
341 }
342 else
343 SendLfgUpdateStatus(guid, LfgUpdateData(LFG_UPDATETYPE_PROPOSAL_BEGIN, GetSelectedDungeons(guid), GetComment(guid)), false);
344 SendLfgUpdateProposal(guid, proposal);
345 }
346
347 if (proposal.state == LFG_PROPOSAL_SUCCESS)
348 UpdateProposal(proposalId, guid, true);
349 }
350 }
351
352 // Update all players status queue info
353 if (m_QueueTimer > LFG_QUEUEUPDATE_INTERVAL)
354 {
355 m_QueueTimer = 0;
356 for (LfgQueueContainer::iterator it = QueuesStore.begin(); it != QueuesStore.end(); ++it)
357 it->second.UpdateQueueTimers(it->first, currTime);
358 }
359 else
360 m_QueueTimer += diff;
361}

363/**
364 Adds the player/group to lfg queue. If player is in a group then it is the leader
365 of the group tying to join the group. Join conditions are checked before adding
366 to the new queue.
367
368 @param[in] player Player trying to join (or leader of group trying to join)
369 @param[in] roles Player selected roles
370 @param[in] dungeons Dungeons the player/group is applying for
371 @param[in] comment Player selected comment
372*/
373void LFGMgr::JoinLfg(Player* player, uint8 roles, LfgDungeonSet& dungeons, const std::string& comment)
374{
375 if (!player || !player->GetSession() || dungeons.empty())
376 return;
377
378 Group* grp = player->GetGroup();
379 ObjectGuid guid = player->GetGUID();
380 ObjectGuid gguid = grp ? grp->GetGUID() : guid;
381 LfgJoinResultData joinData;
382 GuidSet players;
383 uint32 rDungeonId = 0;
384 bool isContinue = grp && grp->isLFGGroup() && GetState(gguid) != LFG_STATE_FINISHED_DUNGEON;
385
386 // Do not allow to change dungeon in the middle of a current dungeon
387 if (isContinue)
388 {
389 dungeons.clear();
390 dungeons.insert(GetDungeon(gguid));
391 }
392
393 // Already in queue?
394 LfgState state = GetState(gguid);
395 if (state == LFG_STATE_QUEUED)
396 {
397 LFGQueue& queue = GetQueue(gguid);
398 queue.RemoveFromQueue(gguid);
399 }
400
401 // Check player or group member restrictions
402 if (!player->GetSession()->HasPermission(rbac::RBAC_PERM_JOIN_DUNGEON_FINDER))
403 joinData.result = LFG_JOIN_NOT_MEET_REQS;
404 else if (player->InBattleground() || player->InArena() || player->InBattlegroundQueue())
405 joinData.result = LFG_JOIN_USING_BG_SYSTEM;
406 else if (player->HasAura(LFG_SPELL_DUNGEON_DESERTER))
407 joinData.result = LFG_JOIN_DESERTER;
408 else if (player->HasAura(LFG_SPELL_DUNGEON_COOLDOWN))
409 joinData.result = LFG_JOIN_RANDOM_COOLDOWN;
410 else if (dungeons.empty())
411 joinData.result = LFG_JOIN_NOT_MEET_REQS;
412 else if (player->HasAura(9454)) // check Freeze debuff
413 joinData.result = LFG_JOIN_NOT_MEET_REQS;
414 else if (grp)
415 {
416 if (grp->GetMembersCount() > MAX_GROUP_SIZE)
417 joinData.result = LFG_JOIN_TOO_MUCH_MEMBERS;
418 else
419 {
420 uint8 memberCount = 0;
421 for (GroupReference* itr = grp->GetFirstMember(); itr != NULL && joinData.result == LFG_JOIN_OK; itr = itr->next())
422 {
423 if (Player* plrg = itr->GetSource())
424 {
425 if (!plrg->GetSession()->HasPermission(rbac::RBAC_PERM_JOIN_DUNGEON_FINDER))
426 joinData.result = LFG_JOIN_INTERNAL_ERROR;
427 if (plrg->HasAura(LFG_SPELL_DUNGEON_DESERTER))
428 joinData.result = LFG_JOIN_PARTY_DESERTER;
429 else if (plrg->HasAura(LFG_SPELL_DUNGEON_COOLDOWN))
430 joinData.result = LFG_JOIN_PARTY_RANDOM_COOLDOWN;
431 else if (plrg->InBattleground() || plrg->InArena() || plrg->InBattlegroundQueue())
432 joinData.result = LFG_JOIN_USING_BG_SYSTEM;
433 else if (plrg->HasAura(9454)) // check Freeze debuff
434 joinData.result = LFG_JOIN_PARTY_NOT_MEET_REQS;
435 ++memberCount;
436 players.insert(plrg->GetGUID());
437 }
438 }
439
440 if (joinData.result == LFG_JOIN_OK && memberCount != grp->GetMembersCount())
441 joinData.result = LFG_JOIN_DISCONNECTED;
442 }
443 }
444 else
445 players.insert(player->GetGUID());
446
447 // Check if all dungeons are valid
448 bool isRaid = false;
449 if (joinData.result == LFG_JOIN_OK)
450 {
451 bool isDungeon = false;
452 for (LfgDungeonSet::const_iterator it = dungeons.begin(); it != dungeons.end() && joinData.result == LFG_JOIN_OK; ++it)
453 {
454 LfgType type = GetDungeonType(*it);
455 switch (type)
456 {
457 case LFG_TYPE_RANDOM:
458 if (dungeons.size() > 1) // Only allow 1 random dungeon
459 joinData.result = LFG_JOIN_DUNGEON_INVALID;
460 else
461 rDungeonId = (*dungeons.begin());
462 // No break on purpose (Random can only be dungeon or heroic dungeon)
463 case LFG_TYPE_HEROIC:
464 case LFG_TYPE_DUNGEON:
465 if (isRaid)
466 joinData.result = LFG_JOIN_MIXED_RAID_DUNGEON;
467 isDungeon = true;
468 break;
469 case LFG_TYPE_RAID:
470 if (isDungeon)
471 joinData.result = LFG_JOIN_MIXED_RAID_DUNGEON;
472 isRaid = true;
473 break;
474 default:
475 joinData.result = LFG_JOIN_DUNGEON_INVALID;
476 break;
477 }
478 }
479
480 // it could be changed
481 if (joinData.result == LFG_JOIN_OK)
482 {
483 // Expand random dungeons and check restrictions
484 if (rDungeonId)
485 dungeons = GetDungeonsByRandom(rDungeonId);
486
487 // if we have lockmap then there are no compatible dungeons
488 GetCompatibleDungeons(dungeons, players, joinData.lockmap, isContinue);
489 if (dungeons.empty())
490 joinData.result = grp ? LFG_JOIN_INTERNAL_ERROR : LFG_JOIN_NOT_MEET_REQS;
491 }
492 }
493
494 // Can't join. Send result
495 if (joinData.result != LFG_JOIN_OK)
496 {
497 TC_LOG_DEBUG("lfg.join", "%s joining with %u members. Result: %u, Dungeons: %s",
498 guid.ToString().c_str(), grp ? grp->GetMembersCount() : 1, joinData.result, ConcatenateDungeons(dungeons).c_str());
499
500 if (!dungeons.empty()) // Only should show lockmap when have no dungeons available
501 joinData.lockmap.clear();
502 player->GetSession()->SendLfgJoinResult(joinData);
503 return;
504 }
505
506 SetComment(guid, comment);
507
508 if (isRaid)
509 {
510 TC_LOG_DEBUG("lfg.join", "%s trying to join raid browser and it's disabled.", guid.ToString().c_str());
511 return;
512 }
513
514 std::string debugNames = "";
515 if (grp) // Begin rolecheck
516 {
517 // Create new rolecheck
518 LfgRoleCheck& roleCheck = RoleChecksStore[gguid];
519 roleCheck.cancelTime = time_t(time(NULL)) + LFG_TIME_ROLECHECK;
520 roleCheck.state = LFG_ROLECHECK_INITIALITING;
521 roleCheck.leader = guid;
522 roleCheck.dungeons = dungeons;
523 roleCheck.rDungeonId = rDungeonId;
524
525 if (rDungeonId)
526 {
527 dungeons.clear();
528 dungeons.insert(rDungeonId);
529 }
530
531 SetState(gguid, LFG_STATE_ROLECHECK);
532 // Send update to player
533 LfgUpdateData updateData = LfgUpdateData(LFG_UPDATETYPE_JOIN_QUEUE, dungeons, comment);
534 for (GroupReference* itr = grp->GetFirstMember(); itr != NULL; itr = itr->next())
535 {
536 if (Player* plrg = itr->GetSource())
537 {
538 ObjectGuid pguid = plrg->GetGUID();
539 plrg->GetSession()->SendLfgUpdateStatus(updateData, true);
540 SetState(pguid, LFG_STATE_ROLECHECK);
541 if (!isContinue)
542 SetSelectedDungeons(pguid, dungeons);
543 roleCheck.roles[pguid] = 0;
544 if (!debugNames.empty())
545 debugNames.append(", ");
546 debugNames.append(plrg->GetName());
547 }
548 }
549 // Update leader role
550 UpdateRoleCheck(gguid, guid, roles);
551 }
552 else // Add player to queue
553 {
554 LfgRolesMap rolesMap;
555 rolesMap[guid] = roles;
556 LFGQueue& queue = GetQueue(guid);
557 queue.AddQueueData(guid, time(NULL), dungeons, rolesMap);
558
559 if (!isContinue)
560 {
561 if (rDungeonId)
562 {
563 dungeons.clear();
564 dungeons.insert(rDungeonId);
565 }
566 SetSelectedDungeons(guid, dungeons);
567 }
568 // Send update to player
569 player->GetSession()->SendLfgJoinResult(joinData);
570 player->GetSession()->SendLfgUpdateStatus(LfgUpdateData(LFG_UPDATETYPE_JOIN_QUEUE, dungeons, comment), false);
571 SetState(gguid, LFG_STATE_QUEUED);
572 SetRoles(guid, roles);
573 debugNames.append(player->GetName());
574 }
575
576 TC_LOG_DEBUG("lfg.join", "%s joined (%s), Members: %s. Dungeons (%u): %s", guid.ToString().c_str(),
577 grp ? "group" : "player", debugNames.c_str(), uint32(dungeons.size()), ConcatenateDungeons(dungeons).c_str());
578}

580/**
581 Leaves Dungeon System. Player/Group is removed from queue, rolechecks, proposals
582 or votekicks. Player or group needs to be not NULL and using Dungeon System
583
584 @param[in] guid Player or group guid
585*/
586void LFGMgr::LeaveLfg(ObjectGuid guid, bool disconnected)
587{
588 ObjectGuid gguid = guid.IsParty() ? guid : GetGroup(guid);
589
590 TC_LOG_DEBUG("lfg.leave", "%s left (%s)", guid.ToString().c_str(), guid == gguid ? "group" : "player");
591
592 LfgState state = GetState(guid);
593 switch (state)
594 {
595 case LFG_STATE_QUEUED:
596 if (!gguid.IsEmpty())
597 {
598 LFGQueue& queue = GetQueue(gguid);
599 queue.RemoveFromQueue(gguid);
600 SetState(gguid, LFG_STATE_NONE);
601 GuidSet const& players = GetPlayers(gguid);
602 for (GuidSet::const_iterator it = players.begin(); it != players.end(); ++it)
603 {
604 SetState(*it, LFG_STATE_NONE);
605 SendLfgUpdateStatus(*it, LfgUpdateData(LFG_UPDATETYPE_REMOVED_FROM_QUEUE), true);
606 }
607 }
608 else
609 {
610 LFGQueue& queue = GetQueue(guid);
611 queue.RemoveFromQueue(guid);
612 SendLfgUpdateStatus(guid, LfgUpdateData(LFG_UPDATETYPE_REMOVED_FROM_QUEUE), false);
613 SetState(guid, LFG_STATE_NONE);
614 }
615 break;
616 case LFG_STATE_ROLECHECK:
617 if (!gguid.IsEmpty())
618 UpdateRoleCheck(gguid); // No player to update role = LFG_ROLECHECK_ABORTED
619 break;
620 case LFG_STATE_PROPOSAL:
621 {
622 // Remove from Proposals
623 LfgProposalContainer::iterator it = ProposalsStore.begin();
624 ObjectGuid pguid = gguid == guid ? GetLeader(gguid) : guid;
625 while (it != ProposalsStore.end())
626 {
627 LfgProposalPlayerContainer::iterator itPlayer = it->second.players.find(pguid);
628 if (itPlayer != it->second.players.end())
629 {
630 // Mark the player/leader of group who left as didn't accept the proposal
631 itPlayer->second.accept = LFG_ANSWER_DENY;
632 break;
633 }
634 ++it;
635 }
636
637 // Remove from queue - if proposal is found, RemoveProposal will call RemoveFromQueue
638 if (it != ProposalsStore.end())
639 RemoveProposal(it, LFG_UPDATETYPE_PROPOSAL_DECLINED);
640 break;
641 }
642 case LFG_STATE_NONE:
643 case LFG_STATE_RAIDBROWSER:

653/**
654 Update the Role check info with the player selected role.
655
656 @param[in] grp Group guid to update rolecheck
657 @param[in] guid Player guid (0 = rolecheck failed)
658 @param[in] roles Player selected roles
659*/
660void LFGMgr::UpdateRoleCheck(ObjectGuid gguid, ObjectGuid guid /* = ObjectGuid::Empty */, uint8 roles /* = PLAYER_ROLE_NONE */)
661{
662 if (!gguid)
663 return;
664
665 LfgRolesMap check_roles;
666 LfgRoleCheckContainer::iterator itRoleCheck = RoleChecksStore.find(gguid);
667 if (itRoleCheck == RoleChecksStore.end())
668 return;
669
670 LfgRoleCheck& roleCheck = itRoleCheck->second;
671 bool sendRoleChosen = roleCheck.state != LFG_ROLECHECK_DEFAULT && !guid.IsEmpty();
672
673 if (!guid)
674 roleCheck.state = LFG_ROLECHECK_ABORTED;
675 else if (roles < PLAYER_ROLE_TANK) // Player selected no role.
676 roleCheck.state = LFG_ROLECHECK_NO_ROLE;
677 else
678 {
679 roleCheck.roles[guid] = roles;
680
681 // Check if all players have selected a role
682 LfgRolesMap::const_iterator itRoles = roleCheck.roles.begin();
683 while (itRoles != roleCheck.roles.end() && itRoles->second != PLAYER_ROLE_NONE)
684 ++itRoles;
685
686 if (itRoles == roleCheck.roles.end())
687 {
688 // use temporal var to check roles, CheckGroupRoles modifies the roles
689 check_roles = roleCheck.roles;
690 roleCheck.state = CheckGroupRoles(check_roles) ? LFG_ROLECHECK_FINISHED : LFG_ROLECHECK_WRONG_ROLES;
691 }
692 }
693
694 LfgDungeonSet dungeons;
695 if (roleCheck.rDungeonId)
696 dungeons.insert(roleCheck.rDungeonId);
697 else
698 dungeons = roleCheck.dungeons;
699
700 LfgJoinResult joinResult = LFG_JOIN_FAILED;
701 if (roleCheck.state == LFG_ROLECHECK_MISSING_ROLE || roleCheck.state == LFG_ROLECHECK_WRONG_ROLES)
702 joinResult = LFG_JOIN_ROLE_CHECK_FAILED;
703
704 LfgJoinResultData joinData = LfgJoinResultData(joinResult, roleCheck.state);
705 for (LfgRolesMap::const_iterator it = roleCheck.roles.begin(); it != roleCheck.roles.end(); ++it)
706 {
707 ObjectGuid pguid = it->first;
708
709 if (sendRoleChosen)
710 SendLfgRoleChosen(pguid, guid, roles);
711
712 SendLfgRoleCheckUpdate(pguid, roleCheck);
713 switch (roleCheck.state)
714 {
715 case LFG_ROLECHECK_INITIALITING:
716 continue;
717 case LFG_ROLECHECK_FINISHED:
718 SetState(pguid, LFG_STATE_QUEUED);
719 SetRoles(pguid, it->second);
720 SendLfgUpdateStatus(pguid, LfgUpdateData(LFG_UPDATETYPE_ADDED_TO_QUEUE, dungeons, GetComment(pguid)), true);
721 break;
722 default:
723 if (roleCheck.leader == pguid)
724 SendLfgJoinResult(pguid, joinData);
725 SendLfgUpdateStatus(pguid, LfgUpdateData(LFG_UPDATETYPE_ROLECHECK_FAILED), true);
726 RestoreState(pguid, "Rolecheck Failed");
727 break;
728 }
729 }
730
731 if (roleCheck.state == LFG_ROLECHECK_FINISHED)
732 {
733 SetState(gguid, LFG_STATE_QUEUED);
734 LFGQueue& queue = GetQueue(gguid);
735 queue.AddQueueData(gguid, time_t(time(NULL)), roleCheck.dungeons, roleCheck.roles);
736 RoleChecksStore.erase(itRoleCheck);
737 }
738 else if (roleCheck.state != LFG_ROLECHECK_INITIALITING)
739 {
740 RestoreState(gguid, "Rolecheck Failed");
741 RoleChecksStore.erase(itRoleCheck);
742 }
743}

749 @param[in] players Set of players to check their dungeon restrictions

752void LFGMgr::GetCompatibleDungeons(LfgDungeonSet& dungeons, GuidSet const& players, LfgLockPartyMap& lockMap, bool isContinue)
753{
754 lockMap.clear();
755
756 std::map<uint32, uint32> lockedDungeons;
757
758 for (GuidSet::const_iterator it = players.begin(); it != players.end() && !dungeons.empty(); ++it)
759 {
760 ObjectGuid guid = (*it);
761 LfgLockMap const& cachedLockMap = GetLockedDungeons(guid);
762 Player* player = ObjectAccessor::FindConnectedPlayer(guid);
763 for (LfgLockMap::const_iterator it2 = cachedLockMap.begin(); it2 != cachedLockMap.end() && !dungeons.empty(); ++it2)
764 {
765 uint32 dungeonId = (it2->first & 0x00FFFFFF); // Compare dungeon ids
766 LfgDungeonSet::iterator itDungeon = dungeons.find(dungeonId);
767 if (itDungeon != dungeons.end())
768 {
769 bool eraseDungeon = true;
770
771 // Don't remove the dungeon if team members are trying to continue a locked instance
772 if (it2->second.lockStatus == LFG_LOCKSTATUS_RAID_LOCKED && isContinue)
773 {
774 LFGDungeonData const* dungeon = GetLFGDungeon(dungeonId);
775 ASSERT(dungeon);
776 ASSERT(player);
777 if (InstancePlayerBind* playerBind = player->GetBoundInstance(dungeon->map, Difficulty(dungeon->difficulty)))

779 if (InstanceSave* playerSave = playerBind->save)
780 {
781 uint32 dungeonInstanceId = playerSave->GetInstanceId();

784 eraseDungeon = false;

787 }
788 }

792 dungeons.erase(itDungeon);
793
794 lockMap[guid][dungeonId] = it2->second;
795 }
796 }
797 }
798 if (!dungeons.empty())
799 lockMap.clear();

802/**
803 Check if a group can be formed with the given group roles
804
805 @param[in] groles Map of roles to check
806 @return True if roles are compatible
807*/
808bool LFGMgr::CheckGroupRoles(LfgRolesMap& groles)
809{
810 if (groles.empty())
811 return false;
812
813 uint8 damage = 0;
814 uint8 tank = 0;
815 uint8 healer = 0;
816
817 for (LfgRolesMap::iterator it = groles.begin(); it != groles.end(); ++it)
818 {
819 uint8 role = it->second & ~PLAYER_ROLE_LEADER;
820 if (role == PLAYER_ROLE_NONE)
821 return false;
822
823 if (role & PLAYER_ROLE_DAMAGE)
824 {
825 if (role != PLAYER_ROLE_DAMAGE)
826 {
827 it->second -= PLAYER_ROLE_DAMAGE;
828 if (CheckGroupRoles(groles))
829 return true;
830 it->second += PLAYER_ROLE_DAMAGE;
831 }
832 else if (damage == LFG_DPS_NEEDED)
833 return false;
834 else
835 damage++;
836 }
837
838 if (role & PLAYER_ROLE_HEALER)
839 {
840 if (role != PLAYER_ROLE_HEALER)
841 {
842 it->second -= PLAYER_ROLE_HEALER;
843 if (CheckGroupRoles(groles))
844 return true;
845 it->second += PLAYER_ROLE_HEALER;
846 }
847 else if (healer == LFG_HEALERS_NEEDED)
848 return false;
849 else
850 healer++;
851 }
852
853 if (role & PLAYER_ROLE_TANK)
854 {
855 if (role != PLAYER_ROLE_TANK)
856 {
857 it->second -= PLAYER_ROLE_TANK;
858 if (CheckGroupRoles(groles))
859 return true;
860 it->second += PLAYER_ROLE_TANK;
861 }
862 else if (tank == LFG_TANKS_NEEDED)
863 return false;
864 else
865 tank++;
866 }
867 }
868 return (tank + healer + damage) == uint8(groles.size());
869}

937 if (Player* player = ObjectAccessor::FindPlayer(*it))

940 // Update group info

951/**
952 Update Proposal info with player answer
953
954 @param[in] proposalId Proposal id to be updated
955 @param[in] guid Player guid to update answer
956 @param[in] accept Player answer
957*/
958void LFGMgr::UpdateProposal(uint32 proposalId, ObjectGuid guid, bool accept)
959{
960 // Check if the proposal exists
961 LfgProposalContainer::iterator itProposal = ProposalsStore.find(proposalId);
962 if (itProposal == ProposalsStore.end())
963 return;
964
965 LfgProposal& proposal = itProposal->second;
966
967 // Check if proposal have the current player
968 LfgProposalPlayerContainer::iterator itProposalPlayer = proposal.players.find(guid);
969 if (itProposalPlayer == proposal.players.end())
970 return;
971
972 LfgProposalPlayer& player = itProposalPlayer->second;
973 player.accept = LfgAnswer(accept);
974
975 TC_LOG_DEBUG("lfg.proposal.update", "%s, Proposal %u, Selection: %u", guid.ToString().c_str(), proposalId, accept);
976 if (!accept)
977 {
978 RemoveProposal(itProposal, LFG_UPDATETYPE_PROPOSAL_DECLINED);
979 return;
980 }
981
982 // check if all have answered and reorder players (leader first)
983 bool allAnswered = true;
984 for (LfgProposalPlayerContainer::const_iterator itPlayers = proposal.players.begin(); itPlayers != proposal.players.end(); ++itPlayers)
985 if (itPlayers->second.accept != LFG_ANSWER_AGREE) // No answer (-1) or not accepted (0)
986 allAnswered = false;
987
988 if (!allAnswered)
989 {
990 for (LfgProposalPlayerContainer::const_iterator it = proposal.players.begin(); it != proposal.players.end(); ++it)
991 SendLfgUpdateProposal(it->first, proposal);
992
993 return;
994 }
995
996 bool sendUpdate = proposal.state != LFG_PROPOSAL_SUCCESS;
997 proposal.state = LFG_PROPOSAL_SUCCESS;
998 time_t joinTime = time(NULL);
999
1000 LFGQueue& queue = GetQueue(guid);
1001 LfgUpdateData updateData = LfgUpdateData(LFG_UPDATETYPE_GROUP_FOUND);
1002 for (LfgProposalPlayerContainer::const_iterator it = proposal.players.begin(); it != proposal.players.end(); ++it)
1003 {
1004 ObjectGuid pguid = it->first;
1005 ObjectGuid gguid = it->second.group;
1006 uint32 dungeonId = (*GetSelectedDungeons(pguid).begin());
1007 int32 waitTime = -1;
1008 if (sendUpdate)
1009 SendLfgUpdateProposal(pguid, proposal);
1010
1011 if (!gguid.IsEmpty())
1012 {
1013 waitTime = int32((joinTime - queue.GetJoinTime(gguid)) / IN_MILLISECONDS);
1014 SendLfgUpdateStatus(pguid, updateData, false);
1015 }
1016 else
1017 {
1018 waitTime = int32((joinTime - queue.GetJoinTime(pguid)) / IN_MILLISECONDS);
1019 SendLfgUpdateStatus(pguid, updateData, false);
1020 }
1021 updateData.updateType = LFG_UPDATETYPE_REMOVED_FROM_QUEUE;
1022 SendLfgUpdateStatus(pguid, updateData, true);
1023 SendLfgUpdateStatus(pguid, updateData, false);
1024
1025 // Update timers
1026 uint8 role = GetRoles(pguid);
1027 role &= ~PLAYER_ROLE_LEADER;
1028 switch (role)
1029 {
1030 case PLAYER_ROLE_DAMAGE:
1031 queue.UpdateWaitTimeDps(waitTime, dungeonId);
1032 break;
1033 case PLAYER_ROLE_HEALER:
1034 queue.UpdateWaitTimeHealer(waitTime, dungeonId);
1035 break;
1036 case PLAYER_ROLE_TANK:
1037 queue.UpdateWaitTimeTank(waitTime, dungeonId);
1038 break;
1039 default:
1040 queue.UpdateWaitTimeAvg(waitTime, dungeonId);
1041 break;
1042 }
1043
1044 SetState(pguid, LFG_STATE_DUNGEON);
1045 }
1046
1047 // Remove players/groups from Queue
1048 for (GuidList::const_iterator it = proposal.queues.begin(); it != proposal.queues.end(); ++it)
1049 queue.RemoveFromQueue(*it);
1050
1051 MakeNewGroup(proposal);
1052 ProposalsStore.erase(itProposal);
1053}

1055/**
1056 Remove a proposal from the pool, remove the group that didn't accept (if needed) and readd the other members to the queue
1057
1058 @param[in] itProposal Iterator to the proposal to remove
1059 @param[in] type Type of removal (LFG_UPDATETYPE_PROPOSAL_FAILED, LFG_UPDATETYPE_PROPOSAL_DECLINED)
1060*/
1061void LFGMgr::RemoveProposal(LfgProposalContainer::iterator itProposal, LfgUpdateType type)
1062{
1063 LfgProposal& proposal = itProposal->second;
1064 proposal.state = LFG_PROPOSAL_FAILED;
1065
1066 TC_LOG_DEBUG("lfg.proposal.remove", "Proposal %u, state FAILED, UpdateType %u", itProposal->first, type);
1067 // Mark all people that didn't answered as no accept
1068 if (type == LFG_UPDATETYPE_PROPOSAL_FAILED)
1069 for (LfgProposalPlayerContainer::iterator it = proposal.players.begin(); it != proposal.players.end(); ++it)
1070 if (it->second.accept == LFG_ANSWER_PENDING)
1071 it->second.accept = LFG_ANSWER_DENY;
1072
1073 // Mark players/groups to be removed
1074 GuidSet toRemove;
1075 for (LfgProposalPlayerContainer::iterator it = proposal.players.begin(); it != proposal.players.end(); ++it)
1076 {
1077 if (it->second.accept == LFG_ANSWER_AGREE)
1078 continue;
1079
1080 ObjectGuid guid = !it->second.group.IsEmpty() ? it->second.group : it->first;
1081 // Player didn't accept or still pending when no secs left
1082 if (it->second.accept == LFG_ANSWER_DENY || type == LFG_UPDATETYPE_PROPOSAL_FAILED)
1083 {
1084 it->second.accept = LFG_ANSWER_DENY;
1085 toRemove.insert(guid);
1086 }
1087 }
1088
1089 // Notify players
1090 for (LfgProposalPlayerContainer::const_iterator it = proposal.players.begin(); it != proposal.players.end(); ++it)
1091 {
1092 ObjectGuid guid = it->first;
1093 ObjectGuid gguid = !it->second.group.IsEmpty() ? it->second.group : guid;
1094
1095 SendLfgUpdateProposal(guid, proposal);
1096
1097 if (toRemove.find(gguid) != toRemove.end()) // Didn't accept or in same group that someone that didn't accept
1098 {
1099 LfgUpdateData updateData;
1100 if (it->second.accept == LFG_ANSWER_DENY)
1101 {
1102 updateData.updateType = type;
1103 TC_LOG_DEBUG("lfg.proposal.remove", "%s didn't accept. Removing from queue and compatible cache", guid.ToString().c_str());
1104 }
1105 else
1106 {
1107 updateData.updateType = LFG_UPDATETYPE_REMOVED_FROM_QUEUE;
1108 TC_LOG_DEBUG("lfg.proposal.remove", "%s in same group that someone that didn't accept. Removing from queue and compatible cache", guid.ToString().c_str());
1109 }
1110
1111 RestoreState(guid, "Proposal Fail (didn't accepted or in group with someone that didn't accept");
1112 if (gguid != guid)
1113 {
1114 RestoreState(it->second.group, "Proposal Fail (someone in group didn't accepted)");
1115 SendLfgUpdateStatus(guid, updateData, true);
1116 }
1117 else
1118 SendLfgUpdateStatus(guid, updateData, false);
1119 }
1120 else
1121 {
1122 TC_LOG_DEBUG("lfg.proposal.remove", "Readding %s to queue.", guid.ToString().c_str());
1123 SetState(guid, LFG_STATE_QUEUED);
1124 if (gguid != guid)
1125 {
1126 SetState(gguid, LFG_STATE_QUEUED);
1127 SendLfgUpdateStatus(guid, LfgUpdateData(LFG_UPDATETYPE_ADDED_TO_QUEUE, GetSelectedDungeons(guid), GetComment(guid)), true);
1128 }
1129 else
1130 SendLfgUpdateStatus(guid, LfgUpdateData(LFG_UPDATETYPE_ADDED_TO_QUEUE, GetSelectedDungeons(guid), GetComment(guid)), false);
1131 }
1132 }
1133
1134 LFGQueue& queue = GetQueue(proposal.players.begin()->first);
1135 // Remove players/groups from queue
1136 for (GuidSet::const_iterator it = toRemove.begin(); it != toRemove.end(); ++it)
1137 {
1138 ObjectGuid guid = *it;
1139 queue.RemoveFromQueue(guid);
1140 proposal.queues.remove(guid);
1141 }
1142
1143 // Readd to queue
1144 for (GuidList::const_iterator it = proposal.queues.begin(); it != proposal.queues.end(); ++it)
1145 {
1146 ObjectGuid guid = *it;
1147 queue.AddToQueue(guid);
1148 }
1149
1150 ProposalsStore.erase(itProposal);
1151}

1188/**
1189 Update Boot info with player answer
1190
1191 @param[in] guid Player who has answered
1192 @param[in] player answer
1193*/
1194void LFGMgr::UpdateBoot(ObjectGuid guid, bool accept)
1195{
1196 ObjectGuid gguid = GetGroup(guid);
1197 if (!gguid)
1198 return;
1199
1200 LfgPlayerBootContainer::iterator itBoot = BootsStore.find(gguid);
1201 if (itBoot == BootsStore.end())
1202 return;
1203
1204 LfgPlayerBoot& boot = itBoot->second;
1205
1206 if (boot.votes[guid] != LFG_ANSWER_PENDING) // Cheat check: Player can't vote twice
1207 return;
1208
1209 boot.votes[guid] = LfgAnswer(accept);
1210
1211 uint8 votesNum = 0;
1212 uint8 agreeNum = 0;
1213 for (LfgAnswerContainer::const_iterator itVotes = boot.votes.begin(); itVotes != boot.votes.end(); ++itVotes)

1217 ++votesNum;
1218 if (itVotes->second == LFG_ANSWER_AGREE)
1219 ++agreeNum;
1220 }
1221 }
1222
1223 // if we don't have enough votes (agree or deny) do nothing
1224 if (agreeNum < LFG_GROUP_KICK_VOTES_NEEDED && (votesNum - agreeNum) < LFG_GROUP_KICK_VOTES_NEEDED)
1225 return;
1226
1227 // Send update info to all players
1228 boot.inProgress = false;
1229 for (LfgAnswerContainer::const_iterator itVotes = boot.votes.begin(); itVotes != boot.votes.end(); ++itVotes)
1230 {
1231 ObjectGuid pguid = itVotes->first;
1232 if (pguid != boot.victim)
1233 SendLfgBootProposalUpdate(pguid, boot);
1234 }
1235
1236 SetVoteKick(gguid, false);
1237 if (agreeNum == LFG_GROUP_KICK_VOTES_NEEDED) // Vote passed - Kick player
1238 {
1239 if (Group* group = sGroupMgr->GetGroupByGUID(gguid))
1240 Player::RemoveFromGroup(group, boot.victim, GROUP_REMOVEMETHOD_KICK_LFG);
1241 DecreaseKicksLeft(gguid);
1242 }
1243 BootsStore.erase(itBoot);
1244}

1280
1281 if (!player->IsAlive())
1282 error = LFG_TELEPORTERROR_PLAYER_DEAD;
1283 else if (player->IsFalling() || player->HasUnitState(UNIT_STATE_JUMPING))
1284 error = LFG_TELEPORTERROR_FALLING;
1285 else if (player->IsMirrorTimerActive(FATIGUE_TIMER))
1286 error = LFG_TELEPORTERROR_FATIGUE;
1287 else if (player->GetVehicle())
1288 error = LFG_TELEPORTERROR_IN_VEHICLE;
1289 else if (!player->GetCharmGUID().IsEmpty())
1290 error = LFG_TELEPORTERROR_CHARMING;
1291 else if (player->HasAura(9454)) // check Freeze debuff
1292 error = LFG_TELEPORTERROR_INVALID_LOCATION;
1293 else if (player->GetMapId() != uint32(dungeon->map)) // Do not teleport players in dungeon to the entrance
1294 {
1295 uint32 mapid = dungeon->map;
1296 float x = dungeon->x;
1297 float y = dungeon->y;
1298 float z = dungeon->z;
1299 float orientation = dungeon->o;
1300
1301 if (!fromOpcode)
1302 {
1303 // Select a player inside to be teleported to
1304 for (GroupReference* itr = group->GetFirstMember(); itr != NULL && !mapid; itr = itr->next())
1305 {
1306 Player* plrg = itr->GetSource();
1307 if (plrg && plrg != player && plrg->GetMapId() == uint32(dungeon->map))
1308 {
1309 mapid = plrg->GetMapId();
1310 x = plrg->GetPositionX();
1311 y = plrg->GetPositionY();
1312 z = plrg->GetPositionZ();
1313 orientation = plrg->GetOrientation();
1314 break;
1315 }
1316 }
1317 }
1318
1319 if (!player->GetMap()->IsDungeon())
1320 player->SetBattlegroundEntryPoint();
1321
1322 if (player->IsInFlight())
1323 {
1324 player->GetMotionMaster()->MovementExpired();
1325 player->CleanupAfterTaxiFlight();
1326 }
1327
1328 if (!player->TeleportTo(mapid, x, y, z, orientation))
1329 error = LFG_TELEPORTERROR_INVALID_LOCATION;
1330 }
1331 else
1332 error = LFG_TELEPORTERROR_INVALID_LOCATION;
1333
1334 if (error != LFG_TELEPORTERROR_OK)
1335 player->GetSession()->SendLfgTeleportError(uint8(error));
1336
1337 TC_LOG_DEBUG("lfg.teleport", "Player %s is being teleported in to map %u "
1338 "(x: %f, y: %f, z: %f) Result: %u", player->GetName().c_str(), dungeon->map,
1339 dungeon->x, dungeon->y, dungeon->z, error);
1340}

1342/**
1343 Finish a dungeon and give reward, if any.
1344
1345 @param[in] guid Group guid
1346 @param[in] dungeonId Dungeonid
1347*/
1348void LFGMgr::FinishDungeon(ObjectGuid gguid, const uint32 dungeonId)
1349{
1350 uint32 gDungeonId = GetDungeon(gguid);
1351 if (gDungeonId != dungeonId)
1352 {
1353 TC_LOG_DEBUG("lfg.dungeon.finish", "Group %s finished dungeon %u but queued for %u", gguid.ToString().c_str(), dungeonId, gDungeonId);
1354 return;
1355 }
1356
1357 if (GetState(gguid) == LFG_STATE_FINISHED_DUNGEON) // Shouldn't happen. Do not reward multiple times
1358 {
1359 TC_LOG_DEBUG("lfg.dungeon.finish", "Group: %s already rewarded", gguid.ToString().c_str());
1360 return;
1361 }
1362
1363 SetState(gguid, LFG_STATE_FINISHED_DUNGEON);
1364
1365 GuidSet const& players = GetPlayers(gguid);
1366 for (GuidSet::const_iterator it = players.begin(); it != players.end(); ++it)
1367 {
1368 ObjectGuid guid = (*it);
1369 if (GetState(guid) == LFG_STATE_FINISHED_DUNGEON)
1370 {
1371 TC_LOG_DEBUG("lfg.dungeon.finish", "Group: %s, Player: %s already rewarded", gguid.ToString().c_str(), guid.ToString().c_str());
1372 continue;
1373 }
1374
1375 uint32 rDungeonId = 0;
1376 const LfgDungeonSet& dungeons = GetSelectedDungeons(guid);
1377 if (!dungeons.empty())
1378 rDungeonId = (*dungeons.begin());
1379
1380 SetState(guid, LFG_STATE_FINISHED_DUNGEON);
1381
1382 // Give rewards only if its a random dungeon
1383 LFGDungeonData const* dungeon = GetLFGDungeon(rDungeonId);
1384
1385 if (!dungeon || (dungeon->type != LFG_TYPE_RANDOM && !dungeon->seasonal))
1386 {
1387 TC_LOG_DEBUG("lfg.dungeon.finish", "Group: %s, Player: %s dungeon %u is not random or seasonal", gguid.ToString().c_str(), guid.ToString().c_str(), rDungeonId);
1388 continue;
1389 }
1390
1391 Player* player = ObjectAccessor::FindPlayer(guid);
1392 if (!player)
1393 {
1394 TC_LOG_DEBUG("lfg.dungeon.finish", "Group: %s, Player: %s not found in world", gguid.ToString().c_str(), guid.ToString().c_str());
1395 continue;
1396 }
1397
1398 LFGDungeonData const* dungeonDone = GetLFGDungeon(dungeonId);
1399 uint32 mapId = dungeonDone ? uint32(dungeonDone->map) : 0;
1400
1401 if (player->GetMapId() != mapId)
1402 {
1403 TC_LOG_DEBUG("lfg.dungeon.finish", "Group: %s, Player: %s is in map %u and should be in %u to get reward", gguid.ToString().c_str(), guid.ToString().c_str(), player->GetMapId(), mapId);
1404 continue;
1405 }
1406
1407 // Update achievements
1408 if (dungeon->difficulty == DIFFICULTY_HEROIC)
1409 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_USE_LFD_TO_GROUP_WITH_PLAYERS, 1);
1410
1411 LfgReward const* reward = GetRandomDungeonReward(rDungeonId, player->getLevel());
1412 if (!reward)
1413 continue;
1414
1415 bool done = false;
1416 Quest const* quest = sObjectMgr->GetQuestTemplate(reward->firstQuest);
1417 if (!quest)
1418 continue;
1419
1420 // if we can take the quest, means that we haven't done this kind of "run", IE: First Heroic Random of Day.
1421 if (player->CanRewardQuest(quest, false))
1422 player->RewardQuest(quest, 0, NULL, false);
1423 else
1424 {
1425 done = true;
1426 quest = sObjectMgr->GetQuestTemplate(reward->otherQuest);
1427 if (!quest)
1428 continue;
1429 // we give reward without informing client (retail does this)
1430 player->RewardQuest(quest, 0, NULL, false);
1431 }
1432
1433 // Give rewards
1434 TC_LOG_DEBUG("lfg.dungeon.finish", "Group: %s, Player: %s done dungeon %u, %s previously done.", gguid.ToString().c_str(), guid.ToString().c_str(), GetDungeon(gguid), done ? " " : " not");
1435 LfgPlayerRewardData data = LfgPlayerRewardData(dungeon->Entry(), GetDungeon(gguid, false), done, quest);
1436 player->GetSession()->SendLfgPlayerReward(data);
1437 }
1438}

1577LfgLockMap const LFGMgr::GetLockedDungeons(ObjectGuid guid)
1578{
1579 TC_LOG_TRACE("lfg.data.player.dungeons.locked.get", "Player: %s, LockedDungeons.", guid.ToString().c_str());
1580 LfgLockMap lock;
1581 Player* player = ObjectAccessor::FindConnectedPlayer(guid);
1582 if (!player)
1583 {
1584 TC_LOG_WARN("lfg.data.player.dungeons.locked.get", "Player: %s not ingame while retrieving his LockedDungeons.", guid.ToString().c_str());
1585 return lock;
1586 }
1587
1588 uint8 level = player->getLevel();
1589 uint8 expansion = player->GetSession()->GetExpansion();
1590 LfgDungeonSet const& dungeons = GetDungeonsByRandom(0);
1591 bool denyJoin = !player->GetSession()->HasPermission(rbac::RBAC_PERM_JOIN_DUNGEON_FINDER);
1592
1593 for (LfgDungeonSet::const_iterator it = dungeons.begin(); it != dungeons.end(); ++it)
1594 {
1595 LFGDungeonData const* dungeon = GetLFGDungeon(*it);
1596 if (!dungeon) // should never happen - We provide a list from sLFGDungeonStore
1597 continue;
1598
1599 uint32 lockStatus = 0;
1600 if (denyJoin)
1601 lockStatus = LFG_LOCKSTATUS_RAID_LOCKED;
1602 else if (dungeon->expansion > expansion)
1603 lockStatus = LFG_LOCKSTATUS_INSUFFICIENT_EXPANSION;
1604 else if (DisableMgr::IsDisabledFor(DISABLE_TYPE_MAP, dungeon->map, player))
1605 lockStatus = LFG_LOCKSTATUS_RAID_LOCKED;
1606 else if (dungeon->difficulty > DIFFICULTY_NORMAL && player->GetBoundInstance(dungeon->map, Difficulty(dungeon->difficulty)))
1607 lockStatus = LFG_LOCKSTATUS_RAID_LOCKED;
1608 else if (dungeon->minlevel > level)
1609 lockStatus = LFG_LOCKSTATUS_TOO_LOW_LEVEL;
1610 else if (dungeon->maxlevel < level)
1611 lockStatus = LFG_LOCKSTATUS_TOO_HIGH_LEVEL;
1612 else if (dungeon->seasonal && !IsSeasonActive(dungeon->id))
1613 lockStatus = LFG_LOCKSTATUS_NOT_IN_SEASON;
1614 else if (dungeon->requiredItemLevel > player->GetAverageItemLevel())
1615 lockStatus = LFG_LOCKSTATUS_TOO_LOW_GEAR_SCORE;
1616 else if (AccessRequirement const* ar = sObjectMgr->GetAccessRequirement(dungeon->map, Difficulty(dungeon->difficulty)))
1617 {
1618 if (ar->achievement && !player->HasAchieved(ar->achievement))
1619 lockStatus = LFG_LOCKSTATUS_MISSING_ACHIEVEMENT;
1620 else if (player->GetTeam() == ALLIANCE && ar->quest_A && !player->GetQuestRewardStatus(ar->quest_A))
1621 lockStatus = LFG_LOCKSTATUS_QUEST_NOT_COMPLETED;
1622 else if (player->GetTeam() == HORDE && ar->quest_H && !player->GetQuestRewardStatus(ar->quest_H))
1623 lockStatus = LFG_LOCKSTATUS_QUEST_NOT_COMPLETED;
1624 else
1625 if (ar->item)
1626 {
1627 if (!player->HasItemCount(ar->item) && (!ar->item2 || !player->HasItemCount(ar->item2)))
1628 lockStatus = LFG_LOCKSTATUS_MISSING_ITEM;
1629 }
1630 else if (ar->item2 && !player->HasItemCount(ar->item2))
1631 lockStatus = LFG_LOCKSTATUS_MISSING_ITEM;
1632 }
1633
1634 /* @todo VoA closed if WG is not under team control (LFG_LOCKSTATUS_RAID_LOCKED)
1635 lockData = LFG_LOCKSTATUS_TOO_HIGH_GEAR_SCORE;
1636 lockData = LFG_LOCKSTATUS_ATTUNEMENT_TOO_LOW_LEVEL;
1637 lockData = LFG_LOCKSTATUS_ATTUNEMENT_TOO_HIGH_LEVEL;
1638 */
1639
1640 if (lockStatus)
1641 lock[dungeon->Entry()] = LfgLockInfoData(lockStatus, dungeon->requiredItemLevel, player->GetAverageItemLevel());
1642 }
1643
1644 return lock;
1645}

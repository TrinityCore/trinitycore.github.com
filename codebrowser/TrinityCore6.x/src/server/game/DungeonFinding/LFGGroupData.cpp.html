24LfgGroupData::LfgGroupData(): m_State(LFG_STATE_NONE), m_OldState(LFG_STATE_NONE),
25 m_Leader(), m_Dungeon(0), m_KicksLeft(LFG_GROUP_MAX_KICKS), m_VoteKickActive(false)
26{ }
27
28LfgGroupData::~LfgGroupData()
29{ }
30
31bool LfgGroupData::IsLfgGroup()
32{
33 return m_OldState != LFG_STATE_NONE;
34}
35
36void LfgGroupData::SetState(LfgState state)
37{
38 switch (state)
39 {
40 case LFG_STATE_NONE:
41 m_Dungeon = 0;
42 m_KicksLeft = LFG_GROUP_MAX_KICKS;
43 case LFG_STATE_FINISHED_DUNGEON:
44 case LFG_STATE_DUNGEON:
45 m_OldState = state;
46 // No break on purpose
47 default:
48 m_State = state;
49 }
50}
51
52void LfgGroupData::RestoreState()
53{
54 m_State = m_OldState;
55}
56
57void LfgGroupData::AddPlayer(ObjectGuid guid)
58{
59 m_Players.insert(guid);
60}
61
62uint8 LfgGroupData::RemovePlayer(ObjectGuid guid)
63{
64 GuidSet::iterator it = m_Players.find(guid);
65 if (it != m_Players.end())
66 m_Players.erase(it);
67 return uint8(m_Players.size());
68}
69
70void LfgGroupData::RemoveAllPlayers()
71{
72 m_Players.clear();
73}

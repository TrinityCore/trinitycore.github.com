23LfgPlayerData::LfgPlayerData(): m_State(LFG_STATE_NONE), m_OldState(LFG_STATE_NONE),
24 m_Team(0), m_Group(), m_Roles(0), m_Comment("")
25{ }

27LfgPlayerData::~LfgPlayerData() { }
28
29void LfgPlayerData::SetState(LfgState state)
30{
31 switch (state)
32 {
33 case LFG_STATE_NONE:
34 case LFG_STATE_FINISHED_DUNGEON:
35 m_Roles = 0;
36 m_SelectedDungeons.clear();
37 m_Comment.clear();
38 // No break on purpose
39 case LFG_STATE_DUNGEON:
40 m_OldState = state;
41 // No break on purpose
42 default:
43 m_State = state;
44 }
45}
46
47void LfgPlayerData::RestoreState()
48{
49 if (m_OldState == LFG_STATE_NONE)
50 {
51 m_SelectedDungeons.clear();
52 m_Roles = 0;
53 }
54 m_State = m_OldState;
55}

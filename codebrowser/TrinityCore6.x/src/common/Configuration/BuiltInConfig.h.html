18#ifndef BUILT_IN_CONFIG_H
19#define BUILT_IN_CONFIG_H
20
21#include <string>
22
23/// Provides helper functions to access built-in values
24/// which can be overwritten in config
25namespace BuiltInConfig
26{
27 /// Returns the CMake command when any is specified in the config,
28 /// returns the built-in path otherwise
29 std::string GetCMakeCommand();
30 /// Returns the build directory path when any is specified in the config,
31 /// returns the built-in one otherwise
32 std::string GetBuildDirectory();
33 /// Returns the source directory path when any is specified in the config,
34 /// returns the built-in one otherwise
35 std::string GetSourceDirectory();
36 /// Returns the path to the mysql executable (`mysql`) when any is specified
37 /// in the config, returns the built-in one otherwise
38 std::string GetMySQLExecutable();
39
40} // namespace BuiltInConfig
41
42#endif // BUILT_IN_CONFIG_H

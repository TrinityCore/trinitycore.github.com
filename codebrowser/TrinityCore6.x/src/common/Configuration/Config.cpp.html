28bool ConfigMgr::LoadInitial(std::string const& file, std::string& error)
29{
30 std::lock_guard<std::mutex> lock(_configLock);
31
32 _filename = file;
33
34 try
35 {
36 ptree fullTree;
37 ini_parser::read_ini(file, fullTree);
38
39 if (fullTree.empty())
40 {
41 error = "empty file (" + file + ")";
42 return false;
43 }
44
45 // Since we're using only one section per config file, we skip the section and have direct property access
46 _config = fullTree.begin()->second;
47 }
48 catch (ini_parser::ini_parser_error const& e)
49 {
50 if (e.line() == 0)
51 error = e.message() + " (" + e.filename() + ")";
52 else
53 error = e.message() + " (" + e.filename() + ":" + std::to_string(e.line()) + ")";
54 return false;
55 }
56
57 return true;
58}

60bool ConfigMgr::Reload(std::string& error)

65std::string ConfigMgr::GetStringDefault(std::string const& name, const std::string& def) const
66{
67 std::string value = _config.get<std::string>(ptree::path_type(name, '/'), def);
68
69 value.erase(std::remove(value.begin(), value.end(), '"'), value.end());
70
71 return value;
72}
73
74bool ConfigMgr::GetBoolDefault(std::string const& name, bool def) const
75{
76 try
77 {
78 std::string val = _config.get<std::string>(ptree::path_type(name, '/'));
79 val.erase(std::remove(val.begin(), val.end(), '"'), val.end());
80 return (val == "true" || val == "TRUE" || val == "yes" || val == "YES" || val == "1");
81 }
82 catch (std::exception const& /*ex*/)
83 {
84 return def;
85 }
86}

104std::list<std::string> ConfigMgr::GetKeysByString(std::string const& name)
105{
106 std::lock_guard<std::mutex> lock(_configLock);
107
108 std::list<std::string> keys;
109
110 for (const ptree::value_type& child : _config)
111 if (child.first.compare(0, name.length(), name) == 0)
112 keys.push_back(child.first);
113
114 return keys;
115}

19#include "HmacHash.h"

23template<HashCreateFn HashCreator, uint32 DigestLength>
24HmacHash<HashCreator, DigestLength>::HmacHash(uint32 len, uint8 const* seed)
25{
26 HMAC_CTX_init(&_ctx);
27 HMAC_Init_ex(&_ctx, seed, len, HashCreator(), NULL);
28 memset(_digest, 0, DigestLength);
29}
30
31template<HashCreateFn HashCreator, uint32 DigestLength>
32HmacHash<HashCreator, DigestLength>::~HmacHash()
33{
34 HMAC_CTX_cleanup(&_ctx);
35}
36
37template<HashCreateFn HashCreator, uint32 DigestLength>
38void HmacHash<HashCreator, DigestLength>::UpdateData(const std::string &str)
39{
40 HMAC_Update(&_ctx, (uint8 const*)str.c_str(), str.length());
41}
42
43template<HashCreateFn HashCreator, uint32 DigestLength>
44void HmacHash<HashCreator, DigestLength>::UpdateData(const uint8* data, size_t len)
45{
46 HMAC_Update(&_ctx, data, len);
47}
48
49template<HashCreateFn HashCreator, uint32 DigestLength>
50void HmacHash<HashCreator, DigestLength>::Finalize()
51{
52 uint32 length = 0;
53 HMAC_Final(&_ctx, _digest, &length);
54 ASSERT(length == DigestLength);
55}
56
57template<HashCreateFn HashCreator, uint32 DigestLength>
58uint8* HmacHash<HashCreator, DigestLength>::ComputeHash(BigNumber* bn)
59{
60 HMAC_Update(&_ctx, bn->AsByteArray().get(), bn->GetNumBytes());
61 Finalize();
62 return _digest;
63}

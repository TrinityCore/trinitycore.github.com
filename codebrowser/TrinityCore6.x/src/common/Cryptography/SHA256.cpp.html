18#include "SHA256.h"
19#include "BigNumber.h"
20#include <cstring>
21#include <stdarg.h>
22
23SHA256Hash::SHA256Hash()
24{
25 SHA256_Init(&mC);
26 memset(mDigest, 0, SHA256_DIGEST_LENGTH * sizeof(uint8));
27}
28
29SHA256Hash::~SHA256Hash()
30{
31 SHA256_Init(&mC);
32}
33
34void SHA256Hash::UpdateData(const uint8 *dta, int len)
35{
36 SHA256_Update(&mC, dta, len);
37}
38
39void SHA256Hash::UpdateData(const std::string &str)
40{
41 UpdateData((uint8 const*)str.c_str(), str.length());
42}
43
44void SHA256Hash::UpdateBigNumbers(BigNumber* bn0, ...)
45{
46 va_list v;
47 BigNumber* bn;
48
49 va_start(v, bn0);
50 bn = bn0;
51 while (bn)
52 {
53 UpdateData(bn->AsByteArray().get(), bn->GetNumBytes());
54 bn = va_arg(v, BigNumber*);
55 }
56 va_end(v);
57}
58
59void SHA256Hash::Initialize()
60{
61 SHA256_Init(&mC);
62}
63
64void SHA256Hash::Finalize(void)
65{
66 SHA256_Final(mDigest, &mC);
67}

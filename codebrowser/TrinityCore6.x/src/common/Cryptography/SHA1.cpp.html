19#include "SHA1.h"
20#include "BigNumber.h"
21#include <cstring>
22#include <stdarg.h>
23
24SHA1Hash::SHA1Hash()
25{
26 SHA1_Init(&mC);
27 memset(mDigest, 0, SHA_DIGEST_LENGTH * sizeof(uint8));
28}
29
30SHA1Hash::~SHA1Hash()
31{
32 SHA1_Init(&mC);
33}
34
35void SHA1Hash::UpdateData(const uint8 *dta, int len)
36{
37 SHA1_Update(&mC, dta, len);
38}
39
40void SHA1Hash::UpdateData(const std::string &str)
41{
42 UpdateData((uint8 const*)str.c_str(), str.length());
43}
44
45void SHA1Hash::UpdateBigNumbers(BigNumber* bn0, ...)
46{
47 va_list v;
48 BigNumber* bn;
49
50 va_start(v, bn0);
51 bn = bn0;
52 while (bn)
53 {
54 UpdateData(bn->AsByteArray().get(), bn->GetNumBytes());
55 bn = va_arg(v, BigNumber*);
56 }
57 va_end(v);
58}
59
60void SHA1Hash::Initialize()
61{
62 SHA1_Init(&mC);
63}
64
65void SHA1Hash::Finalize(void)
66{
67 SHA1_Final(mDigest, &mC);
68}

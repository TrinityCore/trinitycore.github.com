19#ifndef _AUTH_HMAC_H
20#define _AUTH_HMAC_H
21
22#include "Define.h"
23#include <string>
24#include <openssl/hmac.h>
25#include <openssl/sha.h>
26
27class BigNumber;
28
29#define SEED_KEY_SIZE 16
30
31typedef EVP_MD const* (*HashCreateFn)();
32
33template<HashCreateFn HashCreator, uint32 DigestLength>
34class HmacHash
35{
36 public:
37 HmacHash(uint32 len, uint8 const* seed);
38 ~HmacHash();
39 void UpdateData(std::string const& str);
40 void UpdateData(uint8 const* data, size_t len);
41 void Finalize();
42 uint8* ComputeHash(BigNumber* bn);
43 uint8* GetDigest() { return _digest; }
44 uint32 GetLength() const { return DigestLength; }
45 private:
46 HMAC_CTX _ctx;
47 uint8 _digest[DigestLength];
48};
49
50typedef HmacHash<EVP_sha1, SHA_DIGEST_LENGTH> HmacSha1;
51typedef HmacHash<EVP_sha256, SHA256_DIGEST_LENGTH> HmacSha256;

53#endif

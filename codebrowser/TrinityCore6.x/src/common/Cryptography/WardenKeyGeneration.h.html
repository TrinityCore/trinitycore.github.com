19#include "SHA1.h"
20
21#include <cstring>
22
23#ifndef _WARDEN_KEY_GENERATION_H
24#define _WARDEN_KEY_GENERATION_H
25
26class SHA1Randx
27{
28public:
29 SHA1Randx(uint8* buff, uint32 size)
30 {
31 uint32 halfSize = size / 2;
32
33 sh.Initialize();
34 sh.UpdateData(buff, halfSize);
35 sh.Finalize();
36
37 memcpy(o1, sh.GetDigest(), 20);
38
39 sh.Initialize();
40 sh.UpdateData(buff + halfSize, size - halfSize);
41 sh.Finalize();
42
43 memcpy(o2, sh.GetDigest(), 20);
44
45 memset(o0, 0x00, 20);
46
47 FillUp();
48 }
49
50 void Generate(uint8* buf, uint32 sz)
51 {
52 for (uint32 i = 0; i < sz; ++i)
53 {
54 if (taken == 20)
55 FillUp();
56
57 buf[i] = o0[taken];
58 taken++;
59 }
60 }
61
62private:
63 void FillUp()
64 {
65 sh.Initialize();
66 sh.UpdateData(o1, 20);
67 sh.UpdateData(o0, 20);
68 sh.UpdateData(o2, 20);
69 sh.Finalize();
70
71 memcpy(o0, sh.GetDigest(), 20);
72
73 taken = 0;
74 }
75
76 SHA1Hash sh;
77 uint32 taken;
78 uint8 o0[20], o1[20], o2[20];
79};

81#endif

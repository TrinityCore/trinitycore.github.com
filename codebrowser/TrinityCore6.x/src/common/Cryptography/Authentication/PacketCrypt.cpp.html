20PacketCrypt::PacketCrypt(uint32 rc4InitSize)
21 : _clientDecrypt(rc4InitSize), _serverEncrypt(rc4InitSize), _initialized(false)
22{
23}
24
25void PacketCrypt::DecryptRecv(uint8* data, size_t len)
26{
27 if (!_initialized)
28 return;
29
30 _clientDecrypt.UpdateData(len, data);
31}
32
33void PacketCrypt::EncryptSend(uint8* data, size_t len)
34{
35 if (!_initialized)
36 return;
37
38 _serverEncrypt.UpdateData(len, data);
39}

21char const* localeNames[TOTAL_LOCALES] =
22{
23 "enUS",
24 "koKR",
25 "frFR",
26 "deDE",
27 "zhCN",
28 "zhTW",
29 "esES",
30 "esMX",
31 "ruRU",
32 "none",
33 "ptBR",
34 "itIT"
35};
36
37LocaleConstant GetLocaleByName(const std::string& name)
38{
39 for (uint32 i = 0; i < TOTAL_LOCALES; ++i)
40 if (name == localeNames[i])
41 return LocaleConstant(i);
42
43 return LOCALE_enUS; // including enGB case
44}

182//! Hash implementation for std::pair to allow using pairs in unordered_set or as key for unordered_map
183//! Individual types used in pair must be hashable by boost::hash
184namespace std
185{
186 template<class K, class V>
187 struct hash<std::pair<K, V>>
188 {
189 public:
190 size_t operator()(std::pair<K, V> const& key) const
191 {
192 return boost::hash_value(key);
193 }
194 };
195}

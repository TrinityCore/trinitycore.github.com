35template<typename T, class S> struct Finder
36{
37 T val_;
38 T S::* idMember_;
39
40 Finder(T val, T S::* idMember) : val_(val), idMember_(idMember) {}
41 bool operator()(const std::pair<int, S> &obj) { return obj.second.*idMember_ == val_; }
42};

130inline bool isBasicLatinCharacter(wchar_t wchar)
131{
132 if (wchar >= L'a' && wchar <= L'z') // LATIN SMALL LETTER A - LATIN SMALL LETTER Z
133 return true;
134 if (wchar >= L'A' && wchar <= L'Z') // LATIN CAPITAL LETTER A - LATIN CAPITAL LETTER Z
135 return true;
136 return false;
137}
138
139inline bool isExtendedLatinCharacter(wchar_t wchar)
140{
141 if (isBasicLatinCharacter(wchar))
142 return true;
143 if (wchar >= 0x00C0 && wchar <= 0x00D6) // LATIN CAPITAL LETTER A WITH GRAVE - LATIN CAPITAL LETTER O WITH DIAERESIS
144 return true;
145 if (wchar >= 0x00D8 && wchar <= 0x00DE) // LATIN CAPITAL LETTER O WITH STROKE - LATIN CAPITAL LETTER THORN
146 return true;
147 if (wchar == 0x00DF) // LATIN SMALL LETTER SHARP S
148 return true;
149 if (wchar >= 0x00E0 && wchar <= 0x00F6) // LATIN SMALL LETTER A WITH GRAVE - LATIN SMALL LETTER O WITH DIAERESIS
150 return true;
151 if (wchar >= 0x00F8 && wchar <= 0x00FE) // LATIN SMALL LETTER O WITH STROKE - LATIN SMALL LETTER THORN
152 return true;
153 if (wchar >= 0x0100 && wchar <= 0x012F) // LATIN CAPITAL LETTER A WITH MACRON - LATIN SMALL LETTER I WITH OGONEK
154 return true;
155 if (wchar == 0x1E9E) // LATIN CAPITAL LETTER SHARP S
156 return true;
157 return false;
158}

160inline bool isCyrillicCharacter(wchar_t wchar)
161{
162 if (wchar >= 0x0410 && wchar <= 0x044F) // CYRILLIC CAPITAL LETTER A - CYRILLIC SMALL LETTER YA
163 return true;
164 if (wchar == 0x0401 || wchar == 0x0451) // CYRILLIC CAPITAL LETTER IO, CYRILLIC SMALL LETTER IO
165 return true;
166 return false;
167}

169inline bool isEastAsianCharacter(wchar_t wchar)
170{
171 if (wchar >= 0x1100 && wchar <= 0x11F9) // Hangul Jamo
172 return true;
173 if (wchar >= 0x3041 && wchar <= 0x30FF) // Hiragana + Katakana
174 return true;
175 if (wchar >= 0x3131 && wchar <= 0x318E) // Hangul Compatibility Jamo
176 return true;
177 if (wchar >= 0x31F0 && wchar <= 0x31FF) // Katakana Phonetic Ext.
178 return true;
179 if (wchar >= 0x3400 && wchar <= 0x4DB5) // CJK Ideographs Ext. A
180 return true;
181 if (wchar >= 0x4E00 && wchar <= 0x9FC3) // Unified CJK Ideographs
182 return true;
183 if (wchar >= 0xAC00 && wchar <= 0xD7A3) // Hangul Syllables
184 return true;
185 if (wchar >= 0xFF01 && wchar <= 0xFFEE) // Halfwidth forms
186 return true;
187 return false;
188}

246inline wchar_t wcharToUpper(wchar_t wchar)
247{
248 if (wchar >= L'a' && wchar <= L'z') // LATIN SMALL LETTER A - LATIN SMALL LETTER Z
249 return wchar_t(uint16(wchar)-0x0020);
250 if (wchar == 0x00DF) // LATIN SMALL LETTER SHARP S
251 return wchar_t(0x1E9E);
252 if (wchar >= 0x00E0 && wchar <= 0x00F6) // LATIN SMALL LETTER A WITH GRAVE - LATIN SMALL LETTER O WITH DIAERESIS
253 return wchar_t(uint16(wchar)-0x0020);
254 if (wchar >= 0x00F8 && wchar <= 0x00FE) // LATIN SMALL LETTER O WITH STROKE - LATIN SMALL LETTER THORN
255 return wchar_t(uint16(wchar)-0x0020);
256 if (wchar >= 0x0101 && wchar <= 0x012F) // LATIN SMALL LETTER A WITH MACRON - LATIN SMALL LETTER I WITH OGONEK (only %2=1)
257 {
258 if (wchar % 2 == 1)
259 return wchar_t(uint16(wchar)-0x0001);
260 }
261 if (wchar >= 0x0430 && wchar <= 0x044F) // CYRILLIC SMALL LETTER A - CYRILLIC SMALL LETTER YA
262 return wchar_t(uint16(wchar)-0x0020);
263 if (wchar == 0x0451) // CYRILLIC SMALL LETTER IO
264 return wchar_t(0x0401);
265
266 return wchar;
267}

274inline wchar_t wcharToLower(wchar_t wchar)
275{
276 if (wchar >= L'A' && wchar <= L'Z') // LATIN CAPITAL LETTER A - LATIN CAPITAL LETTER Z
277 return wchar_t(uint16(wchar)+0x0020);
278 if (wchar >= 0x00C0 && wchar <= 0x00D6) // LATIN CAPITAL LETTER A WITH GRAVE - LATIN CAPITAL LETTER O WITH DIAERESIS
279 return wchar_t(uint16(wchar)+0x0020);
280 if (wchar >= 0x00D8 && wchar <= 0x00DE) // LATIN CAPITAL LETTER O WITH STROKE - LATIN CAPITAL LETTER THORN
281 return wchar_t(uint16(wchar)+0x0020);
282 if (wchar >= 0x0100 && wchar <= 0x012E) // LATIN CAPITAL LETTER A WITH MACRON - LATIN CAPITAL LETTER I WITH OGONEK (only %2=0)
283 {
284 if (wchar % 2 == 0)
285 return wchar_t(uint16(wchar)+0x0001);
286 }
287 if (wchar == 0x1E9E) // LATIN CAPITAL LETTER SHARP S
288 return wchar_t(0x00DF);
289 if (wchar == 0x0401) // CYRILLIC CAPITAL LETTER IO
290 return wchar_t(0x0451);
291 if (wchar >= 0x0410 && wchar <= 0x042F) // CYRILLIC CAPITAL LETTER A - CYRILLIC CAPITAL LETTER YA
292 return wchar_t(uint16(wchar)+0x0020);
293
294 return wchar;
295}

357class flag128
358{
359private:
360 uint32 part[4];
361
362public:
363 flag128(uint32 p1 = 0, uint32 p2 = 0, uint32 p3 = 0, uint32 p4 = 0)
364 {
365 part[0] = p1;
366 part[1] = p2;
367 part[2] = p3;
368 part[3] = p4;
369 }
370
371 inline bool IsEqual(uint32 p1 = 0, uint32 p2 = 0, uint32 p3 = 0, uint32 p4 = 0) const
372 {
373 return (part[0] == p1 && part[1] == p2 && part[2] == p3 && part[3] == p4);
374 }
375
376 inline bool HasFlag(uint32 p1 = 0, uint32 p2 = 0, uint32 p3 = 0, uint32 p4 = 0) const
377 {
378 return (part[0] & p1 || part[1] & p2 || part[2] & p3 || part[3] & p4);
379 }
380
381 inline void Set(uint32 p1 = 0, uint32 p2 = 0, uint32 p3 = 0, uint32 p4 = 0)
382 {
383 part[0] = p1;
384 part[1] = p2;
385 part[2] = p3;
386 part[3] = p4;
387 }
388
389 inline bool operator <(const flag128 &right) const
390 {
391 for (uint8 i = 4; i > 0; --i)
392 {
393 if (part[i - 1] < right.part[i - 1])
394 return true;
395 else if (part[i - 1] > right.part[i - 1])
396 return false;
397 }
398 return false;
399 }
400
401 inline bool operator ==(const flag128 &right) const
402 {
403 return
404 (
405 part[0] == right.part[0] &&
406 part[1] == right.part[1] &&
407 part[2] == right.part[2] &&
408 part[3] == right.part[3]
409 );
410 }
411
412 inline bool operator !=(const flag128 &right) const
413 {
414 return !this->operator ==(right);
415 }
416
417 inline flag128 & operator =(const flag128 &right)
418 {
419 part[0] = right.part[0];
420 part[1] = right.part[1];
421 part[2] = right.part[2];
422 part[3] = right.part[3];
423 return *this;
424 }
425
426 inline flag128 operator &(const flag128 &right) const
427 {
428 return flag128(part[0] & right.part[0], part[1] & right.part[1],
429 part[2] & right.part[2], part[3] & right.part[3]);
430 }
431
432 inline flag128 & operator &=(const flag128 &right)
433 {
434 part[0] &= right.part[0];
435 part[1] &= right.part[1];
436 part[2] &= right.part[2];
437 part[3] &= right.part[3];
438 return *this;
439 }
440
441 inline flag128 operator |(const flag128 &right) const
442 {
443 return flag128(part[0] | right.part[0], part[1] | right.part[1],
444 part[2] | right.part[2], part[3] | right.part[3]);
445 }
446
447 inline flag128 & operator |=(const flag128 &right)
448 {
449 part[0] |= right.part[0];
450 part[1] |= right.part[1];
451 part[2] |= right.part[2];
452 part[3] |= right.part[3];
453 return *this;
454 }
455
456 inline flag128 operator ~() const
457 {
458 return flag128(~part[0], ~part[1], ~part[2], ~part[3]);
459 }
460
461 inline flag128 operator ^(const flag128 &right) const
462 {
463 return flag128(part[0] ^ right.part[0], part[1] ^ right.part[1],
464 part[2] ^ right.part[2], part[3] ^ right.part[3]);
465 }
466
467 inline flag128 & operator ^=(const flag128 &right)
468 {
469 part[0] ^= right.part[0];
470 part[1] ^= right.part[1];
471 part[2] ^= right.part[2];
472 part[3] ^= right.part[3];
473 return *this;
474 }
475
476 inline operator bool() const
477 {
478 return (part[0] != 0 || part[1] != 0 || part[2] != 0 || part[3] != 0);
479 }
480
481 inline bool operator !() const
482 {
483 return !this->operator bool();
484 }

486 inline uint32 & operator [](uint8 el)
487 {
488 return part[el];
489 }
490
491 inline const uint32 & operator [](uint8 el) const
492 {
493 return part[el];
494 }

497enum ComparisionType
498{
499 COMP_TYPE_EQ = 0,
500 COMP_TYPE_HIGH,
501 COMP_TYPE_LOW,
502 COMP_TYPE_HIGH_EQ,
503 COMP_TYPE_LOW_EQ,
504 COMP_TYPE_MAX
505};
506
507template <class T>
508bool CompareValues(ComparisionType type, T val1, T val2)
509{
510 switch (type)
511 {
512 case COMP_TYPE_EQ:
513 return val1 == val2;
514 case COMP_TYPE_HIGH:
515 return val1 > val2;
516 case COMP_TYPE_LOW:
517 return val1 < val2;
518 case COMP_TYPE_HIGH_EQ:
519 return val1 >= val2;
520 case COMP_TYPE_LOW_EQ:
521 return val1 <= val2;
522 default:
523 // incorrect parameter
524 ABORT();
525 return false;
526 }
527}

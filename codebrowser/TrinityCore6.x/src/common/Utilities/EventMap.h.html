18#ifndef _EVENT_MAP_H_
19#define _EVENT_MAP_H_
20
21#include "Common.h"
22#include "Duration.h"
23#include "Util.h"
24
25class EventMap
26{
27 /**
28 * Internal storage type.
29 * Key: Time as uint32 when the event should occur.
30 * Value: The event data as uint32.
31 *
32 * Structure of event data:
33 * - Bit 0 - 15: Event Id.
34 * - Bit 16 - 23: Group
35 * - Bit 24 - 31: Phase
36 * - Pattern: 0xPPGGEEEE
37 */
38 typedef std::multimap<uint32, uint32> EventStore;
39
40public:
41 EventMap() : _time(0), _phase(0), _lastEvent(0) { }
42
43 /**
44 * @name Reset
45 * @brief Removes all scheduled events and resets time and phase.
46 */
47 void Reset();
48
49 /**
50 * @name Update
51 * @brief Updates the timer of the event map.
52 * @param time Value in ms to be added to time.
53 */
54 void Update(uint32 time)
55 {
56 _time += time;
57 }
58
59 /**
60 * @name GetTimer
61 * @return Current timer in ms value.
62 */
63 uint32 GetTimer() const
64 {
65 return _time;
66 }
67
68 /**
69 * @name GetPhaseMask
70 * @return Active phases as mask.
71 */
72 uint8 GetPhaseMask() const
73 {
74 return _phase;
75 }
76
77 /**
78 * @name Empty
79 * @return True, if there are no events scheduled.
80 */
81 bool Empty() const
82 {
83 return _eventMap.empty();
84 }
85
86 /**
87 * @name SetPhase
88 * @brief Sets the phase of the map (absolute).
89 * @param phase Phase which should be set. Values: 1 - 8. 0 resets phase.
90 */
91 void SetPhase(uint8 phase);
92
93 /**
94 * @name AddPhase
95 * @brief Activates the given phase (bitwise).
96 * @param phase Phase which should be activated. Values: 1 - 8
97 */
98 void AddPhase(uint8 phase)
99 {
100 if (phase && phase <= 8)
101 _phase |= uint8(1 << (phase - 1));
102 }
103
104 /**
105 * @name RemovePhase
106 * @brief Deactivates the given phase (bitwise).
107 * @param phase Phase which should be deactivated. Values: 1 - 8.
108 */
109 void RemovePhase(uint8 phase)
110 {
111 if (phase && phase <= 8)
112 _phase &= uint8(~(1 << (phase - 1)));
113 }
114
115 /**
116 * @name ScheduleEvent
117 * @brief Creates new event entry in map.
118 * @param eventId The id of the new event.
119 * @param time The time in milliseconds as std::chrono::duration until the event occurs.
120 * @param group The group which the event is associated to. Has to be between 1 and 8. 0 means it has no group.
121 * @param phase The phase in which the event can occur. Has to be between 1 and 8. 0 means it can occur in all phases.
122 */
123 void ScheduleEvent(uint32 eventId, Milliseconds const& time, uint32 group = 0, uint8 phase = 0)
124 {
125 ScheduleEvent(eventId, time.count(), group, phase);
126 }
127
128 /**
129 * @name ScheduleEvent
130 * @brief Creates new event entry in map.
131 * @param eventId The id of the new event.
132 * @param time The time in milliseconds until the event occurs.
133 * @param group The group which the event is associated to. Has to be between 1 and 8. 0 means it has no group.
134 * @param phase The phase in which the event can occur. Has to be between 1 and 8. 0 means it can occur in all phases.
135 */
136 void ScheduleEvent(uint32 eventId, uint32 time, uint32 group = 0, uint8 phase = 0);
137
138 /**
139 * @name RescheduleEvent
140 * @brief Cancels the given event and reschedules it.
141 * @param eventId The id of the event.
142 * @param time The time in milliseconds as std::chrono::duration until the event occurs.
143 * @param group The group which the event is associated to. Has to be between 1 and 8. 0 means it has no group.
144 * @param phase The phase in which the event can occur. Has to be between 1 and 8. 0 means it can occur in all phases.
145 */
146 void RescheduleEvent(uint32 eventId, Milliseconds const& time, uint32 group = 0, uint8 phase = 0)
147 {
148 RescheduleEvent(eventId, time.count(), group, phase);
149 }
150
151 /**
152 * @name RescheduleEvent
153 * @brief Cancels the given event and reschedules it.
154 * @param eventId The id of the event.
155 * @param time The time in milliseconds until the event occurs.
156 * @param group The group which the event is associated to. Has to be between 1 and 8. 0 means it has no group.
157 * @param phase The phase in which the event can occur. Has to be between 1 and 8. 0 means it can occur in all phases.
158 */
159 void RescheduleEvent(uint32 eventId, uint32 time, uint32 group = 0, uint8 phase = 0)
160 {
161 CancelEvent(eventId);
162 ScheduleEvent(eventId, time, group, phase);
163 }
164
165 /**
166 * @name RepeatEvent
167 * @brief Repeats the mostly recently executed event.
168 * @param time Time until in milliseconds as std::chrono::duration the event occurs.
169 */
170 void Repeat(Milliseconds const& time)
171 {
172 Repeat(time.count());
173 }
174
175 /**
176 * @name RepeatEvent
177 * @brief Repeats the mostly recently executed event.
178 * @param time Time until the event occurs.
179 */
180 void Repeat(uint32 time)
181 {
182 _eventMap.insert(EventStore::value_type(_time + time, _lastEvent));
183 }
184
185 /**
186 * @name RepeatEvent
187 * @brief Repeats the mostly recently executed event.
188 * @param minTime Minimum time as std::chrono::duration until the event occurs.
189 * @param maxTime Maximum time as std::chrono::duration until the event occurs.
190 */
191 void Repeat(Milliseconds const& minTime, Milliseconds const& maxTime)
192 {
193 Repeat(minTime.count(), maxTime.count());
194 }
195
196 /**
197 * @name RepeatEvent
198 * @brief Repeats the mostly recently executed event, Equivalent to Repeat(urand(minTime, maxTime).
199 * @param minTime Minimum time until the event occurs.
200 * @param maxTime Maximum time until the event occurs.
201 */
202 void Repeat(uint32 minTime, uint32 maxTime)
203 {
204 Repeat(urand(minTime, maxTime));
205 }
206
207 /**
208 * @name ExecuteEvent
209 * @brief Returns the next event to execute and removes it from map.
210 * @return Id of the event to execute.
211 */
212 uint32 ExecuteEvent();
213
214 /**
215 * @name DelayEvents
216 * @brief Delays all events in the map. If delay is greater than or equal internal timer, delay will be 0.
217 * @param delay Amount of delay in ms as std::chrono::duration.
218 */
219 void DelayEvents(Milliseconds const& delay)
220 {
221 DelayEvents(delay.count());
222 }
223
224 /**
225 * @name DelayEvents
226 * @brief Delays all events in the map. If delay is greater than or equal internal timer, delay will be 0.
227 * @param delay Amount of delay.
228 */
229 void DelayEvents(uint32 delay)
230 {
231 _time = delay < _time ? _time - delay : 0;
232 }
233
234 /**
235 * @name DelayEvents
236 * @brief Delay all events of the same group.
237 * @param delay Amount of delay in ms as std::chrono::duration.
238 * @param group Group of the events.
239 */
240 void DelayEvents(Milliseconds const& delay, uint32 group)
241 {
242 DelayEvents(delay.count(), group);
243 }
244
245 /**
246 * @name DelayEvents
247 * @brief Delay all events of the same group.
248 * @param delay Amount of delay.
249 * @param group Group of the events.
250 */
251 void DelayEvents(uint32 delay, uint32 group);
252
253 /**
254 * @name CancelEvent
255 * @brief Cancels all events of the specified id.
256 * @param eventId Event id to cancel.
257 */
258 void CancelEvent(uint32 eventId);
259
260 /**
261 * @name CancelEventGroup
262 * @brief Cancel events belonging to specified group.
263 * @param group Group to cancel.
264 */
265 void CancelEventGroup(uint32 group);
266
267 /**
268 * @name GetNextEventTime
269 * @brief Returns closest occurence of specified event.
270 * @param eventId Wanted event id.
271 * @return Time of found event.
272 */
273 uint32 GetNextEventTime(uint32 eventId) const;
274
275 /**
276 * @name GetNextEventTime
277 * @return Time of next event.
278 */
279 uint32 GetNextEventTime() const
280 {
281 return Empty() ? 0 : _eventMap.begin()->first;
282 }
283
284 /**
285 * @name IsInPhase
286 * @brief Returns whether event map is in specified phase or not.
287 * @param phase Wanted phase.
288 * @return True, if phase of event map contains specified phase.
289 */
290 bool IsInPhase(uint8 phase) const
291 {
292 return phase <= 8 && (!phase || _phase & (1 << (phase - 1)));
293 }
294
295 /**
296 * @name GetTimeUntilEvent
297 * @brief Returns time in milliseconds until next event.
298 * @param eventId of the event.
299 * @return Time of next event.
300 */
301 uint32 GetTimeUntilEvent(uint32 eventId) const;
302
303private:
304 /**
305 * @name _time
306 * @brief Internal timer.
307 *
308 * This does not represent the real date/time value.
309 * It's more like a stopwatch: It can run, it can be stopped,
310 * it can be resetted and so on. Events occur when this timer
311 * has reached their time value. Its value is changed in the
312 * Update method.
313 */
314 uint32 _time;
315
316 /**
317 * @name _phase
318 * @brief Phase mask of the event map.
319 *
320 * Contains the phases the event map is in. Multiple
321 * phases from 1 to 8 can be set with SetPhase or
322 * AddPhase. RemovePhase deactives a phase.
323 */
324 uint8 _phase;
325
326 /**
327 * @name _eventMap
328 * @brief Internal event storage map. Contains the scheduled events.
329 *
330 * See typedef at the beginning of the class for more
331 * details.
332 */
333 EventStore _eventMap;
334
335 /**
336 * @name _lastEvent
337 * @brief Stores information on the most recently executed event
338 */
339 uint32 _lastEvent;
340};
341
342#endif // _EVENT_MAP_H_

190 template <class PMap, class Key>
191 struct ReadablePropertyMapConcept
192 {
193 typedef typename property_traits<PMap>::key_type key_type;
194 typedef typename property_traits<PMap>::reference reference;
195 typedef typename property_traits<PMap>::category Category;
196 typedef boost::readable_property_map_tag ReadableTag;
197 void constraints() {
198 function_requires< ConvertibleConcept<Category, ReadableTag> >();
199
200 val = get(pmap, k);
201 }
202 PMap pmap;
203 Key k;
204 typename property_traits<PMap>::value_type val;
205 };
206 template <typename KeyArchetype, typename ValueArchetype>
207 struct readable_property_map_archetype {
208 typedef KeyArchetype key_type;
209 typedef ValueArchetype value_type;
210 typedef convertible_to_archetype<ValueArchetype> reference;
211 typedef readable_property_map_tag category;
212 };
213 template <typename K, typename V>
214 const typename readable_property_map_archetype<K,V>::reference&
215 get(const readable_property_map_archetype<K,V>&,
216 const typename readable_property_map_archetype<K,V>::key_type&)
217 {
218 typedef typename readable_property_map_archetype<K,V>::reference R;
219 return static_object<R>::get();
220 }

273 template <class PMap, class Key>
274 struct LvaluePropertyMapConcept
275 {
276 typedef typename property_traits<PMap>::category Category;
277 typedef boost::lvalue_property_map_tag LvalueTag;
278 typedef typename property_traits<PMap>::reference reference;
279
280 void constraints() {
281 function_requires< ReadablePropertyMapConcept<PMap, Key> >();
282 function_requires< ConvertibleConcept<Category, LvalueTag> >();
283
284 typedef typename property_traits<PMap>::value_type value_type;
285 BOOST_MPL_ASSERT((boost::mpl::or_<
286 boost::is_same<const value_type&, reference>,
287 boost::is_same<value_type&, reference> >));
288
289 reference ref = pmap[k];
290 ignore_unused_variable_warning(ref);
291 }
292 PMap pmap;
293 Key k;
294 };

302 typedef lvalue_property_map_tag category;
303 const value_type& operator[](const key_type&) const {
304 return static_object<value_type>::get();

367 template <class RandomAccessIterator,
368 class IndexMap
369#ifdef BOOST_NO_STD_ITERATOR_TRAITS
370 , class T, class R
371#else
372 , class T = typename std::iterator_traits<RandomAccessIterator>::value_type
373 , class R = typename std::iterator_traits<RandomAccessIterator>::reference
374#endif
375 >
376 class iterator_property_map
377 : public boost::put_get_helper< R,
378 iterator_property_map<RandomAccessIterator, IndexMap,
379 T, R> >
380 {
381 public:
382 typedef typename property_traits<IndexMap>::key_type key_type;
383 typedef T value_type;
384 typedef R reference;
385 typedef boost::lvalue_property_map_tag category;
386
387 inline iterator_property_map(
388 RandomAccessIterator cc = RandomAccessIterator(),
389 const IndexMap& _id = IndexMap() )
390 : iter(cc), index(_id) { }
391 inline R operator[](key_type v) const { return *(iter + get(index, v)) ; }
392 protected:
393 RandomAccessIterator iter;
394 IndexMap index;
395 };

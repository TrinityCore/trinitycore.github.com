285
286 size_type erase(key_param_type k)
287 {
288 BOOST_MULTI_INDEX_HASHED_INDEX_CHECK_INVARIANT;
289
290 size_type s=0;
291 std::size_t buc=buckets.position(hash_(k));
292 node_impl_pointer x=buckets.at(buc);
293 node_impl_pointer y=x->next();
294 while(y!=x){
295 if(eq_(k,key(node_type::from_impl(y)->value()))){
296 bool b;
297 do{
298 node_impl_pointer z=y->next();
299 b=z!=x&&eq_(
300 key(node_type::from_impl(y)->value()),
301 key(node_type::from_impl(z)->value()));
302 this->final_erase_(
303 static_cast<final_node_type*>(node_type::from_impl(y)));
304 y=z;
305 ++s;
306 }while(b);
307 break;
308 }
309 y=y->next();
310 }
311 return s;
312 }

436 >
437 iterator find(
438 const CompatibleKey& k,
439 const CompatibleHash& hash,const CompatiblePred& eq)const
440 {
441 std::size_t buc=buckets.position(hash(k));
442 node_impl_pointer x=buckets.at(buc);
443 node_impl_pointer y=x->next();
444 while(y!=x){
445 if(eq(k,key(node_type::from_impl(y)->value()))){
446 return make_iterator(node_type::from_impl(y));
447 }
448 y=y->next();
449 }
450 return end();
451 }

459 template<
460 typename CompatibleKey,typename CompatibleHash,typename CompatiblePred
461 >
462 size_type count(
463 const CompatibleKey& k,
464 const CompatibleHash& hash,const CompatiblePred& eq)const
465 {
466 size_type res=0;
467 std::size_t buc=buckets.position(hash(k));
468 node_impl_pointer x=buckets.at(buc);
469 node_impl_pointer y=x->next();
470 while(y!=x){
471 if(eq(k,key(node_type::from_impl(y)->value()))){
472 do{
473 ++res;
474 y=y->next();
475 }while(y!=x&&eq(k,key(node_type::from_impl(y)->value())));
476 break;
477 }
478 y=y->next();
479 }
480 return res;
481 }

489 template<
490 typename CompatibleKey,typename CompatibleHash,typename CompatiblePred
491 >
492 std::pair<iterator,iterator> equal_range(
493 const CompatibleKey& k,
494 const CompatibleHash& hash,const CompatiblePred& eq)const
495 {
496 std::size_t buc=buckets.position(hash(k));
497 node_impl_pointer x=buckets.at(buc);
498 node_impl_pointer y=x->next();
499 while(y!=x){
500 if(eq(k,key(node_type::from_impl(y)->value()))){
501 node_impl_pointer y0=y;
502 do{
503 y=y->next();
504 }while(y!=x&&eq(k,key(node_type::from_impl(y)->value())));
505 if(y==x){
506 do{
507 ++y;
508 }while(y==y->next());
509 y=y->next();
510 }
511 return std::pair<iterator,iterator>(
512 make_iterator(node_type::from_impl(y0)),
513 make_iterator(node_type::from_impl(y)));
514 }
515 y=y->next();
516 }
517 return std::pair<iterator,iterator>(end(),end());
518 }

585 float load_factor()const{return static_cast<float>(size())/bucket_count();}

588
589 void rehash(size_type n)
590 {
591 BOOST_MULTI_INDEX_HASHED_INDEX_CHECK_INVARIANT;
592 if(size()<max_load&&n<=bucket_count())return;
593
594 size_type bc =(std::numeric_limits<size_type>::max)();
595 float fbc=static_cast<float>(1+size()/mlf);
596 if(bc>fbc){
597 bc=static_cast<size_type>(fbc);
598 if(bc<n)bc=n;
599 }
600 unchecked_rehash(bc);
601 }

667 void copy_(
668 const hashed_index<KeyFromValue,Hash,Pred,SuperMeta,TagList,Category>& x,
669 const copy_map_type& map)
670 {
671 for(node_impl_pointer begin_org=x.buckets.begin(),
672 begin_cpy=buckets.begin(),
673 end_org=x.buckets.end();
674 begin_org!=end_org;++begin_org,++begin_cpy){
675
676 node_impl_pointer next_org=begin_org->next();
677 node_impl_pointer cpy=begin_cpy;
678 while(next_org!=begin_org){
679 cpy->next()=
680 static_cast<node_type*>(
681 map.find(
682 static_cast<final_node_type*>(
683 node_type::from_impl(next_org))))->impl();
684 next_org=next_org->next();
685 cpy=cpy->next();
686 }
687 cpy->next()=begin_cpy;
688 }
689
690 super::copy_(x,map);
691 }

693 node_type* insert_(value_param_type v,node_type* x)
694 {
695 reserve(size()+1);
696
697 std::size_t buc=find_bucket(v);
698 node_impl_pointer pos=buckets.at(buc);
699 if(!link_point(v,pos,Category()))return node_type::from_impl(pos);
700
701 node_type* res=static_cast<node_type*>(super::insert_(v,x));
702 if(res==x){
703 link(x,pos);
704 if(first_bucket>buc)first_bucket=buc;
705 }
706 return res;
707 }
708
709 node_type* insert_(value_param_type v,node_type* position,node_type* x)
710 {
711 reserve(size()+1);
712
713 std::size_t buc=find_bucket(v);
714 node_impl_pointer pos=buckets.at(buc);
715 if(!link_point(v,pos,Category()))return node_type::from_impl(pos);

1056 void unchecked_rehash(size_type n)
1057 {
1058 bucket_array_type buckets1(get_allocator(),header()->impl(),n);
1059 auto_space<std::size_t,allocator_type> hashes(get_allocator(),size());
1060
1061 std::size_t i=0;
1062 node_impl_pointer x=buckets.begin();
1063 node_impl_pointer x_end=buckets.end();
1064 for(;x!=x_end;++x){
1065 node_impl_pointer y=x->next();
1066 while(y!=x){
1067 hashes.data()[i++]=hash_(key(node_type::from_impl(y)->value()));
1068 y=y->next();
1069 }
1070 }
1071
1072 i=0;
1073 x=buckets.begin();
1074 for(;x!=x_end;++x){
1075 node_impl_pointer y=x->next();
1076 while(y!=x){
1077 node_impl_pointer z=y->next();
1078 std::size_t buc1=buckets1.position(hashes.data()[i++]);
1079 node_impl_pointer x1=buckets1.at(buc1);
1080 link(y,x1);
1081 y=z;
1082 }
1083 }
1084
1085 buckets.swap(buckets1);
1086 calculate_max_load();

1090 bool in_place(
1091 node_impl_pointer x,key_param_type k,std::size_t buc,
1092 hashed_unique_tag)const
1093 {
1094 std::less_equal<node_impl_pointer> leq;
1095 node_impl_pointer bbegin=buckets.begin();
1096 node_impl_pointer bend=buckets.end();
1097 node_impl_pointer pbuc=x->next();
1098
1099 while(!leq(bbegin,pbuc)||!leq(pbuc,bend))pbuc=pbuc->next();
1100 if(buc!=static_cast<std::size_t>(pbuc-bbegin))return false;
1101
1102 node_impl_pointer y=x;
1103 while(y->next()!=x){
1104 y=y->next();
1105 if(y==pbuc)continue;
1106 if(eq_(k,key(node_type::from_impl(y)->value())))return false;
1107 }
1108 return true;

1121 if(buc!=static_cast<std::size_t>(pbuc-bbegin))return false;

1141 y=y->next();

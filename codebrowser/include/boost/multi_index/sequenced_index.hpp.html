234 void resize(size_type n,value_param_type x=value_type())
235 {
236 BOOST_MULTI_INDEX_SEQ_INDEX_CHECK_INVARIANT;
237 if(n>size())insert(end(),n-size(),x);
238 else if(n<size()){
239 iterator it;
240 if(n<=size()/2){
241 it=begin();
242 std::advance(it,n);
243 }
244 else{
245 it=end();
246 for(size_type m=size()-n;m--;--it){}
247 }
248 erase(it,end());
249 }
250 }

268 std::pair<iterator,bool> insert(iterator position,value_param_type x)
269 {
270 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(position);
271 BOOST_MULTI_INDEX_CHECK_IS_OWNER(position,*this);
272 BOOST_MULTI_INDEX_SEQ_INDEX_CHECK_INVARIANT;
273 std::pair<final_node_type*,bool> p=this->final_insert_(x);
274 if(p.second&&position.get_node()!=header()){
275 relink(position.get_node(),p.first);
276 }
277 return std::pair<iterator,bool>(make_iterator(p.first),p.second);
278 }

382 /* list operations */
383
384 void splice(iterator position,sequenced_index<SuperMeta,TagList>& x)
385 {
386 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(position);
387 BOOST_MULTI_INDEX_CHECK_IS_OWNER(position,*this);
388 BOOST_MULTI_INDEX_CHECK_DIFFERENT_CONTAINER(*this,x);
389 BOOST_MULTI_INDEX_SEQ_INDEX_CHECK_INVARIANT;
390 iterator first=x.begin(),last=x.end();
391 while(first!=last){
392 if(insert(position,*first).second)first=x.erase(first);
393 else ++first;
394 }
395 }
396
397 void splice(iterator position,sequenced_index<SuperMeta,TagList>& x,iterator i)
398 {
399 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(position);
400 BOOST_MULTI_INDEX_CHECK_IS_OWNER(position,*this);
401 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(i);
402 BOOST_MULTI_INDEX_CHECK_DEREFERENCEABLE_ITERATOR(i);
403 BOOST_MULTI_INDEX_CHECK_IS_OWNER(i,x);
404 BOOST_MULTI_INDEX_SEQ_INDEX_CHECK_INVARIANT;
405 if(&x==this){
406 if(position!=i)relink(position.get_node(),i.get_node());
407 }
408 else{
409 if(insert(position,*i).second){
410
411#if defined(BOOST_MULTI_INDEX_ENABLE_SAFE_MODE)
412 /* MSVC++ 6.0 optimizer has a hard time with safe mode, and the following
413 * workaround is needed. Left it for all compilers as it does no
414 * harm.
415 */
416 i.detach();
417 x.erase(x.make_iterator(i.get_node()));
418#else
419 x.erase(i);
420#endif
421
422 }
423 }
424 }

426 void splice(
427 iterator position,sequenced_index<SuperMeta,TagList>& x,
428 iterator first,iterator last)
429 {
430 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(position);
431 BOOST_MULTI_INDEX_CHECK_IS_OWNER(position,*this);
432 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(first);
433 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(last);
434 BOOST_MULTI_INDEX_CHECK_IS_OWNER(first,x);
435 BOOST_MULTI_INDEX_CHECK_IS_OWNER(last,x);
436 BOOST_MULTI_INDEX_CHECK_VALID_RANGE(first,last);
437 BOOST_MULTI_INDEX_SEQ_INDEX_CHECK_INVARIANT;
438 if(&x==this){
439 BOOST_MULTI_INDEX_CHECK_OUTSIDE_RANGE(position,first,last);
440 if(position!=last)relink(
441 position.get_node(),first.get_node(),last.get_node());
442 }
443 else{
444 while(first!=last){
445 if(insert(position,*first).second)first=x.erase(first);
446 else ++first;
447 }
448 }
449 }

577 void copy_(
578 const sequenced_index<SuperMeta,TagList>& x,const copy_map_type& map)
579 {
580 node_type* org=x.header();
581 node_type* cpy=header();
582 do{
583 node_type* next_org=node_type::from_impl(org->next());
584 node_type* next_cpy=map.find(static_cast<final_node_type*>(next_org));
585 cpy->next()=next_cpy->impl();
586 next_cpy->prior()=cpy->impl();
587 org=next_org;
588 cpy=next_cpy;
589 }while(org!=x.header());
590
591 super::copy_(x,map);
592 }

704#if defined(BOOST_MULTI_INDEX_ENABLE_INVARIANT_CHECKING)
705 /* invariant stuff */
706
707 bool invariant_()const
708 {
709 if(size()==0||begin()==end()){
710 if(size()!=0||begin()!=end()||
711 header()->next()!=header()->impl()||
712 header()->prior()!=header()->impl())return false;
713 }
714 else{
715 size_type s=0;
716 for(const_iterator it=begin(),it_end=end();it!=it_end;++it,++s){
717 if(it.get_node()->next()->prior()!=it.get_node()->impl())return false;

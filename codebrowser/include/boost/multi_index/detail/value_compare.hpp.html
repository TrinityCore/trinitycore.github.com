20namespace boost{
21
22namespace multi_index{
23
24namespace detail{
25
26template<typename Value,typename KeyFromValue,typename Compare>
27struct value_comparison:std::binary_function<Value,Value,bool>
28{
29 value_comparison(
30 const KeyFromValue& key_=KeyFromValue(),const Compare& comp_=Compare()):
31 key(key_),comp(comp_)
32 {
33 }
34
35 bool operator()(
36 typename call_traits<Value>::param_type x,
37 typename call_traits<Value>::param_type y)const
38 {
39 return comp(key(x),key(y));
40 }
41
42private:
43 KeyFromValue key;
44 Compare comp;
45};

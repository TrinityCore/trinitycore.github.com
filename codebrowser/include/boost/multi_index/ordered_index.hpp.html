306
307 size_type erase(key_param_type x)
308 {
309 BOOST_MULTI_INDEX_ORD_INDEX_CHECK_INVARIANT;
310 std::pair<iterator,iterator> p=equal_range(x);
311 size_type s=0;
312 while(p.first!=p.second){
313 p.first=erase(p.first);
314 ++s;
315 }
316 return s;
317 }

319 iterator erase(iterator first,iterator last)
320 {
321 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(first);
322 BOOST_MULTI_INDEX_CHECK_VALID_ITERATOR(last);
323 BOOST_MULTI_INDEX_CHECK_IS_OWNER(first,*this);
324 BOOST_MULTI_INDEX_CHECK_IS_OWNER(last,*this);
325 BOOST_MULTI_INDEX_CHECK_VALID_RANGE(first,last);
326 BOOST_MULTI_INDEX_ORD_INDEX_CHECK_INVARIANT;
327 while(first!=last){
328 first=erase(first);
329 }
330 return first;
331 }

573 void copy_(
574 const ordered_index<KeyFromValue,Compare,SuperMeta,TagList,Category>& x,
575 const copy_map_type& map)
576 {
577 if(!x.root()){
578 empty_initialize();
579 }
580 else{
581 header()->color()=x.header()->color();
582
583 node_type* root_cpy=map.find(static_cast<final_node_type*>(x.root()));
584 header()->parent()=root_cpy->impl();
585
586 node_type* leftmost_cpy=map.find(
587 static_cast<final_node_type*>(x.leftmost()));
588 header()->left()=leftmost_cpy->impl();
589
590 node_type* rightmost_cpy=map.find(
591 static_cast<final_node_type*>(x.rightmost()));
592 header()->right()=rightmost_cpy->impl();
593
594 typedef typename copy_map_type::const_iterator copy_map_iterator;
595 for(copy_map_iterator it=map.begin(),it_end=map.end();it!=it_end;++it){
596 node_type* org=it->first;
597 node_type* cpy=it->second;
598
599 cpy->color()=org->color();
600
601 node_impl_pointer parent_org=org->parent();
602 if(parent_org==node_impl_pointer(0))cpy->parent()=node_impl_pointer(0);
603 else{
604 node_type* parent_cpy=map.find(
605 static_cast<final_node_type*>(node_type::from_impl(parent_org)));
606 cpy->parent()=parent_cpy->impl();
607 if(parent_org->left()==org->impl()){
608 parent_cpy->left()=cpy->impl();
609 }
610 else if(parent_org->right()==org->impl()){
611 /* header() does not satisfy this nor the previous check */
612 parent_cpy->right()=cpy->impl();
613 }
614 }
615
616 if(org->left()==node_impl_pointer(0))
617 cpy->left()=node_impl_pointer(0);
618 if(org->right()==node_impl_pointer(0))
619 cpy->right()=node_impl_pointer(0);
620 }
621 }
622
623 super::copy_(x,map);
624 }

692 bool replace_(value_param_type v,node_type* x)
693 {
694 if(in_place(v,x,Category())){
695 return super::replace_(v,x);
696 }
697
698 node_type* next=x;
699 node_type::increment(next);
700
701 node_impl_type::rebalance_for_erase(
702 x->impl(),header()->parent(),header()->left(),header()->right());
703
704 BOOST_TRY{
705 link_info inf;

707 node_impl_type::link(x->impl(),inf.side,inf.pos,header()->impl());

710 node_impl_type::restore(x->impl(),next->impl(),header()->impl());
711 return false;

720 bool modify_(node_type* x)
721 {
722 bool b;
723 BOOST_TRY{
724 b=in_place(x->value(),x,Category());
725 }
726 BOOST_CATCH(...){
727 erase_(x);
728 BOOST_RETHROW;
729 }
730 BOOST_CATCH_END
731 if(!b){
732 node_impl_type::rebalance_for_erase(
733 x->impl(),header()->parent(),header()->left(),header()->right());
734 BOOST_TRY{
735 link_info inf;
736 if(!link_point(key(x->value()),inf,Category())){
737 super::erase_(x);
738
739#if defined(BOOST_MULTI_INDEX_ENABLE_SAFE_MODE)
740 detach_iterators(x);
741#endif
742 return false;
743 }
744 node_impl_type::link(x->impl(),inf.side,inf.pos,header()->impl());
745 }
746 BOOST_CATCH(...){
747 super::erase_(x);
748
749#if defined(BOOST_MULTI_INDEX_ENABLE_SAFE_MODE)
750 detach_iterators(x);
751#endif
752
753 BOOST_RETHROW;
754 }
755 BOOST_CATCH_END
756 }
757
758 BOOST_TRY{
759 if(!super::modify_(x)){
760 node_impl_type::rebalance_for_erase(
761 x->impl(),header()->parent(),header()->left(),header()->right());
762
763#if defined(BOOST_MULTI_INDEX_ENABLE_SAFE_MODE)
764 detach_iterators(x);
765#endif
766
767 return false;
768 }
769 else return true;
770 }
771 BOOST_CATCH(...){
772 node_impl_type::rebalance_for_erase(
773 x->impl(),header()->parent(),header()->left(),header()->right());
774
775#if defined(BOOST_MULTI_INDEX_ENABLE_SAFE_MODE)
776 detach_iterators(x);
777#endif

779 BOOST_RETHROW;
780 }
781 BOOST_CATCH_END

833 bool invariant_()const
834 {
835 if(size()==0||begin()==end()){
836 if(size()!=0||begin()!=end()||
837 header()->left()!=header()->impl()||
838 header()->right()!=header()->impl())return false;
839 }
840 else{
841 if((size_type)std::distance(begin(),end())!=size())return false;
842
843 std::size_t len=node_impl_type::black_count(
844 leftmost()->impl(),root()->impl());
845 for(const_iterator it=begin(),it_end=end();it!=it_end;++it){
846 node_type* x=it.get_node();
847 node_type* left_x=node_type::from_impl(x->left());
848 node_type* right_x=node_type::from_impl(x->right());
849
850 if(x->color()==red){
851 if((left_x&&left_x->color()==red)||
852 (right_x&&right_x->color()==red))return false;
853 }
854 if(left_x&&comp_(key(x->value()),key(left_x->value())))return false;
855 if(right_x&&comp_(key(right_x->value()),key(x->value())))return false;
856 if(!left_x&&!right_x&&
857 node_impl_type::black_count(x->impl(),root()->impl())!=len)
858 return false;
859 }
860
861 if(leftmost()->impl()!=node_impl_type::minimum(root()->impl()))
862 return false;
863 if(rightmost()->impl()!=node_impl_type::maximum(root()->impl()))
864 return false;
865 }
866
867 return super::invariant_();
868 }

903 bool link_point(key_param_type k,link_info& inf,ordered_unique_tag)
904 {
905 node_type* y=header();
906 node_type* x=root();
907 bool c=true;
908 while(x){
909 y=x;
910 c=comp_(k,key(x->value()));
911 x=node_type::from_impl(c?x->left():x->right());
912 }
913 node_type* yy=y;
914 if(c){
915 if(yy==leftmost()){
916 inf.side=to_left;
917 inf.pos=y->impl();
918 return true;
919 }
920 else node_type::decrement(yy);
921 }
922
923 if(comp_(key(yy->value()),k)){
924 inf.side=c?to_left:to_right;
925 inf.pos=y->impl();
926 return true;
927 }
928 else{
929 inf.pos=yy->impl();
930 return false;
931 }
932 }

25 template<class Value>
26 inline iterator_range<counting_iterator<Value> >
27 counting_range(Value first, Value last)
28 {
29 typedef counting_iterator<Value> counting_iterator_t;
30 typedef iterator_range<counting_iterator_t> result_t;
31 return result_t(counting_iterator_t(first),
32 counting_iterator_t(last));
33 }
34
35 template<class Range>
36 inline iterator_range<counting_iterator<BOOST_DEDUCED_TYPENAME range_value<const Range>::type> >
37 counting_range(const Range& rng)
38 {
39 typedef counting_iterator<BOOST_DEDUCED_TYPENAME range_value<const Range>::type> counting_iterator_t;
40 typedef iterator_range<counting_iterator_t> result_t;
41 return boost::empty(rng)
42 ? result_t()
43 : result_t(
44 counting_iterator_t(*boost::begin(rng)),
45 counting_iterator_t(*boost::prior(boost::end(rng))));
46 }

58 counting_iterator_t(*boost::prior(boost::end(rng))));
59 }

65 /** Base class for most exceptions in the library.
66 *
67 * Substitutes the values for the parameter name
68 * placeholders in the template to create the human
69 * readable error message
70 *
71 * Placeholders are surrounded by % signs: %example%
72 * Poor man's version of boost::format
73 *
74 * If a parameter name is absent, perform default substitutions
75 * instead so ugly placeholders are never left in-place.
76 *
77 * Options are displayed in "canonical" form
78 * This is the most unambiguous form of the
79 * *parsed* option name and would correspond to
80 * option_description::format_name()
81 * i.e. what is shown by print_usage()
82 *
83 * The "canonical" form depends on whether the option is
84 * specified in short or long form, using dashes or slashes
85 * or without a prefix (from a configuration file)
86 *
87 * */
88 class BOOST_PROGRAM_OPTIONS_DECL error_with_option_name : public error {
89
90 protected:
91 /** can be
92 * 0 = no prefix (config file options)
93 * allow_long
94 * allow_dash_for_short
95 * allow_slash_for_short
96 * allow_long_disguise */
97 int m_option_style;
98
99
100 /** substitutions
101 * from placeholders to values */
102 std::map<std::string, std::string> m_substitutions;
103 typedef std::pair<std::string, std::string> string_pair;
104 std::map<std::string, string_pair > m_substitution_defaults;
105
106 public:
107 /** template with placeholders */
108 std::string m_error_template;
109
110 error_with_option_name(const std::string& template_,
111 const std::string& option_name = "",
112 const std::string& original_token = "",
113 int option_style = 0);
114
115 /** gcc says that throw specification on dtor is loosened
116 * without this line
117 * */
118 ~error_with_option_name() throw() {}

138 /** Substitute
139 * parameter_name->value to create the error message from
140 * the error template */
141 void set_substitute(const std::string& parameter_name, const std::string& value)
142 { m_substitutions[parameter_name] = value; }
143
144 /** If the parameter is missing, then make the
145 * from->to substitution instead */
146 void set_substitute_default(const std::string& parameter_name,
147 const std::string& from,
148 const std::string& to)
149 {
150 m_substitution_defaults[parameter_name] = std::make_pair(from, to);
151 }
152
153
154 /** Add context to an exception */
155 void add_context(const std::string& option_name,
156 const std::string& original_token,
157 int option_style)
158 {
159 set_option_name(option_name);
160 set_original_token(original_token);
161 set_prefix(option_style);
162 }
163
164 void set_prefix(int option_style)
165 { m_option_style = option_style;}
166
167 /** Overridden in error_with_no_option_name */
168 virtual void set_option_name(const std::string& option_name)
169 { set_substitute("option", option_name);}
170
171 std::string get_option_name() const throw()
172 { return get_canonical_option_name(); }
173
174 void set_original_token(const std::string& original_token)
175 { set_substitute("original_token", original_token);}

178 /** Creates the error_message on the fly
179 * Currently a thin wrapper for substitute_placeholders() */
180 virtual const char* what() const throw();
181
182 protected:
183 /** Used to hold the error text returned by what() */
184 mutable std::string m_message; // For on-demand formatting in 'what'
185
186 /** Makes all substitutions using the template */
187 virtual void substitute_placeholders(const std::string& error_template) const;
188
189 // helper function for substitute_placeholders
190 void replace_token(const std::string& from, const std::string& to) const;
191
192 /** Construct option name in accordance with the appropriate
193 * prefix style: i.e. long dash or short slash etc */
194 std::string get_canonical_option_name() const;
195 std::string get_canonical_option_prefix() const;
196 };

233 /** Base class of unparsable options,
234 * when the desired option cannot be identified.
235 *
236 *
237 * It makes no sense to have an option name, when we can't match an option to the
238 * parameter
239 *
240 * Having this a part of the error_with_option_name hierachy makes error handling
241 * a lot easier, even if the name indicates some sort of conceptual dissonance!
242 *
243 * */
244 class BOOST_PROGRAM_OPTIONS_DECL error_with_no_option_name : public error_with_option_name {
245 public:
246 error_with_no_option_name(const std::string& template_,
247 const std::string& original_token = "")
248 : error_with_option_name(template_, "", original_token)
249 {
250 }
251
252 /** Does NOT set option name, because no option name makes sense */
253 virtual void set_option_name(const std::string&) {}
254
255 ~error_with_no_option_name() throw() {}
256 };

293 /** Class thrown when there's syntax error either for command
294 * line or config file options. See derived children for
295 * concrete classes. */
296 class BOOST_PROGRAM_OPTIONS_DECL invalid_syntax : public error_with_option_name {
297 public:
298 enum kind_t {
299 long_not_allowed = 30,
300 long_adjacent_not_allowed,
301 short_adjacent_not_allowed,
302 empty_adjacent_parameter,
303 missing_parameter,
304 extra_parameter,
305 unrecognized_line
306 };
307
308 invalid_syntax(kind_t kind,
309 const std::string& option_name = "",
310 const std::string& original_token = "",
311 int option_style = 0):
312 error_with_option_name(get_template(kind), option_name, original_token, option_style),
313 m_kind(kind)
314 {
315 }
316
317 ~invalid_syntax() throw() {}
318
319 kind_t kind() const {return m_kind;}
320
321 /** Convenience functions for backwards compatibility */
322 virtual std::string tokens() const {return get_option_name(); }
323 protected:
324 /** Used to convert kind_t to a related error text */
325 std::string get_template(kind_t kind);
326 kind_t m_kind;
327 };

329 class BOOST_PROGRAM_OPTIONS_DECL invalid_config_file_syntax : public invalid_syntax {

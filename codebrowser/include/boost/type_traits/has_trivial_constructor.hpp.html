12#include <boost/type_traits/config.hpp>

15#include <boost/type_traits/detail/ice_or.hpp>

17// should be the last #include
18#include <boost/type_traits/detail/bool_trait_def.hpp>
19
20namespace boost {
21
22namespace detail {
23
24template <typename T>
25struct has_trivial_ctor_impl
26{
27#ifdef BOOST_HAS_TRIVIAL_CONSTRUCTOR
28 BOOST_STATIC_CONSTANT(bool, value =
29 (::boost::type_traits::ice_or<
30 ::boost::is_pod<T>::value,
31 BOOST_HAS_TRIVIAL_CONSTRUCTOR(T)
32 >::value));
33#else
34 BOOST_STATIC_CONSTANT(bool, value =
35 (::boost::type_traits::ice_or<
36 ::boost::is_pod<T>::value,
37 false
38 >::value));
39#endif
40};
41
42} // namespace detail
43
44BOOST_TT_AUX_BOOL_TRAIT_DEF1(has_trivial_constructor,T,::boost::detail::has_trivial_ctor_impl<T>::value)
45BOOST_TT_AUX_BOOL_TRAIT_DEF1(has_trivial_default_constructor,T,::boost::detail::has_trivial_ctor_impl<T>::value)
46
47} // namespace boost

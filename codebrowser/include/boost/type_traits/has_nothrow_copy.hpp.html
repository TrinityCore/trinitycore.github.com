19namespace detail{

21template <class T>
22struct has_nothrow_copy_imp{
23#ifdef BOOST_HAS_NOTHROW_COPY
24 BOOST_STATIC_CONSTANT(bool, value = BOOST_HAS_NOTHROW_COPY(T));
25#else
26 BOOST_STATIC_CONSTANT(bool, value = ::boost::has_trivial_copy<T>::value);
27#endif
28};
29
30}

32BOOST_TT_AUX_BOOL_TRAIT_DEF1(has_nothrow_copy,T,::boost::detail::has_nothrow_copy_imp<T>::value)
33BOOST_TT_AUX_BOOL_TRAIT_DEF1(has_nothrow_copy_constructor,T,::boost::detail::has_nothrow_copy_imp<T>::value)

35BOOST_TT_AUX_BOOL_TRAIT_SPEC1(has_nothrow_copy,void,false)

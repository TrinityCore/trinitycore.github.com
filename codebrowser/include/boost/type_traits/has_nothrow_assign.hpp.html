21template <class T>
22struct has_nothrow_assign_imp{
23#ifndef BOOST_HAS_NOTHROW_ASSIGN
24 BOOST_STATIC_CONSTANT(bool, value = ::boost::has_trivial_assign<T>::value);
25#else
26 BOOST_STATIC_CONSTANT(bool, value = BOOST_HAS_NOTHROW_ASSIGN(T));
27#endif
28};

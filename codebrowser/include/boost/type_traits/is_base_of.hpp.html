21namespace boost {
22
23 namespace detail{
24 template <class B, class D>
25 struct is_base_of_imp
26 {
27 typedef typename remove_cv<B>::type ncvB;
28 typedef typename remove_cv<D>::type ncvD;
29 BOOST_STATIC_CONSTANT(bool, value = (::boost::type_traits::ice_or<
30 (::boost::detail::is_base_and_derived_impl<ncvB,ncvD>::value),
31 (::boost::type_traits::ice_and< ::boost::is_same<ncvB,ncvD>::value, ::boost::is_class<ncvB>::value>::value)>::value));
32 };
33 }
34
35BOOST_TT_AUX_BOOL_TRAIT_DEF2(
36 is_base_of
37 , Base
38 , Derived
39 , (::boost::detail::is_base_of_imp<Base, Derived>::value))

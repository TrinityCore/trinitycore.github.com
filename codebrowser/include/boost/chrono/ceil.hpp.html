21 /**
22 * rounds up
23 */
24 template <class To, class Rep, class Period>
25 To ceil(const duration<Rep, Period>& d)
26 {
27 To t = duration_cast<To>(d);
28 if (t < d)
29 ++t;
30 return t;
31 }

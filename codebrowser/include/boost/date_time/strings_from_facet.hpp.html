19//! This function gathers up all the month strings from a std::locale
20/*! Using the time_put facet, this function creates a collection of
21 * all the month strings from a locale. This is handy when building
22 * custom date parsers or formatters that need to be localized.
23 *
24 *@param charT The type of char to use when gathering typically char
25 * or wchar_t.
26 *@param locale The locale to use when gathering the strings
27 *@param short_strings True(default) to gather short strings,
28 * false for long strings.
29 *@return A vector of strings containing the strings in order. eg:
30 * Jan, Feb, Mar, etc.
31 */
32template<typename charT>
33std::vector<std::basic_string<charT> >
34gather_month_strings(const std::locale& locale, bool short_strings=true)
35{
36 typedef std::basic_string<charT> string_type;
37 typedef std::vector<string_type> collection_type;
38 typedef std::basic_ostringstream<charT> ostream_type;
39 typedef std::ostreambuf_iterator<charT> ostream_iter_type;
40 typedef std::basic_ostringstream<charT> stringstream_type;
41 typedef std::time_put<charT> time_put_facet_type;
42 charT short_fmt[3] = { '%', 'b' };
43 charT long_fmt[3] = { '%', 'B' };
44 collection_type months;
45 string_type outfmt(short_fmt);
46 if (!short_strings) {
47 outfmt = long_fmt;
48 }
49 {
50 //grab the needed strings by using the locale to
51 //output each month
52 const charT* p_outfmt = outfmt.c_str(), *p_outfmt_end = p_outfmt + outfmt.size();
53 tm tm_value;
54 memset(&tm_value, 0, sizeof(tm_value));
55 for (int m=0; m < 12; m++) {
56 tm_value.tm_mon = m;
57 stringstream_type ss;
58 ostream_iter_type oitr(ss);
59 std::use_facet<time_put_facet_type>(locale).put(oitr, ss, ss.fill(),
60 &tm_value,
61 p_outfmt,
62 p_outfmt_end);
63 months.push_back(ss.str());
64 }
65 }
66 return months;
67}

69//! This function gathers up all the weekday strings from a std::locale
70/*! Using the time_put facet, this function creates a collection of
71 * all the weekday strings from a locale starting with the string for
72 * 'Sunday'. This is handy when building custom date parsers or
73 * formatters that need to be localized.
74 *
75 *@param charT The type of char to use when gathering typically char
76 * or wchar_t.
77 *@param locale The locale to use when gathering the strings
78 *@param short_strings True(default) to gather short strings,
79 * false for long strings.
80 *@return A vector of strings containing the weekdays in order. eg:
81 * Sun, Mon, Tue, Wed, Thu, Fri, Sat
82 */
83template<typename charT>
84std::vector<std::basic_string<charT> >
85gather_weekday_strings(const std::locale& locale, bool short_strings=true)
86{
87 typedef std::basic_string<charT> string_type;
88 typedef std::vector<string_type> collection_type;
89 typedef std::basic_ostringstream<charT> ostream_type;
90 typedef std::ostreambuf_iterator<charT> ostream_iter_type;
91 typedef std::basic_ostringstream<charT> stringstream_type;
92 typedef std::time_put<charT> time_put_facet_type;
93 charT short_fmt[3] = { '%', 'a' };
94 charT long_fmt[3] = { '%', 'A' };
95
96 collection_type weekdays;
97
98
99 string_type outfmt(short_fmt);
100 if (!short_strings) {
101 outfmt = long_fmt;
102 }
103 {
104 //grab the needed strings by using the locale to
105 //output each month / weekday
106 const charT* p_outfmt = outfmt.c_str(), *p_outfmt_end = p_outfmt + outfmt.size();
107 tm tm_value;
108 memset(&tm_value, 0, sizeof(tm_value));
109 for (int i=0; i < 7; i++) {
110 tm_value.tm_wday = i;
111 stringstream_type ss;
112 ostream_iter_type oitr(ss);
113 std::use_facet<time_put_facet_type>(locale).put(oitr, ss, ss.fill(),
114 &tm_value,
115 p_outfmt,
116 p_outfmt_end);
117
118 weekdays.push_back(ss.str());
119 }
120 }
121 return weekdays;
122}

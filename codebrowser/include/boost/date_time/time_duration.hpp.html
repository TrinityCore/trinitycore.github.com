23 //! Represents some amount of elapsed time measure to a given resolution
24 /*! This class represents a standard set of capabilities for all
25 counted time durations. Time duration implementations should derive
26 from this class passing their type as the first template parameter.
27 This design allows the subclass duration types to provide custom
28 construction policies or other custom features not provided here.
29
30 @param T The subclass type
31 @param rep_type The time resolution traits for this duration type.
32 */
33 template<class T, typename rep_type>
34 class time_duration : private
35 boost::less_than_comparable<T
36 , boost::equality_comparable<T
37 > >
38 /* dividable, addable, and subtractable operator templates
39 * won't work with this class (MSVC++ 6.0). return type
40 * from '+=' is different than expected return type
41 * from '+'. multipliable probably wont work
42 * either (haven't tried) */
43 {
44 public:
45 typedef T duration_type; //the subclass
46 typedef rep_type traits_type;
47 typedef typename rep_type::day_type day_type;
48 typedef typename rep_type::hour_type hour_type;
49 typedef typename rep_type::min_type min_type;
50 typedef typename rep_type::sec_type sec_type;
51 typedef typename rep_type::fractional_seconds_type fractional_seconds_type;
52 typedef typename rep_type::tick_type tick_type;
53 typedef typename rep_type::impl_type impl_type;
54
55 time_duration() : ticks_(0) {}
56 time_duration(hour_type hours_in,
57 min_type minutes_in,
58 sec_type seconds_in=0,
59 fractional_seconds_type frac_sec_in = 0) :
60 ticks_(rep_type::to_tick_count(hours_in,minutes_in,seconds_in,frac_sec_in))
61 {}
62 // copy constructor required for dividable<>
63 //! Construct from another time_duration (Copy constructor)
64 time_duration(const time_duration<T, rep_type>& other)
65 : ticks_(other.ticks_)
66 {}
67 //! Construct from special_values
68 time_duration(special_values sv) : ticks_(impl_type::from_special(sv))
69 {}
70 //! Returns smallest representable duration
71 static duration_type unit()
72 {
73 return duration_type(0,0,0,1);
74 }
75 //! Return the number of ticks in a second
76 static tick_type ticks_per_second()
77 {
78 return rep_type::res_adjust();
79 }
80 //! Provide the resolution of this duration type
81 static time_resolutions resolution()
82 {
83 return rep_type::resolution();
84 }
85 //! Returns number of hours in the duration
86 hour_type hours() const
87 {
88 return static_cast<hour_type>(ticks() / (3600*ticks_per_second()));
89 }
90 //! Returns normalized number of minutes
91 min_type minutes() const
92 {
93 return static_cast<min_type>((ticks() / (60*ticks_per_second())) % 60);
94 }
95 //! Returns normalized number of seconds (0..60)
96 sec_type seconds() const
97 {
98 return static_cast<sec_type>((ticks()/ticks_per_second()) % 60);
99 }
100 //! Returns total number of seconds truncating any fractional seconds
101 sec_type total_seconds() const
102 {
103 return static_cast<sec_type>(ticks() / ticks_per_second());
104 }
105 //! Returns total number of milliseconds truncating any fractional seconds
106 tick_type total_milliseconds() const
107 {
108 if (ticks_per_second() < 1000) {
109 return ticks() * (static_cast<tick_type>(1000) / ticks_per_second());
110 }
111 return ticks() / (ticks_per_second() / static_cast<tick_type>(1000)) ;
112 }
113 //! Returns total number of nanoseconds truncating any sub millisecond values
114 tick_type total_nanoseconds() const
115 {
116 if (ticks_per_second() < 1000000000) {
117 return ticks() * (static_cast<tick_type>(1000000000) / ticks_per_second());
118 }
119 return ticks() / (ticks_per_second() / static_cast<tick_type>(1000000000)) ;
120 }
121 //! Returns total number of microseconds truncating any sub microsecond values
122 tick_type total_microseconds() const
123 {
124 if (ticks_per_second() < 1000000) {
125 return ticks() * (static_cast<tick_type>(1000000) / ticks_per_second());
126 }
127 return ticks() / (ticks_per_second() / static_cast<tick_type>(1000000)) ;
128 }
129 //! Returns count of fractional seconds at given resolution
130 fractional_seconds_type fractional_seconds() const
131 {
132 return (ticks() % ticks_per_second());
133 }
134 //! Returns number of possible digits in fractional seconds
135 static unsigned short num_fractional_digits()
136 {
137 return rep_type::num_fractional_digits();
138 }
139 duration_type invert_sign() const
140 {
141 return duration_type(ticks_ * (-1));
142 }
143 bool is_negative() const
144 {
145 return ticks_ < 0;
146 }
147 bool operator<(const time_duration& rhs) const
148 {
149 return ticks_ < rhs.ticks_;
150 }
151 bool operator==(const time_duration& rhs) const
152 {
153 return ticks_ == rhs.ticks_;
154 }
155 //! unary- Allows for time_duration td = -td1
156 duration_type operator-()const
157 {
158 return duration_type(ticks_ * (-1));
159 }
160 duration_type operator-(const duration_type& d) const
161 {
162 return duration_type(ticks_ - d.ticks_);
163 }
164 duration_type operator+(const duration_type& d) const
165 {
166 return duration_type(ticks_ + d.ticks_);
167 }
168 duration_type operator/(int divisor) const
169 {
170 return duration_type(ticks_ / divisor);
171 }
172 duration_type operator-=(const duration_type& d)
173 {
174 ticks_ = ticks_ - d.ticks_;
175 return duration_type(ticks_);
176 }
177 duration_type operator+=(const duration_type& d)
178 {
179 ticks_ = ticks_ + d.ticks_;
180 return duration_type(ticks_);
181 }
182 //! Division operations on a duration with an integer.
183 duration_type operator/=(int divisor)
184 {
185 ticks_ = ticks_ / divisor;
186 return duration_type(ticks_);
187 }
188 //! Multiplication operations an a duration with an integer
189 duration_type operator*(int rhs) const
190 {
191 return duration_type(ticks_ * rhs);
192 }
193 duration_type operator*=(int divisor)
194 {
195 ticks_ = ticks_ * divisor;
196 return duration_type(ticks_);
197 }
198 tick_type ticks() const
199 {
200 return traits_type::as_number(ticks_);
201 }
202
203 //! Is ticks_ a special value?
204 bool is_special()const
205 {
206 if(traits_type::is_adapted())
207 {
208 return ticks_.is_special();
209 }
210 else{
211 return false;
212 }
213 }
214 //! Is duration pos-infinity
215 bool is_pos_infinity()const
216 {
217 if(traits_type::is_adapted())
218 {
219 return ticks_.is_pos_infinity();
220 }
221 else{
222 return false;
223 }
224 }
225 //! Is duration neg-infinity
226 bool is_neg_infinity()const
227 {
228 if(traits_type::is_adapted())
229 {
230 return ticks_.is_neg_infinity();
231 }
232 else{
233 return false;
234 }
235 }
236 //! Is duration not-a-date-time
237 bool is_not_a_date_time()const
238 {
239 if(traits_type::is_adapted())
240 {
241 return ticks_.is_nan();
242 }
243 else{
244 return false;
245 }
246 }

248 //! Used for special_values output
249 impl_type get_rep()const
250 {
251 return ticks_;
252 }

257 };

261 //! Template for instantiating derived adjusting durations
262 /* These templates are designed to work with multiples of
263 * 10 for frac_of_second and resoultion adjustment
264 */
265 template<class base_duration, boost::int64_t frac_of_second>
266 class subsecond_duration : public base_duration
267 {
268 public:
269 typedef typename base_duration::impl_type impl_type;
270 typedef typename base_duration::traits_type traits_type;
271
272 private:
273 // To avoid integer overflow we precompute the duration resolution conversion coefficient (ticket #3471)
274 BOOST_STATIC_ASSERT_MSG((traits_type::ticks_per_second >= frac_of_second ? traits_type::ticks_per_second % frac_of_second : frac_of_second % traits_type::ticks_per_second) == 0,\
275 "The base duration resolution must be a multiple of the subsecond duration resolution");
276 BOOST_STATIC_CONSTANT(boost::int64_t, adjustment_ratio = (traits_type::ticks_per_second >= frac_of_second ? traits_type::ticks_per_second / frac_of_second : frac_of_second / traits_type::ticks_per_second));
277
278 public:
279 explicit subsecond_duration(boost::int64_t ss) :
280 base_duration(impl_type(traits_type::ticks_per_second >= frac_of_second ? ss * adjustment_ratio : ss / adjustment_ratio))
281 {
282 }
283 };

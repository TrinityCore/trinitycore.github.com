33namespace boost {

36/// Default service implementation for a sequenced packet socket.
37template <typename Protocol>
38class seq_packet_socket_service
39#if defined(GENERATING_DOCUMENTATION)
40 : public boost::asio::io_service::service
41#else
42 : public boost::asio::detail::service_base<
43 seq_packet_socket_service<Protocol> >
44#endif
45{
46public:
47#if defined(GENERATING_DOCUMENTATION)
48 /// The unique service identifier.
49 static boost::asio::io_service::id id;
50#endif
51
52 /// The protocol type.
53 typedef Protocol protocol_type;
54
55 /// The endpoint type.
56 typedef typename Protocol::endpoint endpoint_type;
57
58private:
59 // The type of the platform-specific implementation.
60#if defined(BOOST_ASIO_HAS_IOCP)
61 typedef detail::win_iocp_socket_service<Protocol> service_impl_type;
62#else
63 typedef detail::reactive_socket_service<Protocol> service_impl_type;
64#endif
65
66public:
67 /// The type of a sequenced packet socket implementation.
68#if defined(GENERATING_DOCUMENTATION)
69 typedef implementation_defined implementation_type;
70#else
71 typedef typename service_impl_type::implementation_type implementation_type;
72#endif
73
74 /// (Deprecated: Use native_handle_type.) The native socket type.
75#if defined(GENERATING_DOCUMENTATION)
76 typedef implementation_defined native_type;
77#else
78 typedef typename service_impl_type::native_handle_type native_type;
79#endif
80
81 /// The native socket type.
82#if defined(GENERATING_DOCUMENTATION)
83 typedef implementation_defined native_handle_type;
84#else
85 typedef typename service_impl_type::native_handle_type native_handle_type;
86#endif
87
88 /// Construct a new sequenced packet socket service for the specified
89 /// io_service.
90 explicit seq_packet_socket_service(boost::asio::io_service& io_service)
91 : boost::asio::detail::service_base<
92 seq_packet_socket_service<Protocol> >(io_service),
93 service_impl_(io_service)
94 {
95 }
96
97 /// Construct a new sequenced packet socket implementation.
98 void construct(implementation_type& impl)
99 {
100 service_impl_.construct(impl);
101 }
102
103#if defined(BOOST_ASIO_HAS_MOVE) || defined(GENERATING_DOCUMENTATION)
104 /// Move-construct a new sequenced packet socket implementation.
105 void move_construct(implementation_type& impl,
106 implementation_type& other_impl)
107 {
108 service_impl_.move_construct(impl, other_impl);
109 }
110
111 /// Move-assign from another sequenced packet socket implementation.
112 void move_assign(implementation_type& impl,
113 seq_packet_socket_service& other_service,
114 implementation_type& other_impl)
115 {
116 service_impl_.move_assign(impl, other_service.service_impl_, other_impl);
117 }
118
119 /// Move-construct a new sequenced packet socket implementation from another
120 /// protocol type.
121 template <typename Protocol1>
122 void converting_move_construct(implementation_type& impl,
123 typename seq_packet_socket_service<
124 Protocol1>::implementation_type& other_impl,
125 typename enable_if<is_convertible<
126 Protocol1, Protocol>::value>::type* = 0)
127 {
128 service_impl_.template converting_move_construct<Protocol1>(
129 impl, other_impl);
130 }
131#endif // defined(BOOST_ASIO_HAS_MOVE) || defined(GENERATING_DOCUMENTATION)
132
133 /// Destroy a sequenced packet socket implementation.
134 void destroy(implementation_type& impl)
135 {
136 service_impl_.destroy(impl);
137 }
138
139 /// Open a sequenced packet socket.
140 boost::system::error_code open(implementation_type& impl,
141 const protocol_type& protocol, boost::system::error_code& ec)
142 {
143 if (protocol.type() == SOCK_SEQPACKET)
144 service_impl_.open(impl, protocol, ec);
145 else
146 ec = boost::asio::error::invalid_argument;
147 return ec;
148 }
149
150 /// Assign an existing native socket to a sequenced packet socket.
151 boost::system::error_code assign(implementation_type& impl,
152 const protocol_type& protocol, const native_handle_type& native_socket,
153 boost::system::error_code& ec)
154 {
155 return service_impl_.assign(impl, protocol, native_socket, ec);
156 }
157
158 /// Determine whether the socket is open.
159 bool is_open(const implementation_type& impl) const
160 {
161 return service_impl_.is_open(impl);
162 }
163
164 /// Close a sequenced packet socket implementation.
165 boost::system::error_code close(implementation_type& impl,
166 boost::system::error_code& ec)
167 {
168 return service_impl_.close(impl, ec);
169 }
170
171 /// (Deprecated: Use native_handle().) Get the native socket implementation.
172 native_type native(implementation_type& impl)
173 {
174 return service_impl_.native_handle(impl);
175 }
176
177 /// Get the native socket implementation.
178 native_handle_type native_handle(implementation_type& impl)
179 {
180 return service_impl_.native_handle(impl);
181 }
182
183 /// Cancel all asynchronous operations associated with the socket.
184 boost::system::error_code cancel(implementation_type& impl,
185 boost::system::error_code& ec)
186 {
187 return service_impl_.cancel(impl, ec);
188 }
189
190 /// Determine whether the socket is at the out-of-band data mark.
191 bool at_mark(const implementation_type& impl,
192 boost::system::error_code& ec) const
193 {
194 return service_impl_.at_mark(impl, ec);
195 }
196
197 /// Determine the number of bytes available for reading.
198 std::size_t available(const implementation_type& impl,
199 boost::system::error_code& ec) const
200 {
201 return service_impl_.available(impl, ec);
202 }
203
204 /// Bind the sequenced packet socket to the specified local endpoint.
205 boost::system::error_code bind(implementation_type& impl,
206 const endpoint_type& endpoint, boost::system::error_code& ec)
207 {
208 return service_impl_.bind(impl, endpoint, ec);
209 }
210
211 /// Connect the sequenced packet socket to the specified endpoint.
212 boost::system::error_code connect(implementation_type& impl,
213 const endpoint_type& peer_endpoint, boost::system::error_code& ec)
214 {
215 return service_impl_.connect(impl, peer_endpoint, ec);
216 }
217
218 /// Start an asynchronous connect.
219 template <typename ConnectHandler>
220 BOOST_ASIO_INITFN_RESULT_TYPE(ConnectHandler,
221 void (boost::system::error_code))
222 async_connect(implementation_type& impl,
223 const endpoint_type& peer_endpoint,
224 BOOST_ASIO_MOVE_ARG(ConnectHandler) handler)
225 {
226 detail::async_result_init<
227 ConnectHandler, void (boost::system::error_code)> init(
228 BOOST_ASIO_MOVE_CAST(ConnectHandler)(handler));
229
230 service_impl_.async_connect(impl, peer_endpoint, init.handler);
231
232 return init.result.get();
233 }
234
235 /// Set a socket option.
236 template <typename SettableSocketOption>
237 boost::system::error_code set_option(implementation_type& impl,
238 const SettableSocketOption& option, boost::system::error_code& ec)
239 {
240 return service_impl_.set_option(impl, option, ec);
241 }
242
243 /// Get a socket option.
244 template <typename GettableSocketOption>
245 boost::system::error_code get_option(const implementation_type& impl,
246 GettableSocketOption& option, boost::system::error_code& ec) const
247 {
248 return service_impl_.get_option(impl, option, ec);
249 }
250
251 /// Perform an IO control command on the socket.
252 template <typename IoControlCommand>
253 boost::system::error_code io_control(implementation_type& impl,
254 IoControlCommand& command, boost::system::error_code& ec)
255 {
256 return service_impl_.io_control(impl, command, ec);
257 }
258
259 /// Gets the non-blocking mode of the socket.
260 bool non_blocking(const implementation_type& impl) const
261 {
262 return service_impl_.non_blocking(impl);
263 }
264
265 /// Sets the non-blocking mode of the socket.
266 boost::system::error_code non_blocking(implementation_type& impl,
267 bool mode, boost::system::error_code& ec)
268 {
269 return service_impl_.non_blocking(impl, mode, ec);
270 }
271
272 /// Gets the non-blocking mode of the native socket implementation.
273 bool native_non_blocking(const implementation_type& impl) const
274 {
275 return service_impl_.native_non_blocking(impl);
276 }
277
278 /// Sets the non-blocking mode of the native socket implementation.
279 boost::system::error_code native_non_blocking(implementation_type& impl,
280 bool mode, boost::system::error_code& ec)
281 {
282 return service_impl_.native_non_blocking(impl, mode, ec);
283 }
284
285 /// Get the local endpoint.
286 endpoint_type local_endpoint(const implementation_type& impl,
287 boost::system::error_code& ec) const
288 {
289 return service_impl_.local_endpoint(impl, ec);
290 }
291
292 /// Get the remote endpoint.
293 endpoint_type remote_endpoint(const implementation_type& impl,
294 boost::system::error_code& ec) const
295 {
296 return service_impl_.remote_endpoint(impl, ec);
297 }
298
299 /// Disable sends or receives on the socket.
300 boost::system::error_code shutdown(implementation_type& impl,
301 socket_base::shutdown_type what, boost::system::error_code& ec)
302 {
303 return service_impl_.shutdown(impl, what, ec);
304 }
305
306 /// Send the given data to the peer.
307 template <typename ConstBufferSequence>
308 std::size_t send(implementation_type& impl,
309 const ConstBufferSequence& buffers,
310 socket_base::message_flags flags, boost::system::error_code& ec)
311 {
312 return service_impl_.send(impl, buffers, flags, ec);
313 }
314
315 /// Start an asynchronous send.
316 template <typename ConstBufferSequence, typename WriteHandler>
317 void async_send(implementation_type& impl,
318 const ConstBufferSequence& buffers,
319 socket_base::message_flags flags,
320 BOOST_ASIO_MOVE_ARG(WriteHandler) handler)
321 {
322 service_impl_.async_send(impl, buffers, flags,
323 BOOST_ASIO_MOVE_CAST(WriteHandler)(handler));
324 }
325
326 /// Receive some data from the peer.
327 template <typename MutableBufferSequence>
328 std::size_t receive(implementation_type& impl,
329 const MutableBufferSequence& buffers, socket_base::message_flags in_flags,
330 socket_base::message_flags& out_flags, boost::system::error_code& ec)
331 {
332 return service_impl_.receive_with_flags(impl,
333 buffers, in_flags, out_flags, ec);
334 }
335
336 /// Start an asynchronous receive.
337 template <typename MutableBufferSequence, typename ReadHandler>
338 void async_receive(implementation_type& impl,
339 const MutableBufferSequence& buffers, socket_base::message_flags in_flags,
340 socket_base::message_flags& out_flags,
341 BOOST_ASIO_MOVE_ARG(ReadHandler) handler)
342 {
343 service_impl_.async_receive_with_flags(impl, buffers, in_flags,
344 out_flags, BOOST_ASIO_MOVE_CAST(ReadHandler)(handler));
345 }
346
347private:
348 // Destroy all user-defined handler objects owned by the service.
349 void shutdown_service()
350 {
351 service_impl_.shutdown_service();
352 }
353
354 // The platform-specific implementation.
355 service_impl_type service_impl_;
356};

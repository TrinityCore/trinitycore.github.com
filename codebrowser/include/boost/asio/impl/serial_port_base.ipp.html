41boost::system::error_code serial_port_base::baud_rate::store(

44#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
45 storage.BaudRate = value_;
46#else
47 speed_t baud;
48 switch (value_)
49 {
50 // Do POSIX-specified rates first.
51 case 0: baud = B0; break;
52 case 50: baud = B50; break;
53 case 75: baud = B75; break;
54 case 110: baud = B110; break;
55 case 134: baud = B134; break;
56 case 150: baud = B150; break;
57 case 200: baud = B200; break;
58 case 300: baud = B300; break;
59 case 600: baud = B600; break;
60 case 1200: baud = B1200; break;
61 case 1800: baud = B1800; break;
62 case 2400: baud = B2400; break;
63 case 4800: baud = B4800; break;
64 case 9600: baud = B9600; break;
65 case 19200: baud = B19200; break;
66 case 38400: baud = B38400; break;
67 // And now the extended ones conditionally.
68# ifdef B7200
69 case 7200: baud = B7200; break;
70# endif
71# ifdef B14400
72 case 14400: baud = B14400; break;
73# endif
74# ifdef B57600
75 case 57600: baud = B57600; break;
76# endif
77# ifdef B115200
78 case 115200: baud = B115200; break;
79# endif
80# ifdef B230400
81 case 230400: baud = B230400; break;
82# endif
83# ifdef B460800
84 case 460800: baud = B460800; break;
85# endif
86# ifdef B500000
87 case 500000: baud = B500000; break;
88# endif
89# ifdef B576000
90 case 576000: baud = B576000; break;
91# endif
92# ifdef B921600
93 case 921600: baud = B921600; break;
94# endif
95# ifdef B1000000

105 case 3000000: baud = B3000000; break;

108 case 3500000: baud = B3500000; break;

112# endif
113 default:
114 baud = B0;
115 ec = boost::asio::error::invalid_argument;
116 return ec;
117 }
118# if defined(_BSD_SOURCE)
119 ::cfsetspeed(&storage, baud);

124#endif

222boost::system::error_code serial_port_base::flow_control::store(
223 BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec) const
224{
225#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
226 storage.fOutxCtsFlow = FALSE;
227 storage.fOutxDsrFlow = FALSE;
228 storage.fTXContinueOnXoff = TRUE;
229 storage.fDtrControl = DTR_CONTROL_ENABLE;
230 storage.fDsrSensitivity = FALSE;
231 storage.fOutX = FALSE;
232 storage.fInX = FALSE;
233 storage.fRtsControl = RTS_CONTROL_ENABLE;
234 switch (value_)
235 {
236 case none:
237 break;
238 case software:
239 storage.fOutX = TRUE;
240 storage.fInX = TRUE;
241 break;
242 case hardware:
243 storage.fOutxCtsFlow = TRUE;
244 storage.fRtsControl = RTS_CONTROL_HANDSHAKE;
245 break;
246 default:
247 break;
248 }
249#else
250 switch (value_)
251 {
252 case none:
253 storage.c_iflag &= ~(IXOFF | IXON);
254# if defined(_BSD_SOURCE)
255 storage.c_cflag &= ~CRTSCTS;
256# elif defined(__QNXNTO__)
257 storage.c_cflag &= ~(IHFLOW | OHFLOW);
258# endif
259 break;
260 case software:
261 storage.c_iflag |= IXOFF | IXON;
262# if defined(_BSD_SOURCE)
263 storage.c_cflag &= ~CRTSCTS;
264# elif defined(__QNXNTO__)
265 storage.c_cflag &= ~(IHFLOW | OHFLOW);
266# endif
267 break;
268 case hardware:
269# if defined(_BSD_SOURCE)
270 storage.c_iflag &= ~(IXOFF | IXON);
271 storage.c_cflag |= CRTSCTS;
272 break;
273# elif defined(__QNXNTO__)
274 storage.c_iflag &= ~(IXOFF | IXON);
275 storage.c_cflag |= (IHFLOW | OHFLOW);
276 break;
277# else
278 ec = boost::asio::error::operation_not_supported;
279 return ec;
280# endif
281 default:
282 break;
283 }
284#endif
285 ec = boost::system::error_code();
286 return ec;
287}

289boost::system::error_code serial_port_base::flow_control::load(
290 const BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec)
291{
292#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
293 if (storage.fOutX && storage.fInX)
294 {
295 value_ = software;
296 }
297 else if (storage.fOutxCtsFlow && storage.fRtsControl == RTS_CONTROL_HANDSHAKE)
298 {
299 value_ = hardware;
300 }
301 else
302 {
303 value_ = none;
304 }
305#else
306 if (storage.c_iflag & (IXOFF | IXON))
307 {
308 value_ = software;
309 }
310# if defined(_BSD_SOURCE)
311 else if (storage.c_cflag & CRTSCTS)
312 {
313 value_ = hardware;
314 }
315# elif defined(__QNXNTO__)
316 else if (storage.c_cflag & IHFLOW && storage.c_cflag & OHFLOW)
317 {
318 value_ = hardware;
319 }
320# endif
321 else
322 {
323 value_ = none;
324 }
325#endif
326 ec = boost::system::error_code();
327 return ec;

340boost::system::error_code serial_port_base::parity::store(
341 BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec) const
342{
343#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
344 switch (value_)
345 {
346 case none:
347 storage.fParity = FALSE;
348 storage.Parity = NOPARITY;
349 break;
350 case odd:
351 storage.fParity = TRUE;
352 storage.Parity = ODDPARITY;
353 break;
354 case even:
355 storage.fParity = TRUE;
356 storage.Parity = EVENPARITY;
357 break;
358 default:
359 break;
360 }
361#else
362 switch (value_)
363 {
364 case none:
365 storage.c_iflag |= IGNPAR;
366 storage.c_cflag &= ~(PARENB | PARODD);
367 break;
368 case even:
369 storage.c_iflag &= ~(IGNPAR | PARMRK);
370 storage.c_iflag |= INPCK;
371 storage.c_cflag |= PARENB;
372 storage.c_cflag &= ~PARODD;
373 break;
374 case odd:
375 storage.c_iflag &= ~(IGNPAR | PARMRK);
376 storage.c_iflag |= INPCK;
377 storage.c_cflag |= (PARENB | PARODD);
378 break;
379 default:
380 break;
381 }
382#endif
383 ec = boost::system::error_code();
384 return ec;
385}

387boost::system::error_code serial_port_base::parity::load(
388 const BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec)
389{
390#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
391 if (storage.Parity == EVENPARITY)
392 {
393 value_ = even;
394 }
395 else if (storage.Parity == ODDPARITY)
396 {
397 value_ = odd;
398 }
399 else
400 {
401 value_ = none;
402 }
403#else
404 if (storage.c_cflag & PARENB)
405 {
406 if (storage.c_cflag & PARODD)
407 {
408 value_ = odd;
409 }
410 else
411 {
412 value_ = even;
413 }
414 }
415 else
416 {
417 value_ = none;
418 }
419#endif
420 ec = boost::system::error_code();
421 return ec;

435boost::system::error_code serial_port_base::stop_bits::store(
436 BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec) const
437{
438#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
439 switch (value_)
440 {
441 case one:
442 storage.StopBits = ONESTOPBIT;
443 break;
444 case onepointfive:
445 storage.StopBits = ONE5STOPBITS;
446 break;
447 case two:
448 storage.StopBits = TWOSTOPBITS;
449 break;
450 default:
451 break;
452 }
453#else
454 switch (value_)
455 {
456 case one:
457 storage.c_cflag &= ~CSTOPB;
458 break;
459 case two:
460 storage.c_cflag |= CSTOPB;
461 break;
462 default:
463 ec = boost::asio::error::operation_not_supported;
464 return ec;
465 }
466#endif
467 ec = boost::system::error_code();
468 return ec;

471boost::system::error_code serial_port_base::stop_bits::load(
472 const BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec)
473{
474#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
475 if (storage.StopBits == ONESTOPBIT)
476 {
477 value_ = one;
478 }
479 else if (storage.StopBits == ONE5STOPBITS)
480 {
481 value_ = onepointfive;
482 }
483 else if (storage.StopBits == TWOSTOPBITS)
484 {
485 value_ = two;
486 }
487 else
488 {
489 value_ = one;
490 }
491#else
492 value_ = (storage.c_cflag & CSTOPB) ? two : one;
493#endif
494 ec = boost::system::error_code();
495 return ec;
496}

508boost::system::error_code serial_port_base::character_size::store(
509 BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec) const
510{
511#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
512 storage.ByteSize = value_;
513#else
514 storage.c_cflag &= ~CSIZE;
515 switch (value_)
516 {
517 case 5: storage.c_cflag |= CS5; break;
518 case 6: storage.c_cflag |= CS6; break;
519 case 7: storage.c_cflag |= CS7; break;
520 case 8: storage.c_cflag |= CS8; break;
521 default: break;
522 }
523#endif
524 ec = boost::system::error_code();
525 return ec;
526}
527
528boost::system::error_code serial_port_base::character_size::load(
529 const BOOST_ASIO_OPTION_STORAGE& storage, boost::system::error_code& ec)
530{
531#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
532 value_ = storage.ByteSize;
533#else
534 if ((storage.c_cflag & CSIZE) == CS5) { value_ = 5; }
535 else if ((storage.c_cflag & CSIZE) == CS6) { value_ = 6; }
536 else if ((storage.c_cflag & CSIZE) == CS7) { value_ = 7; }
537 else if ((storage.c_cflag & CSIZE) == CS8) { value_ = 8; }
538 else
539 {
540 // Hmmm, use 8 for now.
541 value_ = 8;
542 }
543#endif
544 ec = boost::system::error_code();
545 return ec;

73/// Holds a buffer that can be modified.
74/**
75 * The mutable_buffer class provides a safe representation of a buffer that can

381/// Get the total number of bytes in a buffer sequence.
382/**
383 * The @c BufferSequence template parameter may meet either of the @c
384 * ConstBufferSequence or @c MutableBufferSequence type requirements.
385 */
386template <typename BufferSequence>
387inline std::size_t buffer_size(const BufferSequence& b)
388{
389 std::size_t total_buffer_size = 0;
390
391 typename BufferSequence::const_iterator iter = b.begin();
392 typename BufferSequence::const_iterator end = b.end();
393 for (; iter != end; ++iter)
394 total_buffer_size += detail::buffer_size_helper(*iter);
395
396 return total_buffer_size;
397}

541/** @defgroup buffer boost::asio::buffer
542 *
543 * @brief The boost::asio::buffer function is used to create a buffer object to
544 * represent raw memory, an array of POD elements, a vector of POD elements,
545 * or a std::string.
546 *
547 * A buffer object represents a contiguous region of memory as a 2-tuple
548 * consisting of a pointer and size in bytes. A tuple of the form <tt>{void*,
549 * size_t}</tt> specifies a mutable (modifiable) region of memory. Similarly, a
550 * tuple of the form <tt>{const void*, size_t}</tt> specifies a const
551 * (non-modifiable) region of memory. These two forms correspond to the classes
552 * mutable_buffer and const_buffer, respectively. To mirror C++'s conversion
553 * rules, a mutable_buffer is implicitly convertible to a const_buffer, and the
554 * opposite conversion is not permitted.
555 *
556 * The simplest use case involves reading or writing a single buffer of a
557 * specified size:
558 *
559 * @code sock.send(boost::asio::buffer(data, size)); @endcode
560 *
561 * In the above example, the return value of boost::asio::buffer meets the
562 * requirements of the ConstBufferSequence concept so that it may be directly
563 * passed to the socket's write function. A buffer created for modifiable
564 * memory also meets the requirements of the MutableBufferSequence concept.
565 *
566 * An individual buffer may be created from a builtin array, std::vector,
567 * std::array or boost::array of POD elements. This helps prevent buffer
568 * overruns by automatically determining the size of the buffer:
569 *
570 * @code char d1[128];
571 * size_t bytes_transferred = sock.receive(boost::asio::buffer(d1));
572 *
573 * std::vector<char> d2(128);
574 * bytes_transferred = sock.receive(boost::asio::buffer(d2));
575 *
576 * std::array<char, 128> d3;
577 * bytes_transferred = sock.receive(boost::asio::buffer(d3));
578 *
579 * boost::array<char, 128> d4;
580 * bytes_transferred = sock.receive(boost::asio::buffer(d4)); @endcode
581 *
582 * In all three cases above, the buffers created are exactly 128 bytes long.
583 * Note that a vector is @e never automatically resized when creating or using
584 * a buffer. The buffer size is determined using the vector's <tt>size()</tt>
585 * member function, and not its capacity.
586 *
587 * @par Accessing Buffer Contents
588 *
589 * The contents of a buffer may be accessed using the @ref buffer_size and
590 * @ref buffer_cast functions:
591 *
592 * @code boost::asio::mutable_buffer b1 = ...;
593 * std::size_t s1 = boost::asio::buffer_size(b1);
594 * unsigned char* p1 = boost::asio::buffer_cast<unsigned char*>(b1);
595 *
596 * boost::asio::const_buffer b2 = ...;
597 * std::size_t s2 = boost::asio::buffer_size(b2);
598 * const void* p2 = boost::asio::buffer_cast<const void*>(b2); @endcode
599 *
600 * The boost::asio::buffer_cast function permits violations of type safety, so
601 * uses of it in application code should be carefully considered.
602 *
603 * For convenience, the @ref buffer_size function also works on buffer
604 * sequences (that is, types meeting the ConstBufferSequence or
605 * MutableBufferSequence type requirements). In this case, the function returns
606 * the total size of all buffers in the sequence.
607 *
608 * @par Buffer Copying
609 *
610 * The @ref buffer_copy function may be used to copy raw bytes between
611 * individual buffers and buffer sequences.
612 *
613 * In particular, when used with the @ref buffer_size, the @ref buffer_copy
614 * function can be used to linearise a sequence of buffers. For example:
615 *
616 * @code vector<const_buffer> buffers = ...;
617 *
618 * vector<unsigned char> data(boost::asio::buffer_size(buffers));
619 * boost::asio::buffer_copy(boost::asio::buffer(data), buffers); @endcode
620 *
621 * @par Buffer Invalidation
622 *
623 * A buffer object does not have any ownership of the memory it refers to. It
624 * is the responsibility of the application to ensure the memory region remains
625 * valid until it is no longer required for an I/O operation. When the memory
626 * is no longer available, the buffer is said to have been invalidated.
627 *
628 * For the boost::asio::buffer overloads that accept an argument of type
629 * std::vector, the buffer objects returned are invalidated by any vector
630 * operation that also invalidates all references, pointers and iterators
631 * referring to the elements in the sequence (C++ Std, 23.2.4)
632 *
633 * For the boost::asio::buffer overloads that accept an argument of type
634 * std::basic_string, the buffer objects returned are invalidated according to
635 * the rules defined for invalidation of references, pointers and iterators
636 * referring to elements of the sequence (C++ Std, 21.3).
637 *
638 * @par Buffer Arithmetic
639 *
640 * Buffer objects may be manipulated using simple arithmetic in a safe way
641 * which helps prevent buffer overruns. Consider an array initialised as
642 * follows:
643 *
644 * @code boost::array<char, 6> a = { 'a', 'b', 'c', 'd', 'e' }; @endcode
645 *
646 * A buffer object @c b1 created using:
647 *
648 * @code b1 = boost::asio::buffer(a); @endcode
649 *
650 * represents the entire array, <tt>{ 'a', 'b', 'c', 'd', 'e' }</tt>. An
651 * optional second argument to the boost::asio::buffer function may be used to
652 * limit the size, in bytes, of the buffer:
653 *
654 * @code b2 = boost::asio::buffer(a, 3); @endcode
655 *
656 * such that @c b2 represents the data <tt>{ 'a', 'b', 'c' }</tt>. Even if the
657 * size argument exceeds the actual size of the array, the size of the buffer
658 * object created will be limited to the array size.
659 *
660 * An offset may be applied to an existing buffer to create a new one:
661 *
662 * @code b3 = b1 + 2; @endcode
663 *
664 * where @c b3 will set to represent <tt>{ 'c', 'd', 'e' }</tt>. If the offset
665 * exceeds the size of the existing buffer, the newly created buffer will be
666 * empty.
667 *
668 * Both an offset and size may be specified to create a buffer that corresponds
669 * to a specific range of bytes within an existing buffer:
670 *
671 * @code b4 = boost::asio::buffer(b1 + 1, 3); @endcode
672 *
673 * so that @c b4 will refer to the bytes <tt>{ 'b', 'c', 'd' }</tt>.
674 *
675 * @par Buffers and Scatter-Gather I/O
676 *
677 * To read or write using multiple buffers (i.e. scatter-gather I/O), multiple
678 * buffer objects may be assigned into a container that supports the
679 * MutableBufferSequence (for read) or ConstBufferSequence (for write) concepts:
680 *
681 * @code
682 * char d1[128];
683 * std::vector<char> d2(128);
684 * boost::array<char, 128> d3;
685 *
686 * boost::array<mutable_buffer, 3> bufs1 = {
687 * boost::asio::buffer(d1),
688 * boost::asio::buffer(d2),
689 * boost::asio::buffer(d3) };
690 * bytes_transferred = sock.receive(bufs1);
691 *
692 * std::vector<const_buffer> bufs2;
693 * bufs2.push_back(boost::asio::buffer(d1));
694 * bufs2.push_back(boost::asio::buffer(d2));
695 * bufs2.push_back(boost::asio::buffer(d3));
696 * bytes_transferred = sock.send(bufs2); @endcode
697 */
698/*@{*/

1271/// Copies bytes from a source buffer to a target buffer.
1272/**
1273 * @param target A modifiable buffer representing the memory region to which
1274 * the bytes will be copied.
1275 *
1276 * @param source A non-modifiable buffer representing the memory region from
1277 * which the bytes will be copied.
1278 *
1279 * @returns The number of bytes copied.
1280 *
1281 * @note The number of bytes copied is the lesser of:
1282 *
1283 * @li @c buffer_size(target)
1284 *
1285 * @li @c buffer_size(source)
1286 */

1288 const const_buffer& source)
1289{
1290 using namespace std; // For memcpy.
1291 std::size_t target_size = buffer_size(target);
1292 std::size_t source_size = buffer_size(source);
1293 std::size_t n = target_size < source_size ? target_size : source_size;
1294 memcpy(buffer_cast<void*>(target), buffer_cast<const void*>(source), n);
1295 return n;

2078 *

2081 * @returns The number of bytes copied.
2082 *
2083 * @note The number of bytes copied is the lesser of:
2084 *
2085 * @li @c buffer_size(target)
2086 *
2087 * @li @c buffer_size(source)
2088 *
2089 * @li @c max_bytes_to_copy
2090 */
2091template <typename MutableBufferSequence, typename ConstBufferSequence>
2092std::size_t buffer_copy(const MutableBufferSequence& target,
2093 const ConstBufferSequence& source, std::size_t max_bytes_to_copy)
2094{
2095 std::size_t total_bytes_copied = 0;
2096
2097 typename MutableBufferSequence::const_iterator target_iter = target.begin();
2098 typename MutableBufferSequence::const_iterator target_end = target.end();
2099 std::size_t target_buffer_offset = 0;
2100
2101 typename ConstBufferSequence::const_iterator source_iter = source.begin();
2102 typename ConstBufferSequence::const_iterator source_end = source.end();
2103 std::size_t source_buffer_offset = 0;
2104
2105 while (total_bytes_copied != max_bytes_to_copy
2106 && target_iter != target_end && source_iter != source_end)
2107 {
2108 mutable_buffer target_buffer =
2109 mutable_buffer(*target_iter) + target_buffer_offset;
2110
2111 const_buffer source_buffer =
2112 const_buffer(*source_iter) + source_buffer_offset;
2113
2114 std::size_t bytes_copied = buffer_copy(target_buffer,
2115 source_buffer, max_bytes_to_copy - total_bytes_copied);
2116 total_bytes_copied += bytes_copied;
2117
2118 if (bytes_copied == buffer_size(target_buffer))
2119 {
2120 ++target_iter;
2121 target_buffer_offset = 0;
2122 }
2123 else
2124 target_buffer_offset += bytes_copied;
2125
2126 if (bytes_copied == buffer_size(source_buffer))
2127 {
2128 ++source_iter;
2129 source_buffer_offset = 0;
2130 }
2131 else
2132 source_buffer_offset += bytes_copied;
2133 }

29template <typename Handler, typename Arg1>
30class binder1
31{
32public:
33 binder1(const Handler& handler, const Arg1& arg1)
34 : handler_(handler),
35 arg1_(arg1)
36 {
37 }
38
39 binder1(Handler& handler, const Arg1& arg1)
40 : handler_(BOOST_ASIO_MOVE_CAST(Handler)(handler)),
41 arg1_(arg1)
42 {
43 }
44
45 void operator()()
46 {
47 handler_(static_cast<const Arg1&>(arg1_));
48 }
49
50 void operator()() const
51 {
52 handler_(arg1_);
53 }
54
55//private:
56 Handler handler_;
57 Arg1 arg1_;
58};

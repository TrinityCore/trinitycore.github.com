26// Adapts std::chrono clocks for use with a deadline timer.
27template <typename Clock, typename WaitTraits>
28struct chrono_time_traits
29{
30 // The clock type.
31 typedef Clock clock_type;
32
33 // The duration type of the clock.
34 typedef typename clock_type::duration duration_type;
35
36 // The time point type of the clock.
37 typedef typename clock_type::time_point time_type;
38
39 // The period of the clock.
40 typedef typename duration_type::period period_type;
41
42 // Get the current time.
43 static time_type now()
44 {
45 return clock_type::now();
46 }
47
48 // Add a duration to a time.
49 static time_type add(const time_type& t, const duration_type& d)
50 {
51 const time_type epoch;
52 if (t >= epoch)
53 {
54 if ((time_type::max)() - t < d)
55 return (time_type::max)();
56 }
57 else // t < epoch
58 {
59 if (-(t - (time_type::min)()) > d)
60 return (time_type::min)();
61 }
62
63 return t + d;
64 }
65
66 // Subtract one time from another.
67 static duration_type subtract(const time_type& t1, const time_type& t2)
68 {
69 const time_type epoch;
70 if (t1 >= epoch)
71 {
72 if (t2 >= epoch)
73 {
74 return t1 - t2;
75 }
76 else if (t2 == (time_type::min)())
77 {
78 return (duration_type::max)();
79 }
80 else if ((time_type::max)() - t1 < epoch - t2)
81 {
82 return (duration_type::max)();
83 }
84 else
85 {
86 return t1 - t2;
87 }
88 }
89 else // t1 < epoch
90 {
91 if (t2 < epoch)
92 {
93 return t1 - t2;
94 }
95 else if (t1 == (time_type::min)())
96 {
97 return (duration_type::min)();
98 }
99 else if ((time_type::max)() - t2 < epoch - t1)
100 {
101 return (duration_type::min)();
102 }
103 else
104 {
105 return -(t2 - t1);
106 }
107 }
108 }
109
110 // Test whether one time is less than another.
111 static bool less_than(const time_type& t1, const time_type& t2)
112 {
113 return t1 < t2;
114 }

116 // Implement just enough of the posix_time::time_duration interface to supply
117 // what the timer_queue requires.
118 class posix_time_duration
119 {
120 public:
121 explicit posix_time_duration(const duration_type& d)
122 : d_(d)
123 {
124 }
125
126 int64_t ticks() const
127 {
128 return d_.count();
129 }
130
131 int64_t total_seconds() const
132 {
133 return duration_cast<1, 1>();
134 }
135
136 int64_t total_milliseconds() const
137 {
138 return duration_cast<1, 1000>();
139 }
140
141 int64_t total_microseconds() const
142 {
143 return duration_cast<1, 1000000>();
144 }
145
146 private:
147 template <int64_t Num, int64_t Den>
148 int64_t duration_cast() const
149 {
150 const int64_t num = period_type::num * Den;
151 const int64_t den = period_type::den * Num;
152
153 if (num == 1 && den == 1)
154 return ticks();
155 else if (num != 1 && den == 1)
156 return ticks() * num;
157 else if (num == 1 && period_type::den != 1)
158 return ticks() / den;
159 else
160 return ticks() * num / den;
161 }
162
163 duration_type d_;
164 };
165
166 // Convert to POSIX duration type.
167 static posix_time_duration to_posix_duration(const duration_type& d)
168 {
169 return posix_time_duration(WaitTraits::to_wait_duration(d));
170 }
171};

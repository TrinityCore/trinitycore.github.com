34namespace boost {
35namespace asio {
36namespace detail {
37
38class task_io_service
39 : public boost::asio::detail::service_base<task_io_service>
40{
41public:
42 typedef task_io_service_operation operation;
43
44 // Constructor. Specifies the number of concurrent threads that are likely to
45 // run the io_service. If set to 1 certain optimisation are performed.
46 BOOST_ASIO_DECL task_io_service(boost::asio::io_service& io_service,
47 std::size_t concurrency_hint = 0);
48
49 // Destroy all user-defined handler objects owned by the service.
50 BOOST_ASIO_DECL void shutdown_service();
51
52 // Initialise the task, if required.
53 BOOST_ASIO_DECL void init_task();
54
55 // Run the event loop until interrupted or no more work.
56 BOOST_ASIO_DECL std::size_t run(boost::system::error_code& ec);
57
58 // Run until interrupted or one operation is performed.
59 BOOST_ASIO_DECL std::size_t run_one(boost::system::error_code& ec);
60
61 // Poll for operations without blocking.
62 BOOST_ASIO_DECL std::size_t poll(boost::system::error_code& ec);
63
64 // Poll for one operation without blocking.
65 BOOST_ASIO_DECL std::size_t poll_one(boost::system::error_code& ec);
66
67 // Interrupt the event processing loop.
68 BOOST_ASIO_DECL void stop();
69
70 // Determine whether the io_service is stopped.
71 BOOST_ASIO_DECL bool stopped() const;
72
73 // Reset in preparation for a subsequent run invocation.
74 BOOST_ASIO_DECL void reset();
75
76 // Notify that some work has started.
77 void work_started()
78 {
79 ++outstanding_work_;
80 }
81
82 // Notify that some work has finished.
83 void work_finished()
84 {
85 if (--outstanding_work_ == 0)
86 stop();
87 }
88
89 // Return whether a handler can be dispatched immediately.
90 bool can_dispatch()
91 {
92 return thread_call_stack::contains(this) != 0;
93 }
94
95 // Request invocation of the given handler.
96 template <typename Handler>
97 void dispatch(Handler& handler);
98
99 // Request invocation of the given handler and return immediately.
100 template <typename Handler>
101 void post(Handler& handler);
102
103 // Request invocation of the given operation and return immediately. Assumes
104 // that work_started() has not yet been called for the operation.
105 BOOST_ASIO_DECL void post_immediate_completion(
106 operation* op, bool is_continuation);
107
108 // Request invocation of the given operation and return immediately. Assumes
109 // that work_started() was previously called for the operation.
110 BOOST_ASIO_DECL void post_deferred_completion(operation* op);
111
112 // Request invocation of the given operations and return immediately. Assumes
113 // that work_started() was previously called for each operation.
114 BOOST_ASIO_DECL void post_deferred_completions(op_queue<operation>& ops);
115
116 // Process unfinished operations as part of a shutdown_service operation.
117 // Assumes that work_started() was previously called for the operations.
118 BOOST_ASIO_DECL void abandon_operations(op_queue<operation>& ops);
119
120private:
121 // Structure containing information about an idle thread.
122 typedef task_io_service_thread_info thread_info;
123
124 // Enqueue the given operation following a failed attempt to dispatch the
125 // operation for immediate invocation.
126 BOOST_ASIO_DECL void do_dispatch(operation* op);
127
128 // Run at most one operation. May block.
129 BOOST_ASIO_DECL std::size_t do_run_one(mutex::scoped_lock& lock,
130 thread_info& this_thread, const boost::system::error_code& ec);
131
132 // Poll for at most one operation.
133 BOOST_ASIO_DECL std::size_t do_poll_one(mutex::scoped_lock& lock,
134 thread_info& this_thread, const boost::system::error_code& ec);
135
136 // Stop the task and all idle threads.
137 BOOST_ASIO_DECL void stop_all_threads(mutex::scoped_lock& lock);
138
139 // Wakes a single idle thread and unlocks the mutex. Returns true if an idle
140 // thread was found. If there is no idle thread, returns false and leaves the
141 // mutex locked.
142 BOOST_ASIO_DECL bool wake_one_idle_thread_and_unlock(
143 mutex::scoped_lock& lock);
144
145 // Wake a single idle thread, or the task, and always unlock the mutex.
146 BOOST_ASIO_DECL void wake_one_thread_and_unlock(
147 mutex::scoped_lock& lock);
148
149 // Helper class to perform task-related operations on block exit.
150 struct task_cleanup;
151 friend struct task_cleanup;
152
153 // Helper class to call work-related operations on block exit.
154 struct work_cleanup;
155 friend struct work_cleanup;
156
157 // Whether to optimise for single-threaded use cases.
158 const bool one_thread_;
159
160 // Mutex to protect access to internal data.
161 mutable mutex mutex_;
162
163 // The task to be run by this service.
164 reactor* task_;
165
166 // Operation object to represent the position of the task in the queue.
167 struct task_operation : operation
168 {
169 task_operation() : operation(0) {}
170 } task_operation_;
171
172 // Whether the task has been interrupted.
173 bool task_interrupted_;
174
175 // The count of unfinished work.
176 atomic_count outstanding_work_;
177
178 // The queue of handlers that are ready to be delivered.
179 op_queue<operation> op_queue_;
180
181 // Flag to indicate that the dispatcher has been stopped.
182 bool stopped_;
183
184 // Flag to indicate that the dispatcher has been shut down.
185 bool shutdown_;
186
187 // Per-thread call stack to track the state of each thread in the io_service.
188 typedef call_stack<task_io_service, thread_info> thread_call_stack;
189
190 // The threads that are currently idle.
191 thread_info* first_idle_thread_;
192};

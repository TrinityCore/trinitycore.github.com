33// Descriptor state bits.
34enum
35{
36 // The user wants a non-blocking descriptor.
37 user_set_non_blocking = 1,
38
39 // The descriptor has been set non-blocking.
40 internal_non_blocking = 2,
41
42 // Helper "state" used to determine whether the descriptor is non-blocking.
43 non_blocking = user_set_non_blocking | internal_non_blocking,
44
45 // The descriptor may have been dup()-ed.
46 possible_dup = 4
47};
48
49typedef unsigned char state_type;

51template <typename ReturnType>
52inline ReturnType error_wrapper(ReturnType return_value,
53 boost::system::error_code& ec)
54{
55 ec = boost::system::error_code(errno,
56 boost::asio::error::get_system_category());
57 return return_value;
58}

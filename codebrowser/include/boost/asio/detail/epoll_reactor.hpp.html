39namespace boost {
40namespace asio {
41namespace detail {
42
43class epoll_reactor
44 : public boost::asio::detail::service_base<epoll_reactor>
45{
46public:
47 enum op_types { read_op = 0, write_op = 1,
48 connect_op = 1, except_op = 2, max_ops = 3 };
49
50 // Per-descriptor queues.
51 class descriptor_state : operation
52 {
53 friend class epoll_reactor;
54 friend class object_pool_access;
55
56 descriptor_state* next_;
57 descriptor_state* prev_;
58
59 mutex mutex_;
60 epoll_reactor* reactor_;
61 int descriptor_;
62 uint32_t registered_events_;
63 op_queue<reactor_op> op_queue_[max_ops];
64 bool shutdown_;
65
66 BOOST_ASIO_DECL descriptor_state();
67 void set_ready_events(uint32_t events) { task_result_ = events; }
68 BOOST_ASIO_DECL operation* perform_io(uint32_t events);
69 BOOST_ASIO_DECL static void do_complete(
70 io_service_impl* owner, operation* base,
71 const boost::system::error_code& ec, std::size_t bytes_transferred);
72 };
73
74 // Per-descriptor data.
75 typedef descriptor_state* per_descriptor_data;
76
77 // Constructor.
78 BOOST_ASIO_DECL epoll_reactor(boost::asio::io_service& io_service);
79
80 // Destructor.
81 BOOST_ASIO_DECL ~epoll_reactor();
82
83 // Destroy all user-defined handler objects owned by the service.
84 BOOST_ASIO_DECL void shutdown_service();
85
86 // Recreate internal descriptors following a fork.
87 BOOST_ASIO_DECL void fork_service(
88 boost::asio::io_service::fork_event fork_ev);
89
90 // Initialise the task.
91 BOOST_ASIO_DECL void init_task();
92
93 // Register a socket with the reactor. Returns 0 on success, system error
94 // code on failure.
95 BOOST_ASIO_DECL int register_descriptor(socket_type descriptor,
96 per_descriptor_data& descriptor_data);
97
98 // Register a descriptor with an associated single operation. Returns 0 on
99 // success, system error code on failure.
100 BOOST_ASIO_DECL int register_internal_descriptor(
101 int op_type, socket_type descriptor,
102 per_descriptor_data& descriptor_data, reactor_op* op);
103
104 // Move descriptor registration from one descriptor_data object to another.
105 BOOST_ASIO_DECL void move_descriptor(socket_type descriptor,
106 per_descriptor_data& target_descriptor_data,
107 per_descriptor_data& source_descriptor_data);
108
109 // Post a reactor operation for immediate completion.
110 void post_immediate_completion(reactor_op* op, bool is_continuation)
111 {
112 io_service_.post_immediate_completion(op, is_continuation);
113 }
114
115 // Start a new operation. The reactor operation will be performed when the
116 // given descriptor is flagged as ready, or an error has occurred.
117 BOOST_ASIO_DECL void start_op(int op_type, socket_type descriptor,
118 per_descriptor_data& descriptor_data, reactor_op* op,
119 bool is_continuation, bool allow_speculative);
120
121 // Cancel all operations associated with the given descriptor. The
122 // handlers associated with the descriptor will be invoked with the
123 // operation_aborted error.
124 BOOST_ASIO_DECL void cancel_ops(socket_type descriptor,
125 per_descriptor_data& descriptor_data);
126
127 // Cancel any operations that are running against the descriptor and remove
128 // its registration from the reactor.
129 BOOST_ASIO_DECL void deregister_descriptor(socket_type descriptor,
130 per_descriptor_data& descriptor_data, bool closing);
131
132 // Remote the descriptor's registration from the reactor.
133 BOOST_ASIO_DECL void deregister_internal_descriptor(
134 socket_type descriptor, per_descriptor_data& descriptor_data);
135
136 // Add a new timer queue to the reactor.
137 template <typename Time_Traits>
138 void add_timer_queue(timer_queue<Time_Traits>& timer_queue);
139
140 // Remove a timer queue from the reactor.
141 template <typename Time_Traits>
142 void remove_timer_queue(timer_queue<Time_Traits>& timer_queue);
143
144 // Schedule a new operation in the given timer queue to expire at the
145 // specified absolute time.
146 template <typename Time_Traits>
147 void schedule_timer(timer_queue<Time_Traits>& queue,
148 const typename Time_Traits::time_type& time,
149 typename timer_queue<Time_Traits>::per_timer_data& timer, wait_op* op);
150
151 // Cancel the timer operations associated with the given token. Returns the
152 // number of operations that have been posted or dispatched.
153 template <typename Time_Traits>
154 std::size_t cancel_timer(timer_queue<Time_Traits>& queue,
155 typename timer_queue<Time_Traits>::per_timer_data& timer,
156 std::size_t max_cancelled = (std::numeric_limits<std::size_t>::max)());
157
158 // Run epoll once until interrupted or events are ready to be dispatched.
159 BOOST_ASIO_DECL void run(bool block, op_queue<operation>& ops);
160
161 // Interrupt the select loop.
162 BOOST_ASIO_DECL void interrupt();
163
164private:
165 // The hint to pass to epoll_create to size its data structures.
166 enum { epoll_size = 20000 };
167
168 // Create the epoll file descriptor. Throws an exception if the descriptor
169 // cannot be created.
170 BOOST_ASIO_DECL static int do_epoll_create();
171
172 // Create the timerfd file descriptor. Does not throw.
173 BOOST_ASIO_DECL static int do_timerfd_create();
174
175 // Allocate a new descriptor state object.
176 BOOST_ASIO_DECL descriptor_state* allocate_descriptor_state();
177
178 // Free an existing descriptor state object.
179 BOOST_ASIO_DECL void free_descriptor_state(descriptor_state* s);
180
181 // Helper function to add a new timer queue.
182 BOOST_ASIO_DECL void do_add_timer_queue(timer_queue_base& queue);
183
184 // Helper function to remove a timer queue.
185 BOOST_ASIO_DECL void do_remove_timer_queue(timer_queue_base& queue);
186
187 // Called to recalculate and update the timeout.
188 BOOST_ASIO_DECL void update_timeout();
189
190 // Get the timeout value for the epoll_wait call. The timeout value is
191 // returned as a number of milliseconds. A return value of -1 indicates
192 // that epoll_wait should block indefinitely.
193 BOOST_ASIO_DECL int get_timeout();
194
195#if defined(BOOST_ASIO_HAS_TIMERFD)
196 // Get the timeout value for the timer descriptor. The return value is the
197 // flag argument to be used when calling timerfd_settime.
198 BOOST_ASIO_DECL int get_timeout(itimerspec& ts);
199#endif // defined(BOOST_ASIO_HAS_TIMERFD)
200
201 // The io_service implementation used to post completions.
202 io_service_impl& io_service_;
203
204 // Mutex to protect access to internal data.
205 mutex mutex_;
206
207 // The interrupter is used to break a blocking epoll_wait call.
208 select_interrupter interrupter_;
209
210 // The epoll file descriptor.
211 int epoll_fd_;
212
213 // The timer file descriptor.
214 int timer_fd_;
215
216 // The timer queues.
217 timer_queue_set timer_queues_;
218
219 // Whether the service has been shut down.
220 bool shutdown_;
221
222 // Mutex to protect access to the registered descriptors.
223 mutex registered_descriptors_mutex_;
224
225 // Keep track of all registered descriptors.
226 object_pool<descriptor_state> registered_descriptors_;
227
228 // Helper class to do post-perform_io cleanup.
229 struct perform_io_cleanup_on_block_exit;
230 friend struct perform_io_cleanup_on_block_exit;
231};

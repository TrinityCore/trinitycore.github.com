103// If BOOST_ASIO_MOVE_CAST isn't defined, and move support is available, define
104// BOOST_ASIO_MOVE_ARG and BOOST_ASIO_MOVE_CAST to take advantage of rvalue
105// references and perfect forwarding.
106#if defined(BOOST_ASIO_HAS_MOVE) && !defined(BOOST_ASIO_MOVE_CAST)
107# define BOOST_ASIO_MOVE_ARG(type) type&&
108# define BOOST_ASIO_MOVE_CAST(type) static_cast<type&&>
109# define BOOST_ASIO_MOVE_CAST2(type1, type2) static_cast<type1, type2&&>
110#endif // defined(BOOST_ASIO_HAS_MOVE) && !defined(BOOST_ASIO_MOVE_CAST)

112// If BOOST_ASIO_MOVE_CAST still isn't defined, default to a C++03-compatible
113// implementation. Note that older g++ and MSVC versions don't like it when you
114// pass a non-member function through a const reference, so for most compilers
115// we'll play it safe and stick with the old approach of passing the handler by
116// value.
117#if !defined(BOOST_ASIO_MOVE_CAST)
118# if defined(__GNUC__)
119# if ((__GNUC__ == 4) && (__GNUC_MINOR__ >= 1)) || (__GNUC__ > 4)
120# define BOOST_ASIO_MOVE_ARG(type) const type&
121# else // ((__GNUC__ == 4) && (__GNUC_MINOR__ >= 1)) || (__GNUC__ > 4)
122# define BOOST_ASIO_MOVE_ARG(type) type
123# endif // ((__GNUC__ == 4) && (__GNUC_MINOR__ >= 1)) || (__GNUC__ > 4)
124# elif defined(BOOST_ASIO_MSVC)
125# if (_MSC_VER >= 1400)
126# define BOOST_ASIO_MOVE_ARG(type) const type&
127# else // (_MSC_VER >= 1400)
128# define BOOST_ASIO_MOVE_ARG(type) type
129# endif // (_MSC_VER >= 1400)
130# else
131# define BOOST_ASIO_MOVE_ARG(type) type
132# endif
133# define BOOST_ASIO_MOVE_CAST(type) static_cast<const type&>
134# define BOOST_ASIO_MOVE_CAST2(type1, type2) static_cast<const type1, type2&>

35namespace boost {
36namespace asio {
37namespace detail {
38
39template <typename Time_Traits>
40class deadline_timer_service
41{
42public:
43 // The time type.
44 typedef typename Time_Traits::time_type time_type;
45
46 // The duration type.
47 typedef typename Time_Traits::duration_type duration_type;
48
49 // The implementation type of the timer. This type is dependent on the
50 // underlying implementation of the timer service.
51 struct implementation_type
52 : private boost::asio::detail::noncopyable
53 {
54 time_type expiry;
55 bool might_have_pending_waits;
56 typename timer_queue<Time_Traits>::per_timer_data timer_data;
57 };
58
59 // Constructor.
60 deadline_timer_service(boost::asio::io_service& io_service)
61 : scheduler_(boost::asio::use_service<timer_scheduler>(io_service))
62 {
63 scheduler_.init_task();
64 scheduler_.add_timer_queue(timer_queue_);
65 }
66
67 // Destructor.
68 ~deadline_timer_service()
69 {
70 scheduler_.remove_timer_queue(timer_queue_);
71 }
72
73 // Destroy all user-defined handler objects owned by the service.
74 void shutdown_service()
75 {
76 }
77
78 // Construct a new timer implementation.
79 void construct(implementation_type& impl)
80 {
81 impl.expiry = time_type();
82 impl.might_have_pending_waits = false;
83 }
84
85 // Destroy a timer implementation.
86 void destroy(implementation_type& impl)
87 {
88 boost::system::error_code ec;
89 cancel(impl, ec);
90 }
91
92 // Cancel any asynchronous wait operations associated with the timer.
93 std::size_t cancel(implementation_type& impl, boost::system::error_code& ec)
94 {
95 if (!impl.might_have_pending_waits)
96 {
97 ec = boost::system::error_code();
98 return 0;
99 }
100
101 BOOST_ASIO_HANDLER_OPERATION(("deadline_timer", &impl, "cancel"));
102
103 std::size_t count = scheduler_.cancel_timer(timer_queue_, impl.timer_data);
104 impl.might_have_pending_waits = false;
105 ec = boost::system::error_code();
106 return count;
107 }
108
109 // Cancels one asynchronous wait operation associated with the timer.
110 std::size_t cancel_one(implementation_type& impl,
111 boost::system::error_code& ec)
112 {
113 if (!impl.might_have_pending_waits)
114 {
115 ec = boost::system::error_code();
116 return 0;
117 }
118
119 BOOST_ASIO_HANDLER_OPERATION(("deadline_timer", &impl, "cancel_one"));
120
121 std::size_t count = scheduler_.cancel_timer(
122 timer_queue_, impl.timer_data, 1);
123 if (count == 0)
124 impl.might_have_pending_waits = false;
125 ec = boost::system::error_code();
126 return count;
127 }
128
129 // Get the expiry time for the timer as an absolute time.
130 time_type expires_at(const implementation_type& impl) const
131 {
132 return impl.expiry;
133 }
134
135 // Set the expiry time for the timer as an absolute time.
136 std::size_t expires_at(implementation_type& impl,
137 const time_type& expiry_time, boost::system::error_code& ec)
138 {
139 std::size_t count = cancel(impl, ec);
140 impl.expiry = expiry_time;
141 ec = boost::system::error_code();
142 return count;
143 }
144
145 // Get the expiry time for the timer relative to now.
146 duration_type expires_from_now(const implementation_type& impl) const
147 {
148 return Time_Traits::subtract(expires_at(impl), Time_Traits::now());
149 }
150
151 // Set the expiry time for the timer relative to now.
152 std::size_t expires_from_now(implementation_type& impl,
153 const duration_type& expiry_time, boost::system::error_code& ec)
154 {
155 return expires_at(impl,
156 Time_Traits::add(Time_Traits::now(), expiry_time), ec);
157 }
158
159 // Perform a blocking wait on the timer.
160 void wait(implementation_type& impl, boost::system::error_code& ec)
161 {
162 time_type now = Time_Traits::now();
163 ec = boost::system::error_code();
164 while (Time_Traits::less_than(now, impl.expiry) && !ec)
165 {
166 this->do_wait(Time_Traits::to_posix_duration(
167 Time_Traits::subtract(impl.expiry, now)), ec);
168 now = Time_Traits::now();
169 }
170 }
171
172 // Start an asynchronous wait on the timer.
173 template <typename Handler>
174 void async_wait(implementation_type& impl, Handler& handler)
175 {
176 // Allocate and construct an operation to wrap the handler.
177 typedef wait_handler<Handler> op;
178 typename op::ptr p = { boost::asio::detail::addressof(handler),
179 boost_asio_handler_alloc_helpers::allocate(
180 sizeof(op), handler), 0 };
181 p.p = new (p.v) op(handler);
182
183 impl.might_have_pending_waits = true;
184
185 BOOST_ASIO_HANDLER_CREATION((p.p, "deadline_timer", &impl, "async_wait"));
186
187 scheduler_.schedule_timer(timer_queue_, impl.expiry, impl.timer_data, p.p);
188 p.v = p.p = 0;
189 }
190
191private:
192 // Helper function to wait given a duration type. The duration type should
193 // either be of type boost::posix_time::time_duration, or implement the
194 // required subset of its interface.
195 template <typename Duration>
196 void do_wait(const Duration& timeout, boost::system::error_code& ec)
197 {
198 ::timeval tv;
199 tv.tv_sec = timeout.total_seconds();
200 tv.tv_usec = timeout.total_microseconds() % 1000000;
201 socket_ops::select(0, 0, 0, 0, &tv, ec);
202 }
203
204 // The queue of timers.
205 timer_queue<Time_Traits> timer_queue_;
206
207 // The object that schedules and executes timers. Usually a reactor.
208 timer_scheduler& scheduler_;
209};

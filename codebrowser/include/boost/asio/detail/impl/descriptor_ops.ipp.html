41int close(int d, state_type& state, boost::system::error_code& ec)
42{
43 int result = 0;
44 if (d != -1)
45 {
46 errno = 0;
47 result = error_wrapper(::close(d), ec);
48
49 if (result != 0
50 && (ec == boost::asio::error::would_block
51 || ec == boost::asio::error::try_again))
52 {
53 // According to UNIX Network Programming Vol. 1, it is possible for
54 // close() to fail with EWOULDBLOCK under certain circumstances. What
55 // isn't clear is the state of the descriptor after this error. The one
56 // current OS where this behaviour is seen, Windows, says that the socket
57 // remains open. Therefore we'll put the descriptor back into blocking
58 // mode and have another attempt at closing it.
59#if defined(__SYMBIAN32__)
60 int flags = ::fcntl(d, F_GETFL, 0);
61 if (flags >= 0)
62 ::fcntl(d, F_SETFL, flags & ~O_NONBLOCK);
63#else // defined(__SYMBIAN32__)
64 ioctl_arg_type arg = 0;
65 ::ioctl(d, FIONBIO, &arg);
66#endif // defined(__SYMBIAN32__)
67 state &= ~non_blocking;
68
69 errno = 0;
70 result = error_wrapper(::close(d), ec);
71 }
72 }
73
74 if (result == 0)
75 ec = boost::system::error_code();
76 return result;
77}

79bool set_user_non_blocking(int d, state_type& state,
80 bool value, boost::system::error_code& ec)
81{
82 if (d == -1)
83 {
84 ec = boost::asio::error::bad_descriptor;
85 return false;
86 }
87
88 errno = 0;
89#if defined(__SYMBIAN32__)
90 int result = error_wrapper(::fcntl(d, F_GETFL, 0), ec);
91 if (result >= 0)
92 {
93 errno = 0;
94 int flag = (value ? (result | O_NONBLOCK) : (result & ~O_NONBLOCK));
95 result = error_wrapper(::fcntl(d, F_SETFL, flag), ec);
96 }
97#else // defined(__SYMBIAN32__)
98 ioctl_arg_type arg = (value ? 1 : 0);
99 int result = error_wrapper(::ioctl(d, FIONBIO, &arg), ec);
100#endif // defined(__SYMBIAN32__)
101
102 if (result >= 0)
103 {
104 ec = boost::system::error_code();
105 if (value)
106 state |= user_set_non_blocking;
107 else
108 {
109 // Clearing the user-set non-blocking mode always overrides any
110 // internally-set non-blocking flag. Any subsequent asynchronous
111 // operations will need to re-enable non-blocking I/O.
112 state &= ~(user_set_non_blocking | internal_non_blocking);
113 }
114 return true;
115 }
116
117 return false;
118}

120bool set_internal_non_blocking(int d, state_type& state,
121 bool value, boost::system::error_code& ec)
122{
123 if (d == -1)
124 {
125 ec = boost::asio::error::bad_descriptor;
126 return false;
127 }
128
129 if (!value && (state & user_set_non_blocking))
130 {
131 // It does not make sense to clear the internal non-blocking flag if the
132 // user still wants non-blocking behaviour. Return an error and let the
133 // caller figure out whether to update the user-set non-blocking flag.
134 ec = boost::asio::error::invalid_argument;
135 return false;
136 }
137
138 errno = 0;
139#if defined(__SYMBIAN32__)
140 int result = error_wrapper(::fcntl(d, F_GETFL, 0), ec);
141 if (result >= 0)
142 {
143 errno = 0;
144 int flag = (value ? (result | O_NONBLOCK) : (result & ~O_NONBLOCK));
145 result = error_wrapper(::fcntl(d, F_SETFL, flag), ec);
146 }
147#else // defined(__SYMBIAN32__)
148 ioctl_arg_type arg = (value ? 1 : 0);
149 int result = error_wrapper(::ioctl(d, FIONBIO, &arg), ec);
150#endif // defined(__SYMBIAN32__)
151
152 if (result >= 0)
153 {
154 ec = boost::system::error_code();
155 if (value)
156 state |= internal_non_blocking;
157 else
158 state &= ~internal_non_blocking;
159 return true;
160 }
161
162 return false;
163}

165std::size_t sync_read(int d, state_type state, buf* bufs,
166 std::size_t count, bool all_empty, boost::system::error_code& ec)
167{
168 if (d == -1)
169 {
170 ec = boost::asio::error::bad_descriptor;
171 return 0;
172 }
173
174 // A request to read 0 bytes on a stream is a no-op.
175 if (all_empty)
176 {
177 ec = boost::system::error_code();
178 return 0;
179 }
180
181 // Read some data.
182 for (;;)
183 {
184 // Try to complete the operation without blocking.
185 errno = 0;
186 signed_size_type bytes = error_wrapper(::readv(
187 d, bufs, static_cast<int>(count)), ec);
188
189 // Check if operation succeeded.
190 if (bytes > 0)
191 return bytes;
192
193 // Check for EOF.
194 if (bytes == 0)
195 {
196 ec = boost::asio::error::eof;
197 return 0;
198 }
199
200 // Operation failed.
201 if ((state & user_set_non_blocking)
202 || (ec != boost::asio::error::would_block
203 && ec != boost::asio::error::try_again))
204 return 0;
205
206 // Wait for descriptor to become ready.
207 if (descriptor_ops::poll_read(d, 0, ec) < 0)
208 return 0;
209 }
210}

212bool non_blocking_read(int d, buf* bufs, std::size_t count,
213 boost::system::error_code& ec, std::size_t& bytes_transferred)
214{
215 for (;;)
216 {
217 // Read some data.
218 errno = 0;
219 signed_size_type bytes = error_wrapper(::readv(
220 d, bufs, static_cast<int>(count)), ec);
221
222 // Check for end of stream.
223 if (bytes == 0)
224 {
225 ec = boost::asio::error::eof;
226 return true;
227 }
228
229 // Retry operation if interrupted by signal.
230 if (ec == boost::asio::error::interrupted)
231 continue;
232
233 // Check if we need to run the operation again.
234 if (ec == boost::asio::error::would_block
235 || ec == boost::asio::error::try_again)
236 return false;
237
238 // Operation is complete.
239 if (bytes > 0)
240 {
241 ec = boost::system::error_code();
242 bytes_transferred = bytes;
243 }
244 else
245 bytes_transferred = 0;
246
247 return true;
248 }
249}

323int ioctl(int d, state_type& state, long cmd,
324 ioctl_arg_type* arg, boost::system::error_code& ec)
325{
326 if (d == -1)
327 {
328 ec = boost::asio::error::bad_descriptor;
329 return -1;
330 }
331
332 errno = 0;
333 int result = error_wrapper(::ioctl(d, cmd, arg), ec);
334
335 if (result >= 0)
336 {
337 ec = boost::system::error_code();
338
339 // When updating the non-blocking mode we always perform the ioctl syscall,
340 // even if the flags would otherwise indicate that the descriptor is
341 // already in the correct state. This ensures that the underlying
342 // descriptor is put into the state that has been requested by the user. If
343 // the ioctl syscall was successful then we need to update the flags to
344 // match.
345 if (cmd == static_cast<long>(FIONBIO))
346 {
347 if (*arg)
348 {
349 state |= user_set_non_blocking;
350 }
351 else
352 {
353 // Clearing the non-blocking mode always overrides any internally-set
354 // non-blocking flag. Any subsequent asynchronous operations will need
355 // to re-enable non-blocking I/O.
356 state &= ~(user_set_non_blocking | internal_non_blocking);
357 }
358 }
359 }
360
361 return result;
362}

394int poll_read(int d, state_type state, boost::system::error_code& ec)
395{
396 if (d == -1)
397 {
398 ec = boost::asio::error::bad_descriptor;
399 return -1;
400 }
401
402 pollfd fds;
403 fds.fd = d;
404 fds.events = POLLIN;
405 fds.revents = 0;
406 int timeout = (state & user_set_non_blocking) ? 0 : -1;
407 errno = 0;
408 int result = error_wrapper(::poll(&fds, 1, timeout), ec);
409 if (result == 0)
410 ec = (state & user_set_non_blocking)
411 ? boost::asio::error::would_block : boost::system::error_code();
412 else if (result > 0)
413 ec = boost::system::error_code();
414 return result;
415}

426 fds.fd = d;

428 fds.revents = 0;
429 int timeout = (state & user_set_non_blocking) ? 0 : -1;

32struct signal_state
33{
34 // Mutex used for protecting global state.
35 static_mutex mutex_;
36
37 // The read end of the pipe used for signal notifications.
38 int read_descriptor_;
39
40 // The write end of the pipe used for signal notifications.
41 int write_descriptor_;
42
43 // Whether the signal state has been prepared for a fork.
44 bool fork_prepared_;
45
46 // The head of a linked list of all signal_set_service instances.
47 class signal_set_service* service_list_;
48
49 // A count of the number of objects that are registered for each signal.
50 std::size_t registration_count_[max_signal_number];
51};
52
53signal_state* get_signal_state()
54{
55 static signal_state state = {
56 BOOST_ASIO_STATIC_MUTEX_INIT, -1, -1, false, 0, { 0 } };
57 return &state;
58}

60void asio_signal_handler(int signal_number)
61{
62#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
63 signal_set_service::deliver_signal(signal_number);
64#else // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
65 int saved_errno = errno;
66 signal_state* state = get_signal_state();
67 signed_size_type result = ::write(state->write_descriptor_,
68 &signal_number, sizeof(signal_number));
69 (void)result;
70 errno = saved_errno;
71#endif // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
72
73#if defined(BOOST_ASIO_HAS_SIGNAL) && !defined(BOOST_ASIO_HAS_SIGACTION)
74 ::signal(signal_number, asio_signal_handler);
75#endif // defined(BOOST_ASIO_HAS_SIGNAL) && !defined(BOOST_ASIO_HAS_SIGACTION)
76}

78#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
79class signal_set_service::pipe_read_op : public reactor_op
80{
81public:
82 pipe_read_op()
83 : reactor_op(&pipe_read_op::do_perform, pipe_read_op::do_complete)
84 {
85 }
86
87 static bool do_perform(reactor_op*)
88 {
89 signal_state* state = get_signal_state();
90
91 int fd = state->read_descriptor_;
92 int signal_number = 0;
93 while (::read(fd, &signal_number, sizeof(int)) == sizeof(int))
94 if (signal_number >= 0 && signal_number < max_signal_number)
95 signal_set_service::deliver_signal(signal_number);
96
97 return false;
98 }
99
100 static void do_complete(io_service_impl* /*owner*/, operation* base,
101 const boost::system::error_code& /*ec*/,
102 std::size_t /*bytes_transferred*/)
103 {
104 pipe_read_op* o(static_cast<pipe_read_op*>(base));
105 delete o;
106 }
107};
108#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)

110signal_set_service::signal_set_service(
111 boost::asio::io_service& io_service)
112 : io_service_(boost::asio::use_service<io_service_impl>(io_service)),
113#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
114 reactor_(boost::asio::use_service<reactor>(io_service)),
115#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
116 next_(0),
117 prev_(0)
118{
119 get_signal_state()->mutex_.init();
120
121#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
122 reactor_.init_task();
123#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
124
125 for (int i = 0; i < max_signal_number; ++i)
126 registrations_[i] = 0;
127
128 add_service(this);
129}

136void signal_set_service::shutdown_service()
137{
138 remove_service(this);
139
140 op_queue<operation> ops;
141
142 for (int i = 0; i < max_signal_number; ++i)
143 {
144 registration* reg = registrations_[i];
145 while (reg)
146 {
147 ops.push(*reg->queue_);
148 reg = reg->next_in_table_;
149 }
150 }
151
152 io_service_.abandon_operations(ops);
153}

155void signal_set_service::fork_service(
156 boost::asio::io_service::fork_event fork_ev)
157{
158#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
159 signal_state* state = get_signal_state();
160 static_mutex::scoped_lock lock(state->mutex_);
161
162 switch (fork_ev)
163 {
164 case boost::asio::io_service::fork_prepare:
165 {
166 int read_descriptor = state->read_descriptor_;
167 state->fork_prepared_ = true;
168 lock.unlock();
169 reactor_.deregister_internal_descriptor(read_descriptor, reactor_data_);
170 }
171 break;
172 case boost::asio::io_service::fork_parent:
173 if (state->fork_prepared_)
174 {
175 int read_descriptor = state->read_descriptor_;
176 state->fork_prepared_ = false;
177 lock.unlock();
178 reactor_.register_internal_descriptor(reactor::read_op,
179 read_descriptor, reactor_data_, new pipe_read_op);
180 }
181 break;
182 case boost::asio::io_service::fork_child:
183 if (state->fork_prepared_)
184 {
185 boost::asio::detail::signal_blocker blocker;
186 close_descriptors();
187 open_descriptors();
188 int read_descriptor = state->read_descriptor_;
189 state->fork_prepared_ = false;
190 lock.unlock();
191 reactor_.register_internal_descriptor(reactor::read_op,
192 read_descriptor, reactor_data_, new pipe_read_op);
193 }
194 break;
195 default:
196 break;
197 }
198#else // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
199 (void)fork_ev;
200#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
201}

217boost::system::error_code signal_set_service::add(
218 signal_set_service::implementation_type& impl,
219 int signal_number, boost::system::error_code& ec)
220{
221 // Check that the signal number is valid.
222 if (signal_number < 0 || signal_number > max_signal_number)
223 {
224 ec = boost::asio::error::invalid_argument;
225 return ec;
226 }
227
228 signal_state* state = get_signal_state();
229 static_mutex::scoped_lock lock(state->mutex_);
230
231 // Find the appropriate place to insert the registration.
232 registration** insertion_point = &impl.signals_;
233 registration* next = impl.signals_;
234 while (next && next->signal_number_ < signal_number)
235 {
236 insertion_point = &next->next_in_set_;
237 next = next->next_in_set_;
238 }
239
240 // Only do something if the signal is not already registered.
241 if (next == 0 || next->signal_number_ != signal_number)
242 {
243 registration* new_registration = new registration;
244
245#if defined(BOOST_ASIO_HAS_SIGNAL) || defined(BOOST_ASIO_HAS_SIGACTION)
246 // Register for the signal if we're the first.
247 if (state->registration_count_[signal_number] == 0)
248 {
249# if defined(BOOST_ASIO_HAS_SIGACTION)
250 using namespace std; // For memset.
251 struct sigaction sa;
252 memset(&sa, 0, sizeof(sa));
253 sa.sa_handler = asio_signal_handler;
254 sigfillset(&sa.sa_mask);
255 if (::sigaction(signal_number, &sa, 0) == -1)
256# else // defined(BOOST_ASIO_HAS_SIGACTION)
257 if (::signal(signal_number, asio_signal_handler) == SIG_ERR)
258# endif // defined(BOOST_ASIO_HAS_SIGACTION)
259 {
260# if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
261 ec = boost::asio::error::invalid_argument;
262# else // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
263 ec = boost::system::error_code(errno,
264 boost::asio::error::get_system_category());
265# endif // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
266 delete new_registration;
267 return ec;
268 }
269 }
270#endif // defined(BOOST_ASIO_HAS_SIGNAL) || defined(BOOST_ASIO_HAS_SIGACTION)
271
272 // Record the new registration in the set.
273 new_registration->signal_number_ = signal_number;
274 new_registration->queue_ = &impl.queue_;
275 new_registration->next_in_set_ = next;
276 *insertion_point = new_registration;
277
278 // Insert registration into the registration table.
279 new_registration->next_in_table_ = registrations_[signal_number];
280 if (registrations_[signal_number])
281 registrations_[signal_number]->prev_in_table_ = new_registration;
282 registrations_[signal_number] = new_registration;
283
284 ++state->registration_count_[signal_number];
285 }
286
287 ec = boost::system::error_code();
288 return ec;
289}

291boost::system::error_code signal_set_service::remove(
292 signal_set_service::implementation_type& impl,
293 int signal_number, boost::system::error_code& ec)
294{
295 // Check that the signal number is valid.
296 if (signal_number < 0 || signal_number > max_signal_number)
297 {
298 ec = boost::asio::error::invalid_argument;
299 return ec;
300 }
301
302 signal_state* state = get_signal_state();
303 static_mutex::scoped_lock lock(state->mutex_);
304
305 // Find the signal number in the list of registrations.
306 registration** deletion_point = &impl.signals_;
307 registration* reg = impl.signals_;
308 while (reg && reg->signal_number_ < signal_number)
309 {
310 deletion_point = &reg->next_in_set_;
311 reg = reg->next_in_set_;
312 }
313
314 if (reg != 0 && reg->signal_number_ == signal_number)
315 {
316#if defined(BOOST_ASIO_HAS_SIGNAL) || defined(BOOST_ASIO_HAS_SIGACTION)
317 // Set signal handler back to the default if we're the last.
318 if (state->registration_count_[signal_number] == 1)
319 {
320# if defined(BOOST_ASIO_HAS_SIGACTION)
321 using namespace std; // For memset.
322 struct sigaction sa;
323 memset(&sa, 0, sizeof(sa));
324 sa.sa_handler = SIG_DFL;
325 if (::sigaction(signal_number, &sa, 0) == -1)
326# else // defined(BOOST_ASIO_HAS_SIGACTION)
327 if (::signal(signal_number, SIG_DFL) == SIG_ERR)
328# endif // defined(BOOST_ASIO_HAS_SIGACTION)
329 {
330# if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
331 ec = boost::asio::error::invalid_argument;
332# else // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
333 ec = boost::system::error_code(errno,
334 boost::asio::error::get_system_category());
335# endif // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
336 return ec;
337 }
338 }
339#endif // defined(BOOST_ASIO_HAS_SIGNAL) || defined(BOOST_ASIO_HAS_SIGACTION)
340
341 // Remove the registration from the set.
342 *deletion_point = reg->next_in_set_;
343
344 // Remove the registration from the registration table.
345 if (registrations_[signal_number] == reg)
346 registrations_[signal_number] = reg->next_in_table_;
347 if (reg->prev_in_table_)
348 reg->prev_in_table_->next_in_table_ = reg->next_in_table_;
349 if (reg->next_in_table_)
350 reg->next_in_table_->prev_in_table_ = reg->prev_in_table_;
351
352 --state->registration_count_[signal_number];
353
354 delete reg;
355 }
356
357 ec = boost::system::error_code();
358 return ec;
359}

361boost::system::error_code signal_set_service::clear(
362 signal_set_service::implementation_type& impl,
363 boost::system::error_code& ec)
364{
365 signal_state* state = get_signal_state();
366 static_mutex::scoped_lock lock(state->mutex_);
367
368 while (registration* reg = impl.signals_)
369 {
370#if defined(BOOST_ASIO_HAS_SIGNAL) || defined(BOOST_ASIO_HAS_SIGACTION)
371 // Set signal handler back to the default if we're the last.
372 if (state->registration_count_[reg->signal_number_] == 1)
373 {
374# if defined(BOOST_ASIO_HAS_SIGACTION)
375 using namespace std; // For memset.
376 struct sigaction sa;
377 memset(&sa, 0, sizeof(sa));
378 sa.sa_handler = SIG_DFL;
379 if (::sigaction(reg->signal_number_, &sa, 0) == -1)
380# else // defined(BOOST_ASIO_HAS_SIGACTION)
381 if (::signal(reg->signal_number_, SIG_DFL) == SIG_ERR)
382# endif // defined(BOOST_ASIO_HAS_SIGACTION)
383 {
384# if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
385 ec = boost::asio::error::invalid_argument;
386# else // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
387 ec = boost::system::error_code(errno,
388 boost::asio::error::get_system_category());
389# endif // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
390 return ec;
391 }
392 }
393#endif // defined(BOOST_ASIO_HAS_SIGNAL) || defined(BOOST_ASIO_HAS_SIGACTION)
394
395 // Remove the registration from the registration table.
396 if (registrations_[reg->signal_number_] == reg)
397 registrations_[reg->signal_number_] = reg->next_in_table_;
398 if (reg->prev_in_table_)
399 reg->prev_in_table_->next_in_table_ = reg->next_in_table_;
400 if (reg->next_in_table_)
401 reg->next_in_table_->prev_in_table_ = reg->prev_in_table_;
402
403 --state->registration_count_[reg->signal_number_];
404
405 impl.signals_ = reg->next_in_set_;
406 delete reg;
407 }
408
409 ec = boost::system::error_code();
410 return ec;

413boost::system::error_code signal_set_service::cancel(
414 signal_set_service::implementation_type& impl,
415 boost::system::error_code& ec)
416{
417 BOOST_ASIO_HANDLER_OPERATION(("signal_set", &impl, "cancel"));
418
419 op_queue<operation> ops;
420 {
421 signal_state* state = get_signal_state();
422 static_mutex::scoped_lock lock(state->mutex_);
423
424 while (signal_op* op = impl.queue_.front())
425 {
426 op->ec_ = boost::asio::error::operation_aborted;
427 impl.queue_.pop();
428 ops.push(op);
429 }
430 }
431
432 io_service_.post_deferred_completions(ops);
433
434 ec = boost::system::error_code();
435 return ec;
436}

438void signal_set_service::deliver_signal(int signal_number)
439{
440 signal_state* state = get_signal_state();
441 static_mutex::scoped_lock lock(state->mutex_);
442
443 signal_set_service* service = state->service_list_;
444 while (service)
445 {
446 op_queue<operation> ops;
447
448 registration* reg = service->registrations_[signal_number];
449 while (reg)
450 {
451 if (reg->queue_->empty())
452 {
453 ++reg->undelivered_;
454 }
455 else
456 {
457 while (signal_op* op = reg->queue_->front())
458 {
459 op->signal_number_ = signal_number;
460 reg->queue_->pop();
461 ops.push(op);
462 }
463 }
464
465 reg = reg->next_in_table_;
466 }
467
468 service->io_service_.post_deferred_completions(ops);
469
470 service = service->next_;
471 }
472}

474void signal_set_service::add_service(signal_set_service* service)
475{
476 signal_state* state = get_signal_state();
477 static_mutex::scoped_lock lock(state->mutex_);
478
479#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
480 // If this is the first service to be created, open a new pipe.
481 if (state->service_list_ == 0)
482 open_descriptors();
483#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
484
485 // Insert service into linked list of all services.
486 service->next_ = state->service_list_;
487 service->prev_ = 0;
488 if (state->service_list_)
489 state->service_list_->prev_ = service;
490 state->service_list_ = service;
491
492#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
493 // Register for pipe readiness notifications.
494 int read_descriptor = state->read_descriptor_;
495 lock.unlock();
496 service->reactor_.register_internal_descriptor(reactor::read_op,
497 read_descriptor, service->reactor_data_, new pipe_read_op);
498#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
499}

501void signal_set_service::remove_service(signal_set_service* service)
502{
503 signal_state* state = get_signal_state();
504 static_mutex::scoped_lock lock(state->mutex_);
505
506 if (service->next_ || service->prev_ || state->service_list_ == service)
507 {
508#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
509 // Disable the pipe readiness notifications.
510 int read_descriptor = state->read_descriptor_;
511 lock.unlock();
512 service->reactor_.deregister_descriptor(
513 read_descriptor, service->reactor_data_, false);
514 lock.lock();
515#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
516
517 // Remove service from linked list of all services.
518 if (state->service_list_ == service)
519 state->service_list_ = service->next_;
520 if (service->prev_)
521 service->prev_->next_ = service->next_;
522 if (service->next_)
523 service->next_->prev_= service->prev_;
524 service->next_ = 0;
525 service->prev_ = 0;
526
527#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
528 // If this is the last service to be removed, close the pipe.
529 if (state->service_list_ == 0)
530 close_descriptors();
531#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
532 }

535void signal_set_service::open_descriptors()
536{
537#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
538 signal_state* state = get_signal_state();
539
540 int pipe_fds[2];
541 if (::pipe(pipe_fds) == 0)
542 {
543 state->read_descriptor_ = pipe_fds[0];
544 ::fcntl(state->read_descriptor_, F_SETFL, O_NONBLOCK);
545
546 state->write_descriptor_ = pipe_fds[1];
547 ::fcntl(state->write_descriptor_, F_SETFL, O_NONBLOCK);
548
549#if defined(FD_CLOEXEC)
550 ::fcntl(state->read_descriptor_, F_SETFD, FD_CLOEXEC);
551 ::fcntl(state->write_descriptor_, F_SETFD, FD_CLOEXEC);
552#endif // defined(FD_CLOEXEC)
553 }
554 else
555 {
556 boost::system::error_code ec(errno,
557 boost::asio::error::get_system_category());
558 boost::asio::detail::throw_error(ec, "signal_set_service pipe");
559 }
560#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
561}

563void signal_set_service::close_descriptors()
564{
565#if !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
566 signal_state* state = get_signal_state();
567
568 if (state->read_descriptor_ != -1)
569 ::close(state->read_descriptor_);
570 state->read_descriptor_ = -1;
571
572 if (state->write_descriptor_ != -1)
573 ::close(state->write_descriptor_);
574 state->write_descriptor_ = -1;
575#endif // !defined(BOOST_ASIO_WINDOWS) && !defined(__CYGWIN__)
576}

578void signal_set_service::start_wait_op(
579 signal_set_service::implementation_type& impl, signal_op* op)
580{
581 io_service_.work_started();
582
583 signal_state* state = get_signal_state();
584 static_mutex::scoped_lock lock(state->mutex_);
585
586 registration* reg = impl.signals_;
587 while (reg)
588 {
589 if (reg->undelivered_ > 0)
590 {
591 --reg->undelivered_;
592 io_service_.post_deferred_completion(op);
593 return;
594 }
595
596 reg = reg->next_in_set_;
597 }
598
599 impl.queue_.push(op);
600}

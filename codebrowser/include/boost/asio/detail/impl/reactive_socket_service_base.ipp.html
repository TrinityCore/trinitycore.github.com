41void reactive_socket_service_base::construct(

48void reactive_socket_service_base::base_move_construct(
49 reactive_socket_service_base::base_implementation_type& impl,
50 reactive_socket_service_base::base_implementation_type& other_impl)
51{
52 impl.socket_ = other_impl.socket_;
53 other_impl.socket_ = invalid_socket;
54
55 impl.state_ = other_impl.state_;
56 other_impl.state_ = 0;
57
58 reactor_.move_descriptor(impl.socket_,
59 impl.reactor_data_, other_impl.reactor_data_);
60}

62void reactive_socket_service_base::base_move_assign(
63 reactive_socket_service_base::base_implementation_type& impl,
64 reactive_socket_service_base& other_service,
65 reactive_socket_service_base::base_implementation_type& other_impl)
66{
67 destroy(impl);
68
69 impl.socket_ = other_impl.socket_;
70 other_impl.socket_ = invalid_socket;
71
72 impl.state_ = other_impl.state_;
73 other_impl.state_ = 0;
74
75 other_service.reactor_.move_descriptor(impl.socket_,
76 impl.reactor_data_, other_impl.reactor_data_);
77}

79void reactive_socket_service_base::destroy(
80 reactive_socket_service_base::base_implementation_type& impl)
81{
82 if (impl.socket_ != invalid_socket)
83 {
84 BOOST_ASIO_HANDLER_OPERATION(("socket", &impl, "close"));
85
86 reactor_.deregister_descriptor(impl.socket_, impl.reactor_data_,
87 (impl.state_ & socket_ops::possible_dup) == 0);
88
89 boost::system::error_code ignored_ec;
90 socket_ops::close(impl.socket_, impl.state_, true, ignored_ec);
91 }
92}

94boost::system::error_code reactive_socket_service_base::close(
95 reactive_socket_service_base::base_implementation_type& impl,
96 boost::system::error_code& ec)
97{
98 if (is_open(impl))
99 {
100 BOOST_ASIO_HANDLER_OPERATION(("socket", &impl, "close"));
101
102 reactor_.deregister_descriptor(impl.socket_, impl.reactor_data_,
103 (impl.state_ & socket_ops::possible_dup) == 0);
104 }
105
106 socket_ops::close(impl.socket_, impl.state_, false, ec);
107
108 // The descriptor is closed by the OS even if close() returns an error.
109 //
110 // (Actually, POSIX says the state of the descriptor is unspecified. On
111 // Linux the descriptor is apparently closed anyway; e.g. see
112 // http://lkml.org/lkml/2005/9/10/129
113 // We'll just have to assume that other OSes follow the same behaviour. The
114 // known exception is when Windows's closesocket() function fails with
115 // WSAEWOULDBLOCK, but this case is handled inside socket_ops::close().
116 construct(impl);
117
118 return ec;
119}

121boost::system::error_code reactive_socket_service_base::cancel(
122 reactive_socket_service_base::base_implementation_type& impl,
123 boost::system::error_code& ec)
124{
125 if (!is_open(impl))
126 {
127 ec = boost::asio::error::bad_descriptor;
128 return ec;
129 }
130
131 BOOST_ASIO_HANDLER_OPERATION(("socket", &impl, "cancel"));
132
133 reactor_.cancel_ops(impl.socket_, impl.reactor_data_);
134 ec = boost::system::error_code();
135 return ec;
136}

138boost::system::error_code reactive_socket_service_base::do_open(
139 reactive_socket_service_base::base_implementation_type& impl,
140 int af, int type, int protocol, boost::system::error_code& ec)
141{
142 if (is_open(impl))
143 {
144 ec = boost::asio::error::already_open;
145 return ec;
146 }
147
148 socket_holder sock(socket_ops::socket(af, type, protocol, ec));
149 if (sock.get() == invalid_socket)
150 return ec;
151
152 if (int err = reactor_.register_descriptor(sock.get(), impl.reactor_data_))
153 {
154 ec = boost::system::error_code(err,
155 boost::asio::error::get_system_category());
156 return ec;
157 }
158
159 impl.socket_ = sock.release();
160 switch (type)
161 {
162 case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
163 case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
164 default: impl.state_ = 0; break;
165 }
166 ec = boost::system::error_code();
167 return ec;
168}

170boost::system::error_code reactive_socket_service_base::do_assign(
171 reactive_socket_service_base::base_implementation_type& impl, int type,
172 const reactive_socket_service_base::native_handle_type& native_socket,
173 boost::system::error_code& ec)
174{
175 if (is_open(impl))
176 {
177 ec = boost::asio::error::already_open;
178 return ec;
179 }
180
181 if (int err = reactor_.register_descriptor(
182 native_socket, impl.reactor_data_))
183 {
184 ec = boost::system::error_code(err,
185 boost::asio::error::get_system_category());
186 return ec;
187 }
188
189 impl.socket_ = native_socket;
190 switch (type)
191 {
192 case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
193 case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
194 default: impl.state_ = 0; break;
195 }
196 impl.state_ |= socket_ops::possible_dup;
197 ec = boost::system::error_code();
198 return ec;
199}

201void reactive_socket_service_base::start_op(
202 reactive_socket_service_base::base_implementation_type& impl,
203 int op_type, reactor_op* op, bool is_continuation,
204 bool is_non_blocking, bool noop)
205{
206 if (!noop)
207 {
208 if ((impl.state_ & socket_ops::non_blocking)
209 || socket_ops::set_internal_non_blocking(
210 impl.socket_, impl.state_, true, op->ec_))
211 {
212 reactor_.start_op(op_type, impl.socket_,
213 impl.reactor_data_, op, is_continuation, is_non_blocking);
214 return;
215 }
216 }
217
218 reactor_.post_immediate_completion(op, is_continuation);
219}
220
221void reactive_socket_service_base::start_accept_op(
222 reactive_socket_service_base::base_implementation_type& impl,
223 reactor_op* op, bool is_continuation, bool peer_is_open)
224{
225 if (!peer_is_open)
226 start_op(impl, reactor::read_op, op, true, is_continuation, false);
227 else
228 {
229 op->ec_ = boost::asio::error::already_open;
230 reactor_.post_immediate_completion(op, is_continuation);
231 }
232}
233
234void reactive_socket_service_base::start_connect_op(
235 reactive_socket_service_base::base_implementation_type& impl,
236 reactor_op* op, bool is_continuation,
237 const socket_addr_type* addr, size_t addrlen)
238{
239 if ((impl.state_ & socket_ops::non_blocking)
240 || socket_ops::set_internal_non_blocking(
241 impl.socket_, impl.state_, true, op->ec_))
242 {
243 if (socket_ops::connect(impl.socket_, addr, addrlen, op->ec_) != 0)
244 {
245 if (op->ec_ == boost::asio::error::in_progress
246 || op->ec_ == boost::asio::error::would_block)
247 {
248 op->ec_ = boost::system::error_code();
249 reactor_.start_op(reactor::connect_op, impl.socket_,
250 impl.reactor_data_, op, is_continuation, false);
251 return;
252 }
253 }
254 }
255
256 reactor_.post_immediate_completion(op, is_continuation);
257}

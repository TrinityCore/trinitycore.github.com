29class buffer_sequence_adapter_base
30{
31protected:
32#if defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
33 typedef WSABUF native_buffer_type;
34
35 static void init_native_buffer(WSABUF& buf,
36 const boost::asio::mutable_buffer& buffer)
37 {
38 buf.buf = boost::asio::buffer_cast<char*>(buffer);
39 buf.len = static_cast<ULONG>(boost::asio::buffer_size(buffer));
40 }
41
42 static void init_native_buffer(WSABUF& buf,
43 const boost::asio::const_buffer& buffer)
44 {
45 buf.buf = const_cast<char*>(boost::asio::buffer_cast<const char*>(buffer));
46 buf.len = static_cast<ULONG>(boost::asio::buffer_size(buffer));
47 }
48#else // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
49 typedef iovec native_buffer_type;
50
51 static void init_iov_base(void*& base, void* addr)
52 {
53 base = addr;
54 }
55
56 template <typename T>
57 static void init_iov_base(T& base, void* addr)
58 {
59 base = static_cast<T>(addr);
60 }
61
62 static void init_native_buffer(iovec& iov,
63 const boost::asio::mutable_buffer& buffer)
64 {
65 init_iov_base(iov.iov_base, boost::asio::buffer_cast<void*>(buffer));
66 iov.iov_len = boost::asio::buffer_size(buffer);
67 }
68
69 static void init_native_buffer(iovec& iov,
70 const boost::asio::const_buffer& buffer)
71 {
72 init_iov_base(iov.iov_base, const_cast<void*>(
73 boost::asio::buffer_cast<const void*>(buffer)));
74 iov.iov_len = boost::asio::buffer_size(buffer);
75 }
76#endif // defined(BOOST_ASIO_WINDOWS) || defined(__CYGWIN__)
77};

79// Helper class to translate buffers into the native buffer representation.
80template <typename Buffer, typename Buffers>
81class buffer_sequence_adapter
82 : buffer_sequence_adapter_base
83{
84public:
85 explicit buffer_sequence_adapter(const Buffers& buffer_sequence)
86 : count_(0), total_buffer_size_(0)
87 {
88 typename Buffers::const_iterator iter = buffer_sequence.begin();
89 typename Buffers::const_iterator end = buffer_sequence.end();
90 for (; iter != end && count_ < max_buffers; ++iter, ++count_)
91 {
92 Buffer buffer(*iter);
93 init_native_buffer(buffers_[count_], buffer);
94 total_buffer_size_ += boost::asio::buffer_size(buffer);
95 }
96 }
97
98 native_buffer_type* buffers()
99 {
100 return buffers_;
101 }
102
103 std::size_t count() const
104 {
105 return count_;
106 }
107
108 bool all_empty() const
109 {
110 return total_buffer_size_ == 0;
111 }
112
113 static bool all_empty(const Buffers& buffer_sequence)
114 {
115 typename Buffers::const_iterator iter = buffer_sequence.begin();
116 typename Buffers::const_iterator end = buffer_sequence.end();
117 std::size_t i = 0;
118 for (; iter != end && i < max_buffers; ++iter, ++i)
119 if (boost::asio::buffer_size(Buffer(*iter)) > 0)
120 return false;
121 return true;
122 }
123
124 static void validate(const Buffers& buffer_sequence)
125 {
126 typename Buffers::const_iterator iter = buffer_sequence.begin();
127 typename Buffers::const_iterator end = buffer_sequence.end();
128 for (; iter != end; ++iter)
129 {
130 Buffer buffer(*iter);
131 boost::asio::buffer_cast<const void*>(buffer);
132 }
133 }
134
135 static Buffer first(const Buffers& buffer_sequence)
136 {
137 typename Buffers::const_iterator iter = buffer_sequence.begin();
138 typename Buffers::const_iterator end = buffer_sequence.end();
139 for (; iter != end; ++iter)
140 {
141 Buffer buffer(*iter);
142 if (boost::asio::buffer_size(buffer) != 0)
143 return buffer;
144 }
145 return Buffer();
146 }
147
148private:
149 // The maximum number of buffers to support in a single operation.
150 enum { max_buffers = 64 < max_iov_len ? 64 : max_iov_len };
151
152 native_buffer_type buffers_[max_buffers];
153 std::size_t count_;
154 std::size_t total_buffer_size_;
155};

251template <typename Buffer, typename Elem>
252class buffer_sequence_adapter<Buffer, boost::array<Elem, 2> >
253 : buffer_sequence_adapter_base
254{
255public:
256 explicit buffer_sequence_adapter(
257 const boost::array<Elem, 2>& buffer_sequence)
258 {
259 init_native_buffer(buffers_[0], Buffer(buffer_sequence[0]));
260 init_native_buffer(buffers_[1], Buffer(buffer_sequence[1]));
261 total_buffer_size_ = boost::asio::buffer_size(buffer_sequence[0])
262 + boost::asio::buffer_size(buffer_sequence[1]);
263 }
264
265 native_buffer_type* buffers()
266 {
267 return buffers_;
268 }
269
270 std::size_t count() const
271 {
272 return 2;
273 }
274
275 bool all_empty() const
276 {
277 return total_buffer_size_ == 0;
278 }
279
280 static bool all_empty(const boost::array<Elem, 2>& buffer_sequence)
281 {
282 return boost::asio::buffer_size(buffer_sequence[0]) == 0
283 && boost::asio::buffer_size(buffer_sequence[1]) == 0;
284 }
285
286 static void validate(const boost::array<Elem, 2>& buffer_sequence)
287 {
288 boost::asio::buffer_cast<const void*>(buffer_sequence[0]);
289 boost::asio::buffer_cast<const void*>(buffer_sequence[1]);
290 }
291
292 static Buffer first(const boost::array<Elem, 2>& buffer_sequence)
293 {
294 return Buffer(boost::asio::buffer_size(buffer_sequence[0]) != 0
295 ? buffer_sequence[0] : buffer_sequence[1]);
296 }
297
298private:
299 native_buffer_type buffers_[2];
300 std::size_t total_buffer_size_;
301};

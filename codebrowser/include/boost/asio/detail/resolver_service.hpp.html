28namespace boost {

30namespace detail {
31
32template <typename Protocol>
33class resolver_service : public resolver_service_base
34{
35public:
36 // The implementation type of the resolver. A cancellation token is used to
37 // indicate to the background thread that the operation has been cancelled.
38 typedef socket_ops::shared_cancel_token_type implementation_type;
39
40 // The endpoint type.
41 typedef typename Protocol::endpoint endpoint_type;
42
43 // The query type.
44 typedef boost::asio::ip::basic_resolver_query<Protocol> query_type;
45
46 // The iterator type.
47 typedef boost::asio::ip::basic_resolver_iterator<Protocol> iterator_type;
48
49 // Constructor.
50 resolver_service(boost::asio::io_service& io_service)
51 : resolver_service_base(io_service)
52 {
53 }
54
55 // Resolve a query to a list of entries.
56 iterator_type resolve(implementation_type&, const query_type& query,
57 boost::system::error_code& ec)
58 {
59 boost::asio::detail::addrinfo_type* address_info = 0;
60
61 socket_ops::getaddrinfo(query.host_name().c_str(),
62 query.service_name().c_str(), query.hints(), &address_info, ec);
63 auto_addrinfo auto_address_info(address_info);
64
65 return ec ? iterator_type() : iterator_type::create(
66 address_info, query.host_name(), query.service_name());
67 }
68
69 // Asynchronously resolve a query to a list of entries.
70 template <typename Handler>
71 void async_resolve(implementation_type& impl,
72 const query_type& query, Handler& handler)
73 {
74 // Allocate and construct an operation to wrap the handler.
75 typedef resolve_op<Protocol, Handler> op;
76 typename op::ptr p = { boost::asio::detail::addressof(handler),
77 boost_asio_handler_alloc_helpers::allocate(
78 sizeof(op), handler), 0 };
79 p.p = new (p.v) op(impl, query, io_service_impl_, handler);
80
81 BOOST_ASIO_HANDLER_CREATION((p.p, "resolver", &impl, "async_resolve"));
82
83 start_resolve_op(p.p);
84 p.v = p.p = 0;
85 }
86
87 // Resolve an endpoint to a list of entries.
88 iterator_type resolve(implementation_type&,
89 const endpoint_type& endpoint, boost::system::error_code& ec)
90 {
91 char host_name[NI_MAXHOST];
92 char service_name[NI_MAXSERV];
93 socket_ops::sync_getnameinfo(endpoint.data(), endpoint.size(),
94 host_name, NI_MAXHOST, service_name, NI_MAXSERV,
95 endpoint.protocol().type(), ec);
96
97 return ec ? iterator_type() : iterator_type::create(
98 endpoint, host_name, service_name);
99 }
100
101 // Asynchronously resolve an endpoint to a list of entries.
102 template <typename Handler>
103 void async_resolve(implementation_type& impl,
104 const endpoint_type& endpoint, Handler& handler)
105 {
106 // Allocate and construct an operation to wrap the handler.
107 typedef resolve_endpoint_op<Protocol, Handler> op;
108 typename op::ptr p = { boost::asio::detail::addressof(handler),
109 boost_asio_handler_alloc_helpers::allocate(
110 sizeof(op), handler), 0 };
111 p.p = new (p.v) op(impl, endpoint, io_service_impl_, handler);
112
113 BOOST_ASIO_HANDLER_CREATION((p.p, "resolver", &impl, "async_resolve"));
114
115 start_resolve_op(p.p);
116 p.v = p.p = 0;
117 }
118};

35/// Adds buffering to the read-related operations of a stream.
36/**
37 * The buffered_read_stream class template can be used to add buffering to the
38 * synchronous and asynchronous read operations of a stream.
39 *
40 * @par Thread Safety
41 * @e Distinct @e objects: Safe.@n
42 * @e Shared @e objects: Unsafe.
43 *
44 * @par Concepts:
45 * AsyncReadStream, AsyncWriteStream, Stream, Sync_Read_Stream, SyncWriteStream.
46 */
47template <typename Stream>
48class buffered_read_stream
49 : private noncopyable
50{
51public:
52 /// The type of the next layer.
53 typedef typename remove_reference<Stream>::type next_layer_type;
54
55 /// The type of the lowest layer.
56 typedef typename next_layer_type::lowest_layer_type lowest_layer_type;
57
58#if defined(GENERATING_DOCUMENTATION)
59 /// The default buffer size.
60 static const std::size_t default_buffer_size = implementation_defined;
61#else
62 BOOST_ASIO_STATIC_CONSTANT(std::size_t, default_buffer_size = 1024);
63#endif
64
65 /// Construct, passing the specified argument to initialise the next layer.
66 template <typename Arg>
67 explicit buffered_read_stream(Arg& a)
68 : next_layer_(a),
69 storage_(default_buffer_size)
70 {
71 }
72
73 /// Construct, passing the specified argument to initialise the next layer.
74 template <typename Arg>
75 buffered_read_stream(Arg& a, std::size_t buffer_size)
76 : next_layer_(a),
77 storage_(buffer_size)
78 {
79 }
80
81 /// Get a reference to the next layer.
82 next_layer_type& next_layer()
83 {
84 return next_layer_;
85 }
86
87 /// Get a reference to the lowest layer.
88 lowest_layer_type& lowest_layer()
89 {
90 return next_layer_.lowest_layer();
91 }
92
93 /// Get a const reference to the lowest layer.
94 const lowest_layer_type& lowest_layer() const
95 {
96 return next_layer_.lowest_layer();
97 }
98
99 /// Get the io_service associated with the object.
100 boost::asio::io_service& get_io_service()
101 {
102 return next_layer_.get_io_service();
103 }
104
105 /// Close the stream.
106 void close()
107 {
108 next_layer_.close();
109 }
110
111 /// Close the stream.
112 boost::system::error_code close(boost::system::error_code& ec)
113 {
114 return next_layer_.close(ec);
115 }

117 /// Write the given data to the stream. Returns the number of bytes written.
118 /// Throws an exception on failure.
119 template <typename ConstBufferSequence>
120 std::size_t write_some(const ConstBufferSequence& buffers)
121 {
122 return next_layer_.write_some(buffers);
123 }
124
125 /// Write the given data to the stream. Returns the number of bytes written,
126 /// or 0 if an error occurred.
127 template <typename ConstBufferSequence>
128 std::size_t write_some(const ConstBufferSequence& buffers,
129 boost::system::error_code& ec)
130 {
131 return next_layer_.write_some(buffers, ec);
132 }
133
134 /// Start an asynchronous write. The data being written must be valid for the
135 /// lifetime of the asynchronous operation.
136 template <typename ConstBufferSequence, typename WriteHandler>
137 void async_write_some(const ConstBufferSequence& buffers,
138 WriteHandler handler)
139 {
140 next_layer_.async_write_some(buffers, handler);
141 }
142
143 /// Fill the buffer with some data. Returns the number of bytes placed in the
144 /// buffer as a result of the operation. Throws an exception on failure.
145 std::size_t fill()
146 {
147 detail::buffer_resize_guard<detail::buffered_stream_storage>
148 resize_guard(storage_);
149 std::size_t previous_size = storage_.size();
150 storage_.resize(storage_.capacity());
151 storage_.resize(previous_size + next_layer_.read_some(buffer(
152 storage_.data() + previous_size,
153 storage_.size() - previous_size)));
154 resize_guard.commit();
155 return storage_.size() - previous_size;
156 }
157
158 /// Fill the buffer with some data. Returns the number of bytes placed in the
159 /// buffer as a result of the operation, or 0 if an error occurred.
160 std::size_t fill(boost::system::error_code& ec)
161 {
162 detail::buffer_resize_guard<detail::buffered_stream_storage>
163 resize_guard(storage_);
164 std::size_t previous_size = storage_.size();
165 storage_.resize(storage_.capacity());
166 storage_.resize(previous_size + next_layer_.read_some(buffer(
167 storage_.data() + previous_size,
168 storage_.size() - previous_size),
169 ec));
170 resize_guard.commit();
171 return storage_.size() - previous_size;
172 }
173
174 template <typename ReadHandler>
175 class fill_handler
176 {
177 public:
178 fill_handler(boost::asio::io_service& io_service,
179 detail::buffered_stream_storage& storage,
180 std::size_t previous_size, ReadHandler handler)
181 : io_service_(io_service),
182 storage_(storage),
183 previous_size_(previous_size),
184 handler_(handler)
185 {
186 }
187
188 void operator()(const boost::system::error_code& ec,
189 std::size_t bytes_transferred)
190 {
191 storage_.resize(previous_size_ + bytes_transferred);
192 io_service_.dispatch(detail::bind_handler(
193 handler_, ec, bytes_transferred));
194 }
195
196 private:
197 boost::asio::io_service& io_service_;
198 detail::buffered_stream_storage& storage_;
199 std::size_t previous_size_;
200 ReadHandler handler_;
201 };
202
203 /// Start an asynchronous fill.
204 template <typename ReadHandler>
205 void async_fill(ReadHandler handler)
206 {
207 std::size_t previous_size = storage_.size();
208 storage_.resize(storage_.capacity());
209 next_layer_.async_read_some(
210 buffer(
211 storage_.data() + previous_size,
212 storage_.size() - previous_size),
213 fill_handler<ReadHandler>(get_io_service(),
214 storage_, previous_size, handler));
215 }

217 /// Read some data from the stream. Returns the number of bytes read. Throws
218 /// an exception on failure.
219 template <typename MutableBufferSequence>
220 std::size_t read_some(const MutableBufferSequence& buffers)
221 {
222 if (boost::asio::buffer_size(buffers) == 0)
223 return 0;
224
225 if (storage_.empty())
226 fill();
227
228 return copy(buffers);
229 }
230
231 /// Read some data from the stream. Returns the number of bytes read or 0 if
232 /// an error occurred.
233 template <typename MutableBufferSequence>
234 std::size_t read_some(const MutableBufferSequence& buffers,
235 boost::system::error_code& ec)
236 {
237 ec = boost::system::error_code();
238
239 if (boost::asio::buffer_size(buffers) == 0)
240 return 0;
241
242 if (storage_.empty() && !fill(ec))
243 return 0;
244
245 return copy(buffers);
246 }
247
248 template <typename MutableBufferSequence, typename ReadHandler>
249 class read_some_handler
250 {
251 public:
252 read_some_handler(boost::asio::io_service& io_service,
253 detail::buffered_stream_storage& storage,
254 const MutableBufferSequence& buffers, ReadHandler handler)
255 : io_service_(io_service),
256 storage_(storage),
257 buffers_(buffers),
258 handler_(handler)
259 {
260 }
261
262 void operator()(const boost::system::error_code& ec, std::size_t)
263 {
264 if (ec || storage_.empty())
265 {
266 std::size_t length = 0;
267 io_service_.dispatch(detail::bind_handler(handler_, ec, length));
268 }
269 else
270 {
271 std::size_t bytes_copied = boost::asio::buffer_copy(
272 buffers_, storage_.data(), storage_.size());
273 storage_.consume(bytes_copied);
274 io_service_.dispatch(detail::bind_handler(handler_, ec, bytes_copied));
275 }
276 }
277
278 private:
279 boost::asio::io_service& io_service_;
280 detail::buffered_stream_storage& storage_;
281 MutableBufferSequence buffers_;
282 ReadHandler handler_;
283 };
284
285 /// Start an asynchronous read. The buffer into which the data will be read
286 /// must be valid for the lifetime of the asynchronous operation.
287 template <typename MutableBufferSequence, typename ReadHandler>
288 void async_read_some(const MutableBufferSequence& buffers,
289 ReadHandler handler)
290 {
291 if (boost::asio::buffer_size(buffers) == 0)
292 {
293 get_io_service().post(detail::bind_handler(
294 handler, boost::system::error_code(), 0));
295 }
296 else if (storage_.empty())
297 {
298 async_fill(read_some_handler<MutableBufferSequence, ReadHandler>(
299 get_io_service(), storage_, buffers, handler));
300 }
301 else
302 {
303 std::size_t length = copy(buffers);
304 get_io_service().post(detail::bind_handler(
305 handler, boost::system::error_code(), length));
306 }
307 }

309 /// Peek at the incoming data on the stream. Returns the number of bytes read.
310 /// Throws an exception on failure.
311 template <typename MutableBufferSequence>
312 std::size_t peek(const MutableBufferSequence& buffers)
313 {
314 if (storage_.empty())
315 fill();
316 return peek_copy(buffers);
317 }
318
319 /// Peek at the incoming data on the stream. Returns the number of bytes read,
320 /// or 0 if an error occurred.
321 template <typename MutableBufferSequence>
322 std::size_t peek(const MutableBufferSequence& buffers,
323 boost::system::error_code& ec)
324 {
325 ec = boost::system::error_code();
326 if (storage_.empty() && !fill(ec))
327 return 0;
328 return peek_copy(buffers);
329 }

331 /// Determine the amount of data that may be read without blocking.
332 std::size_t in_avail()
333 {
334 return storage_.size();
335 }
336
337 /// Determine the amount of data that may be read without blocking.
338 std::size_t in_avail(boost::system::error_code& ec)
339 {
340 ec = boost::system::error_code();
341 return storage_.size();
342 }
343
344private:
345 /// Copy data out of the internal buffer to the specified target buffer.
346 /// Returns the number of bytes copied.
347 template <typename MutableBufferSequence>
348 std::size_t copy(const MutableBufferSequence& buffers)
349 {
350 std::size_t bytes_copied = boost::asio::buffer_copy(
351 buffers, storage_.data(), storage_.size());
352 storage_.consume(bytes_copied);
353 return bytes_copied;
354 }
355
356 /// Copy data from the internal buffer to the specified target buffer, without
357 /// removing the data from the internal buffer. Returns the number of bytes
358 /// copied.
359 template <typename MutableBufferSequence>
360 std::size_t peek_copy(const MutableBufferSequence& buffers)
361 {
362 return boost::asio::buffer_copy(buffers, storage_.data(), storage_.size());
363 }

365 /// The next layer.
366 Stream next_layer_;
367
368 // The data in the buffer.
369 detail::buffered_stream_storage storage_;
370};

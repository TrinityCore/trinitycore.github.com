91// A separate base class is used to ensure that the streambuf is initialised
92// prior to the basic_socket_iostream's basic_iostream base class.
93template <typename Protocol, typename StreamSocketService,
94 typename Time, typename TimeTraits, typename TimerService>
95class socket_iostream_base
96{
97protected:
98 basic_socket_streambuf<Protocol, StreamSocketService,
99 Time, TimeTraits, TimerService> streambuf_;
100};
101
102}
103
104/// Iostream interface for a socket.
105template <typename Protocol,
106 typename StreamSocketService = stream_socket_service<Protocol>,
107#if defined(BOOST_ASIO_HAS_BOOST_DATE_TIME) \
108 || defined(GENERATING_DOCUMENTATION)
109 typename Time = boost::posix_time::ptime,
110 typename TimeTraits = boost::asio::time_traits<Time>,
111 typename TimerService = deadline_timer_service<Time, TimeTraits> >
112#else
113 typename Time = steady_timer::clock_type,
114 typename TimeTraits = steady_timer::traits_type,
115 typename TimerService = steady_timer::service_type>
116#endif
117class basic_socket_iostream
118 : private detail::socket_iostream_base<Protocol,
119 StreamSocketService, Time, TimeTraits, TimerService>,
120 public std::basic_iostream<char>
121{
122private:
123 // These typedefs are intended keep this class's implementation independent
124 // of whether it's using Boost.DateTime, Boost.Chrono or std::chrono.
125#if defined(BOOST_ASIO_HAS_BOOST_DATE_TIME)
126 typedef TimeTraits traits_helper;
127#else
128 typedef detail::chrono_time_traits<Time, TimeTraits> traits_helper;
129#endif
130
131public:
132 /// The endpoint type.
133 typedef typename Protocol::endpoint endpoint_type;
134
135#if defined(GENERATING_DOCUMENTATION)
136 /// The time type.
137 typedef typename TimeTraits::time_type time_type;
138
139 /// The duration type.
140 typedef typename TimeTraits::duration_type duration_type;
141#else
142 typedef typename traits_helper::time_type time_type;
143 typedef typename traits_helper::duration_type duration_type;
144#endif
145
146 /// Construct a basic_socket_iostream without establishing a connection.
147 basic_socket_iostream()
148 : std::basic_iostream<char>(
149 &this->detail::socket_iostream_base<
150 Protocol, StreamSocketService, Time,
151 TimeTraits, TimerService>::streambuf_)
152 {
153 this->setf(std::ios_base::unitbuf);
154 }
155
156#if defined(GENERATING_DOCUMENTATION)
157 /// Establish a connection to an endpoint corresponding to a resolver query.
158 /**
159 * This constructor automatically establishes a connection based on the
160 * supplied resolver query parameters. The arguments are used to construct
161 * a resolver query object.
162 */
163 template <typename T1, ..., typename TN>
164 explicit basic_socket_iostream(T1 t1, ..., TN tn);
165#elif defined(BOOST_ASIO_HAS_VARIADIC_TEMPLATES)
166 template <typename... T>
167 explicit basic_socket_iostream(T... x)
168 : std::basic_iostream<char>(
169 &this->detail::socket_iostream_base<
170 Protocol, StreamSocketService, Time,
171 TimeTraits, TimerService>::streambuf_)
172 {
173 this->setf(std::ios_base::unitbuf);
174 if (rdbuf()->connect(x...) == 0)
175 this->setstate(std::ios_base::failbit);
176 }
177#else
178 BOOST_PP_REPEAT_FROM_TO(
179 1, BOOST_PP_INC(BOOST_ASIO_SOCKET_IOSTREAM_MAX_ARITY),
180 BOOST_ASIO_PRIVATE_CTR_DEF, _ )
181#endif
182
183#if defined(GENERATING_DOCUMENTATION)
184 /// Establish a connection to an endpoint corresponding to a resolver query.
185 /**
186 * This function automatically establishes a connection based on the supplied
187 * resolver query parameters. The arguments are used to construct a resolver
188 * query object.
189 */
190 template <typename T1, ..., typename TN>
191 void connect(T1 t1, ..., TN tn);
192#elif defined(BOOST_ASIO_HAS_VARIADIC_TEMPLATES)
193 template <typename... T>
194 void connect(T... x)
195 {
196 if (rdbuf()->connect(x...) == 0)
197 this->setstate(std::ios_base::failbit);
198 }
199#else
200 BOOST_PP_REPEAT_FROM_TO(
201 1, BOOST_PP_INC(BOOST_ASIO_SOCKET_IOSTREAM_MAX_ARITY),
202 BOOST_ASIO_PRIVATE_CONNECT_DEF, _ )
203#endif
204
205 /// Close the connection.
206 void close()
207 {
208 if (rdbuf()->close() == 0)
209 this->setstate(std::ios_base::failbit);
210 }
211
212 /// Return a pointer to the underlying streambuf.
213 basic_socket_streambuf<Protocol, StreamSocketService,
214 Time, TimeTraits, TimerService>* rdbuf() const
215 {
216 return const_cast<basic_socket_streambuf<Protocol, StreamSocketService,
217 Time, TimeTraits, TimerService>*>(
218 &this->detail::socket_iostream_base<
219 Protocol, StreamSocketService, Time,
220 TimeTraits, TimerService>::streambuf_);
221 }
222
223 /// Get the last error associated with the stream.
224 /**
225 * @return An \c error_code corresponding to the last error from the stream.
226 *
227 * @par Example
228 * To print the error associated with a failure to establish a connection:
229 * @code tcp::iostream s("www.boost.org", "http");
230 * if (!s)
231 * {
232 * std::cout << "Error: " << s.error().message() << std::endl;
233 * } @endcode
234 */
235 const boost::system::error_code& error() const
236 {
237 return rdbuf()->puberror();
238 }
239
240 /// Get the stream's expiry time as an absolute time.
241 /**
242 * @return An absolute time value representing the stream's expiry time.
243 */
244 time_type expires_at() const
245 {
246 return rdbuf()->expires_at();
247 }
248
249 /// Set the stream's expiry time as an absolute time.
250 /**
251 * This function sets the expiry time associated with the stream. Stream
252 * operations performed after this time (where the operations cannot be
253 * completed using the internal buffers) will fail with the error
254 * boost::asio::error::operation_aborted.
255 *
256 * @param expiry_time The expiry time to be used for the stream.
257 */
258 void expires_at(const time_type& expiry_time)
259 {
260 rdbuf()->expires_at(expiry_time);
261 }
262
263 /// Get the timer's expiry time relative to now.
264 /**
265 * @return A relative time value representing the stream's expiry time.
266 */
267 duration_type expires_from_now() const
268 {
269 return rdbuf()->expires_from_now();
270 }
271
272 /// Set the stream's expiry time relative to now.
273 /**
274 * This function sets the expiry time associated with the stream. Stream
275 * operations performed after this time (where the operations cannot be
276 * completed using the internal buffers) will fail with the error
277 * boost::asio::error::operation_aborted.
278 *
279 * @param expiry_time The expiry time to be used for the timer.
280 */
281 void expires_from_now(const duration_type& expiry_time)
282 {
283 rdbuf()->expires_from_now(expiry_time);
284 }
285};

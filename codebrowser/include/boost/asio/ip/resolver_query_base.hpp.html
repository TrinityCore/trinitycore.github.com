27/// The resolver_query_base class is used as a base for the
28/// basic_resolver_query class templates to provide a common place to define
29/// the flag constants.
30class resolver_query_base
31{
32public:
33#if defined(GENERATING_DOCUMENTATION)
34 /// A bitmask type (C++ Std [lib.bitmask.types]).
35 typedef unspecified flags;
36
37 /// Determine the canonical name of the host specified in the query.
38 static const flags canonical_name = implementation_defined;
39
40 /// Indicate that returned endpoint is intended for use as a locally bound
41 /// socket endpoint.
42 static const flags passive = implementation_defined;
43
44 /// Host name should be treated as a numeric string defining an IPv4 or IPv6
45 /// address and no name resolution should be attempted.
46 static const flags numeric_host = implementation_defined;
47
48 /// Service name should be treated as a numeric string defining a port number
49 /// and no name resolution should be attempted.
50 static const flags numeric_service = implementation_defined;
51
52 /// If the query protocol family is specified as IPv6, return IPv4-mapped
53 /// IPv6 addresses on finding no IPv6 addresses.
54 static const flags v4_mapped = implementation_defined;
55
56 /// If used with v4_mapped, return all matching IPv6 and IPv4 addresses.
57 static const flags all_matching = implementation_defined;
58
59 /// Only return IPv4 addresses if a non-loopback IPv4 address is configured
60 /// for the system. Only return IPv6 addresses if a non-loopback IPv6 address
61 /// is configured for the system.
62 static const flags address_configured = implementation_defined;
63#else
64 enum flags
65 {
66 canonical_name = AI_CANONNAME,
67 passive = AI_PASSIVE,
68 numeric_host = AI_NUMERICHOST,
69# if defined(AI_NUMERICSERV)
70 numeric_service = AI_NUMERICSERV,
71# else
72 numeric_service = 0,
73# endif
74 // Note: QNX Neutrino 6.3 defines AI_V4MAPPED, AI_ALL and AI_ADDRCONFIG but
75 // does not implement them. Therefore they are specifically excluded here.
76# if defined(AI_V4MAPPED) && !defined(__QNXNTO__)
77 v4_mapped = AI_V4MAPPED,
78# else
79 v4_mapped = 0,
80# endif
81# if defined(AI_ALL) && !defined(__QNXNTO__)
82 all_matching = AI_ALL,
83# else
84 all_matching = 0,
85# endif
86# if defined(AI_ADDRCONFIG) && !defined(__QNXNTO__)
87 address_configured = AI_ADDRCONFIG
88# else
89 address_configured = 0
90# endif
91 };
92
93 // Implement bitmask operations as shown in C++ Std [lib.bitmask.types].
94
95 friend flags operator&(flags x, flags y)
96 {
97 return static_cast<flags>(
98 static_cast<unsigned int>(x) & static_cast<unsigned int>(y));
99 }
100
101 friend flags operator|(flags x, flags y)
102 {
103 return static_cast<flags>(
104 static_cast<unsigned int>(x) | static_cast<unsigned int>(y));
105 }
106
107 friend flags operator^(flags x, flags y)
108 {
109 return static_cast<flags>(
110 static_cast<unsigned int>(x) ^ static_cast<unsigned int>(y));
111 }
112
113 friend flags operator~(flags x)
114 {
115 return static_cast<flags>(static_cast<unsigned int>(~x));
116 }
117
118 friend flags& operator&=(flags& x, flags y)
119 {
120 x = x & y;
121 return x;
122 }
123
124 friend flags& operator|=(flags& x, flags y)
125 {
126 x = x | y;
127 return x;
128 }
129
130 friend flags& operator^=(flags& x, flags y)
131 {
132 x = x ^ y;
133 return x;
134 }
135#endif
136
137protected:
138 /// Protected destructor to prevent deletion through this type.
139 ~resolver_query_base()
140 {
141 }
142};

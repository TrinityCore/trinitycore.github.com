35/// An iterator over the entries produced by a resolver.
36/**
37 * The boost::asio::ip::basic_resolver_iterator class template is used to define
38 * iterators over the results returned by a resolver.
39 *
40 * The iterator's value_type, obtained when the iterator is dereferenced, is:
41 * @code const basic_resolver_entry<InternetProtocol> @endcode
42 *
43 * @par Thread Safety
44 * @e Distinct @e objects: Safe.@n
45 * @e Shared @e objects: Unsafe.
46 */
47template <typename InternetProtocol>
48class basic_resolver_iterator
49{
50public:
51 /// The type used for the distance between two iterators.
52 typedef std::ptrdiff_t difference_type;
53
54 /// The type of the value pointed to by the iterator.
55 typedef basic_resolver_entry<InternetProtocol> value_type;
56
57 /// The type of the result of applying operator->() to the iterator.
58 typedef const basic_resolver_entry<InternetProtocol>* pointer;
59
60 /// The type of the result of applying operator*() to the iterator.
61 typedef const basic_resolver_entry<InternetProtocol>& reference;
62
63 /// The iterator category.
64 typedef std::forward_iterator_tag iterator_category;
65
66 /// Default constructor creates an end iterator.
67 basic_resolver_iterator()
68 : index_(0)
69 {
70 }
71
72 /// Create an iterator from an addrinfo list returned by getaddrinfo.
73 static basic_resolver_iterator create(
74 boost::asio::detail::addrinfo_type* address_info,
75 const std::string& host_name, const std::string& service_name)
76 {
77 basic_resolver_iterator iter;
78 if (!address_info)
79 return iter;
80
81 std::string actual_host_name = host_name;
82 if (address_info->ai_canonname)
83 actual_host_name = address_info->ai_canonname;
84
85 iter.values_.reset(new values_type);
86
87 while (address_info)
88 {
89 if (address_info->ai_family == PF_INET
90 || address_info->ai_family == PF_INET6)
91 {
92 using namespace std; // For memcpy.
93 typename InternetProtocol::endpoint endpoint;
94 endpoint.resize(static_cast<std::size_t>(address_info->ai_addrlen));
95 memcpy(endpoint.data(), address_info->ai_addr,
96 address_info->ai_addrlen);
97 iter.values_->push_back(
98 basic_resolver_entry<InternetProtocol>(endpoint,
99 actual_host_name, service_name));
100 }
101 address_info = address_info->ai_next;
102 }
103
104 return iter;
105 }
106
107 /// Create an iterator from an endpoint, host name and service name.
108 static basic_resolver_iterator create(
109 const typename InternetProtocol::endpoint& endpoint,
110 const std::string& host_name, const std::string& service_name)
111 {
112 basic_resolver_iterator iter;
113 iter.values_.reset(new values_type);
114 iter.values_->push_back(
115 basic_resolver_entry<InternetProtocol>(
116 endpoint, host_name, service_name));
117 return iter;
118 }
119
120 /// Dereference an iterator.
121 const basic_resolver_entry<InternetProtocol>& operator*() const
122 {
123 return dereference();
124 }
125
126 /// Dereference an iterator.
127 const basic_resolver_entry<InternetProtocol>* operator->() const
128 {
129 return &dereference();
130 }
131
132 /// Increment operator (prefix).
133 basic_resolver_iterator& operator++()
134 {
135 increment();
136 return *this;
137 }
138
139 /// Increment operator (postfix).
140 basic_resolver_iterator operator++(int)
141 {
142 basic_resolver_iterator tmp(*this);
143 ++*this;
144 return tmp;
145 }
146
147 /// Test two iterators for equality.
148 friend bool operator==(const basic_resolver_iterator& a,
149 const basic_resolver_iterator& b)
150 {
151 return a.equal(b);
152 }
153
154 /// Test two iterators for inequality.
155 friend bool operator!=(const basic_resolver_iterator& a,
156 const basic_resolver_iterator& b)
157 {
158 return !a.equal(b);
159 }
160
161private:
162 void increment()
163 {
164 if (++index_ == values_->size())
165 {
166 // Reset state to match a default constructed end iterator.
167 values_.reset();
168 index_ = 0;
169 }
170 }
171
172 bool equal(const basic_resolver_iterator& other) const
173 {
174 if (!values_ && !other.values_)
175 return true;
176 if (values_ != other.values_)
177 return false;
178 return index_ == other.index_;
179 }
180
181 const basic_resolver_entry<InternetProtocol>& dereference() const
182 {
183 return (*values_)[index_];
184 }
185
186 typedef std::vector<basic_resolver_entry<InternetProtocol> > values_type;
187 boost::asio::detail::shared_ptr<values_type> values_;
188 std::size_t index_;
189};

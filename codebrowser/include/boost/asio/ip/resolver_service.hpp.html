28namespace boost {
29namespace asio {
30namespace ip {
31
32/// Default service implementation for a resolver.
33template <typename InternetProtocol>
34class resolver_service
35#if defined(GENERATING_DOCUMENTATION)
36 : public boost::asio::io_service::service
37#else
38 : public boost::asio::detail::service_base<
39 resolver_service<InternetProtocol> >
40#endif
41{
42public:
43#if defined(GENERATING_DOCUMENTATION)
44 /// The unique service identifier.
45 static boost::asio::io_service::id id;
46#endif
47
48 /// The protocol type.
49 typedef InternetProtocol protocol_type;
50
51 /// The endpoint type.
52 typedef typename InternetProtocol::endpoint endpoint_type;
53
54 /// The query type.
55 typedef basic_resolver_query<InternetProtocol> query_type;
56
57 /// The iterator type.
58 typedef basic_resolver_iterator<InternetProtocol> iterator_type;
59
60private:
61 // The type of the platform-specific implementation.
62 typedef boost::asio::detail::resolver_service<InternetProtocol>
63 service_impl_type;
64
65public:
66 /// The type of a resolver implementation.
67#if defined(GENERATING_DOCUMENTATION)
68 typedef implementation_defined implementation_type;
69#else
70 typedef typename service_impl_type::implementation_type implementation_type;
71#endif
72
73 /// Construct a new resolver service for the specified io_service.
74 explicit resolver_service(boost::asio::io_service& io_service)
75 : boost::asio::detail::service_base<
76 resolver_service<InternetProtocol> >(io_service),
77 service_impl_(io_service)
78 {
79 }
80
81 /// Construct a new resolver implementation.
82 void construct(implementation_type& impl)
83 {
84 service_impl_.construct(impl);
85 }
86
87 /// Destroy a resolver implementation.
88 void destroy(implementation_type& impl)
89 {
90 service_impl_.destroy(impl);
91 }
92
93 /// Cancel pending asynchronous operations.
94 void cancel(implementation_type& impl)
95 {
96 service_impl_.cancel(impl);
97 }
98
99 /// Resolve a query to a list of entries.
100 iterator_type resolve(implementation_type& impl, const query_type& query,
101 boost::system::error_code& ec)
102 {
103 return service_impl_.resolve(impl, query, ec);
104 }
105
106 /// Asynchronously resolve a query to a list of entries.
107 template <typename ResolveHandler>
108 BOOST_ASIO_INITFN_RESULT_TYPE(ResolveHandler,
109 void (boost::system::error_code, iterator_type))
110 async_resolve(implementation_type& impl, const query_type& query,
111 BOOST_ASIO_MOVE_ARG(ResolveHandler) handler)
112 {
113 boost::asio::detail::async_result_init<
114 ResolveHandler, void (boost::system::error_code, iterator_type)> init(
115 BOOST_ASIO_MOVE_CAST(ResolveHandler)(handler));
116
117 service_impl_.async_resolve(impl, query, init.handler);
118
119 return init.result.get();
120 }
121
122 /// Resolve an endpoint to a list of entries.
123 iterator_type resolve(implementation_type& impl,
124 const endpoint_type& endpoint, boost::system::error_code& ec)
125 {
126 return service_impl_.resolve(impl, endpoint, ec);
127 }
128
129 /// Asynchronously resolve an endpoint to a list of entries.
130 template <typename ResolveHandler>
131 BOOST_ASIO_INITFN_RESULT_TYPE(ResolveHandler,
132 void (boost::system::error_code, iterator_type))
133 async_resolve(implementation_type& impl, const endpoint_type& endpoint,
134 BOOST_ASIO_MOVE_ARG(ResolveHandler) handler)
135 {
136 boost::asio::detail::async_result_init<
137 ResolveHandler, void (boost::system::error_code, iterator_type)> init(
138 BOOST_ASIO_MOVE_CAST(ResolveHandler)(handler));
139
140 service_impl_.async_resolve(impl, endpoint, init.handler);
141
142 return init.result.get();
143 }
144
145private:
146 // Destroy all user-defined handler objects owned by the service.
147 void shutdown_service()
148 {
149 service_impl_.shutdown_service();
150 }
151
152 // Perform any fork-related housekeeping.
153 void fork_service(boost::asio::io_service::fork_event event)
154 {
155 service_impl_.fork_service(event);
156 }
157
158 // The platform-specific implementation.
159 service_impl_type service_impl_;
160};

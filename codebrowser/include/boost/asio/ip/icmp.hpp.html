32/// Encapsulates the flags needed for ICMP.
33/**
34 * The boost::asio::ip::icmp class contains flags necessary for ICMP sockets.
35 *
36 * @par Thread Safety
37 * @e Distinct @e objects: Safe.@n
38 * @e Shared @e objects: Safe.
39 *
40 * @par Concepts:
41 * Protocol, InternetProtocol.
42 */
43class icmp
44{
45public:
46 /// The type of a ICMP endpoint.
47 typedef basic_endpoint<icmp> endpoint;
48
49 /// Construct to represent the IPv4 ICMP protocol.
50 static icmp v4()
51 {
52 return icmp(IPPROTO_ICMP, PF_INET);
53 }
54
55 /// Construct to represent the IPv6 ICMP protocol.
56 static icmp v6()
57 {
58 return icmp(IPPROTO_ICMPV6, PF_INET6);
59 }
60
61 /// Obtain an identifier for the type of the protocol.
62 int type() const
63 {
64 return SOCK_RAW;
65 }
66
67 /// Obtain an identifier for the protocol.
68 int protocol() const
69 {
70 return protocol_;
71 }
72
73 /// Obtain an identifier for the protocol family.
74 int family() const
75 {
76 return family_;
77 }
78
79 /// The ICMP socket type.
80 typedef basic_raw_socket<icmp> socket;
81
82 /// The ICMP resolver type.
83 typedef basic_resolver<icmp> resolver;
84
85 /// Compare two protocols for equality.
86 friend bool operator==(const icmp& p1, const icmp& p2)
87 {
88 return p1.protocol_ == p2.protocol_ && p1.family_ == p2.family_;
89 }
90
91 /// Compare two protocols for inequality.
92 friend bool operator!=(const icmp& p1, const icmp& p2)
93 {
94 return p1.protocol_ != p2.protocol_ || p1.family_ != p2.family_;
95 }
96
97private:
98 // Construct with a specific family.
99 explicit icmp(int protocol_id, int protocol_family)
100 : protocol_(protocol_id),
101 family_(protocol_family)
102 {
103 }
104
105 int protocol_;
106 int family_;
107};

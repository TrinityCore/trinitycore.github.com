29/// Socket option to join a multicast group on a specified interface.
30/**
31 * Implements the IPPROTO_IP/IP_ADD_MEMBERSHIP socket option.
32 *
33 * @par Examples
34 * Setting the option to join a multicast group:
35 * @code
36 * boost::asio::ip::udp::socket socket(io_service);
37 * ...
38 * boost::asio::ip::address multicast_address =
39 * boost::asio::ip::address::from_string("225.0.0.1");
40 * boost::asio::ip::multicast::join_group option(multicast_address);
41 * socket.set_option(option);
42 * @endcode
43 *
44 * @par Concepts:
45 * SettableSocketOption.
46 */
47#if defined(GENERATING_DOCUMENTATION)
48typedef implementation_defined join_group;
49#else
50typedef boost::asio::ip::detail::socket_option::multicast_request<
51 IPPROTO_IP, IP_ADD_MEMBERSHIP, IPPROTO_IPV6, IPV6_JOIN_GROUP> join_group;
52#endif

105/// Socket option for time-to-live associated with outgoing multicast packets.
106/**
107 * Implements the IPPROTO_IP/IP_MULTICAST_TTL socket option.

109 * @par Examples

111 * @code

117 *
118 * @par
119 * Getting the current option value:
120 * @code
121 * boost::asio::ip::udp::socket socket(io_service);

123 * boost::asio::ip::multicast::hops option;
124 * socket.get_option(option);
125 * int ttl = option.value();
126 * @endcode
127 *

138/// Socket option determining whether outgoing multicast packets will be
139/// received on the same socket if it is a member of the multicast group.
140/**
141 * Implements the IPPROTO_IP/IP_MULTICAST_LOOP socket option.
142 *
143 * @par Examples
144 * Setting the option:
145 * @code
146 * boost::asio::ip::udp::socket socket(io_service);
147 * ...
148 * boost::asio::ip::multicast::enable_loopback option(true);
149 * socket.set_option(option);
150 * @endcode
151 *
152 * @par
153 * Getting the current option value:
154 * @code
155 * boost::asio::ip::udp::socket socket(io_service);
156 * ...
157 * boost::asio::ip::multicast::enable_loopback option;
158 * socket.get_option(option);
159 * bool is_set = option.value();
160 * @endcode
161 *
162 * @par Concepts:
163 * GettableSocketOption, SettableSocketOption.
164 */
165#if defined(GENERATING_DOCUMENTATION)
166typedef implementation_defined enable_loopback;
167#else
168typedef boost::asio::ip::detail::socket_option::multicast_enable_loopback<
169 IPPROTO_IP, IP_MULTICAST_LOOP, IPPROTO_IPV6, IPV6_MULTICAST_LOOP>
170 enable_loopback;
171#endif

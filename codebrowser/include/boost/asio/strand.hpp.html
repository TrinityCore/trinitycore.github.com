30/// Provides serialised handler execution.
31/**
32 * The io_service::strand class provides the ability to post and dispatch
33 * handlers with the guarantee that none of those handlers will execute
34 * concurrently.
35 *
36 * @par Order of handler invocation
37 * Given:
38 *
39 * @li a strand object @c s
40 *
41 * @li an object @c a meeting completion handler requirements
42 *
43 * @li an object @c a1 which is an arbitrary copy of @c a made by the
44 * implementation
45 *
46 * @li an object @c b meeting completion handler requirements
47 *
48 * @li an object @c b1 which is an arbitrary copy of @c b made by the
49 * implementation
50 *
51 * if any of the following conditions are true:
52 *
53 * @li @c s.post(a) happens-before @c s.post(b)
54 *
55 * @li @c s.post(a) happens-before @c s.dispatch(b), where the latter is
56 * performed outside the strand
57 *
58 * @li @c s.dispatch(a) happens-before @c s.post(b), where the former is
59 * performed outside the strand
60 *
61 * @li @c s.dispatch(a) happens-before @c s.dispatch(b), where both are
62 * performed outside the strand
63 *
64 * then @c asio_handler_invoke(a1, &a1) happens-before
65 * @c asio_handler_invoke(b1, &b1).
66 *
67 * Note that in the following case:
68 * @code async_op_1(..., s.wrap(a));
69 * async_op_2(..., s.wrap(b)); @endcode
70 * the completion of the first async operation will perform @c s.dispatch(a),
71 * and the second will perform @c s.dispatch(b), but the order in which those
72 * are performed is unspecified. That is, you cannot state whether one
73 * happens-before the other. Therefore none of the above conditions are met and
74 * no ordering guarantee is made.
75 *
76 * @par Thread Safety
77 * @e Distinct @e objects: Safe.@n
78 * @e Shared @e objects: Safe.
79 *
80 * @par Concepts:
81 * Dispatcher.
82 */
83class io_service::strand
84{
85public:
86 /// Constructor.
87 /**
88 * Constructs the strand.
89 *
90 * @param io_service The io_service object that the strand will use to
91 * dispatch handlers that are ready to be run.
92 */
93 explicit strand(boost::asio::io_service& io_service)
94 : service_(boost::asio::use_service<
95 boost::asio::detail::strand_service>(io_service))
96 {
97 service_.construct(impl_);
98 }
99
100 /// Destructor.
101 /**
102 * Destroys a strand.
103 *
104 * Handlers posted through the strand that have not yet been invoked will
105 * still be dispatched in a way that meets the guarantee of non-concurrency.
106 */
107 ~strand()
108 {
109 }
110
111 /// Get the io_service associated with the strand.
112 /**
113 * This function may be used to obtain the io_service object that the strand
114 * uses to dispatch handlers for asynchronous operations.
115 *
116 * @return A reference to the io_service object that the strand will use to
117 * dispatch handlers. Ownership is not transferred to the caller.
118 */
119 boost::asio::io_service& get_io_service()
120 {
121 return service_.get_io_service();
122 }
123
124 /// Request the strand to invoke the given handler.
125 /**
126 * This function is used to ask the strand to execute the given handler.
127 *
128 * The strand object guarantees that handlers posted or dispatched through
129 * the strand will not be executed concurrently. The handler may be executed
130 * inside this function if the guarantee can be met. If this function is
131 * called from within a handler that was posted or dispatched through the same
132 * strand, then the new handler will be executed immediately.
133 *
134 * The strand's guarantee is in addition to the guarantee provided by the
135 * underlying io_service. The io_service guarantees that the handler will only
136 * be called in a thread in which the io_service's run member function is
137 * currently being invoked.
138 *
139 * @param handler The handler to be called. The strand will make a copy of the
140 * handler object as required. The function signature of the handler must be:
141 * @code void handler(); @endcode
142 */
143 template <typename CompletionHandler>
144 BOOST_ASIO_INITFN_RESULT_TYPE(CompletionHandler, void ())
145 dispatch(BOOST_ASIO_MOVE_ARG(CompletionHandler) handler)
146 {
147 // If you get an error on the following line it means that your handler does
148 // not meet the documented type requirements for a CompletionHandler.
149 BOOST_ASIO_COMPLETION_HANDLER_CHECK(CompletionHandler, handler) type_check;
150
151 detail::async_result_init<
152 CompletionHandler, void ()> init(
153 BOOST_ASIO_MOVE_CAST(CompletionHandler)(handler));
154
155 service_.dispatch(impl_, init.handler);
156
157 return init.result.get();
158 }
159
160 /// Request the strand to invoke the given handler and return
161 /// immediately.
162 /**
163 * This function is used to ask the strand to execute the given handler, but
164 * without allowing the strand to call the handler from inside this function.
165 *
166 * The strand object guarantees that handlers posted or dispatched through
167 * the strand will not be executed concurrently. The strand's guarantee is in
168 * addition to the guarantee provided by the underlying io_service. The
169 * io_service guarantees that the handler will only be called in a thread in
170 * which the io_service's run member function is currently being invoked.
171 *
172 * @param handler The handler to be called. The strand will make a copy of the
173 * handler object as required. The function signature of the handler must be:
174 * @code void handler(); @endcode
175 */
176 template <typename CompletionHandler>
177 BOOST_ASIO_INITFN_RESULT_TYPE(CompletionHandler, void ())
178 post(BOOST_ASIO_MOVE_ARG(CompletionHandler) handler)
179 {
180 // If you get an error on the following line it means that your handler does
181 // not meet the documented type requirements for a CompletionHandler.
182 BOOST_ASIO_COMPLETION_HANDLER_CHECK(CompletionHandler, handler) type_check;
183
184 detail::async_result_init<
185 CompletionHandler, void ()> init(
186 BOOST_ASIO_MOVE_CAST(CompletionHandler)(handler));
187
188 service_.post(impl_, init.handler);
189
190 return init.result.get();
191 }
192
193 /// Create a new handler that automatically dispatches the wrapped handler
194 /// on the strand.
195 /**
196 * This function is used to create a new handler function object that, when
197 * invoked, will automatically pass the wrapped handler to the strand's
198 * dispatch function.
199 *
200 * @param handler The handler to be wrapped. The strand will make a copy of
201 * the handler object as required. The function signature of the handler must
202 * be: @code void handler(A1 a1, ... An an); @endcode
203 *
204 * @return A function object that, when invoked, passes the wrapped handler to
205 * the strand's dispatch function. Given a function object with the signature:
206 * @code R f(A1 a1, ... An an); @endcode
207 * If this function object is passed to the wrap function like so:
208 * @code strand.wrap(f); @endcode
209 * then the return value is a function object with the signature
210 * @code void g(A1 a1, ... An an); @endcode
211 * that, when invoked, executes code equivalent to:
212 * @code strand.dispatch(boost::bind(f, a1, ... an)); @endcode
213 */
214 template <typename Handler>
215#if defined(GENERATING_DOCUMENTATION)
216 unspecified
217#else
218 detail::wrapped_handler<strand, Handler, detail::is_continuation_if_running>
219#endif
220 wrap(Handler handler)
221 {
222 return detail::wrapped_handler<io_service::strand, Handler,
223 detail::is_continuation_if_running>(*this, handler);
224 }
225
226 /// Determine whether the strand is running in the current thread.
227 /**
228 * @return @c true if the current thread is executing a handler that was
229 * submitted to the strand using post(), dispatch() or wrap(). Otherwise
230 * returns @c false.
231 */
232 bool running_in_this_thread() const
233 {
234 return service_.running_in_this_thread(impl_);
235 }
236
237private:
238 boost::asio::detail::strand_service& service_;
239 boost::asio::detail::strand_service::implementation_type impl_;
240};

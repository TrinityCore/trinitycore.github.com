32namespace boost {
33namespace asio {
34
35/// Default service implementation for a socket acceptor.
36template <typename Protocol>
37class socket_acceptor_service
38#if defined(GENERATING_DOCUMENTATION)
39 : public boost::asio::io_service::service
40#else
41 : public boost::asio::detail::service_base<socket_acceptor_service<Protocol> >
42#endif
43{
44public:
45#if defined(GENERATING_DOCUMENTATION)
46 /// The unique service identifier.
47 static boost::asio::io_service::id id;
48#endif
49
50 /// The protocol type.
51 typedef Protocol protocol_type;
52
53 /// The endpoint type.
54 typedef typename protocol_type::endpoint endpoint_type;
55
56private:
57 // The type of the platform-specific implementation.
58#if defined(BOOST_ASIO_HAS_IOCP)
59 typedef detail::win_iocp_socket_service<Protocol> service_impl_type;
60#else
61 typedef detail::reactive_socket_service<Protocol> service_impl_type;
62#endif
63
64public:
65 /// The native type of the socket acceptor.
66#if defined(GENERATING_DOCUMENTATION)
67 typedef implementation_defined implementation_type;
68#else
69 typedef typename service_impl_type::implementation_type implementation_type;
70#endif
71
72 /// (Deprecated: Use native_handle_type.) The native acceptor type.
73#if defined(GENERATING_DOCUMENTATION)
74 typedef implementation_defined native_type;
75#else
76 typedef typename service_impl_type::native_handle_type native_type;
77#endif
78
79 /// The native acceptor type.
80#if defined(GENERATING_DOCUMENTATION)
81 typedef implementation_defined native_handle_type;
82#else
83 typedef typename service_impl_type::native_handle_type native_handle_type;
84#endif
85
86 /// Construct a new socket acceptor service for the specified io_service.
87 explicit socket_acceptor_service(boost::asio::io_service& io_service)
88 : boost::asio::detail::service_base<
89 socket_acceptor_service<Protocol> >(io_service),
90 service_impl_(io_service)
91 {
92 }
93
94 /// Construct a new socket acceptor implementation.
95 void construct(implementation_type& impl)
96 {
97 service_impl_.construct(impl);
98 }
99
100#if defined(BOOST_ASIO_HAS_MOVE) || defined(GENERATING_DOCUMENTATION)
101 /// Move-construct a new socket acceptor implementation.
102 void move_construct(implementation_type& impl,
103 implementation_type& other_impl)
104 {
105 service_impl_.move_construct(impl, other_impl);
106 }
107
108 /// Move-assign from another socket acceptor implementation.
109 void move_assign(implementation_type& impl,
110 socket_acceptor_service& other_service,
111 implementation_type& other_impl)
112 {
113 service_impl_.move_assign(impl, other_service.service_impl_, other_impl);
114 }
115
116 /// Move-construct a new socket acceptor implementation from another protocol
117 /// type.
118 template <typename Protocol1>
119 void converting_move_construct(implementation_type& impl,
120 typename socket_acceptor_service<
121 Protocol1>::implementation_type& other_impl,
122 typename enable_if<is_convertible<
123 Protocol1, Protocol>::value>::type* = 0)
124 {
125 service_impl_.template converting_move_construct<Protocol1>(
126 impl, other_impl);
127 }
128#endif // defined(BOOST_ASIO_HAS_MOVE) || defined(GENERATING_DOCUMENTATION)
129
130 /// Destroy a socket acceptor implementation.
131 void destroy(implementation_type& impl)
132 {
133 service_impl_.destroy(impl);
134 }
135
136 /// Open a new socket acceptor implementation.
137 boost::system::error_code open(implementation_type& impl,
138 const protocol_type& protocol, boost::system::error_code& ec)
139 {
140 return service_impl_.open(impl, protocol, ec);
141 }
142
143 /// Assign an existing native acceptor to a socket acceptor.
144 boost::system::error_code assign(implementation_type& impl,
145 const protocol_type& protocol, const native_handle_type& native_acceptor,
146 boost::system::error_code& ec)
147 {
148 return service_impl_.assign(impl, protocol, native_acceptor, ec);
149 }
150
151 /// Determine whether the acceptor is open.
152 bool is_open(const implementation_type& impl) const
153 {
154 return service_impl_.is_open(impl);
155 }
156
157 /// Cancel all asynchronous operations associated with the acceptor.
158 boost::system::error_code cancel(implementation_type& impl,
159 boost::system::error_code& ec)
160 {
161 return service_impl_.cancel(impl, ec);
162 }
163
164 /// Bind the socket acceptor to the specified local endpoint.
165 boost::system::error_code bind(implementation_type& impl,
166 const endpoint_type& endpoint, boost::system::error_code& ec)
167 {
168 return service_impl_.bind(impl, endpoint, ec);
169 }
170
171 /// Place the socket acceptor into the state where it will listen for new
172 /// connections.
173 boost::system::error_code listen(implementation_type& impl, int backlog,
174 boost::system::error_code& ec)
175 {
176 return service_impl_.listen(impl, backlog, ec);
177 }
178
179 /// Close a socket acceptor implementation.
180 boost::system::error_code close(implementation_type& impl,
181 boost::system::error_code& ec)
182 {
183 return service_impl_.close(impl, ec);
184 }
185
186 /// (Deprecated: Use native_handle().) Get the native acceptor implementation.
187 native_type native(implementation_type& impl)
188 {
189 return service_impl_.native_handle(impl);
190 }
191
192 /// Get the native acceptor implementation.
193 native_handle_type native_handle(implementation_type& impl)
194 {
195 return service_impl_.native_handle(impl);
196 }
197
198 /// Set a socket option.
199 template <typename SettableSocketOption>
200 boost::system::error_code set_option(implementation_type& impl,
201 const SettableSocketOption& option, boost::system::error_code& ec)
202 {
203 return service_impl_.set_option(impl, option, ec);
204 }
205
206 /// Get a socket option.
207 template <typename GettableSocketOption>
208 boost::system::error_code get_option(const implementation_type& impl,
209 GettableSocketOption& option, boost::system::error_code& ec) const
210 {
211 return service_impl_.get_option(impl, option, ec);
212 }
213
214 /// Perform an IO control command on the socket.
215 template <typename IoControlCommand>
216 boost::system::error_code io_control(implementation_type& impl,
217 IoControlCommand& command, boost::system::error_code& ec)
218 {
219 return service_impl_.io_control(impl, command, ec);
220 }
221
222 /// Gets the non-blocking mode of the acceptor.
223 bool non_blocking(const implementation_type& impl) const
224 {
225 return service_impl_.non_blocking(impl);
226 }
227
228 /// Sets the non-blocking mode of the acceptor.
229 boost::system::error_code non_blocking(implementation_type& impl,
230 bool mode, boost::system::error_code& ec)
231 {
232 return service_impl_.non_blocking(impl, mode, ec);
233 }
234
235 /// Gets the non-blocking mode of the native acceptor implementation.
236 bool native_non_blocking(const implementation_type& impl) const
237 {
238 return service_impl_.native_non_blocking(impl);
239 }
240
241 /// Sets the non-blocking mode of the native acceptor implementation.
242 boost::system::error_code native_non_blocking(implementation_type& impl,
243 bool mode, boost::system::error_code& ec)
244 {
245 return service_impl_.native_non_blocking(impl, mode, ec);
246 }
247
248 /// Get the local endpoint.
249 endpoint_type local_endpoint(const implementation_type& impl,
250 boost::system::error_code& ec) const
251 {
252 return service_impl_.local_endpoint(impl, ec);
253 }
254
255 /// Accept a new connection.
256 template <typename Protocol1, typename SocketService>
257 boost::system::error_code accept(implementation_type& impl,
258 basic_socket<Protocol1, SocketService>& peer,
259 endpoint_type* peer_endpoint, boost::system::error_code& ec,
260 typename enable_if<is_convertible<Protocol, Protocol1>::value>::type* = 0)
261 {
262 return service_impl_.accept(impl, peer, peer_endpoint, ec);
263 }
264
265 /// Start an asynchronous accept.
266 template <typename Protocol1, typename SocketService, typename AcceptHandler>
267 BOOST_ASIO_INITFN_RESULT_TYPE(AcceptHandler,
268 void (boost::system::error_code))
269 async_accept(implementation_type& impl,
270 basic_socket<Protocol1, SocketService>& peer,
271 endpoint_type* peer_endpoint,
272 BOOST_ASIO_MOVE_ARG(AcceptHandler) handler,
273 typename enable_if<is_convertible<Protocol, Protocol1>::value>::type* = 0)
274 {
275 detail::async_result_init<
276 AcceptHandler, void (boost::system::error_code)> init(
277 BOOST_ASIO_MOVE_CAST(AcceptHandler)(handler));
278
279 service_impl_.async_accept(impl, peer, peer_endpoint, init.handler);
280
281 return init.result.get();
282 }
283
284private:
285 // Destroy all user-defined handler objects owned by the service.
286 void shutdown_service()
287 {
288 service_impl_.shutdown_service();
289 }
290
291 // The platform-specific implementation.
292 service_impl_type service_impl_;
293};

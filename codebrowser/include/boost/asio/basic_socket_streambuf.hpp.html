97/// Iostream streambuf for a socket.
98template <typename Protocol,
99 typename StreamSocketService = stream_socket_service<Protocol>,
100#if defined(BOOST_ASIO_HAS_BOOST_DATE_TIME) \
101 || defined(GENERATING_DOCUMENTATION)
102 typename Time = boost::posix_time::ptime,
103 typename TimeTraits = boost::asio::time_traits<Time>,
104 typename TimerService = deadline_timer_service<Time, TimeTraits> >
105#else
106 typename Time = steady_timer::clock_type,
107 typename TimeTraits = steady_timer::traits_type,
108 typename TimerService = steady_timer::service_type>
109#endif
110class basic_socket_streambuf
111 : public std::streambuf,
112 private detail::socket_streambuf_base,
113 public basic_socket<Protocol, StreamSocketService>
114{
115private:
116 // These typedefs are intended keep this class's implementation independent
117 // of whether it's using Boost.DateTime, Boost.Chrono or std::chrono.
118#if defined(BOOST_ASIO_HAS_BOOST_DATE_TIME)
119 typedef TimeTraits traits_helper;
120#else
121 typedef detail::chrono_time_traits<Time, TimeTraits> traits_helper;
122#endif
123
124public:
125 /// The endpoint type.
126 typedef typename Protocol::endpoint endpoint_type;
127
128#if defined(GENERATING_DOCUMENTATION)
129 /// The time type.
130 typedef typename TimeTraits::time_type time_type;
131
132 /// The duration type.
133 typedef typename TimeTraits::duration_type duration_type;
134#else
135 typedef typename traits_helper::time_type time_type;
136 typedef typename traits_helper::duration_type duration_type;
137#endif

139 /// Construct a basic_socket_streambuf without establishing a connection.
140 basic_socket_streambuf()
141 : basic_socket<Protocol, StreamSocketService>(
142 this->detail::socket_streambuf_base::io_service_),
143 unbuffered_(false),
144 timer_service_(0),
145 timer_state_(no_timer)
146 {
147 init_buffers();
148 }
149
150 /// Destructor flushes buffered data.
151 virtual ~basic_socket_streambuf()
152 {
153 if (pptr() != pbase())
154 overflow(traits_type::eof());
155
156 destroy_timer();
157 }

159 /// Establish a connection.
160 /**
161 * This function establishes a connection to the specified endpoint.
162 *
163 * @return \c this if a connection was successfully established, a null
164 * pointer otherwise.
165 */
166 basic_socket_streambuf<Protocol, StreamSocketService,
167 Time, TimeTraits, TimerService>* connect(
168 const endpoint_type& endpoint)
169 {
170 init_buffers();
171
172 this->basic_socket<Protocol, StreamSocketService>::close(ec_);
173
174 if (timer_state_ == timer_has_expired)
175 {
176 ec_ = boost::asio::error::operation_aborted;
177 return 0;
178 }
179
180 io_handler handler = { this };
181 this->basic_socket<Protocol, StreamSocketService>::async_connect(
182 endpoint, handler);
183
184 ec_ = boost::asio::error::would_block;
185 this->get_service().get_io_service().reset();
186 do this->get_service().get_io_service().run_one();
187 while (ec_ == boost::asio::error::would_block);
188
189 return !ec_ ? this : 0;
190 }
191
192#if defined(GENERATING_DOCUMENTATION)
193 /// Establish a connection.
194 /**
195 * This function automatically establishes a connection based on the supplied
196 * resolver query parameters. The arguments are used to construct a resolver
197 * query object.
198 *
199 * @return \c this if a connection was successfully established, a null
200 * pointer otherwise.
201 */
202 template <typename T1, ..., typename TN>
203 basic_socket_streambuf<Protocol, StreamSocketService>* connect(
204 T1 t1, ..., TN tn);
205#elif defined(BOOST_ASIO_HAS_VARIADIC_TEMPLATES)
206 template <typename... T>
207 basic_socket_streambuf<Protocol, StreamSocketService,
208 Time, TimeTraits, TimerService>* connect(T... x)
209 {
210 init_buffers();
211 this->basic_socket<Protocol, StreamSocketService>::close(ec_);
212 typedef typename Protocol::resolver resolver_type;
213 typedef typename resolver_type::query resolver_query;
214 resolver_query query(x...);
215 resolve_and_connect(query);
216 return !ec_ ? this : 0;
217 }
218#else
219 BOOST_PP_REPEAT_FROM_TO(
220 1, BOOST_PP_INC(BOOST_ASIO_SOCKET_STREAMBUF_MAX_ARITY),
221 BOOST_ASIO_PRIVATE_CONNECT_DEF, _ )
222#endif

224 /// Close the connection.
225 /**
226 * @return \c this if a connection was successfully established, a null
227 * pointer otherwise.
228 */
229 basic_socket_streambuf<Protocol, StreamSocketService,
230 Time, TimeTraits, TimerService>* close()
231 {
232 sync();
233 this->basic_socket<Protocol, StreamSocketService>::close(ec_);
234 if (!ec_)
235 init_buffers();
236 return !ec_ ? this : 0;
237 }

249 /// Get the stream buffer's expiry time as an absolute time.
250 /**
251 * @return An absolute time value representing the stream buffer's expiry
252 * time.
253 */
254 time_type expires_at() const
255 {
256 return timer_service_
257 ? timer_service_->expires_at(timer_implementation_)
258 : time_type();
259 }
260
261 /// Set the stream buffer's expiry time as an absolute time.
262 /**
263 * This function sets the expiry time associated with the stream. Stream
264 * operations performed after this time (where the operations cannot be
265 * completed using the internal buffers) will fail with the error
266 * boost::asio::error::operation_aborted.
267 *
268 * @param expiry_time The expiry time to be used for the stream.
269 */
270 void expires_at(const time_type& expiry_time)
271 {
272 construct_timer();
273
274 boost::system::error_code ec;
275 timer_service_->expires_at(timer_implementation_, expiry_time, ec);
276 boost::asio::detail::throw_error(ec, "expires_at");
277
278 start_timer();
279 }
280
281 /// Get the stream buffer's expiry time relative to now.
282 /**
283 * @return A relative time value representing the stream buffer's expiry time.
284 */
285 duration_type expires_from_now() const
286 {
287 return traits_helper::subtract(expires_at(), traits_helper::now());
288 }
289
290 /// Set the stream buffer's expiry time relative to now.
291 /**
292 * This function sets the expiry time associated with the stream. Stream
293 * operations performed after this time (where the operations cannot be
294 * completed using the internal buffers) will fail with the error
295 * boost::asio::error::operation_aborted.
296 *
297 * @param expiry_time The expiry time to be used for the timer.
298 */
299 void expires_from_now(const duration_type& expiry_time)
300 {
301 construct_timer();
302
303 boost::system::error_code ec;
304 timer_service_->expires_from_now(timer_implementation_, expiry_time, ec);
305 boost::asio::detail::throw_error(ec, "expires_from_now");
306
307 start_timer();
308 }

310protected:
311 int_type underflow()
312 {
313 if (gptr() == egptr())
314 {
315 if (timer_state_ == timer_has_expired)
316 {
317 ec_ = boost::asio::error::operation_aborted;
318 return traits_type::eof();
319 }
320
321 io_handler handler = { this };
322 this->get_service().async_receive(this->get_implementation(),
323 boost::asio::buffer(boost::asio::buffer(get_buffer_) + putback_max),
324 0, handler);
325
326 ec_ = boost::asio::error::would_block;
327 this->get_service().get_io_service().reset();
328 do this->get_service().get_io_service().run_one();
329 while (ec_ == boost::asio::error::would_block);
330 if (ec_)
331 return traits_type::eof();
332
333 setg(&get_buffer_[0], &get_buffer_[0] + putback_max,
334 &get_buffer_[0] + putback_max + bytes_transferred_);
335 return traits_type::to_int_type(*gptr());
336 }
337 else
338 {
339 return traits_type::eof();
340 }
341 }
342
343 int_type overflow(int_type c)
344 {
345 if (unbuffered_)
346 {
347 if (traits_type::eq_int_type(c, traits_type::eof()))
348 {
349 // Nothing to do.
350 return traits_type::not_eof(c);
351 }
352 else
353 {
354 if (timer_state_ == timer_has_expired)
355 {
356 ec_ = boost::asio::error::operation_aborted;
357 return traits_type::eof();
358 }
359
360 // Send the single character immediately.
361 char_type ch = traits_type::to_char_type(c);
362 io_handler handler = { this };
363 this->get_service().async_send(this->get_implementation(),
364 boost::asio::buffer(&ch, sizeof(char_type)), 0, handler);
365
366 ec_ = boost::asio::error::would_block;
367 this->get_service().get_io_service().reset();
368 do this->get_service().get_io_service().run_one();
369 while (ec_ == boost::asio::error::would_block);
370 if (ec_)
371 return traits_type::eof();
372
373 return c;
374 }
375 }
376 else
377 {
378 // Send all data in the output buffer.
379 boost::asio::const_buffer buffer =
380 boost::asio::buffer(pbase(), pptr() - pbase());
381 while (boost::asio::buffer_size(buffer) > 0)
382 {
383 if (timer_state_ == timer_has_expired)
384 {
385 ec_ = boost::asio::error::operation_aborted;
386 return traits_type::eof();
387 }
388
389 io_handler handler = { this };
390 this->get_service().async_send(this->get_implementation(),
391 boost::asio::buffer(buffer), 0, handler);
392
393 ec_ = boost::asio::error::would_block;
394 this->get_service().get_io_service().reset();
395 do this->get_service().get_io_service().run_one();
396 while (ec_ == boost::asio::error::would_block);
397 if (ec_)
398 return traits_type::eof();
399
400 buffer = buffer + bytes_transferred_;
401 }
402 setp(&put_buffer_[0], &put_buffer_[0] + put_buffer_.size());
403
404 // If the new character is eof then our work here is done.
405 if (traits_type::eq_int_type(c, traits_type::eof()))
406 return traits_type::not_eof(c);
407
408 // Add the new character to the output buffer.
409 *pptr() = traits_type::to_char_type(c);
410 pbump(1);
411 return c;
412 }
413 }

415 int sync()
416 {
417 return overflow(traits_type::eof());
418 }

420 std::streambuf* setbuf(char_type* s, std::streamsize n)
421 {
422 if (pptr() == pbase() && s == 0 && n == 0)
423 {
424 unbuffered_ = true;
425 setp(0, 0);
426 return this;
427 }
428
429 return 0;
430 }

442private:
443 void init_buffers()
444 {
445 setg(&get_buffer_[0],
446 &get_buffer_[0] + putback_max,
447 &get_buffer_[0] + putback_max);
448 if (unbuffered_)
449 setp(0, 0);
450 else
451 setp(&put_buffer_[0], &put_buffer_[0] + put_buffer_.size());
452 }

454 template <typename ResolverQuery>
455 void resolve_and_connect(const ResolverQuery& query)
456 {
457 typedef typename Protocol::resolver resolver_type;
458 typedef typename resolver_type::iterator iterator_type;
459 resolver_type resolver(detail::socket_streambuf_base::io_service_);
460 iterator_type i = resolver.resolve(query, ec_);
461 if (!ec_)
462 {
463 iterator_type end;
464 ec_ = boost::asio::error::host_not_found;
465 while (ec_ && i != end)
466 {
467 this->basic_socket<Protocol, StreamSocketService>::close(ec_);
468
469 if (timer_state_ == timer_has_expired)
470 {
471 ec_ = boost::asio::error::operation_aborted;
472 return;
473 }
474
475 io_handler handler = { this };
476 this->basic_socket<Protocol, StreamSocketService>::async_connect(
477 *i, handler);
478
479 ec_ = boost::asio::error::would_block;
480 this->get_service().get_io_service().reset();
481 do this->get_service().get_io_service().run_one();
482 while (ec_ == boost::asio::error::would_block);
483
484 ++i;
485 }
486 }

489 struct io_handler;
490 friend struct io_handler;
491 struct io_handler
492 {
493 basic_socket_streambuf* this_;
494
495 void operator()(const boost::system::error_code& ec,
496 std::size_t bytes_transferred = 0)
497 {
498 this_->ec_ = ec;
499 this_->bytes_transferred_ = bytes_transferred;
500 }
501 };
502
503 struct timer_handler;
504 friend struct timer_handler;
505 struct timer_handler
506 {
507 basic_socket_streambuf* this_;
508
509 void operator()(const boost::system::error_code&)
510 {
511 time_type now = traits_helper::now();
512
513 time_type expiry_time = this_->timer_service_->expires_at(
514 this_->timer_implementation_);
515
516 if (traits_helper::less_than(now, expiry_time))
517 {
518 this_->timer_state_ = timer_is_pending;
519 this_->timer_service_->async_wait(this_->timer_implementation_, *this);
520 }
521 else
522 {
523 this_->timer_state_ = timer_has_expired;
524 boost::system::error_code ec;
525 this_->basic_socket<Protocol, StreamSocketService>::close(ec);
526 }
527 }
528 };
529
530 void construct_timer()
531 {
532 if (timer_service_ == 0)
533 {
534 TimerService& timer_service = use_service<TimerService>(
535 detail::socket_streambuf_base::io_service_);
536 timer_service.construct(timer_implementation_);
537 timer_service_ = &timer_service;
538 }
539 }
540
541 void destroy_timer()
542 {
543 if (timer_service_)
544 timer_service_->destroy(timer_implementation_);
545 }
546
547 void start_timer()
548 {
549 if (timer_state_ != timer_is_pending)
550 {
551 timer_handler handler = { this };
552 handler(boost::system::error_code());
553 }
554 }

556 enum { putback_max = 8 };
557 enum { buffer_size = 512 };
558 boost::asio::detail::array<char, buffer_size> get_buffer_;
559 boost::asio::detail::array<char, buffer_size> put_buffer_;
560 bool unbuffered_;
561 boost::system::error_code ec_;
562 std::size_t bytes_transferred_;
563 TimerService* timer_service_;
564 typename TimerService::implementation_type timer_implementation_;
565 enum state { no_timer, timer_is_pending, timer_has_expired } timer_state_;
566};

36/// Default service implementation for a datagram socket.
37template <typename Protocol>
38class datagram_socket_service
39#if defined(GENERATING_DOCUMENTATION)
40 : public boost::asio::io_service::service
41#else
42 : public boost::asio::detail::service_base<datagram_socket_service<Protocol> >
43#endif
44{
45public:
46#if defined(GENERATING_DOCUMENTATION)
47 /// The unique service identifier.
48 static boost::asio::io_service::id id;
49#endif
50
51 /// The protocol type.
52 typedef Protocol protocol_type;
53
54 /// The endpoint type.
55 typedef typename Protocol::endpoint endpoint_type;
56
57private:
58 // The type of the platform-specific implementation.
59#if defined(BOOST_ASIO_HAS_IOCP)
60 typedef detail::win_iocp_socket_service<Protocol> service_impl_type;
61#else
62 typedef detail::reactive_socket_service<Protocol> service_impl_type;
63#endif
64
65public:
66 /// The type of a datagram socket.
67#if defined(GENERATING_DOCUMENTATION)
68 typedef implementation_defined implementation_type;
69#else
70 typedef typename service_impl_type::implementation_type implementation_type;
71#endif
72
73 /// (Deprecated: Use native_handle_type.) The native socket type.
74#if defined(GENERATING_DOCUMENTATION)
75 typedef implementation_defined native_type;
76#else
77 typedef typename service_impl_type::native_handle_type native_type;
78#endif
79
80 /// The native socket type.
81#if defined(GENERATING_DOCUMENTATION)
82 typedef implementation_defined native_handle_type;
83#else
84 typedef typename service_impl_type::native_handle_type native_handle_type;
85#endif
86
87 /// Construct a new datagram socket service for the specified io_service.
88 explicit datagram_socket_service(boost::asio::io_service& io_service)
89 : boost::asio::detail::service_base<
90 datagram_socket_service<Protocol> >(io_service),
91 service_impl_(io_service)
92 {
93 }
94
95 /// Construct a new datagram socket implementation.
96 void construct(implementation_type& impl)
97 {
98 service_impl_.construct(impl);
99 }
100
101#if defined(BOOST_ASIO_HAS_MOVE) || defined(GENERATING_DOCUMENTATION)
102 /// Move-construct a new datagram socket implementation.
103 void move_construct(implementation_type& impl,
104 implementation_type& other_impl)
105 {
106 service_impl_.move_construct(impl, other_impl);
107 }
108
109 /// Move-assign from another datagram socket implementation.
110 void move_assign(implementation_type& impl,
111 datagram_socket_service& other_service,
112 implementation_type& other_impl)
113 {
114 service_impl_.move_assign(impl, other_service.service_impl_, other_impl);
115 }
116
117 /// Move-construct a new datagram socket implementation from another protocol
118 /// type.
119 template <typename Protocol1>
120 void converting_move_construct(implementation_type& impl,
121 typename datagram_socket_service<
122 Protocol1>::implementation_type& other_impl,
123 typename enable_if<is_convertible<
124 Protocol1, Protocol>::value>::type* = 0)
125 {
126 service_impl_.template converting_move_construct<Protocol1>(
127 impl, other_impl);
128 }
129#endif // defined(BOOST_ASIO_HAS_MOVE) || defined(GENERATING_DOCUMENTATION)
130
131 /// Destroy a datagram socket implementation.
132 void destroy(implementation_type& impl)
133 {
134 service_impl_.destroy(impl);
135 }
136
137 // Open a new datagram socket implementation.
138 boost::system::error_code open(implementation_type& impl,
139 const protocol_type& protocol, boost::system::error_code& ec)
140 {
141 if (protocol.type() == SOCK_DGRAM)
142 service_impl_.open(impl, protocol, ec);
143 else
144 ec = boost::asio::error::invalid_argument;
145 return ec;
146 }
147
148 /// Assign an existing native socket to a datagram socket.
149 boost::system::error_code assign(implementation_type& impl,
150 const protocol_type& protocol, const native_handle_type& native_socket,
151 boost::system::error_code& ec)
152 {
153 return service_impl_.assign(impl, protocol, native_socket, ec);
154 }
155
156 /// Determine whether the socket is open.
157 bool is_open(const implementation_type& impl) const
158 {
159 return service_impl_.is_open(impl);
160 }
161
162 /// Close a datagram socket implementation.
163 boost::system::error_code close(implementation_type& impl,
164 boost::system::error_code& ec)
165 {
166 return service_impl_.close(impl, ec);
167 }
168
169 /// (Deprecated: Use native_handle().) Get the native socket implementation.
170 native_type native(implementation_type& impl)
171 {
172 return service_impl_.native_handle(impl);
173 }
174
175 /// Get the native socket implementation.
176 native_handle_type native_handle(implementation_type& impl)
177 {
178 return service_impl_.native_handle(impl);
179 }
180
181 /// Cancel all asynchronous operations associated with the socket.
182 boost::system::error_code cancel(implementation_type& impl,
183 boost::system::error_code& ec)
184 {
185 return service_impl_.cancel(impl, ec);
186 }
187
188 /// Determine whether the socket is at the out-of-band data mark.
189 bool at_mark(const implementation_type& impl,
190 boost::system::error_code& ec) const
191 {
192 return service_impl_.at_mark(impl, ec);
193 }
194
195 /// Determine the number of bytes available for reading.
196 std::size_t available(const implementation_type& impl,
197 boost::system::error_code& ec) const
198 {
199 return service_impl_.available(impl, ec);
200 }
201
202 // Bind the datagram socket to the specified local endpoint.
203 boost::system::error_code bind(implementation_type& impl,
204 const endpoint_type& endpoint, boost::system::error_code& ec)
205 {
206 return service_impl_.bind(impl, endpoint, ec);
207 }
208
209 /// Connect the datagram socket to the specified endpoint.
210 boost::system::error_code connect(implementation_type& impl,
211 const endpoint_type& peer_endpoint, boost::system::error_code& ec)
212 {
213 return service_impl_.connect(impl, peer_endpoint, ec);
214 }

216 /// Start an asynchronous connect.
217 template <typename ConnectHandler>
218 BOOST_ASIO_INITFN_RESULT_TYPE(ConnectHandler,
219 void (boost::system::error_code))
220 async_connect(implementation_type& impl,
221 const endpoint_type& peer_endpoint,
222 BOOST_ASIO_MOVE_ARG(ConnectHandler) handler)
223 {
224 detail::async_result_init<
225 ConnectHandler, void (boost::system::error_code)> init(
226 BOOST_ASIO_MOVE_CAST(ConnectHandler)(handler));
227
228 service_impl_.async_connect(impl, peer_endpoint, init.handler);
229
230 return init.result.get();
231 }

19{
20
21// Defines the same pointer type (raw or smart) to another pointee type
22
23template<class T, class U>
24struct pointer_to_other;
25
26template<class T, class U,
27 template<class> class Sp>
28struct pointer_to_other< Sp<T>, U >
29{
30 typedef Sp<U> type;
31};
32
33template<class T, class T2, class U,
34 template<class, class> class Sp>
35struct pointer_to_other< Sp<T, T2>, U >
36{
37 typedef Sp<U, T2> type;
38};
39
40template<class T, class T2, class T3, class U,
41 template<class, class, class> class Sp>
42struct pointer_to_other< Sp<T, T2, T3>, U >
43{
44 typedef Sp<U, T2, T3> type;
45};
46
47template<class T, class U>
48struct pointer_to_other< T*, U >
49{
50 typedef U* type;
51};

96 BOOST_THREAD_NO_COPYABLE(mutex)

98 mutex()
99 {
100 int const res=pthread_mutex_init(&m,NULL);
101 if(res)
102 {
103 boost::throw_exception(thread_resource_error(res, "boost:: mutex constructor failed in pthread_mutex_init"));
104 }
105 }
106 ~mutex()
107 {
108 BOOST_VERIFY(!posix::pthread_mutex_destroy(&m));
109 }
110
111 void lock()
112 {
113 int res = posix::pthread_mutex_lock(&m);
114 if (res)
115 {
116 boost::throw_exception(lock_error(res,"boost: mutex lock failed in pthread_mutex_lock"));
117 }
118 }
119
120 void unlock()
121 {
122 int res = posix::pthread_mutex_unlock(&m);
123 if (res)
124 {
125 boost::throw_exception(lock_error(res,"boost: mutex unlock failed in pthread_mutex_unlock"));
126 }
127 }

129 bool try_lock()
130 {
131 int res;
132 do
133 {
134 res = pthread_mutex_trylock(&m);
135 } while (res == EINTR);
136 if (res==EBUSY)
137 {
138 return false;
139 }
140
141 return !res;
142 }

159 class timed_mutex
160 {
161 private:
162 pthread_mutex_t m;
163#ifndef BOOST_PTHREAD_HAS_TIMEDLOCK
164 pthread_cond_t cond;
165 bool is_locked;
166#endif
167 public:
168 BOOST_THREAD_NO_COPYABLE(timed_mutex)
169 timed_mutex()
170 {
171 int const res=pthread_mutex_init(&m,NULL);
172 if(res)
173 {
174 boost::throw_exception(thread_resource_error(res, "boost:: timed_mutex constructor failed in pthread_mutex_init"));
175 }
176#ifndef BOOST_PTHREAD_HAS_TIMEDLOCK
177 int const res2=pthread_cond_init(&cond,NULL);
178 if(res2)
179 {
180 BOOST_VERIFY(!posix::pthread_mutex_destroy(&m));
181 //BOOST_VERIFY(!pthread_mutex_destroy(&m));
182 boost::throw_exception(thread_resource_error(res2, "boost:: timed_mutex constructor failed in pthread_cond_init"));
183 }
184 is_locked=false;
185#endif
186 }
187 ~timed_mutex()
188 {
189 BOOST_VERIFY(!posix::pthread_mutex_destroy(&m));
190#ifndef BOOST_PTHREAD_HAS_TIMEDLOCK
191 BOOST_VERIFY(!pthread_cond_destroy(&cond));
192#endif
193 }

279 private:
280 bool do_try_lock_until(struct timespec const &timeout)
281 {
282 boost::pthread::pthread_mutex_scoped_lock const local_lock(&m);
283 while(is_locked)
284 {
285 int const cond_res=pthread_cond_timedwait(&cond,&m,&timeout);
286 if(cond_res==ETIMEDOUT)
287 {
288 return false;
289 }
290 BOOST_ASSERT(!cond_res);
291 }
292 is_locked=true;
293 return true;

309 return try_lock_until(chrono::steady_clock::now() + rel_time);

311 template <class Clock, class Duration>
312 bool try_lock_until(const chrono::time_point<Clock, Duration>& t)
313 {
314 using namespace chrono;
315 system_clock::time_point s_now = system_clock::now();
316 typename Clock::time_point c_now = Clock::now();
317 return try_lock_until(s_now + ceil<nanoseconds>(t - c_now));
318 }
319 template <class Duration>
320 bool try_lock_until(const chrono::time_point<chrono::system_clock, Duration>& t)
321 {
322 using namespace chrono;
323 typedef time_point<system_clock, nanoseconds> nano_sys_tmpt;
324 return try_lock_until(nano_sys_tmpt(ceil<nanoseconds>(t.time_since_epoch())));
325 }

327 {
328 //using namespace chrono;
329 chrono::nanoseconds d = tp.time_since_epoch();

39//
40// Structures used to build the tuple of actual arguments. The
41// tuple is a nested cons-style list of arg_list specializations
42// terminated by an empty_arg_list.
43//
44// Each specialization of arg_list is derived from its successor in
45// the list type. This feature is used along with using
46// declarations to build member function overload sets that can
47// match against keywords.
48//
49
50// MPL sequence support
51struct arg_list_tag;

53// Terminates arg_list<> and represents an empty list. Since this
54// is just the terminating case you might want to look at arg_list
55// first, to get a feel for what's really happening here.

58{
59 empty_arg_list() {}

61 // Constructor taking BOOST_PARAMETER_MAX_ARITY empty_arg_list
62 // arguments; this makes initialization
63 empty_arg_list(
64 BOOST_PP_ENUM_PARAMS(
65 BOOST_PARAMETER_MAX_ARITY, void_ BOOST_PP_INTERCEPT
66 ))
67 {}

168// A tuple of tagged arguments, terminated with empty_arg_list.
169// Every TaggedArg is an instance of tagged_argument<>.
170template <class TaggedArg, class Next = empty_arg_list>
171struct arg_list : Next
172{
173 typedef arg_list<TaggedArg,Next> self;
174 typedef typename TaggedArg::key_type key_type;
175
176 typedef typename is_maybe<typename TaggedArg::value_type>::type holds_maybe;
177
178 typedef typename mpl::eval_if<
179 holds_maybe
180 , get_reference<typename TaggedArg::value_type>
181 , get_reference<TaggedArg>
182 >::type reference;
183
184 typedef typename mpl::if_<
185 holds_maybe
186 , reference
187 , typename TaggedArg::value_type
188 >::type value_type;
189
190 TaggedArg arg; // Stores the argument
191
192 // Store the arguments in successive nodes of this list
193 template< // class A0, class A1, ...
194 BOOST_PP_ENUM_PARAMS(BOOST_PARAMETER_MAX_ARITY, class A)
195 >
196 arg_list( // A0& a0, A1& a1, ...
197 BOOST_PP_ENUM_BINARY_PARAMS(BOOST_PARAMETER_MAX_ARITY, A, & a)
198 )
199 : Next( // a1, a2, ...
200 BOOST_PP_ENUM_SHIFTED_PARAMS(BOOST_PARAMETER_MAX_ARITY, a)
201 , void_reference()
202 )
203 , arg(a0)
204 {}
205
206 // Create a new list by prepending arg to a copy of tail. Used
207 // when incrementally building this structure with the comma
208 // operator.
209 arg_list(TaggedArg head, Next const& tail)
210 : Next(tail)
211 , arg(head)
212 {}

214 // A metafunction class that, given a keyword and a default
215 // type, returns the appropriate result type for a keyword
216 // lookup given that default
217 struct binding
218 {
219 template <class KW, class Default, class Reference>
220 struct apply
221 {
222 typedef typename mpl::eval_if<
223 boost::is_same<KW, key_type>
224 , mpl::if_<Reference, reference, value_type>
225 , mpl::apply_wrap3<typename Next::binding, KW, Default, Reference>
226 >::type type;
227 };
228 };

276 // A metafunction class that, given a keyword, returns the base
277 // sublist whose get() function can produce the value for that
278 // key.
279 struct key_owner
280 {
281 template<class KW>
282 struct apply
283 {
284 typedef typename mpl::eval_if<
285 boost::is_same<KW, key_type>
286 , mpl::identity<arg_list<TaggedArg,Next> >
287 , mpl::apply_wrap1<typename Next::key_owner,KW>
288 >::type type;
289 };
290 };

292 // Outer indexing operators that dispatch to the right node's
293 // get() function.
294 template <class KW>
295 typename mpl::apply_wrap3<binding, KW, void_, mpl::true_>::type
296 operator[](keyword<KW> const& x) const
297 {
298 typename mpl::apply_wrap1<key_owner, KW>::type const& sublist = *this;
299 return sublist.get(x);
300 }
301
302 template <class KW, class Default>
303 typename mpl::apply_wrap3<binding, KW, Default&, mpl::true_>::type
304 operator[](default_<KW, Default> x) const
305 {
306 typename mpl::apply_wrap1<key_owner, KW>::type const& sublist = *this;
307 return sublist.get(x);
308 }
309
310 template <class KW, class F>
311 typename mpl::apply_wrap3<
312 binding,KW
313 , typename result_of0<F>::type
314 , mpl::true_
315 >::type
316 operator[](lazy_default<KW,F> x) const
317 {
318 typename mpl::apply_wrap1<key_owner, KW>::type const& sublist = *this;
319 return sublist.get(x);
320 }

322 // These just return the stored value; when empty_arg_list is
323 // reached, indicating no matching argument was passed, the
324 // default is returned, or if no default_ or lazy_default was
325 // passed, compilation fails.
326 reference get(keyword<key_type> const&) const
327 {
328 BOOST_MPL_ASSERT_NOT((holds_maybe));
329 return arg.value;
330 }
331
332 template <class Default>
333 reference get(default_<key_type,Default> const& d) const
334 {
335 return get_default(d, holds_maybe());

339 reference get(lazy_default<key_type, Default>) const

341 return arg.value;

395 // Comma operator to compose argument list without using parameters<>.
396 // Useful for argument lists with undetermined length.
397 template <class KW, class T2>
398 arg_list<tagged_argument<KW, T2>, self>
399 operator,(tagged_argument<KW,T2> x) const
400 {
401 return arg_list<tagged_argument<KW,T2>, self>(x, *this);
402 }
403

405 typedef self type; // Convenience for users
406 typedef Next tail_type; // For the benefit of iterators
407 typedef arg_list_tag tag; // For dispatching to sequence intrinsics

159template<typename Mode, typename Ch, typename T>
160struct resolve_traits {
161 // Note: test for is_iterator_range must come before test for output
162 // iterator.
163 typedef typename
164 iostreams::select< // Disambiguation for Tru64.
165 is_std_io<T>,
166 mode_adapter<Mode, T>,
167 is_iterator_range<T>,
168 range_adapter<Mode, T>,
169 is_dereferenceable<T>,
170 output_iterator_adapter<Mode, Ch, T>,
171 is_array<T>,
172 array_adapter<Mode, T>,
173 else_,
174 #if !BOOST_WORKAROUND(__BORLANDC__, < 0x600)
175 const T&
176 #else
177 T
178 #endif
179 >::type type;
180};
181
182template<typename Mode, typename Ch, typename T>
183typename resolve_traits<Mode, Ch, T>::type
184resolve(const T& t, mpl::true_)
185{ // Bad overload resolution.
186 typedef typename resolve_traits<Mode, Ch, T>::type return_type;
187 return return_type(wrap(const_cast<T&>(t)));
188}

46// ratio
47template <boost::intmax_t N, boost::intmax_t D = 1> class ratio;
48
49// ratio arithmetic
50template <class R1, class R2> struct ratio_add;
51template <class R1, class R2> struct ratio_subtract;
52template <class R1, class R2> struct ratio_multiply;
53template <class R1, class R2> struct ratio_divide;
54
55// ratio comparison
56template <class R1, class R2> struct ratio_equal;
57template <class R1, class R2> struct ratio_not_equal;
58template <class R1, class R2> struct ratio_less;
59template <class R1, class R2> struct ratio_less_equal;
60template <class R1, class R2> struct ratio_greater;
61template <class R1, class R2> struct ratio_greater_equal;

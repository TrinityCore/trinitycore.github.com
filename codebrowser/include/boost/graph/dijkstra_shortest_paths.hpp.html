41 /**
42 * @brief Updates a particular value in a queue used by Dijkstra's
43 * algorithm.
44 *
45 * This routine is called by Dijkstra's algorithm after it has
46 * decreased the distance from the source vertex to the given @p
47 * vertex. By default, this routine will just call @c
48 * Q.update(vertex). However, other queues may provide more
49 * specialized versions of this routine.
50 *
51 * @param Q the queue that will be updated.
52 * @param vertex the vertex whose distance has been updated
53 * @param old_distance the previous distance to @p vertex
54 */
55 template<typename Buffer, typename Vertex, typename DistanceType>
56 inline void
57 dijkstra_queue_update(Buffer& Q, Vertex vertex, DistanceType old_distance)
58 {
59 (void)old_distance;
60 Q.update(vertex);
61 }

69 template <class Visitor, class Graph>
70 struct DijkstraVisitorConcept {
71 void constraints() {
72 BOOST_CONCEPT_ASSERT(( CopyConstructibleConcept<Visitor> ));
73 vis.initialize_vertex(u, g);
74 vis.discover_vertex(u, g);
75 vis.examine_vertex(u, g);
76 vis.examine_edge(e, g);
77 vis.edge_relaxed(e, g);
78 vis.edge_not_relaxed(e, g);
79 vis.finish_vertex(u, g);
80 }
81 Visitor vis;
82 Graph g;
83 typename graph_traits<Graph>::vertex_descriptor u;
84 typename graph_traits<Graph>::edge_descriptor e;
85 };

87 template <class Visitors = null_visitor>
88 class dijkstra_visitor : public bfs_visitor<Visitors> {
89 public:
90 dijkstra_visitor() { }
91 dijkstra_visitor(Visitors vis)
92 : bfs_visitor<Visitors>(vis) { }

94 template <class Edge, class Graph>
95 void edge_relaxed(Edge e, Graph& g) {
96 invoke_visitors(this->m_vis, e, g, on_edge_relaxed());
97 }
98 template <class Edge, class Graph>
99 void edge_not_relaxed(Edge e, Graph& g) {
100 invoke_visitors(this->m_vis, e, g, on_edge_not_relaxed());
101 }
102 private:
103 template <class Edge, class Graph>
104 void tree_edge(Edge u, Graph& g) { }
105 };

115 template <class UniformCostVisitor, class UpdatableQueue,
116 class WeightMap, class PredecessorMap, class DistanceMap,
117 class BinaryFunction, class BinaryPredicate>
118 struct dijkstra_bfs_visitor
119 {
120 typedef typename property_traits<DistanceMap>::value_type D;
121
122 dijkstra_bfs_visitor(UniformCostVisitor vis, UpdatableQueue& Q,
123 WeightMap w, PredecessorMap p, DistanceMap d,
124 BinaryFunction combine, BinaryPredicate compare,
125 D zero)
126 : m_vis(vis), m_Q(Q), m_weight(w), m_predecessor(p), m_distance(d),
127 m_combine(combine), m_compare(compare), m_zero(zero) { }
128
129 template <class Edge, class Graph>
130 void tree_edge(Edge e, Graph& g) {
131 bool decreased = relax(e, g, m_weight, m_predecessor, m_distance,
132 m_combine, m_compare);
133 if (decreased)
134 m_vis.edge_relaxed(e, g);
135 else
136 m_vis.edge_not_relaxed(e, g);
137 }
138 template <class Edge, class Graph>
139 void gray_target(Edge e, Graph& g) {
140 D old_distance = get(m_distance, target(e, g));
141
142 bool decreased = relax(e, g, m_weight, m_predecessor, m_distance,
143 m_combine, m_compare);
144 if (decreased) {
145 dijkstra_queue_update(m_Q, target(e, g), old_distance);
146 m_vis.edge_relaxed(e, g);
147 } else
148 m_vis.edge_not_relaxed(e, g);
149 }
150
151 template <class Vertex, class Graph>
152 void initialize_vertex(Vertex u, Graph& g)
153 { m_vis.initialize_vertex(u, g); }
154 template <class Edge, class Graph>
155 void non_tree_edge(Edge, Graph&) { }
156 template <class Vertex, class Graph>
157 void discover_vertex(Vertex u, Graph& g) { m_vis.discover_vertex(u, g); }
158 template <class Vertex, class Graph>
159 void examine_vertex(Vertex u, Graph& g) { m_vis.examine_vertex(u, g); }
160 template <class Edge, class Graph>
161 void examine_edge(Edge e, Graph& g) {
162 // Comparison needs to be more complicated because distance and weight
163 // types may not be the same; see bug 8398
164 // (https://svn.boost.org/trac/boost/ticket/8398)
165 D source_dist = get(m_distance, source(e, g));
166 if (m_compare(m_combine(source_dist, get(m_weight, e)), source_dist))
167 boost::throw_exception(negative_edge());
168 m_vis.examine_edge(e, g);
169 }
170 template <class Edge, class Graph>
171 void black_target(Edge, Graph&) { }
172 template <class Vertex, class Graph>
173 void finish_vertex(Vertex u, Graph& g) { m_vis.finish_vertex(u, g); }

177 WeightMap m_weight;
178 PredecessorMap m_predecessor;
179 DistanceMap m_distance;
180 BinaryFunction m_combine;
181 BinaryPredicate m_compare;
182 D m_zero;
183 };

191 template <class Graph, class IndexMap, class Value>
192 struct vertex_property_map_generator_helper<Graph, IndexMap, Value, true> {

195 array_holder.reset(new Value[num_vertices(g)]);
196 std::fill(array_holder.get(), array_holder.get() + num_vertices(g), Value());
197 return make_iterator_property_map(array_holder.get(), index);
198 }

283 // Call breadth first search
284 template <class Graph, class DijkstraVisitor,
285 class PredecessorMap, class DistanceMap,
286 class WeightMap, class IndexMap, class Compare, class Combine,
287 class DistZero, class ColorMap>
288 inline void
289 dijkstra_shortest_paths_no_init
290 (const Graph& g,
291 typename graph_traits<Graph>::vertex_descriptor s,
292 PredecessorMap predecessor, DistanceMap distance, WeightMap weight,
293 IndexMap index_map,
294 Compare compare, Combine combine, DistZero zero,
295 DijkstraVisitor vis, ColorMap color)
296 {
297 typedef indirect_cmp<DistanceMap, Compare> IndirectCmp;
298 IndirectCmp icmp(distance, compare);
299
300 typedef typename graph_traits<Graph>::vertex_descriptor Vertex;
301
302#ifdef BOOST_GRAPH_DIJKSTRA_TESTING
303 if (!dijkstra_relaxed_heap) {
304 typedef mutable_queue<Vertex, std::vector<Vertex>, IndirectCmp, IndexMap>
305 MutableQueue;
306
307 MutableQueue Q(num_vertices(g), icmp, index_map);
308 detail::dijkstra_bfs_visitor<DijkstraVisitor, MutableQueue, WeightMap,
309 PredecessorMap, DistanceMap, Combine, Compare>
310 bfs_vis(vis, Q, weight, predecessor, distance, combine, compare, zero);
311
312 breadth_first_visit(g, s, Q, bfs_vis, color);
313 return;
314 }
315#endif // BOOST_GRAPH_DIJKSTRA_TESTING
316
317#ifdef BOOST_GRAPH_DIJKSTRA_USE_RELAXED_HEAP
318 typedef relaxed_heap<Vertex, IndirectCmp, IndexMap> MutableQueue;
319 MutableQueue Q(num_vertices(g), icmp, index_map);
320#else // Now the default: use a d-ary heap
321 boost::scoped_array<std::size_t> index_in_heap_map_holder;
322 typedef
323 detail::vertex_property_map_generator<Graph, IndexMap, std::size_t>
324 IndexInHeapMapHelper;
325 typedef typename IndexInHeapMapHelper::type IndexInHeapMap;
326 IndexInHeapMap index_in_heap =
327 IndexInHeapMapHelper::build(g, index_map, index_in_heap_map_holder);
328 typedef d_ary_heap_indirect<Vertex, 4, IndexInHeapMap, DistanceMap, Compare>
329 MutableQueue;
330 MutableQueue Q(distance, index_in_heap, compare);
331#endif // Relaxed heap
332
333 detail::dijkstra_bfs_visitor<DijkstraVisitor, MutableQueue, WeightMap,
334 PredecessorMap, DistanceMap, Combine, Compare>
335 bfs_vis(vis, Q, weight, predecessor, distance, combine, compare, zero);
336
337 breadth_first_visit(g, s, Q, bfs_vis, color);
338 }

363 // Initialize distances and call breadth first search
364 template <class VertexListGraph, class DijkstraVisitor,
365 class PredecessorMap, class DistanceMap,
366 class WeightMap, class IndexMap, class Compare, class Combine,
367 class DistInf, class DistZero, class ColorMap>
368 inline void
369 dijkstra_shortest_paths
370 (const VertexListGraph& g,
371 typename graph_traits<VertexListGraph>::vertex_descriptor s,
372 PredecessorMap predecessor, DistanceMap distance, WeightMap weight,
373 IndexMap index_map,
374 Compare compare, Combine combine, DistInf inf, DistZero zero,
375 DijkstraVisitor vis, ColorMap color)
376 {
377 typedef typename property_traits<ColorMap>::value_type ColorValue;
378 typedef color_traits<ColorValue> Color;
379 typename graph_traits<VertexListGraph>::vertex_iterator ui, ui_end;
380 for (boost::tie(ui, ui_end) = vertices(g); ui != ui_end; ++ui) {
381 vis.initialize_vertex(*ui, g);
382 put(distance, *ui, inf);
383 put(predecessor, *ui, *ui);
384 put(color, *ui, Color::white());
385 }
386 put(distance, s, zero);
387
388 dijkstra_shortest_paths_no_init(g, s, predecessor, distance, weight,
389 index_map, compare, combine, zero, vis, color);
390 }

413 // Handle defaults for PredecessorMap and
414 // Distance Compare, Combine, Inf and Zero
415 template <class VertexListGraph, class DistanceMap, class WeightMap,
416 class IndexMap, class Params>
417 inline void
418 dijkstra_dispatch2
419 (const VertexListGraph& g,
420 typename graph_traits<VertexListGraph>::vertex_descriptor s,
421 DistanceMap distance, WeightMap weight, IndexMap index_map,
422 const Params& params)
423 {
424 // Default for predecessor map
425 dummy_property_map p_map;
426
427 typedef typename property_traits<DistanceMap>::value_type D;
428 D inf = choose_param(get_param(params, distance_inf_t()),
429 (std::numeric_limits<D>::max)());
430
431 dijkstra_shortest_paths
432 (g, s,
433 choose_param(get_param(params, vertex_predecessor), p_map),
434 distance, weight, index_map,
435 choose_param(get_param(params, distance_compare_t()),
436 std::less<D>()),
437 choose_param(get_param(params, distance_combine_t()),
438 closed_plus<D>(inf)),
439 inf,
440 choose_param(get_param(params, distance_zero_t()),
441 D()),
442 choose_param(get_param(params, graph_visitor),
443 make_dijkstra_visitor(null_visitor())),
444 params);
445 }

447 template <class VertexListGraph, class DistanceMap, class WeightMap,
448 class IndexMap, class Params>
449 inline void
450 dijkstra_dispatch1
451 (const VertexListGraph& g,
452 typename graph_traits<VertexListGraph>::vertex_descriptor s,
453 DistanceMap distance, WeightMap weight, IndexMap index_map,
454 const Params& params)
455 {
456 // Default for distance map
457 typedef typename property_traits<WeightMap>::value_type D;
458 typename std::vector<D>::size_type
459 n = is_default_param(distance) ? num_vertices(g) : 1;
460 std::vector<D> distance_map(n);
461
462 detail::dijkstra_dispatch2
463 (g, s, choose_param(distance, make_iterator_property_map
464 (distance_map.begin(), index_map,
465 distance_map[0])),
466 weight, index_map, params);
467 }

470 // Named Parameter Variant
471 template <class VertexListGraph, class Param, class Tag, class Rest>
472 inline void
473 dijkstra_shortest_paths
474 (const VertexListGraph& g,
475 typename graph_traits<VertexListGraph>::vertex_descriptor s,
476 const bgl_named_params<Param,Tag,Rest>& params)
477 {
478 // Default for edge weight and vertex index map is to ask for them
479 // from the graph. Default for the visitor is null_visitor.
480 detail::dijkstra_dispatch1
481 (g, s,
482 get_param(params, vertex_distance),
483 choose_const_pmap(get_param(params, edge_weight), g, edge_weight),
484 choose_const_pmap(get_param(params, vertex_index), g, vertex_index),
485 params);
486 }

115 template <typename T, typename Tag, typename Base = no_property>
116 struct bgl_named_params
117 {
118 typedef bgl_named_params self;
119 typedef Base next_type;
120 typedef Tag tag_type;
121 typedef T value_type;
122 bgl_named_params(T v = T()) : m_value(v) { }
123 bgl_named_params(T v, const Base& b) : m_value(v), m_base(b) { }
124 T m_value;
125 Base m_base;
126
127#define BOOST_BGL_ONE_PARAM_REF(name, key) \
128 template <typename PType> \
129 bgl_named_params<boost::reference_wrapper<PType>, BOOST_PP_CAT(key, _t), self> \
130 name(PType& p) const { \
131 typedef bgl_named_params<boost::reference_wrapper<PType>, BOOST_PP_CAT(key, _t), self> Params; \
132 return Params(boost::ref(p), *this); \
133 } \
134
135#define BOOST_BGL_ONE_PARAM_CREF(name, key) \
136 template <typename PType> \
137 bgl_named_params<PType, BOOST_PP_CAT(key, _t), self> \
138 name(const PType& p) const { \
139 typedef bgl_named_params<PType, BOOST_PP_CAT(key, _t), self> Params; \
140 return Params(p, *this); \
141 } \
142
143BOOST_BGL_DECLARE_NAMED_PARAMS
144
145#undef BOOST_BGL_ONE_PARAM_REF
146#undef BOOST_BGL_ONE_PARAM_CREF

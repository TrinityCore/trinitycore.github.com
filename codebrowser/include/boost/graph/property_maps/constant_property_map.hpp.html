18/**
19 * A constant property is one, that regardless of the edge or vertex given,
20 * will always return a constant value.
21 */
22template <typename Key, typename Value>
23struct constant_property_map
24 : public boost::put_get_helper<
25 const Value&,
26 constant_property_map<Key, Value>
27 >
28{
29 typedef Key key_type;
30 typedef Value value_type;
31 typedef const Value& reference;
32 typedef boost::readable_property_map_tag category;
33
34 constant_property_map()
35 : m_value()
36 { }
37
38 constant_property_map(const value_type &value)
39 : m_value(value)
40 { }
41
42 constant_property_map(const constant_property_map& copy)
43 : m_value(copy.m_value)
44 { }
45
46 inline reference operator [](const key_type&) const
47 { return m_value; }
48
49 value_type m_value;
50};
51
52template <typename Key, typename Value>
53inline constant_property_map<Key, Value>
54make_constant_property(const Value& value)
55{ return constant_property_map<Key, Value>(value); }
56
57/**
58 * Same as above, but pretends to be writable as well.
59 */
60template <typename Key, typename Value>
61struct constant_writable_property_map {
62 typedef Key key_type;
63 typedef Value value_type;
64 typedef Value& reference;
65 typedef boost::read_write_property_map_tag category;
66
67 constant_writable_property_map()
68 : m_value()
69 { }
70
71 constant_writable_property_map(const value_type &value)
72 : m_value(value)
73 { }
74
75 constant_writable_property_map(const constant_writable_property_map& copy)
76 : m_value(copy.m_value)
77 { }
78
79 friend Value get(const constant_writable_property_map& me, Key) {return me.m_value;}
80 friend void put(const constant_writable_property_map&, Key, Value) {}
81
82 value_type m_value;
83};
84
85template <typename Key, typename Value>
86inline constant_writable_property_map<Key, Value>
87make_constant_writable_property(const Value& value)
88{ return constant_writable_property_map<Key, Value>(value); }

35 template <class Visitor, class Graph>
36 struct BFSVisitorConcept {
37 void constraints() {
38 BOOST_CONCEPT_ASSERT(( CopyConstructibleConcept<Visitor> ));
39 vis.initialize_vertex(u, g);
40 vis.discover_vertex(u, g);
41 vis.examine_vertex(u, g);
42 vis.examine_edge(e, g);
43 vis.tree_edge(e, g);
44 vis.non_tree_edge(e, g);
45 vis.gray_target(e, g);
46 vis.black_target(e, g);
47 vis.finish_vertex(u, g);
48 }
49 Visitor vis;
50 Graph g;
51 typename graph_traits<Graph>::vertex_descriptor u;
52 typename graph_traits<Graph>::edge_descriptor e;
53 };

56 // Multiple-source version
57 template <class IncidenceGraph, class Buffer, class BFSVisitor,
58 class ColorMap, class SourceIterator>
59 void breadth_first_visit
60 (const IncidenceGraph& g,
61 SourceIterator sources_begin, SourceIterator sources_end,
62 Buffer& Q, BFSVisitor vis, ColorMap color)
63 {
64 BOOST_CONCEPT_ASSERT(( IncidenceGraphConcept<IncidenceGraph> ));
65 typedef graph_traits<IncidenceGraph> GTraits;
66 typedef typename GTraits::vertex_descriptor Vertex;
67 typedef typename GTraits::edge_descriptor Edge;
68 BOOST_CONCEPT_ASSERT(( BFSVisitorConcept<BFSVisitor, IncidenceGraph> ));
69 BOOST_CONCEPT_ASSERT(( ReadWritePropertyMapConcept<ColorMap, Vertex> ));
70 typedef typename property_traits<ColorMap>::value_type ColorValue;
71 typedef color_traits<ColorValue> Color;
72 typename GTraits::out_edge_iterator ei, ei_end;
73
74 for (; sources_begin != sources_end; ++sources_begin) {
75 Vertex s = *sources_begin;
76 put(color, s, Color::gray()); vis.discover_vertex(s, g);
77 Q.push(s);
78 }
79 while (! Q.empty()) {
80 Vertex u = Q.top(); Q.pop(); vis.examine_vertex(u, g);
81 for (boost::tie(ei, ei_end) = out_edges(u, g); ei != ei_end; ++ei) {
82 Vertex v = target(*ei, g); vis.examine_edge(*ei, g);
83 ColorValue v_color = get(color, v);
84 if (v_color == Color::white()) { vis.tree_edge(*ei, g);
85 put(color, v, Color::gray()); vis.discover_vertex(v, g);
86 Q.push(v);
87 } else { vis.non_tree_edge(*ei, g);
88 if (v_color == Color::gray()) vis.gray_target(*ei, g);
89 else vis.black_target(*ei, g);
90 }
91 } // end for
92 put(color, u, Color::black()); vis.finish_vertex(u, g);
93 } // end while
94 } // breadth_first_visit

100 (const IncidenceGraph& g,

109 template <class VertexListGraph, class SourceIterator,
110 class Buffer, class BFSVisitor,
111 class ColorMap>
112 void breadth_first_search
113 (const VertexListGraph& g,
114 SourceIterator sources_begin, SourceIterator sources_end,
115 Buffer& Q, BFSVisitor vis, ColorMap color)
116 {
117 // Initialization
118 typedef typename property_traits<ColorMap>::value_type ColorValue;
119 typedef color_traits<ColorValue> Color;
120 typename boost::graph_traits<VertexListGraph>::vertex_iterator i, i_end;
121 for (boost::tie(i, i_end) = vertices(g); i != i_end; ++i) {
122 vis.initialize_vertex(*i, g);
123 put(color, *i, Color::white());
124 }
125 breadth_first_visit(g, sources_begin, sources_end, Q, vis, color);
126 }

279 // Choose between default color and color parameters. Using
280 // function dispatching so that we don't require vertex index if
281 // the color default is not being used.
282
283 template <class ColorMap>
284 struct bfs_dispatch {
285 template <class VertexListGraph, class P, class T, class R>
286 static void apply
287 (VertexListGraph& g,
288 typename graph_traits<VertexListGraph>::vertex_descriptor s,
289 const bgl_named_params<P, T, R>& params,
290 ColorMap color)
291 {
292 bfs_helper
293 (g, s, color,
294 choose_param(get_param(params, graph_visitor),
295 make_bfs_visitor(null_visitor())),
296 params);
297 }
298 };
299
300 template <>
301 struct bfs_dispatch<param_not_found> {
302 template <class VertexListGraph, class P, class T, class R>
303 static void apply
304 (VertexListGraph& g,
305 typename graph_traits<VertexListGraph>::vertex_descriptor s,
306 const bgl_named_params<P, T, R>& params,
307 param_not_found)
308 {
309 null_visitor null_vis;
310
311 bfs_helper
312 (g, s,
313 make_two_bit_color_map
314 (num_vertices(g),
315 choose_const_pmap(get_param(params, vertex_index),
316 g, vertex_index)),
317 choose_param(get_param(params, graph_visitor),
318 make_bfs_visitor(null_vis)),
319 params);
320 }
321 };

347 template <class IncidenceGraph, class P, class T, class R>
348 void breadth_first_visit
349 (const IncidenceGraph& g,
350 typename graph_traits<IncidenceGraph>::vertex_descriptor s,
351 const bgl_named_params<P, T, R>& params)
352 {
353 // The graph is passed by *const* reference so that graph adaptors
354 // (temporaries) can be passed into this function. However, the
355 // graph is not really const since we may write to property maps
356 // of the graph.
357 IncidenceGraph& ng = const_cast<IncidenceGraph&>(g);
358
359 typedef graph_traits<IncidenceGraph> Traits;
360 // Buffer default
361 typedef typename Traits::vertex_descriptor vertex_descriptor;
362 typedef boost::queue<vertex_descriptor> queue_t;
363 queue_t Q;
364
365 breadth_first_visit
366 (ng, s,
367 choose_param(get_param(params, buffer_param_t()), boost::ref(Q)).get(),
368 choose_param(get_param(params, graph_visitor),
369 make_bfs_visitor(null_visitor())),
370 choose_pmap(get_param(params, vertex_color), ng, vertex_color)
371 );

376 template <typename Graph, typename Source>
377 struct breadth_first_search_impl {
378 typedef void result_type;
379 template <typename ArgPack>
380 void operator()(const Graph& g, const Source& source, const ArgPack& arg_pack) {
381 using namespace boost::graph::keywords;
382 typename boost::graph_traits<Graph>::vertex_descriptor sources[1] = {source};
383 boost::queue<typename boost::graph_traits<Graph>::vertex_descriptor> Q;
384 boost::breadth_first_search(g,
385 &sources[0],
386 &sources[1],
387 boost::unwrap_ref(arg_pack[_buffer | boost::ref(Q)]),
388 arg_pack[_visitor | make_bfs_visitor(null_visitor())],
389 boost::detail::make_color_map_from_arg_pack(g, arg_pack));
390 }
391 };
392 }
393 BOOST_GRAPH_MAKE_FORWARDING_FUNCTION(breadth_first_search, 2, 4)
394 }

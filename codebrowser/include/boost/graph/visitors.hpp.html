104 // The invoke_visitors() function
105
106 namespace detail {
107 template <class Visitor, class T, class Graph>
108 inline void invoke_dispatch(Visitor& v, T x, Graph& g, mpl::true_) {
109 v(x, g);
110 }
111
112 template <class Visitor, class T, class Graph>
113 inline void invoke_dispatch(Visitor&, T, Graph&, mpl::false_)
114 { }
115 } // namespace detail
116
117 template <class Visitor, class Rest, class T, class Graph, class Tag>
118 inline void
119 invoke_visitors(std::pair<Visitor, Rest>& vlist, T x, Graph& g, Tag tag) {
120 typedef typename Visitor::event_filter Category;
121 typedef typename is_same<Category, Tag>::type IsSameTag;
122 detail::invoke_dispatch(vlist.first, x, g, IsSameTag());
123 invoke_visitors(vlist.second, x, g, tag);
124 }
125#if defined(BOOST_MSVC) && BOOST_MSVC <= 1300
126 template <class Visitor, class T, class Graph, class Tag>
127 inline void
128 invoke_visitors(base_visitor<Visitor>& vis, T x, Graph& g, Tag) {
129 typedef typename Visitor::event_filter Category;
130 typedef typename is_same<Category, Tag>::type IsSameTag;
131 Visitor& v = static_cast<Visitor&>(vis);
132 detail::invoke_dispatch(v, x, g, IsSameTag());
133 }
134#else
135 template <class Visitor, class T, class Graph, class Tag>
136 inline void
137 invoke_visitors(Visitor& v, T x, Graph& g, Tag) {
138 typedef typename Visitor::event_filter Category;
139 typedef typename is_same<Category, Tag>::type IsSameTag;
140 detail::invoke_dispatch(v, x, g, IsSameTag());
141 }
142#endif

147 template <class PredecessorMap, class Tag>
148 struct predecessor_recorder
149 : public base_visitor<predecessor_recorder<PredecessorMap, Tag> >
150 {
151 typedef Tag event_filter;
152 predecessor_recorder(PredecessorMap pa) : m_predecessor(pa) { }
153 template <class Edge, class Graph>
154 void operator()(Edge e, const Graph& g) {
155 put(m_predecessor, target(e, g), source(e, g));
156 }
157 PredecessorMap m_predecessor;
158 };
159 template <class PredecessorMap, class Tag>
160 predecessor_recorder<PredecessorMap, Tag>
161 record_predecessors(PredecessorMap pa, Tag) {
162 return predecessor_recorder<PredecessorMap, Tag> (pa);
163 }

189 template <class DistanceMap, class Tag>
190 struct distance_recorder
191 : public base_visitor<distance_recorder<DistanceMap, Tag> >
192 {
193 typedef Tag event_filter;
194 distance_recorder(DistanceMap pa) : m_distance(pa) { }
195 template <class Edge, class Graph>
196 void operator()(Edge e, const Graph& g) {
197 typename graph_traits<Graph>::vertex_descriptor
198 u = source(e, g), v = target(e, g);
199 put(m_distance, v, get(m_distance, u) + 1);
200 }
201 DistanceMap m_distance;
202 };
203 template <class DistanceMap, class Tag>
204 distance_recorder<DistanceMap, Tag>
205 record_distances(DistanceMap pa, Tag) {
206 return distance_recorder<DistanceMap, Tag> (pa);
207 }

255 // property_put
256
257 /**
258 * Functor which just sets a given value to a vertex or edge in a property map.
259 */
260
261 template <typename PropertyMap, typename EventTag>
262 struct property_put
263 {
264 typedef EventTag event_filter;
265
266 property_put (PropertyMap property_map,
267 typename property_traits <PropertyMap>::value_type value) :
268 property_map_ (property_map), value_ (value)
269 {}
270
271 template <typename VertexOrEdge, typename Graph>
272 void operator() (VertexOrEdge v, const Graph&)
273 {
274 put (property_map_, v, value_);
275 }
276
277 private:
278 PropertyMap property_map_;
279 typename property_traits <PropertyMap>::value_type value_;
280 };
281
282 /**
283 * Creates a property_put functor which just sets a given value to a vertex or edge.
284 *
285 * @param property_map Given writeable property map
286 * @param value Fixed value of the map
287 * @param tag Event Filter
288 * @return The functor.
289 */
290
291 template <typename PropertyMap, typename EventTag>
292 inline property_put <PropertyMap, EventTag>
293 put_property (PropertyMap property_map,
294 typename property_traits <PropertyMap>::value_type value,
295 EventTag)
296 {
297 return property_put <PropertyMap, EventTag> (property_map, value);
298 }

300#define BOOST_GRAPH_EVENT_STUB(Event,Kind) \
301 typedef ::boost::Event Event##_type; \
302 template<typename Visitor> \
303 Kind##_visitor<std::pair<detail::functor_to_visitor<Event##_type, \
304 Visitor>, Visitors> > \
305 do_##Event(Visitor visitor) \
306 { \
307 typedef std::pair<detail::functor_to_visitor<Event##_type, Visitor>, \
308 Visitors> visitor_list; \
309 typedef Kind##_visitor<visitor_list> result_type; \
310 return result_type(visitor_list(visitor, m_vis)); \
311 }

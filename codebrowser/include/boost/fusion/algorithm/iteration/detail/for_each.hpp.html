28 template <typename First, typename Last, typename F>
29 inline void
30 for_each_linear(First const& first, Last const& last, F const& f, mpl::false_)
31 {
32 f(*first);
33 detail::for_each_linear(fusion::next(first), last, f,
34 result_of::equal_to<typename result_of::next<First>::type, Last>());

38 template <typename Sequence, typename F, typename Tag>
39 inline void
40 for_each_dispatch(Sequence& seq, F const& f, Tag)
41 {
42 detail::for_each_linear(
43 fusion::begin(seq)
44 , fusion::end(seq)
45 , f
46 , result_of::equal_to<
47 typename result_of::begin<Sequence>::type
48 , typename result_of::end<Sequence>::type>());
49 }

51 template<int N>
52 struct for_each_unrolled
53 {
54 template<typename I0, typename F>
55 static void call(I0 const& i0, F const& f)
56 {
57 f(*i0);
58 typedef typename result_of::next<I0>::type I1;
59 I1 i1(fusion::next(i0));
60 f(*i1);
61 typedef typename result_of::next<I1>::type I2;
62 I2 i2(fusion::next(i1));
63 f(*i2);
64 typedef typename result_of::next<I2>::type I3;
65 I3 i3(fusion::next(i2));
66 f(*i3);
67 for_each_unrolled<N-4>::call(fusion::next(i3), f);
68 }
69 };

119 template <typename Sequence, typename F>
120 inline void
121 for_each_dispatch(Sequence& seq, F const& f, random_access_traversal_tag)
122 {
123 typedef typename result_of::begin<Sequence>::type begin;
124 typedef typename result_of::end<Sequence>::type end;
125 for_each_unrolled<result_of::distance<begin, end>::type::value>::call(fusion::begin(seq), f);
126 }
127
128 template <typename Sequence, typename F>
129 inline void
130 for_each(Sequence& seq, F const& f, mpl::false_) // unsegmented implementation
131 {
132 detail::for_each_dispatch(seq, f, typename traits::category_of<Sequence>::type());
133 }

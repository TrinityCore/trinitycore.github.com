15namespace boost { namespace fusion { namespace detail
16{
17 template <typename Fun>
18 struct segmented_for_each_fun
19 {
20 explicit segmented_for_each_fun(Fun const& f)
21 : fun(f)
22 {}
23
24 Fun const& fun;
25
26 template <typename Sequence, typename State, typename Context>
27 struct apply
28 {
29 typedef void_ type;
30 typedef mpl::true_ continue_type;
31
32 static type call(Sequence& seq, State const&, Context const&, segmented_for_each_fun const& fun)
33 {
34 fusion::for_each(seq, fun.fun);
35 return void_();
36 }
37 };
38 };
39
40 template <typename Sequence, typename F>
41 inline void
42 for_each(Sequence& seq, F const& f, mpl::true_) // segmented implementation
43 {
44 fusion::segmented_fold_until(seq, void_(), segmented_for_each_fun<F>(f));
45 }
46}}}

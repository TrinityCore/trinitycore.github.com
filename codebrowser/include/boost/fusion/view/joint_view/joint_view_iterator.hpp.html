28 struct joint_view_iterator

30 {
31 typedef convert_iterator<First> first_converter;

33 typedef convert_iterator<Concat> concat_converter;
34
35 typedef typename first_converter::type first_type;
36 typedef typename last_converter::type last_type;
37 typedef typename concat_converter::type concat_type;
38
39 typedef joint_view_iterator_tag fusion_tag;
40 typedef Category category;
41 BOOST_STATIC_ASSERT((!result_of::equal_to<first_type, last_type>::value));
42
43 joint_view_iterator(First const& in_first, Concat const& in_concat)
44 : first(first_converter::call(in_first))
45 , concat(concat_converter::call(in_concat))
46 {}

72/* A number random enough not to collide with different errno ranges on */
73/* different OSes. The assumption is that error_t is at least 32-bit type. */
74#define ZMQ_HAUSNUMERO 156384712
75
76/* On Windows platform some of the standard POSIX errnos are not defined. */
77#ifndef ENOTSUP
78#define ENOTSUP (ZMQ_HAUSNUMERO + 1)
79#endif
80#ifndef EPROTONOSUPPORT
81#define EPROTONOSUPPORT (ZMQ_HAUSNUMERO + 2)
82#endif
83#ifndef ENOBUFS
84#define ENOBUFS (ZMQ_HAUSNUMERO + 3)
85#endif
86#ifndef ENETDOWN
87#define ENETDOWN (ZMQ_HAUSNUMERO + 4)
88#endif
89#ifndef EADDRINUSE
90#define EADDRINUSE (ZMQ_HAUSNUMERO + 5)
91#endif
92#ifndef EADDRNOTAVAIL
93#define EADDRNOTAVAIL (ZMQ_HAUSNUMERO + 6)
94#endif
95#ifndef ECONNREFUSED
96#define ECONNREFUSED (ZMQ_HAUSNUMERO + 7)
97#endif
98#ifndef EINPROGRESS
99#define EINPROGRESS (ZMQ_HAUSNUMERO + 8)
100#endif
101#ifndef ENOTSOCK
102#define ENOTSOCK (ZMQ_HAUSNUMERO + 9)

120/******************************************************************************/
121/* 0MQ message definition. */
122/******************************************************************************/
123
124/* Maximal size of "Very Small Message". VSMs are passed by value */
125/* to avoid excessive memory allocation/deallocation. */
126/* If VMSs larger than 255 bytes are required, type of 'vsm_size' */
127/* field in zmq_msg_t structure should be modified accordingly. */
128#define ZMQ_MAX_VSM_SIZE 30
129
130/* Message types. These integers may be stored in 'content' member of the */
131/* message instead of regular pointer to the data. */
132#define ZMQ_DELIMITER 31
133#define ZMQ_VSM 32
134
135/* Message flags. ZMQ_MSG_SHARED is strictly speaking not a message flag */
136/* (it has no equivalent in the wire format), however, making it a flag */
137/* allows us to pack the stucture tigher and thus improve performance. */
138#define ZMQ_MSG_MORE 1
139#define ZMQ_MSG_SHARED 128
140#define ZMQ_MSG_MASK 129 /* Merges all the flags */
141
142/* A message. Note that 'content' is not a pointer to the raw data. */
143/* Rather it is pointer to zmq::msg_content_t structure */
144/* (see src/msg_content.hpp for its definition). */
145typedef struct
146{
147 void *content;
148 unsigned char flags;
149 unsigned char vsm_size;
150 unsigned char vsm_data [ZMQ_MAX_VSM_SIZE];
151} zmq_msg_t;
152

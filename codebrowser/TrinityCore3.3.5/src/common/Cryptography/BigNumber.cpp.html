49void BigNumber::SetQword(uint64 val)
50{
51 BN_set_word(_bn, (uint32)(val >> 32));
52 BN_lshift(_bn, _bn, 32);
53 BN_add_word(_bn, (uint32)(val & 0xFFFFFFFF));
54}

56void BigNumber::SetBinary(uint8 const* bytes, int32 len)
57{
58 uint8* array = new uint8[len];
59
60 for (int i = 0; i < len; i++)
61 array[i] = bytes[len - 1 - i];
62
63 BN_bin2bn(array, len, _bn);
64
65 delete[] array;
66}

171std::unique_ptr<uint8[]> BigNumber::AsByteArray(int32 minSize, bool littleEndian)
172{
173 int numBytes = GetNumBytes();
174 int length = (minSize >= numBytes) ? minSize : numBytes;
175
176 uint8* array = new uint8[length];
177
178 // If we need more bytes than length of BigNumber set the rest to 0
179 if (length > numBytes)
180 memset((void*)array, 0, length);
181
182 BN_bn2bin(_bn, (unsigned char *)array);
183
184 // openssl's BN stores data internally in big endian format, reverse if little endian desired
185 if (littleEndian)
186 std::reverse(array, array + numBytes);
187
188 std::unique_ptr<uint8[]> ret(array);
189 return ret;
190}

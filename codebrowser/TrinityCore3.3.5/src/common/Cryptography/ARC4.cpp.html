19#include "ARC4.h"
20
21ARC4::ARC4(uint8 len) : m_ctx()
22{
23 EVP_CIPHER_CTX_init(&m_ctx);
24 EVP_EncryptInit_ex(&m_ctx, EVP_rc4(), NULL, NULL, NULL);
25 EVP_CIPHER_CTX_set_key_length(&m_ctx, len);
26}
27
28ARC4::ARC4(uint8 *seed, uint8 len) : m_ctx()
29{
30 EVP_CIPHER_CTX_init(&m_ctx);
31 EVP_EncryptInit_ex(&m_ctx, EVP_rc4(), NULL, NULL, NULL);
32 EVP_CIPHER_CTX_set_key_length(&m_ctx, len);
33 EVP_EncryptInit_ex(&m_ctx, NULL, NULL, seed, NULL);
34}
35
36ARC4::~ARC4()
37{
38 EVP_CIPHER_CTX_cleanup(&m_ctx);
39}
40
41void ARC4::Init(uint8 *seed)
42{
43 EVP_EncryptInit_ex(&m_ctx, NULL, NULL, seed, NULL);
44}
45
46void ARC4::UpdateData(int len, uint8 *data)
47{
48 int outlen = 0;
49 EVP_EncryptUpdate(&m_ctx, data, &outlen, data, len);
50 EVP_EncryptFinal_ex(&m_ctx, data, &outlen);
51}

19#include "HMACSHA1.h"
20#include "BigNumber.h"
21#include "Common.h"
22
23HmacHash::HmacHash(uint32 len, uint8 *seed)
24{
25 HMAC_CTX_init(&m_ctx);
26 HMAC_Init_ex(&m_ctx, seed, len, EVP_sha1(), NULL);
27 memset(m_digest, 0, sizeof(m_digest));
28}
29
30HmacHash::~HmacHash()
31{
32 HMAC_CTX_cleanup(&m_ctx);
33}
34
35void HmacHash::UpdateData(const std::string &str)
36{
37 HMAC_Update(&m_ctx, (uint8 const*)str.c_str(), str.length());
38}
39
40void HmacHash::UpdateData(const uint8* data, size_t len)
41{
42 HMAC_Update(&m_ctx, data, len);
43}
44
45void HmacHash::Finalize()
46{
47 uint32 length = 0;
48 HMAC_Final(&m_ctx, (uint8*)m_digest, &length);
49 ASSERT(length == SHA_DIGEST_LENGTH);
50}
51
52uint8 *HmacHash::ComputeHash(BigNumber* bn)
53{
54 HMAC_Update(&m_ctx, bn->AsByteArray().get(), bn->GetNumBytes());
55 Finalize();
56 return (uint8*)m_digest;
57}

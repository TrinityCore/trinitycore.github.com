25AuthCrypt::AuthCrypt() :
26 _clientDecrypt(SHA_DIGEST_LENGTH), _serverEncrypt(SHA_DIGEST_LENGTH),

30void AuthCrypt::Init(BigNumber* K)
31{
32 uint8 ServerEncryptionKey[SEED_KEY_SIZE] = { 0xCC, 0x98, 0xAE, 0x04, 0xE8, 0x97, 0xEA, 0xCA, 0x12, 0xDD, 0xC0, 0x93, 0x42, 0x91, 0x53, 0x57 };
33 HmacHash serverEncryptHmac(SEED_KEY_SIZE, (uint8*)ServerEncryptionKey);
34 uint8 *encryptHash = serverEncryptHmac.ComputeHash(K);
35
36 uint8 ServerDecryptionKey[SEED_KEY_SIZE] = { 0xC2, 0xB3, 0x72, 0x3C, 0xC6, 0xAE, 0xD9, 0xB5, 0x34, 0x3C, 0x53, 0xEE, 0x2F, 0x43, 0x67, 0xCE };
37 HmacHash clientDecryptHmac(SEED_KEY_SIZE, (uint8*)ServerDecryptionKey);
38 uint8 *decryptHash = clientDecryptHmac.ComputeHash(K);
39
40 //ARC4 _serverDecrypt(encryptHash);
41 _clientDecrypt.Init(decryptHash);
42 _serverEncrypt.Init(encryptHash);
43 //ARC4 _clientEncrypt(decryptHash);
44
45 // Drop first 1024 bytes, as WoW uses ARC4-drop1024.
46 uint8 syncBuf[1024];
47 memset(syncBuf, 0, 1024);
48
49 _serverEncrypt.UpdateData(1024, syncBuf);
50 //_clientEncrypt.UpdateData(1024, syncBuf);
51
52 memset(syncBuf, 0, 1024);
53
54 //_serverDecrypt.UpdateData(1024, syncBuf);
55 _clientDecrypt.UpdateData(1024, syncBuf);
56
57 _initialized = true;
58}
59
60void AuthCrypt::DecryptRecv(uint8 *data, size_t len)
61{
62 if (!_initialized)
63 return;
64
65 _clientDecrypt.UpdateData(len, data);
66}
67
68void AuthCrypt::EncryptSend(uint8 *data, size_t len)
69{
70 if (!_initialized)
71 return;
72
73 _serverEncrypt.UpdateData(len, data);
74}

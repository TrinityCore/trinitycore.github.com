36 void MMapManager::InitializeThreadUnsafe(const std::vector<uint32>& mapIds)
37 {
38 // the caller must pass the list of all mapIds that will be used in the VMapManager2 lifetime
39 for (const uint32& mapId : mapIds)
40 loadedMMaps.insert(MMapDataSet::value_type(mapId, nullptr));
41
42 thread_safe_environment = false;
43 }

45 MMapDataSet::const_iterator MMapManager::GetMMapData(uint32 mapId) const
46 {
47 // return the iterator if found or end() if not found/NULL
48 MMapDataSet::const_iterator itr = loadedMMaps.find(mapId);
49 if (itr != loadedMMaps.cend() && !itr->second)
50 itr = loadedMMaps.cend();
51
52 return itr;

55 bool MMapManager::loadMapData(uint32 mapId)
56 {
57 // we already have this map loaded?
58 MMapDataSet::iterator itr = loadedMMaps.find(mapId);
59 if (itr != loadedMMaps.end())
60 {
61 if (itr->second)
62 return true;
63 }
64 else
65 {
66 if (thread_safe_environment)
67 itr = loadedMMaps.insert(MMapDataSet::value_type(mapId, nullptr)).first;
68 else
69 ASSERT(false, "Invalid mapId %u passed to MMapManager after startup in thread unsafe environment", mapId);
70 }
71
72 // load and init dtNavMesh - read parameters from file
73 std::string dataDir = sConfigMgr->GetStringDefault("DataDir", "./");
74 uint32 pathLen = dataDir.length() + strlen("/mmaps/%03i.mmap") + 1;
75 char *fileName = new char[pathLen];
76 snprintf(fileName, pathLen, (dataDir + "/mmaps/%03i.mmap").c_str(), mapId);
77
78 FILE* file = fopen(fileName, "rb");
79 if (!file)
80 {
81 TC_LOG_DEBUG("maps", "MMAP:loadMapData: Error: Could not open mmap file '%s'", fileName);
82 delete [] fileName;
83 return false;
84 }
85
86 dtNavMeshParams params;
87 int count = fread(&params, sizeof(dtNavMeshParams), 1, file);
88 fclose(file);
89 if (count != 1)
90 {
91 TC_LOG_DEBUG("maps", "MMAP:loadMapData: Error: Could not read params from file '%s'", fileName);
92 delete [] fileName;
93 return false;
94 }
95
96 dtNavMesh* mesh = dtAllocNavMesh();
97 ASSERT(mesh);
98 if (dtStatusFailed(mesh->init(&params)))
99 {
100 dtFreeNavMesh(mesh);
101 TC_LOG_ERROR("maps", "MMAP:loadMapData: Failed to initialize dtNavMesh for mmap %03u from file %s", mapId, fileName);
102 delete [] fileName;
103 return false;
104 }
105
106 delete [] fileName;
107
108 TC_LOG_DEBUG("maps", "MMAP:loadMapData: Loaded %03i.mmap", mapId);
109
110 // store inside our map list
111 MMapData* mmap_data = new MMapData(mesh);
112 mmap_data->mmapLoadedTiles.clear();
113
114 itr->second = mmap_data;
115 return true;
116 }

123 bool MMapManager::loadMap(const std::string& basePath, uint32 mapId, int32 x, int32 y)
124 {
125 // make sure the mmap is loaded and ready to load tiles
126 if (!loadMapData(mapId))
127 return false;
128
129 // get this mmap data
130 MMapData* mmap = loadedMMaps[mapId];
131 ASSERT(mmap->navMesh);
132
133 // check if we already have this tile loaded
134 uint32 packedGridPos = packTileID(x, y);
135 if (mmap->mmapLoadedTiles.find(packedGridPos) != mmap->mmapLoadedTiles.end())
136 return false;
137
138 // load this tile :: mmaps/MMMXXYY.mmtile
139 uint32 pathLen = basePath.length() + strlen("/%03i%02i%02i.mmtile") + 1;
140 char *fileName = new char[pathLen];
141
142 snprintf(fileName, pathLen, (basePath + "/%03i%02i%02i.mmtile").c_str(), mapId, x, y);
143
144 FILE* file = fopen(fileName, "rb");
145 if (!file)
146 {
147 TC_LOG_DEBUG("maps", "MMAP:loadMap: Could not open mmtile file '%s'", fileName);
148 delete [] fileName;
149 return false;
150 }
151 delete [] fileName;
152
153 // read header
154 MmapTileHeader fileHeader;
155 if (fread(&fileHeader, sizeof(MmapTileHeader), 1, file) != 1 || fileHeader.mmapMagic != MMAP_MAGIC)
156 {
157 TC_LOG_ERROR("maps", "MMAP:loadMap: Bad header in mmap %03u%02i%02i.mmtile", mapId, x, y);
158 fclose(file);
159 return false;
160 }
161
162 if (fileHeader.mmapVersion != MMAP_VERSION)
163 {
164 TC_LOG_ERROR("maps", "MMAP:loadMap: %03u%02i%02i.mmtile was built with generator v%i, expected v%i",
165 mapId, x, y, fileHeader.mmapVersion, MMAP_VERSION);
166 fclose(file);
167 return false;
168 }
169
170 unsigned char* data = (unsigned char*)dtAlloc(fileHeader.size, DT_ALLOC_PERM);
171 ASSERT(data);
172
173 size_t result = fread(data, fileHeader.size, 1, file);
174 if (!result)
175 {
176 TC_LOG_ERROR("maps", "MMAP:loadMap: Bad header or data in mmap %03u%02i%02i.mmtile", mapId, x, y);
177 fclose(file);
178 return false;
179 }
180
181 fclose(file);
182
183 dtMeshHeader* header = (dtMeshHeader*)data;
184 dtTileRef tileRef = 0;
185
186 // memory allocated for data is now managed by detour, and will be deallocated when the tile is removed
187 if (dtStatusSucceed(mmap->navMesh->addTile(data, fileHeader.size, DT_TILE_FREE_DATA, 0, &tileRef)))
188 {
189 mmap->mmapLoadedTiles.insert(std::pair<uint32, dtTileRef>(packedGridPos, tileRef));
190 ++loadedTiles;
191 TC_LOG_DEBUG("maps", "MMAP:loadMap: Loaded mmtile %03i[%02i, %02i] into %03i[%02i, %02i]", mapId, x, y, mapId, header->x, header->y);
192 return true;
193 }
194 else
195 {
196 TC_LOG_ERROR("maps", "MMAP:loadMap: Could not load %03u%02i%02i.mmtile into navmesh", mapId, x, y);
197 dtFree(data);
198 return false;
199 }
200 }

202 bool MMapManager::unloadMap(uint32 mapId, int32 x, int32 y)
203 {
204 // check if we have this map loaded
205 MMapDataSet::const_iterator itr = GetMMapData(mapId);
206 if (itr == loadedMMaps.end())
207 {
208 // file may not exist, therefore not loaded
209 TC_LOG_DEBUG("maps", "MMAP:unloadMap: Asked to unload not loaded navmesh map. %03u%02i%02i.mmtile", mapId, x, y);
210 return false;
211 }
212
213 MMapData* mmap = itr->second;
214
215 // check if we have this tile loaded
216 uint32 packedGridPos = packTileID(x, y);
217 if (mmap->mmapLoadedTiles.find(packedGridPos) == mmap->mmapLoadedTiles.end())
218 {
219 // file may not exist, therefore not loaded
220 TC_LOG_DEBUG("maps", "MMAP:unloadMap: Asked to unload not loaded navmesh tile. %03u%02i%02i.mmtile", mapId, x, y);
221 return false;
222 }
223
224 dtTileRef tileRef = mmap->mmapLoadedTiles[packedGridPos];
225
226 // unload, and mark as non loaded
227 if (dtStatusFailed(mmap->navMesh->removeTile(tileRef, NULL, NULL)))
228 {
229 // this is technically a memory leak
230 // if the grid is later reloaded, dtNavMesh::addTile will return error but no extra memory is used
231 // we cannot recover from this error - assert out
232 TC_LOG_ERROR("maps", "MMAP:unloadMap: Could not unload %03u%02i%02i.mmtile from navmesh", mapId, x, y);
233 ABORT();
234 }
235 else
236 {
237 mmap->mmapLoadedTiles.erase(packedGridPos);
238 --loadedTiles;
239 TC_LOG_DEBUG("maps", "MMAP:unloadMap: Unloaded mmtile %03i[%02i, %02i] from %03i", mapId, x, y, mapId);
240 return true;
241 }
242
243 return false;
244 }

246 bool MMapManager::unloadMap(uint32 mapId)
247 {
248 MMapDataSet::iterator itr = loadedMMaps.find(mapId);
249 if (itr == loadedMMaps.end() || !itr->second)
250 {
251 // file may not exist, therefore not loaded
252 TC_LOG_DEBUG("maps", "MMAP:unloadMap: Asked to unload not loaded navmesh map %03u", mapId);
253 return false;
254 }
255
256 // unload all tiles from given map
257 MMapData* mmap = itr->second;
258 for (MMapTileSet::iterator i = mmap->mmapLoadedTiles.begin(); i != mmap->mmapLoadedTiles.end(); ++i)
259 {
260 uint32 x = (i->first >> 16);
261 uint32 y = (i->first & 0x0000FFFF);
262 if (dtStatusFailed(mmap->navMesh->removeTile(i->second, NULL, NULL)))
263 TC_LOG_ERROR("maps", "MMAP:unloadMap: Could not unload %03u%02i%02i.mmtile from navmesh", mapId, x, y);
264 else
265 {
266 --loadedTiles;
267 TC_LOG_DEBUG("maps", "MMAP:unloadMap: Unloaded mmtile %03i[%02i, %02i] from %03i", mapId, x, y, mapId);
268 }
269 }
270
271 delete mmap;
272 itr->second = nullptr;
273 TC_LOG_DEBUG("maps", "MMAP:unloadMap: Unloaded %03i.mmap", mapId);
274
275 return true;
276 }

278 bool MMapManager::unloadMapInstance(uint32 mapId, uint32 instanceId)
279 {
280 // check if we have this map loaded
281 MMapDataSet::const_iterator itr = GetMMapData(mapId);
282 if (itr == loadedMMaps.end())
283 {
284 // file may not exist, therefore not loaded
285 TC_LOG_DEBUG("maps", "MMAP:unloadMapInstance: Asked to unload not loaded navmesh map %03u", mapId);
286 return false;
287 }
288
289 MMapData* mmap = itr->second;
290 if (mmap->navMeshQueries.find(instanceId) == mmap->navMeshQueries.end())
291 {
292 TC_LOG_DEBUG("maps", "MMAP:unloadMapInstance: Asked to unload not loaded dtNavMeshQuery mapId %03u instanceId %u", mapId, instanceId);
293 return false;
294 }
295
296 dtNavMeshQuery* query = mmap->navMeshQueries[instanceId];
297
298 dtFreeNavMeshQuery(query);
299 mmap->navMeshQueries.erase(instanceId);
300 TC_LOG_DEBUG("maps", "MMAP:unloadMapInstance: Unloaded mapId %03u instanceId %u", mapId, instanceId);
301
302 return true;
303 }

307 MMapDataSet::const_iterator itr = GetMMapData(mapId);

314 dtNavMeshQuery const* MMapManager::GetNavMeshQuery(uint32 mapId, uint32 instanceId)
315 {
316 MMapDataSet::const_iterator itr = GetMMapData(mapId);
317 if (itr == loadedMMaps.end())
318 return NULL;
319
320 MMapData* mmap = itr->second;
321 if (mmap->navMeshQueries.find(instanceId) == mmap->navMeshQueries.end())
322 {
323 // allocate mesh query
324 dtNavMeshQuery* query = dtAllocNavMeshQuery();
325 ASSERT(query);
326 if (dtStatusFailed(query->init(mmap->navMesh, 1024)))
327 {
328 dtFreeNavMeshQuery(query);
329 TC_LOG_ERROR("maps", "MMAP:GetNavMeshQuery: Failed to initialize dtNavMeshQuery for mapId %03u instanceId %u", mapId, instanceId);
330 return NULL;
331 }
332
333 TC_LOG_DEBUG("maps", "MMAP:GetNavMeshQuery: created dtNavMeshQuery for mapId %03u instanceId %u", mapId, instanceId);
334 mmap->navMeshQueries.insert(std::pair<uint32, dtNavMeshQuery*>(instanceId, query));
335 }
336
337 return mmap->navMeshQueries[instanceId];
338 }

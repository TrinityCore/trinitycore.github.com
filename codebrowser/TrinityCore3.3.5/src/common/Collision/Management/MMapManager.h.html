30// move map related classes
31namespace MMAP
32{
33 typedef std::unordered_map<uint32, dtTileRef> MMapTileSet;
34 typedef std::unordered_map<uint32, dtNavMeshQuery*> NavMeshQuerySet;
35
36 // dummy struct to hold map's mmap data
37 struct MMapData
38 {
39 MMapData(dtNavMesh* mesh) : navMesh(mesh) { }
40 ~MMapData()
41 {
42 for (NavMeshQuerySet::iterator i = navMeshQueries.begin(); i != navMeshQueries.end(); ++i)
43 dtFreeNavMeshQuery(i->second);
44
45 if (navMesh)
46 dtFreeNavMesh(navMesh);
47 }
48
49 dtNavMesh* navMesh;
50
51 // we have to use single dtNavMeshQuery for every instance, since those are not thread safe
52 NavMeshQuerySet navMeshQueries; // instanceId to query
53 MMapTileSet mmapLoadedTiles; // maps [map grid coords] to [dtTile]
54 };
55
56
57 typedef std::unordered_map<uint32, MMapData*> MMapDataSet;
58
59 // singleton class
60 // holds all all access to mmap loading unloading and meshes
61 class MMapManager
62 {
63 public:
64 MMapManager() : loadedTiles(0), thread_safe_environment(true) {}
65 ~MMapManager();
66
67 void InitializeThreadUnsafe(const std::vector<uint32>& mapIds);
68 bool loadMap(const std::string& basePath, uint32 mapId, int32 x, int32 y);
69 bool unloadMap(uint32 mapId, int32 x, int32 y);
70 bool unloadMap(uint32 mapId);
71 bool unloadMapInstance(uint32 mapId, uint32 instanceId);
72
73 // the returned [dtNavMeshQuery const*] is NOT threadsafe
74 dtNavMeshQuery const* GetNavMeshQuery(uint32 mapId, uint32 instanceId);
75 dtNavMesh const* GetNavMesh(uint32 mapId);
76
77 uint32 getLoadedTilesCount() const { return loadedTiles; }
78 uint32 getLoadedMapsCount() const { return loadedMMaps.size(); }
79 private:
80 bool loadMapData(uint32 mapId);
81 uint32 packTileID(int32 x, int32 y);
82
83 MMapDataSet::const_iterator GetMMapData(uint32 mapId) const;
84 MMapDataSet loadedMMaps;
85 uint32 loadedTiles;
86 bool thread_safe_environment;
87 };
88}

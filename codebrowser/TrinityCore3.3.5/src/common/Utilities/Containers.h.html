32 template<class T>
33 void RandomResizeList(std::list<T> &list, uint32 size)
34 {
35 size_t list_size = list.size();
36
37 while (list_size > size)
38 {
39 typename std::list<T>::iterator itr = list.begin();
40 std::advance(itr, urand(0, list_size - 1));
41 list.erase(itr);
42 --list_size;
43 }
44 }
45
46 template<class T, class Predicate>
47 void RandomResizeList(std::list<T> &list, Predicate& predicate, uint32 size)
48 {
49 //! First use predicate filter
50 std::list<T> listCopy;
51 for (typename std::list<T>::iterator itr = list.begin(); itr != list.end(); ++itr)
52 if (predicate(*itr))
53 listCopy.push_back(*itr);
54
55 if (size)
56 RandomResizeList(listCopy, size);
57
58 list = listCopy;
59 }

61 /*
62 * Select a random element from a container.
63 *
64 * Note: container cannot be empty
65 */
66 template <class C>
67 typename C::value_type const& SelectRandomContainerElement(C const& container)
68 {
69 typename C::const_iterator it = container.begin();
70 std::advance(it, urand(0, container.size() - 1));
71 return *it;
72 }

74 /*
75 * Select a random element from a container where each element has a different chance to be selected.
76 *
77 * @param container Container to select an element from
78 * @param weights Chances of each element to be selected, must be in the same order as elements in container.
79 * Caller is responsible for checking that sum of all weights is greater than 0.
80 *
81 * Note: container cannot be empty
82 */
83 template <class C>
84 typename C::const_iterator SelectRandomWeightedContainerElement(C const& container, std::vector<double> weights)
85 {
86 Trinity::discrete_distribution_param<uint32> ddParam(weights.begin(), weights.end());
87 std::discrete_distribution<uint32> dd(ddParam);
88 typename C::const_iterator it = container.begin();
89 std::advance(it, dd(SFMTEngine::Instance()));
90 return it;
91 }
92
93 /*
94 * Select a random element from a container where each element has a different chance to be selected.
95 *
96 * @param container Container to select an element from
97 * @param weightExtractor Function retrieving chance of each element in container, expected to take an element of the container and returning a double
98 *
99 * Note: container cannot be empty
100 */
101 template <class C, class Fn>
102 typename C::const_iterator SelectRandomWeightedContainerElement(C const& container, Fn weightExtractor)
103 {
104 std::vector<double> weights;
105 weights.reserve(container.size());
106 double weightSum = 0.0;
107 for (auto itr = container.begin(); itr != container.end(); ++itr)
108 {
109 double weight = weightExtractor(*itr);
110 weights.push_back(weight);
111 weightSum += weight;
112 }
113 if (weightSum <= 0.0)
114 weights.assign(container.size(), 1.0);
115
116 return SelectRandomWeightedContainerElement(container, weights);
117 }

119 /**
120 * @fn bool Trinity::Containers::Intersects(Iterator first1, Iterator last1, Iterator first2, Iterator last2)
121 *
122 * @brief Checks if two SORTED containers have a common element
123 *
124 * @param first1 Iterator pointing to start of the first container
125 * @param last1 Iterator pointing to end of the first container
126 * @param first2 Iterator pointing to start of the second container
127 * @param last2 Iterator pointing to end of the second container
128 *
129 * @return true if containers have a common element, false otherwise.
130 */
131 template<class Iterator1, class Iterator2>
132 bool Intersects(Iterator1 first1, Iterator1 last1, Iterator2 first2, Iterator2 last2)
133 {
134 while (first1 != last1 && first2 != last2)
135 {
136 if (*first1 < *first2)
137 ++first1;
138 else if (*first2 < *first1)
139 ++first2;
140 else
141 return true;
142 }
143
144 return false;
145 }

147 template<class K, class V, template<class, class, class...> class M, class... Rest>
148 void MultimapErasePair(M<K, V, Rest...>& multimap, K const& key, V const& value)
149 {
150 auto range = multimap.equal_range(key);
151 for (auto itr = range.first; itr != range.second;)
152 {
153 if (itr->second == value)
154 itr = multimap.erase(itr);
155 else
156 ++itr;
157 }
158 }

32Tokenizer::Tokenizer(const std::string &src, const char sep, uint32 vectorReserve)
33{
34 m_str = new char[src.length() + 1];
35 memcpy(m_str, src.c_str(), src.length() + 1);
36
37 if (vectorReserve)
38 m_storage.reserve(vectorReserve);
39
40 char* posold = m_str;
41 char* posnew = m_str;
42
43 for (;;)
44 {
45 if (*posnew == sep)
46 {
47 m_storage.push_back(posold);
48 posold = posnew + 1;
49
50 *posnew = '\0';
51 }
52 else if (*posnew == '\0')
53 {
54 // Hack like, but the old code accepted these kind of broken strings,
55 // so changing it would break other things
56 if (posold != posnew)
57 m_storage.push_back(posold);
58
59 break;
60 }
61
62 ++posnew;
63 }
64}

66void stripLineInvisibleChars(std::string &str)
67{
68 static std::string const invChars = " \t\7\n";
69
70 size_t wpos = 0;
71
72 bool space = false;
73 for (size_t pos = 0; pos < str.size(); ++pos)
74 {
75 if (invChars.find(str[pos])!=std::string::npos)
76 {
77 if (!space)
78 {
79 str[wpos++] = ' ';
80 space = true;
81 }
82 }
83 else
84 {
85 if (wpos!=pos)
86 str[wpos++] = str[pos];
87 else
88 ++wpos;
89 space = false;
90 }
91 }
92
93 if (wpos < str.size())
94 str.erase(wpos, str.size());
95 if (str.find("|TInterface")!=std::string::npos)
96 str.clear();
97
98}

108std::string secsToTimeString(uint64 timeInSecs, bool shortText, bool hoursOnly)
109{
110 uint64 secs = timeInSecs % MINUTE;
111 uint64 minutes = timeInSecs % HOUR / MINUTE;
112 uint64 hours = timeInSecs % DAY / HOUR;
113 uint64 days = timeInSecs / DAY;
114
115 std::ostringstream ss;
116 if (days)
117 ss << days << (shortText ? "d" : " Day(s) ");
118 if (hours || hoursOnly)
119 ss << hours << (shortText ? "h" : " Hour(s) ");
120 if (!hoursOnly)
121 {
122 if (minutes)
123 ss << minutes << (shortText ? "m" : " Minute(s) ");
124 if (secs || (!days && !hours && !minutes) )
125 ss << secs << (shortText ? "s" : " Second(s).");
126 }
127
128 return ss.str();
129}

131int32 MoneyStringToMoney(const std::string& moneyString)
132{
133 int32 money = 0;
134
135 if (!(std::count(moneyString.begin(), moneyString.end(), 'g') == 1 ||
136 std::count(moneyString.begin(), moneyString.end(), 's') == 1 ||
137 std::count(moneyString.begin(), moneyString.end(), 'c') == 1))
138 return 0; // Bad format
139
140 Tokenizer tokens(moneyString, ' ');
141 for (Tokenizer::const_iterator itr = tokens.begin(); itr != tokens.end(); ++itr)
142 {
143 std::string tokenString(*itr);
144 size_t gCount = std::count(tokenString.begin(), tokenString.end(), 'g');
145 size_t sCount = std::count(tokenString.begin(), tokenString.end(), 's');
146 size_t cCount = std::count(tokenString.begin(), tokenString.end(), 'c');
147 if (gCount + sCount + cCount != 1)
148 return 0;
149
150 uint32 amount = atoi(*itr);
151 if (gCount == 1)
152 money += amount * 100 * 100;
153 else if (sCount == 1)
154 money += amount * 100;
155 else if (cCount == 1)
156 money += amount;
157 }
158
159 return money;
160}

162uint32 TimeStringToSecs(const std::string& timestring)
163{
164 uint32 secs = 0;
165 uint32 buffer = 0;
166 uint32 multiplier = 0;
167
168 for (std::string::const_iterator itr = timestring.begin(); itr != timestring.end(); ++itr)
169 {
170 if (isdigit(*itr))
171 {
172 buffer*=10;
173 buffer+= (*itr)-'0';
174 }
175 else
176 {
177 switch (*itr)
178 {
179 case 'd': multiplier = DAY; break;
180 case 'h': multiplier = HOUR; break;
181 case 'm': multiplier = MINUTE; break;
182 case 's': multiplier = 1; break;
183 default : return 0; //bad format
184 }
185 buffer*=multiplier;
186 secs+=buffer;
187 buffer=0;
188 }
189 }
190
191 return secs;
192}

194std::string TimeToTimestampStr(time_t t)
195{
196 tm aTm;
197 localtime_r(&t, &aTm);
198 // YYYY year
199 // MM month (2 digits 01-12)
200 // DD day (2 digits 01-31)
201 // HH hour (2 digits 00-23)
202 // MM minutes (2 digits 00-59)
203 // SS seconds (2 digits 00-59)
204 char buf[20];
205 snprintf(buf, 20, "%04d-%02d-%02d_%02d-%02d-%02d", aTm.tm_year+1900, aTm.tm_mon+1, aTm.tm_mday, aTm.tm_hour, aTm.tm_min, aTm.tm_sec);
206 return std::string(buf);
207}

222{

252void utf8truncate(std::string& utf8str, size_t len)
253{
254 try
255 {
256 size_t wlen = utf8::distance(utf8str.c_str(), utf8str.c_str()+utf8str.size());
257 if (wlen <= len)
258 return;
259
260 std::wstring wstr;
261 wstr.resize(wlen);
262 utf8::utf8to16(utf8str.c_str(), utf8str.c_str()+utf8str.size(), &wstr[0]);
263 wstr.resize(len);
264 char* oend = utf8::utf16to8(wstr.c_str(), wstr.c_str()+wstr.size(), &utf8str[0]);
265 utf8str.resize(oend-(&utf8str[0])); // remove unused tail
266 }
267 catch(std::exception)
268 {
269 utf8str.clear();
270 }
271}

273bool Utf8toWStr(char const* utf8str, size_t csize, wchar_t* wstr, size_t& wsize)
274{
275 try
276 {
277 size_t len = utf8::distance(utf8str, utf8str+csize);
278 if (len > wsize)
279 {
280 if (wsize > 0)
281 wstr[0] = L'\0';
282 wsize = 0;
283 return false;
284 }
285
286 wsize = len;
287 utf8::utf8to16(utf8str, utf8str+csize, wstr);
288 wstr[len] = L'\0';
289 }
290 catch(std::exception)
291 {
292 if (wsize > 0)
293 wstr[0] = L'\0';
294 wsize = 0;
295 return false;
296 }
297
298 return true;
299}

321{
322 try
323 {
324 std::string utf8str2;
325 utf8str2.resize(size*4); // allocate for most long case
326
327 if (size)
328 {
329 char* oend = utf8::utf16to8(wstr, wstr+size, &utf8str2[0]);
330 utf8str2.resize(oend-(&utf8str2[0])); // remove unused tail
331 }
332 utf8str = utf8str2;
333 }
334 catch(std::exception)
335 {
336 utf8str.clear();
337 return false;
338 }
339
340 return true;
341}

368std::wstring GetMainPartOfName(std::wstring const& wname, uint32 declension)
369{
370 // supported only Cyrillic cases
371 if (wname.size() < 1 || !isCyrillicCharacter(wname[0]) || declension > 5)
372 return wname;
373
374 // Important: end length must be <= MAX_INTERNAL_PLAYER_NAME-MAX_PLAYER_NAME (3 currently)
375
376 static wchar_t const a_End[] = { wchar_t(1), wchar_t(0x0430), wchar_t(0x0000)};
377 static wchar_t const o_End[] = { wchar_t(1), wchar_t(0x043E), wchar_t(0x0000)};
378 static wchar_t const ya_End[] = { wchar_t(1), wchar_t(0x044F), wchar_t(0x0000)};
379 static wchar_t const ie_End[] = { wchar_t(1), wchar_t(0x0435), wchar_t(0x0000)};
380 static wchar_t const i_End[] = { wchar_t(1), wchar_t(0x0438), wchar_t(0x0000)};
381 static wchar_t const yeru_End[] = { wchar_t(1), wchar_t(0x044B), wchar_t(0x0000)};
382 static wchar_t const u_End[] = { wchar_t(1), wchar_t(0x0443), wchar_t(0x0000)};
383 static wchar_t const yu_End[] = { wchar_t(1), wchar_t(0x044E), wchar_t(0x0000)};
384 static wchar_t const oj_End[] = { wchar_t(2), wchar_t(0x043E), wchar_t(0x0439), wchar_t(0x0000)};
385 static wchar_t const ie_j_End[] = { wchar_t(2), wchar_t(0x0435), wchar_t(0x0439), wchar_t(0x0000)};
386 static wchar_t const io_j_End[] = { wchar_t(2), wchar_t(0x0451), wchar_t(0x0439), wchar_t(0x0000)};
387 static wchar_t const o_m_End[] = { wchar_t(2), wchar_t(0x043E), wchar_t(0x043C), wchar_t(0x0000)};
388 static wchar_t const io_m_End[] = { wchar_t(2), wchar_t(0x0451), wchar_t(0x043C), wchar_t(0x0000)};
389 static wchar_t const ie_m_End[] = { wchar_t(2), wchar_t(0x0435), wchar_t(0x043C), wchar_t(0x0000)};
390 static wchar_t const soft_End[] = { wchar_t(1), wchar_t(0x044C), wchar_t(0x0000)};
391 static wchar_t const j_End[] = { wchar_t(1), wchar_t(0x0439), wchar_t(0x0000)};
392
393 static wchar_t const* const dropEnds[6][8] = {
394 { &a_End[1], &o_End[1], &ya_End[1], &ie_End[1], &soft_End[1], &j_End[1], NULL, NULL },
395 { &a_End[1], &ya_End[1], &yeru_End[1], &i_End[1], NULL, NULL, NULL, NULL },
396 { &ie_End[1], &u_End[1], &yu_End[1], &i_End[1], NULL, NULL, NULL, NULL },
397 { &u_End[1], &yu_End[1], &o_End[1], &ie_End[1], &soft_End[1], &ya_End[1], &a_End[1], NULL },
398 { &oj_End[1], &io_j_End[1], &ie_j_End[1], &o_m_End[1], &io_m_End[1], &ie_m_End[1], &yu_End[1], NULL },
399 { &ie_End[1], &i_End[1], NULL, NULL, NULL, NULL, NULL, NULL }
400 };
401
402 for (wchar_t const* const* itr = &dropEnds[declension][0]; *itr; ++itr)
403 {
404 size_t len = size_t((*itr)[-1]); // get length from string size field
405
406 if (wname.substr(wname.size()-len, len)==*itr)
407 return wname.substr(0, wname.size()-len);
408 }
409
410 return wname;
411}

445bool Utf8FitTo(const std::string& str, std::wstring const& search)
446{
447 std::wstring temp;
448
449 if (!Utf8toWStr(str, temp))
450 return false;
451
452 // converting to lower case
453 wstrToLower( temp );
454
455 if (temp.find(search) == std::wstring::npos)
456 return false;
457
458 return true;
459}

469void vutf8printf(FILE* out, const char *str, va_list* ap)
470{
471#if PLATFORM == PLATFORM_WINDOWS
472 char temp_buf[32*1024];
473 wchar_t wtemp_buf[32*1024];
474
475 size_t temp_len = vsnprintf(temp_buf, 32*1024, str, *ap);
476 //vsnprintf returns -1 if the buffer is too small
477 if (temp_len == size_t(-1))
478 temp_len = 32*1024-1;
479
480 size_t wtemp_len = 32*1024-1;
481 Utf8toWStr(temp_buf, temp_len, wtemp_buf, wtemp_len);
482
483 CharToOemBuffW(&wtemp_buf[0], &temp_buf[0], wtemp_len+1);
484 fprintf(out, "%s", temp_buf);
485#else
486 vfprintf(out, str, *ap);
487#endif
488}

490std::string ByteArrayToHexStr(uint8 const* bytes, uint32 arrayLen, bool reverse /* = false */)
491{
492 int32 init = 0;
493 int32 end = arrayLen;
494 int8 op = 1;
495
496 if (reverse)
497 {
498 init = arrayLen - 1;
499 end = -1;
500 op = -1;
501 }
502
503 std::ostringstream ss;
504 for (int32 i = init; i != end; i += op)
505 {
506 char buffer[4];
507 sprintf(buffer, "%02X", bytes[i]);
508 ss << buffer;
509 }
510
511 return ss.str();
512}

24namespace Trinity
25{
26 DECLSPEC_NORETURN void Assert(char const* file, int line, char const* function, char const* message) ATTR_NORETURN;
27 DECLSPEC_NORETURN void Assert(char const* file, int line, char const* function, char const* message, char const* format, ...) ATTR_NORETURN ATTR_PRINTF(5, 6);
28
29 DECLSPEC_NORETURN void Fatal(char const* file, int line, char const* function, char const* message, ...) ATTR_NORETURN ATTR_PRINTF(4, 5);
30
31 DECLSPEC_NORETURN void Error(char const* file, int line, char const* function, char const* message) ATTR_NORETURN;
32
33 DECLSPEC_NORETURN void Abort(char const* file, int line, char const* function) ATTR_NORETURN;
34
35 void Warning(char const* file, int line, char const* function, char const* message);
36
37} // namespace Trinity

39#if COMPILER == COMPILER_MICROSOFT
40#define ASSERT_BEGIN __pragma(warning(push)) __pragma(warning(disable: 4127))
41#define ASSERT_END __pragma(warning(pop))
42#else
43#define ASSERT_BEGIN
44#define ASSERT_END
45#endif
46
47#define WPAssert(cond, ...) ASSERT_BEGIN do { if (!(cond)) Trinity::Assert(__FILE__, __LINE__, __FUNCTION__, #cond, ##__VA_ARGS__); } while(0) ASSERT_END
48#define WPFatal(cond, ...) ASSERT_BEGIN do { if (!(cond)) Trinity::Fatal(__FILE__, __LINE__, __FUNCTION__, ##__VA_ARGS__); } while(0) ASSERT_END
49#define WPError(cond, msg) ASSERT_BEGIN do { if (!(cond)) Trinity::Error(__FILE__, __LINE__, __FUNCTION__, (msg)); } while(0) ASSERT_END
50#define WPWarning(cond, msg) ASSERT_BEGIN do { if (!(cond)) Trinity::Warning(__FILE__, __LINE__, __FUNCTION__, (msg)); } while(0) ASSERT_END
51#define WPAbort() ASSERT_BEGIN do { Trinity::Abort(__FILE__, __LINE__, __FUNCTION__); } while(0) ASSERT_END
52
53#define ASSERT WPAssert
54#define ABORT WPAbort
55
56template <typename T> inline T* ASSERT_NOTNULL(T* pointer)
57{
58 ASSERT(pointer);
59 return pointer;
60}

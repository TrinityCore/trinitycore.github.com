18#ifndef APPENDERCONSOLE_H
19#define APPENDERCONSOLE_H
20
21#include <string>
22#include "Appender.h"
23
24enum ColorTypes
25{
26 BLACK,
27 RED,
28 GREEN,
29 BROWN,
30 BLUE,
31 MAGENTA,
32 CYAN,
33 GREY,
34 YELLOW,
35 LRED,
36 LGREEN,
37 LBLUE,
38 LMAGENTA,
39 LCYAN,
40 WHITE
41};
42
43const uint8 MaxColors = uint8(WHITE) + 1;
44
45class AppenderConsole : public Appender
46{
47 public:
48 typedef std::integral_constant<AppenderType, APPENDER_CONSOLE>::type TypeIndex;
49
50 AppenderConsole(uint8 _id, std::string const& name, LogLevel level, AppenderFlags flags, ExtraAppenderArgs extraArgs);
51 void InitColors(const std::string& init_str);
52 AppenderType getType() const override { return TypeIndex::value; }
53
54 private:
55 void SetColor(bool stdout_stream, ColorTypes color);
56 void ResetColor(bool stdout_stream);
57 void _write(LogMessage const* message) override;
58 bool _colored;
59 ColorTypes _colors[MaxLogLevels];
60};

62#endif

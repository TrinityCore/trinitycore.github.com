66 template<typename Format, typename... Args>

68 {
69 if (!ShouldLog("commands.gm", LOG_LEVEL_INFO))
70 return;
71
72 std::unique_ptr<LogMessage> msg =
73 Trinity::make_unique<LogMessage>(LOG_LEVEL_INFO, "commands.gm",
74 Trinity::StringFormat(std::forward<Format>(fmt), std::forward<Args>(args)...));
75
76 msg->param1 = std::to_string(account);
77
78 write(std::move(msg));
79 }

122inline Logger const* Log::GetLoggerByType(std::string const& type) const
123{
124 LoggerMap::const_iterator it = loggers.find(type);
125 if (it != loggers.end())
126 return &(it->second);
127
128 if (type == LOGGER_ROOT)
129 return NULL;
130
131 std::string parentLogger = LOGGER_ROOT;
132 size_t found = type.find_last_of(".");
133 if (found != std::string::npos)
134 parentLogger = type.substr(0,found);
135
136 return GetLoggerByType(parentLogger);
137}

139inline bool Log::ShouldLog(std::string const& type, LogLevel level) const
140{
141 // TODO: Use cache to store "Type.sub1.sub2": "Type" equivalence, should
142 // Speed up in cases where requesting "Type.sub1.sub2" but only configured
143 // Logger "Type"
144
145 // Don't even look for a logger if the LogLevel is lower than lowest log levels across all loggers
146 if (level < lowestLogLevel)
147 return false;
148
149 Logger const* logger = GetLoggerByType(type);
150 if (!logger)
151 return false;
152
153 LogLevel logLevel = logger->getLogLevel();
154 return logLevel != LOG_LEVEL_DISABLED && logLevel <= level;
155}

159#define LOG_EXCEPTION_FREE(filterType__, level__, ...) \
160 { \
161 try \
162 { \
163 sLog->outMessage(filterType__, level__, __VA_ARGS__); \
164 } \
165 catch (std::exception& e) \
166 { \
167 sLog->outMessage("server", LOG_LEVEL_ERROR, "Wrong format occurred (%s) at %s:%u.", \
168 e.what(), __FILE__, __LINE__); \
169 } \
170 }

27AppenderFile::AppenderFile(uint8 id, std::string const& name, LogLevel level, AppenderFlags flags, ExtraAppenderArgs extraArgs) :
28 Appender(id, name, level, flags),
29 logfile(NULL),
30 _logDir(sLog->GetLogsDir()),
31 _fileSize(0)
32{
33 if (extraArgs.empty())
34 throw InvalidAppenderArgsException(Trinity::StringFormat("Log::CreateAppenderFromConfig: Missing file name for appender %s\n", name.c_str()));
35
36 _fileName = extraArgs[0];
37
38 char const* mode = "a";
39 if (extraArgs.size() > 1)
40 mode = extraArgs[1];
41
42 if (flags & APPENDER_FLAGS_USE_TIMESTAMP)
43 {
44 size_t dot_pos = _fileName.find_last_of(".");
45 if (dot_pos != std::string::npos)
46 _fileName.insert(dot_pos, sLog->GetLogsTimestamp());
47 else
48 _fileName += sLog->GetLogsTimestamp();
49 }
50
51 if (extraArgs.size() > 2)
52 _maxFileSize = atoi(extraArgs[2]);
53 else
54 _maxFileSize = 0;
55
56 _dynamicName = std::string::npos != _fileName.find("%s");
57 _backup = (flags & APPENDER_FLAGS_MAKE_FILE_BACKUP) != 0;
58
59 if (!_dynamicName)
60 logfile = OpenFile(_fileName, mode, !strcmp(mode, "w") && _backup);
61}

63AppenderFile::~AppenderFile()
64{
65 CloseFile();
66}

68void AppenderFile::_write(LogMessage const* message)
69{
70 bool exceedMaxSize = _maxFileSize > 0 && (_fileSize.load() + message->Size()) > _maxFileSize;
71
72 if (_dynamicName)
73 {
74 char namebuf[TRINITY_PATH_MAX];
75 snprintf(namebuf, TRINITY_PATH_MAX, _fileName.c_str(), message->param1.c_str());
76 // always use "a" with dynamic name otherwise it could delete the log we wrote in last _write() call
77 FILE* file = OpenFile(namebuf, "a", _backup || exceedMaxSize);
78 if (!file)
79 return;
80 fprintf(file, "%s%s\n", message->prefix.c_str(), message->text.c_str());
81 fflush(file);
82 _fileSize += uint64(message->Size());
83 fclose(file);
84 return;
85 }
86 else if (exceedMaxSize)
87 logfile = OpenFile(_fileName, "w", true);
88
89 if (!logfile)
90 return;
91
92 fprintf(logfile, "%s%s\n", message->prefix.c_str(), message->text.c_str());
93 fflush(logfile);
94 _fileSize += uint64(message->Size());
95}

97FILE* AppenderFile::OpenFile(std::string const& filename, std::string const& mode, bool backup)
98{
99 std::string fullName(_logDir + filename);
100 if (backup)
101 {
102 CloseFile();
103 std::string newName(fullName);
104 newName.push_back('.');
105 newName.append(LogMessage::getTimeStr(time(NULL)));
106 std::replace(newName.begin(), newName.end(), ':', '-');
107 rename(fullName.c_str(), newName.c_str()); // no error handling... if we couldn't make a backup, just ignore
108 }
109
110 if (FILE* ret = fopen(fullName.c_str(), mode.c_str()))
111 {
112 _fileSize = ftell(ret);
113 return ret;
114 }
115
116 return NULL;
117}
118
119void AppenderFile::CloseFile()
120{
121 if (logfile)
122 {
123 fclose(logfile);
124 logfile = NULL;
125 }
126}

81int main(int argc, char** argv)
82{
83 std::string configFile = _TRINITY_REALM_CONFIG;
84 std::string configService;
85 auto vm = GetConsoleArguments(argc, argv, configFile, configService);
86 // exit if help or version is enabled
87 if (vm.count("help") || vm.count("version"))
88 return 0;
89
90#if PLATFORM == PLATFORM_WINDOWS
91 if (configService.compare("install") == 0)
92 return WinServiceInstall() == true ? 0 : 1;
93 else if (configService.compare("uninstall") == 0)
94 return WinServiceUninstall() == true ? 0 : 1;
95 else if (configService.compare("run") == 0)
96 return WinServiceRun() ? 0 : 1;
97#endif
98
99 std::string configError;
100 if (!sConfigMgr->LoadInitial(configFile, configError))
101 {
102 printf("Error in config file: %s\n", configError.c_str());
103 return 1;
104 }
105
106 sLog->RegisterAppender<AppenderDB>();
107 sLog->Initialize(nullptr);
108
109 TC_LOG_INFO("server.authserver", "%s (authserver)", GitRevision::GetFullVersion());
110 TC_LOG_INFO("server.authserver", "<Ctrl-C> to stop.\n");
111 TC_LOG_INFO("server.authserver", "Using configuration file %s.", configFile.c_str());
112 TC_LOG_INFO("server.authserver", "Using SSL version: %s (library: %s)", OPENSSL_VERSION_TEXT, SSLeay_version(SSLEAY_VERSION));
113 TC_LOG_INFO("server.authserver", "Using Boost version: %i.%i.%i", BOOST_VERSION / 100000, BOOST_VERSION / 100 % 1000, BOOST_VERSION % 100);
114
115 // authserver PID file creation
116 std::string pidFile = sConfigMgr->GetStringDefault("PidFile", "");
117 if (!pidFile.empty())
118 {
119 if (uint32 pid = CreatePIDFile(pidFile))
120 TC_LOG_INFO("server.authserver", "Daemon PID: %u\n", pid);
121 else
122 {
123 TC_LOG_ERROR("server.authserver", "Cannot create PID file %s.\n", pidFile.c_str());
124 return 1;
125 }
126 }
127
128 // Initialize the database connection
129 if (!StartDB())
130 return 1;
131
132 _ioService = new boost::asio::io_service();
133
134 // Get the list of realms for the server
135 sRealmList->Initialize(*_ioService, sConfigMgr->GetIntDefault("RealmsStateUpdateDelay", 20));
136
137 if (sRealmList->GetRealms().empty())
138 {
139 TC_LOG_ERROR("server.authserver", "No valid realms specified.");
140 StopDB();
141 delete _ioService;
142 return 1;
143 }
144
145 // Start the listening port (acceptor) for auth connections
146 int32 port = sConfigMgr->GetIntDefault("RealmServerPort", 3724);
147 if (port < 0 || port > 0xFFFF)
148 {
149 TC_LOG_ERROR("server.authserver", "Specified port out of allowed range (1-65535)");
150 StopDB();
151 delete _ioService;
152 return 1;
153 }
154
155 std::string bindIp = sConfigMgr->GetStringDefault("BindIP", "0.0.0.0");
156
157 sAuthSocketMgr.StartNetwork(*_ioService, bindIp, port);
158
159 // Set signal handlers
160 boost::asio::signal_set signals(*_ioService, SIGINT, SIGTERM);
161#if PLATFORM == PLATFORM_WINDOWS
162 signals.add(SIGBREAK);
163#endif
164 signals.async_wait(SignalHandler);
165
166 // Set process priority according to configuration settings
167 SetProcessPriority("server.authserver");
168
169 // Enabled a timed callback for handling the database keep alive ping
170 _dbPingInterval = sConfigMgr->GetIntDefault("MaxPingTime", 30);
171 _dbPingTimer = new boost::asio::deadline_timer(*_ioService);
172 _dbPingTimer->expires_from_now(boost::posix_time::minutes(_dbPingInterval));
173 _dbPingTimer->async_wait(KeepDatabaseAliveHandler);
174
175 _banExpiryCheckInterval = sConfigMgr->GetIntDefault("BanExpiryCheckInterval", 60);
176 _banExpiryCheckTimer = new boost::asio::deadline_timer(*_ioService);
177 _banExpiryCheckTimer->expires_from_now(boost::posix_time::seconds(_banExpiryCheckInterval));
178 _banExpiryCheckTimer->async_wait(BanExpiryHandler);
179
180#if PLATFORM == PLATFORM_WINDOWS
181 if (m_ServiceStatus != -1)
182 {
183 _serviceStatusWatchTimer = new boost::asio::deadline_timer(*_ioService);
184 _serviceStatusWatchTimer->expires_from_now(boost::posix_time::seconds(1));
185 _serviceStatusWatchTimer->async_wait(ServiceStatusWatcher);
186 }
187#endif
188
189 // Start the io service worker loop
190 _ioService->run();
191
192 _banExpiryCheckTimer->cancel();
193 _dbPingTimer->cancel();
194
195 sAuthSocketMgr.StopNetwork();
196
197 sRealmList->Close();
198
199 // Close the Database Pool and library
200 StopDB();
201
202 TC_LOG_INFO("server.authserver", "Halting process...");
203
204 signals.cancel();
205
206 delete _banExpiryCheckTimer;
207 delete _dbPingTimer;
208 delete _ioService;
209 return 0;
210}

212/// Initialize connection to the database
213bool StartDB()
214{
215 MySQL::Library_Init();
216
217 // Load databases
218 // NOTE: While authserver is singlethreaded you should keep synch_threads == 1.
219 // Increasing it is just silly since only 1 will be used ever.
220 DatabaseLoader loader("server.authserver", DatabaseLoader::DATABASE_NONE);
221 loader
222 .AddDatabase(LoginDatabase, "Login");
223
224 if (!loader.Load())
225 return false;
226
227 TC_LOG_INFO("server.authserver", "Started auth database connection pool.");
228 sLog->SetRealmId(0); // Enables DB appenders when realm is set.
229 return true;
230}
231
232/// Close the connection to the database
233void StopDB()
234{
235 LoginDatabase.Close();
236 MySQL::Library_End();
237}

239void SignalHandler(const boost::system::error_code& error, int /*signalNumber*/)
240{
241 if (!error)
242 _ioService->stop();
243}
244
245void KeepDatabaseAliveHandler(const boost::system::error_code& error)
246{
247 if (!error)
248 {
249 TC_LOG_INFO("server.authserver", "Ping MySQL to keep connection alive");
250 LoginDatabase.KeepAlive();
251
252 _dbPingTimer->expires_from_now(boost::posix_time::minutes(_dbPingInterval));
253 _dbPingTimer->async_wait(KeepDatabaseAliveHandler);
254 }
255}
256
257void BanExpiryHandler(boost::system::error_code const& error)
258{
259 if (!error)
260 {
261 LoginDatabase.Execute(LoginDatabase.GetPreparedStatement(LOGIN_DEL_EXPIRED_IP_BANS));
262 LoginDatabase.Execute(LoginDatabase.GetPreparedStatement(LOGIN_UPD_EXPIRED_ACCOUNT_BANS));
263
264 _banExpiryCheckTimer->expires_from_now(boost::posix_time::seconds(_banExpiryCheckInterval));
265 _banExpiryCheckTimer->async_wait(BanExpiryHandler);
266 }
267}

271{
272 if (!error)

274 if (m_ServiceStatus == 0)

278 }

280 {
281 _serviceStatusWatchTimer->expires_from_now(boost::posix_time::seconds(1));
282 _serviceStatusWatchTimer->async_wait(ServiceStatusWatcher);
283 }
284 }

288variables_map GetConsoleArguments(int argc, char** argv, std::string& configFile, std::string& configService)
289{
290 options_description all("Allowed options");
291 all.add_options()
292 ("help,h", "print usage message")
293 ("version,v", "print version build info")
294 ("config,c", value<std::string>(&configFile)->default_value(_TRINITY_REALM_CONFIG), "use <arg> as configuration file")
295 ;
296#if PLATFORM == PLATFORM_WINDOWS
297 options_description win("Windows platform specific options");
298 win.add_options()
299 ("service,s", value<std::string>(&configService)->default_value(""), "Windows service options: [install | uninstall]")
300 ;
301
302 all.add(win);
303#else
304 (void)configService;
305#endif
306 variables_map variablesMap;
307 try
308 {
309 store(command_line_parser(argc, argv).options(all).allow_unregistered().run(), variablesMap);
310 notify(variablesMap);
311 }
312 catch (std::exception& e)
313 {
314 std::cerr << e.what() << "\n";
315 }
316
317 if (variablesMap.count("help"))
318 std::cout << all << "\n";
319 else if (variablesMap.count("version"))
320 std::cout << GitRevision::GetFullVersion() << "\n";
321
322 return variablesMap;
323}

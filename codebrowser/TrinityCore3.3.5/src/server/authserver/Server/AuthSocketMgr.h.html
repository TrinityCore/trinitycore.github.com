18#ifndef AuthSocketMgr_h__
19#define AuthSocketMgr_h__
20
21#include "SocketMgr.h"
22#include "AuthSession.h"
23
24class AuthSocketMgr : public SocketMgr<AuthSession>
25{
26 typedef SocketMgr<AuthSession> BaseSocketMgr;
27
28public:
29 static AuthSocketMgr& Instance()
30 {
31 static AuthSocketMgr instance;
32 return instance;
33 }
34
35 bool StartNetwork(boost::asio::io_service& service, std::string const& bindIp, uint16 port) override
36 {
37 if (!BaseSocketMgr::StartNetwork(service, bindIp, port))
38 return false;
39
40 _acceptor->AsyncAcceptWithCallback<&AuthSocketMgr::OnSocketAccept>();
41 return true;
42 }
43
44protected:
45 NetworkThread<AuthSession>* CreateThreads() const override
46 {
47 return new NetworkThread<AuthSession>[1];
48 }
49
50 static void OnSocketAccept(tcp::socket&& sock, uint32 threadIndex)
51 {
52 Instance().OnSocketOpen(std::forward<tcp::socket>(sock), threadIndex);
53 }
54};
55
56#define sAuthSocketMgr AuthSocketMgr::Instance()
57
58#endif // AuthSocketMgr_h__

117std::unordered_map<uint8, AuthHandler> AuthSession::InitHandlers()
118{
119 std::unordered_map<uint8, AuthHandler> handlers;
120
121 handlers[AUTH_LOGON_CHALLENGE] = { STATUS_CONNECTED, AUTH_LOGON_CHALLENGE_INITIAL_SIZE, &AuthSession::HandleLogonChallenge };
122 handlers[AUTH_LOGON_PROOF] = { STATUS_CONNECTED, sizeof(AUTH_LOGON_PROOF_C), &AuthSession::HandleLogonProof };
123 handlers[AUTH_RECONNECT_CHALLENGE] = { STATUS_CONNECTED, AUTH_LOGON_CHALLENGE_INITIAL_SIZE, &AuthSession::HandleReconnectChallenge };
124 handlers[AUTH_RECONNECT_PROOF] = { STATUS_CONNECTED, sizeof(AUTH_RECONNECT_PROOF_C), &AuthSession::HandleReconnectProof };
125 handlers[REALM_LIST] = { STATUS_AUTHED, REALM_LIST_PACKET_SIZE, &AuthSession::HandleRealmList };
126
127 return handlers;
128}

132void AccountInfo::LoadResult(Field* fields)
133{
134 // 0 1 2 3 4 5 6
135 //SELECT a.id, a.username, a.locked, a.lock_country, a.last_ip, a.failed_logins, ab.unbandate > UNIX_TIMESTAMP() OR ab.unbandate = ab.bandate,
136 // 7 8 9 10 11 12
137 // ab.unbandate = ab.bandate, aa.gmlevel, a.token_key, a.sha_pass_hash, a.v, a.s
138 //FROM account a LEFT JOIN account_access aa ON a.id = aa.id LEFT JOIN account_banned ab ON ab.id = a.id AND ab.active = 1 WHERE a.username = ?
139
140 Id = fields[0].GetUInt32();
141 Login = fields[1].GetString();
142 IsLockedToIP = fields[2].GetBool();
143 LockCountry = fields[3].GetString();
144 LastIP = fields[4].GetString();
145 FailedLogins = fields[5].GetUInt32();
146 IsBanned = fields[6].GetUInt64() != 0;
147 IsPermanenetlyBanned = fields[7].GetUInt64() != 0;
148 SecurityLevel = AccountTypes(fields[8].GetUInt8());
149}

159void AuthSession::Start()
160{
161 std::string ip_address = GetRemoteIpAddress().to_string();
162 TC_LOG_TRACE("session", "Accepted connection from %s", ip_address.c_str());
163
164 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_IP_INFO);
165 stmt->setString(0, ip_address);
166 stmt->setUInt32(1, inet_addr(ip_address.c_str()));
167
168 _queryCallback = std::bind(&AuthSession::CheckIpCallback, this, std::placeholders::_1);
169 _queryFuture = LoginDatabase.AsyncQuery(stmt);
170}

172bool AuthSession::Update()
173{
174 if (!AuthSocket::Update())
175 return false;
176
177 if (_queryFuture.valid() && _queryFuture.wait_for(std::chrono::seconds(0)) == std::future_status::ready)
178 {
179 auto callback = _queryCallback;
180 _queryCallback = nullptr;
181 callback(_queryFuture.get());
182 }
183
184 return true;
185}

187void AuthSession::CheckIpCallback(PreparedQueryResult result)
188{
189 if (result)
190 {
191 bool banned = false;
192 do
193 {
194 Field* fields = result->Fetch();
195 if (fields[0].GetUInt64() != 0)
196 banned = true;
197
198 if (!fields[1].GetString().empty())
199 _ipCountry = fields[1].GetString();
200
201 } while (result->NextRow());
202
203 if (banned)
204 {
205 ByteBuffer pkt;
206 pkt << uint8(AUTH_LOGON_CHALLENGE);
207 pkt << uint8(0x00);
208 pkt << uint8(WOW_FAIL_BANNED);
209 SendPacket(pkt);
210 TC_LOG_DEBUG("session", "[AuthSession::CheckIpCallback] Banned ip '%s:%d' tries to login!", GetRemoteIpAddress().to_string().c_str(), GetRemotePort());
211 return;
212 }
213 }
214
215 AsyncRead();
216}

218void AuthSession::ReadHandler()
219{
220 MessageBuffer& packet = GetReadBuffer();
221 while (packet.GetActiveSize())
222 {
223 uint8 cmd = packet.GetReadPointer()[0];
224 auto itr = Handlers.find(cmd);
225 if (itr == Handlers.end())
226 {
227 // well we dont handle this, lets just ignore it
228 packet.Reset();
229 break;
230 }
231
232 if (_status != itr->second.status)
233 {
234 CloseSocket();
235 return;
236 }
237
238 uint16 size = uint16(itr->second.packetSize);
239 if (packet.GetActiveSize() < size)
240 break;
241
242 if (cmd == AUTH_LOGON_CHALLENGE || cmd == AUTH_RECONNECT_CHALLENGE)
243 {
244 sAuthLogonChallenge_C* challenge = reinterpret_cast<sAuthLogonChallenge_C*>(packet.GetReadPointer());
245 size += challenge->size;
246 if (size > MAX_ACCEPTED_CHALLENGE_SIZE)
247 {
248 CloseSocket();
249 return;
250 }
251 }
252
253 if (packet.GetActiveSize() < size)
254 break;
255
256 if (!(*this.*itr->second.handler)())
257 {
258 CloseSocket();
259 return;
260 }
261
262 packet.ReadCompleted(size);
263 }
264
265 AsyncRead();
266}

268void AuthSession::SendPacket(ByteBuffer& packet)
269{
270 if (!IsOpen())
271 return;
272
273 if (!packet.empty())
274 {
275 MessageBuffer buffer;
276 buffer.Write(packet.contents(), packet.size());
277 QueuePacket(std::move(buffer));
278 }
279}

281bool AuthSession::HandleLogonChallenge()
282{
283 if (_sentChallenge)
284 return false;
285
286 _sentChallenge = true;
287
288 sAuthLogonChallenge_C* challenge = reinterpret_cast<sAuthLogonChallenge_C*>(GetReadBuffer().GetReadPointer());
289 if (challenge->size - (sizeof(sAuthLogonChallenge_C) - AUTH_LOGON_CHALLENGE_INITIAL_SIZE - 1) != challenge->I_len)
290 return false;
291
292 std::string login((const char*)challenge->I, challenge->I_len);
293 TC_LOG_DEBUG("server.authserver", "[AuthChallenge] '%s'", login.c_str());
294
295 if (_queryCallback)
296 {
297 ByteBuffer pkt;
298 pkt << uint8(AUTH_LOGON_CHALLENGE);
299 pkt << uint8(0x00);
300 pkt << uint8(WOW_FAIL_DB_BUSY);
301 SendPacket(pkt);
302
303 TC_LOG_DEBUG("server.authserver", "[AuthChallenge] %s attempted to log too quick after previous attempt!", login.c_str());
304 return true;
305 }
306
307 _build = challenge->build;
308 _expversion = uint8(AuthHelper::IsPostBCAcceptedClientBuild(_build) ? POST_BC_EXP_FLAG : (AuthHelper::IsPreBCAcceptedClientBuild(_build) ? PRE_BC_EXP_FLAG : NO_VALID_EXP_FLAG));
309 std::array<char, 5> os;
310 os.fill('\0');
311 memcpy(os.data(), challenge->os, sizeof(challenge->os));
312 _os = os.data();
313
314 // Restore string order as its byte order is reversed
315 std::reverse(_os.begin(), _os.end());
316
317 _localizationName.resize(4);
318 for (int i = 0; i < 4; ++i)
319 _localizationName[i] = challenge->country[4 - i - 1];
320
321 // Get the account details from the account table
322 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_LOGONCHALLENGE);
323 stmt->setString(0, login);
324
325 _queryCallback = std::bind(&AuthSession::LogonChallengeCallback, this, std::placeholders::_1);
326 _queryFuture = LoginDatabase.AsyncQuery(stmt);
327 return true;
328}

330void AuthSession::LogonChallengeCallback(PreparedQueryResult result)
331{
332 ByteBuffer pkt;
333 pkt << uint8(AUTH_LOGON_CHALLENGE);
334 pkt << uint8(0x00);
335
336 if (!result)
337 {
338 pkt << uint8(WOW_FAIL_UNKNOWN_ACCOUNT);
339 SendPacket(pkt);
340 return;
341 }
342
343 Field* fields = result->Fetch();
344
345 _accountInfo.LoadResult(fields);
346
347 std::string ipAddress = GetRemoteIpAddress().to_string();
348 uint16 port = GetRemotePort();
349
350 // If the IP is 'locked', check that the player comes indeed from the correct IP address
351 if (_accountInfo.IsLockedToIP)
352 {
353 TC_LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is locked to IP - '%s' is logging in from '%s'", _accountInfo.Login.c_str(), _accountInfo.LastIP.c_str(), ipAddress.c_str());
354 if (_accountInfo.LastIP != ipAddress)
355 {
356 pkt << uint8(WOW_FAIL_LOCKED_ENFORCED);
357 SendPacket(pkt);
358 return;
359 }
360 }
361 else
362 {
363 TC_LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is not locked to ip", _accountInfo.Login.c_str());
364 if (_accountInfo.LockCountry.empty() || _accountInfo.LockCountry == "00")
365 TC_LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is not locked to country", _accountInfo.Login.c_str());
366 else if (!_accountInfo.LockCountry.empty() && !_ipCountry.empty())
367 {
368 TC_LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is locked to country: '%s' Player country is '%s'", _accountInfo.Login.c_str(), _accountInfo.LockCountry.c_str(), _ipCountry.c_str());
369 if (_ipCountry != _accountInfo.LockCountry)
370 {
371 pkt << uint8(WOW_FAIL_UNLOCKABLE_LOCK);
372 SendPacket(pkt);
373 return;
374 }
375 }
376 }
377
378 // If the account is banned, reject the logon attempt
379 if (_accountInfo.IsBanned)
380 {
381 if (_accountInfo.IsPermanenetlyBanned)
382 {
383 pkt << uint8(WOW_FAIL_BANNED);
384 SendPacket(pkt);
385 TC_LOG_DEBUG("server.authserver", "'%s:%d' [AuthChallenge] Banned account %s tried to login!", ipAddress.c_str(), port, _accountInfo.Login.c_str());
386 return;
387 }
388 else
389 {
390 pkt << uint8(WOW_FAIL_SUSPENDED);
391 SendPacket(pkt);
392 TC_LOG_DEBUG("server.authserver", "'%s:%d' [AuthChallenge] Temporarily banned account %s tried to login!", ipAddress.c_str(), port, _accountInfo.Login.c_str());
393 return;
394 }
395 }
396
397 // Get the password from the account table, upper it, and make the SRP6 calculation
398 std::string rI = fields[10].GetString();
399
400 // Don't calculate (v, s) if there are already some in the database
401 std::string databaseV = fields[11].GetString();
402 std::string databaseS = fields[12].GetString();
403
404 TC_LOG_DEBUG("network", "database authentication values: v='%s' s='%s'", databaseV.c_str(), databaseS.c_str());
405
406 // multiply with 2 since bytes are stored as hexstring
407 if (databaseV.size() != size_t(BufferSizes::SRP_6_V) * 2 || databaseS.size() != size_t(BufferSizes::SRP_6_S) * 2)
408 SetVSFields(rI);
409 else
410 {
411 s.SetHexStr(databaseS.c_str());
412 v.SetHexStr(databaseV.c_str());
413 }
414
415 b.SetRand(19 * 8);
416 BigNumber gmod = g.ModExp(b, N);
417 B = ((v * 3) + gmod) % N;
418
419 ASSERT(gmod.GetNumBytes() <= 32);
420
421 BigNumber unk3;
422 unk3.SetRand(16 * 8);
423
424 // Fill the response packet with the result
425 if (AuthHelper::IsAcceptedClientBuild(_build))
426 pkt << uint8(WOW_SUCCESS);
427 else
428 pkt << uint8(WOW_FAIL_VERSION_INVALID);
429
430 // B may be calculated < 32B so we force minimal length to 32B
431 pkt.append(B.AsByteArray(32).get(), 32); // 32 bytes
432 pkt << uint8(1);
433 pkt.append(g.AsByteArray(1).get(), 1);
434 pkt << uint8(32);
435 pkt.append(N.AsByteArray(32).get(), 32);
436 pkt.append(s.AsByteArray(int32(BufferSizes::SRP_6_S)).get(), size_t(BufferSizes::SRP_6_S)); // 32 bytes
437 pkt.append(unk3.AsByteArray(16).get(), 16);
438 uint8 securityFlags = 0;
439
440 // Check if token is used
441 _tokenKey = fields[9].GetString();
442 if (!_tokenKey.empty())
443 securityFlags = 4;
444
445 pkt << uint8(securityFlags); // security flags (0x0...0x04)
446
447 if (securityFlags & 0x01) // PIN input
448 {
449 pkt << uint32(0);
450 pkt << uint64(0) << uint64(0); // 16 bytes hash?
451 }
452
453 if (securityFlags & 0x02) // Matrix input
454 {
455 pkt << uint8(0);
456 pkt << uint8(0);
457 pkt << uint8(0);
458 pkt << uint8(0);
459 pkt << uint64(0);
460 }
461
462 if (securityFlags & 0x04) // Security token input
463 pkt << uint8(1);
464
465 TC_LOG_DEBUG("server.authserver", "'%s:%d' [AuthChallenge] account %s is using '%s' locale (%u)",
466 ipAddress.c_str(), port, _accountInfo.Login.c_str(), _localizationName.c_str(), GetLocaleByName(_localizationName));
467
468 SendPacket(pkt);
469}

471// Logon Proof command handler
472bool AuthSession::HandleLogonProof()
473{
474 TC_LOG_DEBUG("server.authserver", "Entering _HandleLogonProof");
475 if (_sentProof)
476 return false;
477
478 _sentProof = true;
479
480 // Read the packet
481 sAuthLogonProof_C *logonProof = reinterpret_cast<sAuthLogonProof_C*>(GetReadBuffer().GetReadPointer());
482
483 // If the client has no valid version
484 if (_expversion == NO_VALID_EXP_FLAG)
485 {
486 // Check if we have the appropriate patch on the disk
487 TC_LOG_DEBUG("network", "Client with invalid version, patching is not implemented");
488 return false;
489 }
490
491 // Continue the SRP6 calculation based on data received from the client
492 BigNumber A;
493
494 A.SetBinary(logonProof->A, 32);
495
496 // SRP safeguard: abort if A == 0
497 if (A.isZero())
498 {
499 return false;
500 }
501
502 SHA1Hash sha;
503 sha.UpdateBigNumbers(&A, &B, NULL);
504 sha.Finalize();
505 BigNumber u;
506 u.SetBinary(sha.GetDigest(), 20);
507 BigNumber S = (A * (v.ModExp(u, N))).ModExp(b, N);
508
509 uint8 t[32];
510 uint8 t1[16];
511 uint8 vK[40];
512 memcpy(t, S.AsByteArray(32).get(), 32);
513
514 for (int i = 0; i < 16; ++i)
515 t1[i] = t[i * 2];
516
517 sha.Initialize();
518 sha.UpdateData(t1, 16);
519 sha.Finalize();
520
521 for (int i = 0; i < 20; ++i)
522 vK[i * 2] = sha.GetDigest()[i];
523
524 for (int i = 0; i < 16; ++i)
525 t1[i] = t[i * 2 + 1];
526
527 sha.Initialize();
528 sha.UpdateData(t1, 16);
529 sha.Finalize();
530
531 for (int i = 0; i < 20; ++i)
532 vK[i * 2 + 1] = sha.GetDigest()[i];
533
534 K.SetBinary(vK, 40);
535
536 uint8 hash[20];
537
538 sha.Initialize();
539 sha.UpdateBigNumbers(&N, NULL);
540 sha.Finalize();
541 memcpy(hash, sha.GetDigest(), 20);
542 sha.Initialize();
543 sha.UpdateBigNumbers(&g, NULL);
544 sha.Finalize();
545
546 for (int i = 0; i < 20; ++i)
547 hash[i] ^= sha.GetDigest()[i];
548
549 BigNumber t3;
550 t3.SetBinary(hash, 20);
551
552 sha.Initialize();
553 sha.UpdateData(_accountInfo.Login);
554 sha.Finalize();
555 uint8 t4[SHA_DIGEST_LENGTH];
556 memcpy(t4, sha.GetDigest(), SHA_DIGEST_LENGTH);
557
558 sha.Initialize();
559 sha.UpdateBigNumbers(&t3, NULL);
560 sha.UpdateData(t4, SHA_DIGEST_LENGTH);
561 sha.UpdateBigNumbers(&s, &A, &B, &K, NULL);
562 sha.Finalize();
563 BigNumber M;
564 M.SetBinary(sha.GetDigest(), sha.GetLength());
565
566 // Check if SRP6 results match (password is correct), else send an error
567 if (!memcmp(M.AsByteArray(sha.GetLength()).get(), logonProof->M1, 20))
568 {
569 TC_LOG_DEBUG("server.authserver", "'%s:%d' User '%s' successfully authenticated", GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), _accountInfo.Login.c_str());
570
571 // Update the sessionkey, last_ip, last login time and reset number of failed logins in the account table for this account
572 // No SQL injection (escaped user name) and IP address as received by socket
573 const char *K_hex = K.AsHexStr();
574
575 PreparedStatement *stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_LOGONPROOF);
576 stmt->setString(0, K_hex);
577 stmt->setString(1, GetRemoteIpAddress().to_string().c_str());
578 stmt->setUInt32(2, GetLocaleByName(_localizationName));
579 stmt->setString(3, _os);
580 stmt->setString(4, _accountInfo.Login);
581 LoginDatabase.DirectExecute(stmt);
582
583 OPENSSL_free((void*)K_hex);
584
585 // Finish SRP6 and send the final result to the client
586 sha.Initialize();
587 sha.UpdateBigNumbers(&A, &M, &K, NULL);
588 sha.Finalize();
589
590 // Check auth token
591 if ((logonProof->securityFlags & 0x04) || !_tokenKey.empty())
592 {
593 uint8 size = *(GetReadBuffer().GetReadPointer() + sizeof(sAuthLogonProof_C));
594 std::string token(reinterpret_cast<char*>(GetReadBuffer().GetReadPointer() + sizeof(sAuthLogonProof_C) + sizeof(size)), size);
595 GetReadBuffer().ReadCompleted(sizeof(size) + size);
596 uint32 validToken = TOTP::GenerateToken(_tokenKey.c_str());
597 _tokenKey.clear();
598 uint32 incomingToken = atoi(token.c_str());
599 if (validToken != incomingToken)
600 {
601 ByteBuffer packet;
602 packet << uint8(AUTH_LOGON_PROOF);
603 packet << uint8(WOW_FAIL_UNKNOWN_ACCOUNT);
604 packet << uint8(3);
605 packet << uint8(0);
606 SendPacket(packet);
607 return true;
608 }
609 }
610
611 ByteBuffer packet;
612 if (_expversion & POST_BC_EXP_FLAG) // 2.x and 3.x clients
613 {
614 sAuthLogonProof_S proof;
615 memcpy(proof.M2, sha.GetDigest(), 20);
616 proof.cmd = AUTH_LOGON_PROOF;
617 proof.error = 0;
618 proof.AccountFlags = 0x00800000; // 0x01 = GM, 0x08 = Trial, 0x00800000 = Pro pass (arena tournament)
619 proof.SurveyId = 0;
620 proof.unk3 = 0;
621
622 packet.resize(sizeof(proof));
623 std::memcpy(packet.contents(), &proof, sizeof(proof));
624 }
625 else
626 {
627 sAuthLogonProof_S_Old proof;
628 memcpy(proof.M2, sha.GetDigest(), 20);
629 proof.cmd = AUTH_LOGON_PROOF;
630 proof.error = 0;
631 proof.unk2 = 0x00;
632
633 packet.resize(sizeof(proof));
634 std::memcpy(packet.contents(), &proof, sizeof(proof));
635 }
636
637 SendPacket(packet);
638 _status = STATUS_AUTHED;
639 }
640 else
641 {
642 ByteBuffer packet;
643 packet << uint8(AUTH_LOGON_PROOF);
644 packet << uint8(WOW_FAIL_UNKNOWN_ACCOUNT);
645 packet << uint8(3);
646 packet << uint8(0);
647 SendPacket(packet);
648
649 TC_LOG_DEBUG("server.authserver", "'%s:%d' [AuthChallenge] account %s tried to login with invalid password!",
650 GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), _accountInfo.Login.c_str());
651
652 uint32 MaxWrongPassCount = sConfigMgr->GetIntDefault("WrongPass.MaxCount", 0);
653
654 // We can not include the failed account login hook. However, this is a workaround to still log this.
655 if (sConfigMgr->GetBoolDefault("WrongPass.Logging", false))
656 {
657 PreparedStatement* logstmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_FALP_IP_LOGGING);
658 logstmt->setString(0, _accountInfo.Login);
659 logstmt->setString(1, GetRemoteIpAddress().to_string());
660 logstmt->setString(2, "Logged on failed AccountLogin due wrong password");
661
662 LoginDatabase.Execute(logstmt);
663 }
664
665 if (MaxWrongPassCount > 0)
666 {
667 //Increment number of failed logins by one and if it reaches the limit temporarily ban that account or IP
668 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_FAILEDLOGINS);
669 stmt->setString(0, _accountInfo.Login);
670 LoginDatabase.Execute(stmt);
671
672 if (_accountInfo.FailedLogins >= MaxWrongPassCount)
673 {
674 uint32 WrongPassBanTime = sConfigMgr->GetIntDefault("WrongPass.BanTime", 600);
675 bool WrongPassBanType = sConfigMgr->GetBoolDefault("WrongPass.BanType", false);
676
677 if (WrongPassBanType)
678 {
679 stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_ACCOUNT_AUTO_BANNED);
680 stmt->setUInt32(0, _accountInfo.Id);
681 stmt->setUInt32(1, WrongPassBanTime);
682 LoginDatabase.Execute(stmt);
683
684 TC_LOG_DEBUG("server.authserver", "'%s:%d' [AuthChallenge] account %s got banned for '%u' seconds because it failed to authenticate '%u' times",
685 GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), _accountInfo.Login.c_str(), WrongPassBanTime, _accountInfo.FailedLogins);
686 }
687 else
688 {
689 stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_IP_AUTO_BANNED);
690 stmt->setString(0, GetRemoteIpAddress().to_string());
691 stmt->setUInt32(1, WrongPassBanTime);
692 LoginDatabase.Execute(stmt);
693
694 TC_LOG_DEBUG("server.authserver", "'%s:%d' [AuthChallenge] IP got banned for '%u' seconds because account %s failed to authenticate '%u' times",
695 GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), WrongPassBanTime, _accountInfo.Login.c_str(), _accountInfo.FailedLogins);
696 }
697 }
698 }
699 }
700
701 return true;
702}

704bool AuthSession::HandleReconnectChallenge()
705{
706 if (_sentChallenge)
707 return false;
708
709 _sentChallenge = true;
710
711 sAuthLogonChallenge_C* challenge = reinterpret_cast<sAuthLogonChallenge_C*>(GetReadBuffer().GetReadPointer());
712 if (challenge->size - (sizeof(sAuthLogonChallenge_C) - AUTH_LOGON_CHALLENGE_INITIAL_SIZE - 1) != challenge->I_len)
713 return false;
714
715 std::string login((const char*)challenge->I, challenge->I_len);
716 TC_LOG_DEBUG("server.authserver", "[ReconnectChallenge] '%s'", login.c_str());
717
718 if (_queryCallback)
719 {
720 ByteBuffer pkt;
721 pkt << uint8(AUTH_RECONNECT_CHALLENGE);
722 pkt << uint8(WOW_FAIL_DB_BUSY);
723 SendPacket(pkt);
724
725 TC_LOG_DEBUG("server.authserver", "[ReconnectChallenge] %s attempted to log too quick after previous attempt!", login.c_str());
726 return true;
727 }
728
729 _build = challenge->build;
730 _expversion = uint8(AuthHelper::IsPostBCAcceptedClientBuild(_build) ? POST_BC_EXP_FLAG : (AuthHelper::IsPreBCAcceptedClientBuild(_build) ? PRE_BC_EXP_FLAG : NO_VALID_EXP_FLAG));
731 std::array<char, 5> os;
732 os.fill('\0');
733 memcpy(os.data(), challenge->os, sizeof(challenge->os));
734 _os = os.data();
735
736 // Restore string order as its byte order is reversed
737 std::reverse(_os.begin(), _os.end());
738
739 _localizationName.resize(4);
740 for (int i = 0; i < 4; ++i)
741 _localizationName[i] = challenge->country[4 - i - 1];
742
743 // Get the account details from the account table
744 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_RECONNECTCHALLENGE);
745 stmt->setString(0, login);
746
747 _queryCallback = std::bind(&AuthSession::ReconnectChallengeCallback, this, std::placeholders::_1);
748 _queryFuture = LoginDatabase.AsyncQuery(stmt);
749 return true;
750}

777bool AuthSession::HandleReconnectProof()
778{
779 TC_LOG_DEBUG("server.authserver", "Entering _HandleReconnectProof");
780 if (_sentProof)
781 return false;
782
783 _sentProof = true;
784
785 sAuthReconnectProof_C *reconnectProof = reinterpret_cast<sAuthReconnectProof_C*>(GetReadBuffer().GetReadPointer());
786
787 if (_accountInfo.Login.empty() || !_reconnectProof.GetNumBytes() || !K.GetNumBytes())
788 return false;
789
790 BigNumber t1;
791 t1.SetBinary(reconnectProof->R1, 16);
792
793 SHA1Hash sha;
794 sha.Initialize();
795 sha.UpdateData(_accountInfo.Login);
796 sha.UpdateBigNumbers(&t1, &_reconnectProof, &K, NULL);
797 sha.Finalize();
798
799 if (!memcmp(sha.GetDigest(), reconnectProof->R2, SHA_DIGEST_LENGTH))
800 {
801 // Sending response
802 ByteBuffer pkt;
803 pkt << uint8(AUTH_RECONNECT_PROOF);
804 pkt << uint8(0x00);
805 pkt << uint16(0x00); // 2 bytes zeros
806 SendPacket(pkt);
807 _status = STATUS_AUTHED;
808 return true;
809 }
810 else
811 {
812 TC_LOG_ERROR("server.authserver", "'%s:%d' [ERROR] user %s tried to login, but session is invalid.", GetRemoteIpAddress().to_string().c_str(),
813 GetRemotePort(), _accountInfo.Login.c_str());
814 return false;
815 }

818tcp::endpoint const GetAddressForClient(Realm const& realm, ip::address const& clientAddr)
819{
820 ip::address realmIp;
821
822 // Attempt to send best address for client
823 if (clientAddr.is_loopback())
824 {
825 // Try guessing if realm is also connected locally
826 if (realm.LocalAddress.is_loopback() || realm.ExternalAddress.is_loopback())
827 realmIp = clientAddr;
828 else
829 {
830 // Assume that user connecting from the machine that authserver is located on
831 // has all realms available in his local network
832 realmIp = realm.LocalAddress;
833 }
834 }
835 else
836 {
837 if (clientAddr.is_v4() &&
838 (clientAddr.to_v4().to_ulong() & realm.LocalSubnetMask.to_v4().to_ulong()) ==
839 (realm.LocalAddress.to_v4().to_ulong() & realm.LocalSubnetMask.to_v4().to_ulong()))
840 {
841 realmIp = realm.LocalAddress;
842 }
843 else
844 realmIp = realm.ExternalAddress;
845 }
846
847 tcp::endpoint endpoint(realmIp, realm.Port);
848
849 // Return external IP
850 return endpoint;
851}

871void AuthSession::RealmListCallback(PreparedQueryResult result)
872{
873 std::map<uint32, uint8> characterCounts;
874 if (result)
875 {
876 do
877 {
878 Field* fields = result->Fetch();
879 characterCounts[fields[0].GetUInt32()] = fields[1].GetUInt8();
880 } while (result->NextRow());
881 }
882
883 // Circle through realms in the RealmList and construct the return packet (including # of user characters in each realm)
884 ByteBuffer pkt;
885
886 size_t RealmListSize = 0;
887 for (RealmList::RealmMap::value_type const& i : sRealmList->GetRealms())
888 {
889 const Realm &realm = i.second;
890 // don't work with realms which not compatible with the client
891 bool okBuild = ((_expversion & POST_BC_EXP_FLAG) && realm.Build == _build) || ((_expversion & PRE_BC_EXP_FLAG) && !AuthHelper::IsPreBCAcceptedClientBuild(realm.Build));
892
893 // No SQL injection. id of realm is controlled by the database.
894 uint32 flag = realm.Flags;
895 RealmBuildInfo const* buildInfo = AuthHelper::GetBuildInfo(realm.Build);
896 if (!okBuild)
897 {
898 if (!buildInfo)
899 continue;
900
901 flag |= REALM_FLAG_OFFLINE | REALM_FLAG_SPECIFYBUILD; // tell the client what build the realm is for
902 }
903
904 if (!buildInfo)
905 flag &= ~REALM_FLAG_SPECIFYBUILD;
906
907 std::string name = realm.Name;
908 if (_expversion & PRE_BC_EXP_FLAG && flag & REALM_FLAG_SPECIFYBUILD)
909 {
910 std::ostringstream ss;
911 ss << name << " (" << buildInfo->MajorVersion << '.' << buildInfo->MinorVersion << '.' << buildInfo->BugfixVersion << ')';
912 name = ss.str();
913 }
914
915 uint8 lock = (realm.AllowedSecurityLevel > _accountInfo.SecurityLevel) ? 1 : 0;
916
917 pkt << uint8(realm.Type); // realm type
918 if (_expversion & POST_BC_EXP_FLAG) // only 2.x and 3.x clients
919 pkt << uint8(lock); // if 1, then realm locked
920 pkt << uint8(flag); // RealmFlags
921 pkt << name;
922 pkt << boost::lexical_cast<std::string>(GetAddressForClient(realm, GetRemoteIpAddress()));
923 pkt << float(realm.PopulationLevel);
924 pkt << uint8(characterCounts[realm.Id.Realm]);
925 pkt << uint8(realm.Timezone); // realm category
926 if (_expversion & POST_BC_EXP_FLAG) // 2.x and 3.x clients
927 pkt << uint8(realm.Id.Realm);
928 else
929 pkt << uint8(0x0); // 1.12.1 and 1.12.2 clients
930
931 if (_expversion & POST_BC_EXP_FLAG && flag & REALM_FLAG_SPECIFYBUILD)
932 {
933 pkt << uint8(buildInfo->MajorVersion);
934 pkt << uint8(buildInfo->MinorVersion);
935 pkt << uint8(buildInfo->BugfixVersion);
936 pkt << uint16(buildInfo->Build);
937 }
938
939 ++RealmListSize;
940 }
941
942 if (_expversion & POST_BC_EXP_FLAG) // 2.x and 3.x clients
943 {
944 pkt << uint8(0x10);
945 pkt << uint8(0x00);
946 }
947 else // 1.12.1 and 1.12.2 clients
948 {
949 pkt << uint8(0x00);
950 pkt << uint8(0x02);
951 }
952
953 // make a ByteBuffer which stores the RealmList's size
954 ByteBuffer RealmListSizeBuffer;
955 RealmListSizeBuffer << uint32(0);
956 if (_expversion & POST_BC_EXP_FLAG) // only 2.x and 3.x clients
957 RealmListSizeBuffer << uint16(RealmListSize);
958 else
959 RealmListSizeBuffer << uint32(RealmListSize);
960
961 ByteBuffer hdr;
962 hdr << uint8(REALM_LIST);
963 hdr << uint16(pkt.size() + RealmListSizeBuffer.size());
964 hdr.append(RealmListSizeBuffer); // append RealmList's size buffer
965 hdr.append(pkt); // append realms in the realmlist
966 SendPacket(hdr);
967}

969// Make the SRP6 calculation from hash in dB
970void AuthSession::SetVSFields(const std::string& rI)
971{
972 s.SetRand(int32(BufferSizes::SRP_6_S) * 8);
973
974 BigNumber I;
975 I.SetHexStr(rI.c_str());
976
977 // In case of leading zeros in the rI hash, restore them
978 uint8 mDigest[SHA_DIGEST_LENGTH];
979 memcpy(mDigest, I.AsByteArray(SHA_DIGEST_LENGTH).get(), SHA_DIGEST_LENGTH);
980
981 std::reverse(mDigest, mDigest + SHA_DIGEST_LENGTH);
982
983 SHA1Hash sha;
984 sha.UpdateData(s.AsByteArray(uint32(BufferSizes::SRP_6_S)).get(), (uint32(BufferSizes::SRP_6_S)));
985 sha.UpdateData(mDigest, SHA_DIGEST_LENGTH);
986 sha.Finalize();
987 BigNumber x;
988 x.SetBinary(sha.GetDigest(), sha.GetLength());
989 v = g.ModExp(x, N);
990
991 // No SQL injection (username escaped)
992 char *v_hex, *s_hex;
993 v_hex = v.AsHexStr();
994 s_hex = s.AsHexStr();
995
996 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_VS);
997 stmt->setString(0, v_hex);
998 stmt->setString(1, s_hex);
999 stmt->setString(2, _accountInfo.Login);
1000 LoginDatabase.Execute(stmt);
1001
1002 OPENSSL_free(v_hex);
1003 OPENSSL_free(s_hex);
1004}

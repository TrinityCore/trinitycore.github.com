26RealmList::RealmList() : _updateInterval(0), _updateTimer(nullptr), _resolver(nullptr) { }
27RealmList::~RealmList()
28{
29 delete _resolver;
30 delete _updateTimer;
31}

33// Load the realm list from the database
34void RealmList::Initialize(boost::asio::io_service& ioService, uint32 updateInterval)
35{
36 _updateInterval = updateInterval;
37 _updateTimer = new boost::asio::deadline_timer(ioService);
38 _resolver = new boost::asio::ip::tcp::resolver(ioService);
39
40 // Get the content of the realmlist table in the database
41 UpdateRealms(true, boost::system::error_code());
42}
43
44void RealmList::Close()
45{
46 _updateTimer->cancel();
47}

49void RealmList::UpdateRealm(RealmHandle const& id, uint32 build, const std::string& name, ip::address const& address, ip::address const& localAddr,
50 ip::address const& localSubmask, uint16 port, uint8 icon, RealmFlags flag, uint8 timezone, AccountTypes allowedSecurityLevel,
51 float population)
52{
53 // Create new if not exist or update existed
54 Realm& realm = _realms[id];
55
56 realm.Id = id;
57 realm.Build = build;
58 realm.Name = name;
59 realm.Type = icon;
60 realm.Flags = flag;
61 realm.Timezone = timezone;
62 realm.AllowedSecurityLevel = allowedSecurityLevel;
63 realm.PopulationLevel = population;
64 realm.ExternalAddress = address;
65 realm.LocalAddress = localAddr;
66 realm.LocalSubnetMask = localSubmask;
67 realm.Port = port;
68}

70void RealmList::UpdateRealms(bool init, boost::system::error_code const& error)
71{
72 if (error)
73 return;
74
75 TC_LOG_INFO("server.authserver", "Updating Realm List...");
76
77 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_REALMLIST);
78 PreparedQueryResult result = LoginDatabase.Query(stmt);
79
80 // Circle through results and add them to the realm map
81 if (result)
82 {
83 do
84 {
85 try
86 {
87 boost::asio::ip::tcp::resolver::iterator end;
88
89 Field* fields = result->Fetch();
90 uint32 realmId = fields[0].GetUInt32();
91 std::string name = fields[1].GetString();
92 boost::asio::ip::tcp::resolver::query externalAddressQuery(ip::tcp::v4(), fields[2].GetString(), "");
93
94 boost::system::error_code ec;
95 boost::asio::ip::tcp::resolver::iterator endPoint = _resolver->resolve(externalAddressQuery, ec);
96 if (endPoint == end || ec)
97 {
98 TC_LOG_ERROR("server.authserver", "Could not resolve address %s", fields[2].GetString().c_str());
99 return;
100 }
101
102 ip::address externalAddress = (*endPoint).endpoint().address();
103
104 boost::asio::ip::tcp::resolver::query localAddressQuery(ip::tcp::v4(), fields[3].GetString(), "");
105 endPoint = _resolver->resolve(localAddressQuery, ec);
106 if (endPoint == end || ec)
107 {
108 TC_LOG_ERROR("server.authserver", "Could not resolve address %s", fields[3].GetString().c_str());
109 return;
110 }
111
112 ip::address localAddress = (*endPoint).endpoint().address();
113
114 boost::asio::ip::tcp::resolver::query localSubmaskQuery(ip::tcp::v4(), fields[4].GetString(), "");
115 endPoint = _resolver->resolve(localSubmaskQuery, ec);
116 if (endPoint == end || ec)
117 {
118 TC_LOG_ERROR("server.authserver", "Could not resolve address %s", fields[4].GetString().c_str());
119 return;
120 }
121
122 ip::address localSubmask = (*endPoint).endpoint().address();
123
124 uint16 port = fields[5].GetUInt16();
125 uint8 icon = fields[6].GetUInt8();
126 if (icon == REALM_TYPE_FFA_PVP)
127 icon = REALM_TYPE_PVP;
128 if (icon >= MAX_CLIENT_REALM_TYPE)
129 icon = REALM_TYPE_NORMAL;
130 RealmFlags flag = RealmFlags(fields[7].GetUInt8());
131 uint8 timezone = fields[8].GetUInt8();
132 uint8 allowedSecurityLevel = fields[9].GetUInt8();
133 float pop = fields[10].GetFloat();
134 uint32 build = fields[11].GetUInt32();
135
136 RealmHandle id{ realmId };
137
138 UpdateRealm(id, build, name, externalAddress, localAddress, localSubmask, port, icon, flag,
139 timezone, (allowedSecurityLevel <= SEC_ADMINISTRATOR ? AccountTypes(allowedSecurityLevel) : SEC_ADMINISTRATOR), pop);
140
141 if (init)
142 TC_LOG_INFO("server.authserver", "Added realm \"%s\" at %s:%u.", name.c_str(), externalAddress.to_string().c_str(), port);
143 }
144 catch (std::exception& ex)
145 {
146 TC_LOG_ERROR("server.authserver", "Realmlist::UpdateRealms has thrown an exception: %s", ex.what());
147 ABORT();
148 }
149 }
150 while (result->NextRow());
151 }
152
153 if (_updateInterval)
154 {
155 _updateTimer->expires_from_now(boost::posix_time::seconds(_updateInterval));
156 _updateTimer->async_wait(std::bind(&RealmList::UpdateRealms, this, false, std::placeholders::_1));
157 }
158}

160Realm const* RealmList::GetRealm(RealmHandle const& id) const
161{
162 auto itr = _realms.find(id);
163 if (itr != _realms.end())
164 return &itr->second;
165
166 return NULL;
167}

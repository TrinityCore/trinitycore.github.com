47 bool IsPreBCAcceptedClientBuild(int build)
48 {
49 for (int i = 0; PreBcAcceptedClientBuilds[i].Build; ++i)
50 if (PreBcAcceptedClientBuilds[i].Build == build)
51 return true;
52
53 return false;
54 }
55
56 bool IsPostBCAcceptedClientBuild(int build)
57 {
58 for (int i = 0; PostBcAcceptedClientBuilds[i].Build; ++i)
59 if (PostBcAcceptedClientBuilds[i].Build == build)
60 return true;
61
62 return false;
63 }
64
65 bool IsAcceptedClientBuild(int build)
66 {
67 return (IsPostBCAcceptedClientBuild(build) || IsPreBCAcceptedClientBuild(build));
68 }
69
70 RealmBuildInfo const* GetBuildInfo(int build)
71 {
72 for (int i = 0; PostBcAcceptedClientBuilds[i].Build; ++i)
73 if (PostBcAcceptedClientBuilds[i].Build == build)
74 return &PostBcAcceptedClientBuilds[i];
75
76 for (int i = 0; PreBcAcceptedClientBuilds[i].Build; ++i)
77 if (PreBcAcceptedClientBuilds[i].Build == build)
78 return &PreBcAcceptedClientBuilds[i];
79
80 return NULL;
81 }
82}

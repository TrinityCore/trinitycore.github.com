21int base32_decode(const char* encoded, char* result, int bufSize)
22{
23 // Base32 implementation
24 // Copyright 2010 Google Inc.
25 // Author: Markus Gutschke
26 // Licensed under the Apache License, Version 2.0
27 int buffer = 0;
28 int bitsLeft = 0;
29 int count = 0;
30 for (const char *ptr = encoded; count < bufSize && *ptr; ++ptr)
31 {
32 char ch = *ptr;
33 if (ch == ' ' || ch == '\t' || ch == '\r' || ch == '\n' || ch == '-')
34 continue;
35 buffer <<= 5;
36
37 // Deal with commonly mistyped characters
38 if (ch == '0')
39 ch = 'O';
40 else if (ch == '1')
41 ch = 'L';
42 else if (ch == '8')
43 ch = 'B';
44
45 // Look up one base32 digit
46 if ((ch >= 'A' && ch <= 'Z') || (ch >= 'a' && ch <= 'z'))
47 ch = (ch & 0x1F) - 1;
48 else if (ch >= '2' && ch <= '7')
49 ch -= '2' - 26;
50 else
51 return -1;
52
53 buffer |= ch;
54 bitsLeft += 5;
55 if (bitsLeft >= 8)
56 {
57 result[count++] = buffer >> (bitsLeft - 8);
58 bitsLeft -= 8;
59 }
60 }
61
62 if (count < bufSize)
63 result[count] = '\000';
64 return count;
65}

67#define HMAC_RES_SIZE 20

69namespace TOTP
70{
71 unsigned int GenerateToken(const char* b32key)
72 {
73 size_t keySize = strlen(b32key);
74 int bufsize = (keySize + 7)/8*5;
75 char* encoded = new char[bufsize];
76 memset(encoded, 0, bufsize);
77 unsigned int hmacResSize = HMAC_RES_SIZE;
78 unsigned char hmacRes[HMAC_RES_SIZE];
79 unsigned long timestamp = time(NULL)/30;
80 unsigned char challenge[8];
81
82 for (int i = 8; i--;timestamp >>= 8)
83 challenge[i] = timestamp;
84
85 base32_decode(b32key, encoded, bufsize);
86 HMAC(EVP_sha1(), encoded, bufsize, challenge, 8, hmacRes, &hmacResSize);
87 unsigned int offset = hmacRes[19] & 0xF;
88 unsigned int truncHash = (hmacRes[offset] << 24) | (hmacRes[offset+1] << 16 )| (hmacRes[offset+2] << 8) | (hmacRes[offset+3]);
89 truncHash &= 0x7FFFFFFF;
90
91 delete[] encoded;
92
93 return truncHash % 1000000;
94 }

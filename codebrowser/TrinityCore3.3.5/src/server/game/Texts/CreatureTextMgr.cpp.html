75void CreatureTextMgr::LoadCreatureTexts()
76{
77 uint32 oldMSTime = getMSTime();
78
79 mTextMap.clear(); // for reload case
80 //all currently used temp texts are NOT reset
81
82 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_CREATURE_TEXT);
83 PreparedQueryResult result = WorldDatabase.Query(stmt);
84
85 if (!result)
86 {
87 TC_LOG_INFO("server.loading", ">> Loaded 0 ceature texts. DB table `creature_text` is empty.");
88
89 return;
90 }
91
92 uint32 textCount = 0;
93
94 do
95 {
96 Field* fields = result->Fetch();
97 CreatureTextEntry temp;
98
99 temp.entry = fields[0].GetUInt32();
100 temp.group = fields[1].GetUInt8();
101 temp.id = fields[2].GetUInt8();
102 temp.text = fields[3].GetString();
103 temp.type = ChatMsg(fields[4].GetUInt8());
104 temp.lang = Language(fields[5].GetUInt8());
105 temp.probability = fields[6].GetFloat();
106 temp.emote = Emote(fields[7].GetUInt32());
107 temp.duration = fields[8].GetUInt32();
108 temp.sound = fields[9].GetUInt32();
109 temp.BroadcastTextId = fields[10].GetUInt32();
110 temp.TextRange = CreatureTextRange(fields[11].GetUInt8());
111
112 if (temp.sound)
113 {
114 if (!sSoundEntriesStore.LookupEntry(temp.sound))
115 {
116 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Entry %u, Group %u in table `creature_text` has Sound %u but sound does not exist.", temp.entry, temp.group, temp.sound);
117 temp.sound = 0;
118 }
119 }
120
121 if (!GetLanguageDescByID(temp.lang))
122 {
123 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Entry %u, Group %u in table `creature_text` using Language %u but Language does not exist.", temp.entry, temp.group, uint32(temp.lang));
124 temp.lang = LANG_UNIVERSAL;
125 }
126
127 if (temp.type >= MAX_CHAT_MSG_TYPE)
128 {
129 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Entry %u, Group %u in table `creature_text` has Type %u but this Chat Type does not exist.", temp.entry, temp.group, uint32(temp.type));
130 temp.type = CHAT_MSG_SAY;
131 }
132
133 if (temp.emote)
134 {
135 if (!sEmotesStore.LookupEntry(temp.emote))
136 {
137 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Entry %u, Group %u in table `creature_text` has Emote %u but emote does not exist.", temp.entry, temp.group, uint32(temp.emote));
138 temp.emote = EMOTE_ONESHOT_NONE;
139 }
140 }
141
142 if (temp.BroadcastTextId)
143 {
144 if (!sObjectMgr->GetBroadcastText(temp.BroadcastTextId))
145 {
146 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Entry %u, Group %u, Id %u in table `creature_text` has non-existing or incompatible BroadcastTextId %u.", temp.entry, temp.group, temp.id, temp.BroadcastTextId);
147 temp.BroadcastTextId = 0;
148 }
149 }
150
151 if (temp.TextRange > TEXT_RANGE_WORLD)
152 {
153 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Entry %u, Group %u, Id %u in table `creature_text` has incorrect TextRange %u.", temp.entry, temp.group, temp.id, temp.TextRange);
154 temp.TextRange = TEXT_RANGE_NORMAL;
155 }
156
157 // add the text into our entry's group
158 mTextMap[temp.entry][temp.group].push_back(temp);
159
160 ++textCount;
161 }
162 while (result->NextRow());
163
164 TC_LOG_INFO("server.loading", ">> Loaded %u creature texts for " SZFMTD " creatures in %u ms", textCount, mTextMap.size(), GetMSTimeDiffToNow(oldMSTime));
165}

167void CreatureTextMgr::LoadCreatureTextLocales()
168{
169 uint32 oldMSTime = getMSTime();
170
171 mLocaleTextMap.clear(); // for reload case
172
173 QueryResult result = WorldDatabase.Query("SELECT entry, groupid, id, text_loc1, text_loc2, text_loc3, text_loc4, text_loc5, text_loc6, text_loc7, text_loc8 FROM locales_creature_text");
174
175 if (!result)
176 return;
177
178 uint32 textCount = 0;
179
180 do
181 {
182 Field* fields = result->Fetch();
183 CreatureTextLocale& loc = mLocaleTextMap[CreatureTextId(fields[0].GetUInt32(), uint32(fields[1].GetUInt8()), uint32(fields[2].GetUInt8()))];
184 for (uint8 i = TOTAL_LOCALES - 1; i > 0; --i)
185 {
186 LocaleConstant locale = LocaleConstant(i);
187 ObjectMgr::AddLocaleString(fields[3 + i - 1].GetString(), locale, loc.Text);
188 }
189
190 ++textCount;
191 } while (result->NextRow());
192
193 TC_LOG_INFO("server.loading", ">> Loaded %u creature localized texts in %u ms", textCount, GetMSTimeDiffToNow(oldMSTime));
194}

196uint32 CreatureTextMgr::SendChat(Creature* source, uint8 textGroup, WorldObject const* whisperTarget /*= nullptr*/, ChatMsg msgType /*= CHAT_MSG_ADDON*/, Language language /*= LANG_ADDON*/, CreatureTextRange range /*= TEXT_RANGE_NORMAL*/, uint32 sound /*= 0*/, Team team /*= TEAM_OTHER*/, bool gmOnly /*= false*/, Player* srcPlr /*= nullptr*/)
197{
198 if (!source)
199 return 0;
200
201 CreatureTextMap::const_iterator sList = mTextMap.find(source->GetEntry());
202 if (sList == mTextMap.end())
203 {
204 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Could not find Text for Creature %s (Entry %u, GUID %u) in 'creature_text' table. Ignoring.", source->GetName().c_str(), source->GetEntry(), source->GetGUID().GetCounter());
205 return 0;
206 }
207
208 CreatureTextHolder const& textHolder = sList->second;
209 CreatureTextHolder::const_iterator itr = textHolder.find(textGroup);
210 if (itr == textHolder.end())
211 {
212 TC_LOG_ERROR("sql.sql", "CreatureTextMgr: Could not find TextGroup %u for Creature %s (Entry %u, GUID %u) in 'creature_text' table. Ignoring.", uint32(textGroup), source->GetName().c_str(), source->GetEntry(), source->GetGUID().GetCounter());
213 return 0;
214 }
215
216 CreatureTextGroup const& textGroupContainer = itr->second; //has all texts in the group
217 CreatureTextRepeatIds repeatGroup = GetRepeatGroup(source, textGroup);//has all textIDs from the group that were already said
218 CreatureTextGroup tempGroup;//will use this to talk after sorting repeatGroup
219
220 for (CreatureTextGroup::const_iterator giter = textGroupContainer.begin(); giter != textGroupContainer.end(); ++giter)
221 if (std::find(repeatGroup.begin(), repeatGroup.end(), giter->id) == repeatGroup.end())
222 tempGroup.push_back(*giter);
223
224 if (tempGroup.empty())
225 {
226 source->ClearTextRepeatGroup(textGroup);
227 tempGroup = textGroupContainer;
228 }
229
230 auto iter = Trinity::Containers::SelectRandomWeightedContainerElement(tempGroup, [](CreatureTextEntry const& t) -> double
231 {
232 return t.probability;
233 });
234
235 ChatMsg finalType = (msgType == CHAT_MSG_ADDON) ? iter->type : msgType;
236 Language finalLang = (language == LANG_ADDON) ? iter->lang : language;
237 uint32 finalSound = sound ? sound : iter->sound;
238
239 if (range == TEXT_RANGE_NORMAL)
240 range = iter->TextRange;
241
242 if (finalSound)
243 SendSound(source, finalSound, finalType, whisperTarget, range, team, gmOnly);
244
245 Unit* finalSource = source;
246 if (srcPlr)
247 finalSource = srcPlr;
248
249 if (iter->emote)
250 SendEmote(finalSource, iter->emote);
251
252 if (srcPlr)
253 {
254 PlayerTextBuilder builder(source, finalSource, finalSource->getGender(), finalType, iter->group, iter->id, finalLang, whisperTarget);
255 SendChatPacket(finalSource, builder, finalType, whisperTarget, range, team, gmOnly);
256 }
257 else
258 {
259 CreatureTextBuilder builder(finalSource, finalSource->getGender(), finalType, iter->group, iter->id, finalLang, whisperTarget);
260 SendChatPacket(finalSource, builder, finalType, whisperTarget, range, team, gmOnly);
261 }
262
263 SetRepeatId(source, textGroup, iter->id);
264 return iter->duration;
265}

267float CreatureTextMgr::GetRangeForChatType(ChatMsg msgType) const
268{
269 float dist = sWorld->getFloatConfig(CONFIG_LISTEN_RANGE_SAY);
270 switch (msgType)
271 {
272 case CHAT_MSG_MONSTER_YELL:
273 dist = sWorld->getFloatConfig(CONFIG_LISTEN_RANGE_YELL);
274 break;
275 case CHAT_MSG_MONSTER_EMOTE:
276 case CHAT_MSG_RAID_BOSS_EMOTE:
277 dist = sWorld->getFloatConfig(CONFIG_LISTEN_RANGE_TEXTEMOTE);
278 break;
279 default:
280 break;
281 }
282
283 return dist;
284}

286void CreatureTextMgr::SendSound(Creature* source, uint32 sound, ChatMsg msgType, WorldObject const* whisperTarget, CreatureTextRange range, Team team, bool gmOnly)
287{

296void CreatureTextMgr::SendNonChatPacket(WorldObject* source, WorldPacket* data, ChatMsg msgType, WorldObject const* whisperTarget, CreatureTextRange range, Team team, bool gmOnly) const
297{
298 switch (msgType)
299 {
300 case CHAT_MSG_MONSTER_PARTY:
301 {
302 if (!whisperTarget)
303 return;
304
305 if (Player const* whisperPlayer = whisperTarget->ToPlayer())
306 {
307 if (Group const* group = whisperPlayer->GetGroup())
308 group->BroadcastWorker([data](Player* player) { player->SendDirectMessage(data); });
309 }
310 return;
311 }
312 case CHAT_MSG_MONSTER_WHISPER:
313 case CHAT_MSG_RAID_BOSS_WHISPER:
314 {
315 if (range == TEXT_RANGE_NORMAL) // ignores team and gmOnly
316 {
317 if (!whisperTarget || whisperTarget->GetTypeId() != TYPEID_PLAYER)
318 return;
319
320 whisperTarget->ToPlayer()->GetSession()->SendPacket(data);
321 return;
322 }
323 break;
324 }
325 default:
326 break;
327 }
328
329 switch (range)
330 {
331 case TEXT_RANGE_AREA:
332 {
333 uint32 areaId = source->GetAreaId();
334 Map::PlayerList const& players = source->GetMap()->GetPlayers();
335 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
336 if (itr->GetSource()->GetAreaId() == areaId && (!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
337 itr->GetSource()->GetSession()->SendPacket(data);
338 return;
339 }
340 case TEXT_RANGE_ZONE:
341 {
342 uint32 zoneId = source->GetZoneId();
343 Map::PlayerList const& players = source->GetMap()->GetPlayers();
344 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
345 if (itr->GetSource()->GetZoneId() == zoneId && (!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
346 itr->GetSource()->GetSession()->SendPacket(data);
347 return;
348 }
349 case TEXT_RANGE_MAP:
350 {
351 Map::PlayerList const& players = source->GetMap()->GetPlayers();
352 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
353 if ((!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
354 itr->GetSource()->GetSession()->SendPacket(data);
355 return;
356 }
357 case TEXT_RANGE_WORLD:
358 {
359 SessionMap const& smap = sWorld->GetAllSessions();
360 for (SessionMap::const_iterator iter = smap.begin(); iter != smap.end(); ++iter)
361 if (Player* player = iter->second->GetPlayer())
362 if (player->GetSession() && (!team || Team(player->GetTeam()) == team) && (!gmOnly || player->IsGameMaster()))
363 player->GetSession()->SendPacket(data);
364 return;
365 }
366 case TEXT_RANGE_NORMAL:
367 default:
368 break;
369 }
370
371 float dist = GetRangeForChatType(msgType);
372 source->SendMessageToSetInRange(data, dist, true);
373}

398bool CreatureTextMgr::TextExist(uint32 sourceEntry, uint8 textGroup)
399{
400 if (!sourceEntry)
401 return false;
402
403 CreatureTextMap::const_iterator sList = mTextMap.find(sourceEntry);
404 if (sList == mTextMap.end())
405 {
406 TC_LOG_DEBUG("entities.unit", "CreatureTextMgr::TextExist: Could not find Text for Creature (entry %u) in 'creature_text' table.", sourceEntry);
407 return false;
408 }
409
410 CreatureTextHolder const& textHolder = sList->second;
411 CreatureTextHolder::const_iterator itr = textHolder.find(textGroup);
412 if (itr == textHolder.end())
413 {
414 TC_LOG_DEBUG("entities.unit", "CreatureTextMgr::TextExist: Could not find TextGroup %u for Creature (entry %u).", uint32(textGroup), sourceEntry);
415 return false;
416 }
417
418 return true;
419}

421std::string CreatureTextMgr::GetLocalizedChatString(uint32 entry, uint8 gender, uint8 textGroup, uint32 id, LocaleConstant locale) const
422{
423 CreatureTextMap::const_iterator mapitr = mTextMap.find(entry);
424 if (mapitr == mTextMap.end())
425 return "";
426
427 CreatureTextHolder::const_iterator holderItr = mapitr->second.find(textGroup);
428 if (holderItr == mapitr->second.end())
429 return "";
430
431 CreatureTextGroup::const_iterator groupItr = holderItr->second.begin();
432 for (; groupItr != holderItr->second.end(); ++groupItr)
433 if (groupItr->id == id)
434 break;
435
436 if (groupItr == holderItr->second.end())
437 return "";
438
439 if (locale > MAX_LOCALES)
440 locale = DEFAULT_LOCALE;
441
442 std::string baseText = "";
443 BroadcastText const* bct = sObjectMgr->GetBroadcastText(groupItr->BroadcastTextId);
444
445 if (bct)
446 baseText = bct->GetText(locale, gender);
447 else
448 baseText = groupItr->text;
449
450 if (locale != DEFAULT_LOCALE && !bct)
451 {
452 LocaleCreatureTextMap::const_iterator locItr = mLocaleTextMap.find(CreatureTextId(entry, uint32(textGroup), id));
453 if (locItr != mLocaleTextMap.end())
454 ObjectMgr::GetLocaleString(locItr->second.Text, locale, baseText);
455 }
456
457 return baseText;
458}

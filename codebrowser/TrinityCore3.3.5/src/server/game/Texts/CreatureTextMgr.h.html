118template<class Builder>
119class CreatureTextLocalizer
120{
121 public:
122 CreatureTextLocalizer(Builder const& builder, ChatMsg msgType) : _builder(builder), _msgType(msgType)
123 {
124 _packetCache.resize(TOTAL_LOCALES, NULL);
125 }
126
127 ~CreatureTextLocalizer()
128 {
129 for (size_t i = 0; i < _packetCache.size(); ++i)
130 {
131 if (_packetCache[i])
132 delete _packetCache[i]->first;
133 delete _packetCache[i];
134 }
135 }
136
137 void operator()(Player* player)
138 {
139 LocaleConstant loc_idx = player->GetSession()->GetSessionDbLocaleIndex();
140 WorldPacket* messageTemplate;
141 size_t whisperGUIDpos;
142
143 // create if not cached yet
144 if (!_packetCache[loc_idx])
145 {
146 messageTemplate = new WorldPacket();
147 whisperGUIDpos = _builder(messageTemplate, loc_idx);
148 ASSERT(messageTemplate->GetOpcode() != MSG_NULL_ACTION);
149 _packetCache[loc_idx] = new std::pair<WorldPacket*, size_t>(messageTemplate, whisperGUIDpos);
150 }
151 else
152 {
153 messageTemplate = _packetCache[loc_idx]->first;
154 whisperGUIDpos = _packetCache[loc_idx]->second;
155 }
156
157 WorldPacket data(*messageTemplate);
158 switch (_msgType)
159 {
160 case CHAT_MSG_MONSTER_WHISPER:
161 case CHAT_MSG_RAID_BOSS_WHISPER:
162 data.put<uint64>(whisperGUIDpos, player->GetGUID().GetRawValue());
163 break;
164 default:
165 break;
166 }
167
168 player->SendDirectMessage(&data);
169 }
170
171 private:
172 std::vector<std::pair<WorldPacket*, size_t>* > _packetCache;
173 Builder const& _builder;
174 ChatMsg _msgType;
175};

177template<class Builder>
178void CreatureTextMgr::SendChatPacket(WorldObject* source, Builder const& builder, ChatMsg msgType, WorldObject const* whisperTarget /*= nullptr*/, CreatureTextRange range /*= TEXT_RANGE_NORMAL*/, Team team /*= TEAM_OTHER*/, bool gmOnly /*= false*/) const
179{
180 if (!source)
181 return;
182
183 CreatureTextLocalizer<Builder> localizer(builder, msgType);
184
185 switch (msgType)
186 {
187 case CHAT_MSG_MONSTER_PARTY:
188 {
189 if (!whisperTarget)
190 return;
191
192 if (Player* whisperPlayer = const_cast<Player*>(whisperTarget->ToPlayer()))
193 {
194 if (Group* group = whisperPlayer->GetGroup())
195 group->BroadcastWorker(localizer);
196 }
197 return;
198 }
199 case CHAT_MSG_MONSTER_WHISPER:
200 case CHAT_MSG_RAID_BOSS_WHISPER:
201 {
202 if (range == TEXT_RANGE_NORMAL) // ignores team and gmOnly
203 {
204 if (!whisperTarget || whisperTarget->GetTypeId() != TYPEID_PLAYER)
205 return;
206
207 localizer(const_cast<Player*>(whisperTarget->ToPlayer()));
208 return;
209 }
210 break;
211 }
212 default:
213 break;
214 }
215
216 switch (range)
217 {
218 case TEXT_RANGE_AREA:
219 {
220 uint32 areaId = source->GetAreaId();
221 Map::PlayerList const& players = source->GetMap()->GetPlayers();
222 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
223 if (itr->GetSource()->GetAreaId() == areaId && (!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
224 localizer(itr->GetSource());
225 return;
226 }
227 case TEXT_RANGE_ZONE:
228 {
229 uint32 zoneId = source->GetZoneId();
230 Map::PlayerList const& players = source->GetMap()->GetPlayers();
231 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
232 if (itr->GetSource()->GetZoneId() == zoneId && (!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
233 localizer(itr->GetSource());
234 return;
235 }
236 case TEXT_RANGE_MAP:
237 {
238 Map::PlayerList const& players = source->GetMap()->GetPlayers();
239 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
240 if ((!team || Team(itr->GetSource()->GetTeam()) == team) && (!gmOnly || itr->GetSource()->IsGameMaster()))
241 localizer(itr->GetSource());
242 return;
243 }
244 case TEXT_RANGE_WORLD:
245 {
246 SessionMap const& smap = sWorld->GetAllSessions();
247 for (SessionMap::const_iterator iter = smap.begin(); iter != smap.end(); ++iter)
248 if (Player* player = iter->second->GetPlayer())
249 if ((!team || Team(player->GetTeam()) == team) && (!gmOnly || player->IsGameMaster()))
250 localizer(player);
251 return;
252 }
253 case TEXT_RANGE_NORMAL:
254 default:
255 break;
256 }
257
258 float dist = GetRangeForChatType(msgType);
259 Trinity::PlayerDistWorker<CreatureTextLocalizer<Builder> > worker(source, dist, localizer);
260 source->VisitNearbyWorldObject(dist, worker);
261}

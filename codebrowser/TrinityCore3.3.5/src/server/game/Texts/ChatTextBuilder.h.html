18#ifndef __CHATTEXT_BUILDER_H
19#define __CHATTEXT_BUILDER_H

21#include "Chat.h"
22#include "ObjectMgr.h"

24namespace Trinity
25{
26 class BroadcastTextBuilder
27 {
28 public:
29 BroadcastTextBuilder(Unit const* obj, ChatMsg msgType, uint32 textId, WorldObject const* target = nullptr, uint32 achievementId = 0)
30 : _source(obj), _msgType(msgType), _textId(textId), _target(target), _achievementId(achievementId) { }
31
32 void operator()(WorldPacket& data, LocaleConstant locale)
33 {
34 BroadcastText const* bct = sObjectMgr->GetBroadcastText(_textId);
35 ChatHandler::BuildChatPacket(data, _msgType, bct ? Language(bct->Language) : LANG_UNIVERSAL, _source, _target, bct ? bct->GetText(locale, _source->getGender()) : "", _achievementId, "", locale);
36 }
37
38 size_t operator()(WorldPacket* data, LocaleConstant locale) const
39 {
40 BroadcastText const* bct = sObjectMgr->GetBroadcastText(_textId);
41 return ChatHandler::BuildChatPacket(*data, _msgType, bct ? Language(bct->Language) : LANG_UNIVERSAL, _source, _target, bct ? bct->GetText(locale, _source->getGender()) : "", _achievementId, "", locale);
42 }
43
44 private:
45 Unit const* _source;
46 ChatMsg _msgType;
47 uint32 _textId;
48 WorldObject const* _target;
49 uint32 _achievementId;
50 };
51
52 class CustomChatTextBuilder
53 {
54 public:
55 CustomChatTextBuilder(WorldObject const* obj, ChatMsg msgType, std::string const& text, Language language = LANG_UNIVERSAL, WorldObject const* target = nullptr)
56 : _source(obj), _msgType(msgType), _text(text), _language(language), _target(target) { }
57
58 void operator()(WorldPacket& data, LocaleConstant locale)
59 {
60 ChatHandler::BuildChatPacket(data, _msgType, _language, _source, _target, _text, 0, "", locale);
61 }
62
63 private:
64 WorldObject const* _source;
65 ChatMsg _msgType;
66 std::string _text;
67 Language _language;
68 WorldObject const* _target;
69 };
70}

46bool Transport::Create(ObjectGuid::LowType guidlow, uint32 entry, uint32 mapid, float x, float y, float z, float ang, uint32 animprogress)
47{
48 Relocate(x, y, z, ang);
49
50 if (!IsPositionValid())
51 {
52 TC_LOG_ERROR("entities.transport", "Transport (GUID: %u) not created. Suggested coordinates isn't valid (X: %f Y: %f)",
53 guidlow, x, y);
54 return false;
55 }
56
57 Object::_Create(guidlow, 0, HighGuid::Mo_Transport);
58
59 GameObjectTemplate const* goinfo = sObjectMgr->GetGameObjectTemplate(entry);
60
61 if (!goinfo)
62 {
63 TC_LOG_ERROR("sql.sql", "Transport not created: entry in `gameobject_template` not found, guidlow: %u map: %u (X: %f Y: %f Z: %f) ang: %f", guidlow, mapid, x, y, z, ang);
64 return false;
65 }
66
67 m_goInfo = goinfo;
68
69 TransportTemplate const* tInfo = sTransportMgr->GetTransportTemplate(entry);
70 if (!tInfo)
71 {
72 TC_LOG_ERROR("sql.sql", "Transport %u (name: %s) will not be created, missing `transport_template` entry.", entry, goinfo->name.c_str());
73 return false;
74 }
75
76 _transportInfo = tInfo;
77
78 // initialize waypoints
79 _nextFrame = tInfo->keyFrames.begin();
80 _currentFrame = _nextFrame++;
81 _triggeredArrivalEvent = false;
82 _triggeredDepartureEvent = false;
83
84 m_goValue.Transport.PathProgress = 0;
85 SetObjectScale(goinfo->size);
86 SetFaction(goinfo->faction);
87 SetUInt32Value(GAMEOBJECT_FLAGS, goinfo->flags);
88 SetPeriod(tInfo->pathTime);
89 SetEntry(goinfo->entry);
90 SetDisplayId(goinfo->displayId);
91 SetGoState(!goinfo->moTransport.canBeStopped ? GO_STATE_READY : GO_STATE_ACTIVE);
92 SetGoType(GAMEOBJECT_TYPE_MO_TRANSPORT);
93 SetGoAnimProgress(animprogress);
94 SetName(goinfo->name);
95 UpdateRotationFields(0.0f, 1.0f);
96
97 m_model = CreateModel();
98 return true;
99}

113void Transport::Update(uint32 diff)
114{
115 uint32 const positionUpdateDelay = 200;
116
117 if (AI())
118 AI()->UpdateAI(diff);
119 else if (!AIM_Initialize())
120 TC_LOG_ERROR("entities.transport", "Could not initialize GameObjectAI for Transport");
121
122 if (GetKeyFrames().size() <= 1)
123 return;
124
125 if (IsMoving() || !_pendingStop)
126 m_goValue.Transport.PathProgress += diff;
127
128 uint32 timer = m_goValue.Transport.PathProgress % GetTransportPeriod();
129
130 // Set current waypoint
131 // Desired outcome: _currentFrame->DepartureTime < timer < _nextFrame->ArriveTime
132 // ... arrive | ... delay ... | departure
133 // event / event /
134 for (;;)
135 {
136 if (timer >= _currentFrame->ArriveTime)
137 {
138 if (!_triggeredArrivalEvent)
139 {
140 DoEventIfAny(*_currentFrame, false);
141 _triggeredArrivalEvent = true;
142 }
143
144 if (timer < _currentFrame->DepartureTime)
145 {
146 SetMoving(false);
147 if (_pendingStop && GetGoState() != GO_STATE_READY)
148 {
149 SetGoState(GO_STATE_READY);
150 m_goValue.Transport.PathProgress = (m_goValue.Transport.PathProgress / GetTransportPeriod());
151 m_goValue.Transport.PathProgress *= GetTransportPeriod();
152 m_goValue.Transport.PathProgress += _currentFrame->ArriveTime;
153 }
154 break; // its a stop frame and we are waiting
155 }
156 }
157
158 if (timer >= _currentFrame->DepartureTime && !_triggeredDepartureEvent)
159 {
160 DoEventIfAny(*_currentFrame, true); // departure event
161 _triggeredDepartureEvent = true;
162 }
163
164 // not waiting anymore
165 SetMoving(true);
166
167 // Enable movement
168 if (GetGOInfo()->moTransport.canBeStopped)
169 SetGoState(GO_STATE_ACTIVE);
170
171 if (timer >= _currentFrame->DepartureTime && timer < _currentFrame->NextArriveTime)
172 break; // found current waypoint
173
174 MoveToNextWaypoint();
175
176 sScriptMgr->OnRelocate(this, _currentFrame->Node->NodeIndex, _currentFrame->Node->MapID, _currentFrame->Node->LocX, _currentFrame->Node->LocY, _currentFrame->Node->LocZ);
177
178 TC_LOG_DEBUG("entities.transport", "Transport %u (%s) moved to node %u %u %f %f %f", GetEntry(), GetName().c_str(), _currentFrame->Node->NodeIndex, _currentFrame->Node->MapID, _currentFrame->Node->LocX, _currentFrame->Node->LocY, _currentFrame->Node->LocZ);
179
180 // Departure event
181 if (_currentFrame->IsTeleportFrame())
182 if (TeleportTransport(_nextFrame->Node->MapID, _nextFrame->Node->LocX, _nextFrame->Node->LocY, _nextFrame->Node->LocZ, _nextFrame->InitialOrientation))
183 return; // Update more in new map thread
184 }
185
186 // Add model to map after we are fully done with moving maps
187 if (_delayedAddModel)
188 {
189 _delayedAddModel = false;
190 if (m_model)
191 GetMap()->InsertGameObjectModel(*m_model);
192 }
193
194 // Set position
195 _positionChangeTimer.Update(diff);
196 if (_positionChangeTimer.Passed())
197 {
198 _positionChangeTimer.Reset(positionUpdateDelay);
199 if (IsMoving())
200 {
201 float t = CalculateSegmentPos(float(timer) * 0.001f);
202 G3D::Vector3 pos, dir;
203 _currentFrame->Spline->evaluate_percent(_currentFrame->Index, t, pos);
204 _currentFrame->Spline->evaluate_derivative(_currentFrame->Index, t, dir);
205 UpdatePosition(pos.x, pos.y, pos.z, std::atan2(dir.y, dir.x) + float(M_PI));
206 }
207 else
208 {
209 /* There are four possible scenarios that trigger loading/unloading passengers:
210 1. transport moves from inactive to active grid
211 2. the grid that transport is currently in becomes active
212 3. transport moves from active to inactive grid
213 4. the grid that transport is currently in unloads
214 */
215 bool gridActive = GetMap()->IsGridLoaded(GetPositionX(), GetPositionY());
216
217 if (_staticPassengers.empty() && gridActive) // 2.
218 LoadStaticPassengers();
219 else if (!_staticPassengers.empty() && !gridActive)
220 // 4. - if transports stopped on grid edge, some passengers can remain in active grids
221 // unload all static passengers otherwise passengers won't load correctly when the grid that transport is currently in becomes active
222 UnloadStaticPassengers();
223 }
224 }
225
226 sScriptMgr->OnTransportUpdate(this, diff);
227}

254void Transport::RemovePassenger(WorldObject* passenger)
255{
256 bool erased = false;
257 if (_passengerTeleportItr != _passengers.end())
258 {
259 PassengerSet::iterator itr = _passengers.find(passenger);
260 if (itr != _passengers.end())
261 {
262 if (itr == _passengerTeleportItr)
263 ++_passengerTeleportItr;
264
265 _passengers.erase(itr);
266 erased = true;
267 }
268 }
269 else
270 erased = _passengers.erase(passenger) > 0;
271
272 if (erased || _staticPassengers.erase(passenger)) // static passenger can remove itself in case of grid unload
273 {
274 passenger->SetTransport(NULL);
275 passenger->m_movementInfo.RemoveMovementFlag(MOVEMENTFLAG_ONTRANSPORT);
276 passenger->m_movementInfo.transport.Reset();
277 TC_LOG_DEBUG("entities.transport", "Object %s removed from transport %s.", passenger->GetName().c_str(), GetName().c_str());
278
279 if (Player* plr = passenger->ToPlayer())
280 sScriptMgr->OnRemovePassenger(this, plr);
281 }

284Creature* Transport::CreateNPCPassenger(ObjectGuid::LowType guid, CreatureData const* data)
285{
286 Map* map = GetMap();
287 Creature* creature = new Creature();
288
289 if (!creature->LoadCreatureFromDB(guid, map, false))
290 {
291 delete creature;
292 return NULL;
293 }
294
295 float x = data->posX;
296 float y = data->posY;
297 float z = data->posZ;
298 float o = data->orientation;
299
300 creature->SetTransport(this);
301 creature->AddUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT);
302 creature->m_movementInfo.transport.guid = GetGUID();
303 creature->m_movementInfo.transport.pos.Relocate(x, y, z, o);
304 CalculatePassengerPosition(x, y, z, &o);
305 creature->Relocate(x, y, z, o);
306 creature->SetHomePosition(creature->GetPositionX(), creature->GetPositionY(), creature->GetPositionZ(), creature->GetOrientation());
307 creature->SetTransportHomePosition(creature->m_movementInfo.transport.pos);
308
309 /// @HACK - transport models are not added to map's dynamic LoS calculations
310 /// because the current GameObjectModel cannot be moved without recreating
311 creature->AddUnitState(UNIT_STATE_IGNORE_PATHFINDING);
312
313 if (!creature->IsPositionValid())
314 {
315 TC_LOG_ERROR("entities.transport", "Creature (guidlow %d, entry %d) not created. Suggested coordinates aren't valid (X: %f Y: %f)",creature->GetGUID().GetCounter(),creature->GetEntry(),creature->GetPositionX(),creature->GetPositionY());
316 delete creature;
317 return NULL;
318 }
319
320 if (!map->AddToMap(creature))
321 {
322 delete creature;
323 return NULL;
324 }
325
326 _staticPassengers.insert(creature);
327 sScriptMgr->OnAddCreaturePassenger(this, creature);
328 return creature;
329}

373TempSummon* Transport::SummonPassenger(uint32 entry, Position const& pos, TempSummonType summonType, SummonPropertiesEntry const* properties /*= NULL*/, uint32 duration /*= 0*/, Unit* summoner /*= NULL*/, uint32 spellId /*= 0*/, uint32 vehId /*= 0*/)
374{
375 Map* map = FindMap();
376 if (!map)
377 return NULL;
378
379 uint32 mask = UNIT_MASK_SUMMON;
380 if (properties)
381 {
382 switch (properties->Category)
383 {
384 case SUMMON_CATEGORY_PET:
385 mask = UNIT_MASK_GUARDIAN;
386 break;
387 case SUMMON_CATEGORY_PUPPET:
388 mask = UNIT_MASK_PUPPET;
389 break;
390 case SUMMON_CATEGORY_VEHICLE:
391 mask = UNIT_MASK_MINION;
392 break;
393 case SUMMON_CATEGORY_WILD:
394 case SUMMON_CATEGORY_ALLY:
395 case SUMMON_CATEGORY_UNK:
396 {
397 switch (properties->Type)
398 {
399 case SUMMON_TYPE_MINION:
400 case SUMMON_TYPE_GUARDIAN:
401 case SUMMON_TYPE_GUARDIAN2:
402 mask = UNIT_MASK_GUARDIAN;
403 break;
404 case SUMMON_TYPE_TOTEM:
405 case SUMMON_TYPE_LIGHTWELL:
406 mask = UNIT_MASK_TOTEM;
407 break;
408 case SUMMON_TYPE_VEHICLE:
409 case SUMMON_TYPE_VEHICLE2:
410 mask = UNIT_MASK_SUMMON;
411 break;
412 case SUMMON_TYPE_MINIPET:
413 mask = UNIT_MASK_MINION;
414 break;
415 default:
416 if (properties->Flags & 512) // Mirror Image, Summon Gargoyle
417 mask = UNIT_MASK_GUARDIAN;
418 break;
419 }
420 break;
421 }
422 default:
423 return NULL;
424 }
425 }
426
427 uint32 phase = PHASEMASK_NORMAL;
428 if (summoner)
429 phase = summoner->GetPhaseMask();
430
431 TempSummon* summon = NULL;
432 switch (mask)
433 {
434 case UNIT_MASK_SUMMON:
435 summon = new TempSummon(properties, summoner, false);
436 break;
437 case UNIT_MASK_GUARDIAN:
438 summon = new Guardian(properties, summoner, false);
439 break;
440 case UNIT_MASK_PUPPET:
441 summon = new Puppet(properties, summoner);
442 break;
443 case UNIT_MASK_TOTEM:
444 summon = new Totem(properties, summoner);
445 break;
446 case UNIT_MASK_MINION:
447 summon = new Minion(properties, summoner, false);
448 break;
449 }
450
451 float x, y, z, o;
452 pos.GetPosition(x, y, z, o);
453 CalculatePassengerPosition(x, y, z, &o);
454
455 if (!summon->Create(map->GenerateLowGuid<HighGuid::Unit>(), map, phase, entry, x, y, z, o, nullptr, vehId))
456 {
457 delete summon;
458 return NULL;
459 }
460
461 summon->SetUInt32Value(UNIT_CREATED_BY_SPELL, spellId);
462
463 summon->SetTransport(this);
464 summon->AddUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT);
465 summon->m_movementInfo.transport.guid = GetGUID();
466 summon->m_movementInfo.transport.pos.Relocate(pos);
467 summon->Relocate(x, y, z, o);
468 summon->SetHomePosition(x, y, z, o);
469 summon->SetTransportHomePosition(pos);
470
471 /// @HACK - transport models are not added to map's dynamic LoS calculations
472 /// because the current GameObjectModel cannot be moved without recreating
473 summon->AddUnitState(UNIT_STATE_IGNORE_PATHFINDING);
474
475 summon->InitStats(duration);
476
477 if (!map->AddToMap<Creature>(summon))
478 {
479 delete summon;
480 return NULL;
481 }
482
483 _staticPassengers.insert(summon);
484
485 summon->InitSummon();
486 summon->SetTempSummonType(summonType);
487
488 return summon;
489}

491void Transport::UpdatePosition(float x, float y, float z, float o)
492{
493 bool newActive = GetMap()->IsGridLoaded(x, y);
494 Cell oldCell(GetPositionX(), GetPositionY());
495
496 Relocate(x, y, z, o);
497 UpdateModelPosition();
498
499 UpdatePassengerPositions(_passengers);
500
501 /* There are four possible scenarios that trigger loading/unloading passengers:
502 1. transport moves from inactive to active grid
503 2. the grid that transport is currently in becomes active
504 3. transport moves from active to inactive grid
505 4. the grid that transport is currently in unloads
506 */
507 if (_staticPassengers.empty() && newActive) // 1.
508 LoadStaticPassengers();
509 else if (!_staticPassengers.empty() && !newActive && oldCell.DiffGrid(Cell(GetPositionX(), GetPositionY()))) // 3.
510 UnloadStaticPassengers();
511 else
512 UpdatePassengerPositions(_staticPassengers);
513 // 4. is handed by grid unload
514}

516void Transport::LoadStaticPassengers()
517{
518 if (uint32 mapId = GetGOInfo()->moTransport.mapID)
519 {
520 CellObjectGuidsMap const& cells = sObjectMgr->GetMapObjectGuids(mapId, GetMap()->GetSpawnMode());
521 CellGuidSet::const_iterator guidEnd;
522 for (CellObjectGuidsMap::const_iterator cellItr = cells.begin(); cellItr != cells.end(); ++cellItr)
523 {
524 // Creatures on transport
525 guidEnd = cellItr->second.creatures.end();
526 for (CellGuidSet::const_iterator guidItr = cellItr->second.creatures.begin(); guidItr != guidEnd; ++guidItr)
527 CreateNPCPassenger(*guidItr, sObjectMgr->GetCreatureData(*guidItr));
528
529 // GameObjects on transport
530 guidEnd = cellItr->second.gameobjects.end();
531 for (CellGuidSet::const_iterator guidItr = cellItr->second.gameobjects.begin(); guidItr != guidEnd; ++guidItr)
532 CreateGOPassenger(*guidItr, sObjectMgr->GetGOData(*guidItr));
533 }
534 }
535}

566float Transport::CalculateSegmentPos(float now)
567{
568 KeyFrame const& frame = *_currentFrame;
569 const float speed = float(m_goInfo->moTransport.moveSpeed);
570 const float accel = float(m_goInfo->moTransport.accelRate);
571 float timeSinceStop = frame.TimeFrom + (now - (1.0f/IN_MILLISECONDS) * frame.DepartureTime);
572 float timeUntilStop = frame.TimeTo - (now - (1.0f/IN_MILLISECONDS) * frame.DepartureTime);
573 float segmentPos, dist;
574 float accelTime = _transportInfo->accelTime;
575 float accelDist = _transportInfo->accelDist;
576 // calculate from nearest stop, less confusing calculation...
577 if (timeSinceStop < timeUntilStop)
578 {
579 if (timeSinceStop < accelTime)
580 dist = 0.5f * accel * timeSinceStop * timeSinceStop;
581 else
582 dist = accelDist + (timeSinceStop - accelTime) * speed;
583 segmentPos = dist - frame.DistSinceStop;
584 }
585 else
586 {
587 if (timeUntilStop < _transportInfo->accelTime)
588 dist = 0.5f * accel * timeUntilStop * timeUntilStop;
589 else
590 dist = accelDist + (timeUntilStop - accelTime) * speed;
591 segmentPos = frame.DistUntilStop - dist;
592 }
593
594 return segmentPos / frame.NextDistFromPrev;
595}

597bool Transport::TeleportTransport(uint32 newMapid, float x, float y, float z, float o)
598{
599 Map const* oldMap = GetMap();
600
601 if (oldMap->GetId() != newMapid)
602 {
603 _delayedTeleport = true;
604 UnloadStaticPassengers();
605 return true;
606 }
607 else
608 {
609 // Teleport players, they need to know it
610 for (PassengerSet::iterator itr = _passengers.begin(); itr != _passengers.end(); ++itr)
611 {
612 if ((*itr)->GetTypeId() == TYPEID_PLAYER)
613 {
614 float destX, destY, destZ, destO;
615 (*itr)->m_movementInfo.transport.pos.GetPosition(destX, destY, destZ, destO);
616 TransportBase::CalculatePassengerPosition(destX, destY, destZ, &destO, x, y, z, o);
617
618 (*itr)->ToUnit()->NearTeleportTo(destX, destY, destZ, destO);
619 }
620 }
621
622 UpdatePosition(x, y, z, o);
623 return false;
624 }
625}

627void Transport::DelayedTeleportTransport()
628{
629 if (!_delayedTeleport)
630 return;
631
632 _delayedTeleport = false;
633 Map* newMap = sMapMgr->CreateBaseMap(_nextFrame->Node->MapID);
634 GetMap()->RemoveFromMap<Transport>(this, false);
635 SetMap(newMap);
636
637 float x = _nextFrame->Node->LocX,
638 y = _nextFrame->Node->LocY,
639 z = _nextFrame->Node->LocZ,
640 o =_nextFrame->InitialOrientation;
641
642 for (_passengerTeleportItr = _passengers.begin(); _passengerTeleportItr != _passengers.end();)
643 {
644 WorldObject* obj = (*_passengerTeleportItr++);
645
646 float destX, destY, destZ, destO;
647 obj->m_movementInfo.transport.pos.GetPosition(destX, destY, destZ, destO);
648 TransportBase::CalculatePassengerPosition(destX, destY, destZ, &destO, x, y, z, o);
649
650 switch (obj->GetTypeId())
651 {
652 case TYPEID_PLAYER:
653 if (!obj->ToPlayer()->TeleportTo(_nextFrame->Node->MapID, destX, destY, destZ, destO, TELE_TO_NOT_LEAVE_TRANSPORT))
654 RemovePassenger(obj);
655 break;
656 case TYPEID_DYNAMICOBJECT:
657 obj->AddObjectToRemoveList();
658 break;
659 default:
660 RemovePassenger(obj);
661 break;
662 }
663 }
664
665 Relocate(x, y, z, o);
666 GetMap()->AddToMap<Transport>(this);
667}

669void Transport::UpdatePassengerPositions(PassengerSet& passengers)
670{
671 for (PassengerSet::iterator itr = passengers.begin(); itr != passengers.end(); ++itr)
672 {
673 WorldObject* passenger = *itr;
674 // transport teleported but passenger not yet (can happen for players)
675 if (passenger->GetMap() != GetMap())
676 continue;
677
678 // if passenger is on vehicle we have to assume the vehicle is also on transport
679 // and its the vehicle that will be updating its passengers
680 if (Unit* unit = passenger->ToUnit())
681 if (unit->GetVehicle())
682 continue;
683
684 // Do not use Unit::UpdatePosition here, we don't want to remove auras
685 // as if regular movement occurred
686 float x, y, z, o;
687 passenger->m_movementInfo.transport.pos.GetPosition(x, y, z, o);
688 CalculatePassengerPosition(x, y, z, &o);
689 switch (passenger->GetTypeId())
690 {
691 case TYPEID_UNIT:
692 {
693 Creature* creature = passenger->ToCreature();
694 GetMap()->CreatureRelocation(creature, x, y, z, o, false);
695 creature->GetTransportHomePosition(x, y, z, o);
696 CalculatePassengerPosition(x, y, z, &o);
697 creature->SetHomePosition(x, y, z, o);
698 break;
699 }
700 case TYPEID_PLAYER:
701 //relocate only passengers in world and skip any player that might be still logging in/teleporting
702 if (passenger->IsInWorld())
703 GetMap()->PlayerRelocation(passenger->ToPlayer(), x, y, z, o);
704 break;
705 case TYPEID_GAMEOBJECT:
706 GetMap()->GameObjectRelocation(passenger->ToGameObject(), x, y, z, o, false);
707 passenger->ToGameObject()->RelocateStationaryPosition(x, y, z, o);
708 break;
709 case TYPEID_DYNAMICOBJECT:
710 GetMap()->DynamicObjectRelocation(passenger->ToDynObject(), x, y, z, o);
711 break;
712 default:
713 break;
714 }
715
716 if (Unit* unit = passenger->ToUnit())
717 if (Vehicle* vehicle = unit->GetVehicleKit())
718 vehicle->RelocatePassengers();
719 }
720}

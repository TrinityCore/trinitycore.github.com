177struct ActionButton
178{
179 ActionButton() : packedData(0), uState(ACTIONBUTTON_NEW) { }
180
181 uint32 packedData;
182 ActionButtonUpdateState uState;
183
184 // helpers
185 ActionButtonType GetType() const { return ActionButtonType(ACTION_BUTTON_TYPE(packedData)); }
186 uint32 GetAction() const { return ACTION_BUTTON_ACTION(packedData); }
187 void SetActionAndType(uint32 action, ActionButtonType type)
188 {
189 uint32 newData = action | (uint32(type) << 24);
190 if (newData != packedData || uState == ACTIONBUTTON_DELETED)
191 {
192 packedData = newData;
193 if (uState != ACTIONBUTTON_NEW)
194 uState = ACTIONBUTTON_CHANGED;
195 }
196 }
197};

2592template <class T> T Player::ApplySpellMod(uint32 spellId, SpellModOp op, T &basevalue, Spell* spell)
2593{
2594 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
2595 if (!spellInfo)
2596 return 0;
2597 float totalmul = 1.0f;
2598 int32 totalflat = 0;
2599
2600 // Drop charges for triggering spells instead of triggered ones
2601 if (m_spellModTakingSpell)
2602 spell = m_spellModTakingSpell;
2603
2604 for (SpellModList::iterator itr = m_spellMods[op].begin(); itr != m_spellMods[op].end(); ++itr)
2605 {
2606 SpellModifier* mod = *itr;
2607
2608 // Charges can be set only for mods with auras
2609 if (!mod->ownerAura)
2610 ASSERT(mod->charges == 0);
2611
2612 if (!IsAffectedBySpellmod(spellInfo, mod, spell))
2613 continue;
2614
2615 if (mod->type == SPELLMOD_FLAT)
2616 totalflat += mod->value;
2617 else if (mod->type == SPELLMOD_PCT)
2618 {
2619 // skip percent mods for null basevalue (most important for spell mods with charges)
2620 if (basevalue == T(0))
2621 continue;
2622
2623 // special case (skip > 10sec spell casts for instant cast setting)
2624 if (mod->op == SPELLMOD_CASTING_TIME && basevalue >= T(10000) && mod->value <= -100)
2625 continue;
2626
2627 totalmul += CalculatePct(1.0f, mod->value);
2628 }
2629
2630 DropModCharge(mod, spell);
2631 }
2632 float diff = (float)basevalue * (totalmul - 1.0f) + (float)totalflat;
2633 basevalue = T((float)basevalue + diff);
2634 return T(diff);
2635}

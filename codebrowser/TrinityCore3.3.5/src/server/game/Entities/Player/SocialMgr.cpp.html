33uint32 PlayerSocial::GetNumberOfSocialsWithFlag(SocialFlag flag)
34{
35 uint32 counter = 0;
36 for (PlayerSocialMap::const_iterator itr = m_playerSocialMap.begin(); itr != m_playerSocialMap.end(); ++itr)
37 if (itr->second.Flags & flag)
38 ++counter;
39
40 return counter;
41}

43bool PlayerSocial::AddToSocialList(ObjectGuid::LowType friendGuid, bool ignore)
44{
45 // check client limits
46 if (ignore)
47 {
48 if (GetNumberOfSocialsWithFlag(SOCIAL_FLAG_IGNORED) >= SOCIALMGR_IGNORE_LIMIT)
49 return false;
50 }
51 else
52 {
53 if (GetNumberOfSocialsWithFlag(SOCIAL_FLAG_FRIEND) >= SOCIALMGR_FRIEND_LIMIT)
54 return false;
55 }
56
57 uint8 flag = SOCIAL_FLAG_FRIEND;
58 if (ignore)
59 flag = SOCIAL_FLAG_IGNORED;
60
61 PlayerSocialMap::const_iterator itr = m_playerSocialMap.find(friendGuid);
62 if (itr != m_playerSocialMap.end())
63 {
64 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ADD_CHARACTER_SOCIAL_FLAGS);
65
66 stmt->setUInt8(0, flag);
67 stmt->setUInt32(1, GetPlayerGUID());
68 stmt->setUInt32(2, friendGuid);
69
70 CharacterDatabase.Execute(stmt);
71
72 m_playerSocialMap[friendGuid].Flags |= flag;
73 }
74 else
75 {
76 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_CHARACTER_SOCIAL);
77
78 stmt->setUInt32(0, GetPlayerGUID());
79 stmt->setUInt32(1, friendGuid);
80 stmt->setUInt8(2, flag);
81
82 CharacterDatabase.Execute(stmt);
83
84 FriendInfo fi;
85 fi.Flags |= flag;
86 m_playerSocialMap[friendGuid] = fi;
87 }
88 return true;
89}

91void PlayerSocial::RemoveFromSocialList(ObjectGuid::LowType friendGuid, bool ignore)
92{
93 PlayerSocialMap::iterator itr = m_playerSocialMap.find(friendGuid);
94 if (itr == m_playerSocialMap.end()) // not exist
95 return;
96
97 uint8 flag = SOCIAL_FLAG_FRIEND;
98 if (ignore)
99 flag = SOCIAL_FLAG_IGNORED;
100
101 itr->second.Flags &= ~flag;
102 if (itr->second.Flags == 0)
103 {
104 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHARACTER_SOCIAL);
105
106 stmt->setUInt32(0, GetPlayerGUID());
107 stmt->setUInt32(1, friendGuid);
108
109 CharacterDatabase.Execute(stmt);
110
111 m_playerSocialMap.erase(itr);
112 }
113 else
114 {
115 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_REM_CHARACTER_SOCIAL_FLAGS);
116
117 stmt->setUInt8(0, flag);
118 stmt->setUInt32(1, GetPlayerGUID());
119 stmt->setUInt32(2, friendGuid);
120
121 CharacterDatabase.Execute(stmt);
122 }
123}

125void PlayerSocial::SetFriendNote(ObjectGuid::LowType friendGuid, std::string note)
126{
127 PlayerSocialMap::const_iterator itr = m_playerSocialMap.find(friendGuid);
128 if (itr == m_playerSocialMap.end()) // not exist
129 return;
130
131 utf8truncate(note, 48); // DB and client size limitation
132
133 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_CHARACTER_SOCIAL_NOTE);
134
135 stmt->setString(0, note);
136 stmt->setUInt32(1, GetPlayerGUID());
137 stmt->setUInt32(2, friendGuid);
138
139 CharacterDatabase.Execute(stmt);
140
141 m_playerSocialMap[friendGuid].Note = note;
142}

144void PlayerSocial::SendSocialList(Player* player)
145{
146 if (!player)
147 return;
148
149 uint32 size = m_playerSocialMap.size();
150
151 WorldPacket data(SMSG_CONTACT_LIST, (4+4+size*25)); // just can guess size
152 data << uint32(7); // 0x1 = Friendlist update. 0x2 = Ignorelist update. 0x4 = Mutelist update.
153 data << uint32(size); // friends count
154
155 for (PlayerSocialMap::iterator itr = m_playerSocialMap.begin(); itr != m_playerSocialMap.end(); ++itr)
156 {
157 sSocialMgr->GetFriendInfo(player, itr->first, itr->second);
158
159 data << uint64(itr->first); // player guid
160 data << uint32(itr->second.Flags); // player flag (0x1 = Friend, 0x2 = Ignored, 0x4 = Muted)
161 data << itr->second.Note; // string note
162 if (itr->second.Flags & SOCIAL_FLAG_FRIEND) // if IsFriend()
163 {
164 data << uint8(itr->second.Status); // online/offline/etc?
165 if (itr->second.Status) // if online
166 {
167 data << uint32(itr->second.Area); // player area
168 data << uint32(itr->second.Level); // player level
169 data << uint32(itr->second.Class); // player class
170 }
171 }
172 }
173
174 player->GetSession()->SendPacket(&data);
175 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_CONTACT_LIST");
176}

198void SocialMgr::GetFriendInfo(Player* player, ObjectGuid::LowType friendGUID, FriendInfo &friendInfo)
199{
200 if (!player)
201 return;
202
203 friendInfo.Status = FRIEND_STATUS_OFFLINE;
204 friendInfo.Area = 0;
205 friendInfo.Level = 0;
206 friendInfo.Class = 0;
207
208 Player* target = ObjectAccessor::FindPlayer(ObjectGuid(HighGuid::Player, friendGUID));
209 if (!target)
210 return;
211
212 PlayerSocialMap::iterator itr = player->GetSocial()->m_playerSocialMap.find(friendGUID);
213 if (itr != player->GetSocial()->m_playerSocialMap.end())
214 friendInfo.Note = itr->second.Note;
215
216 // PLAYER see his team only and PLAYER can't see MODERATOR, GAME MASTER, ADMINISTRATOR characters
217 // MODERATOR, GAME MASTER, ADMINISTRATOR can see all
218
219 if (!player->GetSession()->HasPermission(rbac::RBAC_PERM_WHO_SEE_ALL_SEC_LEVELS) &&
220 target->GetSession()->GetSecurity() > AccountTypes(sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_WHO_LIST)))
221 return;
222
223 // player can see member of other team only if CONFIG_ALLOW_TWO_SIDE_WHO_LIST
224 if (target->GetTeam() != player->GetTeam() && !player->GetSession()->HasPermission(rbac::RBAC_PERM_TWO_SIDE_WHO_LIST))
225 return;
226
227 if (target->IsVisibleGloballyFor(player))
228 {
229 if (target->isDND())
230 friendInfo.Status = FRIEND_STATUS_DND;
231 else if (target->isAFK())
232 friendInfo.Status = FRIEND_STATUS_AFK;
233 else
234 friendInfo.Status = FRIEND_STATUS_ONLINE;
235
236 friendInfo.Area = target->GetZoneId();
237 friendInfo.Level = target->getLevel();
238 friendInfo.Class = target->getClass();
239 }
240}

249void SocialMgr::SendFriendStatus(Player* player, FriendsResult result, ObjectGuid::LowType friendGuid, bool broadcast)
250{
251 FriendInfo fi;
252
253 WorldPacket data;
254 MakeFriendStatusPacket(result, friendGuid, &data);
255 GetFriendInfo(player, friendGuid, fi);
256 switch (result)
257 {
258 case FRIEND_ADDED_OFFLINE:
259 case FRIEND_ADDED_ONLINE:
260 data << fi.Note;
261 break;
262 default:
263 break;
264 }
265
266 switch (result)
267 {
268 case FRIEND_ADDED_ONLINE:
269 case FRIEND_ONLINE:
270 data << uint8(fi.Status);
271 data << uint32(fi.Area);
272 data << uint32(fi.Level);
273 data << uint32(fi.Class);
274 break;
275 default:
276 break;
277 }
278
279 if (broadcast)
280 BroadcastToFriendListers(player, &data);
281 else
282 player->GetSession()->SendPacket(&data);
283}

285void SocialMgr::BroadcastToFriendListers(Player* player, WorldPacket* packet)
286{
287 if (!player)
288 return;
289
290 AccountTypes gmSecLevel = AccountTypes(sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_WHO_LIST));
291 for (SocialMap::const_iterator itr = m_socialMap.begin(); itr != m_socialMap.end(); ++itr)
292 {
293 PlayerSocialMap::const_iterator itr2 = itr->second.m_playerSocialMap.find(player->GetGUID().GetCounter());
294 if (itr2 != itr->second.m_playerSocialMap.end() && (itr2->second.Flags & SOCIAL_FLAG_FRIEND))
295 {
296 Player* target = ObjectAccessor::FindPlayer(ObjectGuid(HighGuid::Player, 0, itr->first));
297 if (!target)
298 continue;
299
300 WorldSession* session = target->GetSession();
301 if (!session->HasPermission(rbac::RBAC_PERM_WHO_SEE_ALL_SEC_LEVELS) && player->GetSession()->GetSecurity() > gmSecLevel)
302 continue;
303
304 if (target->GetTeam() != player->GetTeam() && !session->HasPermission(rbac::RBAC_PERM_TWO_SIDE_WHO_LIST))
305 continue;
306
307 if (player->IsVisibleGloballyFor(target))
308 session->SendPacket(packet);
309 }
310 }

313PlayerSocial* SocialMgr::LoadFromDB(PreparedQueryResult result, ObjectGuid::LowType guid)
314{
315 PlayerSocial *social = &m_socialMap[guid];
316 social->SetPlayerGUID(guid);
317
318 if (!result)
319 return social;
320
321 ObjectGuid::LowType friendGuid = 0;
322 uint8 flags = 0;
323 std::string note = "";
324
325 do
326 {
327 Field* fields = result->Fetch();
328
329 friendGuid = fields[0].GetUInt32();
330 flags = fields[1].GetUInt8();
331 note = fields[2].GetString();
332
333 social->m_playerSocialMap[friendGuid] = FriendInfo(flags, note);
334
335 // client's friends list and ignore list limit
336 if (social->m_playerSocialMap.size() >= (SOCIALMGR_FRIEND_LIMIT + SOCIALMGR_IGNORE_LIMIT))
337 break;
338 }
339 while (result->NextRow());
340
341 return social;
342}

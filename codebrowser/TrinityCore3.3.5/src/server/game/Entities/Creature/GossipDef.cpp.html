37void GossipMenu::AddMenuItem(int32 menuItemId, uint8 icon, std::string const& message, uint32 sender, uint32 action, std::string const& boxMessage, uint32 boxMoney, bool coded /*= false*/)
38{
39 ASSERT(_menuItems.size() <= GOSSIP_MAX_MENU_ITEMS);
40
41 // Find a free new id - script case
42 if (menuItemId == -1)
43 {
44 menuItemId = 0;
45 if (!_menuItems.empty())
46 {
47 for (GossipMenuItemContainer::const_iterator itr = _menuItems.begin(); itr != _menuItems.end(); ++itr)
48 {
49 if (int32(itr->first) > menuItemId)
50 break;
51
52 menuItemId = itr->first + 1;
53 }
54 }
55 }
56
57 GossipMenuItem& menuItem = _menuItems[menuItemId];
58
59 menuItem.MenuItemIcon = icon;
60 menuItem.Message = message;
61 menuItem.IsCoded = coded;
62 menuItem.Sender = sender;
63 menuItem.OptionType = action;
64 menuItem.BoxMessage = boxMessage;
65 menuItem.BoxMoney = boxMoney;
66}

68/**
69 * @name AddMenuItem
70 * @brief Adds a localized gossip menu item from db by menu id and menu item id.
71 * @param menuId Gossip menu id.
72 * @param menuItemId Gossip menu item id.
73 * @param sender Identifier of the current menu.
74 * @param action Custom action given to OnGossipHello.
75 */
76void GossipMenu::AddMenuItem(uint32 menuId, uint32 menuItemId, uint32 sender, uint32 action)
77{
78 /// Find items for given menu id.
79 GossipMenuItemsMapBounds bounds = sObjectMgr->GetGossipMenuItemsMapBounds(menuId);
80 /// Return if there are none.
81 if (bounds.first == bounds.second)
82 return;
83
84 /// Iterate over each of them.
85 for (GossipMenuItemsContainer::const_iterator itr = bounds.first; itr != bounds.second; ++itr)
86 {
87 /// Find the one with the given menu item id.
88 if (itr->second.OptionIndex != menuItemId)
89 continue;
90
91 /// Store texts for localization.
92 std::string strOptionText, strBoxText;
93 BroadcastText const* optionBroadcastText = sObjectMgr->GetBroadcastText(itr->second.OptionBroadcastTextId);
94 BroadcastText const* boxBroadcastText = sObjectMgr->GetBroadcastText(itr->second.BoxBroadcastTextId);
95
96 /// OptionText
97 if (optionBroadcastText)
98 strOptionText = optionBroadcastText->GetText(GetLocale());
99 else
100 strOptionText = itr->second.OptionText;
101
102 /// BoxText
103 if (boxBroadcastText)
104 strBoxText = boxBroadcastText->GetText(GetLocale());
105 else
106 strBoxText = itr->second.BoxText;
107
108 /// Check need of localization.
109 if (GetLocale() != DEFAULT_LOCALE)
110 {
111 if (!optionBroadcastText)
112 {
113 /// Find localizations from database.
114 if (GossipMenuItemsLocale const* gossipMenuLocale = sObjectMgr->GetGossipMenuItemsLocale(MAKE_PAIR32(menuId, menuItemId)))
115 ObjectMgr::GetLocaleString(gossipMenuLocale->OptionText, GetLocale(), strOptionText);
116 }
117
118 if (!boxBroadcastText)
119 {
120 /// Find localizations from database.
121 if (GossipMenuItemsLocale const* gossipMenuLocale = sObjectMgr->GetGossipMenuItemsLocale(MAKE_PAIR32(menuId, menuItemId)))
122 ObjectMgr::GetLocaleString(gossipMenuLocale->BoxText, GetLocale(), strBoxText);
123 }
124 }
125
126 /// Add menu item with existing method. Menu item id -1 is also used in ADD_GOSSIP_ITEM macro.
127 AddMenuItem(-1, itr->second.OptionIcon, strOptionText, sender, action, strBoxText, itr->second.BoxMoney, itr->second.BoxCoded);
128 }
129}

189void PlayerMenu::SendGossipMenu(uint32 titleTextId, ObjectGuid objectGUID)
190{
191 _gossipMenu.SetSenderGUID(objectGUID);
192
193 WorldPacket data(SMSG_GOSSIP_MESSAGE, 100); // guess size
194 data << uint64(objectGUID);
195 data << uint32(_gossipMenu.GetMenuId()); // new 2.4.0
196 data << uint32(titleTextId);
197 data << uint32(_gossipMenu.GetMenuItemCount()); // max count 0x10
198
199 for (GossipMenuItemContainer::const_iterator itr = _gossipMenu.GetMenuItems().begin(); itr != _gossipMenu.GetMenuItems().end(); ++itr)
200 {
201 GossipMenuItem const& item = itr->second;
202 data << uint32(itr->first);
203 data << uint8(item.MenuItemIcon);
204 data << uint8(item.IsCoded); // makes pop up box password
205 data << uint32(item.BoxMoney); // money required to open menu, 2.0.3
206 data << item.Message; // text for gossip item
207 data << item.BoxMessage; // accept text (related to money) pop up box, 2.0.3
208 }
209
210 size_t count_pos = data.wpos();
211 data << uint32(0); // max count 0x20
212 uint32 count = 0;
213
214 // Store this instead of checking the Singleton every loop iteration
215 bool questLevelInTitle = sWorld->getBoolConfig(CONFIG_UI_QUESTLEVELS_IN_DIALOGS);
216
217 for (uint8 i = 0; i < _questMenu.GetMenuItemCount(); ++i)
218 {
219 QuestMenuItem const& item = _questMenu.GetItem(i);
220 uint32 questID = item.QuestId;
221 if (Quest const* quest = sObjectMgr->GetQuestTemplate(questID))
222 {
223 ++count;
224 data << uint32(questID);
225 data << uint32(item.QuestIcon);
226 data << int32(quest->GetQuestLevel());
227 data << uint32(quest->GetFlags()); // 3.3.3 quest flags
228 data << uint8(0); // 3.3.3 changes icon: blue question or yellow exclamation
229 std::string title = quest->GetTitle();
230
231 int32 locale = _session->GetSessionDbLocaleIndex();
232 if (locale >= 0)
233 if (QuestLocale const* localeData = sObjectMgr->GetQuestLocale(questID))
234 ObjectMgr::GetLocaleString(localeData->Title, locale, title);
235
236 if (questLevelInTitle)
237 AddQuestLevelToTitle(title, quest->GetQuestLevel());
238
239 data << title; // max 0x200
240 }
241 }
242
243 data.put<uint8>(count_pos, count);
244 _session->SendPacket(&data);
245}

273 data << float(poi->y);
274 data << uint32(poi->icon);
275 data << uint32(poi->data);
276 data << iconText;
277
278 _session->SendPacket(&data);

382void PlayerMenu::SendQuestGiverQuestDetails(Quest const* quest, ObjectGuid npcGUID, bool activateAccept) const
383{
384 std::string questTitle = quest->GetTitle();
385 std::string questDetails = quest->GetDetails();
386 std::string questObjectives = quest->GetObjectives();
387 std::string questAreaDescription = quest->GetAreaDescription();
388
389 int32 locale = _session->GetSessionDbLocaleIndex();
390 if (locale >= 0)
391 {
392 if (QuestLocale const* localeData = sObjectMgr->GetQuestLocale(quest->GetQuestId()))
393 {
394 ObjectMgr::GetLocaleString(localeData->Title, locale, questTitle);
395 ObjectMgr::GetLocaleString(localeData->Details, locale, questDetails);
396 ObjectMgr::GetLocaleString(localeData->Objectives, locale, questObjectives);
397 ObjectMgr::GetLocaleString(localeData->AreaDescription, locale, questAreaDescription);
398 }
399 }
400
401 if (sWorld->getBoolConfig(CONFIG_UI_QUESTLEVELS_IN_DIALOGS))
402 AddQuestLevelToTitle(questTitle, quest->GetQuestLevel());
403
404 WorldPacket data(SMSG_QUESTGIVER_QUEST_DETAILS, 100); // guess size
405 data << uint64(npcGUID);
406 data << uint64(_session->GetPlayer()->GetDivider());
407 data << uint32(quest->GetQuestId());
408 data << questTitle;
409 data << questDetails;
410 data << questObjectives;
411 data << uint8(activateAccept ? 1 : 0); // CGQuestInfo::m_autoLaunched
412 data << uint32(quest->GetFlags()); // 3.3.3 questFlags
413 data << uint32(quest->GetSuggestedPlayers());
414 data << uint8(0); // CGQuestInfo::m_startQuestCheat
415
416 if (quest->HasFlag(QUEST_FLAGS_HIDDEN_REWARDS))
417 {
418 data << uint32(0); // Rewarded chosen items hidden
419 data << uint32(0); // Rewarded items hidden
420 data << uint32(0); // Rewarded money hidden
421 data << uint32(0); // Rewarded XP hidden
422 }
423 else
424 {
425 data << uint32(quest->GetRewChoiceItemsCount());
426 for (uint32 i=0; i < QUEST_REWARD_CHOICES_COUNT; ++i)
427 {
428 if (!quest->RewardChoiceItemId[i])
429 continue;
430
431 data << uint32(quest->RewardChoiceItemId[i]);
432 data << uint32(quest->RewardChoiceItemCount[i]);
433
434 if (ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(quest->RewardChoiceItemId[i]))
435 data << uint32(itemTemplate->DisplayInfoID);
436 else
437 data << uint32(0x00);
438 }
439
440 data << uint32(quest->GetRewItemsCount());
441
442 for (uint32 i=0; i < QUEST_REWARDS_COUNT; ++i)
443 {
444 if (!quest->RewardItemId[i])
445 continue;
446
447 data << uint32(quest->RewardItemId[i]);
448 data << uint32(quest->RewardItemIdCount[i]);
449
450 if (ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(quest->RewardItemId[i]))
451 data << uint32(itemTemplate->DisplayInfoID);
452 else
453 data << uint32(0);
454 }
455
456 data << uint32(quest->GetRewOrReqMoney());
457 data << uint32(quest->XPValue(_session->GetPlayer()) * sWorld->getRate(RATE_XP_QUEST));
458 }
459
460 // rewarded honor points. Multiply with 10 to satisfy client
461 data << uint32(10 * quest->CalculateHonorGain(_session->GetPlayer()->GetQuestLevel(quest)));
462 data << float(0.0f); // unk, honor multiplier?
463 data << uint32(quest->GetRewSpell()); // reward spell, this spell will display (icon) (cast if RewSpellCast == 0)
464 data << int32(quest->GetRewSpellCast()); // cast spell
465 data << uint32(quest->GetCharTitleId()); // CharTitleId, new 2.4.0, player gets this title (id from CharTitles)
466 data << uint32(quest->GetBonusTalents()); // bonus talents
467 data << uint32(quest->GetRewArenaPoints()); // reward arena points
468 data << uint32(0); // unk
469
470 for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i)
471 data << uint32(quest->RewardFactionId[i]);
472
473 for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i)
474 data << int32(quest->RewardFactionValueId[i]);
475
476 for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i)
477 data << int32(quest->RewardFactionValueIdOverride[i]);
478
479 data << uint32(QUEST_EMOTE_COUNT);
480 for (uint32 i = 0; i < QUEST_EMOTE_COUNT; ++i)
481 {
482 data << uint32(quest->DetailsEmote[i]);
483 data << uint32(quest->DetailsEmoteDelay[i]); // DetailsEmoteDelay (in ms)
484 }
485 _session->SendPacket(&data);
486
487 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_QUESTGIVER_QUEST_DETAILS NPC=%s, questid=%u", npcGUID.ToString().c_str(), quest->GetQuestId());
488}

490void PlayerMenu::SendQuestQueryResponse(Quest const* quest) const
491{
492 std::string questTitle = quest->GetTitle();
493 std::string questDetails = quest->GetDetails();
494 std::string questObjectives = quest->GetObjectives();
495 std::string questAreaDescription = quest->GetAreaDescription();
496 std::string questCompletedText = quest->GetCompletedText();
497
498 std::string questObjectiveText[QUEST_OBJECTIVES_COUNT];
499 for (uint8 i = 0; i < QUEST_OBJECTIVES_COUNT; ++i)
500 questObjectiveText[i] = quest->ObjectiveText[i];
501
502 int32 locale = _session->GetSessionDbLocaleIndex();
503 if (locale >= 0)
504 {
505 if (QuestLocale const* localeData = sObjectMgr->GetQuestLocale(quest->GetQuestId()))
506 {
507 ObjectMgr::GetLocaleString(localeData->Title, locale, questTitle);
508 ObjectMgr::GetLocaleString(localeData->Details, locale, questDetails);
509 ObjectMgr::GetLocaleString(localeData->Objectives, locale, questObjectives);
510 ObjectMgr::GetLocaleString(localeData->AreaDescription, locale, questAreaDescription);
511 ObjectMgr::GetLocaleString(localeData->CompletedText, locale, questCompletedText);
512
513 for (uint8 i = 0; i < QUEST_OBJECTIVES_COUNT; ++i)
514 ObjectMgr::GetLocaleString(localeData->ObjectiveText[i], locale, questObjectiveText[i]);
515 }
516 }
517
518 WorldPacket data(SMSG_QUEST_QUERY_RESPONSE, 100); // guess size
519
520 data << uint32(quest->GetQuestId()); // quest id
521 data << uint32(quest->GetQuestMethod()); // Accepted values: 0, 1 or 2. 0 == IsAutoComplete() (skip objectives/details)
522 data << uint32(quest->GetQuestLevel()); // may be -1, static data, in other cases must be used dynamic level: Player::GetQuestLevel (0 is not known, but assuming this is no longer valid for quest intended for client)
523 data << uint32(quest->GetMinLevel()); // min level
524 data << uint32(quest->GetZoneOrSort()); // zone or sort to display in quest log
525
526 data << uint32(quest->GetType()); // quest type
527 data << uint32(quest->GetSuggestedPlayers()); // suggested players count
528
529 data << uint32(quest->GetRepObjectiveFaction()); // shown in quest log as part of quest objective
530 data << uint32(quest->GetRepObjectiveValue()); // shown in quest log as part of quest objective
531
532 data << uint32(quest->GetRepObjectiveFaction2()); // shown in quest log as part of quest objective OPPOSITE faction
533 data << uint32(quest->GetRepObjectiveValue2()); // shown in quest log as part of quest objective OPPOSITE faction
534
535 data << uint32(quest->GetNextQuestInChain()); // client will request this quest from NPC, if not 0
536 data << uint32(quest->GetXPId()); // used for calculating rewarded experience
537
538 if (quest->HasFlag(QUEST_FLAGS_HIDDEN_REWARDS))
539 data << uint32(0); // Hide money rewarded
540 else
541 data << uint32(quest->GetRewOrReqMoney()); // reward money (below max lvl)
542
543 data << uint32(quest->GetRewMoneyMaxLevel()); // used in XP calculation at client
544 data << uint32(quest->GetRewSpell()); // reward spell, this spell will display (icon) (cast if RewSpellCast == 0)
545 data << int32(quest->GetRewSpellCast()); // cast spell
546
547 // rewarded honor points
548 data << uint32(quest->GetRewHonorAddition());
549 data << float(quest->GetRewHonorMultiplier());
550 data << uint32(quest->GetSrcItemId()); // source item id
551 data << uint32(quest->GetFlags() & 0xFFFF); // quest flags
552 data << uint32(quest->GetCharTitleId()); // CharTitleId, new 2.4.0, player gets this title (id from CharTitles)
553 data << uint32(quest->GetPlayersSlain()); // players slain
554 data << uint32(quest->GetBonusTalents()); // bonus talents
555 data << uint32(quest->GetRewArenaPoints()); // bonus arena points
556 data << uint32(0); // review rep show mask
557
558 if (quest->HasFlag(QUEST_FLAGS_HIDDEN_REWARDS))
559 {
560 for (uint8 i = 0; i < QUEST_REWARDS_COUNT; ++i)
561 data << uint32(0) << uint32(0);
562 for (uint8 i = 0; i < QUEST_REWARD_CHOICES_COUNT; ++i)
563 data << uint32(0) << uint32(0);
564 }
565 else
566 {
567 for (uint8 i = 0; i < QUEST_REWARDS_COUNT; ++i)
568 {
569 data << uint32(quest->RewardItemId[i]);
570 data << uint32(quest->RewardItemIdCount[i]);
571 }
572 for (uint8 i = 0; i < QUEST_REWARD_CHOICES_COUNT; ++i)
573 {
574 data << uint32(quest->RewardChoiceItemId[i]);
575 data << uint32(quest->RewardChoiceItemCount[i]);
576 }
577 }
578
579 for (uint8 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i) // reward factions ids
580 data << uint32(quest->RewardFactionId[i]);
581
582 for (uint8 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i) // columnid+1 QuestFactionReward.dbc?
583 data << int32(quest->RewardFactionValueId[i]);
584
585 for (uint8 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i) // unk (0)
586 data << int32(quest->RewardFactionValueIdOverride[i]);
587
588 data << uint32(quest->GetPOIContinent());
589 data << float(quest->GetPOIx());
590 data << float(quest->GetPOIy());
591 data << uint32(quest->GetPointOpt());
592
593 if (sWorld->getBoolConfig(CONFIG_UI_QUESTLEVELS_IN_DIALOGS))
594 AddQuestLevelToTitle(questTitle, quest->GetQuestLevel());
595
596 data << questTitle;
597 data << questObjectives;
598 data << questDetails;
599 data << questAreaDescription;
600 data << questCompletedText; // display in quest objectives window once all objectives are completed
601
602 for (uint8 i = 0; i < QUEST_OBJECTIVES_COUNT; ++i)
603 {
604 if (quest->RequiredNpcOrGo[i] < 0)
605 data << uint32((quest->RequiredNpcOrGo[i] * (-1)) | 0x80000000); // client expects gameobject template id in form (id|0x80000000)
606 else
607 data << uint32(quest->RequiredNpcOrGo[i]);
608
609 data << uint32(quest->RequiredNpcOrGoCount[i]);
610 data << uint32(quest->ItemDrop[i]);
611 data << uint32(0); // req source count?
612 }
613
614 for (uint8 i = 0; i < QUEST_ITEM_OBJECTIVES_COUNT; ++i)
615 {
616 data << uint32(quest->RequiredItemId[i]);
617 data << uint32(quest->RequiredItemCount[i]);
618 }
619
620 for (uint8 i = 0; i < QUEST_OBJECTIVES_COUNT; ++i)
621 data << questObjectiveText[i];
622
623 _session->SendPacket(&data);
624 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_QUEST_QUERY_RESPONSE questid=%u", quest->GetQuestId());
625}

641
642 if (sWorld->getBoolConfig(CONFIG_UI_QUESTLEVELS_IN_DIALOGS))
643 AddQuestLevelToTitle(questTitle, quest->GetQuestLevel());
644
645 WorldPacket data(SMSG_QUESTGIVER_OFFER_REWARD, 50); // guess size
646 data << uint64(npcGUID);
647 data << uint32(quest->GetQuestId());
648 data << questTitle;
649 data << questOfferRewardText;
650
651 data << uint8(enableNext ? 1 : 0); // Auto Finish
652 data << uint32(quest->GetFlags()); // 3.3.3 questFlags
653 data << uint32(quest->GetSuggestedPlayers()); // SuggestedGroupNum
654
655 uint32 emoteCount = 0;
656 for (uint8 i = 0; i < QUEST_EMOTE_COUNT; ++i)
657 {
658 if (quest->OfferRewardEmote[i] <= 0)
659 break;
660 ++emoteCount;
661 }
662
663 data << emoteCount; // Emote Count
664 for (uint8 i = 0; i < emoteCount; ++i)
665 {
666 data << uint32(quest->OfferRewardEmoteDelay[i]); // Delay Emote
667 data << uint32(quest->OfferRewardEmote[i]);
668 }
669
670 data << uint32(quest->GetRewChoiceItemsCount());
671 for (uint32 i=0; i < quest->GetRewChoiceItemsCount(); ++i)
672 {
673 data << uint32(quest->RewardChoiceItemId[i]);
674 data << uint32(quest->RewardChoiceItemCount[i]);
675
676 if (ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(quest->RewardChoiceItemId[i]))
677 data << uint32(itemTemplate->DisplayInfoID);
678 else
679 data << uint32(0);
680 }
681
682 data << uint32(quest->GetRewItemsCount());
683 for (uint32 i = 0; i < quest->GetRewItemsCount(); ++i)
684 {
685 data << uint32(quest->RewardItemId[i]);
686 data << uint32(quest->RewardItemIdCount[i]);
687
688 if (ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(quest->RewardItemId[i]))
689 data << uint32(itemTemplate->DisplayInfoID);
690 else
691 data << uint32(0);
692 }
693
694 data << uint32(quest->GetRewOrReqMoney());
695 data << uint32(quest->XPValue(_session->GetPlayer()) * sWorld->getRate(RATE_XP_QUEST));
696
697 // rewarded honor points. Multiply with 10 to satisfy client
698 data << uint32(10 * quest->CalculateHonorGain(_session->GetPlayer()->GetQuestLevel(quest)));
699 data << float(0.0f); // unk, honor multiplier?
700 data << uint32(0x08); // unused by client?
701 data << uint32(quest->GetRewSpell()); // reward spell, this spell will display (icon) (cast if RewSpellCast == 0)
702 data << int32(quest->GetRewSpellCast()); // cast spell
703 data << uint32(0); // unknown
704 data << uint32(quest->GetBonusTalents()); // bonus talents
705 data << uint32(quest->GetRewArenaPoints()); // arena points
706 data << uint32(0);
707
708 for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i) // reward factions ids
709 data << uint32(quest->RewardFactionId[i]);
710
711 for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i) // columnid in QuestFactionReward.dbc (zero based)?
712 data << int32(quest->RewardFactionValueId[i]);
713
714 for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i) // reward reputation override?
715 data << uint32(quest->RewardFactionValueIdOverride[i]);
716
717 _session->SendPacket(&data);
718 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_QUESTGIVER_OFFER_REWARD NPC=%s, questid=%u", npcGUID.ToString().c_str(), quest->GetQuestId());
719}

721void PlayerMenu::SendQuestGiverRequestItems(Quest const* quest, ObjectGuid npcGUID, bool canComplete, bool closeOnCancel) const
722{
723 // We can always call to RequestItems, but this packet only goes out if there are actually
724 // items. Otherwise, we'll skip straight to the OfferReward
725
726 std::string questTitle = quest->GetTitle();
727 std::string requestItemsText = quest->GetRequestItemsText();
728
729 int32 locale = _session->GetSessionDbLocaleIndex();
730 if (locale >= 0)
731 {
732 if (QuestLocale const* localeData = sObjectMgr->GetQuestLocale(quest->GetQuestId()))
733 {
734 ObjectMgr::GetLocaleString(localeData->Title, locale, questTitle);
735 ObjectMgr::GetLocaleString(localeData->RequestItemsText, locale, requestItemsText);
736 }
737 }
738
739 if (!quest->GetReqItemsCount() && canComplete)
740 {
741 SendQuestGiverOfferReward(quest, npcGUID, true);
742 return;
743 }
744
745 if (sWorld->getBoolConfig(CONFIG_UI_QUESTLEVELS_IN_DIALOGS))
746 AddQuestLevelToTitle(questTitle, quest->GetQuestLevel());
747
748 WorldPacket data(SMSG_QUESTGIVER_REQUEST_ITEMS, 50); // guess size
749 data << uint64(npcGUID);
750 data << uint32(quest->GetQuestId());
751 data << questTitle;
752 data << requestItemsText;
753
754 data << uint32(0); // unknown
755
756 if (canComplete)
757 data << quest->GetCompleteEmote();
758 else
759 data << quest->GetIncompleteEmote();
760
761 // Close Window after cancel
762 data << uint32(closeOnCancel);
763
764 data << uint32(quest->GetFlags()); // 3.3.3 questFlags
765 data << uint32(quest->GetSuggestedPlayers()); // SuggestedGroupNum
766
767 // Required Money
768 data << uint32(quest->GetRewOrReqMoney() < 0 ? -quest->GetRewOrReqMoney() : 0);
769
770 data << uint32(quest->GetReqItemsCount());
771 for (int i = 0; i < QUEST_ITEM_OBJECTIVES_COUNT; ++i)
772 {
773 if (!quest->RequiredItemId[i])
774 continue;
775
776 data << uint32(quest->RequiredItemId[i]);
777 data << uint32(quest->RequiredItemCount[i]);
778
779 if (ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(quest->RequiredItemId[i]))
780 data << uint32(itemTemplate->DisplayInfoID);
781 else
782 data << uint32(0);
783 }
784
785 if (!canComplete)
786 data << uint32(0x00);
787 else
788 data << uint32(0x03);
789
790 data << uint32(0x04);
791 data << uint32(0x08);
792 data << uint32(0x10);
793
794 _session->SendPacket(&data);
795 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_QUESTGIVER_REQUEST_ITEMS NPC=%s, questid=%u", npcGUID.ToString().c_str(), quest->GetQuestId());
796}

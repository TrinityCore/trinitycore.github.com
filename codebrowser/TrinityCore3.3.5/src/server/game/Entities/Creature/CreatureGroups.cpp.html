33void FormationMgr::AddCreatureToGroup(uint32 leaderGuid, Creature* creature)
34{
35 Map* map = creature->FindMap();
36 if (!map)
37 return;
38
39 CreatureGroupHolderType::iterator itr = map->CreatureGroupHolder.find(leaderGuid);
40
41 //Add member to an existing group
42 if (itr != map->CreatureGroupHolder.end())
43 {
44 TC_LOG_DEBUG("entities.unit", "Group found: %u, inserting creature GUID: %u, Group InstanceID %u", leaderGuid, creature->GetGUID().GetCounter(), creature->GetInstanceId());
45 itr->second->AddMember(creature);
46 }
47 //Create new group
48 else
49 {
50 TC_LOG_DEBUG("entities.unit", "Group not found: %u. Creating new group.", leaderGuid);
51 CreatureGroup* group = new CreatureGroup(leaderGuid);
52 map->CreatureGroupHolder[leaderGuid] = group;
53 group->AddMember(creature);
54 }
55}

57void FormationMgr::RemoveCreatureFromGroup(CreatureGroup* group, Creature* member)
58{
59 TC_LOG_DEBUG("entities.unit", "Deleting member pointer to GUID: %u from group %u", group->GetId(), member->GetSpawnId());
60 group->RemoveMember(member);
61
62 if (group->isEmpty())
63 {
64 Map* map = member->FindMap();
65 if (!map)
66 return;
67
68 TC_LOG_DEBUG("entities.unit", "Deleting group with InstanceID %u", member->GetInstanceId());
69 map->CreatureGroupHolder.erase(group->GetId());
70 delete group;
71 }
72}

74void FormationMgr::LoadCreatureFormations()
75{
76 uint32 oldMSTime = getMSTime();
77
78 for (CreatureGroupInfoType::iterator itr = CreatureGroupMap.begin(); itr != CreatureGroupMap.end(); ++itr) // for reload case
79 delete itr->second;
80 CreatureGroupMap.clear();
81
82 //Get group data
83 QueryResult result = WorldDatabase.Query("SELECT leaderGUID, memberGUID, dist, angle, groupAI, point_1, point_2 FROM creature_formations ORDER BY leaderGUID");
84
85 if (!result)
86 {
87 TC_LOG_ERROR("server.loading", ">> Loaded 0 creatures in formations. DB table `creature_formations` is empty!");
88 return;
89 }
90
91 uint32 count = 0;
92 Field* fields;
93 FormationInfo* group_member;
94
95 do
96 {
97 fields = result->Fetch();
98
99 //Load group member data
100 group_member = new FormationInfo();
101 group_member->leaderGUID = fields[0].GetUInt32();
102 uint32 memberGUID = fields[1].GetUInt32();
103 group_member->groupAI = fields[4].GetUInt32();
104 group_member->point_1 = fields[5].GetUInt16();
105 group_member->point_2 = fields[6].GetUInt16();
106 //If creature is group leader we may skip loading of dist/angle
107 if (group_member->leaderGUID != memberGUID)
108 {
109 group_member->follow_dist = fields[2].GetFloat();
110 group_member->follow_angle = fields[3].GetFloat() * float(M_PI) / 180;
111 }
112 else
113 {
114 group_member->follow_dist = 0;
115 group_member->follow_angle = 0;
116 }
117
118 // check data correctness
119 {
120 if (!sObjectMgr->GetCreatureData(group_member->leaderGUID))
121 {
122 TC_LOG_ERROR("sql.sql", "creature_formations table leader guid %u incorrect (not exist)", group_member->leaderGUID);
123 delete group_member;
124 continue;
125 }
126
127 if (!sObjectMgr->GetCreatureData(memberGUID))
128 {
129 TC_LOG_ERROR("sql.sql", "creature_formations table member guid %u incorrect (not exist)", memberGUID);
130 delete group_member;
131 continue;
132 }
133 }
134
135 CreatureGroupMap[memberGUID] = group_member;
136 ++count;
137 }
138 while (result->NextRow());
139
140 TC_LOG_INFO("server.loading", ">> Loaded %u creatures in formations in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
141}

143void CreatureGroup::AddMember(Creature* member)
144{
145 TC_LOG_DEBUG("entities.unit", "CreatureGroup::AddMember: Adding unit GUID: %u.", member->GetGUID().GetCounter());
146
147 //Check if it is a leader
148 if (member->GetSpawnId() == m_groupID)
149 {
150 TC_LOG_DEBUG("entities.unit", "Unit GUID: %u is formation leader. Adding group.", member->GetGUID().GetCounter());
151 m_leader = member;
152 }
153
154 m_members[member] = sFormationMgr->CreatureGroupMap.find(member->GetSpawnId())->second;
155 member->SetFormation(this);
156}

158void CreatureGroup::RemoveMember(Creature* member)
159{
160 if (m_leader == member)
161 m_leader = NULL;
162
163 m_members.erase(member);
164 member->SetFormation(NULL);
165}

167void CreatureGroup::MemberAttackStart(Creature* member, Unit* target)
168{
169 uint8 groupAI = sFormationMgr->CreatureGroupMap[member->GetSpawnId()]->groupAI;
170 if (!groupAI)
171 return;
172
173 if (groupAI == 1 && member != m_leader)
174 return;
175
176 for (CreatureGroupMemberType::iterator itr = m_members.begin(); itr != m_members.end(); ++itr)
177 {
178 if (m_leader) // avoid crash if leader was killed and reset.
179 TC_LOG_DEBUG("entities.unit", "GROUP ATTACK: group instance id %u calls member instid %u", m_leader->GetInstanceId(), member->GetInstanceId());
180
181 Creature* other = itr->first;
182
183 // Skip self
184 if (other == member)
185 continue;
186
187 if (!other->IsAlive())
188 continue;
189
190 if (other->GetVictim())
191 continue;
192
193 if (other->IsValidAttackTarget(target))
194 other->AI()->AttackStart(target);
195 }
196}

198void CreatureGroup::FormationReset(bool dismiss)
199{
200 for (CreatureGroupMemberType::iterator itr = m_members.begin(); itr != m_members.end(); ++itr)
201 {
202 if (itr->first != m_leader && itr->first->IsAlive())
203 {
204 if (dismiss)
205 itr->first->GetMotionMaster()->Initialize();
206 else
207 itr->first->GetMotionMaster()->MoveIdle();
208 TC_LOG_DEBUG("entities.unit", "Set %s movement for member GUID: %u", dismiss ? "default" : "idle", itr->first->GetGUID().GetCounter());
209 }
210 }
211 m_Formed = !dismiss;
212}

214void CreatureGroup::LeaderMoveTo(float x, float y, float z)
215{
216 //! To do: This should probably get its own movement generator or use WaypointMovementGenerator.
217 //! If the leader's path is known, member's path can be plotted as well using formation offsets.
218 if (!m_leader)
219 return;
220
221 float pathangle = std::atan2(m_leader->GetPositionY() - y, m_leader->GetPositionX() - x);
222
223 for (CreatureGroupMemberType::iterator itr = m_members.begin(); itr != m_members.end(); ++itr)
224 {
225 Creature* member = itr->first;
226 if (member == m_leader || !member->IsAlive() || member->GetVictim())
227 continue;
228
229 if (itr->second->point_1)
230 if (m_leader->GetCurrentWaypointID() == itr->second->point_1 - 1 || m_leader->GetCurrentWaypointID() == itr->second->point_2 - 1)
231 itr->second->follow_angle = float(M_PI) * 2 - itr->second->follow_angle;
232
233 float angle = itr->second->follow_angle;
234 float dist = itr->second->follow_dist;
235
236 float dx = x + std::cos(angle + pathangle) * dist;
237 float dy = y + std::sin(angle + pathangle) * dist;
238 float dz = z;
239
240 Trinity::NormalizeMapCoord(dx);
241 Trinity::NormalizeMapCoord(dy);
242
243 if (!member->IsFlying())
244 member->UpdateGroundPositionZ(dx, dy, dz);
245
246 if (member->IsWithinDist(m_leader, dist + MAX_DESYNC))
247 member->SetUnitMovementFlags(m_leader->GetUnitMovementFlags());
248 else
249 member->SetWalk(false);
250
251 member->GetMotionMaster()->MovePoint(0, dx, dy, dz);
252 member->SetHomePosition(dx, dy, dz, pathangle);
253 }

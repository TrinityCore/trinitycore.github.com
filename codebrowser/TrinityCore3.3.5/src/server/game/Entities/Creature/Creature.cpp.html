148bool AssistDelayEvent::Execute(uint64 /*e_time*/, uint32 /*p_time*/)
149{
150 if (Unit* victim = ObjectAccessor::GetUnit(m_owner, m_victim))
151 {
152 while (!m_assistants.empty())
153 {
154 Creature* assistant = ObjectAccessor::GetCreature(m_owner, *m_assistants.begin());
155 m_assistants.pop_front();
156
157 if (assistant && assistant->CanAssistTo(&m_owner, victim))
158 {
159 assistant->SetNoCallAssistance(true);
160 assistant->CombatStart(victim);
161 if (assistant->IsAIEnabled)
162 assistant->AI()->AttackStart(victim);
163 }
164 }
165 }
166 return true;

308bool Creature::InitEntry(uint32 entry, CreatureData const* data /*= nullptr*/)
309{
310 CreatureTemplate const* normalInfo = sObjectMgr->GetCreatureTemplate(entry);
311 if (!normalInfo)
312 {
313 TC_LOG_ERROR("sql.sql", "Creature::InitEntry creature entry %u does not exist.", entry);
314 return false;
315 }
316
317 // get difficulty 1 mode entry
318 CreatureTemplate const* cinfo = normalInfo;
319 for (uint8 diff = uint8(GetMap()->GetSpawnMode()); diff > 0;)
320 {
321 // we already have valid Map pointer for current creature!
322 if (normalInfo->DifficultyEntry[diff - 1])
323 {
324 cinfo = sObjectMgr->GetCreatureTemplate(normalInfo->DifficultyEntry[diff - 1]);
325 if (cinfo)
326 break; // template found
327
328 // check and reported at startup, so just ignore (restore normalInfo)
329 cinfo = normalInfo;
330 }
331
332 // for instances heroic to normal, other cases attempt to retrieve previous difficulty
333 if (diff >= RAID_DIFFICULTY_10MAN_HEROIC && GetMap()->IsRaid())
334 diff -= 2; // to normal raid difficulty cases
335 else
336 --diff;
337 }
338
339 // Initialize loot duplicate count depending on raid difficulty
340 if (GetMap()->Is25ManRaid())
341 loot.maxDuplicates = 3;
342
343 SetEntry(entry); // normal entry always
344 m_creatureInfo = cinfo; // map mode related always
345
346 // equal to player Race field, but creature does not have race
347 SetByteValue(UNIT_FIELD_BYTES_0, 0, 0);
348
349 // known valid are: CLASS_WARRIOR, CLASS_PALADIN, CLASS_ROGUE, CLASS_MAGE
350 SetByteValue(UNIT_FIELD_BYTES_0, 1, uint8(cinfo->unit_class));
351
352 // Cancel load if no model defined
353 if (!(cinfo->GetFirstValidModelId()))
354 {
355 TC_LOG_ERROR("sql.sql", "Creature (Entry: %u) has no model defined in table `creature_template`, can't load. ", entry);
356 return false;
357 }
358
359 uint32 displayID = ObjectMgr::ChooseDisplayId(GetCreatureTemplate(), data);
360 CreatureModelInfo const* minfo = sObjectMgr->GetCreatureModelRandomGender(&displayID);
361 if (!minfo) // Cancel load if no model defined
362 {
363 TC_LOG_ERROR("sql.sql", "Creature (Entry: %u) has invalid model %u defined in table `creature_template`, can't load.", entry, displayID);
364 return false;
365 }
366
367 SetDisplayId(displayID);
368 SetNativeDisplayId(displayID);
369 SetByteValue(UNIT_FIELD_BYTES_0, 2, minfo->gender);
370
371 // Load creature equipment
372 if (data && data->equipmentId != 0)
373 {
374 m_originalEquipmentId = data->equipmentId;
375 LoadEquipment(data->equipmentId);
376 }
377 else
378 LoadEquipment(0, true);
379
380 SetName(normalInfo->Name); // at normal entry always
381
382 SetFloatValue(UNIT_MOD_CAST_SPEED, 1.0f);
383
384 SetSpeed(MOVE_WALK, cinfo->speed_walk);
385 SetSpeed(MOVE_RUN, cinfo->speed_run);
386 SetSpeed(MOVE_SWIM, 1.0f); // using 1.0 rate
387 SetSpeed(MOVE_FLIGHT, 1.0f); // using 1.0 rate
388
389 // Will set UNIT_FIELD_BOUNDINGRADIUS and UNIT_FIELD_COMBATREACH
390 SetObjectScale(cinfo->scale);
391
392 SetFloatValue(UNIT_FIELD_HOVERHEIGHT, cinfo->HoverHeight);
393
394 // checked at loading
395 m_defaultMovementType = MovementGeneratorType(cinfo->MovementType);
396 if (!m_respawnradius && m_defaultMovementType == RANDOM_MOTION_TYPE)
397 m_defaultMovementType = IDLE_MOTION_TYPE;
398
399 for (uint8 i=0; i < CREATURE_MAX_SPELLS; ++i)
400 m_spells[i] = GetCreatureTemplate()->spells[i];
401
402 return true;
403}

405bool Creature::UpdateEntry(uint32 entry, CreatureData const* data /*= nullptr*/)
406{
407 if (!InitEntry(entry, data))
408 return false;
409
410 CreatureTemplate const* cInfo = GetCreatureTemplate();
411
412 m_regenHealth = cInfo->RegenHealth;
413
414 // creatures always have melee weapon ready if any unless specified otherwise
415 if (!GetCreatureAddon())
416 SetSheath(SHEATH_STATE_MELEE);
417
418 setFaction(cInfo->faction);
419
420 uint32 npcflag, unit_flags, dynamicflags;
421 ObjectMgr::ChooseCreatureFlags(cInfo, npcflag, unit_flags, dynamicflags, data);
422
423 if (cInfo->flags_extra & CREATURE_FLAG_EXTRA_WORLDEVENT)
424 SetUInt32Value(UNIT_NPC_FLAGS, npcflag | sGameEventMgr->GetNPCFlag(this));
425 else
426 SetUInt32Value(UNIT_NPC_FLAGS, npcflag);
427
428 SetUInt32Value(UNIT_FIELD_FLAGS, unit_flags);
429 SetUInt32Value(UNIT_FIELD_FLAGS_2, cInfo->unit_flags2);
430
431 SetUInt32Value(UNIT_DYNAMIC_FLAGS, dynamicflags);
432
433 RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IN_COMBAT);
434
435 SetAttackTime(BASE_ATTACK, cInfo->BaseAttackTime);
436 SetAttackTime(OFF_ATTACK, cInfo->BaseAttackTime);
437 SetAttackTime(RANGED_ATTACK, cInfo->RangeAttackTime);
438
439 SelectLevel();
440
441 SetMeleeDamageSchool(SpellSchools(cInfo->dmgschool));
442 CreatureBaseStats const* stats = sObjectMgr->GetCreatureBaseStats(getLevel(), cInfo->unit_class);
443 float armor = (float)stats->GenerateArmor(cInfo); /// @todo Why is this treated as uint32 when it's a float?
444 SetModifierValue(UNIT_MOD_ARMOR, BASE_VALUE, armor);
445 SetModifierValue(UNIT_MOD_RESISTANCE_HOLY, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_HOLY]));
446 SetModifierValue(UNIT_MOD_RESISTANCE_FIRE, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_FIRE]));
447 SetModifierValue(UNIT_MOD_RESISTANCE_NATURE, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_NATURE]));
448 SetModifierValue(UNIT_MOD_RESISTANCE_FROST, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_FROST]));
449 SetModifierValue(UNIT_MOD_RESISTANCE_SHADOW, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_SHADOW]));
450 SetModifierValue(UNIT_MOD_RESISTANCE_ARCANE, BASE_VALUE, float(cInfo->resistance[SPELL_SCHOOL_ARCANE]));
451
452 SetCanModifyStats(true);
453 UpdateAllStats();
454
455 // checked and error show at loading templates
456 if (FactionTemplateEntry const* factionTemplate = sFactionTemplateStore.LookupEntry(cInfo->faction))
457 {
458 if (factionTemplate->factionFlags & FACTION_TEMPLATE_FLAG_PVP)
459 SetPvP(true);
460 else
461 SetPvP(false);
462 }
463
464 // updates spell bars for vehicles and set player's faction - should be called here, to overwrite faction that is set from the new template
465 if (IsVehicle())
466 {
467 if (Player* owner = Creature::GetCharmerOrOwnerPlayerOrPlayerItself()) // this check comes in case we don't have a player
468 {
469 setFaction(owner->getFaction()); // vehicles should have same as owner faction
470 owner->VehicleSpellInitialize();
471 }
472 }
473
474 // trigger creature is always not selectable and can not be attacked
475 if (IsTrigger())
476 SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
477
478 InitializeReactState();
479
480 if (cInfo->flags_extra & CREATURE_FLAG_EXTRA_NO_TAUNT)
481 {
482 ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, true);
483 ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, true);
484 }
485
486 UpdateMovementFlags();
487 LoadCreaturesAddon();
488 return true;
489}

491void Creature::Update(uint32 diff)
492{
493 if (IsAIEnabled && TriggerJustRespawned)
494 {
495 TriggerJustRespawned = false;
496 AI()->JustRespawned();
497 if (m_vehicleKit)
498 m_vehicleKit->Reset();
499 }
500
501 UpdateMovementFlags();
502
503 switch (m_deathState)
504 {
505 case JUST_RESPAWNED:
506 // Must not be called, see Creature::setDeathState JUST_RESPAWNED -> ALIVE promoting.
507 TC_LOG_ERROR("entities.unit", "Creature (GUID: %u Entry: %u) in wrong state: JUST_RESPAWNED (4)", GetGUID().GetCounter(), GetEntry());
508 break;
509 case JUST_DIED:
510 // Must not be called, see Creature::setDeathState JUST_DIED -> CORPSE promoting.
511 TC_LOG_ERROR("entities.unit", "Creature (GUID: %u Entry: %u) in wrong state: JUST_DEAD (1)", GetGUID().GetCounter(), GetEntry());
512 break;
513 case DEAD:
514 {
515 time_t now = time(NULL);
516 if (m_respawnTime <= now)
517 {
518 bool allowed = IsAIEnabled ? AI()->CanRespawn() : true; // First check if there are any scripts that object to us respawning
519 if (!allowed) // Will be rechecked on next Update call
520 break;
521
522 ObjectGuid dbtableHighGuid(HighGuid::Unit, GetEntry(), m_spawnId);
523 time_t linkedRespawntime = GetMap()->GetLinkedRespawnTime(dbtableHighGuid);
524 if (!linkedRespawntime) // Can respawn
525 Respawn();
526 else // the master is dead
527 {
528 ObjectGuid targetGuid = sObjectMgr->GetLinkedRespawnGuid(dbtableHighGuid);
529 if (targetGuid == dbtableHighGuid) // if linking self, never respawn (check delayed to next day)
530 SetRespawnTime(DAY);
531 else
532 m_respawnTime = (now > linkedRespawntime ? now : linkedRespawntime) + urand(5, MINUTE); // else copy time from master and add a little
533 SaveRespawnTime(); // also save to DB immediately
534 }
535 }
536 break;
537 }
538 case CORPSE:
539 {
540 Unit::Update(diff);
541 // deathstate changed on spells update, prevent problems
542 if (m_deathState != CORPSE)
543 break;
544
545 if (m_groupLootTimer && lootingGroupLowGUID)
546 {
547 if (m_groupLootTimer <= diff)
548 {
549 Group* group = sGroupMgr->GetGroupByGUID(lootingGroupLowGUID);
550 if (group)
551 group->EndRoll(&loot);
552 m_groupLootTimer = 0;
553 lootingGroupLowGUID = 0;
554 }
555 else m_groupLootTimer -= diff;
556 }
557 else if (m_corpseRemoveTime <= time(NULL))
558 {
559 RemoveCorpse(false);
560 TC_LOG_DEBUG("entities.unit", "Removing corpse... %u ", GetUInt32Value(OBJECT_FIELD_ENTRY));
561 }
562 break;
563 }
564 case ALIVE:
565 {
566 Unit::Update(diff);
567
568 // creature can be dead after Unit::Update call
569 // CORPSE/DEAD state will processed at next tick (in other case death timer will be updated unexpectedly)
570 if (!IsAlive())
571 break;
572
573 // if creature is charmed, switch to charmed AI (and back)
574 if (NeedChangeAI)
575 {
576 UpdateCharmAI();
577 NeedChangeAI = false;
578 IsAIEnabled = true;
579 if (!IsInEvadeMode() && LastCharmerGUID)
580 if (Unit* charmer = ObjectAccessor::GetUnit(*this, LastCharmerGUID))
581 if (CanStartAttack(charmer, true))
582 i_AI->AttackStart(charmer);
583
584 LastCharmerGUID.Clear();
585 }
586
587 // periodic check to see if the creature has passed an evade boundary
588 if (IsAIEnabled && !IsInEvadeMode() && IsInCombat())
589 {
590 if (diff >= m_boundaryCheckTime)
591 {
592 AI()->CheckInRoom();
593 m_boundaryCheckTime = 2500;
594 } else
595 m_boundaryCheckTime -= diff;
596 }
597
598 // if periodic combat pulse is enabled and we are both in combat and in a dungeon, do this now
599 if (m_combatPulseDelay > 0 && IsInCombat() && GetMap()->IsDungeon())
600 {
601 if (diff > m_combatPulseTime)
602 m_combatPulseTime = 0;
603 else
604 m_combatPulseTime -= diff;
605
606 if (m_combatPulseTime == 0)
607 {
608 Map::PlayerList const &players = GetMap()->GetPlayers();
609 if (!players.isEmpty())
610 for (Map::PlayerList::const_iterator it = players.begin(); it != players.end(); ++it)
611 {
612 if (Player* player = it->GetSource())
613 {
614 if (player->IsGameMaster())
615 continue;
616
617 if (player->IsAlive() && this->IsHostileTo(player))
618 {
619 if (CanHaveThreatList())
620 AddThreat(player, 0.0f);
621 this->SetInCombatWith(player);
622 player->SetInCombatWith(this);
623 }
624 }
625 }
626
627 m_combatPulseTime = m_combatPulseDelay * IN_MILLISECONDS;
628 }
629 }
630
631 if (!IsInEvadeMode() && IsAIEnabled)
632 {
633 // do not allow the AI to be changed during update
634 m_AI_locked = true;
635
636 i_AI->UpdateAI(diff);
637 m_AI_locked = false;
638 }
639
640 // creature can be dead after UpdateAI call
641 // CORPSE/DEAD state will processed at next tick (in other case death timer will be updated unexpectedly)
642 if (!IsAlive())
643 break;
644
645 if (m_regenTimer > 0)
646 {
647 if (diff >= m_regenTimer)
648 m_regenTimer = 0;
649 else
650 m_regenTimer -= diff;
651 }
652
653 if (m_regenTimer != 0)
654 break;
655
656 bool bInCombat = IsInCombat() && (!GetVictim() || // if IsInCombat() is true and this has no victim
657 !EnsureVictim()->GetCharmerOrOwnerPlayerOrPlayerItself() || // or the victim/owner/charmer is not a player
658 !EnsureVictim()->GetCharmerOrOwnerPlayerOrPlayerItself()->IsGameMaster()); // or the victim/owner/charmer is not a GameMaster
659
660 /*if (m_regenTimer <= diff)
661 {*/
662 if (!IsInEvadeMode() && (!bInCombat || IsPolymorphed())) // regenerate health if not in combat or if polymorphed
663 RegenerateHealth();
664
665 if (HasFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_REGENERATE_POWER))
666 {
667 if (getPowerType() == POWER_ENERGY)
668 Regenerate(POWER_ENERGY);
669 else
670 RegenerateMana();

1499bool Creature::CanStartAttack(Unit const* who, bool force) const
1500{
1501 if (IsCivilian())
1502 return false;
1503
1504 // This set of checks is should be done only for creatures
1505 if ((HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC) && who->GetTypeId() != TYPEID_PLAYER) // flag is valid only for non player characters
1506 || (HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC) && who->GetTypeId() == TYPEID_PLAYER) // immune to PC and target is a player, return false
1507 || (who->GetOwner() && who->GetOwner()->GetTypeId() == TYPEID_PLAYER && HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC))) // player pets are immune to pc as well
1508 return false;
1509
1510 // Do not attack non-combat pets
1511 if (who->GetTypeId() == TYPEID_UNIT && who->GetCreatureType() == CREATURE_TYPE_NON_COMBAT_PET)
1512 return false;
1513
1514 if (!CanFly() && (GetDistanceZ(who) > CREATURE_Z_ATTACK_RANGE + m_CombatDistance))
1515 //|| who->IsControlledByPlayer() && who->IsFlying()))
1516 // we cannot check flying for other creatures, too much map/vmap calculation
1517 /// @todo should switch to range attack
1518 return false;
1519
1520 if (!force)
1521 {
1522 if (!_IsTargetAcceptable(who))
1523 return false;
1524
1525 if (who->IsInCombat() && IsWithinDist(who, ATTACK_DISTANCE))
1526 if (Unit* victim = who->getAttackerForHelper())
1527 if (IsWithinDistInMap(victim, sWorld->getFloatConfig(CONFIG_CREATURE_FAMILY_ASSISTANCE_RADIUS)))
1528 force = true;
1529
1530 if (!force && (IsNeutralToAll() || !IsWithinDistInMap(who, GetAttackDistance(who) + m_CombatDistance)))
1531 return false;
1532 }
1533
1534 if (!CanCreatureAttack(who, force))
1535 return false;
1536
1537 // No aggro from gray creatures
1538 if (CheckNoGrayAggroConfig(who->getLevelForTarget(this), getLevelForTarget(who)))

1560float Creature::GetAttackDistance(Unit const* player) const
1561{
1562 float aggroRate = sWorld->getRate(RATE_CREATURE_AGGRO);
1563 if (aggroRate == 0)
1564 return 0.0f;
1565
1566 uint32 playerlevel = player->getLevelForTarget(this);
1567 uint32 creaturelevel = getLevelForTarget(player);
1568
1569 int32 leveldif = int32(playerlevel) - int32(creaturelevel);
1570
1571 // "The maximum Aggro Radius has a cap of 25 levels under. Example: A level 30 char has the same Aggro Radius of a level 5 char on a level 60 mob."
1572 if (leveldif < - 25)
1573 leveldif = -25;
1574
1575 // "The aggro radius of a mob having the same level as the player is roughly 20 yards"
1576 float RetDistance = 20;
1577
1578 // "Aggro Radius varies with level difference at a rate of roughly 1 yard/level"
1579 // radius grow if playlevel < creaturelevel
1580 RetDistance -= (float)leveldif;
1581
1582 if (creaturelevel+5 <= sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
1583 {
1584 // detect range auras
1585 RetDistance += GetTotalAuraModifier(SPELL_AURA_MOD_DETECT_RANGE);
1586
1587 // detected range auras
1588 RetDistance += player->GetTotalAuraModifier(SPELL_AURA_MOD_DETECTED_RANGE);
1589 }
1590
1591 // "Minimum Aggro Radius for a mob seems to be combat range (5 yards)"
1592 if (RetDistance < 5)
1593 RetDistance = 5;
1594
1595 return (RetDistance*aggroRate);

2633float Creature::GetAggroRange(Unit const* target) const
2634{
2635 // Determines the aggro range for creatures (usually pets), used mainly for aggressive pet target selection.
2636 // Based on data from wowwiki due to lack of 3.3.5a data
2637
2638 if (target && this->IsPet())
2639 {
2640 uint32 targetLevel = 0;
2641
2642 if (target->GetTypeId() == TYPEID_PLAYER)
2643 targetLevel = target->getLevelForTarget(this);
2644 else if (target->GetTypeId() == TYPEID_UNIT)
2645 targetLevel = target->ToCreature()->getLevelForTarget(this);
2646
2647 uint32 myLevel = getLevelForTarget(target);
2648 int32 levelDiff = int32(targetLevel) - int32(myLevel);
2649
2650 // The maximum Aggro Radius is capped at 45 yards (25 level difference)
2651 if (levelDiff < -25)
2652 levelDiff = -25;
2653
2654 // The base aggro radius for mob of same level
2655 float aggroRadius = 20;
2656
2657 // Aggro Radius varies with level difference at a rate of roughly 1 yard/level
2658 aggroRadius -= (float)levelDiff;
2659
2660 // detect range auras
2661 aggroRadius += GetTotalAuraModifier(SPELL_AURA_MOD_DETECT_RANGE);
2662
2663 // detected range auras
2664 aggroRadius += target->GetTotalAuraModifier(SPELL_AURA_MOD_DETECTED_RANGE);
2665
2666 // Just in case, we don't want pets running all over the map
2667 if (aggroRadius > MAX_AGGRO_RADIUS)
2668 aggroRadius = MAX_AGGRO_RADIUS;
2669
2670 // Minimum Aggro Radius for a mob seems to be combat range (5 yards)
2671 // hunter pets seem to ignore minimum aggro radius so we'll default it a little higher
2672 if (aggroRadius < 10)
2673 aggroRadius = 10;
2674
2675 return (aggroRadius);
2676 }
2677
2678 // Default
2679 return 0.0f;

2757bool Creature::FocusTarget(Spell const* focusSpell, WorldObject const* target)
2758{
2759 // already focused
2760 if (_focusSpell)
2761 return false;
2762
2763 if ((!target || target == this) && !focusSpell->GetCastTime()) // instant cast, untargeted (or self-targeted) spell doesn't need any facing updates
2764 return false;
2765
2766 _focusSpell = focusSpell;
2767
2768 // "instant" creature casts that require re-targeting will be delayed by a short moment to prevent facing bugs
2769 bool shouldDelay = false;
2770
2771 // set target, then force send update packet to players if it changed to provide appropriate facing
2772 ObjectGuid newTarget = target ? target->GetGUID() : ObjectGuid::Empty;
2773 if (GetGuidValue(UNIT_FIELD_TARGET) != newTarget)
2774 {
2775 SetGuidValue(UNIT_FIELD_TARGET, newTarget);
2776 if (target)
2777 SetFacingToObject(target);
2778
2779 if ( // here we determine if the (relatively expensive) forced update is worth it, or whether we can afford to wait until the scheduled update tick
2780 ( // only require instant update for spells that actually have a visual
2781 focusSpell->GetSpellInfo()->SpellVisual[0] ||
2782 focusSpell->GetSpellInfo()->SpellVisual[1]
2783 ) && (
2784 !focusSpell->GetCastTime() || // if the spell is instant cast
2785 focusSpell->GetSpellInfo()->HasAttribute(SPELL_ATTR5_DONT_TURN_DURING_CAST) // client gets confused if we attempt to turn at the regularly scheduled update packet
2786 )
2787 )
2788 {
2789 const MapRefManager& mapPlayers = GetMap()->GetPlayers();
2790 for (MapRefManager::const_iterator it = mapPlayers.begin(); it != mapPlayers.end(); ++it)
2791 if (Player* player = (*it).GetSource())
2792 {
2793 // only update players that can both see us, and are actually in combat with us (this is a performance tradeoff)
2794 if (player->CanSeeOrDetect(this, false, true) && IsInCombatWith(player))
2795 {
2796 SendUpdateToPlayer(player);
2797 shouldDelay = true;
2798 }
2799 }
2800 if (shouldDelay)
2801 shouldDelay = !(focusSpell->IsTriggered() || focusSpell->GetCastTime() || focusSpell->GetSpellInfo()->IsChanneled());
2802
2803 }
2804 }
2805
2806 // tell the creature that it should reacquire its current target after the cast is done (this is handled in ::Attack)
2807 MustReacquireTarget();
2808
2809 bool canTurnDuringCast = !focusSpell->GetSpellInfo()->HasAttribute(SPELL_ATTR5_DONT_TURN_DURING_CAST);
2810 // Face the target - we need to do this before the unit state is modified for no-turn spells
2811 if (target)
2812 SetInFront(target);
2813 else if (!canTurnDuringCast)
2814 if(Unit* victim = GetVictim())
2815 SetInFront(victim); // ensure server-side orientation is correct at beginning of cast
2816
2817 if (!canTurnDuringCast)
2818 AddUnitState(UNIT_STATE_CANNOT_TURN);
2819
2820 return shouldDelay;
2821}

2824{

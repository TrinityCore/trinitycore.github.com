30struct EnchStoreItem
31{
32 uint32 ench;
33 float chance;
34
35 EnchStoreItem()
36 : ench(0), chance(0) { }
37
38 EnchStoreItem(uint32 _ench, float _chance)
39 : ench(_ench), chance(_chance) { }
40};
41
42typedef std::vector<EnchStoreItem> EnchStoreList;
43typedef std::unordered_map<uint32, EnchStoreList> EnchantmentStore;
44
45static EnchantmentStore RandomItemEnch;
46
47void LoadRandomEnchantmentsTable()
48{
49 uint32 oldMSTime = getMSTime();
50
51 RandomItemEnch.clear(); // for reload case
52
53 // 0 1 2
54 QueryResult result = WorldDatabase.Query("SELECT entry, ench, chance FROM item_enchantment_template");
55
56 if (result)
57 {
58 uint32 count = 0;
59
60 do
61 {
62 Field* fields = result->Fetch();
63
64 uint32 entry = fields[0].GetUInt32();
65 uint32 ench = fields[1].GetUInt32();
66 float chance = fields[2].GetFloat();
67
68 if (chance > 0.000001f && chance <= 100.0f)
69 RandomItemEnch[entry].push_back(EnchStoreItem(ench, chance));
70
71 ++count;
72 } while (result->NextRow());
73
74 TC_LOG_INFO("server.loading", ">> Loaded %u Item Enchantment definitions in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
75 }
76 else
77 TC_LOG_ERROR("server.loading", ">> Loaded 0 Item Enchantment definitions. DB table `item_enchantment_template` is empty.");
78}

80uint32 GetItemEnchantMod(int32 entry)
81{
82 if (!entry)
83 return 0;
84
85 if (entry == -1)
86 return 0;
87
88 EnchantmentStore::const_iterator tab = RandomItemEnch.find(entry);
89 if (tab == RandomItemEnch.end())
90 {
91 TC_LOG_ERROR("sql.sql", "Item RandomProperty / RandomSuffix id #%u used in `item_template` but it does not have records in `item_enchantment_template` table.", entry);
92 return 0;
93 }
94
95 double dRoll = rand_chance();
96 float fCount = 0;
97
98 for (EnchStoreList::const_iterator ench_iter = tab->second.begin(); ench_iter != tab->second.end(); ++ench_iter)
99 {
100 fCount += ench_iter->chance;
101
102 if (fCount > dRoll)
103 return ench_iter->ench;
104 }
105
106 //we could get here only if sum of all enchantment chances is lower than 100%
107 dRoll = (irand(0, (int)floor(fCount * 100) + 1)) / 100;
108 fCount = 0;
109
110 for (EnchStoreList::const_iterator ench_iter = tab->second.begin(); ench_iter != tab->second.end(); ++ench_iter)
111 {
112 fCount += ench_iter->chance;
113
114 if (fCount > dRoll)
115 return ench_iter->ench;
116 }
117
118 return 0;
119}

121uint32 GenerateEnchSuffixFactor(uint32 item_id)
122{
123 ItemTemplate const* itemProto = sObjectMgr->GetItemTemplate(item_id);
124
125 if (!itemProto)
126 return 0;
127 if (!itemProto->RandomSuffix)
128 return 0;
129
130 RandomPropertiesPointsEntry const* randomProperty = sRandomPropertiesPointsStore.LookupEntry(itemProto->ItemLevel);
131 if (!randomProperty)
132 return 0;
133
134 uint32 suffixFactor;
135 switch (itemProto->InventoryType)
136 {
137 // Items of that type don`t have points
138 case INVTYPE_NON_EQUIP:
139 case INVTYPE_BAG:
140 case INVTYPE_TABARD:
141 case INVTYPE_AMMO:
142 case INVTYPE_QUIVER:
143 case INVTYPE_RELIC:
144 return 0;
145 // Select point coefficient
146 case INVTYPE_HEAD:
147 case INVTYPE_BODY:
148 case INVTYPE_CHEST:
149 case INVTYPE_LEGS:
150 case INVTYPE_2HWEAPON:
151 case INVTYPE_ROBE:
152 suffixFactor = 0;
153 break;
154 case INVTYPE_SHOULDERS:
155 case INVTYPE_WAIST:
156 case INVTYPE_FEET:
157 case INVTYPE_HANDS:
158 case INVTYPE_TRINKET:
159 suffixFactor = 1;
160 break;
161 case INVTYPE_NECK:
162 case INVTYPE_WRISTS:
163 case INVTYPE_FINGER:
164 case INVTYPE_SHIELD:
165 case INVTYPE_CLOAK:
166 case INVTYPE_HOLDABLE:
167 suffixFactor = 2;
168 break;
169 case INVTYPE_WEAPON:
170 case INVTYPE_WEAPONMAINHAND:
171 case INVTYPE_WEAPONOFFHAND:
172 suffixFactor = 3;
173 break;
174 case INVTYPE_RANGED:
175 case INVTYPE_THROWN:
176 case INVTYPE_RANGEDRIGHT:
177 suffixFactor = 4;
178 break;
179 default:
180 return 0;
181 }
182 // Select rare/epic modifier
183 switch (itemProto->Quality)
184 {
185 case ITEM_QUALITY_UNCOMMON:
186 return randomProperty->UncommonPropertiesPoints[suffixFactor];
187 case ITEM_QUALITY_RARE:
188 return randomProperty->RarePropertiesPoints[suffixFactor];
189 case ITEM_QUALITY_EPIC:
190 return randomProperty->EpicPropertiesPoints[suffixFactor];
191 case ITEM_QUALITY_LEGENDARY:
192 case ITEM_QUALITY_ARTIFACT:
193 return 0; // not have random properties
194 default:
195 break;
196 }
197 return 0;
198}

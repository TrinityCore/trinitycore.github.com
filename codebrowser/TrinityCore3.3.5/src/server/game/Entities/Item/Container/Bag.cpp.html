28Bag::Bag(): Item()
29{
30 m_objectType |= TYPEMASK_CONTAINER;
31 m_objectTypeId = TYPEID_CONTAINER;
32
33 m_valuesCount = CONTAINER_END;
34
35 memset(m_bagslot, 0, sizeof(Item*) * MAX_BAG_SIZE);
36}

38Bag::~Bag()
39{
40 for (uint8 i = 0; i < MAX_BAG_SIZE; ++i)
41 if (Item* item = m_bagslot[i])
42 {
43 if (item->IsInWorld())
44 {
45 TC_LOG_FATAL("entities.player.items", "Item %u (slot %u, bag slot %u) in bag %u (slot %u, bag slot %u, m_bagslot %u) is to be deleted but is still in world.",
46 item->GetEntry(), (uint32)item->GetSlot(), (uint32)item->GetBagSlot(),
47 GetEntry(), (uint32)GetSlot(), (uint32)GetBagSlot(), (uint32)i);
48 item->RemoveFromWorld();
49 }
50 delete m_bagslot[i];
51 }
52}

72bool Bag::Create(ObjectGuid::LowType guidlow, uint32 itemid, Player const* owner)
73{
74 ItemTemplate const* itemProto = sObjectMgr->GetItemTemplate(itemid);
75
76 if (!itemProto || itemProto->ContainerSlots > MAX_BAG_SIZE)
77 return false;
78
79 Object::_Create(guidlow, 0, HighGuid::Container);
80
81 SetEntry(itemid);
82 SetObjectScale(1.0f);
83
84 if (owner)
85 {
86 SetGuidValue(ITEM_FIELD_OWNER, owner->GetGUID());
87 SetGuidValue(ITEM_FIELD_CONTAINED, owner->GetGUID());
88 }
89
90 SetUInt32Value(ITEM_FIELD_MAXDURABILITY, itemProto->MaxDurability);
91 SetUInt32Value(ITEM_FIELD_DURABILITY, itemProto->MaxDurability);
92 SetUInt32Value(ITEM_FIELD_STACK_COUNT, 1);
93
94 // Setting the number of Slots the Container has
95 SetUInt32Value(CONTAINER_FIELD_NUM_SLOTS, itemProto->ContainerSlots);
96
97 // Cleaning 20 slots
98 for (uint8 i = 0; i < MAX_BAG_SIZE; ++i)
99 {
100 SetGuidValue(CONTAINER_FIELD_SLOT_1 + (i*2), ObjectGuid::Empty);
101 m_bagslot[i] = NULL;
102 }
103
104 return true;
105}

112bool Bag::LoadFromDB(ObjectGuid::LowType guid, ObjectGuid owner_guid, Field* fields, uint32 entry)
113{
114 if (!Item::LoadFromDB(guid, owner_guid, fields, entry))
115 return false;
116
117 ItemTemplate const* itemProto = GetTemplate(); // checked in Item::LoadFromDB
118 SetUInt32Value(CONTAINER_FIELD_NUM_SLOTS, itemProto->ContainerSlots);
119 // cleanup bag content related item value fields (its will be filled correctly from `character_inventory`)
120 for (uint8 i = 0; i < MAX_BAG_SIZE; ++i)
121 {
122 SetGuidValue(CONTAINER_FIELD_SLOT_1 + (i * 2), ObjectGuid::Empty);
123 delete m_bagslot[i];
124 m_bagslot[i] = NULL;
125 }
126
127 return true;
128}

149void Bag::RemoveItem(uint8 slot, bool /*update*/)
150{
151 ASSERT(slot < MAX_BAG_SIZE);
152
153 if (m_bagslot[slot])
154 m_bagslot[slot]->SetContainer(NULL);
155
156 m_bagslot[slot] = NULL;
157 SetGuidValue(CONTAINER_FIELD_SLOT_1 + (slot * 2), ObjectGuid::Empty);
158}
159
160void Bag::StoreItem(uint8 slot, Item* pItem, bool /*update*/)
161{
162 ASSERT(slot < MAX_BAG_SIZE);
163
164 if (pItem && pItem->GetGUID() != this->GetGUID())
165 {
166 m_bagslot[slot] = pItem;
167 SetGuidValue(CONTAINER_FIELD_SLOT_1 + (slot * 2), pItem->GetGUID());
168 pItem->SetGuidValue(ITEM_FIELD_CONTAINED, GetGUID());
169 pItem->SetGuidValue(ITEM_FIELD_OWNER, GetOwnerGUID());
170 pItem->SetContainer(this);
171 pItem->SetSlot(slot);
172 }
173}

194uint32 Bag::GetItemCount(uint32 item, Item* eItem) const
195{
196 Item* pItem;
197 uint32 count = 0;
198 for (uint32 i=0; i < GetBagSize(); ++i)
199 {
200 pItem = m_bagslot[i];
201 if (pItem && pItem != eItem && pItem->GetEntry() == item)
202 count += pItem->GetCount();
203 }
204
205 if (eItem && eItem->GetTemplate()->GemProperties)
206 {
207 for (uint32 i=0; i < GetBagSize(); ++i)
208 {
209 pItem = m_bagslot[i];
210 if (pItem && pItem != eItem && pItem->GetTemplate()->Socket[0].Color)
211 count += pItem->GetGemCountWithID(item);
212 }
213 }
214
215 return count;
216}

218uint32 Bag::GetItemCountWithLimitCategory(uint32 limitCategory, Item* skipItem) const
219{
220 uint32 count = 0;
221 for (uint32 i = 0; i < GetBagSize(); ++i)
222 if (Item* pItem = m_bagslot[i])
223 if (pItem != skipItem)
224 if (ItemTemplate const* pProto = pItem->GetTemplate())
225 if (pProto->ItemLimitCategory == limitCategory)
226 count += m_bagslot[i]->GetCount();
227
228 return count;
229}

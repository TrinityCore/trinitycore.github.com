28Totem::Totem(SummonPropertiesEntry const* properties, Unit* owner) : Minion(properties, owner, false)
29{
30 m_unitTypeMask |= UNIT_MASK_TOTEM;
31 m_duration = 0;
32 m_type = TOTEM_PASSIVE;
33}

35void Totem::Update(uint32 time)
36{
37 if (!GetOwner()->IsAlive() || !IsAlive())
38 {
39 UnSummon(); // remove self
40 return;
41 }
42
43 if (m_duration <= time)
44 {
45 UnSummon(); // remove self
46 return;
47 }
48 else
49 m_duration -= time;
50
51 Creature::Update(time);
52}

54void Totem::InitStats(uint32 duration)
55{
56 // client requires SMSG_TOTEM_CREATED to be sent before adding to world and before removing old totem
57 if (GetOwner()->GetTypeId() == TYPEID_PLAYER
58 && m_Properties->Slot >= SUMMON_SLOT_TOTEM
59 && m_Properties->Slot < MAX_TOTEM_SLOT)
60 {
61 WorldPacket data(SMSG_TOTEM_CREATED, 1 + 8 + 4 + 4);
62 data << uint8(m_Properties->Slot - 1);
63 data << uint64(GetGUID());
64 data << uint32(duration);
65 data << uint32(GetUInt32Value(UNIT_CREATED_BY_SPELL));
66 GetOwner()->ToPlayer()->SendDirectMessage(&data);
67
68 // set display id depending on caster's race
69 SetDisplayId(GetOwner()->GetModelForTotem(PlayerTotemType(m_Properties->Id)));
70 }
71
72 Minion::InitStats(duration);
73
74 // Get spell cast by totem
75 if (SpellInfo const* totemSpell = sSpellMgr->GetSpellInfo(GetSpell()))
76 if (totemSpell->CalcCastTime()) // If spell has cast time -> its an active totem
77 m_type = TOTEM_ACTIVE;
78
79 if (GetEntry() == SENTRY_TOTEM_ENTRY)
80 SetReactState(REACT_AGGRESSIVE);
81
82 m_duration = duration;
83
84 SetLevel(GetOwner()->getLevel());
85}

87void Totem::InitSummon()
88{
89 if (m_type == TOTEM_PASSIVE && GetSpell())
90 {
91 CastSpell(this, GetSpell(), true);
92 }
93
94 // Some totems can have both instant effect and passive spell
95 if (GetSpell(1))
96 CastSpell(this, GetSpell(1), true);
97}

99void Totem::UnSummon(uint32 msTime)
100{
101 if (msTime)
102 {
103 m_Events.AddEvent(new ForcedUnsummonDelayEvent(*this), m_Events.CalculateTime(msTime));
104 return;
105 }
106
107 CombatStop();
108 RemoveAurasDueToSpell(GetSpell(), GetGUID());
109
110 // clear owner's totem slot
111 for (uint8 i = SUMMON_SLOT_TOTEM; i < MAX_TOTEM_SLOT; ++i)
112 {
113 if (GetOwner()->m_SummonSlot[i] == GetGUID())
114 {
115 GetOwner()->m_SummonSlot[i].Clear();
116 break;
117 }
118 }
119
120 GetOwner()->RemoveAurasDueToSpell(GetSpell(), GetGUID());
121
122 // Remove Sentry Totem Aura
123 if (GetEntry() == SENTRY_TOTEM_ENTRY)
124 GetOwner()->RemoveAurasDueToSpell(SENTRY_TOTEM_SPELLID);
125
126 //remove aura all party members too
127 if (Player* owner = GetOwner()->ToPlayer())
128 {
129 owner->SendAutoRepeatCancel(this);
130
131 if (SpellInfo const* spell = sSpellMgr->GetSpellInfo(GetUInt32Value(UNIT_CREATED_BY_SPELL)))
132 GetSpellHistory()->SendCooldownEvent(spell, 0, nullptr, false);
133
134 if (Group* group = owner->GetGroup())
135 {
136 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
137 {
138 Player* target = itr->GetSource();
139 if (target && group->SameSubGroup(owner, target))
140 target->RemoveAurasDueToSpell(GetSpell(), GetGUID());
141 }
142 }
143 }
144
145 // any totem unsummon look like as totem kill, req. for proper animation
146 if (IsAlive())
147 setDeathState(DEAD);
148
149 AddObjectToRemoveList();
150}

152bool Totem::IsImmunedToSpellEffect(SpellInfo const* spellInfo, uint32 index) const
153{
154 /// @todo possibly all negative auras immune?
155 if (GetEntry() == 5925)
156 return false;
157
158 switch (spellInfo->Effects[index].ApplyAuraName)
159 {
160 case SPELL_AURA_PERIODIC_DAMAGE:
161 case SPELL_AURA_PERIODIC_LEECH:
162 case SPELL_AURA_MOD_FEAR:
163 case SPELL_AURA_TRANSFORM:
164 return true;
165 default:
166 break;
167 }
168
169 return Creature::IsImmunedToSpellEffect(spellInfo, index);
170}

33Vehicle::Vehicle(Unit* unit, VehicleEntry const* vehInfo, uint32 creatureEntry) :
34UsableSeatNum(0), _me(unit), _vehicleInfo(vehInfo), _creatureEntry(creatureEntry), _status(STATUS_NONE), _lastShootPos()
35{
36 for (uint32 i = 0; i < MAX_VEHICLE_SEATS; ++i)
37 {
38 if (uint32 seatId = _vehicleInfo->m_seatID[i])
39 if (VehicleSeatEntry const* veSeat = sVehicleSeatStore.LookupEntry(seatId))
40 {
41 Seats.insert(std::make_pair(i, VehicleSeat(veSeat)));
42 if (veSeat->CanEnterOrExit())
43 ++UsableSeatNum;
44 }
45 }
46
47 // Set or remove correct flags based on available seats. Will overwrite db data (if wrong).
48 if (UsableSeatNum)
49 _me->SetFlag(UNIT_NPC_FLAGS, (_me->GetTypeId() == TYPEID_PLAYER ? UNIT_NPC_FLAG_PLAYER_VEHICLE : UNIT_NPC_FLAG_SPELLCLICK));
50 else
51 _me->RemoveFlag(UNIT_NPC_FLAGS, (_me->GetTypeId() == TYPEID_PLAYER ? UNIT_NPC_FLAG_PLAYER_VEHICLE : UNIT_NPC_FLAG_SPELLCLICK));
52
53 InitMovementInfoForBase();
54}

56Vehicle::~Vehicle()
57{
58 /// @Uninstall must be called before this.
59 ASSERT(_status == STATUS_UNINSTALLING);
60 for (SeatMap::const_iterator itr = Seats.begin(); itr != Seats.end(); ++itr)
61 ASSERT(itr->second.IsEmpty());
62}

88void Vehicle::InstallAllAccessories(bool evading)
89{
90 if (GetBase()->GetTypeId() == TYPEID_PLAYER || !evading)
91 RemoveAllPassengers(); // We might have aura's saved in the DB with now invalid casters - remove
92
93 VehicleAccessoryList const* accessories = sObjectMgr->GetVehicleAccessoryList(this);
94 if (!accessories)
95 return;
96
97 for (VehicleAccessoryList::const_iterator itr = accessories->begin(); itr != accessories->end(); ++itr)
98 if (!evading || itr->IsMinion) // only install minions on evade mode
99 InstallAccessory(itr->AccessoryEntry, itr->SeatId, itr->IsMinion, itr->SummonedType, itr->SummonTime);
100}

102/**
103 * @fn void Vehicle::Uninstall()
104 *
105 * @brief Removes all passengers and sets status to STATUS_UNINSTALLING.
106 * No new passengers can be added to the vehicle after this call.
107 *
108 * @author Machiavelli
109 * @date 17-2-2013
110 */
111
112void Vehicle::Uninstall()
113{
114 /// @Prevent recursive uninstall call. (Bad script in OnUninstall/OnRemovePassenger/PassengerBoarded hook.)
115 if (_status == STATUS_UNINSTALLING && !GetBase()->HasUnitTypeMask(UNIT_MASK_MINION))
116 {
117 TC_LOG_ERROR("entities.vehicle", "Vehicle GuidLow: %u, Entry: %u attempts to uninstall, but already has STATUS_UNINSTALLING! "
118 "Check Uninstall/PassengerBoarded script hooks for errors.", _me->GetGUID().GetCounter(), _me->GetEntry());
119 return;
120 }
121
122 _status = STATUS_UNINSTALLING;
123 TC_LOG_DEBUG("entities.vehicle", "Vehicle::Uninstall Entry: %u, GuidLow: %u", _creatureEntry, _me->GetGUID().GetCounter());
124 RemoveAllPassengers();
125
126 if (GetBase()->GetTypeId() == TYPEID_UNIT)
127 sScriptMgr->OnUninstall(this);
128}

172 if (_me->ToCreature() && _me->ToCreature()->GetCreatureTemplate()->type == CREATURE_TYPE_MECHANICAL && !_me->ToCreature()->isWorldBoss())
173 {
174 // Heal & dispel ...
175 _me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_HEAL, true);

178 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_PERIODIC_HEAL, true);
179
180 // ... Shield & Immunity grant spells ...
181 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_SCHOOL_IMMUNITY, true);
182 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_UNATTACKABLE, true);
183 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_SCHOOL_ABSORB, true);
184 _me->ApplySpellImmune(0, IMMUNITY_MECHANIC, MECHANIC_SHIELD, true);
185 _me->ApplySpellImmune(0, IMMUNITY_MECHANIC, MECHANIC_IMMUNE_SHIELD, true);
186
187 // ... Resistance, Split damage, Change stats ...
188 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_DAMAGE_SHIELD, true);
189 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_SPLIT_DAMAGE_PCT, true);
190 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_RESISTANCE, true);
191 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_STAT, true);
192 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_DAMAGE_PERCENT_TAKEN, true);
193 }
194
195 // If vehicle flag for fixed position set (cannons), or if the following hardcoded units, then set state rooted
196 // 30236 | Argent Cannon
197 // 39759 | Tankbuster Cannon
198 if ((GetVehicleInfo()->m_flags & VEHICLE_FLAG_FIXED_POSITION) || GetBase()->GetEntry() == 30236 || GetBase()->GetEntry() == 39759)
199 _me->SetControlled(true, UNIT_STATE_ROOT);

202 switch (GetVehicleInfo()->m_ID)
203 {
204 // code below prevents a bug with movable cannons
205 case 160: // Strand of the Ancients
206 case 244: // Wintergrasp
207 case 510: // Isle of Conquest
208 case 452: // Isle of Conquest
209 case 543: // Isle of Conquest
210 _me->SetControlled(true, UNIT_STATE_ROOT);
211 // why we need to apply this? we can simple add immunities to slow mechanic in DB
212 _me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_DECREASE_SPEED, true);
213 break;
214 default:
215 break;
216 }

220 * @fn void Vehicle::RemoveAllPassengers()
221 *
222 * @brief Removes all current and pending passengers from the vehicle.
223 *
224 * @author Machiavelli
225 * @date 17-2-2013
226 */
227
228void Vehicle::RemoveAllPassengers()
229{
230 TC_LOG_DEBUG("entities.vehicle", "Vehicle::RemoveAllPassengers. Entry: %u, GuidLow: %u", _creatureEntry, _me->GetGUID().GetCounter());
231
232 /// Setting to_Abort to true will cause @VehicleJoinEvent::Abort to be executed on next @Unit::UpdateEvents call
233 /// This will properly "reset" the pending join process for the passenger.
234 {
235 /// Update vehicle pointer in every pending join event - Abort may be called after vehicle is deleted
236 Vehicle* eventVehicle = _status != STATUS_UNINSTALLING ? this : NULL;
237
238 while (!_pendingJoinEvents.empty())
239 {
240 VehicleJoinEvent* e = _pendingJoinEvents.front();
241 e->to_Abort = true;
242 e->Target = eventVehicle;
243 _pendingJoinEvents.pop_front();
244 }
245 }
246
247 // Passengers always cast an aura with SPELL_AURA_CONTROL_VEHICLE on the vehicle
248 // We just remove the aura and the unapply handler will make the target leave the vehicle.
249 // We don't need to iterate over Seats
250 _me->RemoveAurasByType(SPELL_AURA_CONTROL_VEHICLE);
251
252 // Following the above logic, this assertion should NEVER fail.
253 // Even in 'hacky' cases, there should at least be VEHICLE_SPELL_RIDE_HARDCODED on us.
254 // SeatMap::const_iterator itr;
255 // for (itr = Seats.begin(); itr != Seats.end(); ++itr)
256 // ASSERT(!itr->second.passenger);

302/**
303 * @fn SeatMap::const_iterator Vehicle::GetNextEmptySeat(int8 seatId, bool next) const
304 *
305 * @brief Gets the next empty seat based on current seat.
306 *
307 * @author Machiavelli
308 * @date 17-2-2013
309 *
310 * @param seatId Identifier for the current seat.
311 * @param next true if iterating forward, false means iterating backwards.
312 *
313 * @return The next empty seat.
314 */
315
316SeatMap::const_iterator Vehicle::GetNextEmptySeat(int8 seatId, bool next) const
317{
318 SeatMap::const_iterator seat = Seats.find(seatId);
319 if (seat == Seats.end())
320 return seat;
321
322 while (!seat->second.IsEmpty() || (!seat->second.SeatInfo->CanEnterOrExit() && !seat->second.SeatInfo->IsUsableByOverride()))
323 {
324 if (next)
325 {
326 if (++seat == Seats.end())
327 seat = Seats.begin();
328 }
329 else
330 {
331 if (seat == Seats.begin())
332 seat = Seats.end();
333 --seat;
334 }
335
336 // Make sure we don't loop indefinetly
337 if (seat->first == seatId)
338 return Seats.end();
339 }
340
341 return seat;
342}

344/**
345 * @fn void Vehicle::InstallAccessory(uint32 entry, int8 seatId, bool minion, uint8 type,
346 * uint32 summonTime)
347 *
348 * @brief Installs an accessory.
349 *
350 * @author Machiavelli
351 * @date 17-2-2013
352 *
353 * @param entry The NPC entry of accessory.
354 * @param seatId Identifier for the seat to add the accessory to.
355 * @param minion true if accessory considered a 'minion'. Implies that the accessory will despawn when the vehicle despawns.
356 * Essentially that it has no life without the vehicle. Their fates are bound.
357 * @param type See enum @SummonType.
358 * @param summonTime Time after which the minion is despawned in case of a timed despawn @type specified.
359 */
360
361void Vehicle::InstallAccessory(uint32 entry, int8 seatId, bool minion, uint8 type, uint32 summonTime)
362{
363 /// @Prevent adding accessories when vehicle is uninstalling. (Bad script in OnUninstall/OnRemovePassenger/PassengerBoarded hook.)
364 if (_status == STATUS_UNINSTALLING)
365 {
366 TC_LOG_ERROR("entities.vehicle", "Vehicle (GuidLow: %u, DB GUID: %u, Entry: %u) attempts to install accessory (Entry: %u) on seat %d with STATUS_UNINSTALLING! "
367 "Check Uninstall/PassengerBoarded script hooks for errors.", _me->GetGUID().GetCounter(),
368 (_me->GetTypeId() == TYPEID_UNIT ? _me->ToCreature()->GetSpawnId() : _me->GetGUID().GetCounter()), GetCreatureEntry(), entry, (int32)seatId);
369 return;
370 }
371
372 TC_LOG_DEBUG("entities.vehicle", "Vehicle (GuidLow: %u, DB Guid: %u, Entry %u): installing accessory (Entry: %u) on seat: %d",
373 _me->GetGUID().GetCounter(), (_me->GetTypeId() == TYPEID_UNIT ? _me->ToCreature()->GetSpawnId() : _me->GetGUID().GetCounter()), GetCreatureEntry(),
374 entry, (int32)seatId);
375
376 TempSummon* accessory = _me->SummonCreature(entry, *_me, TempSummonType(type), summonTime);
377 ASSERT(accessory);
378
379 if (minion)
380 accessory->AddUnitTypeMask(UNIT_MASK_ACCESSORY);
381
382 (void)_me->HandleSpellClick(accessory, seatId);
383
384 /// If for some reason adding accessory to vehicle fails it will unsummon in
385 /// @VehicleJoinEvent::Abort
386}

388/**
389 * @fn bool Vehicle::AddPassenger(Unit* unit, int8 seatId)
390 *
391 * @brief Attempts to add a passenger to the vehicle on 'seatId'.
392 *
393 * @author Machiavelli
394 * @date 17-2-2013
395 *
396 * @param [in, out] The prospective passenger.
397 * @param seatId Identifier for the seat. Value of -1 indicates the next available seat.
398 *
399 * @return true if it succeeds, false if it fails.
400 */
401
402bool Vehicle::AddPassenger(Unit* unit, int8 seatId)
403{
404 /// @Prevent adding passengers when vehicle is uninstalling. (Bad script in OnUninstall/OnRemovePassenger/PassengerBoarded hook.)
405 if (_status == STATUS_UNINSTALLING)
406 {
407 TC_LOG_ERROR("entities.vehicle", "Passenger GuidLow: %u, Entry: %u, attempting to board vehicle GuidLow: %u, Entry: %u during uninstall! SeatId: %d",
408 unit->GetGUID().GetCounter(), unit->GetEntry(), _me->GetGUID().GetCounter(), _me->GetEntry(), (int32)seatId);
409 return false;
410 }
411
412 TC_LOG_DEBUG("entities.vehicle", "Unit %s scheduling enter vehicle (entry: %u, vehicleId: %u, guid: %u (dbguid: %u) on seat %d",
413 unit->GetName().c_str(), _me->GetEntry(), _vehicleInfo->m_ID, _me->GetGUID().GetCounter(),
414 (_me->GetTypeId() == TYPEID_UNIT ? _me->ToCreature()->GetSpawnId() : 0), (int32)seatId);
415
416 // The seat selection code may kick other passengers off the vehicle.
417 // While the validity of the following may be arguable, it is possible that when such a passenger
418 // exits the vehicle will dismiss. That's why the actual adding the passenger to the vehicle is scheduled
419 // asynchronously, so it can be cancelled easily in case the vehicle is uninstalled meanwhile.
420 SeatMap::iterator seat;
421 VehicleJoinEvent* e = new VehicleJoinEvent(this, unit);
422 unit->m_Events.AddEvent(e, unit->m_Events.CalculateTime(0));
423
424 if (seatId < 0) // no specific seat requirement
425 {
426 for (seat = Seats.begin(); seat != Seats.end(); ++seat)
427 if (seat->second.IsEmpty() && (seat->second.SeatInfo->CanEnterOrExit() || seat->second.SeatInfo->IsUsableByOverride()))
428 break;
429
430 if (seat == Seats.end()) // no available seat
431 {
432 e->to_Abort = true;
433 return false;
434 }
435
436 e->Seat = seat;
437 _pendingJoinEvents.push_back(e);
438 }
439 else
440 {
441 seat = Seats.find(seatId);
442 if (seat == Seats.end())
443 {
444 e->to_Abort = true;
445 return false;
446 }
447
448 e->Seat = seat;
449 _pendingJoinEvents.push_back(e);
450 if (!seat->second.IsEmpty())
451 {
452 Unit* passenger = ObjectAccessor::GetUnit(*GetBase(), seat->second.Passenger.Guid);
453 ASSERT(passenger);
454 passenger->ExitVehicle();
455 }
456
457 ASSERT(seat->second.IsEmpty());
458 }
459
460 return true;
461}

463/**
464 * @fn void Vehicle::RemovePassenger(Unit* unit)
465 *
466 * @brief Removes the passenger from the vehicle.
467 *
468 * @author Machiavelli
469 * @date 17-2-2013
470 *
471 * @param [in, out] unit The passenger to remove.
472 */
473
474Vehicle* Vehicle::RemovePassenger(Unit* unit)
475{
476 if (unit->GetVehicle() != this)
477 return NULL;
478
479 SeatMap::iterator seat = GetSeatIteratorForPassenger(unit);
480 ASSERT(seat != Seats.end());
481
482 TC_LOG_DEBUG("entities.vehicle", "Unit %s exit vehicle entry %u id %u dbguid %u seat %d",
483 unit->GetName().c_str(), _me->GetEntry(), _vehicleInfo->m_ID, _me->GetGUID().GetCounter(), (int32)seat->first);
484
485 if (seat->second.SeatInfo->CanEnterOrExit() && ++UsableSeatNum)
486 _me->SetFlag(UNIT_NPC_FLAGS, (_me->GetTypeId() == TYPEID_PLAYER ? UNIT_NPC_FLAG_PLAYER_VEHICLE : UNIT_NPC_FLAG_SPELLCLICK));
487
488 // Remove UNIT_FLAG_NOT_SELECTABLE if passenger did not have it before entering vehicle
489 if (seat->second.SeatInfo->m_flags & VEHICLE_SEAT_FLAG_PASSENGER_NOT_SELECTABLE && !seat->second.Passenger.IsUnselectable)
490 unit->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
491
492 seat->second.Passenger.Reset();
493
494 if (_me->GetTypeId() == TYPEID_UNIT && unit->GetTypeId() == TYPEID_PLAYER && seat->second.SeatInfo->m_flags & VEHICLE_SEAT_FLAG_CAN_CONTROL)
495 _me->RemoveCharmedBy(unit);
496
497 if (_me->IsInWorld())
498 {
499 if (!_me->GetTransport())
500 {
501 unit->RemoveUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT);
502 unit->m_movementInfo.transport.Reset();
503 }
504 else
505 unit->m_movementInfo.transport = _me->m_movementInfo.transport;
506 }
507
508 // only for flyable vehicles
509 if (unit->IsFlying())
510 _me->CastSpell(unit, VEHICLE_SPELL_PARACHUTE, true);
511
512 if (_me->GetTypeId() == TYPEID_UNIT && _me->ToCreature()->IsAIEnabled)
513 _me->ToCreature()->AI()->PassengerBoarded(unit, seat->first, false);
514
515 if (GetBase()->GetTypeId() == TYPEID_UNIT)
516 sScriptMgr->OnRemovePassenger(this, unit);
517
518 unit->SetVehicle(NULL);
519 return this;
520}

523 * @fn void Vehicle::RelocatePassengers()
524 *
525 * @brief Relocate passengers. Must be called after m_base::Relocate
526 *
527 * @author Machiavelli
528 * @date 17-2-2013
529 */
530
531void Vehicle::RelocatePassengers()
532{
533 ASSERT(_me->GetMap());
534
535 // not sure that absolute position calculation is correct, it must depend on vehicle pitch angle
536 for (SeatMap::const_iterator itr = Seats.begin(); itr != Seats.end(); ++itr)
537 {
538 if (Unit* passenger = ObjectAccessor::GetUnit(*GetBase(), itr->second.Passenger.Guid))
539 {
540 ASSERT(passenger->IsInWorld());
541
542 float px, py, pz, po;
543 passenger->m_movementInfo.transport.pos.GetPosition(px, py, pz, po);
544 CalculatePassengerPosition(px, py, pz, &po);
545 passenger->UpdatePosition(px, py, pz, po);
546 }
547 }
548}

628 */

631{
632 SeatMap::iterator itr;
633 for (itr = Seats.begin(); itr != Seats.end(); ++itr)
634 if (itr->second.Passenger.Guid == passenger->GetGUID())

637 return Seats.end();

687/**
688 * @fn void Vehicle::RemovePendingEventsForSeat(uint8 seatId)
689 *
690 * @brief Removes any pending events for given seatId. Executed when a @VehicleJoinEvent::Execute is called
691 *
692 * @author Machiavelli
693 * @date 23-2-2013
694 *
695 * @param seatId Identifier for the seat.
696 */

698void Vehicle::RemovePendingEventsForSeat(int8 seatId)
699{
700 for (PendingJoinEventContainer::iterator itr = _pendingJoinEvents.begin(); itr != _pendingJoinEvents.end();)
701 {
702 if ((*itr)->Seat->first == seatId)
703 {
704 (*itr)->to_Abort = true;
705 _pendingJoinEvents.erase(itr++);
706 }
707 else
708 ++itr;
709 }

743/**
744 * @fn bool VehicleJoinEvent::Execute(uint64, uint32)
745 *
746 * @brief Actually adds the passenger @Passenger to vehicle @Target.
747 *
748 * @author Machiavelli
749 * @date 17-2-2013
750 *
751 * @param parameter1 Unused
752 * @param parameter2 Unused.
753 *
754 * @return true, cannot fail.
755 *
756 */
757
758bool VehicleJoinEvent::Execute(uint64, uint32)
759{
760 ASSERT(Passenger->IsInWorld());
761 ASSERT(Target && Target->GetBase()->IsInWorld());
762 ASSERT(Target->GetBase()->HasAuraTypeWithCaster(SPELL_AURA_CONTROL_VEHICLE, Passenger->GetGUID()));
763
764 Target->RemovePendingEventsForSeat(Seat->first);
765 Target->RemovePendingEventsForPassenger(Passenger);
766
767 Passenger->SetVehicle(Target);
768 Seat->second.Passenger.Guid = Passenger->GetGUID();
769 Seat->second.Passenger.IsUnselectable = Passenger->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
770 if (Seat->second.SeatInfo->CanEnterOrExit())
771 {
772 ASSERT(Target->UsableSeatNum);
773 --(Target->UsableSeatNum);
774 if (!Target->UsableSeatNum)
775 {
776 if (Target->GetBase()->GetTypeId() == TYPEID_PLAYER)
777 Target->GetBase()->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_PLAYER_VEHICLE);
778 else
779 Target->GetBase()->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_SPELLCLICK);
780 }
781 }
782
783 Passenger->InterruptNonMeleeSpells(false);
784 Passenger->RemoveAurasByType(SPELL_AURA_MOUNTED);
785
786 VehicleSeatEntry const* veSeat = Seat->second.SeatInfo;
787
788 Player* player = Passenger->ToPlayer();
789 if (player)
790 {
791 // drop flag
792 if (Battleground* bg = player->GetBattleground())
793 bg->EventPlayerDroppedFlag(player);
794
795 player->StopCastingCharm();
796 player->StopCastingBindSight();
797 player->SendOnCancelExpectedVehicleRideAura();
798 if (!(veSeat->m_flagsB & VEHICLE_SEAT_FLAG_B_KEEP_PET))
799 player->UnsummonPetTemporaryIfAny();
800 }
801
802 if (Seat->second.SeatInfo->m_flags & VEHICLE_SEAT_FLAG_PASSENGER_NOT_SELECTABLE)
803 Passenger->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
804
805 Passenger->AddUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT);
806 Passenger->m_movementInfo.transport.pos.Relocate(veSeat->m_attachmentOffsetX, veSeat->m_attachmentOffsetY, veSeat->m_attachmentOffsetZ);
807 Passenger->m_movementInfo.transport.time = 0;
808 Passenger->m_movementInfo.transport.seat = Seat->first;
809 Passenger->m_movementInfo.transport.guid = Target->GetBase()->GetGUID();
810
811 if (Target->GetBase()->GetTypeId() == TYPEID_UNIT && Passenger->GetTypeId() == TYPEID_PLAYER &&
812 Seat->second.SeatInfo->m_flags & VEHICLE_SEAT_FLAG_CAN_CONTROL)
813 ASSERT(Target->GetBase()->SetCharmedBy(Passenger, CHARM_TYPE_VEHICLE)); // SMSG_CLIENT_CONTROL
814
815 Passenger->SendClearTarget(); // SMSG_BREAK_TARGET
816 Passenger->SetControlled(true, UNIT_STATE_ROOT); // SMSG_FORCE_ROOT - In some cases we send SMSG_SPLINE_MOVE_ROOT here (for creatures)
817 // also adds MOVEMENTFLAG_ROOT
818
819 Movement::MoveSplineInit init(Passenger);
820 init.DisableTransportPathTransformations();
821 init.MoveTo(veSeat->m_attachmentOffsetX, veSeat->m_attachmentOffsetY, veSeat->m_attachmentOffsetZ, false, true);
822 init.SetFacing(0.0f);
823 init.SetTransportEnter();
824 init.Launch();
825
826 if (Creature* creature = Target->GetBase()->ToCreature())
827 {
828 if (creature->IsAIEnabled)
829 creature->AI()->PassengerBoarded(Passenger, Seat->first, true);
830
831 sScriptMgr->OnAddPassenger(Target, Passenger, Seat->first);
832
833 // Actually quite a redundant hook. Could just use OnAddPassenger and check for unit typemask inside script.
834 if (Passenger->HasUnitTypeMask(UNIT_MASK_ACCESSORY))
835 sScriptMgr->OnInstallAccessory(Target, Passenger->ToCreature());
836 }
837
838 return true;
839}

841/**
842 * @fn void VehicleJoinEvent::Abort(uint64)
843 *
844 * @brief Aborts the event. Implies that unit @Passenger will not be boarding vehicle @Target after all.
845 *
846 * @author Machiavelli
847 * @date 17-2-2013
848 *
849 * @param parameter1 Unused
850 */
851
852void VehicleJoinEvent::Abort(uint64)
853{
854 /// Check if the Vehicle was already uninstalled, in which case all auras were removed already
855 if (Target)
856 {
857 TC_LOG_DEBUG("entities.vehicle", "Passenger GuidLow: %u, Entry: %u, board on vehicle GuidLow: %u, Entry: %u SeatId: %d cancelled",
858 Passenger->GetGUID().GetCounter(), Passenger->GetEntry(), Target->GetBase()->GetGUID().GetCounter(), Target->GetBase()->GetEntry(), (int32)Seat->first);
859
860 /// @SPELL_AURA_CONTROL_VEHICLE auras can be applied even when the passenger is not (yet) on the vehicle.
861 /// When this code is triggered it means that something went wrong in @Vehicle::AddPassenger, and we should remove
862 /// the aura manually.
863 Target->GetBase()->RemoveAurasByType(SPELL_AURA_CONTROL_VEHICLE, Passenger->GetGUID());
864 }
865 else
866 TC_LOG_DEBUG("entities.vehicle", "Passenger GuidLow: %u, Entry: %u, board on uninstalled vehicle SeatId: %d cancelled",
867 Passenger->GetGUID().GetCounter(), Passenger->GetEntry(), (int32)Seat->first);
868
869 if (Passenger->IsInWorld() && Passenger->HasUnitTypeMask(UNIT_MASK_ACCESSORY))
870 Passenger->ToCreature()->DespawnOrUnsummon();
871}

97class TransportBase
98{
99protected:
100 TransportBase() { }
101 virtual ~TransportBase() { }
102
103public:
104 /// This method transforms supplied transport offsets into global coordinates
105 virtual void CalculatePassengerPosition(float& x, float& y, float& z, float* o = NULL) const = 0;
106
107 /// This method transforms supplied global coordinates into local offsets
108 virtual void CalculatePassengerOffset(float& x, float& y, float& z, float* o = NULL) const = 0;
109
110protected:
111 static void CalculatePassengerPosition(float& x, float& y, float& z, float* o, float transX, float transY, float transZ, float transO)
112 {
113 float inx = x, iny = y, inz = z;
114 if (o)
115 *o = Position::NormalizeOrientation(transO + *o);
116
117 x = transX + inx * std::cos(transO) - iny * std::sin(transO);
118 y = transY + iny * std::cos(transO) + inx * std::sin(transO);
119 z = transZ + inz;
120 }
121
122 static void CalculatePassengerOffset(float& x, float& y, float& z, float* o, float transX, float transY, float transZ, float transO)
123 {
124 if (o)
125 *o = Position::NormalizeOrientation(*o - transO);
126
127 z -= transZ;
128 y -= transY; // y = searchedY * std::cos(o) + searchedX * std::sin(o)
129 x -= transX; // x = searchedX * std::cos(o) + searchedY * std::sin(o + pi)
130 float inx = x, iny = y;
131 y = (iny - inx * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
132 x = (inx + iny * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
133 }
134};

54enum class HighGuid
55{
56 Item = 0x4000, // blizz 4000
57 Container = 0x4000, // blizz 4000
58 Player = 0x0000, // blizz 0000
59 GameObject = 0xF110, // blizz F110
60 Transport = 0xF120, // blizz F120 (for GAMEOBJECT_TYPE_TRANSPORT)
61 Unit = 0xF130, // blizz F130
62 Pet = 0xF140, // blizz F140
63 Vehicle = 0xF150, // blizz F550
64 DynamicObject = 0xF100, // blizz F100
65 Corpse = 0xF101, // blizz F100
66 Mo_Transport = 0x1FC0, // blizz 1FC0 (for GAMEOBJECT_TYPE_MO_TRANSPORT)
67 Instance = 0x1F40, // blizz 1F40
68 Group = 0x1F50,
69};

114class ObjectGuid
115{
116 public:
117 static ObjectGuid const Empty;
118
119 typedef uint32 LowType;
120
121 template<HighGuid type>
122 static typename std::enable_if<ObjectGuidTraits<type>::Global, ObjectGuid>::type Create(LowType counter) { return Global(type, counter); }
123
124 template<HighGuid type>
125 static typename std::enable_if<ObjectGuidTraits<type>::MapSpecific, ObjectGuid>::type Create(uint32 entry, LowType counter) { return MapSpecific(type, entry, counter); }
126
127 ObjectGuid() : _guid(0) { }
128 explicit ObjectGuid(uint64 guid) : _guid(guid) { }
129 ObjectGuid(HighGuid hi, uint32 entry, LowType counter) : _guid(counter ? uint64(counter) | (uint64(entry) << 24) | (uint64(hi) << 48) : 0) { }
130 ObjectGuid(HighGuid hi, LowType counter) : _guid(counter ? uint64(counter) | (uint64(hi) << 48) : 0) { }
131
132 operator uint64() const { return _guid; }
133 PackedGuidReader ReadAsPacked() { return PackedGuidReader(*this); }
134
135 void Set(uint64 guid) { _guid = guid; }
136 void Clear() { _guid = 0; }
137
138 PackedGuid WriteAsPacked() const;
139
140 uint64 GetRawValue() const { return _guid; }
141 HighGuid GetHigh() const { return HighGuid((_guid >> 48) & 0x0000FFFF); }
142 uint32 GetEntry() const { return HasEntry() ? uint32((_guid >> 24) & UI64LIT(0x0000000000FFFFFF)) : 0; }
143 LowType GetCounter() const
144 {
145 return HasEntry()
146 ? LowType(_guid & UI64LIT(0x0000000000FFFFFF))
147 : LowType(_guid & UI64LIT(0x00000000FFFFFFFF));
148 }
149
150 static LowType GetMaxCounter(HighGuid high)
151 {
152 return HasEntry(high)
153 ? LowType(0x00FFFFFF)
154 : LowType(0xFFFFFFFF);
155 }
156
157 ObjectGuid::LowType GetMaxCounter() const { return GetMaxCounter(GetHigh()); }
158
159 bool IsEmpty() const { return _guid == 0; }
160 bool IsCreature() const { return GetHigh() == HighGuid::Unit; }
161 bool IsPet() const { return GetHigh() == HighGuid::Pet; }
162 bool IsVehicle() const { return GetHigh() == HighGuid::Vehicle; }
163 bool IsCreatureOrPet() const { return IsCreature() || IsPet(); }
164 bool IsCreatureOrVehicle() const { return IsCreature() || IsVehicle(); }
165 bool IsAnyTypeCreature() const { return IsCreature() || IsPet() || IsVehicle(); }
166 bool IsPlayer() const { return !IsEmpty() && GetHigh() == HighGuid::Player; }
167 bool IsUnit() const { return IsAnyTypeCreature() || IsPlayer(); }
168 bool IsItem() const { return GetHigh() == HighGuid::Item; }
169 bool IsGameObject() const { return GetHigh() == HighGuid::GameObject; }
170 bool IsDynamicObject() const { return GetHigh() == HighGuid::DynamicObject; }
171 bool IsCorpse() const { return GetHigh() == HighGuid::Corpse; }
172 bool IsTransport() const { return GetHigh() == HighGuid::Transport; }
173 bool IsMOTransport() const { return GetHigh() == HighGuid::Mo_Transport; }
174 bool IsAnyTypeGameObject() const { return IsGameObject() || IsTransport() || IsMOTransport(); }
175 bool IsInstance() const { return GetHigh() == HighGuid::Instance; }
176 bool IsGroup() const { return GetHigh() == HighGuid::Group; }
177
178 static TypeID GetTypeId(HighGuid high)
179 {
180 switch (high)
181 {
182 case HighGuid::Item: return TYPEID_ITEM;
183 //case HighGuid::Container: return TYPEID_CONTAINER; HighGuid::Container==HighGuid::Item currently
184 case HighGuid::Unit: return TYPEID_UNIT;
185 case HighGuid::Pet: return TYPEID_UNIT;
186 case HighGuid::Player: return TYPEID_PLAYER;
187 case HighGuid::GameObject: return TYPEID_GAMEOBJECT;
188 case HighGuid::DynamicObject: return TYPEID_DYNAMICOBJECT;
189 case HighGuid::Corpse: return TYPEID_CORPSE;
190 case HighGuid::Mo_Transport: return TYPEID_GAMEOBJECT;
191 case HighGuid::Vehicle: return TYPEID_UNIT;
192 // unknown
193 case HighGuid::Instance:
194 case HighGuid::Group:
195 default: return TYPEID_OBJECT;
196 }
197 }
198
199 TypeID GetTypeId() const { return GetTypeId(GetHigh()); }
200
201 bool operator!() const { return IsEmpty(); }
202 bool operator== (ObjectGuid const& guid) const { return GetRawValue() == guid.GetRawValue(); }
203 bool operator!= (ObjectGuid const& guid) const { return GetRawValue() != guid.GetRawValue(); }
204 bool operator< (ObjectGuid const& guid) const { return GetRawValue() < guid.GetRawValue(); }
205
206 static char const* GetTypeName(HighGuid high);
207 char const* GetTypeName() const { return !IsEmpty() ? GetTypeName(GetHigh()) : "None"; }
208 std::string ToString() const;
209
210 private:
211 static bool HasEntry(HighGuid high)
212 {
213 switch (high)
214 {
215 case HighGuid::Item:
216 case HighGuid::Player:
217 case HighGuid::DynamicObject:
218 case HighGuid::Corpse:
219 case HighGuid::Mo_Transport:
220 case HighGuid::Instance:
221 case HighGuid::Group:
222 return false;
223 case HighGuid::GameObject:
224 case HighGuid::Transport:
225 case HighGuid::Unit:
226 case HighGuid::Pet:
227 case HighGuid::Vehicle:
228 default:
229 return true;
230 }
231 }
232
233 bool HasEntry() const { return HasEntry(GetHigh()); }
234
235 static ObjectGuid Global(HighGuid type, LowType counter);
236 static ObjectGuid MapSpecific(HighGuid type, uint32 entry, LowType counter);
237
238 explicit ObjectGuid(uint32 const&) = delete; // no implementation, used to catch wrong type assignment
239 ObjectGuid(HighGuid, uint32, uint64 counter) = delete; // no implementation, used to catch wrong type assignment
240 ObjectGuid(HighGuid, uint64 counter) = delete; // no implementation, used to catch wrong type assignment
241
242 uint64 _guid;
243};

252// minimum buffer size for packed guid is 9 bytes
253#define PACKED_GUID_MIN_BUFFER_SIZE 9
254
255class PackedGuid
256{
257 friend ByteBuffer& operator<<(ByteBuffer& buf, PackedGuid const& guid);
258
259 public:
260 explicit PackedGuid() : _packedGuid(PACKED_GUID_MIN_BUFFER_SIZE) { _packedGuid.appendPackGUID(0); }
261 explicit PackedGuid(uint64 guid) : _packedGuid(PACKED_GUID_MIN_BUFFER_SIZE) { _packedGuid.appendPackGUID(guid); }
262 explicit PackedGuid(ObjectGuid guid) : _packedGuid(PACKED_GUID_MIN_BUFFER_SIZE) { _packedGuid.appendPackGUID(guid.GetRawValue()); }
263
264 void Set(uint64 guid) { _packedGuid.wpos(0); _packedGuid.appendPackGUID(guid); }
265 void Set(ObjectGuid guid) { _packedGuid.wpos(0); _packedGuid.appendPackGUID(guid.GetRawValue()); }
266
267 size_t size() const { return _packedGuid.size(); }
268
269 private:
270 ByteBuffer _packedGuid;
271};

274class ObjectGuidGeneratorBase
275{
276public:
277 ObjectGuidGeneratorBase(ObjectGuid::LowType start = 1) : _nextGuid(start) { }
278
279 virtual void Set(ObjectGuid::LowType val) { _nextGuid = val; }
280 virtual ObjectGuid::LowType Generate() = 0;
281 ObjectGuid::LowType GetNextAfterMaxUsed() const { return _nextGuid; }
282
283protected:
284 static void HandleCounterOverflow(HighGuid high);
285 ObjectGuid::LowType _nextGuid;
286};
287
288template<HighGuid high>
289class ObjectGuidGenerator : public ObjectGuidGeneratorBase
290{
291public:
292 explicit ObjectGuidGenerator(ObjectGuid::LowType start = 1) : ObjectGuidGeneratorBase(start) { }
293
294 ObjectGuid::LowType Generate() override
295 {
296 if (_nextGuid >= ObjectGuid::GetMaxCounter(high) - 1)
297 HandleCounterOverflow(high);
298 return _nextGuid++;
299 }
300};

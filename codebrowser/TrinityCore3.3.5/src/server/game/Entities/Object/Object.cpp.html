161void Object::BuildCreateUpdateBlockForPlayer(UpdateData* data, Player* target) const
162{
163 if (!target)
164 return;
165
166 uint8 updateType = UPDATETYPE_CREATE_OBJECT;
167 uint16 flags = m_updateFlag;
168
169 /** lower flag1 **/
170 if (target == this) // building packet for yourself
171 flags |= UPDATEFLAG_SELF;
172
173 if (flags & UPDATEFLAG_STATIONARY_POSITION)
174 {
175 // UPDATETYPE_CREATE_OBJECT2 dynamic objects, corpses...
176 if (isType(TYPEMASK_DYNAMICOBJECT) || isType(TYPEMASK_CORPSE) || isType(TYPEMASK_PLAYER))
177 updateType = UPDATETYPE_CREATE_OBJECT2;
178
179 // UPDATETYPE_CREATE_OBJECT2 for pets...
180 if (target->GetPetGUID() == GetGUID())
181 updateType = UPDATETYPE_CREATE_OBJECT2;
182
183 // UPDATETYPE_CREATE_OBJECT2 for some gameobject types...
184 if (isType(TYPEMASK_GAMEOBJECT))
185 {
186 switch (ToGameObject()->GetGoType())
187 {
188 case GAMEOBJECT_TYPE_TRAP:
189 case GAMEOBJECT_TYPE_DUEL_ARBITER:
190 case GAMEOBJECT_TYPE_FLAGSTAND:
191 case GAMEOBJECT_TYPE_FLAGDROP:
192 updateType = UPDATETYPE_CREATE_OBJECT2;
193 break;
194 default:
195 break;
196 }
197 }
198
199 if (isType(TYPEMASK_UNIT))
200 {
201 if (ToUnit()->GetVictim())
202 flags |= UPDATEFLAG_HAS_TARGET;
203 }
204 }
205
206 //TC_LOG_DEBUG("BuildCreateUpdate: update-type: %u, object-type: %u got flags: %X, flags2: %X", updateType, m_objectTypeId, flags, flags2);
207
208 ByteBuffer buf(500);
209 buf << uint8(updateType);
210 buf << GetPackGUID();
211 buf << uint8(m_objectTypeId);
212
213 BuildMovementUpdate(&buf, flags);
214 BuildValuesUpdate(updateType, &buf, target);
215 data->AddUpdateBlock(buf);
216}

246void Object::DestroyForPlayer(Player* target, bool onDeath) const
247{
248 ASSERT(target);
249
250 if (isType(TYPEMASK_UNIT) || isType(TYPEMASK_PLAYER))
251 {
252 if (Battleground* bg = target->GetBattleground())
253 {
254 if (bg->isArena())
255 {
256 WorldPacket data(SMSG_ARENA_UNIT_DESTROYED, 8);
257 data << uint64(GetGUID());
258 target->GetSession()->SendPacket(&data);
259 }
260 }
261 }
262
263 WorldPacket data(SMSG_DESTROY_OBJECT, 8 + 1);
264 data << uint64(GetGUID());
265 //! If the following bool is true, the client will call "void CGUnit_C::OnDeath()" for this object.
266 //! OnDeath() does for eg trigger death animation and interrupts certain spells/missiles/auras/sounds...
267 data << uint8(onDeath ? 1 : 0);
268 target->GetSession()->SendPacket(&data);
269}

315void Object::BuildMovementUpdate(ByteBuffer* data, uint16 flags) const
316{
317 Unit const* unit = NULL;
318 WorldObject const* object = NULL;
319
320 if (isType(TYPEMASK_UNIT))
321 unit = ToUnit();
322 else
323 object = (WorldObject const*)this;
324
325 *data << uint16(flags); // update flags
326
327 // 0x20
328 if (flags & UPDATEFLAG_LIVING)
329 {
330 ASSERT(unit);
331 unit->BuildMovementPacket(data);
332
333 *data << unit->GetSpeed(MOVE_WALK)
334 << unit->GetSpeed(MOVE_RUN)
335 << unit->GetSpeed(MOVE_RUN_BACK)
336 << unit->GetSpeed(MOVE_SWIM)
337 << unit->GetSpeed(MOVE_SWIM_BACK)
338 << unit->GetSpeed(MOVE_FLIGHT)
339 << unit->GetSpeed(MOVE_FLIGHT_BACK)
340 << unit->GetSpeed(MOVE_TURN_RATE)
341 << unit->GetSpeed(MOVE_PITCH_RATE);
342
343 // 0x08000000
344 if (unit->m_movementInfo.GetMovementFlags() & MOVEMENTFLAG_SPLINE_ENABLED)
345 Movement::PacketBuilder::WriteCreate(*unit->movespline, *data);
346 }
347 else
348 {
349 if (flags & UPDATEFLAG_POSITION)
350 {
351 ASSERT(object);
352 Transport* transport = object->GetTransport();
353
354 if (transport)
355 *data << transport->GetPackGUID();
356 else
357 *data << uint8(0);
358
359 *data << object->GetPositionX();
360 *data << object->GetPositionY();
361 if (isType(TYPEMASK_UNIT))
362 *data << unit->GetPositionZMinusOffset();
363 else
364 *data << object->GetPositionZ();
365
366 if (transport)
367 {
368 *data << object->GetTransOffsetX();
369 *data << object->GetTransOffsetY();
370 *data << object->GetTransOffsetZ();
371 }
372 else
373 {
374 *data << object->GetPositionX();
375 *data << object->GetPositionY();
376 if (isType(TYPEMASK_UNIT))
377 *data << unit->GetPositionZMinusOffset();
378 else
379 *data << object->GetPositionZ();
380 }
381
382 *data << object->GetOrientation();
383
384 if (GetTypeId() == TYPEID_CORPSE)
385 *data << float(object->GetOrientation());
386 else
387 *data << float(0);
388 }
389 else
390 {
391 // 0x40
392 if (flags & UPDATEFLAG_STATIONARY_POSITION)
393 {
394 ASSERT(object);
395 *data << object->GetStationaryX();
396 *data << object->GetStationaryY();
397 *data << object->GetStationaryZ();
398 *data << object->GetStationaryO();
399 }
400 }
401 }
402
403 // 0x8
404 if (flags & UPDATEFLAG_UNKNOWN)
405 {
406 *data << uint32(0);
407 }
408
409 // 0x10
410 if (flags & UPDATEFLAG_LOWGUID)
411 {
412 switch (GetTypeId())
413 {
414 case TYPEID_OBJECT:
415 case TYPEID_ITEM:
416 case TYPEID_CONTAINER:
417 case TYPEID_GAMEOBJECT:
418 case TYPEID_DYNAMICOBJECT:
419 case TYPEID_CORPSE:
420 *data << uint32(GetGUID().GetCounter()); // GetGUID().GetCounter()
421 break;
422 //! Unit, Player and default here are sending wrong values.
423 /// @todo Research the proper formula
424 case TYPEID_UNIT:
425 *data << uint32(0x0000000B); // unk
426 break;
427 case TYPEID_PLAYER:
428 if (flags & UPDATEFLAG_SELF)
429 *data << uint32(0x0000002F); // unk
430 else
431 *data << uint32(0x00000008); // unk
432 break;
433 default:
434 *data << uint32(0x00000000); // unk
435 break;
436 }
437 }
438
439 // 0x4
440 if (flags & UPDATEFLAG_HAS_TARGET)
441 {
442 ASSERT(unit);
443 if (Unit* victim = unit->GetVictim())
444 *data << victim->GetPackGUID();
445 else
446 *data << uint8(0);
447 }
448
449 // 0x2
450 if (flags & UPDATEFLAG_TRANSPORT)
451 {
452 GameObject const* go = ToGameObject();
453 /** @TODO Use IsTransport() to also handle type 11 (TRANSPORT)
454 Currently grid objects are not updated if there are no nearby players,
455 this causes clients to receive different PathProgress
456 resulting in players seeing the object in a different position
457 */
458 if (go && go->ToTransport())
459 *data << uint32(go->GetGOValue()->Transport.PathProgress);
460 else
461 *data << uint32(getMSTime());
462 }
463
464 // 0x80
465 if (flags & UPDATEFLAG_VEHICLE)
466 {
467 /// @todo Allow players to aquire this updateflag.
468 ASSERT(unit);
469 ASSERT(unit->GetVehicleKit());
470 ASSERT(unit->GetVehicleKit()->GetVehicleInfo());
471 *data << uint32(unit->GetVehicleKit()->GetVehicleInfo()->m_ID);
472 if (unit->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT))
473 *data << float(unit->GetTransOffsetO());
474 else
475 *data << float(unit->GetOrientation());
476 }
477
478 // 0x200
479 if (flags & UPDATEFLAG_ROTATION)
480 *data << int64(ToGameObject()->GetRotation());
481}

483void Object::BuildValuesUpdate(uint8 updateType, ByteBuffer* data, Player* target) const
484{
485 if (!target)
486 return;
487
488 ByteBuffer fieldBuffer;
489 UpdateMask updateMask;
490 updateMask.SetCount(m_valuesCount);
491
492 uint32* flags = NULL;
493 uint32 visibleFlag = GetUpdateFieldData(target, flags);
494 ASSERT(flags);
495
496 for (uint16 index = 0; index < m_valuesCount; ++index)
497 {
498 if (_fieldNotifyFlags & flags[index] ||
499 ((updateType == UPDATETYPE_VALUES ? _changesMask.GetBit(index) : m_uint32Values[index]) && (flags[index] & visibleFlag)))
500 {
501 updateMask.SetBit(index);
502 fieldBuffer << m_uint32Values[index];
503 }
504 }
505
506 *data << uint8(updateMask.GetBlockCount());
507 updateMask.AppendToPacket(data);
508 data->append(fieldBuffer);
509}

547uint32 Object::GetUpdateFieldData(Player const* target, uint32*& flags) const
548{
549 uint32 visibleFlag = UF_FLAG_PUBLIC;
550
551 if (target == this)
552 visibleFlag |= UF_FLAG_PRIVATE;
553
554 switch (GetTypeId())
555 {
556 case TYPEID_ITEM:
557 case TYPEID_CONTAINER:
558 flags = ItemUpdateFieldFlags;
559 if (((Item const*)this)->GetOwnerGUID() == target->GetGUID())
560 visibleFlag |= UF_FLAG_OWNER | UF_FLAG_ITEM_OWNER;
561 break;
562 case TYPEID_UNIT:
563 case TYPEID_PLAYER:
564 {
565 Player* plr = ToUnit()->GetCharmerOrOwnerPlayerOrPlayerItself();
566 flags = UnitUpdateFieldFlags;
567 if (ToUnit()->GetOwnerGUID() == target->GetGUID())
568 visibleFlag |= UF_FLAG_OWNER;
569
570 if (HasFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_SPECIALINFO))
571 if (ToUnit()->HasAuraTypeWithCaster(SPELL_AURA_EMPATHY, target->GetGUID()))
572 visibleFlag |= UF_FLAG_SPECIAL_INFO;
573
574 if (plr && plr->IsInSameRaidWith(target))
575 visibleFlag |= UF_FLAG_PARTY_MEMBER;
576 break;
577 }
578 case TYPEID_GAMEOBJECT:
579 flags = GameObjectUpdateFieldFlags;
580 if (ToGameObject()->GetOwnerGUID() == target->GetGUID())
581 visibleFlag |= UF_FLAG_OWNER;
582 break;
583 case TYPEID_DYNAMICOBJECT:
584 flags = DynamicObjectUpdateFieldFlags;
585 if (ToDynObject()->GetCasterGUID() == target->GetGUID())
586 visibleFlag |= UF_FLAG_OWNER;
587 break;
588 case TYPEID_CORPSE:
589 flags = CorpseUpdateFieldFlags;
590 if (ToCorpse()->GetOwnerGUID() == target->GetGUID())
591 visibleFlag |= UF_FLAG_OWNER;
592 break;
593 case TYPEID_OBJECT:
594 break;
595 }
596
597 return visibleFlag;
598}

600void Object::_LoadIntoDataField(std::string const& data, uint32 startOffset, uint32 count)
601{
602 if (data.empty())
603 return;
604
605 Tokenizer tokens(data, ' ', count);
606
607 if (tokens.size() != count)
608 return;
609
610 for (uint32 index = 0; index < count; ++index)
611 {
612 m_uint32Values[startOffset + index] = atoul(tokens[index]);
613 _changesMask.SetBit(startOffset + index);
614 }
615}

713void Object::SetByteValue(uint16 index, uint8 offset, uint8 value)
714{
715 ASSERT(index < m_valuesCount || PrintIndexError(index, true));
716
717 if (offset > 3)
718 {

720 return;
721 }
722

727 _changesMask.SetBit(index);

1604bool WorldObject::CanDetectInvisibilityOf(WorldObject const* obj) const
1605{
1606 uint32 mask = obj->m_invisibility.GetFlags() & m_invisibilityDetect.GetFlags();
1607
1608 // Check for not detected types
1609 if (mask != obj->m_invisibility.GetFlags())
1610 return false;
1611
1612 // It isn't possible in invisibility to detect something that can't detect the invisible object
1613 // (it's at least true for spell: 66)
1614 // It seems like that only Units are affected by this check (couldn't see arena doors with preparation invisibility)
1615 if (obj->ToUnit())
1616 if ((m_invisibility.GetFlags() & obj->m_invisibilityDetect.GetFlags()) != m_invisibility.GetFlags())

1622 continue;

1635bool WorldObject::CanDetectStealthOf(WorldObject const* obj, bool checkAlert) const
1636{
1637 // Combat reach is the minimal distance (both in front and behind),
1638 // and it is also used in the range calculation.
1639 // One stealth point increases the visibility range by 0.3 yard.
1640
1641 if (!obj->m_stealth.GetFlags())
1642 return true;
1643
1644 float distance = GetExactDist(obj);
1645 float combatReach = 0.0f;
1646
1647 Unit const* unit = ToUnit();
1648 if (unit)
1649 combatReach = unit->GetCombatReach();
1650
1651 if (distance < combatReach)
1652 return true;
1653
1654 if (!HasInArc(float(M_PI), obj))
1655 return false;
1656
1657 GameObject const* go = ToGameObject();
1658 for (uint32 i = 0; i < TOTAL_STEALTH_TYPES; ++i)
1659 {
1660 if (!(obj->m_stealth.GetFlags() & (1 << i)))
1661 continue;
1662
1663 if (unit && unit->HasAuraTypeWithMiscvalue(SPELL_AURA_DETECT_STEALTH, i))
1664 return true;
1665
1666 // Starting points
1667 int32 detectionValue = 30;
1668
1669 // Level difference: 5 point / level, starting from level 1.
1670 // There may be spells for this and the starting points too, but
1671 // not in the DBCs of the client.
1672 detectionValue += int32(getLevelForTarget(obj) - 1) * 5;
1673
1674 // Apply modifiers
1675 detectionValue += m_stealthDetect.GetValue(StealthType(i));
1676 if (go)
1677 if (Unit* owner = go->GetOwner())
1678 detectionValue -= int32(owner->getLevelForTarget(this) - 1) * 5;
1679
1680 detectionValue -= obj->m_stealth.GetValue(StealthType(i));
1681
1682 // Calculate max distance
1683 float visibilityRange = float(detectionValue) * 0.3f + combatReach;
1684
1685 // If this unit is an NPC then player detect range doesn't apply
1686 if (unit && unit->GetTypeId() == TYPEID_PLAYER && visibilityRange > MAX_PLAYER_STEALTH_DETECT_RANGE)
1687 visibilityRange = MAX_PLAYER_STEALTH_DETECT_RANGE;
1688
1689 // When checking for alert state, look 8% further, and then 1.5 yards more than that.
1690 if (checkAlert)
1691 visibilityRange += (visibilityRange * 0.08f) + 1.5f;
1692
1693 // If checking for alert, and creature's visibility range is greater than aggro distance, No alert
1694 Unit const* tunit = obj->ToUnit();
1695 if (checkAlert && unit && unit->ToCreature() && visibilityRange >= unit->ToCreature()->GetAttackDistance(tunit) + unit->ToCreature()->m_CombatDistance)
1696 return false;
1697
1698 if (distance > visibilityRange)
1699 return false;
1700 }
1701
1702 return true;
1703}

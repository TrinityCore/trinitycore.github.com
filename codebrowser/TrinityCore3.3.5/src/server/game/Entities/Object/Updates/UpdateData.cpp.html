45void UpdateData::Compress(void* dst, uint32 *dst_size, void* src, int src_size)
46{
47 z_stream c_stream;
48
49 c_stream.zalloc = (alloc_func)nullptr;
50 c_stream.zfree = (free_func)nullptr;
51 c_stream.opaque = (voidpf)nullptr;
52
53 // default Z_BEST_SPEED (1)
54 int z_res = deflateInit(&c_stream, sWorld->getIntConfig(CONFIG_COMPRESSION));
55 if (z_res != Z_OK)
56 {
57 TC_LOG_ERROR("misc", "Can't compress update packet (zlib: deflateInit) Error code: %i (%s)", z_res, zError(z_res));
58 *dst_size = 0;
59 return;
60 }
61
62 c_stream.next_out = (Bytef*)dst;
63 c_stream.avail_out = *dst_size;
64 c_stream.next_in = (Bytef*)src;
65 c_stream.avail_in = (uInt)src_size;
66
67 z_res = deflate(&c_stream, Z_NO_FLUSH);
68 if (z_res != Z_OK)
69 {
70 TC_LOG_ERROR("misc", "Can't compress update packet (zlib: deflate) Error code: %i (%s)", z_res, zError(z_res));
71 *dst_size = 0;
72 return;
73 }
74
75 if (c_stream.avail_in != 0)
76 {
77 TC_LOG_ERROR("misc", "Can't compress update packet (zlib: deflate not greedy)");
78 *dst_size = 0;
79 return;
80 }
81
82 z_res = deflate(&c_stream, Z_FINISH);
83 if (z_res != Z_STREAM_END)
84 {
85 TC_LOG_ERROR("misc", "Can't compress update packet (zlib: deflate should report Z_STREAM_END instead %i (%s)", z_res, zError(z_res));
86 *dst_size = 0;
87 return;
88 }
89
90 z_res = deflateEnd(&c_stream);
91 if (z_res != Z_OK)
92 {
93 TC_LOG_ERROR("misc", "Can't compress update packet (zlib: deflateEnd) Error code: %i (%s)", z_res, zError(z_res));
94 *dst_size = 0;
95 return;
96 }
97
98 *dst_size = c_stream.total_out;
99}

101bool UpdateData::BuildPacket(WorldPacket* packet)
102{
103 ASSERT(packet->empty()); // shouldn't happen
104
105 ByteBuffer buf(4 + (m_outOfRangeGUIDs.empty() ? 0 : 1 + 4 + 9 * m_outOfRangeGUIDs.size()) + m_data.wpos());
106
107 buf << (uint32) (!m_outOfRangeGUIDs.empty() ? m_blockCount + 1 : m_blockCount);
108
109 if (!m_outOfRangeGUIDs.empty())
110 {
111 buf << uint8(UPDATETYPE_OUT_OF_RANGE_OBJECTS);
112 buf << uint32(m_outOfRangeGUIDs.size());
113
114 for (GuidSet::const_iterator i = m_outOfRangeGUIDs.begin(); i != m_outOfRangeGUIDs.end(); ++i)
115 buf << i->WriteAsPacked();
116 }
117
118 buf.append(m_data);
119
120 size_t pSize = buf.wpos(); // use real used data size
121
122 if (pSize > 100) // compress large packets
123 {
124 uint32 destsize = compressBound(pSize);
125 packet->resize(destsize + sizeof(uint32));
126
127 packet->put<uint32>(0, pSize);
128 Compress(const_cast<uint8*>(packet->contents()) + sizeof(uint32), &destsize, (void*)buf.contents(), pSize);
129 if (destsize == 0)
130 return false;
131
132 packet->resize(destsize + sizeof(uint32));
133 packet->SetOpcode(SMSG_COMPRESSED_UPDATE_OBJECT);
134 }
135 else // send small packets without compression
136 {
137 packet->append(buf);
138 packet->SetOpcode(SMSG_UPDATE_OBJECT);
139 }
140
141 return true;
142}

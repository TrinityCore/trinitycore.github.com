19#ifndef __UPDATEMASK_H
20#define __UPDATEMASK_H
21
22#include "UpdateFields.h"
23#include "Errors.h"
24#include "ByteBuffer.h"
25
26class UpdateMask
27{
28 public:
29 /// Type representing how client reads update mask
30 typedef uint32 ClientUpdateMaskType;
31
32 enum UpdateMaskCount
33 {
34 CLIENT_UPDATE_MASK_BITS = sizeof(ClientUpdateMaskType) * 8,
35 };
36
37 UpdateMask() : _fieldCount(0), _blockCount(0), _bits(NULL) { }
38
39 UpdateMask(UpdateMask const& right) : _bits(NULL)
40 {
41 SetCount(right.GetCount());
42 memcpy(_bits, right._bits, sizeof(uint8) * _blockCount * 32);
43 }
44
45 ~UpdateMask() { delete[] _bits; }
46
47 void SetBit(uint32 index) { _bits[index] = 1; }
48 void UnsetBit(uint32 index) { _bits[index] = 0; }
49 bool GetBit(uint32 index) const { return _bits[index] != 0; }
50
51 void AppendToPacket(ByteBuffer* data)
52 {
53 for (uint32 i = 0; i < GetBlockCount(); ++i)
54 {
55 ClientUpdateMaskType maskPart = 0;
56 for (uint32 j = 0; j < CLIENT_UPDATE_MASK_BITS; ++j)
57 if (_bits[CLIENT_UPDATE_MASK_BITS * i + j])
58 maskPart |= 1 << j;
59
60 *data << maskPart;
61 }
62 }
63
64 uint32 GetBlockCount() const { return _blockCount; }
65 uint32 GetCount() const { return _fieldCount; }
66
67 void SetCount(uint32 valuesCount)
68 {
69 delete[] _bits;
70
71 _fieldCount = valuesCount;
72 _blockCount = (valuesCount + CLIENT_UPDATE_MASK_BITS - 1) / CLIENT_UPDATE_MASK_BITS;
73
74 _bits = new uint8[_blockCount * CLIENT_UPDATE_MASK_BITS];
75 memset(_bits, 0, sizeof(uint8) * _blockCount * CLIENT_UPDATE_MASK_BITS);
76 }
77
78 void Clear()
79 {
80 if (_bits)
81 memset(_bits, 0, sizeof(uint8) * _blockCount * CLIENT_UPDATE_MASK_BITS);
82 }
83
84 UpdateMask& operator=(UpdateMask const& right)
85 {
86 if (this == &right)
87 return *this;
88
89 SetCount(right.GetCount());
90 memcpy(_bits, right._bits, sizeof(uint8) * _blockCount * CLIENT_UPDATE_MASK_BITS);
91 return *this;
92 }
93
94 UpdateMask& operator&=(UpdateMask const& right)
95 {
96 ASSERT(right.GetCount() <= GetCount());
97 for (uint32 i = 0; i < _fieldCount; ++i)
98 _bits[i] &= right._bits[i];
99
100 return *this;
101 }
102
103 UpdateMask& operator|=(UpdateMask const& right)
104 {
105 ASSERT(right.GetCount() <= GetCount());
106 for (uint32 i = 0; i < _fieldCount; ++i)
107 _bits[i] |= right._bits[i];
108
109 return *this;
110 }
111
112 UpdateMask operator|(UpdateMask const& right)
113 {
114 UpdateMask ret(*this);
115 ret |= right;
116 return ret;
117 }
118
119 private:
120 uint32 _fieldCount;
121 uint32 _blockCount;
122 uint8* _bits;
123};
124
125#endif

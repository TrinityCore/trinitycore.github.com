29DynamicObject::DynamicObject(bool isWorldObject) : WorldObject(isWorldObject),
30 _aura(NULL), _removedAura(NULL), _caster(NULL), _duration(0), _isViewpoint(false)
31{
32 m_objectType |= TYPEMASK_DYNAMICOBJECT;
33 m_objectTypeId = TYPEID_DYNAMICOBJECT;
34
35 m_updateFlag = (UPDATEFLAG_LOWGUID | UPDATEFLAG_STATIONARY_POSITION | UPDATEFLAG_POSITION);
36
37 m_valuesCount = DYNAMICOBJECT_END;
38}

46 delete _removedAura;

49void DynamicObject::AddToWorld()
50{
51 ///- Register the dynamicObject for guid lookup and for caster
52 if (!IsInWorld())
53 {
54 GetMap()->GetObjectsStore().Insert<DynamicObject>(GetGUID(), this);
55 WorldObject::AddToWorld();
56 BindToCaster();
57 }
58}
59
60void DynamicObject::RemoveFromWorld()
61{
62 ///- Remove the dynamicObject from the accessor and from all lists of objects in world
63 if (IsInWorld())
64 {
65 if (_isViewpoint)
66 RemoveCasterViewpoint();
67
68 if (_aura)
69 RemoveAura();
70
71 // dynobj could get removed in Aura::RemoveAura
72 if (!IsInWorld())
73 return;
74
75 UnbindFromCaster();
76 WorldObject::RemoveFromWorld();
77 GetMap()->GetObjectsStore().Remove<DynamicObject>(GetGUID());
78
79 }
80}

82bool DynamicObject::CreateDynamicObject(ObjectGuid::LowType guidlow, Unit* caster, uint32 spellId, Position const& pos, float radius, DynamicObjectType type)
83{
84 SetMap(caster->GetMap());
85 Relocate(pos);
86 if (!IsPositionValid())
87 {
88 TC_LOG_ERROR("misc", "DynamicObject (spell %u) not created. Suggested coordinates isn't valid (X: %f Y: %f)", spellId, GetPositionX(), GetPositionY());
89 return false;
90 }
91
92 WorldObject::_Create(guidlow, HighGuid::DynamicObject, caster->GetPhaseMask());
93
94 SetEntry(spellId);
95 SetObjectScale(1);
96 SetGuidValue(DYNAMICOBJECT_CASTER, caster->GetGUID());
97
98 // The lower word of DYNAMICOBJECT_BYTES must be 0x0001. This value means that the visual radius will be overriden
99 // by client for most of the "ground patch" visual effect spells and a few "skyfall" ones like Hurricane.
100 // If any other value is used, the client will _always_ use the radius provided in DYNAMICOBJECT_RADIUS, but
101 // precompensation is necessary (eg radius *= 2) for many spells. Anyway, blizz sends 0x0001 for all the spells
102 // I saw sniffed...
103 SetByteValue(DYNAMICOBJECT_BYTES, 0, type);
104 SetUInt32Value(DYNAMICOBJECT_SPELLID, spellId);
105 SetFloatValue(DYNAMICOBJECT_RADIUS, radius);
106 SetUInt32Value(DYNAMICOBJECT_CASTTIME, getMSTime());
107
108 if (IsWorldObject())
109 setActive(true); //must before add to map to be put in world container
110
111 Transport* transport = caster->GetTransport();
112 if (transport)
113 {
114 float x, y, z, o;
115 pos.GetPosition(x, y, z, o);
116 transport->CalculatePassengerOffset(x, y, z, &o);
117 m_movementInfo.transport.pos.Relocate(x, y, z, o);
118
119 // This object must be added to transport before adding to map for the client to properly display it
120 transport->AddPassenger(this);
121 }
122
123 if (!GetMap()->AddToMap(this))
124 {
125 // Returning false will cause the object to be deleted - remove from transport
126 if (transport)
127 transport->RemovePassenger(this);
128 return false;
129 }
130
131 return true;
132}

134void DynamicObject::Update(uint32 p_time)
135{
136 // caster has to be always available and in the same map
137 ASSERT(_caster);
138 ASSERT(_caster->GetMap() == GetMap());
139
140 bool expired = false;
141
142 if (_aura)
143 {
144 if (!_aura->IsRemoved())
145 _aura->UpdateOwner(p_time, this);
146
147 // _aura may be set to null in Aura::UpdateOwner call
148 if (_aura && (_aura->IsRemoved() || _aura->IsExpired()))
149 expired = true;
150 }
151 else
152 {
153 if (GetDuration() > int32(p_time))
154 _duration -= p_time;
155 else
156 expired = true;
157 }
158
159 if (expired)
160 Remove();
161 else
162 sScriptMgr->OnDynamicObjectUpdate(this, p_time);
163}

165void DynamicObject::Remove()
166{
167 if (IsInWorld())
168 {
169 SendObjectDeSpawnAnim(GetGUID());
170 RemoveFromWorld();
171 AddObjectToRemoveList();
172 }
173}

198 ASSERT(!_aura && aura);

202void DynamicObject::RemoveAura()
203{
204 ASSERT(_aura && !_removedAura);
205 _removedAura = _aura;
206 _aura = NULL;
207 if (!_removedAura->IsRemoved())
208 _removedAura->_Remove(AURA_REMOVE_BY_DEFAULT);
209}

211void DynamicObject::SetCasterViewpoint()
212{
213 if (Player* caster = _caster->ToPlayer())
214 {
215 caster->SetViewpoint(this, true);
216 _isViewpoint = true;
217 }
218}
219
220void DynamicObject::RemoveCasterViewpoint()
221{
222 if (Player* caster = _caster->ToPlayer())
223 {
224 caster->SetViewpoint(this, false);
225 _isViewpoint = false;
226 }
227}
228
229void DynamicObject::BindToCaster()
230{
231 ASSERT(!_caster);
232 _caster = ObjectAccessor::GetUnit(*this, GetCasterGUID());
233 ASSERT(_caster);
234 ASSERT(_caster->GetMap() == GetMap());
235 _caster->_RegisterDynObject(this);
236}
237
238void DynamicObject::UnbindFromCaster()
239{
240 ASSERT(_caster);
241 _caster->_UnregisterDynObject(this);
242 _caster = NULL;
243}

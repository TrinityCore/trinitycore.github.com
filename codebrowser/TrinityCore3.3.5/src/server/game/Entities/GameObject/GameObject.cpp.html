103void GameObject::RemoveFromOwner()
104{
105 ObjectGuid ownerGUID = GetOwnerGUID();
106 if (!ownerGUID)
107 return;
108
109 if (Unit* owner = ObjectAccessor::GetUnit(*this, ownerGUID))
110 {
111 owner->RemoveGameObject(this, false);
112 ASSERT(!GetOwnerGUID());
113 return;
114 }
115
116 // This happens when a mage portal is despawned after the caster changes map (for example using the portal)
117 TC_LOG_DEBUG("misc", "Removed GameObject (GUID: %u Entry: %u SpellId: %u LinkedGO: %u) that just lost any reference to the owner (%s) GO list",
118 GetGUID().GetCounter(), GetGOInfo()->entry, m_spellId, GetGOInfo()->GetLinkedGameObjectEntry(), ownerGUID.ToString().c_str());
119 SetOwnerGUID(ObjectGuid::Empty);
120}

122void GameObject::AddToWorld()
123{
124 ///- Register the gameobject for guid lookup
125 if (!IsInWorld())
126 {
127 if (m_zoneScript)
128 m_zoneScript->OnGameObjectCreate(this);
129
130 GetMap()->GetObjectsStore().Insert<GameObject>(GetGUID(), this);
131 if (m_spawnId)
132 GetMap()->GetGameObjectBySpawnIdStore().insert(std::make_pair(m_spawnId, this));
133
134 // The state can be changed after GameObject::Create but before GameObject::AddToWorld
135 bool toggledState = GetGoType() == GAMEOBJECT_TYPE_CHEST ? getLootState() == GO_READY : (GetGoState() == GO_STATE_READY || IsTransport());
136 if (m_model)
137 {
138 if (Transport* trans = ToTransport())
139 trans->SetDelayedAddModelToMap();
140 else
141 GetMap()->InsertGameObjectModel(*m_model);
142 }
143
144 EnableCollision(toggledState);
145 WorldObject::AddToWorld();
146 }

166 GetMap()->GetObjectsStore().Remove<GameObject>(GetGUID());
167 }

170bool GameObject::Create(ObjectGuid::LowType guidlow, uint32 name_id, Map* map, uint32 phaseMask, float x, float y, float z, float ang, float rotation0, float rotation1, float rotation2, float rotation3, uint32 animprogress, GOState go_state, uint32 artKit)
171{
172 ASSERT(map);
173 SetMap(map);
174
175 Relocate(x, y, z, ang);
176 m_stationaryPosition.Relocate(x, y, z, ang);
177 if (!IsPositionValid())
178 {
179 TC_LOG_ERROR("misc", "Gameobject (GUID: %u Entry: %u) not created. Suggested coordinates isn't valid (X: %f Y: %f)", guidlow, name_id, x, y);
180 return false;
181 }
182
183 SetPhaseMask(phaseMask, false);
184
185 SetZoneScript();
186 if (m_zoneScript)
187 {
188 name_id = m_zoneScript->GetGameObjectEntry(guidlow, name_id);
189 if (!name_id)
190 return false;
191 }
192
193 GameObjectTemplate const* goinfo = sObjectMgr->GetGameObjectTemplate(name_id);
194 if (!goinfo)
195 {
196 TC_LOG_ERROR("sql.sql", "Gameobject (GUID: %u Entry: %u) not created: non-existing entry in `gameobject_template`. Map: %u (X: %f Y: %f Z: %f)", guidlow, name_id, map->GetId(), x, y, z);
197 return false;
198 }
199
200 if (goinfo->type == GAMEOBJECT_TYPE_MO_TRANSPORT)
201 {
202 TC_LOG_ERROR("sql.sql", "Gameobject (GUID: %u Entry: %u) not created: gameobject type GAMEOBJECT_TYPE_MO_TRANSPORT cannot be manually created.", guidlow, name_id);
203 return false;
204 }
205
206 if (goinfo->type == GAMEOBJECT_TYPE_TRANSPORT)
207 m_updateFlag = (m_updateFlag | UPDATEFLAG_TRANSPORT) & ~UPDATEFLAG_POSITION;
208
209 Object::_Create(guidlow, goinfo->entry, HighGuid::GameObject);
210
211 m_goInfo = goinfo;
212
213 if (goinfo->type >= MAX_GAMEOBJECT_TYPE)
214 {
215 TC_LOG_ERROR("sql.sql", "Gameobject (GUID: %u Entry: %u) not created: non-existing GO type '%u' in `gameobject_template`. It will crash client if created.", guidlow, name_id, goinfo->type);
216 return false;
217 }
218
219 SetFloatValue(GAMEOBJECT_PARENTROTATION+0, rotation0);
220 SetFloatValue(GAMEOBJECT_PARENTROTATION+1, rotation1);
221
222 UpdateRotationFields(rotation2, rotation3); // GAMEOBJECT_FACING, GAMEOBJECT_ROTATION, GAMEOBJECT_PARENTROTATION+2/3
223
224 SetObjectScale(goinfo->size);
225
226 SetUInt32Value(GAMEOBJECT_FACTION, goinfo->faction);
227 SetUInt32Value(GAMEOBJECT_FLAGS, goinfo->flags);
228
229 SetEntry(goinfo->entry);
230
231 // set name for logs usage, doesn't affect anything ingame
232 SetName(goinfo->name);
233
234 SetDisplayId(goinfo->displayId);
235
236 m_model = CreateModel();
237 // GAMEOBJECT_BYTES_1, index at 0, 1, 2 and 3
238 SetGoType(GameobjectTypes(goinfo->type));
239 SetGoState(go_state);
240 SetGoArtKit(artKit);
241
242 switch (goinfo->type)
243 {
244 case GAMEOBJECT_TYPE_FISHINGHOLE:
245 SetGoAnimProgress(animprogress);
246 m_goValue.FishingHole.MaxOpens = urand(GetGOInfo()->fishinghole.minSuccessOpens, GetGOInfo()->fishinghole.maxSuccessOpens);
247 break;
248 case GAMEOBJECT_TYPE_DESTRUCTIBLE_BUILDING:
249 m_goValue.Building.Health = goinfo->building.intactNumHits + goinfo->building.damagedNumHits;
250 m_goValue.Building.MaxHealth = m_goValue.Building.Health;
251 SetGoAnimProgress(255);
252 break;
253 case GAMEOBJECT_TYPE_TRANSPORT:
254 SetUInt32Value(GAMEOBJECT_LEVEL, goinfo->transport.pause);
255 SetGoState(goinfo->transport.startOpen ? GO_STATE_ACTIVE : GO_STATE_READY);
256 SetGoAnimProgress(animprogress);
257 m_goValue.Transport.PathProgress = 0;
258 m_goValue.Transport.AnimationInfo = sTransportMgr->GetTransportAnimInfo(goinfo->entry);
259 m_goValue.Transport.CurrentSeg = 0;
260 break;
261 case GAMEOBJECT_TYPE_FISHINGNODE:
262 SetGoAnimProgress(0);
263 break;
264 case GAMEOBJECT_TYPE_TRAP:
265 if (GetGOInfo()->trap.stealthed)
266 {
267 m_stealth.AddFlag(STEALTH_TRAP);
268 m_stealth.AddValue(STEALTH_TRAP, 70);
269 }
270
271 if (GetGOInfo()->trap.invisible)
272 {
273 m_invisibility.AddFlag(INVISIBILITY_TRAP);
274 m_invisibility.AddValue(INVISIBILITY_TRAP, 300);
275 }
276 break;
277 default:
278 SetGoAnimProgress(animprogress);
279 break;
280 }
281
282 if (GameObjectAddon const* addon = sObjectMgr->GetGameObjectAddon(GetSpawnId()))
283 {
284 if (addon->InvisibilityValue)
285 {
286 m_invisibility.AddFlag(addon->invisibilityType);
287 m_invisibility.AddValue(addon->invisibilityType, addon->InvisibilityValue);
288 }
289 }
290
291 LastUsedScriptID = GetGOInfo()->ScriptId;
292 AIM_Initialize();
293
294 // Initialize loot duplicate count depending on raid difficulty
295 if (map->Is25ManRaid())
296 loot.maxDuplicates = 3;
297
298 return true;
299}

301void GameObject::Update(uint32 diff)
302{
303 if (AI())
304 AI()->UpdateAI(diff);
305 else if (!AIM_Initialize())
306 TC_LOG_ERROR("misc", "Could not initialize GameObjectAI");
307
308 switch (m_lootState)
309 {
310 case GO_NOT_READY:
311 {
312 switch (GetGoType())
313 {
314 case GAMEOBJECT_TYPE_TRAP:
315 {
316 // Arming Time for GAMEOBJECT_TYPE_TRAP (6)
317 GameObjectTemplate const* goInfo = GetGOInfo();
318 // Bombs
319 if (goInfo->trap.type == 2)
320 // Hardcoded tooltip value
321 m_cooldownTime = time(NULL) + 10;
322 else if (Unit* owner = GetOwner())
323 if (owner->IsInCombat())
324 m_cooldownTime = time(NULL) + goInfo->trap.startDelay;
325
326 SetLootState(GO_READY);
327 break;
328 }
329 case GAMEOBJECT_TYPE_TRANSPORT:
330 {
331 if (!m_goValue.Transport.AnimationInfo)
332 break;
333
334 if (GetGoState() == GO_STATE_READY)
335 {
336 m_goValue.Transport.PathProgress += diff;
337 /* TODO: Fix movement in unloaded grid - currently GO will just disappear
338 uint32 timer = m_goValue.Transport.PathProgress % m_goValue.Transport.AnimationInfo->TotalTime;
339 TransportAnimationEntry const* node = m_goValue.Transport.AnimationInfo->GetAnimNode(timer);
340 if (node && m_goValue.Transport.CurrentSeg != node->TimeSeg)
341 {
342 m_goValue.Transport.CurrentSeg = node->TimeSeg;
343
344 G3D::Quat rotation = m_goValue.Transport.AnimationInfo->GetAnimRotation(timer);
345 G3D::Vector3 pos = rotation.toRotationMatrix()
346 * G3D::Matrix3::fromEulerAnglesZYX(GetOrientation(), 0.0f, 0.0f)
347 * G3D::Vector3(node->X, node->Y, node->Z);
348
349 pos += G3D::Vector3(GetStationaryX(), GetStationaryY(), GetStationaryZ());
350
351 G3D::Vector3 src(GetPositionX(), GetPositionY(), GetPositionZ());
352
353 TC_LOG_DEBUG("misc", "Src: %s Dest: %s", src.toString().c_str(), pos.toString().c_str());
354
355 GetMap()->GameObjectRelocation(this, pos.x, pos.y, pos.z, GetOrientation());
356 }
357 */
358 }
359 break;
360 }
361 case GAMEOBJECT_TYPE_FISHINGNODE:
362 {
363 // fishing code (bobber ready)
364 if (time(NULL) > m_respawnTime - FISHING_BOBBER_READY_TIME)
365 {
366 // splash bobber (bobber ready now)
367 Unit* caster = GetOwner();
368 if (caster && caster->GetTypeId() == TYPEID_PLAYER)
369 {
370 SetGoState(GO_STATE_ACTIVE);
371 SetUInt32Value(GAMEOBJECT_FLAGS, GO_FLAG_NODESPAWN);
372
373 UpdateData udata;
374 WorldPacket packet;
375 BuildValuesUpdateBlockForPlayer(&udata, caster->ToPlayer());
376 udata.BuildPacket(&packet);
377 caster->ToPlayer()->SendDirectMessage(&packet);
378
379 SendCustomAnim(GetGoAnimProgress());
380 }
381
382 m_lootState = GO_READY; // can be successfully open with some chance
383 }
384 return;
385 }
386 default:
387 m_lootState = GO_READY; // for other GOis same switched without delay to GO_READY
388 break;
389 }
390 // NO BREAK for switch (m_lootState)
391 }
392 case GO_READY:
393 {
394 if (m_respawnTime > 0) // timer on
395 {
396 time_t now = time(NULL);
397 if (m_respawnTime <= now) // timer expired
398 {
399 ObjectGuid dbtableHighGuid(HighGuid::GameObject, GetEntry(), m_spawnId);
400 time_t linkedRespawntime = GetMap()->GetLinkedRespawnTime(dbtableHighGuid);
401 if (linkedRespawntime) // Can't respawn, the master is dead
402 {
403 ObjectGuid targetGuid = sObjectMgr->GetLinkedRespawnGuid(dbtableHighGuid);
404 if (targetGuid == dbtableHighGuid) // if linking self, never respawn (check delayed to next day)
405 SetRespawnTime(DAY);
406 else
407 m_respawnTime = (now > linkedRespawntime ? now : linkedRespawntime) + urand(5, MINUTE); // else copy time from master and add a little
408 SaveRespawnTime(); // also save to DB immediately
409 return;
410 }
411
412 m_respawnTime = 0;
413 m_SkillupList.clear();
414 m_usetimes = 0;
415
416 switch (GetGoType())
417 {
418 case GAMEOBJECT_TYPE_FISHINGNODE: // can't fish now
419 {
420 Unit* caster = GetOwner();
421 if (caster && caster->GetTypeId() == TYPEID_PLAYER)
422 {
423 caster->ToPlayer()->RemoveGameObject(this, false);
424
425 WorldPacket data(SMSG_FISH_ESCAPED, 0);
426 caster->ToPlayer()->SendDirectMessage(&data);
427 }
428 // can be delete
429 m_lootState = GO_JUST_DEACTIVATED;
430 return;
431 }
432 case GAMEOBJECT_TYPE_DOOR:
433 case GAMEOBJECT_TYPE_BUTTON:
434 // We need to open doors if they are closed (add there another condition if this code breaks some usage, but it need to be here for battlegrounds)
435 if (GetGoState() != GO_STATE_READY)
436 ResetDoorOrButton();
437 break;
438 case GAMEOBJECT_TYPE_FISHINGHOLE:
439 // Initialize a new max fish count on respawn
440 m_goValue.FishingHole.MaxOpens = urand(GetGOInfo()->fishinghole.minSuccessOpens, GetGOInfo()->fishinghole.maxSuccessOpens);
441 break;
442 default:
443 break;
444 }
445
446 // Despawn timer
447 if (!m_spawnedByDefault)
448 {
449 // Can be despawned or destroyed
450 SetLootState(GO_JUST_DEACTIVATED);
451 return;
452 }
453
454 // Respawn timer
455 uint32 poolid = GetSpawnId() ? sPoolMgr->IsPartOfAPool<GameObject>(GetSpawnId()) : 0;
456 if (poolid)
457 sPoolMgr->UpdatePool<GameObject>(poolid, GetSpawnId());
458 else
459 GetMap()->AddToMap(this);
460 }
461 }
462
463 if (isSpawned())
464 {
465 GameObjectTemplate const* goInfo = GetGOInfo();
466 if (goInfo->type == GAMEOBJECT_TYPE_TRAP)
467 {
468 if (m_cooldownTime >= time(NULL))
469 break;
470
471 // Type 2 (bomb) does not need to be triggered by a unit and despawns after casting its spell.
472 if (goInfo->trap.type == 2)
473 {
474 SetLootState(GO_ACTIVATED);
475 break;
476 }
477
478 // Type 0 despawns after being triggered, type 1 does not.
479 /// @todo This is activation radius. Casting radius must be selected from spell data.
480 float radius;
481 if (!goInfo->trap.diameter)
482 {
483 // Battleground traps: data2 == 0 && data5 == 3
484 if (goInfo->trap.cooldown != 3)
485 break;
486
487 radius = 3.f;
488 }
489 else
490 radius = goInfo->trap.diameter / 2.f;
491
492 // Pointer to appropriate target if found any
493 Unit* target = NULL;
494
495 /// @todo this hack with search required until GO casting not implemented
496 if (Unit* owner = GetOwner())
497 {
498 // Hunter trap: Search units which are unfriendly to the trap's owner
499 Trinity::AnyUnfriendlyNoTotemUnitInObjectRangeCheck checker(this, owner, radius);
500 Trinity::UnitSearcher<Trinity::AnyUnfriendlyNoTotemUnitInObjectRangeCheck> searcher(this, target, checker);
501 VisitNearbyGridObject(radius, searcher);
502 if (!target)
503 VisitNearbyWorldObject(radius, searcher);
504 }
505 else
506 {
507 // Environmental trap: Any player
508 Player* player = NULL;
509 Trinity::AnyPlayerInObjectRangeCheck checker(this, radius);
510 Trinity::PlayerSearcher<Trinity::AnyPlayerInObjectRangeCheck> searcher(this, player, checker);
511 VisitNearbyWorldObject(radius, searcher);
512 target = player;
513 }
514
515 if (target)
516 SetLootState(GO_ACTIVATED, target);
517
518 }
519 else if (uint32 max_charges = goInfo->GetCharges())
520 {
521 if (m_usetimes >= max_charges)
522 {
523 m_usetimes = 0;
524 SetLootState(GO_JUST_DEACTIVATED); // can be despawned or destroyed
525 }
526 }
527 }
528
529 break;
530 }
531 case GO_ACTIVATED:
532 {
533 switch (GetGoType())
534 {
535 case GAMEOBJECT_TYPE_DOOR:
536 case GAMEOBJECT_TYPE_BUTTON:
537 if (m_cooldownTime && (m_cooldownTime < time(NULL)))
538 ResetDoorOrButton();
539 break;
540 case GAMEOBJECT_TYPE_GOOBER:
541 if (m_cooldownTime < time(NULL))
542 {
543 RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_IN_USE);
544
545 SetLootState(GO_JUST_DEACTIVATED);
546 m_cooldownTime = 0;
547 }
548 break;
549 case GAMEOBJECT_TYPE_CHEST:
550 if (m_groupLootTimer)
551 {
552 if (m_groupLootTimer <= diff)
553 {
554 Group* group = sGroupMgr->GetGroupByGUID(lootingGroupLowGUID);
555 if (group)
556 group->EndRoll(&loot);
557 m_groupLootTimer = 0;
558 lootingGroupLowGUID = 0;
559 }
560 else m_groupLootTimer -= diff;
561 }
562 break;
563 case GAMEOBJECT_TYPE_TRAP:
564 {
565 GameObjectTemplate const* goInfo = GetGOInfo();
566 if (goInfo->trap.type == 2 && goInfo->trap.spellId)
567 {
568 /// @todo NULL target won't work for target type 1
569 CastSpell(NULL, goInfo->trap.spellId);
570 SetLootState(GO_JUST_DEACTIVATED);
571 }
572 else if (Unit* target = ObjectAccessor::GetUnit(*this, m_lootStateUnitGUID))
573 {
574 // Some traps do not have a spell but should be triggered
575 if (goInfo->trap.spellId)
576 CastSpell(target, goInfo->trap.spellId);
577
578 // Template value or 4 seconds
579 m_cooldownTime = time(NULL) + (goInfo->trap.cooldown ? goInfo->trap.cooldown : uint32(4));
580
581 if (goInfo->trap.type == 1)
582 SetLootState(GO_JUST_DEACTIVATED);
583 else if (!goInfo->trap.type)
584 SetLootState(GO_READY);
585
586 // Battleground gameobjects have data2 == 0 && data5 == 3
587 if (!goInfo->trap.diameter && goInfo->trap.cooldown == 3)
588 if (Player* player = target->ToPlayer())
589 if (Battleground* bg = player->GetBattleground())
590 bg->HandleTriggerBuff(GetGUID());
591 }
592 break;
593 }
594 default:
595 break;
596 }
597 break;
598 }
599 case GO_JUST_DEACTIVATED:
600 {
601 //if Gameobject should cast spell, then this, but some GOs (type = 10) should be destroyed
602 if (GetGoType() == GAMEOBJECT_TYPE_GOOBER)
603 {
604 uint32 spellId = GetGOInfo()->goober.spellId;
605
606 if (spellId)
607 {
608 for (GuidSet::const_iterator it = m_unique_users.begin(); it != m_unique_users.end(); ++it)
609 // m_unique_users can contain only player GUIDs
610 if (Player* owner = ObjectAccessor::GetPlayer(*this, *it))
611 owner->CastSpell(owner, spellId, false);
612
613 m_unique_users.clear();
614 m_usetimes = 0;
615 }
616
617 SetGoState(GO_STATE_READY);
618
619 //any return here in case battleground traps
620 if (GetGOInfo()->flags & GO_FLAG_NODESPAWN)
621 return;
622 }
623
624 loot.clear();
625
626 //! If this is summoned by a spell with ie. SPELL_EFFECT_SUMMON_OBJECT_WILD, with or without owner, we check respawn criteria based on spell
627 //! The GetOwnerGUID() check is mostly for compatibility with hacky scripts - 99% of the time summoning should be done trough spells.
628 if (GetSpellId() || GetOwnerGUID())
629 {
630 SetRespawnTime(0);
631 Delete();
632 return;
633 }
634
635 SetLootState(GO_READY);
636
637 //burning flags in some battlegrounds, if you find better condition, just add it
638 if (GetGOInfo()->IsDespawnAtAction() || GetGoAnimProgress() > 0)
639 {
640 SendObjectDeSpawnAnim(GetGUID());
641 //reset flags
642 SetUInt32Value(GAMEOBJECT_FLAGS, GetGOInfo()->flags);
643 }
644
645 if (!m_respawnDelayTime)
646 return;
647
648 if (!m_spawnedByDefault)
649 {
650 m_respawnTime = 0;
651 UpdateObjectVisibility();
652 return;
653 }
654
655 m_respawnTime = time(NULL) + m_respawnDelayTime;
656
657 // if option not set then object will be saved at grid unload
658 if (sWorld->getBoolConfig(CONFIG_SAVE_RESPAWN_TIME_IMMEDIATELY))
659 SaveRespawnTime();
660
661 UpdateObjectVisibility();
662
663 break;
664 }
665 }
666 sScriptMgr->OnGameObjectUpdate(this, diff);
667}

702void GameObject::getFishLoot(Loot* fishloot, Player* loot_owner)
703{
704 fishloot->clear();
705
706 uint32 zone, subzone;
707 uint32 defaultzone = 1;
708 GetZoneAndAreaId(zone, subzone);
709

1183void GameObject::Use(Unit* user)
1184{
1185 // by default spell caster is user
1186 Unit* spellCaster = user;
1187 uint32 spellId = 0;
1188 bool triggered = false;
1189
1190 if (Player* playerUser = user->ToPlayer())
1191 {
1192 if (sScriptMgr->OnGossipHello(playerUser, this))
1193 return;
1194
1195 if (AI()->GossipHello(playerUser))
1196 return;
1197 }
1198
1199 // If cooldown data present in template
1200 if (uint32 cooldown = GetGOInfo()->GetCooldown())
1201 {
1202 if (m_cooldownTime > sWorld->GetGameTime())
1203 return;
1204
1205 m_cooldownTime = sWorld->GetGameTime() + cooldown;
1206 }
1207
1208 switch (GetGoType())
1209 {
1210 case GAMEOBJECT_TYPE_DOOR: //0
1211 case GAMEOBJECT_TYPE_BUTTON: //1
1212 //doors/buttons never really despawn, only reset to default state/flags
1213 UseDoorOrButton(0, false, user);
1214 return;
1215 case GAMEOBJECT_TYPE_QUESTGIVER: //2
1216 {
1217 if (user->GetTypeId() != TYPEID_PLAYER)
1218 return;
1219
1220 Player* player = user->ToPlayer();
1221
1222 player->PrepareGossipMenu(this, GetGOInfo()->questgiver.gossipID, true);
1223 player->SendPreparedGossip(this);
1224 return;
1225 }
1226 case GAMEOBJECT_TYPE_TRAP: //6
1227 {
1228 GameObjectTemplate const* goInfo = GetGOInfo();
1229 if (goInfo->trap.spellId)
1230 CastSpell(user, goInfo->trap.spellId);
1231
1232 m_cooldownTime = time(NULL) + (goInfo->trap.cooldown ? goInfo->trap.cooldown : uint32(4)); // template or 4 seconds
1233
1234 if (goInfo->trap.type == 1) // Deactivate after trigger
1235 SetLootState(GO_JUST_DEACTIVATED);
1236
1237 return;
1238 }
1239 //Sitting: Wooden bench, chairs enzz
1240 case GAMEOBJECT_TYPE_CHAIR: //7
1241 {
1242 GameObjectTemplate const* info = GetGOInfo();
1243 if (!info)
1244 return;
1245
1246 if (user->GetTypeId() != TYPEID_PLAYER)
1247 return;
1248
1249 if (ChairListSlots.empty()) // this is called once at first chair use to make list of available slots
1250 {
1251 if (info->chair.slots > 0) // sometimes chairs in DB have error in fields and we dont know number of slots
1252 for (uint32 i = 0; i < info->chair.slots; ++i)
1253 ChairListSlots[i].Clear(); // Last user of current slot set to 0 (none sit here yet)
1254 else
1255 ChairListSlots[0].Clear(); // error in DB, make one default slot
1256 }
1257
1258 Player* player = user->ToPlayer();
1259
1260 // a chair may have n slots. we have to calculate their positions and teleport the player to the nearest one
1261
1262 float lowestDist = DEFAULT_VISIBILITY_DISTANCE;
1263
1264 uint32 nearest_slot = 0;
1265 float x_lowest = GetPositionX();
1266 float y_lowest = GetPositionY();
1267
1268 // the object orientation + 1/2 pi
1269 // every slot will be on that straight line
1270 float orthogonalOrientation = GetOrientation() + float(M_PI) * 0.5f;
1271 // find nearest slot
1272 bool found_free_slot = false;
1273 for (ChairSlotAndUser::iterator itr = ChairListSlots.begin(); itr != ChairListSlots.end(); ++itr)
1274 {
1275 // the distance between this slot and the center of the go - imagine a 1D space
1276 float relativeDistance = (info->size*itr->first)-(info->size*(info->chair.slots-1)/2.0f);
1277
1278 float x_i = GetPositionX() + relativeDistance * std::cos(orthogonalOrientation);
1279 float y_i = GetPositionY() + relativeDistance * std::sin(orthogonalOrientation);
1280
1281 if (itr->second)
1282 {
1283 if (Player* ChairUser = ObjectAccessor::FindPlayer(itr->second))
1284 {
1285 if (ChairUser->IsSitState() && ChairUser->getStandState() != UNIT_STAND_STATE_SIT && ChairUser->GetExactDist2d(x_i, y_i) < 0.1f)
1286 continue; // This seat is already occupied by ChairUser. NOTE: Not sure if the ChairUser->getStandState() != UNIT_STAND_STATE_SIT check is required.
1287 else
1288 itr->second.Clear(); // This seat is unoccupied.
1289 }
1290 else
1291 itr->second.Clear(); // The seat may of had an occupant, but they're offline.
1292 }
1293
1294 found_free_slot = true;
1295
1296 // calculate the distance between the player and this slot
1297 float thisDistance = player->GetDistance2d(x_i, y_i);
1298
1299 if (thisDistance <= lowestDist)
1300 {
1301 nearest_slot = itr->first;
1302 lowestDist = thisDistance;
1303 x_lowest = x_i;
1304 y_lowest = y_i;
1305 }
1306 }
1307
1308 if (found_free_slot)
1309 {
1310 ChairSlotAndUser::iterator itr = ChairListSlots.find(nearest_slot);
1311 if (itr != ChairListSlots.end())
1312 {
1313 itr->second = player->GetGUID(); //this slot in now used by player
1314 player->TeleportTo(GetMapId(), x_lowest, y_lowest, GetPositionZ(), GetOrientation(), TELE_TO_NOT_LEAVE_TRANSPORT | TELE_TO_NOT_LEAVE_COMBAT | TELE_TO_NOT_UNSUMMON_PET);
1315 player->SetStandState(UNIT_STAND_STATE_SIT_LOW_CHAIR+info->chair.height);
1316 return;
1317 }
1318 }
1319
1320 return;
1321 }
1322 //big gun, its a spell/aura
1323 case GAMEOBJECT_TYPE_GOOBER: //10
1324 {
1325 GameObjectTemplate const* info = GetGOInfo();
1326
1327 if (Player* player = user->ToPlayer())
1328 {
1329 if (info->goober.pageId) // show page...
1330 {
1331 WorldPacket data(SMSG_GAMEOBJECT_PAGETEXT, 8);
1332 data << GetGUID();
1333 player->SendDirectMessage(&data);
1334 }
1335 else if (info->goober.gossipID)
1336 {
1337 player->PrepareGossipMenu(this, info->goober.gossipID);
1338 player->SendPreparedGossip(this);
1339 }
1340
1341 if (info->goober.eventId)
1342 {
1343 TC_LOG_DEBUG("maps.script", "Goober ScriptStart id %u for GO entry %u (GUID %u).", info->goober.eventId, GetEntry(), GetSpawnId());
1344 GetMap()->ScriptsStart(sEventScripts, info->goober.eventId, player, this);
1345 EventInform(info->goober.eventId, user);
1346 }
1347
1348 // possible quest objective for active quests
1349 if (info->goober.questId && sObjectMgr->GetQuestTemplate(info->goober.questId))
1350 {
1351 //Quest require to be active for GO using
1352 if (player->GetQuestStatus(info->goober.questId) != QUEST_STATUS_INCOMPLETE)
1353 break;
1354 }
1355
1356 if (Group* group = player->GetGroup())
1357 {
1358 for (GroupReference const* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
1359 if (Player* member = itr->GetSource())
1360 if (member->IsAtGroupRewardDistance(this))
1361 member->KillCreditGO(info->entry, GetGUID());
1362 }
1363 else
1364 player->KillCreditGO(info->entry, GetGUID());
1365 }
1366
1367 if (uint32 trapEntry = info->goober.linkedTrapId)
1368 TriggeringLinkedGameObject(trapEntry, user);

1840bool GameObject::IsInRange(float x, float y, float z, float radius) const
1841{
1842 GameObjectDisplayInfoEntry const* info = sGameObjectDisplayInfoStore.LookupEntry(m_goInfo->displayId);
1843 if (!info)
1844 return IsWithinDist3d(x, y, z, radius);
1845
1846 float sinA = std::sin(GetOrientation());
1847 float cosA = std::cos(GetOrientation());
1848 float dx = x - GetPositionX();
1849 float dy = y - GetPositionY();
1850 float dz = z - GetPositionZ();
1851 float dist = std::sqrt(dx*dx + dy*dy);
1852 //! Check if the distance between the 2 objects is 0, can happen if both objects are on the same position.
1853 //! The code below this check wont crash if dist is 0 because 0/0 in float operations is valid, and returns infinite
1854 if (G3D::fuzzyEq(dist, 0.0f))
1855 return true;
1856
1857 float sinB = dx / dist;
1858 float cosB = dy / dist;
1859 dx = dist * (cosA * cosB + sinA * sinB);
1860 dy = dist * (cosA * sinB - sinA * cosB);
1861 return dx < info->maxX + radius && dx > info->minX - radius
1862 && dy < info->maxY + radius && dy > info->minY - radius
1863 && dz < info->maxZ + radius && dz > info->minZ - radius;
1864}

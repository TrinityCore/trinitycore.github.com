28inline bool _ModifyUInt32(bool apply, uint32& baseValue, int32& amount)
29{
30 // If amount is negative, change sign and value of apply.
31 if (amount < 0)
32 {
33 apply = !apply;
34 amount = -amount;
35 }
36 if (apply)
37 baseValue += amount;
38 else
39 {
40 // Make sure we do not get uint32 overflow.
41 if (amount > int32(baseValue))
42 amount = baseValue;
43 baseValue -= amount;
44 }
45 return apply;
46}

60void Unit::UpdateDamagePhysical(WeaponAttackType attType)
61{
62 float minDamage = 0.0f;
63 float maxDamage = 0.0f;
64
65 CalculateMinMaxDamage(attType, false, true, minDamage, maxDamage);
66
67 switch (attType)
68 {
69 case BASE_ATTACK:
70 default:
71 SetStatFloatValue(UNIT_FIELD_MINDAMAGE, minDamage);
72 SetStatFloatValue(UNIT_FIELD_MAXDAMAGE, maxDamage);
73 break;
74 case OFF_ATTACK:
75 SetStatFloatValue(UNIT_FIELD_MINOFFHANDDAMAGE, minDamage);
76 SetStatFloatValue(UNIT_FIELD_MAXOFFHANDDAMAGE, maxDamage);
77 break;
78 case RANGED_ATTACK:
79 SetStatFloatValue(UNIT_FIELD_MINRANGEDDAMAGE, minDamage);
80 SetStatFloatValue(UNIT_FIELD_MAXRANGEDDAMAGE, maxDamage);
81 break;
82 }
83}

91bool Player::UpdateStats(Stats stat)
92{
93 if (stat > STAT_SPIRIT)
94 return false;
95
96 // value = ((base_value * base_pct) + total_value) * total_pct
97 float value = GetTotalStatValue(stat);
98
99 SetStat(stat, int32(value));
100
101 if (stat == STAT_STAMINA || stat == STAT_INTELLECT || stat == STAT_STRENGTH)
102 {
103 Pet* pet = GetPet();
104 if (pet)
105 pet->UpdateStats(stat);
106 }
107
108 switch (stat)
109 {
110 case STAT_STRENGTH:
111 UpdateShieldBlockValue();
112 break;
113 case STAT_AGILITY:
114 UpdateArmor();
115 UpdateAllCritPercentages();
116 UpdateDodgePercentage();
117 break;
118 case STAT_STAMINA:
119 UpdateMaxHealth();
120 break;
121 case STAT_INTELLECT:
122 UpdateMaxPower(POWER_MANA);
123 UpdateAllSpellCritChances();
124 UpdateArmor(); //SPELL_AURA_MOD_RESISTANCE_OF_INTELLECT_PERCENT, only armor currently
125 break;
126 case STAT_SPIRIT:
127 break;
128 default:
129 break;
130 }
131
132 if (stat == STAT_STRENGTH)
133 {
134 UpdateAttackPowerAndDamage(false);
135 if (HasAuraTypeWithMiscvalue(SPELL_AURA_MOD_RANGED_ATTACK_POWER_OF_STAT_PERCENT, stat))
136 UpdateAttackPowerAndDamage(true);
137 }
138 else if (stat == STAT_AGILITY)
139 {
140 UpdateAttackPowerAndDamage(false);
141 UpdateAttackPowerAndDamage(true);
142 }
143 else
144 {
145 // Need update (exist AP from stat auras)
146 if (HasAuraTypeWithMiscvalue(SPELL_AURA_MOD_ATTACK_POWER_OF_STAT_PERCENT, stat))
147 UpdateAttackPowerAndDamage(false);
148 if (HasAuraTypeWithMiscvalue(SPELL_AURA_MOD_RANGED_ATTACK_POWER_OF_STAT_PERCENT, stat))
149 UpdateAttackPowerAndDamage(true);
150 }
151
152 UpdateSpellDamageAndHealingBonus();
153 UpdateManaRegen();
154
155 // Update ratings in exist SPELL_AURA_MOD_RATING_FROM_STAT and only depends from stat
156 uint32 mask = 0;
157 AuraEffectList const& modRatingFromStat = GetAuraEffectsByType(SPELL_AURA_MOD_RATING_FROM_STAT);
158 for (AuraEffectList::const_iterator i = modRatingFromStat.begin(); i != modRatingFromStat.end(); ++i)
159 if (Stats((*i)->GetMiscValueB()) == stat)
160 mask |= (*i)->GetMiscValue();
161 if (mask)
162 {
163 for (uint32 rating = 0; rating < MAX_COMBAT_RATING; ++rating)
164 if (mask & (1 << rating))
165 ApplyRatingMod(CombatRating(rating), 0, true);
166 }
167 return true;
168}

171{

191bool Player::UpdateAllStats()
192{
193 for (int8 i = STAT_STRENGTH; i < MAX_STATS; ++i)
194 {
195 float value = GetTotalStatValue(Stats(i));
196 SetStat(Stats(i), int32(value));
197 }
198
199 UpdateArmor();
200 // calls UpdateAttackPowerAndDamage() in UpdateArmor for SPELL_AURA_MOD_ATTACK_POWER_OF_ARMOR
201 UpdateAttackPowerAndDamage(true);
202 UpdateMaxHealth();
203
204 for (uint8 i = POWER_MANA; i < MAX_POWERS; ++i)
205 UpdateMaxPower(Powers(i));
206
207 UpdateAllRatings();
208 UpdateAllCritPercentages();
209 UpdateAllSpellCritChances();
210 UpdateDefenseBonusesMod();
211 UpdateShieldBlockValue();
212 UpdateSpellDamageAndHealingBonus();
213 UpdateManaRegen();
214 UpdateExpertise(BASE_ATTACK);
215 UpdateExpertise(OFF_ATTACK);
216 RecalculateRating(CR_ARMOR_PENETRATION);
217 UpdateAllResistances();
218
219 return true;
220}

243void Player::UpdateArmor()
244{
245 UnitMods unitMod = UNIT_MOD_ARMOR;
246
247 float value = GetModifierValue(unitMod, BASE_VALUE); // base armor (from items)
248 value *= GetModifierValue(unitMod, BASE_PCT); // armor percent from items
249 value += GetStat(STAT_AGILITY) * 2.0f; // armor bonus from stats
250 value += GetModifierValue(unitMod, TOTAL_VALUE);
251
252 //add dynamic flat mods
253 AuraEffectList const& mResbyIntellect = GetAuraEffectsByType(SPELL_AURA_MOD_RESISTANCE_OF_STAT_PERCENT);
254 for (AuraEffectList::const_iterator i = mResbyIntellect.begin(); i != mResbyIntellect.end(); ++i)
255 {
256 if ((*i)->GetMiscValue() & SPELL_SCHOOL_MASK_NORMAL)
257 value += CalculatePct(GetStat(Stats((*i)->GetMiscValueB())), (*i)->GetAmount());
258 }
259
260 value *= GetModifierValue(unitMod, TOTAL_PCT);
261
262 SetArmor(int32(value));
263
264 Pet* pet = GetPet();
265 if (pet)
266 pet->UpdateArmor();
267
268 UpdateAttackPowerAndDamage(); // armor dependent auras update for SPELL_AURA_MOD_ATTACK_POWER_OF_ARMOR
269}

271float Player::GetHealthBonusFromStamina()
272{
273 float stamina = GetStat(STAT_STAMINA);
274 float baseStam = std::min(20.0f, stamina);
275 float moreStam = stamina - baseStam;
276
277 return baseStam + (moreStam*10.0f);
278}
279
280float Player::GetManaBonusFromIntellect()
281{
282 float intellect = GetStat(STAT_INTELLECT);
283
284 float baseInt = std::min(20.0f, intellect);
285 float moreInt = intellect - baseInt;
286
287 return baseInt + (moreInt * 15.0f);
288}

290void Player::UpdateMaxHealth()
291{
292 UnitMods unitMod = UNIT_MOD_HEALTH;
293
294 float value = GetModifierValue(unitMod, BASE_VALUE) + GetCreateHealth();
295 value *= GetModifierValue(unitMod, BASE_PCT);
296 value += GetModifierValue(unitMod, TOTAL_VALUE) + GetHealthBonusFromStamina();
297 value *= GetModifierValue(unitMod, TOTAL_PCT);
298
299 SetMaxHealth((uint32)value);
300}

302void Player::UpdateMaxPower(Powers power)
303{
304 UnitMods unitMod = UnitMods(UNIT_MOD_POWER_START + power);
305
306 float bonusPower = (power == POWER_MANA && GetCreatePowers(power) > 0) ? GetManaBonusFromIntellect() : 0;
307
308 float value = GetModifierValue(unitMod, BASE_VALUE) + GetCreatePowers(power);
309 value *= GetModifierValue(unitMod, BASE_PCT);
310 value += GetModifierValue(unitMod, TOTAL_VALUE) + bonusPower;
311 value *= GetModifierValue(unitMod, TOTAL_PCT);
312
313 SetMaxPower(power, uint32(value));
314}

322void Player::UpdateAttackPowerAndDamage(bool ranged)
323{
324 float val2 = 0.0f;
325 float level = float(getLevel());
326
327 UnitMods unitMod = ranged ? UNIT_MOD_ATTACK_POWER_RANGED : UNIT_MOD_ATTACK_POWER;
328
329 uint16 index = UNIT_FIELD_ATTACK_POWER;
330 uint16 index_mod = UNIT_FIELD_ATTACK_POWER_MODS;
331 uint16 index_mult = UNIT_FIELD_ATTACK_POWER_MULTIPLIER;
332
333 if (ranged)
334 {
335 index = UNIT_FIELD_RANGED_ATTACK_POWER;
336 index_mod = UNIT_FIELD_RANGED_ATTACK_POWER_MODS;
337 index_mult = UNIT_FIELD_RANGED_ATTACK_POWER_MULTIPLIER;
338
339 switch (getClass())
340 {
341 case CLASS_HUNTER:
342 val2 = level * 2.0f + GetStat(STAT_AGILITY) - 10.0f;
343 break;
344 case CLASS_ROGUE:
345 val2 = level + GetStat(STAT_AGILITY) - 10.0f;
346 break;
347 case CLASS_WARRIOR:
348 val2 = level + GetStat(STAT_AGILITY) - 10.0f;
349 break;
350 case CLASS_DRUID:
351 switch (GetShapeshiftForm())
352 {
353 case FORM_CAT:
354 case FORM_BEAR:
355 case FORM_DIREBEAR:
356 val2 = 0.0f; break;
357 default:
358 val2 = GetStat(STAT_AGILITY) - 10.0f; break;
359 }
360 break;
361 default: val2 = GetStat(STAT_AGILITY) - 10.0f; break;
362 }
363 }
364 else
365 {
366 switch (getClass())
367 {
368 case CLASS_WARRIOR:
369 val2 = level * 3.0f + GetStat(STAT_STRENGTH) * 2.0f - 20.0f;
370 break;
371 case CLASS_PALADIN:
372 val2 = level * 3.0f + GetStat(STAT_STRENGTH) * 2.0f - 20.0f;
373 break;
374 case CLASS_DEATH_KNIGHT:
375 val2 = level * 3.0f + GetStat(STAT_STRENGTH) * 2.0f - 20.0f;
376 break;
377 case CLASS_ROGUE:
378 val2 = level * 2.0f + GetStat(STAT_STRENGTH) + GetStat(STAT_AGILITY) - 20.0f;
379 break;
380 case CLASS_HUNTER:
381 val2 = level * 2.0f + GetStat(STAT_STRENGTH) + GetStat(STAT_AGILITY) - 20.0f;
382 break;
383 case CLASS_SHAMAN:
384 val2 = level * 2.0f + GetStat(STAT_STRENGTH) + GetStat(STAT_AGILITY) - 20.0f;
385 break;
386 case CLASS_DRUID:
387 {
388 // Check if Predatory Strikes is skilled
389 float mLevelMult = 0.0f;
390 float weapon_bonus = 0.0f;
391 if (IsInFeralForm())
392 {
393 Unit::AuraEffectList const& mDummy = GetAuraEffectsByType(SPELL_AURA_DUMMY);
394 for (Unit::AuraEffectList::const_iterator itr = mDummy.begin(); itr != mDummy.end(); ++itr)
395 {
396 AuraEffect* aurEff = *itr;
397 if (aurEff->GetSpellInfo()->SpellIconID == 1563)
398 {
399 switch (aurEff->GetEffIndex())
400 {
401 case 0: // Predatory Strikes (effect 0)
402 mLevelMult = CalculatePct(1.0f, aurEff->GetAmount());
403 break;
404 case 1: // Predatory Strikes (effect 1)
405 if (Item* mainHand = m_items[EQUIPMENT_SLOT_MAINHAND])
406 {
407 // also gains % attack power from equipped weapon
408 ItemTemplate const* proto = mainHand->GetTemplate();
409 if (!proto)
410 continue;
411
412 weapon_bonus = CalculatePct(float(proto->getFeralBonus()), aurEff->GetAmount());
413 }
414 break;
415 default:
416 break;
417 }
418 }
419 }
420 }
421
422 switch (GetShapeshiftForm())
423 {
424 case FORM_CAT:
425 val2 = getLevel() * (mLevelMult + 2.0f) + GetStat(STAT_STRENGTH) * 2.0f + GetStat(STAT_AGILITY) - 20.0f + weapon_bonus + m_baseFeralAP;
426 break;
427 case FORM_BEAR:
428 case FORM_DIREBEAR:
429 val2 = getLevel() * (mLevelMult + 3.0f) + GetStat(STAT_STRENGTH) * 2.0f - 20.0f + weapon_bonus + m_baseFeralAP;
430 break;
431 case FORM_MOONKIN:
432 val2 = getLevel() * (mLevelMult + 1.5f) + GetStat(STAT_STRENGTH) * 2.0f - 20.0f + m_baseFeralAP;
433 break;
434 default:
435 val2 = GetStat(STAT_STRENGTH) * 2.0f - 20.0f;
436 break;
437 }
438 break;
439 }
440 case CLASS_MAGE:
441 val2 = GetStat(STAT_STRENGTH) - 10.0f;
442 break;
443 case CLASS_PRIEST:
444 val2 = GetStat(STAT_STRENGTH) - 10.0f;
445 break;
446 case CLASS_WARLOCK:
447 val2 = GetStat(STAT_STRENGTH) - 10.0f;
448 break;
449 }
450 }
451
452 SetModifierValue(unitMod, BASE_VALUE, val2);
453
454 float base_attPower = GetModifierValue(unitMod, BASE_VALUE) * GetModifierValue(unitMod, BASE_PCT);
455 float attPowerMod = GetModifierValue(unitMod, TOTAL_VALUE);
456
457 //add dynamic flat mods
458 if (ranged)
459 {
460 if ((getClassMask() & CLASSMASK_WAND_USERS) == 0)
461 {
462 AuraEffectList const& mRAPbyStat = GetAuraEffectsByType(SPELL_AURA_MOD_RANGED_ATTACK_POWER_OF_STAT_PERCENT);
463 for (AuraEffectList::const_iterator i = mRAPbyStat.begin(); i != mRAPbyStat.end(); ++i)
464 attPowerMod += CalculatePct(GetStat(Stats((*i)->GetMiscValue())), (*i)->GetAmount());
465 }
466 }
467 else
468 {
469 AuraEffectList const& mAPbyStat = GetAuraEffectsByType(SPELL_AURA_MOD_ATTACK_POWER_OF_STAT_PERCENT);
470 for (AuraEffectList::const_iterator i = mAPbyStat.begin(); i != mAPbyStat.end(); ++i)
471 attPowerMod += CalculatePct(GetStat(Stats((*i)->GetMiscValue())), (*i)->GetAmount());
472
473 AuraEffectList const& mAPbyArmor = GetAuraEffectsByType(SPELL_AURA_MOD_ATTACK_POWER_OF_ARMOR);
474 for (AuraEffectList::const_iterator iter = mAPbyArmor.begin(); iter != mAPbyArmor.end(); ++iter)
475 // always: ((*i)->GetModifier()->m_miscvalue == 1 == SPELL_SCHOOL_MASK_NORMAL)
476 attPowerMod += int32(GetArmor() / (*iter)->GetAmount());
477 }
478
479 float attPowerMultiplier = GetModifierValue(unitMod, TOTAL_PCT) - 1.0f;
480
481 SetInt32Value(index, (uint32)base_attPower); //UNIT_FIELD_(RANGED)_ATTACK_POWER field
482 SetInt32Value(index_mod, (uint32)attPowerMod); //UNIT_FIELD_(RANGED)_ATTACK_POWER_MODS field
483 SetFloatValue(index_mult, attPowerMultiplier); //UNIT_FIELD_(RANGED)_ATTACK_POWER_MULTIPLIER field
484
485 Pet* pet = GetPet(); //update pet's AP
486 Guardian* guardian = GetGuardianPet();
487 //automatically update weapon damage after attack power modification
488 if (ranged)
489 {
490 UpdateDamagePhysical(RANGED_ATTACK);
491 if (pet && pet->IsHunterPet()) // At ranged attack change for hunter pet
492 pet->UpdateAttackPowerAndDamage();
493 }
494 else
495 {
496 UpdateDamagePhysical(BASE_ATTACK);
497 if (CanDualWield() && haveOffhandWeapon()) //allow update offhand damage only if player knows DualWield Spec and has equipped offhand weapon
498 UpdateDamagePhysical(OFF_ATTACK);
499 if (getClass() == CLASS_SHAMAN || getClass() == CLASS_PALADIN) // mental quickness
500 UpdateSpellDamageAndHealingBonus();
501
502 if (pet && (pet->IsPetGhoul() || pet->IsRisenAlly())) // At melee attack power change for DK pet
503 pet->UpdateAttackPowerAndDamage();
504
505 if (guardian && guardian->IsSpiritWolf()) // At melee attack power change for Shaman feral spirit
506 guardian->UpdateAttackPowerAndDamage();
507 }
508}

515void Player::CalculateMinMaxDamage(WeaponAttackType attType, bool normalized, bool addTotalPct, float& minDamage, float& maxDamage)
516{
517 UnitMods unitMod;
518
519 switch (attType)
520 {
521 case BASE_ATTACK:
522 default:
523 unitMod = UNIT_MOD_DAMAGE_MAINHAND;
524 break;
525 case OFF_ATTACK:
526 unitMod = UNIT_MOD_DAMAGE_OFFHAND;
527 break;
528 case RANGED_ATTACK:
529 unitMod = UNIT_MOD_DAMAGE_RANGED;
530 break;
531 }
532
533 float attackSpeedMod = GetAPMultiplier(attType, normalized);
534
535 float baseValue = GetModifierValue(unitMod, BASE_VALUE) + GetTotalAttackPowerValue(attType) / 14.0f * attackSpeedMod;
536 float basePct = GetModifierValue(unitMod, BASE_PCT);
537 float totalValue = GetModifierValue(unitMod, TOTAL_VALUE);
538 float totalPct = addTotalPct ? GetModifierValue(unitMod, TOTAL_PCT) : 1.0f;
539
540 float weaponMinDamage = GetWeaponDamageRange(attType, MINDAMAGE);
541 float weaponMaxDamage = GetWeaponDamageRange(attType, MAXDAMAGE);
542
543 if (IsInFeralForm()) // check if player is druid and in cat or bear forms
544 {
545 uint8 lvl = getLevel();
546 if (lvl > 60)
547 lvl = 60;
548
549 weaponMinDamage = lvl * 0.85f * attackSpeedMod;
550 weaponMaxDamage = lvl * 1.25f * attackSpeedMod;
551 }
552 else if (!CanUseAttackType(attType)) // check if player not in form but still can't use (disarm case)
553 {
554 // cannot use ranged/off attack, set values to 0
555 if (attType != BASE_ATTACK)
556 {
557 minDamage = 0;
558 maxDamage = 0;
559 return;
560 }
561 weaponMinDamage = BASE_MINDAMAGE;
562 weaponMaxDamage = BASE_MAXDAMAGE;
563 }
564 else if (attType == RANGED_ATTACK) // add ammo DPS to ranged damage
565 {
566 weaponMinDamage += GetAmmoDPS() * attackSpeedMod;
567 weaponMaxDamage += GetAmmoDPS() * attackSpeedMod;
568 }
569
570 minDamage = ((weaponMinDamage + baseValue) * basePct + totalValue) * totalPct;
571 maxDamage = ((weaponMaxDamage + baseValue) * basePct + totalValue) * totalPct;
572}

581void Player::UpdateBlockPercentage()
582{
583 // No block
584 float value = 0.0f;
585 if (CanBlock())
586 {
587 // Base value
588 value = 5.0f;
589 // Modify value from defense skill
590 value += (int32(GetDefenseSkillValue()) - int32(GetMaxSkillValueForLevel())) * 0.04f;
591 // Increase from SPELL_AURA_MOD_BLOCK_PERCENT aura
592 value += GetTotalAuraModifier(SPELL_AURA_MOD_BLOCK_PERCENT);
593 // Increase from rating
594 value += GetRatingBonusValue(CR_BLOCK);
595
596 if (sWorld->getBoolConfig(CONFIG_STATS_LIMITS_ENABLE))
597 value = value > sWorld->getFloatConfig(CONFIG_STATS_LIMITS_BLOCK) ? sWorld->getFloatConfig(CONFIG_STATS_LIMITS_BLOCK) : value;
598
599 value = value < 0.0f ? 0.0f : value;
600 }
601 SetStatFloatValue(PLAYER_BLOCK_PERCENTAGE, value);
602}

604void Player::UpdateCritPercentage(WeaponAttackType attType)
605{
606 BaseModGroup modGroup;
607 uint16 index;
608 CombatRating cr;
609
610 switch (attType)
611 {
612 case OFF_ATTACK:
613 modGroup = OFFHAND_CRIT_PERCENTAGE;
614 index = PLAYER_OFFHAND_CRIT_PERCENTAGE;
615 cr = CR_CRIT_MELEE;
616 break;
617 case RANGED_ATTACK:
618 modGroup = RANGED_CRIT_PERCENTAGE;
619 index = PLAYER_RANGED_CRIT_PERCENTAGE;
620 cr = CR_CRIT_RANGED;
621 break;
622 case BASE_ATTACK:
623 default:
624 modGroup = CRIT_PERCENTAGE;
625 index = PLAYER_CRIT_PERCENTAGE;
626 cr = CR_CRIT_MELEE;
627 break;
628 }
629
630 float value = GetTotalPercentageModValue(modGroup) + GetRatingBonusValue(cr);
631 // Modify crit from weapon skill and maximized defense skill of same level victim difference
632 value += (int32(GetWeaponSkillValue(attType)) - int32(GetMaxSkillValueForLevel())) * 0.04f;
633
634 if (sWorld->getBoolConfig(CONFIG_STATS_LIMITS_ENABLE))
635 value = value > sWorld->getFloatConfig(CONFIG_STATS_LIMITS_CRIT) ? sWorld->getFloatConfig(CONFIG_STATS_LIMITS_CRIT) : value;
636
637 value = value < 0.0f ? 0.0f : value;
638 SetStatFloatValue(index, value);
639}

876void Player::UpdateManaRegen()
877{
878 float Intellect = GetStat(STAT_INTELLECT);
879 // Mana regen from spirit and intellect
880 float power_regen = std::sqrt(Intellect) * OCTRegenMPPerSpirit();
881 // Apply PCT bonus from SPELL_AURA_MOD_POWER_REGEN_PERCENT aura on spirit base regen
882 power_regen *= GetTotalAuraMultiplierByMiscValue(SPELL_AURA_MOD_POWER_REGEN_PERCENT, POWER_MANA);
883
884 // Mana regen from SPELL_AURA_MOD_POWER_REGEN aura
885 float power_regen_mp5 = (GetTotalAuraModifierByMiscValue(SPELL_AURA_MOD_POWER_REGEN, POWER_MANA) + m_baseManaRegen) / 5.0f;
886
887 // Get bonus from SPELL_AURA_MOD_MANA_REGEN_FROM_STAT aura
888 AuraEffectList const& regenAura = GetAuraEffectsByType(SPELL_AURA_MOD_MANA_REGEN_FROM_STAT);
889 for (AuraEffectList::const_iterator i = regenAura.begin(); i != regenAura.end(); ++i)
890 {
891 power_regen_mp5 += GetStat(Stats((*i)->GetMiscValue())) * (*i)->GetAmount() / 500.0f;
892 }
893
894 // Set regen rate in cast state apply only on spirit based regen
895 int32 modManaRegenInterrupt = GetTotalAuraModifier(SPELL_AURA_MOD_MANA_REGEN_INTERRUPT);
896 if (modManaRegenInterrupt > 100)
897 modManaRegenInterrupt = 100;
898 SetStatFloatValue(UNIT_FIELD_POWER_REGEN_INTERRUPTED_FLAT_MODIFIER, power_regen_mp5 + CalculatePct(power_regen, modManaRegenInterrupt));
899
900 SetStatFloatValue(UNIT_FIELD_POWER_REGEN_FLAT_MODIFIER, power_regen_mp5 + power_regen);
901}

1037void Creature::CalculateMinMaxDamage(WeaponAttackType attType, bool normalized, bool addTotalPct, float& minDamage, float& maxDamage)
1038{
1039 float variance = 1.0f;
1040 UnitMods unitMod;
1041 switch (attType)
1042 {
1043 case BASE_ATTACK:
1044 default:
1045 variance = GetCreatureTemplate()->BaseVariance;
1046 unitMod = UNIT_MOD_DAMAGE_MAINHAND;
1047 break;
1048 case OFF_ATTACK:
1049 variance = GetCreatureTemplate()->BaseVariance;
1050 unitMod = UNIT_MOD_DAMAGE_OFFHAND;
1051 break;
1052 case RANGED_ATTACK:
1053 variance = GetCreatureTemplate()->RangeVariance;
1054 unitMod = UNIT_MOD_DAMAGE_RANGED;
1055 break;
1056 }
1057
1058 if (attType == OFF_ATTACK && !haveOffhandWeapon())
1059 {
1060 minDamage = 0.0f;
1061 maxDamage = 0.0f;
1062 return;
1063 }
1064
1065 float weaponMinDamage = GetWeaponDamageRange(attType, MINDAMAGE);
1066 float weaponMaxDamage = GetWeaponDamageRange(attType, MAXDAMAGE);
1067
1068 if (!CanUseAttackType(attType)) // disarm case
1069 {
1070 weaponMinDamage = 0.0f;
1071 weaponMaxDamage = 0.0f;
1072 }
1073
1074 float attackPower = GetTotalAttackPowerValue(attType);
1075 float attackSpeedMulti = GetAPMultiplier(attType, normalized);
1076 float baseValue = GetModifierValue(unitMod, BASE_VALUE) + (attackPower / 14.0f) * variance;
1077 float basePct = GetModifierValue(unitMod, BASE_PCT) * attackSpeedMulti;
1078 float totalValue = GetModifierValue(unitMod, TOTAL_VALUE);
1079 float totalPct = addTotalPct ? GetModifierValue(unitMod, TOTAL_PCT) : 1.0f;
1080 float dmgMultiplier = GetCreatureTemplate()->ModDamage; // = ModDamage * _GetDamageMod(rank);
1081
1082 minDamage = ((weaponMinDamage + baseValue) * dmgMultiplier * basePct + totalValue) * totalPct;
1083 maxDamage = ((weaponMaxDamage + baseValue) * dmgMultiplier * basePct + totalValue) * totalPct;
1084}

1275 float addValue = (power == POWER_MANA) ? GetStat(STAT_INTELLECT) - GetCreateStat(STAT_INTELLECT) : 0.0f;
1276 float multiplicator = 15.0f;
1277
1278 switch (GetEntry())
1279 {
1280 case ENTRY_IMP: multiplicator = 4.95f; break;
1281 case ENTRY_VOIDWALKER:
1282 case ENTRY_SUCCUBUS:
1283 case ENTRY_FELHUNTER:
1284 case ENTRY_FELGUARD: multiplicator = 11.5f; break;
1285 default: multiplicator = 15.0f; break;
1286 }
1287
1288 float value = GetModifierValue(unitMod, BASE_VALUE) + GetCreatePowers(power);
1289 value *= GetModifierValue(unitMod, BASE_PCT);
1290 value += GetModifierValue(unitMod, TOTAL_VALUE) + addValue * multiplicator;
1291 value *= GetModifierValue(unitMod, TOTAL_PCT);
1292

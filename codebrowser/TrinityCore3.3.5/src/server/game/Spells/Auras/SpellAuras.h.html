42class AuraApplication
43{
44 friend void Unit::_ApplyAura(AuraApplication * aurApp, uint8 effMask);

48 friend AuraApplication * Unit::_CreateAuraApplication(Aura* aura, uint8 effMask);
49 private:
50 Unit* const _target;
51 Aura* const _base;

53 uint8 _slot; // Aura slot on unit
54 uint8 _flags; // Aura info flag
55 uint8 _effectsToApply; // Used only at spell hit to determine which effect should be applied
56 bool _needClientUpdate:1;
57
58 explicit AuraApplication(Unit* target, Unit* caster, Aura* base, uint8 effMask);
59 void _Remove();
60 private:
61 void _InitFlags(Unit* caster, uint8 effMask);
62 void _HandleEffect(uint8 effIndex, bool apply);
63 public:
64
65 Unit* GetTarget() const { return _target; }
66 Aura* GetBase() const { return _base; }
67
68 uint8 GetSlot() const { return _slot; }
69 uint8 GetFlags() const { return _flags; }
70 uint8 GetEffectMask() const { return _flags & (AFLAG_EFF_INDEX_0 | AFLAG_EFF_INDEX_1 | AFLAG_EFF_INDEX_2); }
71 bool HasEffect(uint8 effect) const { ASSERT(effect < MAX_SPELL_EFFECTS); return (_flags & (1 << effect)) != 0; }
72 bool IsPositive() const { return (_flags & AFLAG_POSITIVE) != 0; }
73 bool IsSelfcast() const { return (_flags & AFLAG_CASTER) != 0; }
74 uint8 GetEffectsToApply() const { return _effectsToApply; }

76 void SetRemoveMode(AuraRemoveMode mode) { _removeMode = mode; }
77 AuraRemoveMode GetRemoveMode() const {return _removeMode;}

79 void SetNeedClientUpdate() { _needClientUpdate = true;}
80 bool IsNeedClientUpdate() const { return _needClientUpdate;}
81 void BuildUpdatePacket(ByteBuffer& data, bool remove) const;
82 void ClientUpdate(bool remove = false);
83};

317 // Binary predicate for sorting the priority of absorption aura effects
318 class AbsorbAuraOrderPred
319 {
320 public:
321 AbsorbAuraOrderPred() { }
322 bool operator() (AuraEffect* aurEffA, AuraEffect* aurEffB) const
323 {
324 SpellInfo const* spellProtoA = aurEffA->GetSpellInfo();
325 SpellInfo const* spellProtoB = aurEffB->GetSpellInfo();
326
327 // Wards
328 if ((spellProtoA->SpellFamilyName == SPELLFAMILY_MAGE) ||
329 (spellProtoA->SpellFamilyName == SPELLFAMILY_WARLOCK))
330 if (spellProtoA->GetCategory() == 56)
331 return true;
332 if ((spellProtoB->SpellFamilyName == SPELLFAMILY_MAGE) ||
333 (spellProtoB->SpellFamilyName == SPELLFAMILY_WARLOCK))
334 if (spellProtoB->GetCategory() == 56)
335 return false;
336
337 // Sacred Shield
338 if (spellProtoA->Id == 58597)
339 return true;
340 if (spellProtoB->Id == 58597)
341 return false;
342
343 // Fel Blossom
344 if (spellProtoA->Id == 28527)
345 return true;
346 if (spellProtoB->Id == 28527)
347 return false;
348
349 // Divine Aegis
350 if (spellProtoA->Id == 47753)
351 return true;
352 if (spellProtoB->Id == 47753)
353 return false;
354
355 // Ice Barrier
356 if (spellProtoA->GetCategory() == 471)
357 return true;
358 if (spellProtoB->GetCategory() == 471)
359 return false;
360
361 // Sacrifice
362 if ((spellProtoA->SpellFamilyName == SPELLFAMILY_WARLOCK) &&
363 (spellProtoA->SpellIconID == 693))
364 return true;
365 if ((spellProtoB->SpellFamilyName == SPELLFAMILY_WARLOCK) &&
366 (spellProtoB->SpellIconID == 693))
367 return false;
368
369 return false;
370 }
371 };

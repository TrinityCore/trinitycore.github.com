42enum SpellCastTargetFlags
43{
44 TARGET_FLAG_NONE = 0x00000000,
45 TARGET_FLAG_UNUSED_1 = 0x00000001, // not used
46 TARGET_FLAG_UNIT = 0x00000002, // pguid
47 TARGET_FLAG_UNIT_RAID = 0x00000004, // not sent, used to validate target (if raid member)
48 TARGET_FLAG_UNIT_PARTY = 0x00000008, // not sent, used to validate target (if party member)
49 TARGET_FLAG_ITEM = 0x00000010, // pguid
50 TARGET_FLAG_SOURCE_LOCATION = 0x00000020, // pguid, 3 float
51 TARGET_FLAG_DEST_LOCATION = 0x00000040, // pguid, 3 float
52 TARGET_FLAG_UNIT_ENEMY = 0x00000080, // not sent, used to validate target (if enemy)
53 TARGET_FLAG_UNIT_ALLY = 0x00000100, // not sent, used to validate target (if ally)
54 TARGET_FLAG_CORPSE_ENEMY = 0x00000200, // pguid
55 TARGET_FLAG_UNIT_DEAD = 0x00000400, // not sent, used to validate target (if dead creature)
56 TARGET_FLAG_GAMEOBJECT = 0x00000800, // pguid, used with TARGET_GAMEOBJECT_TARGET
57 TARGET_FLAG_TRADE_ITEM = 0x00001000, // pguid
58 TARGET_FLAG_STRING = 0x00002000, // string
59 TARGET_FLAG_GAMEOBJECT_ITEM = 0x00004000, // not sent, used with TARGET_GAMEOBJECT_ITEM_TARGET
60 TARGET_FLAG_CORPSE_ALLY = 0x00008000, // pguid
61 TARGET_FLAG_UNIT_MINIPET = 0x00010000, // pguid, used to validate target (if non combat pet)
62 TARGET_FLAG_GLYPH_SLOT = 0x00020000, // used in glyph spells
63 TARGET_FLAG_DEST_TARGET = 0x00040000, // sometimes appears with DEST_TARGET spells (may appear or not for a given spell)
64 TARGET_FLAG_UNUSED20 = 0x00080000, // uint32 counter, loop { vec3 - screen position (?), guid }, not used so far
65 TARGET_FLAG_UNIT_PASSENGER = 0x00100000, // guessed, used to validate target (if vehicle passenger)
66
67 TARGET_FLAG_UNIT_MASK = TARGET_FLAG_UNIT | TARGET_FLAG_UNIT_RAID | TARGET_FLAG_UNIT_PARTY
68 | TARGET_FLAG_UNIT_ENEMY | TARGET_FLAG_UNIT_ALLY | TARGET_FLAG_UNIT_DEAD | TARGET_FLAG_UNIT_MINIPET | TARGET_FLAG_UNIT_PASSENGER,
69 TARGET_FLAG_GAMEOBJECT_MASK = TARGET_FLAG_GAMEOBJECT | TARGET_FLAG_GAMEOBJECT_ITEM,
70 TARGET_FLAG_CORPSE_MASK = TARGET_FLAG_CORPSE_ALLY | TARGET_FLAG_CORPSE_ENEMY,
71 TARGET_FLAG_ITEM_MASK = TARGET_FLAG_TRADE_ITEM | TARGET_FLAG_ITEM | TARGET_FLAG_GAMEOBJECT_ITEM
72};

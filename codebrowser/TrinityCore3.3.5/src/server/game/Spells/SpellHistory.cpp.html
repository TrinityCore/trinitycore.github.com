27SpellHistory::Clock::duration const SpellHistory::InfinityCooldownDelay = std::chrono::duration_cast<SpellHistory::Clock::duration>(std::chrono::seconds(MONTH));
28SpellHistory::Clock::duration const SpellHistory::InfinityCooldownDelayCheck = std::chrono::duration_cast<SpellHistory::Clock::duration>(std::chrono::seconds(MONTH / 2));

30template<>
31struct SpellHistory::PersistenceHelper<Player>
32{
33 static CharacterDatabaseStatements const CooldownsDeleteStatement = CHAR_DEL_CHAR_SPELL_COOLDOWNS;
34 static CharacterDatabaseStatements const CooldownsInsertStatement = CHAR_INS_CHAR_SPELL_COOLDOWN;
35
36 static void SetIdentifier(PreparedStatement* stmt, uint8 index, Unit* owner) { stmt->setUInt32(index, owner->GetGUID().GetCounter()); }
37
38 static bool ReadCooldown(Field* fields, uint32* spellId, CooldownEntry* cooldownEntry)
39 {
40 *spellId = fields[0].GetUInt32();
41 if (!sSpellMgr->GetSpellInfo(*spellId))
42 return false;
43
44 cooldownEntry->SpellId = *spellId;
45 cooldownEntry->CooldownEnd = Clock::from_time_t(time_t(fields[2].GetUInt32()));
46 cooldownEntry->ItemId = fields[1].GetUInt32();
47 cooldownEntry->CategoryId = fields[3].GetUInt32();
48 cooldownEntry->CategoryEnd = Clock::from_time_t(time_t(fields[4].GetUInt32()));
49 return true;
50 }
51
52 static void WriteCooldown(PreparedStatement* stmt, uint8& index, CooldownStorageType::value_type const& cooldown)
53 {
54 stmt->setUInt32(index++, cooldown.first);
55 stmt->setUInt32(index++, cooldown.second.ItemId);
56 stmt->setUInt32(index++, uint32(Clock::to_time_t(cooldown.second.CooldownEnd)));
57 stmt->setUInt32(index++, cooldown.second.CategoryId);
58 stmt->setUInt32(index++, uint32(Clock::to_time_t(cooldown.second.CategoryEnd)));
59 }
60};

93template<class OwnerType>
94void SpellHistory::LoadFromDB(PreparedQueryResult cooldownsResult)
95{
96 typedef PersistenceHelper<OwnerType> StatementInfo;
97
98 if (cooldownsResult)
99 {
100 do
101 {
102 uint32 spellId;
103 CooldownEntry cooldown;
104 if (StatementInfo::ReadCooldown(cooldownsResult->Fetch(), &spellId, &cooldown))
105 {
106 _spellCooldowns[spellId] = cooldown;
107 if (cooldown.CategoryId)
108 _categoryCooldowns[cooldown.CategoryId] = &_spellCooldowns[spellId];
109 }
110
111 } while (cooldownsResult->NextRow());
112 }
113}

115template<class OwnerType>
116void SpellHistory::SaveToDB(SQLTransaction& trans)
117{
118 typedef PersistenceHelper<OwnerType> StatementInfo;
119
120 uint8 index = 0;
121 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(StatementInfo::CooldownsDeleteStatement);
122 StatementInfo::SetIdentifier(stmt, index++, _owner);
123 trans->Append(stmt);
124
125 for (auto const& p : _spellCooldowns)
126 {
127 if (!p.second.OnHold)
128 {
129 index = 0;
130 stmt = CharacterDatabase.GetPreparedStatement(StatementInfo::CooldownsInsertStatement);
131 StatementInfo::SetIdentifier(stmt, index++, _owner);
132 StatementInfo::WriteCooldown(stmt, index, p);
133 trans->Append(stmt);
134 }
135 }
136}

138void SpellHistory::Update()
139{
140 SQLTransaction t;
141 Clock::time_point now = Clock::now();
142 for (auto itr = _categoryCooldowns.begin(); itr != _categoryCooldowns.end();)
143 {
144 if (itr->second->CategoryEnd < now)
145 itr = _categoryCooldowns.erase(itr);
146 else
147 ++itr;
148 }
149
150 for (auto itr = _spellCooldowns.begin(); itr != _spellCooldowns.end();)
151 {
152 if (itr->second.CooldownEnd < now)
153 itr = EraseCooldown(itr);
154 else
155 ++itr;
156 }
157}

164void SpellHistory::HandleCooldowns(SpellInfo const* spellInfo, uint32 itemID, Spell* spell /*= nullptr*/)
165{
166 if (Player* player = _owner->ToPlayer())
167 {
168 // potions start cooldown until exiting combat
169 if (ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(itemID))
170 {
171 if (itemTemplate->IsPotion() || spellInfo->IsCooldownStartedOnEvent())
172 {
173 player->SetLastPotionId(itemID);
174 return;
175 }
176 }
177 }
178
179 if (spellInfo->IsCooldownStartedOnEvent() || spellInfo->IsPassive() || (spell && spell->IsIgnoringCooldowns()))
180 return;
181
182 StartCooldown(spellInfo, itemID, spell);
183}

185bool SpellHistory::IsReady(SpellInfo const* spellInfo, uint32 itemId /*= 0*/, bool ignoreCategoryCooldown /*= false*/) const
186{
187 if (spellInfo->PreventionType == SPELL_PREVENTION_TYPE_SILENCE)
188 if (IsSchoolLocked(spellInfo->GetSchoolMask()))
189 return false;
190
191 if (HasCooldown(spellInfo->Id, itemId, ignoreCategoryCooldown))
192 return false;
193
194 return true;
195}

197template<>
198void SpellHistory::WritePacket<Pet>(WorldPacket& packet) const
199{
200 Clock::time_point now = Clock::now();
201
202 uint8 cooldownsCount = _spellCooldowns.size();
203 packet << uint8(cooldownsCount);
204
205 for (auto const& spellCooldown : _spellCooldowns)
206 {
207 packet << uint32(spellCooldown.first); // spell ID
208 packet << uint16(spellCooldown.second.CategoryId); // spell category
209 if (!spellCooldown.second.OnHold)
210 {
211 std::chrono::milliseconds cooldownDuration = std::chrono::duration_cast<std::chrono::milliseconds>(spellCooldown.second.CooldownEnd - now);
212 if (cooldownDuration.count() <= 0)
213 {
214 packet << uint32(0);
215 packet << uint32(0);
216 continue;
217 }
218
219 std::chrono::milliseconds categoryDuration = std::chrono::duration_cast<std::chrono::milliseconds>(spellCooldown.second.CategoryEnd - now);
220 if (categoryDuration.count() > 0)
221 {
222 packet << uint32(0);
223 packet << uint32(categoryDuration.count());
224 }
225 else
226 {
227 packet << uint32(cooldownDuration.count());
228 packet << uint32(0);
229 }
230 }
231 }
232}

234template<>
235void SpellHistory::WritePacket<Player>(WorldPacket& packet) const
236{
237 Clock::time_point now = Clock::now();
238 Clock::time_point infTime = now + InfinityCooldownDelayCheck;
239
240 packet << uint16(_spellCooldowns.size());
241
242 for (auto const& spellCooldown : _spellCooldowns)
243 {
244 packet << uint32(spellCooldown.first);
245 packet << uint16(spellCooldown.second.ItemId); // cast item id
246 packet << uint16(spellCooldown.second.CategoryId); // spell category
247
248 // send infinity cooldown in special format
249 if (spellCooldown.second.CooldownEnd >= infTime)
250 {
251 packet << uint32(1); // cooldown
252 packet << uint32(0x80000000); // category cooldown
253 continue;
254 }
255
256 std::chrono::milliseconds cooldownDuration = std::chrono::duration_cast<std::chrono::milliseconds>(spellCooldown.second.CooldownEnd - now);
257 if (cooldownDuration.count() <= 0)
258 {
259 packet << uint32(0);
260 packet << uint32(0);
261 continue;
262 }
263
264 std::chrono::milliseconds categoryDuration = std::chrono::duration_cast<std::chrono::milliseconds>(spellCooldown.second.CategoryEnd - now);
265 if (categoryDuration.count() >= 0)
266 {
267 packet << uint32(0); // cooldown
268 packet << uint32(categoryDuration.count()); // category cooldown
269 }
270 else
271 {
272 packet << uint32(cooldownDuration.count()); // cooldown
273 packet << uint32(0); // category cooldown
274 }
275 }

278void SpellHistory::StartCooldown(SpellInfo const* spellInfo, uint32 itemId, Spell* spell /*= nullptr*/, bool onHold /*= false*/)
279{
280 // init cooldown values
281 uint32 categoryId = 0;
282 int32 cooldown = -1;
283 int32 categoryCooldown = -1;
284
285 GetCooldownDurations(spellInfo, itemId, &cooldown, &categoryId, &categoryCooldown);
286
287 Clock::time_point curTime = Clock::now();
288 Clock::time_point catrecTime;
289 Clock::time_point recTime;
290 bool needsCooldownPacket = false;
291
292 // overwrite time for selected category
293 if (onHold)
294 {
295 // use +MONTH as infinite cooldown marker
296 catrecTime = categoryCooldown > 0 ? (curTime + InfinityCooldownDelay) : curTime;
297 recTime = cooldown > 0 ? (curTime + InfinityCooldownDelay) : catrecTime;
298 }
299 else
300 {
301 // shoot spells used equipped item cooldown values already assigned in GetAttackTime(RANGED_ATTACK)
302 // prevent 0 cooldowns set by another way
303 if (cooldown <= 0 && categoryCooldown <= 0 && (categoryId == 76 || (spellInfo->IsAutoRepeatRangedSpell() && spellInfo->Id != 75)))
304 cooldown = _owner->GetAttackTime(RANGED_ATTACK);
305
306 // Now we have cooldown data (if found any), time to apply mods
307 if (Player* modOwner = _owner->GetSpellModOwner())
308 {
309 if (cooldown > 0)
310 modOwner->ApplySpellMod(spellInfo->Id, SPELLMOD_COOLDOWN, cooldown, spell);
311
312 if (categoryCooldown > 0 && !spellInfo->HasAttribute(SPELL_ATTR6_IGNORE_CATEGORY_COOLDOWN_MODS))
313 modOwner->ApplySpellMod(spellInfo->Id, SPELLMOD_COOLDOWN, categoryCooldown, spell);
314 }
315
316 if (int32 cooldownMod = _owner->GetTotalAuraModifier(SPELL_AURA_MOD_COOLDOWN))
317 {
318 // Apply SPELL_AURA_MOD_COOLDOWN only to own spells
319 Player* playerOwner = GetPlayerOwner();
320 if (!playerOwner || playerOwner->HasSpell(spellInfo->Id))
321 {
322 needsCooldownPacket = true;
323 cooldown += cooldownMod * IN_MILLISECONDS; // SPELL_AURA_MOD_COOLDOWN does not affect category cooldows, verified with shaman shocks
324 }
325 }
326
327 // replace negative cooldowns by 0
328 if (cooldown < 0)
329 cooldown = 0;
330
331 if (categoryCooldown < 0)
332 categoryCooldown = 0;
333
334 // no cooldown after applying spell mods
335 if (cooldown == 0 && categoryCooldown == 0)
336 return;
337
338 catrecTime = categoryCooldown ? curTime + std::chrono::duration_cast<Clock::duration>(std::chrono::milliseconds(categoryCooldown)) : curTime;
339 recTime = cooldown ? curTime + std::chrono::duration_cast<Clock::duration>(std::chrono::milliseconds(cooldown)) : catrecTime;
340 }
341
342 // self spell cooldown
343 if (recTime != curTime)
344 {
345 AddCooldown(spellInfo->Id, itemId, recTime, categoryId, catrecTime, onHold);
346
347 if (needsCooldownPacket)
348 {
349 if (Player* playerOwner = GetPlayerOwner())
350 {
351 WorldPacket spellCooldown;
352 BuildCooldownPacket(spellCooldown, SPELL_COOLDOWN_FLAG_NONE, spellInfo->Id, cooldown);
353 playerOwner->SendDirectMessage(&spellCooldown);
354 }
355 }
356 }
357}

359void SpellHistory::SendCooldownEvent(SpellInfo const* spellInfo, uint32 itemId /*= 0*/, Spell* spell /*= nullptr*/, bool startCooldown /*= true*/)
360{
361 // Send activate cooldown timer (possible 0) at client side
362 if (Player* player = GetPlayerOwner())
363 {
364 uint32 category = spellInfo->GetCategory();
365 GetCooldownDurations(spellInfo, itemId, nullptr, &category, nullptr);
366
367 auto categoryItr = _categoryCooldowns.find(category);
368 if (categoryItr != _categoryCooldowns.end() && categoryItr->second->SpellId != spellInfo->Id)
369 {
370 WorldPacket data(SMSG_COOLDOWN_EVENT, 4 + 8);
371 data << uint32(categoryItr->second->SpellId);
372 data << uint64(_owner->GetGUID());
373 player->SendDirectMessage(&data);
374
375 if (startCooldown)
376 StartCooldown(sSpellMgr->EnsureSpellInfo(categoryItr->second->SpellId), itemId, spell);
377 }
378
379 WorldPacket data(SMSG_COOLDOWN_EVENT, 4 + 8);
380 data << uint32(spellInfo->Id);
381 data << uint64(_owner->GetGUID());
382 player->SendDirectMessage(&data);
383 }
384
385 // start cooldowns at server side, if any
386 if (startCooldown)
387 StartCooldown(spellInfo, itemId, spell);
388}

390void SpellHistory::AddCooldown(uint32 spellId, uint32 itemId, Clock::time_point cooldownEnd, uint32 categoryId, Clock::time_point categoryEnd, bool onHold /*= false*/)
391{
392 CooldownEntry& cooldownEntry = _spellCooldowns[spellId];
393 cooldownEntry.SpellId = spellId;
394 cooldownEntry.CooldownEnd = cooldownEnd;
395 cooldownEntry.ItemId = itemId;
396 cooldownEntry.CategoryId = categoryId;
397 cooldownEntry.CategoryEnd = categoryEnd;
398 cooldownEntry.OnHold = onHold;
399
400 if (categoryId)
401 _categoryCooldowns[categoryId] = &cooldownEntry;
402}

404void SpellHistory::ModifyCooldown(uint32 spellId, int32 cooldownModMs)
405{
406 auto itr = _spellCooldowns.find(spellId);
407 if (!cooldownModMs || itr == _spellCooldowns.end())
408 return;
409
410 Clock::time_point now = Clock::now();
411 Clock::duration offset = std::chrono::duration_cast<Clock::duration>(std::chrono::milliseconds(cooldownModMs));
412 if (itr->second.CooldownEnd + offset > now)
413 itr->second.CooldownEnd += offset;
414 else
415 EraseCooldown(itr);
416
417 if (Player* playerOwner = GetPlayerOwner())
418 {
419 WorldPacket modifyCooldown(SMSG_MODIFY_COOLDOWN, 4 + 8 + 4);
420 modifyCooldown << uint32(spellId);
421 modifyCooldown << uint64(_owner->GetGUID());
422 modifyCooldown << int32(cooldownModMs);
423 playerOwner->SendDirectMessage(&modifyCooldown);
424 }
425}

452void SpellHistory::ResetAllCooldowns()
453{
454 if (GetPlayerOwner())
455 {
456 std::vector<int32> cooldowns;
457 cooldowns.reserve(_spellCooldowns.size());
458 for (auto const& p : _spellCooldowns)

468bool SpellHistory::HasCooldown(SpellInfo const* spellInfo, uint32 itemId /*= 0*/, bool ignoreCategoryCooldown /*= false*/) const
469{
470 if (_spellCooldowns.count(spellInfo->Id) != 0)
471 return true;
472
473 if (ignoreCategoryCooldown)
474 return false;
475
476 uint32 category = 0;
477 GetCooldownDurations(spellInfo, itemId, nullptr, &category, nullptr);
478 if (!category)
479 return false;
480
481 return _categoryCooldowns.count(category) != 0;
482}

489uint32 SpellHistory::GetRemainingCooldown(SpellInfo const* spellInfo) const
490{
491 Clock::time_point end;
492 auto itr = _spellCooldowns.find(spellInfo->Id);
493 if (itr != _spellCooldowns.end())
494 end = itr->second.CooldownEnd;
495 else
496 {
497 auto catItr = _categoryCooldowns.find(spellInfo->GetCategory());
498 if (catItr == _categoryCooldowns.end())
499 return 0;
500
501 end = catItr->second->CategoryEnd;
502 }
503
504 Clock::time_point now = Clock::now();
505 if (end < now)
506 return 0;
507
508 Clock::duration remaining = end - now;
509 return uint32(std::chrono::duration_cast<std::chrono::milliseconds>(remaining).count());
510}

512void SpellHistory::LockSpellSchool(SpellSchoolMask schoolMask, uint32 lockoutTime)
513{
514 Clock::time_point now = Clock::now();
515 Clock::time_point lockoutEnd = now + std::chrono::duration_cast<Clock::duration>(std::chrono::milliseconds(lockoutTime));
516 for (uint32 i = 0; i < MAX_SPELL_SCHOOL; ++i)
517 if (SpellSchoolMask(1 << i) & schoolMask)
518 _schoolLockouts[i] = lockoutEnd;
519
520 std::set<uint32> knownSpells;
521 if (Player* plrOwner = _owner->ToPlayer())
522 {
523 for (auto const& p : plrOwner->GetSpellMap())
524 if (p.second->state != PLAYERSPELL_REMOVED)
525 knownSpells.insert(p.first);
526 }
527 else if (Pet* petOwner = _owner->ToPet())
528 {
529 for (auto const& p : petOwner->m_spells)
530 if (p.second.state != PETSPELL_REMOVED)
531 knownSpells.insert(p.first);
532 }
533 else
534 {
535 Creature* creatureOwner = _owner->ToCreature();
536 for (uint8 i = 0; i < CREATURE_MAX_SPELLS; ++i)
537 if (creatureOwner->m_spells[i])
538 knownSpells.insert(creatureOwner->m_spells[i]);
539 }
540
541 PacketCooldowns cooldowns;
542 WorldPacket spellCooldowns;
543 for (uint32 spellId : knownSpells)
544 {
545 SpellInfo const* spellInfo = sSpellMgr->EnsureSpellInfo(spellId);
546 if (spellInfo->IsCooldownStartedOnEvent())
547 continue;
548
549 if (spellInfo->PreventionType != SPELL_PREVENTION_TYPE_SILENCE)
550 continue;
551
552 if ((schoolMask & spellInfo->GetSchoolMask()) && GetRemainingCooldown(spellInfo) < lockoutTime)
553 {
554 cooldowns[spellId] = lockoutTime;
555 AddCooldown(spellId, 0, lockoutEnd, 0, now);
556 }
557 }
558
559 if (Player* player = GetPlayerOwner())
560 {
561 if (!cooldowns.empty())
562 {
563 BuildCooldownPacket(spellCooldowns, SPELL_COOLDOWN_FLAG_NONE, cooldowns);
564 player->SendDirectMessage(&spellCooldowns);
565 }
566 }
567}

569bool SpellHistory::IsSchoolLocked(SpellSchoolMask schoolMask) const
570{
571 Clock::time_point now = Clock::now();
572 for (uint32 i = 0; i < MAX_SPELL_SCHOOL; ++i)
573 if (SpellSchoolMask(1 << i) & schoolMask)
574 if (_schoolLockouts[i] > now)
575 return true;
576
577 return false;
578}

636void SpellHistory::GetCooldownDurations(SpellInfo const* spellInfo, uint32 itemId, int32* cooldown, uint32* categoryId, int32* categoryCooldown)
637{
638 ASSERT(cooldown || categoryId || categoryCooldown);
639 int32 tmpCooldown = -1;
640 uint32 tmpCategoryId = 0;
641 int32 tmpCategoryCooldown = -1;
642
643 // some special item spells without correct cooldown in SpellInfo
644 // cooldown information stored in item prototype
645 if (itemId)
646 {
647 if (ItemTemplate const* proto = sObjectMgr->GetItemTemplate(itemId))
648 {
649 for (uint8 idx = 0; idx < MAX_ITEM_PROTO_SPELLS; ++idx)
650 {
651 if (uint32(proto->Spells[idx].SpellId) == spellInfo->Id)
652 {
653 tmpCooldown = proto->Spells[idx].SpellCooldown;
654 tmpCategoryId = proto->Spells[idx].SpellCategory;
655 tmpCategoryCooldown = proto->Spells[idx].SpellCategoryCooldown;
656 break;
657 }
658 }
659 }
660 }
661
662 // if no cooldown found above then base at DBC data
663 if (tmpCooldown < 0 && tmpCategoryCooldown < 0)
664 {
665 tmpCooldown = spellInfo->RecoveryTime;
666 tmpCategoryId = spellInfo->GetCategory();
667 tmpCategoryCooldown = spellInfo->CategoryRecoveryTime;
668 }
669
670 if (cooldown)
671 *cooldown = tmpCooldown;
672 if (categoryId)
673 *categoryId = tmpCategoryId;
674 if (categoryCooldown)
675 *categoryCooldown = tmpCategoryCooldown;
676}

683void SpellHistory::RestoreCooldownStateAfterDuel()
684{
685 // category cooldows are not preserved.
686 if (Player* player = _owner->ToPlayer())
687 {
688 // add all profession CDs created while in duel (if any)
689 for (auto itr = _spellCooldowns.begin(); itr != _spellCooldowns.end(); ++itr)
690 {
691 SpellInfo const* spellInfo = sSpellMgr->EnsureSpellInfo(itr->first);
692
693 if (spellInfo->RecoveryTime > 10 * MINUTE * IN_MILLISECONDS ||
694 spellInfo->CategoryRecoveryTime > 10 * MINUTE * IN_MILLISECONDS)
695 _spellCooldownsBeforeDuel[itr->first] = _spellCooldowns[itr->first];
696 }
697
698 // check for spell with onHold active before and during the duel
699 for (auto itr = _spellCooldownsBeforeDuel.begin(); itr != _spellCooldownsBeforeDuel.end(); ++itr)
700 {
701 if (!itr->second.OnHold && !_spellCooldowns[itr->first].OnHold)
702 _spellCooldowns[itr->first] = _spellCooldownsBeforeDuel[itr->first];
703 }
704
705 // update the client: restore old cooldowns
706 PacketCooldowns cooldowns;
707
708 for (auto itr = _spellCooldowns.begin(); itr != _spellCooldowns.end(); ++itr)
709 {
710 Clock::time_point now = Clock::now();
711 uint32 cooldownDuration = itr->second.CooldownEnd > now ? std::chrono::duration_cast<std::chrono::milliseconds>(itr->second.CooldownEnd - now).count() : 0;
712
713 // cooldownDuration must be between 0 and 10 minutes in order to avoid any visual bugs
714 if (cooldownDuration <= 0 || cooldownDuration > 10 * MINUTE * IN_MILLISECONDS || itr->second.OnHold)
715 continue;
716
717 cooldowns[itr->first] = cooldownDuration;
718 }
719
720 WorldPacket data;
721 BuildCooldownPacket(data, SPELL_COOLDOWN_FLAG_INCLUDE_EVENT_COOLDOWNS, cooldowns);
722 player->SendDirectMessage(&data);
723 }
724}

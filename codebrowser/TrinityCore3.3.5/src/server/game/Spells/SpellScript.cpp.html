60_SpellScript::EffectHook::EffectHook(uint8 _effIndex)
61{
62 // effect index must be in range <0;2>, allow use of special effindexes
63 ASSERT(_effIndex == EFFECT_ALL || _effIndex == EFFECT_FIRST_FOUND || _effIndex < MAX_SPELL_EFFECTS);
64 effIndex = _effIndex;
65}
66
67uint8 _SpellScript::EffectHook::GetAffectedEffectsMask(SpellInfo const* spellEntry)
68{
69 uint8 mask = 0;
70 if ((effIndex == EFFECT_ALL) || (effIndex == EFFECT_FIRST_FOUND))
71 {
72 for (uint8 i = 0; i < MAX_SPELL_EFFECTS; ++i)
73 {
74 if ((effIndex == EFFECT_FIRST_FOUND) && mask)
75 return mask;
76 if (CheckEffect(spellEntry, i))
77 mask |= (uint8)1<<i;
78 }
79 }
80 else
81 {
82 if (CheckEffect(spellEntry, effIndex))
83 mask |= (uint8)1<<effIndex;
84 }
85 return mask;
86}

88bool _SpellScript::EffectHook::IsEffectAffected(SpellInfo const* spellEntry, uint8 effIndexToCheck)
89{
90 return (GetAffectedEffectsMask(spellEntry) & 1 << effIndexToCheck) != 0;
91}

216bool SpellScript::TargetHook::CheckEffect(SpellInfo const* spellEntry, uint8 effIndexToCheck)
217{
218 if (!targetType)
219 return false;
220
221 if (spellEntry->Effects[effIndexToCheck].TargetA.GetTarget() != targetType &&
222 spellEntry->Effects[effIndexToCheck].TargetB.GetTarget() != targetType)
223 return false;
224
225 SpellImplicitTargetInfo targetInfo(targetType);
226 switch (targetInfo.GetSelectionCategory())
227 {
228 case TARGET_SELECT_CATEGORY_CHANNEL: // SINGLE
229 return !area;
230 case TARGET_SELECT_CATEGORY_NEARBY: // BOTH
231 return true;
232 case TARGET_SELECT_CATEGORY_CONE: // AREA
233 case TARGET_SELECT_CATEGORY_AREA: // AREA
234 return area;
235 case TARGET_SELECT_CATEGORY_DEFAULT:
236 switch (targetInfo.GetObjectType())
237 {
238 case TARGET_OBJECT_TYPE_SRC: // EMPTY
239 return false;
240 case TARGET_OBJECT_TYPE_DEST: // DEST
241 return dest;
242 default:
243 switch (targetInfo.GetReferenceType())
244 {
245 case TARGET_REFERENCE_TYPE_CASTER: // SINGLE
246 return !area;
247 case TARGET_REFERENCE_TYPE_TARGET: // BOTH
248 return true;
249 default:
250 break;
251 }
252 break;
253 }
254 break;
255 default:
256 break;
257 }
258
259 return false;
260}

295bool SpellScript::_Validate(SpellInfo const* entry)
296{
297 for (std::list<EffectHandler>::iterator itr = OnEffectLaunch.begin(); itr != OnEffectLaunch.end(); ++itr)
298 if (!(*itr).GetAffectedEffectsMask(entry))
299 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnEffectLaunch` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
300
301 for (std::list<EffectHandler>::iterator itr = OnEffectLaunchTarget.begin(); itr != OnEffectLaunchTarget.end(); ++itr)
302 if (!(*itr).GetAffectedEffectsMask(entry))
303 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnEffectLaunchTarget` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
304
305 for (std::list<EffectHandler>::iterator itr = OnEffectHit.begin(); itr != OnEffectHit.end(); ++itr)
306 if (!(*itr).GetAffectedEffectsMask(entry))
307 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnEffectHit` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
308
309 for (std::list<EffectHandler>::iterator itr = OnEffectHitTarget.begin(); itr != OnEffectHitTarget.end(); ++itr)
310 if (!(*itr).GetAffectedEffectsMask(entry))
311 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnEffectHitTarget` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
312
313 for (std::list<EffectHandler>::iterator itr = OnEffectSuccessfulDispel.begin(); itr != OnEffectSuccessfulDispel.end(); ++itr)
314 if (!(*itr).GetAffectedEffectsMask(entry))
315 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnEffectSuccessfulDispel` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
316
317 for (std::list<ObjectAreaTargetSelectHandler>::iterator itr = OnObjectAreaTargetSelect.begin(); itr != OnObjectAreaTargetSelect.end(); ++itr)
318 if (!(*itr).GetAffectedEffectsMask(entry))
319 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnObjectAreaTargetSelect` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
320
321 for (std::list<ObjectTargetSelectHandler>::iterator itr = OnObjectTargetSelect.begin(); itr != OnObjectTargetSelect.end(); ++itr)
322 if (!(*itr).GetAffectedEffectsMask(entry))
323 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnObjectTargetSelect` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
324
325 for (std::list<DestinationTargetSelectHandler>::iterator itr = OnDestinationTargetSelect.begin(); itr != OnDestinationTargetSelect.end(); ++itr)
326 if (!(*itr).GetAffectedEffectsMask(entry))
327 TC_LOG_ERROR("scripts", "Spell `%u` Effect `%s` of script `%s` did not match dbc effect data - handler bound to hook `OnDestinationTargetSelect` of SpellScript won't be executed", entry->Id, (*itr).ToString().c_str(), m_scriptName->c_str());
328
329 return _SpellScript::_Validate(entry);
330}

361bool SpellScript::IsInTargetHook() const
362{
363 switch (m_currentScriptState)
364 {
365 case SPELL_SCRIPT_HOOK_EFFECT_LAUNCH_TARGET:
366 case SPELL_SCRIPT_HOOK_EFFECT_HIT_TARGET:
367 case SPELL_SCRIPT_HOOK_BEFORE_HIT:
368 case SPELL_SCRIPT_HOOK_HIT:
369 case SPELL_SCRIPT_HOOK_AFTER_HIT:
370 return true;
371 }
372 return false;
373}

375{

377}

380{

642bool AuraScript::_Validate(SpellInfo const* entry)
643{
644 for (std::list<CheckAreaTargetHandler>::iterator itr = DoCheckAreaTarget.begin(); itr != DoCheckAreaTarget.end(); ++itr)
645 if (!entry->HasAreaAuraEffect() && !entry->HasEffect(SPELL_EFFECT_PERSISTENT_AREA_AURA) && !entry->HasEffect(SPELL_EFFECT_APPLY_AURA))

913void AuraScript::_PrepareScriptCall(AuraScriptHookType hookType, AuraApplication const* aurApp)
914{
915 m_scriptStates.push(ScriptStateStore(m_currentScriptState, m_auraApplication, m_defaultActionPrevented));
916 m_currentScriptState = hookType;
917 m_defaultActionPrevented = false;
918 m_auraApplication = aurApp;
919}
920
921void AuraScript::_FinishScriptCall()
922{
923 ScriptStateStore stateStore = m_scriptStates.top();
924 m_currentScriptState = stateStore._currentScriptState;
925 m_auraApplication = stateStore._auraApplication;
926 m_defaultActionPrevented = stateStore._defaultActionPrevented;
927 m_scriptStates.pop();
928}

930bool AuraScript::_IsDefaultActionPrevented()
931{
932 switch (m_currentScriptState)
933 {
934 case AURA_SCRIPT_HOOK_EFFECT_APPLY:
935 case AURA_SCRIPT_HOOK_EFFECT_REMOVE:
936 case AURA_SCRIPT_HOOK_EFFECT_PERIODIC:
937 case AURA_SCRIPT_HOOK_EFFECT_ABSORB:
938 case AURA_SCRIPT_HOOK_EFFECT_SPLIT:
939 case AURA_SCRIPT_HOOK_PREPARE_PROC:
940 case AURA_SCRIPT_HOOK_PROC:
941 case AURA_SCRIPT_HOOK_EFFECT_PROC:
942 return m_defaultActionPrevented;
943 default:
944 ASSERT(false && "AuraScript::_IsDefaultActionPrevented is called in a wrong place");
945 return false;
946 }
947}

950{
951 switch (m_currentScriptState)
952 {
953 case AURA_SCRIPT_HOOK_EFFECT_APPLY:
954 case AURA_SCRIPT_HOOK_EFFECT_REMOVE:
955 case AURA_SCRIPT_HOOK_EFFECT_PERIODIC:
956 case AURA_SCRIPT_HOOK_EFFECT_ABSORB:

25void WorldSession::HandleJoinChannel(WorldPacket& recvPacket)
26{
27 uint32 channelId;
28 uint8 unknown1, unknown2;
29 std::string channelName, password;
30
31 recvPacket >> channelId >> unknown1 >> unknown2 >> channelName >> password;
32
33 TC_LOG_DEBUG("chat.system", "CMSG_JOIN_CHANNEL %s Channel: %u, unk1: %u, unk2: %u, channel: %s, password: %s",
34 GetPlayerInfo().c_str(), channelId, unknown1, unknown2, channelName.c_str(), password.c_str());
35
36 if (channelId)
37 {
38 ChatChannelsEntry const* channel = sChatChannelsStore.LookupEntry(channelId);
39 if (!channel)
40 return;
41
42 AreaTableEntry const* zone = sAreaTableStore.LookupEntry(GetPlayer()->GetZoneId());
43 if (!zone || !GetPlayer()->CanJoinConstantChannelInZone(channel, zone))
44 return;
45 }
46
47 if (channelName.empty())
48 return;
49
50 if (isdigit(channelName[0]))
51 return;
52
53 if (ChannelMgr* cMgr = ChannelMgr::forTeam(GetPlayer()->GetTeam()))
54 {
55 cMgr->setTeam(GetPlayer()->GetTeam());
56 if (Channel* channel = cMgr->GetJoinChannel(channelName, channelId))
57 channel->JoinChannel(GetPlayer(), password);
58 }
59}

61void WorldSession::HandleLeaveChannel(WorldPacket& recvPacket)
62{
63 uint32 unk;
64 std::string channelName;
65 recvPacket >> unk >> channelName;
66
67 TC_LOG_DEBUG("chat.system", "CMSG_LEAVE_CHANNEL %s Channel: %s, unk1: %u",
68 GetPlayerInfo().c_str(), channelName.c_str(), unk);
69
70 if (channelName.empty())
71 return;
72
73 if (ChannelMgr* cMgr = ChannelMgr::forTeam(GetPlayer()->GetTeam()))
74 {
75 if (Channel* channel = cMgr->GetChannel(channelName, GetPlayer()))
76 channel->LeaveChannel(GetPlayer(), true);
77 cMgr->LeftChannel(channelName);
78 }
79}

287void WorldSession::HandleGetChannelMemberCount(WorldPacket &recvPacket)
288{
289 std::string channelName;
290 recvPacket >> channelName;
291
292 TC_LOG_DEBUG("chat.system", "CMSG_GET_CHANNEL_MEMBER_COUNT %s Channel: %s",
293 GetPlayerInfo().c_str(), channelName.c_str());
294
295 if (ChannelMgr* cMgr = ChannelMgr::forTeam(GetPlayer()->GetTeam()))
296 {
297 if (Channel* channel = cMgr->GetChannel(channelName, GetPlayer()))
298 {
299 TC_LOG_DEBUG("chat.system", "SMSG_CHANNEL_MEMBER_COUNT %s Channel: %s Count: %u",
300 GetPlayerInfo().c_str(), channelName.c_str(), channel->GetNumPlayers());
301
302 WorldPacket data(SMSG_CHANNEL_MEMBER_COUNT, channel->GetName().size() + 1 + 4);
303 data << channel->GetName();
304 data << uint8(channel->GetFlags());
305 data << uint32(channel->GetNumPlayers());
306 SendPacket(&data);
307 }
308 }

126void WorldSession::SendTrainerList(ObjectGuid guid, const std::string& strTitle)
127{
128 TC_LOG_DEBUG("network", "WORLD: SendTrainerList");
129
130 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
131 if (!unit)
132 {
133 TC_LOG_DEBUG("network", "WORLD: SendTrainerList - %s not found or you can not interact with him.", guid.ToString().c_str());
134 return;
135 }
136
137 // remove fake death
138 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
139 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
140
141 TrainerSpellData const* trainer_spells = unit->GetTrainerSpells();
142 if (!trainer_spells)
143 {
144 TC_LOG_DEBUG("network", "WORLD: SendTrainerList - Training spells not found for %s", guid.ToString().c_str());
145 return;
146 }
147
148 WorldPacket data(SMSG_TRAINER_LIST, 8+4+4+trainer_spells->spellList.size()*38 + strTitle.size()+1);
149 data << guid;
150 data << uint32(trainer_spells->trainerType);
151
152 size_t count_pos = data.wpos();
153 data << uint32(trainer_spells->spellList.size());
154
155 // reputation discount
156 float fDiscountMod = _player->GetReputationPriceDiscount(unit);
157 bool can_learn_primary_prof = GetPlayer()->GetFreePrimaryProfessionPoints() > 0;
158
159 uint32 count = 0;
160 for (TrainerSpellMap::const_iterator itr = trainer_spells->spellList.begin(); itr != trainer_spells->spellList.end(); ++itr)
161 {
162 TrainerSpell const* tSpell = &itr->second;
163
164 bool valid = true;
165 bool primary_prof_first_rank = false;
166 for (uint8 i = 0; i < MAX_SPELL_EFFECTS; ++i)
167 {
168 if (!tSpell->learnedSpell[i])
169 continue;
170 if (!_player->IsSpellFitByClassAndRace(tSpell->learnedSpell[i]))
171 {
172 valid = false;
173 break;
174 }
175 SpellInfo const* learnedSpellInfo = sSpellMgr->GetSpellInfo(tSpell->learnedSpell[i]);
176 if (learnedSpellInfo && learnedSpellInfo->IsPrimaryProfessionFirstRank())
177 primary_prof_first_rank = true;
178 }
179 if (!valid)
180 continue;
181
182 TrainerSpellState state = _player->GetTrainerSpellState(tSpell);
183
184 data << uint32(tSpell->spell); // learned spell (or cast-spell in profession case)
185 data << uint8(state == TRAINER_SPELL_GREEN_DISABLED ? TRAINER_SPELL_GREEN : state);
186 data << uint32(floor(tSpell->spellCost * fDiscountMod));
187
188 data << uint32(primary_prof_first_rank && can_learn_primary_prof ? 1 : 0);
189 // primary prof. learn confirmation dialog
190 data << uint32(primary_prof_first_rank ? 1 : 0); // must be equal prev. field to have learn button in enabled state
191 data << uint8(tSpell->reqLevel);
192 data << uint32(tSpell->reqSkill);
193 data << uint32(tSpell->reqSkillValue);
194 //prev + req or req + 0
195 uint8 maxReq = 0;
196 for (uint8 i = 0; i < MAX_SPELL_EFFECTS; ++i)
197 {
198 if (!tSpell->learnedSpell[i])
199 continue;
200 if (uint32 prevSpellId = sSpellMgr->GetPrevSpellInChain(tSpell->learnedSpell[i]))
201 {
202 data << uint32(prevSpellId);
203 ++maxReq;
204 }
205 if (maxReq == 3)
206 break;
207 SpellsRequiringSpellMapBounds spellsRequired = sSpellMgr->GetSpellsRequiredForSpellBounds(tSpell->learnedSpell[i]);
208 for (SpellsRequiringSpellMap::const_iterator itr2 = spellsRequired.first; itr2 != spellsRequired.second && maxReq < 3; ++itr2)
209 {
210 data << uint32(itr2->second);
211 ++maxReq;
212 }
213 if (maxReq == 3)
214 break;
215 }
216 while (maxReq < 3)
217 {
218 data << uint32(0);
219 ++maxReq;
220 }
221
222 ++count;
223 }
224
225 data << strTitle;
226
227 data.put<uint32>(count_pos, count);
228 SendPacket(&data);
229}

231void WorldSession::HandleTrainerBuySpellOpcode(WorldPacket& recvData)
232{
233 ObjectGuid guid;
234 uint32 spellId = 0;
235
236 recvData >> guid >> spellId;
237 TC_LOG_DEBUG("network", "WORLD: Received CMSG_TRAINER_BUY_SPELL %s, learn spell id is: %u", guid.ToString().c_str(), spellId);
238
239 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
240 if (!unit)
241 {
242 TC_LOG_DEBUG("network", "WORLD: HandleTrainerBuySpellOpcode - %s not found or you can not interact with him.", guid.ToString().c_str());
243 return;
244 }
245
246 // remove fake death
247 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
248 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
249
250 // check present spell in trainer spell list
251 TrainerSpellData const* trainer_spells = unit->GetTrainerSpells();
252 if (!trainer_spells)
253 return;
254
255 // not found, cheat?
256 TrainerSpell const* trainer_spell = trainer_spells->Find(spellId);
257 if (!trainer_spell)
258 return;
259
260 // can't be learn, cheat? Or double learn with lags...
261 if (_player->GetTrainerSpellState(trainer_spell) != TRAINER_SPELL_GREEN)
262 return;
263
264 // apply reputation discount
265 uint32 nSpellCost = uint32(floor(trainer_spell->spellCost * _player->GetReputationPriceDiscount(unit)));
266
267 // check money requirement
268 if (!_player->HasEnoughMoney(nSpellCost))
269 return;
270
271 _player->ModifyMoney(-int32(nSpellCost));
272
273 unit->SendPlaySpellVisual(179); // 53 SpellCastDirected
274 unit->SendPlaySpellImpact(_player->GetGUID(), 362); // 113 EmoteSalute
275
276 // learn explicitly or cast explicitly
277 if (trainer_spell->IsCastable())
278 _player->CastSpell(_player, trainer_spell->spell, true);
279 else
280 _player->LearnSpell(spellId, false);
281
282 WorldPacket data(SMSG_TRAINER_BUY_SUCCEEDED, 12);
283 data << uint64(guid);
284 data << uint32(spellId); // should be same as in packet from client
285 SendPacket(&data);
286}

288void WorldSession::HandleGossipHelloOpcode(WorldPacket& recvData)
289{
290 TC_LOG_DEBUG("network", "WORLD: Received CMSG_GOSSIP_HELLO");
291
292 ObjectGuid guid;
293 recvData >> guid;
294
295 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_GOSSIP);
296 if (!unit)
297 {
298 TC_LOG_DEBUG("network", "WORLD: HandleGossipHelloOpcode - %s not found or you can not interact with him.", guid.ToString().c_str());
299 return;
300 }
301
302 // set faction visible if needed
303 if (FactionTemplateEntry const* factionTemplateEntry = sFactionTemplateStore.LookupEntry(unit->getFaction()))
304 _player->GetReputationMgr().SetVisible(factionTemplateEntry);
305
306 GetPlayer()->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_TALK);
307 // remove fake death
308 //if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
309 // GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
310
311 if (unit->IsArmorer() || unit->IsCivilian() || unit->IsQuestGiver() || unit->IsServiceProvider() || unit->IsGuard())
312 {
313 unit->StopMoving();
314 }
315
316 // If spiritguide, no need for gossip menu, just put player into resurrect queue
317 if (unit->IsSpiritGuide())
318 {
319 Battleground* bg = _player->GetBattleground();
320 if (bg)
321 {
322 bg->AddPlayerToResurrectQueue(unit->GetGUID(), _player->GetGUID());
323 sBattlegroundMgr->SendAreaSpiritHealerQueryOpcode(_player, bg, unit->GetGUID());
324 return;
325 }
326 }
327
328 if (!sScriptMgr->OnGossipHello(_player, unit))
329 {
330// _player->TalkedToCreature(unit->GetEntry(), unit->GetGUID());
331 _player->PrepareGossipMenu(unit, unit->GetCreatureTemplate()->GossipMenuId, true);
332 _player->SendPreparedGossip(unit);
333 }
334 unit->AI()->sGossipHello(_player);
335}

358void WorldSession::SendSpiritResurrect()
359{
360 _player->ResurrectPlayer(0.5f, true);
361 _player->DurabilityLossAll(0.25f, true);
362
363 // get corpse nearest graveyard
364 WorldSafeLocsEntry const* corpseGrave = NULL;
365 WorldLocation corpseLocation = _player->GetCorpseLocation();
366 if (_player->HasCorpse())
367 {
368 corpseGrave = sObjectMgr->GetClosestGraveYard(corpseLocation.GetPositionX(), corpseLocation.GetPositionY(),
369 corpseLocation.GetPositionZ(), corpseLocation.GetMapId(), _player->GetTeam());
370 }
371
372 // now can spawn bones
373 _player->SpawnCorpseBones();
374
375 // teleport to nearest from corpse graveyard, if different from nearest to player ghost
376 if (corpseGrave)
377 {
378 WorldSafeLocsEntry const* ghostGrave = sObjectMgr->GetClosestGraveYard(
379 _player->GetPositionX(), _player->GetPositionY(), _player->GetPositionZ(), _player->GetMapId(), _player->GetTeam());
380
381 if (corpseGrave != ghostGrave)
382 _player->TeleportTo(corpseGrave->map_id, corpseGrave->x, corpseGrave->y, corpseGrave->z, _player->GetOrientation());
383 // or update at original position
384 else
385 _player->UpdateObjectVisibility();
386 }
387 // or update at original position
388 else
389 _player->UpdateObjectVisibility();
390}

466void WorldSession::SendStablePetCallback(PreparedQueryResult result, ObjectGuid guid)
467{
468 if (!GetPlayer())
469 return;
470
471 TC_LOG_DEBUG("network", "WORLD: Recv MSG_LIST_STABLED_PETS Send.");
472
473 WorldPacket data(MSG_LIST_STABLED_PETS, 200); // guess size
474
475 data << uint64(guid);
476
477 Pet* pet = _player->GetPet();
478
479 size_t wpos = data.wpos();
480 data << uint8(0); // place holder for slot show number
481
482 data << uint8(GetPlayer()->m_stableSlots);
483
484 uint8 num = 0; // counter for place holder
485
486 // not let move dead pet in slot
487 if (pet && pet->IsAlive() && pet->getPetType() == HUNTER_PET)
488 {
489 data << uint32(pet->GetCharmInfo()->GetPetNumber());
490 data << uint32(pet->GetEntry());
491 data << uint32(pet->getLevel());
492 data << pet->GetName(); // petname
493 data << uint8(1); // 1 = current, 2/3 = in stable (any from 4, 5, ... create problems with proper show)
494 ++num;
495 }
496
497 if (result)
498 {
499 do
500 {
501 Field* fields = result->Fetch();
502
503 data << uint32(fields[1].GetUInt32()); // petnumber
504 data << uint32(fields[2].GetUInt32()); // creature entry
505 data << uint32(fields[3].GetUInt16()); // level
506 data << fields[4].GetString(); // name
507 data << uint8(2); // 1 = current, 2/3 = in stable (any from 4, 5, ... create problems with proper show)
508
509 ++num;
510 }
511 while (result->NextRow());
512 }
513
514 data.put<uint8>(wpos, num); // set real data to placeholder
515 SendPacket(&data);
516
517}

526void WorldSession::HandleStablePet(WorldPacket& recvData)
527{
528 TC_LOG_DEBUG("network", "WORLD: Recv CMSG_STABLE_PET");
529 ObjectGuid npcGUID;
530
531 recvData >> npcGUID;
532
533 if (!GetPlayer()->IsAlive())
534 {
535 SendStableResult(STABLE_ERR_STABLE);
536 return;
537 }
538
539 if (!CheckStableMaster(npcGUID))
540 {
541 SendStableResult(STABLE_ERR_STABLE);
542 return;
543 }
544
545 // remove fake death
546 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
547 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
548
549 Pet* pet = _player->GetPet();
550
551 // can't place in stable dead pet
552 if (!pet || !pet->IsAlive() || pet->getPetType() != HUNTER_PET)
553 {
554 SendStableResult(STABLE_ERR_STABLE);
555 return;
556 }
557
558 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PET_SLOTS);
559
560 stmt->setUInt32(0, _player->GetGUID().GetCounter());
561 stmt->setUInt8(1, PET_SAVE_FIRST_STABLE_SLOT);
562 stmt->setUInt8(2, PET_SAVE_LAST_STABLE_SLOT);
563
564 _stablePetCallback = CharacterDatabase.AsyncQuery(stmt);
565}

567void WorldSession::HandleStablePetCallback(PreparedQueryResult result)
568{
569 if (!GetPlayer())
570 return;
571
572 uint8 freeSlot = 1;
573 if (result)
574 {
575 do
576 {
577 Field* fields = result->Fetch();
578
579 uint8 slot = fields[1].GetUInt8();
580
581 // slots ordered in query, and if not equal then free
582 if (slot != freeSlot)
583 break;
584
585 // this slot not free, skip
586 ++freeSlot;
587 }
588 while (result->NextRow());
589 }
590
591 WorldPacket data(SMSG_STABLE_RESULT, 1);
592 if (freeSlot > 0 && freeSlot <= GetPlayer()->m_stableSlots)
593 {
594 _player->RemovePet(_player->GetPet(), PetSaveMode(freeSlot));
595 SendStableResult(STABLE_SUCCESS_STABLE);
596 }
597 else
598 SendStableResult(STABLE_ERR_STABLE);
599}

630void WorldSession::HandleUnstablePetCallback(PreparedQueryResult result, uint32 petId)
631{
632 if (!GetPlayer())
633 return;
634
635 uint32 petEntry = 0;
636 if (result)
637 {
638 Field* fields = result->Fetch();
639 petEntry = fields[0].GetUInt32();
640 }
641
642 if (!petEntry)
643 {
644 SendStableResult(STABLE_ERR_STABLE);
645 return;
646 }
647
648 CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate(petEntry);
649 if (!creatureInfo || !creatureInfo->IsTameable(_player->CanTameExoticPets()))
650 {
651 // if problem in exotic pet
652 if (creatureInfo && creatureInfo->IsTameable(true))
653 SendStableResult(STABLE_ERR_EXOTIC);
654 else
655 SendStableResult(STABLE_ERR_STABLE);
656 return;
657 }
658
659 Pet* pet = _player->GetPet();
660 if (pet && pet->IsAlive())
661 {
662 SendStableResult(STABLE_ERR_STABLE);
663 return;
664 }
665
666 // delete dead pet
667 if (pet)
668 _player->RemovePet(pet, PET_SAVE_AS_DELETED);
669
670 Pet* newPet = new Pet(_player, HUNTER_PET);
671 if (!newPet->LoadPetFromDB(_player, petEntry, petId))
672 {
673 delete newPet;
674 newPet = NULL;
675 SendStableResult(STABLE_ERR_STABLE);
676 return;
677 }
678
679 SendStableResult(STABLE_SUCCESS_UNSTABLE);
680}

682void WorldSession::HandleBuyStableSlot(WorldPacket& recvData)
683{
684 TC_LOG_DEBUG("network", "WORLD: Recv CMSG_BUY_STABLE_SLOT.");
685 ObjectGuid npcGUID;
686
687 recvData >> npcGUID;
688
689 if (!CheckStableMaster(npcGUID))
690 {
691 SendStableResult(STABLE_ERR_STABLE);
692 return;
693 }
694
695 // remove fake death
696 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
697 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
698
699 if (GetPlayer()->m_stableSlots < MAX_PET_STABLES)
700 {
701 StableSlotPricesEntry const* SlotPrice = sStableSlotPricesStore.LookupEntry(GetPlayer()->m_stableSlots+1);
702 if (_player->HasEnoughMoney(SlotPrice->Price))
703 {
704 ++GetPlayer()->m_stableSlots;
705 _player->ModifyMoney(-int32(SlotPrice->Price));
706 SendStableResult(STABLE_SUCCESS_BUY_SLOT);

720void WorldSession::HandleStableSwapPet(WorldPacket& recvData)
721{
722 TC_LOG_DEBUG("network", "WORLD: Recv CMSG_STABLE_SWAP_PET.");
723 ObjectGuid npcGUID;
724 uint32 petId;
725
726 recvData >> npcGUID >> petId;
727
728 if (!CheckStableMaster(npcGUID))
729 {
730 SendStableResult(STABLE_ERR_STABLE);
731 return;
732 }
733
734 // remove fake death
735 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
736 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
737
738 Pet* pet = _player->GetPet();
739
740 if (!pet || pet->getPetType() != HUNTER_PET)
741 {
742 SendStableResult(STABLE_ERR_STABLE);
743 return;
744 }
745
746 // Find swapped pet slot in stable
747
748 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PET_SLOT_BY_ID);
749
750 stmt->setUInt32(0, _player->GetGUID().GetCounter());
751 stmt->setUInt32(1, petId);
752
753 _stableSwapCallback.SetParam(petId);
754 _stableSwapCallback.SetFutureResult(CharacterDatabase.AsyncQuery(stmt));
755}

757void WorldSession::HandleStableSwapPetCallback(PreparedQueryResult result, uint32 petId)
758{
759 if (!GetPlayer())
760 return;
761
762 if (!result)
763 {
764 SendStableResult(STABLE_ERR_STABLE);
765 return;
766 }
767
768 Field* fields = result->Fetch();
769
770 uint32 slot = fields[0].GetUInt8();
771 uint32 petEntry = fields[1].GetUInt32();
772
773 if (!petEntry)
774 {
775 SendStableResult(STABLE_ERR_STABLE);
776 return;
777 }
778
779 CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate(petEntry);
780 if (!creatureInfo || !creatureInfo->IsTameable(_player->CanTameExoticPets()))
781 {
782 // if problem in exotic pet
783 if (creatureInfo && creatureInfo->IsTameable(true))
784 SendStableResult(STABLE_ERR_EXOTIC);
785 else
786 SendStableResult(STABLE_ERR_STABLE);
787 return;
788 }
789
790 Pet* pet = _player->GetPet();
791 // The player's pet could have been removed during the delay of the DB callback
792 if (!pet)
793 {
794 SendStableResult(STABLE_ERR_STABLE);
795 return;
796 }
797
798 // move alive pet to slot or delete dead pet
799 _player->RemovePet(pet, pet->IsAlive() ? PetSaveMode(slot) : PET_SAVE_AS_DELETED);
800
801 // summon unstabled pet
802 Pet* newPet = new Pet(_player);
803 if (!newPet->LoadPetFromDB(_player, petEntry, petId))
804 {
805 delete newPet;
806 SendStableResult(STABLE_ERR_STABLE);
807 }
808 else
809 SendStableResult(STABLE_SUCCESS_UNSTABLE);
810}

812void WorldSession::HandleRepairItemOpcode(WorldPacket& recvData)
813{
814 TC_LOG_DEBUG("network", "WORLD: CMSG_REPAIR_ITEM");
815
816 ObjectGuid npcGUID, itemGUID;
817 uint8 guildBank; // new in 2.3.2, bool that means from guild bank money
818
819 recvData >> npcGUID >> itemGUID >> guildBank;
820
821 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(npcGUID, UNIT_NPC_FLAG_REPAIR);
822 if (!unit)
823 {
824 TC_LOG_DEBUG("network", "WORLD: HandleRepairItemOpcode - %s not found or you can not interact with him.", npcGUID.ToString().c_str());
825 return;
826 }
827
828 // remove fake death
829 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
830 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
831
832 // reputation discount
833 float discountMod = _player->GetReputationPriceDiscount(unit);
834
835 if (itemGUID)
836 {
837 TC_LOG_DEBUG("network", "ITEM: Repair %s, at %s", itemGUID.ToString().c_str(), npcGUID.ToString().c_str());
838
839 Item* item = _player->GetItemByGuid(itemGUID);
840 if (item)
841 _player->DurabilityRepair(item->GetPos(), true, discountMod, guildBank != 0);
842 }
843 else
844 {
845 TC_LOG_DEBUG("network", "ITEM: Repair all items at %s", npcGUID.ToString().c_str());
846 _player->DurabilityRepairAll(true, discountMod, guildBank != 0);
847 }
848}

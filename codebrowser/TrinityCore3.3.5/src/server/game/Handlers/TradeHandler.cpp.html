33void WorldSession::SendTradeStatus(TradeStatusInfo const& info)
34{
35 WorldPacket data(SMSG_TRADE_STATUS, 13);
36 data << uint32(info.Status);
37
38 switch (info.Status)
39 {
40 case TRADE_STATUS_BEGIN_TRADE:
41 data << uint64(info.TraderGuid); // CGTradeInfo::m_tradingPlayer
42 break;
43 case TRADE_STATUS_OPEN_WINDOW:
44 data << uint32(0); // CGTradeInfo::m_tradeID
45 break;
46 case TRADE_STATUS_CLOSE_WINDOW:
47 data << uint32(info.Result); // InventoryResult
48 data << uint8(info.IsTargetResult); // bool isTargetError; used for: EQUIP_ERR_BAG_FULL, EQUIP_ERR_CANT_CARRY_MORE_OF_THIS, EQUIP_ERR_MISSING_REAGENT, EQUIP_ERR_ITEM_MAX_LIMIT_CATEGORY_COUNT_EXCEEDED
49 data << uint32(info.ItemLimitCategoryId); // ItemLimitCategory.dbc entry
50 break;
51 case TRADE_STATUS_WRONG_REALM:
52 case TRADE_STATUS_NOT_ON_TAPLIST:
53 data << uint8(info.Slot); // Trade slot; -1 here clears CGTradeInfo::m_tradeMoney
54 break;
55 default:
56 break;
57 }
58
59 SendPacket(&data);
60}

74void WorldSession::SendUpdateTrade(bool trader_data /*= true*/)
75{
76 TradeData* view_trade = trader_data ? _player->GetTradeData()->GetTraderData() : _player->GetTradeData();
77
78 WorldPacket data(SMSG_TRADE_STATUS_EXTENDED, 1+4+4+4+4+4+7*(1+4+4+4+4+8+4+4+4+4+8+4+4+4+4+4+4));
79 data << uint8(trader_data); // 1 means traders data, 0 means own
80 data << uint32(0); // CGTradeInfo::m_tradeID
81 data << uint32(TRADE_SLOT_COUNT); // trade slots count/number?, = next field in most cases
82 data << uint32(TRADE_SLOT_COUNT); // trade slots count/number?, = prev field in most cases
83 data << uint32(view_trade->GetMoney()); // trader gold
84 data << uint32(view_trade->GetSpell()); // spell cast on lowest slot item
85
86 for (uint8 i = 0; i < TRADE_SLOT_COUNT; ++i)
87 {
88 data << uint8(i); // trade slot number, if not specified, then end of packet
89
90 if (Item* item = view_trade->GetItem(TradeSlots(i)))
91 {
92 data << uint32(item->GetTemplate()->ItemId); // entry
93 data << uint32(item->GetTemplate()->DisplayInfoID);// display id
94 data << uint32(item->GetCount()); // stack count
95 // wrapped: hide stats but show giftcreator name
96 data << uint32(item->HasFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_WRAPPED) ? 1 : 0);
97 data << uint64(item->GetGuidValue(ITEM_FIELD_GIFTCREATOR));
98 // perm. enchantment and gems
99 data << uint32(item->GetEnchantmentId(PERM_ENCHANTMENT_SLOT));
100 for (uint32 enchant_slot = SOCK_ENCHANTMENT_SLOT; enchant_slot < SOCK_ENCHANTMENT_SLOT+MAX_GEM_SOCKETS; ++enchant_slot)
101 data << uint32(item->GetEnchantmentId(EnchantmentSlot(enchant_slot)));
102 // creator
103 data << uint64(item->GetGuidValue(ITEM_FIELD_CREATOR));
104 data << uint32(item->GetSpellCharges()); // charges
105 data << uint32(item->GetItemSuffixFactor()); // SuffixFactor
106 data << uint32(item->GetItemRandomPropertyId());// random properties id
107 data << uint32(item->GetTemplate()->LockID); // lock id
108 // max durability
109 data << uint32(item->GetUInt32Value(ITEM_FIELD_MAXDURABILITY));
110 // durability
111 data << uint32(item->GetUInt32Value(ITEM_FIELD_DURABILITY));
112 }
113 else
114 {
115 for (uint8 j = 0; j < 18; ++j)
116 data << uint32(0);
117 }
118 }
119 SendPacket(&data);
120}

125void WorldSession::moveItems(Item* myItems[], Item* hisItems[])
126{
127 Player* trader = _player->GetTrader();
128 if (!trader)
129 return;
130
131 for (uint8 i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
132 {
133 ItemPosCountVec traderDst;
134 ItemPosCountVec playerDst;
135 bool traderCanTrade = (myItems[i] == NULL || trader->CanStoreItem(NULL_BAG, NULL_SLOT, traderDst, myItems[i], false) == EQUIP_ERR_OK);
136 bool playerCanTrade = (hisItems[i] == NULL || _player->CanStoreItem(NULL_BAG, NULL_SLOT, playerDst, hisItems[i], false) == EQUIP_ERR_OK);
137 if (traderCanTrade && playerCanTrade)
138 {
139 // Ok, if trade item exists and can be stored
140 // If we trade in both directions we had to check, if the trade will work before we actually do it
141 // A roll back is not possible after we stored it
142 if (myItems[i])
143 {
144 // logging
145 TC_LOG_DEBUG("network", "partner storing: %u", myItems[i]->GetGUID().GetCounter());
146 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
147 {
148 sLog->outCommand(_player->GetSession()->GetAccountId(), "GM %s (Account: %u) trade: %s (Entry: %d Count: %u) to player: %s (Account: %u)",
149 _player->GetName().c_str(), _player->GetSession()->GetAccountId(),
150 myItems[i]->GetTemplate()->Name1.c_str(), myItems[i]->GetEntry(), myItems[i]->GetCount(),
151 trader->GetName().c_str(), trader->GetSession()->GetAccountId());
152 }
153
154 // adjust time (depends on /played)
155 if (myItems[i]->HasFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_BOP_TRADEABLE))
156 myItems[i]->SetUInt32Value(ITEM_FIELD_CREATE_PLAYED_TIME, trader->GetTotalPlayedTime()-(_player->GetTotalPlayedTime()-myItems[i]->GetUInt32Value(ITEM_FIELD_CREATE_PLAYED_TIME)));
157 // store
158 trader->MoveItemToInventory(traderDst, myItems[i], true, true);
159 }
160 if (hisItems[i])
161 {
162 // logging
163 TC_LOG_DEBUG("network", "player storing: %u", hisItems[i]->GetGUID().GetCounter());
164 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
165 {
166 sLog->outCommand(trader->GetSession()->GetAccountId(), "GM %s (Account: %u) trade: %s (Entry: %d Count: %u) to player: %s (Account: %u)",
167 trader->GetName().c_str(), trader->GetSession()->GetAccountId(),
168 hisItems[i]->GetTemplate()->Name1.c_str(), hisItems[i]->GetEntry(), hisItems[i]->GetCount(),
169 _player->GetName().c_str(), _player->GetSession()->GetAccountId());
170 }
171
172 // adjust time (depends on /played)
173 if (hisItems[i]->HasFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_BOP_TRADEABLE))
174 hisItems[i]->SetUInt32Value(ITEM_FIELD_CREATE_PLAYED_TIME, _player->GetTotalPlayedTime()-(trader->GetTotalPlayedTime()-hisItems[i]->GetUInt32Value(ITEM_FIELD_CREATE_PLAYED_TIME)));
175 // store
176 _player->MoveItemToInventory(playerDst, hisItems[i], true, true);
177 }
178 }
179 else
180 {
181 // in case of fatal error log error message
182 // return the already removed items to the original owner
183 if (myItems[i])
184 {
185 if (!traderCanTrade)
186 TC_LOG_ERROR("network", "trader can't store item: %u", myItems[i]->GetGUID().GetCounter());
187 if (_player->CanStoreItem(NULL_BAG, NULL_SLOT, playerDst, myItems[i], false) == EQUIP_ERR_OK)
188 _player->MoveItemToInventory(playerDst, myItems[i], true, true);
189 else
190 TC_LOG_ERROR("network", "player can't take item back: %u", myItems[i]->GetGUID().GetCounter());
191 }
192 // return the already removed items to the original owner
193 if (hisItems[i])
194 {
195 if (!playerCanTrade)
196 TC_LOG_ERROR("network", "player can't store item: %u", hisItems[i]->GetGUID().GetCounter());
197 if (trader->CanStoreItem(NULL_BAG, NULL_SLOT, traderDst, hisItems[i], false) == EQUIP_ERR_OK)
198 trader->MoveItemToInventory(traderDst, hisItems[i], true, true);
199 else
200 TC_LOG_ERROR("network", "trader can't take item back: %u", hisItems[i]->GetGUID().GetCounter());
201 }
202 }
203 }
204}

208static void setAcceptTradeMode(TradeData* myTrade, TradeData* hisTrade, Item* *myItems, Item* *hisItems)
209{
210 myTrade->SetInAcceptProcess(true);
211 hisTrade->SetInAcceptProcess(true);
212
213 // store items in local list and set 'in-trade' flag
214 for (uint8 i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
215 {
216 if (Item* item = myTrade->GetItem(TradeSlots(i)))
217 {
218 TC_LOG_DEBUG("network", "player trade item %u bag: %u slot: %u", item->GetGUID().GetCounter(), item->GetBagSlot(), item->GetSlot());
219 //Can return NULL
220 myItems[i] = item;
221 myItems[i]->SetInTrade();
222 }
223
224 if (Item* item = hisTrade->GetItem(TradeSlots(i)))
225 {
226 TC_LOG_DEBUG("network", "partner trade item %u bag: %u slot: %u", item->GetGUID().GetCounter(), item->GetBagSlot(), item->GetSlot());
227 hisItems[i] = item;
228 hisItems[i]->SetInTrade();
229 }
230 }
231}

236 hisTrade->SetInAcceptProcess(false);

251void WorldSession::HandleAcceptTradeOpcode(WorldPacket& /*recvPacket*/)
252{
253 TradeData* my_trade = _player->m_trade;
254 if (!my_trade)
255 return;
256
257 Player* trader = my_trade->GetTrader();
258
259 TradeData* his_trade = trader->m_trade;
260 if (!his_trade)
261 return;
262
263 Item* myItems[TRADE_SLOT_TRADED_COUNT] = { NULL, NULL, NULL, NULL, NULL, NULL };
264 Item* hisItems[TRADE_SLOT_TRADED_COUNT] = { NULL, NULL, NULL, NULL, NULL, NULL };
265
266 // set before checks for propertly undo at problems (it already set in to client)
267 my_trade->SetAccepted(true);
268
269 TradeStatusInfo info;
270 if (!_player->IsWithinDistInMap(trader, TRADE_DISTANCE, false))
271 {
272 info.Status = TRADE_STATUS_TARGET_TO_FAR;
273 SendTradeStatus(info);
274 my_trade->SetAccepted(false);
275 return;
276 }
277
278 // not accept case incorrect money amount
279 if (!_player->HasEnoughMoney(my_trade->GetMoney()))
280 {
281 info.Status = TRADE_STATUS_CLOSE_WINDOW;
282 info.Result = EQUIP_ERR_NOT_ENOUGH_MONEY;
283 SendTradeStatus(info);
284 my_trade->SetAccepted(false, true);
285 return;
286 }
287
288 // not accept case incorrect money amount
289 if (!trader->HasEnoughMoney(his_trade->GetMoney()))
290 {
291 info.Status = TRADE_STATUS_CLOSE_WINDOW;
292 info.Result = EQUIP_ERR_NOT_ENOUGH_MONEY;
293 trader->GetSession()->SendTradeStatus(info);
294 his_trade->SetAccepted(false, true);
295 return;
296 }
297
298 if (_player->GetMoney() >= MAX_MONEY_AMOUNT - his_trade->GetMoney())
299 {
300 info.Status = TRADE_STATUS_CLOSE_WINDOW;
301 info.Result = EQUIP_ERR_TOO_MUCH_GOLD;
302 SendTradeStatus(info);
303 my_trade->SetAccepted(false, true);
304 return;
305 }
306
307 if (trader->GetMoney() >= MAX_MONEY_AMOUNT - my_trade->GetMoney())
308 {
309 info.Status = TRADE_STATUS_CLOSE_WINDOW;
310 info.Result = EQUIP_ERR_TOO_MUCH_GOLD;
311 trader->GetSession()->SendTradeStatus(info);
312 his_trade->SetAccepted(false, true);
313 return;
314 }
315
316 // not accept if some items now can't be trade (cheating)
317 for (uint8 i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
318 {
319 if (Item* item = my_trade->GetItem(TradeSlots(i)))
320 {
321 if (!item->CanBeTraded(false, true))
322 {
323 info.Status = TRADE_STATUS_TRADE_CANCELED;
324 SendTradeStatus(info);
325 return;
326 }
327
328 if (item->IsBindedNotWith(trader))
329 {
330 info.Status = TRADE_STATUS_CLOSE_WINDOW;
331 info.Result = EQUIP_ERR_CANNOT_TRADE_THAT;
332 SendTradeStatus(info);
333 return;
334 }
335 }
336
337 if (Item* item = his_trade->GetItem(TradeSlots(i)))
338 {
339 if (!item->CanBeTraded(false, true))
340 {
341 info.Status = TRADE_STATUS_TRADE_CANCELED;
342 SendTradeStatus(info);
343 return;
344 }
345 //if (item->IsBindedNotWith(_player)) // dont mark as invalid when his item isnt good (not exploitable because if item is invalid trade will fail anyway later on the same check)
346 //{
347 // SendTradeStatus(TRADE_STATUS_NOT_ELIGIBLE);
348 // his_trade->SetAccepted(false, true);
349 // return;
350 //}
351 }
352 }
353
354 if (his_trade->IsAccepted())
355 {
356 setAcceptTradeMode(my_trade, his_trade, myItems, hisItems);
357
358 Spell* my_spell = NULL;
359 SpellCastTargets my_targets;
360
361 Spell* his_spell = NULL;
362 SpellCastTargets his_targets;
363
364 // not accept if spell can't be cast now (cheating)
365 if (uint32 my_spell_id = my_trade->GetSpell())
366 {
367 SpellInfo const* spellEntry = sSpellMgr->GetSpellInfo(my_spell_id);
368 Item* castItem = my_trade->GetSpellCastItem();
369
370 if (!spellEntry || !his_trade->GetItem(TRADE_SLOT_NONTRADED) ||
371 (my_trade->HasSpellCastItem() && !castItem))
372 {
373 clearAcceptTradeMode(my_trade, his_trade);
374 clearAcceptTradeMode(myItems, hisItems);
375
376 my_trade->SetSpell(0);
377 return;
378 }
379
380 my_spell = new Spell(_player, spellEntry, TRIGGERED_FULL_MASK);
381 my_spell->m_CastItem = castItem;
382 my_targets.SetTradeItemTarget(_player);
383 my_spell->m_targets = my_targets;
384
385 SpellCastResult res = my_spell->CheckCast(true);
386 if (res != SPELL_CAST_OK)
387 {
388 my_spell->SendCastResult(res);
389
390 clearAcceptTradeMode(my_trade, his_trade);
391 clearAcceptTradeMode(myItems, hisItems);
392
393 delete my_spell;
394 my_trade->SetSpell(0);
395 return;
396 }
397 }
398
399 // not accept if spell can't be cast now (cheating)
400 if (uint32 his_spell_id = his_trade->GetSpell())
401 {
402 SpellInfo const* spellEntry = sSpellMgr->GetSpellInfo(his_spell_id);
403 Item* castItem = his_trade->GetSpellCastItem();
404
405 if (!spellEntry || !my_trade->GetItem(TRADE_SLOT_NONTRADED) || (his_trade->HasSpellCastItem() && !castItem))
406 {
407 delete my_spell;
408 his_trade->SetSpell(0);
409
410 clearAcceptTradeMode(my_trade, his_trade);
411 clearAcceptTradeMode(myItems, hisItems);
412 return;
413 }
414
415 his_spell = new Spell(trader, spellEntry, TRIGGERED_FULL_MASK);
416 his_spell->m_CastItem = castItem;
417 his_targets.SetTradeItemTarget(trader);
418 his_spell->m_targets = his_targets;
419
420 SpellCastResult res = his_spell->CheckCast(true);
421 if (res != SPELL_CAST_OK)
422 {
423 his_spell->SendCastResult(res);
424
425 clearAcceptTradeMode(my_trade, his_trade);
426 clearAcceptTradeMode(myItems, hisItems);
427
428 delete my_spell;
429 delete his_spell;
430
431 his_trade->SetSpell(0);
432 return;
433 }
434 }
435
436 // inform partner client
437 info.Status = TRADE_STATUS_TRADE_ACCEPT;
438 trader->GetSession()->SendTradeStatus(info);
439
440 // test if item will fit in each inventory
441 TradeStatusInfo myCanCompleteInfo, hisCanCompleteInfo;
442 hisCanCompleteInfo.Result = trader->CanStoreItems(myItems, TRADE_SLOT_TRADED_COUNT, &hisCanCompleteInfo.ItemLimitCategoryId);
443 myCanCompleteInfo.Result = _player->CanStoreItems(hisItems, TRADE_SLOT_TRADED_COUNT, &myCanCompleteInfo.ItemLimitCategoryId);
444
445 clearAcceptTradeMode(myItems, hisItems);
446
447 // in case of missing space report error
448 if (myCanCompleteInfo.Result != EQUIP_ERR_OK)
449 {
450 clearAcceptTradeMode(my_trade, his_trade);
451
452 myCanCompleteInfo.Status = TRADE_STATUS_CLOSE_WINDOW;
453 trader->GetSession()->SendTradeStatus(myCanCompleteInfo);
454 myCanCompleteInfo.IsTargetResult = true;
455 SendTradeStatus(myCanCompleteInfo);
456 my_trade->SetAccepted(false);
457 his_trade->SetAccepted(false);
458 delete my_spell;
459 delete his_spell;
460 return;
461 }
462 else if (hisCanCompleteInfo.Result != EQUIP_ERR_OK)
463 {
464 clearAcceptTradeMode(my_trade, his_trade);
465
466 hisCanCompleteInfo.Status = TRADE_STATUS_CLOSE_WINDOW;
467 SendTradeStatus(hisCanCompleteInfo);
468 hisCanCompleteInfo.IsTargetResult = true;
469 trader->GetSession()->SendTradeStatus(hisCanCompleteInfo);
470 my_trade->SetAccepted(false);
471 his_trade->SetAccepted(false);
472 delete my_spell;
473 delete his_spell;
474 return;
475 }
476
477 // execute trade: 1. remove
478 for (uint8 i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
479 {
480 if (myItems[i])
481 {
482 myItems[i]->SetGuidValue(ITEM_FIELD_GIFTCREATOR, _player->GetGUID());
483 _player->MoveItemFromInventory(myItems[i]->GetBagSlot(), myItems[i]->GetSlot(), true);
484 }
485 if (hisItems[i])
486 {
487 hisItems[i]->SetGuidValue(ITEM_FIELD_GIFTCREATOR, trader->GetGUID());
488 trader->MoveItemFromInventory(hisItems[i]->GetBagSlot(), hisItems[i]->GetSlot(), true);
489 }
490 }
491
492 // execute trade: 2. store
493 moveItems(myItems, hisItems);
494
495 // logging money
496 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
497 {
498 if (my_trade->GetMoney() > 0)
499 {
500 sLog->outCommand(_player->GetSession()->GetAccountId(), "GM %s (Account: %u) give money (Amount: %u) to player: %s (Account: %u)",
501 _player->GetName().c_str(), _player->GetSession()->GetAccountId(),
502 my_trade->GetMoney(),
503 trader->GetName().c_str(), trader->GetSession()->GetAccountId());
504 }
505
506 if (his_trade->GetMoney() > 0)
507 {
508 sLog->outCommand(trader->GetSession()->GetAccountId(), "GM %s (Account: %u) give money (Amount: %u) to player: %s (Account: %u)",
509 trader->GetName().c_str(), trader->GetSession()->GetAccountId(),
510 his_trade->GetMoney(),
511 _player->GetName().c_str(), _player->GetSession()->GetAccountId());
512 }
513 }
514
515 // update money
516 _player->ModifyMoney(-int32(my_trade->GetMoney()));
517 _player->ModifyMoney(his_trade->GetMoney());
518 trader->ModifyMoney(-int32(his_trade->GetMoney()));
519 trader->ModifyMoney(my_trade->GetMoney());
520
521 if (my_spell)
522 my_spell->prepare(&my_targets);
523
524 if (his_spell)
525 his_spell->prepare(&his_targets);
526
527 // cleanup
528 clearAcceptTradeMode(my_trade, his_trade);
529 delete _player->m_trade;
530 _player->m_trade = NULL;
531 delete trader->m_trade;
532 trader->m_trade = NULL;
533
534 // desynchronized with the other saves here (SaveInventoryAndGoldToDB() not have own transaction guards)
535 SQLTransaction trans = CharacterDatabase.BeginTransaction();
536 _player->SaveInventoryAndGoldToDB(trans);
537 trader->SaveInventoryAndGoldToDB(trans);
538 CharacterDatabase.CommitTransaction(trans);
539
540 info.Status = TRADE_STATUS_TRADE_COMPLETE;
541 trader->GetSession()->SendTradeStatus(info);
542 SendTradeStatus(info);
543 }
544 else
545 {
546 info.Status = TRADE_STATUS_TRADE_ACCEPT;
547 trader->GetSession()->SendTradeStatus(info);
548 }
549}

589void WorldSession::HandleInitiateTradeOpcode(WorldPacket& recvPacket)
590{
591 ObjectGuid ID;
592 recvPacket >> ID;
593
594 if (GetPlayer()->m_trade)
595 return;
596
597 TradeStatusInfo info;
598 if (!GetPlayer()->IsAlive())
599 {
600 info.Status = TRADE_STATUS_YOU_DEAD;
601 SendTradeStatus(info);
602 return;
603 }
604
605 if (GetPlayer()->HasUnitState(UNIT_STATE_STUNNED))
606 {
607 info.Status = TRADE_STATUS_YOU_STUNNED;
608 SendTradeStatus(info);
609 return;
610 }
611
612 if (isLogingOut())
613 {
614 info.Status = TRADE_STATUS_YOU_LOGOUT;
615 SendTradeStatus(info);
616 return;
617 }
618
619 if (GetPlayer()->IsInFlight())
620 {
621 info.Status = TRADE_STATUS_TARGET_TO_FAR;
622 SendTradeStatus(info);
623 return;
624 }
625
626 if (GetPlayer()->getLevel() < sWorld->getIntConfig(CONFIG_TRADE_LEVEL_REQ))
627 {
628 SendNotification(GetTrinityString(LANG_TRADE_REQ), sWorld->getIntConfig(CONFIG_TRADE_LEVEL_REQ));
629 return;
630 }
631
632 Player* pOther = ObjectAccessor::FindPlayer(ID);
633
634 if (!pOther)
635 {
636 info.Status = TRADE_STATUS_NO_TARGET;
637 SendTradeStatus(info);
638 return;
639 }
640
641 if (pOther == GetPlayer() || pOther->m_trade)
642 {
643 info.Status = TRADE_STATUS_BUSY;
644 SendTradeStatus(info);
645 return;
646 }
647
648 if (!pOther->IsAlive())
649 {
650 info.Status = TRADE_STATUS_TARGET_DEAD;
651 SendTradeStatus(info);
652 return;
653 }
654
655 if (pOther->IsInFlight())
656 {
657 info.Status = TRADE_STATUS_TARGET_TO_FAR;
658 SendTradeStatus(info);
659 return;
660 }
661
662 if (pOther->HasUnitState(UNIT_STATE_STUNNED))
663 {
664 info.Status = TRADE_STATUS_TARGET_STUNNED;
665 SendTradeStatus(info);
666 return;
667 }
668
669 if (pOther->GetSession()->isLogingOut())
670 {
671 info.Status = TRADE_STATUS_TARGET_LOGOUT;
672 SendTradeStatus(info);
673 return;
674 }
675
676 if (pOther->GetSocial()->HasIgnore(GetPlayer()->GetGUID().GetCounter()))
677 {
678 info.Status = TRADE_STATUS_IGNORE_YOU;
679 SendTradeStatus(info);
680 return;
681 }
682
683 if (pOther->GetTeam() != _player->GetTeam() &&
684 (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_TRADE) &&
685 !GetPlayer()->GetSession()->HasPermission(rbac::RBAC_PERM_ALLOW_TWO_SIDE_TRADE)))
686 {
687 info.Status = TRADE_STATUS_WRONG_FACTION;
688 SendTradeStatus(info);
689 return;
690 }
691
692 if (!pOther->IsWithinDistInMap(_player, TRADE_DISTANCE, false))
693 {
694 info.Status = TRADE_STATUS_TARGET_TO_FAR;
695 SendTradeStatus(info);
696 return;
697 }
698
699 if (pOther->getLevel() < sWorld->getIntConfig(CONFIG_TRADE_LEVEL_REQ))
700 {
701 SendNotification(GetTrinityString(LANG_TRADE_OTHER_REQ), sWorld->getIntConfig(CONFIG_TRADE_LEVEL_REQ));
702 return;
703 }
704
705 // OK start trade
706 _player->m_trade = new TradeData(_player, pOther);
707 pOther->m_trade = new TradeData(pOther, _player);
708
709 info.Status = TRADE_STATUS_BEGIN_TRADE;
710 info.TraderGuid = _player->GetGUID();
711 pOther->GetSession()->SendTradeStatus(info);
712}

726void WorldSession::HandleSetTradeItemOpcode(WorldPacket& recvPacket)
727{
728 // send update
729 uint8 tradeSlot;
730 uint8 bag;
731 uint8 slot;
732
733 recvPacket >> tradeSlot;
734 recvPacket >> bag;
735 recvPacket >> slot;
736
737 TradeData* my_trade = _player->GetTradeData();
738 if (!my_trade)
739 return;
740
741 TradeStatusInfo info;
742 // invalid slot number
743 if (tradeSlot >= TRADE_SLOT_COUNT)
744 {
745 info.Status = TRADE_STATUS_TRADE_CANCELED;
746 SendTradeStatus(info);
747 return;
748 }
749
750 // check cheating, can't fail with correct client operations
751 Item* item = _player->GetItemByPos(bag, slot);
752 if (!item || (tradeSlot != TRADE_SLOT_NONTRADED && !item->CanBeTraded(false, true)))
753 {
754 info.Status = TRADE_STATUS_TRADE_CANCELED;
755 SendTradeStatus(info);
756 return;
757 }
758
759 ObjectGuid iGUID = item->GetGUID();
760
761 // prevent place single item into many trade slots using cheating and client bugs
762 if (my_trade->HasItem(iGUID))
763 {
764 // cheating attempt
765 info.Status = TRADE_STATUS_TRADE_CANCELED;
766 SendTradeStatus(info);
767 return;
768 }
769
770 if (tradeSlot != TRADE_SLOT_NONTRADED && item->IsBindedNotWith(my_trade->GetTrader()))
771 {
772 info.Status = TRADE_STATUS_NOT_ON_TAPLIST;
773 info.Slot = tradeSlot;
774 SendTradeStatus(info);
775 return;
776 }
777
778 my_trade->SetItem(TradeSlots(tradeSlot), item);
779}

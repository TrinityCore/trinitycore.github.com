49void WorldSession::HandleRepopRequestOpcode(WorldPacket& recvData)
50{
51 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_REPOP_REQUEST Message");
52
53 recvData.read_skip<uint8>();
54
55 if (GetPlayer()->IsAlive() || GetPlayer()->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_GHOST))
56 return;
57
58 if (GetPlayer()->HasAuraType(SPELL_AURA_PREVENT_RESURRECTION))
59 return; // silently return, client should display the error by itself
60
61 // the world update order is sessions, players, creatures
62 // the netcode runs in parallel with all of these
63 // creatures can kill players
64 // so if the server is lagging enough the player can
65 // release spirit after he's killed but before he is updated
66 if (GetPlayer()->getDeathState() == JUST_DIED)
67 {
68 TC_LOG_DEBUG("network", "HandleRepopRequestOpcode: got request after player %s(%d) was killed and before he was updated",
69 GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().GetCounter());
70 GetPlayer()->KillPlayer();
71 }
72
73 //this is spirit release confirm?
74 GetPlayer()->RemoveGhoul();
75 GetPlayer()->RemovePet(NULL, PET_SAVE_NOT_IN_SLOT, true);
76 GetPlayer()->BuildPlayerRepop();
77 GetPlayer()->RepopAtGraveyard();
78}

80void WorldSession::HandleGossipSelectOptionOpcode(WorldPacket& recvData)
81{
82 TC_LOG_DEBUG("network", "WORLD: CMSG_GOSSIP_SELECT_OPTION");
83
84 uint32 gossipListId;
85 uint32 menuId;
86 ObjectGuid guid;
87 std::string code = "";
88
89 recvData >> guid >> menuId >> gossipListId;
90
91 if (!_player->PlayerTalkClass->GetGossipMenu().GetItem(gossipListId))
92 {
93 recvData.rfinish();
94 return;
95 }
96
97 if (_player->PlayerTalkClass->IsGossipOptionCoded(gossipListId))
98 recvData >> code;
99
100 // Prevent cheating on C++ scripted menus
101 if (_player->PlayerTalkClass->GetGossipMenu().GetSenderGUID() != guid)
102 return;
103
104 Creature* unit = NULL;
105 GameObject* go = NULL;
106 if (guid.IsCreatureOrVehicle())
107 {
108 unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_GOSSIP);
109 if (!unit)
110 {
111 TC_LOG_DEBUG("network", "WORLD: HandleGossipSelectOptionOpcode - %s not found or you can't interact with him.", guid.ToString().c_str());
112 return;
113 }
114 }
115 else if (guid.IsGameObject())
116 {
117 go = _player->GetGameObjectIfCanInteractWith(guid);
118 if (!go)
119 {
120 TC_LOG_DEBUG("network", "WORLD: HandleGossipSelectOptionOpcode - %s not found or you can't interact with it.", guid.ToString().c_str());
121 return;
122 }
123 }
124 else
125 {
126 TC_LOG_DEBUG("network", "WORLD: HandleGossipSelectOptionOpcode - unsupported %s.", guid.ToString().c_str());
127 return;
128 }
129
130 // remove fake death
131 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
132 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
133
134 if ((unit && unit->GetCreatureTemplate()->ScriptID != unit->LastUsedScriptID) || (go && go->GetGOInfo()->ScriptId != go->LastUsedScriptID))
135 {
136 TC_LOG_DEBUG("network", "WORLD: HandleGossipSelectOptionOpcode - Script reloaded while in use, ignoring and set new scipt id");
137 if (unit)
138 unit->LastUsedScriptID = unit->GetCreatureTemplate()->ScriptID;
139 if (go)
140 go->LastUsedScriptID = go->GetGOInfo()->ScriptId;
141 _player->PlayerTalkClass->SendCloseGossip();
142 return;
143 }
144 if (!code.empty())
145 {
146 if (unit)
147 {
148 unit->AI()->sGossipSelectCode(_player, menuId, gossipListId, code.c_str());
149 if (!sScriptMgr->OnGossipSelectCode(_player, unit, _player->PlayerTalkClass->GetGossipOptionSender(gossipListId), _player->PlayerTalkClass->GetGossipOptionAction(gossipListId), code.c_str()))
150 _player->OnGossipSelect(unit, gossipListId, menuId);
151 }
152 else
153 {
154 go->AI()->GossipSelectCode(_player, menuId, gossipListId, code.c_str());
155 if (!sScriptMgr->OnGossipSelectCode(_player, go, _player->PlayerTalkClass->GetGossipOptionSender(gossipListId), _player->PlayerTalkClass->GetGossipOptionAction(gossipListId), code.c_str()))
156 _player->OnGossipSelect(go, gossipListId, menuId);
157 }
158 }
159 else
160 {
161 if (unit)
162 {
163 unit->AI()->sGossipSelect(_player, menuId, gossipListId);
164 if (!sScriptMgr->OnGossipSelect(_player, unit, _player->PlayerTalkClass->GetGossipOptionSender(gossipListId), _player->PlayerTalkClass->GetGossipOptionAction(gossipListId)))
165 _player->OnGossipSelect(unit, gossipListId, menuId);
166 }
167 else
168 {
169 go->AI()->GossipSelect(_player, menuId, gossipListId);
170 if (!sScriptMgr->OnGossipSelect(_player, go, _player->PlayerTalkClass->GetGossipOptionSender(gossipListId), _player->PlayerTalkClass->GetGossipOptionAction(gossipListId)))
171 _player->OnGossipSelect(go, gossipListId, menuId);
172 }
173 }

176void WorldSession::HandleWhoOpcode(WorldPacket& recvData)
177{
178 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_WHO Message");
179
180 uint32 matchcount = 0;
181
182 uint32 level_min, level_max, racemask, classmask, zones_count, str_count;
183 uint32 zoneids[10]; // 10 is client limit
184 std::string player_name, guild_name;
185
186 recvData >> level_min; // maximal player level, default 0
187 recvData >> level_max; // minimal player level, default 100 (MAX_LEVEL)
188 recvData >> player_name; // player name, case sensitive...
189
190 recvData >> guild_name; // guild name, case sensitive...
191
192 recvData >> racemask; // race mask
193 recvData >> classmask; // class mask
194 recvData >> zones_count; // zones count, client limit = 10 (2.0.10)
195
196 if (zones_count > 10)
197 return; // can't be received from real client or broken packet
198
199 for (uint32 i = 0; i < zones_count; ++i)
200 {
201 uint32 temp;
202 recvData >> temp; // zone id, 0 if zone is unknown...
203 zoneids[i] = temp;
204 TC_LOG_DEBUG("network", "Zone %u: %u", i, zoneids[i]);
205 }
206
207 recvData >> str_count; // user entered strings count, client limit=4 (checked on 2.0.10)
208
209 if (str_count > 4)
210 return; // can't be received from real client or broken packet
211
212 TC_LOG_DEBUG("network", "Minlvl %u, maxlvl %u, name %s, guild %s, racemask %u, classmask %u, zones %u, strings %u", level_min, level_max, player_name.c_str(), guild_name.c_str(), racemask, classmask, zones_count, str_count);
213
214 std::wstring str[4]; // 4 is client limit
215 for (uint32 i = 0; i < str_count; ++i)
216 {
217 std::string temp;
218 recvData >> temp; // user entered string, it used as universal search pattern(guild+player name)?
219
220 if (!Utf8toWStr(temp, str[i]))
221 continue;
222
223 wstrToLower(str[i]);
224
225 TC_LOG_DEBUG("network", "String %u: %s", i, temp.c_str());
226 }
227
228 std::wstring wplayer_name;
229 std::wstring wguild_name;
230 if (!(Utf8toWStr(player_name, wplayer_name) && Utf8toWStr(guild_name, wguild_name)))
231 return;
232 wstrToLower(wplayer_name);
233 wstrToLower(wguild_name);
234
235 // client send in case not set max level value 100 but Trinity supports 255 max level,
236 // update it to show GMs with characters after 100 level
237 if (level_max >= MAX_LEVEL)
238 level_max = STRONG_MAX_LEVEL;
239
240 uint32 team = _player->GetTeam();
241
242 uint32 gmLevelInWhoList = sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_WHO_LIST);
243 uint32 displaycount = 0;
244
245 WorldPacket data(SMSG_WHO, 50); // guess size
246 data << uint32(matchcount); // placeholder, count of players matching criteria
247 data << uint32(displaycount); // placeholder, count of players displayed
248
249 boost::shared_lock<boost::shared_mutex> lock(*HashMapHolder<Player>::GetLock());
250
251 HashMapHolder<Player>::MapType const& m = ObjectAccessor::GetPlayers();
252 for (HashMapHolder<Player>::MapType::const_iterator itr = m.begin(); itr != m.end(); ++itr)
253 {
254 Player* target = itr->second;
255 // player can see member of other team only if CONFIG_ALLOW_TWO_SIDE_WHO_LIST
256 if (target->GetTeam() != team && !HasPermission(rbac::RBAC_PERM_TWO_SIDE_WHO_LIST))
257 continue;
258
259 // player can see MODERATOR, GAME MASTER, ADMINISTRATOR only if CONFIG_GM_IN_WHO_LIST
260 if (!HasPermission(rbac::RBAC_PERM_WHO_SEE_ALL_SEC_LEVELS) && target->GetSession()->GetSecurity() > AccountTypes(gmLevelInWhoList))
261 continue;
262
263 // do not process players which are not in world
264 if (!target->IsInWorld())
265 continue;
266
267 // check if target is globally visible for player
268 if (!target->IsVisibleGloballyFor(_player))
269 continue;
270
271 // check if target's level is in level range
272 uint8 lvl = target->getLevel();
273 if (lvl < level_min || lvl > level_max)
274 continue;
275
276 // check if class matches classmask
277 uint8 class_ = target->getClass();
278 if (!(classmask & (1 << class_)))
279 continue;
280
281 // check if race matches racemask
282 uint32 race = target->getRace();
283 if (!(racemask & (1 << race)))
284 continue;
285
286 uint32 pzoneid = target->GetZoneId();
287 uint8 gender = target->GetByteValue(PLAYER_BYTES_3, 0);
288
289 bool z_show = true;
290 for (uint32 i = 0; i < zones_count; ++i)
291 {
292 if (zoneids[i] == pzoneid)
293 {
294 z_show = true;
295 break;
296 }
297
298 z_show = false;
299 }
300 if (!z_show)
301 continue;
302
303 std::string pname = target->GetName();
304 std::wstring wpname;
305 if (!Utf8toWStr(pname, wpname))
306 continue;
307 wstrToLower(wpname);
308
309 if (!(wplayer_name.empty() || wpname.find(wplayer_name) != std::wstring::npos))
310 continue;
311
312 std::string gname = sGuildMgr->GetGuildNameById(target->GetGuildId());
313 std::wstring wgname;
314 if (!Utf8toWStr(gname, wgname))
315 continue;
316 wstrToLower(wgname);
317
318 if (!(wguild_name.empty() || wgname.find(wguild_name) != std::wstring::npos))
319 continue;
320
321 std::string aname;
322 if (AreaTableEntry const* areaEntry = sAreaTableStore.LookupEntry(pzoneid))
323 aname = areaEntry->area_name[GetSessionDbcLocale()];
324
325 bool s_show = true;
326 for (uint32 i = 0; i < str_count; ++i)
327 {
328 if (!str[i].empty())
329 {
330 if (wgname.find(str[i]) != std::wstring::npos ||
331 wpname.find(str[i]) != std::wstring::npos ||
332 Utf8FitTo(aname, str[i]))
333 {
334 s_show = true;
335 break;
336 }
337 s_show = false;
338 }
339 }
340 if (!s_show)
341 continue;
342
343 // 49 is maximum player count sent to client - can be overridden
344 // through config, but is unstable
345 if ((matchcount++) >= sWorld->getIntConfig(CONFIG_MAX_WHO))
346 continue;
347
348 data << pname; // player name
349 data << gname; // guild name
350 data << uint32(lvl); // player level
351 data << uint32(class_); // player class
352 data << uint32(race); // player race
353 data << uint8(gender); // player gender
354 data << uint32(pzoneid); // player zone id
355
356 ++displaycount;
357 }
358
359 data.put(0, displaycount); // insert right count, count displayed
360 data.put(4, matchcount); // insert right count, count of matches
361
362 SendPacket(&data);
363 TC_LOG_DEBUG("network", "WORLD: Send SMSG_WHO Message");
364}

366void WorldSession::HandleLogoutRequestOpcode(WorldPacket& /*recvData*/)
367{
368 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_LOGOUT_REQUEST Message, security - %u", GetSecurity());
369
370 if (ObjectGuid lguid = GetPlayer()->GetLootGUID())
371 DoLootRelease(lguid);
372
373 bool instantLogout = (GetPlayer()->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_RESTING) && !GetPlayer()->IsInCombat()) ||
374 GetPlayer()->IsInFlight() || HasPermission(rbac::RBAC_PERM_INSTANT_LOGOUT);
375
376 /// TODO: Possibly add RBAC permission to log out in combat
377 bool canLogoutInCombat = GetPlayer()->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_RESTING);
378
379 uint32 reason = 0;
380 if (GetPlayer()->IsInCombat() && !canLogoutInCombat)
381 reason = 1;
382 else if (GetPlayer()->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_FALLING | MOVEMENTFLAG_FALLING_FAR))
383 reason = 3; // is jumping or falling
384 else if (GetPlayer()->duel || GetPlayer()->HasAura(9454)) // is dueling or frozen by GM via freeze command
385 reason = 2; // FIXME - Need the correct value
386
387 WorldPacket data(SMSG_LOGOUT_RESPONSE, 1+4);
388 data << uint32(reason);
389 data << uint8(instantLogout);
390 SendPacket(&data);
391
392 if (reason)
393 {
394 LogoutRequest(0);
395 return;
396 }
397
398 //instant logout in taverns/cities or on taxi or for admins, gm's, mod's if its enabled in worldserver.conf
399 if (instantLogout)
400 {
401 LogoutPlayer(true);
402 return;
403 }
404
405 // not set flags if player can't free move to prevent lost state at logout cancel
406 if (GetPlayer()->CanFreeMove())
407 {
408 if (GetPlayer()->getStandState() == UNIT_STAND_STATE_STAND)
409 GetPlayer()->SetStandState(UNIT_STAND_STATE_SIT);
410
411 WorldPacket data(SMSG_FORCE_MOVE_ROOT, (8+4)); // guess size
412 data << GetPlayer()->GetPackGUID();
413 data << (uint32)2;
414 SendPacket(&data);
415 GetPlayer()->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED);
416 }
417
418 LogoutRequest(time(NULL));
419}

563

567

571

574 if (friendGuid)
575 {
576 if (friendGuid == GetPlayer()->GetGUID())
577 friendResult = FRIEND_SELF;
578 else if (GetPlayer()->GetTeam() != team && !HasPermission(rbac::RBAC_PERM_TWO_SIDE_ADD_FRIEND))
579 friendResult = FRIEND_ENEMY;
580 else if (GetPlayer()->GetSocial()->HasFriend(friendGuid.GetCounter()))
581 friendResult = FRIEND_ALREADY;
582 else
583 {
584 Player* pFriend = ObjectAccessor::FindPlayer(friendGuid);
585 if (pFriend && pFriend->IsVisibleGloballyFor(GetPlayer()))
586 friendResult = FRIEND_ADDED_ONLINE;
587 else
588 friendResult = FRIEND_ADDED_OFFLINE;
589 if (!GetPlayer()->GetSocial()->AddToSocialList(friendGuid.GetCounter(), false))
590 {
591 friendResult = FRIEND_LIST_FULL;
592 TC_LOG_DEBUG("network", "WORLD: %s's friend list is full.", GetPlayer()->GetName().c_str());
593 }
594 }
595 GetPlayer()->GetSocial()->SetFriendNote(friendGuid.GetCounter(), friendNote);
596 }
597 }
598 }
599
600 sSocialMgr->SendFriendStatus(GetPlayer(), friendResult, friendGuid.GetCounter(), false);
601
602 TC_LOG_DEBUG("network", "WORLD: Sent (SMSG_FRIEND_STATUS)");
603}

809void WorldSession::HandleAreaTriggerOpcode(WorldPacket& recvData)
810{
811 uint32 triggerId;
812 recvData >> triggerId;
813
814 TC_LOG_DEBUG("network", "CMSG_AREATRIGGER. Trigger ID: %u", triggerId);
815
816 Player* player = GetPlayer();
817 if (player->IsInFlight())
818 {
819 TC_LOG_DEBUG("network", "HandleAreaTriggerOpcode: Player '%s' (GUID: %u) in flight, ignore Area Trigger ID:%u",
820 player->GetName().c_str(), player->GetGUID().GetCounter(), triggerId);
821 return;
822 }
823
824 AreaTriggerEntry const* atEntry = sAreaTriggerStore.LookupEntry(triggerId);
825 if (!atEntry)
826 {
827 TC_LOG_DEBUG("network", "HandleAreaTriggerOpcode: Player '%s' (GUID: %u) send unknown (by DBC) Area Trigger ID:%u",
828 player->GetName().c_str(), player->GetGUID().GetCounter(), triggerId);
829 return;
830 }
831
832 if (!player->IsInAreaTriggerRadius(atEntry))
833 {
834 TC_LOG_DEBUG("network", "HandleAreaTriggerOpcode: Player '%s' (GUID: %u) too far, ignore Area Trigger ID: %u",
835 player->GetName().c_str(), player->GetGUID().GetCounter(), triggerId);
836 return;
837 }
838
839 if (player->isDebugAreaTriggers)
840 ChatHandler(player->GetSession()).PSendSysMessage(LANG_DEBUG_AREATRIGGER_REACHED, triggerId);
841
842 if (sScriptMgr->OnAreaTrigger(player, atEntry))
843 return;
844
845 if (player->IsAlive())
846 if (uint32 questId = sObjectMgr->GetQuestForAreaTrigger(triggerId))
847 if (player->GetQuestStatus(questId) == QUEST_STATUS_INCOMPLETE)
848 player->AreaExploredOrEventHappens(questId);
849
850 if (sObjectMgr->IsTavernAreaTrigger(triggerId))
851 {
852 // set resting flag we are in the inn
853 player->SetRestFlag(REST_FLAG_IN_TAVERN, atEntry->id);
854
855 if (sWorld->IsFFAPvPRealm())
856 player->RemoveByteFlag(UNIT_FIELD_BYTES_2, 1, UNIT_BYTE2_FLAG_FFA_PVP);
857
858 return;
859 }
860
861 if (Battleground* bg = player->GetBattleground())
862 if (bg->GetStatus() == STATUS_IN_PROGRESS)
863 bg->HandleAreaTrigger(player, triggerId);
864
865 if (OutdoorPvP* pvp = player->GetOutdoorPvP())
866 if (pvp->HandleAreaTrigger(_player, triggerId))
867 return;
868
869 AreaTrigger const* at = sObjectMgr->GetAreaTrigger(triggerId);
870 if (!at)
871 return;
872
873 bool teleported = false;
874 if (player->GetMapId() != at->target_mapId)
875 {
876 if (Map::EnterState denyReason = sMapMgr->PlayerCannotEnter(at->target_mapId, player, false))
877 {
878 bool reviveAtTrigger = false; // should we revive the player if he is trying to enter the correct instance?
879 switch (denyReason)
880 {
881 case Map::CANNOT_ENTER_NO_ENTRY:
882 TC_LOG_DEBUG("maps", "MAP: Player '%s' attempted to enter map with id %d which has no entry", player->GetName().c_str(), at->target_mapId);
883 break;
884 case Map::CANNOT_ENTER_UNINSTANCED_DUNGEON:
885 TC_LOG_DEBUG("maps", "MAP: Player '%s' attempted to enter dungeon map %d but no instance template was found", player->GetName().c_str(), at->target_mapId);
886 break;
887 case Map::CANNOT_ENTER_DIFFICULTY_UNAVAILABLE:
888 TC_LOG_DEBUG("maps", "MAP: Player '%s' attempted to enter instance map %d but the requested difficulty was not found", player->GetName().c_str(), at->target_mapId);
889 if (MapEntry const* entry = sMapStore.LookupEntry(at->target_mapId))
890 player->SendTransferAborted(entry->MapID, TRANSFER_ABORT_DIFFICULTY, player->GetDifficulty(entry->IsRaid()));
891 break;
892 case Map::CANNOT_ENTER_NOT_IN_RAID:
893 if (MapEntry const* entry = sMapStore.LookupEntry(at->target_mapId))
894 {
895 char const* mapName = entry->name[player->GetSession()->GetSessionDbcLocale()];
896 TC_LOG_DEBUG("maps", "MAP: Player '%s' must be in a raid group to enter instance '%s'", player->GetName().c_str(), mapName);
897 // probably there must be special opcode, because client has this string constant in GlobalStrings.lua
898 player->GetSession()->SendAreaTriggerMessage(player->GetSession()->GetTrinityString(LANG_INSTANCE_RAID_GROUP_ONLY), mapName);
899 }
900 reviveAtTrigger = true;
901 break;
902 case Map::CANNOT_ENTER_CORPSE_IN_DIFFERENT_INSTANCE:
903 {
904 WorldPacket data(SMSG_CORPSE_NOT_IN_INSTANCE);
905 player->GetSession()->SendPacket(&data);
906 TC_LOG_DEBUG("maps", "MAP: Player '%s' does not have a corpse in instance map %d and cannot enter", player->GetName().c_str(), at->target_mapId);
907 break;
908 }
909 case Map::CANNOT_ENTER_INSTANCE_BIND_MISMATCH:
910 if (MapEntry const* entry = sMapStore.LookupEntry(at->target_mapId))
911 {
912 char const* mapName = entry->name[player->GetSession()->GetSessionDbcLocale()];
913 TC_LOG_DEBUG("maps", "MAP: Player '%s' cannot enter instance map '%s' because their permanent bind is incompatible with their group's", player->GetName().c_str(), mapName);
914 // is there a special opcode for this?
915 // @todo figure out how to get player localized difficulty string (e.g. "10 player", "Heroic" etc)
916 ChatHandler(player->GetSession()).PSendSysMessage(player->GetSession()->GetTrinityString(LANG_INSTANCE_BIND_MISMATCH), mapName);
917 }
918 reviveAtTrigger = true;
919 break;
920 case Map::CANNOT_ENTER_TOO_MANY_INSTANCES:
921 player->SendTransferAborted(at->target_mapId, TRANSFER_ABORT_TOO_MANY_INSTANCES);
922 TC_LOG_DEBUG("maps", "MAP: Player '%s' cannot enter instance map %d because he has exceeded the maximum number of instances per hour.", player->GetName().c_str(), at->target_mapId);
923 reviveAtTrigger = true;
924 break;
925 case Map::CANNOT_ENTER_MAX_PLAYERS:
926 player->SendTransferAborted(at->target_mapId, TRANSFER_ABORT_MAX_PLAYERS);
927 reviveAtTrigger = true;
928 break;
929 case Map::CANNOT_ENTER_ZONE_IN_COMBAT:
930 player->SendTransferAborted(at->target_mapId, TRANSFER_ABORT_ZONE_IN_COMBAT);
931 reviveAtTrigger = true;
932 break;
933 default:
934 break;
935 }
936
937 if (reviveAtTrigger) // check if the player is touching the areatrigger leading to the map his corpse is on
938 if (!player->IsAlive() && player->HasCorpse())
939 if (player->GetCorpseLocation().GetMapId() == at->target_mapId)
940 {
941 player->ResurrectPlayer(0.5f);
942 player->SpawnCorpseBones();
943 }
944
945 return;
946 }
947
948 if (Group* group = player->GetGroup())
949 if (group->isLFGGroup() && player->GetMap()->IsDungeon())
950 teleported = player->TeleportToBGEntryPoint();
951 }
952
953 if (!teleported)
954 player->TeleportTo(at->target_mapId, at->target_X, at->target_Y, at->target_Z, at->target_Orientation, TELE_TO_NOT_LEAVE_TRANSPORT);
955}

957void WorldSession::HandleUpdateAccountData(WorldPacket& recvData)
958{
959 TC_LOG_DEBUG("network", "WORLD: Received CMSG_UPDATE_ACCOUNT_DATA");
960
961 uint32 type, timestamp, decompressedSize;
962 recvData >> type >> timestamp >> decompressedSize;
963
964 TC_LOG_DEBUG("network", "UAD: type %u, time %u, decompressedSize %u", type, timestamp, decompressedSize);
965
966 if (type > NUM_ACCOUNT_DATA_TYPES)
967 return;
968
969 if (decompressedSize == 0) // erase
970 {
971 SetAccountData(AccountDataType(type), 0, "");
972
973 WorldPacket data(SMSG_UPDATE_ACCOUNT_DATA_COMPLETE, 4+4);
974 data << uint32(type);
975 data << uint32(0);
976 SendPacket(&data);
977
978 return;
979 }
980
981 if (decompressedSize > 0xFFFF)
982 {
983 recvData.rfinish(); // unnneded warning spam in this case
984 TC_LOG_ERROR("network", "UAD: Account data packet too big, size %u", decompressedSize);
985 return;
986 }
987
988 ByteBuffer dest;
989 dest.resize(decompressedSize);
990
991 uLongf realSize = decompressedSize;
992 if (uncompress(dest.contents(), &realSize, recvData.contents() + recvData.rpos(), recvData.size() - recvData.rpos()) != Z_OK)
993 {
994 recvData.rfinish(); // unnneded warning spam in this case
995 TC_LOG_ERROR("network", "UAD: Failed to decompress account data");
996 return;
997 }
998
999 recvData.rfinish(); // uncompress read (recvData.size() - recvData.rpos())
1000
1001 std::string adata;
1002 dest >> adata;
1003
1004 SetAccountData(AccountDataType(type), timestamp, adata);
1005
1006 WorldPacket data(SMSG_UPDATE_ACCOUNT_DATA_COMPLETE, 4+4);
1007 data << uint32(type);
1008 data << uint32(0);
1009 SendPacket(&data);
1010}

1012void WorldSession::HandleRequestAccountData(WorldPacket& recvData)
1013{
1014 TC_LOG_DEBUG("network", "WORLD: Received CMSG_REQUEST_ACCOUNT_DATA");
1015
1016 uint32 type;
1017 recvData >> type;
1018
1019 TC_LOG_DEBUG("network", "RAD: type %u", type);
1020
1021 if (type >= NUM_ACCOUNT_DATA_TYPES)
1022 return;
1023
1024 AccountData* adata = GetAccountData(AccountDataType(type));
1025
1026 uint32 size = adata->Data.size();
1027
1028 uLongf destSize = compressBound(size);
1029
1030 ByteBuffer dest;
1031 dest.resize(destSize);
1032
1033 if (size && compress(dest.contents(), &destSize, (uint8 const*)adata->Data.c_str(), size) != Z_OK)
1034 {
1035 TC_LOG_DEBUG("network", "RAD: Failed to compress account data");
1036 return;
1037 }
1038
1039 dest.resize(destSize);
1040
1041 WorldPacket data(SMSG_UPDATE_ACCOUNT_DATA, 8+4+4+4+destSize);
1042 data << uint64(_player ? _player->GetGUID() : ObjectGuid::Empty);
1043 data << uint32(type); // type (0-7)
1044 data << uint32(adata->Time); // unix time
1045 data << uint32(size); // decompressed length
1046 data.append(dest); // compressed data
1047 SendPacket(&data);
1048}

26/**
27 * @fn void WorldSession::SendBfInvitePlayerToWar(uint32 battleId, uint32 zoneId, uint32 acceptTime)
28 *
29 * @brief This send to player windows for invite player to join the war.
30 *
31 * @param battleId The BattleId of Bf
32 * @param zoneId The zone where the battle is (4197 for wg)
33 * @param acceptTime Time in second that the player have for accept
34 */
35void WorldSession::SendBfInvitePlayerToWar(uint32 battleId, uint32 zoneId, uint32 acceptTime)
36{
37 WorldPacket data(SMSG_BATTLEFIELD_MGR_ENTRY_INVITE, 12);
38 data << uint32(battleId);
39 data << uint32(zoneId);
40 data << uint32(time(NULL) + acceptTime);
41 SendPacket(&data);
42}

59/**
60 * @fn void WorldSession::SendBfQueueInviteResponse(uint32 battleId, uint32 zoneId, bool canQueue, bool full)
61 *
62 * @brief This send packet for inform player that he join queue.
63 *
64 * @param battleId The BattleId of Bf
65 * @param zoneId The zone where the battle is (4197 for wg)
66 * @param canQueue if able to queue
67 * @param full on log in is full
68 */
69void WorldSession::SendBfQueueInviteResponse(uint32 battleId, uint32 zoneId, bool canQueue, bool full)
70{
71 WorldPacket data(SMSG_BATTLEFIELD_MGR_QUEUE_REQUEST_RESPONSE, 11);
72 data << uint32(battleId);
73 data << uint32(zoneId);
74 data << uint8(canQueue ? 1 : 0); // Accepted // 0 you cannot queue wg // 1 you are queued
75 data << uint8(full ? 0 : 1); // Logging In // 0 wg full // 1 queue for upcoming
76 data << uint8(1); // Warmup
77 SendPacket(&data);
78}

97/**
98 * @fn void WorldSession::SendBfLeaveMessage(uint32 battleId, BFLeaveReason reason)
99 *
100 * @brief This is call when player leave battlefield zone.
101 *
102 * @param battleId The BattleId of Bf
103 * @param reason Reason why player left battlefield
104 */
105void WorldSession::SendBfLeaveMessage(uint32 battleId, BFLeaveReason reason /*= BF_LEAVE_REASON_EXITED*/)
106{
107 WorldPacket data(SMSG_BATTLEFIELD_MGR_EJECTED, 7);
108 data << uint32(battleId);
109 data << uint8(reason); // byte Reason
110 data << uint8(2); // byte BattleStatus
111 data << uint8(0); // bool Relocated
112 SendPacket(&data);
113}

115/**
116 * @fn void WorldSession::HandleBfQueueInviteResponse(WorldPacket& recvData)
117 *
118 * @brief Send by client when he click on accept for queue.
119 */
120void WorldSession::HandleBfQueueInviteResponse(WorldPacket& recvData)
121{
122 uint32 battleId;
123 uint8 accepted;
124
125 recvData >> battleId >> accepted;
126
127 TC_LOG_DEBUG("misc", "HandleBfQueueInviteResponse: BattleID:%u Accepted:%u", battleId, accepted);
128
129 Battlefield* bf = sBattlefieldMgr->GetBattlefieldByBattleId(battleId);
130 if (!bf)
131 return;
132
133 if (accepted)
134 bf->PlayerAcceptInviteToQueue(_player);
135}

137/**
138 * @fn void WorldSession::HandleBfEntryInviteResponse(WorldPacket& recvData)
139 *
140 * @brief Send by client on clicking in accept or refuse of invitation windows for join game.
141 */
142void WorldSession::HandleBfEntryInviteResponse(WorldPacket& recvData)
143{
144 uint32 battleId;
145 uint8 accepted;
146
147 recvData >> battleId >> accepted;
148
149 TC_LOG_DEBUG("misc", "HandleBfEntryInviteResponse: battleId: %u, accepted: %u", battleId, accepted);
150
151 Battlefield* bf = sBattlefieldMgr->GetBattlefieldByBattleId(battleId);
152 if (!bf)
153 return;
154
155 // If player accept invitation
156 if (accepted)
157 {
158 bf->PlayerAcceptInviteToWar(_player);
159 }
160 else
161 {
162 if (_player->GetZoneId() == bf->GetZoneId())
163 bf->KickPlayerFromBattlefield(_player->GetGUID());
164 }
165}

168 * @fn void WorldSession::HandleBfExitRequest(WorldPacket& recvData)
169 *
170 * @brief Send by client when exited battlefield
171 */
172void WorldSession::HandleBfExitRequest(WorldPacket& recvData)
173{
174 uint32 battleId;
175
176 recvData >> battleId;
177
178 TC_LOG_DEBUG("misc", "HandleBfExitRequest: battleId: %u ", battleId);
179
180 Battlefield* bf = sBattlefieldMgr->GetBattlefieldByBattleId(battleId);
181 if (!bf)
182 return;
183
184 bf->AskToLeaveQueue(_player);
185}

29void WorldSession::HandleInspectArenaTeamsOpcode(WorldPacket& recvData)
30{
31 TC_LOG_DEBUG("network", "MSG_INSPECT_ARENA_TEAMS");
32
33 ObjectGuid guid;
34 recvData >> guid;
35 TC_LOG_DEBUG("network", "Inspect Arena stats %s", guid.ToString().c_str());
36
37 Player* player = ObjectAccessor::FindPlayer(guid);
38
39 if (!player)
40 return;
41
42 if (!GetPlayer()->IsWithinDistInMap(player, INSPECT_DISTANCE, false))
43 return;
44
45 if (GetPlayer()->IsValidAttackTarget(player))
46 return;
47
48 for (uint8 i = 0; i < MAX_ARENA_SLOT; ++i)
49 {
50 if (uint32 a_id = player->GetArenaTeamId(i))
51 {
52 if (ArenaTeam* arenaTeam = sArenaTeamMgr->GetArenaTeamById(a_id))
53 arenaTeam->Inspect(this, player->GetGUID());
54 }
55 }
56}

83void WorldSession::HandleArenaTeamInviteOpcode(WorldPacket& recvData)
84{
85 TC_LOG_DEBUG("network", "CMSG_ARENA_TEAM_INVITE");
86
87 uint32 arenaTeamId; // arena team id
88 std::string invitedName;
89
90 Player* player = NULL;
91
92 recvData >> arenaTeamId >> invitedName;
93
94 if (!invitedName.empty())
95 {
96 if (!normalizePlayerName(invitedName))
97 return;
98
99 player = ObjectAccessor::FindPlayerByName(invitedName);
100 }
101
102 if (!player)
103 {
104 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", invitedName, ERR_ARENA_TEAM_PLAYER_NOT_FOUND_S);
105 return;
106 }
107
108 if (player->getLevel() < sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
109 {
110 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", player->GetName(), ERR_ARENA_TEAM_TARGET_TOO_LOW_S);
111 return;
112 }
113
114 ArenaTeam* arenaTeam = sArenaTeamMgr->GetArenaTeamById(arenaTeamId);
115 if (!arenaTeam)
116 {
117 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_PLAYER_NOT_IN_TEAM);
118 return;
119 }
120
121 if (GetPlayer()->GetArenaTeamId(arenaTeam->GetSlot()) != arenaTeamId)
122 {
123 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_PERMISSIONS);
124 return;
125 }
126
127 // OK result but don't send invite
128 if (player->GetSocial()->HasIgnore(GetPlayer()->GetGUID().GetCounter()))
129 return;
130
131 if (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GUILD) && player->GetTeam() != GetPlayer()->GetTeam())
132 {
133 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", "", ERR_ARENA_TEAM_NOT_ALLIED);
134 return;
135 }
136
137 if (player->GetArenaTeamId(arenaTeam->GetSlot()))
138 {
139 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", player->GetName(), ERR_ALREADY_IN_ARENA_TEAM_S);
140 return;
141 }
142
143 if (player->GetArenaTeamIdInvited())
144 {
145 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", player->GetName(), ERR_ALREADY_INVITED_TO_ARENA_TEAM_S);
146 return;
147 }
148
149 if (arenaTeam->GetMembersSize() >= arenaTeam->GetType() * 2)
150 {
151 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, arenaTeam->GetName(), "", ERR_ARENA_TEAM_TOO_MANY_MEMBERS_S);
152 return;
153 }
154
155 TC_LOG_DEBUG("bg.battleground", "Player %s Invited %s to Join his ArenaTeam", GetPlayer()->GetName().c_str(), invitedName.c_str());
156
157 player->SetArenaTeamIdInvited(arenaTeam->GetId());
158
159 WorldPacket data(SMSG_ARENA_TEAM_INVITE, (8+10));
160 data << GetPlayer()->GetName();
161 data << arenaTeam->GetName();
162 player->GetSession()->SendPacket(&data);
163
164 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_ARENA_TEAM_INVITE");
165}

167void WorldSession::HandleArenaTeamAcceptOpcode(WorldPacket & /*recvData*/)
168{
169 TC_LOG_DEBUG("network", "CMSG_ARENA_TEAM_ACCEPT"); // empty opcode
170
171 ArenaTeam* arenaTeam = sArenaTeamMgr->GetArenaTeamById(_player->GetArenaTeamIdInvited());
172 if (!arenaTeam)
173 return;
174
175 // Check if player is already in another team of the same size
176 if (_player->GetArenaTeamId(arenaTeam->GetSlot()))
177 {
178 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ALREADY_IN_ARENA_TEAM);
179 return;
180 }
181
182 // Only allow members of the other faction to join the team if cross faction interaction is enabled
183 if (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GUILD) && _player->GetTeam() != sObjectMgr->GetPlayerTeamByGUID(arenaTeam->GetCaptain()))
184 {
185 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_NOT_ALLIED);
186 return;
187 }
188
189 // Add player to team
190 if (!arenaTeam->AddMember(_player->GetGUID()))
191 {
192 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_INTERNAL);
193 return;
194 }
195
196 // Broadcast event
197 arenaTeam->BroadcastEvent(ERR_ARENA_TEAM_JOIN_SS, _player->GetGUID(), 2, _player->GetName(), arenaTeam->GetName(), "");
198}

208void WorldSession::HandleArenaTeamLeaveOpcode(WorldPacket& recvData)
209{
210 TC_LOG_DEBUG("network", "CMSG_ARENA_TEAM_LEAVE");
211
212 uint32 arenaTeamId;
213 recvData >> arenaTeamId;
214
215 ArenaTeam* arenaTeam = sArenaTeamMgr->GetArenaTeamById(arenaTeamId);
216 if (!arenaTeam)
217 return;
218
219 // Disallow leave team while in arena
220 if (_player->InArena())
221 {
222 SendArenaTeamCommandResult(ERR_ARENA_TEAM_QUIT_S, "", "", ERR_ARENA_TEAM_INTERNAL);
223 return;
224 }
225
226 // Team captain can't leave the team if other members are still present
227 if (_player->GetGUID() == arenaTeam->GetCaptain() && arenaTeam->GetMembersSize() > 1)
228 {
229 SendArenaTeamCommandResult(ERR_ARENA_TEAM_QUIT_S, "", "", ERR_ARENA_TEAM_LEADER_LEAVE_S);
230 return;
231 }
232
233 // If team consists only of the captain, disband the team
234 if (_player->GetGUID() == arenaTeam->GetCaptain())
235 {
236 arenaTeam->Disband(this);
237 delete arenaTeam;
238 return;
239 }
240 else
241 arenaTeam->DelMember(_player->GetGUID(), true);
242
243 // Broadcast event
244 arenaTeam->BroadcastEvent(ERR_ARENA_TEAM_LEAVE_SS, _player->GetGUID(), 2, _player->GetName(), arenaTeam->GetName(), "");
245
246 // Inform player who left
247 SendArenaTeamCommandResult(ERR_ARENA_TEAM_QUIT_S, arenaTeam->GetName(), "", 0);
248}

250void WorldSession::HandleArenaTeamDisbandOpcode(WorldPacket& recvData)
251{
252 TC_LOG_DEBUG("network", "CMSG_ARENA_TEAM_DISBAND");
253
254 uint32 arenaTeamId;
255 recvData >> arenaTeamId;
256
257 if (ArenaTeam* arenaTeam = sArenaTeamMgr->GetArenaTeamById(arenaTeamId))
258 {
259 // Only captain can disband the team
260 if (arenaTeam->GetCaptain() != _player->GetGUID())
261 return;
262
263 // Teams cannot be disbanded during fights
264 if (arenaTeam->IsFighting())
265 return;
266
267 arenaTeam->Disband(this);
268 delete arenaTeam;
269 }

272void WorldSession::HandleArenaTeamRemoveOpcode(WorldPacket& recvData)
273{
274 TC_LOG_DEBUG("network", "CMSG_ARENA_TEAM_REMOVE");
275
276 uint32 arenaTeamId;
277 std::string name;
278
279 recvData >> arenaTeamId;
280 recvData >> name;
281
282 // Check for valid arena team
283 ArenaTeam* arenaTeam = sArenaTeamMgr->GetArenaTeamById(arenaTeamId);
284 if (!arenaTeam)
285 return;
286
287 // Only captain can remove members
288 if (arenaTeam->GetCaptain() != _player->GetGUID())
289 {
290 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_PERMISSIONS);
291 return;
292 }
293
294 if (!normalizePlayerName(name))
295 return;
296
297 // Check if team member exists
298 ArenaTeamMember* member = arenaTeam->GetMember(name);
299 if (!member)
300 {
301 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", name, ERR_ARENA_TEAM_PLAYER_NOT_FOUND_S);
302 return;
303 }
304
305 // Captain cannot be removed
306 if (arenaTeam->GetCaptain() == member->Guid)
307 {
308 SendArenaTeamCommandResult(ERR_ARENA_TEAM_QUIT_S, "", "", ERR_ARENA_TEAM_LEADER_LEAVE_S);
309 return;
310 }
311
312 // Player cannot be removed during fights
313 if (arenaTeam->IsFighting())
314 return;
315
316 arenaTeam->DelMember(member->Guid, true);
317
318 // Broadcast event
319 arenaTeam->BroadcastEvent(ERR_ARENA_TEAM_REMOVE_SSS, ObjectGuid::Empty, 3, name, arenaTeam->GetName(), _player->GetName());
320}

322void WorldSession::HandleArenaTeamLeaderOpcode(WorldPacket& recvData)
323{
324 TC_LOG_DEBUG("network", "CMSG_ARENA_TEAM_LEADER");
325
326 uint32 arenaTeamId;
327 std::string name;
328
329 recvData >> arenaTeamId;
330 recvData >> name;
331
332 // Check for valid arena team
333 ArenaTeam* arenaTeam = sArenaTeamMgr->GetArenaTeamById(arenaTeamId);
334 if (!arenaTeam)
335 return;
336
337 // Only captain can pass leadership
338 if (arenaTeam->GetCaptain() != _player->GetGUID())
339 {
340 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_PERMISSIONS);
341 return;
342 }
343
344 if (!normalizePlayerName(name))
345 return;
346
347 // Check if team member exists
348 ArenaTeamMember* member = arenaTeam->GetMember(name);
349 if (!member)
350 {
351 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", name, ERR_ARENA_TEAM_PLAYER_NOT_FOUND_S);
352 return;
353 }
354
355 // Check if the target is already team captain
356 if (arenaTeam->GetCaptain() == member->Guid)
357 return;
358
359 arenaTeam->SetCaptain(member->Guid);
360
361 // Broadcast event
362 arenaTeam->BroadcastEvent(ERR_ARENA_TEAM_LEADER_CHANGED_SSS, ObjectGuid::Empty, 3, _player->GetName(), name, arenaTeam->GetName());
363}

38void WorldSession::HandleBattlemasterHelloOpcode(WorldPacket& recvData)
39{
40 ObjectGuid guid;
41 recvData >> guid;
42 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_BATTLEMASTER_HELLO Message from %s", guid.ToString().c_str());
43
44 Creature* unit = GetPlayer()->GetMap()->GetCreature(guid);
45 if (!unit)
46 return;
47
48 if (!unit->IsBattleMaster()) // it's not battlemaster
49 return;
50
51 // Stop the npc if moving
52 unit->StopMoving();
53
54 BattlegroundTypeId bgTypeId = sBattlegroundMgr->GetBattleMasterBG(unit->GetEntry());
55
56 if (!_player->GetBGAccessByLevel(bgTypeId))
57 {
58 // temp, must be gossip message...
59 SendNotification(LANG_YOUR_BG_LEVEL_REQ_ERROR);
60 return;
61 }
62
63 SendBattleGroundList(guid, bgTypeId);
64}

73void WorldSession::HandleBattlemasterJoinOpcode(WorldPacket& recvData)
74{
75 ObjectGuid guid;
76 uint32 bgTypeId_;
77 uint32 instanceId;
78 uint8 joinAsGroup;
79 bool isPremade = false;
80 Group* grp = NULL;
81
82 recvData >> guid; // battlemaster guid
83 recvData >> bgTypeId_; // battleground type id (DBC id)
84 recvData >> instanceId; // instance id, 0 if First Available selected
85 recvData >> joinAsGroup; // join as group
86
87 if (!sBattlemasterListStore.LookupEntry(bgTypeId_))
88 {
89 TC_LOG_ERROR("network", "Battleground: invalid bgtype (%u) received. possible cheater? player guid %u", bgTypeId_, _player->GetGUID().GetCounter());
90 return;
91 }
92
93 if (DisableMgr::IsDisabledFor(DISABLE_TYPE_BATTLEGROUND, bgTypeId_, NULL))
94 {
95 ChatHandler(this).PSendSysMessage(LANG_BG_DISABLED);
96 return;
97 }
98
99 BattlegroundTypeId bgTypeId = BattlegroundTypeId(bgTypeId_);
100
101 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_BATTLEMASTER_JOIN Message from %s", guid.ToString().c_str());
102
103 // can do this, since it's battleground, not arena
104 BattlegroundQueueTypeId bgQueueTypeId = BattlegroundMgr::BGQueueTypeId(bgTypeId, 0);
105 BattlegroundQueueTypeId bgQueueTypeIdRandom = BattlegroundMgr::BGQueueTypeId(BATTLEGROUND_RB, 0);
106
107 // ignore if player is already in BG
108 if (_player->InBattleground())
109 return;
110
111 // get bg instance or bg template if instance not found
112 Battleground* bg = NULL;
113 if (instanceId)
114 bg = sBattlegroundMgr->GetBattlegroundThroughClientInstance(instanceId, bgTypeId);
115
116 if (!bg)
117 bg = sBattlegroundMgr->GetBattlegroundTemplate(bgTypeId);
118 if (!bg)
119 return;
120
121 // expected bracket entry
122 PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->getLevel());
123 if (!bracketEntry)
124 return;
125
126 GroupJoinBattlegroundResult err;
127
128 // check queue conditions
129 if (!joinAsGroup)
130 {
131 if (GetPlayer()->isUsingLfg())
132 {
133 // player is using dungeon finder or raid finder
134 WorldPacket data;
135 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_LFG_CANT_USE_BATTLEGROUND);
136 GetPlayer()->GetSession()->SendPacket(&data);
137 return;
138 }
139
140 // check Deserter debuff
141 if (!_player->CanJoinToBattleground(bg))
142 {
143 WorldPacket data;
144 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_GROUP_JOIN_BATTLEGROUND_DESERTERS);
145 _player->GetSession()->SendPacket(&data);
146 return;
147 }
148
149 if (_player->GetBattlegroundQueueIndex(bgQueueTypeIdRandom) < PLAYER_MAX_BATTLEGROUND_QUEUES)
150 {
151 // player is already in random queue
152 WorldPacket data;
153 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_IN_RANDOM_BG);
154 _player->GetSession()->SendPacket(&data);
155 return;
156 }
157
158 if (_player->InBattlegroundQueue() && bgTypeId == BATTLEGROUND_RB)
159 {
160 // player is already in queue, can't start random queue
161 WorldPacket data;
162 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_IN_NON_RANDOM_BG);
163 _player->GetSession()->SendPacket(&data);
164 return;
165 }
166
167 // check if already in queue
168 if (_player->GetBattlegroundQueueIndex(bgQueueTypeId) < PLAYER_MAX_BATTLEGROUND_QUEUES)
169 // player is already in this queue
170 return;
171
172 // check if has free queue slots
173 if (!_player->HasFreeBattlegroundQueueId())
174 {
175 WorldPacket data;
176 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_BATTLEGROUND_TOO_MANY_QUEUES);
177 _player->GetSession()->SendPacket(&data);
178 return;
179 }
180
181 // check Freeze debuff
182 if (_player->HasAura(9454))
183 return;
184
185 BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
186
187 GroupQueueInfo* ginfo = bgQueue.AddGroup(_player, NULL, bgTypeId, bracketEntry, 0, false, isPremade, 0, 0);
188 uint32 avgTime = bgQueue.GetAverageQueueWaitTime(ginfo, bracketEntry->GetBracketId());
189 // already checked if queueSlot is valid, now just get it
190 uint32 queueSlot = _player->AddBattlegroundQueueId(bgQueueTypeId);
191
192 WorldPacket data;
193 // send status packet (in queue)
194 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, ginfo->ArenaType, 0);
195 SendPacket(&data);
196 TC_LOG_DEBUG("bg.battleground", "Battleground: player joined queue for bg queue type %u bg type %u: GUID %u, NAME %s",
197 bgQueueTypeId, bgTypeId, _player->GetGUID().GetCounter(), _player->GetName().c_str());
198 }
199 else
200 {
201 grp = _player->GetGroup();
202 // no group found, error
203 if (!grp)
204 return;
205 if (grp->GetLeaderGUID() != _player->GetGUID())
206 return;
207 err = grp->CanJoinBattlegroundQueue(bg, bgQueueTypeId, 0, bg->GetMaxPlayersPerTeam(), false, 0);
208 isPremade = (grp->GetMembersCount() >= bg->GetMinPlayersPerTeam());
209
210 BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
211 GroupQueueInfo* ginfo = NULL;
212 uint32 avgTime = 0;
213
214 if (err > 0)
215 {
216 TC_LOG_DEBUG("bg.battleground", "Battleground: the following players are joining as group:");
217 ginfo = bgQueue.AddGroup(_player, grp, bgTypeId, bracketEntry, 0, false, isPremade, 0, 0);
218 avgTime = bgQueue.GetAverageQueueWaitTime(ginfo, bracketEntry->GetBracketId());
219 }
220
221 for (GroupReference* itr = grp->GetFirstMember(); itr != NULL; itr = itr->next())
222 {
223 Player* member = itr->GetSource();
224 if (!member)
225 continue; // this should never happen
226
227 WorldPacket data;
228
229 if (err <= 0)
230 {
231 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
232 member->GetSession()->SendPacket(&data);
233 continue;
234 }
235
236 // add to queue
237 uint32 queueSlot = member->AddBattlegroundQueueId(bgQueueTypeId);
238
239 // send status packet (in queue)
240 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, ginfo->ArenaType, 0);
241 member->GetSession()->SendPacket(&data);
242 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
243 member->GetSession()->SendPacket(&data);
244 TC_LOG_DEBUG("bg.battleground", "Battleground: player joined queue for bg queue type %u bg type %u: GUID %u, NAME %s",
245 bgQueueTypeId, bgTypeId, member->GetGUID().GetCounter(), member->GetName().c_str());
246 }
247 TC_LOG_DEBUG("bg.battleground", "Battleground: group end");
248 }
249 sBattlegroundMgr->ScheduleQueueUpdate(0, 0, bgQueueTypeId, bgTypeId, bracketEntry->GetBracketId());
250}

252void WorldSession::HandleBattlegroundPlayerPositionsOpcode(WorldPacket& /*recvData*/)
253{
254 TC_LOG_DEBUG("network", "WORLD: Recvd MSG_BATTLEGROUND_PLAYER_POSITIONS Message");
255
256 Battleground* bg = _player->GetBattleground();
257 if (!bg) // can't be received if player not in battleground
258 return;
259
260 uint32 flagCarrierCount = 0;
261 Player* allianceFlagCarrier = NULL;
262 Player* hordeFlagCarrier = NULL;
263
264 if (ObjectGuid guid = bg->GetFlagPickerGUID(TEAM_ALLIANCE))
265 {
266 allianceFlagCarrier = ObjectAccessor::FindPlayer(guid);
267 if (allianceFlagCarrier)
268 ++flagCarrierCount;
269 }
270
271 if (ObjectGuid guid = bg->GetFlagPickerGUID(TEAM_HORDE))
272 {
273 hordeFlagCarrier = ObjectAccessor::FindPlayer(guid);
274 if (hordeFlagCarrier)
275 ++flagCarrierCount;
276 }
277
278 WorldPacket data(MSG_BATTLEGROUND_PLAYER_POSITIONS, 4 + 4 + 16 * flagCarrierCount);
279 // Used to send several player positions (found used in AV)
280 data << 0; // CGBattlefieldInfo__m_numPlayerPositions
281 /*
282 for (CGBattlefieldInfo__m_numPlayerPositions)
283 data << guid << posx << posy;
284 */
285 data << flagCarrierCount;
286 if (allianceFlagCarrier)
287 {
288 data << uint64(allianceFlagCarrier->GetGUID());
289 data << float(allianceFlagCarrier->GetPositionX());
290 data << float(allianceFlagCarrier->GetPositionY());
291 }
292
293 if (hordeFlagCarrier)
294 {
295 data << uint64(hordeFlagCarrier->GetGUID());
296 data << float(hordeFlagCarrier->GetPositionX());
297 data << float(hordeFlagCarrier->GetPositionY());
298 }
299
300 SendPacket(&data);
301}

347void WorldSession::HandleBattleFieldPortOpcode(WorldPacket &recvData)
348{
349 uint8 type; // arenatype if arena
350 uint8 unk2; // unk, can be 0x0 (may be if was invited?) and 0x1
351 uint32 bgTypeId_; // type id from dbc
352 uint16 unk; // 0x1F90 constant?
353 uint8 action; // enter battle 0x1, leave queue 0x0
354
355 recvData >> type >> unk2 >> bgTypeId_ >> unk >> action;
356 if (!sBattlemasterListStore.LookupEntry(bgTypeId_))
357 {
358 TC_LOG_DEBUG("bg.battleground", "CMSG_BATTLEFIELD_PORT %s ArenaType: %u, Unk: %u, BgType: %u, Action: %u. Invalid BgType!",
359 GetPlayerInfo().c_str(), type, unk2, bgTypeId_, action);
360 return;
361 }
362
363 if (!_player->InBattlegroundQueue())
364 {
365 TC_LOG_DEBUG("bg.battleground", "CMSG_BATTLEFIELD_PORT %s ArenaType: %u, Unk: %u, BgType: %u, Action: %u. Player not in queue!",
366 GetPlayerInfo().c_str(), type, unk2, bgTypeId_, action);
367 return;
368 }
369
370 //get GroupQueueInfo from BattlegroundQueue
371 BattlegroundTypeId bgTypeId = BattlegroundTypeId(bgTypeId_);
372 BattlegroundQueueTypeId bgQueueTypeId = BattlegroundMgr::BGQueueTypeId(bgTypeId, type);
373 BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
374 //we must use temporary variable, because GroupQueueInfo pointer can be deleted in BattlegroundQueue::RemovePlayer() function
375 GroupQueueInfo ginfo;
376 if (!bgQueue.GetPlayerGroupInfoData(_player->GetGUID(), &ginfo))
377 {
378 TC_LOG_DEBUG("bg.battleground", "CMSG_BATTLEFIELD_PORT %s ArenaType: %u, Unk: %u, BgType: %u, Action: %u. Player not in queue (No player Group Info)!",
379 GetPlayerInfo().c_str(), type, unk2, bgTypeId_, action);
380 return;
381 }
382 // if action == 1, then instanceId is required
383 if (!ginfo.IsInvitedToBGInstanceGUID && action == 1)
384 {
385 TC_LOG_DEBUG("bg.battleground", "CMSG_BATTLEFIELD_PORT %s ArenaType: %u, Unk: %u, BgType: %u, Action: %u. Player is not invited to any bg!",
386 GetPlayerInfo().c_str(), type, unk2, bgTypeId_, action);
387 return;
388 }
389
390 Battleground* bg = sBattlegroundMgr->GetBattleground(ginfo.IsInvitedToBGInstanceGUID, bgTypeId);
391 if (!bg)
392 {
393 if (action)
394 {
395 TC_LOG_DEBUG("bg.battleground", "CMSG_BATTLEFIELD_PORT %s ArenaType: %u, Unk: %u, BgType: %u, Action: %u. Cant find BG with id %u!",
396 GetPlayerInfo().c_str(), type, unk2, bgTypeId_, action, ginfo.IsInvitedToBGInstanceGUID);
397 return;
398 }
399
400 bg = sBattlegroundMgr->GetBattlegroundTemplate(bgTypeId);
401 if (!bg)
402 {
403 TC_LOG_ERROR("network", "BattlegroundHandler: bg_template not found for type id %u.", bgTypeId);
404 return;
405 }
406 }
407
408 TC_LOG_DEBUG("bg.battleground", "CMSG_BATTLEFIELD_PORT %s ArenaType: %u, Unk: %u, BgType: %u, Action: %u.",
409 GetPlayerInfo().c_str(), type, unk2, bgTypeId_, action);
410
411 // expected bracket entry
412 PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->getLevel());
413 if (!bracketEntry)
414 return;
415
416 //some checks if player isn't cheating - it is not exactly cheating, but we cannot allow it
417 if (action == 1 && ginfo.ArenaType == 0)
418 {
419 //if player is trying to enter battleground (not arena!) and he has deserter debuff, we must just remove him from queue
420 if (!_player->CanJoinToBattleground(bg))
421 {
422 //send bg command result to show nice message
423 WorldPacket data2;
424 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data2, ERR_GROUP_JOIN_BATTLEGROUND_DESERTERS);
425 _player->GetSession()->SendPacket(&data2);
426 action = 0;
427 TC_LOG_DEBUG("bg.battleground", "Player %s (%u) has a deserter debuff, do not port him to battleground!", _player->GetName().c_str(), _player->GetGUID().GetCounter());
428 }
429 //if player don't match battleground max level, then do not allow him to enter! (this might happen when player leveled up during his waiting in queue
430 if (_player->getLevel() > bg->GetMaxLevel())
431 {
432 TC_LOG_ERROR("network", "Player %s (%u) has level (%u) higher than maxlevel (%u) of battleground (%u)! Do not port him to battleground!",
433 _player->GetName().c_str(), _player->GetGUID().GetCounter(), _player->getLevel(), bg->GetMaxLevel(), bg->GetTypeID());
434 action = 0;
435 }
436 }
437 uint32 queueSlot = _player->GetBattlegroundQueueIndex(bgQueueTypeId);
438 WorldPacket data;
439 if (action)
440 {
441 // check Freeze debuff
442 if (_player->HasAura(9454))
443 return;
444
445 if (!_player->IsInvitedForBattlegroundQueueType(bgQueueTypeId))
446 return; // cheating?
447
448 if (!_player->InBattleground())
449 _player->SetBattlegroundEntryPoint();
450
451 // resurrect the player
452 if (!_player->IsAlive())
453 {
454 _player->ResurrectPlayer(1.0f);
455 _player->SpawnCorpseBones();
456 }
457 // stop taxi flight at port
458 if (_player->IsInFlight())
459 {
460 _player->GetMotionMaster()->MovementExpired();
461 _player->CleanupAfterTaxiFlight();
462 }
463
464 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, queueSlot, STATUS_IN_PROGRESS, 0, bg->GetStartTime(), bg->GetArenaType(), ginfo.Team);
465 _player->GetSession()->SendPacket(&data);
466
467 // remove battleground queue status from BGmgr
468 bgQueue.RemovePlayer(_player->GetGUID(), false);
469 // this is still needed here if battleground "jumping" shouldn't add deserter debuff
470 // also this is required to prevent stuck at old battleground after SetBattlegroundId set to new
471 if (Battleground* currentBg = _player->GetBattleground())
472 currentBg->RemovePlayerAtLeave(_player->GetGUID(), false, true);
473
474 // set the destination instance id
475 _player->SetBattlegroundId(bg->GetInstanceID(), bgTypeId);
476 // set the destination team
477 _player->SetBGTeam(ginfo.Team);
478
479 // bg->HandleBeforeTeleportToBattleground(_player);
480 sBattlegroundMgr->SendToBattleground(_player, ginfo.IsInvitedToBGInstanceGUID, bgTypeId);
481 // add only in HandleMoveWorldPortAck()
482 // bg->AddPlayer(_player, team);
483 TC_LOG_DEBUG("bg.battleground", "Battleground: player %s (%u) joined battle for bg %u, bgtype %u, queue type %u.", _player->GetName().c_str(), _player->GetGUID().GetCounter(), bg->GetInstanceID(), bg->GetTypeID(), bgQueueTypeId);
484 }
485 else // leave queue
486 {
487 if (bg->isArena() && bg->GetStatus() > STATUS_WAIT_QUEUE)
488 return;
489
490 // if player leaves rated arena match before match start, it is counted as he played but he lost
491 if (ginfo.IsRated && ginfo.IsInvitedToBGInstanceGUID)
492 {
493 ArenaTeam* at = sArenaTeamMgr->GetArenaTeamById(ginfo.Team);
494 if (at)
495 {
496 TC_LOG_DEBUG("bg.battleground", "UPDATING memberLost's personal arena rating for %s by opponents rating: %u, because he has left queue!", _player->GetGUID().ToString().c_str(), ginfo.OpponentsTeamRating);
497 at->MemberLost(_player, ginfo.OpponentsMatchmakerRating);
498 at->SaveToDB();
499 }
500 }
501 _player->RemoveBattlegroundQueueId(bgQueueTypeId); // must be called this way, because if you move this call to queue->removeplayer, it causes bugs
502 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, queueSlot, STATUS_NONE, 0, 0, 0, 0);
503 bgQueue.RemovePlayer(_player->GetGUID(), true);
504 // player left queue, we should update it - do not update Arena Queue
505 if (!ginfo.ArenaType)
506 sBattlegroundMgr->ScheduleQueueUpdate(ginfo.ArenaMatchmakerRating, ginfo.ArenaType, bgQueueTypeId, bgTypeId, bracketEntry->GetBracketId());
507 SendPacket(&data);
508 TC_LOG_DEBUG("bg.battleground", "Battleground: player %s (%u) left queue for bgtype %u, queue type %u.", _player->GetName().c_str(), _player->GetGUID().GetCounter(), bg->GetTypeID(), bgQueueTypeId);
509
510 // track if player refuses to join the BG after being invited
511 if (bg->isBattleground() && sWorld->getBoolConfig(CONFIG_BATTLEGROUND_TRACK_DESERTERS) &&
512 (bg->GetStatus() == STATUS_IN_PROGRESS || bg->GetStatus() == STATUS_WAIT_JOIN))
513 {
514 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_DESERTER_TRACK);
515 stmt->setUInt32(0, _player->GetGUID().GetCounter());
516 stmt->setUInt8(1, BG_DESERTION_TYPE_LEAVE_QUEUE);
517 CharacterDatabase.Execute(stmt);
518 }
519 }
520}

522void WorldSession::HandleBattlefieldLeaveOpcode(WorldPacket& recvData)
523{
524 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_LEAVE_BATTLEFIELD Message");
525
526 recvData.read_skip<uint8>(); // unk1
527 recvData.read_skip<uint8>(); // unk2
528 recvData.read_skip<uint32>(); // BattlegroundTypeId
529 recvData.read_skip<uint16>(); // unk3
530
531 // not allow leave battleground in combat
532 if (_player->IsInCombat())
533 if (Battleground* bg = _player->GetBattleground())
534 if (bg->GetStatus() != STATUS_WAIT_LEAVE)
535 return;
536
537 _player->LeaveBattleground();
538}

540void WorldSession::HandleBattlefieldStatusOpcode(WorldPacket & /*recvData*/)
541{
542 // empty opcode
543 TC_LOG_DEBUG("network", "WORLD: Battleground status");
544
545 WorldPacket data;
546 // we must update all queues here
547 Battleground* bg = NULL;
548 for (uint8 i = 0; i < PLAYER_MAX_BATTLEGROUND_QUEUES; ++i)
549 {
550 BattlegroundQueueTypeId bgQueueTypeId = _player->GetBattlegroundQueueTypeId(i);
551 if (!bgQueueTypeId)
552 continue;
553 BattlegroundTypeId bgTypeId = BattlegroundMgr::BGTemplateId(bgQueueTypeId);
554 uint8 arenaType = BattlegroundMgr::BGArenaType(bgQueueTypeId);
555 if (bgTypeId == _player->GetBattlegroundTypeId())
556 {
557 bg = _player->GetBattleground();
558 //i cannot check any variable from player class because player class doesn't know if player is in 2v2 / 3v3 or 5v5 arena
559 //so i must use bg pointer to get that information
560 if (bg && bg->GetArenaType() == arenaType)
561 {
562 // this line is checked, i only don't know if GetStartTime is changing itself after bg end!
563 // send status in Battleground
564 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, i, STATUS_IN_PROGRESS, bg->GetEndTime(), bg->GetStartTime(), arenaType, _player->GetBGTeam());
565 SendPacket(&data);
566 continue;
567 }
568 }
569 //we are sending update to player about queue - he can be invited there!
570 //get GroupQueueInfo for queue status
571 BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
572 GroupQueueInfo ginfo;
573 if (!bgQueue.GetPlayerGroupInfoData(_player->GetGUID(), &ginfo))
574 continue;
575 if (ginfo.IsInvitedToBGInstanceGUID)
576 {
577 bg = sBattlegroundMgr->GetBattleground(ginfo.IsInvitedToBGInstanceGUID, bgTypeId);
578 if (!bg)
579 continue;
580 uint32 remainingTime = getMSTimeDiff(getMSTime(), ginfo.RemoveInviteTime);
581 // send status invited to Battleground
582 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, i, STATUS_WAIT_JOIN, remainingTime, 0, arenaType, 0);
583 SendPacket(&data);
584 }
585 else
586 {
587 bg = sBattlegroundMgr->GetBattlegroundTemplate(bgTypeId);
588 if (!bg)
589 continue;
590
591 // expected bracket entry
592 PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->getLevel());
593 if (!bracketEntry)
594 continue;
595
596 uint32 avgTime = bgQueue.GetAverageQueueWaitTime(&ginfo, bracketEntry->GetBracketId());
597 // send status in Battleground Queue
598 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, i, STATUS_WAIT_QUEUE, avgTime, getMSTimeDiff(ginfo.JoinTime, getMSTime()), arenaType, 0);
599 SendPacket(&data);
600 }
601 }

604void WorldSession::HandleBattlemasterJoinArena(WorldPacket& recvData)
605{
606 TC_LOG_DEBUG("network", "WORLD: CMSG_BATTLEMASTER_JOIN_ARENA");
607
608 ObjectGuid guid; // arena Battlemaster guid
609 uint8 arenaslot; // 2v2, 3v3 or 5v5
610 uint8 asGroup; // asGroup
611 uint8 isRated; // isRated
612 Group* grp = NULL;
613
614 recvData >> guid >> arenaslot >> asGroup >> isRated;
615
616 // ignore if we already in BG or BG queue
617 if (_player->InBattleground())
618 return;
619
620 Creature* unit = GetPlayer()->GetMap()->GetCreature(guid);
621 if (!unit)
622 return;
623
624 if (!unit->IsBattleMaster()) // it's not battle master
625 return;
626
627 uint8 arenatype = 0;
628 uint32 arenaRating = 0;
629 uint32 matchmakerRating = 0;
630
631 switch (arenaslot)
632 {
633 case 0:
634 arenatype = ARENA_TYPE_2v2;
635 break;
636 case 1:
637 arenatype = ARENA_TYPE_3v3;
638 break;
639 case 2:
640 arenatype = ARENA_TYPE_5v5;
641 break;
642 default:
643 TC_LOG_ERROR("network", "Unknown arena slot %u at HandleBattlemasterJoinArena()", arenaslot);
644 return;
645 }
646
647 //check existance
648 Battleground* bg = sBattlegroundMgr->GetBattlegroundTemplate(BATTLEGROUND_AA);
649 if (!bg)
650 {
651 TC_LOG_ERROR("network", "Battleground: template bg (all arenas) not found");
652 return;
653 }
654
655 if (DisableMgr::IsDisabledFor(DISABLE_TYPE_BATTLEGROUND, BATTLEGROUND_AA, NULL))
656 {
657 ChatHandler(this).PSendSysMessage(LANG_ARENA_DISABLED);
658 return;
659 }
660
661 BattlegroundTypeId bgTypeId = bg->GetTypeID();
662 BattlegroundQueueTypeId bgQueueTypeId = BattlegroundMgr::BGQueueTypeId(bgTypeId, arenatype);
663 PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->getLevel());
664 if (!bracketEntry)
665 return;
666
667 GroupJoinBattlegroundResult err = ERR_GROUP_JOIN_BATTLEGROUND_FAIL;
668
669 if (!asGroup)
670 {
671 // check if already in queue
672 if (_player->GetBattlegroundQueueIndex(bgQueueTypeId) < PLAYER_MAX_BATTLEGROUND_QUEUES)
673 //player is already in this queue
674 return;
675 // check if has free queue slots
676 if (!_player->HasFreeBattlegroundQueueId())
677 return;
678 }
679 else
680 {
681 grp = _player->GetGroup();
682 // no group found, error
683 if (!grp)
684 return;
685 if (grp->GetLeaderGUID() != _player->GetGUID())
686 return;
687 err = grp->CanJoinBattlegroundQueue(bg, bgQueueTypeId, arenatype, arenatype, isRated != 0, arenaslot);
688 }
689
690 uint32 ateamId = 0;
691
692 if (isRated)
693 {
694 ateamId = _player->GetArenaTeamId(arenaslot);
695 // check real arenateam existence only here (if it was moved to group->CanJoin .. () then we would ahve to get it twice)
696 ArenaTeam* at = sArenaTeamMgr->GetArenaTeamById(ateamId);
697 if (!at)
698 {
699 _player->GetSession()->SendNotInArenaTeamPacket(arenatype);
700 return;
701 }
702 // get the team rating for queueing
703 arenaRating = at->GetRating();
704 matchmakerRating = at->GetAverageMMR(grp);
705 // the arenateam id must match for everyone in the group
706
707 if (arenaRating <= 0)
708 arenaRating = 1;
709 }
710
711 BattlegroundQueue &bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
712 if (asGroup)
713 {
714 uint32 avgTime = 0;
715
716 if (err > 0)
717 {
718 TC_LOG_DEBUG("bg.battleground", "Battleground: arena join as group start");
719 if (isRated)
720 {
721 TC_LOG_DEBUG("bg.battleground", "Battleground: arena team id %u, leader %s queued with matchmaker rating %u for type %u", _player->GetArenaTeamId(arenaslot), _player->GetName().c_str(), matchmakerRating, arenatype);
722 bg->SetRated(true);
723 }
724 else
725 bg->SetRated(false);
726
727 GroupQueueInfo* ginfo = bgQueue.AddGroup(_player, grp, bgTypeId, bracketEntry, arenatype, isRated != 0, false, arenaRating, matchmakerRating, ateamId);
728 avgTime = bgQueue.GetAverageQueueWaitTime(ginfo, bracketEntry->GetBracketId());
729 }
730
731 for (GroupReference* itr = grp->GetFirstMember(); itr != NULL; itr = itr->next())
732 {
733 Player* member = itr->GetSource();
734 if (!member)
735 continue;
736
737 WorldPacket data;
738
739 if (err <= 0)
740 {
741 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
742 member->GetSession()->SendPacket(&data);
743 continue;
744 }
745
746 // add to queue
747 uint32 queueSlot = member->AddBattlegroundQueueId(bgQueueTypeId);
748
749 // send status packet (in queue)
750 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, arenatype, 0);
751 member->GetSession()->SendPacket(&data);
752 sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
753 member->GetSession()->SendPacket(&data);
754 TC_LOG_DEBUG("bg.battleground", "Battleground: player joined queue for arena as group bg queue type %u bg type %u: GUID %u, NAME %s", bgQueueTypeId, bgTypeId, member->GetGUID().GetCounter(), member->GetName().c_str());
755 }
756 }
757 else
758 {
759 GroupQueueInfo* ginfo = bgQueue.AddGroup(_player, NULL, bgTypeId, bracketEntry, arenatype, isRated != 0, false, arenaRating, matchmakerRating, ateamId);
760 uint32 avgTime = bgQueue.GetAverageQueueWaitTime(ginfo, bracketEntry->GetBracketId());
761 uint32 queueSlot = _player->AddBattlegroundQueueId(bgQueueTypeId);
762
763 WorldPacket data;
764 // send status packet (in queue)
765 sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, arenatype, 0);
766 SendPacket(&data);
767 TC_LOG_DEBUG("bg.battleground", "Battleground: player joined queue for arena, skirmish, bg queue type %u bg type %u: GUID %u, NAME %s", bgQueueTypeId, bgTypeId, _player->GetGUID().GetCounter(), _player->GetName().c_str());
768 }
769 sBattlegroundMgr->ScheduleQueueUpdate(matchmakerRating, arenatype, bgQueueTypeId, bgTypeId, bracketEntry->GetBracketId());
770}

51void WorldSession::HandleCalendarGetCalendar(WorldPacket& /*recvData*/)
52{
53 ObjectGuid guid = _player->GetGUID();
54 TC_LOG_DEBUG("network", "CMSG_CALENDAR_GET_CALENDAR [%s]", guid.ToString().c_str());
55
56 time_t currTime = time(NULL);
57
58 WorldPacket data(SMSG_CALENDAR_SEND_CALENDAR, 1000); // Average size if no instance
59
60 CalendarInviteStore invites = sCalendarMgr->GetPlayerInvites(guid);
61 data << uint32(invites.size());
62 for (CalendarInviteStore::const_iterator itr = invites.begin(); itr != invites.end(); ++itr)
63 {
64 data << uint64((*itr)->GetEventId());
65 data << uint64((*itr)->GetInviteId());
66 data << uint8((*itr)->GetStatus());
67 data << uint8((*itr)->GetRank());
68
69 if (CalendarEvent* calendarEvent = sCalendarMgr->GetEvent((*itr)->GetEventId()))
70 {
71 data << uint8(calendarEvent->IsGuildEvent());
72 data << calendarEvent->GetCreatorGUID().WriteAsPacked();
73 }
74 else
75 {
76 data << uint8(0);
77 data << (*itr)->GetSenderGUID().WriteAsPacked();
78 }
79 }
80
81 CalendarEventStore playerEvents = sCalendarMgr->GetPlayerEvents(guid);
82 data << uint32(playerEvents.size());
83 for (CalendarEventStore::const_iterator itr = playerEvents.begin(); itr != playerEvents.end(); ++itr)
84 {
85 CalendarEvent* calendarEvent = *itr;
86
87 data << uint64(calendarEvent->GetEventId());
88 data << calendarEvent->GetTitle();
89 data << uint32(calendarEvent->GetType());
90 data.AppendPackedTime(calendarEvent->GetEventTime());
91 data << uint32(calendarEvent->GetFlags());
92 data << int32(calendarEvent->GetDungeonId());
93 data << calendarEvent->GetCreatorGUID().WriteAsPacked();
94 }
95
96 data << uint32(currTime); // server time
97 data.AppendPackedTime(currTime); // zone time
98
99 ByteBuffer dataBuffer;
100 uint32 boundCounter = 0;
101 for (uint8 i = 0; i < MAX_DIFFICULTY; ++i)
102 {
103 Player::BoundInstancesMap boundInstances = _player->GetBoundInstances(Difficulty(i));
104 for (Player::BoundInstancesMap::const_iterator itr = boundInstances.begin(); itr != boundInstances.end(); ++itr)
105 {
106 if (itr->second.perm)
107 {
108 InstanceSave const* save = itr->second.save;
109 dataBuffer << uint32(save->GetMapId());
110 dataBuffer << uint32(save->GetDifficulty());
111 dataBuffer << uint32(save->GetResetTime() - currTime);
112 dataBuffer << uint64(save->GetInstanceId()); // instance save id as unique instance copy id
113 ++boundCounter;
114 }
115 }
116 }
117
118 data << uint32(boundCounter);
119 data.append(dataBuffer);
120
121 data << uint32(1135753200); // Constant date, unk (28.12.2005 07:00)
122
123 // Reuse variables
124 boundCounter = 0;
125 std::set<uint32> sentMaps;
126 dataBuffer.clear();
127
128 ResetTimeByMapDifficultyMap const& resets = sInstanceSaveMgr->GetResetTimeMap();
129 for (ResetTimeByMapDifficultyMap::const_iterator itr = resets.begin(); itr != resets.end(); ++itr)
130 {
131 uint32 mapId = PAIR32_LOPART(itr->first);
132 if (sentMaps.find(mapId) != sentMaps.end())
133 continue;
134
135 MapEntry const* mapEntry = sMapStore.LookupEntry(mapId);
136 if (!mapEntry || !mapEntry->IsRaid())
137 continue;
138
139 sentMaps.insert(mapId);
140
141 dataBuffer << int32(mapId);
142 dataBuffer << int32(itr->second - currTime);
143 dataBuffer << int32(0); // Never seen anything else in sniffs - still unknown
144 ++boundCounter;
145 }
146
147 data << uint32(boundCounter);
148 data.append(dataBuffer);
149
150 /// @todo Fix this, how we do know how many and what holidays to send?
151 uint32 holidayCount = 0;
152 data << uint32(holidayCount);
153 for (uint32 i = 0; i < holidayCount; ++i)
154 {
155 HolidaysEntry const* holiday = sHolidaysStore.LookupEntry(666);
156
157 data << uint32(holiday->Id); // m_ID
158 data << uint32(holiday->Region); // m_region, might be looping
159 data << uint32(holiday->Looping); // m_looping, might be region
160 data << uint32(holiday->Priority); // m_priority
161 data << uint32(holiday->CalendarFilterType); // m_calendarFilterType
162
163 for (uint8 j = 0; j < MAX_HOLIDAY_DATES; ++j)
164 data << uint32(holiday->Date[j]); // 26 * m_date -- WritePackedTime ?
165
166 for (uint8 j = 0; j < MAX_HOLIDAY_DURATIONS; ++j)
167 data << uint32(holiday->Duration[j]); // 10 * m_duration
168
169 for (uint8 j = 0; j < MAX_HOLIDAY_FLAGS; ++j)
170 data << uint32(holiday->CalendarFlags[j]); // 10 * m_calendarFlags
171
172 data << holiday->TextureFilename; // m_textureFilename (holiday name)
173 }
174
175 SendPacket(&data);
176}

218void WorldSession::HandleCalendarAddEvent(WorldPacket& recvData)
219{
220 ObjectGuid guid = _player->GetGUID();
221
222 std::string title;
223 std::string description;
224 uint8 type;
225 uint8 repeatable;
226 uint32 maxInvites;
227 int32 dungeonId;
228 uint32 eventPackedTime;
229 uint32 unkPackedTime;
230 uint32 flags;
231
232 recvData >> title >> description >> type >> repeatable >> maxInvites >> dungeonId;
233 recvData.ReadPackedTime(eventPackedTime);
234 recvData.ReadPackedTime(unkPackedTime);
235 recvData >> flags;
236
237 // prevent events in the past
238 // To Do: properly handle timezones and remove the "- time_t(86400L)" hack
239 if (time_t(eventPackedTime) < (time(NULL) - time_t(86400L)))
240 {
241 recvData.rfinish();
242 return;
243 }
244
245 CalendarEvent* calendarEvent = new CalendarEvent(sCalendarMgr->GetFreeEventId(), guid, 0, CalendarEventType(type), dungeonId,
246 time_t(eventPackedTime), flags, time_t(unkPackedTime), title, description);
247
248 if (calendarEvent->IsGuildEvent() || calendarEvent->IsGuildAnnouncement())
249 if (Player* creator = ObjectAccessor::FindPlayer(guid))
250 calendarEvent->SetGuildId(creator->GetGuildId());
251
252 if (calendarEvent->IsGuildAnnouncement())
253 {
254 // 946684800 is 01/01/2000 00:00:00 - default response time
255 CalendarInvite invite(0, calendarEvent->GetEventId(), ObjectGuid::Empty, guid, 946684800, CALENDAR_STATUS_NOT_SIGNED_UP, CALENDAR_RANK_PLAYER, "");
256 // WARNING: By passing pointer to a local variable, the underlying method(s) must NOT perform any kind
257 // of storage of the pointer as it will lead to memory corruption
258 sCalendarMgr->AddInvite(calendarEvent, &invite);
259 }
260 else
261 {
262 // client limits the amount of players to be invited to 100
263 const uint32 MaxPlayerInvites = 100;
264
265 uint32 inviteCount;
266 ObjectGuid invitee[MaxPlayerInvites];
267 uint8 status[MaxPlayerInvites];
268 uint8 rank[MaxPlayerInvites];
269
270 memset(status, 0, sizeof(status));
271 memset(rank, 0, sizeof(rank));
272
273 try
274 {
275 recvData >> inviteCount;
276
277 for (uint32 i = 0; i < inviteCount && i < MaxPlayerInvites; ++i)
278 {
279 recvData >> invitee[i].ReadAsPacked();
280 recvData >> status[i] >> rank[i];
281 }
282 }
283 catch (ByteBufferException const&)
284 {
285 delete calendarEvent;
286 calendarEvent = NULL;
287 throw;
288 }
289
290 SQLTransaction trans;
291 if (inviteCount > 1)
292 trans = CharacterDatabase.BeginTransaction();
293
294 for (uint32 i = 0; i < inviteCount && i < MaxPlayerInvites; ++i)
295 {
296 // 946684800 is 01/01/2000 00:00:00 - default response time
297 CalendarInvite* invite = new CalendarInvite(sCalendarMgr->GetFreeInviteId(), calendarEvent->GetEventId(), invitee[i], guid, 946684800, CalendarInviteStatus(status[i]), CalendarModerationRank(rank[i]), "");
298 sCalendarMgr->AddInvite(calendarEvent, invite, trans);
299 }
300
301 if (inviteCount > 1)
302 CharacterDatabase.CommitTransaction(trans);
303 }
304
305 sCalendarMgr->AddEvent(calendarEvent, CALENDAR_SENDTYPE_ADD);
306}

308void WorldSession::HandleCalendarUpdateEvent(WorldPacket& recvData)
309{
310 ObjectGuid guid = _player->GetGUID();
311 time_t oldEventTime;
312
313 uint64 eventId;
314 uint64 inviteId;
315 std::string title;
316 std::string description;
317 uint8 type;
318 uint8 repetitionType;
319 uint32 maxInvites;
320 int32 dungeonId;
321 uint32 eventPackedTime;
322 uint32 timeZoneTime;
323 uint32 flags;
324
325 recvData >> eventId >> inviteId >> title >> description >> type >> repetitionType >> maxInvites >> dungeonId;
326 recvData.ReadPackedTime(eventPackedTime);
327 recvData.ReadPackedTime(timeZoneTime);
328 recvData >> flags;
329
330 // prevent events in the past
331 // To Do: properly handle timezones and remove the "- time_t(86400L)" hack
332 if (time_t(eventPackedTime) < (time(NULL) - time_t(86400L)))
333 {
334 recvData.rfinish();
335 return;
336 }
337
338 TC_LOG_DEBUG("network", "CMSG_CALENDAR_UPDATE_EVENT [%s] EventId [" UI64FMTD
339 "], InviteId [" UI64FMTD "] Title %s, Description %s, type %u "
340 "Repeatable %u, MaxInvites %u, Dungeon ID %d, Time %u "
341 "Time2 %u, Flags %u", guid.ToString().c_str(), eventId, inviteId, title.c_str(),
342 description.c_str(), type, repetitionType, maxInvites, dungeonId,
343 eventPackedTime, timeZoneTime, flags);
344
345 if (CalendarEvent* calendarEvent = sCalendarMgr->GetEvent(eventId))
346 {
347 oldEventTime = calendarEvent->GetEventTime();
348
349 calendarEvent->SetType(CalendarEventType(type));
350 calendarEvent->SetFlags(flags);
351 calendarEvent->SetEventTime(time_t(eventPackedTime));
352 calendarEvent->SetTimeZoneTime(time_t(timeZoneTime)); // Not sure, seems constant from the little sniffs we have
353 calendarEvent->SetDungeonId(dungeonId);
354 calendarEvent->SetTitle(title);
355 calendarEvent->SetDescription(description);
356
357 sCalendarMgr->UpdateEvent(calendarEvent);
358 sCalendarMgr->SendCalendarEventUpdateAlert(*calendarEvent, oldEventTime);
359 }
360 else
361 sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_EVENT_INVALID);
362}

375void WorldSession::HandleCalendarCopyEvent(WorldPacket& recvData)
376{
377 ObjectGuid guid = _player->GetGUID();
378 uint64 eventId;
379 uint64 inviteId;
380 uint32 eventTime;
381
382 recvData >> eventId >> inviteId;
383 recvData.ReadPackedTime(eventTime);
384 TC_LOG_DEBUG("network", "CMSG_CALENDAR_COPY_EVENT [%s], EventId [" UI64FMTD
385 "] inviteId [" UI64FMTD "] Time: %u", guid.ToString().c_str(), eventId, inviteId, eventTime);
386
387 // prevent events in the past
388 // To Do: properly handle timezones and remove the "- time_t(86400L)" hack
389 if (time_t(eventTime) < (time(NULL) - time_t(86400L)))
390 {
391 recvData.rfinish();
392 return;
393 }
394
395 if (CalendarEvent* oldEvent = sCalendarMgr->GetEvent(eventId))
396 {
397 CalendarEvent* newEvent = new CalendarEvent(*oldEvent, sCalendarMgr->GetFreeEventId());
398 newEvent->SetEventTime(time_t(eventTime));
399 sCalendarMgr->AddEvent(newEvent, CALENDAR_SENDTYPE_COPY);
400
401 CalendarInviteStore invites = sCalendarMgr->GetEventInvites(eventId);
402 SQLTransaction trans;
403 if (invites.size() > 1)
404 trans = CharacterDatabase.BeginTransaction();
405
406 for (CalendarInviteStore::const_iterator itr = invites.begin(); itr != invites.end(); ++itr)
407 sCalendarMgr->AddInvite(newEvent, new CalendarInvite(**itr, sCalendarMgr->GetFreeInviteId(), newEvent->GetEventId()), trans);
408
409 if (invites.size() > 1)
410 CharacterDatabase.CommitTransaction(trans);
411 // should we change owner when somebody makes a copy of event owned by another person?
412 }
413 else
414 sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_EVENT_INVALID);
415}

417void WorldSession::HandleCalendarEventInvite(WorldPacket& recvData)
418{
419 TC_LOG_DEBUG("network", "CMSG_CALENDAR_EVENT_INVITE");
420
421 ObjectGuid playerGuid = _player->GetGUID();
422
423 uint64 eventId;
424 uint64 inviteId;
425 std::string name;
426 bool isPreInvite;
427 bool isGuildEvent;
428
429 ObjectGuid inviteeGuid;
430 uint32 inviteeTeam = 0;
431 uint32 inviteeGuildId = 0;
432
433 recvData >> eventId >> inviteId >> name >> isPreInvite >> isGuildEvent;
434
435 if (Player* player = ObjectAccessor::FindConnectedPlayerByName(name))
436 {
437 // Invitee is online
438 inviteeGuid = player->GetGUID();
439 inviteeTeam = player->GetTeam();
440 inviteeGuildId = player->GetGuildId();
441 }
442 else
443 {
444 // Invitee offline, get data from database
445 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_GUID_RACE_ACC_BY_NAME);
446 stmt->setString(0, name);
447 if (PreparedQueryResult result = CharacterDatabase.Query(stmt))
448 {
449 Field* fields = result->Fetch();
450 inviteeGuid = ObjectGuid(HighGuid::Player, fields[0].GetUInt32());
451 inviteeTeam = Player::TeamForRace(fields[1].GetUInt8());
452 inviteeGuildId = Player::GetGuildIdFromDB(inviteeGuid);
453 }
454 }
455
456 if (!inviteeGuid)
457 {
458 sCalendarMgr->SendCalendarCommandResult(playerGuid, CALENDAR_ERROR_PLAYER_NOT_FOUND);
459 return;
460 }
461
462 if (_player->GetTeam() != inviteeTeam && !sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_CALENDAR))
463 {
464 sCalendarMgr->SendCalendarCommandResult(playerGuid, CALENDAR_ERROR_NOT_ALLIED);
465 return;
466 }
467
468 if (QueryResult result = CharacterDatabase.PQuery("SELECT flags FROM character_social WHERE guid = %u AND friend = %u", inviteeGuid.GetCounter(), playerGuid.GetCounter()))
469 {
470 Field* fields = result->Fetch();
471 if (fields[0].GetUInt8() & SOCIAL_FLAG_IGNORED)
472 {
473 sCalendarMgr->SendCalendarCommandResult(playerGuid, CALENDAR_ERROR_IGNORING_YOU_S, name.c_str());
474 return;
475 }
476 }
477
478 if (!isPreInvite)
479 {
480 if (CalendarEvent* calendarEvent = sCalendarMgr->GetEvent(eventId))
481 {
482 if (calendarEvent->IsGuildEvent() && calendarEvent->GetGuildId() == inviteeGuildId)
483 {
484 // we can't invite guild members to guild events
485 sCalendarMgr->SendCalendarCommandResult(playerGuid, CALENDAR_ERROR_NO_GUILD_INVITES);
486 return;
487 }
488
489 // 946684800 is 01/01/2000 00:00:00 - default response time
490 CalendarInvite* invite = new CalendarInvite(sCalendarMgr->GetFreeInviteId(), eventId, inviteeGuid, playerGuid, 946684800, CALENDAR_STATUS_INVITED, CALENDAR_RANK_PLAYER, "");
491 sCalendarMgr->AddInvite(calendarEvent, invite);
492 }
493 else
494 sCalendarMgr->SendCalendarCommandResult(playerGuid, CALENDAR_ERROR_EVENT_INVALID);
495 }
496 else
497 {
498 if (isGuildEvent && inviteeGuildId == _player->GetGuildId())
499 {
500 sCalendarMgr->SendCalendarCommandResult(playerGuid, CALENDAR_ERROR_NO_GUILD_INVITES);
501 return;
502 }
503
504 // 946684800 is 01/01/2000 00:00:00 - default response time
505 CalendarInvite invite(inviteId, 0, inviteeGuid, playerGuid, 946684800, CALENDAR_STATUS_INVITED, CALENDAR_RANK_PLAYER, "");
506 sCalendarMgr->SendCalendarEventInvite(invite);
507 }
508}

510void WorldSession::HandleCalendarEventSignup(WorldPacket& recvData)
511{
512 ObjectGuid guid = _player->GetGUID();
513 uint64 eventId;
514 bool tentative;
515
516 recvData >> eventId >> tentative;
517 TC_LOG_DEBUG("network", "CMSG_CALENDAR_EVENT_SIGNUP [%s] EventId [" UI64FMTD "] Tentative %u", guid.ToString().c_str(), eventId, tentative);
518

529 sCalendarMgr->AddInvite(calendarEvent, invite);
530 sCalendarMgr->SendCalendarClearPendingAction(guid);
531 }
532 else
533 sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_EVENT_INVALID);
534}

566 else

568 }
569 else
570 sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_EVENT_INVALID);
571}

602void WorldSession::HandleCalendarEventStatus(WorldPacket& recvData)
603{
604 ObjectGuid guid = _player->GetGUID();
605 ObjectGuid invitee;
606 uint64 eventId;
607 uint64 inviteId;
608 uint64 ownerInviteId; // isn't it sender's inviteId?
609 uint8 status;
610
611 recvData >> invitee.ReadAsPacked();
612 recvData >> eventId >> inviteId >> ownerInviteId >> status;
613 TC_LOG_DEBUG("network", "CMSG_CALENDAR_EVENT_STATUS [%s] EventId ["
614 UI64FMTD "] ownerInviteId [" UI64FMTD "], Invitee ([%s] id: ["
615 UI64FMTD "], status %u", guid.ToString().c_str(), eventId, ownerInviteId, invitee.ToString().c_str(), inviteId, status);
616
617 if (CalendarEvent* calendarEvent = sCalendarMgr->GetEvent(eventId))
618 {
619 if (CalendarInvite* invite = sCalendarMgr->GetInvite(inviteId))
620 {
621 invite->SetStatus((CalendarInviteStatus)status);
622 // not sure if we should set response time when moderator changes invite status
623 //invite->SetStatusTime(time(NULL));
624
625 sCalendarMgr->UpdateInvite(invite);
626 sCalendarMgr->SendCalendarEventStatus(*calendarEvent, *invite);
627 sCalendarMgr->SendCalendarClearPendingAction(invitee);

726void WorldSession::SendCalendarRaidLockout(InstanceSave const* save, bool add)
727{
728 TC_LOG_DEBUG("network", "%s", add ? "SMSG_CALENDAR_RAID_LOCKOUT_ADDED" : "SMSG_CALENDAR_RAID_LOCKOUT_REMOVED");
729 time_t currTime = time(NULL);
730
731 WorldPacket data(SMSG_CALENDAR_RAID_LOCKOUT_REMOVED, (add ? 4 : 0) + 4 + 4 + 4 + 8);
732 if (add)
733 {
734 data.SetOpcode(SMSG_CALENDAR_RAID_LOCKOUT_ADDED);
735 data.AppendPackedTime(currTime);
736 }
737
738 data << uint32(save->GetMapId());
739 data << uint32(save->GetDifficulty());
740 data << uint32(save->GetResetTime() - currTime);
741 data << uint64(save->GetInstanceId());
742 SendPacket(&data);
743}

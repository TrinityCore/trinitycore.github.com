39void WorldSession::SendTaxiStatus(ObjectGuid guid)
40{
41 // cheating checks
42 Creature* unit = GetPlayer()->GetMap()->GetCreature(guid);
43 if (!unit)
44 {
45 TC_LOG_DEBUG("network", "WorldSession::SendTaxiStatus - %s not found.", guid.ToString().c_str());
46 return;
47 }
48
49 uint32 curloc = sObjectMgr->GetNearestTaxiNode(unit->GetPositionX(), unit->GetPositionY(), unit->GetPositionZ(), unit->GetMapId(), GetPlayer()->GetTeam());
50
51 // not found nearest
52 if (curloc == 0)
53 return;
54
55 TC_LOG_DEBUG("network", "WORLD: current location %u ", curloc);
56
57 WorldPacket data(SMSG_TAXINODE_STATUS, 9);
58 data << guid;
59 data << uint8(GetPlayer()->m_taxi.IsTaximaskNodeKnown(curloc) ? 1 : 0);
60 SendPacket(&data);
61 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_TAXINODE_STATUS");
62}

64void WorldSession::HandleTaxiQueryAvailableNodes(WorldPacket& recvData)
65{
66 TC_LOG_DEBUG("network", "WORLD: Received CMSG_TAXIQUERYAVAILABLENODES");
67
68 ObjectGuid guid;
69 recvData >> guid;
70
71 // cheating checks
72 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_FLIGHTMASTER);
73 if (!unit)
74 {
75 TC_LOG_DEBUG("network", "WORLD: HandleTaxiQueryAvailableNodes - %s not found or you can't interact with him.", guid.ToString().c_str());
76 return;
77 }
78
79 // remove fake death
80 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
81 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
82
83 // unknown taxi node case
84 if (SendLearnNewTaxiNode(unit))
85 return;
86
87 // known taxi node case
88 SendTaxiMenu(unit);
89}

91void WorldSession::SendTaxiMenu(Creature* unit)
92{
93 // find current node
94 uint32 curloc = sObjectMgr->GetNearestTaxiNode(unit->GetPositionX(), unit->GetPositionY(), unit->GetPositionZ(), unit->GetMapId(), GetPlayer()->GetTeam());
95
96 if (curloc == 0)
97 return;
98
99 bool lastTaxiCheaterState = GetPlayer()->isTaxiCheater();
100 if (unit->GetEntry() == 29480) GetPlayer()->SetTaxiCheater(true); // Grimwing in Ebon Hold, special case. NOTE: Not perfect, Zul'Aman should not be included according to WoWhead, and I think taxicheat includes it.
101
102 TC_LOG_DEBUG("network", "WORLD: CMSG_TAXINODE_STATUS_QUERY %u ", curloc);
103
104 WorldPacket data(SMSG_SHOWTAXINODES, (4 + 8 + 4 + 8 * 4));
105 data << uint32(1);
106 data << uint64(unit->GetGUID());
107 data << uint32(curloc);
108 GetPlayer()->m_taxi.AppendTaximaskTo(data, GetPlayer()->isTaxiCheater());
109 SendPacket(&data);
110
111 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_SHOWTAXINODES");
112
113 GetPlayer()->SetTaxiCheater(lastTaxiCheaterState);
114}

116void WorldSession::SendDoFlight(uint32 mountDisplayId, uint32 path, uint32 pathNode)
117{
118 // remove fake death
119 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
120 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
121
122 while (GetPlayer()->GetMotionMaster()->GetCurrentMovementGeneratorType() == FLIGHT_MOTION_TYPE)
123 GetPlayer()->GetMotionMaster()->MovementExpired(false);
124
125 if (mountDisplayId)
126 GetPlayer()->Mount(mountDisplayId);
127
128 GetPlayer()->GetMotionMaster()->MoveTaxiFlight(path, pathNode);
129}

131bool WorldSession::SendLearnNewTaxiNode(Creature* unit)
132{
133 // find current node
134 uint32 curloc = sObjectMgr->GetNearestTaxiNode(unit->GetPositionX(), unit->GetPositionY(), unit->GetPositionZ(), unit->GetMapId(), GetPlayer()->GetTeam());
135
136 if (curloc == 0)
137 return true; // `true` send to avoid WorldSession::SendTaxiMenu call with one more curlock seartch with same false result.
138
139 if (GetPlayer()->m_taxi.SetTaximaskNode(curloc))
140 {
141 WorldPacket msg(SMSG_NEW_TAXI_PATH, 0);
142 SendPacket(&msg);
143
144 WorldPacket update(SMSG_TAXINODE_STATUS, 9);
145 update << uint64(unit->GetGUID());
146 update << uint8(1);
147 SendPacket(&update);
148
149 return true;
150 }
151 else
152 return false;
153}

164void WorldSession::HandleActivateTaxiExpressOpcode (WorldPacket& recvData)
165{
166 TC_LOG_DEBUG("network", "WORLD: Received CMSG_ACTIVATETAXIEXPRESS");
167
168 ObjectGuid guid;
169 uint32 node_count;
170
171 recvData >> guid >> node_count;
172
173 Creature* npc = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_FLIGHTMASTER);
174 if (!npc)
175 {
176 TC_LOG_DEBUG("network", "WORLD: HandleActivateTaxiExpressOpcode - %s not found or you can't interact with it.", guid.ToString().c_str());
177 return;
178 }
179 std::vector<uint32> nodes;
180
181 for (uint32 i = 0; i < node_count; ++i)
182 {
183 uint32 node;
184 recvData >> node;
185
186 if (!GetPlayer()->m_taxi.IsTaximaskNodeKnown(node) && !GetPlayer()->isTaxiCheater())
187 {
188 SendActivateTaxiReply(ERR_TAXINOTVISITED);
189 recvData.rfinish();
190 return;
191 }
192
193 nodes.push_back(node);
194 }
195
196 if (nodes.empty())
197 return;
198
199 TC_LOG_DEBUG("network", "WORLD: Received CMSG_ACTIVATETAXIEXPRESS from %d to %d", nodes.front(), nodes.back());
200
201 GetPlayer()->ActivateTaxiPathTo(nodes, npc);
202}

204void WorldSession::HandleMoveSplineDoneOpcode(WorldPacket& recvData)
205{
206 TC_LOG_DEBUG("network", "WORLD: Received CMSG_MOVE_SPLINE_DONE");
207
208 uint64 guid; // used only for proper packet read
209 recvData.readPackGUID(guid);
210
211 MovementInfo movementInfo; // used only for proper packet read
212 ReadMovementInfo(recvData, &movementInfo);
213
214 recvData.read_skip<uint32>(); // spline id
215
216 // in taxi flight packet received in 2 case:
217 // 1) end taxi path in far (multi-node) flight
218 // 2) switch from one map to other in case multim-map taxi path
219 // we need process only (1)
220
221 uint32 curDest = GetPlayer()->m_taxi.GetTaxiDestination();
222 if (curDest)
223 {
224 TaxiNodesEntry const* curDestNode = sTaxiNodesStore.LookupEntry(curDest);
225
226 // far teleport case
227 if (curDestNode && curDestNode->map_id != GetPlayer()->GetMapId())
228 {
229 if (GetPlayer()->GetMotionMaster()->GetCurrentMovementGeneratorType() == FLIGHT_MOTION_TYPE)
230 {
231 // short preparations to continue flight
232 FlightPathMovementGenerator* flight = (FlightPathMovementGenerator*)(GetPlayer()->GetMotionMaster()->top());
233
234 flight->SetCurrentNodeAfterTeleport();
235 TaxiPathNodeEntry const* node = flight->GetPath()[flight->GetCurrentNode()];
236 flight->SkipCurrentNode();
237
238 GetPlayer()->TeleportTo(curDestNode->map_id, node->LocX, node->LocY, node->LocZ, GetPlayer()->GetOrientation());
239 }
240 }
241
242 return;
243 }
244
245 // at this point only 1 node is expected (final destination)
246 if (GetPlayer()->m_taxi.GetPath().size() != 1)
247 return;
248
249 GetPlayer()->CleanupAfterTaxiFlight();
250 GetPlayer()->SetFallInformation(0, GetPlayer()->GetPositionZ());
251 if (GetPlayer()->pvpInfo.IsHostile)
252 GetPlayer()->CastSpell(GetPlayer(), 2479, true);
253}

255void WorldSession::HandleActivateTaxiOpcode(WorldPacket& recvData)
256{
257 TC_LOG_DEBUG("network", "WORLD: Received CMSG_ACTIVATETAXI");
258
259 ObjectGuid guid;
260 std::vector<uint32> nodes;
261 nodes.resize(2);
262
263 recvData >> guid >> nodes[0] >> nodes[1];
264 TC_LOG_DEBUG("network", "WORLD: Received CMSG_ACTIVATETAXI from %d to %d", nodes[0], nodes[1]);
265 Creature* npc = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_FLIGHTMASTER);
266 if (!npc)
267 {
268 TC_LOG_DEBUG("network", "WORLD: HandleActivateTaxiOpcode - %s not found or you can't interact with it.", guid.ToString().c_str());
269 return;
270 }
271
272 if (!GetPlayer()->isTaxiCheater())
273 {
274 if (!GetPlayer()->m_taxi.IsTaximaskNodeKnown(nodes[0]) || !GetPlayer()->m_taxi.IsTaximaskNodeKnown(nodes[1]))
275 {
276 SendActivateTaxiReply(ERR_TAXINOTVISITED);
277 return;
278 }
279 }
280
281 GetPlayer()->ActivateTaxiPathTo(nodes, npc);
282}

28void WorldSession::HandleAttackSwingOpcode(WorldPacket& recvData)
29{
30 ObjectGuid guid;
31 recvData >> guid;
32
33 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_ATTACKSWING Message %s", guid.ToString().c_str());
34
35 Unit* pEnemy = ObjectAccessor::GetUnit(*_player, guid);
36
37 if (!pEnemy)
38 {
39 // stop attack state at client
40 SendAttackStop(NULL);
41 return;
42 }
43
44 if (!_player->IsValidAttackTarget(pEnemy))
45 {
46 // stop attack state at client
47 SendAttackStop(pEnemy);
48 return;
49 }
50
51 //! Client explicitly checks the following before sending CMSG_ATTACKSWING packet,
52 //! so we'll place the same check here. Note that it might be possible to reuse this snippet
53 //! in other places as well.
54 if (Vehicle* vehicle = _player->GetVehicle())
55 {
56 VehicleSeatEntry const* seat = vehicle->GetSeatForPassenger(_player);
57 ASSERT(seat);
58 if (!(seat->m_flags & VEHICLE_SEAT_FLAG_CAN_ATTACK))
59 {
60 SendAttackStop(pEnemy);
61 return;
62 }
63 }
64
65 _player->Attack(pEnemy, true);
66}

73void WorldSession::HandleSetSheathedOpcode(WorldPacket& recvData)
74{
75 uint32 sheathed;
76 recvData >> sheathed;
77
78 //TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_SETSHEATHED Message guidlow:%u value1:%u", GetPlayer()->GetGUID().GetCounter(), sheathed);
79
80 if (sheathed >= MAX_SHEATH_STATE)
81 {
82 TC_LOG_ERROR("network", "Unknown sheath state %u ??", sheathed);
83 return;
84 }
85
86 GetPlayer()->SetSheath(SheathState(sheathed));
87}

89void WorldSession::SendAttackStop(Unit const* enemy)
90{
91 WorldPacket data(SMSG_ATTACKSTOP, (8+8+4)); // we guess size
92 data << GetPlayer()->GetPackGUID();
93 if (enemy)
94 data << enemy->GetPackGUID();
95 else
96 data << uint8(0);
97
98 data << uint32(0); // unk, can be 1 also
99 SendPacket(&data);
100}

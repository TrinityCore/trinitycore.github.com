59void WorldSession::HandleGroupInviteOpcode(WorldPacket& recvData)
60{
61 TC_LOG_DEBUG("network", "WORLD: Received CMSG_GROUP_INVITE");
62
63 std::string membername;
64 recvData >> membername;
65 recvData.read_skip<uint32>();
66
67 // attempt add selected player
68
69 // cheating
70 if (!normalizePlayerName(membername))
71 {
72 SendPartyResult(PARTY_OP_INVITE, membername, ERR_BAD_PLAYER_NAME_S);
73 return;
74 }
75
76 Player* player = ObjectAccessor::FindPlayerByName(membername);
77
78 // no player
79 if (!player)
80 {
81 SendPartyResult(PARTY_OP_INVITE, membername, ERR_BAD_PLAYER_NAME_S);
82 return;
83 }
84
85 // restrict invite to GMs
86 if (!sWorld->getBoolConfig(CONFIG_ALLOW_GM_GROUP) && !GetPlayer()->IsGameMaster() && player->IsGameMaster())
87 {
88 SendPartyResult(PARTY_OP_INVITE, membername, ERR_BAD_PLAYER_NAME_S);
89 return;
90 }
91
92 // can't group with
93 if (!GetPlayer()->IsGameMaster() && !sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GROUP) && GetPlayer()->GetTeam() != player->GetTeam())
94 {
95 SendPartyResult(PARTY_OP_INVITE, membername, ERR_PLAYER_WRONG_FACTION);
96 return;
97 }
98 if (GetPlayer()->GetInstanceId() != 0 && player->GetInstanceId() != 0 && GetPlayer()->GetInstanceId() != player->GetInstanceId() && GetPlayer()->GetMapId() == player->GetMapId())
99 {
100 SendPartyResult(PARTY_OP_INVITE, membername, ERR_TARGET_NOT_IN_INSTANCE_S);
101 return;
102 }
103 // just ignore us
104 if (player->GetInstanceId() != 0 && player->GetDungeonDifficulty() != GetPlayer()->GetDungeonDifficulty())
105 {
106 SendPartyResult(PARTY_OP_INVITE, membername, ERR_IGNORING_YOU_S);
107 return;
108 }
109
110 if (player->GetSocial()->HasIgnore(GetPlayer()->GetGUID().GetCounter()))
111 {
112 SendPartyResult(PARTY_OP_INVITE, membername, ERR_IGNORING_YOU_S);
113 return;
114 }
115
116 Group* group = GetPlayer()->GetGroup();
117 if (group && group->isBGGroup())
118 group = GetPlayer()->GetOriginalGroup();
119
120 Group* group2 = player->GetGroup();
121 if (group2 && group2->isBGGroup())
122 group2 = player->GetOriginalGroup();
123 // player already in another group or invited
124 if (group2 || player->GetGroupInvite())
125 {
126 SendPartyResult(PARTY_OP_INVITE, membername, ERR_ALREADY_IN_GROUP_S);
127
128 if (group2)
129 {
130 // tell the player that they were invited but it failed as they were already in a group
131 WorldPacket data(SMSG_GROUP_INVITE, 10); // guess size
132 data << uint8(0); // invited/already in group flag
133 data << GetPlayer()->GetName(); // max len 48
134 data << uint32(0); // unk
135 data << uint8(0); // count
136 data << uint32(0); // unk
137 player->GetSession()->SendPacket(&data);
138 }
139
140 return;
141 }
142
143 if (group)
144 {
145 // not have permissions for invite
146 if (!group->IsLeader(GetPlayer()->GetGUID()) && !group->IsAssistant(GetPlayer()->GetGUID()))
147 {
148 SendPartyResult(PARTY_OP_INVITE, "", ERR_NOT_LEADER);
149 return;
150 }
151 // not have place
152 if (group->IsFull())
153 {
154 SendPartyResult(PARTY_OP_INVITE, "", ERR_GROUP_FULL);
155 return;
156 }
157 }
158
159 // ok, but group not exist, start a new group
160 // but don't create and save the group to the DB until
161 // at least one person joins
162 if (!group)
163 {
164 group = new Group;
165 // new group: if can't add then delete
166 if (!group->AddLeaderInvite(GetPlayer()))
167 {
168 delete group;
169 return;
170 }
171 if (!group->AddInvite(player))
172 {
173 delete group;
174 return;
175 }
176 }
177 else
178 {
179 // already existed group: if can't add then just leave
180 if (!group->AddInvite(player))
181 {
182 return;
183 }
184 }
185
186 // ok, we do it
187 WorldPacket data(SMSG_GROUP_INVITE, 10); // guess size
188 data << uint8(1); // invited/already in group flag
189 data << GetPlayer()->GetName(); // max len 48
190 data << uint32(0); // unk
191 data << uint8(0); // count
192 data << uint32(0); // unk
193 player->GetSession()->SendPacket(&data);
194
195 SendPartyResult(PARTY_OP_INVITE, membername, ERR_PARTY_RESULT_OK);
196}

198void WorldSession::HandleGroupAcceptOpcode(WorldPacket& recvData)
199{
200 TC_LOG_DEBUG("network", "WORLD: Received CMSG_GROUP_ACCEPT");
201
202 recvData.read_skip<uint32>();
203 Group* group = GetPlayer()->GetGroupInvite();
204
205 if (!group)
206 return;
207
208 // Remove player from invitees in any case
209 group->RemoveInvite(GetPlayer());
210
211 if (group->GetLeaderGUID() == GetPlayer()->GetGUID())
212 {
213 TC_LOG_ERROR("network", "HandleGroupAcceptOpcode: player %s(%d) tried to accept an invite to his own group", GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().GetCounter());
214 return;
215 }
216
217 // Group is full
218 if (group->IsFull())
219 {
220 SendPartyResult(PARTY_OP_INVITE, "", ERR_GROUP_FULL);
221 return;
222 }
223
224 Player* leader = ObjectAccessor::FindPlayer(group->GetLeaderGUID());
225
226 // Forming a new group, create it
227 if (!group->IsCreated())
228 {
229 // This can happen if the leader is zoning. To be removed once delayed actions for zoning are implemented
230 if (!leader)
231 {
232 group->RemoveAllInvites();
233 return;
234 }
235
236 // If we're about to create a group there really should be a leader present
237 ASSERT(leader);
238 group->RemoveInvite(leader);
239 group->Create(leader);
240 sGroupMgr->AddGroup(group);
241 }
242
243 // Everything is fine, do it, PLAYER'S GROUP IS SET IN ADDMEMBER!!!
244 if (!group->AddMember(GetPlayer()))
245 return;
246
247 group->BroadcastGroupUpdate();
248}

273void WorldSession::HandleGroupUninviteGuidOpcode(WorldPacket& recvData)
274{
275 TC_LOG_DEBUG("network", "WORLD: Received CMSG_GROUP_UNINVITE_GUID");
276
277 ObjectGuid guid;
278 std::string reason;
279 recvData >> guid;
280 recvData >> reason;
281
282 //can't uninvite yourself
283 if (guid == GetPlayer()->GetGUID())
284 {
285 TC_LOG_ERROR("network", "WorldSession::HandleGroupUninviteGuidOpcode: leader %s(%d) tried to uninvite himself from the group.",
286 GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().GetCounter());
287 return;
288 }
289
290 PartyResult res = GetPlayer()->CanUninviteFromGroup(guid);
291 if (res != ERR_PARTY_RESULT_OK)
292 {
293 SendPartyResult(PARTY_OP_UNINVITE, "", res);
294 return;
295 }
296
297 Group* grp = GetPlayer()->GetGroup();
298 // grp is checked already above in CanUninviteFromGroup()
299 ASSERT(grp);
300
301 if (grp->IsMember(guid))
302 {
303 Player::RemoveFromGroup(grp, guid, GROUP_REMOVEMETHOD_KICK, GetPlayer()->GetGUID(), reason.c_str());
304 return;
305 }
306
307 if (Player* player = grp->GetInvited(guid))
308 {
309 player->UninviteFromGroup();
310 return;
311 }
312
313 SendPartyResult(PARTY_OP_UNINVITE, "", ERR_TARGET_NOT_IN_GROUP_S);
314}

316void WorldSession::HandleGroupUninviteOpcode(WorldPacket& recvData)
317{
318 TC_LOG_DEBUG("network", "WORLD: Received CMSG_GROUP_UNINVITE");
319
320 std::string membername;
321 recvData >> membername;
322
323 // player not found
324 if (!normalizePlayerName(membername))
325 return;
326
327 // can't uninvite yourself
328 if (GetPlayer()->GetName() == membername)
329 {
330 TC_LOG_ERROR("network", "WorldSession::HandleGroupUninviteOpcode: leader %s(%d) tried to uninvite himself from the group.",
331 GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().GetCounter());
332 return;
333 }
334
335 PartyResult res = GetPlayer()->CanUninviteFromGroup();
336 if (res != ERR_PARTY_RESULT_OK)
337 {
338 SendPartyResult(PARTY_OP_UNINVITE, "", res);
339 return;
340 }
341
342 Group* grp = GetPlayer()->GetGroup();
343 if (!grp)
344 return;
345
346 if (ObjectGuid guid = grp->GetMemberGUID(membername))
347 {
348 Player::RemoveFromGroup(grp, guid, GROUP_REMOVEMETHOD_KICK, GetPlayer()->GetGUID());
349 return;
350 }
351
352 if (Player* player = grp->GetInvited(membername))
353 {
354 player->UninviteFromGroup();
355 return;
356 }
357
358 SendPartyResult(PARTY_OP_UNINVITE, membername, ERR_TARGET_NOT_IN_GROUP_S);

405void WorldSession::HandleLootMethodOpcode(WorldPacket& recvData)
406{
407 TC_LOG_DEBUG("network", "WORLD: Received CMSG_LOOT_METHOD");
408
409 uint32 lootMethod;
410 ObjectGuid lootMaster;
411 uint32 lootThreshold;
412 recvData >> lootMethod >> lootMaster >> lootThreshold;
413
414 Group* group = GetPlayer()->GetGroup();
415 if (!group)
416 return;
417
418 /** error handling **/
419 if (!group->IsLeader(GetPlayer()->GetGUID()))
420 return;
421
422 if (lootMethod > NEED_BEFORE_GREED)
423 return;
424
425 if (lootThreshold < ITEM_QUALITY_UNCOMMON || lootThreshold > ITEM_QUALITY_ARTIFACT)
426 return;
427
428 if (lootMethod == MASTER_LOOT && !group->IsMember(lootMaster))
429 return;
430 /********************/
431
432 // everything's fine, do it
433 group->SetLootMethod((LootMethod)lootMethod);
434 group->SetMasterLooterGuid(lootMaster);
435 group->SetLootThreshold((ItemQualities)lootThreshold);
436 group->SendUpdate();
437}

439void WorldSession::HandleLootRoll(WorldPacket& recvData)
440{
441 ObjectGuid guid;
442 uint32 itemSlot;
443 uint8 rollType;
444 recvData >> guid; // guid of the item rolled
445 recvData >> itemSlot;
446 recvData >> rollType; // 0: pass, 1: need, 2: greed
447
448 Group* group = GetPlayer()->GetGroup();
449 if (!group)
450 return;
451
452 group->CountRollVote(GetPlayer()->GetGUID(), guid, rollType);
453
454 switch (rollType)
455 {
456 case ROLL_NEED:
457 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_ROLL_NEED, 1);
458 break;
459 case ROLL_GREED:
460 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_ROLL_GREED, 1);
461 break;
462 }

579{

581

584 if (!group)
585 return;

636void WorldSession::HandlePartyAssignmentOpcode(WorldPacket& recvData)
637{
638 TC_LOG_DEBUG("network", "WORLD: Received MSG_PARTY_ASSIGNMENT");
639
640 Group* group = GetPlayer()->GetGroup();
641 if (!group)
642 return;
643
644 ObjectGuid senderGuid = GetPlayer()->GetGUID();
645 if (!group->IsLeader(senderGuid) && !group->IsAssistant(senderGuid))
646 return;
647
648 uint8 assignment;
649 bool apply;
650 ObjectGuid guid;
651 recvData >> assignment >> apply;
652 recvData >> guid;
653
654 switch (assignment)
655 {
656 case GROUP_ASSIGN_MAINASSIST:
657 group->RemoveUniqueGroupMemberFlag(MEMBER_FLAG_MAINASSIST);
658 group->SetGroupMemberFlag(guid, apply, MEMBER_FLAG_MAINASSIST);
659 break;
660 case GROUP_ASSIGN_MAINTANK:
661 group->RemoveUniqueGroupMemberFlag(MEMBER_FLAG_MAINTANK); // Remove main assist flag from current if any.
662 group->SetGroupMemberFlag(guid, apply, MEMBER_FLAG_MAINTANK);
663 default:
664 break;
665 }
666
667 group->SendUpdate();
668}

717void WorldSession::BuildPartyMemberStatsChangedPacket(Player* player, WorldPacket* data)
718{
719 uint32 mask = player->GetGroupUpdateFlag();
720
721 if (mask == GROUP_UPDATE_FLAG_NONE)
722 return;
723
724 if (mask & GROUP_UPDATE_FLAG_POWER_TYPE) // if update power type, update current/max power also
725 mask |= (GROUP_UPDATE_FLAG_CUR_POWER | GROUP_UPDATE_FLAG_MAX_POWER);
726
727 if (mask & GROUP_UPDATE_FLAG_PET_POWER_TYPE) // same for pets
728 mask |= (GROUP_UPDATE_FLAG_PET_CUR_POWER | GROUP_UPDATE_FLAG_PET_MAX_POWER);
729
730 uint32 byteCount = 0;
731 for (int i = 1; i < GROUP_UPDATE_FLAGS_COUNT; ++i)
732 if (mask & (1 << i))
733 byteCount += GroupUpdateLength[i];
734
735 data->Initialize(SMSG_PARTY_MEMBER_STATS, 8 + 4 + byteCount);
736 *data << player->GetPackGUID();
737 *data << uint32(mask);
738
739 if (mask & GROUP_UPDATE_FLAG_STATUS)
740 {
741 uint16 playerStatus = MEMBER_STATUS_ONLINE;
742 if (player->IsPvP())
743 playerStatus |= MEMBER_STATUS_PVP;
744
745 if (!player->IsAlive())
746 {
747 if (player->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_GHOST))
748 playerStatus |= MEMBER_STATUS_GHOST;
749 else
750 playerStatus |= MEMBER_STATUS_DEAD;
751 }
752
753 if (player->IsFFAPvP())
754 playerStatus |= MEMBER_STATUS_PVP_FFA;
755
756 if (player->isAFK())
757 playerStatus |= MEMBER_STATUS_AFK;
758
759 if (player->isDND())
760 playerStatus |= MEMBER_STATUS_DND;
761
762 *data << uint16(playerStatus);
763 }
764
765 if (mask & GROUP_UPDATE_FLAG_CUR_HP)
766 *data << uint32(player->GetHealth());
767
768 if (mask & GROUP_UPDATE_FLAG_MAX_HP)
769 *data << uint32(player->GetMaxHealth());
770
771 Powers powerType = player->getPowerType();
772 if (mask & GROUP_UPDATE_FLAG_POWER_TYPE)
773 *data << uint8(powerType);
774
775 if (mask & GROUP_UPDATE_FLAG_CUR_POWER)
776 *data << uint16(player->GetPower(powerType));
777
778 if (mask & GROUP_UPDATE_FLAG_MAX_POWER)
779 *data << uint16(player->GetMaxPower(powerType));
780
781 if (mask & GROUP_UPDATE_FLAG_LEVEL)

902/*this procedure handles clients CMSG_REQUEST_PARTY_MEMBER_STATS request*/
903void WorldSession::HandleRequestPartyMemberStatsOpcode(WorldPacket &recvData)
904{
905 TC_LOG_DEBUG("network", "WORLD: Received CMSG_REQUEST_PARTY_MEMBER_STATS");
906 ObjectGuid Guid;
907 recvData >> Guid;
908
909 Player* player = ObjectAccessor::FindConnectedPlayer(Guid);
910 if (!player)
911 {
912 WorldPacket data(SMSG_PARTY_MEMBER_STATS_FULL, 3+4+2);
913 data << uint8(0); // only for SMSG_PARTY_MEMBER_STATS_FULL, probably arena/bg related
914 data << Guid.WriteAsPacked();
915 data << uint32(GROUP_UPDATE_FLAG_STATUS);
916 data << uint16(MEMBER_STATUS_OFFLINE);
917 SendPacket(&data);
918 return;
919 }
920
921 Pet* pet = player->GetPet();
922 Powers powerType = player->getPowerType();
923
924 WorldPacket data(SMSG_PARTY_MEMBER_STATS_FULL, 4+2+2+2+1+2*6+8+1+8);
925 data << uint8(0); // only for SMSG_PARTY_MEMBER_STATS_FULL, probably arena/bg related
926 data << player->GetPackGUID();
927
928 uint32 updateFlags = GROUP_UPDATE_FLAG_STATUS | GROUP_UPDATE_FLAG_CUR_HP | GROUP_UPDATE_FLAG_MAX_HP
929 | GROUP_UPDATE_FLAG_CUR_POWER | GROUP_UPDATE_FLAG_MAX_POWER | GROUP_UPDATE_FLAG_LEVEL
930 | GROUP_UPDATE_FLAG_ZONE | GROUP_UPDATE_FLAG_POSITION | GROUP_UPDATE_FLAG_AURAS
931 | GROUP_UPDATE_FLAG_PET_NAME | GROUP_UPDATE_FLAG_PET_MODEL_ID | GROUP_UPDATE_FLAG_PET_AURAS;
932
933 if (powerType != POWER_MANA)
934 updateFlags |= GROUP_UPDATE_FLAG_POWER_TYPE;
935
936 if (pet)
937 updateFlags |= GROUP_UPDATE_FLAG_PET_GUID | GROUP_UPDATE_FLAG_PET_CUR_HP | GROUP_UPDATE_FLAG_PET_MAX_HP
938 | GROUP_UPDATE_FLAG_PET_POWER_TYPE | GROUP_UPDATE_FLAG_PET_CUR_POWER | GROUP_UPDATE_FLAG_PET_MAX_POWER;
939
940 if (player->GetVehicle())
941 updateFlags |= GROUP_UPDATE_FLAG_VEHICLE_SEAT;
942
943 uint16 playerStatus = MEMBER_STATUS_ONLINE;
944 if (player->IsPvP())
945 playerStatus |= MEMBER_STATUS_PVP;
946
947 if (!player->IsAlive())
948 {
949 if (player->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_GHOST))
950 playerStatus |= MEMBER_STATUS_GHOST;
951 else
952 playerStatus |= MEMBER_STATUS_DEAD;
953 }
954
955 if (player->IsFFAPvP())
956 playerStatus |= MEMBER_STATUS_PVP_FFA;
957
958 if (player->isAFK())
959 playerStatus |= MEMBER_STATUS_AFK;
960
961 if (player->isDND())
962 playerStatus |= MEMBER_STATUS_DND;
963
964 data << uint32(updateFlags);
965 data << uint16(playerStatus); // GROUP_UPDATE_FLAG_STATUS
966 data << uint32(player->GetHealth()); // GROUP_UPDATE_FLAG_CUR_HP
967 data << uint32(player->GetMaxHealth()); // GROUP_UPDATE_FLAG_MAX_HP
968 if (updateFlags & GROUP_UPDATE_FLAG_POWER_TYPE)
969 data << uint8(powerType);
970
971 data << uint16(player->GetPower(powerType)); // GROUP_UPDATE_FLAG_CUR_POWER
972 data << uint16(player->GetMaxPower(powerType)); // GROUP_UPDATE_FLAG_MAX_POWER
973 data << uint16(player->getLevel()); // GROUP_UPDATE_FLAG_LEVEL
974 data << uint16(player->GetZoneId()); // GROUP_UPDATE_FLAG_ZONE
975 data << uint16(player->GetPositionX()); // GROUP_UPDATE_FLAG_POSITION
976 data << uint16(player->GetPositionY()); // GROUP_UPDATE_FLAG_POSITION
977
978 uint64 auraMask = 0;
979 size_t maskPos = data.wpos();
980 data << uint64(auraMask); // placeholder
981 for (uint8 i = 0; i < MAX_AURAS; ++i)
982 {
983 if (AuraApplication const* aurApp = player->GetVisibleAura(i))
984 {
985 auraMask |= uint64(1) << i;
986 data << uint32(aurApp->GetBase()->GetId());
987 data << uint8(aurApp->GetFlags());
988 }
989 }
990
991 data.put<uint64>(maskPos, auraMask); // GROUP_UPDATE_FLAG_AURAS
992
993 if (updateFlags & GROUP_UPDATE_FLAG_PET_GUID)
994 data << uint64(ASSERT_NOTNULL(pet)->GetGUID());
995
996 data << std::string(pet ? pet->GetName() : ""); // GROUP_UPDATE_FLAG_PET_NAME
997 data << uint16(pet ? pet->GetDisplayId() : 0); // GROUP_UPDATE_FLAG_PET_MODEL_ID
998
999 if (updateFlags & GROUP_UPDATE_FLAG_PET_CUR_HP)
1000 data << uint32(pet->GetHealth());
1001
1002 if (updateFlags & GROUP_UPDATE_FLAG_PET_MAX_HP)
1003 data << uint32(pet->GetMaxHealth());
1004
1005 if (updateFlags & GROUP_UPDATE_FLAG_PET_POWER_TYPE)
1006 data << (uint8)pet->getPowerType();
1007
1008 if (updateFlags & GROUP_UPDATE_FLAG_PET_CUR_POWER)
1009 data << uint16(pet->GetPower(pet->getPowerType()));
1010
1011 if (updateFlags & GROUP_UPDATE_FLAG_PET_MAX_POWER)
1012 data << uint16(pet->GetMaxPower(pet->getPowerType()));
1013
1014 uint64 petAuraMask = 0;
1015 maskPos = data.wpos();
1016 data << uint64(petAuraMask); // placeholder
1017 if (pet)
1018 {
1019 for (uint8 i = 0; i < MAX_AURAS; ++i)
1020 {
1021 if (AuraApplication const* aurApp = pet->GetVisibleAura(i))
1022 {
1023 petAuraMask |= uint64(1) << i;
1024 data << uint32(aurApp->GetBase()->GetId());
1025 data << uint8(aurApp->GetFlags());
1026 }
1027 }
1028 }
1029
1030 data.put<uint64>(maskPos, petAuraMask); // GROUP_UPDATE_FLAG_PET_AURAS
1031
1032 if (updateFlags & GROUP_UPDATE_FLAG_VEHICLE_SEAT)
1033 data << uint32(player->GetVehicle()->GetVehicleInfo()->m_seatID[player->m_movementInfo.transport.seat]);
1034
1035 SendPacket(&data);
1036}

24bool AddonHandler::BuildAddonPacket(WorldPacket* source, WorldPacket* target)
25{
26 ByteBuffer AddOnPacked;
27 uLongf AddonRealSize;
28 uint32 CurrentPosition;
29 uint32 TempValue;
30
31 // broken addon packet, can't be received from real client
32 if (source->rpos() + 4 > source->size())
33 return false;
34
35 *source >> TempValue; // get real size of the packed structure
36
37 // empty addon packet, nothing process, can't be received from real client
38 if (!TempValue)
39 return false;
40
41 AddonRealSize = TempValue; // temp value because ZLIB only excepts uLongf
42
43 CurrentPosition = source->rpos(); // get the position of the pointer in the structure
44
45 AddOnPacked.resize(AddonRealSize); // resize target for zlib action
46
47 if (uncompress(AddOnPacked.contents(), &AddonRealSize, source->contents() + CurrentPosition, source->size() - CurrentPosition) == Z_OK)
48 {
49 target->Initialize(SMSG_ADDON_INFO);
50
51 uint32 addonsCount;
52 AddOnPacked >> addonsCount; // addons count?
53
54 for (uint32 i = 0; i < addonsCount; ++i)
55 {
56 std::string addonName;
57 uint8 enabled;
58 uint32 crc, unk2;
59
60 // check next addon data format correctness
61 if (AddOnPacked.rpos()+1 > AddOnPacked.size())
62 return false;
63
64 AddOnPacked >> addonName;
65
66 // recheck next addon data format correctness
67 if (AddOnPacked.rpos()+1+4+4 > AddOnPacked.size())
68 return false;
69
70 AddOnPacked >> enabled >> crc >> unk2;
71
72 TC_LOG_DEBUG("network", "ADDON: Name: %s, Enabled: 0x%x, CRC: 0x%x, Unknown2: 0x%x", addonName.c_str(), enabled, crc, unk2);
73
74 uint8 state = (enabled ? 2 : 1);
75 *target << uint8(state);
76
77 uint8 unk1 = (enabled ? 1 : 0);
78 *target << uint8(unk1);
79 if (unk1)
80 {
81 uint8 unk = (crc != 0x4c1c776d); // If addon is Standard addon CRC
82 *target << uint8(unk);
83 if (unk)
84 {
85 unsigned char tdata[256] =
86 {
87 0xC3, 0x5B, 0x50, 0x84, 0xB9, 0x3E, 0x32, 0x42, 0x8C, 0xD0, 0xC7, 0x48, 0xFA, 0x0E, 0x5D, 0x54,
88 0x5A, 0xA3, 0x0E, 0x14, 0xBA, 0x9E, 0x0D, 0xB9, 0x5D, 0x8B, 0xEE, 0xB6, 0x84, 0x93, 0x45, 0x75,
89 0xFF, 0x31, 0xFE, 0x2F, 0x64, 0x3F, 0x3D, 0x6D, 0x07, 0xD9, 0x44, 0x9B, 0x40, 0x85, 0x59, 0x34,
90 0x4E, 0x10, 0xE1, 0xE7, 0x43, 0x69, 0xEF, 0x7C, 0x16, 0xFC, 0xB4, 0xED, 0x1B, 0x95, 0x28, 0xA8,
91 0x23, 0x76, 0x51, 0x31, 0x57, 0x30, 0x2B, 0x79, 0x08, 0x50, 0x10, 0x1C, 0x4A, 0x1A, 0x2C, 0xC8,
92 0x8B, 0x8F, 0x05, 0x2D, 0x22, 0x3D, 0xDB, 0x5A, 0x24, 0x7A, 0x0F, 0x13, 0x50, 0x37, 0x8F, 0x5A,
93 0xCC, 0x9E, 0x04, 0x44, 0x0E, 0x87, 0x01, 0xD4, 0xA3, 0x15, 0x94, 0x16, 0x34, 0xC6, 0xC2, 0xC3,
94 0xFB, 0x49, 0xFE, 0xE1, 0xF9, 0xDA, 0x8C, 0x50, 0x3C, 0xBE, 0x2C, 0xBB, 0x57, 0xED, 0x46, 0xB9,
95 0xAD, 0x8B, 0xC6, 0xDF, 0x0E, 0xD6, 0x0F, 0xBE, 0x80, 0xB3, 0x8B, 0x1E, 0x77, 0xCF, 0xAD, 0x22,
96 0xCF, 0xB7, 0x4B, 0xCF, 0xFB, 0xF0, 0x6B, 0x11, 0x45, 0x2D, 0x7A, 0x81, 0x18, 0xF2, 0x92, 0x7E,
97 0x98, 0x56, 0x5D, 0x5E, 0x69, 0x72, 0x0A, 0x0D, 0x03, 0x0A, 0x85, 0xA2, 0x85, 0x9C, 0xCB, 0xFB,
98 0x56, 0x6E, 0x8F, 0x44, 0xBB, 0x8F, 0x02, 0x22, 0x68, 0x63, 0x97, 0xBC, 0x85, 0xBA, 0xA8, 0xF7,
99 0xB5, 0x40, 0x68, 0x3C, 0x77, 0x86, 0x6F, 0x4B, 0xD7, 0x88, 0xCA, 0x8A, 0xD7, 0xCE, 0x36, 0xF0,
100 0x45, 0x6E, 0xD5, 0x64, 0x79, 0x0F, 0x17, 0xFC, 0x64, 0xDD, 0x10, 0x6F, 0xF3, 0xF5, 0xE0, 0xA6,
101 0xC3, 0xFB, 0x1B, 0x8C, 0x29, 0xEF, 0x8E, 0xE5, 0x34, 0xCB, 0xD1, 0x2A, 0xCE, 0x79, 0xC3, 0x9A,
102 0x0D, 0x36, 0xEA, 0x01, 0xE0, 0xAA, 0x91, 0x20, 0x54, 0xF0, 0x72, 0xD8, 0x1E, 0xC7, 0x89, 0xD2
103 };
104 target->append(tdata, sizeof(tdata));
105 }
106
107 *target << uint32(0);
108 }
109
110 uint8 unk3 = (enabled ? 0 : 1);
111 *target << uint8(unk3);
112 if (unk3)
113 {
114 // String, 256 (null terminated?)
115 *target << uint8(0);
116 }
117 }
118
119 uint32 unk4;
120 AddOnPacked >> unk4;
121
122 uint32 count = 0;
123 *target << uint32(count);
124
125 if (AddOnPacked.rpos() != AddOnPacked.size())
126 TC_LOG_DEBUG("network", "packet under read!");
127 }
128 else
129 {
130 TC_LOG_ERROR("network", "Addon packet uncompress error :(");
131 return false;
132 }
133 return true;
134}

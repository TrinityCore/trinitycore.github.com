31void WorldSession::SendNameQueryOpcode(ObjectGuid guid)
32{
33 Player* player = ObjectAccessor::FindConnectedPlayer(guid);
34 CharacterInfo const* nameData = sWorld->GetCharacterInfo(guid);
35
36 WorldPacket data(SMSG_NAME_QUERY_RESPONSE, (8+1+1+1+1+1+10));
37 data << guid.WriteAsPacked();
38 if (!nameData)
39 {
40 data << uint8(1); // name unknown
41 SendPacket(&data);
42 return;
43 }
44
45 data << uint8(0); // name known
46 data << nameData->Name; // played name
47 data << uint8(0); // realm name - only set for cross realm interaction (such as Battlegrounds)
48 data << uint8(nameData->Race);
49 data << uint8(nameData->Sex);
50 data << uint8(nameData->Class);
51
52 if (DeclinedName const* names = (player ? player->GetDeclinedNames() : NULL))
53 {
54 data << uint8(1); // Name is declined
55 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
56 data << names->name[i];
57 }
58 else
59 data << uint8(0); // Name is not declined
60
61 SendPacket(&data);
62}

88/// Only _static_ data is sent in this packet !!!
89void WorldSession::HandleCreatureQueryOpcode(WorldPacket& recvData)
90{
91 uint32 entry;
92 recvData >> entry;
93 ObjectGuid guid;
94 recvData >> guid;
95
96 CreatureTemplate const* ci = sObjectMgr->GetCreatureTemplate(entry);
97 if (ci)
98 {
99 std::string Name, Title;
100 Name = ci->Name;
101 Title = ci->Title;
102
103 int loc_idx = GetSessionDbLocaleIndex();
104 if (loc_idx >= 0)
105 {
106 if (CreatureLocale const* cl = sObjectMgr->GetCreatureLocale(entry))
107 {
108 ObjectMgr::GetLocaleString(cl->Name, loc_idx, Name);
109 ObjectMgr::GetLocaleString(cl->Title, loc_idx, Title);
110 }
111 }
112 TC_LOG_DEBUG("network", "WORLD: CMSG_CREATURE_QUERY '%s' - Entry: %u.", ci->Name.c_str(), entry);
113 // guess size
114 WorldPacket data(SMSG_CREATURE_QUERY_RESPONSE, 100);
115 data << uint32(entry); // creature entry
116 data << Name;
117 data << uint8(0) << uint8(0) << uint8(0); // name2, name3, name4, always empty
118 data << Title;
119 data << ci->IconName; // "Directions" for guard, string for Icons 2.3.0
120 data << uint32(ci->type_flags); // flags
121 data << uint32(ci->type); // CreatureType.dbc
122 data << uint32(ci->family); // CreatureFamily.dbc
123 data << uint32(ci->rank); // Creature Rank (elite, boss, etc)
124 data << uint32(ci->KillCredit[0]); // new in 3.1, kill credit
125 data << uint32(ci->KillCredit[1]); // new in 3.1, kill credit
126 data << uint32(ci->Modelid1); // Modelid1
127 data << uint32(ci->Modelid2); // Modelid2
128 data << uint32(ci->Modelid3); // Modelid3
129 data << uint32(ci->Modelid4); // Modelid4
130 data << float(ci->ModHealth); // dmg/hp modifier
131 data << float(ci->ModMana); // dmg/mana modifier
132 data << uint8(ci->RacialLeader);
133
134 CreatureQuestItemList const* items = sObjectMgr->GetCreatureQuestItemList(entry);
135 if (items)
136 for (size_t i = 0; i < MAX_CREATURE_QUEST_ITEMS; ++i)
137 data << (i < items->size() ? uint32((*items)[i]) : uint32(0));
138 else
139 for (size_t i = 0; i < MAX_CREATURE_QUEST_ITEMS; ++i)
140 data << uint32(0);
141
142 data << uint32(ci->movementId); // CreatureMovementInfo.dbc
143 SendPacket(&data);
144 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_CREATURE_QUERY_RESPONSE");
145 }
146 else
147 {
148 TC_LOG_DEBUG("network", "WORLD: CMSG_CREATURE_QUERY - NO CREATURE INFO! (%s, ENTRY: %u)",
149 guid.ToString().c_str(), entry);
150 WorldPacket data(SMSG_CREATURE_QUERY_RESPONSE, 4);
151 data << uint32(entry | 0x80000000);
152 SendPacket(&data);
153 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_CREATURE_QUERY_RESPONSE");
154 }
155}

157/// Only _static_ data is sent in this packet !!!
158void WorldSession::HandleGameObjectQueryOpcode(WorldPacket& recvData)
159{
160 uint32 entry;
161 recvData >> entry;
162 ObjectGuid guid;
163 recvData >> guid;
164
165 const GameObjectTemplate* info = sObjectMgr->GetGameObjectTemplate(entry);
166 if (info)
167 {
168 std::string Name;
169 std::string IconName;
170 std::string CastBarCaption;
171
172 Name = info->name;
173 IconName = info->IconName;
174 CastBarCaption = info->castBarCaption;
175
176 LocaleConstant localeConstant = GetSessionDbLocaleIndex();
177 if (localeConstant >= LOCALE_enUS)
178 if (GameObjectLocale const* gameObjectLocale = sObjectMgr->GetGameObjectLocale(entry))
179 {
180 ObjectMgr::GetLocaleString(gameObjectLocale->Name, localeConstant, Name);
181 ObjectMgr::GetLocaleString(gameObjectLocale->CastBarCaption, localeConstant, CastBarCaption);
182 }
183
184 TC_LOG_DEBUG("network", "WORLD: CMSG_GAMEOBJECT_QUERY '%s' - Entry: %u. ", info->name.c_str(), entry);
185 WorldPacket data (SMSG_GAMEOBJECT_QUERY_RESPONSE, 150);
186 data << uint32(entry);
187 data << uint32(info->type);
188 data << uint32(info->displayId);
189 data << Name;
190 data << uint8(0) << uint8(0) << uint8(0); // name2, name3, name4
191 data << IconName; // 2.0.3, string. Icon name to use instead of default icon for go's (ex: "Attack" makes sword)
192 data << CastBarCaption; // 2.0.3, string. Text will appear in Cast Bar when using GO (ex: "Collecting")
193 data << info->unk1; // 2.0.3, string
194 data.append(info->raw.data, MAX_GAMEOBJECT_DATA);
195 data << float(info->size); // go size
196
197 GameObjectQuestItemList const* items = sObjectMgr->GetGameObjectQuestItemList(entry);
198 if (items)
199 for (size_t i = 0; i < MAX_GAMEOBJECT_QUEST_ITEMS; ++i)
200 data << (i < items->size() ? uint32((*items)[i]) : uint32(0));
201 else
202 for (size_t i = 0; i < MAX_GAMEOBJECT_QUEST_ITEMS; ++i)
203 data << uint32(0);
204
205 SendPacket(&data);
206 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_GAMEOBJECT_QUERY_RESPONSE");
207 }
208 else
209 {
210 TC_LOG_DEBUG("network", "WORLD: CMSG_GAMEOBJECT_QUERY - Missing gameobject info for (%s, ENTRY: %u)",
211 guid.ToString().c_str(), entry);
212 WorldPacket data (SMSG_GAMEOBJECT_QUERY_RESPONSE, 4);
213 data << uint32(entry | 0x80000000);
214 SendPacket(&data);
215 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_GAMEOBJECT_QUERY_RESPONSE");
216 }

219void WorldSession::HandleCorpseQueryOpcode(WorldPacket & /*recvData*/)
220{
221 if (!_player->HasCorpse())
222 {
223 WorldPacket data(MSG_CORPSE_QUERY, 1);
224 data << uint8(0); // corpse not found
225 SendPacket(&data);
226 return;
227 }
228
229 WorldLocation corpseLocation = _player->GetCorpseLocation();
230 uint32 corpseMapID = corpseLocation.GetMapId();
231 uint32 mapID = corpseLocation.GetMapId();
232 float x = corpseLocation.GetPositionX();
233 float y = corpseLocation.GetPositionY();
234 float z = corpseLocation.GetPositionZ();
235
236 // if corpse at different map
237 if (mapID != _player->GetMapId())
238 {
239 // search entrance map for proper show entrance
240 if (MapEntry const* corpseMapEntry = sMapStore.LookupEntry(mapID))
241 {
242 if (corpseMapEntry->IsDungeon() && corpseMapEntry->entrance_map >= 0)
243 {
244 // if corpse map have entrance
245 if (Map const* entranceMap = sMapMgr->CreateBaseMap(corpseMapEntry->entrance_map))
246 {
247 mapID = corpseMapEntry->entrance_map;
248 x = corpseMapEntry->entrance_x;
249 y = corpseMapEntry->entrance_y;
250 z = entranceMap->GetHeight(GetPlayer()->GetPhaseMask(), x, y, MAX_HEIGHT);
251 }
252 }
253 }
254 }
255
256 WorldPacket data(MSG_CORPSE_QUERY, 1+(6*4));
257 data << uint8(1); // corpse found
258 data << int32(mapID);
259 data << float(x);
260 data << float(y);
261 data << float(z);
262 data << int32(corpseMapID);
263 data << uint32(0); // unknown
264 SendPacket(&data);
265}

267void WorldSession::HandleNpcTextQueryOpcode(WorldPacket& recvData)
268{
269 uint32 textID;
270 uint64 guid;
271
272 recvData >> textID;
273 TC_LOG_DEBUG("network", "WORLD: CMSG_NPC_TEXT_QUERY TextId: %u", textID);
274
275 recvData >> guid;
276
277 GossipText const* gossip = sObjectMgr->GetGossipText(textID);
278
279 WorldPacket data(SMSG_NPC_TEXT_UPDATE, 100); // guess size
280 data << textID;
281
282 if (!gossip)
283 {
284 for (uint8 i = 0; i < MAX_GOSSIP_TEXT_OPTIONS; ++i)
285 {
286 data << float(0);
287 data << "Greetings $N";
288 data << "Greetings $N";
289 data << uint32(0);
290 data << uint32(0);
291 data << uint32(0);
292 data << uint32(0);
293 data << uint32(0);
294 data << uint32(0);
295 data << uint32(0);
296 }
297 }
298 else
299 {
300 std::string text0[MAX_GOSSIP_TEXT_OPTIONS], text1[MAX_GOSSIP_TEXT_OPTIONS];
301 LocaleConstant locale = GetSessionDbLocaleIndex();
302
303 for (uint8 i = 0; i < MAX_GOSSIP_TEXT_OPTIONS; ++i)
304 {
305 BroadcastText const* bct = sObjectMgr->GetBroadcastText(gossip->Options[i].BroadcastTextID);
306 if (bct)
307 {
308 text0[i] = bct->GetText(locale, GENDER_MALE, true);
309 text1[i] = bct->GetText(locale, GENDER_FEMALE, true);
310 }
311 else
312 {
313 text0[i] = gossip->Options[i].Text_0;
314 text1[i] = gossip->Options[i].Text_1;
315 }
316
317 if (locale != DEFAULT_LOCALE && !bct)
318 {
319 if (NpcTextLocale const* npcTextLocale = sObjectMgr->GetNpcTextLocale(textID))
320 {
321 ObjectMgr::GetLocaleString(npcTextLocale->Text_0[i], locale, text0[i]);
322 ObjectMgr::GetLocaleString(npcTextLocale->Text_1[i], locale, text1[i]);
323 }
324 }
325
326 data << gossip->Options[i].Probability;
327
328 if (text0[i].empty())
329 data << text1[i];
330 else
331 data << text0[i];
332
333 if (text1[i].empty())
334 data << text0[i];
335 else
336 data << text1[i];
337
338 data << gossip->Options[i].Language;
339
340 for (uint8 j = 0; j < MAX_GOSSIP_TEXT_EMOTES; ++j)
341 {
342 data << gossip->Options[i].Emotes[j]._Delay;
343 data << gossip->Options[i].Emotes[j]._Emote;
344 }
345 }
346 }
347
348 SendPacket(&data);
349
350 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_NPC_TEXT_UPDATE");
351}

353/// Only _static_ data is sent in this packet !!!
354void WorldSession::HandlePageTextQueryOpcode(WorldPacket& recvData)
355{
356 TC_LOG_DEBUG("network", "WORLD: Received CMSG_PAGE_TEXT_QUERY");
357
358 uint32 pageID;
359 recvData >> pageID;
360 recvData.read_skip<uint64>(); // guid
361
362 while (pageID)
363 {
364 PageText const* pageText = sObjectMgr->GetPageText(pageID);
365 // guess size
366 WorldPacket data(SMSG_PAGE_TEXT_QUERY_RESPONSE, 50);
367 data << pageID;
368
369 if (!pageText)
370 {
371 data << "Item page missing.";
372 data << uint32(0);
373 pageID = 0;
374 }
375 else
376 {
377 std::string Text = pageText->Text;
378
379 int loc_idx = GetSessionDbLocaleIndex();
380 if (loc_idx >= 0)
381 if (PageTextLocale const* player = sObjectMgr->GetPageTextLocale(pageID))
382 ObjectMgr::GetLocaleString(player->Text, loc_idx, Text);
383
384 data << Text;
385 data << uint32(pageText->NextPage);
386 pageID = pageText->NextPage;
387 }
388 SendPacket(&data);
389
390 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_PAGE_TEXT_QUERY_RESPONSE");
391 }

409void WorldSession::HandleQuestPOIQuery(WorldPacket& recvData)
410{
411 uint32 count;
412 recvData >> count; // quest count, max=25
413
414 if (count > MAX_QUEST_LOG_SIZE)
415 {
416 recvData.rfinish();
417 return;
418 }
419
420 // Read quest ids and add the in a unordered_set so we don't send POIs for the same quest multiple times
421 std::unordered_set<uint32> questIds;
422 for (uint32 i = 0; i < count; ++i)
423 questIds.insert(recvData.read<uint32>()); // quest id
424
425 WorldPacket data(SMSG_QUEST_POI_QUERY_RESPONSE, 4 + (4 + 4)*questIds.size());
426 data << uint32(questIds.size()); // count
427
428 for (auto itr = questIds.begin(); itr != questIds.end(); ++itr)
429 {
430 uint32 questId = *itr;
431
432 bool questOk = false;
433
434 uint16 questSlot = _player->FindQuestSlot(questId);
435
436 if (questSlot != MAX_QUEST_LOG_SIZE)
437 questOk =_player->GetQuestSlotQuestId(questSlot) == questId;
438
439 if (questOk)
440 {
441 QuestPOIVector const* POI = sObjectMgr->GetQuestPOIVector(questId);
442
443 if (POI)
444 {
445 data << uint32(questId); // quest ID
446 data << uint32(POI->size()); // POI count
447
448 for (QuestPOIVector::const_iterator itr = POI->begin(); itr != POI->end(); ++itr)
449 {
450 data << uint32(itr->Id); // POI index
451 data << int32(itr->ObjectiveIndex); // objective index
452 data << uint32(itr->MapId); // mapid
453 data << uint32(itr->AreaId); // areaid
454 data << uint32(itr->FloorId); // floorid
455 data << uint32(itr->Unk3); // unknown
456 data << uint32(itr->Unk4); // unknown
457 data << uint32(itr->points.size()); // POI points count
458
459 for (std::vector<QuestPOIPoint>::const_iterator itr2 = itr->points.begin(); itr2 != itr->points.end(); ++itr2)
460 {
461 data << int32(itr2->x); // POI point x
462 data << int32(itr2->y); // POI point y
463 }
464 }
465 }
466 else
467 {
468 data << uint32(questId); // quest ID
469 data << uint32(0); // POI count
470 }
471 }
472 else
473 {
474 data << uint32(questId); // quest ID
475 data << uint32(0); // POI count
476 }
477 }
478
479 SendPacket(&data);
480}

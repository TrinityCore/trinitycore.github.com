37void WorldSession::HandleDismissCritter(WorldPacket& recvData)
38{
39 ObjectGuid guid;
40 recvData >> guid;
41
42 TC_LOG_DEBUG("network", "WORLD: Received CMSG_DISMISS_CRITTER for %s", guid.ToString().c_str());
43
44 Unit* pet = ObjectAccessor::GetCreatureOrPetOrVehicle(*_player, guid);
45
46 if (!pet)
47 {
48 TC_LOG_DEBUG("network", "Vanitypet (%s) does not exist - player '%s' (guid: %u / account: %u) attempted to dismiss it (possibly lagged out)",
49 guid.ToString().c_str(), GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().GetCounter(), GetAccountId());
50 return;
51 }
52
53 if (_player->GetCritterGUID() == pet->GetGUID())
54 {
55 if (pet->GetTypeId() == TYPEID_UNIT && pet->IsSummon())
56 pet->ToTempSummon()->UnSummon();
57 }

60void WorldSession::HandlePetAction(WorldPacket& recvData)
61{
62 ObjectGuid guid1;
63 uint32 data;
64 ObjectGuid guid2;
65 recvData >> guid1; //pet guid
66 recvData >> data;
67 recvData >> guid2; //tag guid
68
69 uint32 spellid = UNIT_ACTION_BUTTON_ACTION(data);
70 uint8 flag = UNIT_ACTION_BUTTON_TYPE(data); //delete = 0x07 CastSpell = C1
71
72 // used also for charmed creature
73 Unit* pet = ObjectAccessor::GetUnit(*_player, guid1);
74 TC_LOG_DEBUG("network", "HandlePetAction: %s - flag: %u, spellid: %u, target: %s.", guid1.ToString().c_str(), uint32(flag), spellid, guid2.ToString().c_str());
75
76 if (!pet)
77 {
78 TC_LOG_DEBUG("network", "HandlePetAction: %s doesn't exist for %s %s", guid1.ToString().c_str(), GetPlayer()->GetGUID().ToString().c_str(), GetPlayer()->GetName().c_str());
79 return;
80 }
81
82 if (pet != GetPlayer()->GetFirstControlled())
83 {
84 TC_LOG_DEBUG("network", "HandlePetAction: %s does not belong to %s %s", guid1.ToString().c_str(), GetPlayer()->GetGUID().ToString().c_str(), GetPlayer()->GetName().c_str());
85 return;
86 }
87
88 if (!pet->IsAlive())
89 {
90 SpellInfo const* spell = (flag == ACT_ENABLED || flag == ACT_PASSIVE) ? sSpellMgr->GetSpellInfo(spellid) : NULL;
91 if (!spell)
92 return;
93 if (!spell->HasAttribute(SPELL_ATTR0_CASTABLE_WHILE_DEAD))
94 return;
95 }
96
97 /// @todo allow control charmed player?
98 if (pet->GetTypeId() == TYPEID_PLAYER && !(flag == ACT_COMMAND && spellid == COMMAND_ATTACK))
99 return;
100
101 if (GetPlayer()->m_Controlled.size() == 1)
102 HandlePetActionHelper(pet, guid1, spellid, flag, guid2);
103 else
104 {
105 //If a pet is dismissed, m_Controlled will change
106 std::vector<Unit*> controlled;
107 for (Unit::ControlList::iterator itr = GetPlayer()->m_Controlled.begin(); itr != GetPlayer()->m_Controlled.end(); ++itr)
108 if ((*itr)->GetEntry() == pet->GetEntry() && (*itr)->IsAlive())
109 controlled.push_back(*itr);
110 for (std::vector<Unit*>::iterator itr = controlled.begin(); itr != controlled.end(); ++itr)
111 HandlePetActionHelper(*itr, guid1, spellid, flag, guid2);
112 }
113}

138 return;
139
140 pet->AttackStop();

143void WorldSession::HandlePetActionHelper(Unit* pet, ObjectGuid guid1, uint32 spellid, uint16 flag, ObjectGuid guid2)
144{
145 CharmInfo* charmInfo = pet->GetCharmInfo();
146 if (!charmInfo)
147 {
148 TC_LOG_DEBUG("network", "WorldSession::HandlePetAction(petGuid: %s, tagGuid: %s, spellId: %u, flag: %u): object (GUID: %u Entry: %u TypeId: %u) is considered pet-like but doesn't have a charminfo!",
149 guid1.ToString().c_str(), guid2.ToString().c_str(), spellid, flag, pet->GetGUID().GetCounter(), pet->GetEntry(), pet->GetTypeId());
150 return;
151 }
152
153 switch (flag)
154 {
155 case ACT_COMMAND: //0x07
156 switch (spellid)
157 {
158 case COMMAND_STAY: //flat=1792 //STAY
159 pet->StopMoving();
160 pet->GetMotionMaster()->Clear(false);
161 pet->GetMotionMaster()->MoveIdle();
162 charmInfo->SetCommandState(COMMAND_STAY);
163
164 charmInfo->SetIsCommandAttack(false);
165 charmInfo->SetIsAtStay(true);
166 charmInfo->SetIsCommandFollow(false);
167 charmInfo->SetIsFollowing(false);
168 charmInfo->SetIsReturning(false);
169 charmInfo->SaveStayPosition();
170 break;
171 case COMMAND_FOLLOW: //spellid=1792 //FOLLOW
172 pet->AttackStop();
173 pet->InterruptNonMeleeSpells(false);
174 pet->GetMotionMaster()->MoveFollow(_player, PET_FOLLOW_DIST, pet->GetFollowAngle());
175 charmInfo->SetCommandState(COMMAND_FOLLOW);
176
177 charmInfo->SetIsCommandAttack(false);
178 charmInfo->SetIsAtStay(false);
179 charmInfo->SetIsReturning(true);
180 charmInfo->SetIsCommandFollow(true);
181 charmInfo->SetIsFollowing(false);
182 break;
183 case COMMAND_ATTACK: //spellid=1792 //ATTACK
184 {
185 // Can't attack if owner is pacified
186 if (_player->HasAuraType(SPELL_AURA_MOD_PACIFY))
187 {
188 //pet->SendPetCastFail(spellid, SPELL_FAILED_PACIFIED);
189 /// @todo Send proper error message to client
190 return;
191 }
192
193 // only place where pet can be player
194 Unit* TargetUnit = ObjectAccessor::GetUnit(*_player, guid2);
195 if (!TargetUnit)
196 return;
197
198 if (Unit* owner = pet->GetOwner())
199 if (!owner->IsValidAttackTarget(TargetUnit))
200 return;
201
202 pet->ClearUnitState(UNIT_STATE_FOLLOW);
203 // This is true if pet has no target or has target but targets differs.
204 if (pet->GetVictim() != TargetUnit || (pet->GetVictim() == TargetUnit && !pet->GetCharmInfo()->IsCommandAttack()))
205 {
206 if (pet->GetVictim())
207 pet->AttackStop();
208
209 if (pet->GetTypeId() != TYPEID_PLAYER && pet->ToCreature()->IsAIEnabled)
210 {
211 charmInfo->SetIsCommandAttack(true);
212 charmInfo->SetIsAtStay(false);
213 charmInfo->SetIsFollowing(false);
214 charmInfo->SetIsCommandFollow(false);
215 charmInfo->SetIsReturning(false);
216
217 pet->ToCreature()->AI()->AttackStart(TargetUnit);
218
219 //10% chance to play special pet attack talk, else growl
220 if (pet->IsPet() && ((Pet*)pet)->getPetType() == SUMMON_PET && pet != TargetUnit && urand(0, 100) < 10)
221 pet->SendPetTalk((uint32)PET_TALK_ATTACK);
222 else
223 {
224 // 90% chance for pet and 100% chance for charmed creature
225 pet->SendPetAIReaction(guid1);
226 }
227 }
228 else // charmed player
229 {
230 if (pet->GetVictim() && pet->GetVictim() != TargetUnit)
231 pet->AttackStop();
232
233 charmInfo->SetIsCommandAttack(true);
234 charmInfo->SetIsAtStay(false);
235 charmInfo->SetIsFollowing(false);
236 charmInfo->SetIsCommandFollow(false);
237 charmInfo->SetIsReturning(false);
238
239 pet->Attack(TargetUnit, true);
240 pet->SendPetAIReaction(guid1);
241 }
242 }
243 break;
244 }
245 case COMMAND_ABANDON: // abandon (hunter pet) or dismiss (summoned pet)
246 if (pet->GetCharmerGUID() == GetPlayer()->GetGUID())
247 _player->StopCastingCharm();
248 else if (pet->GetOwnerGUID() == GetPlayer()->GetGUID())
249 {
250 ASSERT(pet->GetTypeId() == TYPEID_UNIT);
251 if (pet->IsPet())
252 {
253 if (((Pet*)pet)->getPetType() == HUNTER_PET)
254 GetPlayer()->RemovePet((Pet*)pet, PET_SAVE_AS_DELETED);
255 else
256 //dismissing a summoned pet is like killing them (this prevents returning a soulshard...)
257 pet->setDeathState(CORPSE);
258 }
259 else if (pet->HasUnitTypeMask(UNIT_MASK_MINION))
260 {
261 ((Minion*)pet)->UnSummon();
262 }
263 }
264 break;
265 default:
266 TC_LOG_ERROR("network", "WORLD: unknown PET flag Action %i and spellid %i.", uint32(flag), spellid);
267 }
268 break;
269 case ACT_REACTION: // 0x6
270 switch (spellid)
271 {
272 case REACT_PASSIVE: //passive
273 pet->AttackStop();
274 // no break;
275 case REACT_DEFENSIVE: //recovery
276 case REACT_AGGRESSIVE: //activete
277 if (pet->GetTypeId() == TYPEID_UNIT)
278 pet->ToCreature()->SetReactState(ReactStates(spellid));
279 break;
280 }
281 break;
282 case ACT_DISABLED: // 0x81 spell (disabled), ignore
283 case ACT_PASSIVE: // 0x01
284 case ACT_ENABLED: // 0xC1 spell
285 {
286 Unit* unit_target = NULL;
287
288 if (guid2)
289 unit_target = ObjectAccessor::GetUnit(*_player, guid2);
290
291 // do not cast unknown spells
292 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellid);
293 if (!spellInfo)
294 {
295 TC_LOG_ERROR("network", "WORLD: unknown PET spell id %i", spellid);
296 return;
297 }
298
299 for (uint32 i = 0; i < MAX_SPELL_EFFECTS; ++i)
300 {
301 if (spellInfo->Effects[i].TargetA.GetTarget() == TARGET_UNIT_SRC_AREA_ENEMY || spellInfo->Effects[i].TargetA.GetTarget() == TARGET_UNIT_DEST_AREA_ENEMY || spellInfo->Effects[i].TargetA.GetTarget() == TARGET_DEST_DYNOBJ_ENEMY)
302 return;
303 }
304
305 // do not cast not learned spells
306 if (!pet->HasSpell(spellid) || spellInfo->IsPassive())
307 return;
308
309 // Clear the flags as if owner clicked 'attack'. AI will reset them
310 // after AttackStart, even if spell failed
311 if (pet->GetCharmInfo())
312 {
313 pet->GetCharmInfo()->SetIsAtStay(false);
314 pet->GetCharmInfo()->SetIsCommandAttack(true);
315 pet->GetCharmInfo()->SetIsReturning(false);
316 pet->GetCharmInfo()->SetIsFollowing(false);
317 }
318
319 Spell* spell = new Spell(pet, spellInfo, TRIGGERED_NONE);
320
321 SpellCastResult result = spell->CheckPetCast(unit_target);
322
323 //auto turn to target unless possessed
324 if (result == SPELL_FAILED_UNIT_NOT_INFRONT && !pet->isPossessed() && !pet->IsVehicle())
325 {
326 if (unit_target)
327 {
328 pet->SetInFront(unit_target);
329 if (Player* player = unit_target->ToPlayer())
330 pet->SendUpdateToPlayer(player);
331 }
332 else if (Unit* unit_target2 = spell->m_targets.GetUnitTarget())
333 {
334 pet->SetInFront(unit_target2);
335 if (Player* player = unit_target2->ToPlayer())
336 pet->SendUpdateToPlayer(player);
337 }
338
339 if (Unit* powner = pet->GetCharmerOrOwner())
340 if (Player* player = powner->ToPlayer())
341 pet->SendUpdateToPlayer(player);
342
343 result = SPELL_CAST_OK;
344 }
345
346 if (result == SPELL_CAST_OK)
347 {
348 unit_target = spell->m_targets.GetUnitTarget();
349
350 //10% chance to play special pet attack talk, else growl
351 //actually this only seems to happen on special spells, fire shield for imp, torment for voidwalker, but it's stupid to check every spell
352 if (pet->IsPet() && (((Pet*)pet)->getPetType() == SUMMON_PET) && (pet != unit_target) && (urand(0, 100) < 10))
353 pet->SendPetTalk((uint32)PET_TALK_SPECIAL_SPELL);
354 else
355 {
356 pet->SendPetAIReaction(guid1);
357 }
358
359 if (unit_target && !GetPlayer()->IsFriendlyTo(unit_target) && !pet->isPossessed() && !pet->IsVehicle())
360 {
361 // This is true if pet has no target or has target but targets differs.
362 if (pet->GetVictim() != unit_target)
363 {
364 if (pet->GetVictim())
365 pet->AttackStop();
366 pet->GetMotionMaster()->Clear();
367 if (pet->ToCreature()->IsAIEnabled)
368 pet->ToCreature()->AI()->AttackStart(unit_target);
369 }
370 }
371
372 spell->prepare(&(spell->m_targets));
373 }
374 else
375 {
376 if (pet->isPossessed() || pet->IsVehicle()) /// @todo: confirm this check
377 Spell::SendCastResult(GetPlayer(), spellInfo, 0, result);
378 else
379 spell->SendPetCastResult(result);
380
381 if (!pet->GetSpellHistory()->HasCooldown(spellid))
382 pet->GetSpellHistory()->ResetCooldown(spellid, true);
383
384 spell->finish(false);
385 delete spell;
386
387 // reset specific flags in case of spell fail. AI will reset other flags
388 if (pet->GetCharmInfo())
389 pet->GetCharmInfo()->SetIsCommandAttack(false);
390 }
391 break;
392 }
393 default:
394 TC_LOG_ERROR("network", "WORLD: unknown PET flag Action %i and spellid %i.", uint32(flag), spellid);
395 }
396}

411void WorldSession::SendPetNameQuery(ObjectGuid petguid, uint32 petnumber)
412{
413 Creature* pet = ObjectAccessor::GetCreatureOrPetOrVehicle(*_player, petguid);
414 if (!pet)
415 {
416 WorldPacket data(SMSG_PET_NAME_QUERY_RESPONSE, (4+1+4+1));
417 data << uint32(petnumber);
418 data << uint8(0);
419 data << uint32(0);
420 data << uint8(0);
421 _player->GetSession()->SendPacket(&data);
422 return;
423 }
424
425 WorldPacket data(SMSG_PET_NAME_QUERY_RESPONSE, (4+4+pet->GetName().size()+1));
426 data << uint32(petnumber);
427 data << pet->GetName();
428 data << uint32(pet->GetUInt32Value(UNIT_FIELD_PET_NAME_TIMESTAMP));
429
430 if (pet->IsPet() && ((Pet*)pet)->GetDeclinedNames())
431 {
432 data << uint8(1);
433 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
434 data << ((Pet*)pet)->GetDeclinedNames()->name[i];
435 }
436 else
437 data << uint8(0);
438
439 _player->GetSession()->SendPacket(&data);
440}

465void WorldSession::HandlePetSetAction(WorldPacket& recvData)
466{
467 TC_LOG_DEBUG("network", "WORLD: Received CMSG_PET_SET_ACTION");
468
469 ObjectGuid petguid;
470 uint8 count;
471
472 recvData >> petguid;
473
474 Unit* pet = ObjectAccessor::GetUnit(*_player, petguid);
475
476 if (!pet || pet != _player->GetFirstControlled())
477 {
478 TC_LOG_ERROR("network", "HandlePetSetAction: Unknown %s or owner (%s)", petguid.ToString().c_str(), _player->GetGUID().ToString().c_str());
479 return;
480 }
481
482 CharmInfo* charmInfo = pet->GetCharmInfo();
483 if (!charmInfo)
484 {
485 TC_LOG_ERROR("network", "WorldSession::HandlePetSetAction: object (GUID: %u TypeId: %u) is considered pet-like but doesn't have a charminfo!", pet->GetGUID().GetCounter(), pet->GetTypeId());
486 return;
487 }
488
489 count = (recvData.size() == 24) ? 2 : 1;
490
491 uint32 position[2];
492 uint32 data[2];
493 bool move_command = false;
494
495 for (uint8 i = 0; i < count; ++i)
496 {
497 recvData >> position[i];
498 recvData >> data[i];
499
500 uint8 act_state = UNIT_ACTION_BUTTON_TYPE(data[i]);
501
502 //ignore invalid position
503 if (position[i] >= MAX_UNIT_ACTION_BAR_INDEX)
504 return;
505
506 // in the normal case, command and reaction buttons can only be moved, not removed
507 // at moving count == 2, at removing count == 1
508 // ignore attempt to remove command|reaction buttons (not possible at normal case)
509 if (act_state == ACT_COMMAND || act_state == ACT_REACTION)
510 {
511 if (count == 1)
512 return;
513
514 move_command = true;
515 }
516 }
517
518 // check swap (at command->spell swap client remove spell first in another packet, so check only command move correctness)
519 if (move_command)
520 {
521 uint8 act_state_0 = UNIT_ACTION_BUTTON_TYPE(data[0]);
522 if (act_state_0 == ACT_COMMAND || act_state_0 == ACT_REACTION)
523 {
524 uint32 spell_id_0 = UNIT_ACTION_BUTTON_ACTION(data[0]);
525 UnitActionBarEntry const* actionEntry_1 = charmInfo->GetActionBarEntry(position[1]);
526 if (!actionEntry_1 || spell_id_0 != actionEntry_1->GetAction() ||
527 act_state_0 != actionEntry_1->GetType())
528 return;
529 }
530
531 uint8 act_state_1 = UNIT_ACTION_BUTTON_TYPE(data[1]);
532 if (act_state_1 == ACT_COMMAND || act_state_1 == ACT_REACTION)
533 {
534 uint32 spell_id_1 = UNIT_ACTION_BUTTON_ACTION(data[1]);
535 UnitActionBarEntry const* actionEntry_0 = charmInfo->GetActionBarEntry(position[0]);
536 if (!actionEntry_0 || spell_id_1 != actionEntry_0->GetAction() ||
537 act_state_1 != actionEntry_0->GetType())
538 return;
539 }
540 }
541
542 for (uint8 i = 0; i < count; ++i)
543 {
544 uint32 spell_id = UNIT_ACTION_BUTTON_ACTION(data[i]);
545 uint8 act_state = UNIT_ACTION_BUTTON_TYPE(data[i]);
546
547 TC_LOG_DEBUG("network", "Player %s has changed pet spell action. Position: %u, Spell: %u, State: 0x%X",
548 _player->GetName().c_str(), position[i], spell_id, uint32(act_state));
549
550 //if it's act for spell (en/disable/cast) and there is a spell given (0 = remove spell) which pet doesn't know, don't add
551 if (!((act_state == ACT_ENABLED || act_state == ACT_DISABLED || act_state == ACT_PASSIVE) && spell_id && !pet->HasSpell(spell_id)))
552 {
553 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spell_id))
554 {
555 //sign for autocast
556 if (act_state == ACT_ENABLED)
557 {
558 if (pet->GetTypeId() == TYPEID_UNIT && pet->IsPet())
559 ((Pet*)pet)->ToggleAutocast(spellInfo, true);
560 else
561 for (Unit::ControlList::iterator itr = GetPlayer()->m_Controlled.begin(); itr != GetPlayer()->m_Controlled.end(); ++itr)
562 if ((*itr)->GetEntry() == pet->GetEntry())
563 (*itr)->GetCharmInfo()->ToggleCreatureAutocast(spellInfo, true);
564 }
565 //sign for no/turn off autocast
566 else if (act_state == ACT_DISABLED)
567 {
568 if (pet->GetTypeId() == TYPEID_UNIT && pet->IsPet())
569 ((Pet*)pet)->ToggleAutocast(spellInfo, false);
570 else
571 for (Unit::ControlList::iterator itr = GetPlayer()->m_Controlled.begin(); itr != GetPlayer()->m_Controlled.end(); ++itr)
572 if ((*itr)->GetEntry() == pet->GetEntry())
573 (*itr)->GetCharmInfo()->ToggleCreatureAutocast(spellInfo, false);
574 }
575 }
576
577 charmInfo->SetActionBar(position[i], spell_id, ActiveStates(act_state));
578 }
579 }

582void WorldSession::HandlePetRename(WorldPacket& recvData)
583{
584 TC_LOG_DEBUG("network", "WORLD: Received CMSG_PET_RENAME");
585
586 ObjectGuid petguid;
587 uint8 isdeclined;
588
589 std::string name;
590 DeclinedName declinedname;
591
592 recvData >> petguid;
593 recvData >> name;
594 recvData >> isdeclined;
595
596 Pet* pet = ObjectAccessor::GetPet(*_player, petguid);
597 // check it!
598 if (!pet || !pet->IsPet() || ((Pet*)pet)->getPetType()!= HUNTER_PET ||
599 !pet->HasByteFlag(UNIT_FIELD_BYTES_2, 2, UNIT_CAN_BE_RENAMED) ||
600 pet->GetOwnerGUID() != _player->GetGUID() || !pet->GetCharmInfo())
601 return;
602
603 PetNameInvalidReason res = ObjectMgr::CheckPetName(name, GetSessionDbcLocale());
604 if (res != PET_NAME_SUCCESS)
605 {
606 SendPetNameInvalid(res, name, NULL);
607 return;
608 }
609
610 if (sObjectMgr->IsReservedName(name))
611 {
612 SendPetNameInvalid(PET_NAME_RESERVED, name, NULL);
613 return;
614 }
615
616 pet->SetName(name);
617
618 if (pet->GetOwner()->GetGroup())
619 pet->GetOwner()->SetGroupUpdateFlag(GROUP_UPDATE_FLAG_PET_NAME);
620
621 pet->RemoveByteFlag(UNIT_FIELD_BYTES_2, 2, UNIT_CAN_BE_RENAMED);
622
623 if (isdeclined)
624 {
625 for (uint8 i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
626 {
627 recvData >> declinedname.name[i];
628 }
629
630 std::wstring wname;
631 if (!Utf8toWStr(name, wname))
632 return;
633
634 if (!ObjectMgr::CheckDeclinedNames(wname, declinedname))
635 {
636 SendPetNameInvalid(PET_NAME_DECLENSION_DOESNT_MATCH_BASE_NAME, name, &declinedname);
637 return;
638 }
639 }
640
641 SQLTransaction trans = CharacterDatabase.BeginTransaction();
642 if (isdeclined)
643 {
644 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHAR_PET_DECLINEDNAME);
645 stmt->setUInt32(0, pet->GetCharmInfo()->GetPetNumber());
646 trans->Append(stmt);
647
648 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_CHAR_PET_DECLINEDNAME);
649 stmt->setUInt32(0, _player->GetGUID().GetCounter());
650
651 for (uint8 i = 0; i < 5; i++)
652 stmt->setString(i + 1, declinedname.name[i]);
653
654 trans->Append(stmt);
655 }
656
657 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_CHAR_PET_NAME);
658 stmt->setString(0, name);
659 stmt->setUInt32(1, _player->GetGUID().GetCounter());
660 stmt->setUInt32(2, pet->GetCharmInfo()->GetPetNumber());
661 trans->Append(stmt);
662
663 CharacterDatabase.CommitTransaction(trans);
664

697{
698 TC_LOG_DEBUG("network", "WORLD: Received CMSG_PET_SPELL_AUTOCAST");
699 ObjectGuid guid;
700 uint32 spellid;
701 uint8 state; //1 for on, 0 for off
702 recvPacket >> guid >> spellid >> state;
703
704 if (!_player->GetGuardianPet() && !_player->GetCharm())
705 return;
706
707 if (ObjectAccessor::FindPlayer(guid))
708 return;
709
710 Creature* pet=ObjectAccessor::GetCreatureOrPetOrVehicle(*_player, guid);
711
712 if (!pet || (pet != _player->GetGuardianPet() && pet != _player->GetCharm()))
713 {
714 TC_LOG_ERROR("network", "HandlePetSpellAutocastOpcode. %s isn't pet of player %s (%s).", guid.ToString().c_str(), GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().ToString().c_str());
715 return;
716 }
717
718 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellid);

721 TC_LOG_ERROR("network", "WORLD: unknown PET spell id %u", spellid);
722 return;
723 }
724
725 // do not add not learned spells/ passive spells
726 if (!pet->HasSpell(spellid) || !spellInfo->IsAutocastable())
727 return;
728
729 CharmInfo* charmInfo = pet->GetCharmInfo();
730 if (!charmInfo)

732 TC_LOG_ERROR("network", "WorldSession::HandlePetSpellAutocastOpcod: object (GUID: %u TypeId: %u) is considered pet-like but doesn't have a charminfo!", pet->GetGUID().GetCounter(), pet->GetTypeId());
733 return;
734 }
735
736 if (pet->IsPet())
737 ((Pet*)pet)->ToggleAutocast(spellInfo, state != 0);
738 else
739 pet->GetCharmInfo()->ToggleCreatureAutocast(spellInfo, state != 0);

744void WorldSession::HandlePetCastSpellOpcode(WorldPacket& recvPacket)
745{
746 TC_LOG_DEBUG("network", "WORLD: Received CMSG_PET_CAST_SPELL");
747
748 ObjectGuid guid;
749 uint8 castCount;
750 uint32 spellId;
751 uint8 castFlags;
752
753 recvPacket >> guid >> castCount >> spellId >> castFlags;
754
755 TC_LOG_DEBUG("network", "WORLD: CMSG_PET_CAST_SPELL, %s, castCount: %u, spellId %u, castFlags %u", guid.ToString().c_str(), castCount, spellId, castFlags);
756
757 // This opcode is also sent from charmed and possessed units (players and creatures)
758 if (!_player->GetGuardianPet() && !_player->GetCharm())
759 return;
760
761 Unit* caster = ObjectAccessor::GetUnit(*_player, guid);
762
763 if (!caster || (caster != _player->GetGuardianPet() && caster != _player->GetCharm()))
764 {
765 TC_LOG_ERROR("network", "HandlePetCastSpellOpcode: %s isn't pet of player %s (%s).", guid.ToString().c_str(), GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().ToString().c_str());
766 return;
767 }
768
769 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
770 if (!spellInfo)
771 {
772 TC_LOG_ERROR("network", "WORLD: unknown PET spell id %i", spellId);
773 return;
774 }
775
776 // do not cast not learned spells
777 if (!caster->HasSpell(spellId) || spellInfo->IsPassive())
778 return;
779
780 SpellCastTargets targets;
781 targets.Read(recvPacket, caster);
782 HandleClientCastFlags(recvPacket, castFlags, targets);
783
784 caster->ClearUnitState(UNIT_STATE_FOLLOW);
785
786 Spell* spell = new Spell(caster, spellInfo, TRIGGERED_NONE);
787 spell->m_cast_count = castCount; // probably pending spell cast
788 spell->m_targets = targets;
789
790 SpellCastResult result = spell->CheckPetCast(NULL);
791
792 if (result == SPELL_CAST_OK)
793 {
794 if (Creature* creature = caster->ToCreature())
795 {
796 if (Pet* pet = creature->ToPet())
797 {
798 // 10% chance to play special pet attack talk, else growl
799 // actually this only seems to happen on special spells, fire shield for imp, torment for voidwalker, but it's stupid to check every spell
800 if (pet->getPetType() == SUMMON_PET && (urand(0, 100) < 10))
801 pet->SendPetTalk(PET_TALK_SPECIAL_SPELL);
802 else
803 pet->SendPetAIReaction(guid);
804 }
805 }
806
807 spell->prepare(&(spell->m_targets));
808 }
809 else
810 {
811 spell->SendPetCastResult(result);
812
813 if (!caster->GetSpellHistory()->HasCooldown(spellId))
814 caster->GetSpellHistory()->ResetCooldown(spellId, true);
815
816 spell->finish(false);
817 delete spell;
818 }
819}

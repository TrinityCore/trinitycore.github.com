37 if (Guild* guild = sGuildMgr->GetGuildById(guildId))

198void WorldSession::HandleGuildRankOpcode(WorldPacket& recvPacket)
199{
200 uint32 rankId;
201 recvPacket >> rankId;
202
203 uint32 rights;
204 recvPacket >> rights;
205
206 std::string rankName;
207 recvPacket >> rankName;
208
209 uint32 money;
210 recvPacket >> money;
211
212 TC_LOG_DEBUG("guild", "CMSG_GUILD_RANK [%s]: Rank: %s (%u)", GetPlayerInfo().c_str(), rankName.c_str(), rankId);
213
214 Guild* guild = GetPlayer()->GetGuild();
215 if (!guild)
216 {
217 recvPacket.rpos(recvPacket.wpos());
218 return;
219 }
220
221 GuildBankRightsAndSlotsVec rightsAndSlots(GUILD_BANK_MAX_TABS);
222
223 for (uint8 tabId = 0; tabId < GUILD_BANK_MAX_TABS; ++tabId)
224 {
225 uint32 bankRights;
226 uint32 slots;
227
228 recvPacket >> bankRights;
229 recvPacket >> slots;
230
231 rightsAndSlots[tabId] = GuildBankRightsAndSlots(tabId, bankRights, slots);
232 }
233
234 guild->HandleSetRankInfo(this, rankId, rankName, rights, money, rightsAndSlots);
235}

267void WorldSession::HandleSaveGuildEmblemOpcode(WorldPacket& recvPacket)
268{
269 ObjectGuid vendorGuid;
270 recvPacket >> vendorGuid;
271
272 EmblemInfo emblemInfo;
273 emblemInfo.ReadPacket(recvPacket);
274
275 TC_LOG_DEBUG("guild", "MSG_SAVE_GUILD_EMBLEM [%s]: Guid: [%s] Style: %d, Color: %d, BorderStyle: %d, BorderColor: %d, BackgroundColor: %d"
276 , GetPlayerInfo().c_str(), vendorGuid.ToString().c_str(), emblemInfo.GetStyle()
277 , emblemInfo.GetColor(), emblemInfo.GetBorderStyle()
278 , emblemInfo.GetBorderColor(), emblemInfo.GetBackgroundColor());
279
280 if (GetPlayer()->GetNPCIfCanInteractWith(vendorGuid, UNIT_NPC_FLAG_TABARDDESIGNER))
281 {
282 // Remove fake death
283 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
284 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
285
286 if (Guild* guild = GetPlayer()->GetGuild())
287 guild->HandleSetEmblem(this, emblemInfo);
288 else
289 Guild::SendSaveEmblemResult(this, ERR_GUILDEMBLEM_NOGUILD); // "You are not part of a guild!";
290 }
291 else
292 Guild::SendSaveEmblemResult(this, ERR_GUILDEMBLEM_INVALIDVENDOR); // "That's not an emblem vendor!"
293}

385void WorldSession::HandleGuildBankSwapItems(WorldPacket& recvData)
386{
387 TC_LOG_DEBUG("guild", "CMSG_GUILD_BANK_SWAP_ITEMS [%s]", GetPlayerInfo().c_str());
388
389 ObjectGuid GoGuid;
390 recvData >> GoGuid;
391
392 if (!GetPlayer()->GetGameObjectIfCanInteractWith(GoGuid, GAMEOBJECT_TYPE_GUILD_BANK))
393 {
394 recvData.rfinish(); // Prevent additional spam at rejected packet
395 return;
396 }
397
398 Guild* guild = GetPlayer()->GetGuild();
399 if (!guild)
400 {
401 recvData.rfinish(); // Prevent additional spam at rejected packet
402 return;
403 }
404
405 uint8 bankToBank;
406 recvData >> bankToBank;
407
408 uint8 tabId;
409 uint8 slotId;
410 uint32 itemEntry;
411 uint32 splitedAmount = 0;
412
413 if (bankToBank)
414 {
415 uint8 destTabId;
416 recvData >> destTabId;
417
418 uint8 destSlotId;
419 recvData >> destSlotId;
420 recvData.read_skip<uint32>(); // Always 0
421
422 recvData >> tabId;
423 recvData >> slotId;
424 recvData >> itemEntry;
425 recvData.read_skip<uint8>(); // Always 0
426
427 recvData >> splitedAmount;
428
429 guild->SwapItems(GetPlayer(), tabId, slotId, destTabId, destSlotId, splitedAmount);
430 }
431 else
432 {
433 uint8 playerBag = NULL_BAG;
434 uint8 playerSlotId = NULL_SLOT;
435 uint8 toChar = 1;
436
437 recvData >> tabId;
438 recvData >> slotId;
439 recvData >> itemEntry;
440
441 uint8 autoStore;
442 recvData >> autoStore;
443 if (autoStore)
444 {
445 recvData.read_skip<uint32>(); // autoStoreCount
446 recvData.read_skip<uint8>(); // ToChar (?), always and expected to be 1 (autostore only triggered in Bank -> Char)
447 recvData.read_skip<uint32>(); // Always 0
448 }
449 else
450 {
451 recvData >> playerBag;
452 recvData >> playerSlotId;
453 recvData >> toChar;
454 recvData >> splitedAmount;
455 }
456
457 // Player <-> Bank
458 // Allow to work with inventory only
459 if (!Player::IsInventoryPos(playerBag, playerSlotId) && !(playerBag == NULL_BAG && playerSlotId == NULL_SLOT))
460 GetPlayer()->SendEquipError(EQUIP_ERR_NONE, NULL);
461 else
462 guild->SwapItemsWithInventory(GetPlayer(), toChar != 0, tabId, slotId, playerBag, playerSlotId, splitedAmount);
463 }
464}

32bool WorldSession::CanOpenMailBox(ObjectGuid guid)
33{
34 if (guid == _player->GetGUID())
35 {
36 if (!HasPermission(rbac::RBAC_PERM_COMMAND_MAILBOX))
37 {
38 TC_LOG_WARN("cheat", "%s attempted to open mailbox by using a cheat.", _player->GetName().c_str());
39 return false;
40 }
41 }
42 else if (guid.IsGameObject())
43 {
44 if (!_player->GetGameObjectIfCanInteractWith(guid, GAMEOBJECT_TYPE_MAILBOX))
45 return false;
46 }
47 else if (guid.IsAnyTypeCreature())
48 {
49 if (!_player->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_MAILBOX))
50 return false;
51 }
52 else
53 return false;
54
55 return true;
56}

58void WorldSession::HandleSendMail(WorldPacket& recvData)
59{
60 ObjectGuid mailbox, unk3;
61 std::string receiverName, subject, body;
62 uint32 stationery, package, money, COD;
63 uint8 unk4;
64 uint8 items_count;
65 recvData >> mailbox >> receiverName >> subject >> body
66 >> stationery // stationery?
67 >> package // 0x00000000
68 >> items_count; // attached items count
69
70 if (items_count > MAX_MAIL_ITEMS) // client limit
71 {
72 GetPlayer()->SendMailResult(0, MAIL_SEND, MAIL_ERR_TOO_MANY_ATTACHMENTS);
73 recvData.rfinish(); // set to end to avoid warnings spam
74 return;
75 }
76
77 ObjectGuid itemGUIDs[MAX_MAIL_ITEMS];
78
79 for (uint8 i = 0; i < items_count; ++i)
80 {
81 recvData.read_skip<uint8>(); // item slot in mail, not used
82 recvData >> itemGUIDs[i];
83 }
84
85 recvData >> money >> COD; // money and cod
86 recvData >> unk3; // const 0
87 recvData >> unk4; // const 0
88
89 // packet read complete, now do check
90
91 if (!CanOpenMailBox(mailbox))
92 return;
93
94 if (receiverName.empty())
95 return;
96
97 Player* player = _player;
98
99 if (player->getLevel() < sWorld->getIntConfig(CONFIG_MAIL_LEVEL_REQ))
100 {
101 SendNotification(GetTrinityString(LANG_MAIL_SENDER_REQ), sWorld->getIntConfig(CONFIG_MAIL_LEVEL_REQ));
102 return;
103 }
104
105 ObjectGuid receiverGuid;
106 if (normalizePlayerName(receiverName))
107 receiverGuid = sObjectMgr->GetPlayerGUIDByName(receiverName);
108
109 if (!receiverGuid)
110 {
111 TC_LOG_INFO("network", "Player %u is sending mail to %s (GUID: non-existing!) with subject %s "
112 "and body %s includes %u items, %u copper and %u COD copper with unk1 = %u, unk2 = %u",
113 player->GetGUID().GetCounter(), receiverName.c_str(), subject.c_str(), body.c_str(),
114 items_count, money, COD, stationery, package);
115 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_RECIPIENT_NOT_FOUND);
116 return;
117 }
118
119 TC_LOG_INFO("network", "Player %u is sending mail to %s (%s) with subject %s and body %s "
120 "including %u items, %u copper and %u COD copper with unk1 = %u, unk2 = %u",
121 player->GetGUID().GetCounter(), receiverName.c_str(), receiverGuid.ToString().c_str(), subject.c_str(),
122 body.c_str(), items_count, money, COD, stationery, package);
123
124 if (player->GetGUID() == receiverGuid)
125 {
126 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_CANNOT_SEND_TO_SELF);
127 return;
128 }
129
130 uint32 cost = items_count ? 30 * items_count : 30; // price hardcoded in client
131
132 uint32 reqmoney = cost + money;
133
134 // Check for overflow
135 if (reqmoney < money)
136 {
137 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_NOT_ENOUGH_MONEY);
138 return;
139 }
140
141 if (!player->HasEnoughMoney(reqmoney) && !player->IsGameMaster())
142 {
143 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_NOT_ENOUGH_MONEY);
144 return;
145 }
146
147 Player* receiver = ObjectAccessor::FindConnectedPlayer(receiverGuid);
148
149 uint32 receiverTeam = 0;
150 uint8 mailsCount = 0; //do not allow to send to one player more than 100 mails
151 uint8 receiverLevel = 0;
152 uint32 receiverAccountId = 0;
153
154 if (receiver)
155 {
156 receiverTeam = receiver->GetTeam();
157 mailsCount = receiver->GetMailSize();
158 receiverLevel = receiver->getLevel();
159 receiverAccountId = receiver->GetSession()->GetAccountId();
160 }
161 else
162 {
163 receiverTeam = sObjectMgr->GetPlayerTeamByGUID(receiverGuid);
164
165 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_MAIL_COUNT);
166 stmt->setUInt32(0, receiverGuid.GetCounter());
167
168 PreparedQueryResult result = CharacterDatabase.Query(stmt);
169 if (result)
170 {
171 Field* fields = result->Fetch();
172 mailsCount = fields[0].GetUInt64();
173 }
174
175 stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHAR_LEVEL);
176 stmt->setUInt32(0, receiverGuid.GetCounter());
177
178 result = CharacterDatabase.Query(stmt);
179 if (result)
180 {
181 Field* fields = result->Fetch();
182 receiverLevel = fields[0].GetUInt8();
183 }
184
185 receiverAccountId = sObjectMgr->GetPlayerAccountIdByGUID(receiverGuid);
186 }
187
188 // do not allow to have more than 100 mails in mailbox.. mails count is in opcode uint8!!! - so max can be 255..
189 if (mailsCount > 100)
190 {
191 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_RECIPIENT_CAP_REACHED);
192 return;
193 }
194
195 // test the receiver's Faction... or all items are account bound
196 bool accountBound = items_count ? true : false;
197 for (uint8 i = 0; i < items_count; ++i)
198 {
199 if (Item* item = player->GetItemByGuid(itemGUIDs[i]))
200 {
201 ItemTemplate const* itemProto = item->GetTemplate();
202 if (!itemProto || !(itemProto->Flags & ITEM_PROTO_FLAG_BIND_TO_ACCOUNT))
203 {
204 accountBound = false;
205 break;
206 }
207 }
208 }
209
210 if (!accountBound && player->GetTeam() != receiverTeam && !HasPermission(rbac::RBAC_PERM_TWO_SIDE_INTERACTION_MAIL))
211 {
212 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_NOT_YOUR_TEAM);
213 return;
214 }
215
216 if (receiverLevel < sWorld->getIntConfig(CONFIG_MAIL_LEVEL_REQ))
217 {
218 SendNotification(GetTrinityString(LANG_MAIL_RECEIVER_REQ), sWorld->getIntConfig(CONFIG_MAIL_LEVEL_REQ));
219 return;
220 }
221
222 Item* items[MAX_MAIL_ITEMS];
223
224 for (uint8 i = 0; i < items_count; ++i)
225 {
226 if (!itemGUIDs[i])
227 {
228 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_MAIL_ATTACHMENT_INVALID);
229 return;
230 }
231
232 Item* item = player->GetItemByGuid(itemGUIDs[i]);
233
234 // prevent sending bag with items (cheat: can be placed in bag after adding equipped empty bag to mail)
235 if (!item)
236 {
237 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_MAIL_ATTACHMENT_INVALID);
238 return;
239 }
240
241 if (!item->CanBeTraded(true))
242 {
243 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_EQUIP_ERROR, EQUIP_ERR_MAIL_BOUND_ITEM);
244 return;
245 }
246
247 if (item->IsBoundAccountWide() && item->IsSoulBound() && player->GetSession()->GetAccountId() != receiverAccountId)
248 {
249 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_EQUIP_ERROR, EQUIP_ERR_ARTEFACTS_ONLY_FOR_OWN_CHARACTERS);
250 return;
251 }
252
253 if (item->GetTemplate()->Flags & ITEM_PROTO_FLAG_CONJURED || item->GetUInt32Value(ITEM_FIELD_DURATION))
254 {
255 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_EQUIP_ERROR, EQUIP_ERR_MAIL_BOUND_ITEM);
256 return;
257 }
258
259 if (COD && item->HasFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_WRAPPED))
260 {
261 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_CANT_SEND_WRAPPED_COD);
262 return;
263 }
264
265 if (item->IsNotEmptyBag())
266 {
267 player->SendMailResult(0, MAIL_SEND, MAIL_ERR_EQUIP_ERROR, EQUIP_ERR_CAN_ONLY_DO_WITH_EMPTY_BAGS);
268 return;
269 }
270
271 items[i] = item;
272 }
273
274 player->SendMailResult(0, MAIL_SEND, MAIL_OK);
275
276 player->ModifyMoney(-int32(reqmoney));
277 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GOLD_SPENT_FOR_MAIL, cost);
278
279 bool needItemDelay = false;
280
281 MailDraft draft(subject, body);
282
283 SQLTransaction trans = CharacterDatabase.BeginTransaction();
284
285 if (items_count > 0 || money > 0)
286 {
287 bool log = HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE);
288 if (items_count > 0)
289 {
290 for (uint8 i = 0; i < items_count; ++i)
291 {
292 Item* item = items[i];
293 if (log)
294 {
295 sLog->outCommand(GetAccountId(), "GM %s (GUID: %u) (Account: %u) mail item: %s (Entry: %u Count: %u) "
296 "to: %s (%s) (Account: %u)", GetPlayerName().c_str(), GetGUIDLow(), GetAccountId(),
297 item->GetTemplate()->Name1.c_str(), item->GetEntry(), item->GetCount(),
298 receiverName.c_str(), receiverGuid.ToString().c_str(), receiverAccountId);
299 }
300
301 item->SetNotRefundable(GetPlayer()); // makes the item no longer refundable
302 player->MoveItemFromInventory(items[i]->GetBagSlot(), item->GetSlot(), true);
303
304 item->DeleteFromInventoryDB(trans); // deletes item from character's inventory
305 item->SetOwnerGUID(receiverGuid);
306 item->SaveToDB(trans); // recursive and not have transaction guard into self, item not in inventory and can be save standalone
307
308 draft.AddItem(item);
309 }
310
311 // if item send to character at another account, then apply item delivery delay
312 needItemDelay = player->GetSession()->GetAccountId() != receiverAccountId;
313 }
314
315 if (log && money > 0)
316 {
317 sLog->outCommand(GetAccountId(), "GM %s (GUID: %u) (Account: %u) mail money: %u to: %s (%s) (Account: %u)",
318 GetPlayerName().c_str(), GetGUIDLow(), GetAccountId(), money, receiverName.c_str(), receiverGuid.ToString().c_str(), receiverAccountId);
319 }
320 }
321
322 // If theres is an item, there is a one hour delivery delay if sent to another account's character.
323 uint32 deliver_delay = needItemDelay ? sWorld->getIntConfig(CONFIG_MAIL_DELIVERY_DELAY) : 0;
324
325 // don't ask for COD if there are no items
326 if (items_count == 0)
327 COD = 0;
328
329 // will delete item or place to receiver mail list
330 draft
331 .AddMoney(money)
332 .AddCOD(COD)
333 .SendMailTo(trans, MailReceiver(receiver, receiverGuid.GetCounter()), MailSender(player), body.empty() ? MAIL_CHECK_MASK_COPIED : MAIL_CHECK_MASK_HAS_BODY, deliver_delay);
334
335 player->SaveInventoryAndGoldToDB(trans);
336 CharacterDatabase.CommitTransaction(trans);
337}

340void WorldSession::HandleMailMarkAsRead(WorldPacket& recvData)
341{
342 ObjectGuid mailbox;
343 uint32 mailId;
344 recvData >> mailbox;
345 recvData >> mailId;
346
347 if (!CanOpenMailBox(mailbox))
348 return;
349
350 Player* player = _player;
351 Mail* m = player->GetMail(mailId);
352 if (m && m->state != MAIL_STATE_DELETED)
353 {
354 if (player->unReadMails)

356 m->checked = m->checked | MAIL_CHECK_MASK_READ;
357 player->m_mailsUpdated = true;
358 m->state = MAIL_STATE_CHANGED;
359 }

391void WorldSession::HandleMailReturnToSender(WorldPacket& recvData)
392{
393 ObjectGuid mailbox;
394 uint32 mailId;
395 recvData >> mailbox;
396 recvData >> mailId;
397 recvData.read_skip<uint64>(); // original sender GUID for return to, not used
398
399 if (!CanOpenMailBox(mailbox))
400 return;
401
402 Player* player = _player;
403 Mail* m = player->GetMail(mailId);
404 if (!m || m->state == MAIL_STATE_DELETED || m->deliver_time > time(NULL))
405 {
406 player->SendMailResult(mailId, MAIL_RETURNED_TO_SENDER, MAIL_ERR_INTERNAL_ERROR);
407 return;
408 }
409 //we can return mail now
410 //so firstly delete the old one
411 SQLTransaction trans = CharacterDatabase.BeginTransaction();
412
413 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_MAIL_BY_ID);
414 stmt->setUInt32(0, mailId);
415 trans->Append(stmt);
416
417 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_MAIL_ITEM_BY_ID);
418 stmt->setUInt32(0, mailId);
419 trans->Append(stmt);
420
421 player->RemoveMail(mailId);
422
423 // only return mail if the player exists (and delete if not existing)
424 if (m->messageType == MAIL_NORMAL && m->sender)
425 {
426 MailDraft draft(m->subject, m->body);
427 if (m->mailTemplateId)
428 draft = MailDraft(m->mailTemplateId, false); // items already included
429
430 if (m->HasItems())
431 {
432 for (MailItemInfoVec::iterator itr2 = m->items.begin(); itr2 != m->items.end(); ++itr2)
433 {
434 Item* item = player->GetMItem(itr2->item_guid);
435 if (item)
436 draft.AddItem(item);
437 else
438 {
439 //WTF?
440 }
441
442 player->RemoveMItem(itr2->item_guid);
443 }
444 }
445 draft.AddMoney(m->money).SendReturnToSender(GetAccountId(), m->receiver, m->sender, trans);
446 }
447
448 CharacterDatabase.CommitTransaction(trans);
449
450 delete m; //we can deallocate old mail
451 player->SendMailResult(mailId, MAIL_RETURNED_TO_SENDER, MAIL_OK);
452}

454//called when player takes item attached in mail
455void WorldSession::HandleMailTakeItem(WorldPacket& recvData)
456{
457 ObjectGuid mailbox;
458 uint32 mailId;
459 uint32 itemId;
460 recvData >> mailbox;
461 recvData >> mailId;
462 recvData >> itemId; // item guid low
463
464 if (!CanOpenMailBox(mailbox))
465 return;
466
467 Player* player = _player;
468
469 Mail* m = player->GetMail(mailId);
470 if (!m || m->state == MAIL_STATE_DELETED || m->deliver_time > time(NULL))
471 {
472 player->SendMailResult(mailId, MAIL_ITEM_TAKEN, MAIL_ERR_INTERNAL_ERROR);
473 return;
474 }
475
476 // verify that the mail has the item to avoid cheaters taking COD items without paying
477 if (std::find_if(m->items.begin(), m->items.end(), [itemId](MailItemInfo info){ return info.item_guid == itemId; }) == m->items.end())
478 {
479 player->SendMailResult(mailId, MAIL_ITEM_TAKEN, MAIL_ERR_INTERNAL_ERROR);
480 return;
481 }
482
483 // prevent cheating with skip client money check
484 if (!player->HasEnoughMoney(m->COD))
485 {
486 player->SendMailResult(mailId, MAIL_ITEM_TAKEN, MAIL_ERR_NOT_ENOUGH_MONEY);
487 return;
488 }
489
490 Item* it = player->GetMItem(itemId);
491
492 ItemPosCountVec dest;
493 uint8 msg = _player->CanStoreItem(NULL_BAG, NULL_SLOT, dest, it, false);
494 if (msg == EQUIP_ERR_OK)
495 {
496 SQLTransaction trans = CharacterDatabase.BeginTransaction();
497 m->RemoveItem(itemId);
498 m->removedItems.push_back(itemId);
499
500 if (m->COD > 0) //if there is COD, take COD money from player and send them to sender by mail
501 {
502 ObjectGuid sender_guid(HighGuid::Player, m->sender);
503 Player* receiver = ObjectAccessor::FindConnectedPlayer(sender_guid);
504
505 uint32 sender_accId = 0;
506
507 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
508 {
509 std::string sender_name;
510 if (receiver)
511 {
512 sender_accId = receiver->GetSession()->GetAccountId();
513 sender_name = receiver->GetName();
514 }
515 else
516 {
517 // can be calculated early
518 sender_accId = sObjectMgr->GetPlayerAccountIdByGUID(sender_guid);
519
520 if (!sObjectMgr->GetPlayerNameByGUID(sender_guid, sender_name))
521 sender_name = sObjectMgr->GetTrinityStringForDBCLocale(LANG_UNKNOWN);
522 }
523 sLog->outCommand(GetAccountId(), "GM %s (Account: %u) receiver mail item: %s (Entry: %u Count: %u) and send COD money: %u to player: %s (Account: %u)",
524 GetPlayerName().c_str(), GetAccountId(), it->GetTemplate()->Name1.c_str(), it->GetEntry(), it->GetCount(), m->COD, sender_name.c_str(), sender_accId);
525 }
526 else if (!receiver)
527 sender_accId = sObjectMgr->GetPlayerAccountIdByGUID(sender_guid);
528
529 // check player existence
530 if (receiver || sender_accId)
531 {
532 MailDraft(m->subject, "")
533 .AddMoney(m->COD)
534 .SendMailTo(trans, MailReceiver(receiver, m->sender), MailSender(MAIL_NORMAL, m->receiver), MAIL_CHECK_MASK_COD_PAYMENT);
535 }
536
537 player->ModifyMoney(-int32(m->COD));
538 }
539 m->COD = 0;
540 m->state = MAIL_STATE_CHANGED;
541 player->m_mailsUpdated = true;
542 player->RemoveMItem(it->GetGUID().GetCounter());
543
544 uint32 count = it->GetCount(); // save counts before store and possible merge with deleting
545 it->SetState(ITEM_UNCHANGED); // need to set this state, otherwise item cannot be removed later, if neccessary
546 player->MoveItemToInventory(dest, it, true);
547
548 player->SaveInventoryAndGoldToDB(trans);
549 player->_SaveMail(trans);
550 CharacterDatabase.CommitTransaction(trans);
551
552 player->SendMailResult(mailId, MAIL_ITEM_TAKEN, MAIL_OK, 0, itemId, count);
553 }
554 else
555 player->SendMailResult(mailId, MAIL_ITEM_TAKEN, MAIL_ERR_EQUIP_ERROR, msg);
556}

558void WorldSession::HandleMailTakeMoney(WorldPacket& recvData)
559{
560 ObjectGuid mailbox;
561 uint32 mailId;
562 recvData >> mailbox;
563 recvData >> mailId;
564
565 if (!CanOpenMailBox(mailbox))
566 return;
567
568 Player* player = _player;
569
570 Mail* m = player->GetMail(mailId);
571 if (!m || m->state == MAIL_STATE_DELETED || m->deliver_time > time(NULL))
572 {
573 player->SendMailResult(mailId, MAIL_MONEY_TAKEN, MAIL_ERR_INTERNAL_ERROR);
574 return;
575 }
576
577 if (!player->ModifyMoney(m->money, false))
578 {
579 player->SendMailResult(mailId, MAIL_MONEY_TAKEN, MAIL_ERR_EQUIP_ERROR, EQUIP_ERR_TOO_MUCH_GOLD);
580 return;
581 }
582
583 m->money = 0;
584 m->state = MAIL_STATE_CHANGED;
585 player->m_mailsUpdated = true;
586
587 player->SendMailResult(mailId, MAIL_MONEY_TAKEN, MAIL_OK);
588
589 // save money and mail to prevent cheating
590 SQLTransaction trans = CharacterDatabase.BeginTransaction();
591 player->SaveGoldToDB(trans);
592 player->_SaveMail(trans);
593 CharacterDatabase.CommitTransaction(trans);
594}

596//called when player lists his received mails
597void WorldSession::HandleGetMailList(WorldPacket& recvData)
598{
599 ObjectGuid mailbox;
600 recvData >> mailbox;
601
602 if (!CanOpenMailBox(mailbox))
603 return;
604
605 Player* player = _player;
606
607 //load players mails, and mailed items
608 if (!player->m_mailsLoaded)
609 player->_LoadMail();
610
611 // client can't work with packets > max int16 value
612 const uint32 maxPacketSize = 32767;
613
614 uint32 mailsCount = 0; // real send to client mails amount
615 uint32 realCount = 0; // real mails amount
616
617 WorldPacket data(SMSG_MAIL_LIST_RESULT, (200)); // guess size
618 data << uint32(0); // real mail's count
619 data << uint8(0); // mail's count
620 time_t cur_time = time(NULL);
621
622 for (PlayerMails::iterator itr = player->GetMailBegin(); itr != player->GetMailEnd(); ++itr)
623 {
624 // Only first 50 mails are displayed
625 if (mailsCount >= 50)
626 {
627 realCount += 1;
628 continue;
629 }
630
631 // skip deleted or not delivered (deliver delay not expired) mails
632 if ((*itr)->state == MAIL_STATE_DELETED || cur_time < (*itr)->deliver_time)
633 continue;
634
635 uint8 item_count = (*itr)->items.size(); // max count is MAX_MAIL_ITEMS (12)
636
637 size_t next_mail_size = 2+4+1+((*itr)->messageType == MAIL_NORMAL ? 8 : 4)+4*8+((*itr)->subject.size()+1)+((*itr)->body.size()+1)+1+item_count*(1+4+4+MAX_INSPECTED_ENCHANTMENT_SLOT*3*4+4+4+4+4+4+4+1);
638
639 if (data.wpos()+next_mail_size > maxPacketSize)
640 {
641 realCount += 1;
642 continue;
643 }
644
645 data << uint16(next_mail_size); // Message size
646 data << uint32((*itr)->messageID); // Message ID
647 data << uint8((*itr)->messageType); // Message Type
648
649 switch ((*itr)->messageType)
650 {
651 case MAIL_NORMAL: // sender guid
652 data << ObjectGuid(HighGuid::Player, (*itr)->sender);
653 break;
654 case MAIL_CREATURE:
655 case MAIL_GAMEOBJECT:
656 case MAIL_AUCTION:
657 case MAIL_CALENDAR:
658 data << uint32((*itr)->sender); // creature/gameobject entry, auction id, calendar event id?
659 break;
660 }
661
662 data << uint32((*itr)->COD); // COD
663 data << uint32(0); // package (Package.dbc)
664 data << uint32((*itr)->stationery); // stationery (Stationery.dbc)
665 data << uint32((*itr)->money); // Gold
666 data << uint32((*itr)->checked); // flags
667 data << float(float((*itr)->expire_time-time(NULL))/DAY); // Time
668 data << uint32((*itr)->mailTemplateId); // mail template (MailTemplate.dbc)
669 data << (*itr)->subject; // Subject string - once 00, when mail type = 3, max 256
670 data << (*itr)->body; // message? max 8000
671 data << uint8(item_count); // client limit is 0x10
672
673 for (uint8 i = 0; i < item_count; ++i)
674 {
675 Item* item = player->GetMItem((*itr)->items[i].item_guid);
676 // item index (0-6?)
677 data << uint8(i);
678 // item guid low?
679 data << uint32((item ? item->GetGUID().GetCounter() : 0));
680 // entry
681 data << uint32((item ? item->GetEntry() : 0));
682 for (uint8 j = 0; j < MAX_INSPECTED_ENCHANTMENT_SLOT; ++j)
683 {
684 data << uint32((item ? item->GetEnchantmentId((EnchantmentSlot)j) : 0));
685 data << uint32((item ? item->GetEnchantmentDuration((EnchantmentSlot)j) : 0));
686 data << uint32((item ? item->GetEnchantmentCharges((EnchantmentSlot)j) : 0));
687 }
688 // can be negative
689 data << int32((item ? item->GetItemRandomPropertyId() : 0));
690 // unk
691 data << uint32((item ? item->GetItemSuffixFactor() : 0));
692 // stack count
693 data << uint32((item ? item->GetCount() : 0));
694 // charges
695 data << uint32((item ? item->GetSpellCharges() : 0));
696 // durability
697 data << uint32((item ? item->GetUInt32Value(ITEM_FIELD_MAXDURABILITY) : 0));
698 // durability
699 data << uint32((item ? item->GetUInt32Value(ITEM_FIELD_DURABILITY) : 0));
700 // unknown wotlk
701 data << uint8(0);
702 }
703
704 ++realCount;
705 ++mailsCount;
706 }
707
708 data.put<uint32>(0, realCount); // this will display warning about undelivered mail to player if realCount > mailsCount
709 data.put<uint8>(4, mailsCount); // set real send mails to client
710 SendPacket(&data);
711
712 // recalculate m_nextMailDelivereTime and unReadMails
713 _player->UpdateNextMailTimeAndUnreads();
714}

716//used when player copies mail body to his inventory
717void WorldSession::HandleMailCreateTextItem(WorldPacket& recvData)
718{
719 ObjectGuid mailbox;
720 uint32 mailId;
721
722 recvData >> mailbox;
723 recvData >> mailId;
724
725 if (!CanOpenMailBox(mailbox))
726 return;
727
728 Player* player = _player;
729
730 Mail* m = player->GetMail(mailId);
731 if (!m || (m->body.empty() && !m->mailTemplateId) || m->state == MAIL_STATE_DELETED || m->deliver_time > time(NULL) || (m->checked & MAIL_CHECK_MASK_COPIED))
732 {
733 player->SendMailResult(mailId, MAIL_MADE_PERMANENT, MAIL_ERR_INTERNAL_ERROR);
734 return;
735 }
736
737 Item* bodyItem = new Item; // This is not bag and then can be used new Item.
738 if (!bodyItem->Create(sObjectMgr->GetGenerator<HighGuid::Item>().Generate(), MAIL_BODY_ITEM_TEMPLATE, player))
739 {
740 delete bodyItem;
741 return;
742 }
743
744 // in mail template case we need create new item text
745 if (m->mailTemplateId)
746 {
747 MailTemplateEntry const* mailTemplateEntry = sMailTemplateStore.LookupEntry(m->mailTemplateId);
748 if (!mailTemplateEntry)
749 {
750 player->SendMailResult(mailId, MAIL_MADE_PERMANENT, MAIL_ERR_INTERNAL_ERROR);
751 return;
752 }
753
754 bodyItem->SetText(mailTemplateEntry->content[GetSessionDbcLocale()]);
755 }
756 else
757 bodyItem->SetText(m->body);
758
759 if (m->messageType == MAIL_NORMAL)
760 bodyItem->SetGuidValue(ITEM_FIELD_CREATOR, ObjectGuid(HighGuid::Player, m->sender));
761
762 bodyItem->SetFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_MAIL_TEXT_MASK);
763
764 TC_LOG_INFO("network", "HandleMailCreateTextItem mailid=%u", mailId);
765
766 ItemPosCountVec dest;
767 uint8 msg = _player->CanStoreItem(NULL_BAG, NULL_SLOT, dest, bodyItem, false);
768 if (msg == EQUIP_ERR_OK)
769 {
770 m->checked = m->checked | MAIL_CHECK_MASK_COPIED;
771 m->state = MAIL_STATE_CHANGED;
772 player->m_mailsUpdated = true;
773
774 player->StoreItem(dest, bodyItem, true);
775 player->SendMailResult(mailId, MAIL_MADE_PERMANENT, MAIL_OK);
776 }
777 else
778 {
779 player->SendMailResult(mailId, MAIL_MADE_PERMANENT, MAIL_ERR_EQUIP_ERROR, msg);
780 delete bodyItem;
781 }
782}

785void WorldSession::HandleQueryNextMailTime(WorldPacket & /*recvData*/)
786{
787 WorldPacket data(MSG_QUERY_NEXT_MAIL_TIME, 8);
788
789 if (!_player->m_mailsLoaded)
790 _player->_LoadMail();
791
792 if (_player->unReadMails > 0)
793 {
794 data << float(0); // float
795 data << uint32(0); // count
796
797 uint32 count = 0;
798 time_t now = time(NULL);
799 std::set<uint32> sentSenders;
800 for (PlayerMails::iterator itr = _player->GetMailBegin(); itr != _player->GetMailEnd(); ++itr)
801 {
802 Mail* m = (*itr);
803 // must be not checked yet
804 if (m->checked & MAIL_CHECK_MASK_READ)
805 continue;
806
807 // and already delivered
808 if (now < m->deliver_time)
809 continue;
810
811 // only send each mail sender once
812 if (sentSenders.count(m->sender))
813 continue;
814
815 data << uint64(m->messageType == MAIL_NORMAL ? ObjectGuid(HighGuid::Player, m->sender) : ObjectGuid::Empty); // player guid
816 data << uint32(m->messageType != MAIL_NORMAL ? m->sender : 0); // non-player entries
817 data << uint32(m->messageType);
818 data << uint32(m->stationery);
819 data << float(m->deliver_time - now);
820
821 sentSenders.insert(m->sender);
822 ++count;
823 if (count == 2) // do not display more than 2 mails
824 break;
825 }
826
827 data.put<uint32>(4, count);
828 }
829 else
830 {
831 data << float(-DAY);
832 data << uint32(0);
833 }
834
835 SendPacket(&data);
836}

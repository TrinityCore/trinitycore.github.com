33void WorldSession::HandleAuctionHelloOpcode(WorldPacket& recvData)
34{
35 ObjectGuid guid; //NPC guid
36 recvData >> guid;
37
38 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_AUCTIONEER);
39 if (!unit)
40 {
41 TC_LOG_DEBUG("network", "WORLD: HandleAuctionHelloOpcode - Unit (%s) not found or you can't interact with him.", guid.ToString().c_str());

44

48
49 SendAuctionHello(guid, unit);
50}

52//this void causes that auction window is opened
53void WorldSession::SendAuctionHello(ObjectGuid guid, Creature* unit)
54{
55 if (GetPlayer()->getLevel() < sWorld->getIntConfig(CONFIG_AUCTION_LEVEL_REQ))
56 {
57 SendNotification(GetTrinityString(LANG_AUCTION_REQ), sWorld->getIntConfig(CONFIG_AUCTION_LEVEL_REQ));
58 return;
59 }
60
61 AuctionHouseEntry const* ahEntry = AuctionHouseMgr::GetAuctionHouseEntry(unit->getFaction());
62 if (!ahEntry)
63 return;
64
65 WorldPacket data(MSG_AUCTION_HELLO, 12);
66 data << uint64(guid);
67 data << uint32(ahEntry->houseId);
68 data << uint8(1); // 3.3.3: 1 - AH enabled, 0 - AH disabled
69 SendPacket(&data);
70}

112//this void creates new auction and adds auction to some auctionhouse
113void WorldSession::HandleAuctionSellItem(WorldPacket& recvData)
114{
115 ObjectGuid auctioneer;
116 uint32 itemsCount, etime, bid, buyout;
117 recvData >> auctioneer;
118 recvData >> itemsCount;
119
120 ObjectGuid itemGUIDs[MAX_AUCTION_ITEMS]; // 160 slot = 4x 36 slot bag + backpack 16 slot
121 uint32 count[MAX_AUCTION_ITEMS];
122 memset(count, 0, sizeof(count));
123
124 if (itemsCount > MAX_AUCTION_ITEMS)
125 {
126 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
127 recvData.rfinish();
128 return;
129 }
130
131 for (uint32 i = 0; i < itemsCount; ++i)
132 {
133 recvData >> itemGUIDs[i];
134 recvData >> count[i];
135
136 if (!itemGUIDs[i] || !count[i] || count[i] > 1000)
137 {
138 recvData.rfinish();
139 return;
140 }
141 }
142
143 recvData >> bid;
144 recvData >> buyout;
145 recvData >> etime;
146
147 if (!bid || !etime)
148 return;
149
150 if (bid > MAX_MONEY_AMOUNT || buyout > MAX_MONEY_AMOUNT)
151 {
152 TC_LOG_DEBUG("network", "WORLD: HandleAuctionSellItem - Player %s (GUID %u) attempted to sell item with higher price than max gold amount.", _player->GetName().c_str(), _player->GetGUID().GetCounter());
153 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
154 return;
155 }
156
157 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(auctioneer, UNIT_NPC_FLAG_AUCTIONEER);
158 if (!creature)
159 {
160 TC_LOG_DEBUG("network", "WORLD: HandleAuctionSellItem - Unit (%s) not found or you can't interact with him.", auctioneer.ToString().c_str());
161 return;
162 }
163
164 AuctionHouseEntry const* auctionHouseEntry = AuctionHouseMgr::GetAuctionHouseEntry(creature->getFaction());
165 if (!auctionHouseEntry)
166 {
167 TC_LOG_DEBUG("network", "WORLD: HandleAuctionSellItem - Unit (%s) has wrong faction.", auctioneer.ToString().c_str());
168 return;
169 }
170
171 etime *= MINUTE;
172
173 switch (etime)
174 {
175 case 1*MIN_AUCTION_TIME:
176 case 2*MIN_AUCTION_TIME:
177 case 4*MIN_AUCTION_TIME:
178 break;
179 default:
180 return;
181 }
182
183 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
184 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
185
186 Item* items[MAX_AUCTION_ITEMS];
187
188 uint32 finalCount = 0;
189 uint32 itemEntry = 0;
190
191 for (uint32 i = 0; i < itemsCount; ++i)
192 {
193 Item* item = _player->GetItemByGuid(itemGUIDs[i]);
194
195 if (!item)
196 {
197 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_ITEM_NOT_FOUND);
198 return;
199 }
200
201 if (itemEntry == 0)
202 itemEntry = item->GetTemplate()->ItemId;
203
204 if (sAuctionMgr->GetAItem(item->GetGUID().GetCounter()) || !item->CanBeTraded() || item->IsNotEmptyBag() ||
205 item->GetTemplate()->Flags & ITEM_PROTO_FLAG_CONJURED || item->GetUInt32Value(ITEM_FIELD_DURATION) ||
206 item->GetCount() < count[i] || itemEntry != item->GetTemplate()->ItemId)
207 {
208 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
209 return;
210 }
211
212 items[i] = item;
213 finalCount += count[i];
214 }
215
216 if (!finalCount)
217 {
218 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
219 return;
220 }
221
222 // check if there are 2 identical guids, in this case user is most likely cheating
223 for (uint32 i = 0; i < itemsCount - 1; ++i)
224 {
225 for (uint32 j = i + 1; j < itemsCount; ++j)
226 {
227 if (itemGUIDs[i] == itemGUIDs[j])
228 {
229 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
230 return;
231 }
232 }
233 }
234
235 for (uint32 i = 0; i < itemsCount; ++i)
236 {
237 Item* item = items[i];
238
239 if (item->GetMaxStackCount() < finalCount)
240 {
241 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
242 return;
243 }
244 }
245
246 Item* item = items[0];
247
248 uint32 auctionTime = uint32(etime * sWorld->getRate(RATE_AUCTION_TIME));
249 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());
250
251 uint32 deposit = sAuctionMgr->GetAuctionDeposit(auctionHouseEntry, etime, item, finalCount);
252 if (!_player->HasEnoughMoney(deposit))
253 {
254 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_NOT_ENOUGHT_MONEY);
255 return;
256 }
257
258 AuctionEntry* AH = new AuctionEntry();
259
260 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_AUCTION))
261 AH->houseId = AUCTIONHOUSE_NEUTRAL;
262 else
263 {
264 CreatureData const* auctioneerData = sObjectMgr->GetCreatureData(creature->GetSpawnId());
265 if (!auctioneerData)
266 {
267 TC_LOG_ERROR("misc", "Data for auctioneer not found (%s)", auctioneer.ToString().c_str());
268 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
269 delete AH;
270 return;
271 }
272
273 CreatureTemplate const* auctioneerInfo = sObjectMgr->GetCreatureTemplate(auctioneerData->id);
274 if (!auctioneerInfo)
275 {
276 TC_LOG_ERROR("misc", "Non existing auctioneer (%s)", auctioneer.ToString().c_str());
277 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
278 delete AH;
279 return;
280 }
281
282 const AuctionHouseEntry* AHEntry = sAuctionMgr->GetAuctionHouseEntry(auctioneerInfo->faction);
283 AH->houseId = AHEntry->houseId;
284 }
285
286 // Required stack size of auction matches to current item stack size, just move item to auctionhouse
287 if (itemsCount == 1 && item->GetCount() == count[0])
288 {
289 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
290 {
291 sLog->outCommand(GetAccountId(), "GM %s (Account: %u) create auction: %s (Entry: %u Count: %u)",
292 GetPlayerName().c_str(), GetAccountId(), item->GetTemplate()->Name1.c_str(), item->GetEntry(), item->GetCount());
293 }
294
295 AH->Id = sObjectMgr->GenerateAuctionID();
296 AH->itemGUIDLow = item->GetGUID().GetCounter();
297 AH->itemEntry = item->GetEntry();
298 AH->itemCount = item->GetCount();
299 AH->owner = _player->GetGUID().GetCounter();
300 AH->startbid = bid;
301 AH->bidder = 0;
302 AH->bid = 0;
303 AH->buyout = buyout;
304 AH->expire_time = time(NULL) + auctionTime;
305 AH->deposit = deposit;
306 AH->etime = etime;
307 AH->auctionHouseEntry = auctionHouseEntry;
308
309 TC_LOG_INFO("network", "CMSG_AUCTION_SELL_ITEM: Player %s (guid %d) is selling item %s entry %u (guid %d) with count %u with initial bid %u with buyout %u and with time %u (in sec) in auctionhouse %u",
310 _player->GetName().c_str(), _player->GetGUID().GetCounter(), item->GetTemplate()->Name1.c_str(), item->GetEntry(), item->GetGUID().GetCounter(), item->GetCount(), bid, buyout, auctionTime, AH->GetHouseId());
311 sAuctionMgr->AddAItem(item);
312 auctionHouse->AddAuction(AH);
313 sAuctionMgr->PendingAuctionAdd(_player, AH);
314
315 _player->MoveItemFromInventory(item->GetBagSlot(), item->GetSlot(), true);
316
317 SQLTransaction trans = CharacterDatabase.BeginTransaction();
318 item->DeleteFromInventoryDB(trans);
319 item->SaveToDB(trans);
320
321 AH->SaveToDB(trans);
322 _player->SaveInventoryAndGoldToDB(trans);
323 CharacterDatabase.CommitTransaction(trans);
324
325 SendAuctionCommandResult(AH->Id, AUCTION_SELL_ITEM, ERR_AUCTION_OK);
326
327 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_CREATE_AUCTION, 1);
328 }
329 else // Required stack size of auction does not match to current item stack size, clone item and set correct stack size
330 {
331 Item* newItem = item->CloneItem(finalCount, _player);
332 if (!newItem)
333 {
334 TC_LOG_ERROR("network", "CMSG_AUCTION_SELL_ITEM: Could not create clone of item %u", item->GetEntry());
335 SendAuctionCommandResult(0, AUCTION_SELL_ITEM, ERR_AUCTION_DATABASE_ERROR);
336 delete AH;
337 return;
338 }
339
340 if (HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE))
341 {
342 sLog->outCommand(GetAccountId(), "GM %s (Account: %u) create auction: %s (Entry: %u Count: %u)",
343 GetPlayerName().c_str(), GetAccountId(), newItem->GetTemplate()->Name1.c_str(), newItem->GetEntry(), newItem->GetCount());
344 }
345
346 AH->Id = sObjectMgr->GenerateAuctionID();
347 AH->itemGUIDLow = newItem->GetGUID().GetCounter();
348 AH->itemEntry = newItem->GetEntry();
349 AH->itemCount = newItem->GetCount();
350 AH->owner = _player->GetGUID().GetCounter();
351 AH->startbid = bid;
352 AH->bidder = 0;
353 AH->bid = 0;
354 AH->buyout = buyout;
355 AH->expire_time = time(NULL) + auctionTime;
356 AH->deposit = deposit;
357 AH->etime = etime;
358 AH->auctionHouseEntry = auctionHouseEntry;
359
360 TC_LOG_INFO("network", "CMSG_AUCTION_SELL_ITEM: Player %s (guid %d) is selling item %s entry %u (guid %d) with count %u with initial bid %u with buyout %u and with time %u (in sec) in auctionhouse %u",
361 _player->GetName().c_str(), _player->GetGUID().GetCounter(), newItem->GetTemplate()->Name1.c_str(), newItem->GetEntry(), newItem->GetGUID().GetCounter(), newItem->GetCount(), bid, buyout, auctionTime, AH->GetHouseId());
362 sAuctionMgr->AddAItem(newItem);
363 auctionHouse->AddAuction(AH);
364 sAuctionMgr->PendingAuctionAdd(_player, AH);
365
366 for (uint32 j = 0; j < itemsCount; ++j)
367 {
368 Item* item2 = items[j];
369
370 // Item stack count equals required count, ready to delete item - cloned item will be used for auction
371 if (item2->GetCount() == count[j])
372 {
373 _player->MoveItemFromInventory(item2->GetBagSlot(), item2->GetSlot(), true);
374
375 SQLTransaction trans = CharacterDatabase.BeginTransaction();
376 item2->DeleteFromInventoryDB(trans);
377 item2->DeleteFromDB(trans);
378 CharacterDatabase.CommitTransaction(trans);
379 delete item2;
380 }
381 else // Item stack count is bigger than required count, update item stack count and save to database - cloned item will be used for auction
382 {
383 item2->SetCount(item2->GetCount() - count[j]);
384 item2->SetState(ITEM_CHANGED, _player);
385 _player->ItemRemovedQuestCheck(item2->GetEntry(), count[j]);
386 item2->SendUpdateToPlayer(_player);
387
388 SQLTransaction trans = CharacterDatabase.BeginTransaction();
389 item2->SaveToDB(trans);
390 CharacterDatabase.CommitTransaction(trans);
391 }
392 }
393
394 SQLTransaction trans = CharacterDatabase.BeginTransaction();
395 newItem->SaveToDB(trans);
396 AH->SaveToDB(trans);
397 _player->SaveInventoryAndGoldToDB(trans);
398 CharacterDatabase.CommitTransaction(trans);
399
400 SendAuctionCommandResult(AH->Id, AUCTION_SELL_ITEM, ERR_AUCTION_OK);
401
402 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_CREATE_AUCTION, 1);
403 }
404}

406//this function is called when client bids or buys out auction
407void WorldSession::HandleAuctionPlaceBid(WorldPacket& recvData)
408{
409 TC_LOG_DEBUG("network", "WORLD: Received CMSG_AUCTION_PLACE_BID");
410
411 ObjectGuid auctioneer;
412 uint32 auctionId;
413 uint32 price;
414 recvData >> auctioneer;
415 recvData >> auctionId >> price;
416
417 if (!auctionId || !price)
418 return; //check for cheaters
419
420 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(auctioneer, UNIT_NPC_FLAG_AUCTIONEER);
421 if (!creature)
422 {
423 TC_LOG_DEBUG("network", "WORLD: HandleAuctionPlaceBid - %s not found or you can't interact with him.", auctioneer.ToString().c_str());
424 return;
425 }
426
427 // remove fake death
428 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
429 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
430
431 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());
432
433 AuctionEntry* auction = auctionHouse->GetAuction(auctionId);
434 Player* player = GetPlayer();
435
436 if (!auction || auction->owner == player->GetGUID().GetCounter())
437 {
438 //you cannot bid your own auction:
439 SendAuctionCommandResult(0, AUCTION_PLACE_BID, ERR_AUCTION_BID_OWN);
440 return;
441 }
442
443 // impossible have online own another character (use this for speedup check in case online owner)
444 ObjectGuid ownerGuid(HighGuid::Player, auction->owner);
445 Player* auction_owner = ObjectAccessor::FindPlayer(ownerGuid);
446 if (!auction_owner && sObjectMgr->GetPlayerAccountIdByGUID(ownerGuid) == player->GetSession()->GetAccountId())
447 {
448 //you cannot bid your another character auction:
449 SendAuctionCommandResult(0, AUCTION_PLACE_BID, ERR_AUCTION_BID_OWN);
450 return;
451 }
452
453 // cheating
454 if (price <= auction->bid || price < auction->startbid)
455 return;
456
457 // price too low for next bid if not buyout
458 if ((price < auction->buyout || auction->buyout == 0) &&
459 price < auction->bid + auction->GetAuctionOutBid())
460 {
461 //auction has already higher bid, client tests it!
462 return;
463 }
464
465 if (!player->HasEnoughMoney(price))
466 {
467 //you don't have enought money!, client tests!
468 //SendAuctionCommandResult(auction->auctionId, AUCTION_PLACE_BID, ???);
469 return;
470 }
471
472 SQLTransaction trans = CharacterDatabase.BeginTransaction();
473
474 if (price < auction->buyout || auction->buyout == 0)
475 {
476 if (auction->bidder > 0)
477 {
478 if (auction->bidder == player->GetGUID().GetCounter())
479 player->ModifyMoney(-int32(price - auction->bid));
480 else
481 {
482 // mail to last bidder and return money
483 sAuctionMgr->SendAuctionOutbiddedMail(auction, price, GetPlayer(), trans);
484 player->ModifyMoney(-int32(price));
485 }
486 }
487 else
488 player->ModifyMoney(-int32(price));
489
490 auction->bidder = player->GetGUID().GetCounter();
491 auction->bid = price;
492 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_BID, price);
493
494 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_AUCTION_BID);
495 stmt->setUInt32(0, auction->bidder);
496 stmt->setUInt32(1, auction->bid);
497 stmt->setUInt32(2, auction->Id);
498 trans->Append(stmt);
499
500 SendAuctionCommandResult(auction->Id, AUCTION_PLACE_BID, ERR_AUCTION_OK, 0);
501 }
502 else
503 {
504 //buyout:
505 if (player->GetGUID().GetCounter() == auction->bidder)
506 player->ModifyMoney(-int32(auction->buyout - auction->bid));
507 else
508 {
509 player->ModifyMoney(-int32(auction->buyout));
510 if (auction->bidder) //buyout for bidded auction ..
511 sAuctionMgr->SendAuctionOutbiddedMail(auction, auction->buyout, GetPlayer(), trans);
512 }
513 auction->bidder = player->GetGUID().GetCounter();
514 auction->bid = auction->buyout;
515 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_BID, auction->buyout);
516
517 //- Mails must be under transaction control too to prevent data loss
518 sAuctionMgr->SendAuctionSalePendingMail(auction, trans);
519 sAuctionMgr->SendAuctionSuccessfulMail(auction, trans);
520 sAuctionMgr->SendAuctionWonMail(auction, trans);
521
522 SendAuctionCommandResult(auction->Id, AUCTION_PLACE_BID, ERR_AUCTION_OK);
523
524 auction->DeleteFromDB(trans);
525
526 sAuctionMgr->RemoveAItem(auction->itemGUIDLow);
527 auctionHouse->RemoveAuction(auction);
528 }
529 player->SaveInventoryAndGoldToDB(trans);
530 CharacterDatabase.CommitTransaction(trans);
531}

533//this void is called when auction_owner cancels his auction
534void WorldSession::HandleAuctionRemoveItem(WorldPacket& recvData)
535{
536 TC_LOG_DEBUG("network", "WORLD: Received CMSG_AUCTION_REMOVE_ITEM");
537
538 ObjectGuid auctioneer;
539 uint32 auctionId;
540 recvData >> auctioneer;
541 recvData >> auctionId;
542 //TC_LOG_DEBUG("Cancel AUCTION AuctionID: %u", auctionId);
543
544 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(auctioneer, UNIT_NPC_FLAG_AUCTIONEER);
545 if (!creature)
546 {
547 TC_LOG_DEBUG("network", "WORLD: HandleAuctionRemoveItem - %s not found or you can't interact with him.", auctioneer.ToString().c_str());
548 return;
549 }
550
551 // remove fake death
552 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
553 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
554
555 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());
556
557 AuctionEntry* auction = auctionHouse->GetAuction(auctionId);
558 Player* player = GetPlayer();
559
560 SQLTransaction trans = CharacterDatabase.BeginTransaction();
561 if (auction && auction->owner == player->GetGUID().GetCounter())
562 {
563 Item* pItem = sAuctionMgr->GetAItem(auction->itemGUIDLow);
564 if (pItem)
565 {
566 if (auction->bidder > 0) // If we have a bidder, we have to send him the money he paid
567 {
568 uint32 auctionCut = auction->GetAuctionCut();
569 if (!player->HasEnoughMoney(auctionCut)) //player doesn't have enough money, maybe message needed
570 return;
571 //some auctionBidderNotification would be needed, but don't know that parts..
572 sAuctionMgr->SendAuctionCancelledToBidderMail(auction, trans);
573 player->ModifyMoney(-int32(auctionCut));
574 }
575
576 // item will deleted or added to received mail list
577 MailDraft(auction->BuildAuctionMailSubject(AUCTION_CANCELED), AuctionEntry::BuildAuctionMailBody(0, 0, auction->buyout, auction->deposit, 0))
578 .AddItem(pItem)
579 .SendMailTo(trans, player, auction, MAIL_CHECK_MASK_COPIED);
580 }
581 else
582 {
583 TC_LOG_ERROR("network", "Auction id: %u has non-existed item (item guid : %u)!!!", auction->Id, auction->itemGUIDLow);
584 SendAuctionCommandResult(0, AUCTION_CANCEL, ERR_AUCTION_DATABASE_ERROR);
585 return;
586 }
587 }
588 else
589 {
590 SendAuctionCommandResult(0, AUCTION_CANCEL, ERR_AUCTION_DATABASE_ERROR);
591 //this code isn't possible ... maybe there should be assert
592 TC_LOG_ERROR("network", "CHEATER : %u, he tried to cancel auction (id: %u) of another player, or auction is NULL", player->GetGUID().GetCounter(), auctionId);
593 return;
594 }
595
596 //inform player, that auction is removed
597 SendAuctionCommandResult(auction->Id, AUCTION_CANCEL, ERR_AUCTION_OK);
598
599 // Now remove the auction
600
601 player->SaveInventoryAndGoldToDB(trans);
602 auction->DeleteFromDB(trans);
603 CharacterDatabase.CommitTransaction(trans);
604
605 sAuctionMgr->RemoveAItem(auction->itemGUIDLow);
606 auctionHouse->RemoveAuction(auction);
607}

609//called when player lists his bids
610void WorldSession::HandleAuctionListBidderItems(WorldPacket& recvData)
611{
612 TC_LOG_DEBUG("network", "WORLD: Received CMSG_AUCTION_LIST_BIDDER_ITEMS");
613
614 ObjectGuid guid; //NPC guid
615 uint32 listfrom; //page of auctions
616 uint32 outbiddedCount; //count of outbidded auctions
617
618 recvData >> guid;
619 recvData >> listfrom; // not used in fact (this list not have page control in client)
620 recvData >> outbiddedCount;
621 if (recvData.size() != (16 + outbiddedCount * 4))
622 {
623 TC_LOG_ERROR("network", "Client sent bad opcode!!! with count: %u and size : %lu (must be: %u)", outbiddedCount, (unsigned long)recvData.size(), (16 + outbiddedCount * 4));
624 outbiddedCount = 0;
625 }
626
627 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_AUCTIONEER);
628 if (!creature)
629 {
630 TC_LOG_DEBUG("network", "WORLD: HandleAuctionListBidderItems - %s not found or you can't interact with him.", guid.ToString().c_str());
631 recvData.rfinish();

637 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);

639 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(creature->getFaction());

656 }

659 auctionHouse->BuildListBidderItems(data, player, count, totalcount);
660 data.put<uint32>(0, count); // add count to placeholder
661 data << totalcount;
662 data << (uint32)300; //unk 2.3.0

708 std::string searchedname;
709 uint8 levelmin, levelmax, usable, getAll;
710 uint32 listfrom, auctionSlotID, auctionMainCategory, auctionSubCategory, quality;

713 recvData >> guid;

717 recvData >> levelmin >> levelmax;
718 recvData >> auctionSlotID >> auctionMainCategory >> auctionSubCategory;
719 recvData >> quality >> usable;
720
721 recvData >> getAll;

723 // this block looks like it uses some lame byte packing or similar...

728 recvData.read_skip<uint8>();

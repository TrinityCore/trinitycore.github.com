29void WorldSession::HandleSplitItemOpcode(WorldPacket& recvData)
30{
31 //TC_LOG_DEBUG("network", "WORLD: CMSG_SPLIT_ITEM");
32 uint8 srcbag, srcslot, dstbag, dstslot;
33 uint32 count;
34
35 recvData >> srcbag >> srcslot >> dstbag >> dstslot >> count;
36 //TC_LOG_DEBUG("STORAGE: receive srcbag = %u, srcslot = %u, dstbag = %u, dstslot = %u, count = %u", srcbag, srcslot, dstbag, dstslot, count);
37
38 uint16 src = ((srcbag << 8) | srcslot);
39 uint16 dst = ((dstbag << 8) | dstslot);
40
41 if (src == dst)
42 return;
43
44 if (count == 0)
45 return; //check count - if zero it's fake packet
46
47 if (!_player->IsValidPos(srcbag, srcslot, true))
48 {
49 _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
50 return;
51 }
52
53 if (!_player->IsValidPos(dstbag, dstslot, false)) // can be autostore pos
54 {
55 _player->SendEquipError(EQUIP_ERR_ITEM_DOESNT_GO_TO_SLOT, NULL, NULL);
56 return;
57 }
58
59 _player->SplitItem(src, dst, count);
60}

62void WorldSession::HandleSwapInvItemOpcode(WorldPacket& recvData)
63{
64 //TC_LOG_DEBUG("network", "WORLD: CMSG_SWAP_INV_ITEM");
65 uint8 srcslot, dstslot;
66
67 recvData >> dstslot >> srcslot;
68 //TC_LOG_DEBUG("STORAGE: receive srcslot = %u, dstslot = %u", srcslot, dstslot);
69
70 // prevent attempt swap same item to current position generated by client at special checting sequence
71 if (srcslot == dstslot)
72 return;
73
74 if (!_player->IsValidPos(INVENTORY_SLOT_BAG_0, srcslot, true))
75 {
76 _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
77 return;
78 }
79
80 if (!_player->IsValidPos(INVENTORY_SLOT_BAG_0, dstslot, true))
81 {
82 _player->SendEquipError(EQUIP_ERR_ITEM_DOESNT_GO_TO_SLOT, NULL, NULL);
83 return;
84 }
85
86 if (_player->IsBankPos(INVENTORY_SLOT_BAG_0, srcslot) && !CanUseBank())
87 {
88 TC_LOG_DEBUG("network", "WORLD: HandleSwapInvItemOpcode - Unit (%s) not found or you can't interact with him.", m_currentBankerGUID.ToString().c_str());
89 return;
90 }
91
92 if (_player->IsBankPos(INVENTORY_SLOT_BAG_0, dstslot) && !CanUseBank())
93 {
94 TC_LOG_DEBUG("network", "WORLD: HandleSwapInvItemOpcode - Unit (%s) not found or you can't interact with him.", m_currentBankerGUID.ToString().c_str());
95 return;
96 }
97
98 uint16 src = ((INVENTORY_SLOT_BAG_0 << 8) | srcslot);
99 uint16 dst = ((INVENTORY_SLOT_BAG_0 << 8) | dstslot);
100
101 _player->SwapItem(src, dst);
102}

104void WorldSession::HandleAutoEquipItemSlotOpcode(WorldPacket& recvData)
105{
106 ObjectGuid itemguid;
107 uint8 dstslot;
108 recvData >> itemguid >> dstslot;
109
110 // cheating attempt, client should never send opcode in that case
111 if (!Player::IsEquipmentPos(INVENTORY_SLOT_BAG_0, dstslot))
112 return;
113
114 Item* item = _player->GetItemByGuid(itemguid);
115 uint16 dstpos = dstslot | (INVENTORY_SLOT_BAG_0 << 8);
116
117 if (!item || item->GetPos() == dstpos)
118 return;
119
120 _player->SwapItem(item->GetPos(), dstpos);
121}

165void WorldSession::HandleAutoEquipItemOpcode(WorldPacket& recvData)
166{
167 //TC_LOG_DEBUG("network", "WORLD: CMSG_AUTOEQUIP_ITEM");
168 uint8 srcbag, srcslot;
169
170 recvData >> srcbag >> srcslot;
171 //TC_LOG_DEBUG("STORAGE: receive srcbag = %u, srcslot = %u", srcbag, srcslot);
172
173 Item* pSrcItem = _player->GetItemByPos(srcbag, srcslot);
174 if (!pSrcItem)
175 return; // only at cheat
176
177 uint16 dest;
178 InventoryResult msg = _player->CanEquipItem(NULL_SLOT, dest, pSrcItem, !pSrcItem->IsBag());
179 if (msg != EQUIP_ERR_OK)
180 {
181 _player->SendEquipError(msg, pSrcItem, NULL);
182 return;
183 }
184
185 uint16 src = pSrcItem->GetPos();
186 if (dest == src) // prevent equip in same slot, only at cheat
187 return;
188
189 Item* pDstItem = _player->GetItemByPos(dest);
190 if (!pDstItem) // empty slot, simple case
191 {
192 _player->RemoveItem(srcbag, srcslot, true);
193 _player->EquipItem(dest, pSrcItem, true);
194 _player->AutoUnequipOffhandIfNeed();
195 }
196 else // have currently equipped item, not simple case
197 {
198 uint8 dstbag = pDstItem->GetBagSlot();
199 uint8 dstslot = pDstItem->GetSlot();
200
201 msg = _player->CanUnequipItem(dest, !pSrcItem->IsBag());
202 if (msg != EQUIP_ERR_OK)
203 {
204 _player->SendEquipError(msg, pDstItem, NULL);
205 return;
206 }
207
208 // check dest->src move possibility
209 ItemPosCountVec sSrc;
210 uint16 eSrc = 0;
211 if (_player->IsInventoryPos(src))
212 {
213 msg = _player->CanStoreItem(srcbag, srcslot, sSrc, pDstItem, true);
214 if (msg != EQUIP_ERR_OK)
215 msg = _player->CanStoreItem(srcbag, NULL_SLOT, sSrc, pDstItem, true);
216 if (msg != EQUIP_ERR_OK)
217 msg = _player->CanStoreItem(NULL_BAG, NULL_SLOT, sSrc, pDstItem, true);
218 }
219 else if (_player->IsBankPos(src))
220 {
221 msg = _player->CanBankItem(srcbag, srcslot, sSrc, pDstItem, true);
222 if (msg != EQUIP_ERR_OK)
223 msg = _player->CanBankItem(srcbag, NULL_SLOT, sSrc, pDstItem, true);
224 if (msg != EQUIP_ERR_OK)
225 msg = _player->CanBankItem(NULL_BAG, NULL_SLOT, sSrc, pDstItem, true);
226 }
227 else if (_player->IsEquipmentPos(src))
228 {
229 msg = _player->CanEquipItem(srcslot, eSrc, pDstItem, true);
230 if (msg == EQUIP_ERR_OK)
231 msg = _player->CanUnequipItem(eSrc, true);
232 }
233
234 if (msg != EQUIP_ERR_OK)
235 {
236 _player->SendEquipError(msg, pDstItem, pSrcItem);
237 return;
238 }
239
240 // now do moves, remove...
241 _player->RemoveItem(dstbag, dstslot, false);
242 _player->RemoveItem(srcbag, srcslot, false);
243
244 // add to dest
245 _player->EquipItem(dest, pSrcItem, true);
246
247 // add to src
248 if (_player->IsInventoryPos(src))
249 _player->StoreItem(sSrc, pDstItem, true);
250 else if (_player->IsBankPos(src))
251 _player->BankItem(sSrc, pDstItem, true);
252 else if (_player->IsEquipmentPos(src))
253 _player->EquipItem(eSrc, pDstItem, true);
254
255 _player->AutoUnequipOffhandIfNeed();
256 }

259void WorldSession::HandleDestroyItemOpcode(WorldPacket& recvData)
260{
261 //TC_LOG_DEBUG("network", "WORLD: CMSG_DESTROYITEM");
262 uint8 bag, slot, count, data1, data2, data3;
263
264 recvData >> bag >> slot >> count >> data1 >> data2 >> data3;
265 //TC_LOG_DEBUG("STORAGE: receive bag = %u, slot = %u, count = %u", bag, slot, count);
266
267 uint16 pos = (bag << 8) | slot;
268
269 // prevent drop unequipable items (in combat, for example) and non-empty bags
270 if (_player->IsEquipmentPos(pos) || _player->IsBagPos(pos))
271 {
272 InventoryResult msg = _player->CanUnequipItem(pos, false);
273 if (msg != EQUIP_ERR_OK)
274 {
275 _player->SendEquipError(msg, _player->GetItemByPos(pos), NULL);
276 return;
277 }
278 }
279
280 Item* pItem = _player->GetItemByPos(bag, slot);
281 if (!pItem)
282 {
283 _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
284 return;
285 }
286
287 if (pItem->GetTemplate()->Flags & ITEM_PROTO_FLAG_INDESTRUCTIBLE)
288 {
289 _player->SendEquipError(EQUIP_ERR_CANT_DROP_SOULBOUND, NULL, NULL);
290 return;
291 }
292
293 if (count)
294 {
295 uint32 i_count = count;
296 _player->DestroyItemCount(pItem, i_count, true);
297 }
298 else
299 _player->DestroyItem(bag, slot, true);
300}

303void WorldSession::HandleItemQuerySingleOpcode(WorldPacket& recvData)
304{
305 //TC_LOG_DEBUG("network", "WORLD: CMSG_ITEM_QUERY_SINGLE");
306 uint32 item;
307 recvData >> item;
308
309 TC_LOG_INFO("network", "STORAGE: Item Query = %u", item);
310
311 ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(item);
312 if (pProto)
313 {
314 std::string Name = pProto->Name1;
315 std::string Description = pProto->Description;
316
317 int loc_idx = GetSessionDbLocaleIndex();
318 if (loc_idx >= 0)
319 {
320 if (ItemLocale const* il = sObjectMgr->GetItemLocale(pProto->ItemId))
321 {
322 ObjectMgr::GetLocaleString(il->Name, loc_idx, Name);
323 ObjectMgr::GetLocaleString(il->Description, loc_idx, Description);
324 }
325 }
326 // guess size
327 WorldPacket data(SMSG_ITEM_QUERY_SINGLE_RESPONSE, 600);
328 data << pProto->ItemId;
329 data << pProto->Class;
330 data << pProto->SubClass;
331 data << pProto->SoundOverrideSubclass;
332 data << Name;
333 data << uint8(0x00); //pProto->Name2; // blizz not send name there, just uint8(0x00); <-- \0 = empty string = empty name...
334 data << uint8(0x00); //pProto->Name3; // blizz not send name there, just uint8(0x00);
335 data << uint8(0x00); //pProto->Name4; // blizz not send name there, just uint8(0x00);
336 data << pProto->DisplayInfoID;
337 data << pProto->Quality;
338 data << pProto->Flags;
339 data << pProto->Flags2;
340 data << pProto->BuyPrice;
341 data << pProto->SellPrice;
342 data << pProto->InventoryType;
343 data << pProto->AllowableClass;
344 data << pProto->AllowableRace;
345 data << pProto->ItemLevel;
346 data << pProto->RequiredLevel;
347 data << pProto->RequiredSkill;
348 data << pProto->RequiredSkillRank;
349 data << pProto->RequiredSpell;
350 data << pProto->RequiredHonorRank;
351 data << pProto->RequiredCityRank;
352 data << pProto->RequiredReputationFaction;
353 data << pProto->RequiredReputationRank;
354 data << int32(pProto->MaxCount);
355 data << int32(pProto->Stackable);
356 data << pProto->ContainerSlots;
357 data << pProto->StatsCount; // item stats count
358 for (uint32 i = 0; i < pProto->StatsCount; ++i)
359 {
360 data << pProto->ItemStat[i].ItemStatType;
361 data << pProto->ItemStat[i].ItemStatValue;
362 }
363 data << pProto->ScalingStatDistribution; // scaling stats distribution
364 data << pProto->ScalingStatValue; // some kind of flags used to determine stat values column
365 for (int i = 0; i < MAX_ITEM_PROTO_DAMAGES; ++i)
366 {
367 data << pProto->Damage[i].DamageMin;
368 data << pProto->Damage[i].DamageMax;
369 data << pProto->Damage[i].DamageType;
370 }
371
372 // resistances (7)
373 data << pProto->Armor;
374 data << pProto->HolyRes;
375 data << pProto->FireRes;
376 data << pProto->NatureRes;
377 data << pProto->FrostRes;
378 data << pProto->ShadowRes;
379 data << pProto->ArcaneRes;
380
381 data << pProto->Delay;
382 data << pProto->AmmoType;
383 data << pProto->RangedModRange;
384
385 for (int s = 0; s < MAX_ITEM_PROTO_SPELLS; ++s)
386 {
387 // send DBC data for cooldowns in same way as it used in Spell::SendSpellCooldown
388 // use `item_template` or if not set then only use spell cooldowns
389 SpellInfo const* spell = sSpellMgr->GetSpellInfo(pProto->Spells[s].SpellId);
390 if (spell)
391 {
392 bool db_data = pProto->Spells[s].SpellCooldown >= 0 || pProto->Spells[s].SpellCategoryCooldown >= 0;
393
394 data << pProto->Spells[s].SpellId;
395 data << pProto->Spells[s].SpellTrigger;
396 data << uint32(-abs(pProto->Spells[s].SpellCharges));
397
398 if (db_data)
399 {
400 data << uint32(pProto->Spells[s].SpellCooldown);
401 data << uint32(pProto->Spells[s].SpellCategory);
402 data << uint32(pProto->Spells[s].SpellCategoryCooldown);
403 }
404 else
405 {
406 data << uint32(spell->RecoveryTime);
407 data << uint32(spell->GetCategory());
408 data << uint32(spell->CategoryRecoveryTime);
409 }
410 }
411 else
412 {
413 data << uint32(0);
414 data << uint32(0);
415 data << uint32(0);
416 data << uint32(-1);
417 data << uint32(0);
418 data << uint32(-1);
419 }
420 }
421 data << pProto->Bonding;
422 data << Description;
423 data << pProto->PageText;
424 data << pProto->LanguageID;
425 data << pProto->PageMaterial;
426 data << pProto->StartQuest;
427 data << pProto->LockID;
428 data << int32(pProto->Material);
429 data << pProto->Sheath;
430 data << pProto->RandomProperty;
431 data << pProto->RandomSuffix;
432 data << pProto->Block;
433 data << pProto->ItemSet;
434 data << pProto->MaxDurability;
435 data << pProto->Area;
436 data << pProto->Map; // Added in 1.12.x & 2.0.1 client branch
437 data << pProto->BagFamily;
438 data << pProto->TotemCategory;
439 for (int s = 0; s < MAX_ITEM_PROTO_SOCKETS; ++s)
440 {
441 data << pProto->Socket[s].Color;

444 data << pProto->socketBonus;

452 }

454 {
455 TC_LOG_DEBUG("network", "WORLD: CMSG_ITEM_QUERY_SINGLE - NO item INFO! (ENTRY: %u)", item);
456 WorldPacket data(SMSG_ITEM_QUERY_SINGLE_RESPONSE, 4);
457 data << uint32(item | 0x80000000);
458 SendPacket(&data);
459 }

462void WorldSession::HandleReadItem(WorldPacket& recvData)
463{
464 //TC_LOG_DEBUG("network", "WORLD: CMSG_READ_ITEM");
465
466 uint8 bag, slot;
467 recvData >> bag >> slot;
468
469 //TC_LOG_INFO("network", "STORAGE: Read bag = %u, slot = %u", bag, slot);
470 Item* pItem = _player->GetItemByPos(bag, slot);
471
472 if (pItem && pItem->GetTemplate()->PageText)
473 {
474 WorldPacket data;
475
476 InventoryResult msg = _player->CanUseItem(pItem);
477 if (msg == EQUIP_ERR_OK)
478 {
479 data.Initialize (SMSG_READ_ITEM_OK, 8);
480 TC_LOG_INFO("network", "STORAGE: Item page sent");
481 }
482 else
483 {
484 data.Initialize(SMSG_READ_ITEM_FAILED, 8);
485 TC_LOG_INFO("network", "STORAGE: Unable to read item");
486 _player->SendEquipError(msg, pItem, NULL);
487 }
488 data << pItem->GetGUID();
489 SendPacket(&data);
490 }
491 else
492 _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
493}

495void WorldSession::HandleSellItemOpcode(WorldPacket& recvData)
496{
497 TC_LOG_DEBUG("network", "WORLD: Received CMSG_SELL_ITEM");
498 ObjectGuid vendorguid, itemguid;
499 uint32 count;
500
501 recvData >> vendorguid >> itemguid >> count;
502
503 if (!itemguid)
504 return;
505
506 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(vendorguid, UNIT_NPC_FLAG_VENDOR);
507 if (!creature)
508 {
509 TC_LOG_DEBUG("network", "WORLD: HandleSellItemOpcode - %s not found or you can not interact with him.", vendorguid.ToString().c_str());
510 _player->SendSellError(SELL_ERR_CANT_FIND_VENDOR, NULL, itemguid, 0);
511 return;
512 }
513
514 // remove fake death
515 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
516 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
517
518 Item* pItem = _player->GetItemByGuid(itemguid);
519 if (pItem)
520 {
521 // prevent sell not owner item
522 if (_player->GetGUID() != pItem->GetOwnerGUID())
523 {
524 _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, creature, itemguid, 0);
525 return;
526 }
527
528 // prevent sell non empty bag by drag-and-drop at vendor's item list
529 if (pItem->IsNotEmptyBag())
530 {
531 _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, creature, itemguid, 0);
532 return;
533 }
534
535 // prevent sell currently looted item
536 if (_player->GetLootGUID() == pItem->GetGUID())
537 {
538 _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, creature, itemguid, 0);
539 return;
540 }
541
542 // prevent selling item for sellprice when the item is still refundable
543 // this probably happens when right clicking a refundable item, the client sends both
544 // CMSG_SELL_ITEM and CMSG_REFUND_ITEM (unverified)
545 if (pItem->HasFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_REFUNDABLE))
546 return; // Therefore, no feedback to client
547
548 // special case at auto sell (sell all)
549 if (count == 0)
550 count = pItem->GetCount();
551 else
552 {
553 // prevent sell more items that exist in stack (possible only not from client)
554 if (count > pItem->GetCount())
555 {
556 _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, creature, itemguid, 0);
557 return;
558 }
559 }
560
561 ItemTemplate const* pProto = pItem->GetTemplate();
562 if (pProto)
563 {
564 if (pProto->SellPrice > 0)
565 {
566 if (count < pItem->GetCount()) // need split items
567 {
568 Item* pNewItem = pItem->CloneItem(count, _player);
569 if (!pNewItem)
570 {
571 TC_LOG_ERROR("network", "WORLD: HandleSellItemOpcode - could not create clone of item %u; count = %u", pItem->GetEntry(), count);
572 _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, creature, itemguid, 0);
573 return;
574 }
575
576 pItem->SetCount(pItem->GetCount() - count);
577 _player->ItemRemovedQuestCheck(pItem->GetEntry(), count);
578 if (_player->IsInWorld())
579 pItem->SendUpdateToPlayer(_player);
580 pItem->SetState(ITEM_CHANGED, _player);
581
582 _player->AddItemToBuyBackSlot(pNewItem);
583 if (_player->IsInWorld())
584 pNewItem->SendUpdateToPlayer(_player);
585 }
586 else
587 {
588 _player->ItemRemovedQuestCheck(pItem->GetEntry(), pItem->GetCount());
589 _player->RemoveItem(pItem->GetBagSlot(), pItem->GetSlot(), true);
590 pItem->RemoveFromUpdateQueueOf(_player);
591 _player->AddItemToBuyBackSlot(pItem);
592 }
593
594 uint32 money = pProto->SellPrice * count;
595 _player->ModifyMoney(money);
596 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_MONEY_FROM_VENDORS, money);
597 }
598 else
599 _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, creature, itemguid, 0);
600 return;
601 }
602 }
603 _player->SendSellError(SELL_ERR_CANT_FIND_ITEM, creature, itemguid, 0);
604 return;
605}

607void WorldSession::HandleBuybackItem(WorldPacket& recvData)
608{
609 TC_LOG_DEBUG("network", "WORLD: Received CMSG_BUYBACK_ITEM");
610 ObjectGuid vendorguid;
611 uint32 slot;
612
613 recvData >> vendorguid >> slot;
614
615 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(vendorguid, UNIT_NPC_FLAG_VENDOR);
616 if (!creature)
617 {
618 TC_LOG_DEBUG("network", "WORLD: HandleBuybackItem - Unit (%s) not found or you can not interact with him.", vendorguid.ToString().c_str());
619 _player->SendSellError(SELL_ERR_CANT_FIND_VENDOR, NULL, ObjectGuid::Empty, 0);
620 return;
621 }
622
623 // remove fake death
624 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
625 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
626
627 Item* pItem = _player->GetItemFromBuyBackSlot(slot);
628 if (pItem)
629 {
630 uint32 price = _player->GetUInt32Value(PLAYER_FIELD_BUYBACK_PRICE_1 + slot - BUYBACK_SLOT_START);
631 if (!_player->HasEnoughMoney(price))
632 {
633 _player->SendBuyError(BUY_ERR_NOT_ENOUGHT_MONEY, creature, pItem->GetEntry(), 0);
634 return;
635 }
636
637 ItemPosCountVec dest;
638 InventoryResult msg = _player->CanStoreItem(NULL_BAG, NULL_SLOT, dest, pItem, false);
639 if (msg == EQUIP_ERR_OK)
640 {
641 _player->ModifyMoney(-(int32)price);
642 _player->RemoveItemFromBuyBackSlot(slot, false);
643 _player->ItemAddedQuestCheck(pItem->GetEntry(), pItem->GetCount());
644 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_RECEIVE_EPIC_ITEM, pItem->GetEntry(), pItem->GetCount());
645 _player->StoreItem(dest, pItem, true);
646 }
647 else
648 _player->SendEquipError(msg, pItem, NULL);
649 return;
650 }
651 else

729void WorldSession::SendListInventory(ObjectGuid vendorGuid)
730{
731 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_LIST_INVENTORY");
732
733 Creature* vendor = GetPlayer()->GetNPCIfCanInteractWith(vendorGuid, UNIT_NPC_FLAG_VENDOR);
734 if (!vendor)
735 {
736 TC_LOG_DEBUG("network", "WORLD: SendListInventory - %s not found or you can not interact with him.", vendorGuid.ToString().c_str());
737 _player->SendSellError(SELL_ERR_CANT_FIND_VENDOR, NULL, ObjectGuid::Empty, 0);
738 return;
739 }
740
741 // remove fake death
742 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
743 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
744
745 // Stop the npc if moving
746 if (vendor->HasUnitState(UNIT_STATE_MOVING))
747 vendor->StopMoving();
748
749 VendorItemData const* items = vendor->GetVendorItems();
750 if (!items)
751 {
752 WorldPacket data(SMSG_LIST_INVENTORY, 8 + 1 + 1);
753 data << uint64(vendorGuid);
754 data << uint8(0); // count == 0, next will be error code
755 data << uint8(0); // "Vendor has no inventory"
756 SendPacket(&data);
757 return;
758 }
759
760 uint8 itemCount = items->GetItemCount();
761 uint8 count = 0;
762
763 WorldPacket data(SMSG_LIST_INVENTORY, 8 + 1 + itemCount * 8 * 4);
764 data << uint64(vendorGuid);
765
766 size_t countPos = data.wpos();
767 data << uint8(count);
768
769 float discountMod = _player->GetReputationPriceDiscount(vendor);
770
771 for (uint8 slot = 0; slot < itemCount; ++slot)
772 {
773 if (VendorItem const* item = items->GetItem(slot))
774 {
775 if (ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(item->item))
776 {
777 if (!(itemTemplate->AllowableClass & _player->getClassMask()) && itemTemplate->Bonding == BIND_WHEN_PICKED_UP && !_player->IsGameMaster())
778 continue;
779 // Only display items in vendor lists for the team the
780 // player is on. If GM on, display all items.
781 if (!_player->IsGameMaster() && ((itemTemplate->Flags2 & ITEM_FLAGS_EXTRA_HORDE_ONLY && _player->GetTeam() == ALLIANCE) || (itemTemplate->Flags2 == ITEM_FLAGS_EXTRA_ALLIANCE_ONLY && _player->GetTeam() == HORDE)))
782 continue;
783
784 // Items sold out are not displayed in list
785 uint32 leftInStock = !item->maxcount ? 0xFFFFFFFF : vendor->GetVendorItemCurrentCount(item);
786 if (!_player->IsGameMaster() && !leftInStock)
787 continue;
788
789 if (!sConditionMgr->IsObjectMeetingVendorItemConditions(vendor->GetEntry(), item->item, _player, vendor))
790 {
791 TC_LOG_DEBUG("condition", "SendListInventory: conditions not met for creature entry %u item %u", vendor->GetEntry(), item->item);
792 continue;
793 }
794
795 // reputation discount
796 int32 price = item->IsGoldRequired(itemTemplate) ? uint32(floor(itemTemplate->BuyPrice * discountMod)) : 0;
797
798 data << uint32(slot + 1); // client expects counting to start at 1
799 data << uint32(item->item);
800 data << uint32(itemTemplate->DisplayInfoID);
801 data << int32(leftInStock);
802 data << uint32(price);
803 data << uint32(itemTemplate->MaxDurability);
804 data << uint32(itemTemplate->BuyCount);
805 data << uint32(item->ExtendedCost);
806
807 if (++count >= MAX_VENDOR_ITEMS)
808 break;
809 }
810 }
811 }
812
813 if (count == 0)
814 {
815 data << uint8(0);
816 SendPacket(&data);
817 return;
818 }
819
820 data.put<uint8>(countPos, count);
821 SendPacket(&data);
822}

1080void WorldSession::HandleWrapItemOpcode(WorldPacket& recvData)
1081{

1083
1084 uint8 gift_bag, gift_slot, item_bag, item_slot;
1085
1086 recvData >> gift_bag >> gift_slot; // paper
1087 recvData >> item_bag >> item_slot; // item

1089 TC_LOG_DEBUG("network", "WRAP: receive gift_bag = %u, gift_slot = %u, item_bag = %u, item_slot = %u", gift_bag, gift_slot, item_bag, item_slot);
1090
1091 Item* gift = _player->GetItemByPos(gift_bag, gift_slot);
1092 if (!gift)
1093 {
1094 _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, gift, NULL);

1098 if (!(gift->GetTemplate()->Flags & ITEM_PROTO_FLAG_WRAPPER)) // cheating: non-wrapper wrapper

1102 }
1103
1104 Item* item = _player->GetItemByPos(item_bag, item_slot);
1105
1106 if (!item)
1107 {
1108 _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, item, NULL);

1111
1112 if (item == gift) // not possable with pacjket from real client
1113 {
1114 _player->SendEquipError(EQUIP_ERR_WRAPPED_CANT_BE_WRAPPED, item, NULL);
1115 return;
1116 }

1119 {

1121 return;

1126 _player->SendEquipError(EQUIP_ERR_WRAPPED_CANT_BE_WRAPPED, item, NULL);

1138 _player->SendEquipError(EQUIP_ERR_BOUND_CANT_BE_WRAPPED, item, NULL);

1158 stmt->setUInt32(0, item->GetOwnerGUID().GetCounter());

1160 stmt->setUInt32(2, item->GetEntry());

1165

1168 case 5042: item->SetEntry(5043); break;
1169 case 5048: item->SetEntry(5044); break;
1170 case 17303: item->SetEntry(17302); break;
1171 case 17304: item->SetEntry(17305); break;
1172 case 17307: item->SetEntry(17308); break;
1173 case 21830: item->SetEntry(21831); break;
1174 }
1175 item->SetGuidValue(ITEM_FIELD_GIFTCREATOR, _player->GetGUID());
1176 item->SetUInt32Value(ITEM_FIELD_FLAGS, ITEM_FLAG_WRAPPED);
1177 item->SetState(ITEM_CHANGED, _player);

1182 item->RemoveFromUpdateQueueOf(_player);
1183 item->SaveToDB(trans); // item gave inventory record unchanged and can be save standalone
1184 }
1185 CharacterDatabase.CommitTransaction(trans);
1186
1187 uint32 count = 1;
1188 _player->DestroyItemCount(gift, count, true);
1189}

1310 if (ItemLimitCategoryEntry const* limitEntry = sItemLimitCategoryStore.LookupEntry(iGemProto->ItemLimitCategory))
1311 {
1312 // NOTE: limitEntry->mode is not checked because if item has limit then it is applied in equip case
1313 for (int j = 0; j < MAX_GEM_SOCKETS; ++j)
1314 {
1315 if (Gems[j])
1316 {
1317 // new gem
1318 if (iGemProto->ItemLimitCategory == Gems[j]->GetTemplate()->ItemLimitCategory)
1319 ++limit_newcount;
1320 }
1321 else if (OldEnchants[j])
1322 {
1323 // existing gem
1324 if (SpellItemEnchantmentEntry const* enchantEntry = sSpellItemEnchantmentStore.LookupEntry(OldEnchants[j]))
1325 if (ItemTemplate const* jProto = sObjectMgr->GetItemTemplate(enchantEntry->GemID))
1326 if (iGemProto->ItemLimitCategory == jProto->ItemLimitCategory)
1327 ++limit_newcount;
1328 }
1329 }
1330
1331 if (limit_newcount > 0 && uint32(limit_newcount) > limitEntry->maxCount)
1332 {
1333 _player->SendEquipError(EQUIP_ERR_ITEM_UNIQUE_EQUIPPABLE_SOCKETED, itemTarget, NULL);
1334 return;
1335 }
1336 }
1337 }
1338
1339 // for equipped item check all equipment for duplicate equipped gems
1340 if (itemTarget->IsEquipped())
1341 {
1342 if (InventoryResult res = _player->CanEquipUniqueItem(Gems[i], slot, std::max(limit_newcount, 0)))
1343 {
1344 _player->SendEquipError(res, itemTarget, NULL);
1345 return;
1346 }
1347 }
1348 }
1349
1350 bool SocketBonusActivated = itemTarget->GemsFitSockets(); //save state of socketbonus
1351 _player->ToggleMetaGemsActive(slot, false); //turn off all metagems (except for the target item)
1352
1353 //if a meta gem is being equipped, all information has to be written to the item before testing if the conditions for the gem are met
1354
1355 //remove ALL enchants
1356 for (uint32 enchant_slot = SOCK_ENCHANTMENT_SLOT; enchant_slot < SOCK_ENCHANTMENT_SLOT + MAX_GEM_SOCKETS; ++enchant_slot)
1357 _player->ApplyEnchantment(itemTarget, EnchantmentSlot(enchant_slot), false);
1358
1359 for (int i = 0; i < MAX_GEM_SOCKETS; ++i)
1360 {
1361 if (GemEnchants[i])
1362 {
1363 itemTarget->SetEnchantment(EnchantmentSlot(SOCK_ENCHANTMENT_SLOT+i), GemEnchants[i], 0, 0, _player->GetGUID());
1364 if (Item* guidItem = _player->GetItemByGuid(gem_guids[i]))
1365 _player->DestroyItem(guidItem->GetBagSlot(), guidItem->GetSlot(), true);
1366 }
1367 }
1368
1369 for (uint32 enchant_slot = SOCK_ENCHANTMENT_SLOT; enchant_slot < SOCK_ENCHANTMENT_SLOT+MAX_GEM_SOCKETS; ++enchant_slot)
1370 _player->ApplyEnchantment(itemTarget, EnchantmentSlot(enchant_slot), true);
1371
1372 bool SocketBonusToBeActivated = itemTarget->GemsFitSockets();//current socketbonus state
1373 if (SocketBonusActivated ^ SocketBonusToBeActivated) //if there was a change...
1374 {
1375 _player->ApplyEnchantment(itemTarget, BONUS_ENCHANTMENT_SLOT, false);
1376 itemTarget->SetEnchantment(BONUS_ENCHANTMENT_SLOT, (SocketBonusToBeActivated ? itemTarget->GetTemplate()->socketBonus : 0), 0, 0, _player->GetGUID());
1377 _player->ApplyEnchantment(itemTarget, BONUS_ENCHANTMENT_SLOT, true);
1378 //it is not displayed, client has an inbuilt system to determine if the bonus is activated
1379 }
1380
1381 _player->ToggleMetaGemsActive(slot, true); //turn on all metagems (except for target item)
1382
1383 _player->RemoveTradeableItem(itemTarget);
1384 itemTarget->ClearSoulboundTradeable(_player); // clear tradeable flag
1385
1386 itemTarget->SendUpdateSockets();
1387}

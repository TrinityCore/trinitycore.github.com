213void WorldSession::HandleCharEnum(PreparedQueryResult result)
214{
215 WorldPacket data(SMSG_CHAR_ENUM, 100); // we guess size
216
217 uint8 num = 0;
218
219 data << num;
220
221 _legitCharacters.clear();
222 if (result)
223 {
224 do
225 {
226 ObjectGuid guid(HighGuid::Player, (*result)[0].GetUInt32());
227 TC_LOG_INFO("network", "Loading %s from account %u.", guid.ToString().c_str(), GetAccountId());
228 if (Player::BuildEnumData(result, &data))
229 {
230 // Do not allow banned characters to log in
231 if (!(*result)[20].GetUInt32())
232 _legitCharacters.insert(guid);
233
234 if (!sWorld->HasCharacterInfo(guid)) // This can happen if characters are inserted into the database manually. Core hasn't loaded name data yet.
235 sWorld->AddCharacterInfo(guid, GetAccountId(), (*result)[1].GetString(), (*result)[4].GetUInt8(), (*result)[2].GetUInt8(), (*result)[3].GetUInt8(), (*result)[7].GetUInt8());
236 ++num;
237 }
238 }
239 while (result->NextRow());
240 }
241
242 data.put<uint8>(0, num);
243
244 SendPacket(&data);
245}

266void WorldSession::HandleCharCreateOpcode(WorldPacket& recvData)
267{
268 CharacterCreateInfo createInfo;
269
270 recvData >> createInfo.Name
271 >> createInfo.Race
272 >> createInfo.Class
273 >> createInfo.Gender
274 >> createInfo.Skin
275 >> createInfo.Face
276 >> createInfo.HairStyle
277 >> createInfo.HairColor
278 >> createInfo.FacialHair
279 >> createInfo.OutfitId;
280
281 if (!HasPermission(rbac::RBAC_PERM_SKIP_CHECK_CHARACTER_CREATION_TEAMMASK))
282 {
283 if (uint32 mask = sWorld->getIntConfig(CONFIG_CHARACTER_CREATING_DISABLED))
284 {
285 bool disabled = false;
286
287 switch (Player::TeamForRace(createInfo.Race))
288 {
289 case ALLIANCE:
290 disabled = (mask & (1 << 0)) != 0;
291 break;
292 case HORDE:
293 disabled = (mask & (1 << 1)) != 0;
294 break;
295 }
296
297 if (disabled)
298 {
299 SendCharCreate(CHAR_CREATE_DISABLED);
300 return;
301 }
302 }
303 }
304
305 ChrClassesEntry const* classEntry = sChrClassesStore.LookupEntry(createInfo.Class);
306 if (!classEntry)
307 {
308 TC_LOG_ERROR("network", "Class (%u) not found in DBC while creating new char for account (ID: %u): wrong DBC files or cheater?", createInfo.Class, GetAccountId());
309 SendCharCreate(CHAR_CREATE_FAILED);
310 return;
311 }
312
313 ChrRacesEntry const* raceEntry = sChrRacesStore.LookupEntry(createInfo.Race);
314 if (!raceEntry)
315 {
316 TC_LOG_ERROR("network", "Race (%u) not found in DBC while creating new char for account (ID: %u): wrong DBC files or cheater?", createInfo.Race, GetAccountId());
317 SendCharCreate(CHAR_CREATE_FAILED);
318 return;
319 }
320
321 // prevent character creating Expansion race without Expansion account
322 if (raceEntry->expansion > Expansion())
323 {
324 TC_LOG_ERROR("network", "Expansion %u account:[%d] tried to Create character with expansion %u race (%u)", Expansion(), GetAccountId(), raceEntry->expansion, createInfo.Race);
325 SendCharCreate(CHAR_CREATE_EXPANSION);
326 return;
327 }
328
329 // prevent character creating Expansion class without Expansion account
330 if (classEntry->expansion > Expansion())
331 {
332 TC_LOG_ERROR("network", "Expansion %u account:[%d] tried to Create character with expansion %u class (%u)", Expansion(), GetAccountId(), classEntry->expansion, createInfo.Class);
333 SendCharCreate(CHAR_CREATE_EXPANSION_CLASS);
334 return;
335 }
336
337 if (!HasPermission(rbac::RBAC_PERM_SKIP_CHECK_CHARACTER_CREATION_RACEMASK))
338 {
339 uint32 raceMaskDisabled = sWorld->getIntConfig(CONFIG_CHARACTER_CREATING_DISABLED_RACEMASK);
340 if ((1 << (createInfo.Race - 1)) & raceMaskDisabled)
341 {
342 SendCharCreate(CHAR_CREATE_DISABLED);
343 return;
344 }
345 }
346
347 if (!HasPermission(rbac::RBAC_PERM_SKIP_CHECK_CHARACTER_CREATION_CLASSMASK))
348 {
349 uint32 classMaskDisabled = sWorld->getIntConfig(CONFIG_CHARACTER_CREATING_DISABLED_CLASSMASK);
350 if ((1 << (createInfo.Class - 1)) & classMaskDisabled)
351 {
352 SendCharCreate(CHAR_CREATE_DISABLED);
353 return;
354 }
355 }
356
357 // prevent character creating with invalid name
358 if (!normalizePlayerName(createInfo.Name))
359 {
360 TC_LOG_ERROR("network", "Account:[%d] but tried to Create character with empty [name] ", GetAccountId());
361 SendCharCreate(CHAR_NAME_NO_NAME);
362 return;
363 }
364
365 // check name limitations
366 ResponseCodes res = ObjectMgr::CheckPlayerName(createInfo.Name, GetSessionDbcLocale(), true);
367 if (res != CHAR_NAME_SUCCESS)
368 {
369 SendCharCreate(res);
370 return;
371 }
372
373 if (!HasPermission(rbac::RBAC_PERM_SKIP_CHECK_CHARACTER_CREATION_RESERVEDNAME) && sObjectMgr->IsReservedName(createInfo.Name))
374 {
375 SendCharCreate(CHAR_NAME_RESERVED);
376 return;
377 }
378
379 if (createInfo.Class == CLASS_DEATH_KNIGHT && !HasPermission(rbac::RBAC_PERM_SKIP_CHECK_CHARACTER_CREATION_HEROIC_CHARACTER))
380 {
381 // speedup check for heroic class disabled case
382 uint32 heroic_free_slots = sWorld->getIntConfig(CONFIG_HEROIC_CHARACTERS_PER_REALM);
383 if (heroic_free_slots == 0)
384 {
385 SendCharCreate(CHAR_CREATE_UNIQUE_CLASS_LIMIT);
386 return;
387 }
388
389 // speedup check for heroic class disabled case
390 uint32 req_level_for_heroic = sWorld->getIntConfig(CONFIG_CHARACTER_CREATING_MIN_LEVEL_FOR_HEROIC_CHARACTER);
391 if (req_level_for_heroic > sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
392 {
393 SendCharCreate(CHAR_CREATE_LEVEL_REQUIREMENT);
394 return;
395 }
396 }
397
398 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHECK_NAME);
399 stmt->setString(0, createInfo.Name);
400
401 delete _charCreateCallback.GetParam(); // Delete existing if any, to make the callback chain reset to stage 0
402 _charCreateCallback.SetParam(new CharacterCreateInfo(std::move(createInfo)));
403 _charCreateCallback.SetFutureResult(CharacterDatabase.AsyncQuery(stmt));
404}

406void WorldSession::HandleCharCreateCallback(PreparedQueryResult result, CharacterCreateInfo* createInfo)
407{
408 /** This is a series of callbacks executed consecutively as a result from the database becomes available.
409 This is much more efficient than synchronous requests on packet handler, and much less DoS prone.
410 It also prevents data syncrhonisation errors.
411 */
412 switch (_charCreateCallback.GetStage())
413 {
414 case 0:
415 {
416 if (result)
417 {
418 SendCharCreate(CHAR_CREATE_NAME_IN_USE);
419 delete createInfo;
420 _charCreateCallback.Reset();
421 return;
422 }
423
424 ASSERT(_charCreateCallback.GetParam() == createInfo);
425
426 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_SUM_REALM_CHARACTERS);
427 stmt->setUInt32(0, GetAccountId());
428
429 _charCreateCallback.FreeResult();
430 _charCreateCallback.SetFutureResult(LoginDatabase.AsyncQuery(stmt));
431 _charCreateCallback.NextStage();
432 break;
433 }
434 case 1:
435 {
436 uint64 acctCharCount = 0;
437 if (result)
438 {
439 Field* fields = result->Fetch();
440 acctCharCount = uint64(fields[0].GetDouble());
441 }
442
443 if (acctCharCount >= sWorld->getIntConfig(CONFIG_CHARACTERS_PER_ACCOUNT))
444 {
445 SendCharCreate(CHAR_CREATE_ACCOUNT_LIMIT);
446 delete createInfo;
447 _charCreateCallback.Reset();
448 return;
449 }
450
451 ASSERT(_charCreateCallback.GetParam() == createInfo);
452
453 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_SUM_CHARS);
454 stmt->setUInt32(0, GetAccountId());
455
456 _charCreateCallback.FreeResult();
457 _charCreateCallback.SetFutureResult(CharacterDatabase.AsyncQuery(stmt));
458 _charCreateCallback.NextStage();
459 break;
460 }
461 case 2:
462 {
463 if (result)
464 {
465 Field* fields = result->Fetch();
466 createInfo->CharCount = uint8(fields[0].GetUInt64()); // SQL's COUNT() returns uint64 but it will always be less than uint8.Max
467
468 if (createInfo->CharCount >= sWorld->getIntConfig(CONFIG_CHARACTERS_PER_REALM))
469 {
470 SendCharCreate(CHAR_CREATE_SERVER_LIMIT);
471 delete createInfo;
472 _charCreateCallback.Reset();
473 return;
474 }
475 }
476
477 bool allowTwoSideAccounts = !sWorld->IsPvPRealm() || HasPermission(rbac::RBAC_PERM_TWO_SIDE_CHARACTER_CREATION);
478 uint32 skipCinematics = sWorld->getIntConfig(CONFIG_SKIP_CINEMATICS);
479
480 _charCreateCallback.FreeResult();
481
482 if (!allowTwoSideAccounts || skipCinematics == 1 || createInfo->Class == CLASS_DEATH_KNIGHT)
483 {
484 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHAR_CREATE_INFO);
485 stmt->setUInt32(0, GetAccountId());
486 stmt->setUInt32(1, (skipCinematics == 1 || createInfo->Class == CLASS_DEATH_KNIGHT) ? 10 : 1);
487 _charCreateCallback.SetFutureResult(CharacterDatabase.AsyncQuery(stmt));
488 _charCreateCallback.NextStage();
489 return;
490 }
491
492 _charCreateCallback.NextStage();
493 HandleCharCreateCallback(PreparedQueryResult(NULL), createInfo); // Will jump to case 3
494 break;
495 }
496 case 3:
497 {
498 bool haveSameRace = false;
499 uint32 heroicReqLevel = sWorld->getIntConfig(CONFIG_CHARACTER_CREATING_MIN_LEVEL_FOR_HEROIC_CHARACTER);
500 bool hasHeroicReqLevel = (heroicReqLevel == 0);
501 bool allowTwoSideAccounts = !sWorld->IsPvPRealm() || HasPermission(rbac::RBAC_PERM_TWO_SIDE_CHARACTER_CREATION);
502 uint32 skipCinematics = sWorld->getIntConfig(CONFIG_SKIP_CINEMATICS);
503 bool checkHeroicReqs = createInfo->Class == CLASS_DEATH_KNIGHT && !HasPermission(rbac::RBAC_PERM_SKIP_CHECK_CHARACTER_CREATION_HEROIC_CHARACTER);
504
505 if (result)
506 {
507 uint32 team = Player::TeamForRace(createInfo->Race);
508 uint32 freeHeroicSlots = sWorld->getIntConfig(CONFIG_HEROIC_CHARACTERS_PER_REALM);
509
510 Field* field = result->Fetch();
511 uint8 accRace = field[1].GetUInt8();
512
513 if (checkHeroicReqs)
514 {
515 uint8 accClass = field[2].GetUInt8();
516 if (accClass == CLASS_DEATH_KNIGHT)
517 {
518 if (freeHeroicSlots > 0)
519 --freeHeroicSlots;
520
521 if (freeHeroicSlots == 0)
522 {
523 SendCharCreate(CHAR_CREATE_UNIQUE_CLASS_LIMIT);
524 delete createInfo;
525 _charCreateCallback.Reset();
526 return;
527 }
528 }
529
530 if (!hasHeroicReqLevel)
531 {
532 uint8 accLevel = field[0].GetUInt8();
533 if (accLevel >= heroicReqLevel)
534 hasHeroicReqLevel = true;
535 }
536 }
537
538 // need to check team only for first character
539 /// @todo what to if account already has characters of both races?
540 if (!allowTwoSideAccounts)
541 {
542 uint32 accTeam = 0;
543 if (accRace > 0)
544 accTeam = Player::TeamForRace(accRace);
545
546 if (accTeam != team)
547 {
548 SendCharCreate(CHAR_CREATE_PVP_TEAMS_VIOLATION);
549 delete createInfo;
550 _charCreateCallback.Reset();
551 return;
552 }
553 }
554
555 // search same race for cinematic or same class if need
556 /// @todo check if cinematic already shown? (already logged in?; cinematic field)
557 while ((skipCinematics == 1 && !haveSameRace) || createInfo->Class == CLASS_DEATH_KNIGHT)
558 {
559 if (!result->NextRow())
560 break;
561
562 field = result->Fetch();
563 accRace = field[1].GetUInt8();
564
565 if (!haveSameRace)
566 haveSameRace = createInfo->Race == accRace;
567
568 if (checkHeroicReqs)
569 {
570 uint8 acc_class = field[2].GetUInt8();
571 if (acc_class == CLASS_DEATH_KNIGHT)
572 {
573 if (freeHeroicSlots > 0)
574 --freeHeroicSlots;
575
576 if (freeHeroicSlots == 0)
577 {
578 SendCharCreate(CHAR_CREATE_UNIQUE_CLASS_LIMIT);
579 delete createInfo;
580 _charCreateCallback.Reset();
581 return;
582 }
583 }
584
585 if (!hasHeroicReqLevel)
586 {
587 uint8 acc_level = field[0].GetUInt8();
588 if (acc_level >= heroicReqLevel)
589 hasHeroicReqLevel = true;
590 }
591 }
592 }
593 }
594
595 if (checkHeroicReqs && !hasHeroicReqLevel)
596 {
597 SendCharCreate(CHAR_CREATE_LEVEL_REQUIREMENT);
598 delete createInfo;
599 _charCreateCallback.Reset();
600 return;
601 }
602
603 Player newChar(this);
604 newChar.GetMotionMaster()->Initialize();
605 if (!newChar.Create(sObjectMgr->GetGenerator<HighGuid::Player>().Generate(), createInfo))
606
607 {
608 // Player not create (race/class/etc problem?)
609 newChar.CleanupsBeforeDelete();
610
611 SendCharCreate(CHAR_CREATE_ERROR);
612 delete createInfo;
613 _charCreateCallback.Reset();
614 return;
615 }
616
617 if ((haveSameRace && skipCinematics == 1) || skipCinematics == 2)
618 newChar.setCinematic(1); // not show intro
619
620 newChar.SetAtLoginFlag(AT_LOGIN_FIRST); // First login
621
622 // Player created, save it now
623 newChar.SaveToDB(true);
624 createInfo->CharCount += 1;
625
626 SQLTransaction trans = LoginDatabase.BeginTransaction();
627
628 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_DEL_REALM_CHARACTERS_BY_REALM);
629 stmt->setUInt32(0, GetAccountId());
630 stmt->setUInt32(1, realm.Id.Realm);

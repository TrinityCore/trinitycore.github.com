25void WorldSession::HandleDuelAcceptedOpcode(WorldPacket& recvPacket)
26{
27 ObjectGuid guid;
28 Player* player;
29 Player* plTarget;
30
31 recvPacket >> guid;
32
33 if (!GetPlayer()->duel) // ignore accept from duel-sender
34 return;
35
36 player = GetPlayer();
37 plTarget = player->duel->opponent;
38
39 if (player == player->duel->initiator || !plTarget || player == plTarget || player->duel->startTime != 0 || plTarget->duel->startTime != 0)
40 return;
41
42 //TC_LOG_DEBUG("network", "WORLD: Received CMSG_DUEL_ACCEPTED");
43 TC_LOG_DEBUG("network", "Player 1 is: %u (%s)", player->GetGUID().GetCounter(), player->GetName().c_str());
44 TC_LOG_DEBUG("network", "Player 2 is: %u (%s)", plTarget->GetGUID().GetCounter(), plTarget->GetName().c_str());
45
46 time_t now = time(NULL);
47 player->duel->startTimer = now;
48 plTarget->duel->startTimer = now;
49
50 player->SendDuelCountdown(3000);
51 plTarget->SendDuelCountdown(3000);
52}

54void WorldSession::HandleDuelCancelledOpcode(WorldPacket& recvPacket)
55{
56 TC_LOG_DEBUG("network", "WORLD: Received CMSG_DUEL_CANCELLED");
57 ObjectGuid guid;
58 recvPacket >> guid;
59
60 // no duel requested
61 if (!GetPlayer()->duel)
62 return;
63
64 // player surrendered in a duel using /forfeit
65 if (GetPlayer()->duel->startTime != 0)
66 {
67 GetPlayer()->CombatStopWithPets(true);
68 if (GetPlayer()->duel->opponent)
69 GetPlayer()->duel->opponent->CombatStopWithPets(true);
70
71 GetPlayer()->CastSpell(GetPlayer(), 7267, true); // beg
72 GetPlayer()->DuelComplete(DUEL_WON);
73 return;
74 }
75
76 GetPlayer()->DuelComplete(DUEL_INTERRUPTED);
77}

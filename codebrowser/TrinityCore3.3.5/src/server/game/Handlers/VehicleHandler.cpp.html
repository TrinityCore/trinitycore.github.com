25void WorldSession::HandleDismissControlledVehicle(WorldPacket &recvData)
26{
27 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_DISMISS_CONTROLLED_VEHICLE");
28
29 ObjectGuid vehicleGUID = _player->GetCharmGUID();
30
31 if (!vehicleGUID) // something wrong here...
32 {
33 recvData.rfinish(); // prevent warnings spam
34 return;
35 }
36
37 ObjectGuid guid;
38
39 recvData >> guid.ReadAsPacked();
40
41 MovementInfo mi;
42 mi.guid = guid;
43 ReadMovementInfo(recvData, &mi);
44
45 _player->m_movementInfo = mi;
46
47 _player->ExitVehicle();
48}

50void WorldSession::HandleChangeSeatsOnControlledVehicle(WorldPacket &recvData)
51{
52 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_CHANGE_SEATS_ON_CONTROLLED_VEHICLE");
53
54 Unit* vehicle_base = GetPlayer()->GetVehicleBase();
55 if (!vehicle_base)
56 {
57 recvData.rfinish(); // prevent warnings spam
58 return;
59 }
60
61 VehicleSeatEntry const* seat = GetPlayer()->GetVehicle()->GetSeatForPassenger(GetPlayer());
62 if (!seat->CanSwitchFromSeat())
63 {
64 recvData.rfinish(); // prevent warnings spam
65 TC_LOG_ERROR("network", "HandleChangeSeatsOnControlledVehicle, Opcode: %u, Player %u tried to switch seats but current seatflags %u don't permit that.",
66 recvData.GetOpcode(), GetPlayer()->GetGUID().GetCounter(), seat->m_flags);
67 return;
68 }
69
70 switch (recvData.GetOpcode())
71 {
72 case CMSG_REQUEST_VEHICLE_PREV_SEAT:
73 GetPlayer()->ChangeSeat(-1, false);
74 break;
75 case CMSG_REQUEST_VEHICLE_NEXT_SEAT:
76 GetPlayer()->ChangeSeat(-1, true);
77 break;
78 case CMSG_CHANGE_SEATS_ON_CONTROLLED_VEHICLE:
79 {
80 ObjectGuid guid; // current vehicle guid
81 recvData >> guid.ReadAsPacked();
82
83 MovementInfo movementInfo;
84 ReadMovementInfo(recvData, &movementInfo);
85 vehicle_base->m_movementInfo = movementInfo;
86
87 ObjectGuid accessory; // accessory guid
88 recvData >> accessory.ReadAsPacked();
89
90 int8 seatId;
91 recvData >> seatId;
92
93 if (vehicle_base->GetGUID() != guid)
94 return;
95
96 if (!accessory)
97 GetPlayer()->ChangeSeat(-1, seatId > 0); // prev/next
98 else if (Unit* vehUnit = ObjectAccessor::GetUnit(*GetPlayer(), accessory))
99 {
100 if (Vehicle* vehicle = vehUnit->GetVehicleKit())
101 if (vehicle->HasEmptySeat(seatId))
102 vehUnit->HandleSpellClick(GetPlayer(), seatId);
103 }
104 break;
105 }
106 case CMSG_REQUEST_VEHICLE_SWITCH_SEAT:
107 {
108 ObjectGuid guid; // current vehicle guid
109 recvData >> guid.ReadAsPacked();
110
111 int8 seatId;
112 recvData >> seatId;
113
114 if (vehicle_base->GetGUID() == guid)
115 GetPlayer()->ChangeSeat(seatId);
116 else if (Unit* vehUnit = ObjectAccessor::GetUnit(*GetPlayer(), guid))
117 if (Vehicle* vehicle = vehUnit->GetVehicleKit())
118 if (vehicle->HasEmptySeat(seatId))
119 vehUnit->HandleSpellClick(GetPlayer(), seatId);
120 break;
121 }
122 default:
123 break;
124 }
125}

127void WorldSession::HandleEnterPlayerVehicle(WorldPacket &data)
128{
129 // Read guid
130 ObjectGuid guid;
131 data >> guid;
132
133 if (Player* player = ObjectAccessor::FindPlayer(guid))
134 {
135 if (!player->GetVehicleKit())
136 return;
137 if (!player->IsInRaidWith(_player))
138 return;
139 if (!player->IsWithinDistInMap(_player, INTERACTION_DISTANCE))
140 return;
141
142 _player->EnterVehicle(player);
143 }
144}

146void WorldSession::HandleEjectPassenger(WorldPacket &data)
147{
148 Vehicle* vehicle = _player->GetVehicleKit();
149 if (!vehicle)
150 {
151 data.rfinish(); // prevent warnings spam
152 TC_LOG_ERROR("network", "HandleEjectPassenger: %s is not in a vehicle!", GetPlayer()->GetGUID().ToString().c_str());
153 return;
154 }
155
156 ObjectGuid guid;
157 data >> guid;
158
159 if (guid.IsUnit())
160 {
161 Unit* unit = ObjectAccessor::GetUnit(*_player, guid);
162 if (!unit) // creatures can be ejected too from player mounts
163 {
164 TC_LOG_ERROR("network", "%s tried to eject %s from vehicle, but the latter was not found in world!", GetPlayer()->GetGUID().ToString().c_str(), guid.ToString().c_str());
165 return;
166 }
167
168 if (!unit->IsOnVehicle(vehicle->GetBase()))
169 {
170 TC_LOG_ERROR("network", "%s tried to eject %s, but they are not in the same vehicle", GetPlayer()->GetGUID().ToString().c_str(), guid.ToString().c_str());
171 return;
172 }
173
174 VehicleSeatEntry const* seat = vehicle->GetSeatForPassenger(unit);
175 ASSERT(seat);
176 if (seat->IsEjectable())
177 unit->ExitVehicle();
178 else
179 TC_LOG_ERROR("network", "Player %u attempted to eject %s from non-ejectable seat.", GetPlayer()->GetGUID().GetCounter(), guid.ToString().c_str());
180 }
181 else
182 TC_LOG_ERROR("network", "HandleEjectPassenger: %s tried to eject invalid %s ", GetPlayer()->GetGUID().ToString().c_str(), guid.ToString().c_str());
183}

185void WorldSession::HandleRequestVehicleExit(WorldPacket& /*recvData*/)
186{
187 TC_LOG_DEBUG("network", "WORLD: Recvd CMSG_REQUEST_VEHICLE_EXIT");
188
189 if (Vehicle* vehicle = GetPlayer()->GetVehicle())
190 {
191 if (VehicleSeatEntry const* seat = vehicle->GetSeatForPassenger(GetPlayer()))
192 {
193 if (seat->CanEnterOrExit())
194 GetPlayer()->ExitVehicle();
195 else
196 TC_LOG_ERROR("network", "Player %u tried to exit vehicle, but seatflags %u (ID: %u) don't permit that.",
197 GetPlayer()->GetGUID().GetCounter(), seat->m_ID, seat->m_flags);
198 }
199 }

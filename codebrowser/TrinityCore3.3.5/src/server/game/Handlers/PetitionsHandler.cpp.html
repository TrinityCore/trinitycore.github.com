43void WorldSession::HandlePetitionBuyOpcode(WorldPacket& recvData)
44{
45 TC_LOG_DEBUG("network", "Received opcode CMSG_PETITION_BUY");
46
47 ObjectGuid guidNPC;
48 uint32 clientIndex; // 1 for guild and arenaslot+1 for arenas in client
49 std::string name;
50
51 recvData >> guidNPC; // NPC GUID
52 recvData.read_skip<uint32>(); // 0
53 recvData.read_skip<uint64>(); // 0
54 recvData >> name; // name
55 recvData.read_skip<std::string>(); // some string
56 recvData.read_skip<uint32>(); // 0
57 recvData.read_skip<uint32>(); // 0
58 recvData.read_skip<uint32>(); // 0
59 recvData.read_skip<uint32>(); // 0
60 recvData.read_skip<uint32>(); // 0
61 recvData.read_skip<uint32>(); // 0
62 recvData.read_skip<uint32>(); // 0
63 recvData.read_skip<uint16>(); // 0
64 recvData.read_skip<uint32>(); // 0
65 recvData.read_skip<uint32>(); // 0
66 recvData.read_skip<uint32>(); // 0
67
68 for (int i = 0; i < 10; ++i)
69 recvData.read_skip<std::string>();
70
71 recvData >> clientIndex; // index
72 recvData.read_skip<uint32>(); // 0
73
74 TC_LOG_DEBUG("network", "Petitioner %s tried sell petition: name %s", guidNPC.ToString().c_str(), name.c_str());
75
76 // prevent cheating
77 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(guidNPC, UNIT_NPC_FLAG_PETITIONER);
78 if (!creature)
79 {
80 TC_LOG_DEBUG("network", "WORLD: HandlePetitionBuyOpcode - %s not found or you can't interact with him.", guidNPC.ToString().c_str());
81 return;
82 }
83
84 // remove fake death
85 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
86 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
87
88 uint32 charterid = 0;
89 uint32 cost = 0;
90 uint32 type = 0;
91 if (creature->IsTabardDesigner())
92 {
93 // if tabard designer, then trying to buy a guild charter.
94 // do not let if already in guild.
95 if (_player->GetGuildId())
96 return;
97
98 charterid = GUILD_CHARTER;
99 cost = sWorld->getIntConfig(CONFIG_CHARTER_COST_GUILD);
100 type = GUILD_CHARTER_TYPE;
101 }
102 else
103 {
104 /// @todo find correct opcode
105 if (_player->getLevel() < sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
106 {
107 SendNotification(LANG_ARENA_ONE_TOOLOW, sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL));
108 return;
109 }
110
111 switch (clientIndex) // arenaSlot+1 as received from client (1 from 3 case)
112 {
113 case 1:
114 charterid = ARENA_TEAM_CHARTER_2v2;
115 cost = sWorld->getIntConfig(CONFIG_CHARTER_COST_ARENA_2v2);
116 type = ARENA_TEAM_CHARTER_2v2_TYPE;
117 break;
118 case 2:
119 charterid = ARENA_TEAM_CHARTER_3v3;
120 cost = sWorld->getIntConfig(CONFIG_CHARTER_COST_ARENA_3v3);
121 type = ARENA_TEAM_CHARTER_3v3_TYPE;
122 break;
123 case 3:
124 charterid = ARENA_TEAM_CHARTER_5v5;
125 cost = sWorld->getIntConfig(CONFIG_CHARTER_COST_ARENA_5v5);
126 type = ARENA_TEAM_CHARTER_5v5_TYPE;
127 break;
128 default:
129 TC_LOG_DEBUG("network", "unknown selection at buy arena petition: %u", clientIndex);
130 return;
131 }
132
133 if (_player->GetArenaTeamId(clientIndex - 1)) // arenaSlot+1 as received from client
134 {
135 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ALREADY_IN_ARENA_TEAM);
136 return;
137 }
138 }
139
140 if (type == GUILD_CHARTER_TYPE)
141 {
142 if (sGuildMgr->GetGuildByName(name))
143 {
144 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_NAME_EXISTS_S, name);
145 return;
146 }
147
148 if (sObjectMgr->IsReservedName(name) || !ObjectMgr::IsValidCharterName(name))
149 {
150 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_NAME_INVALID, name);
151 return;
152 }
153 }
154 else
155 {
156 if (sArenaTeamMgr->GetArenaTeamByName(name))
157 {
158 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ARENA_TEAM_NAME_EXISTS_S);
159 return;
160 }
161 if (sObjectMgr->IsReservedName(name) || !ObjectMgr::IsValidCharterName(name))
162 {
163 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ARENA_TEAM_NAME_INVALID);
164 return;
165 }
166 }
167
168 ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(charterid);
169 if (!pProto)
170 {
171 _player->SendBuyError(BUY_ERR_CANT_FIND_ITEM, NULL, charterid, 0);
172 return;
173 }
174
175 if (!_player->HasEnoughMoney(cost))
176 { //player hasn't got enough money
177 _player->SendBuyError(BUY_ERR_NOT_ENOUGHT_MONEY, creature, charterid, 0);
178 return;
179 }
180
181 ItemPosCountVec dest;
182 InventoryResult msg = _player->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, charterid, pProto->BuyCount);
183 if (msg != EQUIP_ERR_OK)
184 {
185 _player->SendEquipError(msg, NULL, NULL, charterid);
186 return;
187 }
188
189 _player->ModifyMoney(-(int32)cost);
190 Item* charter = _player->StoreNewItem(dest, charterid, true);
191 if (!charter)
192 return;
193
194 charter->SetUInt32Value(ITEM_FIELD_ENCHANTMENT_1_1, charter->GetGUID().GetCounter());
195 // ITEM_FIELD_ENCHANTMENT_1_1 is guild/arenateam id
196 // ITEM_FIELD_ENCHANTMENT_1_1+1 is current signatures count (showed on item)
197 charter->SetState(ITEM_CHANGED, _player);
198 _player->SendNewItem(charter, 1, true, false);
199
200 // a petition is invalid, if both the owner and the type matches
201 // we checked above, if this player is in an arenateam, so this must be
202 // datacorruption
203 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_BY_OWNER);
204 stmt->setUInt32(0, _player->GetGUID().GetCounter());
205 stmt->setUInt8(1, type);
206 PreparedQueryResult result = CharacterDatabase.Query(stmt);
207
208 std::ostringstream ssInvalidPetitionGUIDs;
209
210 if (result)
211 {
212 do
213 {
214 Field* fields = result->Fetch();
215 ssInvalidPetitionGUIDs << '\'' << fields[0].GetUInt32() << "', ";
216 } while (result->NextRow());
217 }
218
219 // delete petitions with the same guid as this one
220 ssInvalidPetitionGUIDs << '\'' << charter->GetGUID().GetCounter() << '\'';
221
222 TC_LOG_DEBUG("network", "Invalid petition GUIDs: %s", ssInvalidPetitionGUIDs.str().c_str());
223 CharacterDatabase.EscapeString(name);
224 SQLTransaction trans = CharacterDatabase.BeginTransaction();
225 trans->PAppend("DELETE FROM petition WHERE petitionguid IN (%s)", ssInvalidPetitionGUIDs.str().c_str());
226 trans->PAppend("DELETE FROM petition_sign WHERE petitionguid IN (%s)", ssInvalidPetitionGUIDs.str().c_str());
227
228 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_PETITION);
229 stmt->setUInt32(0, _player->GetGUID().GetCounter());
230 stmt->setUInt32(1, charter->GetGUID().GetCounter());
231 stmt->setString(2, name);
232 stmt->setUInt8(3, uint8(type));
233 trans->Append(stmt);
234
235 CharacterDatabase.CommitTransaction(trans);
236}

238void WorldSession::HandlePetitionShowSignOpcode(WorldPacket& recvData)
239{
240 TC_LOG_DEBUG("network", "Received opcode CMSG_PETITION_SHOW_SIGNATURES");
241
242 uint8 signs = 0;
243 ObjectGuid petitionguid;
244 recvData >> petitionguid; // petition guid
245
246 ObjectGuid::LowType petitionGuidLow = petitionguid.GetCounter();
247
248 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_TYPE);
249
250 stmt->setUInt32(0, petitionGuidLow);
251
252 PreparedQueryResult result = CharacterDatabase.Query(stmt);
253
254 if (!result)
255 {
256 TC_LOG_DEBUG("entities.player.items", "Petition %s is not found for player %u %s", petitionguid.ToString().c_str(), GetPlayer()->GetGUID().GetCounter(), GetPlayer()->GetName().c_str());
257 return;
258 }
259 Field* fields = result->Fetch();
260 uint32 type = fields[0].GetUInt8();
261
262 // if guild petition and has guild => error, return;
263 if (type == GUILD_CHARTER_TYPE && _player->GetGuildId())
264 return;
265
266 stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_SIGNATURE);
267
268 stmt->setUInt32(0, petitionGuidLow);
269
270 result = CharacterDatabase.Query(stmt);
271
272 // result == NULL also correct in case no sign yet
273 if (result)
274 signs = uint8(result->GetRowCount());
275
276 TC_LOG_DEBUG("network", "CMSG_PETITION_SHOW_SIGNATURES petition entry: '%u'", petitionGuidLow);
277
278 WorldPacket data(SMSG_PETITION_SHOW_SIGNATURES, (8+8+4+1+signs*12));
279 data << uint64(petitionguid); // petition guid
280 data << uint64(_player->GetGUID()); // owner guid
281 data << uint32(petitionGuidLow); // guild guid
282 data << uint8(signs); // sign's count
283
284 for (uint8 i = 1; i <= signs; ++i)
285 {
286 Field* fields2 = result->Fetch();
287 ObjectGuid::LowType lowGuid = fields2[0].GetUInt32();
288
289 data << ObjectGuid(HighGuid::Player, 0, lowGuid); // Player GUID
290 data << uint32(0); // there 0 ...
291
292 result->NextRow();
293 }
294 SendPacket(&data);
295}

310void WorldSession::SendPetitionQueryOpcode(ObjectGuid petitionguid)
311{
312 ObjectGuid ownerguid;
313 uint32 type;
314 std::string name = "NO_NAME_FOR_GUID";
315
316 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION);
317
318 stmt->setUInt32(0, petitionguid.GetCounter());
319
320 PreparedQueryResult result = CharacterDatabase.Query(stmt);
321
322 if (result)
323 {
324 Field* fields = result->Fetch();
325 ownerguid = ObjectGuid(HighGuid::Player, fields[0].GetUInt32());
326 name = fields[1].GetString();
327 type = fields[2].GetUInt8();
328 }
329 else
330 {
331 TC_LOG_DEBUG("network", "CMSG_PETITION_QUERY failed for petition (%s)", petitionguid.ToString().c_str());
332 return;
333 }
334
335 WorldPacket data(SMSG_PETITION_QUERY_RESPONSE, (4+8+name.size()+1+1+4*12+2+10));
336 data << uint32(petitionguid.GetCounter()); // guild/team guid (in Trinity always same as GUID_LOPART(petition guid)
337 data << uint64(ownerguid); // charter owner guid
338 data << name; // name (guild/arena team)
339 data << uint8(0); // some string
340 if (type == GUILD_CHARTER_TYPE)
341 {
342 uint32 needed = sWorld->getIntConfig(CONFIG_MIN_PETITION_SIGNS);
343 data << uint32(needed);
344 data << uint32(needed);
345 data << uint32(0); // bypass client - side limitation, a different value is needed here for each petition
346 }
347 else
348 {
349 data << uint32(type-1);
350 data << uint32(type-1);
351 data << uint32(type); // bypass client - side limitation, a different value is needed here for each petition
352 }
353 data << uint32(0); // 5
354 data << uint32(0); // 6
355 data << uint32(0); // 7
356 data << uint32(0); // 8
357 data << uint16(0); // 9 2 bytes field
358 data << uint32(0); // 10
359 data << uint32(0); // 11
360 data << uint32(0); // 13 count of next strings?
361
362 for (int i = 0; i < 10; ++i)
363 data << uint8(0); // some string
364
365 data << uint32(0); // 14
366
367 data << uint32(type != GUILD_CHARTER_TYPE); // 15 0 - guild, 1 - arena team
368
369 SendPacket(&data);
370}

372void WorldSession::HandlePetitionRenameOpcode(WorldPacket& recvData)
373{
374 TC_LOG_DEBUG("network", "Received opcode MSG_PETITION_RENAME"); // ok
375
376 ObjectGuid petitionGuid;
377 uint32 type;
378 std::string newName;
379
380 recvData >> petitionGuid; // guid
381 recvData >> newName; // new name
382
383 Item* item = _player->GetItemByGuid(petitionGuid);
384 if (!item)
385 return;
386
387 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_TYPE);
388
389 stmt->setUInt32(0, petitionGuid.GetCounter());
390
391 PreparedQueryResult result = CharacterDatabase.Query(stmt);
392
393 if (result)
394 {
395 Field* fields = result->Fetch();
396 type = fields[0].GetUInt8();
397 }
398 else
399 {
400 TC_LOG_DEBUG("network", "CMSG_PETITION_QUERY failed for petition %s", petitionGuid.ToString().c_str());
401 return;
402 }
403
404 if (type == GUILD_CHARTER_TYPE)
405 {
406 if (sGuildMgr->GetGuildByName(newName))
407 {
408 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_NAME_EXISTS_S, newName);
409 return;
410 }
411 if (sObjectMgr->IsReservedName(newName) || !ObjectMgr::IsValidCharterName(newName))
412 {
413 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_NAME_INVALID, newName);
414 return;
415 }
416 }
417 else
418 {
419 if (sArenaTeamMgr->GetArenaTeamByName(newName))
420 {
421 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, newName, "", ERR_ARENA_TEAM_NAME_EXISTS_S);
422 return;
423 }
424 if (sObjectMgr->IsReservedName(newName) || !ObjectMgr::IsValidCharterName(newName))
425 {
426 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, newName, "", ERR_ARENA_TEAM_NAME_INVALID);
427 return;
428 }
429 }
430
431 stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_PETITION_NAME);
432
433 stmt->setString(0, newName);
434 stmt->setUInt32(1, petitionGuid.GetCounter());
435
436 CharacterDatabase.Execute(stmt);
437
438 TC_LOG_DEBUG("network", "Petition %s renamed to '%s'", petitionGuid.ToString().c_str(), newName.c_str());
439 WorldPacket data(MSG_PETITION_RENAME, (8+newName.size()+1));
440 data << uint64(petitionGuid);
441 data << newName;
442 SendPacket(&data);
443}

445void WorldSession::HandlePetitionSignOpcode(WorldPacket& recvData)
446{
447 TC_LOG_DEBUG("network", "Received opcode CMSG_PETITION_SIGN"); // ok
448
449 Field* fields;
450 ObjectGuid petitionGuid;
451 uint8 unk;
452 recvData >> petitionGuid; // petition guid
453 recvData >> unk;
454
455 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_SIGNATURES);
456
457 stmt->setUInt32(0, petitionGuid.GetCounter());
458 stmt->setUInt32(1, petitionGuid.GetCounter());
459
460 PreparedQueryResult result = CharacterDatabase.Query(stmt);
461
462 if (!result)
463 {
464 TC_LOG_ERROR("network", "Petition %s is not found for player %u %s", petitionGuid.ToString().c_str(), GetPlayer()->GetGUID().GetCounter(), GetPlayer()->GetName().c_str());
465 return;
466 }
467
468 fields = result->Fetch();
469 ObjectGuid ownerGuid(HighGuid::Player, fields[0].GetUInt32());
470 uint64 signs = fields[1].GetUInt64();
471 uint8 type = fields[2].GetUInt8();
472
473 ObjectGuid::LowType playerGuid = _player->GetGUID().GetCounter();
474 if (ownerGuid == _player->GetGUID())
475 return;
476
477 // not let enemies sign guild charter
478 if (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GUILD) && GetPlayer()->GetTeam() != sObjectMgr->GetPlayerTeamByGUID(ownerGuid))
479 {
480 if (type != GUILD_CHARTER_TYPE)
481 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", "", ERR_ARENA_TEAM_NOT_ALLIED);
482 else
483 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_NOT_ALLIED);
484 return;
485 }
486
487 if (type != GUILD_CHARTER_TYPE)
488 {
489 if (_player->getLevel() < sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
490 {
491 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", _player->GetName().c_str(), ERR_ARENA_TEAM_TARGET_TOO_LOW_S);
492 return;
493 }
494
495 uint8 slot = ArenaTeam::GetSlotByType(type);
496 if (slot >= MAX_ARENA_SLOT)
497 return;
498
499 if (_player->GetArenaTeamId(slot))
500 {
501 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", _player->GetName().c_str(), ERR_ALREADY_IN_ARENA_TEAM_S);
502 return;
503 }
504
505 if (_player->GetArenaTeamIdInvited())
506 {
507 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", _player->GetName().c_str(), ERR_ALREADY_INVITED_TO_ARENA_TEAM_S);
508 return;
509 }
510 }
511 else
512 {
513 if (_player->GetGuildId())
514 {
515 Guild::SendCommandResult(this, GUILD_COMMAND_INVITE, ERR_ALREADY_IN_GUILD_S, _player->GetName());
516 return;
517 }
518 if (_player->GetGuildIdInvited())
519 {
520 Guild::SendCommandResult(this, GUILD_COMMAND_INVITE, ERR_ALREADY_INVITED_TO_GUILD_S, _player->GetName());
521 return;
522 }
523 }
524
525 if (++signs > type) // client signs maximum
526 return;
527
528 // Client doesn't allow to sign petition two times by one character, but not check sign by another character from same account
529 // not allow sign another player from already sign player account
530 stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_SIG_BY_ACCOUNT);
531
532 stmt->setUInt32(0, GetAccountId());
533 stmt->setUInt32(1, petitionGuid.GetCounter());
534
535 result = CharacterDatabase.Query(stmt);
536
537 if (result)
538 {
539 WorldPacket data(SMSG_PETITION_SIGN_RESULTS, (8+8+4));
540 data << uint64(petitionGuid);
541 data << uint64(_player->GetGUID());
542 data << (uint32)PETITION_SIGN_ALREADY_SIGNED;
543
544 // close at signer side
545 SendPacket(&data);
546
547 // update for owner if online
548 if (Player* owner = ObjectAccessor::FindConnectedPlayer(ownerGuid))
549 owner->GetSession()->SendPacket(&data);
550 return;
551 }
552
553 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_PETITION_SIGNATURE);
554
555 stmt->setUInt32(0, ownerGuid.GetCounter());
556 stmt->setUInt32(1, petitionGuid.GetCounter());
557 stmt->setUInt32(2, playerGuid);
558 stmt->setUInt32(3, GetAccountId());
559
560 CharacterDatabase.Execute(stmt);
561
562 TC_LOG_DEBUG("network", "PETITION SIGN: %s by player: %s (GUID: %u Account: %u)", petitionGuid.ToString().c_str(), _player->GetName().c_str(), playerGuid, GetAccountId());
563
564 WorldPacket data(SMSG_PETITION_SIGN_RESULTS, (8+8+4));
565 data << uint64(petitionGuid);
566 data << uint64(_player->GetGUID());
567 data << uint32(PETITION_SIGN_OK);
568
569 // close at signer side
570 SendPacket(&data);
571
572 // update signs count on charter, required testing...
573 //Item* item = _player->GetItemByGuid(petitionguid));
574 //if (item)
575 // item->SetUInt32Value(ITEM_FIELD_ENCHANTMENT_1_1+1, signs);
576
577 // update for owner if online
578 if (Player* owner = ObjectAccessor::FindConnectedPlayer(ownerGuid))
579 owner->GetSession()->SendPacket(&data);
580}

611void WorldSession::HandleOfferPetitionOpcode(WorldPacket& recvData)
612{
613 TC_LOG_DEBUG("network", "Received opcode CMSG_OFFER_PETITION"); // ok
614
615 uint8 signs = 0;
616 ObjectGuid petitionguid, plguid;
617 uint32 type, junk;
618 Player* player;
619 recvData >> junk; // this is not petition type!
620 recvData >> petitionguid; // petition guid
621 recvData >> plguid; // player guid
622
623 player = ObjectAccessor::FindConnectedPlayer(plguid);
624 if (!player)
625 return;
626
627 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_TYPE);
628
629 stmt->setUInt32(0, petitionguid.GetCounter());
630
631 PreparedQueryResult result = CharacterDatabase.Query(stmt);
632
633 if (!result)
634 return;
635
636 Field* fields = result->Fetch();
637 type = fields[0].GetUInt8();
638
639 TC_LOG_DEBUG("network", "OFFER PETITION: type %u, %s, to %s", type, petitionguid.ToString().c_str(), plguid.ToString().c_str());
640
641 if (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GUILD) && GetPlayer()->GetTeam() != player->GetTeam())
642 {
643 if (type != GUILD_CHARTER_TYPE)
644 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", "", ERR_ARENA_TEAM_NOT_ALLIED);
645 else
646 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_NOT_ALLIED);
647 return;
648 }
649
650 if (type != GUILD_CHARTER_TYPE)
651 {
652 if (player->getLevel() < sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL))
653 {
654 // player is too low level to join an arena team
655 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, player->GetName().c_str(), "", ERR_ARENA_TEAM_TARGET_TOO_LOW_S);
656 return;
657 }
658
659 uint8 slot = ArenaTeam::GetSlotByType(type);
660 if (slot >= MAX_ARENA_SLOT)
661 return;
662
663 if (player->GetArenaTeamId(slot))
664 {
665 // player is already in an arena team
666 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, player->GetName().c_str(), "", ERR_ALREADY_IN_ARENA_TEAM_S);
667 return;
668 }
669
670 if (player->GetArenaTeamIdInvited())
671 {
672 SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", _player->GetName().c_str(), ERR_ALREADY_INVITED_TO_ARENA_TEAM_S);
673 return;
674 }
675 }
676 else
677 {
678 if (player->GetGuildId())
679 {
680 Guild::SendCommandResult(this, GUILD_COMMAND_INVITE, ERR_ALREADY_IN_GUILD_S, _player->GetName());
681 return;
682 }
683
684 if (player->GetGuildIdInvited())
685 {
686 Guild::SendCommandResult(this, GUILD_COMMAND_INVITE, ERR_ALREADY_INVITED_TO_GUILD_S, _player->GetName());
687 return;
688 }
689 }
690
691
692 stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_SIGNATURE);
693
694 stmt->setUInt32(0, petitionguid.GetCounter());
695
696 result = CharacterDatabase.Query(stmt);
697
698 // result == NULL also correct charter without signs
699 if (result)
700 signs = uint8(result->GetRowCount());
701
702 WorldPacket data(SMSG_PETITION_SHOW_SIGNATURES, (8+8+4+signs+signs*12));
703 data << uint64(petitionguid); // petition guid
704 data << uint64(_player->GetGUID()); // owner guid
705 data << uint32(petitionguid.GetCounter()); // guild guid
706 data << uint8(signs); // sign's count
707
708 for (uint8 i = 1; i <= signs; ++i)
709 {
710 Field* fields2 = result->Fetch();
711 data << uint64(ObjectGuid(HighGuid::Player, fields2[0].GetUInt32())); // Player GUID
712 data << uint32(0); // there 0 ...
713
714 result->NextRow();
715 }
716
717 player->GetSession()->SendPacket(&data);
718}

720void WorldSession::HandleTurnInPetitionOpcode(WorldPacket& recvData)
721{
722 TC_LOG_DEBUG("network", "Received opcode CMSG_TURN_IN_PETITION");
723
724 // Get petition guid from packet
725 WorldPacket data;
726 ObjectGuid petitionGuid;
727
728 recvData >> petitionGuid;
729
730 // Check if player really has the required petition charter
731 Item* item = _player->GetItemByGuid(petitionGuid);
732 if (!item)
733 return;
734
735 TC_LOG_DEBUG("network", "Petition %s turned in by %u", petitionGuid.ToString().c_str(), _player->GetGUID().GetCounter());
736
737 // Get petition data from db
738 ObjectGuid::LowType ownerguidlo;
739 uint32 type;
740 std::string name;
741
742 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION);
743 stmt->setUInt32(0, petitionGuid.GetCounter());
744 PreparedQueryResult result = CharacterDatabase.Query(stmt);
745
746 if (result)
747 {
748 Field* fields = result->Fetch();
749 ownerguidlo = fields[0].GetUInt32();
750 name = fields[1].GetString();
751 type = fields[2].GetUInt8();
752 }
753 else
754 {
755 TC_LOG_ERROR("network", "Player %s (guid: %u) tried to turn in petition (%s) that is not present in the database", _player->GetName().c_str(), _player->GetGUID().GetCounter(), petitionGuid.ToString().c_str());
756 return;
757 }
758
759 // Only the petition owner can turn in the petition
760 if (_player->GetGUID().GetCounter() != ownerguidlo)
761 return;
762
763 // Petition type (guild/arena) specific checks
764 if (type == GUILD_CHARTER_TYPE)
765 {
766 // Check if player is already in a guild
767 if (_player->GetGuildId())
768 {
769 data.Initialize(SMSG_TURN_IN_PETITION_RESULTS, 4);
770 data << (uint32)PETITION_TURN_ALREADY_IN_GUILD;
771 _player->GetSession()->SendPacket(&data);
772 return;
773 }
774
775 // Check if guild name is already taken
776 if (sGuildMgr->GetGuildByName(name))
777 {
778 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_NAME_EXISTS_S, name);
779 return;
780 }
781 }
782 else
783 {
784 // Check for valid arena bracket (2v2, 3v3, 5v5)
785 uint8 slot = ArenaTeam::GetSlotByType(type);
786 if (slot >= MAX_ARENA_SLOT)
787 return;
788
789 // Check if player is already in an arena team
790 if (_player->GetArenaTeamId(slot))
791 {
792 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ALREADY_IN_ARENA_TEAM);
793 return;
794 }
795
796 // Check if arena team name is already taken
797 if (sArenaTeamMgr->GetArenaTeamByName(name))
798 {
799 SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ARENA_TEAM_NAME_EXISTS_S);
800 return;
801 }
802 }
803
804 // Get petition signatures from db
805 uint8 signatures;
806
807 stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PETITION_SIGNATURE);
808 stmt->setUInt32(0, petitionGuid.GetCounter());
809 result = CharacterDatabase.Query(stmt);
810
811 if (result)
812 signatures = uint8(result->GetRowCount());
813 else
814 signatures = 0;
815
816 uint32 requiredSignatures;
817 if (type == GUILD_CHARTER_TYPE)
818 requiredSignatures = sWorld->getIntConfig(CONFIG_MIN_PETITION_SIGNS);
819 else
820 requiredSignatures = type-1;
821
822 // Notify player if signatures are missing
823 if (signatures < requiredSignatures)
824 {
825 data.Initialize(SMSG_TURN_IN_PETITION_RESULTS, 4);
826 data << (uint32)PETITION_TURN_NEED_MORE_SIGNATURES;
827 SendPacket(&data);
828 return;
829 }
830
831 // Proceed with guild/arena team creation
832
833 // Delete charter item
834 _player->DestroyItem(item->GetBagSlot(), item->GetSlot(), true);
835
836 if (type == GUILD_CHARTER_TYPE)
837 {
838 // Create guild
839 Guild* guild = new Guild;
840
841 if (!guild->Create(_player, name))
842 {
843 delete guild;
844 return;
845 }
846
847 // Register guild and add guild master
848 sGuildMgr->AddGuild(guild);
849
850 Guild::SendCommandResult(this, GUILD_COMMAND_CREATE, ERR_GUILD_COMMAND_SUCCESS, name);
851
852 // Add members from signatures
853 for (uint8 i = 0; i < signatures; ++i)
854 {
855 Field* fields = result->Fetch();
856 guild->AddMember(ObjectGuid(HighGuid::Player, fields[0].GetUInt32()));
857 result->NextRow();
858 }
859 }
860 else
861 {
862 // Receive the rest of the packet in arena team creation case
863 uint32 background, icon, iconcolor, border, bordercolor;
864 recvData >> background >> icon >> iconcolor >> border >> bordercolor;
865
866 // Create arena team
867 ArenaTeam* arenaTeam = new ArenaTeam();
868
869 if (!arenaTeam->Create(_player->GetGUID(), type, name, background, icon, iconcolor, border, bordercolor))
870 {
871 delete arenaTeam;
872 return;
873 }
874
875 // Register arena team
876 sArenaTeamMgr->AddArenaTeam(arenaTeam);
877 TC_LOG_DEBUG("network", "PetitonsHandler: Arena team (guid: %u) added to ObjectMgr", arenaTeam->GetId());
878
879 // Add members
880 for (uint8 i = 0; i < signatures; ++i)
881 {
882 Field* fields = result->Fetch();
883 ObjectGuid memberGUID(HighGuid::Player, fields[0].GetUInt32());
884 TC_LOG_DEBUG("network", "PetitionsHandler: Adding arena team (guid: %u) member %s", arenaTeam->GetId(), memberGUID.ToString().c_str());
885 arenaTeam->AddMember(memberGUID);
886 result->NextRow();
887 }
888 }
889
890 SQLTransaction trans = CharacterDatabase.BeginTransaction();
891
892 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_PETITION_BY_GUID);
893 stmt->setUInt32(0, petitionGuid.GetCounter());
894 trans->Append(stmt);
895
896 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_PETITION_SIGNATURE_BY_GUID);
897 stmt->setUInt32(0, petitionGuid.GetCounter());
898 trans->Append(stmt);
899
900 CharacterDatabase.CommitTransaction(trans);
901
902 // created
903 TC_LOG_DEBUG("network", "Player %s (%s) turning in petition %s", _player->GetName().c_str(), _player->GetGUID().ToString().c_str(), petitionGuid.ToString().c_str());
904
905 data.Initialize(SMSG_TURN_IN_PETITION_RESULTS, 4);
906 data << (uint32)PETITION_TURN_OK;
907 SendPacket(&data);
908}

920void WorldSession::SendPetitionShowList(ObjectGuid guid)
921{
922 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_PETITIONER);

924 {

928

932 if (creature->IsTabardDesigner())

945 // 2v2
946 data << uint32(1); // index

948 data << uint32(CHARTER_DISPLAY_ID); // charter display id
949 data << uint32(sWorld->getIntConfig(CONFIG_CHARTER_COST_ARENA_2v2)); // charter cost
950 data << uint32(2); // unknown
951 data << uint32(2); // required signs?
952 // 3v3
953 data << uint32(2); // index
954 data << uint32(ARENA_TEAM_CHARTER_3v3); // charter entry
955 data << uint32(CHARTER_DISPLAY_ID); // charter display id
956 data << uint32(sWorld->getIntConfig(CONFIG_CHARTER_COST_ARENA_3v3)); // charter cost
957 data << uint32(3); // unknown
958 data << uint32(3); // required signs?
959 // 5v5
960 data << uint32(3); // index

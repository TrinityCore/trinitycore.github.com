30void WorldSession::HandleGMTicketCreateOpcode(WorldPacket& recvData)
31{
32 // Don't accept tickets if the ticket queue is disabled. (Ticket UI is greyed out but not fully dependable)
33 if (sTicketMgr->GetStatus() == GMTICKET_QUEUE_STATUS_DISABLED)
34 return;
35
36 if (GetPlayer()->getLevel() < sWorld->getIntConfig(CONFIG_TICKET_LEVEL_REQ))
37 {
38 SendNotification(GetTrinityString(LANG_TICKET_REQ), sWorld->getIntConfig(CONFIG_TICKET_LEVEL_REQ));
39 return;
40 }
41
42 GMTicketResponse response = GMTICKET_RESPONSE_CREATE_ERROR;
43 GmTicket* ticket = sTicketMgr->GetTicketByPlayer(GetPlayer()->GetGUID());
44
45 if (ticket && ticket->IsCompleted())
46 sTicketMgr->CloseTicket(ticket->GetId(), GetPlayer()->GetGUID());
47
48 // Player must not have ticket
49 if (!ticket || ticket->IsClosed())
50 {
51 uint32 mapId;
52 float x, y, z;
53 std::string message;
54 uint32 needResponse;
55 bool needMoreHelp;
56 uint32 count;
57 std::list<uint32> times;
58 uint32 decompressedSize;
59 std::string chatLog;
60
61 recvData >> mapId;
62 recvData >> x >> y >> z;
63 recvData >> message;
64
65 recvData >> needResponse;
66 recvData >> needMoreHelp;
67
68 recvData >> count;
69
70 for (uint32 i = 0; i < count; i++)
71 {
72 uint32 time;
73 recvData >> time;
74 times.push_back(time);
75 }
76
77 recvData >> decompressedSize;
78
79 if (count && decompressedSize && decompressedSize < 0xFFFF)
80 {
81 uint32 pos = recvData.rpos();
82 ByteBuffer dest;
83 dest.resize(decompressedSize);
84
85 uLongf realSize = decompressedSize;
86 if (uncompress(dest.contents(), &realSize, recvData.contents() + pos, recvData.size() - pos) == Z_OK)
87 {
88 dest >> chatLog;
89 }
90 else
91 {
92 TC_LOG_ERROR("network", "CMSG_GMTICKET_CREATE possibly corrupt. Uncompression failed.");
93 recvData.rfinish();
94 return;
95 }
96
97 recvData.rfinish(); // Will still have compressed data in buffer.
98 }
99
100 ticket = new GmTicket(GetPlayer());
101 ticket->SetPosition(mapId, x, y, z);
102 ticket->SetMessage(message);
103 ticket->SetGmAction(needResponse, needMoreHelp);
104
105 if (!chatLog.empty())
106 ticket->SetChatLog(times, chatLog);
107
108 sTicketMgr->AddTicket(ticket);
109 sTicketMgr->UpdateLastChange();
110
111 sWorld->SendGMText(LANG_COMMAND_TICKETNEW, GetPlayer()->GetName().c_str(), ticket->GetId());
112
113 response = GMTICKET_RESPONSE_CREATE_SUCCESS;
114 }
115
116 WorldPacket data(SMSG_GMTICKET_CREATE, 4);
117 data << uint32(response);
118 SendPacket(&data);
119}

121void WorldSession::HandleGMTicketUpdateOpcode(WorldPacket& recvData)
122{
123 std::string message;
124 recvData >> message;
125
126 GMTicketResponse response = GMTICKET_RESPONSE_UPDATE_ERROR;
127 if (GmTicket* ticket = sTicketMgr->GetTicketByPlayer(GetPlayer()->GetGUID()))
128 {
129 SQLTransaction trans = SQLTransaction(NULL);
130 ticket->SetMessage(message);
131 ticket->SaveToDB(trans);
132
133 sWorld->SendGMText(LANG_COMMAND_TICKETUPDATED, GetPlayer()->GetName().c_str(), ticket->GetId());
134
135 response = GMTICKET_RESPONSE_UPDATE_SUCCESS;
136 }
137
138 WorldPacket data(SMSG_GMTICKET_UPDATETEXT, 4);
139 data << uint32(response);
140 SendPacket(&data);
141}

143void WorldSession::HandleGMTicketDeleteOpcode(WorldPacket & /*recvData*/)
144{

146 {

150

152
153 sTicketMgr->CloseTicket(ticket->GetId(), GetPlayer()->GetGUID());
154 sTicketMgr->SendTicket(this, NULL);
155 }

158void WorldSession::HandleGMTicketGetTicketOpcode(WorldPacket & /*recvData*/)
159{
160 SendQueryTimeResponse();
161
162 if (GmTicket* ticket = sTicketMgr->GetTicketByPlayer(GetPlayer()->GetGUID()))
163 {
164 if (ticket->IsCompleted())
165 ticket->SendResponse(this);
166 else
167 sTicketMgr->SendTicket(this, ticket);
168 }
169 else
170 sTicketMgr->SendTicket(this, NULL);
171}

182void WorldSession::HandleGMSurveySubmit(WorldPacket& recvData)
183{
184 uint32 nextSurveyID = sTicketMgr->GetNextSurveyID();
185 // just put the survey into the database
186 uint32 mainSurvey; // GMSurveyCurrentSurvey.dbc, column 1 (all 9) ref to GMSurveySurveys.dbc
187 recvData >> mainSurvey;
188
189 std::unordered_set<uint32> surveyIds;
190 SQLTransaction trans = CharacterDatabase.BeginTransaction();
191 // sub_survey1, r1, comment1, sub_survey2, r2, comment2, sub_survey3, r3, comment3, sub_survey4, r4, comment4, sub_survey5, r5, comment5, sub_survey6, r6, comment6, sub_survey7, r7, comment7, sub_survey8, r8, comment8, sub_survey9, r9, comment9, sub_survey10, r10, comment10,
192 for (uint8 i = 0; i < 10; i++)
193 {
194 uint32 subSurveyId; // ref to i'th GMSurveySurveys.dbc field (all fields in that dbc point to fields in GMSurveyQuestions.dbc)
195 recvData >> subSurveyId;
196 if (!subSurveyId)
197 break;
198
199 uint8 rank; // probably some sort of ref to GMSurveyAnswers.dbc
200 recvData >> rank;
201 std::string comment; // comment ("Usage: GMSurveyAnswerSubmit(question, rank, comment)")
202 recvData >> comment;
203
204 // make sure the same sub survey is not added to DB twice
205 if (!surveyIds.insert(subSurveyId).second)
206 continue;
207
208 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_GM_SUBSURVEY);
209 stmt->setUInt32(0, nextSurveyID);
210 stmt->setUInt32(1, subSurveyId);
211 stmt->setUInt32(2, rank);
212 stmt->setString(3, comment);
213 trans->Append(stmt);
214 }
215
216 std::string comment; // just a guess
217 recvData >> comment;
218
219 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_GM_SURVEY);
220 stmt->setUInt32(0, GetPlayer()->GetGUID().GetCounter());
221 stmt->setUInt32(1, nextSurveyID);
222 stmt->setUInt32(2, mainSurvey);
223 stmt->setString(3, comment);
224
225 trans->Append(stmt);
226
227 CharacterDatabase.CommitTransaction(trans);
228}

230void WorldSession::HandleReportLag(WorldPacket& recvData)
231{
232 // just put the lag report into the database...
233 // can't think of anything else to do with it
234 uint32 lagType, mapId;
235 recvData >> lagType;
236 recvData >> mapId;
237 float x, y, z;
238 recvData >> x;
239 recvData >> y;
240 recvData >> z;
241
242 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_LAG_REPORT);
243 stmt->setUInt32(0, GetPlayer()->GetGUID().GetCounter());
244 stmt->setUInt8 (1, lagType);
245 stmt->setUInt16(2, mapId);
246 stmt->setFloat (3, x);
247 stmt->setFloat (4, y);
248 stmt->setFloat (5, z);
249 stmt->setUInt32(6, GetLatency());
250 stmt->setUInt32(7, time(NULL));
251 CharacterDatabase.Execute(stmt);
252}

254void WorldSession::HandleGMResponseResolve(WorldPacket& /*recvPacket*/)
255{
256 // empty packet
257 if (GmTicket* ticket = sTicketMgr->GetTicketByPlayer(GetPlayer()->GetGUID()))
258 {
259 uint8 getSurvey = 0;
260 if (float(rand_chance()) < sWorld->getFloatConfig(CONFIG_CHANCE_OF_GM_SURVEY))
261 getSurvey = 1;
262
263 WorldPacket data(SMSG_GMRESPONSE_STATUS_UPDATE, 4);
264 data << uint8(getSurvey);
265 SendPacket(&data);
266
267 WorldPacket data2(SMSG_GMTICKET_DELETETICKET, 4);
268 data2 << uint32(GMTICKET_RESPONSE_TICKET_DELETED);
269 SendPacket(&data2);
270
271 sTicketMgr->CloseTicket(ticket->GetId(), GetPlayer()->GetGUID());
272 sTicketMgr->SendTicket(this, NULL);
273 }

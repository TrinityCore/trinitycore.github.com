36void WorldSession::HandleLearnPreviewTalents(WorldPacket& recvPacket)
37{
38 TC_LOG_DEBUG("network", "CMSG_LEARN_PREVIEW_TALENTS");
39
40 uint32 talentsCount;
41 recvPacket >> talentsCount;
42
43 uint32 talentId, talentRank;
44
45 // Client has max 44 talents for tree for 3 trees, rounded up : 150
46 uint32 const MaxTalentsCount = 150;
47
48 for (uint32 i = 0; i < talentsCount && i < MaxTalentsCount; ++i)
49 {
50 recvPacket >> talentId >> talentRank;
51
52 _player->LearnTalent(talentId, talentRank);
53 }
54
55 _player->SendTalentsInfoData(false);
56
57 recvPacket.rfinish();
58}

60void WorldSession::HandleTalentWipeConfirmOpcode(WorldPacket& recvData)
61{
62 TC_LOG_DEBUG("network", "MSG_TALENT_WIPE_CONFIRM");
63 ObjectGuid guid;
64 recvData >> guid;
65
66 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
67 if (!unit)
68 {
69 TC_LOG_DEBUG("network", "WORLD: HandleTalentWipeConfirmOpcode - %s not found or you can't interact with him.", guid.ToString().c_str());
70 return;
71 }
72
73 if (!unit->isCanTrainingAndResetTalentsOf(_player))
74 return;
75
76 // remove fake death
77 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
78 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
79
80 if (!(_player->ResetTalents()))
81 {
82 WorldPacket data(MSG_TALENT_WIPE_CONFIRM, 8+4); //you have not any talent
83 data << uint64(0);
84 data << uint32(0);
85 SendPacket(&data);
86 return;
87 }
88
89 _player->SendTalentsInfoData(false);
90 unit->CastSpell(_player, 14867, true); //spell: "Untalent Visual Effect"
91}

93void WorldSession::HandleUnlearnSkillOpcode(WorldPacket& recvData)
94{
95 uint32 skillId;
96 recvData >> skillId;
97
98 SkillRaceClassInfoEntry const* rcEntry = GetSkillRaceClassInfo(skillId, GetPlayer()->getRace(), GetPlayer()->getClass());
99 if (!rcEntry || !(rcEntry->Flags & SKILL_FLAG_UNLEARNABLE))
100 return;
101
102 GetPlayer()->SetSkill(skillId, 0, 0, 0);
103}

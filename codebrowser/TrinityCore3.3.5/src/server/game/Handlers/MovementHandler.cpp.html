42void WorldSession::HandleMoveWorldportAckOpcode()
43{
44 // ignore unexpected far teleports
45 if (!GetPlayer()->IsBeingTeleportedFar())
46 return;
47
48 GetPlayer()->SetSemaphoreTeleportFar(false);
49
50 // get the teleport destination
51 WorldLocation const& loc = GetPlayer()->GetTeleportDest();
52
53 // possible errors in the coordinate validity check
54 if (!MapManager::IsValidMapCoord(loc))
55 {
56 LogoutPlayer(false);
57 return;
58 }
59
60 // get the destination map entry, not the current one, this will fix homebind and reset greeting
61 MapEntry const* mEntry = sMapStore.LookupEntry(loc.GetMapId());
62 InstanceTemplate const* mInstance = sObjectMgr->GetInstanceTemplate(loc.GetMapId());
63
64 // reset instance validity, except if going to an instance inside an instance
65 if (GetPlayer()->m_InstanceValid == false && !mInstance)
66 GetPlayer()->m_InstanceValid = true;
67
68 Map* oldMap = GetPlayer()->GetMap();
69 Map* newMap = sMapMgr->CreateMap(loc.GetMapId(), GetPlayer());
70
71 if (GetPlayer()->IsInWorld())
72 {
73 TC_LOG_ERROR("network", "%s %s is still in world when teleported from map %s (%u) to new map %s (%u)", GetPlayer()->GetGUID().ToString().c_str(), GetPlayer()->GetName().c_str(), oldMap->GetMapName(), oldMap->GetId(), newMap ? newMap->GetMapName() : "Unknown", loc.GetMapId());
74 oldMap->RemovePlayerFromMap(GetPlayer(), false);
75 }
76
77 // relocate the player to the teleport destination
78 // the CannotEnter checks are done in TeleporTo but conditions may change
79 // while the player is in transit, for example the map may get full
80 if (!newMap || newMap->CannotEnter(GetPlayer()))
81 {
82 TC_LOG_ERROR("network", "Map %d (%s) could not be created for player %d (%s), porting player to homebind", loc.GetMapId(), newMap ? newMap->GetMapName() : "Unknown", GetPlayer()->GetGUID().GetCounter(), GetPlayer()->GetName().c_str());
83 GetPlayer()->TeleportTo(GetPlayer()->m_homebindMapId, GetPlayer()->m_homebindX, GetPlayer()->m_homebindY, GetPlayer()->m_homebindZ, GetPlayer()->GetOrientation());
84 return;
85 }
86
87 float z = loc.GetPositionZ();
88 if (GetPlayer()->HasUnitMovementFlag(MOVEMENTFLAG_HOVER))
89 z += GetPlayer()->GetFloatValue(UNIT_FIELD_HOVERHEIGHT);
90 GetPlayer()->Relocate(loc.GetPositionX(), loc.GetPositionY(), z, loc.GetOrientation());
91
92 GetPlayer()->ResetMap();
93 GetPlayer()->SetMap(newMap);
94
95 GetPlayer()->SendInitialPacketsBeforeAddToMap();
96 if (!GetPlayer()->GetMap()->AddPlayerToMap(GetPlayer()))
97 {
98 TC_LOG_ERROR("network", "WORLD: failed to teleport player %s (%d) to map %d (%s) because of unknown reason!",
99 GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().GetCounter(), loc.GetMapId(), newMap ? newMap->GetMapName() : "Unknown");
100 GetPlayer()->ResetMap();
101 GetPlayer()->SetMap(oldMap);
102 GetPlayer()->TeleportTo(GetPlayer()->m_homebindMapId, GetPlayer()->m_homebindX, GetPlayer()->m_homebindY, GetPlayer()->m_homebindZ, GetPlayer()->GetOrientation());
103 return;
104 }
105
106 // battleground state prepare (in case join to BG), at relogin/tele player not invited
107 // only add to bg group and object, if the player was invited (else he entered through command)
108 if (_player->InBattleground())
109 {
110 // cleanup setting if outdated
111 if (!mEntry->IsBattlegroundOrArena())
112 {
113 // We're not in BG
114 _player->SetBattlegroundId(0, BATTLEGROUND_TYPE_NONE);
115 // reset destination bg team
116 _player->SetBGTeam(0);
117 }
118 // join to bg case
119 else if (Battleground* bg = _player->GetBattleground())
120 {
121 if (_player->IsInvitedForBattlegroundInstance(_player->GetBattlegroundId()))
122 bg->AddPlayer(_player);
123 }
124 }
125
126 GetPlayer()->SendInitialPacketsAfterAddToMap();
127
128 // flight fast teleport case
129 if (GetPlayer()->GetMotionMaster()->GetCurrentMovementGeneratorType() == FLIGHT_MOTION_TYPE)
130 {
131 if (!_player->InBattleground())
132 {
133 // short preparations to continue flight
134 FlightPathMovementGenerator* flight = (FlightPathMovementGenerator*)(GetPlayer()->GetMotionMaster()->top());
135 flight->Initialize(GetPlayer());
136 return;
137 }
138
139 // battleground state prepare, stop flight
140 GetPlayer()->GetMotionMaster()->MovementExpired();
141 GetPlayer()->CleanupAfterTaxiFlight();
142 }
143
144 // resurrect character at enter into instance where his corpse exist after add to map
145
146 if (mEntry->IsDungeon() && !GetPlayer()->IsAlive())
147 if (GetPlayer()->GetCorpseLocation().GetMapId() == mEntry->MapID)
148 {
149 GetPlayer()->ResurrectPlayer(0.5f, false);
150 GetPlayer()->SpawnCorpseBones();
151 }
152
153 bool allowMount = !mEntry->IsDungeon() || mEntry->IsBattlegroundOrArena();
154 if (mInstance)
155 {
156 // check if this instance has a reset time and send it to player if so
157 Difficulty diff = GetPlayer()->GetDifficulty(mEntry->IsRaid());
158 if (MapDifficulty const* mapDiff = GetMapDifficultyData(mEntry->MapID, diff))
159 {
160 if (mapDiff->resetTime)
161 {
162 if (time_t timeReset = sInstanceSaveMgr->GetResetTimeFor(mEntry->MapID, diff))
163 {
164 uint32 timeleft = uint32(timeReset - time(NULL));
165 GetPlayer()->SendInstanceResetWarning(mEntry->MapID, diff, timeleft, true);
166 }
167 }
168 }
169
170 // check if instance is valid
171 if (!GetPlayer()->CheckInstanceValidity(false))
172 GetPlayer()->m_InstanceValid = false;
173
174 // instance mounting is handled in InstanceTemplate
175 allowMount = mInstance->AllowMount;
176 }
177
178 // mount allow check
179 if (!allowMount)
180 _player->RemoveAurasByType(SPELL_AURA_MOUNTED);
181
182 // update zone immediately, otherwise leave channel will cause crash in mtmap
183 uint32 newzone, newarea;
184 GetPlayer()->GetZoneAndAreaId(newzone, newarea);
185 GetPlayer()->UpdateZone(newzone, newarea);
186
187 // honorless target
188 if (GetPlayer()->pvpInfo.IsHostile)
189 GetPlayer()->CastSpell(GetPlayer(), 2479, true);
190
191 // in friendly area
192 else if (GetPlayer()->IsPvP() && !GetPlayer()->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_IN_PVP))
193 GetPlayer()->UpdatePvP(false, false);
194
195 // resummon pet
196 GetPlayer()->ResummonPetTemporaryUnSummonedIfAny();
197
198 //lets process all delayed operations on successful teleport
199 GetPlayer()->ProcessDelayedOperations();
200}

202void WorldSession::HandleMoveTeleportAck(WorldPacket& recvData)
203{
204 TC_LOG_DEBUG("network", "MSG_MOVE_TELEPORT_ACK");
205 ObjectGuid guid;
206
207 recvData >> guid.ReadAsPacked();
208
209 uint32 flags, time;
210 recvData >> flags >> time;
211
212 Player* plMover = _player->m_mover->ToPlayer();
213
214 if (!plMover || !plMover->IsBeingTeleportedNear())
215 return;
216
217 if (guid != plMover->GetGUID())
218 return;
219
220 plMover->SetSemaphoreTeleportNear(false);
221
222 uint32 old_zone = plMover->GetZoneId();
223
224 WorldLocation const& dest = plMover->GetTeleportDest();
225
226 plMover->UpdatePosition(dest, true);
227
228 uint32 newzone, newarea;
229 plMover->GetZoneAndAreaId(newzone, newarea);
230 plMover->UpdateZone(newzone, newarea);
231
232 // new zone
233 if (old_zone != newzone)
234 {
235 // honorless target
236 if (plMover->pvpInfo.IsHostile)
237 plMover->CastSpell(plMover, 2479, true);
238
239 // in friendly area
240 else if (plMover->IsPvP() && !plMover->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_IN_PVP))
241 plMover->UpdatePvP(false, false);
242 }
243
244 // resummon pet
245 GetPlayer()->ResummonPetTemporaryUnSummonedIfAny();
246
247 //lets process all delayed operations on successful teleport
248 GetPlayer()->ProcessDelayedOperations();
249}

251void WorldSession::HandleMovementOpcodes(WorldPacket& recvData)
252{
253 uint16 opcode = recvData.GetOpcode();
254
255 Unit* mover = _player->m_mover;
256
257 ASSERT(mover != NULL); // there must always be a mover
258
259 Player* plrMover = mover->ToPlayer();
260
261 // ignore, waiting processing in WorldSession::HandleMoveWorldportAckOpcode and WorldSession::HandleMoveTeleportAck
262 if (plrMover && plrMover->IsBeingTeleported())
263 {
264 recvData.rfinish(); // prevent warnings spam
265 return;
266 }
267
268 /* extract packet */
269 ObjectGuid guid;
270
271 recvData >> guid.ReadAsPacked();
272
273 MovementInfo movementInfo;
274 movementInfo.guid = guid;
275 ReadMovementInfo(recvData, &movementInfo);
276
277 recvData.rfinish(); // prevent warnings spam
278
279 // prevent tampered movement data
280 if (guid != mover->GetGUID())
281 return;
282
283 if (!movementInfo.pos.IsPositionValid())
284 {
285 recvData.rfinish(); // prevent warnings spam
286 return;
287 }
288
289 /* handle special cases */
290 if (movementInfo.HasMovementFlag(MOVEMENTFLAG_ONTRANSPORT))
291 {
292 // transports size limited
293 // (also received at zeppelin leave by some reason with t_* as absolute in continent coordinates, can be safely skipped)
294 if (movementInfo.transport.pos.GetPositionX() > 50 || movementInfo.transport.pos.GetPositionY() > 50 || movementInfo.transport.pos.GetPositionZ() > 50)
295 {
296 recvData.rfinish(); // prevent warnings spam
297 return;
298 }
299
300 if (!Trinity::IsValidMapCoord(movementInfo.pos.GetPositionX() + movementInfo.transport.pos.GetPositionX(), movementInfo.pos.GetPositionY() + movementInfo.transport.pos.GetPositionY(),
301 movementInfo.pos.GetPositionZ() + movementInfo.transport.pos.GetPositionZ(), movementInfo.pos.GetOrientation() + movementInfo.transport.pos.GetOrientation()))
302 {
303 recvData.rfinish(); // prevent warnings spam
304 return;
305 }
306
307 // if we boarded a transport, add us to it
308 if (plrMover)
309 {
310 if (!plrMover->GetTransport())
311 {
312 if (Transport* transport = plrMover->GetMap()->GetTransport(movementInfo.transport.guid))
313 transport->AddPassenger(plrMover);
314 }
315 else if (plrMover->GetTransport()->GetGUID() != movementInfo.transport.guid)
316 {
317 plrMover->GetTransport()->RemovePassenger(plrMover);
318 if (Transport* transport = plrMover->GetMap()->GetTransport(movementInfo.transport.guid))
319 transport->AddPassenger(plrMover);
320 else
321 movementInfo.transport.Reset();
322 }
323 }
324
325 if (!mover->GetTransport() && !mover->GetVehicle())
326 {
327 GameObject* go = mover->GetMap()->GetGameObject(movementInfo.transport.guid);
328 if (!go || go->GetGoType() != GAMEOBJECT_TYPE_TRANSPORT)
329 movementInfo.RemoveMovementFlag(MOVEMENTFLAG_ONTRANSPORT);
330 }
331 }
332 else if (plrMover && plrMover->GetTransport()) // if we were on a transport, leave
333 {
334 plrMover->GetTransport()->RemovePassenger(plrMover);
335 movementInfo.transport.Reset();
336 }
337
338 // fall damage generation (ignore in flight case that can be triggered also at lags in moment teleportation to another map).
339 if (opcode == MSG_MOVE_FALL_LAND && plrMover && !plrMover->IsInFlight())
340 plrMover->HandleFall(movementInfo);
341
342 if (plrMover && ((movementInfo.flags & MOVEMENTFLAG_SWIMMING) != 0) != plrMover->IsInWater())
343 {
344 // now client not include swimming flag in case jumping under water
345 plrMover->SetInWater(!plrMover->IsInWater() || plrMover->GetBaseMap()->IsUnderWater(movementInfo.pos.GetPositionX(), movementInfo.pos.GetPositionY(), movementInfo.pos.GetPositionZ()));
346 }
347
348 uint32 mstime = getMSTime();
349 /*----------------------*/
350 if (m_clientTimeDelay == 0)
351 m_clientTimeDelay = mstime - movementInfo.time;
352
353 /* process position-change */
354 WorldPacket data(opcode, recvData.size());
355 movementInfo.time = movementInfo.time + m_clientTimeDelay + MOVEMENT_PACKET_TIME_DELAY;
356
357 movementInfo.guid = mover->GetGUID();
358 WriteMovementInfo(&data, &movementInfo);
359 mover->SendMessageToSet(&data, _player);
360
361 mover->m_movementInfo = movementInfo;
362
363 // Some vehicles allow the passenger to turn by himself
364 if (Vehicle* vehicle = mover->GetVehicle())
365 {
366 if (VehicleSeatEntry const* seat = vehicle->GetSeatForPassenger(mover))
367 {
368 if (seat->m_flags & VEHICLE_SEAT_FLAG_ALLOW_TURNING)
369 {
370 if (movementInfo.pos.GetOrientation() != mover->GetOrientation())
371 {
372 mover->SetOrientation(movementInfo.pos.GetOrientation());
373 mover->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_TURNING);
374 }
375 }
376 }
377 return;
378 }
379
380 mover->UpdatePosition(movementInfo.pos);
381
382 if (plrMover) // nothing is charmed, or player charmed
383 {
384 if (plrMover->IsSitState() && (movementInfo.flags & (MOVEMENTFLAG_MASK_MOVING | MOVEMENTFLAG_MASK_TURNING)))
385 plrMover->SetStandState(UNIT_STAND_STATE_STAND);
386
387 plrMover->UpdateFallInformationIfNeed(movementInfo, opcode);
388
389 if (movementInfo.pos.GetPositionZ() < plrMover->GetMap()->GetMinHeight(movementInfo.pos.GetPositionX(), movementInfo.pos.GetPositionY()))
390 {
391 if (!(plrMover->GetBattleground() && plrMover->GetBattleground()->HandlePlayerUnderMap(_player)))
392 {
393 // NOTE: this is actually called many times while falling
394 // even after the player has been teleported away
395 /// @todo discard movement packets after the player is rooted
396 if (plrMover->IsAlive())
397 {
398 plrMover->SetFlag(PLAYER_FLAGS, PLAYER_FLAGS_IS_OUT_OF_BOUNDS);
399 plrMover->EnvironmentalDamage(DAMAGE_FALL_TO_VOID, GetPlayer()->GetMaxHealth());
400 // player can be alive if GM/etc
401 // change the death state to CORPSE to prevent the death timer from
402 // starting in the next player update
403 if (!plrMover->IsAlive())
404 plrMover->KillPlayer();
405 }
406 }
407 }
408 }

411void WorldSession::HandleForceSpeedChangeAck(WorldPacket &recvData)
412{
413 /* extract packet */
414 ObjectGuid guid;
415 uint32 unk1;
416 float newspeed;
417
418 recvData >> guid.ReadAsPacked();
419
420 // now can skip not our packet
421 if (_player->GetGUID() != guid)
422 {
423 recvData.rfinish(); // prevent warnings spam
424 return;
425 }
426
427 // continue parse packet
428
429 recvData >> unk1; // counter or moveEvent
430
431 MovementInfo movementInfo;
432 movementInfo.guid = guid;
433 ReadMovementInfo(recvData, &movementInfo);
434
435 recvData >> newspeed;
436 /*----------------*/
437
438 // client ACK send one packet for mounted/run case and need skip all except last from its
439 // in other cases anti-cheat check can be fail in false case
440 UnitMoveType move_type;
441 UnitMoveType force_move_type;
442
443 static char const* move_type_name[MAX_MOVE_TYPE] = { "Walk", "Run", "RunBack", "Swim", "SwimBack", "TurnRate", "Flight", "FlightBack", "PitchRate" };
444
445 switch (recvData.GetOpcode())
446 {
447 case CMSG_FORCE_WALK_SPEED_CHANGE_ACK: move_type = MOVE_WALK; force_move_type = MOVE_WALK; break;
448 case CMSG_FORCE_RUN_SPEED_CHANGE_ACK: move_type = MOVE_RUN; force_move_type = MOVE_RUN; break;
449 case CMSG_FORCE_RUN_BACK_SPEED_CHANGE_ACK: move_type = MOVE_RUN_BACK; force_move_type = MOVE_RUN_BACK; break;
450 case CMSG_FORCE_SWIM_SPEED_CHANGE_ACK: move_type = MOVE_SWIM; force_move_type = MOVE_SWIM; break;
451 case CMSG_FORCE_SWIM_BACK_SPEED_CHANGE_ACK: move_type = MOVE_SWIM_BACK; force_move_type = MOVE_SWIM_BACK; break;
452 case CMSG_FORCE_TURN_RATE_CHANGE_ACK: move_type = MOVE_TURN_RATE; force_move_type = MOVE_TURN_RATE; break;
453 case CMSG_FORCE_FLIGHT_SPEED_CHANGE_ACK: move_type = MOVE_FLIGHT; force_move_type = MOVE_FLIGHT; break;
454 case CMSG_FORCE_FLIGHT_BACK_SPEED_CHANGE_ACK: move_type = MOVE_FLIGHT_BACK; force_move_type = MOVE_FLIGHT_BACK; break;
455 case CMSG_FORCE_PITCH_RATE_CHANGE_ACK: move_type = MOVE_PITCH_RATE; force_move_type = MOVE_PITCH_RATE; break;
456 default:
457 TC_LOG_ERROR("network", "WorldSession::HandleForceSpeedChangeAck: Unknown move type opcode: %u", recvData.GetOpcode());
458 return;
459 }
460
461 // skip all forced speed changes except last and unexpected
462 // in run/mounted case used one ACK and it must be skipped.m_forced_speed_changes[MOVE_RUN} store both.
463 if (_player->m_forced_speed_changes[force_move_type] > 0)
464 {
465 --_player->m_forced_speed_changes[force_move_type];
466 if (_player->m_forced_speed_changes[force_move_type] > 0)
467 return;
468 }
469
470 if (!_player->GetTransport() && std::fabs(_player->GetSpeed(move_type) - newspeed) > 0.01f)
471 {
472 if (_player->GetSpeed(move_type) > newspeed) // must be greater - just correct
473 {
474 TC_LOG_ERROR("network", "%sSpeedChange player %s is NOT correct (must be %f instead %f), force set to correct value",
475 move_type_name[move_type], _player->GetName().c_str(), _player->GetSpeed(move_type), newspeed);
476 _player->SetSpeed(move_type, _player->GetSpeedRate(move_type), true);
477 }
478 else // must be lesser - cheating
479 {
480 TC_LOG_DEBUG("misc", "Player %s from account id %u kicked for incorrect speed (must be %f instead %f)",
481 _player->GetName().c_str(), _player->GetSession()->GetAccountId(), _player->GetSpeed(move_type), newspeed);
482 _player->GetSession()->KickPlayer();
483 }
484 }

522void WorldSession::HandleMoveKnockBackAck(WorldPacket& recvData)
523{
524 TC_LOG_DEBUG("network", "CMSG_MOVE_KNOCK_BACK_ACK");
525
526 ObjectGuid guid;
527 recvData >> guid.ReadAsPacked();
528
529 if (_player->m_mover->GetGUID() != guid)
530 return;
531
532 recvData.read_skip<uint32>(); // unk
533
534 MovementInfo movementInfo;
535 ReadMovementInfo(recvData, &movementInfo);
536
537 _player->m_movementInfo = movementInfo;
538
539 WorldPacket data(MSG_MOVE_KNOCK_BACK, 66);
540 data << guid.WriteAsPacked();
541 _player->BuildMovementPacket(&data);
542
543 // knockback specific info
544 data << movementInfo.jump.sinAngle;
545 data << movementInfo.jump.cosAngle;
546 data << movementInfo.jump.xyspeed;
547 data << movementInfo.jump.zspeed;
548
549 _player->SendMessageToSet(&data, false);
550}

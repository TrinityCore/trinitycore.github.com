35void WorldSession::HandleClientCastFlags(WorldPacket& recvPacket, uint8 castFlags, SpellCastTargets& targets)
36{
37 // some spell cast packet including more data (for projectiles?)
38 if (castFlags & 0x02)
39 {
40 // not sure about these two
41 float elevation, speed;
42 recvPacket >> elevation;
43 recvPacket >> speed;
44
45 targets.SetElevation(elevation);
46 targets.SetSpeed(speed);
47
48 uint8 hasMovementData;
49 recvPacket >> hasMovementData;
50 if (hasMovementData)
51 {
52 recvPacket.SetOpcode(recvPacket.read<uint32>());
53 HandleMovementOpcodes(recvPacket);
54 }
55 }

58void WorldSession::HandleUseItemOpcode(WorldPacket& recvPacket)
59{
60 /// @todo add targets.read() check
61 Player* pUser = _player;
62
63 // ignore for remote control state
64 if (pUser->m_mover != pUser)
65 return;
66
67 uint8 bagIndex, slot, castFlags;
68 uint8 castCount; // next cast if exists (single or not)
69 ObjectGuid itemGUID;
70 uint32 glyphIndex; // something to do with glyphs?
71 uint32 spellId; // cast spell id
72
73 recvPacket >> bagIndex >> slot >> castCount >> spellId >> itemGUID >> glyphIndex >> castFlags;
74
75 if (glyphIndex >= MAX_GLYPH_SLOT_INDEX)
76 {
77 pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
78 return;
79 }
80
81 Item* pItem = pUser->GetUseableItemByPos(bagIndex, slot);
82 if (!pItem)
83 {
84 pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
85 return;
86 }
87
88 if (pItem->GetGUID() != itemGUID)
89 {
90 pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
91 return;
92 }
93
94 TC_LOG_DEBUG("network", "WORLD: CMSG_USE_ITEM packet, bagIndex: %u, slot: %u, castCount: %u, spellId: %u, Item: %u, glyphIndex: %u, data length = %i", bagIndex, slot, castCount, spellId, pItem->GetEntry(), glyphIndex, (uint32)recvPacket.size());
95
96 ItemTemplate const* proto = pItem->GetTemplate();
97 if (!proto)
98 {
99 pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, NULL);
100 return;
101 }
102
103 // some item classes can be used only in equipped state
104 if (proto->InventoryType != INVTYPE_NON_EQUIP && !pItem->IsEquipped())
105 {
106 pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, NULL);
107 return;
108 }
109
110 InventoryResult msg = pUser->CanUseItem(pItem);
111 if (msg != EQUIP_ERR_OK)
112 {
113 pUser->SendEquipError(msg, pItem, NULL);
114 return;
115 }
116
117 // only allow conjured consumable, bandage, poisons (all should have the 2^21 item flag set in DB)
118 if (proto->Class == ITEM_CLASS_CONSUMABLE && !(proto->Flags & ITEM_PROTO_FLAG_USEABLE_IN_ARENA) && pUser->InArena())
119 {
120 pUser->SendEquipError(EQUIP_ERR_NOT_DURING_ARENA_MATCH, pItem, NULL);
121 return;
122 }
123
124 // don't allow items banned in arena
125 if (proto->Flags & ITEM_PROTO_FLAG_NOT_USEABLE_IN_ARENA && pUser->InArena())
126 {
127 pUser->SendEquipError(EQUIP_ERR_NOT_DURING_ARENA_MATCH, pItem, NULL);
128 return;
129 }
130
131 if (pUser->IsInCombat())
132 {
133 for (int i = 0; i < MAX_ITEM_PROTO_SPELLS; ++i)
134 {
135 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(proto->Spells[i].SpellId))
136 {
137 if (!spellInfo->CanBeUsedInCombat())
138 {
139 pUser->SendEquipError(EQUIP_ERR_NOT_IN_COMBAT, pItem, NULL);
140 return;
141 }
142 }
143 }
144 }
145
146 // check also BIND_WHEN_PICKED_UP and BIND_QUEST_ITEM for .additem or .additemset case by GM (not binded at adding to inventory)
147 if (pItem->GetTemplate()->Bonding == BIND_WHEN_USE || pItem->GetTemplate()->Bonding == BIND_WHEN_PICKED_UP || pItem->GetTemplate()->Bonding == BIND_QUEST_ITEM)
148 {
149 if (!pItem->IsSoulBound())
150 {
151 pItem->SetState(ITEM_CHANGED, pUser);
152 pItem->SetBinding(true);
153 }
154 }
155
156 SpellCastTargets targets;
157 targets.Read(recvPacket, pUser);
158 HandleClientCastFlags(recvPacket, castFlags, targets);
159
160 // Note: If script stop casting it must send appropriate data to client to prevent stuck item in gray state.
161 if (!sScriptMgr->OnItemUse(pUser, pItem, targets))
162 {
163 // no script or script not process request by self
164 pUser->CastItemUseSpell(pItem, targets, castCount, glyphIndex);
165 }
166}

168void WorldSession::HandleOpenItemOpcode(WorldPacket& recvPacket)
169{
170 TC_LOG_DEBUG("network", "WORLD: CMSG_OPEN_ITEM packet, data length = %i", (uint32)recvPacket.size());
171
172 Player* pUser = _player;
173
174 // ignore for remote control state
175 if (pUser->m_mover != pUser)
176 return;
177
178 uint8 bagIndex, slot;
179
180 recvPacket >> bagIndex >> slot;
181
182 TC_LOG_INFO("network", "bagIndex: %u, slot: %u", bagIndex, slot);
183
184 Item* item = pUser->GetItemByPos(bagIndex, slot);
185 if (!item)
186 {
187 pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, NULL, NULL);
188 return;
189 }
190
191 ItemTemplate const* proto = item->GetTemplate();
192 if (!proto)
193 {
194 pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, item, NULL);
195 return;
196 }
197
198 // Verify that the bag is an actual bag or wrapped item that can be used "normally"
199 if (!(proto->Flags & ITEM_PROTO_FLAG_OPENABLE) && !item->HasFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_WRAPPED))
200 {
201 pUser->SendEquipError(EQUIP_ERR_CANT_DO_RIGHT_NOW, item, NULL);
202 TC_LOG_ERROR("network", "Possible hacking attempt: Player %s [guid: %u] tried to open item [guid: %u, entry: %u] which is not openable!",
203 pUser->GetName().c_str(), pUser->GetGUID().GetCounter(), item->GetGUID().GetCounter(), proto->ItemId);
204 return;
205 }
206
207 // locked item
208 uint32 lockId = proto->LockID;
209 if (lockId)
210 {
211 LockEntry const* lockInfo = sLockStore.LookupEntry(lockId);
212
213 if (!lockInfo)
214 {
215 pUser->SendEquipError(EQUIP_ERR_ITEM_LOCKED, item, NULL);
216 TC_LOG_ERROR("network", "WORLD::OpenItem: item [guid = %u] has an unknown lockId: %u!", item->GetGUID().GetCounter(), lockId);
217 return;
218 }
219
220 // was not unlocked yet
221 if (item->IsLocked())
222 {
223 pUser->SendEquipError(EQUIP_ERR_ITEM_LOCKED, item, NULL);
224 return;
225 }
226 }
227
228 if (item->HasFlag(ITEM_FIELD_FLAGS, ITEM_FLAG_WRAPPED))// wrapped?
229 {
230 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHARACTER_GIFT_BY_ITEM);
231
232 stmt->setUInt32(0, item->GetGUID().GetCounter());
233
234 PreparedQueryResult result = CharacterDatabase.Query(stmt);
235
236 if (result)
237 {
238 Field* fields = result->Fetch();
239 uint32 entry = fields[0].GetUInt32();
240 uint32 flags = fields[1].GetUInt32();
241
242 item->SetGuidValue(ITEM_FIELD_GIFTCREATOR, ObjectGuid::Empty);
243 item->SetEntry(entry);
244 item->SetUInt32Value(ITEM_FIELD_FLAGS, flags);
245 item->SetState(ITEM_CHANGED, pUser);
246 }
247 else
248 {
249 TC_LOG_ERROR("network", "Wrapped item %u don't have record in character_gifts table and will deleted", item->GetGUID().GetCounter());
250 pUser->DestroyItem(item->GetBagSlot(), item->GetSlot(), true);
251 return;
252 }
253
254 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_GIFT);
255
256 stmt->setUInt32(0, item->GetGUID().GetCounter());
257
258 CharacterDatabase.Execute(stmt);
259 }
260 else
261 pUser->SendLoot(item->GetGUID(), LOOT_CORPSE);
262}

302void WorldSession::HandleCastSpellOpcode(WorldPacket& recvPacket)
303{
304 uint32 spellId;
305 uint8 castCount, castFlags;
306 recvPacket >> castCount >> spellId >> castFlags;
307
308 TC_LOG_DEBUG("network", "WORLD: got cast spell packet, castCount: %u, spellId: %u, castFlags: %u, data length = %u", castCount, spellId, castFlags, (uint32)recvPacket.size());
309
310 // ignore for remote control state (for player case)
311 Unit* mover = _player->m_mover;
312 if (mover != _player && mover->GetTypeId() == TYPEID_PLAYER)
313 {
314 recvPacket.rfinish(); // prevent spam at ignore packet
315 return;
316 }
317
318 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
319 if (!spellInfo)
320 {
321 TC_LOG_ERROR("network", "WORLD: unknown spell id %u", spellId);
322 recvPacket.rfinish(); // prevent spam at ignore packet
323 return;
324 }
325
326 if (spellInfo->IsPassive())
327 {
328 recvPacket.rfinish(); // prevent spam at ignore packet
329 return;
330 }
331
332 Unit* caster = mover;
333 if (caster->GetTypeId() == TYPEID_UNIT && !caster->ToCreature()->HasSpell(spellId))
334 {
335 // If the vehicle creature does not have the spell but it allows the passenger to cast own spells
336 // change caster to player and let him cast
337 if (!_player->IsOnVehicle(caster) || spellInfo->CheckVehicle(_player) != SPELL_CAST_OK)
338 {
339 recvPacket.rfinish(); // prevent spam at ignore packet
340 return;
341 }
342
343 caster = _player;
344 }
345
346 if (caster->GetTypeId() == TYPEID_PLAYER && !caster->ToPlayer()->HasActiveSpell(spellId))
347 {
348 // not have spell in spellbook
349 recvPacket.rfinish(); // prevent spam at ignore packet
350 return;
351 }
352
353 // Client is resending autoshot cast opcode when other spell is cast during shoot rotation
354 // Skip it to prevent "interrupt" message
355 if (spellInfo->IsAutoRepeatRangedSpell() && caster->GetCurrentSpell(CURRENT_AUTOREPEAT_SPELL)
356 && caster->GetCurrentSpell(CURRENT_AUTOREPEAT_SPELL)->m_spellInfo == spellInfo)
357 {
358 recvPacket.rfinish();
359 return;
360 }
361
362 // can't use our own spells when we're in possession of another unit,
363 if (_player->isPossessing())
364 {
365 recvPacket.rfinish(); // prevent spam at ignore packet
366 return;
367 }
368
369 // client provided targets
370 SpellCastTargets targets;
371 targets.Read(recvPacket, caster);
372 HandleClientCastFlags(recvPacket, castFlags, targets);
373
374 // auto-selection buff level base at target level (in spellInfo)
375 if (targets.GetUnitTarget())
376 {
377 SpellInfo const* actualSpellInfo = spellInfo->GetAuraRankForLevel(targets.GetUnitTarget()->getLevel());
378
379 // if rank not found then function return NULL but in explicit cast case original spell can be cast and later failed with appropriate error message
380 if (actualSpellInfo)
381 spellInfo = actualSpellInfo;
382 }
383
384 Spell* spell = new Spell(caster, spellInfo, TRIGGERED_NONE, ObjectGuid::Empty, false);
385 spell->m_cast_count = castCount; // set count of casts
386 spell->prepare(&targets);
387}

400void WorldSession::HandleCancelAuraOpcode(WorldPacket& recvPacket)
401{
402 uint32 spellId;
403 recvPacket >> spellId;
404
405 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
406 if (!spellInfo)
407 return;
408
409 // not allow remove spells with attr SPELL_ATTR0_CANT_CANCEL
410 if (spellInfo->HasAttribute(SPELL_ATTR0_CANT_CANCEL))
411 return;
412
413 // channeled spell case (it currently cast then)
414 if (spellInfo->IsChanneled())
415 {
416 if (Spell* curSpell = _player->GetCurrentSpell(CURRENT_CHANNELED_SPELL))
417 if (curSpell->m_spellInfo->Id == spellId)
418 _player->InterruptSpell(CURRENT_CHANNELED_SPELL);
419 return;
420 }
421
422 // non channeled case:
423 // don't allow remove non positive spells
424 // don't allow cancelling passive auras (some of them are visible)
425 if (!spellInfo->IsPositive() || spellInfo->IsPassive())
426 return;
427
428 // maybe should only remove one buff when there are multiple?
429 _player->RemoveOwnedAura(spellId, ObjectGuid::Empty, 0, AURA_REMOVE_BY_CANCEL);
430
431 // If spell being removed is a resource tracker, see if player was tracking both (herbs / minerals) and remove the other
432 if (sWorld->getBoolConfig(CONFIG_ALLOW_TRACK_BOTH_RESOURCES) && spellInfo->HasAura(SPELL_AURA_TRACK_RESOURCES))
433 {
434 Unit::AuraEffectList const& auraEffects = _player->GetAuraEffectsByType(SPELL_AURA_TRACK_RESOURCES);
435 if (!auraEffects.empty())
436 {
437 // Build list of spell IDs to cancel. Trying to cancel the aura while iterating
438 // over AuraEffectList caused "incompatible iterator" errors on second pass
439 std::list<uint32> spellIDs;
440
441 for (Unit::AuraEffectList::const_iterator auraEffect = auraEffects.begin(); auraEffect != auraEffects.end(); ++auraEffect)
442 spellIDs.push_back((*auraEffect)->GetId());
443
444 // Remove all auras related to resource tracking (only Herbs and Minerals in 3.3.5a)
445 for (std::list<uint32>::iterator it = spellIDs.begin(); it != spellIDs.end(); ++it)
446 _player->RemoveOwnedAura(*it, ObjectGuid::Empty, 0, AURA_REMOVE_BY_CANCEL);
447 }
448 }
449}

451void WorldSession::HandlePetCancelAuraOpcode(WorldPacket& recvPacket)
452{
453 ObjectGuid guid;
454 uint32 spellId;
455
456 recvPacket >> guid;
457 recvPacket >> spellId;
458
459 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
460 if (!spellInfo)
461 {
462 TC_LOG_ERROR("network", "WORLD: unknown PET spell id %u", spellId);
463 return;
464 }
465
466 Creature* pet=ObjectAccessor::GetCreatureOrPetOrVehicle(*_player, guid);
467
468 if (!pet)
469 {
470 TC_LOG_ERROR("network", "HandlePetCancelAura: Attempt to cancel an aura for non-existant %s by player '%s'", guid.ToString().c_str(), GetPlayer()->GetName().c_str());
471 return;
472 }
473
474 if (pet != GetPlayer()->GetGuardianPet() && pet != GetPlayer()->GetCharm())
475 {
476 TC_LOG_ERROR("network", "HandlePetCancelAura: %s is not a pet of player '%s'", guid.ToString().c_str(), GetPlayer()->GetName().c_str());
477 return;
478 }
479
480 if (!pet->IsAlive())
481 {
482 pet->SendPetActionFeedback(FEEDBACK_PET_DEAD);
483 return;
484 }
485
486 pet->RemoveOwnedAura(spellId, ObjectGuid::Empty, 0, AURA_REMOVE_BY_CANCEL);
487}

510void WorldSession::HandleTotemDestroyed(WorldPacket& recvPacket)
511{
512 // ignore for remote control state
513 if (_player->m_mover != _player)
514 return;
515
516 uint8 slotId;
517
518 recvPacket >> slotId;
519
520 ++slotId;
521 if (slotId >= MAX_TOTEM_SLOT)
522 return;
523
524 if (!_player->m_SummonSlot[slotId])
525 return;
526
527 Creature* totem = GetPlayer()->GetMap()->GetCreature(_player->m_SummonSlot[slotId]);
528
529 if (totem && totem->IsTotem())
530 totem->ToTotem()->UnSummon();

534{
535 TC_LOG_DEBUG("network", "WORLD: CMSG_SELF_RES"); // empty opcode
536
537 if (_player->HasAuraType(SPELL_AURA_PREVENT_RESURRECTION))
538 return; // silent return, client should display error by itself and not send this opcode
539

544 _player->CastSpell(_player, spellInfo, false, nullptr);
545
546 _player->SetUInt32Value(PLAYER_SELF_RES_SPELL, 0);
547 }

568void WorldSession::HandleMirrorImageDataRequest(WorldPacket& recvData)
569{
570 TC_LOG_DEBUG("network", "WORLD: CMSG_GET_MIRRORIMAGE_DATA");
571 ObjectGuid guid;
572 recvData >> guid;
573
574 // Get unit for which data is needed by client
575 Unit* unit = ObjectAccessor::GetUnit(*_player, guid);
576 if (!unit)
577 return;
578
579 if (!unit->HasAuraType(SPELL_AURA_CLONE_CASTER))
580 return;
581
582 // Get creator of the unit (SPELL_AURA_CLONE_CASTER does not stack)
583 Unit* creator = unit->GetAuraEffectsByType(SPELL_AURA_CLONE_CASTER).front()->GetCaster();
584 if (!creator)
585 return;
586
587 WorldPacket data(SMSG_MIRRORIMAGE_DATA, 68);
588 data << uint64(guid);
589 data << uint32(creator->GetDisplayId());
590 data << uint8(creator->getRace());
591 data << uint8(creator->getGender());
592 data << uint8(creator->getClass());
593
594 if (creator->GetTypeId() == TYPEID_PLAYER)
595 {
596 Player* player = creator->ToPlayer();
597 data << uint8(player->GetByteValue(PLAYER_BYTES, 0)); // skin
598 data << uint8(player->GetByteValue(PLAYER_BYTES, 1)); // face
599 data << uint8(player->GetByteValue(PLAYER_BYTES, 2)); // hair
600 data << uint8(player->GetByteValue(PLAYER_BYTES, 3)); // haircolor
601 data << uint8(player->GetByteValue(PLAYER_BYTES_2, 0)); // facialhair
602 data << uint32(player->GetGuildId()); // unk
603
604 static EquipmentSlots const itemSlots[] =
605 {
606 EQUIPMENT_SLOT_HEAD,
607 EQUIPMENT_SLOT_SHOULDERS,
608 EQUIPMENT_SLOT_BODY,
609 EQUIPMENT_SLOT_CHEST,
610 EQUIPMENT_SLOT_WAIST,
611 EQUIPMENT_SLOT_LEGS,
612 EQUIPMENT_SLOT_FEET,
613 EQUIPMENT_SLOT_WRISTS,
614 EQUIPMENT_SLOT_HANDS,
615 EQUIPMENT_SLOT_BACK,
616 EQUIPMENT_SLOT_TABARD,
617 EQUIPMENT_SLOT_END
618 };
619
620 // Display items in visible slots
621 for (EquipmentSlots const* itr = &itemSlots[0]; *itr != EQUIPMENT_SLOT_END; ++itr)
622 {
623 if (*itr == EQUIPMENT_SLOT_HEAD && player->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_HIDE_HELM))
624 data << uint32(0);
625 else if (*itr == EQUIPMENT_SLOT_BACK && player->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_HIDE_CLOAK))
626 data << uint32(0);
627 else if (Item const* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, *itr))
628 data << uint32(item->GetTemplate()->DisplayInfoID);
629 else
630 data << uint32(0);
631 }
632 }
633 else
634 {
635 // Skip player data for creatures
636 data << uint8(0);
637 data << uint32(0);
638 data << uint32(0);
639 data << uint32(0);
640 data << uint32(0);
641 data << uint32(0);
642 data << uint32(0);
643 data << uint32(0);
644 data << uint32(0);
645 data << uint32(0);
646 data << uint32(0);
647 data << uint32(0);
648 data << uint32(0);
649 data << uint32(0);
650 }
651
652 SendPacket(&data);
653}

655void WorldSession::HandleUpdateProjectilePosition(WorldPacket& recvPacket)
656{
657 TC_LOG_DEBUG("network", "WORLD: CMSG_UPDATE_PROJECTILE_POSITION");
658
659 ObjectGuid casterGuid;
660 uint32 spellId;
661 uint8 castCount;
662 float x, y, z; // Position of missile hit
663
664 recvPacket >> casterGuid;
665 recvPacket >> spellId;
666 recvPacket >> castCount;
667 recvPacket >> x;
668 recvPacket >> y;
669 recvPacket >> z;
670
671 Unit* caster = ObjectAccessor::GetUnit(*_player, casterGuid);
672 if (!caster)
673 return;
674
675 Spell* spell = caster->FindCurrentSpellBySpellId(spellId);
676 if (!spell || !spell->m_targets.HasDst())
677 return;
678
679 Position pos = *spell->m_targets.GetDstPos();
680 pos.Relocate(x, y, z);
681 spell->m_targets.ModDst(pos);
682
683 WorldPacket data(SMSG_SET_PROJECTILE_POSITION, 21);

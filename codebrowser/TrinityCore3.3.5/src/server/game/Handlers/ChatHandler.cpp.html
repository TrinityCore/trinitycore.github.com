42void WorldSession::HandleMessagechatOpcode(WorldPacket& recvData)
43{
44 uint32 type;
45 uint32 lang;
46
47 recvData >> type;
48 recvData >> lang;
49
50 if (type >= MAX_CHAT_MSG_TYPE)
51 {
52 TC_LOG_ERROR("network", "CHAT: Wrong message type received: %u", type);
53 recvData.rfinish();
54 return;
55 }
56
57 if (lang == LANG_UNIVERSAL && type != CHAT_MSG_AFK && type != CHAT_MSG_DND)
58 {
59 TC_LOG_ERROR("network", "CMSG_MESSAGECHAT: Possible hacking-attempt: %s tried to send a message in universal language", GetPlayerInfo().c_str());
60 SendNotification(LANG_UNKNOWN_LANGUAGE);
61 recvData.rfinish();
62 return;
63 }
64
65 Player* sender = GetPlayer();
66
67 //TC_LOG_DEBUG("CHAT: packet received. type %u, lang %u", type, lang);
68
69 // prevent talking at unknown language (cheating)
70 LanguageDesc const* langDesc = GetLanguageDescByID(lang);
71 if (!langDesc)
72 {
73 SendNotification(LANG_UNKNOWN_LANGUAGE);
74 recvData.rfinish();
75 return;
76 }
77
78 if (langDesc->skill_id != 0 && !sender->HasSkill(langDesc->skill_id))
79 {
80 // also check SPELL_AURA_COMPREHEND_LANGUAGE (client offers option to speak in that language)
81 Unit::AuraEffectList const& langAuras = sender->GetAuraEffectsByType(SPELL_AURA_COMPREHEND_LANGUAGE);
82 bool foundAura = false;
83 for (Unit::AuraEffectList::const_iterator i = langAuras.begin(); i != langAuras.end(); ++i)
84 {
85 if ((*i)->GetMiscValue() == int32(lang))
86 {
87 foundAura = true;
88 break;
89 }
90 }
91 if (!foundAura)
92 {
93 SendNotification(LANG_NOT_LEARNED_LANGUAGE);
94 recvData.rfinish();
95 return;
96 }
97 }
98
99 if (lang == LANG_ADDON)
100 {
101 // LANG_ADDON is only valid for the following message types
102 switch (type)
103 {
104 case CHAT_MSG_PARTY:
105 case CHAT_MSG_RAID:
106 case CHAT_MSG_GUILD:
107 case CHAT_MSG_BATTLEGROUND:
108 case CHAT_MSG_WHISPER:
109 // check if addon messages are disabled
110 if (!sWorld->getBoolConfig(CONFIG_ADDON_CHANNEL))
111 {
112 recvData.rfinish();
113 return;
114 }
115 break;
116 default:
117 TC_LOG_ERROR("network", "Player %s (GUID: %u) sent a chatmessage with an invalid language/message type combination",
118 GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().GetCounter());
119
120 recvData.rfinish();
121 return;
122 }
123 }
124 // LANG_ADDON should not be changed nor be affected by flood control
125 else
126 {
127 // send in universal language if player in .gmon mode (ignore spell effects)
128 if (sender->IsGameMaster())
129 lang = LANG_UNIVERSAL;
130 else
131 {
132 Unit::AuraEffectList const& ModLangAuras = sender->GetAuraEffectsByType(SPELL_AURA_MOD_LANGUAGE);
133 if (!ModLangAuras.empty())
134 lang = ModLangAuras.front()->GetMiscValue();
135 else if (HasPermission(rbac::RBAC_PERM_TWO_SIDE_INTERACTION_CHAT))
136 lang = LANG_UNIVERSAL;
137 else
138 {
139 switch (type)
140 {
141 case CHAT_MSG_PARTY:
142 case CHAT_MSG_PARTY_LEADER:
143 case CHAT_MSG_RAID:
144 case CHAT_MSG_RAID_LEADER:
145 case CHAT_MSG_RAID_WARNING:
146 // allow two side chat at group channel if two side group allowed
147 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GROUP))
148 lang = LANG_UNIVERSAL;
149 break;
150 case CHAT_MSG_GUILD:
151 case CHAT_MSG_OFFICER:
152 // allow two side chat at guild channel if two side guild allowed
153 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GUILD))
154 lang = LANG_UNIVERSAL;
155 break;
156 }
157 }
158 }
159
160 if (!sender->CanSpeak())
161 {
162 std::string timeStr = secsToTimeString(m_muteTime - time(NULL));
163 SendNotification(GetTrinityString(LANG_WAIT_BEFORE_SPEAKING), timeStr.c_str());
164 recvData.rfinish(); // Prevent warnings
165 return;
166 }
167
168 if (type != CHAT_MSG_AFK && type != CHAT_MSG_DND)
169 sender->UpdateSpeakTime();
170 }
171
172 if (sender->HasAura(1852) && type != CHAT_MSG_WHISPER)
173 {
174 SendNotification(GetTrinityString(LANG_GM_SILENCE), sender->GetName().c_str());
175 recvData.rfinish();
176 return;
177 }
178
179 std::string to, channel, msg;
180 bool ignoreChecks = false;
181 switch (type)
182 {
183 case CHAT_MSG_SAY:
184 case CHAT_MSG_EMOTE:
185 case CHAT_MSG_YELL:
186 case CHAT_MSG_PARTY:
187 case CHAT_MSG_PARTY_LEADER:
188 case CHAT_MSG_GUILD:
189 case CHAT_MSG_OFFICER:
190 case CHAT_MSG_RAID:
191 case CHAT_MSG_RAID_LEADER:
192 case CHAT_MSG_RAID_WARNING:
193 case CHAT_MSG_BATTLEGROUND:
194 case CHAT_MSG_BATTLEGROUND_LEADER:
195 recvData >> msg;
196 break;
197 case CHAT_MSG_WHISPER:
198 recvData >> to;
199 recvData >> msg;
200 break;
201 case CHAT_MSG_CHANNEL:
202 recvData >> channel;
203 recvData >> msg;
204 break;
205 case CHAT_MSG_AFK:
206 case CHAT_MSG_DND:
207 recvData >> msg;
208 ignoreChecks = true;
209 break;
210 }
211
212 if (!ignoreChecks)
213 {
214 if (msg.empty())
215 return;
216
217 if (ChatHandler(this).ParseCommands(msg.c_str()))
218 return;
219
220 if (lang != LANG_ADDON)
221 {
222 // Strip invisible characters for non-addon messages
223 if (sWorld->getBoolConfig(CONFIG_CHAT_FAKE_MESSAGE_PREVENTING))
224 stripLineInvisibleChars(msg);
225
226 if (sWorld->getIntConfig(CONFIG_CHAT_STRICT_LINK_CHECKING_SEVERITY) && !ChatHandler(this).isValidChatMessage(msg.c_str()))
227 {
228 TC_LOG_ERROR("network", "Player %s (GUID: %u) sent a chatmessage with an invalid link: %s", GetPlayer()->GetName().c_str(),
229 GetPlayer()->GetGUID().GetCounter(), msg.c_str());
230
231 if (sWorld->getIntConfig(CONFIG_CHAT_STRICT_LINK_CHECKING_KICK))
232 KickPlayer();
233
234 return;
235 }
236 }
237 }
238
239 switch (type)
240 {
241 case CHAT_MSG_SAY:
242 case CHAT_MSG_EMOTE:
243 case CHAT_MSG_YELL:
244 {
245 // Prevent cheating
246 if (!sender->IsAlive())
247 return;
248
249 if (sender->getLevel() < sWorld->getIntConfig(CONFIG_CHAT_SAY_LEVEL_REQ))
250 {
251 SendNotification(GetTrinityString(LANG_SAY_REQ), sWorld->getIntConfig(CONFIG_CHAT_SAY_LEVEL_REQ));
252 return;
253 }
254
255 if (type == CHAT_MSG_SAY)
256 sender->Say(msg, Language(lang));
257 else if (type == CHAT_MSG_EMOTE)
258 sender->TextEmote(msg);
259 else if (type == CHAT_MSG_YELL)
260 sender->Yell(msg, Language(lang));
261 } break;
262 case CHAT_MSG_WHISPER:
263 {
264 if (!normalizePlayerName(to))
265 {
266 SendPlayerNotFoundNotice(to);
267 break;
268 }
269
270 Player* receiver = ObjectAccessor::FindConnectedPlayerByName(to);
271 if (!receiver || (lang != LANG_ADDON && !receiver->isAcceptWhispers() && receiver->GetSession()->HasPermission(rbac::RBAC_PERM_CAN_FILTER_WHISPERS) && !receiver->IsInWhisperWhiteList(sender->GetGUID())))
272 {
273 SendPlayerNotFoundNotice(to);
274 return;
275 }
276 if (!sender->IsGameMaster() && sender->getLevel() < sWorld->getIntConfig(CONFIG_CHAT_WHISPER_LEVEL_REQ) && !receiver->IsInWhisperWhiteList(sender->GetGUID()))
277 {
278 SendNotification(GetTrinityString(LANG_WHISPER_REQ), sWorld->getIntConfig(CONFIG_CHAT_WHISPER_LEVEL_REQ));
279 return;
280 }
281
282 if (GetPlayer()->GetTeam() != receiver->GetTeam() && !HasPermission(rbac::RBAC_PERM_TWO_SIDE_INTERACTION_CHAT) && !receiver->IsInWhisperWhiteList(sender->GetGUID()))
283 {
284 SendWrongFactionNotice();
285 return;
286 }
287
288 if (GetPlayer()->HasAura(1852) && !receiver->IsGameMaster())
289 {
290 SendNotification(GetTrinityString(LANG_GM_SILENCE), GetPlayer()->GetName().c_str());
291 return;
292 }
293
294 // If player is a Gamemaster and doesn't accept whisper, we auto-whitelist every player that the Gamemaster is talking to
295 // We also do that if a player is under the required level for whispers.
296 if (receiver->getLevel() < sWorld->getIntConfig(CONFIG_CHAT_WHISPER_LEVEL_REQ) ||
297 (HasPermission(rbac::RBAC_PERM_CAN_FILTER_WHISPERS) && !sender->isAcceptWhispers() && !sender->IsInWhisperWhiteList(receiver->GetGUID())))
298 sender->AddWhisperWhiteList(receiver->GetGUID());
299
300 GetPlayer()->Whisper(msg, Language(lang), receiver);
301 } break;
302 case CHAT_MSG_PARTY:
303 case CHAT_MSG_PARTY_LEADER:
304 {
305 // if player is in battleground, he cannot say to battleground members by /p
306 Group* group = GetPlayer()->GetOriginalGroup();
307 if (!group)
308 {
309 group = sender->GetGroup();
310 if (!group || group->isBGGroup())
311 return;
312 }
313
314 if (type == CHAT_MSG_PARTY_LEADER && !group->IsLeader(sender->GetGUID()))
315 return;
316
317 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
318
319 WorldPacket data;
320 ChatHandler::BuildChatPacket(data, ChatMsg(type), Language(lang), sender, NULL, msg);
321 group->BroadcastPacket(&data, false, group->GetMemberGroup(GetPlayer()->GetGUID()));
322 } break;
323 case CHAT_MSG_GUILD:
324 {
325 if (GetPlayer()->GetGuildId())
326 {
327 if (Guild* guild = sGuildMgr->GetGuildById(GetPlayer()->GetGuildId()))
328 {
329 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, guild);
330
331 guild->BroadcastToGuild(this, false, msg, lang == LANG_ADDON ? LANG_ADDON : LANG_UNIVERSAL);
332 }
333 }
334 } break;
335 case CHAT_MSG_OFFICER:
336 {
337 if (GetPlayer()->GetGuildId())
338 {
339 if (Guild* guild = sGuildMgr->GetGuildById(GetPlayer()->GetGuildId()))
340 {
341 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, guild);
342
343 guild->BroadcastToGuild(this, true, msg, lang == LANG_ADDON ? LANG_ADDON : LANG_UNIVERSAL);
344 }
345 }
346 } break;
347 case CHAT_MSG_RAID:
348 {
349 // if player is in battleground, he cannot say to battleground members by /ra
350 Group* group = GetPlayer()->GetOriginalGroup();
351 if (!group)
352 {
353 group = GetPlayer()->GetGroup();
354 if (!group || group->isBGGroup() || !group->isRaidGroup())
355 return;
356 }
357
358 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
359
360 WorldPacket data;
361 ChatHandler::BuildChatPacket(data, CHAT_MSG_RAID, Language(lang), sender, NULL, msg);
362 group->BroadcastPacket(&data, false);
363 } break;
364 case CHAT_MSG_RAID_LEADER:
365 {
366 // if player is in battleground, he cannot say to battleground members by /ra
367 Group* group = GetPlayer()->GetOriginalGroup();
368 if (!group)
369 {
370 group = GetPlayer()->GetGroup();
371 if (!group || group->isBGGroup() || !group->isRaidGroup() || !group->IsLeader(sender->GetGUID()))
372 return;
373 }
374
375 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
376
377 WorldPacket data;
378 ChatHandler::BuildChatPacket(data, CHAT_MSG_RAID_LEADER, Language(lang), sender, NULL, msg);
379 group->BroadcastPacket(&data, false);
380 } break;
381 case CHAT_MSG_RAID_WARNING:
382 {
383 Group* group = GetPlayer()->GetGroup();
384 if (!group || !group->isRaidGroup() || !(group->IsLeader(GetPlayer()->GetGUID()) || group->IsAssistant(GetPlayer()->GetGUID())) || group->isBGGroup())
385 return;
386
387 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
388
389 WorldPacket data;
390 //in battleground, raid warning is sent only to players in battleground - code is ok
391 ChatHandler::BuildChatPacket(data, CHAT_MSG_RAID_WARNING, Language(lang), sender, NULL, msg);
392 group->BroadcastPacket(&data, false);
393 } break;
394 case CHAT_MSG_BATTLEGROUND:
395 {
396 //battleground raid is always in Player->GetGroup(), never in GetOriginalGroup()
397 Group* group = GetPlayer()->GetGroup();
398 if (!group || !group->isBGGroup())
399 return;
400
401 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
402
403 WorldPacket data;
404 ChatHandler::BuildChatPacket(data, CHAT_MSG_BATTLEGROUND, Language(lang), sender, NULL, msg);
405 group->BroadcastPacket(&data, false);
406 } break;
407 case CHAT_MSG_BATTLEGROUND_LEADER:
408 {
409 // battleground raid is always in Player->GetGroup(), never in GetOriginalGroup()
410 Group* group = GetPlayer()->GetGroup();
411 if (!group || !group->isBGGroup() || !group->IsLeader(GetPlayer()->GetGUID()))
412 return;
413
414 sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
415
416 WorldPacket data;
417 ChatHandler::BuildChatPacket(data, CHAT_MSG_BATTLEGROUND_LEADER, Language(lang), sender, NULL, msg);;
418 group->BroadcastPacket(&data, false);
419 } break;
420 case CHAT_MSG_CHANNEL:
421 {
422 if (!HasPermission(rbac::RBAC_PERM_SKIP_CHECK_CHAT_CHANNEL_REQ))
423 {
424 if (sender->getLevel() < sWorld->getIntConfig(CONFIG_CHAT_CHANNEL_LEVEL_REQ))
425 {
426 SendNotification(GetTrinityString(LANG_CHANNEL_REQ), sWorld->getIntConfig(CONFIG_CHAT_CHANNEL_LEVEL_REQ));
427 return;
428 }
429 }
430
431 if (ChannelMgr* cMgr = ChannelMgr::forTeam(sender->GetTeam()))
432 {
433 if (Channel* chn = cMgr->GetChannel(channel, sender))
434 {
435 sScriptMgr->OnPlayerChat(sender, type, lang, msg, chn);
436 chn->Say(sender->GetGUID(), msg.c_str(), lang);
437 }
438 }
439 } break;
440 case CHAT_MSG_AFK:
441 {
442 if (!sender->IsInCombat())
443 {
444 if (sender->isAFK()) // Already AFK
445 {
446 if (msg.empty())
447 sender->ToggleAFK(); // Remove AFK
448 else
449 sender->autoReplyMsg = msg; // Update message
450 }
451 else // New AFK mode
452 {
453 sender->autoReplyMsg = msg.empty() ? GetTrinityString(LANG_PLAYER_AFK_DEFAULT) : msg;
454
455 if (sender->isDND())
456 sender->ToggleDND();
457
458 sender->ToggleAFK();
459 }
460
461 sScriptMgr->OnPlayerChat(sender, type, lang, msg);
462 }
463 break;
464 }
465 case CHAT_MSG_DND:
466 {
467 if (sender->isDND()) // Already DND
468 {
469 if (msg.empty())
470 sender->ToggleDND(); // Remove DND
471 else
472 sender->autoReplyMsg = msg; // Update message
473 }
474 else // New DND mode
475 {
476 sender->autoReplyMsg = msg.empty() ? GetTrinityString(LANG_PLAYER_DND_DEFAULT) : msg;
477
478 if (sender->isAFK())
479 sender->ToggleAFK();
480
481 sender->ToggleDND();
482 }
483
484 sScriptMgr->OnPlayerChat(sender, type, lang, msg);
485 break;
486 }
487 default:
488 TC_LOG_ERROR("network", "CHAT: unknown message type %u, lang: %u", type, lang);
489 break;
490 }
491}

493void WorldSession::HandleEmoteOpcode(WorldPacket& recvData)
494{
495 if (!GetPlayer()->IsAlive() || GetPlayer()->HasUnitState(UNIT_STATE_DIED))
496 return;
497
498 uint32 emote;
499 recvData >> emote;
500 sScriptMgr->OnPlayerEmote(GetPlayer(), emote);
501 GetPlayer()->HandleEmoteCommand(emote);
502}

504namespace Trinity

506 class EmoteChatBuilder
507 {
508 public:
509 EmoteChatBuilder(Player const& player, uint32 text_emote, uint32 emote_num, Unit const* target)
510 : i_player(player), i_text_emote(text_emote), i_emote_num(emote_num), i_target(target) { }
511
512 void operator()(WorldPacket& data, LocaleConstant loc_idx)
513 {
514 std::string const name(i_target ? i_target->GetNameForLocaleIdx(loc_idx) : "");
515 uint32 namlen = name.size();
516
517 data.Initialize(SMSG_TEXT_EMOTE, 20 + namlen);
518 data << i_player.GetGUID();
519 data << uint32(i_text_emote);
520 data << uint32(i_emote_num);
521 data << uint32(namlen);
522 if (namlen > 1)
523 data << name;
524 else
525 data << uint8(0x00);
526 }
527
528 private:
529 Player const& i_player;
530 uint32 i_text_emote;
531 uint32 i_emote_num;
532 Unit const* i_target;
533 };
534} // namespace Trinity

536void WorldSession::HandleTextEmoteOpcode(WorldPacket& recvData)
537{
538 if (!GetPlayer()->IsAlive())
539 return;
540
541 if (!GetPlayer()->CanSpeak())
542 {
543 std::string timeStr = secsToTimeString(m_muteTime - time(NULL));
544 SendNotification(GetTrinityString(LANG_WAIT_BEFORE_SPEAKING), timeStr.c_str());
545 return;
546 }
547
548 uint32 text_emote, emoteNum;
549 ObjectGuid guid;
550
551 recvData >> text_emote;
552 recvData >> emoteNum;
553 recvData >> guid;
554
555 sScriptMgr->OnPlayerTextEmote(GetPlayer(), text_emote, emoteNum, guid);
556
557 EmotesTextEntry const* em = sEmotesTextStore.LookupEntry(text_emote);
558 if (!em)
559 return;
560
561 uint32 emote_anim = em->textid;
562
563 switch (emote_anim)
564 {
565 case EMOTE_STATE_SLEEP:
566 case EMOTE_STATE_SIT:
567 case EMOTE_STATE_KNEEL:
568 case EMOTE_ONESHOT_NONE:
569 break;
570 default:
571 // Only allow text-emotes for "dead" entities (feign death included)
572 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
573 break;
574 GetPlayer()->HandleEmoteCommand(emote_anim);
575 break;
576 }
577
578 Unit* unit = ObjectAccessor::GetUnit(*_player, guid);
579
580 CellCoord p = Trinity::ComputeCellCoord(GetPlayer()->GetPositionX(), GetPlayer()->GetPositionY());
581
582 Cell cell(p);
583 cell.SetNoCreate();
584
585 Trinity::EmoteChatBuilder emote_builder(*GetPlayer(), text_emote, emoteNum, unit);
586 Trinity::LocalizedPacketDo<Trinity::EmoteChatBuilder > emote_do(emote_builder);
587 Trinity::PlayerDistWorker<Trinity::LocalizedPacketDo<Trinity::EmoteChatBuilder > > emote_worker(GetPlayer(), sWorld->getFloatConfig(CONFIG_LISTEN_RANGE_TEXTEMOTE), emote_do);
588 TypeContainerVisitor<Trinity::PlayerDistWorker<Trinity::LocalizedPacketDo<Trinity::EmoteChatBuilder> >, WorldTypeMapContainer> message(emote_worker);
589 cell.Visit(p, message, *GetPlayer()->GetMap(), *GetPlayer(), sWorld->getFloatConfig(CONFIG_LISTEN_RANGE_TEXTEMOTE));
590
591 GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_DO_EMOTE, text_emote, 0, unit);
592
593 //Send scripted event call
594 if (unit && unit->GetTypeId() == TYPEID_UNIT && ((Creature*)unit)->AI())
595 ((Creature*)unit)->AI()->ReceiveEmote(GetPlayer(), text_emote);
596}

598void WorldSession::HandleChatIgnoredOpcode(WorldPacket& recvData)
599{
600 ObjectGuid iguid;
601 uint8 unk;
602 //TC_LOG_DEBUG("network", "WORLD: Received CMSG_CHAT_IGNORED");
603
604 recvData >> iguid;
605 recvData >> unk; // probably related to spam reporting
606
607 Player* player = ObjectAccessor::FindConnectedPlayer(iguid);
608 if (!player || !player->GetSession())
609 return;
610
611 WorldPacket data;
612 ChatHandler::BuildChatPacket(data, CHAT_MSG_IGNORED, LANG_UNIVERSAL, _player, _player, GetPlayer()->GetName());
613 player->GetSession()->SendPacket(&data);
614}

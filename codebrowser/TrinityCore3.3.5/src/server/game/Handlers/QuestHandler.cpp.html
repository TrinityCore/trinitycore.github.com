34void WorldSession::HandleQuestgiverStatusQueryOpcode(WorldPacket& recvData)
35{
36 ObjectGuid guid;
37 recvData >> guid;
38 uint32 questStatus = DIALOG_STATUS_NONE;
39
40 Object* questGiver = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
41 if (!questGiver)
42 {
43 TC_LOG_INFO("network", "Error in CMSG_QUESTGIVER_STATUS_QUERY, called for non-existing questgiver (%s)", guid.ToString().c_str());
44 return;
45 }
46
47 switch (questGiver->GetTypeId())
48 {
49 case TYPEID_UNIT:
50 {
51 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_STATUS_QUERY for npc, guid = %u", questGiver->GetGUID().GetCounter());
52 if (!questGiver->ToCreature()->IsHostileTo(_player)) // do not show quest status to enemies
53 questStatus = _player->GetQuestDialogStatus(questGiver);
54 break;
55 }
56 case TYPEID_GAMEOBJECT:
57 {
58 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_STATUS_QUERY for GameObject guid = %u", questGiver->GetGUID().GetCounter());
59 questStatus = _player->GetQuestDialogStatus(questGiver);
60 break;
61 }
62 default:
63 TC_LOG_ERROR("network", "QuestGiver called for unexpected type %u", questGiver->GetTypeId());
64 break;
65 }
66
67 // inform client about status of quest
68 _player->PlayerTalkClass->SendQuestGiverStatus(uint8(questStatus), guid);
69}

71void WorldSession::HandleQuestgiverHelloOpcode(WorldPacket& recvData)
72{
73 ObjectGuid guid;
74 recvData >> guid;
75
76 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_HELLO %s", guid.ToString().c_str());
77
78 Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_QUESTGIVER);
79 if (!creature)
80 {
81 TC_LOG_DEBUG("network", "WORLD: HandleQuestgiverHelloOpcode - %s not found or you can't interact with him.",
82 guid.ToString().c_str());
83 return;
84 }
85
86 // remove fake death
87 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
88 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
89 // Stop the npc if moving
90 creature->StopMoving();
91
92 if (sScriptMgr->OnGossipHello(_player, creature))
93 return;
94
95 _player->PrepareGossipMenu(creature, creature->GetCreatureTemplate()->GossipMenuId, true);
96 _player->SendPreparedGossip(creature);
97
98 creature->AI()->sGossipHello(_player);
99}

101void WorldSession::HandleQuestgiverAcceptQuestOpcode(WorldPacket& recvData)
102{
103 ObjectGuid guid;
104 uint32 questId;
105 uint32 startCheat;
106 recvData >> guid >> questId >> startCheat;
107
108 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_ACCEPT_QUEST %s, quest = %u, startCheat = %u", guid.ToString().c_str(), questId, startCheat);
109
110 Object* object;
111 if (!guid.IsPlayer())
112 object = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT | TYPEMASK_ITEM);
113 else
114 object = ObjectAccessor::FindPlayer(guid);
115
116#define CLOSE_GOSSIP_CLEAR_DIVIDER() \
117 do { \
118 _player->PlayerTalkClass->SendCloseGossip(); \
119 _player->SetDivider(ObjectGuid::Empty); \
120 } while (0)
121
122 // no or incorrect quest giver
123 if (!object)
124 {
125 CLOSE_GOSSIP_CLEAR_DIVIDER();
126 return;
127 }
128
129 if (Player* playerQuestObject = object->ToPlayer())
130 {
131 if ((_player->GetDivider() && _player->GetDivider() != guid) || !playerQuestObject->CanShareQuest(questId))
132 {
133 CLOSE_GOSSIP_CLEAR_DIVIDER();
134 return;
135 }
136 if (!_player->IsInSameRaidWith(playerQuestObject))
137 {
138 CLOSE_GOSSIP_CLEAR_DIVIDER();
139 return;
140 }
141 }
142 else
143 {
144 if (!object->hasQuest(questId))
145 {
146 CLOSE_GOSSIP_CLEAR_DIVIDER();
147 return;
148 }
149 }
150
151 // some kind of WPE protection
152 if (!_player->CanInteractWithQuestGiver(object))
153 {
154 CLOSE_GOSSIP_CLEAR_DIVIDER();
155 return;
156 }
157
158 if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
159 {
160 // prevent cheating
161 if (!GetPlayer()->CanTakeQuest(quest, true))
162 {
163 CLOSE_GOSSIP_CLEAR_DIVIDER();
164 return;
165 }
166
167 if (!_player->GetDivider().IsEmpty())
168 {
169 Player* player = ObjectAccessor::FindPlayer(_player->GetDivider());
170 if (player)
171 {
172 player->SendPushToPartyResponse(_player, QUEST_PARTY_MSG_ACCEPT_QUEST);
173 _player->SetDivider(ObjectGuid::Empty);
174 }
175 }
176
177 if (_player->CanAddQuest(quest, true))
178 {
179 _player->AddQuestAndCheckCompletion(quest, object);
180
181 if (quest->HasFlag(QUEST_FLAGS_PARTY_ACCEPT))
182 {
183 if (Group* group = _player->GetGroup())
184 {
185 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
186 {
187 Player* player = itr->GetSource();
188
189 if (!player || player == _player) // not self
190 continue;
191
192 if (player->CanTakeQuest(quest, true))
193 {
194 player->SetDivider(_player->GetGUID());
195
196 // need confirmation that any gossip window will close
197 player->PlayerTalkClass->SendCloseGossip();
198
199 _player->SendQuestConfirmAccept(quest, player);
200 }
201 }
202 }
203 }
204
205 _player->PlayerTalkClass->SendCloseGossip();
206
207 if (quest->GetSrcSpell() > 0)
208 _player->CastSpell(_player, quest->GetSrcSpell(), true);
209
210 return;
211 }
212 }
213
214 CLOSE_GOSSIP_CLEAR_DIVIDER();
215
216#undef CLOSE_GOSSIP_CLEAR_DIVIDER
217}

219void WorldSession::HandleQuestgiverQueryQuestOpcode(WorldPacket& recvData)
220{
221 ObjectGuid guid;
222 uint32 questId;
223 uint8 unk1;
224 recvData >> guid >> questId >> unk1;
225 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_QUERY_QUEST npc = %s, quest = %u, unk1 = %u", guid.ToString().c_str(), questId, unk1);
226
227 // Verify that the guid is valid and is a questgiver or involved in the requested quest
228 Object* object = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT | TYPEMASK_ITEM);
229 if (!object || (!object->hasQuest(questId) && !object->hasInvolvedQuest(questId)))
230 {
231 _player->PlayerTalkClass->SendCloseGossip();
232 return;
233 }
234
235 if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
236 {
237 // not sure here what should happen to quests with QUEST_FLAGS_AUTOCOMPLETE
238 // if this breaks them, add && object->GetTypeId() == TYPEID_ITEM to this check
239 // item-started quests never have that flag
240 if (!_player->CanTakeQuest(quest, true))
241 return;
242
243 if (quest->IsAutoAccept() && _player->CanAddQuest(quest, true))
244 _player->AddQuestAndCheckCompletion(quest, object);
245
246 if (quest->HasFlag(QUEST_FLAGS_AUTOCOMPLETE))
247 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, object->GetGUID(), _player->CanCompleteQuest(quest->GetQuestId()), true);
248 else
249 _player->PlayerTalkClass->SendQuestGiverQuestDetails(quest, object->GetGUID(), true);
250 }

266void WorldSession::HandleQuestgiverChooseRewardOpcode(WorldPacket& recvData)
267{
268 uint32 questId, reward;
269 ObjectGuid guid;
270 recvData >> guid >> questId >> reward;
271
272 if (reward >= QUEST_REWARD_CHOICES_COUNT)
273 {
274 TC_LOG_ERROR("network", "Error in CMSG_QUESTGIVER_CHOOSE_REWARD: player %s (guid %d) tried to get invalid reward (%u) (possible packet-hacking detected)", _player->GetName().c_str(), _player->GetGUID().GetCounter(), reward);
275 return;
276 }
277
278 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_CHOOSE_REWARD npc = %s, quest = %u, reward = %u", guid.ToString().c_str(), questId, reward);
279
280 Object* object = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
281 if (!object || !object->hasInvolvedQuest(questId))
282 return;
283
284 // some kind of WPE protection
285 if (!_player->CanInteractWithQuestGiver(object))
286 return;
287
288 if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
289 {
290 if ((!_player->CanSeeStartQuest(quest) && _player->GetQuestStatus(questId) == QUEST_STATUS_NONE) ||
291 (_player->GetQuestStatus(questId) != QUEST_STATUS_COMPLETE && !quest->IsAutoComplete()))
292 {
293 TC_LOG_ERROR("network", "Error in QUEST_STATUS_COMPLETE: player %s (guid %u) tried to complete quest %u, but is not allowed to do so (possible packet-hacking or high latency)",
294 _player->GetName().c_str(), _player->GetGUID().GetCounter(), questId);
295 return;
296 }
297 if (_player->CanRewardQuest(quest, reward, true))
298 {
299 _player->RewardQuest(quest, reward, object);
300
301 switch (object->GetTypeId())
302 {
303 case TYPEID_UNIT:
304 {
305 Creature* questgiver = object->ToCreature();
306 if (!sScriptMgr->OnQuestReward(_player, questgiver, quest, reward))
307 {
308 // Send next quest
309 if (Quest const* nextQuest = _player->GetNextQuest(guid, quest))
310 {
311 // Only send the quest to the player if the conditions are met
312 if (_player->CanTakeQuest(nextQuest, false))
313 {
314 if (nextQuest->IsAutoAccept() && _player->CanAddQuest(nextQuest, true))
315 _player->AddQuestAndCheckCompletion(nextQuest, object);
316
317 _player->PlayerTalkClass->SendQuestGiverQuestDetails(nextQuest, guid, true);
318 }
319 }
320
321 questgiver->AI()->sQuestReward(_player, quest, reward);
322 }
323 break;
324 }
325 case TYPEID_GAMEOBJECT:
326 {
327 GameObject* questGiver = object->ToGameObject();
328 if (!sScriptMgr->OnQuestReward(_player, questGiver, quest, reward))
329 {
330 // Send next quest
331 if (Quest const* nextQuest = _player->GetNextQuest(guid, quest))
332 {
333 // Only send the quest to the player if the conditions are met
334 if (_player->CanTakeQuest(nextQuest, false))
335 {
336 if (nextQuest->IsAutoAccept() && _player->CanAddQuest(nextQuest, true))
337 _player->AddQuestAndCheckCompletion(nextQuest, object);
338
339 _player->PlayerTalkClass->SendQuestGiverQuestDetails(nextQuest, guid, true);
340 }
341 }
342
343 questGiver->AI()->QuestReward(_player, quest, reward);
344 }
345 break;
346 }
347 default:
348 break;
349 }
350 }
351 else
352 _player->PlayerTalkClass->SendQuestGiverOfferReward(quest, guid, true);
353 }
354}

402void WorldSession::HandleQuestLogRemoveQuest(WorldPacket& recvData)
403{
404 uint8 slot;
405 recvData >> slot;
406
407 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTLOG_REMOVE_QUEST slot = %u", slot);
408
409 if (slot < MAX_QUEST_LOG_SIZE)
410 {
411 if (uint32 questId = _player->GetQuestSlotQuestId(slot))
412 {
413 if (!_player->TakeQuestSourceItem(questId, true))
414 return; // can't un-equip some items, reject quest cancel
415
416 if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
417 {
418 if (quest->HasSpecialFlag(QUEST_SPECIAL_FLAGS_TIMED))
419 _player->RemoveTimedQuest(questId);
420
421 if (quest->HasFlag(QUEST_FLAGS_FLAGS_PVP))
422 {
423 _player->pvpInfo.IsHostile = _player->pvpInfo.IsInHostileArea || _player->HasPvPForcingQuest();
424 _player->UpdatePvPState();
425 }
426 }
427
428 _player->TakeQuestSourceItem(questId, true); // remove quest src item from player
429 _player->RemoveActiveQuest(questId);
430 _player->RemoveTimedAchievement(ACHIEVEMENT_TIMED_TYPE_QUEST, questId);
431
432 TC_LOG_INFO("network", "Player %u abandoned quest %u", _player->GetGUID().GetCounter(), questId);
433
434 if (sWorld->getBoolConfig(CONFIG_QUEST_ENABLE_QUEST_TRACKER)) // check if Quest Tracker is enabled
435 {
436 // prepare Quest Tracker datas
437 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_QUEST_TRACK_ABANDON_TIME);
438 stmt->setUInt32(0, questId);
439 stmt->setUInt32(1, _player->GetGUID().GetCounter());
440
441 // add to Quest Tracker
442 CharacterDatabase.Execute(stmt);
443 }
444 }
445
446 _player->SetQuestSlot(slot, 0);
447
448 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_QUEST_ABANDONED, 1);
449 }

452void WorldSession::HandleQuestConfirmAccept(WorldPacket& recvData)
453{
454 uint32 questId;
455 recvData >> questId;
456
457 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUEST_CONFIRM_ACCEPT quest = %u", questId);
458
459 if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
460 {
461 if (!quest->HasFlag(QUEST_FLAGS_PARTY_ACCEPT))
462 return;
463
464 Player* originalPlayer = ObjectAccessor::FindPlayer(_player->GetDivider());
465 if (!originalPlayer)
466 return;
467
468 if (!_player->IsInSameRaidWith(originalPlayer))
469 return;
470
471 if (!originalPlayer->IsActiveQuest(questId))
472 return;
473
474 if (!_player->CanTakeQuest(quest, true))
475 return;
476
477 if (_player->CanAddQuest(quest, true))
478 {
479 _player->AddQuestAndCheckCompletion(quest, NULL); // NULL, this prevent DB script from duplicate running
480
481 if (quest->GetSrcSpell() > 0)
482 _player->CastSpell(_player, quest->GetSrcSpell(), true);
483 }
484 }
485
486 _player->SetDivider(ObjectGuid::Empty);
487}

489void WorldSession::HandleQuestgiverCompleteQuest(WorldPacket& recvData)
490{
491 uint32 questId;
492 ObjectGuid guid;
493
494 recvData >> guid >> questId;
495
496 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_COMPLETE_QUEST npc = %s, quest = %u", guid.ToString().c_str(), questId);
497
498 Quest const* quest = sObjectMgr->GetQuestTemplate(questId);
499 if (!quest)
500 return;
501
502 Object* object = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
503 if (!object || !object->hasInvolvedQuest(questId))
504 return;
505
506 // some kind of WPE protection
507 if (!_player->CanInteractWithQuestGiver(object))
508 return;
509
510 if (!_player->CanSeeStartQuest(quest) && _player->GetQuestStatus(questId) == QUEST_STATUS_NONE)
511 {
512 TC_LOG_ERROR("network", "Possible hacking attempt: Player %s [guid: %u] tried to complete quest [entry: %u] without being in possession of the quest!",
513 _player->GetName().c_str(), _player->GetGUID().GetCounter(), questId);
514 return;
515 }
516
517 if (Battleground* bg = _player->GetBattleground())
518 bg->HandleQuestComplete(questId, _player);
519
520 if (_player->GetQuestStatus(questId) != QUEST_STATUS_COMPLETE)
521 {
522 if (quest->IsRepeatable())
523 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, guid, _player->CanCompleteRepeatableQuest(quest), false);
524 else
525 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, guid, _player->CanRewardQuest(quest, false), false);
526 }
527 else
528 {
529 if (quest->GetReqItemsCount()) // some items required
530 _player->PlayerTalkClass->SendQuestGiverRequestItems(quest, guid, _player->CanRewardQuest(quest, false), false);
531 else // no items required
532 _player->PlayerTalkClass->SendQuestGiverOfferReward(quest, guid, true);
533 }
534}

541void WorldSession::HandlePushQuestToParty(WorldPacket& recvPacket)
542{
543 uint32 questId;
544 recvPacket >> questId;
545
546 if (!_player->CanShareQuest(questId))
547 return;
548
549 TC_LOG_DEBUG("network", "WORLD: Received CMSG_PUSHQUESTTOPARTY questId = %u", questId);
550
551 Quest const* quest = sObjectMgr->GetQuestTemplate(questId);
552 if (!quest)
553 return;
554
555 Player * const sender = GetPlayer();
556
557 Group* group = sender->GetGroup();
558 if (!group)
559 return;
560
561 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
562 {
563 Player* receiver = itr->GetSource();
564
565 if (!receiver || receiver == sender)
566 continue;
567
568 if (!receiver->SatisfyQuestStatus(quest, false))
569 {
570 sender->SendPushToPartyResponse(receiver, QUEST_PARTY_MSG_HAVE_QUEST);
571 continue;
572 }
573
574 if (receiver->GetQuestStatus(questId) == QUEST_STATUS_COMPLETE)
575 {
576 sender->SendPushToPartyResponse(receiver, QUEST_PARTY_MSG_FINISH_QUEST);
577 continue;
578 }
579
580 if (!receiver->CanTakeQuest(quest, false))
581 {
582 sender->SendPushToPartyResponse(receiver, QUEST_PARTY_MSG_CANT_TAKE_QUEST);
583 continue;
584 }
585
586 if (!receiver->SatisfyQuestLog(false))
587 {
588 sender->SendPushToPartyResponse(receiver, QUEST_PARTY_MSG_LOG_FULL);
589 continue;
590 }
591
592 if (receiver->GetDivider())
593 {
594 sender->SendPushToPartyResponse(receiver, QUEST_PARTY_MSG_BUSY);
595 continue;
596 }
597
598 sender->SendPushToPartyResponse(receiver, QUEST_PARTY_MSG_SHARING_QUEST);
599
600 if (quest->IsAutoAccept() && receiver->CanAddQuest(quest, true) && receiver->CanTakeQuest(quest, true))
601 receiver->AddQuestAndCheckCompletion(quest, sender);
602
603 if ((quest->IsAutoComplete() && quest->IsRepeatable() && !quest->IsDailyOrWeekly()) || quest->HasFlag(QUEST_FLAGS_AUTOCOMPLETE))
604 receiver->PlayerTalkClass->SendQuestGiverRequestItems(quest, sender->GetGUID(), receiver->CanCompleteRepeatableQuest(quest), true);
605 else
606 {
607 receiver->SetDivider(sender->GetGUID());
608 receiver->PlayerTalkClass->SendQuestGiverQuestDetails(quest, receiver->GetGUID(), true);
609 }
610 }
611}

613void WorldSession::HandleQuestPushResult(WorldPacket& recvPacket)
614{
615 ObjectGuid guid;
616 uint32 questId;
617 uint8 msg;
618 recvPacket >> guid >> questId >> msg;
619
620 TC_LOG_DEBUG("network", "WORLD: Received MSG_QUEST_PUSH_RESULT");

623 {

632 player->SendDirectMessage(&data);
633 }
634 }
635 _player->SetDivider(ObjectGuid::Empty);
636 }

639void WorldSession::HandleQuestgiverStatusMultipleQuery(WorldPacket& /*recvPacket*/)
640{
641 TC_LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_STATUS_MULTIPLE_QUERY");
642
643 uint32 count = 0;
644
645 WorldPacket data(SMSG_QUESTGIVER_STATUS_MULTIPLE, 4);
646 data << uint32(count); // placeholder
647
648 for (auto itr = _player->m_clientGUIDs.begin(); itr != _player->m_clientGUIDs.end(); ++itr)
649 {
650 uint32 questStatus = DIALOG_STATUS_NONE;
651
652 if (itr->IsAnyTypeCreature())
653 {
654 // need also pet quests case support
655 Creature* questgiver = ObjectAccessor::GetCreatureOrPetOrVehicle(*GetPlayer(), *itr);
656 if (!questgiver || questgiver->IsHostileTo(_player))
657 continue;
658 if (!questgiver->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER))
659 continue;
660
661 questStatus = _player->GetQuestDialogStatus(questgiver);
662
663 data << uint64(questgiver->GetGUID());
664 data << uint8(questStatus);
665 ++count;
666 }
667 else if (itr->IsGameObject())
668 {
669 GameObject* questgiver = GetPlayer()->GetMap()->GetGameObject(*itr);
670 if (!questgiver || questgiver->GetGoType() != GAMEOBJECT_TYPE_QUESTGIVER)
671 continue;
672
673 questStatus = _player->GetQuestDialogStatus(questgiver);
674
675 data << uint64(questgiver->GetGUID());
676 data << uint8(questStatus);
677 ++count;
678 }
679 }
680
681 data.put<uint32>(0, count); // write real count
682 SendPacket(&data);
683}

685void WorldSession::HandleQueryQuestsCompleted(WorldPacket & /*recvData*/)
686{
687 size_t rew_count = _player->GetRewardedQuestCount();
688
689 WorldPacket data(SMSG_QUERY_QUESTS_COMPLETED_RESPONSE, 4 + 4 * rew_count);

691
692 const RewardedQuestSet &rewQuests = _player->getRewardedQuests();
693 for (RewardedQuestSet::const_iterator itr = rewQuests.begin(); itr != rewQuests.end(); ++itr)
694 data << uint32(*itr);
695
696 SendPacket(&data);

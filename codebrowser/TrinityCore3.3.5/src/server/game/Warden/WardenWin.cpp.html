40void WardenWin::Init(WorldSession* session, BigNumber* k)
41{
42 _session = session;
43 // Generate Warden Key
44 SHA1Randx WK(k->AsByteArray().get(), k->GetNumBytes());
45 WK.Generate(_inputKey, 16);
46 WK.Generate(_outputKey, 16);
47
48 memcpy(_seed, Module.Seed, 16);
49
50 _inputCrypto.Init(_inputKey);
51 _outputCrypto.Init(_outputKey);
52 TC_LOG_DEBUG("warden", "Server side warden for client %u initializing...", session->GetAccountId());
53 TC_LOG_DEBUG("warden", "C->S Key: %s", ByteArrayToHexStr(_inputKey, 16).c_str());
54 TC_LOG_DEBUG("warden", "S->C Key: %s", ByteArrayToHexStr(_outputKey, 16).c_str());
55 TC_LOG_DEBUG("warden", " Seed: %s", ByteArrayToHexStr(_seed, 16).c_str());
56 TC_LOG_DEBUG("warden", "Loading Module...");
57
58 _module = GetModuleForClient();
59
60 TC_LOG_DEBUG("warden", "Module Key: %s", ByteArrayToHexStr(_module->Key, 16).c_str());
61 TC_LOG_DEBUG("warden", "Module ID: %s", ByteArrayToHexStr(_module->Id, 16).c_str());
62 RequestModule();
63}

65ClientWardenModule* WardenWin::GetModuleForClient()
66{
67 ClientWardenModule *mod = new ClientWardenModule;
68
69 uint32 length = sizeof(Module.Module);
70
71 // data assign
72 mod->CompressedSize = length;
73 mod->CompressedData = new uint8[length];
74 memcpy(mod->CompressedData, Module.Module, length);
75 memcpy(mod->Key, Module.ModuleKey, 16);
76
77 // md5 hash
78 MD5_CTX ctx;
79 MD5_Init(&ctx);
80 MD5_Update(&ctx, mod->CompressedData, length);
81 MD5_Final((uint8*)&mod->Id, &ctx);
82
83 return mod;
84}

86void WardenWin::InitializeModule()
87{
88 TC_LOG_DEBUG("warden", "Initialize module");
89
90 // Create packet structure
91 WardenInitModuleRequest Request;
92 Request.Command1 = WARDEN_SMSG_MODULE_INITIALIZE;
93 Request.Size1 = 20;
94 Request.Unk1 = 1;
95 Request.Unk2 = 0;
96 Request.Type = 1;
97 Request.String_library1 = 0;
98 Request.Function1[0] = 0x00024F80; // 0x00400000 + 0x00024F80 SFileOpenFile
99 Request.Function1[1] = 0x000218C0; // 0x00400000 + 0x000218C0 SFileGetFileSize
100 Request.Function1[2] = 0x00022530; // 0x00400000 + 0x00022530 SFileReadFile
101 Request.Function1[3] = 0x00022910; // 0x00400000 + 0x00022910 SFileCloseFile
102 Request.CheckSumm1 = BuildChecksum(&Request.Unk1, 20);
103
104 Request.Command2 = WARDEN_SMSG_MODULE_INITIALIZE;
105 Request.Size2 = 8;
106 Request.Unk3 = 4;
107 Request.Unk4 = 0;
108 Request.String_library2 = 0;
109 Request.Function2 = 0x00419D40; // 0x00400000 + 0x00419D40 FrameScript::GetText
110 Request.Function2_set = 1;
111 Request.CheckSumm2 = BuildChecksum(&Request.Unk2, 8);
112
113 Request.Command3 = WARDEN_SMSG_MODULE_INITIALIZE;
114 Request.Size3 = 8;
115 Request.Unk5 = 1;
116 Request.Unk6 = 1;
117 Request.String_library3 = 0;
118 Request.Function3 = 0x0046AE20; // 0x00400000 + 0x0046AE20 PerformanceCounter
119 Request.Function3_set = 1;
120 Request.CheckSumm3 = BuildChecksum(&Request.Unk5, 8);
121
122 // Encrypt with warden RC4 key.
123 EncryptData((uint8*)&Request, sizeof(WardenInitModuleRequest));
124
125 WorldPacket pkt(SMSG_WARDEN_DATA, sizeof(WardenInitModuleRequest));
126 pkt.append((uint8*)&Request, sizeof(WardenInitModuleRequest));
127 _session->SendPacket(&pkt);
128}

147void WardenWin::HandleHashResult(ByteBuffer &buff)
148{
149 buff.rpos(buff.wpos());
150
151 // Verify key
152 if (memcmp(buff.contents() + 1, Module.ClientKeySeedHash, 20) != 0)
153 {
154 TC_LOG_WARN("warden", "%s failed hash reply. Action: %s", _session->GetPlayerInfo().c_str(), Penalty().c_str());
155 return;
156 }
157
158 TC_LOG_DEBUG("warden", "Request hash reply: succeed");
159
160 // Change keys here
161 memcpy(_inputKey, Module.ClientKeySeed, 16);
162 memcpy(_outputKey, Module.ServerKeySeed, 16);
163
164 _inputCrypto.Init(_inputKey);
165 _outputCrypto.Init(_outputKey);
166
167 _initialized = true;
168
169 _previousTimestamp = getMSTime();
170}

172void WardenWin::RequestData()
173{
174 TC_LOG_DEBUG("warden", "Request data");
175
176 // If all checks were done, fill the todo list again
177 if (_memChecksTodo.empty())
178 _memChecksTodo.assign(sWardenCheckMgr->MemChecksIdPool.begin(), sWardenCheckMgr->MemChecksIdPool.end());
179
180 if (_otherChecksTodo.empty())
181 _otherChecksTodo.assign(sWardenCheckMgr->OtherChecksIdPool.begin(), sWardenCheckMgr->OtherChecksIdPool.end());
182
183 _serverTicks = getMSTime();
184
185 uint16 id;
186 uint8 type;
187 WardenCheck* wd;
188 _currentChecks.clear();
189
190 // Build check request
191 for (uint32 i = 0; i < sWorld->getIntConfig(CONFIG_WARDEN_NUM_MEM_CHECKS); ++i)
192 {
193 // If todo list is done break loop (will be filled on next Update() run)
194 if (_memChecksTodo.empty())
195 break;
196
197 // Get check id from the end and remove it from todo
198 id = _memChecksTodo.back();
199 _memChecksTodo.pop_back();
200
201 // Add the id to the list sent in this cycle
202 _currentChecks.push_back(id);
203 }
204
205 ByteBuffer buff;
206 buff << uint8(WARDEN_SMSG_CHEAT_CHECKS_REQUEST);
207
208 boost::shared_lock<boost::shared_mutex> lock(sWardenCheckMgr->_checkStoreLock);
209
210 for (uint32 i = 0; i < sWorld->getIntConfig(CONFIG_WARDEN_NUM_OTHER_CHECKS); ++i)
211 {
212 // If todo list is done break loop (will be filled on next Update() run)
213 if (_otherChecksTodo.empty())
214 break;
215
216 // Get check id from the end and remove it from todo
217 id = _otherChecksTodo.back();
218 _otherChecksTodo.pop_back();
219
220 // Add the id to the list sent in this cycle
221 _currentChecks.push_back(id);
222
223 wd = sWardenCheckMgr->GetWardenDataById(id);
224
225 switch (wd->Type)
226 {
227 case MPQ_CHECK:
228 case LUA_STR_CHECK:
229 case DRIVER_CHECK:
230 buff << uint8(wd->Str.size());
231 buff.append(wd->Str.c_str(), wd->Str.size());
232 break;
233 default:
234 break;
235 }
236 }
237
238 uint8 xorByte = _inputKey[0];
239
240 // Add TIMING_CHECK
241 buff << uint8(0x00);
242 buff << uint8(TIMING_CHECK ^ xorByte);
243
244 uint8 index = 1;
245
246 for (std::list<uint16>::iterator itr = _currentChecks.begin(); itr != _currentChecks.end(); ++itr)
247 {
248 wd = sWardenCheckMgr->GetWardenDataById(*itr);
249
250 type = wd->Type;
251 buff << uint8(type ^ xorByte);
252 switch (type)
253 {
254 case MEM_CHECK:
255 {
256 buff << uint8(0x00);
257 buff << uint32(wd->Address);
258 buff << uint8(wd->Length);
259 break;
260 }
261 case PAGE_CHECK_A:
262 case PAGE_CHECK_B:
263 {
264 buff.append(wd->Data.AsByteArray(0, false).get(), wd->Data.GetNumBytes());
265 buff << uint32(wd->Address);
266 buff << uint8(wd->Length);
267 break;
268 }
269 case MPQ_CHECK:
270 case LUA_STR_CHECK:
271 {
272 buff << uint8(index++);
273 break;
274 }
275 case DRIVER_CHECK:
276 {
277 buff.append(wd->Data.AsByteArray(0, false).get(), wd->Data.GetNumBytes());
278 buff << uint8(index++);
279 break;
280 }
281 case MODULE_CHECK:
282 {
283 uint32 seed = rand32();
284 buff << uint32(seed);
285 HmacHash hmac(4, (uint8*)&seed);
286 hmac.UpdateData(wd->Str);
287 hmac.Finalize();
288 buff.append(hmac.GetDigest(), hmac.GetLength());
289 break;
290 }
291 /*case PROC_CHECK:
292 {
293 buff.append(wd->i.AsByteArray(0, false).get(), wd->i.GetNumBytes());
294 buff << uint8(index++);
295 buff << uint8(index++);
296 buff << uint32(wd->Address);
297 buff << uint8(wd->Length);
298 break;
299 }*/
300 default:
301 break; // Should never happen
302 }
303 }
304 buff << uint8(xorByte);
305 buff.hexlike();
306
307 // Encrypt with warden RC4 key
308 EncryptData(buff.contents(), buff.size());
309
310 WorldPacket pkt(SMSG_WARDEN_DATA, buff.size());
311 pkt.append(buff);
312 _session->SendPacket(&pkt);
313
314 _dataSent = true;
315
316 std::stringstream stream;
317 stream << "Sent check id's: ";
318 for (std::list<uint16>::iterator itr = _currentChecks.begin(); itr != _currentChecks.end(); ++itr)
319 stream << *itr << " ";
320
321 TC_LOG_DEBUG("warden", "%s", stream.str().c_str());
322}

324void WardenWin::HandleData(ByteBuffer &buff)
325{
326 TC_LOG_DEBUG("warden", "Handle data");
327
328 _dataSent = false;
329 _clientResponseTimer = 0;
330
331 uint16 Length;
332 buff >> Length;
333 uint32 Checksum;
334 buff >> Checksum;
335
336 if (!IsValidCheckSum(Checksum, buff.contents() + buff.rpos(), Length))
337 {
338 buff.rpos(buff.wpos());
339 TC_LOG_WARN("warden", "%s failed checksum. Action: %s", _session->GetPlayerInfo().c_str(), Penalty().c_str());
340 return;
341 }
342
343 // TIMING_CHECK
344 {
345 uint8 result;
346 buff >> result;
347 /// @todo test it.
348 if (result == 0x00)
349 {
350 TC_LOG_WARN("warden", "%s failed timing check. Action: %s", _session->GetPlayerInfo().c_str(), Penalty().c_str());
351 return;
352 }
353
354 uint32 newClientTicks;
355 buff >> newClientTicks;
356
357 uint32 ticksNow = getMSTime();
358 uint32 ourTicks = newClientTicks + (ticksNow - _serverTicks);
359
360 TC_LOG_DEBUG("warden", "ServerTicks %u", ticksNow); // Now
361 TC_LOG_DEBUG("warden", "RequestTicks %u", _serverTicks); // At request
362 TC_LOG_DEBUG("warden", "Ticks %u", newClientTicks); // At response
363 TC_LOG_DEBUG("warden", "Ticks diff %u", ourTicks - newClientTicks);
364 }
365
366 WardenCheckResult* rs;
367 WardenCheck *rd;
368 uint8 type;
369 uint16 checkFailed = 0;
370
371 boost::shared_lock<boost::shared_mutex> lock(sWardenCheckMgr->_checkStoreLock);
372
373 for (std::list<uint16>::iterator itr = _currentChecks.begin(); itr != _currentChecks.end(); ++itr)
374 {
375 rd = sWardenCheckMgr->GetWardenDataById(*itr);
376 rs = sWardenCheckMgr->GetWardenResultById(*itr);
377
378 type = rd->Type;
379 switch (type)
380 {
381 case MEM_CHECK:
382 {
383 uint8 Mem_Result;
384 buff >> Mem_Result;
385
386 if (Mem_Result != 0)
387 {
388 TC_LOG_DEBUG("warden", "RESULT MEM_CHECK not 0x00, CheckId %u account Id %u", *itr, _session->GetAccountId());
389 checkFailed = *itr;
390 continue;
391 }
392
393 if (memcmp(buff.contents() + buff.rpos(), rs->Result.AsByteArray(0, false).get(), rd->Length) != 0)
394 {
395 TC_LOG_DEBUG("warden", "RESULT MEM_CHECK fail CheckId %u account Id %u", *itr, _session->GetAccountId());
396 checkFailed = *itr;
397 buff.rpos(buff.rpos() + rd->Length);
398 continue;
399 }
400
401 buff.rpos(buff.rpos() + rd->Length);
402 TC_LOG_DEBUG("warden", "RESULT MEM_CHECK passed CheckId %u account Id %u", *itr, _session->GetAccountId());
403 break;
404 }
405 case PAGE_CHECK_A:
406 case PAGE_CHECK_B:
407 case DRIVER_CHECK:
408 case MODULE_CHECK:
409 {
410 const uint8 byte = 0xE9;
411 if (memcmp(buff.contents() + buff.rpos(), &byte, sizeof(uint8)) != 0)
412 {
413 if (type == PAGE_CHECK_A || type == PAGE_CHECK_B)
414 TC_LOG_DEBUG("warden", "RESULT PAGE_CHECK fail, CheckId %u account Id %u", *itr, _session->GetAccountId());
415 if (type == MODULE_CHECK)
416 TC_LOG_DEBUG("warden", "RESULT MODULE_CHECK fail, CheckId %u account Id %u", *itr, _session->GetAccountId());
417 if (type == DRIVER_CHECK)
418 TC_LOG_DEBUG("warden", "RESULT DRIVER_CHECK fail, CheckId %u account Id %u", *itr, _session->GetAccountId());
419 checkFailed = *itr;
420 buff.rpos(buff.rpos() + 1);
421 continue;
422 }
423
424 buff.rpos(buff.rpos() + 1);
425 if (type == PAGE_CHECK_A || type == PAGE_CHECK_B)
426 TC_LOG_DEBUG("warden", "RESULT PAGE_CHECK passed CheckId %u account Id %u", *itr, _session->GetAccountId());
427 else if (type == MODULE_CHECK)
428 TC_LOG_DEBUG("warden", "RESULT MODULE_CHECK passed CheckId %u account Id %u", *itr, _session->GetAccountId());
429 else if (type == DRIVER_CHECK)
430 TC_LOG_DEBUG("warden", "RESULT DRIVER_CHECK passed CheckId %u account Id %u", *itr, _session->GetAccountId());
431 break;
432 }
433 case LUA_STR_CHECK:
434 {
435 uint8 Lua_Result;
436 buff >> Lua_Result;
437
438 if (Lua_Result != 0)
439 {
440 TC_LOG_DEBUG("warden", "RESULT LUA_STR_CHECK fail, CheckId %u account Id %u", *itr, _session->GetAccountId());
441 checkFailed = *itr;
442 continue;
443 }
444
445 uint8 luaStrLen;
446 buff >> luaStrLen;
447
448 if (luaStrLen != 0)
449 {
450 char *str = new char[luaStrLen + 1];
451 memcpy(str, buff.contents() + buff.rpos(), luaStrLen);
452 str[luaStrLen] = '\0'; // null terminator
453 TC_LOG_DEBUG("warden", "Lua string: %s", str);
454 delete[] str;
455 }
456 buff.rpos(buff.rpos() + luaStrLen); // Skip string
457 TC_LOG_DEBUG("warden", "RESULT LUA_STR_CHECK passed, CheckId %u account Id %u", *itr, _session->GetAccountId());
458 break;
459 }
460 case MPQ_CHECK:
461 {
462 uint8 Mpq_Result;
463 buff >> Mpq_Result;
464
465 if (Mpq_Result != 0)
466 {
467 TC_LOG_DEBUG("warden", "RESULT MPQ_CHECK not 0x00 account id %u", _session->GetAccountId());
468 checkFailed = *itr;
469 continue;
470 }
471
472 if (memcmp(buff.contents() + buff.rpos(), rs->Result.AsByteArray(0, false).get(), 20) != 0) // SHA1
473 {
474 TC_LOG_DEBUG("warden", "RESULT MPQ_CHECK fail, CheckId %u account Id %u", *itr, _session->GetAccountId());
475 checkFailed = *itr;
476 buff.rpos(buff.rpos() + 20); // 20 bytes SHA1
477 continue;
478 }
479
480 buff.rpos(buff.rpos() + 20); // 20 bytes SHA1
481 TC_LOG_DEBUG("warden", "RESULT MPQ_CHECK passed, CheckId %u account Id %u", *itr, _session->GetAccountId());
482 break;
483 }
484 default: // Should never happen
485 break;
486 }
487 }
488
489 if (checkFailed > 0)
490 {
491 WardenCheck* check = sWardenCheckMgr->GetWardenDataById(checkFailed);
492 TC_LOG_WARN("warden", "%s failed Warden check %u. Action: %s", _session->GetPlayerInfo().c_str(), checkFailed, Penalty(check).c_str());
493 }
494
495 // Set hold off timer, minimum timer should at least be 1 second
496 uint32 holdOff = sWorld->getIntConfig(CONFIG_WARDEN_CLIENT_CHECK_HOLDOFF);
497 _checkTimer = (holdOff < 1 ? 1 : holdOff) * IN_MILLISECONDS;
498}

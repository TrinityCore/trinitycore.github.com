29const int32 ReputationMgr::PointsInRank[MAX_REPUTATION_RANK] = {36000, 3000, 3000, 3000, 6000, 12000, 21000, 1000};

31const int32 ReputationMgr::Reputation_Cap = 42999;
32const int32 ReputationMgr::Reputation_Bottom = -42000;

34ReputationRank ReputationMgr::ReputationToRank(int32 standing)
35{
36 int32 limit = Reputation_Cap + 1;
37 for (int i = MAX_REPUTATION_RANK-1; i >= MIN_REPUTATION_RANK; --i)
38 {
39 limit -= PointsInRank[i];
40 if (standing >= limit)
41 return ReputationRank(i);
42 }
43 return MIN_REPUTATION_RANK;
44}

82int32 ReputationMgr::GetBaseReputation(FactionEntry const* factionEntry) const
83{
84 if (!factionEntry)
85 return 0;
86
87 uint32 raceMask = _player->getRaceMask();
88 uint32 classMask = _player->getClassMask();
89 for (int i=0; i < 4; i++)
90 {
91 if ((factionEntry->BaseRepRaceMask[i] & raceMask ||
92 (factionEntry->BaseRepRaceMask[i] == 0 &&
93 factionEntry->BaseRepClassMask[i] != 0)) &&
94 (factionEntry->BaseRepClassMask[i] & classMask ||
95 factionEntry->BaseRepClassMask[i] == 0))
96
97 return factionEntry->BaseRepValue[i];
98 }
99
100 // in faction.dbc exist factions with (RepListId >=0, listed in character reputation list) with all BaseRepRaceMask[i] == 0
101 return 0;
102}

169void ReputationMgr::SendState(FactionState const* faction)
170{
171 uint32 count = 1;
172
173 WorldPacket data(SMSG_SET_FACTION_STANDING, 17);
174 data << float(0);
175 data << uint8(_sendFactionIncreased);
176 _sendFactionIncreased = false; // Reset
177
178 size_t p_count = data.wpos();
179 data << uint32(count);
180
181 data << uint32(faction->ReputationListID);
182 data << uint32(faction->Standing);
183
184 for (FactionStateList::iterator itr = _factions.begin(); itr != _factions.end(); ++itr)
185 {
186 if (itr->second.needSend)
187 {
188 itr->second.needSend = false;
189 if (itr->second.ReputationListID != faction->ReputationListID)
190 {
191 data << uint32(itr->second.ReputationListID);
192 data << uint32(itr->second.Standing);
193 ++count;
194 }
195 }
196 }
197
198 data.put<uint32>(p_count, count);
199 _player->SendDirectMessage(&data);
200}

202void ReputationMgr::SendInitialReputations()
203{
204 uint8 count = 128;
205 WorldPacket data(SMSG_INITIALIZE_FACTIONS, 4 + count * 5);
206 data << uint32(count);
207
208 RepListID a = 0;
209
210 for (FactionStateList::iterator itr = _factions.begin(); itr != _factions.end(); ++itr)
211 {
212 // fill in absent fields
213 for (; a != itr->first; ++a)
214 {
215 data << uint8(0);
216 data << uint32(0);
217 }
218
219 // fill in encountered data
220 data << uint8(itr->second.Flags);
221 data << uint32(itr->second.Standing);
222
223 itr->second.needSend = false;
224
225 ++a;
226 }
227
228 // fill in absent fields
229 for (; a != count; ++a)
230 {
231 data << uint8(0);
232 data << uint32(0);
233 }
234
235 _player->SendDirectMessage(&data);
236}

255void ReputationMgr::Initialize()
256{
257 _factions.clear();
258 _visibleFactionCount = 0;
259 _honoredFactionCount = 0;
260 _reveredFactionCount = 0;
261 _exaltedFactionCount = 0;
262 _sendFactionIncreased = false;
263
264 for (unsigned int i = 1; i < sFactionStore.GetNumRows(); i++)
265 {
266 FactionEntry const* factionEntry = sFactionStore.LookupEntry(i);
267
268 if (factionEntry && (factionEntry->reputationListID >= 0))
269 {
270 FactionState newFaction;
271 newFaction.ID = factionEntry->ID;
272 newFaction.ReputationListID = factionEntry->reputationListID;
273 newFaction.Standing = 0;
274 newFaction.Flags = GetDefaultStateFlags(factionEntry);
275 newFaction.needSend = true;
276 newFaction.needSave = true;
277
278 if (newFaction.Flags & FACTION_FLAG_VISIBLE)
279 ++_visibleFactionCount;
280
281 UpdateRankCounters(REP_HOSTILE, GetBaseRank(factionEntry));
282
283 _factions[newFaction.ReputationListID] = newFaction;
284 }
285 }

288bool ReputationMgr::SetReputation(FactionEntry const* factionEntry, int32 standing, bool incremental)
289{
290 sScriptMgr->OnPlayerReputationChange(_player, factionEntry->ID, standing, incremental);
291 bool res = false;
292 // if spillover definition exists in DB, override DBC
293 if (const RepSpilloverTemplate* repTemplate = sObjectMgr->GetRepSpilloverTemplate(factionEntry->ID))
294 {
295 for (uint32 i = 0; i < MAX_SPILLOVER_FACTIONS; ++i)
296 {
297 if (repTemplate->faction[i])
298 {
299 if (_player->GetReputationRank(repTemplate->faction[i]) <= ReputationRank(repTemplate->faction_rank[i]))
300 {
301 // bonuses are already given, so just modify standing by rate
302 int32 spilloverRep = int32(standing * repTemplate->faction_rate[i]);
303 SetOneFactionReputation(sFactionStore.AssertEntry(repTemplate->faction[i]), spilloverRep, incremental);
304 }
305 }
306 }
307 }
308 else
309 {
310 float spillOverRepOut = float(standing);
311 // check for sub-factions that receive spillover
312 SimpleFactionsList const* flist = GetFactionTeamList(factionEntry->ID);
313 // if has no sub-factions, check for factions with same parent
314 if (!flist && factionEntry->team && factionEntry->spilloverRateOut != 0.0f)
315 {
316 spillOverRepOut *= factionEntry->spilloverRateOut;
317 if (FactionEntry const* parent = sFactionStore.LookupEntry(factionEntry->team))
318 {
319 FactionStateList::iterator parentState = _factions.find(parent->reputationListID);
320 // some team factions have own reputation standing, in this case do not spill to other sub-factions
321 if (parentState != _factions.end() && (parentState->second.Flags & FACTION_FLAG_SPECIAL))
322 {
323 SetOneFactionReputation(parent, int32(spillOverRepOut), incremental);
324 }
325 else // spill to "sister" factions
326 {
327 flist = GetFactionTeamList(factionEntry->team);
328 }
329 }
330 }
331 if (flist)
332 {
333 // Spillover to affiliated factions
334 for (SimpleFactionsList::const_iterator itr = flist->begin(); itr != flist->end(); ++itr)
335 {
336 if (FactionEntry const* factionEntryCalc = sFactionStore.LookupEntry(*itr))
337 {
338 if (factionEntryCalc == factionEntry || GetRank(factionEntryCalc) > ReputationRank(factionEntryCalc->spilloverMaxRankIn))
339 continue;
340 int32 spilloverRep = int32(spillOverRepOut * factionEntryCalc->spilloverRateIn);
341 if (spilloverRep != 0 || !incremental)
342 res = SetOneFactionReputation(factionEntryCalc, spilloverRep, incremental);
343 }
344 }
345 }
346 }
347
348 // spillover done, update faction itself
349 FactionStateList::iterator faction = _factions.find(factionEntry->reputationListID);
350 if (faction != _factions.end())
351 {
352 res = SetOneFactionReputation(factionEntry, standing, incremental);
353 // only this faction gets reported to client, even if it has no own visible standing
354 SendState(&faction->second);
355 }
356 return res;
357}

359bool ReputationMgr::SetOneFactionReputation(FactionEntry const* factionEntry, int32 standing, bool incremental)
360{
361 FactionStateList::iterator itr = _factions.find(factionEntry->reputationListID);
362 if (itr != _factions.end())
363 {
364 int32 BaseRep = GetBaseReputation(factionEntry);
365
366 if (incremental)
367 {
368 // int32 *= float cause one point loss?
369 standing = int32(floor((float)standing * sWorld->getRate(RATE_REPUTATION_GAIN) + 0.5f));
370 standing += itr->second.Standing + BaseRep;
371 }
372
373 if (standing > Reputation_Cap)
374 standing = Reputation_Cap;
375 else if (standing < Reputation_Bottom)
376 standing = Reputation_Bottom;
377
378 ReputationRank old_rank = ReputationToRank(itr->second.Standing + BaseRep);
379 ReputationRank new_rank = ReputationToRank(standing);
380
381 itr->second.Standing = standing - BaseRep;
382 itr->second.needSend = true;
383 itr->second.needSave = true;
384
385 SetVisible(&itr->second);
386
387 if (new_rank <= REP_HOSTILE)
388 SetAtWar(&itr->second, true);
389
390 if (new_rank > old_rank)
391 _sendFactionIncreased = true;
392
393 UpdateRankCounters(old_rank, new_rank);
394
395 _player->ReputationChanged(factionEntry);
396 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_KNOWN_FACTIONS, factionEntry->ID);
397 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_REPUTATION, factionEntry->ID);
398 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_EXALTED_REPUTATION, factionEntry->ID);
399 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_REVERED_REPUTATION, factionEntry->ID);
400 _player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_HONORED_REPUTATION, factionEntry->ID);
401
402 return true;
403 }
404 return false;
405}

430void ReputationMgr::SetVisible(FactionState* faction)
431{
432 // always invisible or hidden faction can't be make visible
433 // except if faction has FACTION_FLAG_SPECIAL
434 if (faction->Flags & (FACTION_FLAG_INVISIBLE_FORCED|FACTION_FLAG_HIDDEN) && !(faction->Flags & FACTION_FLAG_SPECIAL))
435 return;
436
437 // already set
438 if (faction->Flags & FACTION_FLAG_VISIBLE)
439 return;
440
441 faction->Flags |= FACTION_FLAG_VISIBLE;
442 faction->needSend = true;
443 faction->needSave = true;
444
445 ++_visibleFactionCount;
446
447 SendVisible(faction);
448}

463void ReputationMgr::SetAtWar(FactionState* faction, bool atWar) const
464{
465 // not allow declare war to own faction
466 if (atWar && (faction->Flags & FACTION_FLAG_PEACE_FORCED))
467 return;
468
469 // already set
470 if (((faction->Flags & FACTION_FLAG_AT_WAR) != 0) == atWar)
471 return;
472
473 if (atWar)
474 faction->Flags |= FACTION_FLAG_AT_WAR;
475 else
476 faction->Flags &= ~FACTION_FLAG_AT_WAR;
477
478 faction->needSend = true;
479 faction->needSave = true;
480}

491void ReputationMgr::SetInactive(FactionState* faction, bool inactive) const
492{
493 // always invisible or hidden faction can't be inactive
494 if (inactive && ((faction->Flags & (FACTION_FLAG_INVISIBLE_FORCED|FACTION_FLAG_HIDDEN)) || !(faction->Flags & FACTION_FLAG_VISIBLE)))
495 return;
496
497 // already set
498 if (((faction->Flags & FACTION_FLAG_INACTIVE) != 0) == inactive)
499 return;
500
501 if (inactive)
502 faction->Flags |= FACTION_FLAG_INACTIVE;
503 else
504 faction->Flags &= ~FACTION_FLAG_INACTIVE;
505
506 faction->needSend = true;
507 faction->needSave = true;
508}

510void ReputationMgr::LoadFromDB(PreparedQueryResult result)
511{
512 // Set initial reputations (so everything is nifty before DB data load)
513 Initialize();
514
515 //QueryResult* result = CharacterDatabase.PQuery("SELECT faction, standing, flags FROM character_reputation WHERE guid = '%u'", GetGUID().GetCounter());
516
517 if (result)
518 {
519 do
520 {
521 Field* fields = result->Fetch();
522
523 FactionEntry const* factionEntry = sFactionStore.LookupEntry(fields[0].GetUInt16());
524 if (factionEntry && (factionEntry->reputationListID >= 0))
525 {
526 FactionState* faction = &_factions[factionEntry->reputationListID];
527
528 // update standing to current
529 faction->Standing = fields[1].GetInt32();
530
531 // update counters
532 int32 BaseRep = GetBaseReputation(factionEntry);
533 ReputationRank old_rank = ReputationToRank(BaseRep);
534 ReputationRank new_rank = ReputationToRank(BaseRep + faction->Standing);
535 UpdateRankCounters(old_rank, new_rank);
536
537 uint32 dbFactionFlags = fields[2].GetUInt16();
538
539 if (dbFactionFlags & FACTION_FLAG_VISIBLE)
540 SetVisible(faction); // have internal checks for forced invisibility
541
542 if (dbFactionFlags & FACTION_FLAG_INACTIVE)
543 SetInactive(faction, true); // have internal checks for visibility requirement
544
545 if (dbFactionFlags & FACTION_FLAG_AT_WAR) // DB at war
546 SetAtWar(faction, true); // have internal checks for FACTION_FLAG_PEACE_FORCED
547 else // DB not at war
548 {
549 // allow remove if visible (and then not FACTION_FLAG_INVISIBLE_FORCED or FACTION_FLAG_HIDDEN)
550 if (faction->Flags & FACTION_FLAG_VISIBLE)
551 SetAtWar(faction, false); // have internal checks for FACTION_FLAG_PEACE_FORCED
552 }
553
554 // set atWar for hostile
555 if (GetRank(factionEntry) <= REP_HOSTILE)
556 SetAtWar(faction, true);
557
558 // reset changed flag if values similar to saved in DB
559 if (faction->Flags == dbFactionFlags)
560 {
561 faction->needSend = false;
562 faction->needSave = false;
563 }
564 }
565 }
566 while (result->NextRow());
567 }
568}

570void ReputationMgr::SaveToDB(SQLTransaction& trans)
571{
572 for (FactionStateList::iterator itr = _factions.begin(); itr != _factions.end(); ++itr)
573 {
574 if (itr->second.needSave)
575 {
576 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHAR_REPUTATION_BY_FACTION);
577 stmt->setUInt32(0, _player->GetGUID().GetCounter());
578 stmt->setUInt16(1, uint16(itr->second.ID));
579 trans->Append(stmt);
580
581 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_CHAR_REPUTATION_BY_FACTION);
582 stmt->setUInt32(0, _player->GetGUID().GetCounter());
583 stmt->setUInt16(1, uint16(itr->second.ID));
584 stmt->setInt32(2, itr->second.Standing);
585 stmt->setUInt16(3, uint16(itr->second.Flags));
586 trans->Append(stmt);
587
588 itr->second.needSave = false;
589 }
590 }

593void ReputationMgr::UpdateRankCounters(ReputationRank old_rank, ReputationRank new_rank)
594{
595 if (old_rank >= REP_EXALTED)
596 --_exaltedFactionCount;
597 if (old_rank >= REP_REVERED)
598 --_reveredFactionCount;
599 if (old_rank >= REP_HONORED)
600 --_honoredFactionCount;
601
602 if (new_rank >= REP_EXALTED)
603 ++_exaltedFactionCount;
604 if (new_rank >= REP_REVERED)
605 ++_reveredFactionCount;
606 if (new_rank >= REP_HONORED)
607 ++_honoredFactionCount;
608}

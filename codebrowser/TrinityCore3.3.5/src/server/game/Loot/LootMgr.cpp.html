55// Selects invalid loot items to be removed from group possible entries (before rolling)
56struct LootGroupInvalidSelector : public std::unary_function<LootStoreItem*, bool>
57{
58 explicit LootGroupInvalidSelector(Loot const& loot, uint16 lootMode) : _loot(loot), _lootMode(lootMode) { }
59
60 bool operator()(LootStoreItem* item) const
61 {
62 if (!(item->lootmode & _lootMode))
63 return true;
64
65 uint8 foundDuplicates = 0;
66 for (std::vector<LootItem>::const_iterator itr = _loot.items.begin(); itr != _loot.items.end(); ++itr)
67 if (itr->itemid == item->itemid)
68 if (++foundDuplicates == _loot.maxDuplicates)
69 return true;
70
71 return false;
72 }
73
74private:
75 Loot const& _loot;
76 uint16 _lootMode;
77};

126// Loads a *_loot_template DB table into loot store
127// All checks of the loaded template are called from here, no error reports at loot generation required
128uint32 LootStore::LoadLootTable()
129{
130 LootTemplateMap::const_iterator tab;
131
132 // Clearing store (for reloading case)
133 Clear();
134
135 // 0 1 2 3 4 5 6
136 QueryResult result = WorldDatabase.PQuery("SELECT Entry, Item, Reference, Chance, QuestRequired, LootMode, GroupId, MinCount, MaxCount FROM %s", GetName());
137
138 if (!result)
139 return 0;
140
141 uint32 count = 0;
142
143 do
144 {
145 Field* fields = result->Fetch();
146
147 uint32 entry = fields[0].GetUInt32();
148 uint32 item = fields[1].GetUInt32();
149 uint32 reference = fields[2].GetUInt32();
150 float chance = fields[3].GetFloat();
151 bool needsquest = fields[4].GetBool();
152 uint16 lootmode = fields[5].GetUInt16();
153 uint8 groupid = fields[6].GetUInt8();
154 uint8 mincount = fields[7].GetUInt8();
155 uint8 maxcount = fields[8].GetUInt8();
156
157 if (groupid >= 1 << 7) // it stored in 7 bit field
158 {
159 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: GroupId (%u) must be less %u - skipped", GetName(), entry, item, groupid, 1 << 7);
160 return 0;
161 }
162
163 LootStoreItem* storeitem = new LootStoreItem(item, reference, chance, needsquest, lootmode, groupid, mincount, maxcount);
164
165 if (!storeitem->IsValid(*this, entry)) // Validity checks
166 {
167 delete storeitem;
168 continue;
169 }
170
171 // Looking for the template of the entry
172 // often entries are put together
173 if (m_LootTemplates.empty() || tab->first != entry)
174 {
175 // Searching the template (in case template Id changed)
176 tab = m_LootTemplates.find(entry);
177 if (tab == m_LootTemplates.end())
178 {
179 std::pair< LootTemplateMap::iterator, bool > pr = m_LootTemplates.insert(LootTemplateMap::value_type(entry, new LootTemplate()));
180 tab = pr.first;
181 }
182 }
183 // else is empty - template Id and iter are the same
184 // finally iter refers to already existed or just created <entry, LootTemplate>
185
186 // Adds current row to the template
187 tab->second->AddEntry(storeitem);
188 ++count;
189 }
190 while (result->NextRow());
191
192 Verify(); // Checks validity of the loot store
193
194 return count;
195}

278// Checks if the entry (quest, non-quest, reference) takes it's chance (at loot generation)
279// RATE_DROP_ITEMS is no longer used for all types of entries
280bool LootStoreItem::Roll(bool rate) const
281{
282 if (chance >= 100.0f)
283 return true;
284
285 if (reference > 0) // reference case
286 return roll_chance_f(chance* (rate ? sWorld->getRate(RATE_DROP_ITEM_REFERENCED) : 1.0f));
287
288 ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(itemid);
289
290 float qualityModifier = pProto && rate ? sWorld->getRate(qualityToRate[pProto->Quality]) : 1.0f;
291
292 return roll_chance_f(chance*qualityModifier);
293}

295// Checks correctness of values
296bool LootStoreItem::IsValid(LootStore const& store, uint32 entry) const
297{
298 if (mincount == 0)
299 {
300 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: wrong MinCount (%d) - skipped", store.GetName(), entry, itemid, mincount);
301 return false;
302 }
303
304 if (reference == 0) // item (quest or non-quest) entry, maybe grouped
305 {
306 ItemTemplate const* proto = sObjectMgr->GetItemTemplate(itemid);
307 if (!proto)
308 {
309 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: item entry not listed in `item_template` - skipped", store.GetName(), entry, itemid);
310 return false;
311 }
312
313 if (chance == 0 && groupid == 0) // Zero chance is allowed for grouped entries only
314 {
315 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: equal-chanced grouped entry, but group not defined - skipped", store.GetName(), entry, itemid);
316 return false;
317 }
318
319 if (chance != 0 && chance < 0.000001f) // loot with low chance
320 {
321 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: low chance (%f) - skipped",
322 store.GetName(), entry, itemid, chance);
323 return false;
324 }
325
326 if (maxcount < mincount) // wrong max count
327 {
328 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: MaxCount (%u) less that MinCount (%i) - skipped", store.GetName(), entry, itemid, int32(maxcount), mincount);
329 return false;
330 }
331 }
332 else // if reference loot
333 {
334 if (needs_quest)
335 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: quest required will be ignored", store.GetName(), entry, itemid);
336 else if (chance == 0) // no chance for the reference
337 {
338 TC_LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: zero chance is specified for a reference, skipped", store.GetName(), entry, itemid);
339 return false;
340 }
341 }
342 return true; // Referenced template existence is checked at whole store level
343}

371// Basic checks for player/item compatibility - if false no chance to see the item in the loot
372bool LootItem::AllowedForPlayer(Player const* player) const
373{
374 // DB conditions check
375 if (!sConditionMgr->IsObjectMeetToConditions(const_cast<Player*>(player), conditions))
376 return false;
377
378 ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(itemid);
379 if (!pProto)
380 return false;
381
382 // not show loot for players without profession or those who already know the recipe
383 if ((pProto->Flags & ITEM_PROTO_FLAG_SMART_LOOT) && (!player->HasSkill(pProto->RequiredSkill) || player->HasSpell(pProto->Spells[1].SpellId)))
384 return false;
385
386 // not show loot for not own team
387 if ((pProto->Flags2 & ITEM_FLAGS_EXTRA_HORDE_ONLY) && player->GetTeam() != HORDE)
388 return false;
389
390 if ((pProto->Flags2 & ITEM_FLAGS_EXTRA_ALLIANCE_ONLY) && player->GetTeam() != ALLIANCE)
391 return false;
392
393 // check quest requirements
394 if (!(pProto->FlagsCu & ITEM_FLAGS_CU_IGNORE_QUEST_STATUS) && ((needs_quest || (pProto->StartQuest && player->GetQuestStatus(pProto->StartQuest) != QUEST_STATUS_NONE)) && !player->HasQuestForItem(itemid)))
395 return false;
396
397 return true;
398}

409// Inserts the item into the loot (called by LootTemplate processors)
410void Loot::AddItem(LootStoreItem const& item)
411{
412 ItemTemplate const* proto = sObjectMgr->GetItemTemplate(item.itemid);
413 if (!proto)
414 return;
415
416 uint32 count = urand(item.mincount, item.maxcount);
417 uint32 stacks = count / proto->GetMaxStackSize() + ((count % proto->GetMaxStackSize()) ? 1 : 0);
418
419 std::vector<LootItem>& lootItems = item.needs_quest ? quest_items : items;
420 uint32 limit = item.needs_quest ? MAX_NR_QUEST_ITEMS : MAX_NR_LOOT_ITEMS;
421
422 for (uint32 i = 0; i < stacks && lootItems.size() < limit; ++i)
423 {
424 LootItem generatedLoot(item);
425 generatedLoot.count = std::min(count, proto->GetMaxStackSize());
426 lootItems.push_back(generatedLoot);
427 count -= proto->GetMaxStackSize();
428
429 // non-conditional one-player only items are counted here,
430 // free for all items are counted in FillFFALoot(),
431 // non-ffa conditionals are counted in FillNonQuestNonFFAConditionalLoot()
432 if (!item.needs_quest && item.conditions.empty() && !(proto->Flags & ITEM_PROTO_FLAG_PARTY_LOOT))
433 ++unlootedCount;
434 }
435}

437// Calls processor of corresponding LootTemplate (which handles everything including references)
438bool Loot::FillLoot(uint32 lootId, LootStore const& store, Player* lootOwner, bool personal, bool noEmptyError, uint16 lootMode /*= LOOT_MODE_DEFAULT*/)
439{
440 // Must be provided
441 if (!lootOwner)
442 return false;
443
444 LootTemplate const* tab = store.GetLootFor(lootId);
445
446 if (!tab)
447 {
448 if (!noEmptyError)
449 TC_LOG_ERROR("sql.sql", "Table '%s' loot id #%u used but it doesn't have records.", store.GetName(), lootId);
450 return false;
451 }
452
453 items.reserve(MAX_NR_LOOT_ITEMS);
454 quest_items.reserve(MAX_NR_QUEST_ITEMS);
455
456 tab->Process(*this, store.IsRatesAllowed(), lootMode); // Processing is done there, callback via Loot::AddItem()
457
458 // Setting access rights for group loot case
459 Group* group = lootOwner->GetGroup();
460 if (!personal && group)
461 {
462 roundRobinPlayer = lootOwner->GetGUID();
463
464 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
465 if (Player* player = itr->GetSource()) // should actually be looted object instead of lootOwner but looter has to be really close so doesnt really matter
466 FillNotNormalLootFor(player, player->IsAtGroupRewardDistance(lootOwner));
467
468 for (uint8 i = 0; i < items.size(); ++i)
469 {
470 if (ItemTemplate const* proto = sObjectMgr->GetItemTemplate(items[i].itemid))
471 if (proto->Quality < uint32(group->GetLootThreshold()))
472 items[i].is_underthreshold = true;
473 }
474 }
475 // ... for personal loot
476 else
477 FillNotNormalLootFor(lootOwner, true);
478
479 return true;
480}

482void Loot::FillNotNormalLootFor(Player* player, bool presentAtLooting)
483{
484 ObjectGuid::LowType plguid = player->GetGUID().GetCounter();
485
486 QuestItemMap::const_iterator qmapitr = PlayerQuestItems.find(plguid);
487 if (qmapitr == PlayerQuestItems.end())
488 FillQuestLoot(player);
489
490 qmapitr = PlayerFFAItems.find(plguid);
491 if (qmapitr == PlayerFFAItems.end())
492 FillFFALoot(player);
493
494 qmapitr = PlayerNonQuestNonFFAConditionalItems.find(plguid);
495 if (qmapitr == PlayerNonQuestNonFFAConditionalItems.end())
496 FillNonQuestNonFFAConditionalLoot(player, presentAtLooting);
497
498 // if not auto-processed player will have to come and pick it up manually
499 if (!presentAtLooting)
500 return;
501
502 // Process currency items
503 uint32 max_slot = GetMaxSlotInLootFor(player);
504 LootItem const* item = NULL;
505 uint32 itemsSize = uint32(items.size());
506 for (uint32 i = 0; i < max_slot; ++i)
507 {
508 if (i < items.size())
509 item = &items[i];
510 else
511 item = &quest_items[i-itemsSize];
512
513 if (!item->is_looted && item->freeforall && item->AllowedForPlayer(player))
514 if (ItemTemplate const* proto = sObjectMgr->GetItemTemplate(item->itemid))
515 if (proto->IsCurrencyToken())
516 player->StoreLootItem(i, this);
517 }
518}

543QuestItemList* Loot::FillQuestLoot(Player* player)
544{
545 if (items.size() == MAX_NR_LOOT_ITEMS)
546 return NULL;
547
548 QuestItemList* ql = new QuestItemList();
549
550 for (uint8 i = 0; i < quest_items.size(); ++i)
551 {
552 LootItem &item = quest_items[i];
553
554 if (!item.is_looted && (item.AllowedForPlayer(player) || (item.follow_loot_rules && player->GetGroup() && ((player->GetGroup()->GetLootMethod() == MASTER_LOOT && player->GetGroup()->GetMasterLooterGuid() == player->GetGUID()) || player->GetGroup()->GetLootMethod() != MASTER_LOOT))))
555 {
556 ql->push_back(QuestItem(i));
557
558 // quest items get blocked when they first appear in a
559 // player's quest vector
560 //
561 // increase once if one looter only, looter-times if free for all
562 if (item.freeforall || !item.is_blocked)
563 ++unlootedCount;
564 if (!player->GetGroup() || (player->GetGroup()->GetLootMethod() != GROUP_LOOT && player->GetGroup()->GetLootMethod() != ROUND_ROBIN))
565 item.is_blocked = true;
566
567 if (items.size() + ql->size() == MAX_NR_LOOT_ITEMS)
568 break;
569 }
570 }
571 if (ql->empty())
572 {
573 delete ql;
574 return NULL;
575 }
576
577 PlayerQuestItems[player->GetGUID().GetCounter()] = ql;
578 return ql;
579}

581QuestItemList* Loot::FillNonQuestNonFFAConditionalLoot(Player* player, bool presentAtLooting)
582{
583 QuestItemList* ql = new QuestItemList();
584
585 for (uint8 i = 0; i < items.size(); ++i)
586 {
587 LootItem &item = items[i];
588 if (!item.is_looted && !item.freeforall && (item.AllowedForPlayer(player) || (item.follow_loot_rules && player->GetGroup() && ((player->GetGroup()->GetLootMethod() == MASTER_LOOT && player->GetGroup()->GetMasterLooterGuid() == player->GetGUID()) || player->GetGroup()->GetLootMethod() != MASTER_LOOT))))
589 {
590 if (presentAtLooting)
591 item.AddAllowedLooter(player);
592 if (!item.conditions.empty())
593 {
594 ql->push_back(QuestItem(i));
595 if (!item.is_counted)
596 {
597 ++unlootedCount;
598 item.is_counted = true;
599 }
600 }
601 }
602 }
603 if (ql->empty())
604 {
605 delete ql;
606 return NULL;
607 }
608
609 PlayerNonQuestNonFFAConditionalItems[player->GetGUID().GetCounter()] = ql;
610 return ql;
611}

862 b << uint8(0); // item count

866 Loot &l = lv.loot;
867
868 uint8 itemsShown = 0;
869
870 b << uint32(l.gold); //gold
871
872 size_t count_pos = b.wpos(); // pos of item count byte
873 b << uint8(0); // item count placeholder
874
875 switch (lv.permission)
876 {
877 case GROUP_PERMISSION:
878 case MASTER_PERMISSION:
879 case RESTRICTED_PERMISSION:
880 {
881 // if you are not the round-robin group looter, you can only see
882 // blocked rolled items and quest items, and !ffa items
883 for (uint8 i = 0; i < l.items.size(); ++i)
884 {
885 if (!l.items[i].is_looted && !l.items[i].freeforall && l.items[i].conditions.empty() && l.items[i].AllowedForPlayer(lv.viewer))
886 {
887 uint8 slot_type;
888

890 {
891 switch (lv.permission)
892 {
893 case GROUP_PERMISSION:
894 slot_type = LOOT_SLOT_TYPE_ROLL_ONGOING;
895 break;
896 case MASTER_PERMISSION:
897 {

899 slot_type = LOOT_SLOT_TYPE_MASTER;
900 else
901 slot_type = LOOT_SLOT_TYPE_LOCKED;
902 break;
903 }
904 case RESTRICTED_PERMISSION:
905 slot_type = LOOT_SLOT_TYPE_LOCKED;
906 break;
907 default:
908 continue;
909 }
910 }

912 {

914 // or it IS the round robin group owner
915 // => item is lootable
916 slot_type = LOOT_SLOT_TYPE_ALLOW_LOOT;
917 }
918 else
919 // item shall not be displayed.
920 continue;
921
922 b << uint8(i) << l.items[i];
923 b << uint8(slot_type);
924 ++itemsShown;
925 }
926 }
927 break;
928 }
929 case ROUND_ROBIN_PERMISSION:
930 {
931 for (uint8 i = 0; i < l.items.size(); ++i)

942 }
943 }
944 break;

952 if (!l.items[i].is_looted && !l.items[i].freeforall && l.items[i].conditions.empty() && l.items[i].AllowedForPlayer(lv.viewer))

954 b << uint8(i) << l.items[i];

956 ++itemsShown;
957 }
958 }
959 break;
960 }
961 default:
962 return b;
963 }
964
965 LootSlotType slotType = lv.permission == OWNER_PERMISSION ? LOOT_SLOT_TYPE_OWNER : LOOT_SLOT_TYPE_ALLOW_LOOT;

987 break;
988 case GROUP_PERMISSION:
989 case ROUND_ROBIN_PERMISSION:
990 if (!item.is_blocked)

1101LootStoreItem const* LootTemplate::LootGroup::Roll(Loot& loot, uint16 lootMode) const
1102{

1104 possibleLoot.remove_if(LootGroupInvalidSelector(loot, lootMode));

1109
1110 for (LootStoreItemList::const_iterator itr = possibleLoot.begin(); itr != possibleLoot.end(); ++itr) // check each explicitly chanced entry in the template and modify its chance based on quality.

1114 return item;
1115
1116 roll -= item->chance;
1117 if (roll < 0)
1118 return item;
1119 }
1120 }
1121
1122 possibleLoot = EqualChanced;
1123 possibleLoot.remove_if(LootGroupInvalidSelector(loot, lootMode));
1124 if (!possibleLoot.empty()) // If nothing selected yet - an item is taken from equal-chanced part

1303 for (LootStoreItemList::const_iterator i = Entries.begin(); i != Entries.end(); ++i)
1304 {
1305 LootStoreItem* item = *i;
1306 if (!(item->lootmode & lootMode)) // Do not add if mode mismatch
1307 continue;
1308
1309 if (!item->Roll(rate))
1310 continue; // Bad luck for the entry
1311
1312 if (item->reference > 0) // References processing
1313 {
1314 LootTemplate const* Referenced = LootTemplates_Reference.GetLootFor(item->reference);
1315 if (!Referenced)
1316 continue; // Error message already printed at loading stage
1317
1318 uint32 maxcount = uint32(float(item->maxcount) * sWorld->getRate(RATE_DROP_ITEM_REFERENCED_AMOUNT));
1319 for (uint32 loop = 0; loop < maxcount; ++loop) // Ref multiplicator
1320 Referenced->Process(loot, rate, lootMode, item->groupid);
1321 }
1322 else // Plain entries (not a reference, not grouped)
1323 loot.AddItem(*item); // Chance is already checked, just add
1324 }
1325
1326 // Now processing groups
1327 for (LootGroups::const_iterator i = Groups.begin(); i != Groups.end(); ++i)
1328 if (LootGroup* group = *i)
1329 group->Process(loot, lootMode);
1330}

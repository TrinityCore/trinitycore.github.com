95enum LootError
96{
97 LOOT_ERROR_DIDNT_KILL = 0, // You don't have permission to loot that corpse.
98 LOOT_ERROR_TOO_FAR = 4, // You are too far away to loot that corpse.
99 LOOT_ERROR_BAD_FACING = 5, // You must be facing the corpse to loot it.
100 LOOT_ERROR_LOCKED = 6, // Someone is already looting that corpse.
101 LOOT_ERROR_NOTSTANDING = 8, // You need to be standing up to loot something!
102 LOOT_ERROR_STUNNED = 9, // You can't loot anything while stunned!
103 LOOT_ERROR_PLAYER_NOT_FOUND = 10, // Player not found
104 LOOT_ERROR_PLAY_TIME_EXCEEDED = 11, // Maximum play time exceeded
105 LOOT_ERROR_MASTER_INV_FULL = 12, // That player's inventory is full
106 LOOT_ERROR_MASTER_UNIQUE_ITEM = 13, // Player has too many of that item already
107 LOOT_ERROR_MASTER_OTHER = 14, // Can't assign item to that player
108 LOOT_ERROR_ALREADY_PICKPOCKETED = 15, // Your target has already had its pockets picked
109 LOOT_ERROR_NOT_WHILE_SHAPESHIFTED = 16 // You can't do that while shapeshifted.

125struct LootStoreItem
126{
127 uint32 itemid; // id of the item
128 uint32 reference; // referenced TemplateleId
129 float chance; // chance to drop for both quest and non-quest items, chance to be used for refs
130 uint16 lootmode;
131 bool needs_quest : 1; // quest drop (quest is required for item to drop)
132 uint8 groupid : 7;
133 uint8 mincount; // mincount for drop items
134 uint8 maxcount; // max drop count for the item mincount or Ref multiplicator
135 ConditionContainer conditions; // additional loot condition
136
137 // Constructor
138 // displayid is filled in IsValid() which must be called after
139 LootStoreItem(uint32 _itemid, uint32 _reference, float _chance, bool _needs_quest, uint16 _lootmode, uint8 _groupid, int32 _mincount, uint8 _maxcount)
140 : itemid(_itemid), reference(_reference), chance(_chance), lootmode(_lootmode),
141 needs_quest(_needs_quest), groupid(_groupid), mincount(_mincount), maxcount(_maxcount)
142 { }
143
144 bool Roll(bool rate) const; // Checks if the entry takes it's chance (at loot generation)
145 bool IsValid(LootStore const& store, uint32 entry) const;
146 // Checks correctness of values
147};

151struct LootItem
152{
153 uint32 itemid;
154 uint32 randomSuffix;
155 int32 randomPropertyId;
156 ConditionContainer conditions; // additional loot condition
157 AllowedLooterSet allowedGUIDs;
158 uint8 count : 8;
159 bool is_looted : 1;
160 bool is_blocked : 1;
161 bool freeforall : 1; // free for all
162 bool is_underthreshold : 1;
163 bool is_counted : 1;
164 bool needs_quest : 1; // quest drop
165 bool follow_loot_rules : 1;
166 bool canSave;
167
168 // Constructor, copies most fields from LootStoreItem, generates random count and random suffixes/properties
169 // Should be called for non-reference LootStoreItem entries only (reference = 0)
170 explicit LootItem(LootStoreItem const& li);
171
172 // Empty constructor for creating an empty LootItem to be filled in with DB data
173 LootItem() : itemid(0), randomSuffix(0), randomPropertyId(0), count(0), is_looted(false), is_blocked(false),
174 freeforall(false), is_underthreshold(false), is_counted(false), needs_quest(false), follow_loot_rules(false),
175 canSave(true){ };
176
177 // Basic checks for player/item compatibility - if false no chance to see the item in the loot
178 bool AllowedForPlayer(Player const* player) const;
179 void AddAllowedLooter(Player const* player);
180 const AllowedLooterSet & GetAllowedLooters() const { return allowedGUIDs; }
181};

318 std::vector<LootItem> items;
319 std::vector<LootItem> quest_items;
320 uint32 gold;
321 uint8 unlootedCount;
322 ObjectGuid roundRobinPlayer; // GUID of the player having the Round-Robin ownership for the loot. If 0, round robin owner has released.
323 LootType loot_type; // required for achievement system
324 uint8 maxDuplicates; // Max amount of items with the same entry that can drop (default is 1; on 25 man raid mode 3)
325
326 // GUIDLow of container that holds this loot (item_instance.entry)
327 // Only set for inventory items that can be right-click looted
328 uint32 containerID;
329
330 Loot(uint32 _gold = 0) : gold(_gold), unlootedCount(0), roundRobinPlayer(), loot_type(LOOT_CORPSE), maxDuplicates(1), containerID(0) { }
331 ~Loot() { clear(); }
332
333 // For deleting items at loot removal since there is no backward interface to the Item()
334 void DeleteLootItemFromContainerItemDB(uint32 itemID);
335 void DeleteLootMoneyFromContainerItemDB();
336
337 // if loot becomes invalid this reference is used to inform the listener
338 void addLootValidatorRef(LootValidatorRef* pLootValidatorRef)
339 {
340 i_LootValidatorRefManager.insertFirst(pLootValidatorRef);
341 }
342
343 // void clear();
344 void clear()
345 {
346 for (QuestItemMap::const_iterator itr = PlayerQuestItems.begin(); itr != PlayerQuestItems.end(); ++itr)
347 delete itr->second;
348 PlayerQuestItems.clear();
349
350 for (QuestItemMap::const_iterator itr = PlayerFFAItems.begin(); itr != PlayerFFAItems.end(); ++itr)
351 delete itr->second;
352 PlayerFFAItems.clear();
353
354 for (QuestItemMap::const_iterator itr = PlayerNonQuestNonFFAConditionalItems.begin(); itr != PlayerNonQuestNonFFAConditionalItems.end(); ++itr)
355 delete itr->second;
356 PlayerNonQuestNonFFAConditionalItems.clear();
357
358 PlayersLooting.clear();
359 items.clear();
360 quest_items.clear();
361 gold = 0;
362 unlootedCount = 0;
363 roundRobinPlayer.Clear();
364 loot_type = LOOT_NONE;
365 i_LootValidatorRefManager.clearReferences();
366 }
367
368 bool empty() const { return items.empty() && gold == 0; }
369 bool isLooted() const { return gold == 0 && unlootedCount == 0; }
370
371 void NotifyItemRemoved(uint8 lootIndex);
372 void NotifyQuestItemRemoved(uint8 questIndex);
373 void NotifyMoneyRemoved();
374 void AddLooter(ObjectGuid GUID) { PlayersLooting.insert(GUID); }
375 void RemoveLooter(ObjectGuid GUID) { PlayersLooting.erase(GUID); }

385 bool hasItemFor(Player* player) const;

388 private:
389 void FillNotNormalLootFor(Player* player, bool presentAtLooting);

392 QuestItemList* FillNonQuestNonFFAConditionalLoot(Player* player, bool presentAtLooting);

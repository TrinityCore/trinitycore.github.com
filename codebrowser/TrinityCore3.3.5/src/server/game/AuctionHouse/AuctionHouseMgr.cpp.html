47AuctionHouseObject* AuctionHouseMgr::GetAuctionsMap(uint32 factionTemplateId)
48{
49 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_AUCTION))
50 return &mNeutralAuctions;
51
52 // teams have linked auction houses
53 FactionTemplateEntry const* uEntry = sFactionTemplateStore.LookupEntry(factionTemplateId);
54 if (!uEntry)
55 return &mNeutralAuctions;
56 else if (uEntry->ourMask & FACTION_MASK_ALLIANCE)
57 return &mAllianceAuctions;
58 else if (uEntry->ourMask & FACTION_MASK_HORDE)
59 return &mHordeAuctions;
60 else
61 return &mNeutralAuctions;
62}

77uint32 AuctionHouseMgr::GetAuctionDeposit(AuctionHouseEntry const* entry, uint32 time, Item* pItem, uint32 count)
78{
79 uint32 MSV = pItem->GetTemplate()->SellPrice;
80
81 if (MSV <= 0)
82 return AH_MINIMUM_DEPOSIT * sWorld->getRate(RATE_AUCTION_DEPOSIT);
83
84 float multiplier = CalculatePct(float(entry->depositPercent), 3);
85 uint32 timeHr = (((time / 60) / 60) / 12);
86 uint32 deposit = uint32(MSV * multiplier * sWorld->getRate(RATE_AUCTION_DEPOSIT));
87 float remainderbase = float(MSV * multiplier * sWorld->getRate(RATE_AUCTION_DEPOSIT)) - deposit;
88
89 deposit *= timeHr * count;
90
91 int i = count;
92 while (i > 0 && (remainderbase * i) != uint32(remainderbase * i))
93 i--;
94
95 if (i)
96 deposit += remainderbase * i * timeHr;
97
98 TC_LOG_DEBUG("auctionHouse", "MSV: %u", MSV);
99 TC_LOG_DEBUG("auctionHouse", "Items: %u", count);
100 TC_LOG_DEBUG("auctionHouse", "Multiplier: %f", multiplier);
101 TC_LOG_DEBUG("auctionHouse", "Deposit: %u", deposit);
102 TC_LOG_DEBUG("auctionHouse", "Deposit rm: %f", remainderbase * count);
103
104 if (deposit < AH_MINIMUM_DEPOSIT * sWorld->getRate(RATE_AUCTION_DEPOSIT))
105 return AH_MINIMUM_DEPOSIT * sWorld->getRate(RATE_AUCTION_DEPOSIT);
106 else
107 return deposit;
108}

111void AuctionHouseMgr::SendAuctionWonMail(AuctionEntry* auction, SQLTransaction& trans)
112{
113 Item* pItem = GetAItem(auction->itemGUIDLow);
114 if (!pItem)
115 return;
116
117 uint32 bidderAccId = 0;
118 ObjectGuid bidderGuid(HighGuid::Player, auction->bidder);
119 Player* bidder = ObjectAccessor::FindConnectedPlayer(bidderGuid);
120 // data for gm.log
121 std::string bidderName;
122 bool logGmTrade = false;
123
124 if (bidder)
125 {
126 bidderAccId = bidder->GetSession()->GetAccountId();
127 bidderName = bidder->GetName();
128 logGmTrade = bidder->GetSession()->HasPermission(rbac::RBAC_PERM_LOG_GM_TRADE);
129 }
130 else
131 {
132 bidderAccId = sObjectMgr->GetPlayerAccountIdByGUID(bidderGuid);
133 logGmTrade = AccountMgr::HasPermission(bidderAccId, rbac::RBAC_PERM_LOG_GM_TRADE, realm.Id.Realm);
134
135 if (logGmTrade && !sObjectMgr->GetPlayerNameByGUID(bidderGuid, bidderName))
136 bidderName = sObjectMgr->GetTrinityStringForDBCLocale(LANG_UNKNOWN);
137 }
138
139 if (logGmTrade)
140 {
141 ObjectGuid ownerGuid = ObjectGuid(HighGuid::Player, auction->owner);
142 std::string ownerName;
143 if (!sObjectMgr->GetPlayerNameByGUID(ownerGuid, ownerName))
144 ownerName = sObjectMgr->GetTrinityStringForDBCLocale(LANG_UNKNOWN);
145
146 uint32 ownerAccId = sObjectMgr->GetPlayerAccountIdByGUID(ownerGuid);
147
148 sLog->outCommand(bidderAccId, "GM %s (Account: %u) won item in auction: %s (Entry: %u Count: %u) and pay money: %u. Original owner %s (Account: %u)",
149 bidderName.c_str(), bidderAccId, pItem->GetTemplate()->Name1.c_str(), pItem->GetEntry(), pItem->GetCount(), auction->bid, ownerName.c_str(), ownerAccId);
150 }
151
152 // receiver exist
153 if (bidder || bidderAccId)
154 {
155 // set owner to bidder (to prevent delete item with sender char deleting)
156 // owner in `data` will set at mail receive and item extracting
157 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ITEM_OWNER);
158 stmt->setUInt32(0, auction->bidder);
159 stmt->setUInt32(1, pItem->GetGUID().GetCounter());
160 trans->Append(stmt);
161
162 if (bidder)
163 {
164 bidder->GetSession()->SendAuctionBidderNotification(auction->GetHouseId(), auction->Id, bidderGuid, 0, 0, auction->itemEntry);
165 // FIXME: for offline player need also
166 bidder->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_WON_AUCTIONS, 1);
167 }
168
169 MailDraft(auction->BuildAuctionMailSubject(AUCTION_WON), AuctionEntry::BuildAuctionMailBody(auction->owner, auction->bid, auction->buyout, 0, 0))
170 .AddItem(pItem)
171 .SendMailTo(trans, MailReceiver(bidder, auction->bidder), auction, MAIL_CHECK_MASK_COPIED);
172 }
173 else
174 {
175 // bidder doesn't exist, delete the item
176 sAuctionMgr->RemoveAItem(auction->itemGUIDLow, true);
177 }
178}

191//call this method to send mail to auction owner, when auction is successful, it does not clear ram
192void AuctionHouseMgr::SendAuctionSuccessfulMail(AuctionEntry* auction, SQLTransaction& trans)
193{
194 ObjectGuid owner_guid(HighGuid::Player, auction->owner);
195 Player* owner = ObjectAccessor::FindConnectedPlayer(owner_guid);
196 uint32 owner_accId = sObjectMgr->GetPlayerAccountIdByGUID(owner_guid);
197 // owner exist
198 if (owner || owner_accId)
199 {
200 uint32 profit = auction->bid + auction->deposit - auction->GetAuctionCut();
201
202 //FIXME: what do if owner offline
203 if (owner)
204 {
205 owner->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GOLD_EARNED_BY_AUCTIONS, profit);
206 owner->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_SOLD, auction->bid);
207 //send auction owner notification, bidder must be current!
208 owner->GetSession()->SendAuctionOwnerNotification(auction);
209 }
210
211 MailDraft(auction->BuildAuctionMailSubject(AUCTION_SUCCESSFUL), AuctionEntry::BuildAuctionMailBody(auction->bidder, auction->bid, auction->buyout, auction->deposit, auction->GetAuctionCut()))
212 .AddMoney(profit)
213 .SendMailTo(trans, MailReceiver(owner, auction->owner), auction, MAIL_CHECK_MASK_COPIED, sWorld->getIntConfig(CONFIG_MAIL_DELIVERY_DELAY));
214 }

284void AuctionHouseMgr::LoadAuctionItems()
285{
286 uint32 oldMSTime = getMSTime();
287
288 // need to clear in case we are reloading
289 if (!mAitems.empty())
290 {
291 for (ItemMap::iterator itr = mAitems.begin(); itr != mAitems.end(); ++itr)
292 delete itr->second;
293
294 mAitems.clear();
295 }
296
297 // data needs to be at first place for Item::LoadFromDB
298 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_AUCTION_ITEMS);
299 PreparedQueryResult result = CharacterDatabase.Query(stmt);
300
301 if (!result)
302 {
303 TC_LOG_INFO("server.loading", ">> Loaded 0 auction items. DB table `auctionhouse` or `item_instance` is empty!");
304
305 return;
306 }
307
308 uint32 count = 0;
309
310 do
311 {
312 Field* fields = result->Fetch();
313
314 ObjectGuid::LowType item_guid = fields[11].GetUInt32();
315 uint32 itemEntry = fields[12].GetUInt32();
316
317 ItemTemplate const* proto = sObjectMgr->GetItemTemplate(itemEntry);
318 if (!proto)
319 {
320 TC_LOG_ERROR("misc", "AuctionHouseMgr::LoadAuctionItems: Unknown item (GUID: %u item entry: #%u) in auction, skipped.", item_guid, itemEntry);
321 continue;
322 }
323
324 Item* item = NewItemOrBag(proto);
325 if (!item->LoadFromDB(item_guid, ObjectGuid::Empty, fields, itemEntry))
326 {
327 delete item;
328 continue;
329 }
330 AddAItem(item);
331
332 ++count;
333 }
334 while (result->NextRow());
335
336 TC_LOG_INFO("server.loading", ">> Loaded %u auction items in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
337
338}

340void AuctionHouseMgr::LoadAuctions()
341{
342 uint32 oldMSTime = getMSTime();
343
344 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_AUCTIONS);
345 PreparedQueryResult result = CharacterDatabase.Query(stmt);
346
347 if (!result)
348 {
349 TC_LOG_INFO("server.loading", ">> Loaded 0 auctions. DB table `auctionhouse` is empty.");
350
351 return;
352 }
353
354 uint32 count = 0;
355
356 SQLTransaction trans = CharacterDatabase.BeginTransaction();
357 do
358 {
359 Field* fields = result->Fetch();
360
361 AuctionEntry* aItem = new AuctionEntry();
362 if (!aItem->LoadFromDB(fields))
363 {
364 aItem->DeleteFromDB(trans);
365 delete aItem;
366 continue;
367 }
368
369 GetAuctionsMapByHouseId(aItem->houseId)->AddAuction(aItem);
370 ++count;
371 } while (result->NextRow());
372
373 CharacterDatabase.CommitTransaction(trans);
374
375 TC_LOG_INFO("server.loading", ">> Loaded %u auctions in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
376
377}

426void AuctionHouseMgr::PendingAuctionProcess(Player* player)
427{
428 auto iterMap = pendingAuctionMap.find(player->GetGUID());
429 if (iterMap == pendingAuctionMap.end())
430 return;
431
432 PlayerAuctions* thisAH = iterMap->second.first;
433
434 SQLTransaction trans = CharacterDatabase.BeginTransaction();
435
436 uint32 totalItems = 0;
437 for (auto itrAH = thisAH->begin(); itrAH != thisAH->end(); ++itrAH)
438 {
439 AuctionEntry* AH = (*itrAH);
440 totalItems += AH->itemCount;
441 }
442
443 uint32 totaldeposit = 0;
444 auto itr = (*thisAH->begin());
445
446 if (Item* item = GetAItem(itr->itemGUIDLow))
447 totaldeposit = GetAuctionDeposit(itr->auctionHouseEntry, itr->etime, item, totalItems);
448
449 uint32 depositremain = totaldeposit;
450 for (auto itr = thisAH->begin(); itr != thisAH->end(); ++itr)
451 {
452 AuctionEntry* AH = (*itr);
453
454 if (next(itr) == thisAH->end())
455 AH->deposit = depositremain;
456 else
457 {
458 AH->deposit = totaldeposit / thisAH->size();
459 depositremain -= AH->deposit;
460 }
461
462 AH->DeleteFromDB(trans);
463 AH->SaveToDB(trans);
464 }
465
466 CharacterDatabase.CommitTransaction(trans);
467 pendingAuctionMap.erase(player->GetGUID());
468 delete thisAH;
469 player->ModifyMoney(-int32(totaldeposit));
470}

472void AuctionHouseMgr::UpdatePendingAuctions()
473{
474 for (auto itr = pendingAuctionMap.begin(); itr != pendingAuctionMap.end();)
475 {
476 ObjectGuid playerGUID = itr->first;
477 if (Player* player = ObjectAccessor::FindConnectedPlayer(playerGUID))
478 {
479 // Check if there were auctions since last update process if not
480 if (PendingAuctionCount(player) == itr->second.second)
481 {
482 ++itr;
483 PendingAuctionProcess(player);
484 }
485 else
486 {
487 ++itr;
488 pendingAuctionMap[playerGUID].second = PendingAuctionCount(player);
489 }
490 }
491 else
492 {
493 // Expire any auctions that we couldn't get a deposit for
494 TC_LOG_WARN("auctionHouse", "Player %s was offline, unable to retrieve deposit!", playerGUID.ToString().c_str());
495 PlayerAuctions* thisAH = itr->second.first;
496 ++itr;
497 SQLTransaction trans = CharacterDatabase.BeginTransaction();
498 for (auto AHitr = thisAH->begin(); AHitr != thisAH->end();)
499 {
500 AuctionEntry* AH = (*AHitr);
501 ++AHitr;
502 AH->expire_time = time(NULL);
503 AH->DeleteFromDB(trans);
504 AH->SaveToDB(trans);
505 }
506 CharacterDatabase.CommitTransaction(trans);
507 pendingAuctionMap.erase(playerGUID);
508 delete thisAH;
509 }
510 }

520AuctionHouseEntry const* AuctionHouseMgr::GetAuctionHouseEntry(uint32 factionTemplateId)
521{
522 uint32 houseid = AUCTIONHOUSE_NEUTRAL; // goblin auction house
523
524 if (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_AUCTION))
525 {
526 // FIXME: found way for proper auctionhouse selection by another way
527 // AuctionHouse.dbc have faction field with _player_ factions associated with auction house races.
528 // but no easy way convert creature faction to player race faction for specific city
529 FactionTemplateEntry const* u_entry = sFactionTemplateStore.LookupEntry(factionTemplateId);
530 if (!u_entry)
531 houseid = AUCTIONHOUSE_NEUTRAL; // goblin auction house
532 else if (u_entry->ourMask & FACTION_MASK_ALLIANCE)
533 houseid = AUCTIONHOUSE_ALLIANCE; // human auction house
534 else if (u_entry->ourMask & FACTION_MASK_HORDE)
535 houseid = AUCTIONHOUSE_HORDE; // orc auction house
536 else
537 houseid = AUCTIONHOUSE_NEUTRAL; // goblin auction house
538 }
539
540 return sAuctionHouseStore.LookupEntry(houseid);
541}

567void AuctionHouseObject::Update()
568{
569 time_t curTime = sWorld->GetGameTime();
570 ///- Handle expired auctions
571
572 // If storage is empty, no need to update. next == NULL in this case.
573 if (AuctionsMap.empty())
574 return;
575
576 // Clear expired throttled players
577 for (PlayerGetAllThrottleMap::const_iterator itr = GetAllThrottleMap.begin(); itr != GetAllThrottleMap.end();)
578 {
579 if (itr->second <= curTime)
580 itr = GetAllThrottleMap.erase(itr);
581 else
582 ++itr;
583 }
584
585 SQLTransaction trans = CharacterDatabase.BeginTransaction();
586
587 for (AuctionEntryMap::iterator it = AuctionsMap.begin(); it != AuctionsMap.end();)
588 {
589 // from auctionhousehandler.cpp, creates auction pointer & player pointer
590 AuctionEntry* auction = it->second;
591 // Increment iterator due to AuctionEntry deletion
592 ++it;
593
594 ///- filter auctions expired on next update
595 if (auction->expire_time > curTime + 60)
596 continue;
597
598 ///- Either cancel the auction if there was no bidder
599 if (auction->bidder == 0 && auction->bid == 0)
600 {
601 sAuctionMgr->SendAuctionExpiredMail(auction, trans);
602 sScriptMgr->OnAuctionExpire(this, auction);
603 }
604 ///- Or perform the transaction
605 else
606 {
607 //we should send an "item sold" message if the seller is online
608 //we send the item to the winner
609 //we send the money to the seller
610 sAuctionMgr->SendAuctionSuccessfulMail(auction, trans);
611 sAuctionMgr->SendAuctionWonMail(auction, trans);
612 sScriptMgr->OnAuctionSuccessful(this, auction);
613 }
614
615 ///- In any case clear the auction
616 auction->DeleteFromDB(trans);
617
618 sAuctionMgr->RemoveAItem(auction->itemGUIDLow);
619 RemoveAuction(auction);
620 }
621
622 // Run DB changes
623 CharacterDatabase.CommitTransaction(trans);
624}

656void AuctionHouseObject::BuildListAuctionItems(WorldPacket& data, Player* player,
657 std::wstring const& wsearchedname, uint32 listfrom, uint8 levelmin, uint8 levelmax, uint8 usable,
658 uint32 inventoryType, uint32 itemClass, uint32 itemSubClass, uint32 quality,
659 uint32& count, uint32& totalcount, bool getall)
660{
661 int loc_idx = player->GetSession()->GetSessionDbLocaleIndex();
662 int locdbc_idx = player->GetSession()->GetSessionDbcLocale();
663
664 time_t curTime = sWorld->GetGameTime();
665
666 PlayerGetAllThrottleMap::const_iterator itr = GetAllThrottleMap.find(player->GetGUID());
667 time_t throttleTime = itr != GetAllThrottleMap.end() ? itr->second : curTime;
668
669 if (getall && throttleTime <= curTime)
670 {
671 for (AuctionEntryMap::const_iterator itr = AuctionsMap.begin(); itr != AuctionsMap.end(); ++itr)
672 {
673 AuctionEntry* Aentry = itr->second;
674 // Skip expired auctions
675 if (Aentry->expire_time < curTime)
676 continue;
677
678 Item* item = sAuctionMgr->GetAItem(Aentry->itemGUIDLow);
679 if (!item)
680 continue;
681
682 ++count;
683 ++totalcount;
684 Aentry->BuildAuctionInfo(data, item);
685
686 if (count >= MAX_GETALL_RETURN)
687 break;
688 }
689 GetAllThrottleMap[player->GetGUID()] = curTime + sWorld->getIntConfig(CONFIG_AUCTION_GETALL_DELAY);
690 return;
691 }
692
693 for (AuctionEntryMap::const_iterator itr = AuctionsMap.begin(); itr != AuctionsMap.end(); ++itr)
694 {
695 AuctionEntry* Aentry = itr->second;
696 // Skip expired auctions
697 if (Aentry->expire_time < curTime)
698 continue;
699
700 Item* item = sAuctionMgr->GetAItem(Aentry->itemGUIDLow);
701 if (!item)
702 continue;
703
704 ItemTemplate const* proto = item->GetTemplate();
705
706 if (itemClass != 0xffffffff && proto->Class != itemClass)
707 continue;
708
709 if (itemSubClass != 0xffffffff && proto->SubClass != itemSubClass)
710 continue;
711
712 if (inventoryType != 0xffffffff && proto->InventoryType != inventoryType)
713 continue;
714
715 if (quality != 0xffffffff && proto->Quality != quality)
716 continue;
717
718 if (levelmin != 0x00 && (proto->RequiredLevel < levelmin || (levelmax != 0x00 && proto->RequiredLevel > levelmax)))
719 continue;
720
721 if (usable != 0x00 && player->CanUseItem(item) != EQUIP_ERR_OK)
722 continue;
723
724 // Allow search by suffix (ie: of the Monkey) or partial name (ie: Monkey)
725 // No need to do any of this if no search term was entered
726 if (!wsearchedname.empty())
727 {
728 std::string name = proto->Name1;
729 if (name.empty())
730 continue;
731
732 // local name
733 if (loc_idx >= 0)
734 if (ItemLocale const* il = sObjectMgr->GetItemLocale(proto->ItemId))
735 ObjectMgr::GetLocaleString(il->Name, loc_idx, name);
736
737 // DO NOT use GetItemEnchantMod(proto->RandomProperty) as it may return a result
738 // that matches the search but it may not equal item->GetItemRandomPropertyId()
739 // used in BuildAuctionInfo() which then causes wrong items to be listed
740 int32 propRefID = item->GetItemRandomPropertyId();
741
742 if (propRefID)
743 {
744 // Append the suffix to the name (ie: of the Monkey) if one exists
745 // These are found in ItemRandomSuffix.dbc and ItemRandomProperties.dbc
746 // even though the DBC names seem misleading
747
748 char* const* suffix = nullptr;
749
750 if (propRefID < 0)
751 {
752 const ItemRandomSuffixEntry* itemRandSuffix = sItemRandomSuffixStore.LookupEntry(-propRefID);
753 if (itemRandSuffix)
754 suffix = itemRandSuffix->nameSuffix;
755 }
756 else
757 {
758 const ItemRandomPropertiesEntry* itemRandProp = sItemRandomPropertiesStore.LookupEntry(propRefID);
759 if (itemRandProp)
760 suffix = itemRandProp->nameSuffix;
761 }
762
763 // dbc local name
764 if (suffix)
765 {
766 // Append the suffix (ie: of the Monkey) to the name using localization
767 // or default enUS if localization is invalid
768 name += ' ';
769 name += suffix[locdbc_idx >= 0 ? locdbc_idx : LOCALE_enUS];
770 }
771 }
772
773 // Perform the search (with or without suffix)
774 if (!Utf8FitTo(name, wsearchedname))
775 continue;
776 }
777
778 // Add the item if no search term or if entered search term was found
779 if (count < 50 && totalcount >= listfrom)
780 {
781 ++count;
782 Aentry->BuildAuctionInfo(data, item);
783 }
784 ++totalcount;
785 }
786}

788//this function inserts to WorldPacket auction's data
789bool AuctionEntry::BuildAuctionInfo(WorldPacket& data, Item* sourceItem) const
790{
791 Item* item = (sourceItem) ? sourceItem : sAuctionMgr->GetAItem(itemGUIDLow);
792 if (!item)
793 {
794 TC_LOG_ERROR("misc", "AuctionEntry::BuildAuctionInfo: Auction %u has a non-existent item: %u", Id, itemGUIDLow);
795 return false;
796 }
797 data << uint32(Id);
798 data << uint32(item->GetEntry());
799
800 for (uint8 i = 0; i < MAX_INSPECTED_ENCHANTMENT_SLOT; ++i)
801 {
802 data << uint32(item->GetEnchantmentId(EnchantmentSlot(i)));
803 data << uint32(item->GetEnchantmentDuration(EnchantmentSlot(i)));
804 data << uint32(item->GetEnchantmentCharges(EnchantmentSlot(i)));
805 }
806
807 data << int32(item->GetItemRandomPropertyId()); // Random item property id
808 data << uint32(item->GetItemSuffixFactor()); // SuffixFactor
809 data << uint32(item->GetCount()); // item->count
810 data << uint32(item->GetSpellCharges()); // item->charge FFFFFFF
811 data << uint32(0); // Unknown
812 data << uint64(owner); // Auction->owner
813 data << uint32(startbid); // Auction->startbid (not sure if useful)
814 data << uint32(bid ? GetAuctionOutBid() : 0);
815 // Minimal outbid
816 data << uint32(buyout); // Auction->buyout
817 data << uint32((expire_time - time(NULL)) * IN_MILLISECONDS); // time left
818 data << uint64(bidder); // auction->bidder current
819 data << uint32(bid); // current bid
820 return true;
821}

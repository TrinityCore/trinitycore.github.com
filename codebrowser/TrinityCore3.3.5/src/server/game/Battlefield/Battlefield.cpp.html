69// Called when a player enters the zone
70void Battlefield::HandlePlayerEnterZone(Player* player, uint32 /*zone*/)
71{
72 // If battle is started,
73 // If not full of players > invite player to join the war
74 // If full of players > announce to player that BF is full and kick him after a few second if he desn't leave
75 if (IsWarTime())
76 {
77 if (m_PlayersInWar[player->GetTeamId()].size() + m_InvitedPlayers[player->GetTeamId()].size() < m_MaxPlayer) // Vacant spaces
78 InvitePlayerToWar(player);
79 else // No more vacant places
80 {
81 /// @todo Send a packet to announce it to player
82 m_PlayersWillBeKick[player->GetTeamId()][player->GetGUID()] = time(NULL) + 10;
83 InvitePlayerToQueue(player);
84 }
85 }
86 else
87 {
88 // If time left is < 15 minutes invite player to join queue
89 if (m_Timer <= m_StartGroupingTimer)
90 InvitePlayerToQueue(player);
91 }
92
93 // Add player in the list of player in zone
94 m_players[player->GetTeamId()].insert(player->GetGUID());
95 OnPlayerEnterZone(player);
96}

98// Called when a player leave the zone
99void Battlefield::HandlePlayerLeaveZone(Player* player, uint32 /*zone*/)
100{
101 if (IsWarTime())
102 {
103 // If the player is participating to the battle
104 if (m_PlayersInWar[player->GetTeamId()].find(player->GetGUID()) != m_PlayersInWar[player->GetTeamId()].end())
105 {
106 m_PlayersInWar[player->GetTeamId()].erase(player->GetGUID());
107 player->GetSession()->SendBfLeaveMessage(m_BattleId);
108 if (Group* group = player->GetGroup()) // Remove the player from the raid group
109 group->RemoveMember(player->GetGUID());
110
111 OnPlayerLeaveWar(player);
112 }
113 }
114
115 for (BfCapturePointMap::iterator itr = m_capturePoints.begin(); itr != m_capturePoints.end(); ++itr)
116 itr->second->HandlePlayerLeave(player);
117
118 m_InvitedPlayers[player->GetTeamId()].erase(player->GetGUID());
119 m_PlayersWillBeKick[player->GetTeamId()].erase(player->GetGUID());
120 m_players[player->GetTeamId()].erase(player->GetGUID());
121 SendRemoveWorldStates(player);
122 RemovePlayerFromResurrectQueue(player->GetGUID());
123 OnPlayerLeaveZone(player);
124}

126bool Battlefield::Update(uint32 diff)
127{
128 if (m_Timer <= diff)
129 {
130 // Battlefield ends on time
131 if (IsWarTime())
132 EndBattle(true);
133 else // Time to start a new battle!
134 StartBattle();
135 }
136 else
137 m_Timer -= diff;
138
139 // Invite players a few minutes before the battle's beginning
140 if (!IsWarTime() && !m_StartGrouping && m_Timer <= m_StartGroupingTimer)
141 {
142 m_StartGrouping = true;
143 InvitePlayersInZoneToQueue();
144 OnStartGrouping();
145 }
146
147 bool objective_changed = false;
148 if (IsWarTime())
149 {
150 if (m_uiKickAfkPlayersTimer <= diff)
151 {
152 m_uiKickAfkPlayersTimer = 1000;
153 KickAfkPlayers();
154 }
155 else
156 m_uiKickAfkPlayersTimer -= diff;
157
158 // Kick players who chose not to accept invitation to the battle
159 if (m_uiKickDontAcceptTimer <= diff)
160 {
161 time_t now = time(NULL);
162 for (int team = 0; team < 2; team++)
163 for (PlayerTimerMap::iterator itr = m_InvitedPlayers[team].begin(); itr != m_InvitedPlayers[team].end(); ++itr)
164 if (itr->second <= now)
165 KickPlayerFromBattlefield(itr->first);
166
167 InvitePlayersInZoneToWar();
168 for (int team = 0; team < 2; team++)
169 for (PlayerTimerMap::iterator itr = m_PlayersWillBeKick[team].begin(); itr != m_PlayersWillBeKick[team].end(); ++itr)
170 if (itr->second <= now)
171 KickPlayerFromBattlefield(itr->first);
172
173 m_uiKickDontAcceptTimer = 1000;
174 }
175 else
176 m_uiKickDontAcceptTimer -= diff;
177
178 for (BfCapturePointMap::iterator itr = m_capturePoints.begin(); itr != m_capturePoints.end(); ++itr)
179 if (itr->second->Update(diff))
180 objective_changed = true;
181 }
182
183
184 if (m_LastResurrectTimer <= diff)
185 {
186 for (uint8 i = 0; i < m_GraveyardList.size(); i++)
187 if (GetGraveyardById(i))
188 m_GraveyardList[i]->Resurrect();
189 m_LastResurrectTimer = RESURRECTION_INTERVAL;
190 }
191 else
192 m_LastResurrectTimer -= diff;
193
194 return objective_changed;
195}

206{

208 return;
209
210 if (m_PlayersInQueue[player->GetTeamId()].size() <= m_MinPlayer || m_PlayersInQueue[GetOtherTeam(player->GetTeamId())].size() >= m_MinPlayer)
211 player->GetSession()->SendBfInvitePlayerToQueue(m_BattleId);

234void Battlefield::InvitePlayersInZoneToWar()
235{
236 for (uint8 team = 0; team < BG_TEAMS_COUNT; ++team)
237 for (GuidSet::const_iterator itr = m_players[team].begin(); itr != m_players[team].end(); ++itr)
238 {
239 if (Player* player = ObjectAccessor::FindPlayer(*itr))
240 {
241 if (m_PlayersInWar[player->GetTeamId()].count(player->GetGUID()) || m_InvitedPlayers[player->GetTeamId()].count(player->GetGUID()))
242 continue;
243 if (m_PlayersInWar[player->GetTeamId()].size() + m_InvitedPlayers[player->GetTeamId()].size() < m_MaxPlayer)
244 InvitePlayerToWar(player);
245 else // Battlefield is full of players
246 m_PlayersWillBeKick[player->GetTeamId()][player->GetGUID()] = time(NULL) + 10;
247 }
248 }

251void Battlefield::InvitePlayerToWar(Player* player)
252{
253 if (!player)
254 return;
255
256 /// @todo needed ?
257 if (player->IsInFlight())
258 return;
259
260 if (player->InArena() || player->GetBattleground())
261 {
262 m_PlayersInQueue[player->GetTeamId()].erase(player->GetGUID());
263 return;
264 }
265
266 // If the player does not match minimal level requirements for the battlefield, kick him
267 if (player->getLevel() < m_MinLevel)
268 {
269 if (m_PlayersWillBeKick[player->GetTeamId()].count(player->GetGUID()) == 0)
270 m_PlayersWillBeKick[player->GetTeamId()][player->GetGUID()] = time(NULL) + 10;
271 return;
272 }
273
274 // Check if player is not already in war
275 if (m_PlayersInWar[player->GetTeamId()].count(player->GetGUID()) || m_InvitedPlayers[player->GetTeamId()].count(player->GetGUID()))
276 return;
277
278 m_PlayersWillBeKick[player->GetTeamId()].erase(player->GetGUID());
279 m_InvitedPlayers[player->GetTeamId()][player->GetGUID()] = time(NULL) + m_TimeForAcceptInvite;
280 player->GetSession()->SendBfInvitePlayerToWar(m_BattleId, m_ZoneId, m_TimeForAcceptInvite);
281}

307void Battlefield::StartBattle()
308{
309 if (m_isActive)
310 return;
311
312 for (int team = 0; team < BG_TEAMS_COUNT; team++)
313 {
314 m_PlayersInWar[team].clear();
315 m_Groups[team].clear();
316 }
317
318 m_Timer = m_BattleTime;
319 m_isActive = true;
320
321 InvitePlayersInZoneToWar();
322 InvitePlayersInQueueToWar();
323
324 OnBattleStart();
325}
326
327void Battlefield::EndBattle(bool endByTimer)
328{
329 if (!m_isActive)
330 return;
331
332 m_isActive = false;
333
334 m_StartGrouping = false;
335
336 if (!endByTimer)
337 SetDefenderTeam(GetAttackerTeam());
338
339 OnBattleEnd(endByTimer);
340
341 // Reset battlefield timer
342 m_Timer = m_NoWarBattleTime;
343 SendInitWorldStatesToAll();
344}

503bool Battlefield::AddOrSetPlayerToCorrectBfGroup(Player* player)
504{
505 if (!player->IsInWorld())
506 return false;
507
508 if (Group* group = player->GetGroup())
509 group->RemoveMember(player->GetGUID());
510
511 Group* group = GetFreeBfRaid(player->GetTeamId());
512 if (!group)
513 {
514 group = new Group;
515 group->SetBattlefieldGroup(this);
516 group->Create(player);
517 sGroupMgr->AddGroup(group);
518 m_Groups[player->GetTeamId()].insert(group->GetGUID());
519 }
520 else if (group->IsMember(player->GetGUID()))
521 {
522 uint8 subgroup = group->GetMemberGroup(player->GetGUID());
523 player->SetBattlegroundOrBattlefieldRaid(group, subgroup);
524 }
525 else
526 group->AddMember(player);
527
528 return true;
529}

555WorldSafeLocsEntry const* Battlefield::GetClosestGraveYard(Player* player)
556{
557 BfGraveyard* closestGY = NULL;
558 float maxdist = -1;
559 for (uint8 i = 0; i < m_GraveyardList.size(); i++)
560 {
561 if (m_GraveyardList[i])
562 {
563 if (m_GraveyardList[i]->GetControlTeamId() != player->GetTeamId())
564 continue;
565
566 float dist = m_GraveyardList[i]->GetDistance(player);
567 if (dist < maxdist || maxdist < 0)
568 {
569 closestGY = m_GraveyardList[i];
570 maxdist = dist;
571 }
572 }
573 }
574
575 if (closestGY)
576 return sWorldSafeLocsStore.LookupEntry(closestGY->GetGraveyardId());
577
578 return NULL;
579}

675void BfGraveyard::Resurrect()
676{
677 if (m_ResurrectQueue.empty())
678 return;
679
680 for (GuidSet::const_iterator itr = m_ResurrectQueue.begin(); itr != m_ResurrectQueue.end(); ++itr)
681 {
682 // Get player object from his guid
683 Player* player = ObjectAccessor::FindPlayer(*itr);
684 if (!player)
685 continue;
686
687 // Check if the player is in world and on the good graveyard
688 if (player->IsInWorld())
689 if (Creature* spirit = m_Bf->GetCreature(m_SpiritGuide[m_ControlTeam]))
690 spirit->CastSpell(spirit, SPELL_SPIRIT_HEAL, true);
691
692 // Resurrect player
693 player->CastSpell(player, SPELL_RESURRECTION_VISUAL, true);
694 player->ResurrectPlayer(1.0f);
695 player->CastSpell(player, 6962, true);
696 player->CastSpell(player, SPELL_SPIRIT_HEAL_MANA, true);
697
698 player->SpawnCorpseBones(false);
699 }
700
701 m_ResurrectQueue.clear();
702}

898bool BfCapturePoint::SetCapturePointData(GameObject* capturePoint)
899{
900 ASSERT(capturePoint);
901
902 TC_LOG_DEBUG("bg.battlefield", "Creating capture point %u", capturePoint->GetEntry());
903
904 m_capturePointGUID = ObjectGuid(HighGuid::GameObject, capturePoint->GetEntry(), capturePoint->GetGUID().GetCounter());
905
906 // check info existence
907 GameObjectTemplate const* goinfo = capturePoint->GetGOInfo();

911 return false;
912 }
913
914 // get the needed values from goinfo
915 m_maxValue = goinfo->capturePoint.maxTime;
916 m_maxSpeed = m_maxValue / (goinfo->capturePoint.minTime ? goinfo->capturePoint.minTime : 60);
917 m_neutralValuePct = goinfo->capturePoint.neutralPercent;
918 m_minValue = m_maxValue * goinfo->capturePoint.neutralPercent / 100;
919 m_capturePointEntry = capturePoint->GetEntry();
920 if (m_team == TEAM_ALLIANCE)
921 {
922 m_value = m_maxValue;
923 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_ALLIANCE;
924 }
925 else
926 {
927 m_value = -m_maxValue;
928 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_HORDE;
929 }

955bool BfCapturePoint::Update(uint32 diff)
956{
957 if (!m_capturePointGUID)
958 return false;
959
960 if (GameObject* capturePoint = m_Bf->GetGameObject(m_capturePointGUID))
961 {
962 float radius = capturePoint->GetGOInfo()->capturePoint.radius;
963
964 for (uint8 team = 0; team < 2; ++team)
965 {
966 for (GuidSet::iterator itr = m_activePlayers[team].begin(); itr != m_activePlayers[team].end();)
967 {
968 if (Player* player = ObjectAccessor::FindPlayer(*itr))
969 {
970 if (!capturePoint->IsWithinDistInMap(player, radius) || !player->IsOutdoorPvPActive())
971 itr = HandlePlayerLeave(player);
972 else
973 ++itr;
974 }
975 else
976 ++itr;
977 }
978 }
979
980 std::list<Player*> players;
981 Trinity::AnyPlayerInObjectRangeCheck checker(capturePoint, radius);
982 Trinity::PlayerListSearcher<Trinity::AnyPlayerInObjectRangeCheck> searcher(capturePoint, players, checker);
983 capturePoint->VisitNearbyWorldObject(radius, searcher);
984
985 for (std::list<Player*>::iterator itr = players.begin(); itr != players.end(); ++itr)
986 if ((*itr)->IsOutdoorPvPActive())
987 if (m_activePlayers[(*itr)->GetTeamId()].insert((*itr)->GetGUID()).second)
988 HandlePlayerEnter(*itr);
989 }
990
991 // get the difference of numbers
992 float fact_diff = ((float) m_activePlayers[0].size() - (float) m_activePlayers[1].size()) * diff / BATTLEFIELD_OBJECTIVE_UPDATE_INTERVAL;
993 if (G3D::fuzzyEq(fact_diff, 0.0f))
994 return false;
995
996 uint32 Challenger = 0;
997 float maxDiff = m_maxSpeed * diff;
998
999 if (fact_diff < 0)
1000 {
1001 // horde is in majority, but it's already horde-controlled -> no change
1002 if (m_State == BF_CAPTUREPOINT_OBJECTIVESTATE_HORDE && m_value <= -m_maxValue)
1003 return false;
1004
1005 if (fact_diff < -maxDiff)
1006 fact_diff = -maxDiff;
1007
1008 Challenger = HORDE;
1009 }
1010 else
1011 {
1012 // ally is in majority, but it's already ally-controlled -> no change
1013 if (m_State == BF_CAPTUREPOINT_OBJECTIVESTATE_ALLIANCE && m_value >= m_maxValue)
1014 return false;
1015
1016 if (fact_diff > maxDiff)
1017 fact_diff = maxDiff;
1018
1019 Challenger = ALLIANCE;
1020 }
1021
1022 float oldValue = m_value;
1023 TeamId oldTeam = m_team;
1024
1025 m_OldState = m_State;
1026
1027 m_value += fact_diff;
1028
1029 if (m_value < -m_minValue) // red
1030 {
1031 if (m_value < -m_maxValue)
1032 m_value = -m_maxValue;
1033 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_HORDE;
1034 m_team = TEAM_HORDE;
1035 }
1036 else if (m_value > m_minValue) // blue
1037 {
1038 if (m_value > m_maxValue)
1039 m_value = m_maxValue;
1040 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_ALLIANCE;
1041 m_team = TEAM_ALLIANCE;
1042 }
1043 else if (oldValue * m_value <= 0) // grey, go through mid point
1044 {
1045 // if challenger is ally, then n->a challenge
1046 if (Challenger == ALLIANCE)
1047 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE;
1048 // if challenger is horde, then n->h challenge
1049 else if (Challenger == HORDE)
1050 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE;
1051 m_team = TEAM_NEUTRAL;
1052 }
1053 else // grey, did not go through mid point
1054 {
1055 // old phase and current are on the same side, so one team challenges the other
1056 if (Challenger == ALLIANCE && (m_OldState == BF_CAPTUREPOINT_OBJECTIVESTATE_HORDE || m_OldState == BF_CAPTUREPOINT_OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE))
1057 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE;
1058 else if (Challenger == HORDE && (m_OldState == BF_CAPTUREPOINT_OBJECTIVESTATE_ALLIANCE || m_OldState == BF_CAPTUREPOINT_OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE))
1059 m_State = BF_CAPTUREPOINT_OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE;
1060 m_team = TEAM_NEUTRAL;
1061 }
1062
1063 if (G3D::fuzzyNe(m_value, oldValue))
1064 SendChangePhase();
1065
1066 if (m_OldState != m_State)
1067 {
1068 //TC_LOG_ERROR("bg.battlefield", "%u->%u", m_OldState, m_State);
1069 if (oldTeam != m_team)
1070 ChangeTeam(oldTeam);
1071 return true;
1072 }
1073
1074 return false;
1075}

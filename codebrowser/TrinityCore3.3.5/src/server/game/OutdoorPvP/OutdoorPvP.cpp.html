30class DefenseMessageBuilder
31{
32 public:
33 DefenseMessageBuilder(uint32 zoneId, uint32 id)
34 : _zoneId(zoneId), _id(id) { }
35
36 void operator()(WorldPacket& data, LocaleConstant locale) const
37 {
38 std::string text = sOutdoorPvPMgr->GetDefenseMessage(_zoneId, _id, locale);
39
40 data.Initialize(SMSG_DEFENSE_MESSAGE, 4 + 4 + text.length());
41 data.append<uint32>(_zoneId);
42 data.append<uint32>(text.length());
43 data << text;
44 }
45
46 private:
47 uint32 _zoneId; ///< ZoneId
48 uint32 _id; ///< BroadcastTextId
49};

57bool OPvPCapturePoint::HandlePlayerEnter(Player* player)
58{
59 if (m_capturePoint)
60 {
61 player->SendUpdateWorldState(m_capturePoint->GetGOInfo()->capturePoint.worldState1, 1);
62 player->SendUpdateWorldState(m_capturePoint->GetGOInfo()->capturePoint.worldstate2, (uint32)ceil((m_value + m_maxValue) / (2 * m_maxValue) * 100.0f));
63 player->SendUpdateWorldState(m_capturePoint->GetGOInfo()->capturePoint.worldstate3, m_neutralValuePct);
64 }
65 return m_activePlayers[player->GetTeamId()].insert(player->GetGUID()).second;
66}
67
68void OPvPCapturePoint::HandlePlayerLeave(Player* player)
69{
70 if (m_capturePoint)
71 player->SendUpdateWorldState(m_capturePoint->GetGOInfo()->capturePoint.worldState1, 0);
72 m_activePlayers[player->GetTeamId()].erase(player->GetGUID());
73}

138bool OPvPCapturePoint::SetCapturePointData(uint32 entry, uint32 map, float x, float y, float z, float o, float rotation0, float rotation1, float rotation2, float rotation3)
139{
140 TC_LOG_DEBUG("outdoorpvp", "Creating capture point %u", entry);
141
142 // check info existence
143 GameObjectTemplate const* goinfo = sObjectMgr->GetGameObjectTemplate(entry);
144 if (!goinfo || goinfo->type != GAMEOBJECT_TYPE_CAPTURE_POINT)
145 {
146 TC_LOG_ERROR("outdoorpvp", "OutdoorPvP: GO %u is not capture point!", entry);
147 return false;
148 }
149
150 m_capturePointSpawnId = sObjectMgr->AddGOData(entry, map, x, y, z, o, 0, rotation0, rotation1, rotation2, rotation3);
151
152 if (m_capturePointSpawnId == 0)
153 return false;
154
155 // get the needed values from goinfo
156 m_maxValue = (float)goinfo->capturePoint.maxTime;
157 m_maxSpeed = m_maxValue / (goinfo->capturePoint.minTime ? goinfo->capturePoint.minTime : 60);
158 m_neutralValuePct = goinfo->capturePoint.neutralPercent;
159 m_minValue = CalculatePct(m_maxValue, m_neutralValuePct);
160
161 return true;
162}

164bool OPvPCapturePoint::DelCreature(uint32 type)
165{
166 uint32 spawnId = m_Creatures[type];
167 if (!spawnId)
168 {
169 TC_LOG_DEBUG("outdoorpvp", "opvp creature type %u was already deleted", type);
170 return false;
171 }
172
173 auto bounds = m_PvP->GetMap()->GetCreatureBySpawnIdStore().equal_range(spawnId);
174 for (auto itr = bounds.first; itr != bounds.second;)
175 {
176 Creature* c = itr->second;
177 ++itr;
178 // Don't save respawn time
179 c->SetRespawnTime(0);
180 c->RemoveCorpse();
181 c->AddObjectToRemoveList();
182 }
183
184 TC_LOG_DEBUG("outdoorpvp", "deleting opvp creature type %u", type);
185 // explicit removal from map
186 // beats me why this is needed, but with the recent removal "cleanup" some creatures stay in the map if "properly" deleted
187 // so this is a big fat workaround, if AddObjectToRemoveList and DoDelayedMovesAndRemoves worked correctly, this wouldn't be needed
188 //if (Map* map = sMapMgr->FindMap(cr->GetMapId()))
189 // map->Remove(cr, false);
190 // delete respawn time for this creature
191 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CREATURE_RESPAWN);
192 stmt->setUInt32(0, spawnId);
193 stmt->setUInt16(1, m_PvP->GetMap()->GetId()); stmt->setUInt32(2, 0); // instance id, always 0 for world maps
194 CharacterDatabase.Execute(stmt);
195
196 sObjectMgr->DeleteCreatureData(spawnId);
197 m_CreatureTypes[m_Creatures[type]] = 0;
198 m_Creatures[type] = 0;
199 return true;
200}

202bool OPvPCapturePoint::DelObject(uint32 type)
203{

205 return false;
206
207 uint32 spawnId = m_Objects[type];

246void OutdoorPvP::DeleteSpawns()
247{
248 // Remove script from any registered gameobjects/creatures
249 for (auto itr = m_GoScriptStore.begin(); itr != m_GoScriptStore.end(); ++itr)
250 {
251 if (GameObject* go = itr->second)
252 go->ClearZoneScript();
253 }
254 m_GoScriptStore.clear();
255
256 for (auto itr = m_CreatureScriptStore.begin(); itr != m_CreatureScriptStore.end(); ++itr)
257 {
258 if (Creature* creature = itr->second)
259 creature->ClearZoneScript();
260 }
261 m_CreatureScriptStore.clear();
262
263 for (OPvPCapturePointMap::iterator itr = m_capturePoints.begin(); itr != m_capturePoints.end(); ++itr)
264 {
265 itr->second->DeleteSpawns();
266 delete itr->second;
267 }
268 m_capturePoints.clear();
269}

308bool OPvPCapturePoint::Update(uint32 diff)
309{
310 if (!m_capturePoint)
311 return false;
312
313 float radius = (float)m_capturePoint->GetGOInfo()->capturePoint.radius;
314
315 for (uint32 team = 0; team < 2; ++team)
316 {
317 for (GuidSet::iterator itr = m_activePlayers[team].begin(); itr != m_activePlayers[team].end();)
318 {
319 ObjectGuid playerGuid = *itr;
320 ++itr;
321
322 if (Player* player = ObjectAccessor::FindPlayer(playerGuid))
323 if (!m_capturePoint->IsWithinDistInMap(player, radius) || !player->IsOutdoorPvPActive())
324 HandlePlayerLeave(player);
325 }
326 }
327
328 std::list<Player*> players;
329 Trinity::AnyPlayerInObjectRangeCheck checker(m_capturePoint, radius);
330 Trinity::PlayerListSearcher<Trinity::AnyPlayerInObjectRangeCheck> searcher(m_capturePoint, players, checker);
331 m_capturePoint->VisitNearbyWorldObject(radius, searcher);
332
333 for (std::list<Player*>::iterator itr = players.begin(); itr != players.end(); ++itr)
334 {
335 Player* const player = *itr;
336 if (player->IsOutdoorPvPActive())
337 {
338 if (m_activePlayers[player->GetTeamId()].insert(player->GetGUID()).second)
339 HandlePlayerEnter(*itr);
340 }
341 }
342
343 // get the difference of numbers
344 float fact_diff = ((float)m_activePlayers[0].size() - (float)m_activePlayers[1].size()) * diff / OUTDOORPVP_OBJECTIVE_UPDATE_INTERVAL;
345 if (!fact_diff)
346 return false;
347
348 uint32 Challenger = 0;
349 float maxDiff = m_maxSpeed * diff;
350
351 if (fact_diff < 0)
352 {
353 // horde is in majority, but it's already horde-controlled -> no change
354 if (m_State == OBJECTIVESTATE_HORDE && m_value <= -m_maxValue)
355 return false;
356
357 if (fact_diff < -maxDiff)
358 fact_diff = -maxDiff;
359
360 Challenger = HORDE;
361 }
362 else
363 {
364 // ally is in majority, but it's already ally-controlled -> no change
365 if (m_State == OBJECTIVESTATE_ALLIANCE && m_value >= m_maxValue)
366 return false;
367
368 if (fact_diff > maxDiff)
369 fact_diff = maxDiff;
370
371 Challenger = ALLIANCE;
372 }
373
374 float oldValue = m_value;
375 TeamId oldTeam = m_team;
376
377 m_OldState = m_State;
378
379 m_value += fact_diff;
380
381 if (m_value < -m_minValue) // red
382 {
383 if (m_value < -m_maxValue)
384 m_value = -m_maxValue;
385 m_State = OBJECTIVESTATE_HORDE;
386 m_team = TEAM_HORDE;
387 }
388 else if (m_value > m_minValue) // blue
389 {
390 if (m_value > m_maxValue)
391 m_value = m_maxValue;
392 m_State = OBJECTIVESTATE_ALLIANCE;
393 m_team = TEAM_ALLIANCE;
394 }
395 else if (oldValue * m_value <= 0) // grey, go through mid point
396 {
397 // if challenger is ally, then n->a challenge
398 if (Challenger == ALLIANCE)
399 m_State = OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE;
400 // if challenger is horde, then n->h challenge
401 else if (Challenger == HORDE)
402 m_State = OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE;
403 m_team = TEAM_NEUTRAL;
404 }
405 else // grey, did not go through mid point
406 {
407 // old phase and current are on the same side, so one team challenges the other
408 if (Challenger == ALLIANCE && (m_OldState == OBJECTIVESTATE_HORDE || m_OldState == OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE))
409 m_State = OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE;
410 else if (Challenger == HORDE && (m_OldState == OBJECTIVESTATE_ALLIANCE || m_OldState == OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE))
411 m_State = OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE;
412 m_team = TEAM_NEUTRAL;
413 }
414
415 if (m_value != oldValue)
416 SendChangePhase();
417
418 if (m_OldState != m_State)
419 {
420 //TC_LOG_ERROR("outdoorpvp", "%u->%u", m_OldState, m_State);
421 if (oldTeam != m_team)
422 ChangeTeam(oldTeam);
423 ChangeState();
424 return true;
425 }
426
427 return false;
428}

471void OutdoorPvP::HandleKill(Player* killer, Unit* killed)
472{
473 if (Group* group = killer->GetGroup())
474 {
475 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
476 {
477 Player* groupGuy = itr->GetSource();
478
479 if (!groupGuy)
480 continue;
481
482 // skip if too far away
483 if (!groupGuy->IsAtGroupRewardDistance(killed))
484 continue;
485
486 // creature kills must be notified, even if not inside objective / not outdoor pvp active
487 // player kills only count if active and inside objective
488 if ((groupGuy->IsOutdoorPvPActive() && IsInsideObjective(groupGuy)) || killed->GetTypeId() == TYPEID_UNIT)
489 HandleKillImpl(groupGuy, killed);
490 }
491 }
492 else
493 {
494 // creature kills must be notified, even if not inside objective / not outdoor pvp active
495 if ((killer->IsOutdoorPvPActive() && IsInsideObjective(killer)) || killed->GetTypeId() == TYPEID_UNIT)
496 HandleKillImpl(killer, killed);
497 }
498}

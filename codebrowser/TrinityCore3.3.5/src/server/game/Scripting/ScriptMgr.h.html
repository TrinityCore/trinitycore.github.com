84/*
85 Standard procedure when adding new script type classes:
86
87 First of all, define the actual class, and have it inherit from ScriptObject, like so:
88
89 class MyScriptType : public ScriptObject
90 {
91 uint32 _someId;
92
93 private:
94
95 void RegisterSelf();
96
97 protected:
98
99 MyScriptType(const char* name, uint32 someId)
100 : ScriptObject(name), _someId(someId)
101 {
102 ScriptRegistry<MyScriptType>::AddScript(this);
103 }
104
105 public:
106
107 // If a virtual function in your script type class is not necessarily
108 // required to be overridden, just declare it virtual with an empty
109 // body. If, on the other hand, it's logical only to override it (i.e.
110 // if it's the only method in the class), make it pure virtual, by adding
111 // = 0 to it.
112 virtual void OnSomeEvent(uint32 someArg1, std::string& someArg2) { }
113
114 // This is a pure virtual function:
115 virtual void OnAnotherEvent(uint32 someArg) = 0;
116 }
117
118 Next, you need to add a specialization for ScriptRegistry. Put this in the bottom of
119 ScriptMgr.cpp:
120
121 template class ScriptRegistry<MyScriptType>;
122
123 Now, add a cleanup routine in ScriptMgr::~ScriptMgr:
124
125 SCR_CLEAR(MyScriptType);
126
127 Now your script type is good to go with the script system. What you need to do now
128 is add functions to ScriptMgr that can be called from the core to actually trigger
129 certain events. For example, in ScriptMgr.h:
130
131 void OnSomeEvent(uint32 someArg1, std::string& someArg2);
132 void OnAnotherEvent(uint32 someArg);
133
134 In ScriptMgr.cpp:
135
136 void ScriptMgr::OnSomeEvent(uint32 someArg1, std::string& someArg2)
137 {
138 FOREACH_SCRIPT(MyScriptType)->OnSomeEvent(someArg1, someArg2);
139 }
140
141 void ScriptMgr::OnAnotherEvent(uint32 someArg)
142 {
143 FOREACH_SCRIPT(MyScriptType)->OnAnotherEvent(someArg1, someArg2);
144 }
145
146 Now you simply call these two functions from anywhere in the core to trigger the
147 event on all registered scripts of that type.
148*/

204class ServerScript : public ScriptObject

213 virtual void OnNetworkStart() { }
214

216 virtual void OnNetworkStop() { }
217
218 // Called when a remote socket establishes a connection to the server. Do not store the socket object.
219 virtual void OnSocketOpen(std::shared_ptr<WorldSocket> /*socket*/) { }
220
221 // Called when a socket is closed. Do not store the socket object, and do not rely on the connection
222 // being open; it is not.
223 virtual void OnSocketClose(std::shared_ptr<WorldSocket> /*socket*/) { }
224
225 // Called when a packet is sent to a client. The packet object is a copy of the original packet, so reading
226 // and modifying it is safe.
227 virtual void OnPacketSend(WorldSession* /*session*/, WorldPacket& /*packet*/) { }
228
229 // Called when a (valid) packet is received by a client. The packet object is a copy of the original packet, so
230 // reading and modifying it is safe. Make sure to check WorldSession pointer before usage, it might be null in case of auth packets
231 virtual void OnPacketReceive(WorldSession* /*session*/, WorldPacket& /*packet*/) { }
232
233 // Called when an invalid (unknown opcode) packet is received by a client. The packet is a reference to the orignal
234 // packet; not a copy. This allows you to actually handle unknown packets (for whatever purpose).
235 virtual void OnUnknownPacketReceive(WorldSession* /*session*/, WorldPacket& /*packet*/) { }
236};

25// Packet logging structures in PKT 3.1 format
26struct LogHeader
27{
28 char Signature[3];
29 uint16 FormatVersion;
30 uint8 SnifferId;
31 uint32 Build;
32 char Locale[4];
33 uint8 SessionKey[40];
34 uint32 SniffStartUnixtime;
35 uint32 SniffStartTicks;
36 uint32 OptionalDataSize;
37};
38
39struct PacketHeader
40{
41 // used to uniquely identify a connection
42 struct OptionalData
43 {
44 uint8 SocketIPBytes[16];
45 uint32 SocketPort;
46 };
47
48 char Direction[4];
49 uint32 ConnectionId;
50 uint32 ArrivalTicks;
51 uint32 OptionalDataSize;
52 uint32 Length;
53 OptionalData OptionalData;
54 uint32 Opcode;
55};

59PacketLog::PacketLog() : _file(NULL)
60{
61 std::call_once(_initializeFlag, &PacketLog::Initialize, this);
62}
63
64PacketLog::~PacketLog()
65{
66 if (_file)
67 fclose(_file);
68
69 _file = NULL;
70}

72void PacketLog::Initialize()
73{
74 std::string logsDir = sConfigMgr->GetStringDefault("LogsDir", "");
75
76 if (!logsDir.empty())
77 if ((logsDir.at(logsDir.length() - 1) != '/') && (logsDir.at(logsDir.length() - 1) != '\\'))
78 logsDir.push_back('/');
79
80 std::string logname = sConfigMgr->GetStringDefault("PacketLogFile", "");
81 if (!logname.empty())
82 {
83 _file = fopen((logsDir + logname).c_str(), "wb");
84
85 LogHeader header;
86 header.Signature[0] = 'P'; header.Signature[1] = 'K'; header.Signature[2] = 'T';
87 header.FormatVersion = 0x0301;
88 header.SnifferId = 'T';
89 header.Build = 12340;
90 header.Locale[0] = 'e'; header.Locale[1] = 'n'; header.Locale[2] = 'U'; header.Locale[3] = 'S';
91 std::memset(header.SessionKey, 0, sizeof(header.SessionKey));
92 header.SniffStartUnixtime = time(NULL);
93 header.SniffStartTicks = getMSTime();
94 header.OptionalDataSize = 0;
95
96 if (CanLogPacket())
97 fwrite(&header, sizeof(header), 1, _file);
98 }
99}

101void PacketLog::LogPacket(WorldPacket const& packet, Direction direction, boost::asio::ip::address const& addr, uint16 port)
102{
103 std::lock_guard<std::mutex> lock(_logPacketLock);
104
105 PacketHeader header;
106 *reinterpret_cast<uint32*>(header.Direction) = direction == CLIENT_TO_SERVER ? 0x47534d43 : 0x47534d53;
107 header.ConnectionId = 0;
108 header.ArrivalTicks = getMSTime();
109
110 header.OptionalDataSize = sizeof(header.OptionalData);
111 memset(header.OptionalData.SocketIPBytes, 0, sizeof(header.OptionalData.SocketIPBytes));
112 if (addr.is_v4())
113 {
114 auto bytes = addr.to_v4().to_bytes();
115 memcpy(header.OptionalData.SocketIPBytes, bytes.data(), bytes.size());
116 }
117 else if (addr.is_v6())
118 {
119 auto bytes = addr.to_v6().to_bytes();
120 memcpy(header.OptionalData.SocketIPBytes, bytes.data(), bytes.size());
121 }
122
123 header.OptionalData.SocketPort = port;
124 header.Length = packet.size() + sizeof(header.Opcode);
125 header.Opcode = packet.GetOpcode();
126
127 fwrite(&header, sizeof(header), 1, _file);
128 if (!packet.empty())
129 fwrite(packet.contents(), 1, packet.size(), _file);
130
131 fflush(_file);
132}

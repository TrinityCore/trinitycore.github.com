145//class to deal with packet processing
146//allows to determine if next packet is safe to be processed
147class PacketFilter
148{
149public:
150 explicit PacketFilter(WorldSession* pSession) : m_pSession(pSession) { }
151 virtual ~PacketFilter() { }
152
153 virtual bool Process(WorldPacket* /*packet*/) { return true; }
154 virtual bool ProcessLogout() const { return true; }
155
156protected:
157 WorldSession* const m_pSession;
158
159private:
160 PacketFilter(PacketFilter const& right) = delete;
161 PacketFilter& operator=(PacketFilter const& right) = delete;
162};
163//process only thread-safe packets in Map::Update()
164class MapSessionFilter : public PacketFilter
165{
166public:
167 explicit MapSessionFilter(WorldSession* pSession) : PacketFilter(pSession) { }
168 ~MapSessionFilter() { }
169
170 virtual bool Process(WorldPacket* packet) override;
171 //in Map::Update() we do not process player logout!
172 virtual bool ProcessLogout() const override { return false; }
173};

179public:
180 explicit WorldSessionFilter(WorldSession* pSession) : PacketFilter(pSession) { }
181 ~WorldSessionFilter() { }
182
183 virtual bool Process(WorldPacket* packet) override;
184};

27static void OnSocketAccept(tcp::socket&& sock, uint32 threadIndex)
28{
29 sWorldSocketMgr.OnSocketOpen(std::forward<tcp::socket>(sock), threadIndex);
30}

32class WorldSocketThread : public NetworkThread<WorldSocket>
33{
34public:
35 void SocketAdded(std::shared_ptr<WorldSocket> sock) override
36 {
37 sScriptMgr->OnSocketOpen(sock);
38 }
39
40 void SocketRemoved(std::shared_ptr<WorldSocket> sock) override
41 {
42 sScriptMgr->OnSocketClose(sock);

50bool WorldSocketMgr::StartNetwork(boost::asio::io_service& service, std::string const& bindIp, uint16 port)
51{
52 _tcpNoDelay = sConfigMgr->GetBoolDefault("Network.TcpNodelay", true);
53
54 int const max_connections = boost::asio::socket_base::max_connections;
55 TC_LOG_DEBUG("misc", "Max allowed socket connections %d", max_connections);
56
57 // -1 means use default
58 _socketSendBufferSize = sConfigMgr->GetIntDefault("Network.OutKBuff", -1);
59
60 m_SockOutUBuff = sConfigMgr->GetIntDefault("Network.OutUBuff", 65536);
61
62 if (m_SockOutUBuff <= 0)
63 {
64 TC_LOG_ERROR("misc", "Network.OutUBuff is wrong in your config file");
65 return false;
66 }
67
68 BaseSocketMgr::StartNetwork(service, bindIp, port);
69
70 _acceptor->SetSocketFactory(std::bind(&BaseSocketMgr::GetSocketForAccept, this));
71
72 _acceptor->AsyncAcceptWithCallback<&OnSocketAccept>();
73
74 sScriptMgr->OnNetworkStart();
75 return true;
76}

85void WorldSocketMgr::OnSocketOpen(tcp::socket&& sock, uint32 threadIndex)
86{
87 // set some options here
88 if (_socketSendBufferSize >= 0)
89 {
90 boost::system::error_code err;
91 sock.set_option(boost::asio::socket_base::send_buffer_size(_socketSendBufferSize), err);
92 if (err && err != boost::system::errc::not_supported)
93 {
94 TC_LOG_ERROR("misc", "WorldSocketMgr::OnSocketOpen sock.set_option(boost::asio::socket_base::send_buffer_size) err = %s", err.message().c_str());
95 return;
96 }
97 }
98
99 // Set TCP_NODELAY.
100 if (_tcpNoDelay)
101 {
102 boost::system::error_code err;
103 sock.set_option(boost::asio::ip::tcp::no_delay(true), err);
104 if (err)
105 {
106 TC_LOG_ERROR("misc", "WorldSocketMgr::OnSocketOpen sock.set_option(boost::asio::ip::tcp::no_delay) err = %s", err.message().c_str());
107 return;
108 }
109 }
110
111 //sock->m_OutBufferSize = static_cast<size_t> (m_SockOutUBuff);
112
113 BaseSocketMgr::OnSocketOpen(std::forward<tcp::socket>(sock), threadIndex);
114}

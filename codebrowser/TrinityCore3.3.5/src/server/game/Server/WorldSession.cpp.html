55bool MapSessionFilter::Process(WorldPacket* packet)
56{
57 OpcodeHandler const& opHandle = opcodeTable[packet->GetOpcode()];
58
59 //let's check if our opcode can be really processed in Map::Update()
60 if (opHandle.packetProcessing == PROCESS_INPLACE)
61 return true;
62
63 //we do not process thread-unsafe packets
64 if (opHandle.packetProcessing == PROCESS_THREADUNSAFE)
65 return false;
66
67 Player* player = m_pSession->GetPlayer();
68 if (!player)
69 return false;
70
71 //in Map::Update() we do not process packets where player is not in world!
72 return player->IsInWorld();
73}

75//we should process ALL packets when player is not in world/logged in
76//OR packet handler is not thread-safe!
77bool WorldSessionFilter::Process(WorldPacket* packet)
78{
79 OpcodeHandler const& opHandle = opcodeTable[packet->GetOpcode()];
80 //check if packet handler is supposed to be safe
81 if (opHandle.packetProcessing == PROCESS_INPLACE)
82 return true;
83
84 //thread-unsafe packets should be processed in World::UpdateSessions()
85 if (opHandle.packetProcessing == PROCESS_THREADUNSAFE)
86 return true;
87
88 //no player attached? -> our client! ^^
89 Player* player = m_pSession->GetPlayer();
90 if (!player)
91 return true;
92
93 //lets process all packets for non-in-the-world player
94 return (player->IsInWorld() == false);
95}

189/// Send a packet to the client
190void WorldSession::SendPacket(WorldPacket* packet)
191{
192 if (!m_Socket)
193 return;
194
195#ifdef TRINITY_DEBUG
196 // Code for network use statistic
197 static uint64 sendPacketCount = 0;
198 static uint64 sendPacketBytes = 0;
199
200 static time_t firstTime = time(NULL);
201 static time_t lastTime = firstTime; // next 60 secs start time
202
203 static uint64 sendLastPacketCount = 0;
204 static uint64 sendLastPacketBytes = 0;
205
206 time_t cur_time = time(NULL);
207
208 if ((cur_time - lastTime) < 60)
209 {
210 sendPacketCount+=1;
211 sendPacketBytes+=packet->size();
212
213 sendLastPacketCount+=1;
214 sendLastPacketBytes+=packet->size();
215 }
216 else
217 {
218 uint64 minTime = uint64(cur_time - lastTime);
219 uint64 fullTime = uint64(lastTime - firstTime);
220 TC_LOG_DEBUG("misc", "Send all time packets count: " UI64FMTD " bytes: " UI64FMTD " avr.count/sec: %f avr.bytes/sec: %f time: %u", sendPacketCount, sendPacketBytes, float(sendPacketCount)/fullTime, float(sendPacketBytes)/fullTime, uint32(fullTime));
221 TC_LOG_DEBUG("misc", "Send last min packets count: " UI64FMTD " bytes: " UI64FMTD " avr.count/sec: %f avr.bytes/sec: %f", sendLastPacketCount, sendLastPacketBytes, float(sendLastPacketCount)/minTime, float(sendLastPacketBytes)/minTime);
222
223 lastTime = cur_time;
224 sendLastPacketCount = 1;
225 sendLastPacketBytes = packet->wpos(); // wpos is real written size
226 }
227#endif // !TRINITY_DEBUG
228
229 sScriptMgr->OnPacketSend(this, *packet);
230
231 TC_LOG_TRACE("network.opcode", "S->C: %s %s", GetPlayerInfo().c_str(), GetOpcodeNameForLogging(packet->GetOpcode()).c_str());
232 m_Socket->SendPacket(*packet);
233}

259/// Update the WorldSession (triggered by World update)
260bool WorldSession::Update(uint32 diff, PacketFilter& updater)
261{
262 /// Update Timeout timer.
263 UpdateTimeOutTime(diff);
264
265 ///- Before we process anything:
266 /// If necessary, kick the player from the character select screen
267 if (IsConnectionIdle())
268 m_Socket->CloseSocket();
269
270 ///- Retrieve packets from the receive queue and call the appropriate handlers
271 /// not process packets if socket already closed
272 WorldPacket* packet = NULL;
273 //! Delete packet after processing by default
274 bool deletePacket = true;
275 //! To prevent infinite loop
276 WorldPacket* firstDelayedPacket = NULL;
277 //! If _recvQueue.peek() == firstDelayedPacket it means that in this Update call, we've processed all
278 //! *properly timed* packets, and we're now at the part of the queue where we find
279 //! delayed packets that were re-enqueued due to improper timing. To prevent an infinite
280 //! loop caused by re-enqueueing the same packets over and over again, we stop updating this session
281 //! and continue updating others. The re-enqueued packets will be handled in the next Update call for this session.
282 uint32 processedPackets = 0;
283 time_t currentTime = time(NULL);
284
285 while (m_Socket && !_recvQueue.empty() && _recvQueue.peek(true) != firstDelayedPacket && _recvQueue.next(packet, updater))
286 {
287 if (packet->GetOpcode() >= NUM_MSG_TYPES)
288 {
289 TC_LOG_ERROR("network.opcode", "Received non-existed opcode %s from %s", GetOpcodeNameForLogging(packet->GetOpcode()).c_str()
290 , GetPlayerInfo().c_str());
291 sScriptMgr->OnUnknownPacketReceive(this, *packet);
292 }
293 else
294 {
295 OpcodeHandler& opHandle = opcodeTable[packet->GetOpcode()];
296 try
297 {
298 switch (opHandle.status)
299 {
300 case STATUS_LOGGEDIN:
301 if (!_player)
302 {
303 // skip STATUS_LOGGEDIN opcode unexpected errors if player logout sometime ago - this can be network lag delayed packets
304 //! If player didn't log out a while ago, it means packets are being sent while the server does not recognize
305 //! the client to be in world yet. We will re-add the packets to the bottom of the queue and process them later.
306 if (!m_playerRecentlyLogout)
307 {
308 //! Prevent infinite loop
309 if (!firstDelayedPacket)
310 firstDelayedPacket = packet;
311 //! Because checking a bool is faster than reallocating memory
312 deletePacket = false;
313 QueuePacket(packet);
314 //! Log
315 TC_LOG_DEBUG("network", "Re-enqueueing packet with opcode %s with with status STATUS_LOGGEDIN. "
316 "Player is currently not in world yet.", GetOpcodeNameForLogging(packet->GetOpcode()).c_str());
317 }
318 }
319 else if (_player->IsInWorld() && AntiDOS.EvaluateOpcode(*packet, currentTime))
320 {
321 sScriptMgr->OnPacketReceive(this, *packet);
322 (this->*opHandle.handler)(*packet);
323 LogUnprocessedTail(packet);
324 }
325 // lag can cause STATUS_LOGGEDIN opcodes to arrive after the player started a transfer
326 break;
327 case STATUS_LOGGEDIN_OR_RECENTLY_LOGGOUT:
328 if (!_player && !m_playerRecentlyLogout && !m_playerLogout) // There's a short delay between _player = null and m_playerRecentlyLogout = true during logout
329 LogUnexpectedOpcode(packet, "STATUS_LOGGEDIN_OR_RECENTLY_LOGGOUT",
330 "the player has not logged in yet and not recently logout");
331 else if (AntiDOS.EvaluateOpcode(*packet, currentTime))
332 {
333 // not expected _player or must checked in packet handler
334 sScriptMgr->OnPacketReceive(this, *packet);
335 (this->*opHandle.handler)(*packet);
336 LogUnprocessedTail(packet);
337 }
338 break;
339 case STATUS_TRANSFER:
340 if (!_player)
341 LogUnexpectedOpcode(packet, "STATUS_TRANSFER", "the player has not logged in yet");
342 else if (_player->IsInWorld())
343 LogUnexpectedOpcode(packet, "STATUS_TRANSFER", "the player is still in world");
344 else if (AntiDOS.EvaluateOpcode(*packet, currentTime))
345 {
346 sScriptMgr->OnPacketReceive(this, *packet);
347 (this->*opHandle.handler)(*packet);
348 LogUnprocessedTail(packet);
349 }
350 break;
351 case STATUS_AUTHED:
352 // prevent cheating with skip queue wait
353 if (m_inQueue)
354 {
355 LogUnexpectedOpcode(packet, "STATUS_AUTHED", "the player not pass queue yet");
356 break;
357 }
358
359 // some auth opcodes can be recieved before STATUS_LOGGEDIN_OR_RECENTLY_LOGGOUT opcodes
360 // however when we recieve CMSG_CHAR_ENUM we are surely no longer during the logout process.
361 if (packet->GetOpcode() == CMSG_CHAR_ENUM)
362 m_playerRecentlyLogout = false;
363
364 if (AntiDOS.EvaluateOpcode(*packet, currentTime))
365 {
366 sScriptMgr->OnPacketReceive(this, *packet);
367 (this->*opHandle.handler)(*packet);
368 LogUnprocessedTail(packet);
369 }
370 break;
371 case STATUS_NEVER:
372 TC_LOG_ERROR("network.opcode", "Received not allowed opcode %s from %s", GetOpcodeNameForLogging(packet->GetOpcode()).c_str()
373 , GetPlayerInfo().c_str());
374 break;
375 case STATUS_UNHANDLED:
376 TC_LOG_DEBUG("network.opcode", "Received not handled opcode %s from %s", GetOpcodeNameForLogging(packet->GetOpcode()).c_str()
377 , GetPlayerInfo().c_str());
378 break;
379 }
380 }
381 catch (ByteBufferException const&)
382 {
383 TC_LOG_ERROR("misc", "WorldSession::Update ByteBufferException occured while parsing a packet (opcode: %u) from client %s, accountid=%i. Skipped packet.",
384 packet->GetOpcode(), GetRemoteAddress().c_str(), GetAccountId());
385 packet->hexlike();
386 }
387 }
388
389 if (deletePacket)
390 delete packet;
391
392 deletePacket = true;
393
394#define MAX_PROCESSED_PACKETS_IN_SAME_WORLDSESSION_UPDATE 100
395 processedPackets++;
396
397 //process only a max amout of packets in 1 Update() call.
398 //Any leftover will be processed in next update
399 if (processedPackets > MAX_PROCESSED_PACKETS_IN_SAME_WORLDSESSION_UPDATE)
400 break;
401 }
402
403 if (m_Socket && m_Socket->IsOpen() && _warden)
404 _warden->Update();
405
406 ProcessQueryCallbacks();
407
408 //check if we are safe to proceed with logout
409 //logout procedure should happen only in World::UpdateSessions() method!!!
410 if (updater.ProcessLogout())
411 {
412 time_t currTime = time(NULL);
413 ///- If necessary, log the player out
414 if (ShouldLogOut(currTime) && !m_playerLoading)
415 LogoutPlayer(true);
416
417 if (m_Socket && GetPlayer() && _warden)
418 _warden->Update();
419
420 ///- Cleanup socket pointer if need
421 if (m_Socket && !m_Socket->IsOpen())
422 {
423 expireTime -= expireTime > diff ? diff : expireTime;
424 if (expireTime < diff || forceExit || !GetPlayer())
425 {
426 m_Socket = nullptr;
427 }
428 }
429
430 if (!m_Socket)
431 return false; //Will remove this session from the world session map
432 }
433
434 return true;
435}

437/// %Log the player out
438void WorldSession::LogoutPlayer(bool save)
439{
440 // finish pending transfers before starting the logout
441 while (_player && _player->IsBeingTeleportedFar())
442 HandleMoveWorldportAckOpcode();
443
444 m_playerLogout = true;
445 m_playerSave = save;
446
447 if (_player)
448 {
449 if (ObjectGuid lguid = _player->GetLootGUID())
450 DoLootRelease(lguid);
451
452 ///- If the player just died before logging out, make him appear as a ghost
453 if (_player->GetDeathTimer())
454 {
455 _player->getHostileRefManager().deleteReferences();
456 _player->BuildPlayerRepop();
457 _player->RepopAtGraveyard();
458 }
459 else if (_player->HasAuraType(SPELL_AURA_SPIRIT_OF_REDEMPTION))
460 {
461 // this will kill character by SPELL_AURA_SPIRIT_OF_REDEMPTION
462 _player->RemoveAurasByType(SPELL_AURA_MOD_SHAPESHIFT);
463 _player->KillPlayer();
464 _player->BuildPlayerRepop();
465 _player->RepopAtGraveyard();
466 }
467 else if (_player->HasPendingBind())
468 {
469 _player->RepopAtGraveyard();
470 _player->SetPendingBind(0, 0);
471 }
472
473 //drop a flag if player is carrying it
474 if (Battleground* bg = _player->GetBattleground())
475 bg->EventPlayerLoggedOut(_player);
476
477 ///- Teleport to home if the player is in an invalid instance
478 if (!_player->m_InstanceValid && !_player->IsGameMaster())
479 _player->TeleportTo(_player->m_homebindMapId, _player->m_homebindX, _player->m_homebindY, _player->m_homebindZ, _player->GetOrientation());
480
481 sOutdoorPvPMgr->HandlePlayerLeaveZone(_player, _player->GetZoneId());
482
483 for (int i=0; i < PLAYER_MAX_BATTLEGROUND_QUEUES; ++i)
484 {
485 if (BattlegroundQueueTypeId bgQueueTypeId = _player->GetBattlegroundQueueTypeId(i))
486 {
487 // track if player logs out after invited to join BG
488 if (_player->IsInvitedForBattlegroundQueueType(bgQueueTypeId) && sWorld->getBoolConfig(CONFIG_BATTLEGROUND_TRACK_DESERTERS))
489 {
490 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_DESERTER_TRACK);
491 stmt->setUInt32(0, _player->GetGUID().GetCounter());
492 stmt->setUInt8(1, BG_DESERTION_TYPE_INVITE_LOGOUT);
493 CharacterDatabase.Execute(stmt);
494 }
495
496 _player->RemoveBattlegroundQueueId(bgQueueTypeId);
497 BattlegroundQueue& queue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
498 queue.RemovePlayer(_player->GetGUID(), true);
499 }
500 }
501
502 // Repop at GraveYard or other player far teleport will prevent saving player because of not present map
503 // Teleport player immediately for correct player save
504 while (_player->IsBeingTeleportedFar())
505 HandleMoveWorldportAckOpcode();
506
507 ///- If the player is in a guild, update the guild roster and broadcast a logout message to other guild members
508 if (Guild* guild = sGuildMgr->GetGuildById(_player->GetGuildId()))
509 guild->HandleMemberLogout(this);
510
511 ///- Remove pet
512 _player->RemovePet(NULL, PET_SAVE_AS_CURRENT, true);
513
514 ///- Clear whisper whitelist
515 _player->ClearWhisperWhiteList();
516
517 ///- empty buyback items and save the player in the database
518 // some save parts only correctly work in case player present in map/player_lists (pets, etc)
519 if (save)
520 {
521 uint32 eslot;
522 for (int j = BUYBACK_SLOT_START; j < BUYBACK_SLOT_END; ++j)
523 {
524 eslot = j - BUYBACK_SLOT_START;
525 _player->SetGuidValue(PLAYER_FIELD_VENDORBUYBACK_SLOT_1 + (eslot * 2), ObjectGuid::Empty);
526 _player->SetUInt32Value(PLAYER_FIELD_BUYBACK_PRICE_1 + eslot, 0);
527 _player->SetUInt32Value(PLAYER_FIELD_BUYBACK_TIMESTAMP_1 + eslot, 0);
528 }
529 _player->SaveToDB();
530 }
531
532 ///- Leave all channels before player delete...
533 _player->CleanupChannels();
534
535 ///- If the player is in a group (or invited), remove him. If the group if then only 1 person, disband the group.
536 _player->UninviteFromGroup();
537
538 // remove player from the group if he is:
539 // a) in group; b) not in raid group; c) logging out normally (not being kicked or disconnected)
540 if (_player->GetGroup() && !_player->GetGroup()->isRaidGroup() && m_Socket)
541 _player->RemoveFromGroup();
542
543 //! Send update to group and reset stored max enchanting level
544 if (_player->GetGroup())
545 {
546 _player->GetGroup()->SendUpdate();
547 _player->GetGroup()->ResetMaxEnchantingLevel();
548 }
549
550 //! Broadcast a logout message to the player's friends
551 sSocialMgr->SendFriendStatus(_player, FRIEND_OFFLINE, _player->GetGUID().GetCounter(), true);
552 sSocialMgr->RemovePlayerSocial(_player->GetGUID().GetCounter());
553
554 //! Call script hook before deletion
555 sScriptMgr->OnPlayerLogout(_player);
556
557 //! Remove the player from the world
558 // the player may not be in the world when logging out
559 // e.g if he got disconnected during a transfer to another map
560 // calls to GetMap in this case may cause crashes
561 _player->CleanupsBeforeDelete();
562 TC_LOG_INFO("entities.player.character", "Account: %d (IP: %s) Logout Character:[%s] (GUID: %u) Level: %d",
563 GetAccountId(), GetRemoteAddress().c_str(), _player->GetName().c_str(), _player->GetGUID().GetCounter(), _player->getLevel());
564 if (Map* _map = _player->FindMap())
565 _map->RemovePlayerFromMap(_player, true);
566
567 SetPlayer(NULL); //! Pointer already deleted during RemovePlayerFromMap
568
569 //! Send the 'logout complete' packet to the client
570 //! Client will respond by sending 3x CMSG_CANCEL_TRADE, which we currently dont handle
571 WorldPacket data(SMSG_LOGOUT_COMPLETE, 0);
572 SendPacket(&data);
573 TC_LOG_DEBUG("network", "SESSION: Sent SMSG_LOGOUT_COMPLETE Message");
574
575 //! Since each account can only have one online character at any given time, ensure all characters for active account are marked as offline
576 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ACCOUNT_ONLINE);
577 stmt->setUInt32(0, GetAccountId());
578 CharacterDatabase.Execute(stmt);
579 }
580
581 m_playerLogout = false;
582 m_playerSave = false;
583 m_playerRecentlyLogout = true;
584 LogoutRequest(0);
585}

681 for (uint32 i = 0; i < NUM_ACCOUNT_DATA_TYPES; ++i)
682 if (mask & (1 << i))
683 m_accountData[i] = AccountData();
684
685 if (!result)
686 return;
687
688 do
689 {
690 Field* fields = result->Fetch();
691 uint32 type = fields[0].GetUInt8();
692 if (type >= NUM_ACCOUNT_DATA_TYPES)
693 {
694 TC_LOG_ERROR("misc", "Table `%s` have invalid account data type (%u), ignore.",
695 mask == GLOBAL_CACHE_MASK ? "account_data" : "character_account_data", type);
696 continue;
697 }
698
699 if ((mask & (1 << type)) == 0)
700 {
701 TC_LOG_ERROR("misc", "Table `%s` have non appropriate for table account data type (%u), ignore.",
702 mask == GLOBAL_CACHE_MASK ? "account_data" : "character_account_data", type);
703 continue;
704 }
705
706 m_accountData[type].Time = time_t(fields[1].GetUInt32());
707 m_accountData[type].Data = fields[2].GetString();
708 }
709 while (result->NextRow());
710}

792{
793 data >> mi->flags;
794 data >> mi->flags2;
795 data >> mi->time;
796 data >> mi->pos.PositionXYZOStream();
797
798 if (mi->HasMovementFlag(MOVEMENTFLAG_ONTRANSPORT))
799 {
800 data >> mi->transport.guid.ReadAsPacked();
801 data >> mi->transport.pos.PositionXYZOStream();
802 data >> mi->transport.time;
803 data >> mi->transport.seat;
804
805 if (mi->HasExtraMovementFlag(MOVEMENTFLAG2_INTERPOLATED_MOVEMENT))
806 data >> mi->transport.time2;
807 }
808
809 if (mi->HasMovementFlag(MovementFlags(MOVEMENTFLAG_SWIMMING | MOVEMENTFLAG_FLYING)) || (mi->HasExtraMovementFlag(MOVEMENTFLAG2_ALWAYS_ALLOW_PITCHING)))
810 data >> mi->pitch;
811
812 data >> mi->fallTime;
813
814 if (mi->HasMovementFlag(MOVEMENTFLAG_FALLING))
815 {
816 data >> mi->jump.zspeed;
817 data >> mi->jump.sinAngle;
818 data >> mi->jump.cosAngle;
819 data >> mi->jump.xyspeed;
820 }
821
822 if (mi->HasMovementFlag(MOVEMENTFLAG_SPLINE_ELEVATION))
823 data >> mi->splineElevation;
824
825 //! Anti-cheat checks. Please keep them in seperate if () blocks to maintain a clear overview.
826 //! Might be subject to latency, so just remove improper flags.
827 #ifdef TRINITY_DEBUG
828 #define REMOVE_VIOLATING_FLAGS(check, maskToRemove) \
829 { \
830 if (check) \
831 { \
832 TC_LOG_DEBUG("entities.unit", "WorldSession::ReadMovementInfo: Violation of MovementFlags found (%s). " \
833 "MovementFlags: %u, MovementFlags2: %u for player GUID: %u. Mask %u will be removed.", \
834 STRINGIZE(check), mi->GetMovementFlags(), mi->GetExtraMovementFlags(), GetPlayer()->GetGUID().GetCounter(), maskToRemove); \
835 mi->RemoveMovementFlag((maskToRemove)); \
836 } \
837 }
838 #else
839 #define REMOVE_VIOLATING_FLAGS(check, maskToRemove) \
840 if (check) \
841 mi->RemoveMovementFlag((maskToRemove));
842 #endif
843
844 /*! This must be a packet spoofing attempt. MOVEMENTFLAG_ROOT sent from the client is not valid
845 in conjunction with any of the moving movement flags such as MOVEMENTFLAG_FORWARD.
846 It will freeze clients that receive this player's movement info.
847 */
848 // Only adjust movement flag removal for vehicles with the VEHICLE_FLAG_FIXED_POSITION flag, or the hard coded exceptions below:
849 // 30236 | Argent Cannon
850 // 39759 | Tankbuster Cannon
851 if (GetPlayer()->GetVehicleBase() && ((GetPlayer()->GetVehicle()->GetVehicleInfo()->m_flags & VEHICLE_FLAG_FIXED_POSITION) || GetPlayer()->GetVehicleBase()->GetEntry() == 30236 || GetPlayer()->GetVehicleBase()->GetEntry() == 39759))
852 {
853 // Actually players in rooted vehicles still send commands, don't clear root for these!
854 // Check specifically for the following conditions:
855 // MOVEMENTFLAG_ROOT + no other flags (0x800)
856 // MOVEMENTFLAG_ROOT + MOVEMENTFLAG_LEFT (0x810)
857 // MOVEMENTFLAG_ROOT + MOVEMENTFLAG_RIGHT (0x820)
858 // MOVEMENTFLAG_ROOT + MOVEMENTFLAG_PITCH_UP (0x840)
859 // MOVEMENTFLAG_ROOT + MOVEMENTFLAG_PITCH_DOWN (0x880)
860 // If none of these are true, clear the root
861 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_ROOT) && mi->HasMovementFlag(MOVEMENTFLAG_LEFT | MOVEMENTFLAG_RIGHT | MOVEMENTFLAG_PITCH_UP | MOVEMENTFLAG_PITCH_DOWN),
862 MOVEMENTFLAG_MASK_MOVING);
863 }
864 else
865 {
866 // Only remove here for non vehicles
867 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_ROOT),
868 MOVEMENTFLAG_ROOT);
869 }
870
871 //! Cannot hover without SPELL_AURA_HOVER
872 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_HOVER) && !GetPlayer()->HasAuraType(SPELL_AURA_HOVER),
873 MOVEMENTFLAG_HOVER);
874
875 //! Cannot ascend and descend at the same time
876 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_ASCENDING) && mi->HasMovementFlag(MOVEMENTFLAG_DESCENDING),
877 MOVEMENTFLAG_ASCENDING | MOVEMENTFLAG_DESCENDING);
878
879 //! Cannot move left and right at the same time
880 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_LEFT) && mi->HasMovementFlag(MOVEMENTFLAG_RIGHT),
881 MOVEMENTFLAG_LEFT | MOVEMENTFLAG_RIGHT);
882
883 //! Cannot strafe left and right at the same time
884 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_STRAFE_LEFT) && mi->HasMovementFlag(MOVEMENTFLAG_STRAFE_RIGHT),
885 MOVEMENTFLAG_STRAFE_LEFT | MOVEMENTFLAG_STRAFE_RIGHT);
886
887 //! Cannot pitch up and down at the same time
888 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_PITCH_UP) && mi->HasMovementFlag(MOVEMENTFLAG_PITCH_DOWN),
889 MOVEMENTFLAG_PITCH_UP | MOVEMENTFLAG_PITCH_DOWN);
890
891 //! Cannot move forwards and backwards at the same time
892 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_FORWARD) && mi->HasMovementFlag(MOVEMENTFLAG_BACKWARD),
893 MOVEMENTFLAG_FORWARD | MOVEMENTFLAG_BACKWARD);
894
895 //! Cannot walk on water without SPELL_AURA_WATER_WALK
896 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_WATERWALKING) && !GetPlayer()->HasAuraType(SPELL_AURA_WATER_WALK),
897 MOVEMENTFLAG_WATERWALKING);
898
899 //! Cannot feather fall without SPELL_AURA_FEATHER_FALL
900 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_FALLING_SLOW) && !GetPlayer()->HasAuraType(SPELL_AURA_FEATHER_FALL),
901 MOVEMENTFLAG_FALLING_SLOW);
902
903 /*! Cannot fly if no fly auras present. Exception is being a GM.
904 Note that we check for account level instead of Player::IsGameMaster() because in some
905 situations it may be feasable to use .gm fly on as a GM without having .gm on,
906 e.g. aerial combat.
907 */
908
909 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_FLYING | MOVEMENTFLAG_CAN_FLY) && GetSecurity() == SEC_PLAYER &&
910 !GetPlayer()->m_mover->HasAuraType(SPELL_AURA_FLY) &&
911 !GetPlayer()->m_mover->HasAuraType(SPELL_AURA_MOD_INCREASE_MOUNTED_FLIGHT_SPEED),
912 MOVEMENTFLAG_FLYING | MOVEMENTFLAG_CAN_FLY);
913
914 //! Cannot fly and fall at the same time
915 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_CAN_FLY | MOVEMENTFLAG_DISABLE_GRAVITY) && mi->HasMovementFlag(MOVEMENTFLAG_FALLING),
916 MOVEMENTFLAG_FALLING);
917
918 REMOVE_VIOLATING_FLAGS(mi->HasMovementFlag(MOVEMENTFLAG_SPLINE_ENABLED) &&
919 (!GetPlayer()->movespline->Initialized() || GetPlayer()->movespline->Finalized()), MOVEMENTFLAG_SPLINE_ENABLED);
920
921 #undef REMOVE_VIOLATING_FLAGS
922}

960void WorldSession::ReadAddonsInfo(ByteBuffer &data)
961{
962 if (data.rpos() + 4 > data.size())
963 return;
964
965 uint32 size;
966 data >> size;
967
968 if (!size)
969 return;
970
971 if (size > 0xFFFFF)
972 {
973 TC_LOG_DEBUG("addon", "WorldSession::ReadAddonsInfo: AddOnInfo too big, size %u", size);
974 return;
975 }
976
977 uLongf uSize = size;
978
979 uint32 pos = data.rpos();
980
981 ByteBuffer addonInfo;
982 addonInfo.resize(size);
983
984 if (uncompress(addonInfo.contents(), &uSize, data.contents() + pos, data.size() - pos) == Z_OK)
985 {
986 uint32 addonsCount;
987 addonInfo >> addonsCount; // addons count
988
989 for (uint32 i = 0; i < addonsCount; ++i)
990 {
991 std::string addonName;
992 uint8 enabled;
993 uint32 crc, unk1;
994
995 // check next addon data format correctness
996 if (addonInfo.rpos() + 1 > addonInfo.size())
997 return;
998
999 addonInfo >> addonName;
1000
1001 addonInfo >> enabled >> crc >> unk1;
1002
1003 TC_LOG_DEBUG("addon", "AddOn: %s (CRC: 0x%x) - enabled: 0x%x - Unknown2: 0x%x", addonName.c_str(), crc, enabled, unk1);
1004
1005 AddonInfo addon(addonName, enabled, crc, 2, true);
1006
1007 SavedAddon const* savedAddon = AddonMgr::GetAddonInfo(addonName);
1008 if (savedAddon)
1009 {
1010 if (addon.CRC != savedAddon->CRC)
1011 TC_LOG_WARN("addon", " Addon: %s: modified (CRC: 0x%x) - accountID %d)", addon.Name.c_str(), savedAddon->CRC, GetAccountId());
1012 else
1013 TC_LOG_DEBUG("addon", "Addon: %s: validated (CRC: 0x%x) - accountID %d", addon.Name.c_str(), savedAddon->CRC, GetAccountId());
1014 }
1015 else
1016 {
1017 AddonMgr::SaveAddon(addon);
1018 TC_LOG_WARN("addon", "Addon: %s: unknown (CRC: 0x%x) - accountId %d (storing addon name and checksum to database)", addon.Name.c_str(), addon.CRC, GetAccountId());
1019 }
1020
1021 /// @todo Find out when to not use CRC/pubkey, and other possible states.
1022 m_addonsList.push_back(addon);
1023 }
1024
1025 uint32 currentTime;
1026 addonInfo >> currentTime;
1027 TC_LOG_DEBUG("addon", "AddOn: CurrentTime: %u", currentTime);
1028 }
1029 else
1030 TC_LOG_DEBUG("addon", "AddOn: Addon packet uncompress error!");
1031}

1326bool WorldSession::DosProtection::EvaluateOpcode(WorldPacket& p, time_t time) const
1327{
1328 uint32 maxPacketCounterAllowed = GetMaxPacketCounterAllowed(p.GetOpcode());
1329
1330 // Return true if there no limit for the opcode
1331 if (!maxPacketCounterAllowed)
1332 return true;
1333
1334 PacketCounter& packetCounter = _PacketThrottlingMap[p.GetOpcode()];
1335 if (packetCounter.lastReceiveTime != time)
1336 {
1337 packetCounter.lastReceiveTime = time;
1338 packetCounter.amountCounter = 0;
1339 }
1340
1341 // Check if player is flooding some packets
1342 if (++packetCounter.amountCounter <= maxPacketCounterAllowed)
1343 return true;
1344
1345 TC_LOG_WARN("network", "AntiDOS: Account %u, IP: %s, Ping: %u, Character: %s, flooding packet (opc: %s (0x%X), count: %u)",
1346 Session->GetAccountId(), Session->GetRemoteAddress().c_str(), Session->GetLatency(), Session->GetPlayerName().c_str(),
1347 opcodeTable[p.GetOpcode()].name, p.GetOpcode(), packetCounter.amountCounter);
1348
1349 switch (_policy)
1350 {
1351 case POLICY_LOG:
1352 return true;
1353 case POLICY_KICK:
1354 {
1355 TC_LOG_WARN("network", "AntiDOS: Player kicked!");
1356 Session->KickPlayer();
1357 return false;
1358 }
1359 case POLICY_BAN:
1360 {
1361 BanMode bm = (BanMode)sWorld->getIntConfig(CONFIG_PACKET_SPOOF_BANMODE);
1362 uint32 duration = sWorld->getIntConfig(CONFIG_PACKET_SPOOF_BANDURATION); // in seconds
1363 std::string nameOrIp = "";
1364 switch (bm)
1365 {
1366 case BAN_CHARACTER: // not supported, ban account
1367 case BAN_ACCOUNT: (void)sAccountMgr->GetName(Session->GetAccountId(), nameOrIp); break;
1368 case BAN_IP: nameOrIp = Session->GetRemoteAddress(); break;
1369 }
1370 sWorld->BanAccount(bm, nameOrIp, duration, "DOS (Packet Flooding/Spoofing", "Server: AutoDOS");
1371 TC_LOG_WARN("network", "AntiDOS: Player automatically banned for %u seconds.", duration);
1372 Session->KickPlayer();
1373 return false;
1374 }
1375 default: // invalid policy
1376 return true;
1377 }

123 struct MessageDistDeliverer
124 {
125 WorldObject* i_source;
126 WorldPacket* i_message;
127 uint32 i_phaseMask;
128 float i_distSq;
129 uint32 team;
130 Player const* skipped_receiver;
131 MessageDistDeliverer(WorldObject* src, WorldPacket* msg, float dist, bool own_team_only = false, Player const* skipped = NULL)
132 : i_source(src), i_message(msg), i_phaseMask(src->GetPhaseMask()), i_distSq(dist * dist)
133 , team(0)
134 , skipped_receiver(skipped)
135 {
136 if (own_team_only)
137 if (Player* player = src->ToPlayer())
138 team = player->GetTeam();
139 }
140
141 void Visit(PlayerMapType &m);
142 void Visit(CreatureMapType &m);
143 void Visit(DynamicObjectMapType &m);
144 template<class SKIP> void Visit(GridRefManager<SKIP> &) { }
145
146 void SendPacket(Player* player)
147 {
148 // never send packet to self
149 if (player == i_source || (team && player->GetTeam() != team) || skipped_receiver == player)
150 return;
151
152 if (!player->HaveAtClient(i_source))
153 return;
154
155 if (WorldSession* session = player->GetSession())
156 session->SendPacket(i_message);
157 }
158 };

644 return false;

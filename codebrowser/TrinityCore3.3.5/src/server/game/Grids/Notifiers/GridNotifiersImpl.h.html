31template<class T>
32inline void Trinity::VisibleNotifier::Visit(GridRefManager<T> &m)
33{
34 for (typename GridRefManager<T>::iterator iter = m.begin(); iter != m.end(); ++iter)
35 {
36 vis_guids.erase(iter->GetSource()->GetGUID());
37 i_player.UpdateVisibilityOf(iter->GetSource(), i_data, i_visibleNow);
38 }
39}

45template<class Check>
46void Trinity::WorldObjectSearcher<Check>::Visit(GameObjectMapType &m)
47{
48 if (!(i_mapTypeMask & GRID_MAP_TYPE_MASK_GAMEOBJECT))
49 return;
50
51 // already found
52 if (i_object)
53 return;
54
55 for (GameObjectMapType::iterator itr=m.begin(); itr != m.end(); ++itr)
56 {
57 if (!itr->GetSource()->InSamePhase(i_phaseMask))
58 continue;
59
60 if (i_check(itr->GetSource()))
61 {
62 i_object = itr->GetSource();
63 return;
64 }
65 }
66}

512template<class Builder>
513void Trinity::LocalizedPacketDo<Builder>::operator()(Player* p)
514{
515 LocaleConstant loc_idx = p->GetSession()->GetSessionDbLocaleIndex();
516 uint32 cache_idx = loc_idx+1;
517 WorldPacket* data;
518
519 // create if not cached yet
520 if (i_data_cache.size() < cache_idx + 1 || !i_data_cache[cache_idx])
521 {
522 if (i_data_cache.size() < cache_idx + 1)
523 i_data_cache.resize(cache_idx + 1);
524
525 data = new WorldPacket();
526
527 i_builder(*data, loc_idx);
528
529 ASSERT(data->GetOpcode() != MSG_NULL_ACTION);
530
531 i_data_cache[cache_idx] = data;
532 }
533 else
534 data = i_data_cache[cache_idx];
535
536 p->SendDirectMessage(data);
537}
538
539template<class Builder>
540void Trinity::LocalizedPacketListDo<Builder>::operator()(Player* p)
541{
542 LocaleConstant loc_idx = p->GetSession()->GetSessionDbLocaleIndex();
543 uint32 cache_idx = loc_idx+1;
544 WorldPacketList* data_list;
545
546 // create if not cached yet
547 if (i_data_cache.size() < cache_idx+1 || i_data_cache[cache_idx].empty())
548 {
549 if (i_data_cache.size() < cache_idx+1)
550 i_data_cache.resize(cache_idx+1);
551
552 data_list = &i_data_cache[cache_idx];
553
554 i_builder(*data_list, loc_idx);
555 }
556 else
557 data_list = &i_data_cache[cache_idx];
558
559 for (size_t i = 0; i < data_list->size(); ++i)
560 p->SendDirectMessage((*data_list)[i]);
561}

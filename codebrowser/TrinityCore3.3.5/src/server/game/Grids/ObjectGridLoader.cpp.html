30void ObjectGridEvacuator::Visit(CreatureMapType &m)
31{
32 // creature in unloading grid can have respawn point in another grid
33 // if it will be unloaded then it will not respawn in original grid until unload/load original grid
34 // move to respawn point to prevent this case. For player view in respawn grid this will be normal respawn.
35 for (CreatureMapType::iterator iter = m.begin(); iter != m.end();)
36 {
37 Creature* c = iter->GetSource();
38 ++iter;
39
40 ASSERT(!c->IsPet() && "ObjectGridRespawnMover must not be called for pets");
41 c->GetMap()->CreatureRespawnRelocation(c, true);
42 }
43}

45void ObjectGridEvacuator::Visit(GameObjectMapType &m)
46{
47 // gameobject in unloading grid can have respawn point in another grid
48 // if it will be unloaded then it will not respawn in original grid until unload/load original grid
49 // move to respawn point to prevent this case. For player view in respawn grid this will be normal respawn.
50 for (GameObjectMapType::iterator iter = m.begin(); iter != m.end();)
51 {
52 GameObject* go = iter->GetSource();
53 ++iter;
54
55 go->GetMap()->GameObjectRespawnRelocation(go, true);
56 }

94template <class T>
95void AddObjectHelper(CellCoord &cell, GridRefManager<T> &m, uint32 &count, Map* /*map*/, T *obj)
96{
97 obj->AddToGrid(m);
98 ObjectGridLoader::SetObjectCell(obj, cell);
99 obj->AddToWorld();
100 ++count;
101}
102
103template <>
104void AddObjectHelper(CellCoord &cell, CreatureMapType &m, uint32 &count, Map* map, Creature *obj)
105{
106 obj->AddToGrid(m);
107 ObjectGridLoader::SetObjectCell(obj, cell);
108 obj->AddToWorld();
109 if (obj->isActiveObject())
110 map->AddToActive(obj);
111
112 ++count;
113}

115template <class T>
116void LoadHelper(CellGuidSet const& guid_set, CellCoord &cell, GridRefManager<T> &m, uint32 &count, Map* map)
117{
118 for (CellGuidSet::const_iterator i_guid = guid_set.begin(); i_guid != guid_set.end(); ++i_guid)
119 {
120 T* obj = new T;
121 ObjectGuid::LowType guid = *i_guid;
122 //TC_LOG_INFO("misc", "DEBUG: LoadHelper from table: %s for (guid: %u) Loading", table, guid);
123 if (!obj->LoadFromDB(guid, map))
124 {
125 delete obj;
126 continue;
127 }
128
129 AddObjectHelper(cell, m, count, map, obj);
130 }
131}

147void ObjectWorldLoader::Visit(CorpseMapType& /*m*/)
148{
149 CellCoord cellCoord = i_cell.GetCellCoord();
150 if (std::unordered_set<Corpse*> const* corpses = i_map->GetCorpsesInCell(cellCoord.GetId()))
151 {
152 for (Corpse* corpse : *corpses)
153 {
154 corpse->AddToWorld();
155 GridType& cell = i_grid.GetGridType(i_cell.CellX(), i_cell.CellY());
156 if (corpse->IsWorldObject())
157 cell.AddWorldObject(corpse);
158 else
159 cell.AddGridObject(corpse);
160
161 ++i_corpses;
162 }
163 }
164}

166void ObjectGridLoader::LoadN(void)
167{
168 i_gameObjects = 0; i_creatures = 0; i_corpses = 0;
169 i_cell.data.Part.cell_y = 0;
170 for (uint32 x = 0; x < MAX_NUMBER_OF_CELLS; ++x)
171 {
172 i_cell.data.Part.cell_x = x;
173 for (uint32 y = 0; y < MAX_NUMBER_OF_CELLS; ++y)
174 {
175 i_cell.data.Part.cell_y = y;
176
177 //Load creatures and game objects
178 {
179 TypeContainerVisitor<ObjectGridLoader, GridTypeMapContainer> visitor(*this);
180 i_grid.VisitGrid(x, y, visitor);
181 }
182
183 //Load corpses (not bones)
184 {
185 ObjectWorldLoader worker(*this);
186 TypeContainerVisitor<ObjectWorldLoader, WorldTypeMapContainer> visitor(worker);
187 i_grid.VisitGrid(x, y, visitor);
188 }
189 }
190 }
191 TC_LOG_DEBUG("maps", "%u GameObjects, %u Creatures, and %u Corpses/Bones loaded for grid %u on map %u", i_gameObjects, i_creatures, i_corpses, i_grid.GetGridId(), i_map->GetId());
192}

194template<class T>
195void ObjectGridUnloader::Visit(GridRefManager<T> &m)
196{
197 while (!m.isEmpty())
198 {
199 T *obj = m.getFirst()->GetSource();
200 // if option set then object already saved at this moment
201 if (!sWorld->getBoolConfig(CONFIG_SAVE_RESPAWN_TIME_IMMEDIATELY))
202 obj->SaveRespawnTime();
203 //Some creatures may summon other temp summons in CleanupsBeforeDelete()
204 //So we need this even after cleaner (maybe we can remove cleaner)
205 //Example: Flame Leviathan Turret 33139 is summoned when a creature is deleted
206 /// @todo Check if that script has the correct logic. Do we really need to summons something before deleting?
207 obj->CleanupsBeforeDelete();
208 ///- object will get delinked from the manager when deleted
209 delete obj;
210 }
211}

213void ObjectGridStoper::Visit(CreatureMapType &m)
214{
215 // stop any fights at grid de-activation and remove dynobjects created at cast by creatures
216 for (CreatureMapType::iterator iter = m.begin(); iter != m.end(); ++iter)
217 {
218 iter->GetSource()->RemoveAllDynObjects();
219 if (iter->GetSource()->IsInCombat())
220 {
221 iter->GetSource()->CombatStop();
222 iter->GetSource()->DeleteThreatList();
223 iter->GetSource()->AI()->EnterEvadeMode();
224 }
225 }
226}

232 iter->GetSource()->CleanupsBeforeDelete();

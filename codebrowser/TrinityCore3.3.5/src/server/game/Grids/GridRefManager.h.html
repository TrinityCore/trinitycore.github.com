19#ifndef _GRIDREFMANAGER
20#define _GRIDREFMANAGER
21
22#include "RefManager.h"
23
24template<class OBJECT>
25class GridReference;
26
27template<class OBJECT>
28class GridRefManager : public RefManager<GridRefManager<OBJECT>, OBJECT>
29{
30 public:
31 typedef LinkedListHead::Iterator< GridReference<OBJECT> > iterator;
32
33 GridReference<OBJECT>* getFirst() { return (GridReference<OBJECT>*)RefManager<GridRefManager<OBJECT>, OBJECT>::getFirst(); }
34 GridReference<OBJECT>* getLast() { return (GridReference<OBJECT>*)RefManager<GridRefManager<OBJECT>, OBJECT>::getLast(); }
35
36 iterator begin() { return iterator(getFirst()); }
37 iterator end() { return iterator(NULL); }
38 iterator rbegin() { return iterator(getLast()); }
39 iterator rend() { return iterator(NULL); }
40};
41#endif

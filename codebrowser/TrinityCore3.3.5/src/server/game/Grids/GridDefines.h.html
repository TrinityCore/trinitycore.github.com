87template<uint32 LIMIT>
88struct CoordPair
89{
90 CoordPair(uint32 x=0, uint32 y=0)
91 : x_coord(x), y_coord(y)
92 { }
93
94 CoordPair(const CoordPair<LIMIT> &obj)
95 : x_coord(obj.x_coord), y_coord(obj.y_coord)
96 { }
97
98 CoordPair<LIMIT> & operator=(const CoordPair<LIMIT> &obj)
99 {
100 x_coord = obj.x_coord;
101 y_coord = obj.y_coord;
102 return *this;
103 }
104
105 void dec_x(uint32 val)
106 {
107 if (x_coord > val)
108 x_coord -= val;
109 else
110 x_coord = 0;
111 }
112
113 void inc_x(uint32 val)
114 {
115 if (x_coord + val < LIMIT)
116 x_coord += val;
117 else
118 x_coord = LIMIT - 1;
119 }
120
121 void dec_y(uint32 val)
122 {
123 if (y_coord > val)
124 y_coord -= val;
125 else
126 y_coord = 0;
127 }
128
129 void inc_y(uint32 val)
130 {
131 if (y_coord + val < LIMIT)
132 y_coord += val;
133 else
134 y_coord = LIMIT - 1;
135 }
136
137 bool IsCoordValid() const
138 {
139 return x_coord < LIMIT && y_coord < LIMIT;
140 }
141
142 CoordPair& normalize()
143 {
144 x_coord = std::min(x_coord, LIMIT - 1);
145 y_coord = std::min(y_coord, LIMIT - 1);
146 return *this;
147 }
148
149 uint32 GetId() const
150 {
151 return y_coord * LIMIT + x_coord;
152 }
153
154 uint32 x_coord;
155 uint32 y_coord;
156};
157
158template<uint32 LIMIT>
159bool operator==(const CoordPair<LIMIT> &p1, const CoordPair<LIMIT> &p2)
160{
161 return (p1.x_coord == p2.x_coord && p1.y_coord == p2.y_coord);
162}
163
164template<uint32 LIMIT>
165bool operator!=(const CoordPair<LIMIT> &p1, const CoordPair<LIMIT> &p2)
166{
167 return !(p1 == p2);
168}

175 template<class RET_TYPE, int CENTER_VAL>
176 inline RET_TYPE Compute(float x, float y, float center_offset, float size)
177 {
178 // calculate and store temporary values in double format for having same result as same mySQL calculations
179 double x_offset = (double(x) - center_offset)/size;
180 double y_offset = (double(y) - center_offset)/size;
181
182 int x_val = int(x_offset + CENTER_VAL + 0.5f);
183 int y_val = int(y_offset + CENTER_VAL + 0.5f);
184 return RET_TYPE(x_val, y_val);
185 }
186
187 inline GridCoord ComputeGridCoord(float x, float y)
188 {
189 return Compute<GridCoord, CENTER_GRID_ID>(x, y, CENTER_GRID_OFFSET, SIZE_OF_GRIDS);
190 }
191
192 inline CellCoord ComputeCellCoord(float x, float y)
193 {
194 return Compute<CellCoord, CENTER_GRID_CELL_ID>(x, y, CENTER_GRID_CELL_OFFSET, SIZE_OF_GRID_CELL);
195 }
196
197 inline CellCoord ComputeCellCoord(float x, float y, float &x_off, float &y_off)
198 {
199 double x_offset = (double(x) - CENTER_GRID_CELL_OFFSET)/SIZE_OF_GRID_CELL;
200 double y_offset = (double(y) - CENTER_GRID_CELL_OFFSET)/SIZE_OF_GRID_CELL;
201
202 int x_val = int(x_offset + CENTER_GRID_CELL_ID + 0.5f);
203 int y_val = int(y_offset + CENTER_GRID_CELL_ID + 0.5f);
204 x_off = (float(x_offset) - float(x_val) + CENTER_GRID_CELL_ID) * SIZE_OF_GRID_CELL;
205 y_off = (float(y_offset) - float(y_val) + CENTER_GRID_CELL_ID) * SIZE_OF_GRID_CELL;
206 return CellCoord(x_val, y_val);
207 }

209 inline void NormalizeMapCoord(float &c)
210 {
211 if (c > MAP_HALFSIZE - 0.5f)
212 c = MAP_HALFSIZE - 0.5f;
213 else if (c < -(MAP_HALFSIZE - 0.5f))
214 c = -(MAP_HALFSIZE - 0.5f);
215 }
216
217 inline bool IsValidMapCoord(float c)
218 {
219 return std::isfinite(c) && (std::fabs(c) <= MAP_HALFSIZE - 0.5f);
220 }
221
222 inline bool IsValidMapCoord(float x, float y)
223 {
224 return IsValidMapCoord(x) && IsValidMapCoord(y);
225 }
226
227 inline bool IsValidMapCoord(float x, float y, float z)
228 {
229 return IsValidMapCoord(x, y) && IsValidMapCoord(z);
230 }
231
232 inline bool IsValidMapCoord(float x, float y, float z, float o)
233 {
234 return IsValidMapCoord(x, y, z) && std::isfinite(o);
235 }

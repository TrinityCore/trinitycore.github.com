30void SmartWaypointMgr::LoadFromDB()
31{
32 uint32 oldMSTime = getMSTime();
33
34 for (std::unordered_map<uint32, WPPath*>::iterator itr = waypoint_map.begin(); itr != waypoint_map.end(); ++itr)
35 {
36 for (WPPath::iterator pathItr = itr->second->begin(); pathItr != itr->second->end(); ++pathItr)
37 delete pathItr->second;
38
39 delete itr->second;
40 }
41
42 waypoint_map.clear();
43
44 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_SMARTAI_WP);
45 PreparedQueryResult result = WorldDatabase.Query(stmt);
46
47 if (!result)
48 {
49 TC_LOG_INFO("server.loading", ">> Loaded 0 SmartAI Waypoint Paths. DB table `waypoints` is empty.");
50
51 return;
52 }
53
54 uint32 count = 0;
55 uint32 total = 0;
56 uint32 last_entry = 0;
57 uint32 last_id = 1;
58
59 do
60 {
61 Field* fields = result->Fetch();
62 uint32 entry = fields[0].GetUInt32();
63 uint32 id = fields[1].GetUInt32();
64 float x, y, z;
65 x = fields[2].GetFloat();
66 y = fields[3].GetFloat();
67 z = fields[4].GetFloat();
68
69 if (last_entry != entry)
70 {
71 waypoint_map[entry] = new WPPath();
72 last_id = 1;
73 count++;
74 }
75
76 if (last_id != id)
77 TC_LOG_ERROR("sql.sql", "SmartWaypointMgr::LoadFromDB: Path entry %u, unexpected point id %u, expected %u.", entry, id, last_id);
78
79 last_id++;
80 (*waypoint_map[entry])[id] = new WayPoint(id, x, y, z);
81
82 last_entry = entry;
83 total++;
84 }
85 while (result->NextRow());
86
87 TC_LOG_INFO("server.loading", ">> Loaded %u SmartAI waypoint paths (total %u waypoints) in %u ms", count, total, GetMSTimeDiffToNow(oldMSTime));
88}

101void SmartAIMgr::LoadSmartAIFromDB()
102{
103 LoadHelperStores();
104
105 uint32 oldMSTime = getMSTime();
106
107 for (uint8 i = 0; i < SMART_SCRIPT_TYPE_MAX; i++)
108 mEventMap[i].clear(); //Drop Existing SmartAI List
109
110 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_SMART_SCRIPTS);
111 PreparedQueryResult result = WorldDatabase.Query(stmt);
112
113 if (!result)
114 {
115 TC_LOG_INFO("server.loading", ">> Loaded 0 SmartAI scripts. DB table `smartai_scripts` is empty.");
116 return;
117 }
118
119 uint32 count = 0;
120
121 do
122 {
123 Field* fields = result->Fetch();
124
125 SmartScriptHolder temp;
126
127 temp.entryOrGuid = fields[0].GetInt32();
128 SmartScriptType source_type = (SmartScriptType)fields[1].GetUInt8();
129 if (source_type >= SMART_SCRIPT_TYPE_MAX)
130 {
131 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: invalid source_type (%u), skipped loading.", uint32(source_type));
132 continue;
133 }
134 if (temp.entryOrGuid >= 0)
135 {
136 switch (source_type)
137 {
138 case SMART_SCRIPT_TYPE_CREATURE:
139 {
140 CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate((uint32)temp.entryOrGuid);
141 if (!creatureInfo)
142 {
143 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) does not exist, skipped loading.", uint32(temp.entryOrGuid));
144 continue;
145 }
146
147 if (creatureInfo->AIName != "SmartAI")
148 {
149 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) is not using SmartAI, skipped loading.", uint32(temp.entryOrGuid));
150 continue;
151 }
152 break;
153 }
154 case SMART_SCRIPT_TYPE_GAMEOBJECT:
155 {
156 GameObjectTemplate const* gameObjectInfo = sObjectMgr->GetGameObjectTemplate((uint32)temp.entryOrGuid);
157 if (!gameObjectInfo)
158 {
159 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: GameObject entry (%u) does not exist, skipped loading.", uint32(temp.entryOrGuid));
160 continue;
161 }
162
163 if (gameObjectInfo->AIName != "SmartGameObjectAI")
164 {
165 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: GameObject entry (%u) is not using SmartGameObjectAI, skipped loading.", uint32(temp.entryOrGuid));
166 continue;
167 }
168 break;
169 }
170 case SMART_SCRIPT_TYPE_AREATRIGGER:
171 {
172 if (!sAreaTriggerStore.LookupEntry((uint32)temp.entryOrGuid))
173 {
174 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: AreaTrigger entry (%u) does not exist, skipped loading.", uint32(temp.entryOrGuid));
175 continue;
176 }
177 break;
178 }
179 case SMART_SCRIPT_TYPE_TIMED_ACTIONLIST:
180 break;//nothing to check, really
181 default:
182 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: not yet implemented source_type %u", (uint32)source_type);
183 continue;
184 }
185 }
186 else
187 {
188 CreatureData const* creature = sObjectMgr->GetCreatureData(uint32(std::abs(temp.entryOrGuid)));
189 if (!creature)
190 {
191 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature guid (%u) does not exist, skipped loading.", uint32(std::abs(temp.entryOrGuid)));
192 continue;
193 }
194
195 CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate(creature->id);
196 if (!creatureInfo)
197 {
198 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) guid (%u) does not exist, skipped loading.", creature->id, uint32(std::abs(temp.entryOrGuid)));
199 continue;
200 }
201
202 if (creatureInfo->AIName != "SmartAI")
203 {
204 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry (%u) guid (%u) is not using SmartAI, skipped loading.", creature->id, uint32(std::abs(temp.entryOrGuid)));
205 continue;
206 }
207 }
208
209 temp.source_type = source_type;
210 temp.event_id = fields[2].GetUInt16();
211 temp.link = fields[3].GetUInt16();
212 temp.event.type = (SMART_EVENT)fields[4].GetUInt8();
213 temp.event.event_phase_mask = fields[5].GetUInt8();
214 temp.event.event_chance = fields[6].GetUInt8();
215 temp.event.event_flags = fields[7].GetUInt8();
216
217 temp.event.raw.param1 = fields[8].GetUInt32();
218 temp.event.raw.param2 = fields[9].GetUInt32();
219 temp.event.raw.param3 = fields[10].GetUInt32();
220 temp.event.raw.param4 = fields[11].GetUInt32();
221
222 temp.action.type = (SMART_ACTION)fields[12].GetUInt8();
223 temp.action.raw.param1 = fields[13].GetUInt32();
224 temp.action.raw.param2 = fields[14].GetUInt32();
225 temp.action.raw.param3 = fields[15].GetUInt32();
226 temp.action.raw.param4 = fields[16].GetUInt32();
227 temp.action.raw.param5 = fields[17].GetUInt32();
228 temp.action.raw.param6 = fields[18].GetUInt32();
229
230 temp.target.type = (SMARTAI_TARGETS)fields[19].GetUInt8();
231 temp.target.raw.param1 = fields[20].GetUInt32();
232 temp.target.raw.param2 = fields[21].GetUInt32();
233 temp.target.raw.param3 = fields[22].GetUInt32();
234 temp.target.x = fields[23].GetFloat();
235 temp.target.y = fields[24].GetFloat();
236 temp.target.z = fields[25].GetFloat();
237 temp.target.o = fields[26].GetFloat();
238
239 //check target
240 if (!IsTargetValid(temp))
241 continue;
242
243 // check all event and action params
244 if (!IsEventValid(temp))
245 continue;
246
247 // creature entry / guid not found in storage, create empty event list for it and increase counters
248 if (mEventMap[source_type].find(temp.entryOrGuid) == mEventMap[source_type].end())
249 {
250 ++count;
251 SmartAIEventList eventList;
252 mEventMap[source_type][temp.entryOrGuid] = eventList;
253 }
254 // store the new event
255 mEventMap[source_type][temp.entryOrGuid].push_back(temp);
256 }
257 while (result->NextRow());
258
259 // Post Loading Validation
260 for (uint8 i = 0; i < SMART_SCRIPT_TYPE_MAX; ++i)
261 {
262 for (SmartAIEventMap::iterator itr = mEventMap[i].begin(); itr != mEventMap[i].end(); ++itr)
263 {
264 for (SmartScriptHolder const& e : itr->second)
265 {
266 if (e.link)
267 {
268 if (!FindLinkedEvent(itr->second, e.link))
269 {
270 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Entry %d SourceType %u, Event %u, Link Event %u not found or invalid.",
271 e.entryOrGuid, e.GetScriptType(), e.event_id, e.link);
272 }
273 }
274
275 if (e.GetEventType() == SMART_EVENT_LINK)
276 {
277 if (!FindLinkedSourceEvent(itr->second, e.event_id))
278 {
279 TC_LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Entry %d SourceType %u, Event %u, Link Source Event not found or invalid. Event will never trigger.",
280 e.entryOrGuid, e.GetScriptType(), e.event_id);
281 }
282 }
283 }
284 }
285 }
286
287 TC_LOG_INFO("server.loading", ">> Loaded %u SmartAI scripts in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
288
289 UnLoadHelperStores();
290}

292bool SmartAIMgr::IsTargetValid(SmartScriptHolder const& e)
293{
294 if (std::abs(e.target.o) > 2 * float(M_PI))
295 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u has abs(`target.o` = %f) > 2*PI (orientation is expressed in radians)",
296 e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.o);
297
298 if (e.GetActionType() == SMART_ACTION_INSTALL_AI_TEMPLATE)
299 return true; // AI template has special handling
300
301 switch (e.GetTargetType())
302 {
303 case SMART_TARGET_CREATURE_DISTANCE:
304 case SMART_TARGET_CREATURE_RANGE:
305 {
306 if (e.target.unitDistance.creature && !sObjectMgr->GetCreatureTemplate(e.target.unitDistance.creature))
307 {
308 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses non-existent Creature entry %u as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.unitDistance.creature);
309 return false;
310 }
311 break;
312 }
313 case SMART_TARGET_GAMEOBJECT_DISTANCE:
314 case SMART_TARGET_GAMEOBJECT_RANGE:
315 {
316 if (e.target.goDistance.entry && !sObjectMgr->GetGameObjectTemplate(e.target.goDistance.entry))
317 {
318 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses non-existent GameObject entry %u as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.goDistance.entry);
319 return false;
320 }
321 break;
322 }
323 case SMART_TARGET_CREATURE_GUID:
324 {
325 if (e.target.unitGUID.entry && !IsCreatureValid(e, e.target.unitGUID.entry))
326 return false;
327 break;
328 }
329 case SMART_TARGET_GAMEOBJECT_GUID:
330 {
331 if (e.target.goGUID.entry && !IsGameObjectValid(e, e.target.goGUID.entry))
332 return false;
333 break;
334 }
335 case SMART_TARGET_PLAYER_DISTANCE:
336 case SMART_TARGET_CLOSEST_PLAYER:
337 {
338 if (e.target.playerDistance.dist == 0)
339 {
340 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u has maxDist 0 as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType());
341 return false;
342 }
343 break;
344 }
345 case SMART_TARGET_PLAYER_RANGE:
346 case SMART_TARGET_SELF:
347 case SMART_TARGET_VICTIM:
348 case SMART_TARGET_HOSTILE_SECOND_AGGRO:
349 case SMART_TARGET_HOSTILE_LAST_AGGRO:
350 case SMART_TARGET_HOSTILE_RANDOM:
351 case SMART_TARGET_HOSTILE_RANDOM_NOT_TOP:
352 case SMART_TARGET_ACTION_INVOKER:
353 case SMART_TARGET_INVOKER_PARTY:
354 case SMART_TARGET_POSITION:
355 case SMART_TARGET_NONE:
356 case SMART_TARGET_ACTION_INVOKER_VEHICLE:
357 case SMART_TARGET_OWNER_OR_SUMMONER:
358 case SMART_TARGET_THREAT_LIST:
359 case SMART_TARGET_CLOSEST_GAMEOBJECT:
360 case SMART_TARGET_CLOSEST_CREATURE:
361 case SMART_TARGET_CLOSEST_ENEMY:
362 case SMART_TARGET_CLOSEST_FRIENDLY:
363 case SMART_TARGET_STORED:
364 break;
365 default:
366 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Not handled target_type(%u), Entry %d SourceType %u Event %u Action %u, skipped.", e.GetTargetType(), e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType());
367 return false;
368 }
369 return true;
370}

372bool SmartAIMgr::IsEventValid(SmartScriptHolder& e)
373{
374 if (e.event.type >= SMART_EVENT_END)
375 {
376 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid %d using event(%u) has invalid event type (%u), skipped.", e.entryOrGuid, e.event_id, e.GetEventType());
377 return false;
378 }
379
380 // in SMART_SCRIPT_TYPE_TIMED_ACTIONLIST all event types are overriden by core
381 if (e.GetScriptType() != SMART_SCRIPT_TYPE_TIMED_ACTIONLIST && !(SmartAIEventMask[e.event.type][1] & SmartAITypeMask[e.GetScriptType()][1]))
382 {
383 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid %d, event type %u can not be used for Script type %u", e.entryOrGuid, e.GetEventType(), e.GetScriptType());
384 return false;
385 }
386
387 if (e.action.type <= 0 || e.action.type >= SMART_ACTION_END)
388 {
389 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid %d using event(%u) has invalid action type (%u), skipped.", e.entryOrGuid, e.event_id, e.GetActionType());
390 return false;
391 }
392
393 if (e.event.event_phase_mask > SMART_EVENT_PHASE_ALL)
394 {
395 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid %d using event(%u) has invalid phase mask (%u), skipped.", e.entryOrGuid, e.event_id, e.event.event_phase_mask);
396 return false;
397 }
398
399 if (e.event.event_flags > SMART_EVENT_FLAGS_ALL)
400 {
401 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid %d using event(%u) has invalid event flags (%u), skipped.", e.entryOrGuid, e.event_id, e.event.event_flags);
402 return false;
403 }
404
405 if (e.link && e.link == e.event_id)
406 {
407 TC_LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid %d SourceType %u, Event %u, Event is linking self (infinite loop), skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id);
408 return false;
409 }
410
411 if (e.GetScriptType() == SMART_SCRIPT_TYPE_TIMED_ACTIONLIST)
412 {
413 e.event.type = SMART_EVENT_UPDATE_OOC;//force default OOC, can change when calling the script!
414 if (!IsMinMaxValid(e, e.event.minMaxRepeat.min, e.event.minMaxRepeat.max))
415 return false;
416
417 if (!IsMinMaxValid(e, e.event.minMaxRepeat.repeatMin, e.event.minMaxRepeat.repeatMax))
418 return false;
419 }
420 else
421 {
422 switch (e.GetEventType())
423 {
424 case SMART_EVENT_UPDATE:
425 case SMART_EVENT_UPDATE_IC:
426 case SMART_EVENT_UPDATE_OOC:
427 case SMART_EVENT_HEALT_PCT:
428 case SMART_EVENT_MANA_PCT:
429 case SMART_EVENT_TARGET_HEALTH_PCT:
430 case SMART_EVENT_TARGET_MANA_PCT:
431 case SMART_EVENT_RANGE:
432 case SMART_EVENT_DAMAGED:
433 case SMART_EVENT_DAMAGED_TARGET:
434 case SMART_EVENT_RECEIVE_HEAL:
435 if (!IsMinMaxValid(e, e.event.minMaxRepeat.min, e.event.minMaxRepeat.max))
436 return false;
437
438 if (!IsMinMaxValid(e, e.event.minMaxRepeat.repeatMin, e.event.minMaxRepeat.repeatMax))
439 return false;
440 break;
441 case SMART_EVENT_SPELLHIT:
442 case SMART_EVENT_SPELLHIT_TARGET:
443 if (e.event.spellHit.spell)
444 {
445 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(e.event.spellHit.spell);
446 if (!spellInfo)
447 {
448 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses non-existent Spell entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.spellHit.spell);
449 return false;
450 }
451 if (e.event.spellHit.school && (e.event.spellHit.school & spellInfo->SchoolMask) != spellInfo->SchoolMask)
452 {
453 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses Spell entry %u with invalid school mask, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.spellHit.spell);
454 return false;
455 }
456 }
457 if (!IsMinMaxValid(e, e.event.spellHit.cooldownMin, e.event.spellHit.cooldownMax))
458 return false;
459 break;
460 case SMART_EVENT_OOC_LOS:
461 case SMART_EVENT_IC_LOS:
462 if (!IsMinMaxValid(e, e.event.los.cooldownMin, e.event.los.cooldownMax))
463 return false;
464 break;
465 case SMART_EVENT_RESPAWN:
466 if (e.event.respawn.type == SMART_SCRIPT_RESPAWN_CONDITION_MAP && !sMapStore.LookupEntry(e.event.respawn.map))
467 {
468 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses non-existent Map entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.respawn.map);
469 return false;
470 }
471 if (e.event.respawn.type == SMART_SCRIPT_RESPAWN_CONDITION_AREA && !sAreaTableStore.LookupEntry(e.event.respawn.area))
472 {
473 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses non-existent Area entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.respawn.area);
474 return false;
475 }
476 break;
477 case SMART_EVENT_FRIENDLY_HEALTH:
478 if (!NotNULL(e, e.event.friendlyHealth.radius))
479 return false;
480
481 if (!IsMinMaxValid(e, e.event.friendlyHealth.repeatMin, e.event.friendlyHealth.repeatMax))
482 return false;
483 break;
484 case SMART_EVENT_FRIENDLY_IS_CC:
485 if (!IsMinMaxValid(e, e.event.friendlyCC.repeatMin, e.event.friendlyCC.repeatMax))
486 return false;
487 break;
488 case SMART_EVENT_FRIENDLY_MISSING_BUFF:
489 {
490 if (!IsSpellValid(e, e.event.missingBuff.spell))
491 return false;
492
493 if (!NotNULL(e, e.event.missingBuff.radius))
494 return false;
495
496 if (!IsMinMaxValid(e, e.event.missingBuff.repeatMin, e.event.missingBuff.repeatMax))
497 return false;
498 break;
499 }
500 case SMART_EVENT_KILL:
501 if (!IsMinMaxValid(e, e.event.kill.cooldownMin, e.event.kill.cooldownMax))
502 return false;
503
504 if (e.event.kill.creature && !IsCreatureValid(e, e.event.kill.creature))
505 return false;
506 break;
507 case SMART_EVENT_VICTIM_CASTING:
508 if (e.event.targetCasting.spellId > 0 && !sSpellMgr->GetSpellInfo(e.event.targetCasting.spellId))
509 {
510 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses non-existent Spell entry %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.spellHit.spell);
511 return false;
512 }
513
514 if (!IsMinMaxValid(e, e.event.targetCasting.repeatMin, e.event.targetCasting.repeatMax))
515 return false;
516 break;
517 case SMART_EVENT_PASSENGER_BOARDED:
518 case SMART_EVENT_PASSENGER_REMOVED:
519 if (!IsMinMaxValid(e, e.event.minMax.repeatMin, e.event.minMax.repeatMax))
520 return false;
521 break;
522 case SMART_EVENT_SUMMON_DESPAWNED:
523 case SMART_EVENT_SUMMONED_UNIT:
524 if (e.event.summoned.creature && !IsCreatureValid(e, e.event.summoned.creature))
525 return false;
526
527 if (!IsMinMaxValid(e, e.event.summoned.cooldownMin, e.event.summoned.cooldownMax))
528 return false;
529 break;
530 case SMART_EVENT_ACCEPTED_QUEST:
531 case SMART_EVENT_REWARD_QUEST:
532 if (e.event.quest.quest && !IsQuestValid(e, e.event.quest.quest))
533 return false;
534 break;
535 case SMART_EVENT_RECEIVE_EMOTE:
536 {
537 if (e.event.emote.emote && !IsTextEmoteValid(e, e.event.emote.emote))
538 return false;
539
540 if (!IsMinMaxValid(e, e.event.emote.cooldownMin, e.event.emote.cooldownMax))
541 return false;
542 break;
543 }
544 case SMART_EVENT_HAS_AURA:
545 case SMART_EVENT_TARGET_BUFFED:
546 {
547 if (!IsSpellValid(e, e.event.aura.spell))
548 return false;
549
550 if (!IsMinMaxValid(e, e.event.aura.repeatMin, e.event.aura.repeatMax))
551 return false;
552 break;
553 }
554 case SMART_EVENT_TRANSPORT_ADDCREATURE:
555 {
556 if (e.event.transportAddCreature.creature && !IsCreatureValid(e, e.event.transportAddCreature.creature))
557 return false;
558 break;
559 }
560 case SMART_EVENT_MOVEMENTINFORM:
561 {
562 if (e.event.movementInform.type > NULL_MOTION_TYPE)
563 {
564 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u uses invalid Motion type %u, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.event.movementInform.type);
565 return false;
566 }
567 break;
568 }
569 case SMART_EVENT_DATA_SET:
570 {
571 if (!IsMinMaxValid(e, e.event.dataSet.cooldownMin, e.event.dataSet.cooldownMax))
572 return false;
573 break;
574 }
575 case SMART_EVENT_AREATRIGGER_ONTRIGGER:
576 {
577 if (e.event.areatrigger.id && !IsAreaTriggerValid(e, e.event.areatrigger.id))
578 return false;
579 break;
580 }
581 case SMART_EVENT_TEXT_OVER:
582 if (!IsTextValid(e, e.event.textOver.textGroupID))

1227bool SmartAIMgr::IsTextValid(SmartScriptHolder const& e, uint32 id)
1228{
1229 if (e.GetScriptType() != SMART_SCRIPT_TYPE_CREATURE)
1230 return true;
1231
1232 uint32 entry = 0;
1233
1234 if (e.GetEventType() == SMART_EVENT_TEXT_OVER)
1235 {
1236 entry = e.event.textOver.creatureEntry;
1237 }
1238 else
1239 {
1240 switch (e.GetTargetType())
1241 {
1242 case SMART_TARGET_CREATURE_DISTANCE:
1243 case SMART_TARGET_CREATURE_RANGE:
1244 case SMART_TARGET_CLOSEST_CREATURE:
1245 return true; // ignore
1246 default:
1247 if (e.entryOrGuid < 0)
1248 {
1249 ObjectGuid::LowType guid = ObjectGuid::LowType(-e.entryOrGuid);
1250 CreatureData const* data = sObjectMgr->GetCreatureData(guid);
1251 if (!data)
1252 {
1253 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u using non-existent Creature guid %d, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), guid);
1254 return false;
1255 }
1256 else
1257 entry = data->id;
1258 }
1259 else
1260 entry = uint32(e.entryOrGuid);
1261 break;
1262 }
1263 }
1264
1265 if (!entry || !sCreatureTextMgr->TextExist(entry, uint8(id)))
1266 {
1267 TC_LOG_ERROR("sql.sql", "SmartAIMgr: Entry %d SourceType %u Event %u Action %u using non-existent Text id %d, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), id);
1268 return false;

1274void SmartAIMgr::LoadHelperStores()
1275{
1276 uint32 oldMSTime = getMSTime();
1277
1278 SpellInfo const* spellInfo = NULL;
1279 for (uint32 i = 0; i < sSpellMgr->GetSpellInfoStoreSize(); ++i)
1280 {
1281 spellInfo = sSpellMgr->GetSpellInfo(i);
1282 if (!spellInfo)
1283 continue;
1284
1285 for (uint32 j = 0; j < MAX_SPELL_EFFECTS; ++j)
1286 {
1287 if (spellInfo->Effects[j].IsEffect(SPELL_EFFECT_SUMMON))
1288 SummonCreatureSpellStore.insert(std::make_pair(uint32(spellInfo->Effects[j].MiscValue), std::make_pair(i, SpellEffIndex(j))));
1289
1290 else if (spellInfo->Effects[j].IsEffect(SPELL_EFFECT_SUMMON_OBJECT_WILD))
1291 SummonGameObjectSpellStore.insert(std::make_pair(uint32(spellInfo->Effects[j].MiscValue), std::make_pair(i, SpellEffIndex(j))));
1292
1293 else if (spellInfo->Effects[j].IsEffect(SPELL_EFFECT_KILL_CREDIT) || spellInfo->Effects[j].IsEffect(SPELL_EFFECT_KILL_CREDIT2))
1294 KillCreditSpellStore.insert(std::make_pair(uint32(spellInfo->Effects[j].MiscValue), std::make_pair(i, SpellEffIndex(j))));
1295
1296 else if (spellInfo->Effects[j].IsEffect(SPELL_EFFECT_CREATE_ITEM))
1297 CreateItemSpellStore.insert(std::make_pair(uint32(spellInfo->Effects[j].ItemType), std::make_pair(i, SpellEffIndex(j))));
1298 }
1299 }
1300
1301 TC_LOG_INFO("server.loading", ">> Loaded SmartAIMgr Helpers in %u ms", GetMSTimeDiffToNow(oldMSTime));
1302}

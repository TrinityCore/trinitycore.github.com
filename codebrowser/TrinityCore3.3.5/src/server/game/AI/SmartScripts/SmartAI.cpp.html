72void SmartAI::UpdateDespawn(const uint32 diff)
73{
74 if (mDespawnState <= 1 || mDespawnState > 3)
75 return;
76
77 if (mDespawnTime < diff)
78 {
79 if (mDespawnState == 2)
80 {
81 me->SetVisible(false);
82 mDespawnTime = 5000;
83 mDespawnState++;
84 }
85 else
86 me->DespawnOrUnsummon();
87 } else mDespawnTime -= diff;
88}

90WayPoint* SmartAI::GetNextWayPoint()
91{
92 if (!mWayPoints || mWayPoints->empty())
93 return NULL;
94
95 mCurrentWPID++;
96 WPPath::const_iterator itr = mWayPoints->find(mCurrentWPID);
97 if (itr != mWayPoints->end())
98 {
99 mLastWP = (*itr).second;
100 if (mLastWP->id != mCurrentWPID)
101 {
102 TC_LOG_ERROR("misc", "SmartAI::GetNextWayPoint: Got not expected waypoint id %u, expected %u", mLastWP->id, mCurrentWPID);
103 }
104 return (*itr).second;
105 }
106 return NULL;
107}

109void SmartAI::StartPath(bool run, uint32 path, bool repeat, Unit* /*invoker*/)
110{
111 if (me->IsInCombat())// no wp movement in combat
112 {
113 TC_LOG_ERROR("misc", "SmartAI::StartPath: Creature entry %u wanted to start waypoint movement while in combat, ignoring.", me->GetEntry());
114 return;
115 }
116 if (HasEscortState(SMART_ESCORT_ESCORTING))
117 StopPath();
118 if (path)
119 if (!LoadPath(path))
120 return;
121 if (!mWayPoints || mWayPoints->empty())
122 return;
123
124 AddEscortState(SMART_ESCORT_ESCORTING);
125 mCanRepeatPath = repeat;
126
127 SetRun(run);
128
129 if (WayPoint* wp = GetNextWayPoint())
130 {
131 mLastOOCPos = me->GetPosition();
132 me->GetMotionMaster()->MovePoint(wp->id, wp->x, wp->y, wp->z);
133 GetScript()->ProcessEventsFor(SMART_EVENT_WAYPOINT_START, NULL, wp->id, GetScript()->GetPathId());
134 }
135}

138{

147 GetScript()->SetPathId(entry);
148 return true;

151void SmartAI::PausePath(uint32 delay, bool forced)
152{
153 if (!HasEscortState(SMART_ESCORT_ESCORTING))
154 return;
155 if (HasEscortState(SMART_ESCORT_PAUSED))
156 {
157 TC_LOG_ERROR("misc", "SmartAI::StartPath: Creature entry %u wanted to pause waypoint movement while already paused, ignoring.", me->GetEntry());
158 return;
159 }
160 mForcedPaused = forced;
161 mLastOOCPos = me->GetPosition();
162 AddEscortState(SMART_ESCORT_PAUSED);
163 mWPPauseTimer = delay;
164 if (forced)
165 {
166 SetRun(mRun);
167 me->StopMoving();//force stop
168 me->GetMotionMaster()->MoveIdle();//force stop
169 }
170 GetScript()->ProcessEventsFor(SMART_EVENT_WAYPOINT_PAUSED, NULL, mLastWP->id, GetScript()->GetPathId());
171}

173void SmartAI::StopPath(uint32 DespawnTime, uint32 quest, bool fail)
174{
175 if (!HasEscortState(SMART_ESCORT_ESCORTING))
176 return;
177
178 if (quest)
179 mEscortQuestID = quest;
180 SetDespawnTime(DespawnTime);
181 //mDespawnTime = DespawnTime;
182
183 mLastOOCPos = me->GetPosition();
184 me->StopMoving();//force stop
185 me->GetMotionMaster()->MoveIdle();
186 GetScript()->ProcessEventsFor(SMART_EVENT_WAYPOINT_STOPPED, NULL, mLastWP->id, GetScript()->GetPathId());
187 EndPath(fail);
188}

190void SmartAI::EndPath(bool fail)
191{
192 GetScript()->ProcessEventsFor(SMART_EVENT_WAYPOINT_ENDED, NULL, mLastWP->id, GetScript()->GetPathId());
193
194 RemoveEscortState(SMART_ESCORT_ESCORTING | SMART_ESCORT_PAUSED | SMART_ESCORT_RETURNING);
195 mWayPoints = NULL;
196 mCurrentWPID = 0;
197 mWPPauseTimer = 0;
198 mLastWP = NULL;
199
200 if (mCanRepeatPath)
201 StartPath(mRun, GetScript()->GetPathId(), true);
202 else
203 GetScript()->SetPathId(0);
204
205 ObjectList* targets = GetScript()->GetTargetList(SMART_ESCORT_TARGETS);
206 if (targets && mEscortQuestID)
207 {
208 if (targets->size() == 1 && GetScript()->IsPlayer((*targets->begin())))
209 {
210 Player* player = (*targets->begin())->ToPlayer();
211 if (!fail && player->IsAtGroupRewardDistance(me) && !player->HasCorpse())
212 player->GroupEventHappens(mEscortQuestID, me);
213
214 if (fail && player->GetQuestStatus(mEscortQuestID) == QUEST_STATUS_INCOMPLETE)
215 player->FailQuest(mEscortQuestID);
216
217 if (Group* group = player->GetGroup())
218 {
219 for (GroupReference* groupRef = group->GetFirstMember(); groupRef != NULL; groupRef = groupRef->next())
220 {
221 Player* groupGuy = groupRef->GetSource();
222
223 if (!fail && groupGuy->IsAtGroupRewardDistance(me) && !groupGuy->HasCorpse())
224 groupGuy->AreaExploredOrEventHappens(mEscortQuestID);
225 if (fail && groupGuy->GetQuestStatus(mEscortQuestID) == QUEST_STATUS_INCOMPLETE)
226 groupGuy->FailQuest(mEscortQuestID);
227 }
228 }
229 }else
230 {
231 for (ObjectList::iterator iter = targets->begin(); iter != targets->end(); ++iter)
232 {
233 if (GetScript()->IsPlayer((*iter)))
234 {
235 Player* player = (*iter)->ToPlayer();
236 if (!fail && player->IsAtGroupRewardDistance(me) && !player->HasCorpse())
237 player->AreaExploredOrEventHappens(mEscortQuestID);
238 if (fail && player->GetQuestStatus(mEscortQuestID) == QUEST_STATUS_INCOMPLETE)
239 player->FailQuest(mEscortQuestID);
240 }
241 }
242 }
243 }
244 if (mDespawnState == 1)
245 StartDespawn();
246}

262void SmartAI::UpdatePath(const uint32 diff)
263{
264 if (!HasEscortState(SMART_ESCORT_ESCORTING))
265 return;
266 if (mEscortInvokerCheckTimer < diff)
267 {
268 if (!IsEscortInvokerInRange())
269 {
270 StopPath(mDespawnTime, mEscortQuestID, true);
271 }
272 mEscortInvokerCheckTimer = 1000;
273 } else mEscortInvokerCheckTimer -= diff;
274 // handle pause
275 if (HasEscortState(SMART_ESCORT_PAUSED))
276 {
277 if (mWPPauseTimer < diff)
278 {
279 if (!me->IsInCombat() && !HasEscortState(SMART_ESCORT_RETURNING) && (mWPReached || mLastWPIDReached == SMART_ESCORT_LAST_OOC_POINT || mForcedPaused))
280 {
281 GetScript()->ProcessEventsFor(SMART_EVENT_WAYPOINT_RESUMED, NULL, mLastWP->id, GetScript()->GetPathId());
282 RemoveEscortState(SMART_ESCORT_PAUSED);
283 if (mForcedPaused)// if paused between 2 wps resend movement
284 {
285 ResumePath();
286 mWPReached = false;
287 mForcedPaused = false;
288 }
289 if (mLastWPIDReached == SMART_ESCORT_LAST_OOC_POINT)
290 mWPReached = true;
291 }
292 mWPPauseTimer = 0;
293 } else {
294 mWPPauseTimer -= diff;
295 }
296 }
297 if (HasEscortState(SMART_ESCORT_RETURNING))
298 {
299 if (mWPReached)//reached OOC WP
300 {
301 RemoveEscortState(SMART_ESCORT_RETURNING);
302 if (!HasEscortState(SMART_ESCORT_PAUSED))
303 ResumePath();
304 mWPReached = false;
305 }
306 }
307 if ((!me->HasReactState(REACT_PASSIVE) && me->IsInCombat()) || HasEscortState(SMART_ESCORT_PAUSED | SMART_ESCORT_RETURNING))
308 return;
309 // handle next wp
310 if (mWPReached)//reached WP
311 {
312 mWPReached = false;
313 if (mCurrentWPID == GetWPCount())
314 {
315 EndPath();
316 }
317 else if (WayPoint* wp = GetNextWayPoint())
318 {
319 SetRun(mRun);
320 me->GetMotionMaster()->MovePoint(wp->id, wp->x, wp->y, wp->z);
321 }
322 }
323}

325void SmartAI::UpdateAI(uint32 diff)
326{
327 GetScript()->OnUpdate(diff);
328 UpdatePath(diff);
329 UpdateDespawn(diff);
330
331 /// @todo move to void
332 if (mFollowGuid)
333 {
334 if (mFollowArrivedTimer < diff)
335 {
336 if (me->FindNearestCreature(mFollowArrivedEntry, INTERACTION_DISTANCE, true))
337 {
338 StopFollow();
339 return;
340 }
341
342 mFollowArrivedTimer = 1000;
343 }
344 else
345 mFollowArrivedTimer -= diff;
346 }
347
348 if (!UpdateVictim())
349 return;
350
351 if (mCanAutoAttack)
352 DoMeleeAttackIfReady();
353}

355bool SmartAI::IsEscortInvokerInRange()
356{
357 ObjectList* targets = GetScript()->GetTargetList(SMART_ESCORT_TARGETS);
358 if (targets)
359 {
360 if (targets->size() == 1 && GetScript()->IsPlayer((*targets->begin())))
361 {
362 Player* player = (*targets->begin())->ToPlayer();
363 if (me->GetDistance(player) <= SMART_ESCORT_MAX_PLAYER_DIST)
364 return true;
365
366 if (Group* group = player->GetGroup())
367 {
368 for (GroupReference* groupRef = group->GetFirstMember(); groupRef != NULL; groupRef = groupRef->next())
369 {
370 Player* groupGuy = groupRef->GetSource();
371
372 if (me->GetDistance(groupGuy) <= SMART_ESCORT_MAX_PLAYER_DIST)
373 return true;
374 }
375 }
376 }else
377 {
378 for (ObjectList::iterator iter = targets->begin(); iter != targets->end(); ++iter)
379 {
380 if (GetScript()->IsPlayer((*iter)))
381 {
382 if (me->GetDistance((*iter)->ToPlayer()) <= SMART_ESCORT_MAX_PLAYER_DIST)
383 return true;
384 }
385 }
386 }
387 }
388 return true;//escort targets were not set, ignore range check
389}

411void SmartAI::EnterEvadeMode(EvadeReason /*why*/)
412{
413 if (!me->IsAlive() || me->IsInEvadeMode())
414 return;
415
416 me->RemoveAllAurasExceptType(SPELL_AURA_CONTROL_VEHICLE, SPELL_AURA_CLONE_CASTER);
417
418 me->AddUnitState(UNIT_STATE_EVADE);
419 me->DeleteThreatList();
420 me->CombatStop(true);
421 me->LoadCreaturesAddon();
422 me->SetLootRecipient(NULL);
423 me->ResetPlayerDamageReq();
424 me->SetLastDamagedTime(0);
425
426 GetScript()->ProcessEventsFor(SMART_EVENT_EVADE);//must be after aura clear so we can cast spells from db
427
428 SetRun(mRun);
429 if (HasEscortState(SMART_ESCORT_ESCORTING))
430 {
431 AddEscortState(SMART_ESCORT_RETURNING);
432 ReturnToLastOOCPos();
433 }
434 else if (mFollowGuid)
435 {
436 if (Unit* target = ObjectAccessor::GetUnit(*me, mFollowGuid))
437 me->GetMotionMaster()->MoveFollow(target, mFollowDist, mFollowAngle);
438
439 // evade is not cleared in MoveFollow, so we can't keep it
440 me->ClearUnitState(UNIT_STATE_EVADE);
441 }
442 else
443 me->GetMotionMaster()->MoveTargetedHome();
444
445 if (!HasEscortState(SMART_ESCORT_ESCORTING))//dont mess up escort movement after combat
446 SetRun(mRun);
447}

466
467bool SmartAI::AssistPlayerInCombat(Unit* who)
468{
469 if (me->HasReactState(REACT_PASSIVE))
470 return false;
471
472 if (!who || !who->GetVictim())
473 return false;
474
475 //experimental (unknown) flag not present
476 if (!(me->GetCreatureTemplate()->type_flags & CREATURE_TYPEFLAGS_AID_PLAYERS))
477 return false;
478
479 //not a player
480 if (!who->EnsureVictim()->GetCharmerOrOwnerPlayerOrPlayerItself())
481 return false;
482
483 //never attack friendly
484 if (me->IsFriendlyTo(who))
485 return false;
486
487 //too far away and no free sight?
488 if (me->IsWithinDistInMap(who, SMART_MAX_AID_DIST) && me->IsWithinLOSInMap(who))
489 {
490 //already fighting someone?
491 if (!me->GetVictim())
492 {
493 AttackStart(who);
494 return true;
495 }
496 else
497 {
498 who->SetInCombatWith(me);
499 me->AddThreat(who, 0.0f);
500 return true;
501 }
502 }
503
504 return false;
505}

507void SmartAI::JustRespawned()
508{
509 mDespawnTime = 0;
510 mDespawnState = 0;
511 mEscortState = SMART_ESCORT_NONE;
512 me->SetVisible(true);
513 if (me->getFaction() != me->GetCreatureTemplate()->faction)
514 me->RestoreFaction();
515 mJustReset = true;
516 JustReachedHome();
517 GetScript()->ProcessEventsFor(SMART_EVENT_RESPAWN);
518 mFollowGuid.Clear();//do not reset follower on Reset(), we need it after combat evade
519 mFollowDist = 0;
520 mFollowAngle = 0;
521 mFollowCredit = 0;
522 mFollowArrivedTimer = 1000;
523 mFollowArrivedEntry = 0;
524 mFollowCreditType = 0;
525}

727void SmartAI::SetCombatMove(bool on)
728{
729 if (mCanCombatMove == on)
730 return;
731 mCanCombatMove = on;
732 if (!HasEscortState(SMART_ESCORT_ESCORTING))
733 {
734 if (on && me->GetVictim())
735 {
736 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == IDLE_MOTION_TYPE)
737 {
738 SetRun(mRun);
739 me->GetMotionMaster()->MoveChase(me->GetVictim());
740 me->CastStop();
741 }
742 }
743 else
744 {
745 if (me->HasUnitState(UNIT_STATE_CONFUSED_MOVE | UNIT_STATE_FLEEING_MOVE))
746 return;
747
748 me->GetMotionMaster()->MovementExpired();
749 me->GetMotionMaster()->Clear(true);
750 me->StopMoving();
751 me->GetMotionMaster()->MoveIdle();
752 }
753 }

756void SmartAI::SetFollow(Unit* target, float dist, float angle, uint32 credit, uint32 end, uint32 creditType)
757{
758 if (!target)
759 {
760 StopFollow();
761 return;
762 }
763
764 SetRun(mRun);
765 mFollowGuid = target->GetGUID();
766 mFollowDist = dist >= 0.0f ? dist : PET_FOLLOW_DIST;
767 mFollowAngle = angle >= 0.0f ? angle : me->GetFollowAngle();
768 mFollowArrivedTimer = 1000;
769 mFollowCredit = credit;
770 mFollowArrivedEntry = end;
771 me->GetMotionMaster()->MoveFollow(target, mFollowDist, mFollowAngle);
772 mFollowCreditType = creditType;
773}

775void SmartAI::StopFollow()
776{
777 if (Player* player = ObjectAccessor::GetPlayer(*me, mFollowGuid))
778 {
779 if (!mFollowCreditType)
780 player->RewardPlayerAndGroupAtEvent(mFollowCredit, me);
781 else
782 player->GroupEventHappens(mFollowCredit, me);
783 }
784
785 mFollowGuid.Clear();
786 mFollowDist = 0;
787 mFollowAngle = 0;
788 mFollowCredit = 0;
789 mFollowArrivedTimer = 1000;
790 mFollowArrivedEntry = 0;
791 mFollowCreditType = 0;
792 SetDespawnTime(5000);
793 me->StopMoving();
794 me->GetMotionMaster()->MoveIdle();
795 StartDespawn();
796 GetScript()->ProcessEventsFor(SMART_EVENT_FOLLOW_COMPLETED);
797}

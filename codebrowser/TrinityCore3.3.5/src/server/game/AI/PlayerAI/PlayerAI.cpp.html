51bool PlayerAI::IsPlayerHealer(Player const* who)
52{
53 switch (who->getClass())
54 {
55 case CLASS_WARRIOR:
56 case CLASS_HUNTER:
57 case CLASS_ROGUE:
58 case CLASS_DEATH_KNIGHT:
59 case CLASS_MAGE:
60 case CLASS_WARLOCK:
61 default:
62 return false;
63 case CLASS_PALADIN:
64 return who->HasSpell(PASSIVE_ILLUMINATION);
65 case CLASS_PRIEST:
66 return who->HasSpell(SPELL_SOUL_WARDING) || who->HasSpell(SPELL_SPIRIT_REDEMPTION);
67 case CLASS_SHAMAN:
68 return who->HasSpell(SPELL_MANA_TIDE_TOTEM) || who->HasSpell(SPELL_SHA_NATURE_SWIFT) || who->HasSpell(SPELL_TIDAL_FORCE);
69 case CLASS_DRUID:
70 return who->HasSpell(SPELL_SWIFTMEND) || who->HasSpell(SPELL_DRU_NATURE_SWIFT) || who->HasSpell(SPELL_TREE_OF_LIFE);
71 }
72}

74bool PlayerAI::IsPlayerRangedAttacker(Player const* who)
75{
76 switch (who->getClass())
77 {
78 case CLASS_WARRIOR:
79 case CLASS_PALADIN:
80 case CLASS_ROGUE:
81 case CLASS_DEATH_KNIGHT:
82 default:
83 return false;
84 case CLASS_MAGE:
85 case CLASS_WARLOCK:
86 return true;
87 case CLASS_HUNTER:
88 {
89 // check if we have a ranged weapon equipped
90 Item const* rangedSlot = who->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_RANGED);
91 if (ItemTemplate const* rangedTemplate = rangedSlot ? rangedSlot->GetTemplate() : nullptr)
92 if ((1 << rangedTemplate->SubClass) & ITEM_SUBCLASS_MASK_WEAPON_RANGED)
93 return true;
94 return false;
95 }
96 case CLASS_PRIEST:
97 return who->HasSpell(SPELL_SHADOWFORM);
98 case CLASS_SHAMAN:
99 return !who->HasSpell(SPELL_STORMSTRIKE);
100 case CLASS_DRUID:
101 return who->HasSpell(SPELL_MOONKIN_FORM);
102 }
103}

105void PlayerAI::DoRangedAttackIfReady()
106{
107 if (me->HasUnitState(UNIT_STATE_CASTING))
108 return;
109
110 if (!me->isAttackReady(RANGED_ATTACK))
111 return;
112
113 Unit* victim = me->GetVictim();
114 if (!victim)
115 return;
116
117 uint32 rangedAttackSpell = 0;
118
119 Item const* rangedItem = me->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_RANGED);
120 if (ItemTemplate const* rangedTemplate = rangedItem ? rangedItem->GetTemplate() : nullptr)
121 {
122 switch (rangedTemplate->SubClass)
123 {
124 case ITEM_SUBCLASS_WEAPON_BOW:
125 case ITEM_SUBCLASS_WEAPON_GUN:
126 case ITEM_SUBCLASS_WEAPON_CROSSBOW:
127 rangedAttackSpell = SPELL_SHOOT;
128 break;
129 case ITEM_SUBCLASS_WEAPON_THROWN:
130 rangedAttackSpell = SPELL_THROW;
131 break;
132 case ITEM_SUBCLASS_WEAPON_WAND:
133 rangedAttackSpell = SPELL_SHOOT_WAND;
134 break;
135 }
136 }
137
138 if (!rangedAttackSpell)
139 return;
140
141 me->CastSpell(victim, rangedAttackSpell, TRIGGERED_CAST_DIRECTLY);
142 me->resetAttackTimer(RANGED_ATTACK);
143}

145void PlayerAI::DoAutoAttackIfReady()
146{
147 if (IsRangedAttacker())
148 DoRangedAttackIfReady();
149 else
150 DoMeleeAttackIfReady();
151}

153struct UncontrolledTargetSelectPredicate : public std::unary_function<Unit*, bool>
154{
155 bool operator()(Unit const* target) const
156 {
157 return !target->HasBreakableByDamageCrowdControlAura();
158 }
159};
160Unit* SimpleCharmedPlayerAI::SelectAttackTarget() const
161{
162 if (Unit* charmer = me->GetCharmer())
163 return charmer->IsAIEnabled ? charmer->GetAI()->SelectTarget(SELECT_TARGET_RANDOM, 0, UncontrolledTargetSelectPredicate()) : charmer->GetVictim();
164 return nullptr;
165}

167void SimpleCharmedPlayerAI::UpdateAI(const uint32 /*diff*/)
168{
169 Creature* charmer = me->GetCharmer() ? me->GetCharmer()->ToCreature() : nullptr;
170 if (!charmer)
171 return;
172
173 //kill self if charm aura has infinite duration
174 if (charmer->IsInEvadeMode())
175 {
176 Player::AuraEffectList const& auras = me->GetAuraEffectsByType(SPELL_AURA_MOD_CHARM);
177 for (Player::AuraEffectList::const_iterator iter = auras.begin(); iter != auras.end(); ++iter)
178 if ((*iter)->GetCasterGUID() == charmer->GetGUID() && (*iter)->GetBase()->IsPermanent())
179 {
180 me->Kill(me);
181 return;
182 }
183 }
184
185 if (charmer->IsInCombat())
186 {
187 Unit* target = me->GetVictim();
188 if (!target || !charmer->IsValidAttackTarget(target) || target->HasBreakableByDamageCrowdControlAura())
189 {
190 target = SelectAttackTarget();
191 if (!target)
192 return;
193
194 if (IsRangedAttacker())
195 AttackStartCaster(target, 28.0f);
196 else
197 AttackStart(target);
198 }
199 DoAutoAttackIfReady();
200 }
201 else
202 {
203 me->AttackStop();
204 me->CastStop();
205 me->StopMoving();
206 me->GetMotionMaster()->Clear();
207 me->GetMotionMaster()->MoveFollow(charmer, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
208 }
209}

211void SimpleCharmedPlayerAI::OnCharmed(bool apply)
212{
213 if (apply)
214 {
215 me->CastStop();
216 me->AttackStop();
217 }
218 else
219 {
220 me->CastStop();
221 me->AttackStop();
222 // @todo only voluntary movement (don't cancel stuff like death grip or charge mid-animation)
223 me->GetMotionMaster()->Clear();
224 me->StopMoving();
225 }

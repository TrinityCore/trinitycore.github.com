22PassiveAI::PassiveAI(Creature* c) : CreatureAI(c) { me->SetReactState(REACT_PASSIVE); }
23PossessedAI::PossessedAI(Creature* c) : CreatureAI(c) { me->SetReactState(REACT_PASSIVE); }
24NullCreatureAI::NullCreatureAI(Creature* c) : CreatureAI(c) { me->SetReactState(REACT_PASSIVE); }
25
26void PassiveAI::UpdateAI(uint32)
27{
28 if (me->IsInCombat() && me->getAttackers().empty())
29 EnterEvadeMode(EVADE_REASON_NO_HOSTILES);
30}
31
32void PossessedAI::AttackStart(Unit* target)
33{
34 me->Attack(target, true);
35}
36
37void PossessedAI::UpdateAI(uint32 /*diff*/)
38{
39 if (me->GetVictim())
40 {
41 if (!me->IsValidAttackTarget(me->GetVictim()))
42 me->AttackStop();
43 else
44 DoMeleeAttackIfReady();
45 }
46}
47
48void PossessedAI::JustDied(Unit* /*u*/)
49{
50 // We died while possessed, disable our loot
51 me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
52}
53
54void PossessedAI::KilledUnit(Unit* victim)
55{
56 // We killed a creature, disable victim's loot
57 if (victim->GetTypeId() == TYPEID_UNIT)
58 victim->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
59}
60
61void CritterAI::DamageTaken(Unit* /*done_by*/, uint32&)
62{
63 if (!me->HasUnitState(UNIT_STATE_FLEEING))
64 me->SetControlled(true, UNIT_STATE_FLEEING);
65}
66
67void CritterAI::EnterEvadeMode(EvadeReason why)
68{
69 if (me->HasUnitState(UNIT_STATE_FLEEING))
70 me->SetControlled(false, UNIT_STATE_FLEEING);
71 CreatureAI::EnterEvadeMode(why);
72}
73
74void TriggerAI::IsSummonedBy(Unit* summoner)
75{
76 if (me->m_spells[0])
77 me->CastSpell(me, me->m_spells[0], false, nullptr, nullptr, summoner->GetGUID());
78}

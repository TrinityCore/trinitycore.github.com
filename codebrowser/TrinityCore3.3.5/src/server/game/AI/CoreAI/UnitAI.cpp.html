29void UnitAI::AttackStart(Unit* victim)
30{
31 if (victim && me->Attack(victim, true))
32 {

49void UnitAI::DoMeleeAttackIfReady()
50{
51 if (me->HasUnitState(UNIT_STATE_CASTING))
52 return;
53
54 Unit* victim = me->GetVictim();
55
56 if (!me->IsWithinMeleeRange(victim))
57 return;
58
59 //Make sure our attack is ready and we aren't currently casting before checking distance
60 if (me->isAttackReady())
61 {
62 me->AttackerStateUpdate(victim);
63 me->resetAttackTimer();
64 }
65
66 if (me->haveOffhandWeapon() && me->isAttackReady(OFF_ATTACK))
67 {
68 me->AttackerStateUpdate(victim, OFF_ATTACK);
69 me->resetAttackTimer(OFF_ATTACK);
70 }
71}

73bool UnitAI::DoSpellAttackIfReady(uint32 spell)
74{
75 if (me->HasUnitState(UNIT_STATE_CASTING) || !me->isAttackReady())
76 return true;
77
78 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spell))
79 {
80 if (me->IsWithinCombatRange(me->GetVictim(), spellInfo->GetMaxRange(false)))
81 {
82 me->CastSpell(me->GetVictim(), spell, false);
83 me->resetAttackTimer();
84 return true;
85 }
86 }
87
88 return false;
89}

135void UnitAI::DoCast(uint32 spellId)
136{
137 Unit* target = NULL;
138
139 switch (AISpellInfo[spellId].target)
140 {
141 default:
142 case AITARGET_SELF:
143 target = me;
144 break;
145 case AITARGET_VICTIM:
146 target = me->GetVictim();
147 break;
148 case AITARGET_ENEMY:
149 {
150 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId))
151 {
152 bool playerOnly = spellInfo->HasAttribute(SPELL_ATTR3_ONLY_TARGET_PLAYERS);
153 target = SelectTarget(SELECT_TARGET_RANDOM, 0, spellInfo->GetMaxRange(false), playerOnly);
154 }
155 break;
156 }
157 case AITARGET_ALLY:
158 target = me;
159 break;
160 case AITARGET_BUFF:
161 target = me;
162 break;
163 case AITARGET_DEBUFF:
164 {
165 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId))
166 {
167 bool playerOnly = spellInfo->HasAttribute(SPELL_ATTR3_ONLY_TARGET_PLAYERS);
168 float range = spellInfo->GetMaxRange(false);
169
170 DefaultTargetSelector targetSelector(me, range, playerOnly, -(int32)spellId);
171 if (!(spellInfo->AuraInterruptFlags & AURA_INTERRUPT_FLAG_NOT_VICTIM)
172 && targetSelector(me->GetVictim()))
173 target = me->GetVictim();
174 else
175 target = SelectTarget(SELECT_TARGET_RANDOM, 0, targetSelector);
176 }
177 break;
178 }
179 }
180
181 if (target)
182 me->CastSpell(target, spellId, false);
183}

211void UnitAI::FillAISpellInfo()
212{
213 AISpellInfo = new AISpellInfoType[sSpellMgr->GetSpellInfoStoreSize()];
214
215 AISpellInfoType* AIInfo = AISpellInfo;
216 const SpellInfo* spellInfo;
217
218 for (uint32 i = 0; i < sSpellMgr->GetSpellInfoStoreSize(); ++i, ++AIInfo)
219 {
220 spellInfo = sSpellMgr->GetSpellInfo(i);
221 if (!spellInfo)
222 continue;
223
224 if (spellInfo->HasAttribute(SPELL_ATTR0_CASTABLE_WHILE_DEAD))
225 AIInfo->condition = AICOND_DIE;
226 else if (spellInfo->IsPassive() || spellInfo->GetDuration() == -1)
227 AIInfo->condition = AICOND_AGGRO;
228 else
229 AIInfo->condition = AICOND_COMBAT;
230
231 if (AIInfo->cooldown < spellInfo->RecoveryTime)
232 AIInfo->cooldown = spellInfo->RecoveryTime;
233
234 if (!spellInfo->GetMaxRange(false))
235 UPDATE_TARGET(AITARGET_SELF)
236 else
237 {
238 for (uint32 j = 0; j < MAX_SPELL_EFFECTS; ++j)
239 {
240 uint32 targetType = spellInfo->Effects[j].TargetA.GetTarget();
241
242 if (targetType == TARGET_UNIT_TARGET_ENEMY
243 || targetType == TARGET_DEST_TARGET_ENEMY)
244 UPDATE_TARGET(AITARGET_VICTIM)
245 else if (targetType == TARGET_UNIT_DEST_AREA_ENEMY)
246 UPDATE_TARGET(AITARGET_ENEMY)
247
248 if (spellInfo->Effects[j].Effect == SPELL_EFFECT_APPLY_AURA)
249 {
250 if (targetType == TARGET_UNIT_TARGET_ENEMY)
251 UPDATE_TARGET(AITARGET_DEBUFF)
252 else if (spellInfo->IsPositive())
253 UPDATE_TARGET(AITARGET_BUFF)
254 }
255 }
256 }
257 AIInfo->realCooldown = spellInfo->RecoveryTime + spellInfo->StartRecoveryTime;
258 AIInfo->maxRange = spellInfo->GetMaxRange(false) * 3 / 4;
259 }
260}

268bool SpellTargetSelector::operator()(Unit const* target) const
269{
270 if (!target)
271 return false;
272
273 if (_spellInfo->CheckTarget(_caster, target) != SPELL_CAST_OK)
274 return false;
275
276 // copypasta from Spell::CheckRange
277 uint32 range_type = _spellInfo->RangeEntry ? _spellInfo->RangeEntry->type : 0;
278 float max_range = _caster->GetSpellMaxRangeForTarget(target, _spellInfo);
279 float min_range = _caster->GetSpellMinRangeForTarget(target, _spellInfo);
280
281
282 if (target && target != _caster)
283 {
284 if (range_type == SPELL_RANGE_MELEE)
285 {
286 // Because of lag, we can not check too strictly here.
287 if (!_caster->IsWithinMeleeRange(target, max_range))
288 return false;
289 }
290 else if (!_caster->IsWithinCombatRange(target, max_range))
291 return false;
292
293 if (range_type == SPELL_RANGE_RANGED)
294 {
295 if (_caster->IsWithinMeleeRange(target))
296 return false;
297 }
298 else if (min_range && _caster->IsWithinCombatRange(target, min_range)) // skip this check if min_range = 0
299 return false;
300 }
301
302 return true;
303}

305bool NonTankTargetSelector::operator()(Unit const* target) const
306{
307 if (!target)
308 return false;
309
310 if (_playerOnly && target->GetTypeId() != TYPEID_PLAYER)
311 return false;
312
313 return target != _source->GetVictim();

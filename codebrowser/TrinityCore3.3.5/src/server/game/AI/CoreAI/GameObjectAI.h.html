37 virtual void UpdateAI(uint32 /*diff*/) { }

41 virtual void Reset() { }

44 virtual void DoAction(int32 /*param = 0 */) { }
45 virtual void SetGUID(uint64 /*guid*/, int32 /*id = 0 */) { }

57 virtual uint32 GetData(uint32 /*id*/) const { return 0; }

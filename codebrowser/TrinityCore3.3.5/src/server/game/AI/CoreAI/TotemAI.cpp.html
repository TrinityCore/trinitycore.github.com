48void TotemAI::UpdateAI(uint32 /*diff*/)
49{
50 if (me->ToTotem()->GetTotemType() != TOTEM_ACTIVE)
51 return;
52
53 if (!me->IsAlive() || me->IsNonMeleeSpellCast(false))
54 return;
55
56 // Search spell
57 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(me->ToTotem()->GetSpell());
58 if (!spellInfo)
59 return;
60
61 // Get spell range
62 float max_range = spellInfo->GetMaxRange(false);
63
64 // SPELLMOD_RANGE not applied in this place just because not existence range mods for attacking totems
65
66 // pointer to appropriate target if found any
67 Unit* victim = i_victimGuid ? ObjectAccessor::GetUnit(*me, i_victimGuid) : NULL;
68
69 // Search victim if no, not attackable, or out of range, or friendly (possible in case duel end)
70 if (!victim ||
71 !victim->isTargetableForAttack() || !me->IsWithinDistInMap(victim, max_range) ||
72 me->IsFriendlyTo(victim) || !me->CanSeeOrDetect(victim))
73 {
74 victim = NULL;
75 Trinity::NearestAttackableUnitInObjectRangeCheck u_check(me, me, max_range);
76 Trinity::UnitLastSearcher<Trinity::NearestAttackableUnitInObjectRangeCheck> checker(me, victim, u_check);
77 me->VisitNearbyObject(max_range, checker);
78 }
79
80 // If have target
81 if (victim)
82 {
83 // remember
84 i_victimGuid = victim->GetGUID();
85
86 // attack
87 me->SetInFront(victim); // client change orientation by self
88 me->CastSpell(victim, me->ToTotem()->GetSpell(), false);
89 }
90 else
91 i_victimGuid.Clear();
92}

94void TotemAI::AttackStart(Unit* /*victim*/)
95{
96 // Sentry totem sends ping on attack
97 if (me->GetEntry() == SENTRY_TOTEM_ENTRY)
98 if (Unit* owner = me->GetOwner())
99 if (Player* player = owner->ToPlayer())
100 {
101 WorldPacket data(MSG_MINIMAP_PING, (8+4+4));
102 data << me->GetGUID();
103 data << me->GetPositionX();
104 data << me->GetPositionY();
105 player->SendDirectMessage(&data);
106 }
107}

83// Defined fuctions to use with player.

85// This fuction add's a menu item,
86// a - Icon Id
87// b - Text
88// c - Sender(this is to identify the current Menu with this item)
89// d - Action (identifys this Menu Item)
90// e - Text to be displayed in pop up box
91// f - Money value in pop up box
92// g - Coded
93// h - Menu ID from DB
94// i - Menu item ID from DB
95#define ADD_GOSSIP_ITEM(a, b, c, d) PlayerTalkClass->GetGossipMenu().AddMenuItem(-1, a, b, c, d, "", 0)
96#define ADD_GOSSIP_ITEM_DB(h, i, c, d) PlayerTalkClass->GetGossipMenu().AddMenuItem(h, i, c, d)
97#define ADD_GOSSIP_ITEM_EXTENDED(a, b, c, d, e, f, g) PlayerTalkClass->GetGossipMenu().AddMenuItem(-1, a, b, c, d, e, f, g)
98
99// This fuction Sends the current menu to show to client, a - NPCTEXTID(uint32), b - npc guid(uint64)
100#define SEND_GOSSIP_MENU(a, b) PlayerTalkClass->SendGossipMenu(a, b)
101
102// Closes the Menu
103#define CLOSE_GOSSIP_MENU() PlayerTalkClass->SendCloseGossip()

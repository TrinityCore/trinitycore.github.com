44template <class T>
45class HashMapHolder
46{
47 public:
48 static_assert(std::is_same<Player, T>::value
49 || std::is_same<Transport, T>::value,
50 "Only Player and Transport can be registered in global HashMapHolder");
51
52 typedef std::unordered_map<ObjectGuid, T*> MapType;
53
54 static void Insert(T* o)
55 {
56 boost::unique_lock<boost::shared_mutex> lock(_lock);
57
58 _objectMap[o->GetGUID()] = o;
59 }
60
61 static void Remove(T* o)
62 {
63 boost::unique_lock<boost::shared_mutex> lock(_lock);
64
65 _objectMap.erase(o->GetGUID());
66 }
67
68 static T* Find(ObjectGuid guid)
69 {
70 boost::shared_lock<boost::shared_mutex> lock(_lock);
71
72 typename MapType::iterator itr = _objectMap.find(guid);
73 return (itr != _objectMap.end()) ? itr->second : NULL;
74 }
75
76 static MapType& GetContainer() { return _objectMap; }
77
78 static boost::shared_mutex* GetLock() { return &_lock; }
79
80 private:
81 //Non instanceable only static
82 HashMapHolder() { }
83
84 static boost::shared_mutex _lock;
85 static MapType _objectMap;
86};

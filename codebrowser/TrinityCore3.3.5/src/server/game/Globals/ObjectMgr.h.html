63/// Key for storing temp summon data in TempSummonDataContainer
64struct TempSummonGroupKey
65{
66 TempSummonGroupKey(uint32 summonerEntry, SummonerType summonerType, uint8 group)
67 : _summonerEntry(summonerEntry), _summonerType(summonerType), _summonGroup(group)
68 {
69 }
70
71 bool operator<(TempSummonGroupKey const& rhs) const
72 {
73 return std::tie(_summonerEntry, _summonerType, _summonGroup) <
74 std::tie(rhs._summonerEntry, rhs._summonerType, rhs._summonGroup);
75 }
76
77private:
78 uint32 _summonerEntry; ///< Summoner's entry
79 SummonerType _summonerType; ///< Summoner's type, see SummonerType for available types
80 uint8 _summonGroup; ///< Summon's group id
81};

85// DB scripting commands
86enum ScriptCommands
87{
88 SCRIPT_COMMAND_TALK = 0, // source/target = Creature, target = any, datalong = talk type (0=say, 1=whisper, 2=yell, 3=emote text, 4=boss emote text), datalong2 & 1 = player talk (instead of creature), dataint = string_id
89 SCRIPT_COMMAND_EMOTE = 1, // source/target = Creature, datalong = emote id, datalong2 = 0: set emote state; > 0: play emote state
90 SCRIPT_COMMAND_FIELD_SET = 2, // source/target = Creature, datalong = field id, datalog2 = value
91 SCRIPT_COMMAND_MOVE_TO = 3, // source/target = Creature, datalong2 = time to reach, x/y/z = destination
92 SCRIPT_COMMAND_FLAG_SET = 4, // source/target = Creature, datalong = field id, datalog2 = bitmask
93 SCRIPT_COMMAND_FLAG_REMOVE = 5, // source/target = Creature, datalong = field id, datalog2 = bitmask
94 SCRIPT_COMMAND_TELEPORT_TO = 6, // source/target = Creature/Player (see datalong2), datalong = map_id, datalong2 = 0: Player; 1: Creature, x/y/z = destination, o = orientation
95 SCRIPT_COMMAND_QUEST_EXPLORED = 7, // target/source = Player, target/source = GO/Creature, datalong = quest id, datalong2 = distance or 0
96 SCRIPT_COMMAND_KILL_CREDIT = 8, // target/source = Player, datalong = creature entry, datalong2 = 0: personal credit, 1: group credit
97 SCRIPT_COMMAND_RESPAWN_GAMEOBJECT = 9, // source = WorldObject (summoner), datalong = GO guid, datalong2 = despawn delay
98 SCRIPT_COMMAND_TEMP_SUMMON_CREATURE = 10, // source = WorldObject (summoner), datalong = creature entry, datalong2 = despawn delay, x/y/z = summon position, o = orientation
99 SCRIPT_COMMAND_OPEN_DOOR = 11, // source = Unit, datalong = GO guid, datalong2 = reset delay (min 15)
100 SCRIPT_COMMAND_CLOSE_DOOR = 12, // source = Unit, datalong = GO guid, datalong2 = reset delay (min 15)
101 SCRIPT_COMMAND_ACTIVATE_OBJECT = 13, // source = Unit, target = GO
102 SCRIPT_COMMAND_REMOVE_AURA = 14, // source (datalong2 != 0) or target (datalong2 == 0) = Unit, datalong = spell id
103 SCRIPT_COMMAND_CAST_SPELL = 15, // source and/or target = Unit, datalong2 = cast direction (0: s->t 1: s->s 2: t->t 3: t->s 4: s->creature with dataint entry), dataint & 1 = triggered flag
104 SCRIPT_COMMAND_PLAY_SOUND = 16, // source = WorldObject, target = none/Player, datalong = sound id, datalong2 (bitmask: 0/1=anyone/player, 0/2=without/with distance dependency, so 1|2 = 3 is target with distance dependency)
105 SCRIPT_COMMAND_CREATE_ITEM = 17, // target/source = Player, datalong = item entry, datalong2 = amount
106 SCRIPT_COMMAND_DESPAWN_SELF = 18, // target/source = Creature, datalong = despawn delay
107
108 SCRIPT_COMMAND_LOAD_PATH = 20, // source = Unit, datalong = path id, datalong2 = is repeatable
109 SCRIPT_COMMAND_CALLSCRIPT_TO_UNIT = 21, // source = WorldObject (if present used as a search center), datalong = script id, datalong2 = unit lowguid, dataint = script table to use (see ScriptsType)
110 SCRIPT_COMMAND_KILL = 22, // source/target = Creature, dataint = remove corpse attribute
111
112 // TrinityCore only
113 SCRIPT_COMMAND_ORIENTATION = 30, // source = Unit, target (datalong > 0) = Unit, datalong = > 0 turn source to face target, o = orientation
114 SCRIPT_COMMAND_EQUIP = 31, // soucre = Creature, datalong = equipment id
115 SCRIPT_COMMAND_MODEL = 32, // source = Creature, datalong = model id
116 SCRIPT_COMMAND_CLOSE_GOSSIP = 33, // source = Player
117 SCRIPT_COMMAND_PLAYMOVIE = 34 // source = Player, datalong = movie id
118};

429 std::string const& GetText(LocaleConstant locale = DEFAULT_LOCALE, uint8 gender = GENDER_MALE, bool forceGender = false) const
430 {
431 if (gender == GENDER_FEMALE && (forceGender || !FemaleText[DEFAULT_LOCALE].empty()))
432 {
433 if (FemaleText.size() > size_t(locale) && !FemaleText[locale].empty())
434 return FemaleText[locale];
435 return FemaleText[DEFAULT_LOCALE];
436 }
437 // else if (gender == GENDER_MALE)
438 {
439 if (MaleText.size() > size_t(locale) && !MaleText[locale].empty())
440 return MaleText[locale];
441 return MaleText[DEFAULT_LOCALE];
442 }
443 }

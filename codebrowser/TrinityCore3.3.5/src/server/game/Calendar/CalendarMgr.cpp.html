48void CalendarMgr::LoadFromDB()
49{
50 uint32 count = 0;
51 _maxEventId = 0;
52 _maxInviteId = 0;
53
54 // 0 1 2 3 4 5 6 7 8
55 if (QueryResult result = CharacterDatabase.Query("SELECT id, creator, title, description, type, dungeon, eventtime, flags, time2 FROM calendar_events"))
56 do
57 {
58 Field* fields = result->Fetch();
59
60 uint64 eventId = fields[0].GetUInt64();
61 ObjectGuid creatorGUID = ObjectGuid(HighGuid::Player, fields[1].GetUInt32());
62 std::string title = fields[2].GetString();
63 std::string description = fields[3].GetString();
64 CalendarEventType type = CalendarEventType(fields[4].GetUInt8());
65 int32 dungeonId = fields[5].GetInt32();
66 uint32 eventTime = fields[6].GetUInt32();
67 uint32 flags = fields[7].GetUInt32();
68 uint32 timezoneTime = fields[8].GetUInt32();
69 ObjectGuid::LowType guildId = 0;
70
71 if (flags & CALENDAR_FLAG_GUILD_EVENT || flags & CALENDAR_FLAG_WITHOUT_INVITES)
72 guildId = Player::GetGuildIdFromDB(creatorGUID);
73
74 CalendarEvent* calendarEvent = new CalendarEvent(eventId, creatorGUID, guildId, type, dungeonId, time_t(eventTime), flags, time_t(timezoneTime), title, description);
75 _events.insert(calendarEvent);
76
77 _maxEventId = std::max(_maxEventId, eventId);
78
79 ++count;
80 }
81 while (result->NextRow());
82
83 TC_LOG_INFO("server.loading", ">> Loaded %u calendar events", count);
84 count = 0;
85
86 // 0 1 2 3 4 5 6 7
87 if (QueryResult result = CharacterDatabase.Query("SELECT id, event, invitee, sender, status, statustime, rank, text FROM calendar_invites"))
88 do
89 {
90 Field* fields = result->Fetch();
91
92 uint64 inviteId = fields[0].GetUInt64();
93 uint64 eventId = fields[1].GetUInt64();
94 ObjectGuid invitee = ObjectGuid(HighGuid::Player, fields[2].GetUInt32());
95 ObjectGuid senderGUID = ObjectGuid(HighGuid::Player, fields[3].GetUInt32());
96 CalendarInviteStatus status = CalendarInviteStatus(fields[4].GetUInt8());
97 uint32 statusTime = fields[5].GetUInt32();
98 CalendarModerationRank rank = CalendarModerationRank(fields[6].GetUInt8());
99 std::string text = fields[7].GetString();
100
101 CalendarInvite* invite = new CalendarInvite(inviteId, eventId, invitee, senderGUID, time_t(statusTime), status, rank, text);
102 _invites[eventId].push_back(invite);
103
104 _maxInviteId = std::max(_maxInviteId, inviteId);
105
106 ++count;
107 }
108 while (result->NextRow());
109
110 TC_LOG_INFO("server.loading", ">> Loaded %u calendar invites", count);
111
112 for (uint64 i = 1; i < _maxEventId; ++i)
113 if (!GetEvent(i))
114 _freeEventIds.push_back(i);
115
116 for (uint64 i = 1; i < _maxInviteId; ++i)
117 if (!GetInvite(i))
118 _freeInviteIds.push_back(i);
119}

149void CalendarMgr::RemoveEvent(uint64 eventId, ObjectGuid remover)
150{
151 CalendarEvent* calendarEvent = GetEvent(eventId);
152
153 if (!calendarEvent)
154 {
155 SendCalendarCommandResult(remover, CALENDAR_ERROR_EVENT_INVALID);
156 return;
157 }
158
159 SendCalendarEventRemovedAlert(*calendarEvent);
160
161 SQLTransaction trans = CharacterDatabase.BeginTransaction();
162 PreparedStatement* stmt;
163 MailDraft mail(calendarEvent->BuildCalendarMailSubject(remover), calendarEvent->BuildCalendarMailBody());
164
165 CalendarInviteStore& eventInvites = _invites[eventId];
166 for (size_t i = 0; i < eventInvites.size(); ++i)
167 {
168 CalendarInvite* invite = eventInvites[i];
169 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_INVITE);
170 stmt->setUInt64(0, invite->GetInviteId());
171 trans->Append(stmt);
172
173 // guild events only? check invite status here?
174 // When an event is deleted, all invited (accepted/declined? - verify) guildies are notified via in-game mail. (wowwiki)
175 if (remover && invite->GetInviteeGUID() != remover)
176 mail.SendMailTo(trans, MailReceiver(invite->GetInviteeGUID().GetCounter()), calendarEvent, MAIL_CHECK_MASK_COPIED);
177
178 delete invite;
179 }
180
181 _invites.erase(eventId);
182
183 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_EVENT);
184 stmt->setUInt64(0, eventId);
185 trans->Append(stmt);
186 CharacterDatabase.CommitTransaction(trans);
187
188 delete calendarEvent;
189 _events.erase(calendarEvent);
190}

192void CalendarMgr::RemoveInvite(uint64 inviteId, uint64 eventId, ObjectGuid /*remover*/)
193{
194 CalendarEvent* calendarEvent = GetEvent(eventId);
195
196 if (!calendarEvent)
197 return;
198
199 CalendarInviteStore::iterator itr = _invites[eventId].begin();
200 for (; itr != _invites[eventId].end(); ++itr)
201 if ((*itr)->GetInviteId() == inviteId)
202 break;
203
204 if (itr == _invites[eventId].end())
205 return;
206
207 SQLTransaction trans = CharacterDatabase.BeginTransaction();
208 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_INVITE);
209 stmt->setUInt64(0, (*itr)->GetInviteId());
210 trans->Append(stmt);
211 CharacterDatabase.CommitTransaction(trans);
212
213 if (!calendarEvent->IsGuildEvent())
214 SendCalendarEventInviteRemoveAlert((*itr)->GetInviteeGUID(), *calendarEvent, CALENDAR_STATUS_REMOVED);
215
216 SendCalendarEventInviteRemove(*calendarEvent, **itr, calendarEvent->GetFlags());
217
218 // we need to find out how to use CALENDAR_INVITE_REMOVED_MAIL_SUBJECT to force client to display different mail
219 //if ((*itr)->GetInviteeGUID() != remover)
220 // MailDraft(calendarEvent->BuildCalendarMailSubject(remover), calendarEvent->BuildCalendarMailBody())
221 // .SendMailTo(trans, MailReceiver((*itr)->GetInvitee()), calendarEvent, MAIL_CHECK_MASK_COPIED);
222
223 delete *itr;
224 _invites[eventId].erase(itr);
225}

262void CalendarMgr::RemoveAllPlayerEventsAndInvites(ObjectGuid guid)
263{
264 for (CalendarEventStore::const_iterator itr = _events.begin(); itr != _events.end(); ++itr)
265 if ((*itr)->GetCreatorGUID() == guid)
266 RemoveEvent((*itr)->GetEventId(), ObjectGuid::Empty); // don't send mail if removing a character
267
268 CalendarInviteStore playerInvites = GetPlayerInvites(guid);
269 for (CalendarInviteStore::const_iterator itr = playerInvites.begin(); itr != playerInvites.end(); ++itr)
270 RemoveInvite((*itr)->GetInviteId(), (*itr)->GetEventId(), guid);
271}

273void CalendarMgr::RemovePlayerGuildEventsAndSignups(ObjectGuid guid, ObjectGuid::LowType guildId)
274{
275 for (CalendarEventStore::const_iterator itr = _events.begin(); itr != _events.end(); ++itr)
276 if ((*itr)->GetCreatorGUID() == guid && ((*itr)->IsGuildEvent() || (*itr)->IsGuildAnnouncement()))
277 RemoveEvent((*itr)->GetEventId(), guid);
278
279 CalendarInviteStore playerInvites = GetPlayerInvites(guid);
280 for (CalendarInviteStore::const_iterator itr = playerInvites.begin(); itr != playerInvites.end(); ++itr)
281 if (CalendarEvent* calendarEvent = GetEvent((*itr)->GetEventId()))
282 if (calendarEvent->IsGuildEvent() && calendarEvent->GetGuildId() == guildId)
283 RemoveInvite((*itr)->GetInviteId(), (*itr)->GetEventId(), guid);
284}

343CalendarEventStore CalendarMgr::GetPlayerEvents(ObjectGuid guid)
344{
345 CalendarEventStore events;
346
347 for (CalendarEventInviteStore::const_iterator itr = _invites.begin(); itr != _invites.end(); ++itr)
348 for (CalendarInviteStore::const_iterator itr2 = itr->second.begin(); itr2 != itr->second.end(); ++itr2)

352
353 if (Player* player = ObjectAccessor::FindConnectedPlayer(guid))
354 for (CalendarEventStore::const_iterator itr = _events.begin(); itr != _events.end(); ++itr)
355 if ((*itr)->GetGuildId() == player->GetGuildId())
356 events.insert(*itr);

379{

394 }

396

420void CalendarMgr::SendCalendarEventInvite(CalendarInvite const& invite)
421{
422 CalendarEvent* calendarEvent = GetEvent(invite.GetEventId());
423 time_t statusTime = invite.GetStatusTime();
424 bool hasStatusTime = statusTime != 946684800; // 01/01/2000 00:00:00
425
426 ObjectGuid invitee = invite.GetInviteeGUID();
427 Player* player = ObjectAccessor::FindConnectedPlayer(invitee);
428
429 uint8 level = player ? player->getLevel() : Player::GetLevelFromDB(invitee);
430
431 WorldPacket data(SMSG_CALENDAR_EVENT_INVITE, 8 + 8 + 8 + 1 + 1 + 1 + (statusTime ? 4 : 0) + 1);
432 data << invitee.WriteAsPacked();
433 data << uint64(invite.GetEventId());
434 data << uint64(invite.GetInviteId());
435 data << uint8(level);
436 data << uint8(invite.GetStatus());
437 data << uint8(hasStatusTime);
438 if (hasStatusTime)
439 data.AppendPackedTime(statusTime);
440 data << uint8(invite.GetSenderGUID() != invite.GetInviteeGUID()); // false only if the invite is sign-up
441
442 if (!calendarEvent) // Pre-invite
443 {
444 if (Player* playerSender = ObjectAccessor::FindConnectedPlayer(invite.GetSenderGUID()))
445 playerSender->SendDirectMessage(&data);
446 }
447 else
448 {
449 if (calendarEvent->GetCreatorGUID() != invite.GetInviteeGUID()) // correct?
450 SendPacketToAllEventRelatives(data, *calendarEvent);
451 }
452}

545void CalendarMgr::SendCalendarEvent(ObjectGuid guid, CalendarEvent const& calendarEvent, CalendarSendEventType sendType)
546{
547 Player* player = ObjectAccessor::FindConnectedPlayer(guid);
548 if (!player)
549 return;
550
551 CalendarInviteStore const& eventInviteeList = _invites[calendarEvent.GetEventId()];
552
553 WorldPacket data(SMSG_CALENDAR_SEND_EVENT, 60 + eventInviteeList.size() * 32);
554 data << uint8(sendType);
555 data << calendarEvent.GetCreatorGUID().WriteAsPacked();
556 data << uint64(calendarEvent.GetEventId());
557 data << calendarEvent.GetTitle();
558 data << calendarEvent.GetDescription();
559 data << uint8(calendarEvent.GetType());
560 data << uint8(CALENDAR_REPEAT_NEVER); // repeatable
561 data << uint32(CALENDAR_MAX_INVITES);
562 data << int32(calendarEvent.GetDungeonId());
563 data << uint32(calendarEvent.GetFlags());
564 data.AppendPackedTime(calendarEvent.GetEventTime());
565 data.AppendPackedTime(calendarEvent.GetTimeZoneTime());
566 data << uint32(calendarEvent.GetGuildId());
567
568 data << uint32(eventInviteeList.size());
569 for (CalendarInviteStore::const_iterator itr = eventInviteeList.begin(); itr != eventInviteeList.end(); ++itr)
570 {
571 CalendarInvite const* calendarInvite = (*itr);
572 ObjectGuid inviteeGuid = calendarInvite->GetInviteeGUID();
573 Player* invitee = ObjectAccessor::FindPlayer(inviteeGuid);
574
575 uint8 inviteeLevel = invitee ? invitee->getLevel() : Player::GetLevelFromDB(inviteeGuid);
576 ObjectGuid::LowType inviteeGuildId = invitee ? invitee->GetGuildId() : Player::GetGuildIdFromDB(inviteeGuid);
577
578 data << inviteeGuid.WriteAsPacked();
579 data << uint8(inviteeLevel);
580 data << uint8(calendarInvite->GetStatus());
581 data << uint8(calendarInvite->GetRank());
582 data << uint8(calendarEvent.IsGuildEvent() && calendarEvent.GetGuildId() == inviteeGuildId);
583 data << uint64(calendarInvite->GetInviteId());
584 data.AppendPackedTime(calendarInvite->GetStatusTime());
585 data << calendarInvite->GetText();
586 }
587
588 player->SendDirectMessage(&data);
589}

614void CalendarMgr::SendCalendarCommandResult(ObjectGuid guid, CalendarError err, char const* param /*= NULL*/)
615{
616 if (Player* player = ObjectAccessor::FindConnectedPlayer(guid))
617 {
618 WorldPacket data(SMSG_CALENDAR_COMMAND_RESULT, 0);
619 data << uint32(0);
620 data << uint8(0);
621 switch (err)
622 {
623 case CALENDAR_ERROR_OTHER_INVITES_EXCEEDED:
624 case CALENDAR_ERROR_ALREADY_INVITED_TO_EVENT_S:
625 case CALENDAR_ERROR_IGNORING_YOU_S:
626 data << param;
627 break;
628 default:
629 data << uint8(0);
630 break;
631 }

633 data << uint32(err);
634
635 player->SendDirectMessage(&data);
636 }

639void CalendarMgr::SendPacketToAllEventRelatives(WorldPacket& packet, CalendarEvent const& calendarEvent)
640{
641 // Send packet to all guild members
642 if (calendarEvent.IsGuildEvent() || calendarEvent.IsGuildAnnouncement())
643 if (Guild* guild = sGuildMgr->GetGuildById(calendarEvent.GetGuildId()))
644 guild->BroadcastPacket(&packet);
645
646 // Send packet to all invitees if event is non-guild, in other case only to non-guild invitees (packet was broadcasted for them)
647 CalendarInviteStore invites = _invites[calendarEvent.GetEventId()];
648 for (CalendarInviteStore::iterator itr = invites.begin(); itr != invites.end(); ++itr)
649 if (Player* player = ObjectAccessor::FindConnectedPlayer((*itr)->GetInviteeGUID()))
650 if (!calendarEvent.IsGuildEvent() || (calendarEvent.IsGuildEvent() && player->GetGuildId() != calendarEvent.GetGuildId()))
651 player->SendDirectMessage(&packet);
652}

32#define WEATHER_SEASONS 4
33struct WeatherSeasonChances
34{
35 uint32 rainChance;
36 uint32 snowChance;
37 uint32 stormChance;
38};
39
40struct WeatherData
41{
42 WeatherSeasonChances data[WEATHER_SEASONS];
43 uint32 ScriptId;
44};
45
46enum WeatherState
47{
48 WEATHER_STATE_FINE = 0,
49 WEATHER_STATE_FOG = 1,
50 WEATHER_STATE_LIGHT_RAIN = 3,
51 WEATHER_STATE_MEDIUM_RAIN = 4,
52 WEATHER_STATE_HEAVY_RAIN = 5,
53 WEATHER_STATE_LIGHT_SNOW = 6,
54 WEATHER_STATE_MEDIUM_SNOW = 7,
55 WEATHER_STATE_HEAVY_SNOW = 8,
56 WEATHER_STATE_LIGHT_SANDSTORM = 22,
57 WEATHER_STATE_MEDIUM_SANDSTORM = 41,
58 WEATHER_STATE_HEAVY_SANDSTORM = 42,
59 WEATHER_STATE_THUNDERS = 86,
60 WEATHER_STATE_BLACKRAIN = 90,
61 WEATHER_STATE_BLACKSNOW = 106
62};
63
64/// Weather for one zone
65class Weather
66{
67 public:
68
69 Weather(uint32 zone, WeatherData const* weatherChances);
70 ~Weather() { };
71
72 bool Update(uint32 diff);
73 bool ReGenerate();
74 bool UpdateWeather();
75
76 void SendWeatherUpdateToPlayer(Player* player);
77 void SetWeather(WeatherType type, float grade);
78
79 /// For which zone is this weather?
80 uint32 GetZone() const { return m_zone; };
81 uint32 GetScriptId() const { return m_weatherChances->ScriptId; }
82
83 private:
84
85 WeatherState GetWeatherState() const;
86 uint32 m_zone;
87 WeatherType m_type;
88 float m_grade;
89 IntervalTimer m_timer;
90 WeatherData const* m_weatherChances;
91};

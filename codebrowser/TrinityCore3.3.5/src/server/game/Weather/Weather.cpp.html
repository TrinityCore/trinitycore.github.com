32/// Create the Weather object
33Weather::Weather(uint32 zone, WeatherData const* weatherChances)
34 : m_zone(zone), m_weatherChances(weatherChances)
35{
36 m_timer.SetInterval(sWorld->getIntConfig(CONFIG_INTERVAL_CHANGEWEATHER));
37 m_type = WEATHER_TYPE_FINE;
38 m_grade = 0;
39
40 TC_LOG_INFO("misc", "WORLD: Starting weather system for zone %u (change every %u minutes).", m_zone, (uint32)(m_timer.GetInterval() / (MINUTE*IN_MILLISECONDS)));
41}

43/// Launch a weather update
44bool Weather::Update(uint32 diff)
45{
46 if (m_timer.GetCurrent() >= 0)
47 m_timer.Update(diff);
48 else
49 m_timer.SetCurrent(0);
50
51 ///- If the timer has passed, ReGenerate the weather
52 if (m_timer.Passed())
53 {
54 m_timer.Reset();
55 // update only if Regenerate has changed the weather
56 if (ReGenerate())
57 {
58 ///- Weather will be removed if not updated (no players in zone anymore)
59 if (!UpdateWeather())
60 return false;
61 }
62 }
63
64 sScriptMgr->OnWeatherUpdate(this, diff);
65 return true;
66}

68/// Calculate the new weather
69bool Weather::ReGenerate()
70{
71 if (!m_weatherChances)
72 {
73 m_type = WEATHER_TYPE_FINE;
74 m_grade = 0.0f;
75 return false;
76 }
77
78 /// Weather statistics:
79 ///- 30% - no change
80 ///- 30% - weather gets better (if not fine) or change weather type
81 ///- 30% - weather worsens (if not fine)
82 ///- 10% - radical change (if not fine)
83 uint32 u = urand(0, 99);
84
85 if (u < 30)
86 return false;
87
88 // remember old values
89 WeatherType old_type = m_type;
90 float old_grade = m_grade;
91
92 //78 days between January 1st and March 20nd; 365/4=91 days by season
93 // season source http://aa.usno.navy.mil/data/docs/EarthSeasons.html
94 time_t gtime = sWorld->GetGameTime();
95 struct tm ltime;
96 localtime_r(&gtime, &ltime);
97 uint32 season = ((ltime.tm_yday - 78 + 365)/91)%4;
98
99 static char const* seasonName[WEATHER_SEASONS] = { "spring", "summer", "fall", "winter" };
100
101 TC_LOG_INFO("misc", "Generating a change in %s weather for zone %u.", seasonName[season], m_zone);
102
103 if ((u < 60) && (m_grade < 0.33333334f)) // Get fair
104 {
105 m_type = WEATHER_TYPE_FINE;
106 m_grade = 0.0f;
107 }
108
109 if ((u < 60) && (m_type != WEATHER_TYPE_FINE)) // Get better
110 {
111 m_grade -= 0.33333334f;
112 return true;
113 }
114
115 if ((u < 90) && (m_type != WEATHER_TYPE_FINE)) // Get worse
116 {
117 m_grade += 0.33333334f;
118 return true;
119 }
120
121 if (m_type != WEATHER_TYPE_FINE)
122 {
123 /// Radical change:
124 ///- if light -> heavy
125 ///- if medium -> change weather type
126 ///- if heavy -> 50% light, 50% change weather type
127
128 if (m_grade < 0.33333334f)
129 {
130 m_grade = 0.9999f; // go nuts
131 return true;
132 }
133 else
134 {
135 if (m_grade > 0.6666667f)
136 {
137 // Severe change, but how severe?
138 uint32 rnd = urand(0, 99);
139 if (rnd < 50)
140 {
141 m_grade -= 0.6666667f;
142 return true;
143 }
144 }
145 m_type = WEATHER_TYPE_FINE; // clear up
146 m_grade = 0;
147 }
148 }
149
150 // At this point, only weather that isn't doing anything remains but that have weather data
151 uint32 chance1 = m_weatherChances->data[season].rainChance;
152 uint32 chance2 = chance1+ m_weatherChances->data[season].snowChance;
153 uint32 chance3 = chance2+ m_weatherChances->data[season].stormChance;
154
155 uint32 rnd = urand(1, 100);
156 if (rnd <= chance1)
157 m_type = WEATHER_TYPE_RAIN;
158 else if (rnd <= chance2)
159 m_type = WEATHER_TYPE_SNOW;
160 else if (rnd <= chance3)
161 m_type = WEATHER_TYPE_STORM;
162 else
163 m_type = WEATHER_TYPE_FINE;
164
165 /// New weather statistics (if not fine):
166 ///- 85% light
167 ///- 7% medium
168 ///- 7% heavy
169 /// If fine 100% sun (no fog)
170
171 if (m_type == WEATHER_TYPE_FINE)
172 {
173 m_grade = 0.0f;
174 }
175 else if (u < 90)
176 {
177 m_grade = (float)rand_norm() * 0.3333f;
178 }
179 else
180 {
181 // Severe change, but how severe?
182 rnd = urand(0, 99);
183 if (rnd < 50)
184 m_grade = (float)rand_norm() * 0.3333f + 0.3334f;
185 else
186 m_grade = (float)rand_norm() * 0.3333f + 0.6667f;
187 }
188
189 // return true only in case weather changes
190 return m_type != old_type || m_grade != old_grade;
191}

193void Weather::SendWeatherUpdateToPlayer(Player* player)
194{
195 WorldPacket data(SMSG_WEATHER, (4 + 4 + 1));
196 data << uint32(GetWeatherState());
197 data << (float)m_grade;
198 data << uint8(0);
199 player->GetSession()->SendPacket(&data);
200}

202/// Send the new weather to all players in the zone
203bool Weather::UpdateWeather()
204{
205 ///- Send the weather packet to all players in this zone
206 if (m_grade >= 1)
207 m_grade = 0.9999f;
208 else if (m_grade < 0)
209 m_grade = 0.0001f;
210
211 WeatherState state = GetWeatherState();
212
213 WorldPacket data(SMSG_WEATHER, (4 + 4 + 1));
214 data << uint32(state);
215 data << (float)m_grade;
216 data << uint8(0);
217
218 //- Returns false if there were no players found to update
219 if (!sWorld->SendZoneMessage(m_zone, &data))
220 return false;
221
222 ///- Log the event
223 char const* wthstr;
224 switch (state)
225 {
226 case WEATHER_STATE_FOG:
227 wthstr = "fog";
228 break;
229 case WEATHER_STATE_LIGHT_RAIN:
230 wthstr = "light rain";
231 break;
232 case WEATHER_STATE_MEDIUM_RAIN:
233 wthstr = "medium rain";
234 break;
235 case WEATHER_STATE_HEAVY_RAIN:
236 wthstr = "heavy rain";
237 break;
238 case WEATHER_STATE_LIGHT_SNOW:
239 wthstr = "light snow";
240 break;
241 case WEATHER_STATE_MEDIUM_SNOW:
242 wthstr = "medium snow";
243 break;
244 case WEATHER_STATE_HEAVY_SNOW:
245 wthstr = "heavy snow";
246 break;
247 case WEATHER_STATE_LIGHT_SANDSTORM:
248 wthstr = "light sandstorm";
249 break;
250 case WEATHER_STATE_MEDIUM_SANDSTORM:
251 wthstr = "medium sandstorm";
252 break;
253 case WEATHER_STATE_HEAVY_SANDSTORM:
254 wthstr = "heavy sandstorm";
255 break;
256 case WEATHER_STATE_THUNDERS:
257 wthstr = "thunders";
258 break;
259 case WEATHER_STATE_BLACKRAIN:
260 wthstr = "blackrain";
261 break;
262 case WEATHER_STATE_FINE:
263 default:
264 wthstr = "fine";
265 break;
266 }
267
268 TC_LOG_INFO("misc", "Change the weather of zone %u to %s.", m_zone, wthstr);
269 sScriptMgr->OnWeatherChange(this, state, m_grade);
270 return true;
271}

273/// Set the weather
274void Weather::SetWeather(WeatherType type, float grade)
275{
276 if (m_type == type && m_grade == grade)
277 return;
278
279 m_type = type;
280 m_grade = grade;
281 UpdateWeather();
282}

284/// Get the sound number associated with the current weather
285WeatherState Weather::GetWeatherState() const
286{
287 if (m_grade<0.27f)
288 return WEATHER_STATE_FINE;
289
290 switch (m_type)
291 {
292 case WEATHER_TYPE_RAIN:
293 if (m_grade<0.40f)
294 return WEATHER_STATE_LIGHT_RAIN;
295 else if (m_grade<0.70f)
296 return WEATHER_STATE_MEDIUM_RAIN;
297 else
298 return WEATHER_STATE_HEAVY_RAIN;
299 case WEATHER_TYPE_SNOW:
300 if (m_grade<0.40f)
301 return WEATHER_STATE_LIGHT_SNOW;
302 else if (m_grade<0.70f)
303 return WEATHER_STATE_MEDIUM_SNOW;
304 else
305 return WEATHER_STATE_HEAVY_SNOW;
306 case WEATHER_TYPE_STORM:
307 if (m_grade<0.40f)
308 return WEATHER_STATE_LIGHT_SANDSTORM;
309 else if (m_grade<0.70f)
310 return WEATHER_STATE_MEDIUM_SANDSTORM;
311 else
312 return WEATHER_STATE_HEAVY_SANDSTORM;
313 case WEATHER_TYPE_BLACKRAIN:
314 return WEATHER_STATE_BLACKRAIN;
315 case WEATHER_TYPE_THUNDERS:
316 return WEATHER_STATE_THUNDERS;
317 case WEATHER_TYPE_FINE:
318 default:
319 return WEATHER_STATE_FINE;
320 }

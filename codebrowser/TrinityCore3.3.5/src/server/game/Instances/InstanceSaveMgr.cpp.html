41void InstanceSaveManager::Unload()
42{
43 lock_instLists = true;
44 for (InstanceSaveHashMap::iterator itr = m_instanceSaveById.begin(); itr != m_instanceSaveById.end(); ++itr)
45 {
46 InstanceSave* save = itr->second;
47
48 for (InstanceSave::PlayerListType::iterator itr2 = save->m_playerList.begin(), next = itr2; itr2 != save->m_playerList.end(); itr2 = next)
49 {
50 ++next;
51 (*itr2)->UnbindInstance(save->GetMapId(), save->GetDifficulty(), true);
52 }
53
54 for (InstanceSave::GroupListType::iterator itr2 = save->m_groupList.begin(), next = itr2; itr2 != save->m_groupList.end(); itr2 = next)
55 {
56 ++next;
57 (*itr2)->UnbindInstance(save->GetMapId(), save->GetDifficulty(), true);
58 }
59
60 delete save;
61 }
62}

68InstanceSave* InstanceSaveManager::AddInstanceSave(uint32 mapId, uint32 instanceId, Difficulty difficulty, time_t resetTime, bool canReset, bool load)
69{
70 if (InstanceSave* old_save = GetInstanceSave(instanceId))
71 return old_save;
72
73 const MapEntry* entry = sMapStore.LookupEntry(mapId);
74 if (!entry)
75 {
76 TC_LOG_ERROR("misc", "InstanceSaveManager::AddInstanceSave: wrong mapid = %d, instanceid = %d!", mapId, instanceId);
77 return NULL;
78 }
79
80 if (instanceId == 0)
81 {
82 TC_LOG_ERROR("misc", "InstanceSaveManager::AddInstanceSave: mapid = %d, wrong instanceid = %d!", mapId, instanceId);
83 return NULL;
84 }
85
86 if (difficulty >= (entry->IsRaid() ? MAX_RAID_DIFFICULTY : MAX_DUNGEON_DIFFICULTY))
87 {
88 TC_LOG_ERROR("misc", "InstanceSaveManager::AddInstanceSave: mapid = %d, instanceid = %d, wrong dificalty %u!", mapId, instanceId, difficulty);
89 return NULL;
90 }
91
92 if (!resetTime)
93 {
94 // initialize reset time
95 // for normal instances if no creatures are killed the instance will reset in two hours
96 if (entry->map_type == MAP_RAID || difficulty > DUNGEON_DIFFICULTY_NORMAL)
97 resetTime = GetResetTimeFor(mapId, difficulty);
98 else
99 {
100 resetTime = time(NULL) + 2 * HOUR;
101 // normally this will be removed soon after in InstanceMap::Add, prevent error
102 ScheduleReset(true, resetTime, InstResetEvent(0, mapId, difficulty, instanceId));
103 }
104 }
105
106 TC_LOG_DEBUG("maps", "InstanceSaveManager::AddInstanceSave: mapid = %d, instanceid = %d", mapId, instanceId);
107
108 InstanceSave* save = new InstanceSave(mapId, instanceId, difficulty, resetTime, canReset);
109 if (!load)
110 save->SaveToDB();
111
112 m_instanceSaveById[instanceId] = save;
113 return save;
114}

142void InstanceSaveManager::RemoveInstanceSave(uint32 InstanceId)
143{
144 InstanceSaveHashMap::iterator itr = m_instanceSaveById.find(InstanceId);
145 if (itr != m_instanceSaveById.end())
146 {
147 // save the resettime for normal instances only when they get unloaded
148 if (time_t resettime = itr->second->GetResetTimeForDB())
149 {
150 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_INSTANCE_RESETTIME);
151
152 stmt->setUInt32(0, uint32(resettime));
153 stmt->setUInt32(1, InstanceId);
154
155 CharacterDatabase.Execute(stmt);
156 }
157
158 itr->second->SetToDelete(true);
159 m_instanceSaveById.erase(itr);
160 }
161}

186void InstanceSave::SaveToDB()
187{
188 // save instance data too
189 std::string data;
190 uint32 completedEncounters = 0;
191
192 Map* map = sMapMgr->FindMap(GetMapId(), m_instanceid);
193 if (map)
194 {
195 ASSERT(map->IsDungeon());
196 if (InstanceScript* instanceScript = ((InstanceMap*)map)->GetInstanceScript())
197 {
198 data = instanceScript->GetSaveData();
199 completedEncounters = instanceScript->GetCompletedEncounterMask();
200 }
201 }
202
203 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_INSTANCE_SAVE);
204 stmt->setUInt32(0, m_instanceid);
205 stmt->setUInt16(1, GetMapId());
206 stmt->setUInt32(2, uint32(GetResetTimeForDB()));
207 stmt->setUInt8(3, uint8(GetDifficulty()));
208 stmt->setUInt32(4, completedEncounters);
209 stmt->setString(5, data);
210 CharacterDatabase.Execute(stmt);
211}

239/* true if the instance save is still valid */
240bool InstanceSave::UnloadIfEmpty()
241{
242 if (m_playerList.empty() && m_groupList.empty())
243 {
244 // don't remove the save if there are still players inside the map
245 if (Map* map = sMapMgr->FindMap(GetMapId(), GetInstanceId()))
246 if (map->HavePlayers())
247 return true;
248
249 if (!sInstanceSaveMgr->lock_instLists)
250 sInstanceSaveMgr->RemoveInstanceSave(GetInstanceId());
251
252 return false;
253 }
254 else
255 return true;
256}

258void InstanceSaveManager::LoadInstances()
259{
260 uint32 oldMSTime = getMSTime();
261
262 // Delete expired instances (Instance related spawns are removed in the following cleanup queries)
263 CharacterDatabase.DirectExecute("DELETE i FROM instance i LEFT JOIN instance_reset ir ON mapid = map AND i.difficulty = ir.difficulty "
264 "WHERE (i.resettime > 0 AND i.resettime < UNIX_TIMESTAMP()) OR (ir.resettime IS NOT NULL AND ir.resettime < UNIX_TIMESTAMP())");
265
266 // Delete invalid character_instance and group_instance references
267 CharacterDatabase.DirectExecute("DELETE ci.* FROM character_instance AS ci LEFT JOIN characters AS c ON ci.guid = c.guid WHERE c.guid IS NULL");
268 CharacterDatabase.DirectExecute("DELETE gi.* FROM group_instance AS gi LEFT JOIN groups AS g ON gi.guid = g.guid WHERE g.guid IS NULL");
269
270 // Delete invalid instance references
271 CharacterDatabase.DirectExecute("DELETE i.* FROM instance AS i LEFT JOIN character_instance AS ci ON i.id = ci.instance LEFT JOIN group_instance AS gi ON i.id = gi.instance WHERE ci.guid IS NULL AND gi.guid IS NULL");
272
273 // Delete invalid references to instance
274 CharacterDatabase.DirectExecute("DELETE FROM creature_respawn WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");
275 CharacterDatabase.DirectExecute("DELETE FROM gameobject_respawn WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");
276 CharacterDatabase.DirectExecute("DELETE tmp.* FROM character_instance AS tmp LEFT JOIN instance ON tmp.instance = instance.id WHERE tmp.instance > 0 AND instance.id IS NULL");
277 CharacterDatabase.DirectExecute("DELETE tmp.* FROM group_instance AS tmp LEFT JOIN instance ON tmp.instance = instance.id WHERE tmp.instance > 0 AND instance.id IS NULL");
278
279 // Clean invalid references to instance
280 CharacterDatabase.DirectExecute("UPDATE corpse SET instanceId = 0 WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");
281 CharacterDatabase.DirectExecute("UPDATE characters AS tmp LEFT JOIN instance ON tmp.instance_id = instance.id SET tmp.instance_id = 0 WHERE tmp.instance_id > 0 AND instance.id IS NULL");
282
283 // Initialize instance id storage (Needs to be done after the trash has been clean out)
284 sMapMgr->InitInstanceIds();
285
286 // Load reset times and clean expired instances
287 sInstanceSaveMgr->LoadResetTimes();
288
289 TC_LOG_INFO("server.loading", ">> Loaded instances in %u ms", GetMSTimeDiffToNow(oldMSTime));
290
291}

293void InstanceSaveManager::LoadResetTimes()
294{
295 time_t now = time(NULL);
296 time_t today = (now / DAY) * DAY;
297
298 // NOTE: Use DirectPExecute for tables that will be queried later
299
300 // get the current reset times for normal instances (these may need to be updated)
301 // these are only kept in memory for InstanceSaves that are loaded later
302 // resettime = 0 in the DB for raid/heroic instances so those are skipped
303 typedef std::pair<uint32 /*PAIR32(map, difficulty)*/, time_t> ResetTimeMapDiffType;
304 typedef std::map<uint32, ResetTimeMapDiffType> InstResetTimeMapDiffType;
305 InstResetTimeMapDiffType instResetTime;
306
307 // index instance ids by map/difficulty pairs for fast reset warning send
308 typedef std::multimap<uint32 /*PAIR32(map, difficulty)*/, uint32 /*instanceid*/ > ResetTimeMapDiffInstances;
309 typedef std::pair<ResetTimeMapDiffInstances::const_iterator, ResetTimeMapDiffInstances::const_iterator> ResetTimeMapDiffInstancesBounds;
310 ResetTimeMapDiffInstances mapDiffResetInstances;
311
312 QueryResult result = CharacterDatabase.Query("SELECT id, map, difficulty, resettime FROM instance ORDER BY id ASC");
313 if (result)
314 {
315 do
316 {
317 Field* fields = result->Fetch();
318
319 uint32 instanceId = fields[0].GetUInt32();
320
321 // Instances are pulled in ascending order from db and nextInstanceId is initialized with 1,
322 // so if the instance id is used, increment until we find the first unused one for a potential new instance
323 if (sMapMgr->GetNextInstanceId() == instanceId)
324 sMapMgr->SetNextInstanceId(instanceId + 1);
325
326 // Mark instance id as being used
327 sMapMgr->RegisterInstanceId(instanceId);
328
329 if (time_t resettime = time_t(fields[3].GetUInt32()))
330 {
331 uint32 mapid = fields[1].GetUInt16();
332 uint32 difficulty = fields[2].GetUInt8();
333
334 instResetTime[instanceId] = ResetTimeMapDiffType(MAKE_PAIR32(mapid, difficulty), resettime);
335 mapDiffResetInstances.insert(ResetTimeMapDiffInstances::value_type(MAKE_PAIR32(mapid, difficulty), instanceId));
336 }
337 }
338 while (result->NextRow());
339
340 // update reset time for normal instances with the max creature respawn time + X hours
341 if (PreparedQueryResult result2 = CharacterDatabase.Query(CharacterDatabase.GetPreparedStatement(CHAR_SEL_MAX_CREATURE_RESPAWNS)))
342 {
343 do
344 {
345 Field* fields = result2->Fetch();
346 uint32 instance = fields[1].GetUInt32();
347 time_t resettime = time_t(fields[0].GetUInt32() + 2 * HOUR);
348 InstResetTimeMapDiffType::iterator itr = instResetTime.find(instance);
349 if (itr != instResetTime.end() && itr->second.second != resettime)
350 {
351 CharacterDatabase.DirectPExecute("UPDATE instance SET resettime = '" UI64FMTD "' WHERE id = '%u'", uint64(resettime), instance);
352 itr->second.second = resettime;
353 }
354 }
355 while (result->NextRow());
356 }
357
358 // schedule the reset times
359 for (InstResetTimeMapDiffType::iterator itr = instResetTime.begin(); itr != instResetTime.end(); ++itr)
360 if (itr->second.second > now)
361 ScheduleReset(true, itr->second.second, InstResetEvent(0, PAIR32_LOPART(itr->second.first), Difficulty(PAIR32_HIPART(itr->second.first)), itr->first));
362 }
363
364 // load the global respawn times for raid/heroic instances
365 uint32 diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
366 result = CharacterDatabase.Query("SELECT mapid, difficulty, resettime FROM instance_reset");
367 if (result)
368 {
369 do
370 {
371 Field* fields = result->Fetch();
372 uint32 mapid = fields[0].GetUInt16();
373 Difficulty difficulty = Difficulty(fields[1].GetUInt8());
374 uint64 oldresettime = fields[2].GetUInt32();
375
376 MapDifficulty const* mapDiff = GetMapDifficultyData(mapid, difficulty);
377 if (!mapDiff)
378 {
379 TC_LOG_ERROR("misc", "InstanceSaveManager::LoadResetTimes: invalid mapid(%u)/difficulty(%u) pair in instance_reset!", mapid, difficulty);
380 CharacterDatabase.DirectPExecute("DELETE FROM instance_reset WHERE mapid = '%u' AND difficulty = '%u'", mapid, difficulty);
381 continue;
382 }
383
384 // update the reset time if the hour in the configs changes
385 uint64 newresettime = (oldresettime / DAY) * DAY + diff;
386 if (oldresettime != newresettime)
387 CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '%u' WHERE mapid = '%u' AND difficulty = '%u'", uint32(newresettime), mapid, difficulty);
388
389 InitializeResetTimeFor(mapid, difficulty, newresettime);
390 } while (result->NextRow());
391 }
392
393 // calculate new global reset times for expired instances and those that have never been reset yet
394 // add the global reset times to the priority queue
395 for (MapDifficultyMap::const_iterator itr = sMapDifficultyMap.begin(); itr != sMapDifficultyMap.end(); ++itr)
396 {
397 uint32 map_diff_pair = itr->first;
398 uint32 mapid = PAIR32_LOPART(map_diff_pair);
399 Difficulty difficulty = Difficulty(PAIR32_HIPART(map_diff_pair));
400 MapDifficulty const* mapDiff = &itr->second;
401 if (!mapDiff->resetTime)
402 continue;
403
404 // the reset_delay must be at least one day
405 uint32 period = uint32(((mapDiff->resetTime * sWorld->getRate(RATE_INSTANCE_RESET_TIME))/DAY) * DAY);
406 if (period < DAY)
407 period = DAY;
408
409 time_t t = GetResetTimeFor(mapid, difficulty);
410 if (!t)
411 {
412 // initialize the reset time
413 t = today + period + diff;
414 CharacterDatabase.DirectPExecute("INSERT INTO instance_reset VALUES ('%u', '%u', '%u')", mapid, difficulty, (uint32)t);
415 }
416
417 if (t < now)
418 {
419 // assume that expired instances have already been cleaned
420 // calculate the next reset time
421 t = (t / DAY) * DAY;
422 t += ((today - t) / period + 1) * period + diff;
423 CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '" UI64FMTD "' WHERE mapid = '%u' AND difficulty= '%u'", (uint64)t, mapid, difficulty);
424 }
425
426 InitializeResetTimeFor(mapid, difficulty, t);
427
428 // schedule the global reset/warning
429 uint8 type;
430 for (type = 1; type < 4; ++type)
431 if (t - ResetTimeDelay[type-1] > now)
432 break;
433
434 ScheduleReset(true, t - ResetTimeDelay[type-1], InstResetEvent(type, mapid, difficulty, 0));
435
436 ResetTimeMapDiffInstancesBounds range = mapDiffResetInstances.equal_range(map_diff_pair);
437 for (; range.first != range.second; ++range.first)
438 ScheduleReset(true, t - ResetTimeDelay[type-1], InstResetEvent(type, mapid, difficulty, range.first->second));
439 }
440}

442time_t InstanceSaveManager::GetSubsequentResetTime(uint32 mapid, Difficulty difficulty, time_t resetTime) const
443{
444 MapDifficulty const* mapDiff = GetMapDifficultyData(mapid, difficulty);
445 if (!mapDiff || !mapDiff->resetTime)
446 {
447 TC_LOG_ERROR("misc", "InstanceSaveManager::GetSubsequentResetTime: not valid difficulty or no reset delay for map %u", mapid);
448 return 0;
449 }
450
451 time_t diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
452 time_t period = uint32(((mapDiff->resetTime * sWorld->getRate(RATE_INSTANCE_RESET_TIME)) / DAY) * DAY);
453 if (period < DAY)
454 period = DAY;
455
456 return ((resetTime + MINUTE) / DAY * DAY) + period + diff;
457}

459void InstanceSaveManager::ScheduleReset(bool add, time_t time, InstResetEvent event)
460{
461 if (!add)
462 {
463 // find the event in the queue and remove it
464 ResetTimeQueue::iterator itr;
465 std::pair<ResetTimeQueue::iterator, ResetTimeQueue::iterator> range;
466 range = m_resetTimeQueue.equal_range(time);
467 for (itr = range.first; itr != range.second; ++itr)
468 {
469 if (itr->second == event)
470 {
471 m_resetTimeQueue.erase(itr);
472 return;
473 }
474 }
475
476 // in case the reset time changed (should happen very rarely), we search the whole queue
477 if (itr == range.second)
478 {
479 for (itr = m_resetTimeQueue.begin(); itr != m_resetTimeQueue.end(); ++itr)
480 {
481 if (itr->second == event)
482 {
483 m_resetTimeQueue.erase(itr);
484 return;
485 }
486 }
487
488 if (itr == m_resetTimeQueue.end())
489 TC_LOG_ERROR("misc", "InstanceSaveManager::ScheduleReset: cannot cancel the reset, the event(%d, %d, %d) was not found!", event.type, event.mapid, event.instanceId);
490 }
491 }
492 else
493 m_resetTimeQueue.insert(std::pair<time_t, InstResetEvent>(time, event));
494}

496void InstanceSaveManager::ForceGlobalReset(uint32 mapId, Difficulty difficulty)
497{
498 if (!GetDownscaledMapDifficultyData(mapId, difficulty))
499 return;
500 // remove currently scheduled reset times
501 ScheduleReset(false, 0, InstResetEvent(1, mapId, difficulty, 0));
502 ScheduleReset(false, 0, InstResetEvent(4, mapId, difficulty, 0));
503 // force global reset on the instance
504 _ResetOrWarnAll(mapId, difficulty, false, time(nullptr));
505}

507void InstanceSaveManager::Update()
508{
509 time_t now = time(NULL);
510 time_t t;
511
512 while (!m_resetTimeQueue.empty())
513 {
514 t = m_resetTimeQueue.begin()->first;
515 if (t >= now)
516 break;
517
518 InstResetEvent &event = m_resetTimeQueue.begin()->second;
519 if (event.type == 0)
520 {
521 // for individual normal instances, max creature respawn + X hours
522 _ResetInstance(event.mapid, event.instanceId);
523 m_resetTimeQueue.erase(m_resetTimeQueue.begin());
524 }
525 else
526 {
527 // global reset/warning for a certain map
528 time_t resetTime = GetResetTimeFor(event.mapid, event.difficulty);
529 _ResetOrWarnAll(event.mapid, event.difficulty, event.type != 4, resetTime);
530 if (event.type != 4)
531 {
532 // schedule the next warning/reset
533 ++event.type;
534 ScheduleReset(true, resetTime - ResetTimeDelay[event.type-1], event);
535 }
536 m_resetTimeQueue.erase(m_resetTimeQueue.begin());
537 }
538 }

541void InstanceSaveManager::_ResetSave(InstanceSaveHashMap::iterator &itr)
542{
543 // unbind all players bound to the instance
544 // do not allow UnbindInstance to automatically unload the InstanceSaves
545 lock_instLists = true;
546
547 bool shouldDelete = true;
548 InstanceSave::PlayerListType &pList = itr->second->m_playerList;
549 std::vector<Player*> temp; // list of expired binds that should be unbound
550 for (Player* player : pList)
551 {
552 if (InstancePlayerBind* bind = player->GetBoundInstance(itr->second->GetMapId(), itr->second->GetDifficulty()))
553 {
554 ASSERT(bind->save == itr->second);
555 if (bind->perm && bind->extendState) // permanent and not already expired
556 {
557 // actual promotion in DB already happened in caller
558 bind->extendState = bind->extendState == EXTEND_STATE_EXTENDED ? EXTEND_STATE_NORMAL : EXTEND_STATE_EXPIRED;
559 shouldDelete = false;
560 continue;
561 }
562 }
563 temp.push_back(player);
564 }
565 for (Player* player : temp)
566 {
567 player->UnbindInstance(itr->second->GetMapId(), itr->second->GetDifficulty(), true);
568 }
569
570 InstanceSave::GroupListType &gList = itr->second->m_groupList;
571 while (!gList.empty())
572 {
573 Group* group = *(gList.begin());
574 group->UnbindInstance(itr->second->GetMapId(), itr->second->GetDifficulty(), true);
575 }
576
577 if (shouldDelete)
578 {
579 delete itr->second;
580 itr = m_instanceSaveById.erase(itr);
581 }
582 else
583 ++itr;
584
585 lock_instLists = false;
586}

588void InstanceSaveManager::_ResetInstance(uint32 mapid, uint32 instanceId)
589{
590 TC_LOG_DEBUG("maps", "InstanceSaveMgr::_ResetInstance %u, %u", mapid, instanceId);
591 Map const* map = sMapMgr->CreateBaseMap(mapid);
592 if (!map->Instanceable())
593 return;
594
595 InstanceSaveHashMap::iterator itr = m_instanceSaveById.find(instanceId);
596 if (itr != m_instanceSaveById.end())
597 _ResetSave(itr);
598
599 DeleteInstanceFromDB(instanceId); // even if save not loaded
600
601 Map* iMap = ((MapInstanced*)map)->FindInstanceMap(instanceId);
602
603 if (iMap && iMap->IsDungeon())
604 ((InstanceMap*)iMap)->Reset(INSTANCE_RESET_RESPAWN_DELAY);
605
606 if (iMap)
607 {
608 iMap->DeleteRespawnTimes();
609 iMap->DeleteCorpseData();
610 }
611 else
612 Map::DeleteRespawnTimesInDB(mapid, instanceId);
613
614 // Free up the instance id and allow it to be reused
615 sMapMgr->FreeInstanceId(instanceId);
616}

618void InstanceSaveManager::_ResetOrWarnAll(uint32 mapid, Difficulty difficulty, bool warn, time_t resetTime)
619{
620 // global reset for all instances of the given map
621 MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
622 if (!mapEntry->Instanceable())
623 return;
624 TC_LOG_DEBUG("misc", "InstanceSaveManager::ResetOrWarnAll: Processing map %s (%u) on difficulty %u (warn? %u)", mapEntry->name[0], mapid, uint8(difficulty), warn);
625
626 time_t now = time(NULL);
627
628 if (!warn)
629 {
630 // calculate the next reset time
631 time_t next_reset = GetSubsequentResetTime(mapid, difficulty, resetTime);
632 if (!next_reset)
633 return;
634
635 // delete/promote instance binds from the DB, even if not loaded
636 SQLTransaction trans = CharacterDatabase.BeginTransaction();
637
638 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_EXPIRED_CHAR_INSTANCE_BY_MAP_DIFF);
639 stmt->setUInt16(0, uint16(mapid));
640 stmt->setUInt8(1, uint8(difficulty));
641 trans->Append(stmt);
642
643 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_GROUP_INSTANCE_BY_MAP_DIFF);
644 stmt->setUInt16(0, uint16(mapid));
645 stmt->setUInt8(1, uint8(difficulty));
646 trans->Append(stmt);
647
648 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_EXPIRED_INSTANCE_BY_MAP_DIFF);
649 stmt->setUInt16(0, uint16(mapid));
650 stmt->setUInt8(1, uint8(difficulty));
651 trans->Append(stmt);
652
653 stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_EXPIRE_CHAR_INSTANCE_BY_MAP_DIFF);
654 stmt->setUInt16(0, uint16(mapid));
655 stmt->setUInt8(1, uint8(difficulty));
656 trans->Append(stmt);
657
658 CharacterDatabase.CommitTransaction(trans);
659
660 // promote loaded binds to instances of the given map
661 for (InstanceSaveHashMap::iterator itr = m_instanceSaveById.begin(); itr != m_instanceSaveById.end();)
662 {
663 if (itr->second->GetMapId() == mapid && itr->second->GetDifficulty() == difficulty)
664 _ResetSave(itr);
665 else
666 ++itr;
667 }
668
669 SetResetTimeFor(mapid, difficulty, next_reset);
670 ScheduleReset(true, time_t(next_reset-3600), InstResetEvent(1, mapid, difficulty, 0));
671
672 // Update it in the DB
673 stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_GLOBAL_INSTANCE_RESETTIME);
674
675 stmt->setUInt32(0, next_reset);
676 stmt->setUInt16(1, uint16(mapid));
677 stmt->setUInt8(2, uint8(difficulty));
678
679 CharacterDatabase.Execute(stmt);
680 }
681
682 // note: this isn't fast but it's meant to be executed very rarely
683 Map const* map = sMapMgr->CreateBaseMap(mapid); // _not_ include difficulty
684 MapInstanced::InstancedMaps &instMaps = ((MapInstanced*)map)->GetInstancedMaps();
685 MapInstanced::InstancedMaps::iterator mitr;
686 uint32 timeLeft;
687
688 for (mitr = instMaps.begin(); mitr != instMaps.end(); ++mitr)
689 {
690 Map* map2 = mitr->second;
691 if (!map2->IsDungeon())
692 continue;
693
694 if (warn)
695 {
696 if (now >= resetTime)
697 timeLeft = 0;
698 else
699 timeLeft = uint32(resetTime - now);
700
701 ((InstanceMap*)map2)->SendResetWarnings(timeLeft);
702 }
703 else
704 ((InstanceMap*)map2)->Reset(INSTANCE_RESET_GLOBAL);
705 }
706
707 /// @todo delete creature/gameobject respawn times even if the maps are not loaded
708}

29 UnitMoveType SelectSpeedType(uint32 moveFlags)
30 {
31 if (moveFlags & MOVEMENTFLAG_FLYING)
32 {
33 if (moveFlags & MOVEMENTFLAG_BACKWARD /*&& speed_obj.flight >= speed_obj.flight_back*/)
34 return MOVE_FLIGHT_BACK;
35 else
36 return MOVE_FLIGHT;
37 }
38 else if (moveFlags & MOVEMENTFLAG_SWIMMING)
39 {
40 if (moveFlags & MOVEMENTFLAG_BACKWARD /*&& speed_obj.swim >= speed_obj.swim_back*/)
41 return MOVE_SWIM_BACK;
42 else
43 return MOVE_SWIM;
44 }
45 else if (moveFlags & MOVEMENTFLAG_WALKING)
46 {
47 //if (speed_obj.run > speed_obj.walk)
48 return MOVE_WALK;
49 }
50 else if (moveFlags & MOVEMENTFLAG_BACKWARD /*&& speed_obj.run >= speed_obj.run_back*/)
51 return MOVE_RUN_BACK;
52
53 // Flying creatures use MOVEMENTFLAG_CAN_FLY or MOVEMENTFLAG_DISABLE_GRAVITY
54 // Run speed is their default flight speed.
55 return MOVE_RUN;
56 }

58 int32 MoveSplineInit::Launch()
59 {
60 MoveSpline& move_spline = *unit->movespline;
61
62 // Elevators also use MOVEMENTFLAG_ONTRANSPORT but we do not keep track of their position changes (movementInfo.transport.guid is 0 in that case)
63 bool transport = unit->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT) && unit->GetTransGUID();
64 Location real_position;
65 // there is a big chance that current position is unknown if current state is not finalized, need compute it
66 // this also allows CalculatePath spline position and update map position in much greater intervals
67 // Don't compute for transport movement if the unit is in a motion between two transports
68 if (!move_spline.Finalized() && move_spline.onTransport == transport)
69 real_position = move_spline.ComputePosition();
70 else
71 {
72 Position const* pos;
73 if (!transport)
74 pos = unit;
75 else
76 pos = &unit->m_movementInfo.transport.pos;
77
78 real_position.x = pos->GetPositionX();
79 real_position.y = pos->GetPositionY();
80 real_position.z = pos->GetPositionZ();
81 real_position.orientation = unit->GetOrientation();
82 }
83
84 // should i do the things that user should do? - no.
85 if (args.path.empty())
86 return 0;
87
88 // corrent first vertex
89 args.path[0] = real_position;
90 args.initialOrientation = real_position.orientation;
91 move_spline.onTransport = transport;
92
93 uint32 moveFlags = unit->m_movementInfo.GetMovementFlags();
94 moveFlags |= (MOVEMENTFLAG_SPLINE_ENABLED|MOVEMENTFLAG_FORWARD);
95
96 if (moveFlags & MOVEMENTFLAG_ROOT)
97 moveFlags &= ~MOVEMENTFLAG_MASK_MOVING;
98
99 if (!args.HasVelocity)
100 {
101 // If spline is initialized with SetWalk method it only means we need to select
102 // walk move speed for it but not add walk flag to unit
103 uint32 moveFlagsForSpeed = moveFlags;
104 if (args.flags.walkmode)
105 moveFlagsForSpeed |= MOVEMENTFLAG_WALKING;
106 else
107 moveFlagsForSpeed &= ~MOVEMENTFLAG_WALKING;
108
109 args.velocity = unit->GetSpeed(SelectSpeedType(moveFlagsForSpeed));
110 }
111
112 if (!args.Validate(unit))
113 return 0;
114
115 unit->m_movementInfo.SetMovementFlags(moveFlags);
116 move_spline.Initialize(args);
117
118 WorldPacket data(SMSG_MONSTER_MOVE, 64);
119 data << unit->GetPackGUID();
120 if (transport)
121 {
122 data.SetOpcode(SMSG_MONSTER_MOVE_TRANSPORT);
123 data << unit->GetTransGUID().WriteAsPacked();
124 data << int8(unit->GetTransSeat());
125 }
126
127 PacketBuilder::WriteMonsterMove(move_spline, data);
128 unit->SendMessageToSet(&data, true);
129
130 return move_spline.Duration();
131 }

133 void MoveSplineInit::Stop()
134 {
135 MoveSpline& move_spline = *unit->movespline;
136
137 // No need to stop if we are not moving
138 if (move_spline.Finalized())
139 return;
140
141 bool transport = unit->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT) && unit->GetTransGUID();
142 Location loc;
143 if (move_spline.onTransport == transport)
144 loc = move_spline.ComputePosition();
145 else
146 {
147 Position const* pos;
148 if (!transport)
149 pos = unit;
150 else
151 pos = &unit->m_movementInfo.transport.pos;
152
153 loc.x = pos->GetPositionX();
154 loc.y = pos->GetPositionY();
155 loc.z = pos->GetPositionZ();
156 loc.orientation = unit->GetOrientation();
157 }
158
159 args.flags = MoveSplineFlag::Done;
160 unit->m_movementInfo.RemoveMovementFlag(MOVEMENTFLAG_FORWARD | MOVEMENTFLAG_SPLINE_ENABLED);
161 move_spline.onTransport = transport;
162 move_spline.Initialize(args);
163
164 WorldPacket data(SMSG_MONSTER_MOVE, 64);
165 data << unit->GetPackGUID();
166 if (transport)
167 {
168 data.SetOpcode(SMSG_MONSTER_MOVE_TRANSPORT);
169 data << unit->GetTransGUID().WriteAsPacked();
170 data << int8(unit->GetTransSeat());
171 }
172
173 PacketBuilder::WriteStopMovement(loc, args.splineId, data);
174 unit->SendMessageToSet(&data, true);
175 }

177 MoveSplineInit::MoveSplineInit(Unit* m) : unit(m)
178 {
179 args.splineId = splineIdGen.NewId();
180 // Elevators also use MOVEMENTFLAG_ONTRANSPORT but we do not keep track of their position changes
181 args.TransformForTransport = unit->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT) && unit->GetTransGUID();
182 // mix existing state into new
183 args.flags.walkmode = unit->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_WALKING);
184 args.flags.flying = unit->m_movementInfo.HasMovementFlag((MovementFlags)(MOVEMENTFLAG_CAN_FLY | MOVEMENTFLAG_DISABLE_GRAVITY));
185 }

193 void MoveSplineInit::SetFacing(float angle)
194 {
195 if (args.TransformForTransport)
196 {
197 if (Unit* vehicle = unit->GetVehicleBase())
198 angle -= vehicle->GetOrientation();
199 else if (Transport* transport = unit->GetTransport())
200 angle -= transport->GetOrientation();
201 }
202
203 args.facing.angle = G3D::wrap(angle, 0.f, (float)G3D::twoPi());
204 args.flags.EnableFacingAngle();
205 }

207 void MoveSplineInit::MoveTo(const Vector3& dest, bool generatePath, bool forceDestination)
208 {
209 if (generatePath)
210 {
211 PathGenerator path(unit);
212 bool result = path.CalculatePath(dest.x, dest.y, dest.z, forceDestination);
213 if (result && !(path.GetPathType() & PATHFIND_NOPATH))
214 {
215 MovebyPath(path.GetPath());
216 return;
217 }
218 }
219
220 args.path_Idx_offset = 0;
221 args.path.resize(2);
222 TransportPathTransform transform(unit, args.TransformForTransport);
223 args.path[1] = transform(dest);
224 }

227 {
228 if (_transformForTransport)
229 if (TransportBase* transport = _owner->GetDirectTransport())

231
232 return input;

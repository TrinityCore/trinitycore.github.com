27 struct Location : public Vector3
28 {
29 Location() : orientation(0) { }
30 Location(float x, float y, float z, float o) : Vector3(x, y, z), orientation(o) { }

32 Location(const Vector3& v, float o) : Vector3(v), orientation(o) { }

37 // MoveSpline represents smooth catmullrom or linear curve and point that moves belong it
38 // curve can be cyclic - in this case movement will be cyclic
39 // point can have vertical acceleration motion componemt(used in fall, parabolic movement)
40 class MoveSpline
41 {
42 public:
43 typedef Spline<int32> MySpline;
44 enum UpdateResult
45 {
46 Result_None = 0x01,
47 Result_Arrived = 0x02,
48 Result_NextCycle = 0x04,
49 Result_NextSegment = 0x08
50 };
51 friend class PacketBuilder;
52
53 protected:
54 MySpline spline;
55
56 FacingInfo facing;
57
58 uint32 m_Id;
59
60 MoveSplineFlag splineflags;
61
62 int32 time_passed;
63 // currently duration mods are unused, but its _currently_
64 //float duration_mod;
65 //float duration_mod_next;
66 float vertical_acceleration;
67 float initialOrientation;
68 int32 effect_start_time;
69 int32 point_Idx;
70 int32 point_Idx_offset;
71
72 void init_spline(const MoveSplineInitArgs& args);
73
74 protected:
75 MySpline::ControlArray const& getPath() const { return spline.getPoints(); }
76 void computeParabolicElevation(float& el) const;
77 void computeFallElevation(float& el) const;
78
79 UpdateResult _updateState(int32& ms_time_diff);
80 int32 next_timestamp() const { return spline.length(point_Idx + 1); }
81 int32 segment_time_elapsed() const { return next_timestamp() - time_passed; }
82 int32 timeElapsed() const { return Duration() - time_passed; }
83 int32 timePassed() const { return time_passed; }
84
85 public:
86 int32 Duration() const { return spline.length(); }
87 MySpline const& _Spline() const { return spline; }
88 int32 _currentSplineIdx() const { return point_Idx; }
89 void _Finalize();
90 void _Interrupt() { splineflags.done = true; }
91
92 public:
93 void Initialize(const MoveSplineInitArgs&);
94 bool Initialized() const { return !spline.empty(); }
95
96 MoveSpline();
97
98 template<class UpdateHandler>
99 void updateState(int32 difftime, UpdateHandler& handler)
100 {
101 ASSERT(Initialized());
102 do
103 handler(_updateState(difftime));
104 while (difftime > 0);
105 }
106
107 void updateState(int32 difftime)
108 {
109 ASSERT(Initialized());
110 do _updateState(difftime);
111 while (difftime > 0);
112 }
113
114 Location ComputePosition() const;
115
116 uint32 GetId() const { return m_Id; }
117 bool Finalized() const { return splineflags.done; }
118 bool isCyclic() const { return splineflags.cyclic; }
119 bool isFalling() const { return splineflags.falling; }
120 Vector3 FinalDestination() const { return Initialized() ? spline.getPoint(spline.last()) : Vector3(); }
121 Vector3 CurrentDestination() const { return Initialized() ? spline.getPoint(point_Idx + 1) : Vector3(); }
122 int32 currentPathIdx() const;
123
124 bool onTransport;
125 std::string ToString() const;
126 };
127}
128#endif // TRINITYSERVER_MOVEPLINE_H

48 template<class T, T limit>
49 class counter
50 {
51 public:
52 counter() { init(); }
53
54 void Increase()
55 {
56 if (m_counter == limit)
57 init();
58 else
59 ++m_counter;
60 }
61
62 T NewId() { Increase(); return m_counter; }
63 T getCurrent() const { return m_counter; }
64
65 private:
66 void init() { m_counter = 0; }
67 T m_counter;
68 };
69
70 typedef counter<uint32, 0xFFFFFFFF> UInt32Counter;

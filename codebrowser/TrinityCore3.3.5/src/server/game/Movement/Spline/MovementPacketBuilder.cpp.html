36 enum MonsterMoveType
37 {
38 MonsterMoveNormal = 0,
39 MonsterMoveStop = 1,
40 MonsterMoveFacingSpot = 2,
41 MonsterMoveFacingTarget = 3,
42 MonsterMoveFacingAngle = 4
43 };
44
45 void PacketBuilder::WriteCommonMonsterMovePart(const MoveSpline& move_spline, ByteBuffer& data)
46 {
47 MoveSplineFlag splineflags = move_spline.splineflags;
48
49 data << uint8(0); // sets/unsets MOVEMENTFLAG2_UNK7 (0x40)
50 data << move_spline.spline.getPoint(move_spline.spline.first());
51 data << move_spline.GetId();
52
53 switch (splineflags & MoveSplineFlag::Mask_Final_Facing)
54 {
55 case MoveSplineFlag::Final_Target:
56 data << uint8(MonsterMoveFacingTarget);
57 data << move_spline.facing.target;
58 break;
59 case MoveSplineFlag::Final_Angle:
60 data << uint8(MonsterMoveFacingAngle);
61 data << move_spline.facing.angle;
62 break;
63 case MoveSplineFlag::Final_Point:
64 data << uint8(MonsterMoveFacingSpot);
65 data << move_spline.facing.f.x << move_spline.facing.f.y << move_spline.facing.f.z;
66 break;
67 default:
68 data << uint8(MonsterMoveNormal);
69 break;
70 }
71
72 // add fake Enter_Cycle flag - needed for client-side cyclic movement (client will erase first spline vertex after first cycle done)
73 splineflags.enter_cycle = move_spline.isCyclic();
74 data << uint32(splineflags & uint32(~MoveSplineFlag::Mask_No_Monster_Move));
75
76 if (splineflags.animation)
77 {
78 data << splineflags.getAnimationId();
79 data << move_spline.effect_start_time;
80 }
81
82 data << move_spline.Duration();
83
84 if (splineflags.parabolic)
85 {
86 data << move_spline.vertical_acceleration;
87 data << move_spline.effect_start_time;
88 }
89 }

91 void PacketBuilder::WriteStopMovement(Vector3 const& pos, uint32 splineId, ByteBuffer& data)
92 {
93 data << uint8(0); // sets/unsets MOVEMENTFLAG2_UNK7 (0x40)
94 data << pos;
95 data << splineId;
96 data << uint8(MonsterMoveStop);
97 }

99 void WriteLinearPath(const Spline<int32>& spline, ByteBuffer& data)
100 {
101 uint32 last_idx = spline.getPointCount() - 3;
102 const Vector3 * real_path = &spline.getPoint(1);
103
104 data << last_idx;
105 data << real_path[last_idx]; // destination
106 if (last_idx > 1)
107 {
108 Vector3 middle = (real_path[0] + real_path[last_idx]) / 2.f;
109 Vector3 offset;
110 // first and last points already appended
111 for (uint32 i = 1; i < last_idx; ++i)
112 {
113 offset = middle - real_path[i];
114 data.appendPackXYZ(offset.x, offset.y, offset.z);
115 }
116 }
117 }

119 void WriteCatmullRomPath(const Spline<int32>& spline, ByteBuffer& data)
120 {
121 uint32 count = spline.getPointCount() - 3;
122 data << count;
123 data.append<Vector3>(&spline.getPoint(2), count);
124 }
125
126 void WriteCatmullRomCyclicPath(const Spline<int32>& spline, ByteBuffer& data)
127 {
128 uint32 count = spline.getPointCount() - 3;
129 data << uint32(count + 1);
130 data << spline.getPoint(1); // fake point, client will erase it from the spline after first cycle done
131 data.append<Vector3>(&spline.getPoint(1), count);
132 }

134 void PacketBuilder::WriteMonsterMove(const MoveSpline& move_spline, ByteBuffer& data)
135 {
136 WriteCommonMonsterMovePart(move_spline, data);
137
138 const Spline<int32>& spline = move_spline.spline;
139 MoveSplineFlag splineflags = move_spline.splineflags;
140 if (splineflags & MoveSplineFlag::Mask_CatmullRom)
141 {
142 if (splineflags.cyclic)
143 WriteCatmullRomCyclicPath(spline, data);
144 else
145 WriteCatmullRomPath(spline, data);
146 }
147 else
148 WriteLinearPath(spline, data);
149 }

151 void PacketBuilder::WriteCreate(const MoveSpline& move_spline, ByteBuffer& data)
152 {
153 //WriteClientStatus(mov, data);
154 //data.append<float>(&mov.m_float_values[SpeedWalk], SpeedMaxCount);
155 //if (mov.SplineEnabled())
156 {
157 MoveSplineFlag const& splineFlags = move_spline.splineflags;
158
159 data << splineFlags.raw();
160
161 if (splineFlags.final_angle)
162 {
163 data << move_spline.facing.angle;
164 }
165 else if (splineFlags.final_target)
166 {
167 data << move_spline.facing.target;
168 }
169 else if (splineFlags.final_point)
170 {
171 data << move_spline.facing.f.x << move_spline.facing.f.y << move_spline.facing.f.z;
172 }
173
174 data << move_spline.timePassed();
175 data << move_spline.Duration();
176 data << move_spline.GetId();
177
178 data << float(1.f); // splineInfo.duration_mod; added in 3.1
179 data << float(1.f); // splineInfo.duration_mod_next; added in 3.1
180
181 data << move_spline.vertical_acceleration; // added in 3.1
182 data << move_spline.effect_start_time; // added in 3.1
183
184 uint32 nodes = move_spline.getPath().size();
185 data << nodes;
186 data.append<Vector3>(&move_spline.getPath()[0], nodes);
187 data << uint8(move_spline.spline.mode()); // added in 3.1
188 data << (move_spline.isCyclic() ? Vector3::zero() : move_spline.FinalDestination());
189 }

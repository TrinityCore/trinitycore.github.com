36template<class T, class P>
37class PathMovementBase

39 public:

50template<class T>
51class WaypointMovementGenerator;
52
53template<>
54class WaypointMovementGenerator<Creature> : public MovementGeneratorMedium< Creature, WaypointMovementGenerator<Creature> >,
55 public PathMovementBase<Creature, WaypointPath const*>
56{
57 public:
58 WaypointMovementGenerator(uint32 _path_id = 0, bool _repeating = true)
59 : i_nextMoveTime(0), m_isArrivalDone(false), path_id(_path_id), repeating(_repeating) { }
60 ~WaypointMovementGenerator() { i_path = NULL; }
61 void DoInitialize(Creature*);
62 void DoFinalize(Creature*);
63 void DoReset(Creature*);
64 bool DoUpdate(Creature*, uint32 diff);
65
66 void MovementInform(Creature*);
67
68 MovementGeneratorType GetMovementGeneratorType() const override { return WAYPOINT_MOTION_TYPE; }
69
70 // now path movement implmementation
71 void LoadPath(Creature*);
72
73 bool GetResetPos(Creature*, float& x, float& y, float& z);
74
75 private:
76
77 void Stop(int32 time) { i_nextMoveTime.Reset(time);}
78
79 bool Stopped() { return !i_nextMoveTime.Passed();}
80
81 bool CanMove(int32 diff)
82 {
83 i_nextMoveTime.Update(diff);
84 return i_nextMoveTime.Passed();
85 }
86
87 void OnArrived(Creature*);
88 bool StartMove(Creature*);
89
90 void StartMoveNow(Creature* creature)
91 {
92 i_nextMoveTime.Reset(0);
93 StartMove(creature);
94 }
95
96 TimeTrackerSmall i_nextMoveTime;
97 bool m_isArrivalDone;
98 uint32 path_id;
99 bool repeating;
100};

102/** FlightPathMovementGenerator generates movement of the player for the paths
103 * and hence generates ground and activities for the player.
104 */
105class FlightPathMovementGenerator : public MovementGeneratorMedium< Player, FlightPathMovementGenerator >,
106 public PathMovementBase<Player, TaxiPathNodeList>
107{
108 public:
109 explicit FlightPathMovementGenerator(uint32 startNode = 0)
110 {
111 i_currentNode = startNode;
112 _endGridX = 0.0f;
113 _endGridY = 0.0f;
114 _endMapId = 0;
115 _preloadTargetNode = 0;
116 }
117 void LoadPath(Player* player);
118 void DoInitialize(Player*);
119 void DoReset(Player*);
120 void DoFinalize(Player*);
121 bool DoUpdate(Player*, uint32);
122 MovementGeneratorType GetMovementGeneratorType() const override { return FLIGHT_MOTION_TYPE; }
123
124 TaxiPathNodeList const& GetPath() { return i_path; }
125 uint32 GetPathAtMapEnd() const;
126 bool HasArrived() const { return (i_currentNode >= i_path.size()); }
127 void SetCurrentNodeAfterTeleport();
128 void SkipCurrentNode() { ++i_currentNode; }
129 void DoEventIfAny(Player* player, TaxiPathNodeEntry const* node, bool departure);
130
131 bool GetResetPos(Player*, float& x, float& y, float& z);
132
133 void InitEndGridInfo();
134 void PreloadEndGrid();
135
136 private:
137
138 float _endGridX; //! X coord of last node location
139 float _endGridY; //! Y coord of last node location
140 uint32 _endMapId; //! map Id of last node location
141 uint32 _preloadTargetNode; //! node index where preloading starts

145 uint32 PathIndex;
146 int32 Cost;
147 };
148
149 std::deque<TaxiNodeChangeInfo> _pointsForPathSwitch; //! node indexes and costs where TaxiPath changes
150};

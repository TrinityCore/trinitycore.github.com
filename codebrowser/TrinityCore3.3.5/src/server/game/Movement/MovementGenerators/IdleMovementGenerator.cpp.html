25// StopMoving is needed to make unit stop if its last movement generator expires
26// But it should not be sent otherwise there are many redundent packets
27void IdleMovementGenerator::Initialize(Unit* owner)
28{
29 Reset(owner);
30}

32void IdleMovementGenerator::Reset(Unit* owner)
33{
34 if (!owner->IsStopped())
35 owner->StopMoving();

38void RotateMovementGenerator::Initialize(Unit* owner)
39{
40 if (!owner->IsStopped())
41 owner->StopMoving();
42
43 if (owner->GetVictim())
44 owner->SetInFront(owner->GetVictim());
45
46 owner->AddUnitState(UNIT_STATE_ROTATING);
47
48 owner->AttackStop();
49}
50
51bool RotateMovementGenerator::Update(Unit* owner, uint32 diff)
52{
53 float angle = owner->GetOrientation();
54 if (m_direction == ROTATE_DIRECTION_LEFT)
55 {
56 angle += (float)diff * static_cast<float>(M_PI * 2) / m_maxDuration;
57 while (angle >= static_cast<float>(M_PI * 2)) angle -= static_cast<float>(M_PI * 2);
58 }
59 else
60 {
61 angle -= (float)diff * static_cast<float>(M_PI * 2) / m_maxDuration;
62 while (angle < 0) angle += static_cast<float>(M_PI * 2);
63 }
64
65 owner->SetFacingTo(angle);
66
67 if (m_duration > diff)
68 m_duration -= diff;
69 else
70 return false;
71
72 return true;
73}

75void RotateMovementGenerator::Finalize(Unit* unit)
76{
77 unit->ClearUnitState(UNIT_STATE_ROTATING);
78 if (unit->GetTypeId() == TYPEID_UNIT)
79 unit->ToCreature()->AI()->MovementInform(ROTATE_MOTION_TYPE, 0);
80}

82void DistractMovementGenerator::Initialize(Unit* owner)
83{
84 // Distracted creatures stand up if not standing
85 if (!owner->IsStandState())
86 owner->SetStandState(UNIT_STAND_STATE_STAND);
87
88 owner->AddUnitState(UNIT_STATE_DISTRACTED);
89}
90
91void DistractMovementGenerator::Finalize(Unit* owner)
92{
93 owner->ClearUnitState(UNIT_STATE_DISTRACTED);
94
95 // If this is a creature, then return orientation to original position (for idle movement creatures)
96 if (owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature())
97 {
98 float angle = owner->ToCreature()->GetHomePosition().GetOrientation();
99 owner->SetFacingTo(angle);
100 }
101}
102
103bool DistractMovementGenerator::Update(Unit* /*owner*/, uint32 time_diff)
104{
105 if (time_diff > m_timer)
106 return false;
107
108 m_timer -= time_diff;
109 return true;
110}

112void AssistanceDistractMovementGenerator::Finalize(Unit* unit)
113{
114 unit->ClearUnitState(UNIT_STATE_DISTRACTED);
115 unit->ToCreature()->SetReactState(REACT_AGGRESSIVE);
116}

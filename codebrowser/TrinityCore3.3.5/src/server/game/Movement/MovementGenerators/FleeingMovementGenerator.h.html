19#ifndef TRINITY_FLEEINGMOVEMENTGENERATOR_H
20#define TRINITY_FLEEINGMOVEMENTGENERATOR_H
21
22#include "MovementGenerator.h"
23
24template<class T>
25class FleeingMovementGenerator : public MovementGeneratorMedium< T, FleeingMovementGenerator<T> >
26{
27 public:
28 FleeingMovementGenerator(ObjectGuid fright) : i_frightGUID(fright), i_nextCheckTime(0) { }
29
30 void DoInitialize(T*);
31 void DoFinalize(T*);
32 void DoReset(T*);
33 bool DoUpdate(T*, uint32);
34
35 MovementGeneratorType GetMovementGeneratorType() const override { return FLEEING_MOTION_TYPE; }
36
37 private:
38 void _setTargetLocation(T*);
39 void _getPoint(T*, float &x, float &y, float &z);
40
41 ObjectGuid i_frightGUID;
42 TimeTracker i_nextCheckTime;
43};
44
45class TimedFleeingMovementGenerator : public FleeingMovementGenerator<Creature>
46{
47 public:
48 TimedFleeingMovementGenerator(ObjectGuid fright, uint32 time) :
49 FleeingMovementGenerator<Creature>(fright),
50 i_totalFleeTime(time) { }
51
52 MovementGeneratorType GetMovementGeneratorType() const override { return TIMED_FLEEING_MOTION_TYPE; }
53 bool Update(Unit*, uint32) override;
54 void Finalize(Unit*) override;
55
56 private:
57 TimeTracker i_totalFleeTime;
58};

19#ifndef TRINITY_WAYPOINTMANAGER_H
20#define TRINITY_WAYPOINTMANAGER_H
21
22#include <vector>
23
24enum WaypointMoveType
25{
26 WAYPOINT_MOVE_TYPE_WALK,
27 WAYPOINT_MOVE_TYPE_RUN,
28 WAYPOINT_MOVE_TYPE_LAND,
29 WAYPOINT_MOVE_TYPE_TAKEOFF,
30
31 WAYPOINT_MOVE_TYPE_MAX
32};
33
34struct WaypointData
35{
36 uint32 id;
37 float x, y, z, orientation;
38 uint32 delay;
39 uint32 event_id;
40 uint32 move_type;
41 uint8 event_chance;
42};
43
44typedef std::vector<WaypointData*> WaypointPath;
45typedef std::unordered_map<uint32, WaypointPath> WaypointPathContainer;
46
47class WaypointMgr
48{
49 public:
50 static WaypointMgr* instance()
51 {
52 static WaypointMgr instance;
53 return &instance;
54 }
55
56 // Attempts to reload a single path from database
57 void ReloadPath(uint32 id);
58
59 // Loads all paths from database, should only run on startup
60 void Load();
61
62 // Returns the path from a given id
63 WaypointPath const* GetPath(uint32 id) const
64 {
65 WaypointPathContainer::const_iterator itr = _waypointStore.find(id);
66 if (itr != _waypointStore.end())
67 return &itr->second;
68
69 return NULL;
70 }
71
72 private:
73 WaypointMgr();
74 ~WaypointMgr();
75
76 WaypointPathContainer _waypointStore;
77};
78
79#define sWaypointMgr WaypointMgr::instance()
80
81#endif

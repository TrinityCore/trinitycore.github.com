34inline bool isStatic(MovementGenerator *mv)
35{
36 return (mv == &si_idleMovement);
37}

39void MotionMaster::Initialize()
40{
41 // clear ALL movement generators (including default)
42 while (!empty())
43 {
44 MovementGenerator *curr = top();
45 pop();
46 if (curr)
47 DirectDelete(curr);
48 }
49
50 InitDefault();
51}

53// set new default movement generator
54void MotionMaster::InitDefault()
55{
56 if (_owner->GetTypeId() == TYPEID_UNIT)
57 {
58 MovementGenerator* movement = FactorySelector::selectMovementGenerator(_owner->ToCreature());
59 Mutate(movement == NULL ? &si_idleMovement : movement, MOTION_SLOT_IDLE);
60 }
61 else
62 {
63 Mutate(&si_idleMovement, MOTION_SLOT_IDLE);
64 }
65}

67MotionMaster::~MotionMaster()
68{
69 // clear ALL movement generators (including default)
70 while (!empty())
71 {
72 MovementGenerator *curr = top();
73 pop();
74 if (curr && !isStatic(curr))
75 delete curr; // Skip finalizing on delete, it might launch new movement
76 }
77}

79void MotionMaster::UpdateMotion(uint32 diff)
80{
81 if (!_owner)
82 return;
83
84 if (_owner->HasUnitState(UNIT_STATE_ROOT | UNIT_STATE_STUNNED)) // what about UNIT_STATE_DISTRACTED? Why is this not included?
85 return;
86
87 ASSERT(!empty());
88
89 _cleanFlag |= MMCF_UPDATE;
90 if (!top()->Update(_owner, diff))
91 {
92 _cleanFlag &= ~MMCF_UPDATE;
93 MovementExpired();
94 }
95 else
96 _cleanFlag &= ~MMCF_UPDATE;
97
98 if (_expList)
99 {
100 for (size_t i = 0; i < _expList->size(); ++i)
101 {
102 MovementGenerator* mg = (*_expList)[i];
103 DirectDelete(mg);
104 }
105
106 delete _expList;
107 _expList = NULL;
108
109 if (empty())
110 Initialize();
111 else if (needInitTop())
112 InitTop();
113 else if (_cleanFlag & MMCF_RESET)
114 top()->Reset(_owner);
115
116 _cleanFlag &= ~MMCF_RESET;
117 }
118
119 // probably not the best place to pu this but im not really sure where else to put it.
120 _owner->UpdateUnderwaterState(_owner->GetMap(), _owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ());
121}

123void MotionMaster::DirectClean(bool reset)
124{
125 while (size() > 1)
126 {
127 MovementGenerator *curr = top();
128 pop();
129 if (curr) DirectDelete(curr);
130 }
131
132 if (empty())
133 return;
134
135 if (needInitTop())
136 InitTop();
137 else if (reset)
138 top()->Reset(_owner);
139}

141void MotionMaster::DelayedClean()

145 MovementGenerator *curr = top();
146 pop();

148 DelayedDelete(curr);
149 }
150}
151
152void MotionMaster::DirectExpire(bool reset)
153{
154 if (size() > 1)
155 {
156 MovementGenerator *curr = top();
157 pop();
158 DirectDelete(curr);
159 }
160
161 while (!empty() && !top())
162 --_top;
163
164 if (empty())
165 Initialize();
166 else if (needInitTop())
167 InitTop();
168 else if (reset)
169 top()->Reset(_owner);
170}
171
172void MotionMaster::DelayedExpire()
173{
174 if (size() > 1)
175 {
176 MovementGenerator *curr = top();
177 pop();
178 DelayedDelete(curr);
179 }
180
181 while (!empty() && !top())
182 --_top;
183}

333void MotionMaster::MoveKnockbackFrom(float srcX, float srcY, float speedXY, float speedZ)
334{
335 //this function may make players fall below map
336 if (_owner->GetTypeId() == TYPEID_PLAYER)
337 return;
338
339 if (speedXY <= 0.1f)
340 return;
341
342 float x, y, z;
343 float moveTimeHalf = speedZ / Movement::gravity;
344 float dist = 2 * moveTimeHalf * speedXY;
345 float max_height = -Movement::computeFallElevation(moveTimeHalf, false, -speedZ);
346
347 _owner->GetNearPoint(_owner, x, y, z, _owner->GetObjectSize(), dist, _owner->GetAngle(srcX, srcY) + float(M_PI));
348
349 Movement::MoveSplineInit init(_owner);
350 init.MoveTo(x, y, z);
351 init.SetParabolic(max_height, 0);
352 init.SetOrientationFixed(true);
353 init.SetVelocity(speedXY);
354 init.Launch();
355 Mutate(new EffectMovementGenerator(0), MOTION_SLOT_CONTROLLED);
356}

391void MotionMaster::MoveCirclePath(float x, float y, float z, float radius, bool clockwise, uint8 stepCount)
392{
393 float step = 2 * float(M_PI) / stepCount * (clockwise ? -1.0f : 1.0f);
394 Position const& pos = { x, y, z, 0.0f };
395 float angle = pos.GetAngle(_owner->GetPositionX(), _owner->GetPositionY());
396
397 Movement::MoveSplineInit init(_owner);
398
399 for (uint8 i = 0; i < stepCount; angle += step, ++i)
400 {
401 G3D::Vector3 point;
402 point.x = x + radius * cosf(angle);
403 point.y = y + radius * sinf(angle);
404
405 if (_owner->IsFlying())
406 point.z = z;
407 else
408 point.z = _owner->GetMap()->GetHeight(_owner->GetPhaseMask(), point.x, point.y, z);
409
410 init.Path().push_back(point);
411 }
412
413 if (_owner->IsFlying())
414 {
415 init.SetFly();
416 init.SetCyclic();
417 init.SetAnimation(Movement::ToFly);
418 }
419 else
420 {
421 init.SetWalk(true);
422 init.SetCyclic();
423 }
424
425 init.Launch();
426}

446void MotionMaster::MoveFall(uint32 id /*=0*/)
447{
448 // use larger distance for vmap height search than in most other cases
449 float tz = _owner->GetMap()->GetHeight(_owner->GetPhaseMask(), _owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ(), true, MAX_FALL_DISTANCE);
450 if (tz <= INVALID_HEIGHT)
451 {
452 TC_LOG_DEBUG("misc", "MotionMaster::MoveFall: unable to retrieve a proper height at map %u (x: %f, y: %f, z: %f).",
453 _owner->GetMap()->GetId(), _owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ());
454 return;
455 }
456
457 // Abort too if the ground is very near
458 if (std::fabs(_owner->GetPositionZ() - tz) < 0.1f)
459 return;
460
461 if (_owner->GetTypeId() == TYPEID_PLAYER)
462 {
463 _owner->AddUnitMovementFlag(MOVEMENTFLAG_FALLING);
464 _owner->m_movementInfo.SetFallTime(0);
465 }
466
467 Movement::MoveSplineInit init(_owner);
468 init.MoveTo(_owner->GetPositionX(), _owner->GetPositionY(), tz, false);
469 init.SetFall();
470 init.Launch();
471 Mutate(new EffectMovementGenerator(id), MOTION_SLOT_CONTROLLED);
472}

604void MotionMaster::Mutate(MovementGenerator *m, MovementSlot slot)
605{
606 if (MovementGenerator *curr = Impl[slot])
607 {
608 Impl[slot] = NULL; // in case a new one is generated in this slot during directdelete
609 if (_top == slot && (_cleanFlag & MMCF_UPDATE))
610 DelayedDelete(curr);
611 else
612 DirectDelete(curr);
613 }
614 else if (_top < slot)
615 {
616 _top = slot;
617 }
618
619 Impl[slot] = m;
620 if (_top > slot)
621 _needInit[slot] = true;
622 else
623 {
624 _needInit[slot] = false;
625 m->Initialize(_owner);
626 }
627}

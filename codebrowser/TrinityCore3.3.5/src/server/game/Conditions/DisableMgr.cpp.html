32 struct DisableData
33 {
34 uint8 flags;
35 std::set<uint32> params[2]; // params0, params1
36 };
37
38 // single disables here with optional data
39 typedef std::map<uint32, DisableData> DisableTypeMap;
40 // global disable map by source
41 typedef std::map<DisableType, DisableTypeMap> DisableMap;
42
43 DisableMap m_DisableMap;
44
45 uint8 MAX_DISABLE_TYPES = 8;
46}

48void LoadDisables()
49{
50 uint32 oldMSTime = getMSTime();
51
52 // reload case
53 for (DisableMap::iterator itr = m_DisableMap.begin(); itr != m_DisableMap.end(); ++itr)
54 itr->second.clear();
55
56 m_DisableMap.clear();
57
58 QueryResult result = WorldDatabase.Query("SELECT sourceType, entry, flags, params_0, params_1 FROM disables");
59
60 uint32 total_count = 0;
61
62 if (!result)
63 {
64 TC_LOG_INFO("server.loading", ">> Loaded 0 disables. DB table `disables` is empty!");
65 return;
66 }
67
68 Field* fields;
69 do
70 {
71 fields = result->Fetch();
72 DisableType type = DisableType(fields[0].GetUInt32());
73 if (type >= MAX_DISABLE_TYPES)
74 {
75 TC_LOG_ERROR("sql.sql", "Invalid type %u specified in `disables` table, skipped.", type);
76 continue;
77 }
78
79 uint32 entry = fields[1].GetUInt32();
80 uint8 flags = fields[2].GetUInt8();
81 std::string params_0 = fields[3].GetString();
82 std::string params_1 = fields[4].GetString();
83
84 DisableData data;
85 data.flags = flags;
86
87 switch (type)
88 {
89 case DISABLE_TYPE_SPELL:
90 if (!(sSpellMgr->GetSpellInfo(entry) || flags & SPELL_DISABLE_DEPRECATED_SPELL))
91 {
92 TC_LOG_ERROR("sql.sql", "Spell entry %u from `disables` doesn't exist in dbc, skipped.", entry);
93 continue;
94 }
95
96 if (!flags || flags > MAX_SPELL_DISABLE_TYPE)
97 {
98 TC_LOG_ERROR("sql.sql", "Disable flags for spell %u are invalid, skipped.", entry);
99 continue;
100 }
101
102 if (flags & SPELL_DISABLE_MAP)
103 {
104 Tokenizer tokens(params_0, ',');
105 for (uint8 i = 0; i < tokens.size(); )
106 data.params[0].insert(atoi(tokens[i++]));
107 }
108
109 if (flags & SPELL_DISABLE_AREA)
110 {
111 Tokenizer tokens(params_1, ',');
112 for (uint8 i = 0; i < tokens.size(); )
113 data.params[1].insert(atoi(tokens[i++]));
114 }
115
116 break;
117 // checked later
118 case DISABLE_TYPE_QUEST:
119 break;
120 case DISABLE_TYPE_MAP:
121 {
122 MapEntry const* mapEntry = sMapStore.LookupEntry(entry);
123 if (!mapEntry)
124 {
125 TC_LOG_ERROR("sql.sql", "Map entry %u from `disables` doesn't exist in dbc, skipped.", entry);
126 continue;
127 }
128 bool isFlagInvalid = false;
129 switch (mapEntry->map_type)
130 {
131 case MAP_COMMON:
132 if (flags)
133 isFlagInvalid = true;
134 break;
135 case MAP_INSTANCE:
136 case MAP_RAID:
137 if (flags & DUNGEON_STATUSFLAG_HEROIC && !GetMapDifficultyData(entry, DUNGEON_DIFFICULTY_HEROIC))
138 flags -= DUNGEON_STATUSFLAG_HEROIC;
139 if (flags & RAID_STATUSFLAG_10MAN_HEROIC && !GetMapDifficultyData(entry, RAID_DIFFICULTY_10MAN_HEROIC))
140 flags -= RAID_STATUSFLAG_10MAN_HEROIC;
141 if (flags & RAID_STATUSFLAG_25MAN_HEROIC && !GetMapDifficultyData(entry, RAID_DIFFICULTY_25MAN_HEROIC))
142 flags -= RAID_STATUSFLAG_25MAN_HEROIC;
143 if (!flags)
144 isFlagInvalid = true;
145 break;
146 case MAP_BATTLEGROUND:
147 case MAP_ARENA:
148 TC_LOG_ERROR("sql.sql", "Battleground map %u specified to be disabled in map case, skipped.", entry);
149 continue;
150 }
151 if (isFlagInvalid)
152 {
153 TC_LOG_ERROR("sql.sql", "Disable flags for map %u are invalid, skipped.", entry);
154 continue;
155 }
156 break;
157 }
158 case DISABLE_TYPE_BATTLEGROUND:
159 if (!sBattlemasterListStore.LookupEntry(entry))
160 {
161 TC_LOG_ERROR("sql.sql", "Battleground entry %u from `disables` doesn't exist in dbc, skipped.", entry);
162 continue;
163 }
164 if (flags)
165 TC_LOG_ERROR("sql.sql", "Disable flags specified for battleground %u, useless data.", entry);
166 break;
167 case DISABLE_TYPE_OUTDOORPVP:
168 if (entry > MAX_OUTDOORPVP_TYPES)
169 {
170 TC_LOG_ERROR("sql.sql", "OutdoorPvPTypes value %u from `disables` is invalid, skipped.", entry);
171 continue;
172 }
173 if (flags)
174 TC_LOG_ERROR("sql.sql", "Disable flags specified for outdoor PvP %u, useless data.", entry);
175 break;
176 case DISABLE_TYPE_ACHIEVEMENT_CRITERIA:
177 if (!sAchievementMgr->GetAchievementCriteria(entry))
178 {
179 TC_LOG_ERROR("sql.sql", "Achievement Criteria entry %u from `disables` doesn't exist in dbc, skipped.", entry);
180 continue;
181 }
182 if (flags)
183 TC_LOG_ERROR("sql.sql", "Disable flags specified for Achievement Criteria %u, useless data.", entry);
184 break;
185 case DISABLE_TYPE_VMAP:
186 {
187 MapEntry const* mapEntry = sMapStore.LookupEntry(entry);
188 if (!mapEntry)
189 {
190 TC_LOG_ERROR("sql.sql", "Map entry %u from `disables` doesn't exist in dbc, skipped.", entry);
191 continue;
192 }
193 switch (mapEntry->map_type)
194 {
195 case MAP_COMMON:
196 if (flags & VMAP::VMAP_DISABLE_AREAFLAG)
197 TC_LOG_INFO("misc", "Areaflag disabled for world map %u.", entry);
198 if (flags & VMAP::VMAP_DISABLE_LIQUIDSTATUS)
199 TC_LOG_INFO("misc", "Liquid status disabled for world map %u.", entry);
200 break;
201 case MAP_INSTANCE:
202 case MAP_RAID:
203 if (flags & VMAP::VMAP_DISABLE_HEIGHT)
204 TC_LOG_INFO("misc", "Height disabled for instance map %u.", entry);
205 if (flags & VMAP::VMAP_DISABLE_LOS)
206 TC_LOG_INFO("misc", "LoS disabled for instance map %u.", entry);
207 break;
208 case MAP_BATTLEGROUND:
209 if (flags & VMAP::VMAP_DISABLE_HEIGHT)
210 TC_LOG_INFO("misc", "Height disabled for battleground map %u.", entry);
211 if (flags & VMAP::VMAP_DISABLE_LOS)
212 TC_LOG_INFO("misc", "LoS disabled for battleground map %u.", entry);
213 break;
214 case MAP_ARENA:
215 if (flags & VMAP::VMAP_DISABLE_HEIGHT)
216 TC_LOG_INFO("misc", "Height disabled for arena map %u.", entry);
217 if (flags & VMAP::VMAP_DISABLE_LOS)
218 TC_LOG_INFO("misc", "LoS disabled for arena map %u.", entry);
219 break;
220 default:
221 break;
222 }
223 break;
224 }
225 case DISABLE_TYPE_MMAP:
226 {
227 MapEntry const* mapEntry = sMapStore.LookupEntry(entry);
228 if (!mapEntry)
229 {
230 TC_LOG_ERROR("sql.sql", "Map entry %u from `disables` doesn't exist in dbc, skipped.", entry);
231 continue;
232 }
233 switch (mapEntry->map_type)
234 {
235 case MAP_COMMON:
236 TC_LOG_INFO("misc", "Pathfinding disabled for world map %u.", entry);
237 break;
238 case MAP_INSTANCE:
239 case MAP_RAID:
240 TC_LOG_INFO("misc", "Pathfinding disabled for instance map %u.", entry);
241 break;
242 case MAP_BATTLEGROUND:
243 TC_LOG_INFO("misc", "Pathfinding disabled for battleground map %u.", entry);
244 break;
245 case MAP_ARENA:
246 TC_LOG_INFO("misc", "Pathfinding disabled for arena map %u.", entry);
247 break;
248 default:
249 break;
250 }
251 break;
252 }
253 default:
254 break;
255 }
256
257 m_DisableMap[type].insert(DisableTypeMap::value_type(entry, data));
258 ++total_count;
259 }
260 while (result->NextRow());
261
262 TC_LOG_INFO("server.loading", ">> Loaded %u disables in %u ms", total_count, GetMSTimeDiffToNow(oldMSTime));
263}

265void CheckQuestDisables()
266{
267 uint32 oldMSTime = getMSTime();
268
269 uint32 count = m_DisableMap[DISABLE_TYPE_QUEST].size();
270 if (!count)
271 {
272 TC_LOG_INFO("server.loading", ">> Checked 0 quest disables.");
273 return;
274 }
275
276 // check only quests, rest already done at startup
277 for (DisableTypeMap::iterator itr = m_DisableMap[DISABLE_TYPE_QUEST].begin(); itr != m_DisableMap[DISABLE_TYPE_QUEST].end();)
278 {
279 const uint32 entry = itr->first;
280 if (!sObjectMgr->GetQuestTemplate(entry))
281 {
282 TC_LOG_ERROR("sql.sql", "Quest entry %u from `disables` doesn't exist, skipped.", entry);
283 m_DisableMap[DISABLE_TYPE_QUEST].erase(itr++);
284 continue;
285 }
286 if (itr->second.flags)
287 TC_LOG_ERROR("sql.sql", "Disable flags specified for quest %u, useless data.", entry);
288 ++itr;
289 }
290
291 TC_LOG_INFO("server.loading", ">> Checked %u quest disables in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
292}

294bool IsDisabledFor(DisableType type, uint32 entry, Unit const* unit, uint8 flags)
295{
296 ASSERT(type < MAX_DISABLE_TYPES);
297 if (m_DisableMap[type].empty())
298 return false;
299
300 DisableTypeMap::iterator itr = m_DisableMap[type].find(entry);
301 if (itr == m_DisableMap[type].end()) // not disabled
302 return false;
303
304 switch (type)
305 {
306 case DISABLE_TYPE_SPELL:
307 {
308 uint8 spellFlags = itr->second.flags;
309 if (unit)
310 {
311 if ((spellFlags & SPELL_DISABLE_PLAYER && unit->GetTypeId() == TYPEID_PLAYER) ||
312 (unit->GetTypeId() == TYPEID_UNIT && ((unit->IsPet() && spellFlags & SPELL_DISABLE_PET) || spellFlags & SPELL_DISABLE_CREATURE)))
313 {
314 if (spellFlags & SPELL_DISABLE_MAP)
315 {
316 std::set<uint32> const& mapIds = itr->second.params[0];
317 if (mapIds.find(unit->GetMapId()) != mapIds.end())
318 return true; // Spell is disabled on current map
319
320 if (!(spellFlags & SPELL_DISABLE_AREA))
321 return false; // Spell is disabled on another map, but not this one, return false
322
323 // Spell is disabled in an area, but not explicitly our current mapId. Continue processing.
324 }
325
326 if (spellFlags & SPELL_DISABLE_AREA)
327 {
328 std::set<uint32> const& areaIds = itr->second.params[1];
329 if (areaIds.find(unit->GetAreaId()) != areaIds.end())
330 return true; // Spell is disabled in this area
331 return false; // Spell is disabled in another area, but not this one, return false
332 }
333 else
334 return true; // Spell disabled for all maps
335 }
336
337 return false;
338 }
339 else if (spellFlags & SPELL_DISABLE_DEPRECATED_SPELL) // call not from spellcast
340 return true;
341 else if (flags & SPELL_DISABLE_LOS)
342 return (spellFlags & SPELL_DISABLE_LOS) != 0;
343
344 break;
345 }
346 case DISABLE_TYPE_MAP:
347 if (Player const* player = unit->ToPlayer())
348 {
349 MapEntry const* mapEntry = sMapStore.LookupEntry(entry);
350 if (mapEntry->IsDungeon())
351 {
352 uint8 disabledModes = itr->second.flags;
353 Difficulty targetDifficulty = player->GetDifficulty(mapEntry->IsRaid());
354 GetDownscaledMapDifficultyData(entry, targetDifficulty);
355 switch (targetDifficulty)
356 {
357 case DUNGEON_DIFFICULTY_NORMAL:
358 return (disabledModes & DUNGEON_STATUSFLAG_NORMAL) != 0;
359 case DUNGEON_DIFFICULTY_HEROIC:
360 return (disabledModes & DUNGEON_STATUSFLAG_HEROIC) != 0;
361 case RAID_DIFFICULTY_10MAN_HEROIC:
362 return (disabledModes & RAID_STATUSFLAG_10MAN_HEROIC) != 0;
363 case RAID_DIFFICULTY_25MAN_HEROIC:
364 return (disabledModes & RAID_STATUSFLAG_25MAN_HEROIC) != 0;
365 }
366 }
367 else if (mapEntry->map_type == MAP_COMMON)
368 return true;
369 }
370 return false;
371 case DISABLE_TYPE_QUEST:
372 if (!unit)
373 return true;
374 if (Player const* player = unit->ToPlayer())
375 if (player->IsGameMaster())
376 return false;
377 return true;
378 case DISABLE_TYPE_BATTLEGROUND:
379 case DISABLE_TYPE_OUTDOORPVP:
380 case DISABLE_TYPE_ACHIEVEMENT_CRITERIA:
381 case DISABLE_TYPE_MMAP:
382 return true;
383 case DISABLE_TYPE_VMAP:
384 return (flags & itr->second.flags) != 0;
385 }
386
387 return false;
388}

390bool IsVMAPDisabledFor(uint32 entry, uint8 flags)
391{
392 return IsDisabledFor(DISABLE_TYPE_VMAP, entry, NULL, flags);
393}

395bool IsPathfindingEnabled(uint32 mapId)
396{
397 return sWorld->getBoolConfig(CONFIG_ENABLE_MMAPS)
398 && !IsDisabledFor(DISABLE_TYPE_MMAP, mapId, NULL, MMAP_DISABLE_PATHFINDING);
399}

31enum ConditionTypes
32{ // value1 value2 value3

37 CONDITION_ZONEID = 4, // zone_id 0 0 true if in zone_id
38 CONDITION_REPUTATION_RANK = 5, // faction_id rankMask 0 true if has min_rank for faction_id
39 CONDITION_TEAM = 6, // player_team 0, 0 469 - Alliance, 67 - Horde)
40 CONDITION_SKILL = 7, // skill_id skill_value 0 true if has skill_value for skill_id
41 CONDITION_QUESTREWARDED = 8, // quest_id 0 0 true if quest_id was rewarded before
42 CONDITION_QUESTTAKEN = 9, // quest_id 0, 0 true while quest active
43 CONDITION_DRUNKENSTATE = 10, // DrunkenState 0, 0 true if player is drunk enough
44 CONDITION_WORLD_STATE = 11, // index value 0 true if world has the value for the index
45 CONDITION_ACTIVE_EVENT = 12, // event_id 0 0 true if event is active
46 CONDITION_INSTANCE_INFO = 13, // entry data type true if the instance info defined by type (enum InstanceInfo) equals data.
47 CONDITION_QUEST_NONE = 14, // quest_id 0 0 true if doesn't have quest saved
48 CONDITION_CLASS = 15, // class 0 0 true if player's class is equal to class
49 CONDITION_RACE = 16, // race 0 0 true if player's race is equal to race
50 CONDITION_ACHIEVEMENT = 17, // achievement_id 0 0 true if achievement is complete
51 CONDITION_TITLE = 18, // title id 0 0 true if player has title
52 CONDITION_SPAWNMASK = 19, // spawnMask 0 0 true if in spawnMask
53 CONDITION_GENDER = 20, // gender 0 0 true if player's gender is equal to gender
54 CONDITION_UNIT_STATE = 21, // unitState 0 0 true if unit has unitState
55 CONDITION_MAPID = 22, // map_id 0 0 true if in map_id
56 CONDITION_AREAID = 23, // area_id 0 0 true if in area_id
57 CONDITION_CREATURE_TYPE = 24, // cinfo.type 0 0 true if creature_template.type = value1
58 CONDITION_SPELL = 25, // spell_id 0 0 true if player has learned spell
59 CONDITION_PHASEMASK = 26, // phasemask 0 0 true if object is in phasemask
60 CONDITION_LEVEL = 27, // level ComparisonType 0 true if unit's level is equal to param1 (param2 can modify the statement)
61 CONDITION_QUEST_COMPLETE = 28, // quest_id 0 0 true if player has quest_id with all objectives complete, but not yet rewarded
62 CONDITION_NEAR_CREATURE = 29, // creature entry distance dead (0/1) true if there is a creature of entry in range
63 CONDITION_NEAR_GAMEOBJECT = 30, // gameobject entry distance 0 true if there is a gameobject of entry in range
64 CONDITION_OBJECT_ENTRY_GUID = 31, // TypeID entry guid true if object is type TypeID and the entry is 0 or matches entry of the object or matches guid of the object
65 CONDITION_TYPE_MASK = 32, // TypeMask 0 0 true if object is type object's TypeMask matches provided TypeMask
66 CONDITION_RELATION_TO = 33, // ConditionTarget RelationType 0 true if object is in given relation with object specified by ConditionTarget
67 CONDITION_REACTION_TO = 34, // ConditionTarget rankMask 0 true if object's reaction matches rankMask object specified by ConditionTarget
68 CONDITION_DISTANCE_TO = 35, // ConditionTarget distance ComparisonType true if object and ConditionTarget are within distance given by parameters
69 CONDITION_ALIVE = 36, // 0 0 0 true if unit is alive
70 CONDITION_HP_VAL = 37, // hpVal ComparisonType 0 true if unit's hp matches given value
71 CONDITION_HP_PCT = 38, // hpPct ComparisonType 0 true if unit's hp matches given pct
72 CONDITION_REALM_ACHIEVEMENT = 39, // achievement_id 0 0 true if realm achievement is complete
73 CONDITION_IN_WATER = 40, // 0 0 0 true if unit in water
74 CONDITION_TERRAIN_SWAP = 41, // only for 6.x
75 CONDITION_STAND_STATE = 42, // stateType state 0 true if unit matches specified sitstate (0,x: has exactly state x; 1,0: any standing state; 1,1: any sitting state;)
76 CONDITION_MAX = 43 // MAX
77};

79/*! Documentation on implementing a new ConditionSourceType:
80 Step 1: Check for the lowest free ID. Look for CONDITION_SOURCE_TYPE_UNUSED_XX in the enum.
81 Then define the new source type.
82
83 Step 2: Determine and map the parameters for the new condition type.
84
85 Step 3: Add a case block to ConditionMgr::isSourceTypeValid with the new condition type
86 and validate the parameters.
87
88 Step 4: If your condition can be grouped (determined in step 2), add a rule for it in
89 ConditionMgr::CanHaveSourceGroupSet, following the example of the existing types.
90
91 Step 5: Define the maximum available condition targets in ConditionMgr::GetMaxAvailableConditionTargets.
92
93 The following steps only apply if your condition can be grouped:
94
95 Step 6: Determine how you are going to store your conditions. You need to add a new storage container
96 for it in ConditionMgr class, along with a function like:
97 ConditionList GetConditionsForXXXYourNewSourceTypeXXX(parameters...)
98
99 The above function should be placed in upper level (practical) code that actually
100 checks the conditions.
101
102 Step 7: Implement loading for your source type in ConditionMgr::LoadConditions.
103
104 Step 8: Implement memory cleaning for your source type in ConditionMgr::Clean.
105*/

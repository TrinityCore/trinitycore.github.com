232template<class T>
233inline void LoadDBC(uint32& availableDbcLocales, StoreProblemList& errors, DBCStorage<T>& storage, std::string const& dbcPath, std::string const& filename, std::string const* customFormat = NULL, std::string const* customIndexName = NULL)
234{
235 // compatibility format and C++ structure sizes
236 ASSERT(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()) == sizeof(T) || LoadDBC_assert_print(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()), sizeof(T), filename));
237
238 ++DBCFileCount;
239 std::string dbcFilename = dbcPath + filename;
240 SqlDbc * sql = NULL;
241 if (customFormat)
242 sql = new SqlDbc(&filename, customFormat, customIndexName, storage.GetFormat());
243
244 if (storage.Load(dbcFilename.c_str(), sql))
245 {
246 for (uint8 i = 0; i < TOTAL_LOCALES; ++i)
247 {
248 if (!(availableDbcLocales & (1 << i)))
249 continue;
250
251 std::string localizedName(dbcPath);
252 localizedName.append(localeNames[i]);
253 localizedName.push_back('/');
254 localizedName.append(filename);
255
256 if (!storage.LoadStringsFrom(localizedName.c_str()))
257 availableDbcLocales &= ~(1<<i); // mark as not available for speedup next checks
258 }
259 }
260 else
261 {
262 // sort problematic dbc to (1) non compatible and (2) non-existed
263 if (FILE* f = fopen(dbcFilename.c_str(), "rb"))
264 {
265 std::ostringstream stream;
266 stream << dbcFilename << " exists, and has " << storage.GetFieldCount() << " field(s) (expected " << strlen(storage.GetFormat()) << "). Extracted file might be from wrong client version or a database-update has been forgotten.";
267 std::string buf = stream.str();
268 errors.push_back(buf);
269 fclose(f);
270 }
271 else
272 errors.push_back(dbcFilename);
273 }
274
275 delete sql;
276}

891 if (PvPDifficultyEntry const* entry = sPvPDifficultyStore.LookupEntry(i))
892 {
893 // skip unrelated and too-high brackets
894 if (entry->mapId != mapid || entry->minLevel > level)
895 continue;
896
897 // exactly fit
898 if (entry->maxLevel >= level)
899 return entry;
900
901 // remember for possible out-of-range case (search higher from existed)
902 if (!maxEntry || maxEntry->maxLevel < entry->maxLevel)
903 maxEntry = entry;
904 }
905 }
906
907 return maxEntry;
908}

1015 return CHAR_NAME_SUCCESS;

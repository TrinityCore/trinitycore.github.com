40void Arena::AddPlayer(Player* player)
41{
42 Battleground::AddPlayer(player);
43 PlayerScores[player->GetGUID().GetCounter()] = new ArenaScore(player->GetGUID(), player->GetBGTeam());
44
45 if (player->GetBGTeam() == ALLIANCE) // gold
46 {
47 if (player->GetTeam() == HORDE)
48 player->CastSpell(player, SPELL_HORDE_GOLD_FLAG, true);
49 else
50 player->CastSpell(player, SPELL_ALLIANCE_GOLD_FLAG, true);
51 }
52 else // green
53 {
54 if (player->GetTeam() == HORDE)
55 player->CastSpell(player, SPELL_HORDE_GREEN_FLAG, true);
56 else
57 player->CastSpell(player, SPELL_ALLIANCE_GREEN_FLAG, true);
58 }
59
60 UpdateArenaWorldState();
61}

63void Arena::RemovePlayer(Player* /*player*/, ObjectGuid /*guid*/, uint32 /*team*/)
64{
65 if (GetStatus() == STATUS_WAIT_LEAVE)
66 return;
67
68 UpdateArenaWorldState();
69 CheckWinConditions();
70}

95void Arena::RemovePlayerAtLeave(ObjectGuid guid, bool transport, bool sendPacket)
96{
97 if (isRated() && GetStatus() == STATUS_IN_PROGRESS)
98 {
99 BattlegroundPlayerMap::const_iterator itr = m_Players.find(guid);
100 if (itr != m_Players.end()) // check if the player was a participant of the match, or only entered through gm command (appear)
101 {
102 // if the player was a match participant, calculate rating
103 uint32 team = itr->second.Team;
104
105 ArenaTeam* winnerArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(GetOtherTeam(team)));
106 ArenaTeam* loserArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(team));
107
108 // left a rated match while the encounter was in progress, consider as loser
109 if (winnerArenaTeam && loserArenaTeam && winnerArenaTeam != loserArenaTeam)
110 {
111 if (Player* player = _GetPlayer(itr->first, itr->second.OfflineRemoveTime != 0, "Arena::RemovePlayerAtLeave"))
112 loserArenaTeam->MemberLost(player, GetArenaMatchmakerRating(GetOtherTeam(team)));
113 else
114 loserArenaTeam->OfflineMemberLost(guid, GetArenaMatchmakerRating(GetOtherTeam(team)));
115 }
116 }
117 }
118
119 // remove player
120 Battleground::RemovePlayerAtLeave(guid, transport, sendPacket);
121}

123void Arena::CheckWinConditions()
124{
125 if (!GetAlivePlayersCountByTeam(ALLIANCE) && GetPlayersCountByTeam(HORDE))
126 EndBattleground(HORDE);
127 else if (GetPlayersCountByTeam(ALLIANCE) && !GetAlivePlayersCountByTeam(HORDE))
128 EndBattleground(ALLIANCE);
129}

131void Arena::EndBattleground(uint32 winner)
132{
133 // arena rating calculation
134 if (isRated())
135 {
136 uint32 loserTeamRating = 0;
137 uint32 loserMatchmakerRating = 0;
138 int32 loserChange = 0;
139 int32 loserMatchmakerChange = 0;
140 uint32 winnerTeamRating = 0;
141 uint32 winnerMatchmakerRating = 0;
142 int32 winnerChange = 0;
143 int32 winnerMatchmakerChange = 0;
144
145 ArenaTeam* winnerArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(winner));
146 ArenaTeam* loserArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(GetOtherTeam(winner)));
147
148 if (winnerArenaTeam && loserArenaTeam && winnerArenaTeam != loserArenaTeam)
149 {
150 loserTeamRating = loserArenaTeam->GetRating();
151 loserMatchmakerRating = GetArenaMatchmakerRating(GetOtherTeam(winner));
152 winnerTeamRating = winnerArenaTeam->GetRating();
153 winnerMatchmakerRating = GetArenaMatchmakerRating(winner);
154
155 if (winner != 0)
156 {
157 winnerMatchmakerChange = winnerArenaTeam->WonAgainst(winnerMatchmakerRating, loserMatchmakerRating, winnerChange);
158 loserMatchmakerChange = loserArenaTeam->LostAgainst(loserMatchmakerRating, winnerMatchmakerRating, loserChange);
159
160 TC_LOG_DEBUG("bg.arena", "match Type: %u --- Winner: old rating: %u, rating gain: %d, old MMR: %u, MMR gain: %d --- Loser: old rating: %u, rating loss: %d, old MMR: %u, MMR loss: %d ---",
161 GetArenaType(), winnerTeamRating, winnerChange, winnerMatchmakerRating, winnerMatchmakerChange,
162 loserTeamRating, loserChange, loserMatchmakerRating, loserMatchmakerChange);
163
164 SetArenaMatchmakerRating(winner, winnerMatchmakerRating + winnerMatchmakerChange);
165 SetArenaMatchmakerRating(GetOtherTeam(winner), loserMatchmakerRating + loserMatchmakerChange);
166
167 // bg team that the client expects is different to TeamId
168 // alliance 1, horde 0
169 uint8 winnerTeam = winner == ALLIANCE ? BG_TEAM_ALLIANCE : BG_TEAM_HORDE;
170 uint8 loserTeam = winner == ALLIANCE ? BG_TEAM_HORDE : BG_TEAM_ALLIANCE;
171
172 _arenaTeamScores[winnerTeam].Assign(winnerChange, winnerMatchmakerRating, winnerArenaTeam->GetName());
173 _arenaTeamScores[loserTeam].Assign(loserChange, loserMatchmakerRating, loserArenaTeam->GetName());
174
175 TC_LOG_DEBUG("bg.arena", "Arena match Type: %u for Team1Id: %u - Team2Id: %u ended. WinnerTeamId: %u. Winner rating: +%d, Loser rating: %d",
176 GetArenaType(), GetArenaTeamIdByIndex(TEAM_ALLIANCE), GetArenaTeamIdByIndex(TEAM_HORDE), winnerArenaTeam->GetId(), winnerChange, loserChange);
177
178 if (sWorld->getBoolConfig(CONFIG_ARENA_LOG_EXTENDED_INFO))
179 for (auto const& score : PlayerScores)
180 if (Player* player = ObjectAccessor::FindConnectedPlayer(ObjectGuid(HighGuid::Player, score.first)))
181 {
182 TC_LOG_DEBUG("bg.arena", "Statistics match Type: %u for %s (GUID: %u, Team: %d, IP: %s): %s",
183 GetArenaType(), player->GetName().c_str(), score.first, player->GetArenaTeamId(GetArenaType() == 5 ? 2 : GetArenaType() == 3),
184 player->GetSession()->GetRemoteAddress().c_str(), score.second->ToString().c_str());
185 }
186 }
187 // Deduct 16 points from each teams arena-rating if there are no winners after 45+2 minutes
188 else
189 {
190 _arenaTeamScores[BG_TEAM_ALLIANCE].Assign(ARENA_TIMELIMIT_POINTS_LOSS, winnerMatchmakerRating, winnerArenaTeam->GetName());
191 _arenaTeamScores[BG_TEAM_HORDE].Assign(ARENA_TIMELIMIT_POINTS_LOSS, loserMatchmakerRating, loserArenaTeam->GetName());
192
193 winnerArenaTeam->FinishGame(ARENA_TIMELIMIT_POINTS_LOSS);
194 loserArenaTeam->FinishGame(ARENA_TIMELIMIT_POINTS_LOSS);
195 }
196
197 uint8 aliveWinners = GetAlivePlayersCountByTeam(winner);
198
199 for (auto const& i : GetPlayers())
200 {
201 uint32 team = i.second.Team;
202
203 if (i.second.OfflineRemoveTime)
204 {
205 // if rated arena match - make member lost!
206 if (team == winner)
207 winnerArenaTeam->OfflineMemberLost(i.first, loserMatchmakerRating, winnerMatchmakerChange);
208 else
209 loserArenaTeam->OfflineMemberLost(i.first, winnerMatchmakerRating, loserMatchmakerChange);
210 continue;
211 }
212
213 Player* player = _GetPlayer(i.first, i.second.OfflineRemoveTime != 0, "Arena::EndBattleground");
214 if (!player)
215 continue;
216
217 // per player calculation
218 if (team == winner)
219 {
220 // update achievement BEFORE personal rating update
221 uint32 rating = player->GetArenaPersonalRating(winnerArenaTeam->GetSlot());
222 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_WIN_RATED_ARENA, rating ? rating : 1);
223 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_WIN_ARENA, GetMapId());
224
225 // Last standing - Rated 5v5 arena & be solely alive player
226 if (GetArenaType() == ARENA_TYPE_5v5 && aliveWinners == 1 && player->IsAlive())
227 player->CastSpell(player, SPELL_LAST_MAN_STANDING, true);
228
229 winnerArenaTeam->MemberWon(player, loserMatchmakerRating, winnerMatchmakerChange);
230 }
231 else
232 {
233 loserArenaTeam->MemberLost(player, winnerMatchmakerRating, loserMatchmakerChange);
234
235 // Arena lost => reset the win_rated_arena having the "no_lose" condition
236 player->ResetAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_WIN_RATED_ARENA, ACHIEVEMENT_CRITERIA_CONDITION_NO_LOSE);
237 }
238 }
239
240 // save the stat changes
241 winnerArenaTeam->SaveToDB();
242 loserArenaTeam->SaveToDB();
243 // send updated arena team stats to players
244 // this way all arena team members will get notified, not only the ones who participated in this match
245 winnerArenaTeam->NotifyStatsChanged();
246 loserArenaTeam->NotifyStatsChanged();
247 }
248 }
249
250 // end battleground
251 Battleground::EndBattleground(winner);
252}

40 class BattlegroundChatBuilder
41 {
42 public:
43 BattlegroundChatBuilder(ChatMsg msgtype, uint32 textId, Player const* source, va_list* args = NULL)
44 : _msgtype(msgtype), _textId(textId), _source(source), _args(args) { }
45
46 void operator()(WorldPacket& data, LocaleConstant loc_idx)
47 {
48 char const* text = sObjectMgr->GetTrinityString(_textId, loc_idx);
49 if (_args)
50 {
51 // we need copy va_list before use or original va_list will corrupted
52 va_list ap;
53 va_copy(ap, *_args);
54
55 char str[2048];
56 vsnprintf(str, 2048, text, ap);
57 va_end(ap);
58
59 do_helper(data, &str[0]);
60 }
61 else
62 do_helper(data, text);
63 }
64
65 private:
66 void do_helper(WorldPacket& data, char const* text)
67 {
68 ChatHandler::BuildChatPacket(data, _msgtype, LANG_UNIVERSAL, _source, _source, text);
69 }
70
71 ChatMsg _msgtype;
72 uint32 _textId;
73 Player const* _source;
74 va_list* _args;
75 };

209void Battleground::Update(uint32 diff)
210{
211 if (!PreUpdateImpl(diff))
212 return;
213
214 if (!GetPlayersSize())
215 {
216 //BG is empty
217 // if there are no players invited, delete BG
218 // this will delete arena or bg object, where any player entered
219 // [[ but if you use battleground object again (more battles possible to be played on 1 instance)
220 // then this condition should be removed and code:
221 // if (!GetInvitedCount(HORDE) && !GetInvitedCount(ALLIANCE))
222 // this->AddToFreeBGObjectsQueue(); // not yet implemented
223 // should be used instead of current
224 // ]]
225 // Battleground Template instance cannot be updated, because it would be deleted
226 if (!GetInvitedCount(HORDE) && !GetInvitedCount(ALLIANCE))
227 m_SetDeleteThis = true;
228 return;
229 }
230
231 switch (GetStatus())
232 {
233 case STATUS_WAIT_JOIN:
234 if (GetPlayersSize())
235 {
236 _ProcessJoin(diff);
237 _CheckSafePositions(diff);
238 }
239 break;
240 case STATUS_IN_PROGRESS:
241 _ProcessOfflineQueue();
242 // after 47 minutes without one team losing, the arena closes with no winner and no rating change
243 if (isArena())
244 {
245 if (GetStartTime() >= 47 * MINUTE*IN_MILLISECONDS)
246 {
247 EndBattleground(0);
248 return;
249 }
250 }
251 else
252 {
253 _ProcessResurrect(diff);
254 if (sBattlegroundMgr->GetPrematureFinishTime() && (GetPlayersCountByTeam(ALLIANCE) < GetMinPlayersPerTeam() || GetPlayersCountByTeam(HORDE) < GetMinPlayersPerTeam()))
255 _ProcessProgress(diff);
256 else if (m_PrematureCountDown)
257 m_PrematureCountDown = false;
258 }
259 break;
260 case STATUS_WAIT_LEAVE:
261 _ProcessLeave(diff);
262 break;
263 default:
264 break;
265 }
266
267 // Update start time and reset stats timer
268 m_StartTime += diff;
269 m_ResetStatTimer += diff;
270
271 PostUpdateImpl(diff);
272}

274inline void Battleground::_CheckSafePositions(uint32 diff)
275{
276 float maxDist = GetStartMaxDist();
277 if (!maxDist)
278 return;
279
280 m_ValidStartPositionTimer += diff;
281 if (m_ValidStartPositionTimer >= CHECK_PLAYER_POSITION_INVERVAL)
282 {
283 m_ValidStartPositionTimer = 0;
284
285 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
286 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
287 {
288 Position pos = player->GetPosition();
289 Position const* startPos = GetTeamStartPosition(Battleground::GetTeamIndexByTeamId(player->GetBGTeam()));
290 if (pos.GetExactDistSq(startPos) > maxDist)
291 {
292 TC_LOG_DEBUG("bg.battleground", "BATTLEGROUND: Sending %s back to start location (map: %u) (possible exploit)", player->GetName().c_str(), GetMapId());
293 player->TeleportTo(GetMapId(), startPos->GetPositionX(), startPos->GetPositionY(), startPos->GetPositionZ(), startPos->GetOrientation());
294 }
295 }
296 }

299inline void Battleground::_ProcessOfflineQueue()
300{
301 // remove offline players from bg after 5 minutes
302 if (!m_OfflineQueue.empty())
303 {
304 BattlegroundPlayerMap::iterator itr = m_Players.find(*(m_OfflineQueue.begin()));
305 if (itr != m_Players.end())
306 {
307 if (itr->second.OfflineRemoveTime <= sWorld->GetGameTime())
308 {
309 if (isBattleground() && sWorld->getBoolConfig(CONFIG_BATTLEGROUND_TRACK_DESERTERS) &&
310 (GetStatus() == STATUS_IN_PROGRESS || GetStatus() == STATUS_WAIT_JOIN))
311 {
312 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_DESERTER_TRACK);
313 stmt->setUInt32(0, itr->first.GetCounter());
314 stmt->setUInt8(1, BG_DESERTION_TYPE_OFFLINE);
315 CharacterDatabase.Execute(stmt);
316 }
317
318 RemovePlayerAtLeave(itr->first, true, true);// remove player from BG
319 m_OfflineQueue.pop_front(); // remove from offline queue
320 //do not use itr for anything, because it is erased in RemovePlayerAtLeave()
321 }
322 }
323 }

326inline void Battleground::_ProcessResurrect(uint32 diff)
327{
328 // *********************************************************
329 // *** BATTLEGROUND RESURRECTION SYSTEM ***
330 // *********************************************************
331 // this should be handled by spell system
332 m_LastResurrectTime += diff;
333 if (m_LastResurrectTime >= RESURRECTION_INTERVAL)
334 {
335 if (GetReviveQueueSize())
336 {
337 for (std::map<ObjectGuid, GuidVector>::iterator itr = m_ReviveQueue.begin(); itr != m_ReviveQueue.end(); ++itr)
338 {
339 Creature* sh = NULL;
340 for (GuidVector::const_iterator itr2 = (itr->second).begin(); itr2 != (itr->second).end(); ++itr2)
341 {
342 Player* player = ObjectAccessor::FindPlayer(*itr2);
343 if (!player)
344 continue;
345
346 if (!sh && player->IsInWorld())
347 {
348 sh = player->GetMap()->GetCreature(itr->first);
349 // only for visual effect
350 if (sh)
351 // Spirit Heal, effect 117
352 sh->CastSpell(sh, SPELL_SPIRIT_HEAL, true);
353 }
354
355 // Resurrection visual
356 player->CastSpell(player, SPELL_RESURRECTION_VISUAL, true);
357 m_ResurrectQueue.push_back(*itr2);
358 }
359 (itr->second).clear();
360 }
361
362 m_ReviveQueue.clear();
363 m_LastResurrectTime = 0;
364 }
365 else
366 // queue is clear and time passed, just update last resurrection time
367 m_LastResurrectTime = 0;
368 }
369 else if (m_LastResurrectTime > 500) // Resurrect players only half a second later, to see spirit heal effect on NPC
370 {
371 for (GuidVector::const_iterator itr = m_ResurrectQueue.begin(); itr != m_ResurrectQueue.end(); ++itr)
372 {
373 Player* player = ObjectAccessor::FindPlayer(*itr);
374 if (!player)
375 continue;
376 player->ResurrectPlayer(1.0f);
377 player->CastSpell(player, 6962, true);
378 player->CastSpell(player, SPELL_SPIRIT_HEAL_MANA, true);
379 player->SpawnCorpseBones(false);
380 }
381 m_ResurrectQueue.clear();
382 }

396inline void Battleground::_ProcessProgress(uint32 diff)
397{
398 // *********************************************************
399 // *** BATTLEGROUND BALLANCE SYSTEM ***
400 // *********************************************************
401 // if less then minimum players are in on one side, then start premature finish timer
402 if (!m_PrematureCountDown)
403 {
404 m_PrematureCountDown = true;
405 m_PrematureCountDownTimer = sBattlegroundMgr->GetPrematureFinishTime();
406 }
407 else if (m_PrematureCountDownTimer < diff)
408 {
409 // time's up!
410 EndBattleground(GetPrematureWinner());
411 m_PrematureCountDown = false;
412 }
413 else if (!sBattlegroundMgr->isTesting())
414 {
415 uint32 newtime = m_PrematureCountDownTimer - diff;
416 // announce every minute
417 if (newtime > (MINUTE * IN_MILLISECONDS))
418 {
419 if (newtime / (MINUTE * IN_MILLISECONDS) != m_PrematureCountDownTimer / (MINUTE * IN_MILLISECONDS))
420 PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING, CHAT_MSG_SYSTEM, NULL, (uint32)(m_PrematureCountDownTimer / (MINUTE * IN_MILLISECONDS)));
421 }
422 else
423 {
424 //announce every 15 seconds
425 if (newtime / (15 * IN_MILLISECONDS) != m_PrematureCountDownTimer / (15 * IN_MILLISECONDS))
426 PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING_SECS, CHAT_MSG_SYSTEM, NULL, (uint32)(m_PrematureCountDownTimer / IN_MILLISECONDS));
427 }
428 m_PrematureCountDownTimer = newtime;
429 }

432inline void Battleground::_ProcessJoin(uint32 diff)
433{
434 // *********************************************************
435 // *** BATTLEGROUND STARTING SYSTEM ***
436 // *********************************************************
437 ModifyStartDelayTime(diff);
438
439 if (m_ResetStatTimer > 5000)
440 {
441 m_ResetStatTimer = 0;
442 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
443 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
444 player->ResetAllPowers();
445 }
446
447 if (!(m_Events & BG_STARTING_EVENT_1))
448 {
449 m_Events |= BG_STARTING_EVENT_1;
450
451 if (!FindBgMap())
452 {
453 TC_LOG_ERROR("bg.battleground", "Battleground::_ProcessJoin: map (map id: %u, instance id: %u) is not created!", m_MapId, m_InstanceID);
454 EndNow();
455 return;
456 }
457
458 // Setup here, only when at least one player has ported to the map
459 if (!SetupBattleground())
460 {
461 EndNow();
462 return;
463 }
464
465 StartingEventCloseDoors();
466 SetStartDelayTime(StartDelayTimes[BG_STARTING_EVENT_FIRST]);
467 // First start warning - 2 or 1 minute
468 SendMessageToAll(StartMessageIds[BG_STARTING_EVENT_FIRST], CHAT_MSG_BG_SYSTEM_NEUTRAL);
469 }
470 // After 1 minute or 30 seconds, warning is signaled
471 else if (GetStartDelayTime() <= StartDelayTimes[BG_STARTING_EVENT_SECOND] && !(m_Events & BG_STARTING_EVENT_2))
472 {
473 m_Events |= BG_STARTING_EVENT_2;
474 SendMessageToAll(StartMessageIds[BG_STARTING_EVENT_SECOND], CHAT_MSG_BG_SYSTEM_NEUTRAL);
475 }
476 // After 30 or 15 seconds, warning is signaled
477 else if (GetStartDelayTime() <= StartDelayTimes[BG_STARTING_EVENT_THIRD] && !(m_Events & BG_STARTING_EVENT_3))
478 {
479 m_Events |= BG_STARTING_EVENT_3;
480 SendMessageToAll(StartMessageIds[BG_STARTING_EVENT_THIRD], CHAT_MSG_BG_SYSTEM_NEUTRAL);
481 }
482 // Delay expired (after 2 or 1 minute)
483 else if (GetStartDelayTime() <= 0 && !(m_Events & BG_STARTING_EVENT_4))
484 {
485 m_Events |= BG_STARTING_EVENT_4;
486
487 StartingEventOpenDoors();
488
489 SendWarningToAll(StartMessageIds[BG_STARTING_EVENT_FOURTH]);
490 SetStatus(STATUS_IN_PROGRESS);
491 SetStartDelayTime(StartDelayTimes[BG_STARTING_EVENT_FOURTH]);
492
493 // Remove preparation
494 if (isArena())
495 {
496 /// @todo add arena sound PlaySoundToAll(SOUND_ARENA_START);
497 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
498 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
499 {
500 // BG Status packet
501 WorldPacket status;
502 BattlegroundQueueTypeId bgQueueTypeId = sBattlegroundMgr->BGQueueTypeId(m_TypeID, GetArenaType());
503 uint32 queueSlot = player->GetBattlegroundQueueIndex(bgQueueTypeId);
504 sBattlegroundMgr->BuildBattlegroundStatusPacket(&status, this, queueSlot, STATUS_IN_PROGRESS, 0, GetStartTime(), GetArenaType(), player->GetBGTeam());
505 player->SendDirectMessage(&status);
506
507 player->RemoveAurasDueToSpell(SPELL_ARENA_PREPARATION);
508 player->ResetAllPowers();
509 if (!player->IsGameMaster())
510 {
511 // remove auras with duration lower than 30s
512 Unit::AuraApplicationMap & auraMap = player->GetAppliedAuras();
513 for (Unit::AuraApplicationMap::iterator iter = auraMap.begin(); iter != auraMap.end();)
514 {
515 AuraApplication * aurApp = iter->second;
516 Aura* aura = aurApp->GetBase();
517 if (!aura->IsPermanent()
518 && aura->GetDuration() <= 30*IN_MILLISECONDS
519 && aurApp->IsPositive()
520 && (!aura->GetSpellInfo()->HasAttribute(SPELL_ATTR0_UNAFFECTED_BY_INVULNERABILITY))
521 && (!aura->HasEffectType(SPELL_AURA_MOD_INVISIBILITY)))
522 player->RemoveAura(iter);
523 else
524 ++iter;
525 }
526 }
527 }
528
529 CheckWinConditions();
530 }
531 else
532 {
533 PlaySoundToAll(SOUND_BG_START);
534
535 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
536 if (Player* player = ObjectAccessor::FindPlayer(itr->first))
537 {
538 player->RemoveAurasDueToSpell(SPELL_PREPARATION);
539 player->ResetAllPowers();
540 }
541 // Announce BG starting
542 if (sWorld->getBoolConfig(CONFIG_BATTLEGROUND_QUEUE_ANNOUNCER_ENABLE))
543 sWorld->SendWorldText(LANG_BG_STARTED_ANNOUNCE_WORLD, GetName().c_str(), GetMinLevel(), GetMaxLevel());
544 }
545 }
546}

548inline void Battleground::_ProcessLeave(uint32 diff)
549{
550 // *********************************************************
551 // *** BATTLEGROUND ENDING SYSTEM ***
552 // *********************************************************
553 // remove all players from battleground after 2 minutes
554 m_EndTime -= diff;
555 if (m_EndTime <= 0)
556 {
557 m_EndTime = 0;
558 BattlegroundPlayerMap::iterator itr, next;
559 for (itr = m_Players.begin(); itr != m_Players.end(); itr = next)
560 {
561 next = itr;
562 ++next;
563 //itr is erased here!
564 RemovePlayerAtLeave(itr->first, true, true);// remove player from BG
565 // do not change any battleground's private variables
566 }
567 }

707{

748 stmt->setUInt8(1, GetWinner());
749 stmt->setUInt8(2, GetUniqueBracketId());
750 stmt->setUInt8(3, GetTypeID(true));

1174// get the number of free slots for team
1175// returns the number how many players can join battleground to MaxPlayersPerTeam
1176uint32 Battleground::GetFreeSlotsForTeam(uint32 Team) const
1177{
1178 // if BG is starting and CONFIG_BATTLEGROUND_INVITATION_TYPE == BG_QUEUE_INVITATION_TYPE_NO_BALANCE, invite anyone
1179 if (GetStatus() == STATUS_WAIT_JOIN && sWorld->getIntConfig(CONFIG_BATTLEGROUND_INVITATION_TYPE) == BG_QUEUE_INVITATION_TYPE_NO_BALANCE)
1180 return (GetInvitedCount(Team) < GetMaxPlayersPerTeam()) ? GetMaxPlayersPerTeam() - GetInvitedCount(Team) : 0;
1181
1182 // if BG is already started or CONFIG_BATTLEGROUND_INVITATION_TYPE != BG_QUEUE_INVITATION_TYPE_NO_BALANCE, do not allow to join too much players of one faction
1183 uint32 otherTeamInvitedCount;
1184 uint32 thisTeamInvitedCount;
1185 uint32 otherTeamPlayersCount;
1186 uint32 thisTeamPlayersCount;
1187
1188 if (Team == ALLIANCE)
1189 {
1190 thisTeamInvitedCount = GetInvitedCount(ALLIANCE);
1191 otherTeamInvitedCount = GetInvitedCount(HORDE);
1192 thisTeamPlayersCount = GetPlayersCountByTeam(ALLIANCE);
1193 otherTeamPlayersCount = GetPlayersCountByTeam(HORDE);
1194 }
1195 else
1196 {
1197 thisTeamInvitedCount = GetInvitedCount(HORDE);
1198 otherTeamInvitedCount = GetInvitedCount(ALLIANCE);
1199 thisTeamPlayersCount = GetPlayersCountByTeam(HORDE);
1200 otherTeamPlayersCount = GetPlayersCountByTeam(ALLIANCE);
1201 }
1202 if (GetStatus() == STATUS_IN_PROGRESS || GetStatus() == STATUS_WAIT_JOIN)
1203 {
1204 // difference based on ppl invited (not necessarily entered battle)
1205 // default: allow 0
1206 uint32 diff = 0;
1207
1208 // allow join one person if the sides are equal (to fill up bg to minPlayerPerTeam)
1209 if (otherTeamInvitedCount == thisTeamInvitedCount)
1210 diff = 1;
1211 // allow join more ppl if the other side has more players
1212 else if (otherTeamInvitedCount > thisTeamInvitedCount)
1213 diff = otherTeamInvitedCount - thisTeamInvitedCount;
1214
1215 // difference based on max players per team (don't allow inviting more)
1216 uint32 diff2 = (thisTeamInvitedCount < GetMaxPlayersPerTeam()) ? GetMaxPlayersPerTeam() - thisTeamInvitedCount : 0;
1217
1218 // difference based on players who already entered
1219 // default: allow 0
1220 uint32 diff3 = 0;
1221 // allow join one person if the sides are equal (to fill up bg minPlayerPerTeam)
1222 if (otherTeamPlayersCount == thisTeamPlayersCount)
1223 diff3 = 1;
1224 // allow join more ppl if the other side has more players
1225 else if (otherTeamPlayersCount > thisTeamPlayersCount)
1226 diff3 = otherTeamPlayersCount - thisTeamPlayersCount;
1227 // or other side has less than minPlayersPerTeam
1228 else if (thisTeamInvitedCount <= GetMinPlayersPerTeam())
1229 diff3 = GetMinPlayersPerTeam() - thisTeamInvitedCount + 1;
1230
1231 // return the minimum of the 3 differences
1232
1233 // min of diff and diff 2
1234 diff = std::min(diff, diff2);
1235 // min of diff, diff2 and diff3
1236 return std::min(diff, diff3);
1237 }
1238 return 0;
1239}

18#ifndef TRINITY_ARENA_SCORE_H
19#define TRINITY_ARENA_SCORE_H
20
21#include "BattlegroundScore.h"
22#include "SharedDefines.h"
23
24struct ArenaScore : public BattlegroundScore
25{
26 friend class Arena;
27
28 protected:
29 ArenaScore(ObjectGuid playerGuid, uint32 team) : BattlegroundScore(playerGuid), TeamId(team == ALLIANCE ? BG_TEAM_ALLIANCE : BG_TEAM_HORDE) { }
30
31 void AppendToPacket(WorldPacket& data) final override
32 {
33 data << uint64(PlayerGuid);
34
35 data << uint32(KillingBlows);
36 data << uint8(TeamId);
37 data << uint32(DamageDone);
38 data << uint32(HealingDone);
39
40 BuildObjectivesBlock(data);
41 }
42
43 void BuildObjectivesBlock(WorldPacket& data) final override
44 {
45 data << uint32(0); // Objectives Count
46 }
47
48 // For Logging purpose
49 std::string ToString() const override
50 {
51 std::ostringstream stream;
52 stream << "Damage done: " << DamageDone << ", Healing done: " << HealingDone << ", Killing blows: " << KillingBlows;
53 return stream.str();
54 }
55
56 uint8 TeamId; // BattlegroundTeamId
57};
58
59struct ArenaTeamScore
60{
61 friend class Arena;
62 friend class Battleground;
63
64 protected:
65 ArenaTeamScore() : RatingChange(0), MatchmakerRating(0) { }
66
67 virtual ~ArenaTeamScore() { }
68
69 void Reset()
70 {
71 RatingChange = 0;
72 MatchmakerRating = 0;
73 TeamName.clear();
74 }
75
76 void Assign(int32 ratingChange, uint32 matchMakerRating, std::string const& teamName)
77 {
78 RatingChange = ratingChange;
79 MatchmakerRating = matchMakerRating;
80 TeamName = teamName;
81 }
82
83 void BuildRatingInfoBlock(WorldPacket& data)
84 {
85 uint32 ratingLost = std::abs(std::min(RatingChange, 0));
86 uint32 ratingWon = std::max(RatingChange, 0);
87
88 // should be old rating, new rating, and client will calculate rating change itself
89 data << uint32(ratingLost);
90 data << uint32(ratingWon);
91 data << uint32(MatchmakerRating);
92 }
93
94 void BuildTeamInfoBlock(WorldPacket& data)
95 {
96 data << TeamName;
97 }
98
99 int32 RatingChange;
100 uint32 MatchmakerRating;
101 std::string TeamName;
102};

104#endif // TRINITY_ARENA_SCORE_H

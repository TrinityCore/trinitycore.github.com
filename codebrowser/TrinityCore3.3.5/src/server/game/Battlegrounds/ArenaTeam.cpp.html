44bool ArenaTeam::Create(ObjectGuid captainGuid, uint8 type, std::string const& teamName, uint32 backgroundColor, uint8 emblemStyle, uint32 emblemColor, uint8 borderStyle, uint32 borderColor)
45{
46 // Check if captain is present
47 if (!ObjectAccessor::FindPlayer(captainGuid))
48 return false;
49
50 // Check if arena team name is already taken
51 if (sArenaTeamMgr->GetArenaTeamByName(teamName))
52 return false;
53
54 // Generate new arena team id
55 TeamId = sArenaTeamMgr->GenerateArenaTeamId();
56
57 // Assign member variables
58 CaptainGuid = captainGuid;
59 Type = type;
60 TeamName = teamName;
61 BackgroundColor = backgroundColor;
62 EmblemStyle = emblemStyle;
63 EmblemColor = emblemColor;
64 BorderStyle = borderStyle;
65 BorderColor = borderColor;
66 ObjectGuid::LowType captainLowGuid = captainGuid.GetCounter();
67
68 // Save arena team to db
69 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_ARENA_TEAM);
70 stmt->setUInt32(0, TeamId);
71 stmt->setString(1, TeamName);
72 stmt->setUInt32(2, captainLowGuid);
73 stmt->setUInt8(3, Type);
74 stmt->setUInt16(4, Stats.Rating);
75 stmt->setUInt32(5, BackgroundColor);
76 stmt->setUInt8(6, EmblemStyle);
77 stmt->setUInt32(7, EmblemColor);
78 stmt->setUInt8(8, BorderStyle);
79 stmt->setUInt32(9, BorderColor);
80 CharacterDatabase.Execute(stmt);
81
82 // Add captain as member
83 AddMember(CaptainGuid);
84
85 TC_LOG_DEBUG("bg.arena", "New ArenaTeam created [Id: %u, Name: %s] [Type: %u] [Captain low GUID: %u]", GetId(), GetName().c_str(), GetType(), captainLowGuid);
86 return true;
87}

89bool ArenaTeam::AddMember(ObjectGuid playerGuid)
90{
91 std::string playerName;
92 uint8 playerClass;
93
94 // Check if arena team is full (Can't have more than type * 2 players)
95 if (GetMembersSize() >= GetType() * 2)
96 return false;
97
98 // Get player name and class either from db or ObjectMgr
99 Player* player = ObjectAccessor::FindPlayer(playerGuid);
100 if (player)
101 {
102 playerClass = player->getClass();
103 playerName = player->GetName();
104 }
105 else
106 {
107 // 0 1
108 // SELECT name, class FROM characters WHERE guid = ?
109 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHARACTER_NAME_CLASS);
110 stmt->setUInt32(0, playerGuid.GetCounter());
111 PreparedQueryResult result = CharacterDatabase.Query(stmt);
112
113 if (!result)
114 return false;
115
116 playerName = (*result)[0].GetString();
117 playerClass = (*result)[1].GetUInt8();
118 }
119
120 // Check if player is already in a similar arena team
121 if ((player && player->GetArenaTeamId(GetSlot())) || Player::GetArenaTeamIdFromDB(playerGuid, GetType()) != 0)
122 {
123 TC_LOG_DEBUG("bg.arena", "Arena: %s %s already has an arena team of type %u", playerGuid.ToString().c_str(), playerName.c_str(), GetType());
124 return false;
125 }
126
127 // Set player's personal rating
128 uint32 personalRating = 0;
129
130 if (sWorld->getIntConfig(CONFIG_ARENA_START_PERSONAL_RATING) > 0)
131 personalRating = sWorld->getIntConfig(CONFIG_ARENA_START_PERSONAL_RATING);
132 else if (GetRating() >= 1000)
133 personalRating = 1000;
134
135 // Try to get player's match maker rating from db and fall back to config setting if not found
136 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_MATCH_MAKER_RATING);
137 stmt->setUInt32(0, playerGuid.GetCounter());
138 stmt->setUInt8(1, GetSlot());
139 PreparedQueryResult result = CharacterDatabase.Query(stmt);
140
141 uint32 matchMakerRating;
142 if (result)
143 matchMakerRating = (*result)[0].GetUInt16();
144 else
145 matchMakerRating = sWorld->getIntConfig(CONFIG_ARENA_START_MATCHMAKER_RATING);
146
147 // Remove all player signatures from other petitions
148 // This will prevent player from joining too many arena teams and corrupt arena team data integrity
149 Player::RemovePetitionsAndSigns(playerGuid, GetType());
150
151 // Feed data to the struct
152 ArenaTeamMember newMember;
153 newMember.Name = playerName;
154 newMember.Guid = playerGuid;
155 newMember.Class = playerClass;
156 newMember.SeasonGames = 0;
157 newMember.WeekGames = 0;
158 newMember.SeasonWins = 0;
159 newMember.WeekWins = 0;
160 newMember.PersonalRating = personalRating;
161 newMember.MatchMakerRating = matchMakerRating;
162
163 Members.push_back(newMember);
164
165 // Save player's arena team membership to db
166 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_ARENA_TEAM_MEMBER);
167 stmt->setUInt32(0, TeamId);
168 stmt->setUInt32(1, playerGuid.GetCounter());
169 CharacterDatabase.Execute(stmt);
170
171 // Inform player if online
172 if (player)
173 {
174 player->SetInArenaTeam(TeamId, GetSlot(), GetType());
175 player->SetArenaTeamIdInvited(0);
176
177 // Hide promote/remove buttons
178 if (CaptainGuid != playerGuid)
179 player->SetArenaTeamInfoField(GetSlot(), ARENA_TEAM_MEMBER, 1);
180 }
181
182 TC_LOG_DEBUG("bg.arena", "Player: %s [%s] joined arena team type: %u [Id: %u, Name: %s].", playerName.c_str(), playerGuid.ToString().c_str(), GetType(), GetId(), GetName().c_str());
183
184 return true;
185}

187bool ArenaTeam::LoadArenaTeamFromDB(QueryResult result)
188{
189 if (!result)
190 return false;
191
192 Field* fields = result->Fetch();
193
194 TeamId = fields[0].GetUInt32();
195 TeamName = fields[1].GetString();
196 CaptainGuid = ObjectGuid(HighGuid::Player, fields[2].GetUInt32());
197 Type = fields[3].GetUInt8();
198 BackgroundColor = fields[4].GetUInt32();
199 EmblemStyle = fields[5].GetUInt8();
200 EmblemColor = fields[6].GetUInt32();
201 BorderStyle = fields[7].GetUInt8();

205 Stats.WeekWins = fields[11].GetUInt16();
206 Stats.SeasonGames = fields[12].GetUInt16();
207 Stats.SeasonWins = fields[13].GetUInt16();
208 Stats.Rank = fields[14].GetUInt32();
209
210 return true;
211}

213bool ArenaTeam::LoadMembersFromDB(QueryResult result)
214{
215 if (!result)
216 return false;
217
218 bool captainPresentInTeam = false;
219
220 do
221 {
222 Field* fields = result->Fetch();
223
224 // Prevent crash if db records are broken when all members in result are already processed and current team doesn't have any members
225 if (!fields)
226 break;
227
228 uint32 arenaTeamId = fields[0].GetUInt32();
229
230 // We loaded all members for this arena_team already, break cycle
231 if (arenaTeamId > TeamId)
232 break;
233
234 ArenaTeamMember newMember;
235 newMember.Guid = ObjectGuid(HighGuid::Player, fields[1].GetUInt32());
236 newMember.WeekGames = fields[2].GetUInt16();
237 newMember.WeekWins = fields[3].GetUInt16();
238 newMember.SeasonGames = fields[4].GetUInt16();
239 newMember.SeasonWins = fields[5].GetUInt16();
240 newMember.Name = fields[6].GetString();
241 newMember.Class = fields[7].GetUInt8();
242 newMember.PersonalRating = fields[8].GetUInt16();
243 newMember.MatchMakerRating = fields[9].GetUInt16() > 0 ? fields[9].GetUInt16() : sWorld->getIntConfig(CONFIG_ARENA_START_MATCHMAKER_RATING);
244
245 // Delete member if character information is missing
246 if (newMember.Name.empty())
247 {
248 TC_LOG_ERROR("sql.sql", "ArenaTeam %u has member with empty name - probably %s doesn't exist, deleting him from memberlist!", arenaTeamId, newMember.Guid.ToString().c_str());
249 DelMember(newMember.Guid, true);
250 continue;
251 }
252
253 // Check if team team has a valid captain
254 if (newMember.Guid == GetCaptain())
255 captainPresentInTeam = true;
256
257 // Put the player in the team
258 Members.push_back(newMember);
259 }
260 while (result->NextRow());
261
262 if (Empty() || !captainPresentInTeam)
263 {
264 // Arena team is empty or captain is not in team, delete from db
265 TC_LOG_DEBUG("bg.arena", "ArenaTeam %u does not have any members or its captain is not in team, disbanding it...", TeamId);
266 return false;
267 }
268
269 return true;
270}

314void ArenaTeam::DelMember(ObjectGuid guid, bool cleanDb)
315{
316 // Remove member from team
317 for (MemberList::iterator itr = Members.begin(); itr != Members.end(); ++itr)
318 if (itr->Guid == guid)
319 {
320 Members.erase(itr);
321 break;
322 }
323
324 // Inform player and remove arena team info from player data
325 if (Player* player = ObjectAccessor::FindPlayer(guid))
326 {
327 player->GetSession()->SendArenaTeamCommandResult(ERR_ARENA_TEAM_QUIT_S, GetName(), "", 0);
328 // delete all info regarding this team
329 for (uint32 i = 0; i < ARENA_TEAM_END; ++i)
330 player->SetArenaTeamInfoField(GetSlot(), ArenaTeamInfoType(i), 0);
331 TC_LOG_DEBUG("bg.arena", "Player: %s [GUID: %u] left arena team type: %u [Id: %u, Name: %s].", player->GetName().c_str(), player->GetGUID().GetCounter(), GetType(), GetId(), GetName().c_str());
332 }
333
334 // Only used for single member deletion, for arena team disband we use a single query for more efficiency
335 if (cleanDb)
336 {
337 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_ARENA_TEAM_MEMBER);
338 stmt->setUInt32(0, GetId());
339 stmt->setUInt32(1, guid.GetCounter());
340 CharacterDatabase.Execute(stmt);
341 }
342}

344void ArenaTeam::Disband(WorldSession* session)
345{
346 // Remove all members from arena team
347 while (!Members.empty())
348 DelMember(Members.front().Guid, false);
349
350 // Broadcast update
351 if (session)
352 {
353 BroadcastEvent(ERR_ARENA_TEAM_DISBANDED_S, ObjectGuid::Empty, 2, session->GetPlayerName(), GetName(), "");
354
355 if (Player* player = session->GetPlayer())
356 TC_LOG_DEBUG("bg.arena", "Player: %s [GUID: %u] disbanded arena team type: %u [Id: %u, Name: %s].", player->GetName().c_str(), player->GetGUID().GetCounter(), GetType(), GetId(), GetName().c_str());
357 }
358
359 // Update database
360 SQLTransaction trans = CharacterDatabase.BeginTransaction();
361
362 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_ARENA_TEAM);
363 stmt->setUInt32(0, TeamId);
364 trans->Append(stmt);
365
366 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_ARENA_TEAM_MEMBERS);
367 stmt->setUInt32(0, TeamId);
368 trans->Append(stmt);
369
370 CharacterDatabase.CommitTransaction(trans);
371
372 // Remove arena team from ObjectMgr
373 sArenaTeamMgr->RemoveArenaTeam(TeamId);
374}

529 break;
530 case 1:
531 data << str1;
532 break;
533 case 2:
534 data << str1 << str2;
535 break;
536 case 3:
537 data << str1 << str2 << str3;
538 break;
539 default:
540 TC_LOG_ERROR("bg.arena", "Unhandled strCount %u in ArenaTeam::BroadcastEvent", strCount);
541 return;
542 }
543
544 if (guid)
545 data << uint64(guid);
546
547 BroadcastPacket(&data);
548
549 TC_LOG_DEBUG("network", "WORLD: Sent SMSG_ARENA_TEAM_EVENT");
550}

592uint32 ArenaTeam::GetPoints(uint32 memberRating)
593{
594 // Returns how many points would be awarded with this team type with this rating
595 float points;
596
597 uint32 rating = memberRating + 150 < Stats.Rating ? memberRating : Stats.Rating;
598
599 if (rating <= 1500)
600 {
601 if (sWorld->getIntConfig(CONFIG_ARENA_SEASON_ID) < 6)
602 points = (float)rating * 0.22f + 14.0f;
603 else
604 points = 344;
605 }
606 else
607 points = 1511.26f / (1.0f + 1639.28f * std::exp(-0.00412f * float(rating)));
608
609 // Type penalties for teams < 5v5
610 if (Type == ARENA_TEAM_2v2)
611 points *= 0.76f;
612 else if (Type == ARENA_TEAM_3v3)
613 points *= 0.88f;
614
615 points *= sWorld->getRate(RATE_ARENA_POINTS);
616
617 return (uint32) points;
618}

620uint32 ArenaTeam::GetAverageMMR(Group* group) const
621{
622 if (!group)
623 return 0;
624
625 uint32 matchMakerRating = 0;
626 uint32 playerDivider = 0;
627 for (MemberList::const_iterator itr = Members.begin(); itr != Members.end(); ++itr)
628 {
629 // Skip if player is not online
630 if (!ObjectAccessor::FindConnectedPlayer(itr->Guid))
631 continue;
632
633 // Skip if player is not member of group
634 if (!group->IsMember(itr->Guid))
635 continue;
636
637 matchMakerRating += itr->MatchMakerRating;
638 ++playerDivider;
639 }
640
641 // x/0 = crash
642 if (playerDivider == 0)
643 playerDivider = 1;
644
645 matchMakerRating /= playerDivider;
646
647 return matchMakerRating;
648}

650float ArenaTeam::GetChanceAgainst(uint32 ownRating, uint32 opponentRating)
651{
652 // Returns the chance to win against a team with the given rating, used in the rating adjustment calculation
653 // ELO system
654 return 1.0f / (1.0f + std::exp(std::log(10.0f) * (float(opponentRating) - float(ownRating)) / 650.0f));
655}

657int32 ArenaTeam::GetMatchmakerRatingMod(uint32 ownRating, uint32 opponentRating, bool won /*, float& confidence_factor*/)
658{
659 // 'Chance' calculation - to beat the opponent
660 // This is a simulation. Not much info on how it really works
661 float chance = GetChanceAgainst(ownRating, opponentRating);
662 float won_mod = (won) ? 1.0f : 0.0f;
663 float mod = won_mod - chance;
664
665 // Work in progress:
666 /*
667 // This is a simulation, as there is not much info on how it really works
668 float confidence_mod = min(1.0f - fabs(mod), 0.5f);
669
670 // Apply confidence factor to the mod:
671 mod *= confidence_factor
672
673 // And only after that update the new confidence factor
674 confidence_factor -= ((confidence_factor - 1.0f) * confidence_mod) / confidence_factor;
675 */
676
677 // Real rating modification
678 mod *= sWorld->getFloatConfig(CONFIG_ARENA_MATCHMAKER_RATING_MODIFIER);
679
680 return (int32)ceil(mod);
681}

683int32 ArenaTeam::GetRatingMod(uint32 ownRating, uint32 opponentRating, bool won /*, float confidence_factor*/)
684{
685 // 'Chance' calculation - to beat the opponent
686 // This is a simulation. Not much info on how it really works
687 float chance = GetChanceAgainst(ownRating, opponentRating);
688
689 // Calculate the rating modification
690 float mod;
691
692 /// @todo Replace this hack with using the confidence factor (limiting the factor to 2.0f)
693 if (won)
694 {
695 if (ownRating < 1300)
696 {
697 float win_rating_modifier1 = sWorld->getFloatConfig(CONFIG_ARENA_WIN_RATING_MODIFIER_1);
698
699 if (ownRating < 1000)
700 mod = win_rating_modifier1 * (1.0f - chance);
701 else
702 mod = ((win_rating_modifier1 / 2.0f) + ((win_rating_modifier1 / 2.0f) * (1300.0f - float(ownRating)) / 300.0f)) * (1.0f - chance);
703 }
704 else
705 mod = sWorld->getFloatConfig(CONFIG_ARENA_WIN_RATING_MODIFIER_2) * (1.0f - chance);
706 }
707 else
708 mod = sWorld->getFloatConfig(CONFIG_ARENA_LOSE_RATING_MODIFIER) * (-chance);
709
710 return (int32)ceil(mod);
711}

713void ArenaTeam::FinishGame(int32 mod)
714{
715 // Rating can only drop to 0
716 if (int32(Stats.Rating) + mod < 0)
717 Stats.Rating = 0;
718 else
719 {
720 Stats.Rating += mod;
721
722 // Check if rating related achivements are met
723 for (MemberList::iterator itr = Members.begin(); itr != Members.end(); ++itr)
724 if (Player* member = ObjectAccessor::FindConnectedPlayer(itr->Guid))
725 member->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_TEAM_RATING, Stats.Rating, Type);
726 }
727
728 // Update number of games played per season or week
729 Stats.WeekGames += 1;
730 Stats.SeasonGames += 1;
731
732 // Update team's rank, start with rank 1 and increase until no team with more rating was found
733 Stats.Rank = 1;
734 ArenaTeamMgr::ArenaTeamContainer::const_iterator i = sArenaTeamMgr->GetArenaTeamMapBegin();
735 for (; i != sArenaTeamMgr->GetArenaTeamMapEnd(); ++i)
736 {
737 if (i->second->GetType() == Type && i->second->GetStats().Rating > Stats.Rating)
738 ++Stats.Rank;
739 }
740}

853void ArenaTeam::UpdateArenaPointsHelper(std::map<uint32, uint32>& playerPoints)
854{
855 // Called after a match has ended and the stats are already modified
856 // Helper function for arena point distribution (this way, when distributing, no actual calculation is required, just a few comparisons)
857 // 10 played games per week is a minimum
858 if (Stats.WeekGames < 10)
859 return;
860
861 // To get points, a player has to participate in at least 30% of the matches
862 uint32 requiredGames = (uint32)ceil(Stats.WeekGames * 0.3f);
863
864 for (MemberList::const_iterator itr = Members.begin(); itr != Members.end(); ++itr)
865 {
866 // The player participated in enough games, update his points
867 uint32 pointsToAdd = 0;
868 if (itr->WeekGames >= requiredGames)
869 pointsToAdd = GetPoints(itr->PersonalRating);
870
871 std::map<uint32, uint32>::iterator plr_itr = playerPoints.find(itr->Guid.GetCounter());
872 if (plr_itr != playerPoints.end())
873 {
874 // Check if there is already more points
875 if (plr_itr->second < pointsToAdd)
876 playerPoints[itr->Guid.GetCounter()] = pointsToAdd;
877 }
878 else
879 playerPoints[itr->Guid.GetCounter()] = pointsToAdd;
880 }

33void BattlegroundDS::PostUpdateImpl(uint32 diff)
34{
35 if (GetStatus() != STATUS_IN_PROGRESS)
36 return;
37
38 _events.Update(diff);
39
40 while (uint32 eventId = _events.ExecuteEvent())
41 {
42 switch (eventId)
43 {
44 case BG_DS_EVENT_WATERFALL_WARNING:
45 // Add the water
46 DoorClose(BG_DS_OBJECT_WATER_2);
47 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_ON, BG_DS_WATERFALL_WARNING_DURATION);
48 break;
49 case BG_DS_EVENT_WATERFALL_ON:
50 // Active collision and start knockback timer
51 DoorClose(BG_DS_OBJECT_WATER_1);
52 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_OFF, BG_DS_WATERFALL_DURATION);
53 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_KNOCKBACK, BG_DS_WATERFALL_KNOCKBACK_TIMER);
54 break;
55 case BG_DS_EVENT_WATERFALL_OFF:
56 // Remove collision and water
57 DoorOpen(BG_DS_OBJECT_WATER_1);
58 DoorOpen(BG_DS_OBJECT_WATER_2);
59 _events.CancelEvent(BG_DS_EVENT_WATERFALL_KNOCKBACK);
60 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_WARNING, urand(BG_DS_WATERFALL_TIMER_MIN, BG_DS_WATERFALL_TIMER_MAX));
61 break;
62 case BG_DS_EVENT_WATERFALL_KNOCKBACK:
63 // Repeat knockback while the waterfall still active
64 if (Creature* waterSpout = GetBGCreature(BG_DS_NPC_WATERFALL_KNOCKBACK))
65 waterSpout->CastSpell(waterSpout, BG_DS_SPELL_WATER_SPOUT, true);
66 _events.ScheduleEvent(eventId, BG_DS_WATERFALL_KNOCKBACK_TIMER);
67 break;
68 case BG_DS_EVENT_PIPE_KNOCKBACK:
69 for (uint32 i = BG_DS_NPC_PIPE_KNOCKBACK_1; i <= BG_DS_NPC_PIPE_KNOCKBACK_2; ++i)
70 if (Creature* waterSpout = GetBGCreature(i))
71 waterSpout->CastSpell(waterSpout, BG_DS_SPELL_FLUSH, true);
72 break;
73 }
74 }
75
76 if (_pipeKnockBackCount < BG_DS_PIPE_KNOCKBACK_TOTAL_COUNT)
77 {
78 if (_pipeKnockBackTimer < diff)
79 {
80 for (uint32 i = BG_DS_NPC_PIPE_KNOCKBACK_1; i <= BG_DS_NPC_PIPE_KNOCKBACK_2; ++i)
81 if (Creature* waterSpout = GetBGCreature(i))
82 waterSpout->CastSpell(waterSpout, BG_DS_SPELL_FLUSH, true);
83
84 ++_pipeKnockBackCount;
85 _pipeKnockBackTimer = BG_DS_PIPE_KNOCKBACK_DELAY;
86 }
87 else
88 _pipeKnockBackTimer -= diff;
89 }
90}

98void BattlegroundDS::StartingEventOpenDoors()
99{
100 for (uint32 i = BG_DS_OBJECT_DOOR_1; i <= BG_DS_OBJECT_DOOR_2; ++i)
101 DoorOpen(i);
102
103 for (uint32 i = BG_DS_OBJECT_BUFF_1; i <= BG_DS_OBJECT_BUFF_2; ++i)
104 SpawnBGObject(i, 60);
105
106 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_WARNING, urand(BG_DS_WATERFALL_TIMER_MIN, BG_DS_WATERFALL_TIMER_MAX));
107 //for (uint8 i = 0; i < BG_DS_PIPE_KNOCKBACK_TOTAL_COUNT; ++i)
108 // _events.ScheduleEvent(BG_DS_EVENT_PIPE_KNOCKBACK, BG_DS_PIPE_KNOCKBACK_FIRST_DELAY + i * BG_DS_PIPE_KNOCKBACK_DELAY);
109
110 _pipeKnockBackCount = 0;
111 _pipeKnockBackTimer = BG_DS_PIPE_KNOCKBACK_FIRST_DELAY;
112
113 SpawnBGObject(BG_DS_OBJECT_WATER_2, RESPAWN_IMMEDIATELY);
114
115 DoorOpen(BG_DS_OBJECT_WATER_1); // Turn off collision
116 DoorOpen(BG_DS_OBJECT_WATER_2);
117
118 // Remove effects of Demonic Circle Summon
119 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
120 if (Player* player = _GetPlayer(itr, "BattlegroundDS::StartingEventOpenDoors"))
121 player->RemoveAurasDueToSpell(SPELL_WARL_DEMONIC_CIRCLE);
122}

124void BattlegroundDS::HandleAreaTrigger(Player* player, uint32 trigger)
125{
126 if (GetStatus() != STATUS_IN_PROGRESS)
127 return;
128
129 switch (trigger)
130 {
131 case 5347:
132 case 5348:
133 // Remove effects of Demonic Circle Summon
134 player->RemoveAurasDueToSpell(SPELL_WARL_DEMONIC_CIRCLE);
135
136 // Someone has get back into the pipes and the knockback has already been performed,
137 // so we reset the knockback count for kicking the player again into the arena.
138 if (_pipeKnockBackCount >= BG_DS_PIPE_KNOCKBACK_TOTAL_COUNT)
139 _pipeKnockBackCount = 0;
140 break;
141 default:
142 Battleground::HandleAreaTrigger(player, trigger);
143 break;
144 }
145}

240struct BattlegroundABScore final : public BattlegroundScore
241{
242 friend class BattlegroundAB;
243
244 protected:
245 BattlegroundABScore(ObjectGuid playerGuid) : BattlegroundScore(playerGuid), BasesAssaulted(0), BasesDefended(0) { }
246
247 void UpdateScore(uint32 type, uint32 value) override
248 {
249 switch (type)
250 {
251 case SCORE_BASES_ASSAULTED:
252 BasesAssaulted += value;
253 break;
254 case SCORE_BASES_DEFENDED:
255 BasesDefended += value;
256 break;
257 default:
258 BattlegroundScore::UpdateScore(type, value);
259 break;
260 }
261 }
262
263 void BuildObjectivesBlock(WorldPacket& data) final override
264 {
265 data << uint32(2);
266 data << uint32(BasesAssaulted);
267 data << uint32(BasesDefended);
268 }
269
270 uint32 GetAttr1() const final override { return BasesAssaulted; }
271 uint32 GetAttr2() const final override { return BasesDefended; }
272
273 uint32 BasesAssaulted;
274 uint32 BasesDefended;
275};

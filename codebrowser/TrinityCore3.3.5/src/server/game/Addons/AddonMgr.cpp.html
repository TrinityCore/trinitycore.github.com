41void LoadFromDB()
42{
43 uint32 oldMSTime = getMSTime();
44
45 QueryResult result = CharacterDatabase.Query("SELECT name, crc FROM addons");
46 if (result)
47 {
48 uint32 count = 0;
49
50 do
51 {
52 Field* fields = result->Fetch();
53
54 std::string name = fields[0].GetString();
55 uint32 crc = fields[1].GetUInt32();
56
57 m_knownAddons.push_back(SavedAddon(name, crc));
58
59 ++count;
60 }
61 while (result->NextRow());
62
63 TC_LOG_INFO("server.loading", ">> Loaded %u known addons in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
64 }
65 else
66 TC_LOG_INFO("server.loading", ">> Loaded 0 known addons. DB table `addons` is empty!");
67
68 oldMSTime = getMSTime();
69 result = CharacterDatabase.Query("SELECT id, name, version, UNIX_TIMESTAMP(timestamp) FROM banned_addons");
70 if (result)
71 {
72 uint32 count = 0;
73 uint32 dbcMaxBannedAddon = sBannedAddOnsStore.GetNumRows();
74
75 do
76 {
77 Field* fields = result->Fetch();
78
79 BannedAddon addon;
80 addon.Id = fields[0].GetUInt32() + dbcMaxBannedAddon;
81 addon.Timestamp = uint32(fields[3].GetUInt64());
82
83 std::string name = fields[1].GetString();
84 std::string version = fields[2].GetString();
85
86 MD5(reinterpret_cast<uint8 const*>(name.c_str()), name.length(), addon.NameMD5);
87 MD5(reinterpret_cast<uint8 const*>(version.c_str()), version.length(), addon.VersionMD5);
88
89 m_bannedAddons.push_back(addon);
90
91 ++count;
92 }
93 while (result->NextRow());
94
95 TC_LOG_INFO("server.loading", ">> Loaded %u banned addons in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
96 }
97}

99void SaveAddon(AddonInfo const& addon)
100{
101 std::string name = addon.Name;
102
103 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_ADDON);
104
105 stmt->setString(0, name);
106 stmt->setUInt32(1, addon.CRC);
107
108 CharacterDatabase.Execute(stmt);
109
110 m_knownAddons.push_back(SavedAddon(addon.Name, addon.CRC));
111}
112
113SavedAddon const* GetAddonInfo(const std::string& name)
114{
115 for (SavedAddonsList::const_iterator it = m_knownAddons.begin(); it != m_knownAddons.end(); ++it)
116 {
117 SavedAddon const& addon = (*it);
118 if (addon.Name == name)
119 return &addon;
120 }
121
122 return NULL;
123}
124
125BannedAddonList const* GetBannedAddons()
126{
127 return &m_bannedAddons;
128}

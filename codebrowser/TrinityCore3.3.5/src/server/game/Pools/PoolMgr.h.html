172// Method that tell if the creature is part of a pool and return the pool id if yes
173template<>
174inline uint32 PoolMgr::IsPartOfAPool<Creature>(uint32 db_guid) const
175{
176 SearchMap::const_iterator itr = mCreatureSearchMap.find(db_guid);
177 if (itr != mCreatureSearchMap.end())
178 return itr->second;
179
180 return 0;
181}
182

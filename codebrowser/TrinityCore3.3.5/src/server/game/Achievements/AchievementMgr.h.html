48enum AchievementCriteriaDataType
49{ // value1 value2 comment
50 ACHIEVEMENT_CRITERIA_DATA_TYPE_NONE = 0, // 0 0
51 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_CREATURE = 1, // creature_id 0
52 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_CLASS_RACE = 2, // class_id race_id
53 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_LESS_HEALTH= 3, // health_percent 0
54 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_DEAD = 4, // own_team 0 not corpse (not released body), own_team == false if enemy team expected
55 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AURA = 5, // spell_id effect_idx
56 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AREA = 6, // area id 0
57 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_AURA = 7, // spell_id effect_idx
58 ACHIEVEMENT_CRITERIA_DATA_TYPE_VALUE = 8, // minvalue value provided with achievement update must be not less that limit
59 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_LEVEL = 9, // minlevel minlevel of target
60 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_GENDER = 10, // gender 0=male; 1=female
61 ACHIEVEMENT_CRITERIA_DATA_TYPE_SCRIPT = 11, // scripted requirement
62 ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_DIFFICULTY = 12, // difficulty normal/heroic difficulty for current event map
63 ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_PLAYER_COUNT = 13, // count "with less than %u people in the zone"
64 ACHIEVEMENT_CRITERIA_DATA_TYPE_T_TEAM = 14, // team HORDE(67), ALLIANCE(469)
65 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_DRUNK = 15, // drunken_state 0 (enum DrunkenState) of player
66 ACHIEVEMENT_CRITERIA_DATA_TYPE_HOLIDAY = 16, // holiday_id 0 event in holiday time
67 ACHIEVEMENT_CRITERIA_DATA_TYPE_BG_LOSS_TEAM_SCORE = 17, // min_score max_score player's team win bg and opposition team have team score in range
68 ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT = 18, // 0 0 maker instance script call for check current criteria requirements fit
69 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_EQUIPED_ITEM = 19, // item_level item_quality for equipped item in slot to check item level and quality
70 ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_ID = 20, // map_id 0 player must be on map with id in map_id
71 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_PLAYER_CLASS_RACE = 21, // class_id race_id
72 ACHIEVEMENT_CRITERIA_DATA_TYPE_NTH_BIRTHDAY = 22, // N login on day of N-th Birthday
73 ACHIEVEMENT_CRITERIA_DATA_TYPE_S_KNOWN_TITLE = 23 // title_id known (pvp) title, values from dbc
74};

78struct AchievementCriteriaData
79{
80 AchievementCriteriaDataType dataType;
81 union
82 {
83 // ACHIEVEMENT_CRITERIA_DATA_TYPE_NONE = 0 (no data)
84 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_CREATURE = 1
85 struct
86 {
87 uint32 id;
88 } creature;
89 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_CLASS_RACE = 2
90 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_PLAYER_CLASS_RACE = 21
91 struct
92 {
93 uint32 class_id;
94 uint32 race_id;
95 } classRace;
96 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_LESS_HEALTH = 3
97 struct
98 {
99 uint32 percent;
100 } health;
101 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_DEAD = 4
102 struct
103 {
104 uint32 own_team_flag;
105 } player_dead;
106 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AURA = 5
107 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_AURA = 7
108 struct
109 {
110 uint32 spell_id;
111 uint32 effect_idx;
112 } aura;
113 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AREA = 6
114 struct
115 {
116 uint32 id;
117 } area;
118 // ACHIEVEMENT_CRITERIA_DATA_TYPE_VALUE = 8
119 struct
120 {
121 uint32 value;
122 uint32 compType;
123 } value;
124 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_LEVEL = 9
125 struct
126 {
127 uint32 minlevel;
128 } level;
129 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_GENDER = 10
130 struct
131 {
132 uint32 gender;
133 } gender;
134 // ACHIEVEMENT_CRITERIA_DATA_TYPE_SCRIPT = 11 (no data)
135 // ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_DIFFICULTY = 12
136 struct
137 {
138 uint32 difficulty;
139 } difficulty;
140 // ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_PLAYER_COUNT = 13
141 struct
142 {
143 uint32 maxcount;
144 } map_players;
145 // ACHIEVEMENT_CRITERIA_DATA_TYPE_T_TEAM = 14
146 struct
147 {
148 uint32 team;
149 } team;
150 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_DRUNK = 15
151 struct
152 {
153 uint32 state;
154 } drunk;
155 // ACHIEVEMENT_CRITERIA_DATA_TYPE_HOLIDAY = 16
156 struct
157 {
158 uint32 id;
159 } holiday;
160 // ACHIEVEMENT_CRITERIA_DATA_TYPE_BG_LOSS_TEAM_SCORE= 17
161 struct
162 {
163 uint32 min_score;
164 uint32 max_score;
165 } bg_loss_team_score;
166 // ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT = 18 (no data)
167 // ACHIEVEMENT_CRITERIA_DATA_TYPE_S_EQUIPED_ITEM = 19
168 struct
169 {
170 uint32 item_level;
171 uint32 item_quality;
172 } equipped_item;
173 // ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_ID = 20
174 struct
175 {
176 uint32 mapId;
177 } map_id;
178 // ACHIEVEMENT_CRITERIA_DATA_TYPE_NTH_BIRTHDAY = 21
179 struct
180 {
181 uint32 nth_birthday;
182 } birthday_login;
183 // ACHIEVEMENT_CRITERIA_DATA_TYPE_KNOWN_TITLE = 22
184 struct
185 {
186 uint32 title_id;
187 } known_title;
188 // ...
189 struct
190 {
191 uint32 value1;
192 uint32 value2;
193 } raw;
194 };
195 uint32 ScriptId;
196
197 AchievementCriteriaData() : dataType(ACHIEVEMENT_CRITERIA_DATA_TYPE_NONE)
198 {
199 raw.value1 = 0;
200 raw.value2 = 0;
201 ScriptId = 0;
202 }
203
204 AchievementCriteriaData(uint32 _dataType, uint32 _value1, uint32 _value2, uint32 _scriptId) : dataType(AchievementCriteriaDataType(_dataType))
205 {
206 raw.value1 = _value1;
207 raw.value2 = _value2;
208 ScriptId = _scriptId;
209 }
210
211 bool IsValid(AchievementCriteriaEntry const* criteria);
212 bool Meets(uint32 criteria_id, Player const* source, Unit const* target, uint32 miscValue1 = 0) const;
213};

343 return itr != m_AchievementListByReferencedId.end() ? &itr->second : NULL;
344 }
345
346 AchievementReward const* GetAchievementReward(AchievementEntry const* achievement) const
347 {
348 AchievementRewards::const_iterator iter = m_achievementRewards.find(achievement->ID);
349 return iter != m_achievementRewards.end() ? &iter->second : NULL;
350 }
351
352 AchievementRewardLocale const* GetAchievementRewardLocale(AchievementEntry const* achievement) const
353 {
354 AchievementRewardLocales::const_iterator iter = m_achievementRewardLocales.find(achievement->ID);
355 return iter != m_achievementRewardLocales.end() ? &iter->second : NULL;
356 }
357
358 AchievementCriteriaDataSet const* GetCriteriaDataSet(AchievementCriteriaEntry const* achievementCriteria) const
359 {
360 AchievementCriteriaDataMap::const_iterator iter = m_criteriaDataMap.find(achievementCriteria->ID);
361 return iter != m_criteriaDataMap.end() ? &iter->second : NULL;
362 }
363
364 bool IsRealmCompleted(AchievementEntry const* achievement, uint32 instanceId) const
365 {
366 AllCompletedAchievements::const_iterator itr = m_allCompletedAchievements.find(achievement->ID);
367 if (itr == m_allCompletedAchievements.end())
368 return false;
369
370 if (achievement->flags & ACHIEVEMENT_FLAG_REALM_FIRST_KILL)
371 return itr->second != instanceId;
372
373 return true;
374 }
375
376 void SetRealmCompleted(AchievementEntry const* achievement, uint32 instanceId)
377 {
378 if (IsRealmCompleted(achievement, instanceId))
379 return;
380
381 m_allCompletedAchievements[achievement->ID] = instanceId;
382 }

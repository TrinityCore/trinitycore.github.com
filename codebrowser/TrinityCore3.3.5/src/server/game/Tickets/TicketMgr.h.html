50// from Blizzard LUA:
51// GMTICKET_ASSIGNEDTOGM_STATUS_NOT_ASSIGNED = 0; -- ticket is not currently assigned to a gm
52// GMTICKET_ASSIGNEDTOGM_STATUS_ASSIGNED = 1; -- ticket is assigned to a normal gm
53// GMTICKET_ASSIGNEDTOGM_STATUS_ESCALATED = 2; -- ticket is in the escalation queue
54// 3 is a custom value and should never actually be sent
55enum GMTicketEscalationStatus
56{
57 TICKET_UNASSIGNED = 0,
58 TICKET_ASSIGNED = 1,
59 TICKET_IN_ESCALATION_QUEUE = 2,

61};

89public:

94 bool IsClosed() const { return _type != TICKET_TYPE_OPEN; }
95 bool IsCompleted() const { return _completed; }
96 bool IsFromPlayer(ObjectGuid guid) const { return guid == _playerGuid; }
97 bool IsAssigned() const { return !_assignedTo.IsEmpty(); }
98 bool IsAssignedTo(ObjectGuid guid) const { return guid == _assignedTo; }
99 bool IsAssignedNotTo(ObjectGuid guid) const { return IsAssigned() && !IsAssignedTo(guid); }

101 uint32 GetId() const { return _id; }

103 std::string const& GetPlayerName() const { return _playerName; }
104 std::string const& GetMessage() const { return _message; }
105 Player* GetAssignedPlayer() const { return ObjectAccessor::FindPlayer(_assignedTo); }
106 ObjectGuid GetAssignedToGUID() const { return _assignedTo; }
107 std::string GetAssignedToName() const
108 {
109 std::string name;
110 // save queries if ticket is not assigned
111 if (_assignedTo)
112 sObjectMgr->GetPlayerNameByGUID(_assignedTo, name);
113
114 return name;
115 }
116 uint64 GetLastModifiedTime() const { return _lastModifiedTime; }
117 GMTicketEscalationStatus GetEscalatedStatus() const { return _escalatedStatus; }
118
119 void SetEscalatedStatus(GMTicketEscalationStatus escalatedStatus) { _escalatedStatus = escalatedStatus; }
120 void SetAssignedTo(ObjectGuid guid, bool isAdmin)
121 {
122 _assignedTo = guid;
123 if (isAdmin && _escalatedStatus == TICKET_IN_ESCALATION_QUEUE)
124 _escalatedStatus = TICKET_ESCALATED_ASSIGNED;
125 else if (_escalatedStatus == TICKET_UNASSIGNED)
126 _escalatedStatus = TICKET_ASSIGNED;

197 void LoadTickets();
198 void LoadSurveys();
199
200 GmTicket* GetTicket(uint32 ticketId)
201 {
202 GmTicketList::iterator itr = _ticketList.find(ticketId);
203 if (itr != _ticketList.end())
204 return itr->second;
205
206 return NULL;
207 }
208
209 GmTicket* GetTicketByPlayer(ObjectGuid playerGuid)
210 {
211 for (GmTicketList::const_iterator itr = _ticketList.begin(); itr != _ticketList.end(); ++itr)
212 if (itr->second && itr->second->IsFromPlayer(playerGuid) && !itr->second->IsClosed())
213 return itr->second;
214
215 return NULL;
216 }
217
218 GmTicket* GetOldestOpenTicket()
219 {
220 for (GmTicketList::const_iterator itr = _ticketList.begin(); itr != _ticketList.end(); ++itr)
221 if (itr->second && !itr->second->IsClosed() && !itr->second->IsCompleted())
222 return itr->second;
223
224 return NULL;
225 }

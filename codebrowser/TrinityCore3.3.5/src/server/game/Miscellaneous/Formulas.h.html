31 inline float hk_honor_at_level_f(uint8 level, float multiplier = 1.0f)
32 {
33 float honor = multiplier * level * 1.55f;
34 sScriptMgr->OnHonorCalculation(honor, level, multiplier);
35 return honor;
36 }

38 inline uint32 hk_honor_at_level(uint8 level, float multiplier = 1.0f)
39 {
40 return uint32(ceil(hk_honor_at_level_f(level, multiplier)));

46 inline uint8 GetGrayLevel(uint8 pl_level)
47 {
48 uint8 level;
49
50 if (pl_level <= 5)
51 level = 0;
52 else if (pl_level <= 39)
53 level = pl_level - 5 - pl_level / 10;
54 else if (pl_level <= 59)
55 level = pl_level - 1 - pl_level / 5;
56 else
57 level = pl_level - 9;
58
59 sScriptMgr->OnGrayLevelCalculation(level, pl_level);
60 return level;
61 }

63 inline XPColorChar GetColorCode(uint8 pl_level, uint8 mob_level)
64 {
65 XPColorChar color;
66
67 if (mob_level >= pl_level + 5)
68 color = XP_RED;
69 else if (mob_level >= pl_level + 3)
70 color = XP_ORANGE;
71 else if (mob_level >= pl_level - 2)
72 color = XP_YELLOW;
73 else if (mob_level > GetGrayLevel(pl_level))
74 color = XP_GREEN;
75 else
76 color = XP_GRAY;
77
78 sScriptMgr->OnColorCodeCalculation(color, pl_level, mob_level);
79 return color;
80 }

82 inline uint8 GetZeroDifference(uint8 pl_level)
83 {
84 uint8 diff;
85
86 if (pl_level < 8)
87 diff = 5;
88 else if (pl_level < 10)
89 diff = 6;
90 else if (pl_level < 12)
91 diff = 7;
92 else if (pl_level < 16)
93 diff = 8;
94 else if (pl_level < 20)
95 diff = 9;
96 else if (pl_level < 30)
97 diff = 11;
98 else if (pl_level < 40)
99 diff = 12;
100 else if (pl_level < 45)
101 diff = 13;
102 else if (pl_level < 50)
103 diff = 14;
104 else if (pl_level < 55)
105 diff = 15;
106 else if (pl_level < 60)
107 diff = 16;
108 else
109 diff = 17;
110
111 sScriptMgr->OnZeroDifferenceCalculation(diff, pl_level);
112 return diff;
113 }

115 inline uint32 BaseGain(uint8 pl_level, uint8 mob_level, ContentLevels content)
116 {
117 uint32 baseGain;
118 uint32 nBaseExp;
119
120 switch (content)
121 {
122 case CONTENT_1_60:
123 nBaseExp = 45;
124 break;
125 case CONTENT_61_70:
126 nBaseExp = 235;
127 break;
128 case CONTENT_71_80:
129 nBaseExp = 580;
130 break;
131 default:
132 TC_LOG_ERROR("misc", "BaseGain: Unsupported content level %u", content);
133 nBaseExp = 45;
134 break;
135 }
136
137 if (mob_level >= pl_level)
138 {
139 uint8 nLevelDiff = mob_level - pl_level;
140 if (nLevelDiff > 4)
141 nLevelDiff = 4;
142
143 baseGain = ((pl_level * 5 + nBaseExp) * (20 + nLevelDiff) / 10 + 1) / 2;
144 }
145 else
146 {
147 uint8 gray_level = GetGrayLevel(pl_level);
148 if (mob_level > gray_level)
149 {
150 uint8 ZD = GetZeroDifference(pl_level);
151 baseGain = (pl_level * 5 + nBaseExp) * (ZD + mob_level - pl_level) / ZD;
152 }
153 else
154 baseGain = 0;
155 }
156
157 sScriptMgr->OnBaseGainCalculation(baseGain, pl_level, mob_level, content);
158 return baseGain;
159 }

161 inline uint32 Gain(Player* player, Unit* u, bool isBattleGround = false)
162 {
163 Creature* creature = u->ToCreature();
164 uint32 gain = 0;
165
166 if (!creature || (!creature->IsTotem() && !creature->IsPet() && !creature->IsCritter() &&
167 !(creature->GetCreatureTemplate()->flags_extra & CREATURE_FLAG_EXTRA_NO_XP_AT_KILL)))
168 {
169 float xpMod = 1.0f;
170
171 gain = BaseGain(player->getLevel(), u->getLevel(), GetContentLevelsForMapAndZone(u->GetMapId(), u->GetZoneId()));
172
173 if (gain && creature)
174 {
175 if (creature->isElite())
176 {
177 // Elites in instances have a 2.75x XP bonus instead of the regular 2x world bonus.
178 if (u->GetMap()->IsDungeon())
179 xpMod *= 2.75f;
180 else
181 xpMod *= 2.0f;
182 }
183
184 xpMod *= creature->GetCreatureTemplate()->ModExperience;
185 }
186
187 xpMod *= isBattleGround ? sWorld->getRate(RATE_XP_BG_KILL) : sWorld->getRate(RATE_XP_KILL);
188 if (creature && creature->m_PlayerDamageReq) // if players dealt less than 50% of the damage and were credited anyway (due to CREATURE_FLAG_EXTRA_NO_PLAYER_DAMAGE_REQ), scale XP gained appropriately (linear scaling)
189 xpMod *= 1.0f - 2.0f*creature->m_PlayerDamageReq / creature->GetMaxHealth();
190
191 gain = uint32(gain * xpMod);
192 }
193
194 sScriptMgr->OnGainCalculation(gain, player, u);
195 return gain;
196 }

198 inline float xp_in_group_rate(uint32 count, bool isRaid)
199 {
200 float rate;
201
202 if (isRaid)
203 {
204 // FIXME: Must apply decrease modifiers depending on raid size.
205 rate = 1.0f;
206 }
207 else
208 {
209 switch (count)
210 {
211 case 0:
212 case 1:
213 case 2:
214 rate = 1.0f;
215 break;
216 case 3:
217 rate = 1.166f;
218 break;
219 case 4:
220 rate = 1.3f;
221 break;
222 case 5:
223 default:
224 rate = 1.4f;
225 }
226 }
227
228 sScriptMgr->OnGroupRateCalculation(rate, count, isRaid);
229 return rate;
230 }

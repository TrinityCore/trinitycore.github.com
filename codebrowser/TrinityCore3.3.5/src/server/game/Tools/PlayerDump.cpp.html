66// Low level functions
67static bool FindTokNth(std::string const& str, uint32 n, std::string::size_type& s, std::string::size_type& e)
68{
69 s = e = 0;
70
71 uint32 i = 1;
72 for (; s < str.size() && i < n; ++s)
73 if (str[s] == ' ')
74 ++i;
75
76 if (i < n)
77 return false;
78
79 e = str.find(' ', s);
80
81 return e != std::string::npos;
82}
83
84std::string GetTokNth(std::string const& str, uint32 n)
85{
86 std::string::size_type s = 0, e = 0;
87 if (!FindTokNth(str, n, s, e))
88 return "";
89
90 return str.substr(s, e - s);
91}
92
93bool FindNth(std::string const& str, uint32 n, std::string::size_type& s, std::string::size_type& e)
94{
95 s = str.find("VALUES ('") + 9;
96 if (s == std::string::npos)
97 return false;
98
99 do
100 {
101 e = str.find('\'', s);
102 if (e == std::string::npos)
103 return false;
104 } while (str[e - 1] == '\\');
105
106 for (uint32 i = 1; i < n; ++i)
107 {
108 do
109 {
110 s = e + 4;
111 e = str.find('\'', s);
112 if (e == std::string::npos)
113 return false;
114 } while (str[e - 1] == '\\');
115 }
116 return true;
117}

119std::string GetTableName(std::string const& str)
120{
121 static std::string::size_type const s = 13;
122 std::string::size_type e = str.find(_TABLE_SIM_, s);
123 if (e == std::string::npos)
124 return "";
125
126 return str.substr(s, e - s);
127}
128
129bool ChangeNth(std::string& str, uint32 n, char const* with, bool insert = false, bool allowZero = false)
130{
131 std::string::size_type s, e;
132 if (!FindNth(str, n, s, e))
133 return false;
134
135 if (allowZero && str.substr(s, e - s) == "0")
136 return true; // not an error
137
138 if (!insert)
139 str.replace(s, e - s, with);
140 else
141 str.insert(s, with);
142
143 return true;
144}

172ObjectGuid::LowType RegisterNewGuid(ObjectGuid::LowType oldGuid, PlayerDump::DumpGuidMap& guidMap, ObjectGuid::LowType guidOffset)
173{
174 PlayerDumpWriter::DumpGuidMap::const_iterator itr = guidMap.find(oldGuid);
175 if (itr != guidMap.end())
176 return itr->second;
177
178 ObjectGuid::LowType newguid = guidOffset + guidMap.size();
179 guidMap[oldGuid] = newguid;
180 return newguid;
181}
182
183bool ChangeGuid(std::string& str, uint32 n, PlayerDump::DumpGuidMap& guidMap, ObjectGuid::LowType guidOffset, bool allowZero = false)
184{
185 ObjectGuid::LowType oldGuid = strtoull(GetNth(str, n).c_str(), nullptr, 10);
186 if (allowZero && !oldGuid)
187 return true; // not an error
188
189 char chritem[20];
190 ObjectGuid::LowType newGuid = RegisterNewGuid(oldGuid, guidMap, guidOffset);
191 snprintf(chritem, 20, "%u", newGuid);
192
193 return ChangeNth(str, n, chritem, false, allowZero);
194}

196std::string CreateDumpString(char const* tableName, QueryResult result)
197{
198 if (!tableName || !result) return "";
199 std::ostringstream ss;
200 ss << "INSERT INTO " << _TABLE_SIM_ << tableName << _TABLE_SIM_ << " VALUES (";
201 Field* fields = result->Fetch();
202 for (uint32 i = 0; i < result->GetFieldCount(); ++i)
203 {
204 if (i == 0) ss << '\'';
205 else ss << ", '";
206
207 std::string s = fields[i].GetString();
208 CharacterDatabase.EscapeString(s);
209 ss << s;
210
211 ss << '\'';
212 }
213 ss << ");";
214 return ss.str();
215}

217std::string PlayerDumpWriter::GenerateWhereStr(char const* field, ObjectGuid::LowType guid)
218{
219 std::ostringstream wherestr;
220 wherestr << field << " = '" << guid << '\'';
221 return wherestr.str();

224std::string PlayerDumpWriter::GenerateWhereStr(char const* field, DumpGuidSet const& guids, DumpGuidSet::const_iterator& itr)
225{
226 std::ostringstream wherestr;
227 wherestr << field << " IN ('";
228 for (; itr != guids.end();)
229 {
230 wherestr << *itr;
231 ++itr;
232
233 if (wherestr.str().size() > MAX_QUERY_LEN - 50) // near to max query
234 break;
235
236 if (itr != guids.end())
237 wherestr << "', '";
238 }
239 wherestr << "')";
240 return wherestr.str();
241}

245 Field* fields = result->Fetch();

247 if (guid)

251void StoreGUID(QueryResult result, uint32 data, uint32 field, PlayerDump::DumpGuidSet& guids)

255 ObjectGuid::LowType guid = strtoull(GetTokNth(dataStr, field).c_str(), nullptr, 10);
256 if (guid)
257 guids.insert(guid);
258}

260// Writing - High-level functions
261bool PlayerDumpWriter::DumpTable(std::string& dump, ObjectGuid::LowType guid, char const* tableFrom, char const* tableTo, DumpTableType type)
262{
263 DumpGuidSet const* guids = nullptr;
264 char const* fieldname = nullptr;
265
266 switch (type)
267 {
268 case DTT_ITEM: fieldname = "guid"; guids = &items; break;
269 case DTT_ITEM_GIFT: fieldname = "item_guid"; guids = &items; break;
270 case DTT_PET: fieldname = "owner"; break;
271 case DTT_PET_TABLE: fieldname = "guid"; guids = &pets; break;
272 case DTT_MAIL: fieldname = "receiver"; break;
273 case DTT_MAIL_ITEM: fieldname = "mail_id"; guids = &mails; break;
274 default: fieldname = "guid"; break;
275 }
276
277 // for guid set stop if set is empty
278 if (guids && guids->empty())
279 return true; // nothing to do
280
281 // setup for guids case start position
282 DumpGuidSet::const_iterator guidsItr;
283 if (guids)
284 guidsItr = guids->begin();
285
286 do
287 {
288 std::string wherestr;
289
290 if (guids) // set case, get next guids string
291 wherestr = GenerateWhereStr(fieldname, *guids, guidsItr);
292 else // not set case, get single guid string
293 wherestr = GenerateWhereStr(fieldname, guid);
294
295 QueryResult result = CharacterDatabase.PQuery("SELECT * FROM %s WHERE %s", tableFrom, wherestr.c_str());
296 if (!result)
297 return true;
298
299 do
300 {
301 // collect guids
302 switch (type)
303 {
304 case DTT_INVENTORY:
305 StoreGUID(result, 3, items); // item guid collection (character_inventory.item)
306 break;
307 case DTT_PET:
308 StoreGUID(result, 0, pets); // pet petnumber collection (character_pet.id)
309 break;
310 case DTT_MAIL:
311 StoreGUID(result, 0, mails); // mail id collection (mail.id)
312 break;
313 case DTT_MAIL_ITEM:
314 StoreGUID(result, 1, items); // item guid collection (mail_items.item_guid)
315 break;
316 case DTT_CHARACTER:
317 {
318 if (result->GetFieldCount() <= 68) // avoid crashes on next check
319 TC_LOG_FATAL("misc", "PlayerDumpWriter::DumpTable - Trying to access non-existing or wrong positioned field (`deleteInfos_Account`) in `characters` table.");
320
321 if (result->Fetch()[68].GetUInt32()) // characters.deleteInfos_Account - if filled error
322 return false;
323 break;
324 }
325 default:
326 break;
327 }
328
329 dump += CreateDumpString(tableTo, result);
330 dump += "\n";
331 }
332 while (result->NextRow());
333 }
334 while (guids && guidsItr != guids->end()); // not set case iterate single time, set case iterate for all guids
335 return true;
336}

353DumpReturn PlayerDumpWriter::WriteDump(const std::string& file, ObjectGuid::LowType guid)
354{
355 if (sWorld->getBoolConfig(CONFIG_PDUMP_NO_PATHS))
356 if (strstr(file.c_str(), "\\") || strstr(file.c_str(), "/"))
357 return DUMP_FILE_OPEN_ERROR;
358
359 if (sWorld->getBoolConfig(CONFIG_PDUMP_NO_OVERWRITE))
360 if (FILE* f = fopen(file.c_str(), "r"))
361 {
362 fclose(f);
363 return DUMP_FILE_OPEN_ERROR;
364 }
365
366 FILE* fout = fopen(file.c_str(), "w");
367 if (!fout)
368 return DUMP_FILE_OPEN_ERROR;
369
370 DumpReturn ret = DUMP_SUCCESS;
371 std::string dump;
372 if (!GetDump(guid, dump))
373 ret = DUMP_CHARACTER_DELETED;
374
375 fprintf(fout, "%s\n", dump.c_str());
376 fclose(fout);
377 return ret;
378}

383void fixNULLfields(std::string& line)
384{
385 static std::string const nullString("'NULL'");
386 size_t pos = line.find(nullString);
387 while (pos != std::string::npos)
388 {
389 line.replace(pos, nullString.length(), "NULL");
390 pos = line.find(nullString);
391 }
392}

394DumpReturn PlayerDumpReader::LoadDump(std::string const& file, uint32 account, std::string name, ObjectGuid::LowType guid)
395{
396 uint32 charcount = AccountMgr::GetCharactersCount(account);
397 if (charcount >= 10)
398 return DUMP_TOO_MANY_CHARS;
399
400 FILE* fin = fopen(file.c_str(), "r");
401 if (!fin)
402 return DUMP_FILE_OPEN_ERROR;
403
404 char newguid[20], chraccount[20];
405
406 // make sure the same guid doesn't already exist and is safe to use
407 bool incHighest = true;
408 if (guid && guid < sObjectMgr->GetGenerator<HighGuid::Player>().GetNextAfterMaxUsed())
409
410 {
411 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHECK_GUID);
412 stmt->setUInt32(0, guid);
413 PreparedQueryResult result = CharacterDatabase.Query(stmt);
414
415 if (result)
416 guid = sObjectMgr->GetGenerator<HighGuid::Player>().GetNextAfterMaxUsed(); // use first free if exists
417
418 else
419 incHighest = false;
420 }
421 else
422 guid = sObjectMgr->GetGenerator<HighGuid::Player>().GetNextAfterMaxUsed();
423
424
425 // normalize the name if specified and check if it exists
426 if (!normalizePlayerName(name))
427 name.clear();
428
429 if (ObjectMgr::CheckPlayerName(name, sWorld->GetDefaultDbcLocale(), true) == CHAR_NAME_SUCCESS)
430 {
431 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHECK_NAME);
432 stmt->setString(0, name);
433 PreparedQueryResult result = CharacterDatabase.Query(stmt);
434
435 if (result)
436 name.clear(); // use the one from the dump
437 }
438 else
439 name.clear();
440
441 // name encoded or empty
442
443 snprintf(newguid, 20, "%u", guid);
444 snprintf(chraccount, 20, "%u", account);
445
446 DumpGuidMap items;
447 DumpGuidMap mails;
448 char buf[32000];
449 memset(buf, 0, sizeof(buf));
450
451 typedef std::map<uint32 /*old*/, uint32 /*new*/> PetIds;
452 PetIds petIds;
453
454 uint8 gender = GENDER_NONE;
455 uint8 race = RACE_NONE;
456 uint8 playerClass = 0;
457 uint8 level = 1;
458
459 ObjectGuid::LowType itemLowGuidOffset = sObjectMgr->GetGenerator<HighGuid::Item>().GetNextAfterMaxUsed();
460
461 SQLTransaction trans = CharacterDatabase.BeginTransaction();
462 while (!feof(fin))
463 {
464 if (!fgets(buf, 32000, fin))
465 {
466 if (feof(fin))
467 break;
468 ROLLBACK(DUMP_FILE_BROKEN);
469 }
470
471 std::string line; line.assign(buf);
472
473 // skip empty strings
474 size_t nw_pos = line.find_first_not_of(" \t\n\r\7");
475 if (nw_pos == std::string::npos)
476 continue;
477
478 // skip logfile-side dump start notice, the important notes and dump end notices
479 if ((line.substr(nw_pos, 16) == "== START DUMP ==") ||
480 (line.substr(nw_pos, 15) == "IMPORTANT NOTE:") ||
481 (line.substr(nw_pos, 14) == "== END DUMP =="))
482 continue;
483
484 // add required_ check
485 /*
486 if (line.substr(nw_pos, 41) == "UPDATE character_db_version SET required_")
487 {
488 if (!CharacterDatabase.Execute(line.c_str()))
489 ROLLBACK(DUMP_FILE_BROKEN);
490
491 continue;
492 }
493 */
494
495 // determine table name and load type
496 std::string tn = GetTableName(line);
497 if (tn.empty())
498 {
499 TC_LOG_ERROR("misc", "LoadPlayerDump: Can't extract table name from line: '%s'!", line.c_str());
500 ROLLBACK(DUMP_FILE_BROKEN);
501 }
502
503 DumpTableType type = DumpTableType(0);
504 uint8 i;
505 for (i = 0; i < DUMP_TABLE_COUNT; ++i)
506 {
507 if (tn == dumpTables[i].name)
508 {
509 type = dumpTables[i].type;
510 break;
511 }
512 }
513
514 if (i == DUMP_TABLE_COUNT)
515 {
516 TC_LOG_ERROR("misc", "LoadPlayerDump: Unknown table: '%s'!", tn.c_str());
517 ROLLBACK(DUMP_FILE_BROKEN);
518 }
519
520 // change the data to server values
521 switch (type)
522 {
523 case DTT_CHARACTER:
524 {
525 if (!ChangeNth(line, 1, newguid)) // characters.guid update
526 ROLLBACK(DUMP_FILE_BROKEN);
527
528 if (!ChangeNth(line, 2, chraccount)) // characters.account update
529 ROLLBACK(DUMP_FILE_BROKEN);
530
531 race = uint8(atoul(GetNth(line, 4).c_str()));
532 playerClass = uint8(atoul(GetNth(line, 5).c_str()));
533 gender = uint8(atoul(GetNth(line, 6).c_str()));
534 level = uint8(atoul(GetNth(line, 7).c_str()));
535 if (name.empty())
536 {
537 // check if the original name already exists
538 name = GetNth(line, 3);
539
540 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHECK_NAME);
541 stmt->setString(0, name);
542 PreparedQueryResult result = CharacterDatabase.Query(stmt);
543
544 if (result)
545 if (!ChangeNth(line, 37, "1")) // characters.at_login set to "rename on login"
546 ROLLBACK(DUMP_FILE_BROKEN);
547 }
548 else if (!ChangeNth(line, 3, name.c_str())) // characters.name
549 ROLLBACK(DUMP_FILE_BROKEN);
550
551 const char null[5] = "NULL";
552 if (!ChangeNth(line, 69, null)) // characters.deleteInfos_Account
553 ROLLBACK(DUMP_FILE_BROKEN);
554 if (!ChangeNth(line, 70, null)) // characters.deleteInfos_Name
555 ROLLBACK(DUMP_FILE_BROKEN);
556 if (!ChangeNth(line, 71, null)) // characters.deleteDate
557 ROLLBACK(DUMP_FILE_BROKEN);
558 break;
559 }
560 case DTT_CHAR_TABLE:
561 {
562 if (!ChangeNth(line, 1, newguid)) // character_*.guid update
563 ROLLBACK(DUMP_FILE_BROKEN);
564 break;
565 }
566 case DTT_EQSET_TABLE:
567 {
568 if (!ChangeNth(line, 1, newguid))
569 ROLLBACK(DUMP_FILE_BROKEN); // character_equipmentsets.guid
570
571 char newSetGuid[24];
572 snprintf(newSetGuid, 24, UI64FMTD, sObjectMgr->GenerateEquipmentSetGuid());
573 if (!ChangeNth(line, 2, newSetGuid))
574 ROLLBACK(DUMP_FILE_BROKEN); // character_equipmentsets.setguid
575
576 for (uint8 slot = 0; slot < EQUIPMENT_SLOT_END; ++slot)
577 if (!ChangeGuid(line, 7 + slot, items, itemLowGuidOffset, true))
578 ROLLBACK(DUMP_FILE_BROKEN); // character_equipmentsets.item
579 break;
580 }
581 case DTT_INVENTORY:
582 {
583 if (!ChangeNth(line, 1, newguid)) // character_inventory.guid update
584 ROLLBACK(DUMP_FILE_BROKEN);
585
586 if (!ChangeGuid(line, 2, items, itemLowGuidOffset, true))
587 ROLLBACK(DUMP_FILE_BROKEN); // character_inventory.bag update
588 if (!ChangeGuid(line, 4, items, itemLowGuidOffset))
589 ROLLBACK(DUMP_FILE_BROKEN); // character_inventory.item update
590 break;
591 }
592 case DTT_MAIL: // mail
593 {
594 if (!ChangeGuid(line, 1, mails, sObjectMgr->_mailId))
595 ROLLBACK(DUMP_FILE_BROKEN); // mail.id update
596 if (!ChangeNth(line, 6, newguid)) // mail.receiver update
597 ROLLBACK(DUMP_FILE_BROKEN);
598 break;
599 }
600 case DTT_MAIL_ITEM: // mail_items
601 {
602 if (!ChangeGuid(line, 1, mails, sObjectMgr->_mailId))
603 ROLLBACK(DUMP_FILE_BROKEN); // mail_items.id
604 if (!ChangeGuid(line, 2, items, itemLowGuidOffset))
605 ROLLBACK(DUMP_FILE_BROKEN); // mail_items.item_guid
606 if (!ChangeNth(line, 3, newguid)) // mail_items.receiver
607 ROLLBACK(DUMP_FILE_BROKEN);
608 break;
609 }
610 case DTT_ITEM:
611 {
612 // item, owner, data field:item, owner guid
613 if (!ChangeGuid(line, 1, items, itemLowGuidOffset))
614 ROLLBACK(DUMP_FILE_BROKEN); // item_instance.guid update
615 if (!ChangeNth(line, 3, newguid)) // item_instance.owner_guid update
616 ROLLBACK(DUMP_FILE_BROKEN);
617 break;
618 }
619 case DTT_ITEM_GIFT:
620 {
621 if (!ChangeNth(line, 1, newguid)) // character_gifts.guid update
622 ROLLBACK(DUMP_FILE_BROKEN);
623 if (!ChangeGuid(line, 2, items, itemLowGuidOffset))
624 ROLLBACK(DUMP_FILE_BROKEN); // character_gifts.item_guid update
625 break;
626 }
627 case DTT_PET:
628 {
629 // store a map of old pet id to new inserted pet id for use by DTT_PET_TABLE tables
630 std::string petIdStr = GetNth(line, 1);
631
632 uint32 currentPetId = atoul(petIdStr.c_str());
633
634 PetIds::const_iterator petIdsItr = petIds.find(currentPetId);
635 if (petIdsItr != petIds.end()) // duplicate pets
636 ROLLBACK(DUMP_FILE_BROKEN);
637
638 uint32 newPetId = sObjectMgr->GeneratePetNumber();
639 petIds[currentPetId] = newPetId;
640
641 if (!ChangeNth(line, 1, std::to_string(newPetId).c_str())) // character_pet.id update
642 ROLLBACK(DUMP_FILE_BROKEN);
643 if (!ChangeNth(line, 3, newguid)) // character_pet.owner update
644 ROLLBACK(DUMP_FILE_BROKEN);
645
646 break;
647 }
648 case DTT_PET_TABLE: // pet_aura, pet_spell, pet_spell_cooldown
649 {
650 std::string petIdStr = GetNth(line, 1);
651
652 // lookup currpetid and match to new inserted pet id
653 PetIds::const_iterator petIdsItr = petIds.find(atoul(petIdStr.c_str()));
654 if (petIdsItr == petIds.end()) // couldn't find new inserted id
655 ROLLBACK(DUMP_FILE_BROKEN);
656
657 if (!ChangeNth(line, 1, std::to_string(petIdsItr->second).c_str()))
658 ROLLBACK(DUMP_FILE_BROKEN);
659
660 break;
661 }
662 default:
663 TC_LOG_ERROR("misc", "Unknown dump table type: %u", type);
664 break;
665 }
666
667 fixNULLfields(line);
668
669 trans->Append(line.c_str());
670 }
671
672 CharacterDatabase.CommitTransaction(trans);
673
674 // in case of name conflict player has to rename at login anyway
675 sWorld->AddCharacterInfo(ObjectGuid(HighGuid::Player, guid), account, name, gender, race, playerClass, level);
676
677 sObjectMgr->GetGenerator<HighGuid::Item>().Set(sObjectMgr->GetGenerator<HighGuid::Item>().GetNextAfterMaxUsed() + items.size());
678
679 sObjectMgr->_mailId += mails.size();
680
681 if (incHighest)
682 sObjectMgr->GetGenerator<HighGuid::Player>().Generate();
683
684
685 fclose(fin);
686
687 return DUMP_SUCCESS;

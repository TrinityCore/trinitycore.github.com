123 struct PlayerInfo
124 {
125 ObjectGuid player;
126 uint8 flags;
127
128 bool HasFlag(uint8 flag) const { return (flags & flag) != 0; }
129 void SetFlag(uint8 flag) { flags |= flag; }
130 bool IsOwner() const { return (flags & MEMBER_FLAG_OWNER) != 0; }
131 void SetOwner(bool state)
132 {
133 if (state) flags |= MEMBER_FLAG_OWNER;
134 else flags &= ~MEMBER_FLAG_OWNER;
135 }
136 bool IsModerator() const { return (flags & MEMBER_FLAG_MODERATOR) != 0; }
137 void SetModerator(bool state)
138 {
139 if (state) flags |= MEMBER_FLAG_MODERATOR;
140 else flags &= ~MEMBER_FLAG_MODERATOR;
141 }
142 bool IsMuted() const { return (flags & MEMBER_FLAG_MUTED) != 0; }
143 void SetMuted(bool state)
144 {
145 if (state) flags |= MEMBER_FLAG_MUTED;
146 else flags &= ~MEMBER_FLAG_MUTED;
147 }
148 };

245 void SetModerator(ObjectGuid guid, bool set)
246 {
247 if (playersStore[guid].IsModerator() != set)
248 {
249 uint8 oldFlag = GetPlayerFlags(guid);
250 playersStore[guid].SetModerator(set);
251
252 WorldPacket data;
253 MakeModeChange(&data, guid, oldFlag);
254 SendToAll(&data);
255 }
256 }
257
258 void SetMute(ObjectGuid guid, bool set)
259 {
260 if (playersStore[guid].IsMuted() != set)
261 {
262 uint8 oldFlag = GetPlayerFlags(guid);
263 playersStore[guid].SetMuted(set);
264
265 WorldPacket data;
266 MakeModeChange(&data, guid, oldFlag);
267 SendToAll(&data);
268 }

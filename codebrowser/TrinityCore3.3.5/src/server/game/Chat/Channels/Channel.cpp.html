28Channel::Channel(std::string const& name, uint32 channelId, uint32 team):
29 _announce(true),
30 _ownership(true),
31 _IsSaved(false),
32 _flags(0),
33 _channelId(channelId),
34 _Team(team),
35 _ownerGUID(),
36 _name(name),
37 _password("")
38{
39 // set special flags if built-in channel
40 if (ChatChannelsEntry const* ch = sChatChannelsStore.LookupEntry(channelId)) // check whether it's a built-in channel
41 {
42 _announce = false; // no join/leave announces
43 _ownership = false; // no ownership handout
44
45 _flags |= CHANNEL_FLAG_GENERAL; // for all built-in channels
46
47 if (ch->flags & CHANNEL_DBC_FLAG_TRADE) // for trade channel
48 _flags |= CHANNEL_FLAG_TRADE;
49
50 if (ch->flags & CHANNEL_DBC_FLAG_CITY_ONLY2) // for city only channels
51 _flags |= CHANNEL_FLAG_CITY;
52
53 if (ch->flags & CHANNEL_DBC_FLAG_LFG) // for LFG channel
54 _flags |= CHANNEL_FLAG_LFG;
55 else // for all other channels
56 _flags |= CHANNEL_FLAG_NOT_LFG;
57 }
58 else // it's custom channel
59 {
60 _flags |= CHANNEL_FLAG_CUSTOM;
61
62 // If storing custom channels in the db is enabled either load or save the channel
63 if (sWorld->getBoolConfig(CONFIG_PRESERVE_CUSTOM_CHANNELS))
64 {
65 PreparedStatement *stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHANNEL);
66 stmt->setString(0, name);
67 stmt->setUInt32(1, _Team);
68 PreparedQueryResult result = CharacterDatabase.Query(stmt);
69
70 if (result) //load
71 {
72 Field* fields = result->Fetch();
73 _announce = fields[0].GetBool();
74 _ownership = fields[1].GetBool();
75 _password = fields[2].GetString();
76 const char* db_BannedList = fields[3].GetCString();
77
78 if (db_BannedList)
79 {
80 Tokenizer tokens(db_BannedList, ' ');
81 for (Tokenizer::const_iterator i = tokens.begin(); i != tokens.end(); ++i)
82 {
83 ObjectGuid banned_guid(uint64(strtoull(*i, NULL, 10)));
84 if (banned_guid)
85 {
86 TC_LOG_DEBUG("chat.system", "Channel(%s) loaded bannedStore %s", name.c_str(), banned_guid.ToString().c_str());
87 bannedStore.insert(banned_guid);
88 }
89 }
90 }
91 }
92 else // save
93 {
94 stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_CHANNEL);
95 stmt->setString(0, name);
96 stmt->setUInt32(1, _Team);
97 CharacterDatabase.Execute(stmt);
98 TC_LOG_DEBUG("chat.system", "Channel(%s) saved in database", name.c_str());
99 }
100
101 _IsSaved = true;
102 }
103 }
104}

106void Channel::UpdateChannelInDB() const
107{
108 if (_IsSaved)
109 {
110 std::ostringstream banlist;
111 BannedContainer::const_iterator iter;
112 for (iter = bannedStore.begin(); iter != bannedStore.end(); ++iter)
113 banlist << iter->GetRawValue() << ' ';
114
115 std::string banListStr = banlist.str();
116
117 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_CHANNEL);
118 stmt->setBool(0, _announce);
119 stmt->setBool(1, _ownership);
120 stmt->setString(2, _password);
121 stmt->setString(3, banListStr);
122 stmt->setString(4, _name);
123 stmt->setUInt32(5, _Team);
124 CharacterDatabase.Execute(stmt);
125
126 TC_LOG_DEBUG("chat.system", "Channel(%s) updated in database", _name.c_str());
127 }
128}

150void Channel::JoinChannel(Player* player, std::string const& pass)
151{
152 ObjectGuid guid = player->GetGUID();
153 if (IsOn(guid))
154 {
155 // Do not send error message for built-in channels
156 if (!IsConstant())
157 {
158 WorldPacket data;
159 MakePlayerAlreadyMember(&data, guid);
160 SendToOne(&data, guid);
161 }
162 return;
163 }
164
165 if (IsBanned(guid))
166 {
167 WorldPacket data;
168 MakeBanned(&data);
169 SendToOne(&data, guid);
170 return;
171 }
172
173 if (!_password.empty() && pass != _password)
174 {
175 WorldPacket data;
176 MakeWrongPassword(&data);
177 SendToOne(&data, guid);
178 return;
179 }
180
181 if (HasFlag(CHANNEL_FLAG_LFG) &&
182 sWorld->getBoolConfig(CONFIG_RESTRICTED_LFG_CHANNEL) &&
183 AccountMgr::IsPlayerAccount(player->GetSession()->GetSecurity()) && //FIXME: Move to RBAC
184 player->GetGroup())
185 {
186 WorldPacket data;
187 MakeNotInLfg(&data);
188 SendToOne(&data, guid);
189 return;
190 }
191
192 player->JoinedChannel(this);
193
194 if (_announce && !player->GetSession()->HasPermission(rbac::RBAC_PERM_SILENTLY_JOIN_CHANNEL))
195 {
196 WorldPacket data;
197 MakeJoined(&data, guid);
198 SendToAll(&data);
199 }
200
201 PlayerInfo pinfo;
202 pinfo.player = guid;
203 pinfo.flags = MEMBER_FLAG_NONE;
204 playersStore[guid] = pinfo;
205
206 WorldPacket data;
207 MakeYouJoined(&data);
208 SendToOne(&data, guid);
209
210 JoinNotify(guid);
211
212 // Custom channel handling
213 if (!IsConstant())
214 {
215 // Update last_used timestamp in db
216 if (!playersStore.empty())
217 UpdateChannelUseageInDB();
218
219 // If the channel has no owner yet and ownership is allowed, set the new owner.
220 if (!_ownerGUID && _ownership)
221 {
222 SetOwner(guid, playersStore.size() > 1);
223 playersStore[guid].SetModerator(true);
224 }
225 }

228void Channel::LeaveChannel(Player* player, bool send)
229{
230 ObjectGuid guid = player->GetGUID();
231 if (!IsOn(guid))
232 {
233 if (send)
234 {
235 WorldPacket data;
236 MakeNotMember(&data);
237 SendToOne(&data, guid);
238 }
239 return;
240 }
241
242 if (send)
243 {
244 WorldPacket data;
245 MakeYouLeft(&data);
246 SendToOne(&data, guid);
247 player->LeftChannel(this);
248 data.clear();
249 }
250
251 bool changeowner = playersStore[guid].IsOwner();
252
253 playersStore.erase(guid);
254
255 if (_announce && !player->GetSession()->HasPermission(rbac::RBAC_PERM_SILENTLY_JOIN_CHANNEL))
256 {
257 WorldPacket data;
258 MakeLeft(&data, guid);
259 SendToAll(&data);
260 }
261
262 LeaveNotify(guid);
263
264 if (!IsConstant())
265 {
266 // Update last_used timestamp in db
267 UpdateChannelUseageInDB();
268
269 // If the channel owner left and there are still playersStore inside, pick a new owner
270 if (changeowner && _ownership && !playersStore.empty())
271 {
272 ObjectGuid newowner = playersStore.begin()->second.player;
273 playersStore[newowner].SetModerator(true);
274 SetOwner(newowner);
275 }
276 }

279void Channel::KickOrBan(Player const* player, std::string const& badname, bool ban)
280{
281 ObjectGuid good = player->GetGUID();
282
283 if (!IsOn(good))
284 {
285 WorldPacket data;
286 MakeNotMember(&data);
287 SendToOne(&data, good);
288 return;
289 }
290
291 if (!playersStore[good].IsModerator() && !player->GetSession()->HasPermission(rbac::RBAC_PERM_CHANGE_CHANNEL_NOT_MODERATOR))
292 {
293 WorldPacket data;
294 MakeNotModerator(&data);
295 SendToOne(&data, good);
296 return;
297 }
298
299 Player* bad = ObjectAccessor::FindConnectedPlayerByName(badname);
300 ObjectGuid victim = bad ? bad->GetGUID() : ObjectGuid::Empty;
301 if (!victim || !IsOn(victim))
302 {
303 WorldPacket data;
304 MakePlayerNotFound(&data, badname);
305 SendToOne(&data, good);
306 return;
307 }
308
309 bool changeowner = _ownerGUID == victim;
310
311 if (!player->GetSession()->HasPermission(rbac::RBAC_PERM_CHANGE_CHANNEL_NOT_MODERATOR) && changeowner && good != _ownerGUID)
312 {
313 WorldPacket data;
314 MakeNotOwner(&data);
315 SendToOne(&data, good);
316 return;
317 }
318
319 if (ban && !IsBanned(victim))
320 {
321 bannedStore.insert(victim);
322 UpdateChannelInDB();
323
324 if (!player->GetSession()->HasPermission(rbac::RBAC_PERM_SILENTLY_JOIN_CHANNEL))
325 {
326 WorldPacket data;
327 MakePlayerBanned(&data, victim, good);
328 SendToAll(&data);
329 }
330 }
331 else if (!player->GetSession()->HasPermission(rbac::RBAC_PERM_SILENTLY_JOIN_CHANNEL))
332 {
333 WorldPacket data;
334 MakePlayerKicked(&data, victim, good);
335 SendToAll(&data);
336 }
337
338 playersStore.erase(victim);
339 bad->LeftChannel(this);
340
341 if (changeowner && _ownership && !playersStore.empty())
342 {
343 ObjectGuid newowner = good;
344 playersStore[newowner].SetModerator(true);
345 SetOwner(newowner);
346 }
347}

419void Channel::SetMode(Player const* player, std::string const& p2n, bool mod, bool set)
420{
421 ObjectGuid guid = player->GetGUID();
422
423 if (!IsOn(guid))
424 {
425 WorldPacket data;
426 MakeNotMember(&data);
427 SendToOne(&data, guid);
428 return;
429 }
430
431 if (!playersStore[guid].IsModerator() && !player->GetSession()->HasPermission(rbac::RBAC_PERM_CHANGE_CHANNEL_NOT_MODERATOR))
432 {
433 WorldPacket data;
434 MakeNotModerator(&data);
435 SendToOne(&data, guid);
436 return;
437 }
438
439 if (guid == _ownerGUID && std::string(p2n) == player->GetName() && mod)
440 return;
441
442 Player* newp = ObjectAccessor::FindConnectedPlayerByName(p2n);
443 ObjectGuid victim = newp ? newp->GetGUID() : ObjectGuid::Empty;
444
445 if (!victim || !IsOn(victim) ||
446 (player->GetTeam() != newp->GetTeam() &&
447 (!player->GetSession()->HasPermission(rbac::RBAC_PERM_TWO_SIDE_INTERACTION_CHANNEL) ||
448 !newp->GetSession()->HasPermission(rbac::RBAC_PERM_TWO_SIDE_INTERACTION_CHANNEL))))
449 {
450 WorldPacket data;
451 MakePlayerNotFound(&data, p2n);
452 SendToOne(&data, guid);
453 return;
454 }
455
456 if (_ownerGUID == victim && _ownerGUID != guid)
457 {
458 WorldPacket data;
459 MakeNotOwner(&data);
460 SendToOne(&data, guid);
461 return;
462 }
463
464 if (mod)
465 SetModerator(newp->GetGUID(), set);
466 else
467 SetMute(newp->GetGUID(), set);
468}

518void Channel::List(Player const* player)
519{
520 ObjectGuid guid = player->GetGUID();
521
522 if (!IsOn(guid))
523 {
524 WorldPacket data;
525 MakeNotMember(&data);
526 SendToOne(&data, guid);
527 return;
528 }
529
530 TC_LOG_DEBUG("chat.system", "SMSG_CHANNEL_LIST %s Channel: %s",
531 player->GetSession()->GetPlayerInfo().c_str(), GetName().c_str());
532
533 WorldPacket data(SMSG_CHANNEL_LIST, 1+(GetName().size()+1)+1+4+playersStore.size()*(8+1));
534 data << uint8(1); // channel type?
535 data << GetName(); // channel name
536 data << uint8(GetFlags()); // channel flags?
537
538 size_t pos = data.wpos();
539 data << uint32(0); // size of list, placeholder
540
541 uint32 gmLevelInWhoList = sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_WHO_LIST);
542
543 uint32 count = 0;
544 for (PlayerContainer::const_iterator i = playersStore.begin(); i != playersStore.end(); ++i)
545 {
546 Player* member = ObjectAccessor::FindConnectedPlayer(i->first);
547
548 // PLAYER can't see MODERATOR, GAME MASTER, ADMINISTRATOR characters
549 // MODERATOR, GAME MASTER, ADMINISTRATOR can see all
550 if (member &&
551 (player->GetSession()->HasPermission(rbac::RBAC_PERM_WHO_SEE_ALL_SEC_LEVELS) ||
552 member->GetSession()->GetSecurity() <= AccountTypes(gmLevelInWhoList)) &&
553 member->IsVisibleGloballyFor(player))
554 {
555 data << uint64(i->first);
556 data << uint8(i->second.flags); // flags seems to be changed...
557 ++count;
558 }
559 }
560
561 data.put<uint32>(pos, count);
562
563 SendToOne(&data, guid);
564}

598void Channel::Say(ObjectGuid guid, std::string const& what, uint32 lang)
599{
600 if (what.empty())
601 return;
602
603 // TODO: Add proper RBAC check
604 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_CHANNEL))
605 lang = LANG_UNIVERSAL;
606
607 if (!IsOn(guid))
608 {
609 WorldPacket data;
610 MakeNotMember(&data);
611 SendToOne(&data, guid);
612 return;
613 }
614
615 if (playersStore[guid].IsMuted())
616 {
617 WorldPacket data;
618 MakeMuted(&data);
619 SendToOne(&data, guid);
620 return;
621 }
622
623 WorldPacket data;
624 if (Player* player = ObjectAccessor::FindConnectedPlayer(guid))
625 ChatHandler::BuildChatPacket(data, CHAT_MSG_CHANNEL, Language(lang), player, player, what, 0, _name);
626 else
627 ChatHandler::BuildChatPacket(data, CHAT_MSG_CHANNEL, Language(lang), guid, guid, what, 0, "", "", 0, false, _name);
628
629 SendToAll(&data, !playersStore[guid].IsModerator() ? guid : ObjectGuid::Empty);
630}

632void Channel::Invite(Player const* player, std::string const& newname)
633{
634 ObjectGuid guid = player->GetGUID();
635
636 if (!IsOn(guid))
637 {
638 WorldPacket data;
639 MakeNotMember(&data);
640 SendToOne(&data, guid);
641 return;
642 }
643
644 Player* newp = ObjectAccessor::FindConnectedPlayerByName(newname);
645 if (!newp || !newp->isGMVisible())
646 {
647 WorldPacket data;
648 MakePlayerNotFound(&data, newname);
649 SendToOne(&data, guid);
650 return;
651 }
652
653 if (IsBanned(newp->GetGUID()))
654 {
655 WorldPacket data;
656 MakePlayerInviteBanned(&data, newname);
657 SendToOne(&data, guid);
658 return;
659 }
660
661 if (newp->GetTeam() != player->GetTeam() &&
662 (!player->GetSession()->HasPermission(rbac::RBAC_PERM_TWO_SIDE_INTERACTION_CHANNEL) ||
663 !newp->GetSession()->HasPermission(rbac::RBAC_PERM_TWO_SIDE_INTERACTION_CHANNEL)))
664 {
665 WorldPacket data;
666 MakeInviteWrongFaction(&data);
667 SendToOne(&data, guid);
668 return;
669 }
670
671 if (IsOn(newp->GetGUID()))
672 {
673 WorldPacket data;
674 MakePlayerAlreadyMember(&data, newp->GetGUID());
675 SendToOne(&data, guid);
676 return;
677 }
678
679 if (!newp->GetSocial()->HasIgnore(guid.GetCounter()))
680 {
681 WorldPacket data;
682 MakeInvite(&data, guid);
683 SendToOne(&data, newp->GetGUID());
684 data.clear();
685 }
686
687 WorldPacket data;
688 MakePlayerInvited(&data, newp->GetName());
689 SendToOne(&data, guid);
690}

692void Channel::SetOwner(ObjectGuid guid, bool exclaim)
693{
694 if (_ownerGUID)
695 {
696 // [] will re-add player after it possible removed
697 PlayerContainer::iterator p_itr = playersStore.find(_ownerGUID);
698 if (p_itr != playersStore.end())
699 p_itr->second.SetOwner(false);
700 }
701
702 _ownerGUID = guid;
703 if (_ownerGUID)
704 {
705 uint8 oldFlag = GetPlayerFlags(_ownerGUID);
706 playersStore[_ownerGUID].SetModerator(true);
707 playersStore[_ownerGUID].SetOwner(true);
708
709 WorldPacket data;
710 MakeModeChange(&data, _ownerGUID, oldFlag);
711 SendToAll(&data);
712
713 if (exclaim)
714 {
715 MakeOwnerChanged(&data, _ownerGUID);
716 SendToAll(&data);
717 }
718
719 UpdateChannelInDB();
720 }
721}

966void Channel::JoinNotify(ObjectGuid guid)
967{
968 WorldPacket data(IsConstant() ? SMSG_USERLIST_ADD : SMSG_USERLIST_UPDATE, 8 + 1 + 1 + 4 + GetName().size());
969 data << uint64(guid);
970 data << uint8(GetPlayerFlags(guid));
971 data << uint8(GetFlags());
972 data << uint32(GetNumPlayers());
973 data << GetName();
974
975 if (IsConstant())
976 SendToAllButOne(&data, guid);
977 else
978 SendToAll(&data);
979}

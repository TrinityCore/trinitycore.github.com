23ChannelMgr::~ChannelMgr()
24{
25 for (ChannelMap::iterator itr = channels.begin(); itr != channels.end(); ++itr)
26 delete itr->second;
27}

29ChannelMgr* ChannelMgr::forTeam(uint32 team)
30{
31 static ChannelMgr allianceChannelMgr;
32 static ChannelMgr hordeChannelMgr;
33 if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_CHANNEL))
34 return &allianceChannelMgr; // cross-faction
35
36 if (team == ALLIANCE)
37 return &allianceChannelMgr;
38
39 if (team == HORDE)
40 return &hordeChannelMgr;
41
42 return NULL;
43}

45Channel* ChannelMgr::GetJoinChannel(std::string const& name, uint32 channelId)
46{
47 std::wstring wname;
48 if (!Utf8toWStr(name, wname))
49 return NULL;
50
51 wstrToLower(wname);
52
53 ChannelMap::const_iterator i = channels.find(wname);
54
55 if (i == channels.end())
56 {
57 Channel* nchan = new Channel(name, channelId, team);
58 channels[wname] = nchan;
59 return nchan;
60 }
61
62 return i->second;
63}

65Channel* ChannelMgr::GetChannel(std::string const& name, Player* player, bool pkt)
66{
67 std::wstring wname;
68 if (!Utf8toWStr(name, wname))
69 return NULL;
70
71 wstrToLower(wname);
72
73 ChannelMap::const_iterator i = channels.find(wname);
74
75 if (i == channels.end())
76 {
77 if (pkt)
78 {
79 WorldPacket data;
80 MakeNotOnPacket(&data, name);
81 player->GetSession()->SendPacket(&data);
82 }
83
84 return NULL;
85 }
86
87 return i->second;
88}

90void ChannelMgr::LeftChannel(std::string const& name)
91{
92 std::wstring wname;
93 if (!Utf8toWStr(name, wname))
94 return;
95
96 wstrToLower(wname);
97
98 ChannelMap::const_iterator i = channels.find(wname);
99
100 if (i == channels.end())
101 return;
102
103 Channel* channel = i->second;
104
105 if (!channel->GetNumPlayers() && !channel->IsConstant())
106 {
107 channels.erase(wname);
108 delete channel;
109 }
110}

112void ChannelMgr::MakeNotOnPacket(WorldPacket* data, std::string const& name)
113{
114 data->Initialize(SMSG_CHANNEL_NOTIFY, 1 + name.size());
115 (*data) << uint8(5) << name;
116}

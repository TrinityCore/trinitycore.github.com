49inline bool ReadUInt32(std::istringstream& iss, uint32& res)
50{
51 iss >> std::dec >> res;
52 return !iss.fail() && !iss.eof();
53}
54
55inline bool ReadInt32(std::istringstream& iss, int32& res)
56{
57 iss >> std::dec >> res;
58 return !iss.fail() && !iss.eof();
59}
60
61inline std::string ReadSkip(std::istringstream& iss, char term)
62{
63 std::string res;
64 char c = iss.peek();
65 while (c != term && c != '\0')
66 {
67 res += c;
68 iss.ignore(1);
69 c = iss.peek();
70 }
71 return res;
72}
73
74inline bool CheckDelimiter(std::istringstream& iss, char delimiter, const char* context)
75{
76 char c = iss.peek();
77 if (c != delimiter)
78 {
79 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): invalid %s link structure ('%c' expected, '%c' found)", iss.str().c_str(), context, delimiter, c);
80 return false;
81 }
82 iss.ignore(1);
83 return true;
84}
85
86inline bool ReadHex(std::istringstream& iss, uint32& res, uint32 length)
87{
88 std::istringstream::pos_type pos = iss.tellg();
89 iss >> std::hex >> res;
90 //uint32 size = uint32(iss.gcount());
91 if (length && uint32(iss.tellg() - pos) != length)
92 return false;
93 return !iss.fail() && !iss.eof();
94}

105// |color|Hitem:item_id:perm_ench_id:gem1:gem2:gem3:0:random_property:0:reporter_level|h[name]|h|r
106// |cffa335ee|Hitem:812:0:0:0:0:0:0:0:70|h[Glowing Brightwood Staff]|h|r
107bool ItemChatLink::Initialize(std::istringstream& iss)
108{
109 // Read item entry
110 uint32 itemEntry = 0;
111 if (!ReadUInt32(iss, itemEntry))
112 {
113 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): sequence finished unexpectedly while reading item entry", iss.str().c_str());
114 return false;
115 }
116 // Validate item
117 _item = sObjectMgr->GetItemTemplate(itemEntry);
118 if (!_item)
119 {
120 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): got invalid itemEntry %u in |item command", iss.str().c_str(), itemEntry);
121 return false;
122 }
123 // Validate item's color
124 if (_color != ItemQualityColors[_item->Quality])
125 {
126 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): linked item has color %u, but user claims %u", iss.str().c_str(), ItemQualityColors[_item->Quality], _color);
127 return false;
128 }
129 // Number of various item properties after item entry
130 const uint8 propsCount = 8;
131 const uint8 randomPropertyPosition = 5;
132 for (uint8 index = 0; index < propsCount; ++index)
133 {
134 if (!CheckDelimiter(iss, DELIMITER, "item"))
135 return false;
136
137 int32 id = 0;
138 if (!ReadInt32(iss, id))
139 {
140 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): sequence finished unexpectedly while reading item property (%u)", iss.str().c_str(), index);
141 return false;
142 }
143 if (id && (index == randomPropertyPosition))
144 {
145 // Validate random property
146 if (id > 0)
147 {
148 _property = sItemRandomPropertiesStore.LookupEntry(id);
149 if (!_property)
150 {
151 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): got invalid item property id %u in |item command", iss.str().c_str(), id);
152 return false;
153 }
154 }
155 else if (id < 0)
156 {
157 _suffix = sItemRandomSuffixStore.LookupEntry(-id);
158 if (!_suffix)
159 {
160 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): got invalid item suffix id %u in |item command", iss.str().c_str(), -id);
161 return false;
162 }
163 }
164 }
165 _data[index] = id;
166 }
167 return true;
168}

170inline std::string ItemChatLink::FormatName(uint8 index, ItemLocale const* locale, char* const* suffixStrings) const
171{
172 std::stringstream ss;
173 if (locale == NULL || index >= locale->Name.size())
174 ss << _item->Name1;
175 else
176 ss << locale->Name[index];
177 if (suffixStrings)
178 ss << ' ' << suffixStrings[index];
179 return ss.str();
180}

182bool ItemChatLink::ValidateName(char* buffer, const char* context)
183{
184 ChatLink::ValidateName(buffer, context);
185
186 char* const* suffixStrings = _suffix ? _suffix->nameSuffix : (_property ? _property->nameSuffix : NULL);
187
188 bool res = (FormatName(LOCALE_enUS, NULL, suffixStrings) == buffer);
189 if (!res)
190 {
191 ItemLocale const* il = sObjectMgr->GetItemLocale(_item->ItemId);
192 for (uint8 index = LOCALE_koKR; index < TOTAL_LOCALES; ++index)
193 {
194 if (FormatName(index, il, suffixStrings) == buffer)
195 {
196 res = true;
197 break;
198 }
199 }
200 }
201 if (!res)
202 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): linked item (id: %u) name wasn't found in any localization", context, _item->ItemId);
203 return res;
204}

283bool SpellChatLink::ValidateName(char* buffer, const char* context)
284{
285 ChatLink::ValidateName(buffer, context);
286
287 // spells with that flag have a prefix of "$PROFESSION: "
288 if (_spell->HasAttribute(SPELL_ATTR0_TRADESPELL))
289 {
290 SkillLineAbilityMapBounds bounds = sSpellMgr->GetSkillLineAbilityMapBounds(_spell->Id);
291 if (bounds.first == bounds.second)
292 {
293 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): skill line not found for spell %u", context, _spell->Id);
294 return false;
295 }
296 SkillLineAbilityEntry const* skillInfo = bounds.first->second;
297 if (!skillInfo)
298 {
299 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): skill line ability not found for spell %u", context, _spell->Id);
300 return false;
301 }
302 SkillLineEntry const* skillLine = sSkillLineStore.LookupEntry(skillInfo->skillId);
303 if (!skillLine)
304 {
305 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): skill line not found for skill %u", context, skillInfo->skillId);
306 return false;
307 }
308
309 for (uint8 i = 0; i < TOTAL_LOCALES; ++i)
310 {
311 uint32 skillLineNameLength = strlen(skillLine->name[i]);
312 if (skillLineNameLength > 0 && strncmp(skillLine->name[i], buffer, skillLineNameLength) == 0)
313 {
314 // found the prefix, remove it to perform spellname validation below
315 // -2 = strlen(": ")
316 uint32 spellNameLength = strlen(buffer) - skillLineNameLength - 2;
317 memmove(buffer, buffer + skillLineNameLength + 2, spellNameLength + 1);
318 break;
319 }
320 }
321 }
322
323 for (uint8 i = 0; i < TOTAL_LOCALES; ++i)
324 if (*_spell->SpellName[i] && strcmp(_spell->SpellName[i], buffer) == 0)
325 return true;
326
327 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): linked spell (id: %u) name wasn't found in any localization", context, _spell->Id);
328 return false;
329}

331// |color|Hachievement:achievement_id:player_guid:0:0:0:0:0:0:0:0|h[name]|h|r
332// |cffffff00|Hachievement:546:0000000000000001:0:0:0:-1:0:0:0:0|h[Safe Deposit]|h|r
333bool AchievementChatLink::Initialize(std::istringstream& iss)
334{
335 if (_color != CHAT_LINK_COLOR_ACHIEVEMENT)
336 return false;
337 // Read achievemnt Id
338 uint32 achievementId = 0;
339 if (!ReadUInt32(iss, achievementId))
340 {
341 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): sequence finished unexpectedly while reading achievement entry", iss.str().c_str());
342 return false;
343 }
344 // Validate achievement
345 _achievement = sAchievementMgr->GetAchievement(achievementId);
346 if (!_achievement)
347 {
348 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): got invalid achivement id %u in |achievement command", iss.str().c_str(), achievementId);
349 return false;
350 }
351 // Check delimiter
352 if (!CheckDelimiter(iss, DELIMITER, "achievement"))
353 return false;
354 // Read HEX
355 if (!ReadHex(iss, _guid, 0))
356 {
357 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): invalid hexadecimal number while reading char's guid", iss.str().c_str());
358 return false;
359 }
360 // Skip progress
361 const uint8 propsCount = 8;
362 for (uint8 index = 0; index < propsCount; ++index)
363 {
364 if (!CheckDelimiter(iss, DELIMITER, "achievement"))
365 return false;
366
367 if (!ReadUInt32(iss, _data[index]))
368 {
369 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): sequence finished unexpectedly while reading achievement property (%u)", iss.str().c_str(), index);
370 return false;
371 }
372 }
373 return true;
374}

549bool LinkExtractor::IsValidMessage()
550{
551 const char validSequence[6] = "cHhhr";
552 const char* validSequenceIterator = validSequence;
553
554 char buffer[256];
555
556 std::istringstream::pos_type startPos = 0;
557 uint32 color = 0;
558
559 ChatLink* link = NULL;
560 while (!_iss.eof())
561 {
562 if (validSequence == validSequenceIterator)
563 {
564 link = NULL;
565 _iss.ignore(255, PIPE_CHAR);
566 startPos = _iss.tellg() - std::istringstream::pos_type(1);
567 }
568 else if (_iss.get() != PIPE_CHAR)
569 {
570 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): sequence aborted unexpectedly", _iss.str().c_str());
571 return false;
572 }
573
574 // pipe has always to be followed by at least one char
575 if (_iss.peek() == '\0')
576 {
577 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): pipe followed by '\\0'", _iss.str().c_str());
578 return false;
579 }
580
581 // no further pipe commands
582 if (_iss.eof())
583 break;
584
585 char commandChar;
586 _iss >> commandChar;
587
588 // | in normal messages is escaped by ||
589 if (commandChar != PIPE_CHAR)
590 {
591 if (commandChar == *validSequenceIterator)
592 {
593 if (validSequenceIterator == validSequence+4)
594 validSequenceIterator = validSequence;
595 else
596 ++validSequenceIterator;
597 }
598 else
599 {
600 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): invalid sequence, expected '%c' but got '%c'", _iss.str().c_str(), *validSequenceIterator, commandChar);
601 return false;
602 }
603 }
604 else if (validSequence != validSequenceIterator)
605 {
606 // no escaped pipes in sequences
607 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): got escaped pipe in sequence", _iss.str().c_str());
608 return false;
609 }
610
611 switch (commandChar)
612 {
613 case 'c':
614 if (!ReadHex(_iss, color, 8))
615 {
616 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): invalid hexadecimal number while reading color", _iss.str().c_str());
617 return false;
618 }
619 break;
620 case 'H':
621 // read chars up to colon = link type
622 _iss.getline(buffer, 256, DELIMITER);
623 if (_iss.eof())
624 {
625 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): sequence finished unexpectedly", _iss.str().c_str());
626 return false;
627 }
628
629 if (strcmp(buffer, "item") == 0)
630 link = new ItemChatLink();
631 else if (strcmp(buffer, "quest") == 0)
632 link = new QuestChatLink();
633 else if (strcmp(buffer, "trade") == 0)
634 link = new TradeChatLink();
635 else if (strcmp(buffer, "talent") == 0)
636 link = new TalentChatLink();
637 else if (strcmp(buffer, "spell") == 0)
638 link = new SpellChatLink();
639 else if (strcmp(buffer, "enchant") == 0)
640 link = new EnchantmentChatLink();
641 else if (strcmp(buffer, "achievement") == 0)
642 link = new AchievementChatLink();
643 else if (strcmp(buffer, "glyph") == 0)
644 link = new GlyphChatLink();
645 else
646 {
647 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): user sent unsupported link type '%s'", _iss.str().c_str(), buffer);
648 return false;
649 }
650 _links.push_back(link);
651 link->SetColor(color);
652 if (!link->Initialize(_iss))
653 return false;
654 break;
655 case 'h':
656 // if h is next element in sequence, this one must contain the linked text :)
657 if (*validSequenceIterator == 'h')
658 {
659 // links start with '['
660 if (_iss.get() != '[')
661 {
662 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): link caption doesn't start with '['", _iss.str().c_str());
663 return false;
664 }
665 _iss.getline(buffer, 256, ']');
666 if (_iss.eof())
667 {
668 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): sequence finished unexpectedly", _iss.str().c_str());
669 return false;
670 }
671
672 if (!link)
673 return false;
674
675 if (!link->ValidateName(buffer, _iss.str().c_str()))
676 return false;
677 }
678 break;
679 case 'r':
680 if (link)
681 link->SetBounds(startPos, _iss.tellg());
682 case '|':
683 // no further payload
684 break;
685 default:
686 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): got invalid command |%c", _iss.str().c_str(), commandChar);
687 return false;
688 }
689 }
690
691 // check if every opened sequence was also closed properly
692 if (validSequence != validSequenceIterator)
693 {
694 TC_LOG_TRACE("chat.system", "ChatHandler::isValidChatMessage('%s'): EOF in active sequence", _iss.str().c_str());
695 return false;
696 }
697
698 return true;
699}

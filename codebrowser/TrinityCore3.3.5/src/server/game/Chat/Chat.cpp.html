37std::vector<ChatCommand> const& ChatHandler::getCommandTable()
38{
39 static std::vector<ChatCommand> commandTableCache;
40
41 if (LoadCommandTable())
42 {
43 SetLoadCommandTable(false);
44
45 std::vector<ChatCommand> cmds = sScriptMgr->GetChatCommands();
46 commandTableCache.swap(cmds);
47
48 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_COMMANDS);
49 PreparedQueryResult result = WorldDatabase.Query(stmt);
50 if (result)
51 {
52 do
53 {
54 Field* fields = result->Fetch();
55 std::string name = fields[0].GetString();
56
57 SetDataForCommandInTable(commandTableCache, name.c_str(), fields[1].GetUInt16(), fields[2].GetString(), name);
58 }
59 while (result->NextRow());
60 }
61 }
62
63 return commandTableCache;
64}

96bool ChatHandler::HasLowerSecurityAccount(WorldSession* target, uint32 target_account, bool strong)
97{
98 uint32 target_sec;
99
100 // allow everything from console and RA console
101 if (!m_session)
102 return false;
103
104 // ignore only for non-players for non strong checks (when allow apply command at least to same sec level)
105 if (m_session->HasPermission(rbac::RBAC_PERM_CHECK_FOR_LOWER_SECURITY) && !strong && !sWorld->getBoolConfig(CONFIG_GM_LOWER_SECURITY))
106 return false;
107
108 if (target)
109 target_sec = target->GetSecurity();
110 else if (target_account)
111 target_sec = AccountMgr::GetSecurity(target_account, realm.Id.Realm);
112 else
113 return true; // caller must report error for (target == NULL && target_account == 0)
114
115 AccountTypes target_ac_sec = AccountTypes(target_sec);
116 if (m_session->GetSecurity() < target_ac_sec || (strong && m_session->GetSecurity() <= target_ac_sec))
117 {
118 SendSysMessage(LANG_YOURS_SECURITY_IS_LOW);
119 SetSentErrorMessage(true);
120 return true;
121 }
122
123 return false;
124}

126bool ChatHandler::hasStringAbbr(const char* name, const char* part)
127{
128 // non "" command
129 if (*name)
130 {
131 // "" part from non-"" command
132 if (!*part)
133 return false;
134
135 while (true)
136 {
137 if (!*part)
138 return true;
139 else if (!*name)
140 return false;
141 else if (tolower(*name) != tolower(*part))
142 return false;
143 ++name; ++part;
144 }
145 }
146 // allow with any for ""
147
148 return true;
149}

151void ChatHandler::SendSysMessage(const char *str, bool escapeCharacters)
152{
153 WorldPacket data;
154
155 // need copy to prevent corruption by strtok call in LineFromMessage original string
156 char* buf;
157 char* pos;
158
159 if (escapeCharacters && strchr(str, '|'))
160 {
161 size_t startPos = 0;
162 std::ostringstream o;
163 while (const char* charPos = strchr(str + startPos, '|'))
164 {
165 o.write(str + startPos, charPos - str - startPos);
166 o << "||";
167 startPos = charPos - str + 1;
168 }
169 o.write(str + startPos, strlen(str) - startPos);
170 buf = strdup(o.str().c_str());
171 }
172 else
173 {
174 buf = strdup(str);
175 }
176
177 pos = buf;
178
179 while (char* line = LineFromMessage(pos))
180 {
181 BuildChatPacket(data, CHAT_MSG_SYSTEM, LANG_UNIVERSAL, NULL, NULL, line);
182 m_session->SendPacket(&data);
183 }
184
185 free(buf);
186}

229bool ChatHandler::ExecuteCommandInTable(std::vector<ChatCommand> const& table, const char* text, std::string const& fullcmd)
230{
231 char const* oldtext = text;
232 std::string cmd = "";
233
234 while (*text != ' ' && *text != '\0')
235 {
236 cmd += *text;
237 ++text;
238 }
239
240 while (*text == ' ') ++text;
241
242 for (uint32 i = 0; i < table.size(); ++i)
243 {
244 if (!hasStringAbbr(table[i].Name, cmd.c_str()))
245 continue;
246
247 bool match = false;
248 if (strlen(table[i].Name) > cmd.length())
249 {
250 for (uint32 j = 0; j < table.size(); ++j)
251 {
252 if (!hasStringAbbr(table[j].Name, cmd.c_str()))
253 continue;
254
255 if (strcmp(table[j].Name, cmd.c_str()) == 0)
256 {
257 match = true;
258 break;
259 }
260 }
261 }
262 if (match)
263 continue;
264
265 // select subcommand from child commands list
266 if (!table[i].ChildCommands.empty())
267 {
268 if (!ExecuteCommandInTable(table[i].ChildCommands, text, fullcmd))
269 {
270 if (text[0] != '\0')
271 SendSysMessage(LANG_NO_SUBCMD);
272 else
273 SendSysMessage(LANG_CMD_SYNTAX);
274
275 ShowHelpForCommand(table[i].ChildCommands, text);
276 }
277
278 return true;
279 }
280
281 // must be available and have handler
282 if (!table[i].Handler || !isAvailable(table[i]))
283 continue;
284
285 SetSentErrorMessage(false);
286 // table[i].Name == "" is special case: send original command to handler
287 if ((table[i].Handler)(this, table[i].Name[0] != '\0' ? text : oldtext))
288 {
289 if (!m_session) // ignore console
290 return true;
291
292 Player* player = m_session->GetPlayer();
293 if (!AccountMgr::IsPlayerAccount(m_session->GetSecurity()))
294 {
295 ObjectGuid guid = player->GetTarget();
296 uint32 areaId = player->GetAreaId();
297 std::string areaName = "Unknown";
298 std::string zoneName = "Unknown";
299 if (AreaTableEntry const* area = sAreaTableStore.LookupEntry(areaId))
300 {
301 int locale = GetSessionDbcLocale();
302 areaName = area->area_name[locale];
303 if (AreaTableEntry const* zone = sAreaTableStore.LookupEntry(area->zone))
304 zoneName = zone->area_name[locale];
305 }
306
307 sLog->outCommand(m_session->GetAccountId(), "Command: %s [Player: %s (%s) (Account: %u) X: %f Y: %f Z: %f Map: %u (%s) Area: %u (%s) Zone: %s Selected: %s (%s)]",
308 fullcmd.c_str(), player->GetName().c_str(), player->GetGUID().ToString().c_str(),
309 m_session->GetAccountId(), player->GetPositionX(), player->GetPositionY(),
310 player->GetPositionZ(), player->GetMapId(),
311 player->FindMap() ? player->FindMap()->GetMapName() : "Unknown",
312 areaId, areaName.c_str(), zoneName.c_str(),
313 (player->GetSelectedUnit()) ? player->GetSelectedUnit()->GetName().c_str() : "",
314 guid.ToString().c_str());
315 }
316 }
317 // some commands have custom error messages. Don't send the default one in these cases.
318 else if (!HasSentErrorMessage())
319 {
320 if (!table[i].Help.empty())
321 SendSysMessage(table[i].Help.c_str());
322 else
323 SendSysMessage(LANG_CMD_SYNTAX);
324 }
325
326 return true;
327 }
328
329 return false;
330}

332bool ChatHandler::SetDataForCommandInTable(std::vector<ChatCommand>& table, char const* text, uint32 permission, std::string const& help, std::string const& fullcommand)
333{
334 std::string cmd = "";
335
336 while (*text != ' ' && *text != '\0')
337 {
338 cmd += *text;
339 ++text;
340 }
341
342 while (*text == ' ') ++text;
343
344 for (uint32 i = 0; i < table.size(); i++)
345 {
346 // for data fill use full explicit command names
347 if (table[i].Name != cmd)
348 continue;
349
350 // select subcommand from child commands list (including "")
351 if (!table[i].ChildCommands.empty())
352 {
353 if (SetDataForCommandInTable(table[i].ChildCommands, text, permission, help, fullcommand))
354 return true;
355 else if (*text)
356 return false;
357
358 // fail with "" subcommands, then use normal level up command instead
359 }
360 // expected subcommand by full name DB content
361 else if (*text)
362 {
363 TC_LOG_ERROR("sql.sql", "Table `command` contains an unexpected subcommand '%s' in command '%s', skipped.", text, fullcommand.c_str());
364 return false;
365 }
366
367 if (table[i].Permission != permission)
368 TC_LOG_INFO("misc", "Table `command` overwrite for command '%s' default permission (%u) by %u", fullcommand.c_str(), table[i].Permission, permission);
369
370 table[i].Permission = permission;
371 table[i].Help = help;
372 return true;
373 }
374
375 // in case "" command let process by caller
376 if (!cmd.empty())
377 {
378 if (&table == &getCommandTable())
379 TC_LOG_ERROR("sql.sql", "Table `command` contains a non-existing command '%s', skipped.", cmd.c_str());
380 else
381 TC_LOG_ERROR("sql.sql", "Table `command` contains a non-existing subcommand '%s' in command '%s', skipped.", cmd.c_str(), fullcommand.c_str());
382 }
383
384 return false;
385}

387bool ChatHandler::ParseCommands(char const* text)
388{
389 ASSERT(text);
390 ASSERT(*text);
391
392 std::string fullcmd = text;
393
394 /// chat case (.command or !command format)
395 if (m_session)
396 {
397 if (text[0] != '!' && text[0] != '.')
398 return false;
399 }
400
401 /// ignore single . and ! in line
402 if (strlen(text) < 2)
403 return false;
404 // original `text` can't be used. It content destroyed in command code processing.
405
406 /// ignore messages staring from many dots.
407 if ((text[0] == '.' && text[1] == '.') || (text[0] == '!' && text[1] == '!'))
408 return false;
409
410 /// skip first . or ! (in console allowed use command with . and ! and without its)
411 if (text[0] == '!' || text[0] == '.')
412 ++text;
413
414 if (!ExecuteCommandInTable(getCommandTable(), text, fullcmd))
415 {
416 if (m_session && !m_session->HasPermission(rbac::RBAC_PERM_COMMANDS_NOTIFY_COMMAND_NOT_FOUND_ERROR))
417 return false;
418
419 SendSysMessage(LANG_NO_CMD);
420 }
421 return true;
422}

424bool ChatHandler::isValidChatMessage(char const* message)
425{
426/*
427Valid examples:
428|cffa335ee|Hitem:812:0:0:0:0:0:0:0:70|h[Glowing Brightwood Staff]|h|r
429|cff808080|Hquest:2278:47|h[The Platinum Discs]|h|r
430|cffffd000|Htrade:4037:1:150:1:6AAAAAAAAAAAAAAAAAAAAAAOAADAAAAAAAAAAAAAAAAIAAAAAAAAA|h[Engineering]|h|r
431|cff4e96f7|Htalent:2232:-1|h[Taste for Blood]|h|r
432|cff71d5ff|Hspell:21563|h[Command]|h|r
433|cffffd000|Henchant:3919|h[Engineering: Rough Dynamite]|h|r
434|cffffff00|Hachievement:546:0000000000000001:0:0:0:-1:0:0:0:0|h[Safe Deposit]|h|r
435|cff66bbff|Hglyph:21:762|h[Glyph of Bladestorm]|h|r
436
437| will be escaped to ||
438*/
439
440 if (strlen(message) > 255)
441 return false;
442
443 // more simple checks
444 if (sWorld->getIntConfig(CONFIG_CHAT_STRICT_LINK_CHECKING_SEVERITY) < 3)
445 {
446 const char validSequence[6] = "cHhhr";
447 const char* validSequenceIterator = validSequence;
448 const std::string validCommands = "cHhr|";
449
450 while (*message)
451 {
452 // find next pipe command
453 message = strchr(message, '|');
454
455 if (!message)
456 return true;
457
458 ++message;
459 char commandChar = *message;
460 if (validCommands.find(commandChar) == std::string::npos)
461 return false;
462
463 ++message;
464 // validate sequence
465 if (sWorld->getIntConfig(CONFIG_CHAT_STRICT_LINK_CHECKING_SEVERITY) == 2)
466 {
467 if (commandChar == *validSequenceIterator)
468 {
469 if (validSequenceIterator == validSequence + 4)
470 validSequenceIterator = validSequence;
471 else
472 ++validSequenceIterator;
473 }
474 else
475 return false;
476 }
477 }
478 return true;
479 }
480
481 return LinkExtractor(message).IsValidMessage();
482}

484bool ChatHandler::ShowHelpForSubCommands(std::vector<ChatCommand> const& table, char const* cmd, char const* subcmd)
485{
486 std::string list;
487 for (uint32 i = 0; i < table.size(); ++i)
488 {
489 // must be available (ignore handler existence to show command with possible available subcommands)
490 if (!isAvailable(table[i]))
491 continue;
492
493 // for empty subcmd show all available
494 if (*subcmd && !hasStringAbbr(table[i].Name, subcmd))
495 continue;
496
497 if (m_session)
498 list += "\n ";
499 else
500 list += "\n\r ";
501
502 list += table[i].Name;
503
504 if (!table[i].ChildCommands.empty())
505 list += " ...";
506 }
507
508 if (list.empty())
509 return false;
510
511 if (&table == &getCommandTable())
512 {
513 SendSysMessage(LANG_AVIABLE_CMD);
514 PSendSysMessage("%s", list.c_str());
515 }
516 else
517 PSendSysMessage(LANG_SUBCMDS_LIST, cmd, list.c_str());
518
519 return true;
520}

522bool ChatHandler::ShowHelpForCommand(std::vector<ChatCommand> const& table, const char* cmd)
523{
524 if (*cmd)
525 {
526 for (uint32 i = 0; i < table.size(); ++i)
527 {
528 // must be available (ignore handler existence to show command with possible available subcommands)
529 if (!isAvailable(table[i]))
530 continue;
531
532 if (!hasStringAbbr(table[i].Name, cmd))
533 continue;
534
535 // have subcommand
536 char const* subcmd = (*cmd) ? strtok(NULL, " ") : "";
537
538 if (!table[i].ChildCommands.empty() && subcmd && *subcmd)
539 {
540 if (ShowHelpForCommand(table[i].ChildCommands, subcmd))
541 return true;
542 }
543
544 if (!table[i].Help.empty())
545 SendSysMessage(table[i].Help.c_str());
546
547 if (!table[i].ChildCommands.empty())
548 if (ShowHelpForSubCommands(table[i].ChildCommands, table[i].Name, subcmd ? subcmd : ""))
549 return true;
550
551 return !table[i].Help.empty();
552 }
553 }
554 else
555 {
556 for (uint32 i = 0; i < table.size(); ++i)
557 {
558 // must be available (ignore handler existence to show command with possible available subcommands)
559 if (!isAvailable(table[i]))
560 continue;
561
562 if (strlen(table[i].Name))
563 continue;
564
565 if (!table[i].Help.empty())
566 SendSysMessage(table[i].Help.c_str());
567
568 if (!table[i].ChildCommands.empty())
569 if (ShowHelpForSubCommands(table[i].ChildCommands, "", ""))
570 return true;
571
572 return !table[i].Help.empty();
573 }
574 }
575
576 return ShowHelpForSubCommands(table, "", cmd);
577}

579size_t ChatHandler::BuildChatPacket(WorldPacket& data, ChatMsg chatType, Language language, ObjectGuid senderGUID, ObjectGuid receiverGUID, std::string const& message, uint8 chatTag,
580 std::string const& senderName /*= ""*/, std::string const& receiverName /*= ""*/,
581 uint32 achievementId /*= 0*/, bool gmMessage /*= false*/, std::string const& channelName /*= ""*/)
582{
583 size_t receiverGUIDPos = 0;
584 data.Initialize(!gmMessage ? SMSG_MESSAGECHAT : SMSG_GM_MESSAGECHAT);
585 data << uint8(chatType);
586 data << int32(language);
587 data << uint64(senderGUID);
588 data << uint32(0); // some flags
589 switch (chatType)
590 {
591 case CHAT_MSG_MONSTER_SAY:
592 case CHAT_MSG_MONSTER_PARTY:
593 case CHAT_MSG_MONSTER_YELL:
594 case CHAT_MSG_MONSTER_WHISPER:
595 case CHAT_MSG_MONSTER_EMOTE:
596 case CHAT_MSG_RAID_BOSS_EMOTE:
597 case CHAT_MSG_RAID_BOSS_WHISPER:
598 case CHAT_MSG_BATTLENET:
599 data << uint32(senderName.length() + 1);
600 data << senderName;
601 receiverGUIDPos = data.wpos();
602 data << uint64(receiverGUID);
603 if (receiverGUID && !receiverGUID.IsPlayer() && !receiverGUID.IsPet())
604 {
605 data << uint32(receiverName.length() + 1);
606 data << receiverName;
607 }
608 break;
609 case CHAT_MSG_WHISPER_FOREIGN:
610 data << uint32(senderName.length() + 1);
611 data << senderName;
612 receiverGUIDPos = data.wpos();
613 data << uint64(receiverGUID);
614 break;
615 case CHAT_MSG_BG_SYSTEM_NEUTRAL:
616 case CHAT_MSG_BG_SYSTEM_ALLIANCE:
617 case CHAT_MSG_BG_SYSTEM_HORDE:
618 receiverGUIDPos = data.wpos();
619 data << uint64(receiverGUID);
620 if (receiverGUID && !receiverGUID.IsPlayer())
621 {
622 data << uint32(receiverName.length() + 1);
623 data << receiverName;
624 }
625 break;
626 case CHAT_MSG_ACHIEVEMENT:
627 case CHAT_MSG_GUILD_ACHIEVEMENT:
628 receiverGUIDPos = data.wpos();
629 data << uint64(receiverGUID);
630 break;
631 default:
632 if (gmMessage)
633 {
634 data << uint32(senderName.length() + 1);
635 data << senderName;
636 }
637
638 if (chatType == CHAT_MSG_CHANNEL)
639 {
640 ASSERT(channelName.length() > 0);
641 data << channelName;
642 }
643
644 receiverGUIDPos = data.wpos();
645 data << uint64(receiverGUID);
646 break;
647 }
648
649 data << uint32(message.length() + 1);
650 data << message;
651 data << uint8(chatTag);
652
653 if (chatType == CHAT_MSG_ACHIEVEMENT || chatType == CHAT_MSG_GUILD_ACHIEVEMENT)
654 data << uint32(achievementId);
655
656 return receiverGUIDPos;
657}

659size_t ChatHandler::BuildChatPacket(WorldPacket& data, ChatMsg chatType, Language language, WorldObject const* sender, WorldObject const* receiver, std::string const& message,
660 uint32 achievementId /*= 0*/, std::string const& channelName /*= ""*/, LocaleConstant locale /*= DEFAULT_LOCALE*/)
661{
662 ObjectGuid senderGUID;
663 std::string senderName = "";
664 uint8 chatTag = 0;
665 bool gmMessage = false;
666 ObjectGuid receiverGUID;
667 std::string receiverName = "";
668 if (sender)
669 {
670 senderGUID = sender->GetGUID();
671 senderName = sender->GetNameForLocaleIdx(locale);
672 if (Player const* playerSender = sender->ToPlayer())
673 {
674 chatTag = playerSender->GetChatTag();
675 gmMessage = playerSender->GetSession()->HasPermission(rbac::RBAC_PERM_COMMAND_GM_CHAT);
676 }
677 }
678
679 if (receiver)
680 {
681 receiverGUID = receiver->GetGUID();
682 receiverName = receiver->GetNameForLocaleIdx(locale);
683 }
684
685 return BuildChatPacket(data, chatType, language, senderGUID, receiverGUID, message, chatTag, senderName, receiverName, achievementId, gmMessage, channelName);
686}

750char* ChatHandler::extractKeyFromLink(char* text, char const* linkType, char** something1)
751{
752 // skip empty
753 if (!text)
754 return NULL;
755
756 // skip spaces
757 while (*text == ' '||*text == '\t'||*text == '\b')
758 ++text;
759
760 if (!*text)
761 return NULL;
762
763 // return non link case
764 if (text[0] != '|')
765 return strtok(text, " ");
766
767 // [name] Shift-click form |color|linkType:key|h[name]|h|r
768 // or
769 // [name] Shift-click form |color|linkType:key:something1:...:somethingN|h[name]|h|r
770
771 char* check = strtok(text, "|"); // skip color
772 if (!check)
773 return NULL; // end of data
774
775 char* cLinkType = strtok(NULL, ":"); // linktype
776 if (!cLinkType)
777 return NULL; // end of data
778
779 if (strcmp(cLinkType, linkType) != 0)
780 {
781 strtok(NULL, " "); // skip link tail (to allow continue strtok(NULL, s) use after retturn from function
782 SendSysMessage(LANG_WRONG_LINK_TYPE);
783 return NULL;
784 }

786 char* cKeys = strtok(NULL, "|"); // extract keys and values
787 char* cKeysTail = strtok(NULL, "");
788
789 char* cKey = strtok(cKeys, ":|"); // extract key
790 if (something1)
791 *something1 = strtok(NULL, ":|"); // extract something
792
793 strtok(cKeysTail, "]"); // restart scan tail and skip name with possible spaces
794 strtok(NULL, " "); // skip link tail (to allow continue strtok(NULL, s) use after return from function
795 return cKey;
796}

798char* ChatHandler::extractKeyFromLink(char* text, char const* const* linkTypes, int* found_idx, char** something1)
799{
800 // skip empty
801 if (!text)
802 return NULL;
803
804 // skip spaces
805 while (*text == ' '||*text == '\t'||*text == '\b')
806 ++text;
807
808 if (!*text)
809 return NULL;
810
811 // return non link case
812 if (text[0] != '|')
813 return strtok(text, " ");
814
815 // [name] Shift-click form |color|linkType:key|h[name]|h|r
816 // or
817 // [name] Shift-click form |color|linkType:key:something1:...:somethingN|h[name]|h|r
818 // or
819 // [name] Shift-click form |linkType:key|h[name]|h|r
820
821 char* tail;
822
823 if (text[1] == 'c')
824 {
825 char* check = strtok(text, "|"); // skip color
826 if (!check)
827 return NULL; // end of data
828
829 tail = strtok(NULL, ""); // tail
830 }
831 else
832 tail = text+1; // skip first |
833
834 char* cLinkType = strtok(tail, ":"); // linktype
835 if (!cLinkType)
836 return NULL; // end of data
837
838 for (int i = 0; linkTypes[i]; ++i)
839 {
840 if (strcmp(cLinkType, linkTypes[i]) == 0)
841 {
842 char* cKeys = strtok(NULL, "|"); // extract keys and values
843 char* cKeysTail = strtok(NULL, "");
844
845 char* cKey = strtok(cKeys, ":|"); // extract key
846 if (something1)
847 *something1 = strtok(NULL, ":|"); // extract something
848
849 strtok(cKeysTail, "]"); // restart scan tail and skip name with possible spaces
850 strtok(NULL, " "); // skip link tail (to allow continue strtok(NULL, s) use after return from function
851 if (found_idx)
852 *found_idx = i;
853 return cKey;
854 }
855 }
856
857 strtok(NULL, " "); // skip link tail (to allow continue strtok(NULL, s) use after return from function
858 SendSysMessage(LANG_WRONG_LINK_TYPE);
859 return NULL;
860}

915uint32 ChatHandler::extractSpellIdFromLink(char* text)
916{
917 // number or [name] Shift-click form |color|Henchant:recipe_spell_id|h[prof_name: recipe_name]|h|r
918 // number or [name] Shift-click form |color|Hglyph:glyph_slot_id:glyph_prop_id|h[%s]|h|r
919 // number or [name] Shift-click form |color|Hspell:spell_id|h[name]|h|r
920 // number or [name] Shift-click form |color|Htalent:talent_id, rank|h[name]|h|r
921 // number or [name] Shift-click form |color|Htrade:spell_id, skill_id, max_value, cur_value|h[name]|h|r
922 int type = 0;
923 char* param1_str = NULL;
924 char* idS = extractKeyFromLink(text, spellKeys, &type, &param1_str);
925 if (!idS)
926 return 0;
927
928 uint32 id = atoul(idS);
929
930 switch (type)
931 {
932 case SPELL_LINK_SPELL:
933 return id;
934 case SPELL_LINK_TALENT:
935 {
936 // talent
937 TalentEntry const* talentEntry = sTalentStore.LookupEntry(id);
938 if (!talentEntry)
939 return 0;
940
941 uint32 rank = param1_str ? atol(param1_str) : 0u;
942 if (rank >= MAX_TALENT_RANK)
943 return 0;
944
945 return talentEntry->RankID[rank];
946 }
947 case SPELL_LINK_ENCHANT:
948 case SPELL_LINK_TRADE:
949 return id;
950 case SPELL_LINK_GLYPH:
951 {
952 uint32 glyph_prop_id = param1_str ? atoul(param1_str) : 0;
953
954 GlyphPropertiesEntry const* glyphPropEntry = sGlyphPropertiesStore.LookupEntry(glyph_prop_id);
955 if (!glyphPropEntry)
956 return 0;
957
958 return glyphPropEntry->SpellId;
959 }
960 }
961
962 // unknown type?
963 return 0;
964}

1132char* ChatHandler::extractQuotedArg(char* args)
1133{
1134 if (!args || !*args)
1135 return NULL;
1136
1137 if (*args == '"')
1138 return strtok(args+1, "\"");
1139 else
1140 {
1141 // skip spaces
1142 while (*args == ' ')
1143 {
1144 args += 1;
1145 continue;
1146 }
1147
1148 // return NULL if we reached the end of the string
1149 if (!*args)
1150 return NULL;
1151
1152 // since we skipped all spaces, we expect another token now
1153 if (*args == '"')
1154 {
1155 // return an empty string if there are 2 "" in a row.
1156 // strtok doesn't handle this case
1157 if (*(args + 1) == '"')
1158 {
1159 strtok(args, " ");
1160 static char arg[1];
1161 arg[0] = '\0';
1162 return arg;
1163 }
1164 else
1165 return strtok(args + 1, "\"");
1166 }
1167 else
1168 return NULL;
1169 }

501/// Storage class for commands issued for delayed execution
502struct CliCommandHolder
503{
504 typedef void Print(void*, const char*);
505 typedef void CommandFinished(void*, bool success);
506
507 void* m_callbackArg;
508 char *m_command;
509 Print* m_print;
510
511 CommandFinished* m_commandFinished;
512
513 CliCommandHolder(void* callbackArg, const char *command, Print* zprint, CommandFinished* commandFinished)
514 : m_callbackArg(callbackArg), m_command(strdup(command)), m_print(zprint), m_commandFinished(commandFinished)
515 {
516 }
517
518 ~CliCommandHolder() { free(m_command); }
519
520private:
521 CliCommandHolder(CliCommandHolder const& right) = delete;
522 CliCommandHolder& operator=(CliCommandHolder const& right) = delete;
523};

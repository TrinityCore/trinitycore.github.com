35bool AuctionBotSeller::Initialize()
36{
37 std::vector<uint32> npcItems;
38 std::vector<uint32> lootItems;
39 std::vector<uint32> includeItems;
40 std::vector<uint32> excludeItems;
41
42 TC_LOG_DEBUG("ahbot", "AHBot seller filters:");
43
44 {
45 std::stringstream includeStream(sAuctionBotConfig->GetAHBotIncludes());
46 std::string temp;
47 while (std::getline(includeStream, temp, ','))
48 includeItems.push_back(atoi(temp.c_str()));
49 }
50
51 {
52 std::stringstream excludeStream(sAuctionBotConfig->GetAHBotExcludes());
53 std::string temp;
54 while (std::getline(excludeStream, temp, ','))
55 excludeItems.push_back(atoi(temp.c_str()));
56 }
57
58 TC_LOG_DEBUG("ahbot", "Forced Inclusion %u items", (uint32)includeItems.size());
59 TC_LOG_DEBUG("ahbot", "Forced Exclusion %u items", (uint32)excludeItems.size());
60
61 TC_LOG_DEBUG("ahbot", "Loading npc vendor items for filter..");
62 const CreatureTemplateContainer* creatures = sObjectMgr->GetCreatureTemplates();
63 std::set<uint32> tempItems;
64 for (CreatureTemplateContainer::const_iterator it = creatures->begin(); it != creatures->end(); ++it)
65 {
66 if (const VendorItemData* data = sObjectMgr->GetNpcVendorItemList(it->first))
67 {
68 for (VendorItemList::const_iterator it2 = data->m_items.begin(); it2 != data->m_items.end(); ++it2)
69 tempItems.insert((*it2)->item);
70 }
71 }
72 for (std::set<uint32>::const_iterator it = tempItems.begin(); it != tempItems.end(); ++it)
73 npcItems.push_back(*it);
74
75 TC_LOG_DEBUG("ahbot", "Npc vendor filter has %u items", (uint32)npcItems.size());
76
77 TC_LOG_DEBUG("ahbot", "Loading loot items for filter..");
78 QueryResult result = WorldDatabase.PQuery(
79 "SELECT `item` FROM `creature_loot_template` WHERE `Reference` = 0 UNION "
80 "SELECT `item` FROM `disenchant_loot_template` WHERE `Reference` = 0 UNION "
81 "SELECT `item` FROM `fishing_loot_template` WHERE `Reference` = 0 UNION "
82 "SELECT `item` FROM `gameobject_loot_template` WHERE `Reference` = 0 UNION "
83 "SELECT `item` FROM `item_loot_template` WHERE `Reference` = 0 UNION "
84 "SELECT `item` FROM `milling_loot_template` WHERE `Reference` = 0 UNION "
85 "SELECT `item` FROM `pickpocketing_loot_template` WHERE `Reference` = 0 UNION "
86 "SELECT `item` FROM `prospecting_loot_template` WHERE `Reference` = 0 UNION "
87 "SELECT `item` FROM `reference_loot_template` WHERE `Reference` = 0 UNION "
88 "SELECT `item` FROM `skinning_loot_template` WHERE `Reference` = 0 UNION "
89 "SELECT `item` FROM `spell_loot_template` WHERE `Reference` = 0");
90
91 if (result)
92 {
93 do
94 {
95 Field* fields = result->Fetch();
96
97 uint32 entry = fields[0].GetUInt32();
98 if (!entry)
99 continue;
100
101 lootItems.push_back(entry);
102 } while (result->NextRow());
103 }
104
105 TC_LOG_DEBUG("ahbot", "Loot filter has %u items", (uint32)lootItems.size());
106 TC_LOG_DEBUG("ahbot", "Sorting and cleaning items for AHBot seller...");
107
108 uint32 itemsAdded = 0;
109
110 for (uint32 itemId = 0; itemId < sItemStore.GetNumRows(); ++itemId)
111 {
112 ItemTemplate const* prototype = sObjectMgr->GetItemTemplate(itemId);
113
114 if (!prototype)
115 continue;
116
117 // skip items with too high quality (code can't properly work with its)
118 if (prototype->Quality >= MAX_AUCTION_QUALITY)
119 continue;
120
121 // forced exclude filter
122 bool isExcludeItem = false;
123 for (size_t i = 0; i < excludeItems.size() && !isExcludeItem; ++i)
124 if (itemId == excludeItems[i])
125 isExcludeItem = true;
126
127 if (isExcludeItem)
128 continue;
129
130 // forced include filter
131 bool isForcedIncludeItem = false;
132 for (size_t i = 0; i < includeItems.size() && !isForcedIncludeItem; ++i)
133 if (itemId == includeItems[i])
134 isForcedIncludeItem = true;
135
136 if (isForcedIncludeItem)
137 {
138 _itemPool[prototype->Quality][prototype->Class].push_back(itemId);
139 ++itemsAdded;
140 continue;
141 }
142
143 // bounding filters
144 switch (prototype->Bonding)
145 {
146 case NO_BIND:
147 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BIND_NO))
148 continue;
149 break;
150 case BIND_WHEN_PICKED_UP:
151 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BIND_PICKUP))
152 continue;
153 break;
154 case BIND_WHEN_EQUIPED:
155 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BIND_EQUIP))
156 continue;
157 break;
158 case BIND_WHEN_USE:
159 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BIND_USE))
160 continue;
161 break;
162 case BIND_QUEST_ITEM:
163 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BIND_QUEST))
164 continue;
165 break;
166 default:
167 continue;
168 }
169
170 bool allowZero = false;
171 switch (prototype->Class)
172 {
173 case ITEM_CLASS_CONSUMABLE:
174 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_CONSUMABLE_ALLOW_ZERO); break;
175 case ITEM_CLASS_CONTAINER:
176 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_CONTAINER_ALLOW_ZERO); break;
177 case ITEM_CLASS_WEAPON:
178 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_WEAPON_ALLOW_ZERO); break;
179 case ITEM_CLASS_GEM:
180 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_GEM_ALLOW_ZERO); break;
181 case ITEM_CLASS_ARMOR:
182 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_ARMOR_ALLOW_ZERO); break;
183 case ITEM_CLASS_REAGENT:
184 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_REAGENT_ALLOW_ZERO); break;
185 case ITEM_CLASS_PROJECTILE:
186 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_PROJECTILE_ALLOW_ZERO); break;
187 case ITEM_CLASS_TRADE_GOODS:
188 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_TRADEGOOD_ALLOW_ZERO); break;
189 case ITEM_CLASS_RECIPE:
190 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_RECIPE_ALLOW_ZERO); break;
191 case ITEM_CLASS_QUIVER:
192 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_QUIVER_ALLOW_ZERO); break;
193 case ITEM_CLASS_QUEST:
194 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_QUEST_ALLOW_ZERO); break;
195 case ITEM_CLASS_KEY:
196 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_KEY_ALLOW_ZERO); break;
197 case ITEM_CLASS_MISC:
198 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_MISC_ALLOW_ZERO); break;
199 case ITEM_CLASS_GLYPH:
200 allowZero = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_GLYPH_ALLOW_ZERO); break;
201 default:
202 allowZero = false;
203 }
204
205 // Filter out items with no buy/sell price unless otherwise flagged in the config.
206 if (!allowZero)
207 {
208 if (sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BUYPRICE_SELLER))
209 {
210 if (prototype->SellPrice == 0)
211 continue;
212 }
213 else
214 {
215 if (prototype->BuyPrice == 0)
216 continue;
217 }
218 }
219
220 // vendor filter
221 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEMS_VENDOR))
222 {
223 bool isVendorItem = false;
224 for (size_t i = 0; i < npcItems.size() && !isVendorItem; ++i)
225 if (itemId == npcItems[i])
226 isVendorItem = true;
227
228 if (isVendorItem)
229 continue;
230 }
231
232 // loot filter
233 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEMS_LOOT))
234 {
235 bool isLootItem = false;
236 for (size_t i = 0; i < lootItems.size() && !isLootItem; ++i)
237 if (itemId == lootItems[i])
238 isLootItem = true;
239
240 if (isLootItem)
241 continue;
242 }
243
244 // not vendor/loot filter
245 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEMS_MISC))
246 {
247 bool isVendorItem = false;
248 bool isLootItem = false;
249
250 for (size_t i = 0; i < npcItems.size() && !isVendorItem; ++i)
251 if (itemId == npcItems[i])
252 isVendorItem = true;
253
254 for (size_t i = 0; i < lootItems.size() && !isLootItem; ++i)
255 if (itemId == lootItems[i])
256 isLootItem = true;
257
258 if (!isLootItem && !isVendorItem)
259 continue;
260 }
261
262 // item class/subclass specific filters
263 switch (prototype->Class)
264 {
265 case ITEM_CLASS_ARMOR:
266 case ITEM_CLASS_WEAPON:
267 {
268 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MIN_ITEM_LEVEL))
269 if (prototype->ItemLevel < value)
270 continue;
271 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MAX_ITEM_LEVEL))
272 if (prototype->ItemLevel > value)
273 continue;
274 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MIN_REQ_LEVEL))
275 if (prototype->RequiredLevel < value)
276 continue;
277 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MAX_REQ_LEVEL))
278 if (prototype->RequiredLevel > value)
279 continue;
280 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MIN_SKILL_RANK))
281 if (prototype->RequiredSkillRank < value)
282 continue;
283 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MAX_SKILL_RANK))
284 if (prototype->RequiredSkillRank > value)
285 continue;
286 break;
287 }
288 case ITEM_CLASS_RECIPE:
289 case ITEM_CLASS_CONSUMABLE:
290 case ITEM_CLASS_PROJECTILE:
291 {
292 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MIN_REQ_LEVEL))
293 if (prototype->RequiredLevel < value)
294 continue;
295 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MAX_REQ_LEVEL))
296 if (prototype->RequiredLevel > value)
297 continue;
298 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MIN_SKILL_RANK))
299 if (prototype->RequiredSkillRank < value)
300 continue;
301 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_ITEM_MAX_SKILL_RANK))
302 if (prototype->RequiredSkillRank > value)
303 continue;
304 break;
305 }
306 case ITEM_CLASS_MISC:
307 if (prototype->SubClass == ITEM_SUBCLASS_JUNK_MOUNT)
308 {
309 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_MISC_MOUNT_MIN_REQ_LEVEL))
310 if (prototype->RequiredLevel < value)
311 continue;
312 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_MISC_MOUNT_MAX_REQ_LEVEL))
313 if (prototype->RequiredLevel > value)
314 continue;
315 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_MISC_MOUNT_MIN_SKILL_RANK))
316 if (prototype->RequiredSkillRank < value)
317 continue;
318 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_MISC_MOUNT_MAX_SKILL_RANK))
319 if (prototype->RequiredSkillRank > value)
320 continue;
321 }
322
323 if (prototype->Flags & ITEM_FLAG_UNLOCKED)
324 {
325 // skip any not locked lootable items (mostly quest specific or reward cases)
326 if (!prototype->LockID)
327 continue;
328
329 if (!sAuctionBotConfig->GetConfig(CONFIG_AHBOT_LOCKBOX_ENABLED))
330 continue;
331 }
332
333 break;
334 case ITEM_CLASS_GLYPH:
335 {
336 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_GLYPH_MIN_REQ_LEVEL))
337 if (prototype->RequiredLevel < value)
338 continue;
339 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_GLYPH_MAX_REQ_LEVEL))
340 if (prototype->RequiredLevel > value)
341 continue;
342 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_GLYPH_MIN_ITEM_LEVEL))
343 if (prototype->RequiredLevel < value)
344 continue;
345 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_GLYPH_MAX_ITEM_LEVEL))
346 if (prototype->RequiredLevel > value)
347 continue;
348 break;
349 }
350 case ITEM_CLASS_TRADE_GOODS:
351 {
352 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_TRADEGOOD_MIN_ITEM_LEVEL))
353 if (prototype->ItemLevel < value)
354 continue;
355 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_TRADEGOOD_MAX_ITEM_LEVEL))
356 if (prototype->ItemLevel > value)
357 continue;
358 break;
359 }
360 case ITEM_CLASS_CONTAINER:
361 case ITEM_CLASS_QUIVER:
362 {
363 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_CONTAINER_MIN_ITEM_LEVEL))
364 if (prototype->ItemLevel < value)
365 continue;
366 if (uint32 value = sAuctionBotConfig->GetConfig(CONFIG_AHBOT_CLASS_CONTAINER_MAX_ITEM_LEVEL))
367 if (prototype->ItemLevel > value)
368 continue;
369 break;
370 }
371 }
372
373 _itemPool[prototype->Quality][prototype->Class].push_back(itemId);
374 ++itemsAdded;
375 }
376
377 if (!itemsAdded)
378 {
379 TC_LOG_ERROR("ahbot", "AuctionHouseBot seller not have items, disabled.");
380 sAuctionBotConfig->SetConfig(CONFIG_AHBOT_ALLIANCE_ITEM_AMOUNT_RATIO, 0);
381 sAuctionBotConfig->SetConfig(CONFIG_AHBOT_HORDE_ITEM_AMOUNT_RATIO, 0);
382 sAuctionBotConfig->SetConfig(CONFIG_AHBOT_NEUTRAL_ITEM_AMOUNT_RATIO, 0);
383 return false;
384 }
385
386 TC_LOG_DEBUG("ahbot", "AuctionHouseBot seller will use %u items to fill auction house (according your config choices)", itemsAdded);
387
388 LoadConfig();
389
390 TC_LOG_DEBUG("ahbot", "Items loaded \tGray\tWhite\tGreen\tBlue\tPurple\tOrange\tYellow");
391 for (uint32 i = 0; i < MAX_ITEM_CLASS; ++i)
392 TC_LOG_DEBUG("ahbot", "\t\t%u\t%u\t%u\t%u\t%u\t%u\t%u",
393 (uint32)_itemPool[0][i].size(), (uint32)_itemPool[1][i].size(), (uint32)_itemPool[2][i].size(),
394 (uint32)_itemPool[3][i].size(), (uint32)_itemPool[4][i].size(), (uint32)_itemPool[5][i].size(),
395 (uint32)_itemPool[6][i].size());
396
397 TC_LOG_DEBUG("ahbot", "AHBot seller configuration data loaded and initialized");
398 return true;
399}

699// Set items price. All important value are passed by address.
700void AuctionBotSeller::SetPricesOfItem(ItemTemplate const* itemProto, SellerConfiguration& config, uint32& buyp, uint32& bidp, uint32 stackCount)
701{
702 uint32 classRatio = config.GetPriceRatioPerClass(ItemClass(itemProto->Class));
703 uint32 qualityRatio = config.GetPriceRatioPerQuality(AuctionQuality(itemProto->Quality));
704 uint32 priceRatio = (classRatio * qualityRatio) / 100;
705
706 uint32 buyPrice = itemProto->BuyPrice;
707 uint32 sellPrice = itemProto->SellPrice;
708
709 if (buyPrice == 0)
710 {
711 if (sellPrice > 0)
712 buyPrice = sellPrice * GetSellModifier(itemProto);
713 else
714 {
715 uint32 divisor = ((itemProto->Class == 2 || itemProto->Class == 4) ? 284 : 80);
716 uint32 tempLevel = (itemProto->ItemLevel == 0 ? 1 : itemProto->ItemLevel);
717 uint32 tempQuality = (itemProto->Quality == 0 ? 1 : itemProto->Quality);
718
719 buyPrice = tempLevel * tempQuality * GetBuyModifier(itemProto)* tempLevel / divisor;
720 }
721 }
722
723 if (sellPrice == 0)
724 sellPrice = (buyPrice > 10 ? buyPrice / GetSellModifier(itemProto) : buyPrice);
725
726 if (sAuctionBotConfig->GetConfig(CONFIG_AHBOT_BUYPRICE_SELLER))
727 buyPrice = sellPrice;
728
729 uint32 basePrice = (buyPrice * stackCount * priceRatio) / (itemProto->Class == 6 ? 200 : itemProto->BuyCount) / 100;
730 uint32 range = basePrice * 0.04;
731
732 buyp = urand(basePrice - range, basePrice + range) + 1;
733
734 basePrice = buyp * .5;
735 range = buyp * .4;
736 bidp = urand(basePrice - range, basePrice + range) + 1;
737}

918// Add new auction to one of the factions.
919// Faction and setting associated is defined passed argument ( config )
920void AuctionBotSeller::AddNewAuctions(SellerConfiguration& config)
921{
922 uint32 count = 0;
923 uint32 items = 0;
924
925 // If there is large amount of items missed we can use boost value to get fast filled AH
926 if (config.LastMissedItem > sAuctionBotConfig->GetItemPerCycleBoost())
927 {
928 items = sAuctionBotConfig->GetItemPerCycleBoost();
929 TC_LOG_DEBUG("ahbot", "AHBot: Boost value used to fill AH! (if this happens often adjust both ItemsPerCycle in worldserver.conf)");
930 }
931 else
932 items = sAuctionBotConfig->GetItemPerCycleNormal();
933
934 uint32 houseid = 0;
935 switch (config.GetHouseType())
936 {
937 case AUCTION_HOUSE_ALLIANCE:
938 houseid = AUCTIONHOUSE_ALLIANCE;
939 break;
940 case AUCTION_HOUSE_HORDE:
941 houseid = AUCTIONHOUSE_HORDE;
942 break;
943 default:
944 houseid = AUCTIONHOUSE_NEUTRAL;
945 break;
946 }
947
948 AuctionHouseEntry const* ahEntry = sAuctionHouseStore.LookupEntry(houseid);
949
950 AuctionHouseObject* auctionHouse = sAuctionMgr->GetAuctionsMap(config.GetHouseType());
951
952 ItemsToSellArray itemsToSell;
953 AllItemsArray allItems(MAX_AUCTION_QUALITY, std::vector<uint32>(MAX_ITEM_CLASS));
954 // Main loop
955 // getRandomArray will give what categories of items should be added (return true if there is at least 1 items missed)
956 SQLTransaction trans = CharacterDatabase.BeginTransaction();
957 while (GetItemsToSell(config, itemsToSell, allItems) && items > 0)
958 {
959 --items;
960
961 // Select random position from missed items table
962 uint32 pos = urand(0, itemsToSell.size() - 1);
963
964 // Set itemId with random item ID for selected categories and color, from _itemPool table
965 uint32 itemId = _itemPool[itemsToSell[pos].Color][itemsToSell[pos].Itemclass][urand(0, _itemPool[itemsToSell[pos].Color][itemsToSell[pos].Itemclass].size() - 1)];
966 ++allItems[itemsToSell[pos].Color][itemsToSell[pos].Itemclass]; // Helper table to avoid rescan from DB in this loop. (has we add item in random orders)
967
968 if (!itemId)
969 {
970 TC_LOG_DEBUG("ahbot", "AHBot: Item entry 0 auction creating attempt.");
971 continue;
972 }
973
974 ItemTemplate const* prototype = sObjectMgr->GetItemTemplate(itemId);
975 if (!prototype)
976 {
977 TC_LOG_DEBUG("ahbot", "AHBot: Unknown item %u auction creating attempt.", itemId);
978 continue;
979 }
980
981 uint32 stackCount = GetStackSizeForItem(prototype, config);
982
983 Item* item = Item::CreateItem(itemId, stackCount);
984 if (!item)
985 {
986 TC_LOG_ERROR("ahbot", "AHBot: Item::CreateItem() returned NULL for item %u (stack: %u)", itemId, stackCount);
987 return;
988 }
989
990 // Update the just created item so that if it needs random properties it has them.
991 // Ex: Notched Shortsword of Stamina will only generate as a Notched Shortsword without this.
992 if (int32 randomPropertyId = Item::GenerateItemRandomPropertyId(itemId))
993 item->SetItemRandomProperties(randomPropertyId);
994
995 uint32 buyoutPrice;
996 uint32 bidPrice = 0;
997
998 // Price of items are set here
999 SetPricesOfItem(prototype, config, buyoutPrice, bidPrice, stackCount);
1000
1001 // Deposit time
1002 uint32 etime = urand(1, 3);
1003 switch (etime)
1004 {
1005 case 1:
1006 etime = 43200;
1007 break;
1008 case 2:
1009 etime = 86400;
1010 break;
1011 case 3:
1012 etime = 172800;
1013 break;
1014 default:
1015 etime = 86400;
1016 break;
1017 }
1018
1019 AuctionEntry* auctionEntry = new AuctionEntry();
1020 auctionEntry->Id = sObjectMgr->GenerateAuctionID();
1021 auctionEntry->owner = 0;
1022 auctionEntry->itemGUIDLow = item->GetGUID().GetCounter();
1023 auctionEntry->itemEntry = item->GetEntry();
1024 auctionEntry->startbid = bidPrice;
1025 auctionEntry->buyout = buyoutPrice;
1026 auctionEntry->houseId = houseid;
1027 auctionEntry->bidder = 0;
1028 auctionEntry->bid = 0;
1029 auctionEntry->deposit = sAuctionMgr->GetAuctionDeposit(ahEntry, etime, item, stackCount);
1030 auctionEntry->auctionHouseEntry = ahEntry;
1031 auctionEntry->expire_time = time(NULL) + urand(config.GetMinTime(), config.GetMaxTime()) * HOUR;
1032
1033 item->SaveToDB(trans);
1034 sAuctionMgr->AddAItem(item);
1035 auctionHouse->AddAuction(auctionEntry);
1036 auctionEntry->SaveToDB(trans);
1037
1038 auctionHouse->AddAuction(auctionEntry);
1039
1040 ++count;
1041 }
1042 CharacterDatabase.CommitTransaction(trans);
1043
1044 TC_LOG_DEBUG("ahbot", "AHBot: Added %u items to auction", count);
1045}

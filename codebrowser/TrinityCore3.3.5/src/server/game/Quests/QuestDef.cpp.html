182uint32 Quest::XPValue(Player* player) const
183{
184 if (player)
185 {
186 int32 quest_level = (Level == -1 ? player->getLevel() : Level);
187 const QuestXPEntry* xpentry = sQuestXPStore.LookupEntry(quest_level);
188 if (!xpentry)
189 return 0;
190
191 int32 diffFactor = 2 * (quest_level - player->getLevel()) + 20;
192 if (diffFactor < 1)
193 diffFactor = 1;
194 else if (diffFactor > 10)
195 diffFactor = 10;
196
197 uint32 xp = diffFactor * xpentry->Exp[RewardXPDifficulty] / 10;
198 if (xp <= 100)
199 xp = 5 * ((xp + 2) / 5);
200 else if (xp <= 500)
201 xp = 10 * ((xp + 5) / 10);
202 else if (xp <= 1000)
203 xp = 25 * ((xp + 12) / 25);
204 else
205 xp = 50 * ((xp + 25) / 50);
206
207 return xp;
208 }
209
210 return 0;
211}

213int32 Quest::GetRewOrReqMoney() const
214{
215 // RequiredMoney: the amount is the negative copper sum.
216 if (RewardMoney <= 0)
217 return RewardMoney;
218
219 // RewardMoney: the positive amount
220 return int32(RewardMoney * sWorld->getRate(RATE_MONEY_QUEST));
221}

223uint32 Quest::GetRewMoneyMaxLevel() const
224{
225 // If Quest has flag to not give money on max level, it's 0
226 if (HasFlag(QUEST_FLAGS_NO_MONEY_FROM_XP))
227 return 0;
228
229 // Else, return the rewarded copper sum modified by the rate
230 return uint32(RewardBonusMoney * sWorld->getRate(RATE_MONEY_MAX_LEVEL_QUEST));
231}

243bool Quest::IsRaidQuest(Difficulty difficulty) const
244{
245 switch (Type)
246 {
247 case QUEST_TYPE_RAID:
248 return true;
249 case QUEST_TYPE_RAID_10:
250 return !(difficulty & RAID_DIFFICULTY_MASK_25MAN);
251 case QUEST_TYPE_RAID_25:
252 return difficulty & RAID_DIFFICULTY_MASK_25MAN;
253 default:
254 break;
255 }
256
257 return false;
258}

260bool Quest::IsAllowedInRaid(Difficulty difficulty) const
261{
262 if (IsRaidQuest(difficulty))
263 return true;
264
265 return sWorld->getBoolConfig(CONFIG_QUEST_IGNORE_RAID);
266}

268uint32 Quest::CalculateHonorGain(uint8 level) const
269{
270 if (level > GT_MAX_LEVEL)
271 level = GT_MAX_LEVEL;
272
273 uint32 honor = 0;
274
275 if (GetRewHonorAddition() > 0 || GetRewHonorMultiplier() > 0.0f)
276 {
277 // values stored from 0.. for 1...
278 TeamContributionPointsEntry const* tc = sTeamContributionPointsStore.LookupEntry(level);
279 if (!tc)
280 return 0;
281
282 honor = uint32(tc->value * GetRewHonorMultiplier() * 0.1f);
283 honor += GetRewHonorAddition();
284 }
285
286 return honor;
287}

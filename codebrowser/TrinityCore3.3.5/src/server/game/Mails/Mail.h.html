184 void AddItem(ObjectGuid::LowType itemGuidLow, uint32 item_template)
185 {
186 MailItemInfo mii;
187 mii.item_guid = itemGuidLow;
188 mii.item_template = item_template;
189 items.push_back(mii);
190 }
191
192 bool RemoveItem(ObjectGuid::LowType item_guid)
193 {
194 for (MailItemInfoVec::iterator itr = items.begin(); itr != items.end(); ++itr)
195 {
196 if (itr->item_guid == item_guid)
197 {
198 items.erase(itr);
199 return true;
200 }
201 }
202 return false;
203 }
204
205 bool HasItems() const { return !items.empty(); }
206};

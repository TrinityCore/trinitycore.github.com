22// ---== RECTANGLE ==---
23RectangleBoundary::RectangleBoundary(float southX, float northX, float eastY, float westY, bool isInverted) :
24 AreaBoundary(BoundaryType::BOUNDARY_RECTANGLE, isInverted), _minX(southX), _maxX(northX), _minY(eastY), _maxY(westY) { }
25bool RectangleBoundary::IsWithinBoundaryArea(const Position* pos) const
26{
27 if (!pos)
28 return false;
29
30 return !(
31 pos->GetPositionX() < _minX ||
32 pos->GetPositionX() > _maxX ||
33 pos->GetPositionY() < _minY ||
34 pos->GetPositionY() > _maxY
35 );
36}

40CircleBoundary::CircleBoundary(Position const& center, double radius, bool isInverted) :
41 CircleBoundary(DoublePosition(center), radius, isInverted) { }
42CircleBoundary::CircleBoundary(DoublePosition const& center, double radius, bool isInverted) :
43 AreaBoundary(BoundaryType::BOUNDARY_CIRCLE, isInverted), _center(center), _radiusSq(radius*radius) { }
44CircleBoundary::CircleBoundary(Position const& center, Position const& pointOnCircle, bool isInverted) :
45 CircleBoundary(DoublePosition(center), DoublePosition(pointOnCircle), isInverted) { }
46CircleBoundary::CircleBoundary(DoublePosition const& center, DoublePosition const& pointOnCircle, bool isInverted) :
47 AreaBoundary(BoundaryType::BOUNDARY_CIRCLE, isInverted), _center(center), _radiusSq(center.GetDoubleExactDist2dSq(pointOnCircle)) { }
48bool CircleBoundary::IsWithinBoundaryArea(const Position* pos) const
49{
50 if (!pos)
51 return false;
52
53 double offX = _center.GetDoublePositionX() - pos->GetPositionX();
54 double offY = _center.GetDoublePositionY() - pos->GetPositionY();
55 return offX*offX+offY*offY <= _radiusSq;
56}

59// ---== ELLIPSE ==---
60EllipseBoundary::EllipseBoundary(Position const& center, double radiusX, double radiusY, bool isInverted) :
61 EllipseBoundary(DoublePosition(center), radiusX, radiusY, isInverted) { }
62EllipseBoundary::EllipseBoundary(DoublePosition const& center, double radiusX, double radiusY, bool isInverted) :
63 AreaBoundary(BoundaryType::BOUNDARY_ELLIPSE, isInverted), _center(center), _radiusYSq(radiusY*radiusY), _scaleXSq(_radiusYSq / (radiusX*radiusX)) { }
64bool EllipseBoundary::IsWithinBoundaryArea(const Position* pos) const
65{
66 if (!pos)
67 return false;
68
69 double offX = _center.GetDoublePositionX()-pos->GetPositionX(), offY = _center.GetDoublePositionY()-pos->GetPositionY();
70 return (offX*offX)*_scaleXSq + (offY*offY) <= _radiusYSq;
71}

74// ---== TRIANGLE ==---
75TriangleBoundary::TriangleBoundary(Position const& pointA, Position const& pointB, Position const& pointC, bool isInverted) :
76 TriangleBoundary(DoublePosition(pointA), DoublePosition(pointB), DoublePosition(pointC), isInverted) { }
77TriangleBoundary::TriangleBoundary(DoublePosition const& pointA, DoublePosition const& pointB, DoublePosition const& pointC, bool isInverted) :
78 AreaBoundary(BoundaryType::BOUNDARY_TRIANGLE, isInverted), _a(pointA), _b(pointB), _c(pointC), _abx(_b.GetDoublePositionX()-_a.GetDoublePositionX()), _bcx(_c.GetDoublePositionX()-_b.GetDoublePositionX()), _cax(_a.GetDoublePositionX() - _c.GetDoublePositionX()), _aby(_b.GetDoublePositionY()-_a.GetDoublePositionY()), _bcy(_c.GetDoublePositionY()-_b.GetDoublePositionY()), _cay(_a.GetDoublePositionY() - _c.GetDoublePositionY()) { }
79bool TriangleBoundary::IsWithinBoundaryArea(const Position* pos) const
80{
81 if (!pos)
82 return false;
83
84 // half-plane signs
85 bool sign1 = ((-_b.GetDoublePositionX() + pos->GetPositionX()) * _aby - (-_b.GetDoublePositionY() + pos->GetPositionY()) * _abx) < 0;
86 bool sign2 = ((-_c.GetDoublePositionX() + pos->GetPositionX()) * _bcy - (-_c.GetDoublePositionY() + pos->GetPositionY()) * _bcx) < 0;
87 bool sign3 = ((-_a.GetDoublePositionX() + pos->GetPositionX()) * _cay - (-_a.GetDoublePositionY() + pos->GetPositionY()) * _cax) < 0;
88
89 // if all signs are the same, the point is inside the triangle
90 return ((sign1 == sign2) && (sign2 == sign3));
91}

94// ---== PARALLELOGRAM ==---
95ParallelogramBoundary::ParallelogramBoundary(Position const& cornerA, Position const& cornerB, Position const& cornerD, bool isInverted) :
96 ParallelogramBoundary(DoublePosition(cornerA), DoublePosition(cornerB), DoublePosition(cornerD), isInverted) { }
97ParallelogramBoundary::ParallelogramBoundary(DoublePosition const& cornerA, DoublePosition const& cornerB, DoublePosition const& cornerD, bool isInverted) :
98 AreaBoundary(BoundaryType::BOUNDARY_PARALLELOGRAM, isInverted), _a(cornerA), _b(cornerB), _d(cornerD), _c(DoublePosition(_d.GetDoublePositionX() + (_b.GetDoublePositionX() - _a.GetDoublePositionX()), _d.GetDoublePositionY() + (_b.GetDoublePositionY() - _a.GetDoublePositionY()))), _abx(_b.GetDoublePositionX() - _a.GetDoublePositionX()), _dax(_a.GetDoublePositionX() - _d.GetDoublePositionX()), _aby(_b.GetDoublePositionY() - _a.GetDoublePositionY()), _day(_a.GetDoublePositionY() - _d.GetDoublePositionY()) { }
99bool ParallelogramBoundary::IsWithinBoundaryArea(const Position* pos) const
100{
101 if (!pos)
102 return false;
103
104 // half-plane signs
105 bool sign1 = ((-_b.GetDoublePositionX() + pos->GetPositionX()) * _aby - (-_b.GetDoublePositionY() + pos->GetPositionY()) * _abx) < 0;
106 bool sign2 = ((-_a.GetDoublePositionX() + pos->GetPositionX()) * _day - (-_a.GetDoublePositionY() + pos->GetPositionY()) * _dax) < 0;
107 bool sign3 = ((-_d.GetDoublePositionY() + pos->GetPositionY()) * _abx - (-_d.GetDoublePositionX() + pos->GetPositionX()) * _aby) < 0; // AB = -CD
108 bool sign4 = ((-_c.GetDoublePositionY() + pos->GetPositionY()) * _dax - (-_c.GetDoublePositionX() + pos->GetPositionX()) * _day) < 0; // DA = -BC
109
110 // if all signs are equal, the point is inside
111 return ((sign1 == sign2) && (sign2 == sign3) && (sign3 == sign4));
112}

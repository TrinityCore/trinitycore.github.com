65Map* MapManager::CreateBaseMap(uint32 id)
66{
67 Map* map = FindBaseMap(id);
68
69 if (map == NULL)
70 {
71 std::lock_guard<std::mutex> lock(_mapsLock);
72
73 MapEntry const* entry = sMapStore.LookupEntry(id);
74 ASSERT(entry);
75
76 if (entry->Instanceable())
77 map = new MapInstanced(id, i_gridCleanUpDelay);
78 else
79 {
80 map = new Map(id, i_gridCleanUpDelay, 0, REGULAR_DIFFICULTY);
81 map->LoadRespawnTimes();
82 map->LoadCorpseData();
83 }
84
85 i_maps[id] = map;
86 }
87
88 ASSERT(map);
89 return map;
90}

122Map::EnterState MapManager::PlayerCannotEnter(uint32 mapid, Player* player, bool loginCheck)
123{
124 MapEntry const* entry = sMapStore.LookupEntry(mapid);
125 if (!entry)
126 return Map::CANNOT_ENTER_NO_ENTRY;
127
128 if (!entry->IsDungeon())
129 return Map::CAN_ENTER;
130
131 InstanceTemplate const* instance = sObjectMgr->GetInstanceTemplate(mapid);
132 if (!instance)
133 return Map::CANNOT_ENTER_UNINSTANCED_DUNGEON;
134
135 Difficulty targetDifficulty, requestedDifficulty;
136 targetDifficulty = requestedDifficulty = player->GetDifficulty(entry->IsRaid());
137 // Get the highest available difficulty if current setting is higher than the instance allows
138 MapDifficulty const* mapDiff = GetDownscaledMapDifficultyData(entry->MapID, targetDifficulty);
139 if (!mapDiff)
140 return Map::CANNOT_ENTER_DIFFICULTY_UNAVAILABLE;
141
142 //Bypass checks for GMs
143 if (player->IsGameMaster())
144 return Map::CAN_ENTER;
145
146 char const* mapName = entry->name[player->GetSession()->GetSessionDbcLocale()];
147
148 Group* group = player->GetGroup();
149 if (entry->IsRaid()) // can only enter in a raid group
150 if ((!group || !group->isRaidGroup()) && !sWorld->getBoolConfig(CONFIG_INSTANCE_IGNORE_RAID))
151 return Map::CANNOT_ENTER_NOT_IN_RAID;
152
153 if (!player->IsAlive())
154 {
155 if (player->HasCorpse())
156 {
157 // let enter in ghost mode in instance that connected to inner instance with corpse
158 uint32 corpseMap = player->GetCorpseLocation().GetMapId();
159 do
160 {
161 if (corpseMap == mapid)
162 break;
163
164 InstanceTemplate const* corpseInstance = sObjectMgr->GetInstanceTemplate(corpseMap);
165 corpseMap = corpseInstance ? corpseInstance->Parent : 0;
166 } while (corpseMap);
167
168 if (!corpseMap)
169 return Map::CANNOT_ENTER_CORPSE_IN_DIFFERENT_INSTANCE;
170
171 TC_LOG_DEBUG("maps", "MAP: Player '%s' has corpse in instance '%s' and can enter.", player->GetName().c_str(), mapName);
172 }
173 else
174 TC_LOG_DEBUG("maps", "Map::CanPlayerEnter - player '%s' is dead but does not have a corpse!", player->GetName().c_str());
175 }
176
177 //Get instance where player's group is bound & its map
178 if (!loginCheck && group)
179 {
180 InstanceGroupBind* boundInstance = group->GetBoundInstance(entry);
181 if (boundInstance && boundInstance->save)
182 if (Map* boundMap = sMapMgr->FindMap(mapid, boundInstance->save->GetInstanceId()))
183 if (Map::EnterState denyReason = boundMap->CannotEnter(player))
184 return denyReason;
185 }
186
187 // players are only allowed to enter 5 instances per hour
188 if (entry->IsDungeon() && (!player->GetGroup() || (player->GetGroup() && !player->GetGroup()->isLFGGroup())))
189 {
190 uint32 instanceIdToCheck = 0;
191 if (InstanceSave* save = player->GetInstanceSave(mapid, entry->IsRaid()))
192 instanceIdToCheck = save->GetInstanceId();
193
194 // instanceId can never be 0 - will not be found
195 if (!player->CheckInstanceCount(instanceIdToCheck) && !player->isDead())
196 return Map::CANNOT_ENTER_TOO_MANY_INSTANCES;
197 }
198
199 //Other requirements
200 if (player->Satisfy(sObjectMgr->GetAccessRequirement(mapid, targetDifficulty), mapid, true))
201 return Map::CAN_ENTER;
202 else
203 return Map::CANNOT_ENTER_UNSPECIFIED_REASON;
204}

206void MapManager::Update(uint32 diff)
207{
208 i_timer.Update(diff);
209 if (!i_timer.Passed())
210 return;
211
212 MapMapType::iterator iter = i_maps.begin();
213 for (; iter != i_maps.end(); ++iter)
214 {
215 if (m_updater.activated())
216 m_updater.schedule_update(*iter->second, uint32(i_timer.GetCurrent()));
217 else
218 iter->second->Update(uint32(i_timer.GetCurrent()));
219 }
220 if (m_updater.activated())
221 m_updater.wait();
222
223 for (iter = i_maps.begin(); iter != i_maps.end(); ++iter)
224 iter->second->DelayedUpdate(uint32(i_timer.GetCurrent()));
225
226 i_timer.SetCurrent(0);
227}

231bool MapManager::ExistMapAndVMap(uint32 mapid, float x, float y)
232{
233 GridCoord p = Trinity::ComputeGridCoord(x, y);
234
235 int gx = (MAX_NUMBER_OF_GRIDS - 1) - p.x_coord;
236 int gy = (MAX_NUMBER_OF_GRIDS - 1) - p.y_coord;
237
238 return Map::ExistMap(mapid, gx, gy) && Map::ExistVMap(mapid, gx, gy);
239}

268uint32 MapManager::GetNumInstances()
269{
270 std::lock_guard<std::mutex> lock(_mapsLock);
271
272 uint32 ret = 0;
273 for (MapMapType::iterator itr = i_maps.begin(); itr != i_maps.end(); ++itr)
274 {
275 Map* map = itr->second;
276 if (!map->Instanceable())
277 continue;
278 MapInstanced::InstancedMaps &maps = ((MapInstanced*)map)->GetInstancedMaps();
279 for (MapInstanced::InstancedMaps::iterator mitr = maps.begin(); mitr != maps.end(); ++mitr)
280 if (mitr->second->IsDungeon()) ret++;
281 }
282 return ret;
283}

303void MapManager::InitInstanceIds()
304{
305 _nextInstanceId = 1;
306
307 QueryResult result = CharacterDatabase.Query("SELECT MAX(id) FROM instance");
308 if (result)
309 {
310 uint32 maxId = (*result)[0].GetUInt32();
311
312 // Resize to multiples of 32 (vector<bool> allocates memory the same way)
313 _instanceIds.resize((maxId / 32) * 32 + (maxId % 32 > 0 ? 32 : 0));
314 }
315}

323uint32 MapManager::GenerateInstanceId()
324{
325 uint32 newInstanceId = _nextInstanceId;
326
327 // Find the lowest available id starting from the current NextInstanceId (which should be the lowest according to the logic in FreeInstanceId()
328 for (uint32 i = ++_nextInstanceId; i < 0xFFFFFFFF; ++i)
329 {
330 if ((i < _instanceIds.size() && !_instanceIds[i]) || i >= _instanceIds.size())
331 {
332 _nextInstanceId = i;
333 break;
334 }
335 }
336
337 if (newInstanceId == _nextInstanceId)
338 {
339 TC_LOG_ERROR("maps", "Instance ID overflow!! Can't continue, shutting down server. ");
340 World::StopNow(ERROR_EXIT_CODE);
341 }
342
343 // Allocate space if necessary
344 if (newInstanceId >= uint32(_instanceIds.size()))
345 {
346 // Due to the odd memory allocation behavior of vector<bool> we match size to capacity before triggering a new allocation
347 if (_instanceIds.size() < _instanceIds.capacity())
348 {
349 _instanceIds.resize(_instanceIds.capacity());
350 }
351 else
352 _instanceIds.resize((newInstanceId / 32) * 32 + (newInstanceId % 32 > 0 ? 32 : 0));
353 }
354
355 _instanceIds[newInstanceId] = true;
356
357 return newInstanceId;
358}

360void MapManager::FreeInstanceId(uint32 instanceId)
361{
362 // If freed instance id is lower than the next id available for new instances, use the freed one instead
363 if (instanceId < _nextInstanceId)
364 SetNextInstanceId(instanceId);
365
366 _instanceIds[instanceId] = false;
367 sAchievementMgr->OnInstanceDestroyed(instanceId);
368}

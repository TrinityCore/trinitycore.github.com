366 enum EnterState
367 {
368 CAN_ENTER = 0,
369 CANNOT_ENTER_ALREADY_IN_MAP = 1, // Player is already in the map
370 CANNOT_ENTER_NO_ENTRY, // No map entry was found for the target map ID
371 CANNOT_ENTER_UNINSTANCED_DUNGEON, // No instance template was found for dungeon map
372 CANNOT_ENTER_DIFFICULTY_UNAVAILABLE, // Requested instance difficulty is not available for target map
373 CANNOT_ENTER_NOT_IN_RAID, // Target instance is a raid instance and the player is not in a raid group
374 CANNOT_ENTER_CORPSE_IN_DIFFERENT_INSTANCE, // Player is dead and their corpse is not in target instance
375 CANNOT_ENTER_INSTANCE_BIND_MISMATCH, // Player's permanent instance save is not compatible with their group's current instance bind
376 CANNOT_ENTER_TOO_MANY_INSTANCES, // Player has entered too many instances recently
377 CANNOT_ENTER_MAX_PLAYERS, // Target map already has the maximum number of players allowed
378 CANNOT_ENTER_ZONE_IN_COMBAT, // A boss encounter is currently in progress on the target map
379 CANNOT_ENTER_UNSPECIFIED_REASON
380 };

808template<class T, class CONTAINER>
809inline void Map::Visit(Cell const& cell, TypeContainerVisitor<T, CONTAINER>& visitor)
810{
811 const uint32 x = cell.GridX();
812 const uint32 y = cell.GridY();
813 const uint32 cell_x = cell.CellX();
814 const uint32 cell_y = cell.CellY();
815
816 if (!cell.NoCreate() || IsGridLoaded(GridCoord(x, y)))
817 {
818 EnsureGridLoaded(cell);
819 getNGrid(x, y)->VisitGrid(cell_x, cell_y, visitor);
820 }
821}

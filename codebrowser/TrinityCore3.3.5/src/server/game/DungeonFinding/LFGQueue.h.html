26enum LfgCompatibility
27{
28 LFG_COMPATIBILITY_PENDING,
29 LFG_INCOMPATIBLES_WRONG_GROUP_SIZE,
30 LFG_INCOMPATIBLES_TOO_MUCH_PLAYERS,
31 LFG_INCOMPATIBLES_MULTIPLE_LFG_GROUPS,
32 LFG_INCOMPATIBLES_HAS_IGNORES,
33 LFG_INCOMPATIBLES_NO_ROLES,
34 LFG_INCOMPATIBLES_NO_DUNGEONS,
35 LFG_COMPATIBLES_WITH_LESS_PLAYERS, // Values under this = not compatible (do not modify order)
36 LFG_COMPATIBLES_BAD_STATES,
37 LFG_COMPATIBLES_MATCH // Must be the last one
38};
39
40struct LfgCompatibilityData
41{
42 LfgCompatibilityData(): compatibility(LFG_COMPATIBILITY_PENDING) { }
43 LfgCompatibilityData(LfgCompatibility _compatibility): compatibility(_compatibility) { }
44 LfgCompatibilityData(LfgCompatibility _compatibility, LfgRolesMap const& _roles):
45 compatibility(_compatibility), roles(_roles) { }
46
47 LfgCompatibility compatibility;
48 LfgRolesMap roles;
49};

51/// Stores player or group queue info
52struct LfgQueueData
53{
54 LfgQueueData(): joinTime(time_t(time(NULL))), tanks(LFG_TANKS_NEEDED),
55 healers(LFG_HEALERS_NEEDED), dps(LFG_DPS_NEEDED)
56 { }
57
58 LfgQueueData(time_t _joinTime, LfgDungeonSet const& _dungeons, LfgRolesMap const& _roles):
59 joinTime(_joinTime), tanks(LFG_TANKS_NEEDED), healers(LFG_HEALERS_NEEDED),
60 dps(LFG_DPS_NEEDED), dungeons(_dungeons), roles(_roles)
61 { }
62
63 time_t joinTime; ///< Player queue join time (to calculate wait times)
64 uint8 tanks; ///< Tanks needed
65 uint8 healers; ///< Healers needed
66 uint8 dps; ///< Dps needed
67 LfgDungeonSet dungeons; ///< Selected Player/Group Dungeon/s
68 LfgRolesMap roles; ///< Selected Player Role/s
69 std::string bestCompatible; ///< Best compatible combination of people queued
70};
71
72struct LfgWaitTime
73{
74 LfgWaitTime(): time(-1), number(0) { }
75 int32 time; ///< Wait time
76 uint32 number; ///< Number of people used to get that wait time
77};

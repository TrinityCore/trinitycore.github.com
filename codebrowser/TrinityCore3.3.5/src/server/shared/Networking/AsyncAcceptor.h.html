18#ifndef __ASYNCACCEPT_H_
19#define __ASYNCACCEPT_H_
20
21#include "Log.h"
22#include <boost/asio.hpp>
23#include <functional>
24
25using boost::asio::ip::tcp;
26
27class AsyncAcceptor
28{
29public:
30 typedef void(*AcceptCallback)(tcp::socket&& newSocket, uint32 threadIndex);
31
32 AsyncAcceptor(boost::asio::io_service& ioService, std::string const& bindIp, uint16 port) :
33 _acceptor(ioService, tcp::endpoint(boost::asio::ip::address::from_string(bindIp), port)),
34 _socket(ioService), _closed(false), _socketFactory(std::bind(&AsyncAcceptor::DefeaultSocketFactory, this))
35 {
36 }
37
38 template<class T>
39 void AsyncAccept();
40
41 template<AcceptCallback acceptCallback>
42 void AsyncAcceptWithCallback()
43 {
44 tcp::socket* socket;
45 uint32 threadIndex;
46 std::tie(socket, threadIndex) = _socketFactory();
47 _acceptor.async_accept(*socket, [this, socket, threadIndex](boost::system::error_code error)
48 {
49 if (!error)
50 {
51 try
52 {
53 socket->non_blocking(true);
54
55 acceptCallback(std::move(*socket), threadIndex);
56 }
57 catch (boost::system::system_error const& err)
58 {
59 TC_LOG_INFO("network", "Failed to initialize client's socket %s", err.what());
60 }
61 }
62
63 if (!_closed)
64 this->AsyncAcceptWithCallback<acceptCallback>();
65 });
66 }
67
68 void Close()
69 {
70 if (_closed.exchange(true))
71 return;
72
73 boost::system::error_code err;
74 _acceptor.close(err);
75 }
76
77 void SetSocketFactory(std::function<std::pair<tcp::socket*, uint32>()> func) { _socketFactory = func; }
78
79private:
80 std::pair<tcp::socket*, uint32> DefeaultSocketFactory() { return std::make_pair(&_socket, 0); }
81
82 tcp::acceptor _acceptor;
83 tcp::socket _socket;
84 std::atomic<bool> _closed;
85 std::function<std::pair<tcp::socket*, uint32>()> _socketFactory;
86};
87
88template<class T>
89void AsyncAcceptor::AsyncAccept()
90{
91 _acceptor.async_accept(_socket, [this](boost::system::error_code error)
92 {
93 if (!error)
94 {
95 try
96 {
97 // this-> is required here to fix an segmentation fault in gcc 4.7.2 - reason is lambdas in a templated class
98 std::make_shared<T>(std::move(this->_socket))->Start();
99 }
100 catch (boost::system::system_error const& err)
101 {
102 TC_LOG_INFO("network", "Failed to retrieve client's remote address %s", err.what());
103 }
104 }
105
106 // lets slap some more this-> on this so we can fix this bug with gcc 4.7.2 throwing internals in yo face
107 if (!_closed)
108 this->AsyncAccept<T>();
109 });
110}
111
112#endif /* __ASYNCACCEPT_H_ */

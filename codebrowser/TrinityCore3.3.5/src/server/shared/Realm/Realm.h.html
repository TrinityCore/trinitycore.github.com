27enum RealmFlags
28{
29 REALM_FLAG_NONE = 0x00,
30 REALM_FLAG_VERSION_MISMATCH = 0x01,
31 REALM_FLAG_OFFLINE = 0x02,
32 REALM_FLAG_SPECIFYBUILD = 0x04,
33 REALM_FLAG_UNK1 = 0x08,
34 REALM_FLAG_UNK2 = 0x10,
35 REALM_FLAG_RECOMMENDED = 0x20,
36 REALM_FLAG_NEW = 0x40,
37 REALM_FLAG_FULL = 0x80
38};
39
40struct RealmHandle
41{
42 RealmHandle() : Realm(0) { }
43 RealmHandle(uint32 index) : Realm(index) { }
44
45 uint32 Realm; // primary key in `realmlist` table
46
47 bool operator<(RealmHandle const& r) const
48 {
49 return Realm < r.Realm;
50 }
51};
52
53/// Type of server, this is values from second column of Cfg_Configs.dbc
54enum RealmType
55{
56 REALM_TYPE_NORMAL = 0,
57 REALM_TYPE_PVP = 1,
58 REALM_TYPE_NORMAL2 = 4,
59 REALM_TYPE_RP = 6,
60 REALM_TYPE_RPPVP = 8,
61
62 MAX_CLIENT_REALM_TYPE = 14,
63
64 REALM_TYPE_FFA_PVP = 16 // custom, free for all pvp mode like arena PvP in all zones except rest activated places and sanctuaries
65 // replaced by REALM_PVP in realm list
66};
67
68// Storage object for a realm
69struct Realm
70{
71 RealmHandle Id;
72 uint32 Build;
73 ip::address ExternalAddress;
74 ip::address LocalAddress;
75 ip::address LocalSubnetMask;
76 uint16 Port;
77 std::string Name;
78 uint8 Type; // icon
79 RealmFlags Flags;
80 uint8 Timezone;
81 AccountTypes AllowedSecurityLevel;
82 float PopulationLevel;
83};

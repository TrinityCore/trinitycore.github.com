360 void readPackGUID(uint64& guid)
361 {
362 if (rpos() + 1 > size())
363 throw ByteBufferPositionException(false, _rpos, 1, size());
364
365 guid = 0;
366
367 uint8 guidmark = 0;
368 (*this) >> guidmark;
369
370 for (int i = 0; i < 8; ++i)
371 {
372 if (guidmark & (uint8(1) << i))
373 {
374 if (rpos() + 1 > size())
375 throw ByteBufferPositionException(false, _rpos, 1, size());
376
377 uint8 bit;
378 (*this) >> bit;
379 guid |= (uint64(bit) << (i * 8));
380 }
381 }

384 uint32 ReadPackedTime()
385 {
386 uint32 packedDate = read<uint32>();
387 tm lt = tm();
388
389 lt.tm_min = packedDate & 0x3F;
390 lt.tm_hour = (packedDate >> 6) & 0x1F;
391 //lt.tm_wday = (packedDate >> 11) & 7;
392 lt.tm_mday = ((packedDate >> 14) & 0x3F) + 1;
393 lt.tm_mon = (packedDate >> 20) & 0xF;
394 lt.tm_year = ((packedDate >> 24) & 0x1F) + 100;
395
396 return uint32(mktime(&lt));
397 }

445 void append(const uint8 *src, size_t cnt)
446 {
447 if (!cnt)
448 throw ByteBufferSourceException(_wpos, size(), cnt);
449
450 if (!src)
451 throw ByteBufferSourceException(_wpos, size(), cnt);
452
453 ASSERT(size() < 10000000);
454
455 if (_storage.size() < _wpos + cnt)
456 _storage.resize(_wpos + cnt);
457 std::memcpy(&_storage[_wpos], src, cnt);
458 _wpos += cnt;
459 }

467 // can be used in SMSG_MONSTER_MOVE opcode
468 void appendPackXYZ(float x, float y, float z)
469 {
470 uint32 packed = 0;
471 packed |= ((int)(x / 0.25f) & 0x7FF);
472 packed |= ((int)(y / 0.25f) & 0x7FF) << 11;
473 packed |= ((int)(z / 0.25f) & 0x3FF) << 22;
474 *this << packed;
475 }

477 void appendPackGUID(uint64 guid)
478 {
479 uint8 packGUID[8+1];
480 packGUID[0] = 0;
481 size_t size = 1;
482 for (uint8 i = 0;guid != 0;++i)
483 {
484 if (guid & 0xFF)
485 {
486 packGUID[0] |= uint8(1 << i);
487 packGUID[size] = uint8(guid & 0xFF);
488 ++size;
489 }
490
491 guid >>= 8;
492 }
493 append(packGUID, size);
494 }

496 void AppendPackedTime(time_t time)
497 {
498 tm lt;
499 localtime_r(&time, &lt);
500 append<uint32>((lt.tm_year - 100) << 24 | lt.tm_mon << 20 | (lt.tm_mday - 1) << 14 | lt.tm_wday << 11 | lt.tm_hour << 6 | lt.tm_min);
501 }

503 void put(size_t pos, const uint8 *src, size_t cnt)
504 {
505 if (pos + cnt > size())
506 throw ByteBufferPositionException(true, pos, cnt, size());
507
508 if (!src)
509 throw ByteBufferSourceException(_wpos, size(), cnt);

590{
591 uint32 msize;
592 b >> msize;
593 m.clear();
594 while (msize--)
595 {
596 K k;
597 V v;
598 b >> k >> v;
599 m.insert(make_pair(k, v));
600 }
601 return b;

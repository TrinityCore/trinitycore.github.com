19#ifndef TRINITYCORE_WORLDPACKET_H
20#define TRINITYCORE_WORLDPACKET_H
21
22#include "Common.h"
23#include "ByteBuffer.h"
24
25class WorldPacket : public ByteBuffer
26{
27 public:
28 // just container for later use
29 WorldPacket() : ByteBuffer(0), m_opcode(0)
30 {
31 }
32
33 explicit WorldPacket(uint16 opcode, size_t res=200) : ByteBuffer(res), m_opcode(opcode) { }
34
35 WorldPacket(WorldPacket&& packet) : ByteBuffer(std::move(packet)), m_opcode(packet.m_opcode)
36 {
37 }
38
39 WorldPacket(WorldPacket const& right) : ByteBuffer(right), m_opcode(right.m_opcode)
40 {
41 }
42
43 WorldPacket& operator=(WorldPacket const& right)
44 {
45 if (this != &right)
46 {
47 m_opcode = right.m_opcode;
48 ByteBuffer::operator =(right);
49 }
50
51 return *this;
52 }
53
54 WorldPacket(uint16 opcode, MessageBuffer&& buffer) : ByteBuffer(std::move(buffer)), m_opcode(opcode) { }
55
56 void Initialize(uint16 opcode, size_t newres=200)
57 {
58 clear();
59 _storage.reserve(newres);
60 m_opcode = opcode;
61 }
62
63 uint16 GetOpcode() const { return m_opcode; }
64 void SetOpcode(uint16 opcode) { m_opcode = opcode; }
65
66 protected:
67 uint16 m_opcode;
68};
69
70#endif

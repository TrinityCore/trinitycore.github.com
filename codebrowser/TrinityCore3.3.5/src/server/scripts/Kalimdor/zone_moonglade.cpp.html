271 void IsSummonedBy(Unit* /*summoner*/) override
272 {
273 std::list<Player*> playerOnQuestList;
274 Trinity::AnyPlayerInObjectRangeCheck checker(me, 5.0f);
275 Trinity::PlayerListSearcher<Trinity::AnyPlayerInObjectRangeCheck> searcher(me, playerOnQuestList, checker);
276 me->VisitNearbyWorldObject(5.0f, searcher);
277 for (std::list<Player*>::const_iterator itr = playerOnQuestList.begin(); itr != playerOnQuestList.end(); ++itr)
278 {
279 // Check if found player target has active quest
280 if (Player* player = (*itr))
281 {
282 if (player->GetQuestStatus(10965) == QUEST_STATUS_INCOMPLETE)
283 {
284 StartEvent(player);
285 break;
286 }
287 }
288 else
289 break;
290 }
291 }

293 void JustDied(Unit* /*killer*/) override
294 {
295 if (!PlayerGUID)
296 return;
297
298 Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
299 if (player && player->GetQuestStatus(10965) == QUEST_STATUS_INCOMPLETE)
300 {
301 player->FailQuest(10965);
302 PlayerGUID.Clear();
303 Reset();
304 }
305 }

307 void EnterEvadeMode(EvadeReason why) override
308 {
309 Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
310 if (player && player->IsInCombat() && player->getAttackerForHelper())
311 {
312 AttackStart(player->getAttackerForHelper());
313 return;
314 }
315 npc_escortAI::EnterEvadeMode(why);
316 }

318 void StartEvent(Player* player)
319 {
320 if (player && player->GetQuestStatus(10965) == QUEST_STATUS_INCOMPLETE)
321 {
322 for (uint8 i = 0; i < 41; ++i)
323 {
324 AddWaypoint(i, Clintar_spirit_WP[i][0], Clintar_spirit_WP[i][1], Clintar_spirit_WP[i][2], (uint32)Clintar_spirit_WP[i][4]);

328 me->SetDisplayId(me->GetCreatureTemplate()->Modelid1);
329 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);

405 break;
406 }
407 break;
408 case 16:

533 struct npc_omenAI : public ScriptedAI
534 {
535 npc_omenAI(Creature* creature) : ScriptedAI(creature)
536 {
537 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
538 me->GetMotionMaster()->MovePoint(1, 7549.977f, -2855.137f, 456.9678f);

541 EventMap events;
542
543 void MovementInform(uint32 type, uint32 pointId) override
544 {
545 if (type != POINT_MOTION_TYPE)
546 return;
547
548 if (pointId == 1)
549 {
550 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
551 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
552 if (Player* player = me->SelectNearestPlayer(40.0f))
553 AttackStart(player);
554 }
555 }
556
557 void EnterCombat(Unit* /*attacker*/) override
558 {
559 events.Reset();
560 events.ScheduleEvent(EVENT_CAST_CLEAVE, urand(3000, 5000));
561 events.ScheduleEvent(EVENT_CAST_STARFALL, urand(8000, 10000));
562 }
563
564 void JustDied(Unit* /*killer*/) override
565 {
566 DoCast(SPELL_OMEN_SUMMON_SPOTLIGHT);
567 }
568
569 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
570 {
571 if (spell->Id == SPELL_ELUNE_CANDLE)
572 {
573 if (me->HasAura(SPELL_OMEN_STARFALL))
574 me->RemoveAurasDueToSpell(SPELL_OMEN_STARFALL);
575
576 events.RescheduleEvent(EVENT_CAST_STARFALL, urand(14000, 16000));
577 }
578 }
579
580 void UpdateAI(uint32 diff) override
581 {
582 if (!UpdateVictim())
583 return;
584
585 events.Update(diff);
586
587 switch (events.ExecuteEvent())
588 {
589 case EVENT_CAST_CLEAVE:
590 DoCastVictim(SPELL_OMEN_CLEAVE);
591 events.ScheduleEvent(EVENT_CAST_CLEAVE, urand(8000, 10000));
592 break;
593 case EVENT_CAST_STARFALL:
594 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
595 DoCast(target, SPELL_OMEN_STARFALL);
596 events.ScheduleEvent(EVENT_CAST_STARFALL, urand(14000, 16000));
597 break;
598 }
599
600 DoMeleeAttackIfReady();
601 }
602 };
603
604 CreatureAI* GetAI(Creature* creature) const override
605 {
606 return new npc_omenAI(creature);
607 }

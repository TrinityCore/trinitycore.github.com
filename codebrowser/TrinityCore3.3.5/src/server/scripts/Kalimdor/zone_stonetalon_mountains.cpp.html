47class npc_braug_dimspirit : public CreatureScript
48{
49public:
50 npc_braug_dimspirit() : CreatureScript("npc_braug_dimspirit") { }
51
52 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
53 {
54 player->PlayerTalkClass->ClearMenus();
55 if (action == GOSSIP_ACTION_INFO_DEF+1)
56 {
57 player->CLOSE_GOSSIP_MENU();
58 creature->CastSpell(player, 6766, false);
59
60 }
61 if (action == GOSSIP_ACTION_INFO_DEF+2)
62 {
63 player->CLOSE_GOSSIP_MENU();
64 player->AreaExploredOrEventHappens(6627);
65 }
66 return true;
67 }
68
69 bool OnGossipHello(Player* player, Creature* creature) override
70 {
71 if (creature->IsQuestGiver())
72 player->PrepareQuestMenu(creature->GetGUID());
73
74 if (player->GetQuestStatus(6627) == QUEST_STATUS_INCOMPLETE)
75 {
76 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HBD1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
77 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HBD2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+2);
78 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HBD3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
79 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HBD4, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
80 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HBD5, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
81
82 player->SEND_GOSSIP_MENU(5820, creature->GetGUID());
83 }
84 else
85 player->SEND_GOSSIP_MENU(5819, creature->GetGUID());
86
87 return true;
88 }

106 SAY_END = 2,

111class npc_kaya_flathoof : public CreatureScript
112{
113public:
114 npc_kaya_flathoof() : CreatureScript("npc_kaya_flathoof") { }
115
116 struct npc_kaya_flathoofAI : public npc_escortAI
117 {
118 npc_kaya_flathoofAI(Creature* creature) : npc_escortAI(creature) { }
119
120 void WaypointReached(uint32 waypointId) override
121 {
122 Player* player = GetPlayerForEscort();
123 if (!player)
124 return;
125
126 switch (waypointId)
127 {
128 case 16:
129 Talk(SAY_AMBUSH);
130 me->SummonCreature(NPC_GRIMTOTEM_BRUTE, -48.53f, -503.34f, -46.31f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
131 me->SummonCreature(NPC_GRIMTOTEM_RUFFIAN, -38.85f, -503.77f, -45.90f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
132 me->SummonCreature(NPC_GRIMTOTEM_SORCERER, -36.37f, -496.23f, -45.71f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
133 break;
134 case 18:
135 me->SetInFront(player);
136 Talk(SAY_END);
137 player->GroupEventHappens(QUEST_PROTECT_KAYA, me);
138 break;
139 }
140 }
141
142 void JustSummoned(Creature* summoned) override
143 {
144 summoned->AI()->AttackStart(me);
145 }
146
147 void Reset() override { }
148 };
149
150 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
151 {
152 if (quest->GetQuestId() == QUEST_PROTECT_KAYA)
153 {
154 if (npc_escortAI* pEscortAI = CAST_AI(npc_kaya_flathoof::npc_kaya_flathoofAI, creature->AI()))
155 pEscortAI->Start(true, false, player->GetGUID());
156
157 creature->AI()->Talk(SAY_START);
158 creature->setFaction(113);
159 creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
160 }
161 return true;
162 }
163
164 CreatureAI* GetAI(Creature* creature) const override
165 {
166 return new npc_kaya_flathoofAI(creature);
167 }
168
169};

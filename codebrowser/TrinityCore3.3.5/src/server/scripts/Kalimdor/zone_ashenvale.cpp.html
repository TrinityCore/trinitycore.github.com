102 Cage->SetGoState(GO_STATE_ACTIVE);
103 break;
104 case 13:
105 me->SummonCreature(NPC_THISTLEFUR_TOTEMIC, RuulSnowhoofSummonsCoord[0], TEMPSUMMON_DEAD_DESPAWN, 60000);
106 me->SummonCreature(NPC_THISTLEFUR_URSA, RuulSnowhoofSummonsCoord[1], TEMPSUMMON_DEAD_DESPAWN, 60000);
107 me->SummonCreature(NPC_THISTLEFUR_PATHFINDER, RuulSnowhoofSummonsCoord[2], TEMPSUMMON_DEAD_DESPAWN, 60000);

177class npc_muglash : public CreatureScript
178{
179public:
180 npc_muglash() : CreatureScript("npc_muglash") { }
181
182 struct npc_muglashAI : public npc_escortAI
183 {
184 npc_muglashAI(Creature* creature) : npc_escortAI(creature)
185 {
186 Initialize();
187 }
188
189 void Initialize()
190 {
191 eventTimer = 10000;
192 waveId = 0;
193 _isBrazierExtinguished = false;
194 }
195
196 void Reset() override
197 {
198 Initialize();
199 }
200
201 void EnterCombat(Unit* /*who*/) override
202 {
203 if (Player* player = GetPlayerForEscort())
204 if (HasEscortState(STATE_ESCORT_PAUSED))
205 {
206 if (urand(0, 1))
207 Talk(SAY_MUG_ON_GUARD, player);
208 return;
209 }
210 }
211
212 void JustDied(Unit* /*killer*/) override
213 {
214 if (HasEscortState(STATE_ESCORT_ESCORTING))
215 if (Player* player = GetPlayerForEscort())
216 player->FailQuest(QUEST_VORSHA);
217 }
218
219 void JustSummoned(Creature* summoned) override
220 {
221 summoned->AI()->AttackStart(me);
222 }
223
224 void sQuestAccept(Player* player, Quest const* quest) override
225 {
226 if (quest->GetQuestId() == QUEST_VORSHA)
227 {
228 Talk(SAY_MUG_START1);
229 me->setFaction(FACTION_QUEST);
230 npc_escortAI::Start(true, false, player->GetGUID());
231 }
232 }
233
234 void WaypointReached(uint32 waypointId) override
235 {
236 if (Player* player = GetPlayerForEscort())
237 {
238 switch (waypointId)
239 {
240 case 0:
241 Talk(SAY_MUG_START2, player);
242 break;
243 case 24:
244 Talk(SAY_MUG_BRAZIER, player);
245
246 if (GameObject* go = GetClosestGameObjectWithEntry(me, GO_NAGA_BRAZIER, INTERACTION_DISTANCE*2))
247 {
248 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
249 SetEscortPaused(true);
250 }
251 break;
252 case 25:
253 Talk(SAY_MUG_GRATITUDE);
254 player->GroupEventHappens(QUEST_VORSHA, me);
255 break;
256 case 26:
257 Talk(SAY_MUG_PATROL);
258 break;
259 case 27:
260 Talk(SAY_MUG_RETURN);
261 break;
262 }
263 }
264 }
265
266 void DoWaveSummon()
267 {
268 switch (waveId)
269 {
270 case 1:
271 me->SummonCreature(NPC_WRATH_RIDER, FirstNagaCoord[0], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
272 me->SummonCreature(NPC_WRATH_SORCERESS, FirstNagaCoord[1], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
273 me->SummonCreature(NPC_WRATH_RAZORTAIL, FirstNagaCoord[2], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
274 break;
275 case 2:
276 me->SummonCreature(NPC_WRATH_PRIESTESS, SecondNagaCoord[0], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
277 me->SummonCreature(NPC_WRATH_MYRMIDON, SecondNagaCoord[1], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
278 me->SummonCreature(NPC_WRATH_SEAWITCH, SecondNagaCoord[2], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
279 break;
280 case 3:
281 me->SummonCreature(NPC_VORSHA, VorshaCoord, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
282 break;
283 case 4:
284 SetEscortPaused(false);
285 Talk(SAY_MUG_DONE);
286 break;
287 }
288 }
289
290 void UpdateAI(uint32 diff) override
291 {
292 npc_escortAI::UpdateAI(diff);
293
294 if (!me->GetVictim())
295 {
296 if (HasEscortState(STATE_ESCORT_PAUSED) && _isBrazierExtinguished)
297 {
298 if (eventTimer < diff)
299 {
300 ++waveId;
301 DoWaveSummon();
302 eventTimer = 10000;
303 }
304 else
305 eventTimer -= diff;
306 }
307 return;
308 }
309 DoMeleeAttackIfReady();
310 }
311
312 private:
313 uint32 eventTimer;
314 uint8 waveId;
315 public:
316 bool _isBrazierExtinguished;
317
318 };
319
320 CreatureAI* GetAI(Creature* creature) const override
321 {
322 return new npc_muglashAI(creature);
323 }

331 bool OnGossipHello(Player* /*player*/, GameObject* go) override
332 {

334 {
335 if (npc_muglash::npc_muglashAI* pEscortAI = CAST_AI(npc_muglash::npc_muglashAI, creature->AI()))
336 {
337 creature->AI()->Talk(SAY_MUG_BRAZIER_WAIT);
338
339 pEscortAI->_isBrazierExtinguished = true;
340 return false;
341 }
342 }
343
344 return true;
345 }

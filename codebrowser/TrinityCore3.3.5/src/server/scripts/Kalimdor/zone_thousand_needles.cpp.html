61 {
62 if (quest->GetQuestId() == QUEST_PROTECT_KANATI)
63 if (npc_kanatiAI* pEscortAI = CAST_AI(npc_kanati::npc_kanatiAI, creature->AI()))
64 pEscortAI->Start(false, false, player->GetGUID(), quest, true);
65

75 {
76 npc_kanatiAI(Creature* creature) : npc_escortAI(creature) { }
77
78 void Reset() override { }
79
80 void WaypointReached(uint32 waypointId) override
81 {
82 switch (waypointId)
83 {
84 case 0:
85 Talk(SAY_KAN_START);
86 DoSpawnGalak();
87 break;
88 case 1:
89 if (Player* player = GetPlayerForEscort())
90 player->GroupEventHappens(QUEST_PROTECT_KANATI, me);
91 break;
92 }
93 }
94
95 void DoSpawnGalak()
96 {
97 for (int i = 0; i < 3; ++i)
98 me->SummonCreature(NPC_GALAK_ASS, GalakLoc, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
99 }
100
101 void JustSummoned(Creature* summoned) override
102 {
103 summoned->AI()->AttackStart(me);
104 }

292class npc_plucky : public CreatureScript
293{
294public:
295 npc_plucky() : CreatureScript("npc_plucky") { }
296
297 bool OnGossipSelect(Player* player, Creature* /*creature*/, uint32 /*sender*/, uint32 action) override
298 {
299 player->PlayerTalkClass->ClearMenus();
300 switch (action)
301 {
302 case GOSSIP_ACTION_INFO_DEF+1:
303 player->CLOSE_GOSSIP_MENU();
304 player->CompleteQuest(QUEST_SCOOP);
305 break;
306 }
307 return true;
308 }
309
310 bool OnGossipHello(Player* player, Creature* creature) override
311 {
312 if (player->GetQuestStatus(QUEST_SCOOP) == QUEST_STATUS_INCOMPLETE)
313 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_P, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
314
315 player->SEND_GOSSIP_MENU(738, creature->GetGUID());
316
317 return true;
318 }
319
320 CreatureAI* GetAI(Creature* creature) const override
321 {
322 return new npc_pluckyAI(creature);
323 }
324
325 struct npc_pluckyAI : public ScriptedAI
326 {
327 npc_pluckyAI(Creature* creature) : ScriptedAI(creature)
328 {
329 Initialize();
330 NormFaction = creature->getFaction();
331 }
332
333 void Initialize()
334 {
335 ResetTimer = 120000;
336 }
337
338 uint32 NormFaction;
339 uint32 ResetTimer;
340
341 void Reset() override
342 {
343 Initialize();
344
345 if (me->getFaction() != NormFaction)
346 me->setFaction(NormFaction);
347
348 if (me->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP))
349 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
350
351 DoCast(me, SPELL_PLUCKY_CHICKEN, false);
352 }
353
354 void ReceiveEmote(Player* player, uint32 TextEmote) override
355 {
356 if (player->GetQuestStatus(QUEST_SCOOP) == QUEST_STATUS_INCOMPLETE)
357 {
358 if (TextEmote == TEXT_EMOTE_BECKON)
359 {
360 me->setFaction(FACTION_FRIENDLY);
361 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
362 DoCast(me, SPELL_PLUCKY_HUMAN, false);
363 }
364 }
365
366 if (TextEmote == TEXT_EMOTE_CHICKEN)
367 {
368 if (me->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP))
369 return;
370 else
371 {
372 me->setFaction(FACTION_FRIENDLY);
373 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
374 DoCast(me, SPELL_PLUCKY_HUMAN, false);
375 me->HandleEmoteCommand(EMOTE_ONESHOT_WAVE);
376 }
377 }
378 }
379
380 void UpdateAI(uint32 Diff) override
381 {
382 if (me->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP))
383 {
384 if (ResetTimer <= Diff)
385 {
386 if (!me->GetVictim())
387 EnterEvadeMode();
388 else
389 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
390
391 return;
392 }
393 else
394 ResetTimer -= Diff;
395 }
396
397 if (!UpdateVictim())
398 return;
399
400 DoMeleeAttackIfReady();
401 }
402 };

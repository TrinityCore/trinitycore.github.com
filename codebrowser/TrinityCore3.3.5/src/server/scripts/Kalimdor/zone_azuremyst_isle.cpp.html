62 struct npc_draenei_survivorAI : public ScriptedAI
63 {
64 npc_draenei_survivorAI(Creature* creature) : ScriptedAI(creature)
65 {
66 Initialize();
67 }
68
69 void Initialize()
70 {
71 pCaster.Clear();
72
73 SayThanksTimer = 0;
74 RunAwayTimer = 0;
75 SayHelpTimer = 10000;
76
77 CanSayHelp = true;
78 }
79
80 ObjectGuid pCaster;
81
82 uint32 SayThanksTimer;
83 uint32 RunAwayTimer;
84 uint32 SayHelpTimer;
85
86 bool CanSayHelp;
87
88 void Reset() override
89 {
90 Initialize();
91
92 DoCast(me, SPELL_IRRIDATION, true);
93
94 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PVP_ATTACKABLE);
95 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IN_COMBAT);
96 me->SetHealth(me->CountPctFromMaxHealth(10));
97 me->SetStandState(UNIT_STAND_STATE_SLEEP);
98 }
99
100 void EnterCombat(Unit* /*who*/) override { }
101
102 void MoveInLineOfSight(Unit* who) override
103 {
104 if (CanSayHelp && who->GetTypeId() == TYPEID_PLAYER && me->IsFriendlyTo(who) && me->IsWithinDistInMap(who, 25.0f))
105 {
106 //Random switch between 4 texts
107 Talk(SAY_HELP, who);
108
109 SayHelpTimer = 20000;
110 CanSayHelp = false;
111 }
112 }
113
114 void SpellHit(Unit* Caster, const SpellInfo* Spell) override
115 {
116 if (Spell->SpellFamilyFlags[2] & 0x080000000)
117 {
118 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PVP_ATTACKABLE);
119 me->SetStandState(UNIT_STAND_STATE_STAND);
120
121 DoCast(me, SPELL_STUNNED, true);
122
123 pCaster = Caster->GetGUID();
124
125 SayThanksTimer = 5000;
126 }
127 }
128
129 void UpdateAI(uint32 diff) override
130 {
131 if (SayThanksTimer)
132 {
133 if (SayThanksTimer <= diff)
134 {
135 me->RemoveAurasDueToSpell(SPELL_IRRIDATION);
136
137 if (Player* player = ObjectAccessor::GetPlayer(*me, pCaster))
138 {
139 Talk(SAY_HEAL, player);
140
141 player->TalkedToCreature(me->GetEntry(), me->GetGUID());
142 }
143
144 me->GetMotionMaster()->Clear();
145 me->GetMotionMaster()->MovePoint(0, -4115.053711f, -13754.831055f, 73.508949f);
146
147 RunAwayTimer = 10000;
148 SayThanksTimer = 0;
149 } else SayThanksTimer -= diff;
150
151 return;
152 }
153
154 if (RunAwayTimer)
155 {
156 if (RunAwayTimer <= diff)
157 me->DespawnOrUnsummon();
158 else
159 RunAwayTimer -= diff;
160
161 return;
162 }
163
164 if (SayHelpTimer <= diff)
165 {
166 CanSayHelp = true;
167 SayHelpTimer = 20000;
168 } else SayHelpTimer -= diff;
169 }
170 };
171
172 CreatureAI* GetAI(Creature* creature) const override
173 {
174 return new npc_draenei_survivorAI(creature);
175 }

425 struct npc_geezleAI : public ScriptedAI
426 {
427 npc_geezleAI(Creature* creature) : ScriptedAI(creature)
428 {
429 Initialize();
430 }
431
432 void Initialize()
433 {
434 SparkGUID.Clear();
435 Step = 0;
436 EventStarted = false;
437 SayTimer = 0;
438 }
439
440 ObjectGuid SparkGUID;
441
442 uint8 Step;
443 uint32 SayTimer;
444
445 bool EventStarted;
446
447 void Reset() override
448 {
449 Initialize();
450 StartEvent();
451 }
452
453 void EnterCombat(Unit* /*who*/) override { }
454
455 void StartEvent()
456 {
457 Step = 0;
458 EventStarted = true;
459 if (Creature* Spark = me->SummonCreature(NPC_SPARK, SparkPos, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 1000))
460 {
461 SparkGUID = Spark->GetGUID();
462 Spark->setActive(true);
463 Spark->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
464 }
465 SayTimer = 8000;
466 }
467
468 uint32 NextStep(uint8 step)
469 {
470 Creature* Spark = ObjectAccessor::GetCreature(*me, SparkGUID);
471 if (!Spark)
472 return 99999999;
473
474 switch (step)
475 {
476 case 0:
477 Spark->GetMotionMaster()->MovePoint(0, -5080.70f, -11253.61f, 0.56f);
478 me->GetMotionMaster()->MovePoint(0, -5092.26f, -11252, 0.71f);
479 return 9000;
480 case 1:
481 DespawnNagaFlag(true);
482 Spark->AI()->Talk(EMOTE_SPARK);
483 return 1000;
484 case 2:
485 Talk(GEEZLE_SAY_1, Spark);
486 Spark->SetInFront(me);
487 me->SetInFront(Spark);
488 return 5000;
489 case 3:
490 Spark->AI()->Talk(SPARK_SAY_2);
491 return 7000;
492 case 4:
493 Spark->AI()->Talk(SPARK_SAY_3);
494 return 8000;
495 case 5:
496 Talk(GEEZLE_SAY_4, Spark);
497 return 8000;
498 case 6:
499 Spark->AI()->Talk(SPARK_SAY_5);
500 return 9000;
501 case 7:
502 Spark->AI()->Talk(SPARK_SAY_6);
503 return 8000;
504 case 8:
505 Talk(GEEZLE_SAY_7, Spark);
506 return 2000;
507 case 9:
508 me->GetMotionMaster()->MoveTargetedHome();
509 Spark->GetMotionMaster()->MovePoint(0, SparkPos);
510 CompleteQuest();
511 return 9000;
512 case 10:
513 Spark->DisappearAndDie();
514 DespawnNagaFlag(false);
515 me->DisappearAndDie();
516 default: return 99999999;
517 }
518 }

520 // will complete Tree's company quest for all nearby players that are disguised as trees
521 void CompleteQuest()
522 {
523 float radius = 50.0f;
524 std::list<Player*> players;
525 Trinity::AnyPlayerInObjectRangeCheck checker(me, radius);
526 Trinity::PlayerListSearcher<Trinity::AnyPlayerInObjectRangeCheck> searcher(me, players, checker);
527 me->VisitNearbyWorldObject(radius, searcher);
528
529 for (std::list<Player*>::const_iterator itr = players.begin(); itr != players.end(); ++itr)
530 if ((*itr)->GetQuestStatus(QUEST_TREES_COMPANY) == QUEST_STATUS_INCOMPLETE && (*itr)->HasAura(SPELL_TREE_DISGUISE))
531 (*itr)->KilledMonsterCredit(NPC_SPARK);
532 }
533
534 void DespawnNagaFlag(bool despawn)
535 {
536 std::list<GameObject*> FlagList;
537 me->GetGameObjectListWithEntryInGrid(FlagList, GO_NAGA_FLAG, 100.0f);
538
539 if (!FlagList.empty())
540 {
541 for (std::list<GameObject*>::const_iterator itr = FlagList.begin(); itr != FlagList.end(); ++itr)
542 {
543 if (despawn)
544 (*itr)->SetLootState(GO_JUST_DEACTIVATED);
545 else
546 (*itr)->Respawn();
547 }
548 }
549 else
550 TC_LOG_ERROR("scripts", "SD2 ERROR: FlagList is empty!");
551 }
552
553 void UpdateAI(uint32 diff) override
554 {
555 if (SayTimer <= diff)
556 {
557 if (EventStarted)
558 SayTimer = NextStep(Step++);
559 }
560 else
561 SayTimer -= diff;
562 }
563 };
564
565 CreatureAI* GetAI(Creature* creature) const override
566 {
567 return new npc_geezleAI(creature);
568 }

717 {
718 if (type != POINT_MOTION_TYPE || id != POINT_INIT)
719 return;
720
721 if (Player* _player = ObjectAccessor::GetPlayer(*me, _playerGUID))
722 _player->KilledMonsterCredit(me->GetEntry(), me->GetGUID());
723
724 _movementComplete = true;
725 _events.ScheduleEvent(EVENT_DESPAWN, 3500);
726 }
727
728 void UpdateAI(uint32 diff) override
729 {
730 if (!_movementComplete)
731 return;
732
733 _events.Update(diff);
734
735 if (_events.ExecuteEvent() == EVENT_DESPAWN)
736 me->DespawnOrUnsummon();
737 }
738
739 private:

741 EventMap _events;
742 bool _movementComplete;
743 };

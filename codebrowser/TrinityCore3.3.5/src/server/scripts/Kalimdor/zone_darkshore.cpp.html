60/// @todo make concept similar as "ringo" -escort. Find a way to run the scripted attacks, _if_ player are choosing road.
61class npc_kerlonian : public CreatureScript
62{
63public:
64 npc_kerlonian() : CreatureScript("npc_kerlonian") { }
65
66 struct npc_kerlonianAI : public FollowerAI
67 {
68 npc_kerlonianAI(Creature* creature) : FollowerAI(creature)
69 {
70 Initialize();
71 }
72
73 void Initialize()
74 {
75 FallAsleepTimer = urand(10000, 45000);
76 }
77
78 uint32 FallAsleepTimer;
79
80 void Reset() override
81 {
82 Initialize();
83 }
84
85 void MoveInLineOfSight(Unit* who) override
86
87 {
88 FollowerAI::MoveInLineOfSight(who);
89
90 if (!me->GetVictim() && !HasFollowState(STATE_FOLLOW_COMPLETE) && who->GetEntry() == NPC_LILADRIS)
91 {
92 if (me->IsWithinDistInMap(who, INTERACTION_DISTANCE*5))
93 {
94 if (Player* player = GetLeaderForFollower())
95 {
96 if (player->GetQuestStatus(QUEST_SLEEPER_AWAKENED) == QUEST_STATUS_INCOMPLETE)
97 player->GroupEventHappens(QUEST_SLEEPER_AWAKENED, me);
98
99 Talk(SAY_KER_END);
100 }
101
102 SetFollowComplete();
103 }
104 }
105 }
106
107 void SpellHit(Unit* /*pCaster*/, const SpellInfo* pSpell) override
108 {
109 if (HasFollowState(STATE_FOLLOW_INPROGRESS | STATE_FOLLOW_PAUSED) && pSpell->Id == SPELL_AWAKEN)
110 ClearSleeping();
111 }
112
113 void SetSleeping()
114 {
115 SetFollowPaused(true);
116
117 Talk(EMOTE_KER_SLEEP);
118
119 Talk(SAY_KER_SLEEP);
120
121 me->SetStandState(UNIT_STAND_STATE_SLEEP);
122 DoCast(me, SPELL_SLEEP_VISUAL, false);
123 }
124
125 void ClearSleeping()
126 {
127 me->RemoveAurasDueToSpell(SPELL_SLEEP_VISUAL);
128 me->SetStandState(UNIT_STAND_STATE_STAND);
129
130 Talk(EMOTE_KER_AWAKEN);
131
132 SetFollowPaused(false);
133 }
134
135 void UpdateFollowerAI(uint32 diff) override
136 {
137 if (!UpdateVictim())
138 {
139 if (!HasFollowState(STATE_FOLLOW_INPROGRESS))
140 return;
141
142 if (!HasFollowState(STATE_FOLLOW_PAUSED))
143 {
144 if (FallAsleepTimer <= diff)
145 {
146 SetSleeping();
147 FallAsleepTimer = urand(25000, 90000);
148 }
149 else
150 FallAsleepTimer -= diff;
151 }
152
153 return;
154 }
155
156 DoMeleeAttackIfReady();
157 }
158 };
159
160 bool OnQuestAccept(Player* player, Creature* creature, const Quest* quest) override
161 {
162 if (quest->GetQuestId() == QUEST_SLEEPER_AWAKENED)
163 {
164 if (npc_kerlonianAI* pKerlonianAI = CAST_AI(npc_kerlonian::npc_kerlonianAI, creature->AI()))
165 {
166 creature->SetStandState(UNIT_STAND_STATE_STAND);
167 creature->AI()->Talk(SAY_KER_START, player);
168 pKerlonianAI->StartFollow(player, FACTION_KER_ESCORTEE, quest);
169 }
170 }
171
172 return true;
173 }
174
175 CreatureAI* GetAI(Creature* creature) const override
176 {
177 return new npc_kerlonianAI(creature);
178 }
179};

334 npc_threshwackonatorAI(Creature* creature) : FollowerAI(creature) { }

336 void Reset() override { }
337
338 void MoveInLineOfSight(Unit* who) override
339
340 {
341 FollowerAI::MoveInLineOfSight(who);
342
343 if (!me->GetVictim() && !HasFollowState(STATE_FOLLOW_COMPLETE) && who->GetEntry() == NPC_GELKAK)
344 {
345 if (me->IsWithinDistInMap(who, 10.0f))
346 {
347 Talk(SAY_AT_CLOSE, who);
348 DoAtEnd();
349 }
350 }
351 }
352
353 void DoAtEnd()
354 {
355 me->setFaction(FACTION_HOSTILE);
356
357 if (Player* pHolder = GetLeaderForFollower())
358 AttackStart(pHolder);
359
360 SetFollowComplete(true);
361 }
362 };

364 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
365 {
366 player->PlayerTalkClass->ClearMenus();
367 if (action == GOSSIP_ACTION_INFO_DEF+1)
368 {
369 player->CLOSE_GOSSIP_MENU();
370
371 if (npc_threshwackonatorAI* pThreshAI = CAST_AI(npc_threshwackonator::npc_threshwackonatorAI, creature->AI()))
372 {
373 creature->AI()->Talk(EMOTE_START);
374 pThreshAI->StartFollow(player);
375 }
376 }
377
378 return true;
379 }

382 {

386 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());

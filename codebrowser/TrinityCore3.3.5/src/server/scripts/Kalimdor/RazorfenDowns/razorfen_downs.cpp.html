83class npc_belnistrasz : public CreatureScript
84{
85public:
86 npc_belnistrasz() : CreatureScript("npc_belnistrasz") { }
87
88 struct npc_belnistraszAI : public ScriptedAI
89 {
90 npc_belnistraszAI(Creature* creature) : ScriptedAI(creature)
91 {
92 instance = creature->GetInstanceScript();
93 eventInProgress = false;
94 channeling = false;
95 eventProgress = 0;
96 spawnerCount = 0;
97 }
98
99 void Reset() override
100 {
101 if (!eventInProgress)
102 {
103 if (!me->HasAura(SPELL_ARCANE_INTELLECT))
104 DoCast(me, SPELL_ARCANE_INTELLECT);
105
106 channeling = false;
107 eventProgress = 0;
108 spawnerCount = 0;
109 me->SetFlag(UNIT_NPC_FLAGS, GOSSIP_OPTION_QUESTGIVER);
110 }
111 }
112
113 void EnterCombat(Unit* who) override
114 {
115 if (channeling)
116 Talk(SAY_WATCH_OUT, who);
117 else
118 {
119 events.ScheduleEvent(EVENT_FIREBALL, 1000);
120 events.ScheduleEvent(EVENT_FROST_NOVA, urand(8000, 12000));
121 if (urand(0, 100) > 40)
122 Talk(SAY_AGGRO, who);
123 }
124 }
125
126 void JustDied(Unit* /*killer*/) override
127 {
128 instance->SetBossState(DATA_EXTINGUISHING_THE_IDOL, DONE);
129 me->DespawnOrUnsummon(5000);
130 }
131
132 void sQuestAccept(Player* /*player*/, Quest const* quest) override
133 {
134 if (quest->GetQuestId() == QUEST_EXTINGUISHING_THE_IDOL)
135 {
136 eventInProgress = true;
137 Talk(SAY_QUEST_ACCEPTED);
138 me->RemoveFlag(UNIT_NPC_FLAGS, GOSSIP_OPTION_QUESTGIVER);
139 me->setFaction(FACTION_ESCORT);
140 me->GetMotionMaster()->MovePath(PATH_ESCORT, false);
141 }
142 }
143
144 void MovementInform(uint32 type, uint32 id) override
145 {
146 if (type == WAYPOINT_MOTION_TYPE && id == POINT_REACH_IDOL)
147 {
148 channeling = true;
149 events.ScheduleEvent(EVENT_CHANNEL, 2000);
150 }
151 }
152
153 void UpdateAI(uint32 diff) override
154 {
155 if (!eventInProgress)
156 return;
157
158 events.Update(diff);
159
160 while (uint32 eventId = events.ExecuteEvent())
161 {
162 switch (eventId)
163 {
164 case EVENT_CHANNEL:
165 Talk(SAY_EVENT_START);
166 DoCast(me, SPELL_IDOL_SHUTDOWN_VISUAL);
167 events.ScheduleEvent(EVENT_IDOL_ROOM_SPAWNER, 100);
168 events.ScheduleEvent(EVENT_PROGRESS, 120000);
169 break;
170 case EVENT_IDOL_ROOM_SPAWNER:
171 if (Creature* creature = me->SummonCreature(NPC_IDOL_ROOM_SPAWNER, PosSummonSpawner[urand(0,2)], TEMPSUMMON_TIMED_DESPAWN, 4000))
172 creature->AI()->SetData(0,spawnerCount);
173 if (++spawnerCount < 8)
174 events.ScheduleEvent(EVENT_IDOL_ROOM_SPAWNER, 35000);
175 break;
176 case EVENT_PROGRESS:
177 {
178 switch (eventProgress)
179 {
180 case 0:
181 Talk(SAY_EVENT_THREE_MIN_LEFT);
182 ++eventProgress;
183 events.ScheduleEvent(EVENT_PROGRESS, 60000);
184 break;
185 case 1:
186 Talk(SAY_EVENT_TWO_MIN_LEFT);
187 ++eventProgress;
188 events.ScheduleEvent(EVENT_PROGRESS, 60000);
189 break;
190 case 2:
191 Talk(SAY_EVENT_ONE_MIN_LEFT);
192 ++eventProgress;
193 events.ScheduleEvent(EVENT_PROGRESS, 60000);
194 break;
195 case 3:
196 events.CancelEvent(EVENT_IDOL_ROOM_SPAWNER);
197 me->InterruptSpell(CURRENT_CHANNELED_SPELL);
198 Talk(SAY_EVENT_END);
199 events.ScheduleEvent(EVENT_COMPLETE, 3000);
200 break;
201 }
202 break;
203 }
204 case EVENT_COMPLETE:
205 {
206 DoCast(me, SPELL_IDOM_ROOM_CAMERA_SHAKE);
207 me->SummonGameObject(GO_BELNISTRASZS_BRAZIER, 2577.196f, 947.0781f, 53.16757f, 2.356195f, 0, 0, 0.9238796f, 0.3826832f, 3600);
208 std::list<WorldObject*> ClusterList;
209 Trinity::AllWorldObjectsInRange objects(me, 50.0f);
210 Trinity::WorldObjectListSearcher<Trinity::AllWorldObjectsInRange> searcher(me, ClusterList, objects);
211 me->VisitNearbyObject(50.0f, searcher);
212 for (std::list<WorldObject*>::const_iterator itr = ClusterList.begin(); itr != ClusterList.end(); ++itr)
213 {
214 if (Player* player = (*itr)->ToPlayer())
215 {
216 if (player->GetQuestStatus(QUEST_EXTINGUISHING_THE_IDOL) == QUEST_STATUS_INCOMPLETE)
217 player->CompleteQuest(QUEST_EXTINGUISHING_THE_IDOL);
218 }
219 else if (GameObject* go = (*itr)->ToGameObject())
220 {
221 if (go->GetEntry() == GO_IDOL_OVEN_FIRE || go->GetEntry() == GO_IDOL_CUP_FIRE || go->GetEntry() == GO_IDOL_MOUTH_FIRE)
222 go->Delete();
223 }
224 }
225 instance->SetBossState(DATA_EXTINGUISHING_THE_IDOL, DONE);
226 me->DespawnOrUnsummon();
227 break;
228 }
229 case EVENT_FIREBALL:
230 if (me->HasUnitState(UNIT_STATE_CASTING) || !UpdateVictim())
231 return;
232 DoCastVictim(SPELL_FIREBALL);
233 events.ScheduleEvent(EVENT_FIREBALL, 8000);
234 break;
235 case EVENT_FROST_NOVA:
236 if (me->HasUnitState(UNIT_STATE_CASTING) || !UpdateVictim())
237 return;
238 DoCast(me, SPELL_FROST_NOVA);
239 events.ScheduleEvent(EVENT_FROST_NOVA, 15000);
240 break;
241 }
242 }
243 if (!channeling)
244 DoMeleeAttackIfReady();
245 }
246
247 private:
248 InstanceScript* instance;
249 EventMap events;
250 bool eventInProgress;
251 bool channeling;
252 uint8 eventProgress;
253 uint8 spawnerCount;
254 };
255
256 CreatureAI* GetAI(Creature* creature) const override
257 {
258 return GetInstanceAI<npc_belnistraszAI>(creature);
259 }

267 struct npc_idol_room_spawnerAI : public ScriptedAI
268 {
269 npc_idol_room_spawnerAI(Creature* creature) : ScriptedAI(creature)
270 {
271 instance = creature->GetInstanceScript();
272 }
273
274 void Reset() override { }
275
276 void SetData(uint32 /*type*/, uint32 data) override
277 {
278 if (data < 7)
279 {
280 me->SummonCreature(NPC_WITHERED_BATTLE_BOAR, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
281 if (data > 0 && me->GetOrientation() < 4.0f)
282 me->SummonCreature(NPC_WITHERED_BATTLE_BOAR, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
283 me->SummonCreature(NPC_DEATHS_HEAD_GEOMANCER, me->GetPositionX() + (std::cos(me->GetOrientation() - (float(M_PI) / 2)) * 2), me->GetPositionY() + (std::sin(me->GetOrientation() - (float(M_PI) / 2)) * 2), me->GetPositionZ(), me->GetOrientation());
284 me->SummonCreature(NPC_WITHERED_QUILGUARD, me->GetPositionX() + (std::cos(me->GetOrientation() + (float(M_PI) / 2)) * 2), me->GetPositionY() + (std::sin(me->GetOrientation() + (float(M_PI) / 2)) * 2), me->GetPositionZ(), me->GetOrientation());
285 }
286 else if (data == 7)
287 me->SummonCreature(NPC_PLAGUEMAW_THE_ROTTING, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
288 }
289
290 private:
291 InstanceScript* instance;
292 };
293
294 CreatureAI* GetAI(Creature* creature) const override

296 return GetInstanceAI<npc_idol_room_spawnerAI>(creature);

308class npc_tomb_creature : public CreatureScript
309{
310public:
311 npc_tomb_creature() : CreatureScript("npc_tomb_creature") { }
312
313 struct npc_tomb_creatureAI : public ScriptedAI
314 {
315 npc_tomb_creatureAI(Creature* creature) : ScriptedAI(creature)
316 {
317 instance = creature->GetInstanceScript();
318 }
319
320 void Reset() override
321 {
322 if (!me->HasAura(SPELL_POISON_PROC) && me->GetEntry() == NPC_TOMB_FIEND)
323 DoCast(me, SPELL_POISON_PROC);
324
325 if (!me->HasAura(SPELL_VIRULENT_POISON_PROC) && me->GetEntry() == NPC_TOMB_REAVER)
326 DoCast(me, SPELL_VIRULENT_POISON_PROC);
327 }
328
329 void JustDied(Unit* /*killer*/) override
330 {
331 instance->SetData(DATA_WAVE, me->GetEntry());
332 }
333
334 void EnterCombat(Unit* /*who*/) override
335 {
336 events.ScheduleEvent(EVENT_WEB, urand(5000, 8000));
337 }
338
339 void UpdateAI(uint32 diff) override
340 {
341 if (!UpdateVictim())
342 return;
343
344 events.Update(diff);
345
346 while (uint32 eventId = events.ExecuteEvent())
347 {
348 switch (eventId)
349 {
350 case EVENT_WEB:
351 DoCastVictim(SPELL_WEB);
352 events.ScheduleEvent(EVENT_WEB, urand(7000, 16000));
353 break;
354 }
355 }
356 DoMeleeAttackIfReady();
357 }
358
359 private:
360 InstanceScript* instance;
361 EventMap events;
362 };
363
364 CreatureAI* GetAI(Creature* creature) const override
365 {
366 return GetInstanceAI<npc_tomb_creatureAI>(creature);
367 }

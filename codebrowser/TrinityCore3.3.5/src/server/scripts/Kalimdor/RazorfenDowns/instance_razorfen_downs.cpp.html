47class instance_razorfen_downs : public InstanceMapScript
48{
49public:
50 instance_razorfen_downs() : InstanceMapScript(RFDScriptName, 129) { }
51
52 struct instance_razorfen_downs_InstanceMapScript : public InstanceScript
53 {
54 instance_razorfen_downs_InstanceMapScript(Map* map) : InstanceScript(map)
55 {
56 SetHeaders(DataHeader);
57 SetBossNumber(EncounterCount);
58 gongWave = 0;
59 fiendsKilled = 0;
60 reaversKilled = 0;
61 summonLowRange = 0;
62 summonHighRange = 0;
63 summonCreature = 0;
64 }
65
66 void OnGameObjectCreate(GameObject* gameObject) override
67 {
68 switch (gameObject->GetEntry())
69 {
70 case GO_GONG:
71 goGongGUID = gameObject->GetGUID();
72 if (GetBossState(DATA_TUTEN_KASH) == DONE)
73 gameObject->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
74 break;
75 case GO_IDOL_OVEN_FIRE:
76 case GO_IDOL_CUP_FIRE:
77 case GO_IDOL_MOUTH_FIRE:
78 if (GetBossState(DATA_EXTINGUISHING_THE_IDOL) == DONE)
79 gameObject->Delete();
80 break;
81 default:
82 break;
83 }
84 }
85
86 bool SetBossState(uint32 type, EncounterState state) override
87 {
88 if (!InstanceScript::SetBossState(type, state))
89 return false;
90
91 switch (type)
92 {
93 case DATA_TUTEN_KASH:
94 case DATA_MORDRESH_FIRE_EYE:
95 case DATA_GLUTTON:
96 case DATA_AMNENNAR_THE_COLD_BRINGER:
97 case DATA_GONG:
98 case DATA_WAVE:
99 case DATA_EXTINGUISHING_THE_IDOL:
100 break;
101 default:
102 break;
103 }
104 return true;
105 }
106
107 void SetData(uint32 type, uint32 data) override
108 {
109 if (type == DATA_WAVE)
110 {
111 switch (data)
112 {
113 case IN_PROGRESS:
114 {
115 if (GameObject* go = instance->GetGameObject(goGongGUID))
116 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
117
118 switch (gongWave)
119 {
120 case 0:
121 summonLowRange = 0;
122 summonHighRange = 10;
123 summonCreature = NPC_TOMB_FIEND;
124 break;
125 case 1:
126 summonLowRange = 10;
127 summonHighRange = 14;
128 summonCreature = NPC_TOMB_REAVER;
129 break;
130 case 2:
131 summonLowRange = 14;
132 summonHighRange = 15;
133 summonCreature = NPC_TUTEN_KASH;
134 break;
135 }
136
137 if (GameObject* go = instance->GetGameObject(goGongGUID))
138 {
139 for (uint8 i = summonLowRange; i < summonHighRange; ++i)
140 {
141 Creature* creature = go->SummonCreature(summonCreature, PosSummonTutenkash[i]);
142 creature->GetMotionMaster()->MovePoint(0, 2533.479f + float(irand(-5, 5)), 870.020f + float(irand(-5, 5)), 47.678f);
143 }
144 }
145
146 ++gongWave;
147 break;
148 }
149 case NPC_TOMB_FIEND:
150 if (++fiendsKilled == 10)
151 {
152 fiendsKilled = 0;
153 if (GameObject* go = instance->GetGameObject(goGongGUID))
154 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
155 }
156 break;
157 case NPC_TOMB_REAVER:
158 if (++reaversKilled == 4)
159 {
160 reaversKilled = 0;
161 if (GameObject* go = instance->GetGameObject(goGongGUID))
162 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
163 }
164 break;
165 }
166 }
167 }
168
169 protected:
170 ObjectGuid goGongGUID;
171 uint16 gongWave;
172 uint8 fiendsKilled;
173 uint8 reaversKilled;
174 uint8 summonLowRange;
175 uint8 summonHighRange;
176 uint32 summonCreature;
177 };
178
179 InstanceScript* GetInstanceScript(InstanceMap* map) const override
180 {
181 return new instance_razorfen_downs_InstanceMapScript(map);
182 }
183};

187 new instance_razorfen_downs();

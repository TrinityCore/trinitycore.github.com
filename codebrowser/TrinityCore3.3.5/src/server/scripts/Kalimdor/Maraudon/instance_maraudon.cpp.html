18/*
19This placeholder for the instance is needed for dungeon finding to be able
20to give credit after the boss defined in lastEncounterDungeon is killed.
21Without it, the party doing random dungeon won't get satchel of spoils and
22gets instead the deserter debuff.
23*/

25#include "ScriptMgr.h"
26#include "InstanceScript.h"

28class instance_maraudon : public InstanceMapScript
29{
30public:

32
33 InstanceScript* GetInstanceScript(InstanceMap* map) const override
34 {

36 }
37
38 struct instance_maraudon_InstanceMapScript : public InstanceScript
39 {
40 instance_maraudon_InstanceMapScript(Map* map) : InstanceScript(map) { }
41 };
42};
43
44void AddSC_instance_maraudon()
45{
46 new instance_maraudon();
47}

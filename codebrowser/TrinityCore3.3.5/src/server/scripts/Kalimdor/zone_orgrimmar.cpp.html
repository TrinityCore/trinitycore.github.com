40enum Shenthul

45class npc_shenthul : public CreatureScript
46{
47public:
48 npc_shenthul() : CreatureScript("npc_shenthul") { }
49
50 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
51 {
52 if (quest->GetQuestId() == QUEST_SHATTERED_SALUTE)
53 {
54 ENSURE_AI(npc_shenthul::npc_shenthulAI, creature->AI())->CanTalk = true;
55 ENSURE_AI(npc_shenthul::npc_shenthulAI, creature->AI())->PlayerGUID = player->GetGUID();
56 }
57 return true;
58 }
59
60 CreatureAI* GetAI(Creature* creature) const override
61 {
62 return new npc_shenthulAI(creature);
63 }
64
65 struct npc_shenthulAI : public ScriptedAI
66 {
67 npc_shenthulAI(Creature* creature) : ScriptedAI(creature)
68 {
69 Initialize();
70 }
71
72 void Initialize()
73 {
74 CanTalk = false;
75 CanEmote = false;
76 SaluteTimer = 6000;
77 ResetTimer = 0;
78 PlayerGUID.Clear();
79 }
80
81 bool CanTalk;
82 bool CanEmote;
83 uint32 SaluteTimer;
84 uint32 ResetTimer;
85 ObjectGuid PlayerGUID;
86
87 void Reset() override
88 {
89 Initialize();
90 }
91
92 void EnterCombat(Unit* /*who*/) override { }
93
94 void UpdateAI(uint32 diff) override
95 {
96 if (CanEmote)
97 {
98 if (ResetTimer <= diff)
99 {
100 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
101 {
102 if (player->GetTypeId() == TYPEID_PLAYER && player->GetQuestStatus(QUEST_SHATTERED_SALUTE) == QUEST_STATUS_INCOMPLETE)
103 player->FailQuest(QUEST_SHATTERED_SALUTE);
104 }
105 Reset();
106 } else ResetTimer -= diff;
107 }
108
109 if (CanTalk && !CanEmote)
110 {
111 if (SaluteTimer <= diff)
112 {
113 me->HandleEmoteCommand(EMOTE_ONESHOT_SALUTE);
114 CanEmote = true;
115 ResetTimer = 60000;
116 } else SaluteTimer -= diff;
117 }
118
119 if (!UpdateVictim())
120 return;
121
122 DoMeleeAttackIfReady();
123 }
124
125 void ReceiveEmote(Player* player, uint32 emote) override
126 {
127 if (emote == TEXT_EMOTE_SALUTE && player->GetQuestStatus(QUEST_SHATTERED_SALUTE) == QUEST_STATUS_INCOMPLETE)
128 {
129 if (CanEmote)
130 {
131 player->AreaExploredOrEventHappens(QUEST_SHATTERED_SALUTE);
132 Reset();
133 }
134 }
135 }
136 };
137
138};

171class npc_thrall_warchief : public CreatureScript
172{
173public:
174 npc_thrall_warchief() : CreatureScript("npc_thrall_warchief") { }
175
176 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
177 {
178 player->PlayerTalkClass->ClearMenus();
179 switch (action)
180 {
181 case GOSSIP_ACTION_INFO_DEF+1:
182 player->ADD_GOSSIP_ITEM_DB(OPTION_WHAT_DISCOVERIES, GOSSIP_MENU_OPTION_ID_ALL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+2);
183 player->SEND_GOSSIP_MENU(GOSSIP_THE_SHATTERED_HAND, creature->GetGUID());
184 break;
185 case GOSSIP_ACTION_INFO_DEF+2:
186 player->ADD_GOSSIP_ITEM_DB(OPTION_USURPER, GOSSIP_MENU_OPTION_ID_ALL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+3);
187 player->SEND_GOSSIP_MENU(GOSSIP_IT_WOULD_APPEAR_AS, creature->GetGUID());
188 break;
189 case GOSSIP_ACTION_INFO_DEF+3:
190 player->ADD_GOSSIP_ITEM_DB(OPTION_WITH_ALL_DUE_RESPECT, GOSSIP_MENU_OPTION_ID_ALL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+4);
191 player->SEND_GOSSIP_MENU(GOSSIP_THE_BROOD_MOTHER, creature->GetGUID());
192 break;
193 case GOSSIP_ACTION_INFO_DEF+4:
194 player->ADD_GOSSIP_ITEM_DB(OPTION_I_I_DID_NOT_THINK_OF, GOSSIP_MENU_OPTION_ID_ALL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+5);
195 player->SEND_GOSSIP_MENU(GOSSIP_SO_MUCH_TO_LEARN, creature->GetGUID());
196 break;
197 case GOSSIP_ACTION_INFO_DEF+5:
198 player->ADD_GOSSIP_ITEM_DB(OPTION_I_LIVE_ONLY_TO_SERVE, GOSSIP_MENU_OPTION_ID_ALL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+6);
199 player->SEND_GOSSIP_MENU(GOSSIP_I_DO_NOT_FAULT_YOU, creature->GetGUID());
200 break;
201 case GOSSIP_ACTION_INFO_DEF+6:
202 player->ADD_GOSSIP_ITEM_DB(OPTION_OF_COURSE_WARCHIEF, GOSSIP_MENU_OPTION_ID_ALL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+7);
203 player->SEND_GOSSIP_MENU(GOSSIP_NOW_PAY_ATTENTION, creature->GetGUID());
204 break;
205 case GOSSIP_ACTION_INFO_DEF+7:
206 player->CLOSE_GOSSIP_MENU();
207 player->AreaExploredOrEventHappens(QUEST_WHAT_THE_WIND_CARRIES);
208 break;
209 }
210 return true;
211 }

213 bool OnGossipHello(Player* player, Creature* creature) override
214 {
215 if (creature->IsQuestGiver())
216 player->PrepareQuestMenu(creature->GetGUID());
217
218 if (player->GetQuestStatus(QUEST_WHAT_THE_WIND_CARRIES) == QUEST_STATUS_INCOMPLETE)
219 player->ADD_GOSSIP_ITEM_DB(OPTION_PLEASE_SHARE_YOUR, GOSSIP_MENU_OPTION_ID_ALL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
220
221 player->SEND_GOSSIP_MENU(GOSSIP_MEMBERS_OF_THE_HORDE, creature->GetGUID());
222 return true;
223 }

253 void UpdateAI(uint32 diff) override
254 {
255 if (!UpdateVictim())
256 return;
257
258 if (ChainLightningTimer <= diff)
259 {
260 DoCastVictim(SPELL_CHAIN_LIGHTNING);
261 ChainLightningTimer = 9000;
262 } else ChainLightningTimer -= diff;
263
264 if (ShockTimer <= diff)
265 {
266 DoCastVictim(SPELL_SHOCK);
267 ShockTimer = 15000;
268 } else ShockTimer -= diff;
269
270 DoMeleeAttackIfReady();
271 }

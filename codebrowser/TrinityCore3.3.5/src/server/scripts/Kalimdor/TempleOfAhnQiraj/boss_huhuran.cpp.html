29enum Huhuran

35 SPELL_BERSERK = 26068,
36 SPELL_POISONBOLT = 26052,
37 SPELL_NOXIOUSPOISON = 26053,
38 SPELL_WYVERNSTING = 26180,
39 SPELL_ACIDSPIT = 26050
40};
41
42class boss_huhuran : public CreatureScript
43{
44public:
45 boss_huhuran() : CreatureScript("boss_huhuran") { }
46
47 CreatureAI* GetAI(Creature* creature) const override
48 {
49 return new boss_huhuranAI(creature);
50 }
51
52 struct boss_huhuranAI : public ScriptedAI
53 {
54 boss_huhuranAI(Creature* creature) : ScriptedAI(creature)
55 {
56 Initialize();
57 }
58
59 void Initialize()
60 {
61 Frenzy_Timer = urand(25000, 35000);
62 Wyvern_Timer = urand(18000, 28000);
63 Spit_Timer = 8000;
64 PoisonBolt_Timer = 4000;
65 NoxiousPoison_Timer = urand(10000, 20000);
66 FrenzyBack_Timer = 15000;
67
68 Frenzy = false;
69 Berserk = false;
70 }
71
72 uint32 Frenzy_Timer;
73 uint32 Wyvern_Timer;
74 uint32 Spit_Timer;
75 uint32 PoisonBolt_Timer;
76 uint32 NoxiousPoison_Timer;
77 uint32 FrenzyBack_Timer;
78
79 bool Frenzy;
80 bool Berserk;
81
82 void Reset() override
83 {
84 Initialize();
85 }
86
87 void EnterCombat(Unit* /*who*/) override
88 {
89 }
90
91 void UpdateAI(uint32 diff) override
92 {
93 //Return since we have no target
94 if (!UpdateVictim())
95 return;
96
97 //Frenzy_Timer
98 if (!Frenzy && Frenzy_Timer <= diff)
99 {
100 DoCast(me, SPELL_FRENZY);
101 Talk(EMOTE_FRENZY_KILL);
102 Frenzy = true;
103 PoisonBolt_Timer = 3000;
104 Frenzy_Timer = urand(25000, 35000);
105 } else Frenzy_Timer -= diff;
106
107 // Wyvern Timer
108 if (Wyvern_Timer <= diff)
109 {
110 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
111 DoCast(target, SPELL_WYVERNSTING);
112 Wyvern_Timer = urand(15000, 32000);
113 } else Wyvern_Timer -= diff;
114
115 //Spit Timer
116 if (Spit_Timer <= diff)
117 {
118 DoCastVictim(SPELL_ACIDSPIT);
119 Spit_Timer = urand(5000, 10000);
120 } else Spit_Timer -= diff;
121
122 //NoxiousPoison_Timer
123 if (NoxiousPoison_Timer <= diff)
124 {
125 DoCastVictim(SPELL_NOXIOUSPOISON);
126 NoxiousPoison_Timer = urand(12000, 24000);
127 } else NoxiousPoison_Timer -= diff;
128
129 //PoisonBolt only if frenzy or berserk
130 if (Frenzy || Berserk)
131 {
132 if (PoisonBolt_Timer <= diff)
133 {
134 DoCastVictim(SPELL_POISONBOLT);
135 PoisonBolt_Timer = 3000;
136 } else PoisonBolt_Timer -= diff;
137 }
138
139 //FrenzyBack_Timer
140 if (Frenzy && FrenzyBack_Timer <= diff)
141 {
142 me->InterruptNonMeleeSpells(false);
143 Frenzy = false;
144 FrenzyBack_Timer = 15000;
145 } else FrenzyBack_Timer -= diff;
146
147 if (!Berserk && HealthBelowPct(31))
148 {
149 me->InterruptNonMeleeSpells(false);
150 Talk(EMOTE_BERSERK);
151 DoCast(me, SPELL_BERSERK);
152 Berserk = true;
153 }
154
155 DoMeleeAttackIfReady();
156 }
157 };
158
159};

30enum Spells
31{
32 SPELL_SWEEP = 26103,
33 SPELL_SANDBLAST = 26102,
34 SPELL_GROUND_RUPTURE = 26100,
35 SPELL_BIRTH = 26262, // The Birth Animation
36 SPELL_DIRTMOUND_PASSIVE = 26092
37};
38
39class boss_ouro : public CreatureScript
40{
41public:
42 boss_ouro() : CreatureScript("boss_ouro") { }
43
44 CreatureAI* GetAI(Creature* creature) const override
45 {
46 return new boss_ouroAI(creature);
47 }
48
49 struct boss_ouroAI : public ScriptedAI
50 {
51 boss_ouroAI(Creature* creature) : ScriptedAI(creature)
52 {
53 Initialize();
54 }
55
56 void Initialize()
57 {
58 Sweep_Timer = urand(5000, 10000);
59 SandBlast_Timer = urand(20000, 35000);
60 Submerge_Timer = urand(90000, 150000);
61 Back_Timer = urand(30000, 45000);
62 ChangeTarget_Timer = urand(5000, 8000);
63 Spawn_Timer = urand(10000, 20000);
64
65 Enrage = false;
66 Submerged = false;
67 }
68
69 uint32 Sweep_Timer;
70 uint32 SandBlast_Timer;
71 uint32 Submerge_Timer;
72 uint32 Back_Timer;
73 uint32 ChangeTarget_Timer;
74 uint32 Spawn_Timer;
75
76 bool Enrage;
77 bool Submerged;
78
79 void Reset() override
80 {
81 Initialize();
82 }
83
84 void EnterCombat(Unit* /*who*/) override
85 {
86 DoCastVictim(SPELL_BIRTH);
87 }
88
89 void UpdateAI(uint32 diff) override
90 {
91 //Return since we have no target
92 if (!UpdateVictim())
93 return;
94
95 //Sweep_Timer
96 if (!Submerged && Sweep_Timer <= diff)
97 {
98 DoCastVictim(SPELL_SWEEP);
99 Sweep_Timer = urand(15000, 30000);
100 } else Sweep_Timer -= diff;
101
102 //SandBlast_Timer
103 if (!Submerged && SandBlast_Timer <= diff)
104 {
105 DoCastVictim(SPELL_SANDBLAST);
106 SandBlast_Timer = urand(20000, 35000);
107 } else SandBlast_Timer -= diff;
108
109 //Submerge_Timer
110 if (!Submerged && Submerge_Timer <= diff)
111 {
112 //Cast
113 me->HandleEmoteCommand(EMOTE_ONESHOT_SUBMERGE);
114 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
115 me->setFaction(35);
116 DoCast(me, SPELL_DIRTMOUND_PASSIVE);
117
118 Submerged = true;
119 Back_Timer = urand(30000, 45000);
120 } else Submerge_Timer -= diff;
121
122 //ChangeTarget_Timer
123 if (Submerged && ChangeTarget_Timer <= diff)
124 {
125 Unit* target = NULL;
126 target = SelectTarget(SELECT_TARGET_RANDOM, 0);
127
128 if (target)
129 DoTeleportTo(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ());
130
131 ChangeTarget_Timer = urand(10000, 20000);
132 } else ChangeTarget_Timer -= diff;
133
134 //Back_Timer
135 if (Submerged && Back_Timer <= diff)
136 {
137 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
138 me->setFaction(14);
139
140 DoCastVictim(SPELL_GROUND_RUPTURE);
141
142 Submerged = false;
143 Submerge_Timer = urand(60000, 120000);
144 } else Back_Timer -= diff;
145
146 DoMeleeAttackIfReady();
147 }
148 };
149
150};
151
152void AddSC_boss_ouro()

154 new boss_ouro();

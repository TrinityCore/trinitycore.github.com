31/*
32 * This is a 2 phases events. Here follows an explanation of the main events and transition between phases and sub-phases.
33 *
34 * The first phase is the EYE phase: the Eye of C'Thun is active and C'thun is not active.
35 * During this phase, the "Eye of C'Thun" alternates between 2 sub-phases:
36 * - PHASE_EYE_GREEN_BEAM:
37 * 50 sec phase during which the Eye mainly casts its Green Beam every 3 sec.
38 * - PHASE_EYE_RED_BEAM:
39 * 35 sec phase during which the Eye casts its red beam every sec.
40 * This EYE phase ends when the "Eye of C'Thun" is killed. Then starts the CTHUN phase.
41 *
42 * The second phase is the CTHUN phase. The Eye of C'Thun is not active and C'Thun is active.
43 * This phase starts with the transformation of the Eye into C'Thun (PHASE_CTHUN_TRANSITION).
44 * After the transformation, C'Thun alternates between 2 sub-phases:
45 * - PHASE_CTHUN_STOMACH:
46 * - C'Thun is almost insensible to all damage (99% damage reduction).
47 * - It spawns 2 tentacles in its stomach.
48 * - C'Thun swallows players.
49 * - This sub-phase ends when the 2 tentacles are killed. Swallowed players are regurgitate.
50 *
51 * - PHASE_CTHUN_WEAK:
52 * - weakened C'Thun takes normal damage.
53 * - This sub-phase ends after 45 secs.
54 *
55 * This CTHUN phase ends when C'Thun is killed
56 *
57 * Note:
58 * - the current phase is stored in the instance data to be easily shared between the eye and cthun.
59 */

235 void UpdateAI(uint32 diff) override
236 {
237 //Check if we have a target
238 if (!UpdateVictim())
239 return;
240
241 uint32 currentPhase = instance->GetData(DATA_CTHUN_PHASE);
242 if (currentPhase == PHASE_EYE_GREEN_BEAM || currentPhase == PHASE_EYE_RED_BEAM)
243 {
244 // EyeTentacleTimer
245 if (EyeTentacleTimer <= diff)
246 {
247 //Spawn the 8 Eye Tentacles in the corret spots
248 SpawnEyeTentacle(0, 20); //south
249 SpawnEyeTentacle(10, 10); //south west
250 SpawnEyeTentacle(20, 0); //west
251 SpawnEyeTentacle(10, -10); //north west
252
253 SpawnEyeTentacle(0, -20); //north
254 SpawnEyeTentacle(-10, -10); //north east
255 SpawnEyeTentacle(-20, 0); // east
256 SpawnEyeTentacle(-10, 10); // south east
257
258 EyeTentacleTimer = 45000;
259 } else EyeTentacleTimer -= diff;
260 }
261
262 switch (currentPhase)
263 {
264 case PHASE_EYE_GREEN_BEAM:
265 //BeamTimer
266 if (BeamTimer <= diff)
267 {
268 //SPELL_GREEN_BEAM
269 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
270 {
271 me->InterruptNonMeleeSpells(false);
272 DoCast(target, SPELL_GREEN_BEAM);
273
274 //Correctly update our target
275 me->SetTarget(target->GetGUID());
276 }
277
278 //Beam every 3 seconds
279 BeamTimer = 3000;
280 } else BeamTimer -= diff;
281
282 //ClawTentacleTimer
283 if (ClawTentacleTimer <= diff)
284 {
285 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
286 {
287 Creature* Spawned = NULL;
288
289 //Spawn claw tentacle on the random target
290 Spawned = me->SummonCreature(NPC_CLAW_TENTACLE, *target, TEMPSUMMON_CORPSE_DESPAWN, 500);
291
292 if (Spawned && Spawned->AI())
293 Spawned->AI()->AttackStart(target);
294 }
295
296 //One claw tentacle every 12.5 seconds
297 ClawTentacleTimer = 12500;
298 } else ClawTentacleTimer -= diff;
299
300 //PhaseTimer
301 if (PhaseTimer <= diff)
302 {
303 //Switch to Dark Beam
304 instance->SetData(DATA_CTHUN_PHASE, PHASE_EYE_RED_BEAM);
305
306 me->InterruptNonMeleeSpells(false);
307 me->SetReactState(REACT_PASSIVE);
308
309 //Remove any target
310 me->SetTarget(ObjectGuid::Empty);
311
312 //Select random target for dark beam to start on
313 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
314 {
315 //Face our target
316 DarkGlareAngle = me->GetAngle(target);
317 DarkGlareTickTimer = 1000;
318 DarkGlareTick = 0;
319 ClockWise = RAND(true, false);
320 }
321
322 //Add red coloration to C'thun
323 DoCast(me, SPELL_RED_COLORATION, true);
324
325 //Freeze animation
326 DoCast(me, SPELL_FREEZE_ANIM);
327 me->SetOrientation(DarkGlareAngle);
328 me->StopMoving();
329
330 //Darkbeam for 35 seconds
331 PhaseTimer = 35000;
332 } else PhaseTimer -= diff;
333
334 break;
335
336 case PHASE_EYE_RED_BEAM:
337 if (DarkGlareTick < 35)
338 {
339 if (DarkGlareTickTimer <= diff)
340 {
341 //Set angle and cast
342 if (ClockWise)
343 me->SetOrientation(DarkGlareAngle + DarkGlareTick * float(M_PI) / 35);
344 else
345 me->SetOrientation(DarkGlareAngle - DarkGlareTick * float(M_PI) / 35);
346
347 me->StopMoving();
348
349 //Actual dark glare cast, maybe something missing here?
350 DoCast(me, SPELL_DARK_GLARE, false);
351
352 //Increase tick
353 ++DarkGlareTick;
354
355 //1 second per tick
356 DarkGlareTickTimer = 1000;
357 } else DarkGlareTickTimer -= diff;
358 }
359
360 //PhaseTimer
361 if (PhaseTimer <= diff)
362 {
363 //Switch to Eye Beam
364 instance->SetData(DATA_CTHUN_PHASE, PHASE_EYE_GREEN_BEAM);
365
366 BeamTimer = 3000;
367 ClawTentacleTimer = 12500; //4 per Eye beam phase (unsure if they spawn during Dark beam)
368
369 me->InterruptNonMeleeSpells(false);
370
371 //Remove Red coloration from c'thun
372 me->RemoveAurasDueToSpell(SPELL_RED_COLORATION);
373 me->RemoveAurasDueToSpell(SPELL_FREEZE_ANIM);
374
375 //set it back to aggressive
376 me->SetReactState(REACT_AGGRESSIVE);
377
378 //Eye Beam for 50 seconds
379 PhaseTimer = 50000;
380 } else PhaseTimer -= diff;
381
382 break;
383
384 //Transition phase
385 case PHASE_CTHUN_TRANSITION:
386 //Remove any target
387 me->SetTarget(ObjectGuid::Empty);
388 me->SetHealth(0);
389 me->SetVisible(false);
390 break;
391
392 //Dead phase
393 case PHASE_CTHUN_DONE:
394 Creature* pPortal= me->FindNearestCreature(NPC_CTHUN_PORTAL, 10);
395 if (pPortal)
396 pPortal->DespawnOrUnsummon();
397
398 me->DespawnOrUnsummon();
399 break;
400 }
401 }
402
403 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override
404 {
405 switch (instance->GetData(DATA_CTHUN_PHASE))
406 {
407 case PHASE_EYE_GREEN_BEAM:
408 case PHASE_EYE_RED_BEAM:
409 //Only if it will kill
410 if (damage < me->GetHealth())
411 return;
412
413 //Fake death in phase 0 or 1 (green beam or dark glare phase)
414 me->InterruptNonMeleeSpells(false);
415
416 //Remove Red coloration from c'thun
417 me->RemoveAurasDueToSpell(SPELL_RED_COLORATION);
418
419 //Reset to normal emote state and prevent select and attack
420 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
421
422 //Remove Target field
423 me->SetTarget(ObjectGuid::Empty);
424
425 //Death animation/respawning;
426 instance->SetData(DATA_CTHUN_PHASE, PHASE_CTHUN_TRANSITION);
427
428 me->SetHealth(0);
429 damage = 0;
430
431 me->InterruptNonMeleeSpells(true);
432 me->RemoveAllAuras();
433 break;
434
435 case PHASE_CTHUN_DONE:
436 //Allow death here
437 return;
438
439 default:
440 //Prevent death in these phases
441 damage = 0;
442 return;
443 }
444 }
445 };

545 Unit* SelectRandomNotStomach()
546 {
547 if (Stomach_Map.empty())
548 return NULL;
549
550 std::unordered_map<ObjectGuid, bool>::const_iterator i = Stomach_Map.begin();
551
552 std::list<Unit*> temp;
553 std::list<Unit*>::const_iterator j;
554
555 //Get all players in map
556 while (i != Stomach_Map.end())
557 {
558 //Check for valid player
559 Unit* unit = ObjectAccessor::GetUnit(*me, i->first);
560
561 //Only units out of stomach
562 if (unit && i->second == false)
563 temp.push_back(unit);
564
565 ++i;
566 }
567
568 if (temp.empty())
569 return NULL;
570
571 j = temp.begin();
572
573 //Get random but only if we have more than one unit on threat list
574 if (temp.size() > 1)
575 advance(j, rand32() % (temp.size() - 1));
576
577 return (*j);
578 }

580 void UpdateAI(uint32 diff) override
581 {
582 //Check if we have a target
583 if (!UpdateVictim())
584 {
585 //No target so we'll use this section to do our random wispers instance wide
586 //WisperTimer

589 //Play random sound to the zone

591 for (Map::PlayerList::const_iterator itr = PlayerList.begin(); itr != PlayerList.end(); ++itr)
592 me->PlayDirectSound(RANDOM_SOUND_WHISPER, itr->GetSource());
593
594 //One random wisper every 90 - 300 seconds

621 } else EyeTentacleTimer -= diff;
622 }

624 switch (currentPhase)
625 {

627 case PHASE_CTHUN_TRANSITION:
628 //PhaseTimer

642 //Emerging phase
643 //AttackStart(ObjectAccessor::GetUnit(*me, HoldpPlayer));
644 DoZoneInCombat();
645
646 //Place all units in threat list on outside of stomach
647 Stomach_Map.clear();
648
649 for (std::list<HostileReference*>::const_iterator i = me->getThreatManager().getThreatList().begin(); i != me->getThreatManager().getThreatList().end(); ++i)

656 for (uint8 i = 0; i < 2; i++)
657 {

659 if (!spawned)
660 ++FleshTentaclesKilled;
661 }
662
663 PhaseTimer = 0;
664 } else PhaseTimer -= diff;
665
666 break;
667
668 //Body Phase
669 case PHASE_CTHUN_STOMACH:
670 //Remove Target field
671 me->SetTarget(ObjectGuid::Empty);
672
673 //Weaken
674 if (FleshTentaclesKilled > 1)
675 {
676 instance->SetData(DATA_CTHUN_PHASE, PHASE_CTHUN_WEAK);
677
678 Talk(EMOTE_WEAKENED);
679 PhaseTimer = 45000;
680
681 DoCast(me, SPELL_PURPLE_COLORATION, true);
682
683 std::unordered_map<ObjectGuid, bool>::iterator i = Stomach_Map.begin();
684
685 //Kick all players out of stomach
686 while (i != Stomach_Map.end())
687 {
688 //Check for valid player
689 Unit* unit = ObjectAccessor::GetUnit(*me, i->first);
690
691 //Only move units in stomach
692 if (unit && i->second == true)
693 {
694 //Teleport each player out

712 if (StomachAcidTimer <= diff)
713 {
714 //Apply aura to all players in stomach
715 std::unordered_map<ObjectGuid, bool>::iterator i = Stomach_Map.begin();
716
717 while (i != Stomach_Map.end())
718 {
719 //Check for valid player
720 Unit* unit = ObjectAccessor::GetUnit(*me, i->first);
721
722 //Only apply to units in stomach
723 if (unit && i->second == true)
724 {
725 //Cast digestive acid on them
726 DoCast(unit, SPELL_DIGESTIVE_ACID, true);

744 }

746 StomachAcidTimer = 4000;

750 if (StomachEnterTimer <= diff)
751 {
752 if (Unit* target = SelectRandomNotStomach())
753 {
754 //Set target in stomach
755 Stomach_Map[target->GetGUID()] = true;
756 target->InterruptNonMeleeSpells(false);
757 target->CastSpell(target, SPELL_MOUTH_TENTACLE, true, NULL, NULL, me->GetGUID());

765 if (StomachEnterVisTimer && StomachEnterTarget)
766 {
767 if (StomachEnterVisTimer <= diff)
768 {
769 //Check for valid player
770 Unit* unit = ObjectAccessor::GetUnit(*me, StomachEnterTarget);
771
772 if (unit)
773 {
774 DoTeleportPlayer(unit, STOMACH_X, STOMACH_Y, STOMACH_Z, STOMACH_O);
775 }
776
777 StomachEnterTarget.Clear();
778 StomachEnterVisTimer = 0;
779 } else StomachEnterVisTimer -= diff;
780 }
781
782 //GientClawTentacleTimer
783 if (GiantClawTentacleTimer <= diff)
784 {
785 if (Unit* target = SelectRandomNotStomach())
786 {
787 //Spawn claw tentacle on the random target
788 if (Creature* spawned = me->SummonCreature(NPC_GIANT_CLAW_TENTACLE, *target, TEMPSUMMON_CORPSE_DESPAWN, 500))
789 if (spawned->AI())
790 spawned->AI()->AttackStart(target);
791 }
792
793 //One giant claw tentacle every minute
794 GiantClawTentacleTimer = 60000;
795 } else GiantClawTentacleTimer -= diff;
796
797 //GiantEyeTentacleTimer
798 if (GiantEyeTentacleTimer <= diff)
799 {
800 if (Unit* target = SelectRandomNotStomach())
801 {
802 //Spawn claw tentacle on the random target
803 if (Creature* spawned = me->SummonCreature(NPC_GIANT_EYE_TENTACLE, *target, TEMPSUMMON_CORPSE_DESPAWN, 500))
804 if (spawned->AI())
805 spawned->AI()->AttackStart(target);
806 }
807
808 //One giant eye tentacle every minute
809 GiantEyeTentacleTimer = 60000;
810 } else GiantEyeTentacleTimer -= diff;
811
812 break;
813
814 //Weakened state
815 case PHASE_CTHUN_WEAK:
816 //PhaseTimer
817 if (PhaseTimer <= diff)
818 {
819 //Switch
820 instance->SetData(DATA_CTHUN_PHASE, PHASE_CTHUN_STOMACH);
821
822 //Remove purple coloration
823 me->RemoveAurasDueToSpell(SPELL_PURPLE_COLORATION);
824
825 //Spawn 2 flesh tentacles
826 FleshTentaclesKilled = 0;
827
828 //Spawn flesh tentacle
829 for (uint8 i = 0; i < 2; i++)
830 {
831 Creature* spawned = me->SummonCreature(NPC_FLESH_TENTACLE, FleshTentaclePos[i], TEMPSUMMON_CORPSE_DESPAWN);
832 if (!spawned)
833 ++FleshTentaclesKilled;
834 }
835
836 PhaseTimer = 0;
837 } else PhaseTimer -= diff;
838
839 break;
840 }
841 }
842
843 void JustDied(Unit* /*killer*/) override
844 {
845 instance->SetData(DATA_CTHUN_PHASE, PHASE_CTHUN_DONE);
846 }
847
848 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override
849 {
850 switch (instance->GetData(DATA_CTHUN_PHASE))
851 {
852 case PHASE_CTHUN_STOMACH:
853 //Not weakened so reduce damage by 99%
854 damage /= 100;
855 if (damage == 0)
856 damage = 1;
857
858 //Prevent death in non-weakened state
859 if (damage >= me->GetHealth())
860 damage = 0;
861
862 return;
863
864 case PHASE_CTHUN_WEAK:
865 //Weakened - takes normal damage
866 return;
867
868 default:
869 damage = 0;
870 break;
871 }
872 }

875 {
876 switch (param)
877 {
878 case ACTION_FLESH_TENTACLE_KILLED:
879 ++FleshTentaclesKilled;
880 break;
881 }

29enum Sartura
30{
31 SAY_AGGRO = 0,
32 SAY_SLAY = 1,
33 SAY_DEATH = 2,
34
35 SPELL_WHIRLWIND = 26083,
36 SPELL_ENRAGE = 28747, //Not sure if right ID.
37 SPELL_ENRAGEHARD = 28798,
38
39//Guard Spell
40 SPELL_WHIRLWINDADD = 26038,
41 SPELL_KNOCKBACK = 26027
42};

44class boss_sartura : public CreatureScript
45{
46public:
47 boss_sartura() : CreatureScript("boss_sartura") { }
48
49 CreatureAI* GetAI(Creature* creature) const override
50 {
51 return new boss_sarturaAI(creature);
52 }
53
54 struct boss_sarturaAI : public ScriptedAI
55 {
56 boss_sarturaAI(Creature* creature) : ScriptedAI(creature)
57 {
58 Initialize();
59 }
60
61 void Initialize()
62 {
63 WhirlWind_Timer = 30000;
64 WhirlWindRandom_Timer = urand(3000, 7000);
65 WhirlWindEnd_Timer = 15000;
66 AggroReset_Timer = urand(45000, 55000);
67 AggroResetEnd_Timer = 5000;
68 EnrageHard_Timer = 10 * 60000;
69
70 WhirlWind = false;
71 AggroReset = false;
72 Enraged = false;
73 EnragedHard = false;
74 }
75
76 uint32 WhirlWind_Timer;
77 uint32 WhirlWindRandom_Timer;
78 uint32 WhirlWindEnd_Timer;
79 uint32 AggroReset_Timer;
80 uint32 AggroResetEnd_Timer;
81 uint32 EnrageHard_Timer;
82
83 bool Enraged;
84 bool EnragedHard;
85 bool WhirlWind;
86 bool AggroReset;
87
88 void Reset() override
89 {
90 Initialize();
91 }
92
93 void EnterCombat(Unit* /*who*/) override
94 {
95 Talk(SAY_AGGRO);
96 }
97
98 void JustDied(Unit* /*killer*/) override
99 {
100 Talk(SAY_DEATH);
101 }
102
103 void KilledUnit(Unit* /*victim*/) override
104 {
105 Talk(SAY_SLAY);
106 }
107
108 void UpdateAI(uint32 diff) override
109 {
110 //Return since we have no target
111 if (!UpdateVictim())
112 return;
113
114 if (WhirlWind)
115 {
116 if (WhirlWindRandom_Timer <= diff)
117 {
118 //Attack random Gamers
119 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100.0f, true))
120 {
121 me->AddThreat(target, 1.0f);
122 me->TauntApply(target);
123 AttackStart(target);
124 }
125 WhirlWindRandom_Timer = urand(3000, 7000);
126 } else WhirlWindRandom_Timer -= diff;
127
128 if (WhirlWindEnd_Timer <= diff)
129 {
130 WhirlWind = false;
131 WhirlWind_Timer = urand(25000, 40000);
132 } else WhirlWindEnd_Timer -= diff;
133 }
134
135 if (!WhirlWind)
136 {
137 if (WhirlWind_Timer <= diff)
138 {
139 DoCast(me, SPELL_WHIRLWIND);
140 WhirlWind = true;
141 WhirlWindEnd_Timer = 15000;
142 } else WhirlWind_Timer -= diff;
143
144 if (AggroReset_Timer <= diff)
145 {
146 //Attack random Gamers
147 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100.0f, true))
148 {
149 me->AddThreat(target, 1.0f);
150 me->TauntApply(target);
151 AttackStart(target);
152 }
153 AggroReset = true;
154 AggroReset_Timer = urand(2000, 5000);
155 } else AggroReset_Timer -= diff;
156
157 if (AggroReset)
158 {
159 if (AggroResetEnd_Timer <= diff)
160 {
161 AggroReset = false;
162 AggroResetEnd_Timer = 5000;
163 AggroReset_Timer = urand(35000, 45000);
164 } else AggroResetEnd_Timer -= diff;
165 }
166
167 //If she is 20% enrage
168 if (!Enraged)
169 {
170 if (!HealthAbovePct(20) && !me->IsNonMeleeSpellCast(false))
171 {
172 DoCast(me, SPELL_ENRAGE);
173 Enraged = true;
174 }
175 }
176
177 //After 10 minutes hard enrage
178 if (!EnragedHard)
179 {
180 if (EnrageHard_Timer <= diff)
181 {
182 DoCast(me, SPELL_ENRAGEHARD);
183 EnragedHard = true;
184 } else EnrageHard_Timer -= diff;
185 }
186
187 DoMeleeAttackIfReady();
188 }
189 }
190 };
191
192};

252 WhirlWind = true;

254 WhirlWindEnd_Timer = 15000;
255 } else WhirlWind_Timer -= diff;
256
257 if (WhirlWind)
258 {
259 if (WhirlWindRandom_Timer <= diff)
260 {
261 //Attack random Gamers
262 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100.0f, true))
263 {
264 me->AddThreat(target, 1.0f);
265 me->TauntApply(target);
266 AttackStart(target);
267 }
268
269 WhirlWindRandom_Timer = urand(3000, 7000);
270 } else WhirlWindRandom_Timer -= diff;
271
272 if (WhirlWindEnd_Timer <= diff)
273 {
274 WhirlWind = false;
275 } else WhirlWindEnd_Timer -= diff;
276 }
277
278 if (!WhirlWind)
279 {
280 if (AggroReset_Timer <= diff)
281 {
282 //Attack random Gamers
283 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100.0f, true))
284 {
285 me->AddThreat(target, 1.0f);
286 me->TauntApply(target);
287 AttackStart(target);
288 }
289

293

308 } else AggroResetEnd_Timer -= diff;
309 }
310

312 }

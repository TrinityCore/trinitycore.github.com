30enum Spells
31{
32 SPELL_CLEAVE = 26350,
33 SPELL_TOXIC_VOLLEY = 25812,
34 SPELL_POISON_CLOUD = 38718, //Only Spell with right dmg.

44class boss_kri : public CreatureScript
45{
46public:
47 boss_kri() : CreatureScript("boss_kri") { }
48
49 CreatureAI* GetAI(Creature* creature) const override
50 {
51 return GetInstanceAI<boss_kriAI>(creature);
52 }
53
54 struct boss_kriAI : public ScriptedAI
55 {
56 boss_kriAI(Creature* creature) : ScriptedAI(creature)
57 {
58 Initialize();
59 instance = creature->GetInstanceScript();
60 }
61
62 void Initialize()
63 {
64 Cleave_Timer = urand(4000, 8000);
65 ToxicVolley_Timer = urand(6000, 12000);
66 Check_Timer = 2000;
67
68 VemDead = false;
69 Death = false;
70 }
71
72 InstanceScript* instance;
73
74 uint32 Cleave_Timer;
75 uint32 ToxicVolley_Timer;
76 uint32 Check_Timer;
77
78 bool VemDead;
79 bool Death;
80
81 void Reset() override
82 {
83 Initialize();
84 }
85
86 void EnterCombat(Unit* /*who*/) override
87 {
88 }
89
90 void JustDied(Unit* /*killer*/) override
91 {
92 if (instance->GetData(DATA_BUG_TRIO_DEATH) < 2)// Unlootable if death
93 me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
94
95 instance->SetData(DATA_BUG_TRIO_DEATH, 1);
96 }
97 void UpdateAI(uint32 diff) override
98 {
99 //Return since we have no target
100 if (!UpdateVictim())
101 return;
102
103 //Cleave_Timer
104 if (Cleave_Timer <= diff)
105 {
106 DoCastVictim(SPELL_CLEAVE);
107 Cleave_Timer = urand(5000, 12000);
108 } else Cleave_Timer -= diff;
109
110 //ToxicVolley_Timer
111 if (ToxicVolley_Timer <= diff)
112 {
113 DoCastVictim(SPELL_TOXIC_VOLLEY);
114 ToxicVolley_Timer = urand(10000, 15000);
115 } else ToxicVolley_Timer -= diff;
116
117 if (!HealthAbovePct(5) && !Death)
118 {
119 DoCastVictim(SPELL_POISON_CLOUD);
120 Death = true;
121 }
122
123 if (!VemDead)
124 {
125 //Checking if Vem is dead. If yes we will enrage.
126 if (Check_Timer <= diff)
127 {
128 if (instance->GetData(DATA_VEMISDEAD))
129 {
130 DoCast(me, SPELL_ENRAGE);
131 VemDead = true;
132 }
133 Check_Timer = 2000;
134 } else Check_Timer -=diff;
135 }
136
137 DoMeleeAttackIfReady();
138 }
139 };

153 struct boss_vemAI : public ScriptedAI
154 {
155 boss_vemAI(Creature* creature) : ScriptedAI(creature)
156 {
157 Initialize();
158 instance = creature->GetInstanceScript();
159 }
160
161 void Initialize()
162 {
163 Charge_Timer = urand(15000, 27000);
164 KnockBack_Timer = urand(8000, 20000);
165 Enrage_Timer = 120000;
166
167 Enraged = false;
168 }
169
170 InstanceScript* instance;
171
172 uint32 Charge_Timer;
173 uint32 KnockBack_Timer;
174 uint32 Enrage_Timer;
175
176 bool Enraged;
177
178 void Reset() override
179 {
180 Initialize();
181 }
182
183 void JustDied(Unit* /*killer*/) override
184 {
185 instance->SetData(DATA_VEM_DEATH, 0);
186 if (instance->GetData(DATA_BUG_TRIO_DEATH) < 2)// Unlootable if death
187 me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
188 instance->SetData(DATA_BUG_TRIO_DEATH, 1);
189 }
190
191 void EnterCombat(Unit* /*who*/) override
192 {
193 }
194
195 void UpdateAI(uint32 diff) override
196 {
197 //Return since we have no target
198 if (!UpdateVictim())
199 return;
200
201 //Charge_Timer
202 if (Charge_Timer <= diff)
203 {
204 Unit* target = NULL;
205 target = SelectTarget(SELECT_TARGET_RANDOM, 0);
206 if (target)
207 {
208 DoCast(target, SPELL_CHARGE);
209 //me->SendMonsterMove(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, true, 1);
210 AttackStart(target);
211 }
212
213 Charge_Timer = urand(8000, 16000);
214 } else Charge_Timer -= diff;
215
216 //KnockBack_Timer
217 if (KnockBack_Timer <= diff)
218 {
219 DoCastVictim(SPELL_KNOCKBACK);
220 if (DoGetThreat(me->GetVictim()))
221 DoModifyThreatPercent(me->GetVictim(), -80);
222 KnockBack_Timer = urand(15000, 25000);
223 } else KnockBack_Timer -= diff;
224
225 //Enrage_Timer
226 if (!Enraged && Enrage_Timer <= diff)
227 {
228 DoCast(me, SPELL_ENRAGE);
229 Enraged = true;
230 } else Charge_Timer -= diff;
231
232 DoMeleeAttackIfReady();
233 }
234 };

238class boss_yauj : public CreatureScript
239{
240public:
241 boss_yauj() : CreatureScript("boss_yauj") { }
242
243 CreatureAI* GetAI(Creature* creature) const override
244 {
245 return GetInstanceAI<boss_yaujAI>(creature);
246 }
247
248 struct boss_yaujAI : public ScriptedAI
249 {
250 boss_yaujAI(Creature* creature) : ScriptedAI(creature)
251 {
252 Initialize();
253 instance = creature->GetInstanceScript();
254 }
255
256 void Initialize()
257 {
258 Heal_Timer = urand(25000, 40000);
259 Fear_Timer = urand(12000, 24000);
260 Check_Timer = 2000;
261
262 VemDead = false;
263 }
264
265 InstanceScript* instance;
266
267 uint32 Heal_Timer;
268 uint32 Fear_Timer;
269 uint32 Check_Timer;
270
271 bool VemDead;
272
273 void Reset() override
274 {
275 Initialize();
276 }
277
278 void JustDied(Unit* /*killer*/) override
279 {
280 if (instance->GetData(DATA_BUG_TRIO_DEATH) < 2)// Unlootable if death
281 me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
282 instance->SetData(DATA_BUG_TRIO_DEATH, 1);
283
284 for (uint8 i = 0; i < 10; ++i)
285 {
286 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
287 {
288 if (Creature* Summoned = me->SummonCreature(15621, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 90000))
289 Summoned->AI()->AttackStart(target);
290 }
291 }
292 }
293
294 void EnterCombat(Unit* /*who*/) override
295 {
296 }
297
298 void UpdateAI(uint32 diff) override
299 {
300 //Return since we have no target
301 if (!UpdateVictim())
302 return;
303
304 //Fear_Timer
305 if (Fear_Timer <= diff)
306 {
307 DoCastVictim(SPELL_FEAR);
308 DoResetThreat();
309 Fear_Timer = 20000;
310 } else Fear_Timer -= diff;
311
312 //Casting Heal to other twins or herself.
313 if (Heal_Timer <= diff)
314 {
315 switch (urand(0, 2))
316 {
317 case 0:
318 if (Creature* kri = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_KRI)))
319 DoCast(kri, SPELL_HEAL);
320 break;
321 case 1:
322 if (Creature* vem = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_VEM)))
323 DoCast(vem, SPELL_HEAL);
324 break;
325 case 2:
326 DoCast(me, SPELL_HEAL);
327 break;
328 }
329
330 Heal_Timer = 15000 + rand32() % 15000;
331 } else Heal_Timer -= diff;
332
333 //Checking if Vem is dead. If yes we will enrage.
334 if (Check_Timer <= diff)
335 {
336 if (!VemDead)
337 {
338 if (instance->GetData(DATA_VEMISDEAD))
339 {
340 DoCast(me, SPELL_ENRAGE);
341 VemDead = true;
342 }
343 }
344 Check_Timer = 2000;
345 } else Check_Timer -= diff;
346
347 DoMeleeAttackIfReady();
348 }
349 };

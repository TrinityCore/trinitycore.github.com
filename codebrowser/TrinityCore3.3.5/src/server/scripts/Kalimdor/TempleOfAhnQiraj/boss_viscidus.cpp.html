87class boss_viscidus : public CreatureScript
88{
89 public:
90 boss_viscidus() : CreatureScript("boss_viscidus") { }
91
92 struct boss_viscidusAI : public BossAI
93 {
94 boss_viscidusAI(Creature* creature) : BossAI(creature, DATA_VISCIDUS)
95 {
96 Initialize();
97 }
98
99 void Initialize()
100 {
101 _hitcounter = 0;
102 _phase = PHASE_FROST;
103 }
104
105 void Reset() override
106 {
107 _Reset();
108 Initialize();
109 }
110
111 void DamageTaken(Unit* attacker, uint32& /*damage*/) override
112 {
113 if (_phase != PHASE_MELEE)
114 return;
115
116 ++_hitcounter;
117
118 if (attacker->HasUnitState(UNIT_STATE_MELEE_ATTACKING) && _hitcounter >= HITCOUNTER_EXPLODE)
119 {
120 Talk(EMOTE_EXPLODE);
121 events.Reset();
122 _phase = PHASE_GLOB;
123 DoCast(me, SPELL_VISCIDUS_EXPLODE);
124 me->SetVisible(false);
125 me->RemoveAura(SPELL_TOXIN);
126 me->RemoveAura(SPELL_VISCIDUS_FREEZE);
127
128 uint8 NumGlobes = me->GetHealthPct() / 5.0f;
129 for (uint8 i = 0; i < NumGlobes; ++i)
130 {
131 float Angle = i * 2 * float(M_PI) / NumGlobes;
132 float X = ViscidusCoord.GetPositionX() + std::cos(Angle) * RoomRadius;
133 float Y = ViscidusCoord.GetPositionY() + std::sin(Angle) * RoomRadius;
134 float Z = -35.0f;
135
136 if (TempSummon* Glob = me->SummonCreature(NPC_GLOB_OF_VISCIDUS, X, Y, Z))
137 {
138 Glob->UpdateAllowedPositionZ(X, Y, Z);
139 Glob->NearTeleportTo(X, Y, Z, 0.0f);
140 Glob->GetMotionMaster()->MovePoint(ROOM_CENTER, ViscidusCoord);
141 }
142 }
143 }
144 else if (_hitcounter == HITCOUNTER_SHATTER)
145 Talk(EMOTE_SHATTER);
146 else if (_hitcounter == HITCOUNTER_CRACK)
147 Talk(EMOTE_CRACK);
148 }
149
150 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
151 {
152 if ((spell->GetSchoolMask() & SPELL_SCHOOL_MASK_FROST) && _phase == PHASE_FROST && me->GetHealthPct() > 5.0f)
153 {
154 ++_hitcounter;
155
156 if (_hitcounter >= HITCOUNTER_FREEZE)
157 {
158 _hitcounter = 0;
159 Talk(EMOTE_FROZEN);
160 _phase = PHASE_MELEE;
161 DoCast(me, SPELL_VISCIDUS_FREEZE);
162 me->RemoveAura(SPELL_VISCIDUS_SLOWED_MORE);
163 events.ScheduleEvent(EVENT_RESET_PHASE, 15000);
164 }
165 else if (_hitcounter >= HITCOUNTER_SLOW_MORE)
166 {
167 Talk(EMOTE_FREEZE);
168 me->RemoveAura(SPELL_VISCIDUS_SLOWED);
169 DoCast(me, SPELL_VISCIDUS_SLOWED_MORE);
170 }
171 else if (_hitcounter >= HITCOUNTER_SLOW)
172 {
173 Talk(EMOTE_SLOW);
174 DoCast(me, SPELL_VISCIDUS_SLOWED);
175 }
176 }
177 }
178
179 void EnterCombat(Unit* /*who*/) override
180 {
181 _EnterCombat();
182 events.Reset();
183 InitSpells();
184 }
185
186 void InitSpells()
187 {
188 DoCast(me, SPELL_TOXIN);
189 events.ScheduleEvent(EVENT_POISONBOLT_VOLLEY, urand(10000, 15000));
190 events.ScheduleEvent(EVENT_POISON_SHOCK, urand(7000, 12000));
191 }
192
193 void EnterEvadeMode(EvadeReason why) override
194 {
195 summons.DespawnAll();
196 ScriptedAI::EnterEvadeMode(why);
197 }
198
199 void JustDied(Unit* /*killer*/) override
200 {
201 DoCast(me, SPELL_VISCIDUS_SUICIDE);
202 summons.DespawnAll();
203 }
204
205 void UpdateAI(uint32 diff) override
206 {
207 if (!UpdateVictim())
208 return;
209
210 if (_phase == PHASE_GLOB && summons.empty())
211 {
212 DoResetThreat();
213 me->NearTeleportTo(ViscidusCoord.GetPositionX(),
214 ViscidusCoord.GetPositionY(),
215 ViscidusCoord.GetPositionZ(),
216 ViscidusCoord.GetOrientation());
217
218 _hitcounter = 0;
219 _phase = PHASE_FROST;
220 InitSpells();
221 me->SetVisible(true);
222 }
223
224 events.Update(diff);
225
226 while (uint32 eventId = events.ExecuteEvent())
227 {
228 switch (eventId)
229 {
230 case EVENT_POISONBOLT_VOLLEY:
231 DoCast(me, SPELL_POISONBOLT_VOLLEY);
232 events.ScheduleEvent(EVENT_POISONBOLT_VOLLEY, urand(10000, 15000));
233 break;
234 case EVENT_POISON_SHOCK:
235 DoCast(me, SPELL_POISON_SHOCK);
236 events.ScheduleEvent(EVENT_POISON_SHOCK, urand(7000, 12000));
237 break;
238 case EVENT_RESET_PHASE:
239 _hitcounter = 0;
240 _phase = PHASE_FROST;
241 break;
242 default:
243 break;
244 }
245 }
246
247 if (_phase != PHASE_GLOB)
248 DoMeleeAttackIfReady();
249 }
250
251 private:
252 uint8 _hitcounter;
253 Phases _phase;
254 };
255
256 CreatureAI* GetAI(Creature* creature) const override
257 {
258 return new boss_viscidusAI(creature);
259 }

268 {
269 npc_glob_of_viscidusAI(Creature* creature) : ScriptedAI(creature) { }
270
271 void JustDied(Unit* /*killer*/) override
272 {
273 InstanceScript* Instance = me->GetInstanceScript();
274
275 if (Creature* Viscidus = me->GetMap()->GetCreature(Instance->GetGuidData(DATA_VISCIDUS)))
276 {
277 if (BossAI* ViscidusAI = dynamic_cast<BossAI*>(Viscidus->GetAI()))
278 ViscidusAI->SummonedCreatureDespawn(me);
279
280 if (Viscidus->IsAlive() && Viscidus->GetHealthPct() < 5.0f)
281 {
282 Viscidus->SetVisible(true);
283 if (Viscidus->GetVictim())
284 Viscidus->EnsureVictim()->Kill(Viscidus);
285 }
286 else
287 {
288 Viscidus->SetHealth(Viscidus->GetHealth() - Viscidus->GetMaxHealth() / 20);
289 Viscidus->CastSpell(Viscidus, SPELL_VISCIDUS_SHRINKS);
290 }
291 }
292 }
293
294 void MovementInform(uint32 /*type*/, uint32 id) override
295 {
296 if (id == ROOM_CENTER)
297 {
298 DoCast(me, SPELL_REJOIN_VISCIDUS);
299 if (TempSummon* summon = me->ToTempSummon())
300 summon->UnSummon();
301 }
302 }
303 };

305 CreatureAI* GetAI(Creature* creature) const override
306 {
307 return GetInstanceAI<npc_glob_of_viscidusAI>(creature);
308 }

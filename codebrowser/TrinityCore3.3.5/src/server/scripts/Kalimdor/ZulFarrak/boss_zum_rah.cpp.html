27enum Says
28{
29 SAY_SANCT_INVADE = 0,
30 SAY_WARD = 1,
31 SAY_KILL = 2
32};
33
34enum Spells
35{
36 SPELL_SHADOW_BOLT = 12739,
37 SPELL_SHADOWBOLT_VOLLEY = 15245,
38 SPELL_WARD_OF_ZUM_RAH = 11086,
39 SPELL_HEALING_WAVE = 12491
40};
41
42enum Events
43{
44 EVENT_SHADOW_BOLT = 1,
45 EVENT_SHADOWBOLT_VOLLEY = 2,
46 EVENT_WARD_OF_ZUM_RAH = 3,
47 EVENT_HEALING_WAVE = 4
48};
49
50enum Faction
51{
52 ZUMRAH_FRIENDLY_FACTION = 35
53};
54
55class boss_zum_rah : public CreatureScript
56{
57public:
58 boss_zum_rah() : CreatureScript("boss_zum_rah") { }
59
60 struct boss_zum_rahAI : public BossAI
61 {
62 boss_zum_rahAI(Creature* creature) : BossAI(creature, DATA_ZUM_RAH)
63 {
64 Initialize();
65 }
66
67 void Initialize()
68 {
69 _ward80 = false;
70 _ward40 = false;
71 _heal30 = false;
72 }
73
74 void Reset() override
75 {
76 me->setFaction(ZUMRAH_FRIENDLY_FACTION); // areatrigger sets faction to enemy
77 Initialize();
78 }
79
80 void EnterCombat(Unit* /*who*/) override
81 {
82 Talk(SAY_SANCT_INVADE);
83 events.ScheduleEvent(EVENT_SHADOW_BOLT, 1000);
84 events.ScheduleEvent(EVENT_SHADOWBOLT_VOLLEY, 10000);
85 }
86
87 void JustDied(Unit* /*killer*/) override
88 {
89 instance->SetData(DATA_ZUM_RAH, DONE);
90 }
91
92 void KilledUnit(Unit* /*victim*/) override
93 {
94 Talk(SAY_KILL);
95 }
96
97 void UpdateAI(uint32 diff) override
98 {
99 if (!UpdateVictim())
100 return;
101
102 events.Update(diff);
103
104 while (uint32 eventId = events.ExecuteEvent())
105 {
106 switch (eventId)
107 {
108 case EVENT_SHADOW_BOLT:
109 DoCastVictim(SPELL_SHADOW_BOLT);
110 events.ScheduleEvent(EVENT_SHADOW_BOLT, 4000);
111 break;
112 case EVENT_WARD_OF_ZUM_RAH:
113 DoCast(me,SPELL_WARD_OF_ZUM_RAH);
114 break;
115 case EVENT_HEALING_WAVE:
116 DoCast(me,SPELL_HEALING_WAVE);
117 break;
118 case EVENT_SHADOWBOLT_VOLLEY:
119 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
120 DoCast(target, SPELL_SHADOWBOLT_VOLLEY);
121 events.ScheduleEvent(EVENT_SHADOWBOLT_VOLLEY, 9000);
122 break;
123 default:
124 break;
125 }
126 }
127
128 if (!_ward80 && HealthBelowPct(80))
129 {
130 _ward80 = true;
131 Talk(SAY_WARD);
132 events.ScheduleEvent(EVENT_WARD_OF_ZUM_RAH, 1000);
133 }
134
135 if (!_ward40 && HealthBelowPct(40))
136 {
137 _ward40 = true;
138 Talk(SAY_WARD);
139 events.ScheduleEvent(EVENT_WARD_OF_ZUM_RAH, 1000);
140 }
141
142 if (!_heal30 && HealthBelowPct(30))
143 {
144 _heal30 = true;
145 events.ScheduleEvent(EVENT_HEALING_WAVE, 3000);
146 }
147
148 DoMeleeAttackIfReady();
149 }
150
151 private:
152 bool _ward80;
153 bool _ward40;
154 bool _heal30;
155
156 };
157
158 CreatureAI* GetAI(Creature* creature) const override
159 {
160 return GetInstanceAI<boss_zum_rahAI>(creature);
161 }
162};
163
164void AddSC_boss_zum_rah()
165{
166 new boss_zum_rah();
167}

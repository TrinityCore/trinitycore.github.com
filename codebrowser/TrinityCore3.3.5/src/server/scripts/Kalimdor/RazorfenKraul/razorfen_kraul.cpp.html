27enum Willix
28{
29 SAY_READY = 0,
30 SAY_POINT = 1,
31 SAY_AGGRO1 = 2,
32 SAY_BLUELEAF = 3,
33 SAY_DANGER = 4,
34 SAY_BAD = 5,
35 SAY_THINK = 6,
36 SAY_SOON = 7,
37 SAY_FINALY = 8,
38 SAY_WIN = 9,
39 SAY_END = 10,
40
41 QUEST_WILLIX_THE_IMPORTER = 1144,
42 ENTRY_BOAR = 4514
43};
44
45class npc_willix : public CreatureScript
46{
47public:
48 npc_willix() : CreatureScript("npc_willix") { }
49
50 struct npc_willixAI : public npc_escortAI
51 {
52 npc_willixAI(Creature* creature) : npc_escortAI(creature) { }
53
54 void sQuestAccept(Player* player, Quest const* quest) override
55 {
56 if (quest->GetQuestId() == QUEST_WILLIX_THE_IMPORTER)
57 {
58 Start(true, false, player->GetGUID());
59 Talk(SAY_READY, player);
60 me->setFaction(113);
61 }
62 }
63
64 void WaypointReached(uint32 waypointId) override
65 {
66 Player* player = GetPlayerForEscort();
67 if (!player)
68 return;
69
70 switch (waypointId)
71 {
72 case 3:
73 me->HandleEmoteCommand(EMOTE_STATE_POINT);
74 Talk(SAY_POINT, player);
75 break;
76 case 4:
77 me->SummonCreature(ENTRY_BOAR, 2137.66f, 1843.98f, 48.08f, 1.54f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
78 break;
79 case 8:
80 Talk(SAY_BLUELEAF, player);
81 break;
82 case 9:
83 Talk(SAY_DANGER, player);
84 break;
85 case 13:
86 Talk(SAY_BAD, player);
87 break;
88 case 14:
89 me->SummonCreature(ENTRY_BOAR, 2078.91f, 1704.54f, 56.77f, 1.54f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
90 break;
91 case 25:
92 Talk(SAY_THINK, player);
93 break;
94 case 31:
95 Talk(SAY_SOON, player);
96 break;
97 case 42:
98 Talk(SAY_FINALY, player);
99 break;
100 case 43:
101 me->SummonCreature(ENTRY_BOAR, 1956.43f, 1596.97f, 81.75f, 1.54f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
102 break;
103 case 45:
104 Talk(SAY_WIN, player);
105 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
106 player->GroupEventHappens(QUEST_WILLIX_THE_IMPORTER, me);
107 break;
108 case 46:
109 Talk(SAY_END, player);
110 break;
111 }
112 }
113
114 void Reset() override { }
115
116 void EnterCombat(Unit* /*who*/) override
117 {
118 Talk(SAY_AGGRO1);
119 }
120
121 void JustSummoned(Creature* summoned) override
122 {
123 summoned->AI()->AttackStart(me);
124 }
125
126 void JustDied(Unit* /*killer*/) override
127 {
128 if (Player* player = GetPlayerForEscort())
129 player->FailQuest(QUEST_WILLIX_THE_IMPORTER);
130 }
131 };
132
133 CreatureAI* GetAI(Creature* creature) const override
134 {
135 return new npc_willixAI(creature);
136 }
137};
138
139void AddSC_razorfen_kraul()

141 new npc_willix();
142}

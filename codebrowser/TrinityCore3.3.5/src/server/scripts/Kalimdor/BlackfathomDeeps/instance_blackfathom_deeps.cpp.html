38const Position SpawnsLocation[] =
39{
40 {-775.431f, -153.853f, -25.871f, 3.207f},
41 {-775.404f, -174.132f, -25.871f, 3.185f},
42 {-862.430f, -154.937f, -25.871f, 0.060f},

47class instance_blackfathom_deeps : public InstanceMapScript
48{
49public:
50 instance_blackfathom_deeps() : InstanceMapScript("instance_blackfathom_deeps", 48) { }
51
52 InstanceScript* GetInstanceScript(InstanceMap* map) const override
53 {
54 return new instance_blackfathom_deeps_InstanceMapScript(map);
55 }
56
57 struct instance_blackfathom_deeps_InstanceMapScript : public InstanceScript
58 {
59 instance_blackfathom_deeps_InstanceMapScript(Map* map) : InstanceScript(map)
60 {
61 SetHeaders(DataHeader);
62 SetBossNumber(EncounterCount);
63
64 countFires = 0;
65 deathTimes = 0;
66 }
67
68 void OnCreatureCreate(Creature* creature) override
69 {
70 switch (creature->GetEntry())
71 {
72 case NPC_TWILIGHT_LORD_KELRIS:
73 twilightLordKelrisGUID = creature->GetGUID();
74 break;
75 case NPC_LORGUS_JETT:
76 creature->SetHomePosition(LorgusPosition[urand(0, 3)]);
77 break;
78 }
79 }
80
81 void OnGameObjectCreate(GameObject* go) override
82 {
83 switch (go->GetEntry())
84 {
85 case GO_FIRE_OF_AKU_MAI_1:
86 shrine1GUID = go->GetGUID();
87 break;
88 case GO_FIRE_OF_AKU_MAI_2:
89 shrine2GUID = go->GetGUID();
90 break;
91 case GO_FIRE_OF_AKU_MAI_3:
92 shrine3GUID = go->GetGUID();
93 break;
94 case GO_FIRE_OF_AKU_MAI_4:
95 shrine4GUID = go->GetGUID();
96 break;
97 case GO_SHRINE_OF_GELIHAST:
98 shrineOfGelihastGUID = go->GetGUID();
99 if (GetBossState(DATA_GELIHAST) != DONE)
100 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
101 break;
102 case GO_ALTAR_OF_THE_DEEPS:
103 altarOfTheDeepsGUID = go->GetGUID();
104 if (GetBossState(DATA_AKU_MAI) != DONE)
105 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
106 break;
107 case GO_AKU_MAI_DOOR:
108 if (GetBossState(DATA_AKU_MAI) == DONE)
109 HandleGameObject(ObjectGuid::Empty, true, go);
110 mainDoorGUID = go->GetGUID();
111 break;
112 }
113 }
114
115 void SetData(uint32 type, uint32 data) override
116 {
117 switch (type)
118 {
119 case DATA_FIRE:
120 countFires = data;
121 switch (countFires)
122 {
123 case 1:
124 if (GameObject* go = instance->GetGameObject(shrine1GUID))
125 {
126 for (uint8 i = 0; i < 4; ++i)
127 go->SummonCreature(NPC_AKU_MAI_SNAPJAW, SpawnsLocation[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
128 }
129 break;
130 case 2:
131 if (GameObject* go = instance->GetGameObject(shrine1GUID))
132 {
133 for (uint8 i = 0; i < 2; ++i)
134 {
135 go->SummonCreature(NPC_MURKSHALLOW_SOFTSHELL, SpawnsLocation[0], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
136 go->SummonCreature(NPC_MURKSHALLOW_SOFTSHELL, SpawnsLocation[1], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
137 go->SummonCreature(NPC_MURKSHALLOW_SOFTSHELL, SpawnsLocation[2], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
138 go->SummonCreature(NPC_MURKSHALLOW_SOFTSHELL, SpawnsLocation[3], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
139 }
140 }
141 break;
142 case 3:
143 if (GameObject* go = instance->GetGameObject(shrine1GUID))
144 {
145 go->SummonCreature(NPC_AKU_MAI_SERVANT, SpawnsLocation[1], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
146 go->SummonCreature(NPC_AKU_MAI_SERVANT, SpawnsLocation[2], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
147 }
148 break;
149 case 4:
150 if (GameObject* go = instance->GetGameObject(shrine1GUID))
151 {
152 for (uint8 i = 0; i < 4; ++i)
153 go->SummonCreature(NPC_BARBED_CRUSTACEAN, SpawnsLocation[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
154 }
155 break;
156 }
157 break;
158 case DATA_EVENT:
159 deathTimes = data;
160 if (deathTimes == 18)
161 HandleGameObject(mainDoorGUID, true);
162 break;
163 }
164 }
165
166 bool SetBossState(uint32 type, EncounterState state) override
167 {
168 if (!InstanceScript::SetBossState(type, state))
169 return false;
170
171 switch (type)
172 {
173 case DATA_GELIHAST:
174 if (state == DONE)
175 if (GameObject* go = instance->GetGameObject(shrineOfGelihastGUID))
176 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
177 break;
178 case DATA_AKU_MAI:
179 if (state == DONE)
180 if (GameObject* go = instance->GetGameObject(altarOfTheDeepsGUID))
181 {
182 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
183 go->SummonCreature(NPC_MORRIDUNE, SpawnsLocation[4], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
184 }
185 break;
186 default:
187 break;
188 }
189
190 return true;
191 }
192
193 uint32 GetData(uint32 type) const override
194 {
195 switch (type)
196 {
197 case DATA_FIRE:
198 return countFires;
199 case DATA_EVENT:
200 return deathTimes;
201 }
202
203 return 0;
204 }
205
206 ObjectGuid GetGuidData(uint32 data) const override
207 {
208 switch (data)
209 {
210 case DATA_TWILIGHT_LORD_KELRIS:
211 return twilightLordKelrisGUID;
212 case DATA_SHRINE1:
213 return shrine1GUID;
214 case DATA_SHRINE2:
215 return shrine2GUID;
216 case DATA_SHRINE3:
217 return shrine3GUID;
218 case DATA_SHRINE4:
219 return shrine4GUID;
220 case DATA_SHRINE_OF_GELIHAST:
221 return shrineOfGelihastGUID;
222 case DATA_MAINDOOR:
223 return mainDoorGUID;
224 }
225
226 return ObjectGuid::Empty;
227 }
228
229 private:
230 ObjectGuid twilightLordKelrisGUID;
231 ObjectGuid shrine1GUID;
232 ObjectGuid shrine2GUID;
233 ObjectGuid shrine3GUID;
234 ObjectGuid shrine4GUID;
235 ObjectGuid shrineOfGelihastGUID;
236 ObjectGuid altarOfTheDeepsGUID;
237 ObjectGuid mainDoorGUID;
238 uint8 countFires;
239 uint8 deathTimes;
240 };
241};

36class go_blackfathom_altar : public GameObjectScript
37{

49class go_blackfathom_fire : public GameObjectScript

52 go_blackfathom_fire() : GameObjectScript("go_blackfathom_fire") { }
53
54 bool OnGossipHello(Player* /*player*/, GameObject* go) override
55 {
56 InstanceScript* instance = go->GetInstanceScript();
57
58 if (instance)
59 {
60 go->SetGoState(GO_STATE_ACTIVE);
61 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
62 instance->SetData(DATA_FIRE, instance->GetData(DATA_FIRE) + 1);
63 return true;
64 }
65 return false;
66 }

69class npc_blackfathom_deeps_event : public CreatureScript
70{
71public:
72 npc_blackfathom_deeps_event() : CreatureScript("npc_blackfathom_deeps_event") { }
73
74 CreatureAI* GetAI(Creature* creature) const override

76 return GetInstanceAI<npc_blackfathom_deeps_eventAI>(creature);

79 struct npc_blackfathom_deeps_eventAI : public ScriptedAI
80 {
81 npc_blackfathom_deeps_eventAI(Creature* creature) : ScriptedAI(creature)
82 {
83 Initialize();
84 if (creature->IsSummon())
85 {
86 creature->SetHomePosition(HomePosition);
87 AttackPlayer();
88 }
89
90 instance = creature->GetInstanceScript();
91 }
92
93 void Initialize()
94 {
95 Flee = false;
96
97 ravageTimer = urand(5000, 8000);
98 frostNovaTimer = urand(9000, 12000);
99 frostBoltVolleyTimer = urand(2000, 4000);
100 }
101
102 InstanceScript* instance;
103
104 uint32 ravageTimer;
105 uint32 frostNovaTimer;
106 uint32 frostBoltVolleyTimer;
107
108 bool Flee;
109
110 void Reset() override
111 {
112 Initialize();
113 }
114
115 void AttackPlayer()
116 {
117 Map::PlayerList const &PlList = me->GetMap()->GetPlayers();
118
119 if (PlList.isEmpty())
120 return;
121
122 for (Map::PlayerList::const_iterator i = PlList.begin(); i != PlList.end(); ++i)
123 {
124 if (Player* player = i->GetSource())
125 {
126 if (player->IsGameMaster())
127 continue;
128
129 if (player->IsAlive())
130 {
131 me->SetInCombatWith(player);
132 player->SetInCombatWith(me);
133 me->AddThreat(player, 0.0f);
134 }
135 }
136 }
137 }
138
139 void UpdateAI(uint32 diff) override
140 {
141 if (!UpdateVictim())
142 return;
143
144 switch (me->GetEntry())
145 {
146 case NPC_AKU_MAI_SNAPJAW:
147 {
148 if (ravageTimer <= diff)
149 {
150 DoCastVictim(SPELL_RAVAGE);
151 ravageTimer = urand(9000, 14000);
152 } else ravageTimer -= diff;
153 break;
154 }
155 case NPC_MURKSHALLOW_SOFTSHELL:
156 case NPC_BARBED_CRUSTACEAN:
157 {
158 if (!Flee && HealthBelowPct(15))
159 {
160 Flee = true;
161 me->DoFleeToGetAssistance();
162 }
163 break;
164 }
165 case NPC_AKU_MAI_SERVANT:
166 {
167 if (frostBoltVolleyTimer <= diff)
168 {
169 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
170 DoCast(target, SPELL_FROST_BOLT_VOLLEY);
171 frostBoltVolleyTimer = urand(5000, 8000);
172 }
173 else frostBoltVolleyTimer -= diff;
174
175 if (frostNovaTimer <= diff)
176 {
177 DoCastAOE(SPELL_FROST_NOVA, false);
178 frostNovaTimer = urand(25000, 30000);
179 }
180 else frostNovaTimer -= diff;
181 break;
182 }
183 }
184
185 DoMeleeAttackIfReady();
186 }
187
188 void JustDied(Unit* /*killer*/) override
189 {
190 if (me->IsSummon()) //we are not a normal spawn.
191 instance->SetData(DATA_EVENT, instance->GetData(DATA_EVENT) + 1);
192 }
193 };
194};

202class npc_morridune : public CreatureScript
203{
204public:
205 npc_morridune() : CreatureScript("npc_morridune") { }
206
207 struct npc_morriduneAI : public npc_escortAI
208 {
209 npc_morriduneAI(Creature* creature) : npc_escortAI(creature)
210 {
211 Talk(SAY_MORRIDUNE_1);
212 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
213 Start(false);
214 }
215
216 void WaypointReached(uint32 waypointId) override
217 {
218 switch (waypointId)
219 {
220 case 4:
221 SetEscortPaused(true);
222 me->SetFacingTo(1.775791f);
223 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
224 Talk(SAY_MORRIDUNE_2);
225 break;
226 }
227 }
228
229 void sGossipSelect(Player* player, uint32 /*menuId*/, uint32 /*gossipListId*/) override
230 {
231 DoCast(player, SPELL_TELEPORT_DARNASSUS);
232 }
233 };
234
235 CreatureAI* GetAI(Creature* creature) const override
236 {
237 return new npc_morriduneAI(creature);
238 }
239};

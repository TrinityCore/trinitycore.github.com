462 {
463 Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
464 if (!player)
465 return;
466
467 Creature* Fandral = player->FindNearestCreature(C_FANDRAL_STAGHELM, 100.0f);

472 if (!Fandral || !Arygos || !Caelestrasz || !Merithra)
473 return;

475 AnimationTimer = EventAnim[AnimationCount].Timer;
476 if (eventEnd == false)
477 {
478 switch (AnimationCount)
479 {
480 case 0:
481 Talk(ANACHRONOS_SAY_1, Fandral);

492 Merithra->AI()->Talk(MERITHRA_SAY_1);

496 break;
497 case 5:
498 Caelestrasz->SetTarget(Fandral->GetGUID());
499 Caelestrasz->AI()->Talk(CAELESTRASZ_SAY_1);
500 break;
501 case 6:
502 Merithra->AI()->Talk(MERITHRA_SAY_2);
503 break;
504 case 7:
505 Caelestrasz->SetTarget(ObjectGuid::Empty);
506 Merithra->GetMotionMaster()->MoveCharge(-8065, 1530, 2.61f, 10);
507 break;
508 case 8:
509 Merithra->AI()->Talk(MERITHRA_YELL_1);
510 break;
511 case 9:
512 Merithra->CastSpell(Merithra, 25105, true);
513 break;
514 case 10:

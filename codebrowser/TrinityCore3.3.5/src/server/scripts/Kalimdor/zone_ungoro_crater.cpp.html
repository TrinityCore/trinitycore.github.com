55class npc_ame : public CreatureScript
56{
57public:
58 npc_ame() : CreatureScript("npc_ame") { }
59
60 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
61 {
62 if (quest->GetQuestId() == QUEST_CHASING_AME)
63 {
64 ENSURE_AI(npc_escortAI, (creature->AI()))->Start(false, false, player->GetGUID());
65 creature->AI()->Talk(SAY_READY, player);
66 creature->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
67 // Change faction so mobs attack
68 creature->setFaction(113);
69 }
70 return true;
71 }
72
73 CreatureAI* GetAI(Creature* creature) const override
74 {
75 return new npc_ameAI(creature);
76 }
77
78 struct npc_ameAI : public npc_escortAI
79 {
80 npc_ameAI(Creature* creature) : npc_escortAI(creature)
81 {
82 Initialize();
83 }
84
85 void Initialize()
86 {
87 DemoralizingShoutTimer = 5000;
88 }
89
90 uint32 DemoralizingShoutTimer;
91
92 void WaypointReached(uint32 waypointId) override
93 {
94 if (Player* player = GetPlayerForEscort())
95 {
96 switch (waypointId)
97 {
98 case 19:
99 me->SummonCreature(ENTRY_STOMPER, -6391.69f, -1730.49f, -272.83f, 4.96f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
100 Talk(SAY_AGGRO1, player);
101 break;
102 case 28:
103 Talk(SAY_SEARCH, player);
104 break;
105 case 38:
106 me->SummonCreature(ENTRY_TARLORD, -6370.75f, -1382.84f, -270.51f, 6.06f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
107 Talk(SAY_AGGRO2, player);
108 break;
109 case 49:
110 me->SummonCreature(ENTRY_TARLORD1, -6324.44f, -1181.05f, -270.17f, 4.34f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 25000);
111 Talk(SAY_AGGRO3, player);
112 break;
113 case 55:
114 Talk(SAY_FINISH, player);
115 player->GroupEventHappens(QUEST_CHASING_AME, me);
116 break;
117 }
118 }
119 }
120
121 void Reset() override
122 {
123 Initialize();
124 }
125
126 void JustSummoned(Creature* summoned) override
127 {
128 summoned->AI()->AttackStart(me);
129 }
130
131 void JustDied(Unit* /*killer*/) override
132 {
133 if (Player* player = GetPlayerForEscort())
134 player->FailQuest(QUEST_CHASING_AME);
135 }
136
137 void UpdateAI(uint32 diff) override
138 {
139 npc_escortAI::UpdateAI(diff);
140 if (!UpdateVictim())
141 return;
142
143 if (DemoralizingShoutTimer <= diff)
144 {
145 DoCastVictim(SPELL_DEMORALIZINGSHOUT);
146 DemoralizingShoutTimer = 70000;
147 } else DemoralizingShoutTimer -= diff;
148 }
149 };

179class npc_ringo : public CreatureScript
180{
181public:
182 npc_ringo() : CreatureScript("npc_ringo") { }
183
184 bool OnQuestAccept(Player* player, Creature* creature, const Quest* quest) override
185 {
186 if (quest->GetQuestId() == QUEST_A_LITTLE_HELP)
187 {
188 if (npc_ringoAI* ringoAI = CAST_AI(npc_ringo::npc_ringoAI, creature->AI()))
189 {
190 creature->SetStandState(UNIT_STAND_STATE_STAND);
191 ringoAI->StartFollow(player, FACTION_ESCORTEE, quest);
192 }
193 }
194
195 return true;
196 }
197
198 CreatureAI* GetAI(Creature* creature) const override
199 {
200 return new npc_ringoAI(creature);
201 }
202
203 struct npc_ringoAI : public FollowerAI
204 {
205 npc_ringoAI(Creature* creature) : FollowerAI(creature)
206 {
207 Initialize();
208 }
209
210 void Initialize()
211 {
212 FaintTimer = urand(30000, 60000);
213 EndEventProgress = 0;
214 EndEventTimer = 1000;
215 SpraggleGUID.Clear();
216 }
217
218 uint32 FaintTimer;
219 uint32 EndEventProgress;
220 uint32 EndEventTimer;
221
222 ObjectGuid SpraggleGUID;
223
224 void Reset() override
225 {
226 Initialize();
227 }
228
229 void MoveInLineOfSight(Unit* who) override
230
231 {
232 FollowerAI::MoveInLineOfSight(who);
233
234 if (!me->GetVictim() && !HasFollowState(STATE_FOLLOW_COMPLETE) && who->GetEntry() == NPC_SPRAGGLE)
235 {
236 if (me->IsWithinDistInMap(who, INTERACTION_DISTANCE))
237 {
238 if (Player* player = GetLeaderForFollower())
239 {
240 if (player->GetQuestStatus(QUEST_A_LITTLE_HELP) == QUEST_STATUS_INCOMPLETE)
241 player->GroupEventHappens(QUEST_A_LITTLE_HELP, me);
242 }
243
244 SpraggleGUID = who->GetGUID();
245 SetFollowComplete(true);
246 }
247 }
248 }
249
250 void SpellHit(Unit* /*pCaster*/, const SpellInfo* pSpell) override
251 {
252 if (HasFollowState(STATE_FOLLOW_INPROGRESS | STATE_FOLLOW_PAUSED) && pSpell->Id == SPELL_REVIVE_RINGO)
253 ClearFaint();
254 }
255
256 void SetFaint()
257 {
258 if (!HasFollowState(STATE_FOLLOW_POSTEVENT))
259 {
260 SetFollowPaused(true);
261
262 Talk(SAY_FAINT);
263 }
264
265 //what does actually happen here? Emote? Aura?
266 me->SetStandState(UNIT_STAND_STATE_SLEEP);
267 }
268
269 void ClearFaint()
270 {
271 me->SetStandState(UNIT_STAND_STATE_STAND);
272
273 if (HasFollowState(STATE_FOLLOW_POSTEVENT))
274 return;
275
276 Talk(SAY_WAKE);
277
278 SetFollowPaused(false);
279 }
280
281 void UpdateFollowerAI(uint32 Diff) override
282 {
283 if (!UpdateVictim())
284 {
285 if (HasFollowState(STATE_FOLLOW_POSTEVENT))
286 {
287 if (EndEventTimer <= Diff)
288 {
289 Creature* spraggle = ObjectAccessor::GetCreature(*me, SpraggleGUID);
290 if (!spraggle || !spraggle->IsAlive())
291 {
292 SetFollowComplete();
293 return;
294 }
295
296 switch (EndEventProgress)
297 {
298 case 1:
299 Talk(SAY_RIN_END_1);
300 EndEventTimer = 3000;
301 break;
302 case 2:
303 spraggle->AI()->Talk(SAY_SPR_END_2);
304 EndEventTimer = 5000;
305 break;
306 case 3:
307 Talk(SAY_RIN_END_3);
308 EndEventTimer = 1000;
309 break;
310 case 4:
311 Talk(EMOTE_RIN_END_4);
312 SetFaint();
313 EndEventTimer = 9000;
314 break;
315 case 5:
316 Talk(EMOTE_RIN_END_5);
317 ClearFaint();
318 EndEventTimer = 1000;
319 break;
320 case 6:
321 Talk(SAY_RIN_END_6);
322 EndEventTimer = 3000;
323 break;
324 case 7:
325 spraggle->AI()->Talk(SAY_SPR_END_7);
326 EndEventTimer = 10000;
327 break;
328 case 8:
329 Talk(EMOTE_RIN_END_8);
330 EndEventTimer = 5000;
331 break;
332 case 9:
333 SetFollowComplete();
334 break;
335 }
336
337 ++EndEventProgress;
338 }
339 else
340 EndEventTimer -= Diff;
341 }
342 else if (HasFollowState(STATE_FOLLOW_INPROGRESS) && !HasFollowState(STATE_FOLLOW_PAUSED))
343 {
344 if (FaintTimer <= Diff)
345 {
346 SetFaint();
347 FaintTimer = urand(60000, 120000);
348 }
349 else
350 FaintTimer -= Diff;
351 }
352
353 return;
354 }
355
356 DoMeleeAttackIfReady();
357 }
358 };
359};

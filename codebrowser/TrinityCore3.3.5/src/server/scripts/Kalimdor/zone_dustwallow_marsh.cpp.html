53class npc_lady_jaina_proudmoore : public CreatureScript
54{
55public:
56 npc_lady_jaina_proudmoore() : CreatureScript("npc_lady_jaina_proudmoore") { }
57
58 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
59 {
60 player->PlayerTalkClass->ClearMenus();
61 if (action == GOSSIP_SENDER_INFO)
62 {
63 player->SEND_GOSSIP_MENU(7012, creature->GetGUID());
64 player->CastSpell(player, SPELL_JAINAS_AUTOGRAPH, false);
65 }
66 return true;
67 }
68
69 bool OnGossipHello(Player* player, Creature* creature) override
70 {
71 if (creature->IsQuestGiver())
72 player->PrepareQuestMenu(creature->GetGUID());
73
74 if (player->GetQuestStatus(QUEST_JAINAS_AUTOGRAPH) == QUEST_STATUS_INCOMPLETE)
75 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_JAINA, GOSSIP_SENDER_MAIN, GOSSIP_SENDER_INFO);
76
77 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
78
79 return true;
80 }

222
223 struct npc_zelfraxAI : public ScriptedAI
224 {
225 npc_zelfraxAI(Creature* creature) : ScriptedAI(creature)
226 {
227 MoveToDock();
228 }
229
230 void AttackStart(Unit* who) override
231 {
232 if (!who)
233 return;
234
235 if (me->Attack(who, true))
236 {
237 me->SetInCombatWith(who);
238 who->SetInCombatWith(me);
239
240 if (IsCombatMovementAllowed())
241 me->GetMotionMaster()->MoveChase(who);
242 }
243 }
244
245 void MovementInform(uint32 Type, uint32 /*Id*/) override
246 {
247 if (Type != POINT_MOTION_TYPE)
248 return;
249
250 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
251 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
252 SetCombatMovement(true);
253
254 if (me->IsInCombat())
255 if (Unit* unit = me->GetVictim())
256 me->GetMotionMaster()->MoveChase(unit);
257 }
258
259 void MoveToDock()
260 {
261 SetCombatMovement(false);
262 me->GetMotionMaster()->MovePoint(0, MovePosition);
263 Talk(SAY_ZELFRAX1);
264 Talk(SAY_ZELFRAX2);
265 }
266
267 void UpdateAI(uint32 /*Diff*/) override
268 {
269 if (!UpdateVictim())
270 return;
271
272 DoMeleeAttackIfReady();
273 }
274 };

291
292 class spell_ooze_zap_SpellScript : public SpellScript
293 {
294 PrepareSpellScript(spell_ooze_zap_SpellScript);
295
296 bool Validate(SpellInfo const* /*spellInfo*/) override
297 {
298 if (!sSpellMgr->GetSpellInfo(SPELL_OOZE_ZAP))
299 return false;
300 return true;
301 }
302
303 SpellCastResult CheckRequirement()
304 {
305 if (!GetCaster()->HasAura(GetSpellInfo()->Effects[EFFECT_1].CalcValue()))
306 return SPELL_FAILED_CANT_DO_THAT_RIGHT_NOW; // This is actually correct
307
308 if (!GetExplTargetUnit())
309 return SPELL_FAILED_BAD_TARGETS;
310
311 return SPELL_CAST_OK;
312 }
313
314 void HandleDummy(SpellEffIndex effIndex)
315 {
316 PreventHitDefaultEffect(effIndex);
317 if (GetHitUnit())
318 GetCaster()->CastSpell(GetHitUnit(), uint32(GetEffectValue()), true);
319 }
320
321 void Register() override
322 {
323 OnEffectHitTarget += SpellEffectFn(spell_ooze_zap_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
324 OnCheckCast += SpellCheckCastFn(spell_ooze_zap_SpellScript::CheckRequirement);
325 }
326 };
327
328 SpellScript* GetSpellScript() const override
329 {
330 return new spell_ooze_zap_SpellScript();

370class spell_energize_aoe : public SpellScriptLoader
371{
372 public:
373 spell_energize_aoe() : SpellScriptLoader("spell_energize_aoe") { }
374
375 class spell_energize_aoe_SpellScript : public SpellScript
376 {
377 PrepareSpellScript(spell_energize_aoe_SpellScript);
378
379 bool Validate(SpellInfo const* /*spellInfo*/) override
380 {
381 if (!sSpellMgr->GetSpellInfo(SPELL_ENERGIZED))
382 return false;
383 return true;
384 }
385
386 void FilterTargets(std::list<WorldObject*>& targets)
387 {
388 for (std::list<WorldObject*>::iterator itr = targets.begin(); itr != targets.end();)
389 {
390 if ((*itr)->GetTypeId() == TYPEID_PLAYER && (*itr)->ToPlayer()->GetQuestStatus(GetSpellInfo()->Effects[EFFECT_1].CalcValue()) == QUEST_STATUS_INCOMPLETE)
391 ++itr;
392 else
393 targets.erase(itr++);
394 }
395 targets.push_back(GetCaster());
396 }
397
398 void HandleScript(SpellEffIndex effIndex)
399 {
400 PreventHitDefaultEffect(effIndex);
401 GetCaster()->CastSpell(GetCaster(), uint32(GetEffectValue()), true);
402 }
403
404 void Register() override
405 {
406 OnEffectHitTarget += SpellEffectFn(spell_energize_aoe_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
407 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_energize_aoe_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENTRY);
408 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_energize_aoe_SpellScript::FilterTargets, EFFECT_1, TARGET_UNIT_SRC_AREA_ENTRY);
409 }
410 };
411
412 SpellScript* GetSpellScript() const override
413 {
414 return new spell_energize_aoe_SpellScript();

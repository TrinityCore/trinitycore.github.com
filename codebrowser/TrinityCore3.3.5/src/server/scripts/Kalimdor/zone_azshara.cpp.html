54class npc_spitelashes : public CreatureScript
55{
56public:
57 npc_spitelashes() : CreatureScript("npc_spitelashes") { }
58
59 struct npc_spitelashesAI : public ScriptedAI
60 {
61 npc_spitelashesAI(Creature* creature) : ScriptedAI(creature)
62 {
63 Initialize();
64 }
65
66 void Initialize()
67 {
68 morphtimer = 0;
69 spellhit = false;
70 }
71
72 uint32 morphtimer;
73 bool spellhit;
74
75 void Reset() override
76 {
77 Initialize();
78 }
79
80 void EnterCombat(Unit* /*who*/) override { }
81
82 void SpellHit(Unit* unit, const SpellInfo* spell) override
83 {
84 if (spellhit)
85 return;
86
87 switch (spell->Id)
88 {
89 case SPELL_POLYMORPH_RANK1:
90 case SPELL_POLYMORPH_RANK2:
91 case SPELL_POLYMORPH_RANK3:
92 case SPELL_POLYMORPH_RANK4:
93 if (Player* player = unit->ToPlayer())
94 if (player->GetQuestStatus(9364) == QUEST_STATUS_INCOMPLETE)
95 {
96 spellhit = true;
97 DoCast(me, SPELL_POLYMORPH);
98 }
99 break;
100 default:
101 break;
102 }
103 }
104
105 void UpdateAI(uint32 diff) override
106 {
107 // we mustn't remove the Creature in the same round in which we cast the summon spell, otherwise there will be no summons
108 if (spellhit && morphtimer >= 5000)
109 {
110 me->DespawnOrUnsummon();
111 return;
112 }
113 // walk 5 seconds before summoning
114 if (spellhit && morphtimer<5000)
115 {
116 morphtimer+=diff;
117 if (morphtimer >= 5000)
118 {
119 DoCast(me, SPELL_POLYMORPH_BACKFIRE); // summon copies
120 DoCast(me, SPELL_REMOVE_POLYMORPH); // visual explosion
121 }
122 }
123 if (!UpdateVictim())
124 return;
125
126 /// @todo add abilities for the different creatures
127 DoMeleeAttackIfReady();
128 }
129 };
130
131 CreatureAI* GetAI(Creature* creature) const override
132 {
133 return new npc_spitelashesAI(creature);
134 }

296class npc_rizzle_sprysprocket : public CreatureScript
297{
298public:
299 npc_rizzle_sprysprocket() : CreatureScript("npc_rizzle_sprysprocket") { }
300
301 struct npc_rizzle_sprysprocketAI : public ScriptedAI
302 {
303 npc_rizzle_sprysprocketAI(Creature* creature) : ScriptedAI(creature)
304 {
305 Initialize();
306 }
307
308 void Initialize()
309 {
310 SpellEscapeTimer = 1300;
311 TeleportTimer = 3500;
312 CheckTimer = 10000;
313 GrenadeTimer = 30000;
314 MustDieTimer = 3000;
315 CurrWP = 0;
316
317 PlayerGUID.Clear();
318
319 MustDie = false;
320 Escape = false;
321 ContinueWP = false;
322 Reached = false;
323 }
324
325 void Reset() override
326 {
327 Initialize();
328 }
329
330 void EnterCombat(Unit* /*who*/) override { }
331
332 void AttackStart(Unit* who) override
333 {
334 if (!who || PlayerGUID)
335 return;
336
337 Player* player = who->ToPlayer();
338
339 if (player && player->GetQuestStatus(QUEST_CHASING_THE_MOONSTONE) == QUEST_STATUS_INCOMPLETE)
340 {
341 PlayerGUID = who->GetGUID();
342 Talk(SAY_RIZZLE_START);
343 DoCast(who, SPELL_RIZZLE_BLACKJACK, false);
344 return;
345 }
346 }
347
348 void sGossipSelect(Player* player, uint32 /*menuId*/, uint32 /*gossipListId*/) override
349 {
350 player->CLOSE_GOSSIP_MENU();
351 me->CastSpell(player, SPELL_GIVE_SOUTHFURY_MOONSTONE, true);
352 MustDieTimer = 3000;
353 MustDie = true;
354 }
355
356 void MovementInform(uint32 type, uint32 id) override
357 {
358 if (type != POINT_MOTION_TYPE)
359 return;
360
361 if (id == 57)
362 {
363 me->DespawnOrUnsummon();
364 return;
365 }
366
367 ++CurrWP;
368 ContinueWP = true;
369 }
370
371 void UpdateAI(uint32 diff) override
372 {
373 if (MustDie)
374 {
375 if (MustDieTimer <= diff)
376 {
377 me->DespawnOrUnsummon();
378 return;
379 } else MustDieTimer -= diff;
380 }
381
382 if (!Escape)
383 {
384 if (!PlayerGUID)
385 return;
386
387 if (SpellEscapeTimer <= diff)
388 {
389 DoCast(me, SPELL_RIZZLE_ESCAPE, false);
390 SpellEscapeTimer = 10000;
391 } else SpellEscapeTimer -= diff;
392
393 if (TeleportTimer <= diff)
394 {
395 // temp solution - unit can't be teleported by core using spelleffect 5, only players
396 DoTeleportTo(3706.39f, -3969.15f, 35.9118f);
397
398 //begin swimming and summon depth charges
399 Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
400 if (!player)
401 return;
402
403 Talk(MSG_ESCAPE_NOTICE, player);
404 DoCast(me, SPELL_PERIODIC_DEPTH_CHARGE);
405 me->SetHover(true);
406 me->SetSwim(true);
407 me->SetSpeed(MOVE_RUN, 0.85f, true);
408 me->GetMotionMaster()->MovementExpired();
409 me->GetMotionMaster()->MovePoint(CurrWP, WPs[CurrWP]);
410 Escape = true;
411 } else TeleportTimer -= diff;
412
413 return;
414 }
415
416 if (ContinueWP)
417 {
418 me->GetMotionMaster()->MovePoint(CurrWP, WPs[CurrWP]);
419 ContinueWP = false;
420 }
421
422 if (GrenadeTimer <= diff)
423 {
424 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
425 {
426 Talk(SAY_RIZZLE_GRENADE, player);
427 DoCast(player, SPELL_RIZZLE_FROST_GRENADE, true);
428 }
429 GrenadeTimer = 30000;
430 } else GrenadeTimer -= diff;
431
432 if (CheckTimer <= diff)
433 {
434 Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
435 if (!player)
436 {
437 me->DespawnOrUnsummon();
438 return;
439 }
440
441 if (me->IsWithinDist(player, 10) && me->GetPositionX() > player->GetPositionX() && !Reached)
442 {
443 Talk(SAY_RIZZLE_FINAL);
444 me->SetUInt32Value(UNIT_NPC_FLAGS, 1);
445 me->setFaction(35);
446 me->GetMotionMaster()->MoveIdle();
447 me->RemoveAurasDueToSpell(SPELL_PERIODIC_DEPTH_CHARGE);
448 Reached = true;
449 }
450
451 CheckTimer = 1000;
452 } else CheckTimer -= diff;
453 }
454
455 private:
456 ObjectGuid PlayerGUID;
457 uint32 SpellEscapeTimer;
458 uint32 TeleportTimer;
459 uint32 CheckTimer;
460 uint32 GrenadeTimer;
461 uint32 MustDieTimer;
462 uint32 CurrWP;
463 bool MustDie;
464 bool Escape;
465 bool ContinueWP;
466 bool Reached;
467 };

475 return true;
476 }

480 return new npc_rizzle_sprysprocketAI(creature);
481 }

39#define GOSSIP_ITEM_BEACON "Please make me a Cenarion Beacon"

41enum RiverbreezeAndSilversky
42{
43 SPELL_CENARION_BEACON = 15120,
44
45 NPC_ARATHANDRIS_SILVERSKY = 9528,
46 NPC_MAYBESS_RIVERBREEZE = 9529,
47
48 QUEST_CLEASING_FELWOOD_A = 4101,
49 QUEST_CLEASING_FELWOOD_H = 4102
50};
51
52class npcs_riverbreeze_and_silversky : public CreatureScript
53{
54public:
55 npcs_riverbreeze_and_silversky() : CreatureScript("npcs_riverbreeze_and_silversky") { }
56
57 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
58 {
59 player->PlayerTalkClass->ClearMenus();
60 if (action == GOSSIP_ACTION_INFO_DEF+1)
61 {
62 player->CLOSE_GOSSIP_MENU();
63 creature->CastSpell(player, SPELL_CENARION_BEACON, false);
64 }
65 return true;
66 }
67
68 bool OnGossipHello(Player* player, Creature* creature) override
69 {
70 if (creature->IsQuestGiver())
71 player->PrepareQuestMenu(creature->GetGUID());
72
73 uint32 creatureId = creature->GetEntry();
74
75 if (creatureId == NPC_ARATHANDRIS_SILVERSKY)
76 {
77 if (player->GetQuestRewardStatus(QUEST_CLEASING_FELWOOD_A))
78 {
79 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_BEACON, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
80 player->SEND_GOSSIP_MENU(2848, creature->GetGUID());
81 } else if (player->GetTeam() == HORDE)
82 player->SEND_GOSSIP_MENU(2845, creature->GetGUID());
83 else
84 player->SEND_GOSSIP_MENU(2844, creature->GetGUID());
85 }
86
87 if (creatureId == NPC_MAYBESS_RIVERBREEZE)
88 {
89 if (player->GetQuestRewardStatus(QUEST_CLEASING_FELWOOD_H))
90 {
91 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_BEACON, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
92 player->SEND_GOSSIP_MENU(2849, creature->GetGUID());
93 } else if (player->GetTeam() == ALLIANCE)
94 player->SEND_GOSSIP_MENU(2843, creature->GetGUID());
95 else
96 player->SEND_GOSSIP_MENU(2842, creature->GetGUID());
97 }
98
99 return true;
100 }
101};

107enum AncientMisc
108{
109 QUEST_ANCIENT_LEAF = 7632,
110 NPC_VARTRUS = 14524,
111 NPC_STOMA = 14525,
112 NPC_HASTAT = 14526,
113 CREATURE_GROUP_ANCIENTS = 1
114};
115
116class at_ancient_leaf : public AreaTriggerScript
117{
118 public:
119 at_ancient_leaf() : AreaTriggerScript("at_ancient_leaf") { }
120
121 bool OnTrigger(Player* player, AreaTriggerEntry const* /*trigger*/) override
122 {
123 if (player->IsGameMaster() || !player->IsAlive())
124 return false;
125
126 // Handle Call Ancients event start - The area trigger summons 3 ancients
127 if ((player->GetQuestStatus(QUEST_ANCIENT_LEAF) == QUEST_STATUS_COMPLETE) || (player->GetQuestStatus(QUEST_ANCIENT_LEAF) == QUEST_STATUS_REWARDED))
128 {
129 // If ancients are already spawned, skip the rest
130 if (GetClosestCreatureWithEntry(player, NPC_VARTRUS, 50.0f) || GetClosestCreatureWithEntry(player, NPC_STOMA, 50.0f) || GetClosestCreatureWithEntry(player, NPC_HASTAT, 50.0f))
131 return true;
132
133 player->GetMap()->SummonCreatureGroup(CREATURE_GROUP_ANCIENTS);
134 }
135 return false;
136 }
137};

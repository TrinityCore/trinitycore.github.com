34enum EpochHunter
35{
36 SAY_ENTER = 0,
37 SAY_AGGRO = 1,
38 SAY_SLAY = 2,
39 SAY_BREATH = 3,
40 SAY_DEATH = 4,
41
42 SPELL_SAND_BREATH = 31914,
43 SPELL_IMPENDING_DEATH = 31916,
44 SPELL_MAGIC_DISRUPTION_AURA = 33834,
45 SPELL_WING_BUFFET = 31475
46};
47
48class boss_epoch_hunter : public CreatureScript
49{
50public:
51 boss_epoch_hunter() : CreatureScript("boss_epoch_hunter") { }
52
53 CreatureAI* GetAI(Creature* creature) const override
54 {
55 return GetInstanceAI<boss_epoch_hunterAI>(creature);
56 }
57
58 struct boss_epoch_hunterAI : public ScriptedAI
59 {
60 boss_epoch_hunterAI(Creature* creature) : ScriptedAI(creature)
61 {
62 Initialize();
63 instance = creature->GetInstanceScript();
64 }
65
66 void Initialize()
67 {
68 SandBreath_Timer = urand(8000, 16000);
69 ImpendingDeath_Timer = urand(25000, 30000);
70 WingBuffet_Timer = 35000;
71 Mda_Timer = 40000;
72 }
73
74 InstanceScript* instance;
75
76 uint32 SandBreath_Timer;
77 uint32 ImpendingDeath_Timer;
78 uint32 WingBuffet_Timer;
79 uint32 Mda_Timer;
80
81 void Reset() override
82 {
83 Initialize();
84 }
85
86 void EnterCombat(Unit* /*who*/) override
87 {
88 Talk(SAY_AGGRO);
89 }
90
91 void KilledUnit(Unit* /*victim*/) override
92 {
93 Talk(SAY_SLAY);
94 }
95
96 void JustDied(Unit* /*killer*/) override
97 {
98 Talk(SAY_DEATH);
99
100 if (instance->GetData(TYPE_THRALL_EVENT) == IN_PROGRESS)
101 instance->SetData(TYPE_THRALL_PART4, DONE);
102 }
103
104 void UpdateAI(uint32 diff) override
105 {
106 //Return since we have no target
107 if (!UpdateVictim())
108 return;
109
110 //Sand Breath
111 if (SandBreath_Timer <= diff)
112 {
113 if (me->IsNonMeleeSpellCast(false))
114 me->InterruptNonMeleeSpells(false);
115
116 DoCastVictim(SPELL_SAND_BREATH);
117
118 Talk(SAY_BREATH);
119
120 SandBreath_Timer = urand(10000, 20000);
121 } else SandBreath_Timer -= diff;
122
123 if (ImpendingDeath_Timer <= diff)
124 {
125 DoCastVictim(SPELL_IMPENDING_DEATH);
126 ImpendingDeath_Timer = 25000 + rand32() % 5000;
127 } else ImpendingDeath_Timer -= diff;
128
129 if (WingBuffet_Timer <= diff)
130 {
131 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
132 DoCast(target, SPELL_WING_BUFFET);
133 WingBuffet_Timer = 25000 + rand32() % 10000;
134 } else WingBuffet_Timer -= diff;
135
136 if (Mda_Timer <= diff)
137 {
138 DoCast(me, SPELL_MAGIC_DISRUPTION_AURA);
139 Mda_Timer = 15000;
140 } else Mda_Timer -= diff;
141
142 DoMeleeAttackIfReady();
143 }
144 };
145
146};
147
148void AddSC_boss_epoch_hunter()
149{
150 new boss_epoch_hunter();
151}

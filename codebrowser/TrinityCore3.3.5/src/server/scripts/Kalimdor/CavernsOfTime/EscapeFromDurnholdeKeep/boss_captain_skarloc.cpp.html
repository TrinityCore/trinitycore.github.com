50class boss_captain_skarloc : public CreatureScript
51{
52public:
53 boss_captain_skarloc() : CreatureScript("boss_captain_skarloc") { }
54
55 CreatureAI* GetAI(Creature* creature) const override
56 {
57 return GetInstanceAI<boss_captain_skarlocAI>(creature);
58 }
59
60 struct boss_captain_skarlocAI : public ScriptedAI
61 {
62 boss_captain_skarlocAI(Creature* creature) : ScriptedAI(creature)
63 {
64 Initialize();
65 instance = creature->GetInstanceScript();
66 }
67
68 void Initialize()
69 {
70 Holy_Light_Timer = urand(20000, 30000);
71 Cleanse_Timer = 10000;
72 HammerOfJustice_Timer = urand(20000, 35000);
73 HolyShield_Timer = 240000;
74 DevotionAura_Timer = 3000;
75 Consecration_Timer = 8000;
76 }
77
78 InstanceScript* instance;
79
80 uint32 Holy_Light_Timer;
81 uint32 Cleanse_Timer;
82 uint32 HammerOfJustice_Timer;
83 uint32 HolyShield_Timer;
84 uint32 DevotionAura_Timer;
85 uint32 Consecration_Timer;
86
87 void Reset() override
88 {
89 Initialize();
90 }
91
92 void EnterCombat(Unit* /*who*/) override
93 {
94 //This is not correct. Should taunt Thrall before engage in combat
95 Talk(SAY_TAUNT1);
96 Talk(SAY_TAUNT2);
97 }
98
99 void KilledUnit(Unit* /*victim*/) override
100 {
101 Talk(SAY_SLAY);
102 }
103
104 void JustDied(Unit* /*killer*/) override
105 {
106 Talk(SAY_DEATH);
107
108 if (instance->GetData(TYPE_THRALL_EVENT) == IN_PROGRESS)
109 instance->SetData(TYPE_THRALL_PART1, DONE);
110 }
111
112 void UpdateAI(uint32 diff) override
113 {
114 //Return since we have no target
115 if (!UpdateVictim())
116 return;
117
118 //Holy_Light
119 if (Holy_Light_Timer <= diff)
120 {
121 DoCast(me, SPELL_HOLY_LIGHT);
122 Holy_Light_Timer = 30000;
123 } else Holy_Light_Timer -= diff;
124
125 //Cleanse
126 if (Cleanse_Timer <= diff)
127 {
128 DoCast(me, SPELL_CLEANSE);
129 Cleanse_Timer = 10000;
130 } else Cleanse_Timer -= diff;
131
132 //Hammer of Justice
133 if (HammerOfJustice_Timer <= diff)
134 {
135 DoCastVictim(SPELL_HAMMER_OF_JUSTICE);
136 HammerOfJustice_Timer = 60000;
137 } else HammerOfJustice_Timer -= diff;
138
139 //Holy Shield
140 if (HolyShield_Timer <= diff)
141 {
142 DoCast(me, SPELL_HOLY_SHIELD);
143 HolyShield_Timer = 240000;
144 } else HolyShield_Timer -= diff;
145
146 //Devotion_Aura
147 if (DevotionAura_Timer <= diff)
148 {
149 DoCast(me, SPELL_DEVOTION_AURA);
150 DevotionAura_Timer = urand(45000, 55000);
151 } else DevotionAura_Timer -= diff;
152
153 //Consecration
154 if (Consecration_Timer <= diff)
155 {
156 //DoCastVictim(SPELL_CONSECRATION);
157 Consecration_Timer = urand(5000, 10000);
158 } else Consecration_Timer -= diff;
159
160 DoMeleeAttackIfReady();
161 }
162 };
163
164};

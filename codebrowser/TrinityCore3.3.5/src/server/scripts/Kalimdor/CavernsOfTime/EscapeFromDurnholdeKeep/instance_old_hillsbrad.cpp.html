32#define MAX_ENCOUNTER 6

34#define THRALL_ENTRY 17876
35#define TARETHA_ENTRY 18887
36#define EPOCH_ENTRY 18096
37
38#define DRAKE_ENTRY 17848
39
40#define QUEST_ENTRY_DIVERSION 10283
41#define LODGE_QUEST_TRIGGER 20155
42
43class instance_old_hillsbrad : public InstanceMapScript
44{
45public:
46 instance_old_hillsbrad() : InstanceMapScript("instance_old_hillsbrad", 560) { }
47
48 InstanceScript* GetInstanceScript(InstanceMap* map) const override
49 {
50 return new instance_old_hillsbrad_InstanceMapScript(map);
51 }
52
53 struct instance_old_hillsbrad_InstanceMapScript : public InstanceScript
54 {
55 instance_old_hillsbrad_InstanceMapScript(Map* map) : InstanceScript(map)
56 {
57 SetHeaders(DataHeader);
58 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
59
60 mBarrelCount = 0;
61 mThrallEventCount = 0;
62 }
63
64 uint32 m_auiEncounter[MAX_ENCOUNTER];
65 uint32 mBarrelCount;
66 uint32 mThrallEventCount;
67
68 ObjectGuid ThrallGUID;
69 ObjectGuid TarethaGUID;
70 ObjectGuid EpochGUID;
71
72 Player* GetPlayerInMap()
73 {
74 Map::PlayerList const& players = instance->GetPlayers();
75
76 if (!players.isEmpty())
77 {
78 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
79 {
80 if (Player* player = itr->GetSource())
81 return player;
82 }
83 }
84
85 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: GetPlayerInMap, but PlayerList is empty!");
86 return NULL;
87 }
88
89 void UpdateQuestCredit()
90 {
91 Map::PlayerList const& players = instance->GetPlayers();
92
93 if (!players.isEmpty())
94 {
95 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
96 {
97 if (Player* player = itr->GetSource())
98 player->KilledMonsterCredit(LODGE_QUEST_TRIGGER);
99 }
100 }
101 }
102
103 void OnCreatureCreate(Creature* creature) override
104 {
105 switch (creature->GetEntry())
106 {
107 case THRALL_ENTRY:
108 ThrallGUID = creature->GetGUID();
109 break;
110 case TARETHA_ENTRY:
111 TarethaGUID = creature->GetGUID();
112 break;
113 case EPOCH_ENTRY:
114 EpochGUID = creature->GetGUID();
115 break;
116 }
117 }
118
119 void SetData(uint32 type, uint32 data) override
120 {
121 Player* player = GetPlayerInMap();
122
123 if (!player)
124 {
125 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: SetData (Type: %u Data %u) cannot find any player.", type, data);
126 return;
127 }
128
129 switch (type)
130 {
131 case TYPE_BARREL_DIVERSION:
132 {
133 if (data == IN_PROGRESS)
134 {
135 if (mBarrelCount >= 5)
136 return;
137
138 ++mBarrelCount;
139 DoUpdateWorldState(WORLD_STATE_OH, mBarrelCount);
140
141 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: go_barrel_old_hillsbrad count %u", mBarrelCount);
142
143 m_auiEncounter[0] = IN_PROGRESS;
144
145 if (mBarrelCount == 5)
146 {
147 UpdateQuestCredit();
148 player->SummonCreature(DRAKE_ENTRY, 2128.43f, 71.01f, 64.42f, 1.74f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 1800000);
149 m_auiEncounter[0] = DONE;
150 }
151 }
152 break;
153 }
154 case TYPE_THRALL_EVENT:
155 {
156 if (data == FAIL)
157 {
158 if (mThrallEventCount <= 20)
159 {
160 ++mThrallEventCount;
161 m_auiEncounter[1] = NOT_STARTED;
162 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: Thrall event failed %u times. Resetting all sub-events.", mThrallEventCount);
163 m_auiEncounter[2] = NOT_STARTED;
164 m_auiEncounter[3] = NOT_STARTED;
165 m_auiEncounter[4] = NOT_STARTED;
166 m_auiEncounter[5] = NOT_STARTED;
167 }
168 else if (mThrallEventCount > 20)
169 {
170 m_auiEncounter[1] = data;
171 m_auiEncounter[2] = data;
172 m_auiEncounter[3] = data;
173 m_auiEncounter[4] = data;
174 m_auiEncounter[5] = data;
175 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: Thrall event failed %u times. Resetting all sub-events.", mThrallEventCount);
176 }
177 }
178 else
179 m_auiEncounter[1] = data;
180 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: Thrall escort event adjusted to data %u.", data);
181 break;
182 }
183 case TYPE_THRALL_PART1:
184 m_auiEncounter[2] = data;
185 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: Thrall event part I adjusted to data %u.", data);
186 break;
187 case TYPE_THRALL_PART2:
188 m_auiEncounter[3] = data;
189 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: Thrall event part II adjusted to data %u.", data);
190 break;
191 case TYPE_THRALL_PART3:
192 m_auiEncounter[4] = data;
193 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: Thrall event part III adjusted to data %u.", data);
194 break;
195 case TYPE_THRALL_PART4:
196 m_auiEncounter[5] = data;
197 TC_LOG_DEBUG("scripts", "Instance Old Hillsbrad: Thrall event part IV adjusted to data %u.", data);
198 break;
199 }
200 }
201
202 uint32 GetData(uint32 data) const override
203 {
204 switch (data)
205 {
206 case TYPE_BARREL_DIVERSION:
207 return m_auiEncounter[0];
208 case TYPE_THRALL_EVENT:
209 return m_auiEncounter[1];
210 case TYPE_THRALL_PART1:
211 return m_auiEncounter[2];
212 case TYPE_THRALL_PART2:
213 return m_auiEncounter[3];
214 case TYPE_THRALL_PART3:
215 return m_auiEncounter[4];
216 case TYPE_THRALL_PART4:
217 return m_auiEncounter[5];
218 }
219 return 0;
220 }
221
222 ObjectGuid GetGuidData(uint32 data) const override
223 {
224 switch (data)
225 {
226 case DATA_THRALL:
227 return ThrallGUID;
228 case DATA_TARETHA:
229 return TarethaGUID;
230 case DATA_EPOCH:
231 return EpochGUID;
232 }
233 return ObjectGuid::Empty;
234 }
235 };
236
237};

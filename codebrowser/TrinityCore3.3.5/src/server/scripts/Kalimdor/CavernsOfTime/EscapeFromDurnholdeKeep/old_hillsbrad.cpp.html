208

210 pEscortAI->Start(true, true, player->GetGUID());

212 ENSURE_AI(npc_escortAI, (creature->AI()))->SetMaxPlayerDistance(100.0f);//not really needed, because it will not despawn if player is too far

214 ENSURE_AI(npc_escortAI, (creature->AI()))->SetDespawnAtFar(false);

217 case GOSSIP_ACTION_INFO_DEF+2:

219 player->SEND_GOSSIP_MENU(GOSSIP_ID_SKARLOC2, creature->GetGUID());
220 break;
221
222 case GOSSIP_ACTION_INFO_DEF+20:
223 player->SEND_GOSSIP_MENU(GOSSIP_ID_SKARLOC3, creature->GetGUID());
224 creature->SummonCreature(SKARLOC_MOUNT, 2038.81f, 270.26f, 63.20f, 5.41f, TEMPSUMMON_TIMED_DESPAWN, 12000);

228 creature->AI()->Talk(SAY_TH_START_EVENT_PART2);

230 ENSURE_AI(npc_thrall_old_hillsbrad::npc_thrall_old_hillsbradAI, creature->AI())->StartWP();

234 player->CLOSE_GOSSIP_MENU();

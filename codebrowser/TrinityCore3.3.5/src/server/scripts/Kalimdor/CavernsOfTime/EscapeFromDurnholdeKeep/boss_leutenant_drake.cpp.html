97class boss_lieutenant_drake : public CreatureScript
98{
99public:
100 boss_lieutenant_drake() : CreatureScript("boss_lieutenant_drake") { }
101
102 CreatureAI* GetAI(Creature* creature) const override
103 {
104 return new boss_lieutenant_drakeAI(creature);
105 }
106
107 struct boss_lieutenant_drakeAI : public ScriptedAI
108 {
109 boss_lieutenant_drakeAI(Creature* creature) : ScriptedAI(creature)
110 {
111 Initialize();
112 }
113
114 void Initialize()
115 {
116 CanPatrol = true;
117 wpId = 0;
118
119 Whirlwind_Timer = 20000;
120 Fear_Timer = 30000;
121 MortalStrike_Timer = 45000;
122 ExplodingShout_Timer = 25000;
123 }
124
125 bool CanPatrol;
126 uint32 wpId;
127
128 uint32 Whirlwind_Timer;
129 uint32 Fear_Timer;
130 uint32 MortalStrike_Timer;
131 uint32 ExplodingShout_Timer;
132
133 void Reset() override
134 {
135 Initialize();
136 }
137
138 void EnterCombat(Unit* /*who*/) override
139 {
140 Talk(SAY_AGGRO);
141 }
142
143 void KilledUnit(Unit* /*victim*/) override
144 {
145 Talk(SAY_SLAY);
146 }
147
148 void JustDied(Unit* /*killer*/) override
149 {
150 Talk(SAY_DEATH);
151 }
152
153 void UpdateAI(uint32 diff) override
154 {
155 /// @todo make this work
156 if (CanPatrol && wpId == 0)
157 {
158 me->GetMotionMaster()->MovePoint(wpId, DrakeWP[wpId].x, DrakeWP[wpId].y, DrakeWP[wpId].z);
159 ++wpId;
160 }
161
162 //Return since we have no target
163 if (!UpdateVictim())
164 return;
165
166 //Whirlwind
167 if (Whirlwind_Timer <= diff)
168 {
169 DoCastVictim(SPELL_WHIRLWIND);
170 Whirlwind_Timer = 20000 + rand32() % 5000;
171 } else Whirlwind_Timer -= diff;
172
173 //Fear
174 if (Fear_Timer <= diff)
175 {
176 Talk(SAY_SHOUT);
177 DoCastVictim(SPELL_FRIGHTENING_SHOUT);
178 Fear_Timer = 25000 + rand32() % 10000;
179 } else Fear_Timer -= diff;
180
181 //Mortal Strike
182 if (MortalStrike_Timer <= diff)
183 {
184 Talk(SAY_MORTAL);
185 DoCastVictim(SPELL_MORTAL_STRIKE);
186 MortalStrike_Timer = 20000 + rand32() % 10000;
187 } else MortalStrike_Timer -= diff;
188
189 DoMeleeAttackIfReady();
190 }
191 };
192
193};

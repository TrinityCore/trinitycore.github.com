166 void SetData(uint32 type, uint32 data) override
167 {
168 switch (type)
169 {
170 case TYPE_MEDIVH:
171 if (data == SPECIAL && m_auiEncounter[0] == IN_PROGRESS)
172 {
173 --mShieldPercent;
174
175 DoUpdateWorldState(WORLD_STATE_BM_SHIELD, mShieldPercent);
176
177 if (!mShieldPercent)
178 {
179 if (Creature* medivh = instance->GetCreature(_medivhGUID))
180 {
181 if (medivh->IsAlive())
182 {
183 medivh->DealDamage(medivh, medivh->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
184 m_auiEncounter[0] = FAIL;
185 m_auiEncounter[1] = NOT_STARTED;
186 }
187 }
188 }
189 }
190 else
191 {
192 if (data == IN_PROGRESS)
193 {
194 TC_LOG_DEBUG("scripts", "Instance The Black Morass: Starting event.");
195 InitWorldState();
196 m_auiEncounter[1] = IN_PROGRESS;
197 ScheduleEventNextPortal(15000);
198 }
199
200 if (data == DONE)
201 {
202 //this may be completed further out in the post-event
203 TC_LOG_DEBUG("scripts", "Instance The Black Morass: Event completed.");
204 Map::PlayerList const& players = instance->GetPlayers();
205
206 if (!players.isEmpty())
207 {
208 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
209 {
210 if (Player* player = itr->GetSource())
211 {
212 if (player->GetQuestStatus(QUEST_OPENING_PORTAL) == QUEST_STATUS_INCOMPLETE)
213 player->AreaExploredOrEventHappens(QUEST_OPENING_PORTAL);
214
215 if (player->GetQuestStatus(QUEST_MASTER_TOUCH) == QUEST_STATUS_INCOMPLETE)
216 player->AreaExploredOrEventHappens(QUEST_MASTER_TOUCH);
217 }
218 }
219 }
220 }
221
222 m_auiEncounter[0] = data;
223 }
224 break;
225 case TYPE_RIFT:
226 if (data == SPECIAL)
227 {
228 if (mRiftPortalCount < 7)
229 ScheduleEventNextPortal(5000);
230 }
231 else
232 m_auiEncounter[1] = data;
233 break;
234 }
235 }

261 Creature* SummonedPortalBoss(Creature* me)
262 {
263 uint32 entry = RiftWaves[GetRiftWaveId()].PortalBoss;
264
265 if (entry == RIFT_BOSS)
266 entry = RandRiftBoss();
267
268 TC_LOG_DEBUG("scripts", "Instance The Black Morass: Summoning rift boss entry %u.", entry);
269
270 Position pos = me->GetRandomNearPosition(10.0f);
271
272 //normalize Z-level if we can, if rift is not at ground level.
273 pos.m_positionZ = std::max(me->GetMap()->GetHeight(pos.m_positionX, pos.m_positionY, MAX_HEIGHT), me->GetMap()->GetWaterLevel(pos.m_positionX, pos.m_positionY));
274
275 if (Creature* summon = me->SummonCreature(entry, pos, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000))
276 return summon;
277
278 TC_LOG_DEBUG("scripts", "Instance The Black Morass: What just happened there? No boss, no loot, no fun...");
279 return NULL;
280 }

282 void DoSpawnPortal()
283 {
284 if (Creature* medivh = instance->GetCreature(_medivhGUID))
285 {
286 uint8 tmp = urand(0, 2);
287
288 if (tmp >= _currentRiftId)
289 ++tmp;
290
291 TC_LOG_DEBUG("scripts", "Instance The Black Morass: Creating Time Rift at locationId %i (old locationId was %u).", tmp, _currentRiftId);
292
293 _currentRiftId = tmp;
294
295 Creature* temp = medivh->SummonCreature(NPC_TIME_RIFT,
296 PortalLocation[tmp][0], PortalLocation[tmp][1], PortalLocation[tmp][2], PortalLocation[tmp][3],
297 TEMPSUMMON_CORPSE_DESPAWN, 0);
298 if (temp)
299 {
300 temp->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
301 temp->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
302
303 if (Creature* boss = SummonedPortalBoss(temp))
304 {
305 if (boss->GetEntry() == NPC_AEONUS)
306 boss->AddThreat(medivh, 0.0f);
307 else
308 {
309 boss->AddThreat(temp, 0.0f);
310 temp->CastSpell(boss, SPELL_RIFT_CHANNEL, false);
311 }
312 }
313 }
314 }
315 }
316
317 void Update(uint32 diff) override
318 {
319 if (m_auiEncounter[1] != IN_PROGRESS)
320 return;
321
322 //add delay timer?
323 if (!CanProgressEvent())
324 {
325 Clear();
326 return;
327 }
328
329 Events.Update(diff);
330
331 if (Events.ExecuteEvent() == EVENT_NEXT_PORTAL)
332 {
333 ++mRiftPortalCount;
334 DoUpdateWorldState(WORLD_STATE_BM_RIFT, mRiftPortalCount);
335 DoSpawnPortal();
336 ScheduleEventNextPortal(RiftWaves[GetRiftWaveId()].NextPortalTime);
337 }
338 }
339
340 void ScheduleEventNextPortal(uint32 nextPortalTime)
341 {
342 if (nextPortalTime > 0)
343 Events.RescheduleEvent(EVENT_NEXT_PORTAL, nextPortalTime);
344 }

346 protected:
347 EventMap Events;
348 };

254/* Finally, Archimonde's script. His script isn't extremely complex, most are simply spells on timers.
255 The only complicated aspect of the battle is Finger of Death and Doomfire, with Doomfire being the
256 hardest bit to code. Finger of Death is simply a distance check - if no one is in melee range, then
257 select a random target and cast the spell on them. However, if someone IS in melee range, and this
258 is NOT the main tank (creature's victim), then we aggro that player and they become the new victim.
259 For Doomfire, we summon a mob (Doomfire Spirit) for the Doomfire mob to follow. It's spirit will
260 randomly select it's target to follow and then we create the random movement making it unpredictable. */
261
262class boss_archimonde : public CreatureScript
263{
264public:
265 boss_archimonde() : CreatureScript("boss_archimonde") { }
266
267 struct boss_archimondeAI : public BossAI
268 {
269 boss_archimondeAI(Creature* creature) : BossAI(creature, DATA_ARCHIMONDE)
270 {
271 Initialize();
272 }
273
274 void Initialize()
275 {
276 DoomfireSpiritGUID.Clear();
277
278 SoulChargeCount = 0;
279 WispCount = 0; // When ~30 wisps are summoned, Archimonde dies
280 _unleashSpell = 0;
281 _chargeSpell = 0;
282
283 Enraged = false;
284 HasProtected = false;
285 }
286
287 void InitializeAI() override
288 {
289 BossAI::InitializeAI();
290 DoAction(ACTION_CHANNEL_WORLD_TREE);
291 }
292
293 void Reset() override
294 {
295 Initialize();
296 _Reset();
297 me->RemoveAllAuras(); // Reset Soul Charge auras.
298 }
299
300 void EnterCombat(Unit* /*who*/) override
301 {
302 Talk(SAY_AGGRO);
303 _EnterCombat();
304 events.ScheduleEvent(EVENT_FEAR, 42000);
305 events.ScheduleEvent(EVENT_AIR_BURST, 30000);
306 events.ScheduleEvent(EVENT_GRIP_OF_THE_LEGION, urand(5000, 25000));
307 events.ScheduleEvent(EVENT_DOOMFIRE, 20000);
308 events.ScheduleEvent(EVENT_UNLEASH_SOUL_CHARGE, urand(2000, 30000));
309 events.ScheduleEvent(EVENT_FINGER_OF_DEATH, 15000);
310 events.ScheduleEvent(EVENT_HAND_OF_DEATH, 600000);
311 events.ScheduleEvent(EVENT_DISTANCE_CHECK, 30000);
312 }
313
314 void ExecuteEvent(uint32 eventId) override
315 {
316 switch (eventId)
317 {
318 case EVENT_HAND_OF_DEATH:
319 DoCastAOE(SPELL_HAND_OF_DEATH);
320 events.ScheduleEvent(EVENT_HAND_OF_DEATH, 2000);
321 break;
322 case EVENT_UNLEASH_SOUL_CHARGE:
323 _chargeSpell = 0;
324 _unleashSpell = 0;
325 me->InterruptNonMeleeSpells(false);
326 switch (urand(0, 2))
327 {
328 case 0:
329 _chargeSpell = SPELL_SOUL_CHARGE_RED;
330 _unleashSpell = SPELL_UNLEASH_SOUL_RED;
331 break;
332 case 1:
333 _chargeSpell = SPELL_SOUL_CHARGE_YELLOW;
334 _unleashSpell = SPELL_UNLEASH_SOUL_YELLOW;
335 break;
336 case 2:
337 _chargeSpell = SPELL_SOUL_CHARGE_GREEN;
338 _unleashSpell = SPELL_UNLEASH_SOUL_GREEN;
339 break;
340 }
341
342 if (me->HasAura(_chargeSpell))
343 {
344 me->RemoveAuraFromStack(_chargeSpell);
345 DoCastVictim(_unleashSpell);
346 SoulChargeCount--;
347 events.ScheduleEvent(EVENT_UNLEASH_SOUL_CHARGE, urand(2000, 30000));
348 }
349 break;
350 case EVENT_FINGER_OF_DEATH:
351 if (!SelectTarget(SELECT_TARGET_RANDOM, 0, 5.0f)) // Checks if there are no targets in melee range
352 {
353 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0), SPELL_FINGER_OF_DEATH);
354 events.ScheduleEvent(EVENT_FINGER_OF_DEATH, 1000);
355 }
356 else
357 events.ScheduleEvent(EVENT_FINGER_OF_DEATH, 5000);
358 break;
359 case EVENT_GRIP_OF_THE_LEGION:
360 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
361 DoCast(target, SPELL_GRIP_OF_THE_LEGION);
362 events.ScheduleEvent(EVENT_GRIP_OF_THE_LEGION, urand(5000, 25000));
363 break;
364 case EVENT_AIR_BURST:
365 Talk(SAY_AIR_BURST);
366 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
367 DoCast(target, SPELL_AIR_BURST); //not on tank
368 events.ScheduleEvent(EVENT_AIR_BURST, urand(25000, 40000));
369 break;
370 case EVENT_FEAR:
371 DoCastAOE(SPELL_FEAR);
372 events.ScheduleEvent(EVENT_FEAR, 42000);
373 break;
374 case EVENT_DOOMFIRE:
375 Talk(SAY_DOOMFIRE);
376 if (Unit* temp = SelectTarget(SELECT_TARGET_RANDOM, 1))
377 SummonDoomfire(temp);
378 else
379 SummonDoomfire(me->GetVictim());
380 events.ScheduleEvent(EVENT_DOOMFIRE, 20000);
381 break;
382 case EVENT_DISTANCE_CHECK:
383 if (Creature* channelTrigger = instance->GetCreature(DATA_CHANNEL_TARGET))
384 if (me->IsWithinDistInMap(channelTrigger, 75.0f))
385 DoAction(ACTION_ENRAGE);
386 events.ScheduleEvent(EVENT_DISTANCE_CHECK, 5000);
387 break;
388 case EVENT_SUMMON_WHISP:
389 DoSpawnCreature(NPC_ANCIENT_WISP, float(rand32() % 40), float(rand32() % 40), 0, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000);
390 ++WispCount;
391 if (WispCount >= 30)
392 me->DealDamage(me, me->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
393 events.ScheduleEvent(EVENT_SUMMON_WHISP, 1500);
394 break;
395 default:
396 break;
397 }
398 }
399
400 void DamageTaken(Unit* /*attacker*/, uint32 &damage) override
401 {
402 if (me->HealthBelowPctDamaged(10, damage))
403 {
404 if (!Enraged)
405 DoAction(ACTION_ENRAGE);
406
407 if (!HasProtected)
408 {
409 me->GetMotionMaster()->Clear(false);
410 me->GetMotionMaster()->MoveIdle();
411
412 // All members of raid must get this buff
413 DoCastAOE(SPELL_PROTECTION_OF_ELUNE, true);
414 HasProtected = true;
415 events.ScheduleEvent(EVENT_SUMMON_WHISP, 1500);
416 }
417 }
418 }
419
420 void KilledUnit(Unit* victim) override
421 {
422 Talk(SAY_SLAY);
423
424 if (victim->GetTypeId() == TYPEID_PLAYER)
425 {
426 switch (victim->getClass())
427 {
428 case CLASS_PRIEST:
429 case CLASS_PALADIN:
430 case CLASS_WARLOCK:
431 victim->CastSpell(me, SPELL_SOUL_CHARGE_RED, true);
432 break;
433 case CLASS_MAGE:
434 case CLASS_ROGUE:
435 case CLASS_WARRIOR:
436 victim->CastSpell(me, SPELL_SOUL_CHARGE_YELLOW, true);
437 break;
438 case CLASS_DRUID:
439 case CLASS_SHAMAN:
440 case CLASS_HUNTER:
441 victim->CastSpell(me, SPELL_SOUL_CHARGE_GREEN, true);
442 break;
443 }
444
445 events.ScheduleEvent(EVENT_UNLEASH_SOUL_CHARGE, urand(2000, 30000));
446 ++SoulChargeCount;
447 }
448 }
449
450 void JustReachedHome() override
451 {
452 DoAction(ACTION_CHANNEL_WORLD_TREE);
453 }

455 void JustDied(Unit* /*killer*/) override
456 {
457 Talk(SAY_DEATH);
458 _JustDied();
459 // @todo: remove this when instance script gets updated, kept for compatibility only
460 instance->SetData(DATA_ARCHIMONDE, DONE);
461 }
462
463 void JustSummoned(Creature* summoned) override
464 {
465 switch (summoned->GetEntry())
466 {
467 case NPC_ANCIENT_WISP:
468 summoned->AI()->AttackStart(me);
469 break;
470 case NPC_DOOMFIRE_SPIRIT:
471 DoomfireSpiritGUID = summoned->GetGUID();
472 break;
473 case NPC_DOOMFIRE:
474 summoned->CastSpell(summoned, SPELL_DOOMFIRE_SPAWN, false);
475 summoned->CastSpell(summoned, SPELL_DOOMFIRE, true, 0, 0, me->GetGUID());
476
477 if (Unit* DoomfireSpirit = ObjectAccessor::GetUnit(*me, DoomfireSpiritGUID))
478 {
479 summoned->GetMotionMaster()->MoveFollow(DoomfireSpirit, 0.0f, 0.0f);
480 DoomfireSpiritGUID.Clear();
481 }
482 break;
483 default:
484 break;
485 }
486 }
487
488 void DoAction(int32 actionId) override
489 {
490 switch (actionId)
491 {
492 case ACTION_ENRAGE:
493 me->GetMotionMaster()->Clear(false);
494 me->GetMotionMaster()->MoveIdle();
495 Enraged = true;
496 Talk(SAY_ENRAGE);
497 break;
498 case ACTION_CHANNEL_WORLD_TREE:
499 DoCastAOE(SPELL_DRAIN_WORLD_TREE, true);
500 break;
501 default:
502 break;
503 }
504 }
505
506 //this is code doing close to what the summoning spell would do (spell 31903)
507 void SummonDoomfire(Unit* target)
508 {
509 if (!target)
510 return;
511
512 me->SummonCreature(NPC_DOOMFIRE_SPIRIT,
513 target->GetPositionX()+15.0f, target->GetPositionY()+15.0f, target->GetPositionZ(), 0,
514 TEMPSUMMON_TIMED_DESPAWN, 27000);
515
516 me->SummonCreature(NPC_DOOMFIRE,
517 target->GetPositionX()-15.0f, target->GetPositionY()-15.0f, target->GetPositionZ(), 0,
518 TEMPSUMMON_TIMED_DESPAWN, 27000);
519 }
520
521 private:
522 ObjectGuid DoomfireSpiritGUID;
523 uint8 SoulChargeCount;
524 uint8 WispCount;
525 uint32 _chargeSpell;
526 uint32 _unleashSpell;
527 bool Enraged;
528 bool HasProtected;
529 };
530
531 CreatureAI* GetAI(Creature* creature) const override
532 {
533 return GetInstanceAI<boss_archimondeAI>(creature);
534 }

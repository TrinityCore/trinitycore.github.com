44ObjectData const creatureData[] =
45{
46 { NPC_CHANNEL_TARGET, DATA_CHANNEL_TARGET },
47 { 0, 0 } // END

50class instance_hyjal : public InstanceMapScript
51{
52public:
53 instance_hyjal() : InstanceMapScript("instance_hyjal", 534) { }
54
55 InstanceScript* GetInstanceScript(InstanceMap* map) const override
56 {
57 return new instance_mount_hyjal_InstanceMapScript(map);
58 }
59
60 struct instance_mount_hyjal_InstanceMapScript : public InstanceScript
61 {
62 instance_mount_hyjal_InstanceMapScript(Map* map) : InstanceScript(map)
63 {
64 SetHeaders(DataHeader);
65 LoadObjectData(creatureData, nullptr);
66 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
67
68 RaidDamage = 0;
69 Trash = 0;
70 hordeRetreat = 0;
71 allianceRetreat = 0;
72
73 ArchiYell = false;
74 }
75
76 bool IsEncounterInProgress() const override
77 {
78 for (uint8 i = 0; i < EncounterCount; ++i)
79 if (m_auiEncounter[i] == IN_PROGRESS)
80 return true;
81
82 return false;
83 }
84
85 void OnGameObjectCreate(GameObject* go) override
86 {
87 switch (go->GetEntry())
88 {
89 case GO_HORDE_ENCAMPMENT_PORTAL:
90 HordeGate = go->GetGUID();
91 if (allianceRetreat)
92 HandleGameObject(ObjectGuid::Empty, true, go);
93 else
94 HandleGameObject(ObjectGuid::Empty, false, go);
95 break;
96 case GO_NIGHT_ELF_VILLAGE_PORTAL:
97 ElfGate = go->GetGUID();
98 if (hordeRetreat)
99 HandleGameObject(ObjectGuid::Empty, true, go);
100 else
101 HandleGameObject(ObjectGuid::Empty, false, go);
102 break;
103 case GO_ANCIENT_GEM:
104 m_uiAncientGemGUID.push_back(go->GetGUID());
105 break;
106 }
107 }
108
109 void OnCreatureCreate(Creature* creature) override
110 {
111 switch (creature->GetEntry())
112 {
113 case RAGE_WINTERCHILL:
114 RageWinterchill = creature->GetGUID();
115 break;
116 case ANETHERON:
117 Anetheron = creature->GetGUID();
118 break;
119 case KAZROGAL:
120 Kazrogal = creature->GetGUID();
121 break;
122 case AZGALOR:
123 Azgalor = creature->GetGUID();
124 break;
125 case ARCHIMONDE:
126 Archimonde = creature->GetGUID();
127 if (GetData(DATA_AZGALOREVENT) != DONE)
128 creature->SetVisible(false);
129 break;
130 case JAINA:
131 JainaProudmoore = creature->GetGUID();
132 break;
133 case THRALL:
134 Thrall = creature->GetGUID();
135 break;
136 case TYRANDE:
137 TyrandeWhisperwind = creature->GetGUID();
138 break;
139 }
140
141 InstanceScript::OnCreatureCreate(creature);
142 }
143
144 ObjectGuid GetGuidData(uint32 identifier) const override
145 {
146 switch (identifier)
147 {
148 case DATA_RAGEWINTERCHILL: return RageWinterchill;
149 case DATA_ANETHERON: return Anetheron;
150 case DATA_KAZROGAL: return Kazrogal;
151 case DATA_AZGALOR: return Azgalor;
152 case DATA_ARCHIMONDE: return Archimonde;
153 case DATA_JAINAPROUDMOORE: return JainaProudmoore;
154 case DATA_THRALL: return Thrall;
155 case DATA_TYRANDEWHISPERWIND: return TyrandeWhisperwind;
156 }
157
158 return ObjectGuid::Empty;
159 }
160
161 void SetData(uint32 type, uint32 data) override
162 {
163 switch (type)
164 {
165 case DATA_RAGEWINTERCHILLEVENT:
166 m_auiEncounter[0] = data;
167 break;
168 case DATA_ANETHERONEVENT:
169 m_auiEncounter[1] = data;
170 break;
171 case DATA_KAZROGALEVENT:
172 m_auiEncounter[2] = data;
173 break;
174 case DATA_AZGALOREVENT:
175 m_auiEncounter[3] = data;
176 if (data == DONE)
177 {
178 instance->LoadGrid(5581.49f, -3445.63f);
179 if (Creature* archimonde = instance->GetCreature(Archimonde))
180 {
181 archimonde->SetVisible(true);
182
183 if (!ArchiYell)
184 {
185 ArchiYell = true;
186 archimonde->AI()->Talk(YELL_ARCHIMONDE_INTRO);
187 }
188 }
189 }
190 break;
191 case DATA_ARCHIMONDEEVENT:
192 m_auiEncounter[4] = data;
193 break;
194 case DATA_RESET_TRASH_COUNT:
195 Trash = 0;
196 break;
197 case DATA_TRASH:
198 if (data)
199 Trash = data;
200 else
201 Trash--;
202 DoUpdateWorldState(WORLD_STATE_ENEMYCOUNT, Trash);
203 break;
204 case TYPE_RETREAT:
205 if (data == SPECIAL)
206 {
207 if (!m_uiAncientGemGUID.empty())
208 {
209 for (GuidList::const_iterator itr = m_uiAncientGemGUID.begin(); itr != m_uiAncientGemGUID.end(); ++itr)
210 {
211 //don't know how long it expected
212 DoRespawnGameObject(*itr, DAY);
213 }
214 }
215 }
216 break;
217 case DATA_ALLIANCE_RETREAT:
218 allianceRetreat = data;
219 HandleGameObject(HordeGate, true);
220 SaveToDB();
221 break;
222 case DATA_HORDE_RETREAT:
223 hordeRetreat = data;
224 HandleGameObject(ElfGate, true);
225 SaveToDB();
226 break;
227 case DATA_RAIDDAMAGE:
228 RaidDamage += data;
229 if (RaidDamage >= MINRAIDDAMAGE)
230 RaidDamage = MINRAIDDAMAGE;
231 break;
232 case DATA_RESET_RAIDDAMAGE:
233 RaidDamage = 0;
234 break;
235 }
236
237 TC_LOG_DEBUG("scripts", "Instance Hyjal: Instance data updated for event %u (Data=%u)", type, data);
238
239 if (data == DONE)
240 {
241 OUT_SAVE_INST_DATA;
242
243 std::ostringstream saveStream;
244 saveStream << m_auiEncounter[0] << ' ' << m_auiEncounter[1] << ' ' << m_auiEncounter[2] << ' '
245 << m_auiEncounter[3] << ' ' << m_auiEncounter[4]
246 << ' ' << allianceRetreat << ' ' << hordeRetreat
247 << ' ' << RaidDamage;
248
249 str_data = saveStream.str();
250
251 SaveToDB();
252 OUT_SAVE_INST_DATA_COMPLETE;
253 }
254
255 }
256
257 uint32 GetData(uint32 type) const override
258 {
259 switch (type)
260 {
261 case DATA_RAGEWINTERCHILLEVENT: return m_auiEncounter[0];
262 case DATA_ANETHERONEVENT: return m_auiEncounter[1];
263 case DATA_KAZROGALEVENT: return m_auiEncounter[2];
264 case DATA_AZGALOREVENT: return m_auiEncounter[3];
265 case DATA_ARCHIMONDEEVENT: return m_auiEncounter[4];
266 case DATA_TRASH: return Trash;
267 case DATA_ALLIANCE_RETREAT: return allianceRetreat;
268 case DATA_HORDE_RETREAT: return hordeRetreat;
269 case DATA_RAIDDAMAGE: return RaidDamage;
270 }
271 return 0;
272 }
273
274 std::string GetSaveData() override
275 {
276 return str_data;
277 }
278
279 void Load(const char* in) override
280 {
281 if (!in)
282 {
283 OUT_LOAD_INST_DATA_FAIL;
284 return;
285 }
286
287 OUT_LOAD_INST_DATA(in);
288 std::istringstream loadStream(in);
289 loadStream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3] >> m_auiEncounter[4] >> allianceRetreat >> hordeRetreat >> RaidDamage;
290 for (uint8 i = 0; i < EncounterCount; ++i)
291 if (m_auiEncounter[i] == IN_PROGRESS) // Do not load an encounter as IN_PROGRESS - reset it instead.
292 m_auiEncounter[i] = NOT_STARTED;
293 OUT_LOAD_INST_DATA_COMPLETE;
294 }
295
296 protected:
297 uint32 m_auiEncounter[EncounterCount];
298 std::string str_data;
299 GuidList m_uiAncientGemGUID;
300 ObjectGuid RageWinterchill;
301 ObjectGuid Anetheron;
302 ObjectGuid Kazrogal;
303 ObjectGuid Azgalor;
304 ObjectGuid Archimonde;
305 ObjectGuid JainaProudmoore;
306 ObjectGuid Thrall;
307 ObjectGuid TyrandeWhisperwind;
308 ObjectGuid HordeGate;
309 ObjectGuid ElfGate;
310 uint32 Trash;
311 uint32 hordeRetreat;
312 uint32 allianceRetreat;
313 uint32 RaidDamage;
314 bool ArchiYell;
315 };
316};

23enum Spells
24{
25 SPELL_FROST_ARMOR = 31256,
26 SPELL_DEATH_AND_DECAY = 31258,
27 SPELL_FROST_NOVA = 31250,
28 SPELL_ICEBOLT = 31249
29};
30
31enum Texts
32{
33 SAY_ONDEATH = 0,
34 SAY_ONSLAY = 1,
35 SAY_DECAY = 2,
36 SAY_NOVA = 3,
37 SAY_ONAGGRO = 4
38};
39
40class boss_rage_winterchill : public CreatureScript
41{
42public:
43 boss_rage_winterchill() : CreatureScript("boss_rage_winterchill") { }
44
45 CreatureAI* GetAI(Creature* creature) const override
46 {
47 return GetInstanceAI<boss_rage_winterchillAI>(creature);
48 }
49
50 struct boss_rage_winterchillAI : public hyjal_trashAI
51 {
52 boss_rage_winterchillAI(Creature* creature) : hyjal_trashAI(creature)
53 {
54 Initialize();
55 instance = creature->GetInstanceScript();
56 go = false;
57 }
58
59 void Initialize()
60 {
61 damageTaken = 0;
62 FrostArmorTimer = 37000;
63 DecayTimer = 45000;
64 NovaTimer = 15000;
65 IceboltTimer = 10000;
66 }
67
68 uint32 FrostArmorTimer;
69 uint32 DecayTimer;
70 uint32 NovaTimer;
71 uint32 IceboltTimer;
72 bool go;
73
74 void Reset() override
75 {
76 Initialize();
77
78 if (IsEvent)
79 instance->SetData(DATA_RAGEWINTERCHILLEVENT, NOT_STARTED);
80 }
81
82 void EnterCombat(Unit* /*who*/) override
83 {
84 if (IsEvent)
85 instance->SetData(DATA_RAGEWINTERCHILLEVENT, IN_PROGRESS);
86 Talk(SAY_ONAGGRO);
87 }
88
89 void KilledUnit(Unit* /*victim*/) override
90 {
91 Talk(SAY_ONSLAY);
92 }
93
94 void WaypointReached(uint32 waypointId) override
95 {
96 if (waypointId == 7 && instance)
97 {
98 Unit* target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_JAINAPROUDMOORE));
99 if (target && target->IsAlive())
100 me->AddThreat(target, 0.0f);
101 }
102 }
103
104 void JustDied(Unit* killer) override
105 {
106 hyjal_trashAI::JustDied(killer);
107 if (IsEvent)
108 instance->SetData(DATA_RAGEWINTERCHILLEVENT, DONE);
109 Talk(SAY_ONDEATH);
110 }
111
112 void UpdateAI(uint32 diff) override
113 {
114 if (IsEvent)
115 {
116 //Must update npc_escortAI
117 npc_escortAI::UpdateAI(diff);
118 if (!go)
119 {
120 go = true;
121 AddWaypoint(0, 4896.08f, -1576.35f, 1333.65f);
122 AddWaypoint(1, 4898.68f, -1615.02f, 1329.48f);
123 AddWaypoint(2, 4907.12f, -1667.08f, 1321.00f);
124 AddWaypoint(3, 4963.18f, -1699.35f, 1340.51f);
125 AddWaypoint(4, 4989.16f, -1716.67f, 1335.74f);
126 AddWaypoint(5, 5026.27f, -1736.89f, 1323.02f);
127 AddWaypoint(6, 5037.77f, -1770.56f, 1324.36f);
128 AddWaypoint(7, 5067.23f, -1789.95f, 1321.17f);
129 Start(false, true);
130 SetDespawnAtEnd(false);
131 }
132 }
133
134 //Return since we have no target
135 if (!UpdateVictim())
136 return;
137
138 if (FrostArmorTimer <= diff)
139 {
140 DoCast(me, SPELL_FROST_ARMOR);
141 FrostArmorTimer = 40000 + rand32() % 20000;
142 } else FrostArmorTimer -= diff;
143 if (DecayTimer <= diff)
144 {
145 DoCastVictim(SPELL_DEATH_AND_DECAY);
146 DecayTimer = 60000 + rand32() % 20000;
147 Talk(SAY_DECAY);
148 } else DecayTimer -= diff;
149 if (NovaTimer <= diff)
150 {
151 DoCastVictim(SPELL_FROST_NOVA);
152 NovaTimer = 30000 + rand32() % 15000;
153 Talk(SAY_NOVA);
154 } else NovaTimer -= diff;
155 if (IceboltTimer <= diff)
156 {
157 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true), SPELL_ICEBOLT);
158 IceboltTimer = 11000 + rand32() % 20000;
159 } else IceboltTimer -= diff;
160
161 DoMeleeAttackIfReady();
162 }
163 };
164
165};
166
167void AddSC_boss_rage_winterchill()
168{
169 new boss_rage_winterchill();
170}

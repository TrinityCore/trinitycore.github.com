54class boss_moam : public CreatureScript
55{
56 public:
57 boss_moam() : CreatureScript("boss_moam") { }
58
59 struct boss_moamAI : public BossAI
60 {
61 boss_moamAI(Creature* creature) : BossAI(creature, DATA_MOAM)
62 {
63 Initialize();
64 }
65
66 void Initialize()
67 {
68 _isStonePhase = false;
69 }
70
71 void Reset() override
72 {
73 _Reset();
74 me->SetPower(POWER_MANA, 0);
75 Initialize();
76 events.ScheduleEvent(EVENT_STONE_PHASE, 90000);
77 //events.ScheduleEvent(EVENT_WIDE_SLASH, 11000);
78 }
79
80 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
81 {
82 if (!_isStonePhase && HealthBelowPct(45))
83 {
84 _isStonePhase = true;
85 DoAction(ACTION_STONE_PHASE_START);
86 }
87 }
88
89 void DoAction(int32 action) override
90 {
91 switch (action)
92 {
93 case ACTION_STONE_PHASE_END:
94 {
95 me->RemoveAurasDueToSpell(SPELL_ENERGIZE);
96 events.ScheduleEvent(EVENT_STONE_PHASE, 90000);
97 _isStonePhase = false;
98 break;
99 }
100 case ACTION_STONE_PHASE_START:
101 {
102 DoCast(me, SPELL_SUMMON_MANA_FIEND_1);
103 DoCast(me, SPELL_SUMMON_MANA_FIEND_2);
104 DoCast(me, SPELL_SUMMON_MANA_FIEND_3);
105 DoCast(me, SPELL_ENERGIZE);
106 events.ScheduleEvent(EVENT_STONE_PHASE_END, 90000);
107 break;
108 }
109 default:
110 break;
111 }
112 }
113
114 void UpdateAI(uint32 diff) override
115 {
116 if (!UpdateVictim())
117 return;
118
119 events.Update(diff);
120
121 if (me->GetPower(POWER_MANA) == me->GetMaxPower(POWER_MANA))
122 {
123 if (_isStonePhase)
124 DoAction(ACTION_STONE_PHASE_END);
125 DoCastAOE(SPELL_ARCANE_ERUPTION);
126 me->SetPower(POWER_MANA, 0);
127 }
128
129 if (_isStonePhase)
130 {
131 if (events.ExecuteEvent() == EVENT_STONE_PHASE_END)
132 DoAction(ACTION_STONE_PHASE_END);
133 return;
134 }
135
136 // Messing up mana-drain channel
137 //if (me->HasUnitState(UNIT_STATE_CASTING))
138 // return;
139
140 while (uint32 eventId = events.ExecuteEvent())
141 {
142 switch (eventId)
143 {
144 case EVENT_STONE_PHASE:
145 DoAction(ACTION_STONE_PHASE_START);
146 break;
147 case EVENT_DRAIN_MANA:
148 {
149 std::list<Unit*> targetList;
150 {
151 const std::list<HostileReference*>& threatlist = me->getThreatManager().getThreatList();
152 for (std::list<HostileReference*>::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
153 if ((*itr)->getTarget()->GetTypeId() == TYPEID_PLAYER && (*itr)->getTarget()->getPowerType() == POWER_MANA)
154 targetList.push_back((*itr)->getTarget());
155 }
156
157 Trinity::Containers::RandomResizeList(targetList, 5);
158
159 for (std::list<Unit*>::iterator itr = targetList.begin(); itr != targetList.end(); ++itr)
160 DoCast(*itr, SPELL_DRAIN_MANA);
161
162 events.ScheduleEvent(EVENT_DRAIN_MANA, urand(5000, 15000));
163 break;
164 }/*
165 case EVENT_WIDE_SLASH:
166 DoCast(me, SPELL_WIDE_SLASH);
167 events.ScheduleEvent(EVENT_WIDE_SLASH, 11000);
168 break;
169 case EVENT_TRASH:
170 DoCast(me, SPELL_TRASH);
171 events.ScheduleEvent(EVENT_WIDE_SLASH, 16000);
172 break;*/
173 default:
174 break;
175 }
176 }
177
178 DoMeleeAttackIfReady();
179 }
180 private:
181 bool _isStonePhase;
182 };
183
184 CreatureAI* GetAI(Creature* creature) const override
185 {
186 return new boss_moamAI(creature);
187 }
188};
189
190void AddSC_boss_moam()
191{
192 new boss_moam();
193}

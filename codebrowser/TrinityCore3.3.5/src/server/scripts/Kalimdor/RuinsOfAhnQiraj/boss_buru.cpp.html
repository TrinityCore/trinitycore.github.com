63class boss_buru : public CreatureScript
64{
65 public:
66 boss_buru() : CreatureScript("boss_buru") { }
67
68 struct boss_buruAI : public BossAI
69 {
70 boss_buruAI(Creature* creature) : BossAI(creature, DATA_BURU)
71 {
72 _phase = 0;
73 }
74
75 void EnterEvadeMode(EvadeReason why) override
76 {
77 BossAI::EnterEvadeMode(why);
78
79 for (ObjectGuid eggGuid : Eggs)
80 if (Creature* egg = me->GetMap()->GetCreature(eggGuid))
81 egg->Respawn();
82
83 Eggs.clear();
84 }
85
86 void EnterCombat(Unit* who) override
87 {
88 _EnterCombat();
89 Talk(EMOTE_TARGET, who);
90 DoCast(me, SPELL_THORNS);
91
92 events.ScheduleEvent(EVENT_DISMEMBER, 5000);
93 events.ScheduleEvent(EVENT_GATHERING_SPEED, 9000);
94 events.ScheduleEvent(EVENT_FULL_SPEED, 60000);
95
96 _phase = PHASE_EGG;
97 }
98
99 void DoAction(int32 action) override
100 {
101 if (action == ACTION_EXPLODE)
102 if (_phase == PHASE_EGG)
103 me->DealDamage(me, 45000);
104 }
105
106 void KilledUnit(Unit* victim) override
107 {
108 if (victim->GetTypeId() == TYPEID_PLAYER)
109 ChaseNewVictim();
110 }
111
112 void ChaseNewVictim()
113 {
114 if (_phase != PHASE_EGG)
115 return;
116
117 me->RemoveAurasDueToSpell(SPELL_FULL_SPEED);
118 me->RemoveAurasDueToSpell(SPELL_GATHERING_SPEED);
119 events.ScheduleEvent(EVENT_GATHERING_SPEED, 9000);
120 events.ScheduleEvent(EVENT_FULL_SPEED, 60000);
121
122 if (Unit* victim = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
123 {
124 DoResetThreat();
125 AttackStart(victim);
126 Talk(EMOTE_TARGET, victim);
127 }
128 }
129
130 void ManageRespawn(ObjectGuid EggGUID)
131 {
132 ChaseNewVictim();
133 Eggs.push_back(EggGUID);
134 events.ScheduleEvent(EVENT_RESPAWN_EGG, 100000);
135 }
136
137 void UpdateAI(uint32 diff) override
138 {
139 if (!UpdateVictim())
140 return;
141
142 events.Update(diff);
143
144 while (uint32 eventId = events.ExecuteEvent())
145 {
146 switch (eventId)
147 {
148 case EVENT_DISMEMBER:
149 DoCastVictim(SPELL_DISMEMBER);
150 events.ScheduleEvent(EVENT_DISMEMBER, 5000);
151 break;
152 case EVENT_GATHERING_SPEED:
153 DoCast(me, SPELL_GATHERING_SPEED);
154 events.ScheduleEvent(EVENT_GATHERING_SPEED, 9000);
155 break;
156 case EVENT_FULL_SPEED:
157 DoCast(me, SPELL_FULL_SPEED);
158 break;
159 case EVENT_CREEPING_PLAGUE:
160 DoCast(me, SPELL_CREEPING_PLAGUE);
161 events.ScheduleEvent(EVENT_CREEPING_PLAGUE, 6000);
162 break;
163 case EVENT_RESPAWN_EGG:
164 if (Creature* egg = me->GetMap()->GetCreature(*Eggs.begin()))
165 {
166 egg->Respawn();
167 Eggs.pop_front();
168 }
169 break;
170 default:
171 break;
172 }
173 }
174
175 if (me->GetHealthPct() < 20.0f && _phase == PHASE_EGG)
176 {
177 DoCast(me, SPELL_BURU_TRANSFORM); // Enrage
178 DoCast(me, SPELL_FULL_SPEED, true);
179 me->RemoveAurasDueToSpell(SPELL_THORNS);
180 _phase = PHASE_TRANSFORM;
181 }
182
183 DoMeleeAttackIfReady();
184 }
185 private:
186 GuidList Eggs;
187 uint8 _phase;
188 };
189
190 CreatureAI* GetAI(Creature* creature) const override
191 {
192 return new boss_buruAI(creature);
193 }
194};

196class npc_buru_egg : public CreatureScript
197{
198 public:
199 npc_buru_egg() : CreatureScript("npc_buru_egg") { }
200
201 struct npc_buru_eggAI : public ScriptedAI
202 {
203 npc_buru_eggAI(Creature* creature) : ScriptedAI(creature)
204 {
205 _instance = me->GetInstanceScript();
206 SetCombatMovement(false);
207 }
208
209 void EnterCombat(Unit* attacker) override
210 {
211 if (Creature* buru = me->GetMap()->GetCreature(_instance->GetGuidData(DATA_BURU)))
212 if (!buru->IsInCombat())
213 buru->AI()->AttackStart(attacker);
214 }
215
216 void JustSummoned(Creature* who) override
217 {
218 if (who->GetEntry() == NPC_HATCHLING)
219 if (Creature* buru = me->GetMap()->GetCreature(_instance->GetGuidData(DATA_BURU)))
220 if (Unit* target = buru->AI()->SelectTarget(SELECT_TARGET_RANDOM))
221 who->AI()->AttackStart(target);
222 }
223
224 void JustDied(Unit* /*killer*/) override
225 {
226 DoCastAOE(SPELL_EXPLODE, true);
227 DoCastAOE(SPELL_EXPLODE_2, true); // Unknown purpose
228 DoCast(me, SPELL_SUMMON_HATCHLING, true);
229
230 if (Creature* buru = me->GetMap()->GetCreature(_instance->GetGuidData(DATA_BURU)))
231 if (boss_buru::boss_buruAI* buruAI = dynamic_cast<boss_buru::boss_buruAI*>(buru->AI()))
232 buruAI->ManageRespawn(me->GetGUID());
233 }
234 private:
235 InstanceScript* _instance;
236 };
237
238 CreatureAI* GetAI(Creature* creature) const override
239 {
240 return GetInstanceAI<npc_buru_eggAI>(creature);
241 }
242};

244class spell_egg_explosion : public SpellScriptLoader
245{
246 public:
247 spell_egg_explosion() : SpellScriptLoader("spell_egg_explosion") { }
248
249 class spell_egg_explosion_SpellScript : public SpellScript
250 {
251 PrepareSpellScript(spell_egg_explosion_SpellScript);
252
253 void HandleAfterCast()
254 {
255 if (Creature* buru = GetCaster()->FindNearestCreature(NPC_BURU, 5.f))
256 buru->AI()->DoAction(ACTION_EXPLODE);
257 }
258
259 void HandleDummyHitTarget(SpellEffIndex /*effIndex*/)
260 {
261 if (Unit* target = GetHitUnit())
262 GetCaster()->DealDamage(target, -16 * GetCaster()->GetDistance(target) + 500);
263 }
264
265 void Register() override
266 {
267 AfterCast += SpellCastFn(spell_egg_explosion_SpellScript::HandleAfterCast);
268 OnEffectHitTarget += SpellEffectFn(spell_egg_explosion_SpellScript::HandleDummyHitTarget, EFFECT_0, SPELL_EFFECT_DUMMY);
269 }
270 };
271
272 SpellScript* GetSpellScript() const override
273 {
274 return new spell_egg_explosion_SpellScript();
275 }

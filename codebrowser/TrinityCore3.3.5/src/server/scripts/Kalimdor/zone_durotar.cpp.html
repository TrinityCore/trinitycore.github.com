52 struct npc_lazy_peonAI : public ScriptedAI
53 {
54 npc_lazy_peonAI(Creature* creature) : ScriptedAI(creature)
55 {
56 Initialize();
57 }
58
59 void Initialize()
60 {
61 RebuffTimer = 0;
62 work = false;
63 }
64
65 uint32 RebuffTimer;
66 bool work;
67
68 void Reset() override
69 {
70 Initialize();
71 }
72
73 void MovementInform(uint32 /*type*/, uint32 id) override
74 {
75 if (id == 1)
76 work = true;
77 }
78
79 void SpellHit(Unit* caster, const SpellInfo* spell) override
80 {
81 if (spell->Id != SPELL_AWAKEN_PEON)
82 return;
83
84 Player* player = caster->ToPlayer();
85 if (player && player->GetQuestStatus(QUEST_LAZY_PEONS) == QUEST_STATUS_INCOMPLETE)
86 {
87 player->KilledMonsterCredit(me->GetEntry(), me->GetGUID());
88 Talk(SAY_SPELL_HIT, caster);
89 me->RemoveAllAuras();
90 if (GameObject* Lumberpile = me->FindNearestGameObject(GO_LUMBERPILE, 20))
91 me->GetMotionMaster()->MovePoint(1, Lumberpile->GetPositionX()-1, Lumberpile->GetPositionY(), Lumberpile->GetPositionZ());
92 }
93 }
94
95 void UpdateAI(uint32 diff) override
96 {
97 if (work == true)
98 me->HandleEmoteCommand(EMOTE_ONESHOT_WORK_CHOPWOOD);
99 if (RebuffTimer <= diff)
100 {
101 DoCast(me, SPELL_BUFF_SLEEP);
102 RebuffTimer = 300000; //Rebuff agian in 5 minutes
103 }
104 else
105 RebuffTimer -= diff;
106 if (!UpdateVictim())
107 return;
108 DoMeleeAttackIfReady();
109 }
110 };

181class npc_tiger_matriarch_credit : public CreatureScript
182{
183 public:
184 npc_tiger_matriarch_credit() : CreatureScript("npc_tiger_matriarch_credit") { }
185
186 struct npc_tiger_matriarch_creditAI : public ScriptedAI
187 {
188 npc_tiger_matriarch_creditAI(Creature* creature) : ScriptedAI(creature)
189 {
190 SetCombatMovement(false);
191 events.ScheduleEvent(EVENT_CHECK_SUMMON_AURA, 2000);
192 }
193
194 void UpdateAI(uint32 diff) override
195 {
196 events.Update(diff);
197
198 if (events.ExecuteEvent() == EVENT_CHECK_SUMMON_AURA)
199 {
200 std::list<Creature*> tigers;
201 GetCreatureListWithEntryInGrid(tigers, me, NPC_TIGER_VEHICLE, 15.0f);
202 if (!tigers.empty())
203 {
204 for (std::list<Creature*>::iterator itr = tigers.begin(); itr != tigers.end(); ++itr)
205 {
206 if (!(*itr)->IsSummon())
207 continue;
208
209 if (Unit* summoner = (*itr)->ToTempSummon()->GetSummoner())
210 if (!summoner->HasAura(SPELL_NO_SUMMON_AURA) && !summoner->HasAura(SPELL_SUMMON_ZENTABRA_TRIGGER)
211 && !summoner->IsInCombat())
212 {
213 me->AddAura(SPELL_NO_SUMMON_AURA, summoner);
214 me->AddAura(SPELL_DETECT_INVIS, summoner);
215 summoner->CastSpell(summoner, SPELL_SUMMON_MATRIARCH, true);
216 Talk(SAY_MATRIARCH_AGGRO, summoner);
217 }
218 }
219 }
220
221 events.ScheduleEvent(EVENT_CHECK_SUMMON_AURA, 5000);
222 }
223 }
224
225 private:
226 EventMap events;
227 };
228
229 CreatureAI* GetAI(Creature* creature) const override
230 {
231 return new npc_tiger_matriarch_creditAI(creature);
232 }

235class npc_tiger_matriarch : public CreatureScript
236{
237 public:
238 npc_tiger_matriarch() : CreatureScript("npc_tiger_matriarch") { }
239
240 struct npc_tiger_matriarchAI : public ScriptedAI
241 {
242 npc_tiger_matriarchAI(Creature* creature) : ScriptedAI(creature)
243 {
244 }
245
246 void EnterCombat(Unit* /*target*/) override
247 {
248 _events.Reset();
249 _events.ScheduleEvent(EVENT_POUNCE, 100);
250 _events.ScheduleEvent(EVENT_NOSUMMON, 50000);
251 }
252
253 void IsSummonedBy(Unit* summoner) override
254 {
255 if (summoner->GetTypeId() != TYPEID_PLAYER || !summoner->GetVehicle())
256 return;
257
258 _tigerGuid = summoner->GetVehicle()->GetBase()->GetGUID();
259 if (Unit* tiger = ObjectAccessor::GetUnit(*me, _tigerGuid))
260 {
261 me->AddThreat(tiger, 500000.0f);
262 DoCast(me, SPELL_FURIOUS_BITE);
263 }
264 }
265
266 void KilledUnit(Unit* victim) override
267 {
268 if (victim->GetTypeId() != TYPEID_UNIT || !victim->IsSummon())
269 return;
270
271 if (Unit* vehSummoner = victim->ToTempSummon()->GetSummoner())
272 {
273 vehSummoner->RemoveAurasDueToSpell(SPELL_NO_SUMMON_AURA);
274 vehSummoner->RemoveAurasDueToSpell(SPELL_DETECT_INVIS);
275 vehSummoner->RemoveAurasDueToSpell(SPELL_SPIRIT_OF_THE_TIGER_RIDER);
276 vehSummoner->RemoveAurasDueToSpell(SPELL_SUMMON_ZENTABRA_TRIGGER);
277 }
278 me->DespawnOrUnsummon();
279 }
280
281 void DamageTaken(Unit* attacker, uint32& damage) override
282 {
283 if (!attacker->IsSummon())
284 return;
285
286 if (HealthBelowPct(20))
287 {
288 damage = 0;
289 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
290 if (Unit* vehSummoner = attacker->ToTempSummon()->GetSummoner())
291 {
292 vehSummoner->AddAura(SPELL_SUMMON_ZENTABRA_TRIGGER, vehSummoner);
293 vehSummoner->CastSpell(vehSummoner, SPELL_SUMMON_ZENTABRA, true);
294 attacker->CastSpell(attacker, SPELL_EJECT_PASSENGERS, true);
295 vehSummoner->RemoveAurasDueToSpell(SPELL_NO_SUMMON_AURA);
296 vehSummoner->RemoveAurasDueToSpell(SPELL_DETECT_INVIS);
297 vehSummoner->RemoveAurasDueToSpell(SPELL_SPIRIT_OF_THE_TIGER_RIDER);
298 vehSummoner->RemoveAurasDueToSpell(SPELL_SUMMON_ZENTABRA_TRIGGER);
299 }
300
301 me->DespawnOrUnsummon();
302 }
303 }
304
305 void UpdateAI(uint32 diff) override
306 {
307 if (!UpdateVictim())
308 return;
309
310 if (!_tigerGuid)
311 return;
312
313 _events.Update(diff);
314
315 while (uint32 eventId = _events.ExecuteEvent())
316 {
317 switch (eventId)
318 {
319 case EVENT_POUNCE:
320 DoCastVictim(SPELL_POUNCE);
321 _events.ScheduleEvent(EVENT_POUNCE, 30000);
322 break;
323 case EVENT_NOSUMMON: // Reapply SPELL_NO_SUMMON_AURA
324 if (Unit* tiger = ObjectAccessor::GetUnit(*me, _tigerGuid))
325 {
326 if (tiger->IsSummon())
327 if (Unit* vehSummoner = tiger->ToTempSummon()->GetSummoner())
328 me->AddAura(SPELL_NO_SUMMON_AURA, vehSummoner);
329 }
330 _events.ScheduleEvent(EVENT_NOSUMMON, 50000);
331 break;
332 default:
333 break;
334 }
335 }
336
337 DoMeleeAttackIfReady();
338 }
339
340 private:
341 EventMap _events;
342 ObjectGuid _tigerGuid;
343 };

345 CreatureAI* GetAI(Creature* creature) const override
346 {
347 return new npc_tiger_matriarchAI(creature);

358class npc_troll_volunteer : public CreatureScript
359{
360 public:
361 npc_troll_volunteer() : CreatureScript("npc_troll_volunteer") { }
362
363 struct npc_troll_volunteerAI : public ScriptedAI
364 {
365 npc_troll_volunteerAI(Creature* creature) : ScriptedAI(creature)
366 {
367 Initialize();
368 _mountModel = 0;
369 }
370
371 void Initialize()
372 {
373 _complete = false;
374 }
375
376 void InitializeAI() override
377 {
378 if (me->isDead() || !me->GetOwner())
379 return;
380
381 Reset();
382
383 switch (urand(0, 3))
384 {
385 case 0:
386 _mountModel = 6471;
387 break;
388 case 1:
389 _mountModel = 6473;
390 break;
391 case 2:
392 _mountModel = 6469;
393 break;
394 default:
395 _mountModel = 6472;
396 break;
397 }
398 me->SetDisplayId(trollmodel[urand(0, 39)]);
399 if (Player* player = me->GetOwner()->ToPlayer())
400 me->GetMotionMaster()->MoveFollow(player, 5.0f, float(rand_norm() + 1.0f) * float(M_PI) / 3.0f * 4.0f);
401 }
402
403 void Reset() override
404 {
405 Initialize();
406 me->AddAura(SPELL_VOLUNTEER_AURA, me);
407 me->AddAura(SPELL_MOUNTING_CHECK, me);
408 DoCast(me, SPELL_PETACT_AURA);
409 me->SetReactState(REACT_PASSIVE);
410 Talk(SAY_VOLUNTEER_START);
411 }
412
413 // This is needed for mount check aura to know what mountmodel the npc got stored
414 uint32 GetMountId()
415 {
416 return _mountModel;
417 }
418
419 void MovementInform(uint32 type, uint32 id) override
420 {
421 if (type != POINT_MOTION_TYPE)
422 return;
423 if (id == POINT_URUZIN)
424 me->DespawnOrUnsummon();
425 }
426
427 void SpellHit(Unit* caster, SpellInfo const* spell) override
428 {
429 if (spell->Id == SPELL_AOE_TURNIN && caster->GetEntry() == NPC_URUZIN && !_complete)
430 {
431 _complete = true; // Preventing from giving credit twice
432 DoCast(me, SPELL_TURNIN);
433 DoCast(me, SPELL_QUEST_CREDIT);
434 me->RemoveAurasDueToSpell(SPELL_MOUNTING_CHECK);
435 me->Dismount();
436 Talk(SAY_VOLUNTEER_END);
437 me->GetMotionMaster()->MovePoint(POINT_URUZIN, caster->GetPositionX(), caster->GetPositionY(), caster->GetPositionZ());
438 }
439 }
440
441 private:
442 uint32 _mountModel;
443 bool _complete;
444 };
445
446 CreatureAI* GetAI(Creature* creature) const override
447 {
448 return new npc_troll_volunteerAI(creature);
449 }

454class spell_mount_check : public SpellScriptLoader
455{
456 public:
457 spell_mount_check() : SpellScriptLoader("spell_mount_check") { }
458
459 class spell_mount_check_AuraScript : public AuraScript
460 {
461 PrepareAuraScript(spell_mount_check_AuraScript);
462
463 bool Validate(SpellInfo const* /*spellInfo*/) override
464 {
465 if (!sSpellMgr->GetSpellInfo(SPELL_MOUNTING_CHECK))
466 return false;
467 return true;
468 }
469
470 void HandleEffectPeriodic(AuraEffect const* /*aurEff*/)
471 {
472 Unit* target = GetTarget();
473 Unit* owner = target->GetOwner();
474
475 if (!owner)
476 return;
477
478 if (owner->IsMounted() && !target->IsMounted())
479 {
480 if (VolunteerAI* volunteerAI = CAST_AI(VolunteerAI, target->GetAI()))
481 target->Mount(volunteerAI->GetMountId());
482 }
483 else if (!owner->IsMounted() && target->IsMounted())
484 target->Dismount();
485
486 target->SetSpeed(MOVE_RUN, owner->GetSpeedRate(MOVE_RUN));
487 target->SetSpeed(MOVE_WALK, owner->GetSpeedRate(MOVE_WALK));
488 }
489
490 void Register() override
491 {
492 OnEffectPeriodic += AuraEffectPeriodicFn(spell_mount_check_AuraScript::HandleEffectPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
493 }
494 };
495
496 AuraScript* GetAuraScript() const override
497 {
498 return new spell_mount_check_AuraScript();
499 }

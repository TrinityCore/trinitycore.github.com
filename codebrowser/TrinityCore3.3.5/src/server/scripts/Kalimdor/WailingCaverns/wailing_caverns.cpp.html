86
87 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
88 {
89 player->PlayerTalkClass->ClearMenus();
90 InstanceScript* instance = creature->GetInstanceScript();
91 if (action == GOSSIP_ACTION_INFO_DEF + 1)
92 {
93 player->CLOSE_GOSSIP_MENU();
94 if (instance)
95 instance->SetData(TYPE_NARALEX_EVENT, IN_PROGRESS);
96
97 creature->AI()->Talk(SAY_MAKE_PREPARATIONS);
98
99 creature->setFaction(250);
100 creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
101
102 ENSURE_AI(npc_escortAI, (creature->AI()))->Start(false, false, player->GetGUID());
103 ENSURE_AI(npc_escortAI, (creature->AI()))->SetDespawnAtFar(false);
104 ENSURE_AI(npc_escortAI, (creature->AI()))->SetDespawnAtEnd(false);
105 }
106 return true;
107 }
108
109 bool OnGossipHello(Player* player, Creature* creature) override
110 {
111 InstanceScript* instance = creature->GetInstanceScript();
112
113 if (instance)
114 {
115 creature->CastSpell(player, SPELL_MARK_OF_THE_WILD_RANK_2, true);
116 if ((instance->GetData(TYPE_LORD_COBRAHN) == DONE) && (instance->GetData(TYPE_LORD_PYTHAS) == DONE) &&
117 (instance->GetData(TYPE_LADY_ANACONDRA) == DONE) && (instance->GetData(TYPE_LORD_SERPENTIS) == DONE))
118 {
119 player->ADD_GOSSIP_ITEM_DB(GOSSIP_OPTION_LET_EVENT_BEGIN, 0, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
120 player->SEND_GOSSIP_MENU(NPC_TEXT_FANGLORDS_ARE_DEAD, creature->GetGUID());
121
122 if (!instance->GetData(TYPE_NARALEX_YELLED))
123 {
124 creature->AI()->Talk(SAY_AT_LAST);
125 instance->SetData(TYPE_NARALEX_YELLED, 1);
126 }
127 }
128 else
129 {
130 player->SEND_GOSSIP_MENU(NPC_TEXT_NARALEX_SLEEPS_AGAIN, creature->GetGUID());
131 }
132 }
133 return true;
134 }

153 void WaypointReached(uint32 waypointId) override
154 {
155 switch (waypointId)
156 {
157 case 4:
158 eventProgress = 1;
159 currentEvent = TYPE_NARALEX_PART1;
160 instance->SetData(TYPE_NARALEX_PART1, IN_PROGRESS);
161 break;
162 case 5:
163 Talk(SAY_MUST_CONTINUE);
164 instance->SetData(TYPE_NARALEX_PART1, DONE);
165 break;
166 case 11:
167 eventProgress = 1;
168 currentEvent = TYPE_NARALEX_PART2;
169 instance->SetData(TYPE_NARALEX_PART2, IN_PROGRESS);
170 break;
171 case 19:
172 Talk(SAY_BEYOND_THIS_CORRIDOR);
173 break;
174 case 24:
175 eventProgress = 1;
176 currentEvent = TYPE_NARALEX_PART3;
177 instance->SetData(TYPE_NARALEX_PART3, IN_PROGRESS);
178 break;
179 }
180 }

205 void UpdateAI(uint32 diff) override
206 {
207 if (currentEvent != TYPE_NARALEX_PART3)
208 npc_escortAI::UpdateAI(diff);
209
210 if (eventTimer <= diff)
211 {
212 eventTimer = 0;
213 if (instance->GetData(currentEvent) == IN_PROGRESS)
214 {
215 switch (currentEvent)
216 {
217 case TYPE_NARALEX_PART1:
218 if (eventProgress == 1)
219 {
220 ++eventProgress;
221 Talk(SAY_TEMPLE_OF_PROMISE);
222 me->SummonCreature(NPC_DEVIATE_RAVAGER, -82.1763f, 227.874f, -93.3233f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000);
223 me->SummonCreature(NPC_DEVIATE_RAVAGER, -72.9506f, 216.645f, -93.6756f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000);
224 }
225 break;
226 case TYPE_NARALEX_PART2:
227 if (eventProgress == 1)
228 {
229 ++eventProgress;
230 Talk(SAY_BANISH_THE_SPIRITS);
231 DoCast(me, SPELL_SERPENTINE_CLEANSING);
232 //CAST_AI(npc_escort::npc_escortAI, me->AI())->SetCanDefend(false);
233 eventTimer = 30000;
234 me->SummonCreature(NPC_DEVIATE_VIPER, -61.5261f, 273.676f, -92.8442f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000);
235 me->SummonCreature(NPC_DEVIATE_VIPER, -58.4658f, 280.799f, -92.8393f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000);
236 me->SummonCreature(NPC_DEVIATE_VIPER, -50.002f, 278.578f, -92.8442f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000);
237 }
238 else
239 if (eventProgress == 2)
240 {
241 //CAST_AI(npc_escort::npc_escortAI, me->AI())->SetCanDefend(true);
242 Talk(SAY_CAVERNS_PURIFIED);
243 instance->SetData(TYPE_NARALEX_PART2, DONE);
244 if (me->HasAura(SPELL_SERPENTINE_CLEANSING))
245 me->RemoveAura(SPELL_SERPENTINE_CLEANSING);
246 }
247 break;
248 case TYPE_NARALEX_PART3:
249 if (eventProgress == 1)
250 {
251 ++eventProgress;
252 eventTimer = 4000;
253 me->SetStandState(UNIT_STAND_STATE_KNEEL);
254 Talk(SAY_EMERALD_DREAM);
255 }
256 else
257 if (eventProgress == 2)
258 {
259 ++eventProgress;
260 eventTimer = 15000;
261 //CAST_AI(npc_escort::npc_escortAI, me->AI())->SetCanDefend(false);
262 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
263 DoCast(naralex, SPELL_NARALEXS_AWAKENING, true);
264 Talk(EMOTE_AWAKENING_RITUAL);
265 }
266 else
267 if (eventProgress == 3)
268 {
269 ++eventProgress;
270 eventTimer = 15000;
271 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
272 naralex->AI()->Talk(EMOTE_TROUBLED_SLEEP);
273 me->SummonCreature(NPC_DEVIATE_MOCCASIN, 135.943f, 199.701f, -103.529f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);
274 me->SummonCreature(NPC_DEVIATE_MOCCASIN, 151.08f, 221.13f, -103.609f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);
275 me->SummonCreature(NPC_DEVIATE_MOCCASIN, 128.007f, 227.428f, -97.421f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);
276 }
277 else
278 if (eventProgress == 4)
279 {
280 ++eventProgress;
281 eventTimer = 30000;
282 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
283 naralex->AI()->Talk(EMOTE_WRITHE_IN_AGONY);
284 me->SummonCreature(NPC_NIGHTMARE_ECTOPLASM, 133.413f, 207.188f, -102.469f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);
285 me->SummonCreature(NPC_NIGHTMARE_ECTOPLASM, 142.857f, 218.645f, -102.905f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);
286 me->SummonCreature(NPC_NIGHTMARE_ECTOPLASM, 105.102f, 227.211f, -102.752f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);
287 me->SummonCreature(NPC_NIGHTMARE_ECTOPLASM, 153.372f, 235.149f, -102.826f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);

290 me->SummonCreature(NPC_NIGHTMARE_ECTOPLASM, 126.167f, 274.759f, -102.962f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000);
291 }
292 else
293 if (eventProgress == 5)
294 {
295 ++eventProgress;
296 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
297 naralex->AI()->Talk(EMOTE_HORRENDOUS_VISION);
298 me->SummonCreature(NPC_MUTANUS_THE_DEVOURER, 150.872f, 262.905f, -103.503f, 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
299 Talk(SAY_MUTANUS_THE_DEVOURER);

301 }
302 else
303 if (eventProgress == 6 && instance->GetData(TYPE_MUTANUS_THE_DEVOURER) == DONE)
304 {
305 ++eventProgress;
306 eventTimer = 3000;
307 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
308 {
309 if (me->HasAura(SPELL_NARALEXS_AWAKENING))
310 me->RemoveAura(SPELL_NARALEXS_AWAKENING);
311 naralex->SetStandState(UNIT_STAND_STATE_STAND);
312 naralex->AI()->Talk(SAY_I_AM_AWAKE);

315 }
316 else
317 if (eventProgress == 7)

319 ++eventProgress;
320 eventTimer = 6000;

322 naralex->AI()->Talk(SAY_THANK_YOU);
323 }
324 else
325 if (eventProgress == 8)

327 ++eventProgress;
328 eventTimer = 8000;
329 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))

331 naralex->AI()->Talk(SAY_FAREWELL);
332 naralex->AddAura(SPELL_FLIGHT_FORM, naralex);
333 }
334 SetRun();
335 me->SetStandState(UNIT_STAND_STATE_STAND);
336 me->AddAura(SPELL_FLIGHT_FORM, me);
337 }
338 else
339 if (eventProgress == 9)
340 {
341 ++eventProgress;
342 eventTimer = 1500;
343 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
344 naralex->GetMotionMaster()->MovePoint(25, naralex->GetPositionX(), naralex->GetPositionY(), naralex->GetPositionZ());
345 }
346 else
347 if (eventProgress == 10)
348 {
349 ++eventProgress;
350 eventTimer = 2500;
351 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
352 {
353 naralex->GetMotionMaster()->MovePoint(0, 117.095512f, 247.107971f, -96.167870f);
354 naralex->GetMotionMaster()->MovePoint(1, 90.388809f, 276.135406f, -83.389801f);
355 }
356 me->GetMotionMaster()->MovePoint(26, 117.095512f, 247.107971f, -96.167870f);
357 me->GetMotionMaster()->MovePoint(27, 144.375443f, 281.045837f, -82.477135f);
358 }
359 else
360 if (eventProgress == 11)
361 {
362 if (Creature* naralex = instance->instance->GetCreature(instance->GetGuidData(DATA_NARALEX)))
363 naralex->SetVisible(false);
364 me->SetVisible(false);
365 instance->SetData(TYPE_NARALEX_PART3, DONE);
366 }
367 break;
368 }
369 }
370 } else eventTimer -= diff;
371 }
372 };

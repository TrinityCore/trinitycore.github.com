65 //Eruption is a BFS graph problem
66 //One map to remember all floor, one map to keep floor that still need to erupt and one queue to know what needs to be removed

68 void OnCreatureCreate(Creature* creature) override
69 {
70 switch (creature->GetEntry())

72 case NPC_ONYXIA:

74 break;

77
78 void OnGameObjectCreate(GameObject* go) override
79 {
80 if ((go->GetGOInfo()->displayId == 4392 || go->GetGOInfo()->displayId == 4472) && go->GetGOInfo()->trap.spellId == 17731)
81 {
82 FloorEruptionGUID[0].insert(std::make_pair(go->GetGUID(), 0));
83 return;
84 }
85
86 switch (go->GetEntry())
87 {
88 case GO_WHELP_SPAWNER:
89 Position goPos = go->GetPosition();
90 if (Creature* temp = go->SummonCreature(NPC_WHELP, goPos, TEMPSUMMON_CORPSE_DESPAWN))
91 {
92 temp->SetInCombatWithZone();
93 ++manyWhelpsCounter;
94 }
95 break;
96 }
97 }
98
99 void OnGameObjectRemove(GameObject* go) override
100 {
101 if ((go->GetGOInfo()->displayId == 4392 || go->GetGOInfo()->displayId == 4472) && go->GetGOInfo()->trap.spellId == 17731)
102 {
103 FloorEruptionGUID[0].erase(go->GetGUID());
104 return;
105 }
106 }

108 void FloorEruption(ObjectGuid floorEruptedGUID)
109 {
110 if (GameObject* floorEruption = instance->GetGameObject(floorEruptedGUID))
111 {
112 //THIS GOB IS A TRAP - What shall i do? =(
113 //Cast it spell? Copyed Heigan method
114 floorEruption->SendCustomAnim(floorEruption->GetGoAnimProgress());
115 floorEruption->CastSpell(NULL, Difficulty(instance->GetSpawnMode()) == RAID_DIFFICULTY_10MAN_NORMAL ? 17731 : 69294); //pFloorEruption->GetGOInfo()->trap.spellId
116
117 //Get all immediatly nearby floors
118 std::list<GameObject*> nearFloorList;
119 Trinity::GameObjectInRangeCheck check(floorEruption->GetPositionX(), floorEruption->GetPositionY(), floorEruption->GetPositionZ(), 15);
120 Trinity::GameObjectListSearcher<Trinity::GameObjectInRangeCheck> searcher(floorEruption, nearFloorList, check);
121 floorEruption->VisitNearbyGridObject(999, searcher);
122 //remove all that are not present on FloorEruptionGUID[1] and update treeLen on each GUID
123 for (std::list<GameObject*>::const_iterator itr = nearFloorList.begin(); itr != nearFloorList.end(); ++itr)
124 {
125 if (((*itr)->GetGOInfo()->displayId == 4392 || (*itr)->GetGOInfo()->displayId == 4472) && (*itr)->GetGOInfo()->trap.spellId == 17731)
126 {
127 ObjectGuid nearFloorGUID = (*itr)->GetGUID();
128 if (FloorEruptionGUID[1].find(nearFloorGUID) != FloorEruptionGUID[1].end() && (*FloorEruptionGUID[1].find(nearFloorGUID)).second == 0)
129 {
130 (*FloorEruptionGUID[1].find(nearFloorGUID)).second = (*FloorEruptionGUID[1].find(floorEruptedGUID)).second+1;
131 FloorEruptionGUIDQueue.push(nearFloorGUID);
132 }
133 }
134 }
135 }
136 FloorEruptionGUID[1].erase(floorEruptedGUID);
137 }

140 {

142 return false;

155 {

202 void Update(uint32 diff) override
203 {
204 if (GetBossState(DATA_ONYXIA) == IN_PROGRESS)
205 {
206 if (onyxiaLiftoffTimer && onyxiaLiftoffTimer <= diff)
207 {
208 onyxiaLiftoffTimer = 0;
209 if (manyWhelpsCounter >= 50)
210 achievManyWhelpsHandleIt = true;
211 } else onyxiaLiftoffTimer -= diff;
212 }
213
214 if (!FloorEruptionGUIDQueue.empty())
215 {
216 if (eruptTimer <= diff)
217 {
218 ObjectGuid frontGuid = FloorEruptionGUIDQueue.front();
219 std::map<ObjectGuid, uint32>::iterator itr = FloorEruptionGUID[1].find(frontGuid);
220 if (itr != FloorEruptionGUID[1].end())
221 {
222 uint32 treeHeight = itr->second;
223
224 do
225 {
226 FloorEruption(frontGuid);
227 FloorEruptionGUIDQueue.pop();
228 if (FloorEruptionGUIDQueue.empty())
229 break;
230
231 frontGuid = FloorEruptionGUIDQueue.front();
232 itr = FloorEruptionGUID[1].find(frontGuid);
233 } while (itr != FloorEruptionGUID[1].end() && itr->second == treeHeight);
234 }
235
236 eruptTimer = 1000;
237 }
238 else
239 eruptTimer -= diff;
240 }
241 }

243 bool CheckAchievementCriteriaMeet(uint32 criteriaId, Player const* /*source*/, Unit const* /*target = NULL*/, uint32 /*miscValue1 = 0*/) override
244 {
245 switch (criteriaId)
246 {
247 case ACHIEV_CRITERIA_MANY_WHELPS_10_PLAYER: // Criteria for achievement 4403: Many Whelps! Handle It! (10 player) Hatch 50 eggs in 10s
248 case ACHIEV_CRITERIA_MANY_WHELPS_25_PLAYER: // Criteria for achievement 4406: Many Whelps! Handle It! (25 player) Hatch 50 eggs in 10s
249 return achievManyWhelpsHandleIt;
250 case ACHIEV_CRITERIA_DEEP_BREATH_10_PLAYER: // Criteria for achievement 4404: She Deep Breaths More (10 player) Everybody evade Deep Breath
251 case ACHIEV_CRITERIA_DEEP_BREATH_25_PLAYER: // Criteria for achievement 4407: She Deep Breaths More (25 player) Everybody evade Deep Breath
252 return achievSheDeepBreathMore;
253 }
254 return false;
255 }

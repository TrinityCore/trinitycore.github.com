238class npc_sironas : public CreatureScript

240public:
241 npc_sironas() : CreatureScript("npc_sironas") { }

247
248 struct npc_sironasAI : public ScriptedAI
249 {
250 npc_sironasAI(Creature* creature) : ScriptedAI(creature) { }
251
252 void Reset() override
253 {
254 _events.Reset();
255 me->SetDisplayId(me->GetCreatureTemplate()->Modelid2);
256 }
257
258 void EnterCombat(Unit* /*who*/) override
259 {
260 _events.ScheduleEvent(EVENT_UPPERCUT, 15 * IN_MILLISECONDS);
261 _events.ScheduleEvent(EVENT_IMMOLATE, 10 * IN_MILLISECONDS);
262 _events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, 5 * IN_MILLISECONDS);
263 }
264
265 void JustDied(Unit* killer) override
266 {
267 me->SetObjectScale(1.0f);
268 _events.Reset();
269 if (Creature* legoso = me->FindNearestCreature(NPC_LEGOSO, SIZE_OF_GRIDS))
270 {
271 Group* group = me->GetLootRecipientGroup();
272
273 if (killer->GetGUID() == legoso->GetGUID() ||
274 (group && group->IsMember(killer->GetGUID())) ||
275 killer->GetGUID().GetCounter() == legoso->AI()->GetData(DATA_EVENT_STARTER_GUID))
276 legoso->AI()->DoAction(ACTION_LEGOSO_SIRONAS_KILLED);
277 }
278 }
279
280 void UpdateAI(uint32 diff) override
281 {
282 if (!UpdateVictim())
283 return;
284
285 _events.Update(diff);
286
287 while (uint32 eventId = _events.ExecuteEvent())
288 {
289 switch (eventId)
290 {
291 case EVENT_UPPERCUT:
292 DoCastVictim(SPELL_UPPERCUT);
293 _events.ScheduleEvent(EVENT_UPPERCUT, urand(10, 12) * IN_MILLISECONDS);
294 break;
295 case EVENT_IMMOLATE:
296 DoCastVictim(SPELL_IMMOLATE);
297 _events.ScheduleEvent(EVENT_IMMOLATE, urand(15, 20) * IN_MILLISECONDS);
298 break;
299 case EVENT_CURSE_OF_BLOOD:
300 DoCastVictim(SPELL_CURSE_OF_BLOOD);
301 _events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, urand(20, 25) * IN_MILLISECONDS);
302 break;
303 default:
304 break;
305 }
306 }
307
308 DoMeleeAttackIfReady();
309 }
310
311 void DoAction(int32 param) override
312 {
313 switch (param)
314 {
315 case ACTION_SIRONAS_CHANNEL_START:
316 {
317 DoCast(me, SPELL_SIRONAS_CHANNELING);
318 std::list<Creature*> BeamList;
319 _beamGuidList.clear();
320 me->GetCreatureListWithEntryInGrid(BeamList, NPC_BLOODMYST_TESLA_COIL, SIZE_OF_GRIDS);
321 for (std::list<Creature*>::iterator itr = BeamList.begin(); itr != BeamList.end(); ++itr)
322 {
323 _beamGuidList.push_back((*itr)->GetGUID());
324 (*itr)->CastSpell(*itr, SPELL_BLOODMYST_TESLA);
325 }
326 break;
327 }
328 case ACTION_SIRONAS_CHANNEL_STOP:
329 {
330 me->InterruptNonMeleeSpells(true, SPELL_SIRONAS_CHANNELING);
331 std::list<Creature*> creatureList;
332 GetCreatureListWithEntryInGrid(creatureList, me, NPC_BLOODMYST_TESLA_COIL, 500.0f);
333 if (!creatureList.empty())
334 for (std::list<Creature*>::iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
335 (*itr)->InterruptNonMeleeSpells(true, SPELL_BLOODMYST_TESLA);
336 }
337 default:
338 break;
339 }
340 }
341
342 private:
343 GuidList _beamGuidList;
344 EventMap _events;
345 };

408 void Reset() override
409 {
410 Initialize();
411 me->SetCanDualWield(true);
412
413 _events.Reset();
414 _events.ScheduleEvent(EVENT_FROST_SHOCK, 1 * IN_MILLISECONDS);
415 _events.ScheduleEvent(EVENT_HEALING_SURGE, 5 * IN_MILLISECONDS);
416 _events.ScheduleEvent(EVENT_SEARING_TOTEM, 15 * IN_MILLISECONDS);
417 _events.ScheduleEvent(EVENT_STRENGTH_OF_EARTH_TOTEM, 20 * IN_MILLISECONDS);
418 }
419
420 void UpdateAI(uint32 diff) override
421 {
422 _events.Update(diff);
423
424 if (UpdateVictim())
425 {
426 while (uint32 eventId = _events.ExecuteEvent())
427 {
428 switch (eventId)
429 {
430 case EVENT_FROST_SHOCK:
431 DoCastVictim(SPELL_FROST_SHOCK);
432 _events.DelayEvents(1 * IN_MILLISECONDS);
433 _events.ScheduleEvent(EVENT_FROST_SHOCK, urand(10, 15) * IN_MILLISECONDS);
434 break;
435 case EVENT_SEARING_TOTEM:
436 DoCast(me, SPELL_SEARING_TOTEM);
437 _events.DelayEvents(1 * IN_MILLISECONDS);
438 _events.ScheduleEvent(EVENT_SEARING_TOTEM, urand(110, 130) * IN_MILLISECONDS);
439 break;
440 case EVENT_STRENGTH_OF_EARTH_TOTEM:
441 DoCast(me, SPELL_STRENGTH_OF_EARTH_TOTEM);
442 _events.DelayEvents(1 * IN_MILLISECONDS);
443 _events.ScheduleEvent(EVENT_STRENGTH_OF_EARTH_TOTEM, urand(110, 130) * IN_MILLISECONDS);
444 break;
445 case EVENT_HEALING_SURGE:
446 {
447 Unit* target = NULL;
448 if (me->GetHealthPct() < 85)
449 target = me;
450 else if (Player* player = GetPlayerForEscort())
451 if (player->GetHealthPct() < 85)
452 target = player;
453 if (target)
454 {
455 DoCast(target, SPELL_HEALING_SURGE);
456 _events.ScheduleEvent(EVENT_HEALING_SURGE, 10 * IN_MILLISECONDS);
457 }
458 else
459 _events.ScheduleEvent(EVENT_HEALING_SURGE, 2 * IN_MILLISECONDS);
460 break;
461 }
462 default:
463 break;
464 }
465 }
466
467 DoMeleeAttackIfReady();
468 }
469
470 if (HasEscortState(STATE_ESCORT_NONE))
471 return;
472
473 npc_escortAI::UpdateAI(diff);
474
475 if (_phase)
476 {
477 if (_moveTimer <= diff)
478 {
479 switch (_phase)
480 {
481 case PHASE_WP_26: //debug skip path to point 26, buggy path calculation
482 me->GetMotionMaster()->MovePoint(WP_DEBUG_2, -2021.77f, -10648.8f, 129.903f, false);
483 _moveTimer = 2 * IN_MILLISECONDS;
484 _phase = PHASE_CONTINUE;
485 break;
486 case PHASE_CONTINUE: // continue escort
487 SetEscortPaused(false);
488 _moveTimer = 0 * IN_MILLISECONDS;
489 _phase = PHASE_NONE;
490 break;
491 case PHASE_WP_22: //debug skip path to point 22, buggy path calculation
492 me->GetMotionMaster()->MovePoint(WP_EXPLOSIVES_FIRST_PLANT, -1958.026f, -10660.465f, 111.547f, false);
493 Talk(SAY_LEGOSO_3);
494 _moveTimer = 2 * IN_MILLISECONDS;
495 _phase = PHASE_PLANT_FIRST_KNEEL;
496 break;
497 case PHASE_PLANT_FIRST_KNEEL: // plant first explosives stage 1 kneel
498 me->SetStandState(UNIT_STAND_STATE_KNEEL);
499 _moveTimer = 10 * IN_MILLISECONDS;
500 _phase = PHASE_PLANT_FIRST_STAND;
501 break;
502 case PHASE_PLANT_FIRST_STAND: // plant first explosives stage 1 stand
503 me->SetStandState(UNIT_STAND_STATE_STAND);
504 _moveTimer = 0.5* IN_MILLISECONDS;
505 _phase = PHASE_PLANT_FIRST_WORK;
506 break;
507 case PHASE_PLANT_FIRST_WORK: // plant first explosives stage 2 work
508 Talk(SAY_LEGOSO_4);
509 _moveTimer = 17.5 * IN_MILLISECONDS;
510 _phase = PHASE_PLANT_FIRST_FINISH;
511 break;
512 case PHASE_PLANT_FIRST_FINISH: // plant first explosives finish
513 _explosivesGuids.clear();
514 for (uint8 i = 0; i != MAX_EXPLOSIVES; ++i)
515 {

517 _explosivesGuids.push_back(explosive->GetGUID());
518 }
519 me->HandleEmoteCommand(EMOTE_ONESHOT_NONE); // reset anim state

617 _phase = PHASE_PLANT_SECOND_WAIT;
618 break;
619 case PHASE_PLANT_SECOND_WAIT: // plant second explosives finish - proceed to next point
620 _moveTimer = 1 * IN_MILLISECONDS;
621 _phase = PHASE_CONTINUE;

623 case PHASE_PLANT_SECOND_TIMER_1: // second explosives detonate timer 1
624 Talk(SAY_LEGOSO_16);
625 _moveTimer = 1 * IN_MILLISECONDS;
626 _phase = PHASE_PLANT_SECOND_TIMER_2;
627 break;
628 case PHASE_PLANT_SECOND_TIMER_2: // second explosives detonate timer 2
629 Talk(SAY_LEGOSO_17);
630 _moveTimer = 1 * IN_MILLISECONDS;
631 _phase = PHASE_PLANT_SECOND_TIMER_3;
632 break;
633 case PHASE_PLANT_SECOND_TIMER_3: // second explosives detonate timer 3
634 Talk(SAY_LEGOSO_18);
635 _moveTimer = 1 * IN_MILLISECONDS;
636 _phase = PHASE_PLANT_SECOND_DETONATE;
637 break;
638 case PHASE_PLANT_SECOND_DETONATE: // second explosives detonate finish
639 for (GuidList::iterator itr = _explosivesGuids.begin(); itr != _explosivesGuids.end(); ++itr)
640 {
641 if (GameObject* explosive = ObjectAccessor::GetGameObject(*me, *itr))
642 me->RemoveGameObject(explosive, true);
643 }
644 _explosivesGuids.clear();
645 if (Creature* sironas = me->FindNearestCreature(NPC_SIRONAS, SIZE_OF_GRIDS))
646 {
647 sironas->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
648 me->SetFacingToObject(sironas);
649 }

651 _phase = PHASE_FIGHT_SIRONAS_STOP;
652 break;
653 case PHASE_FIGHT_SIRONAS_STOP: // sironas channel stop
654 if (Creature* sironas = me->FindNearestCreature(NPC_SIRONAS, SIZE_OF_GRIDS))
655 sironas->AI()->DoAction(ACTION_SIRONAS_CHANNEL_STOP);
656 _moveTimer = 1 * IN_MILLISECONDS;
657 _phase = PHASE_FIGHT_SIRONAS_SPEECH_1;
658 break;
659 case PHASE_FIGHT_SIRONAS_SPEECH_1: // sironas exclamation before aggro
660 if (Creature* sironas = me->FindNearestCreature(NPC_SIRONAS, SIZE_OF_GRIDS))
661 sironas->AI()->Talk(SAY_SIRONAS_1);
662 _moveTimer = 1 * IN_MILLISECONDS;
663 _phase = PHASE_FIGHT_SIRONAS_SPEECH_2;
664 break;
665 case PHASE_FIGHT_SIRONAS_SPEECH_2: // legoso exclamation before aggro
666 if (Creature* sironas = me->FindNearestCreature(NPC_SIRONAS, SIZE_OF_GRIDS))
667 sironas->SetObjectScale(3.0f);
668 Talk(SAY_LEGOSO_19);
669 _moveTimer = 1 * IN_MILLISECONDS;
670 _phase = PHASE_FIGHT_SIRONAS_START;
671 break;
672 case PHASE_FIGHT_SIRONAS_START: // legoso exclamation at aggro
673 if (Creature* sironas = me->FindNearestCreature(NPC_SIRONAS, SIZE_OF_GRIDS))
674 {
675 Unit* target = GetPlayerForEscort();
676 if (!target)
677 target = me;
678
679 target->AddThreat(sironas, 0.001f);
680 sironas->Attack(target, true);
681 sironas->GetMotionMaster()->MoveChase(target);
682 }
683 _moveTimer = 10 * IN_MILLISECONDS;
684 _phase = PHASE_CONTINUE;
685 break;
686 case PHASE_SIRONAS_SLAIN_SPEECH_1: // legoso exclamation after battle - stage 1.1
687 Talk(SAY_LEGOSO_20);
688 _moveTimer = 2 * IN_MILLISECONDS;
689 _phase = PHASE_SIRONAS_SLAIN_EMOTE_1;
690 break;
691 case PHASE_SIRONAS_SLAIN_EMOTE_1: // legoso exclamation after battle - stage 1.2
692 me->HandleEmoteCommand(EMOTE_ONESHOT_EXCLAMATION);
693 _moveTimer = 2 * IN_MILLISECONDS;
694 _phase = PHASE_SIRONAS_SLAIN_EMOTE_2;
695 break;
696 case PHASE_SIRONAS_SLAIN_EMOTE_2: // legoso exclamation after battle - stage 1.3
697 if (Player* player = GetPlayerForEscort())
698 player->GroupEventHappens(QUEST_ENDING_THEIR_WORLD, me);
699 me->HandleEmoteCommand(EMOTE_ONESHOT_CHEER);
700 _moveTimer = 5 * IN_MILLISECONDS;
701 _phase = PHASE_SIRONAS_SLAIN_SPEECH_2;
702 break;
703 case PHASE_SIRONAS_SLAIN_SPEECH_2: // legoso exclamation after battle - stage 2
704 Talk(SAY_LEGOSO_21);
705 _moveTimer = 30 * IN_MILLISECONDS;
706 _phase = PHASE_CONTINUE;
707 break;
708 default:
709 break;
710 }
711 }
712 else if (!me->IsInCombat())
713 _moveTimer -= diff;

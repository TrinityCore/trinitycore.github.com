22DoorData const doorData[] =
23{
24 { GO_MAULGAR_DOOR, DATA_MAULGAR, DOOR_TYPE_PASSAGE },
25 { GO_GRUUL_DOOR, DATA_GRUUL, DOOR_TYPE_ROOM },
26 { 0, 0, DOOR_TYPE_ROOM } // END
27};

29MinionData const minionData[] =
30{
31 { NPC_MAULGAR, DATA_MAULGAR },
32 { NPC_KROSH_FIREHAND, DATA_MAULGAR },
33 { NPC_OLM_THE_SUMMONER, DATA_MAULGAR },
34 { NPC_KIGGLER_THE_CRAZED, DATA_MAULGAR },
35 { NPC_BLINDEYE_THE_SEER, DATA_MAULGAR }

37
38class instance_gruuls_lair : public InstanceMapScript
39{
40 public:
41 instance_gruuls_lair() : InstanceMapScript(GLScriptName, 565) { }
42
43 struct instance_gruuls_lair_InstanceMapScript : public InstanceScript
44 {
45 instance_gruuls_lair_InstanceMapScript(Map* map) : InstanceScript(map)
46 {
47 SetHeaders(DataHeader);
48 SetBossNumber(EncounterCount);
49 LoadDoorData(doorData);
50 LoadMinionData(minionData);
51 }
52
53 void OnCreatureCreate(Creature* creature) override
54 {
55 switch (creature->GetEntry())
56 {
57 case NPC_MAULGAR:
58 MaulgarGUID = creature->GetGUID();
59 // no break;
60 case NPC_KROSH_FIREHAND:
61 case NPC_OLM_THE_SUMMONER:
62 case NPC_KIGGLER_THE_CRAZED:
63 case NPC_BLINDEYE_THE_SEER:
64 AddMinion(creature, true);
65 break;
66 default:
67 break;
68 }
69 }
70
71 void OnCreatureRemove(Creature* creature) override
72 {
73 switch (creature->GetEntry())
74 {
75 case NPC_MAULGAR:
76 case NPC_KROSH_FIREHAND:
77 case NPC_OLM_THE_SUMMONER:
78 case NPC_KIGGLER_THE_CRAZED:
79 case NPC_BLINDEYE_THE_SEER:
80 AddMinion(creature, false);
81 break;
82 default:
83 break;
84 }
85 }
86
87 void OnGameObjectCreate(GameObject* go) override
88 {
89 switch (go->GetEntry())
90 {
91 case GO_MAULGAR_DOOR:
92 case GO_GRUUL_DOOR:
93 AddDoor(go, true);
94 break;
95 default:
96 break;
97 }
98 }
99
100 void OnGameObjectRemove(GameObject* go) override
101 {
102 switch (go->GetEntry())
103 {
104 case GO_MAULGAR_DOOR:
105 case GO_GRUUL_DOOR:
106 AddDoor(go, false);
107 break;
108 default:
109 break;
110 }
111 }
112
113 ObjectGuid GetGuidData(uint32 type) const override
114 {
115 switch (type)
116 {
117 case DATA_MAULGAR:
118 return MaulgarGUID;
119 default:
120 break;
121 }
122 return ObjectGuid::Empty;
123 }
124
125 protected:
126 ObjectGuid MaulgarGUID;
127 };
128
129 InstanceScript* GetInstanceScript(InstanceMap* map) const override
130 {
131 return new instance_gruuls_lair_InstanceMapScript(map);
132 }
133};
134
135void AddSC_instance_gruuls_lair()
136{
137 new instance_gruuls_lair();
138}

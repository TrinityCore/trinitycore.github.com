124 void SpellHitTarget(Unit* target, const SpellInfo* pSpell) override
125 {
126 //This to emulate effect1 (77) of SPELL_GROUND_SLAM, knock back to any direction
127 //It's initially wrong, since this will cause fall damage, which is by comments, not intended.
128 if (pSpell->Id == SPELL_GROUND_SLAM)
129 {
130 if (target->GetTypeId() == TYPEID_PLAYER)
131 {
132 switch (urand(0, 1))
133 {
134 case 0:
135 target->CastSpell(target, SPELL_MAGNETIC_PULL, true, NULL, NULL, me->GetGUID());
136 break;
137
138 case 1:
139 target->CastSpell(target, SPELL_KNOCK_BACK, true, NULL, NULL, me->GetGUID());
140 break;
141 }
142 }
143 }
144
145 //this part should be in the core
146 if (pSpell->Id == SPELL_SHATTER)
147 {
148 /// @todo use eventmap to kill this stuff
149 //clear this, if we are still performing
150 if (m_bPerformingGroundSlam)
151 {
152 m_bPerformingGroundSlam = false;
153
154 //and correct movement, if not already
155 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() != CHASE_MOTION_TYPE)
156 {
157 if (me->GetVictim())
158 me->GetMotionMaster()->MoveChase(me->GetVictim());
159 }
160 }
161 }
162 }
163
164 void UpdateAI(uint32 diff) override
165 {
166 if (!UpdateVictim())
167 return;
168
169 events.Update(diff);
170
171 if (me->HasUnitState(UNIT_STATE_CASTING))
172 return;
173
174 /// @todo: convert this shit to eventmap
175
176 // Growth
177 // Gruul can cast this spell up to 30 times
178 if (m_uiGrowth_Timer <= diff)
179 {
180 Talk(EMOTE_GROW);
181 DoCast(me, SPELL_GROWTH);
182 m_uiGrowth_Timer = 30000;
183 }
184 else
185 m_uiGrowth_Timer -= diff;
186
187 if (m_bPerformingGroundSlam)
188 {
189 if (m_uiGroundSlamTimer <= diff)
190 {
191 m_uiGroundSlamTimer =120000;
192 m_uiHurtfulStrike_Timer= 8000;
193
194 if (m_uiReverberation_Timer < 10000) //Give a little time to the players to undo the damage from shatter
195 m_uiReverberation_Timer += 10000;
196
197 DoCast(me, SPELL_SHATTER);
198 }
199 else
200 m_uiGroundSlamTimer -= diff;
201 }
202 else
203 {
204 // Hurtful Strike
205 if (m_uiHurtfulStrike_Timer <= diff)
206 {
207 Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 1);
208
209 if (target && me->IsWithinMeleeRange(me->GetVictim()))
210 DoCast(target, SPELL_HURTFUL_STRIKE);
211 else
212 DoCastVictim(SPELL_HURTFUL_STRIKE);
213
214 m_uiHurtfulStrike_Timer= 8000;
215 }
216 else
217 m_uiHurtfulStrike_Timer -= diff;
218
219 // Reverberation
220 if (m_uiReverberation_Timer <= diff)
221 {
222 DoCastVictim(SPELL_REVERBERATION, true);
223 m_uiReverberation_Timer = urand(15000, 25000);
224 }
225 else
226 m_uiReverberation_Timer -= diff;
227
228 // Cave In
229 if (m_uiCaveIn_Timer <= diff)
230 {
231 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
232 DoCast(target, SPELL_CAVE_IN);
233
234 if (m_uiCaveIn_StaticTimer >= 4000)
235 m_uiCaveIn_StaticTimer -= 2000;
236
237 m_uiCaveIn_Timer = m_uiCaveIn_StaticTimer;
238 }
239 else
240 m_uiCaveIn_Timer -= diff;
241
242 // Ground Slam, Gronn Lord's Grasp, Stoned, Shatter
243 if (m_uiGroundSlamTimer <= diff)
244 {
245 me->GetMotionMaster()->Clear();
246 me->GetMotionMaster()->MoveIdle();
247
248 m_bPerformingGroundSlam= true;
249 m_uiGroundSlamTimer = 10000;
250
251 DoCast(me, SPELL_GROUND_SLAM);
252 }
253 else
254 m_uiGroundSlamTimer -= diff;
255
256 DoMeleeAttackIfReady();
257 }
258 }
259 };

267class spell_gruul_shatter : public SpellScriptLoader
268{
269 public:
270 spell_gruul_shatter() : SpellScriptLoader("spell_gruul_shatter") { }
271
272 class spell_gruul_shatter_SpellScript : public SpellScript
273 {
274 PrepareSpellScript(spell_gruul_shatter_SpellScript);
275
276 bool Validate(SpellInfo const* /*spell*/) override
277 {
278 if (!sSpellMgr->GetSpellInfo(SPELL_STONED))
279 return false;
280 if (!sSpellMgr->GetSpellInfo(SPELL_SHATTER_EFFECT))
281 return false;
282 return true;
283 }
284
285 void HandleScript(SpellEffIndex /*effIndex*/)
286 {
287 if (Unit* target = GetHitUnit())
288 {
289 target->RemoveAurasDueToSpell(SPELL_STONED);
290 target->CastSpell((Unit*)NULL, SPELL_SHATTER_EFFECT, true);
291 }
292 }
293
294 void Register() override
295 {
296 OnEffectHitTarget += SpellEffectFn(spell_gruul_shatter_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
297 }
298 };
299
300 SpellScript* GetSpellScript() const override
301 {
302 return new spell_gruul_shatter_SpellScript();
303 }

306class spell_gruul_shatter_effect : public SpellScriptLoader

308 public:
309 spell_gruul_shatter_effect() : SpellScriptLoader("spell_gruul_shatter_effect") { }
310
311 class spell_gruul_shatter_effect_SpellScript : public SpellScript
312 {
313 PrepareSpellScript(spell_gruul_shatter_effect_SpellScript);
314
315 void CalculateDamage()
316 {
317 if (!GetHitUnit())
318 return;
319
320 float radius = GetSpellInfo()->Effects[EFFECT_0].CalcRadius(GetCaster());
321 if (!radius)
322 return;
323
324 float distance = GetCaster()->GetDistance2d(GetHitUnit());
325 if (distance > 1.0f)
326 SetHitDamage(int32(GetHitDamage() * ((radius - distance) / radius)));
327 }
328
329 void Register() override
330 {
331 OnHit += SpellHitFn(spell_gruul_shatter_effect_SpellScript::CalculateDamage);
332 }
333 };
334
335 SpellScript* GetSpellScript() const override
336 {
337 return new spell_gruul_shatter_effect_SpellScript();
338 }

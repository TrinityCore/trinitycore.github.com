99 struct npc_abyssalAI : public ScriptedAI
100 {
101 npc_abyssalAI(Creature* creature) : ScriptedAI(creature)
102 {
103 Initialize();
104 trigger = 0;
105 Despawn_Timer = 60000;
106 }
107
108 void Initialize()
109 {
110 FireBlast_Timer = 6000;
111 }

113 uint32 FireBlast_Timer;
114 uint32 Despawn_Timer;
115 uint32 trigger;
116
117 void Reset() override
118 {
119 Initialize();
120 }
121
122 void SpellHit(Unit*, const SpellInfo* spell) override
123 {
124 if (trigger == 2 && spell->Id == SPELL_BLAZE_TARGET)
125 {
126 DoCast(me, SPELL_BLAZE_TRAP, true);
127 me->SetVisible(false);
128 Despawn_Timer = 130000;
129 }
130 }
131
132 void SetTrigger(uint32 _trigger)
133 {
134 trigger = _trigger;
135 me->SetDisplayId(11686);
136 if (trigger == 1) //debris
137 {
138 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
139 DoCast(me, SPELL_DEBRIS_VISUAL, true);
140 FireBlast_Timer = 5000;
141 Despawn_Timer = 10000;
142 }
143 }
144
145 void EnterCombat(Unit* /*who*/) override
146 {
147 DoZoneInCombat();
148 }
149 void AttackStart(Unit* who) override
150 {
151 if (!trigger)
152 ScriptedAI::AttackStart(who);
153 }
154 void MoveInLineOfSight(Unit* who) override
155
156 {
157 if (!trigger)
158 ScriptedAI::MoveInLineOfSight(who);
159 }
160
161 void UpdateAI(uint32 diff) override
162 {
163 if (trigger)
164 {
165 if (trigger == 1)
166 {
167 if (FireBlast_Timer <= diff)
168 {
169 DoCast(me, SPELL_DEBRIS_DAMAGE, true);
170 trigger = 3;
171 }
172 else FireBlast_Timer -= diff;
173 }
174 return;
175 }
176
177 if (Despawn_Timer <= diff)
178 {
179 me->DespawnOrUnsummon();
180 }
181 else Despawn_Timer -= diff;
182
183 if (!UpdateVictim())
184 return;
185
186 if (FireBlast_Timer <= diff)
187 {
188 DoCastVictim(SPELL_FIRE_BLAST);
189 FireBlast_Timer = 5000 + rand32() % 10000;
190 }
191 else FireBlast_Timer -= diff;
192
193 DoMeleeAttackIfReady();
194 }
195 };
196
197 CreatureAI* GetAI(Creature* creature) const override
198 {
199 return new npc_abyssalAI(creature);
200 }

214 boss_magtheridonAI(Creature* creature) : ScriptedAI(creature)
215 {
216 Initialize();
217 instance = creature->GetInstanceScript();
218 me->SetFloatValue(UNIT_FIELD_BOUNDINGRADIUS, 10);
219 me->SetFloatValue(UNIT_FIELD_COMBATREACH, 10);
220 }
221
222 void Initialize()

224 Berserk_Timer = 1320000;
225 Quake_Timer = 40000;
226 Debris_Timer = 10000;
227 Blaze_Timer = 10000 + rand32() % 20000;
228 BlastNova_Timer = 60000;
229 Cleave_Timer = 15000;
230 RandChat_Timer = 90000;
231
232 Phase3 = false;
233 NeedCheckCube = false;
234 }

236 CubeMap Cube;
237
238 InstanceScript* instance;
239
240 uint32 Berserk_Timer;
241 uint32 Quake_Timer;
242 uint32 Cleave_Timer;
243 uint32 BlastNova_Timer;
244 uint32 Blaze_Timer;
245 uint32 Debris_Timer;
246 uint32 RandChat_Timer;
247
248 bool Phase3;
249 bool NeedCheckCube;
250
251 void Reset() override
252 {
253 Initialize();
254
255 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
256 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
257 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
258 me->AddUnitState(UNIT_STATE_STUNNED);
259 DoCast(me, SPELL_SHADOW_CAGE_C, true);
260 }
261
262 void JustReachedHome() override
263 {
264 instance->SetData(DATA_MAGTHERIDON_EVENT, NOT_STARTED);
265 instance->SetData(DATA_COLLAPSE, false);
266 }
267
268 void SetClicker(ObjectGuid cubeGUID, ObjectGuid clickerGUID)
269 {
270 // to avoid multiclicks from 1 cube
271 if (ObjectGuid guid = Cube[cubeGUID])
272 DebuffClicker(ObjectAccessor::GetUnit(*me, guid));
273 Cube[cubeGUID] = clickerGUID;
274 NeedCheckCube = true;
275 }
276
277 //function to interrupt channeling and debuff clicker with mind exh(used if second person clicks with same cube or after dispeling/ending shadow grasp DoT)
278 void DebuffClicker(Unit* clicker)
279 {
280 if (!clicker || !clicker->IsAlive())
281 return;
282
283 clicker->RemoveAurasDueToSpell(SPELL_SHADOW_GRASP); // cannot interrupt triggered spells
284 clicker->InterruptNonMeleeSpells(false);
285 clicker->CastSpell(clicker, SPELL_MIND_EXHAUSTION, true);
286 }
287
288 void NeedCheckCubeStatus()
289 {
290 uint32 ClickerNum = 0;
291 // now checking if every clicker has debuff from manticron(it is dispelable atm rev 6110 : S)
292 // if not - apply mind exhaustion and delete from clicker's list
293 for (CubeMap::iterator i = Cube.begin(); i != Cube.end(); ++i)
294 {
295 Unit* clicker = ObjectAccessor::GetUnit(*me, (*i).second);
296 if (!clicker || !clicker->HasAura(SPELL_SHADOW_GRASP))
297 {
298 DebuffClicker(clicker);
299 (*i).second.Clear();
300 }
301 else
302 ++ClickerNum;
303 }
304
305 // if 5 clickers from other cubes apply shadow cage
306 if (ClickerNum >= CLICKERS_COUNT && !me->HasAura(SPELL_SHADOW_CAGE))
307 {
308 Talk(SAY_BANISH);
309 DoCast(me, SPELL_SHADOW_CAGE, true);
310 }
311 else
312 if (ClickerNum < CLICKERS_COUNT && me->HasAura(SPELL_SHADOW_CAGE))
313 me->RemoveAurasDueToSpell(SPELL_SHADOW_CAGE);
314
315 if (!ClickerNum)
316 NeedCheckCube = false;
317 }

342 instance->SetData(DATA_MAGTHERIDON_EVENT, IN_PROGRESS);
343 DoZoneInCombat();
344
345 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
346 me->RemoveAurasDueToSpell(SPELL_SHADOW_CAGE_C);
347
348 Talk(SAY_FREED);
349 }
350
351 void UpdateAI(uint32 diff) override
352 {
353 if (!me->IsInCombat())
354 {
355 if (RandChat_Timer <= diff)
356 {
357 Talk(SAY_TAUNT);
358 RandChat_Timer = 90000;
359 }
360 else
361 RandChat_Timer -= diff;
362 }
363
364 if (!UpdateVictim())
365 return;
366
367 if (NeedCheckCube) NeedCheckCubeStatus();
368
369 if (Berserk_Timer <= diff)
370 {
371 DoCast(me, SPELL_BERSERK, true);
372 Talk(EMOTE_BERSERK);
373 Berserk_Timer = 60000;
374 }
375 else
376 Berserk_Timer -= diff;
377
378 if (Cleave_Timer <= diff)
379 {
380 DoCastVictim(SPELL_CLEAVE);
381 Cleave_Timer = 10000;
382 }
383 else
384 Cleave_Timer -= diff;
385
386 if (BlastNova_Timer <= diff)
387 {
388 // to avoid earthquake interruption
389 if (!me->HasUnitState(UNIT_STATE_STUNNED))
390 {
391 Talk(EMOTE_BLASTNOVA);
392 DoCast(me, SPELL_BLASTNOVA);
393 BlastNova_Timer = 60000;
394 }
395 }
396 else
397 BlastNova_Timer -= diff;
398
399 if (Quake_Timer <= diff)
400 {
401 // to avoid blastnova interruption
402 if (!me->IsNonMeleeSpellCast(false))
403 {
404 DoCast(me, SPELL_QUAKE_TRIGGER, true);
405 Quake_Timer = 50000;
406 }
407 }
408 else
409 Quake_Timer -= diff;
410
411 if (Blaze_Timer <= diff)
412 {
413 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
414 {
415 float x, y, z;
416 target->GetPosition(x, y, z);
417 Creature* summon = me->SummonCreature(NPC_ABYSSAL, x, y, z, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
418 if (summon)
419 {
420 ENSURE_AI(npc_abyssal::npc_abyssalAI, summon->AI())->SetTrigger(2);
421 DoCast(summon, SPELL_BLAZE_TARGET, true);
422 summon->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
423 }
424 }
425 Blaze_Timer = urand(20000, 40000);
426 }
427 else
428 Blaze_Timer -= diff;
429
430 if (!Phase3 && HealthBelowPct(30)
431 && !me->IsNonMeleeSpellCast(false) // blast nova
432 && !me->HasUnitState(UNIT_STATE_STUNNED)) // shadow cage and earthquake
433 {
434 Phase3 = true;
435 Talk(SAY_CHAMBER_DESTROY);
436 DoCast(me, SPELL_CAMERA_SHAKE, true);
437 DoCast(me, SPELL_DEBRIS_KNOCKDOWN, true);
438
439 instance->SetData(DATA_COLLAPSE, true);
440 }
441
442 if (Phase3)
443 {
444 if (Debris_Timer <= diff)
445 {
446 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
447 {
448 float x, y, z;
449 target->GetPosition(x, y, z);
450 Creature* summon = me->SummonCreature(NPC_ABYSSAL, x, y, z, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
451 if (summon)
452 ENSURE_AI(npc_abyssal::npc_abyssalAI, summon->AI())->SetTrigger(1);
453 }
454 Debris_Timer = 10000;
455 }
456 else
457 Debris_Timer -= diff;
458 }
459
460 DoMeleeAttackIfReady();
461 }
462 };

466 return GetInstanceAI<boss_magtheridonAI>(creature);
467 }

593class go_manticron_cube : public GameObjectScript
594{
595public:
596 go_manticron_cube() : GameObjectScript("go_manticron_cube")
597 {
598 }
599
600 bool OnGossipHello(Player* player, GameObject* go) override
601 {
602 InstanceScript* instance = go->GetInstanceScript();
603
604 if (!instance)
605 return true;
606
607 if (instance->GetData(DATA_MAGTHERIDON_EVENT) != IN_PROGRESS)
608 return true;
609 Creature* Magtheridon =ObjectAccessor::GetCreature(*go, instance->GetGuidData(DATA_MAGTHERIDON));
610 if (!Magtheridon || !Magtheridon->IsAlive())
611 return true;
612
613 // if exhausted or already channeling return
614 if (player->HasAura(SPELL_MIND_EXHAUSTION) || player->HasAura(SPELL_SHADOW_GRASP))
615 return true;
616
617 player->InterruptNonMeleeSpells(false);
618 player->CastSpell(player, SPELL_SHADOW_GRASP, true);
619 player->CastSpell(player, SPELL_SHADOW_GRASP_VISUAL, false);
620 ENSURE_AI(boss_magtheridon::boss_magtheridonAI, Magtheridon->AI())->SetClicker(go->GetGUID(), player->GetGUID());
621 return true;
622 }
623};

54class boss_watchkeeper_gargolmar : public CreatureScript
55{
56 public:
57 boss_watchkeeper_gargolmar() : CreatureScript("boss_watchkeeper_gargolmar") { }
58
59 struct boss_watchkeeper_gargolmarAI : public BossAI
60 {
61 boss_watchkeeper_gargolmarAI(Creature* creature) : BossAI(creature, DATA_WATCHKEEPER_GARGOLMAR)
62 {
63 Initialize();
64 }
65
66 void Initialize()
67 {
68 hasTaunted = false;
69 yelledForHeal = false;
70 retaliation = false;
71 }
72
73 void Reset() override
74 {
75 Initialize();
76 _Reset();
77 }
78
79 void EnterCombat(Unit* /*who*/) override
80 {
81 Talk(SAY_AGGRO);
82 events.ScheduleEvent(EVENT_MORTAL_WOUND, 5000);
83 events.ScheduleEvent(EVENT_SURGE, 4000);
84 _EnterCombat();
85 }
86
87 void MoveInLineOfSight(Unit* who) override
88 {
89 if (!me->GetVictim() && me->CanCreatureAttack(who))
90 {
91 if (!me->CanFly() && me->GetDistanceZ(who) > CREATURE_Z_ATTACK_RANGE)
92 return;
93
94 float attackRadius = me->GetAttackDistance(who);
95 if (me->IsWithinDistInMap(who, attackRadius) && me->IsWithinLOSInMap(who))
96 {
97 //who->RemoveSpellsCausingAura(SPELL_AURA_MOD_STEALTH);
98 AttackStart(who);
99 }
100 else if (!hasTaunted && me->IsWithinDistInMap(who, 60.0f))
101 {
102 Talk(SAY_TAUNT);
103 hasTaunted = true;
104 }
105 }
106 }
107
108 void KilledUnit(Unit* /*victim*/) override
109 {
110 Talk(SAY_KILL);
111 }
112
113 void JustDied(Unit* /*killer*/) override
114 {
115 Talk(SAY_DIE);
116 _JustDied();
117 }
118
119 void UpdateAI(uint32 diff) override
120 {
121 if (!UpdateVictim())
122 return;
123
124 events.Update(diff);
125
126 while (uint32 eventId = events.ExecuteEvent())
127 {
128 switch (eventId)
129 {
130 case EVENT_MORTAL_WOUND:
131 DoCastVictim(SPELL_MORTAL_WOUND);
132 events.ScheduleEvent(EVENT_MORTAL_WOUND, urand (5000, 13000));
133 break;
134 case EVENT_SURGE:
135 Talk(SAY_SURGE);
136 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
137 DoCast(target, SPELL_SURGE);
138 events.ScheduleEvent(EVENT_SURGE, urand (5000, 13000));
139 break;
140 case EVENT_RETALIATION:
141 DoCast(me, SPELL_RETALIATION);
142 events.ScheduleEvent(EVENT_RETALIATION, 30000);
143 break;
144 default:
145 break;
146 }
147 }
148
149 if (!retaliation)
150 {
151 if (HealthBelowPct(20))
152 {
153 events.ScheduleEvent(EVENT_RETALIATION, 1000);
154 retaliation = true;
155 }
156 }
157
158 if (!yelledForHeal)
159 {
160 if (HealthBelowPct(40))
161 {
162 Talk(SAY_HEAL);
163 yelledForHeal = true;
164 }
165 }
166
167 DoMeleeAttackIfReady();
168 }
169
170 private:
171 bool hasTaunted;
172 bool yelledForHeal;
173 bool retaliation;
174 };
175
176 CreatureAI* GetAI(Creature* creature) const override
177 {
178 return new boss_watchkeeper_gargolmarAI(creature);
179 }
180};

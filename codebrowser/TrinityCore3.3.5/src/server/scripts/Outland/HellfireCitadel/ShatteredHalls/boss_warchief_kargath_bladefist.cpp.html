64class boss_warchief_kargath_bladefist : public CreatureScript

69 struct boss_warchief_kargath_bladefistAI : public BossAI
70 {
71 boss_warchief_kargath_bladefistAI(Creature* creature) : BossAI(creature, DATA_KARGATH)
72 {
73 Initialize();
74 target_num = 0;
75 }
76
77 void Initialize()
78 {
79 summoned = 2;
80 InBlade = false;
81 Wait_Timer = 0;
82
83 Charge_timer = 0;
84 Blade_Dance_Timer = 45000;
85 Summon_Assistant_Timer = 30000;
86 Assassins_Timer = 5000;
87 resetcheck_timer = 5000;
88 }
89
90 void DoAction(int32 action) override
91 {
92 if (action == ACTION_EXECUTIONER_TAUNT)
93 {
94 switch (instance->GetData(DATA_TEAM_IN_INSTANCE))
95 {
96 case ALLIANCE:
97 Talk(SAY_CALL_EXECUTIONER_A);
98 break;
99 case HORDE:
100 Talk(SAY_CALL_EXECUTIONER_H);
101 break;
102 default:
103 break;
104 }
105 }
106 }
107
108 void Reset() override
109 {
110 removeAdds();
111 _Reset();
112 me->SetSpeed(MOVE_RUN, 2);
113 me->SetWalk(false);
114
115 Initialize();
116 }
117
118 void JustDied(Unit* /*killer*/) override
119 {
120 _JustDied();
121 Talk(SAY_DEATH);
122 removeAdds();
123 }
124
125 void EnterCombat(Unit* /*who*/) override
126 {
127 Talk(SAY_AGGRO);
128 }
129
130 void JustSummoned(Creature* summon) override
131 {
132 switch (summon->GetEntry())
133 {
134 case NPC_HEARTHEN_GUARD:
135 case NPC_SHARPSHOOTER_GUARD:
136 case NPC_REAVER_GUARD:
137 summon->AI()->AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0));
138 adds.push_back(summon->GetGUID());
139 break;
140 case NPC_SHATTERED_ASSASSIN:
141 assassins.push_back(summon->GetGUID());
142 break;
143 }
144 }
145
146 void KilledUnit(Unit* victim) override
147 {
148 if (victim->GetTypeId() == TYPEID_PLAYER)
149 {
150 Talk(SAY_SLAY);
151 }
152 }
153
154 void MovementInform(uint32 type, uint32 id) override
155 {
156 if (InBlade)
157 {
158 if (type != POINT_MOTION_TYPE)
159 return;
160
161 if (id != 1)
162 return;
163
164 if (target_num > 0) // to prevent loops
165 {
166 Wait_Timer = 1;
167 DoCast(me, SPELL_BLADE_DANCE, true);
168 target_num--;
169 }
170 }
171 }
172
173 void removeAdds()
174 {
175 for (GuidVector::const_iterator itr = adds.begin(); itr!= adds.end(); ++itr)
176 {
177 Creature* creature = ObjectAccessor::GetCreature(*me, *itr);
178 if (creature && creature->IsAlive())
179 {
180 creature->GetMotionMaster()->Clear(true);
181 me->DealDamage(creature, creature->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
182 creature->RemoveCorpse();
183 }
184 }
185 adds.clear();
186
187 for (GuidVector::const_iterator itr = assassins.begin(); itr!= assassins.end(); ++itr)
188 {
189 Creature* creature = ObjectAccessor::GetCreature(*me, *itr);
190 if (creature && creature->IsAlive())
191 {
192 creature->GetMotionMaster()->Clear(true);
193 me->DealDamage(creature, creature->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
194 creature->RemoveCorpse();
195 }
196 }
197 assassins.clear();
198 }

200 {
201 me->SummonCreature(NPC_SHATTERED_ASSASSIN, AssassEntrance[0], AssassEntrance[1]+8, AssassEntrance[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
202 me->SummonCreature(NPC_SHATTERED_ASSASSIN, AssassEntrance[0], AssassEntrance[1]-8, AssassEntrance[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
203 me->SummonCreature(NPC_SHATTERED_ASSASSIN, AssassExit[0], AssassExit[1]+8, AssassExit[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
204 me->SummonCreature(NPC_SHATTERED_ASSASSIN, AssassExit[0], AssassExit[1]-8, AssassExit[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
205 }

207 void UpdateAI(uint32 diff) override
208 {
209 //Return since we have no target
210 if (!UpdateVictim())
211 return;
212
213 if (Assassins_Timer)
214 {
215 if (Assassins_Timer <= diff)
216 {
217 SpawnAssassin();
218 Assassins_Timer = 0;
219 }
220 else
221 Assassins_Timer -= diff;
222 }
223
224 if (InBlade)
225 {
226 if (Wait_Timer)
227 {
228 if (Wait_Timer <= diff)
229 {
230 if (target_num <= 0)
231 {
232 // stop bladedance
233 InBlade = false;
234 me->SetSpeed(MOVE_RUN, 2);
235 me->GetMotionMaster()->MoveChase(me->GetVictim());
236 Blade_Dance_Timer = 30000;
237 Wait_Timer = 0;
238 if (IsHeroic())
239 Charge_timer = 5000;
240 }
241 else
242 {
243 //move in bladedance
244 float x, y, randx, randy;
245 randx = 0.0f + rand32() % 40;
246 randy = 0.0f + rand32() % 40;
247 x = 210+ randx;
248 y = -60- randy;
249 me->GetMotionMaster()->MovePoint(1, x, y, me->GetPositionZ());
250 Wait_Timer = 0;
251 }
252 }
253 else
254 Wait_Timer -= diff;
255 }
256 }
257 else
258 {
259 if (Blade_Dance_Timer)
260 {
261 if (Blade_Dance_Timer <= diff)
262 {
263 target_num = TARGET_NUM;
264 Wait_Timer = 1;
265 InBlade = true;
266 Blade_Dance_Timer = 0;
267 me->SetSpeed(MOVE_RUN, 4);
268 return;
269 }
270 else
271 Blade_Dance_Timer -= diff;
272 }
273
274 if (Charge_timer)
275 {
276 if (Charge_timer <= diff)
277 {
278 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0), H_SPELL_CHARGE);
279 Charge_timer = 0;
280 }
281 else
282 Charge_timer -= diff;
283 }
284
285 if (Summon_Assistant_Timer <= diff)
286 {
287 for (uint8 i = 0; i < summoned; ++i)
288 {
289 switch (urand(0, 2))
290 {
291 case 0:
292 me->SummonCreature(NPC_HEARTHEN_GUARD, AddsEntrance[0], AddsEntrance[1], AddsEntrance[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
293 break;
294 case 1:
295 me->SummonCreature(NPC_SHARPSHOOTER_GUARD, AddsEntrance[0], AddsEntrance[1], AddsEntrance[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
296 break;
297 case 2:
298 me->SummonCreature(NPC_REAVER_GUARD, AddsEntrance[0], AddsEntrance[1], AddsEntrance[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
299 break;
300 }
301 }
302 if (urand(0, 9) < 2)
303 ++summoned;
304 Summon_Assistant_Timer = urand(25000, 35000);
305 }
306 else
307 Summon_Assistant_Timer -= diff;
308
309 DoMeleeAttackIfReady();
310 }
311
312 if (resetcheck_timer <= diff)
313 {
314 uint32 tempx = uint32(me->GetPositionX());
315 if (tempx > 255 || tempx < 205)
316 {
317 EnterEvadeMode();
318 return;
319 }
320 resetcheck_timer = 5000;
321 }
322 else
323 resetcheck_timer -= diff;
324 }

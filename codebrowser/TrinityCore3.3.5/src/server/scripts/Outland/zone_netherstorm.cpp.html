122 // Emote Ardonis and Pathaleon
123 void Turn_to_Pathaleons_Image()
124 {
125 Creature* ardonis = ObjectAccessor::GetCreature(*me, ardonisGUID);
126 Creature* pathaleon = ObjectAccessor::GetCreature(*me, pathaleonGUID);
127
128 if (!ardonis || !pathaleon)
129 return;
130
131 // Turn Dawnforge
132 me->SetFacingToObject(pathaleon);
133
134 // Turn Ardonis
135 ardonis->SetFacingToObject(pathaleon);
136
137 //Set them to kneel
138 me->SetStandState(UNIT_STAND_STATE_KNEEL);
139 ardonis->SetStandState(UNIT_STAND_STATE_KNEEL);
140 }
141
142 //Set them back to each other
143 void Turn_to_eachother()
144 {
145 if (Unit* ardonis = ObjectAccessor::GetUnit(*me, ardonisGUID))
146 {
147 // Turn Dawnforge
148 me->SetFacingToObject(ardonis);
149
150 // Turn Ardonis
151 ardonis->SetFacingToObject(me);
152
153 //Set state
154 me->SetStandState(UNIT_STAND_STATE_STAND);
155 ardonis->SetStandState(UNIT_STAND_STATE_STAND);
156 }
157 }
158
159 bool CanStartEvent(Player* player)
160 {
161 if (!isEvent)
162 {
163 Creature* ardonis = me->FindNearestCreature(CreatureEntry[0], 10.0f);
164 if (!ardonis)
165 return false;
166
167 ardonisGUID = ardonis->GetGUID();
168 PlayerGUID = player->GetGUID();
169
170 isEvent = true;
171
172 Turn_to_eachother();
173 return true;
174 }
175
176 TC_LOG_DEBUG("scripts", "npc_commander_dawnforge event already in progress, need to wait.");
177 return false;
178 }
179
180 void UpdateAI(uint32 diff) override
181 {
182 //Is event even running?
183 if (!isEvent)
184 return;
185
186 //Phase timing
187 if (Phase_Timer >= diff)
188 {
189 Phase_Timer -= diff;
190 return;
191 }
192
193 Creature* ardonis = ObjectAccessor::GetCreature(*me, ardonisGUID);
194 Creature* pathaleon = ObjectAccessor::GetCreature(*me, pathaleonGUID);
195 Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
196
197 if (!ardonis || !player)
198 {
199 Reset();
200 return;
201 }
202
203 if (Phase > 4 && !pathaleon)
204 {
205 Reset();
206 return;
207 }
208
209 //Phase 1 Dawnforge say
210 switch (Phase)
211 {
212 case 1:
213 Talk(SAY_COMMANDER_DAWNFORGE_1);
214 ++Phase;
215 Phase_Timer = 16000;
216 break;
217 //Phase 2 Ardonis say
218 case 2:
219 ardonis->AI()->Talk(SAY_ARCANIST_ARDONIS_1);
220 ++Phase;
221 Phase_Timer = 16000;
222 break;
223 //Phase 3 Dawnforge say
224 case 3:
225 Talk(SAY_COMMANDER_DAWNFORGE_2);
226 ++Phase;
227 Phase_Timer = 16000;
228 break;
229 //Phase 4 Pathaleon spawns up to phase 9
230 case 4:
231 //spawn pathaleon's image
232 me->SummonCreature(CreatureEntry[2], 2325.851563f, 2799.534668f, 133.084229f, 6.038996f, TEMPSUMMON_TIMED_DESPAWN, 90000);
233 ++Phase;
234 Phase_Timer = 500;
235 break;
236 //Phase 5 Pathaleon say
237 case 5:
238 pathaleon->AI()->Talk(SAY_PATHALEON_CULATOR_IMAGE_1);
239 ++Phase;
240 Phase_Timer = 6000;
241 break;
242 //Phase 6
243 case 6:
244 switch (PhaseSubphase)
245 {
246 //Subphase 1: Turn Dawnforge and Ardonis
247 case 0:
248 Turn_to_Pathaleons_Image();
249 ++PhaseSubphase;
250 Phase_Timer = 8000;
251 break;
252 //Subphase 2 Dawnforge say
253 case 1:
254 Talk(SAY_COMMANDER_DAWNFORGE_3);
255 PhaseSubphase = 0;
256 ++Phase;
257 Phase_Timer = 8000;
258 break;
259 }
260 break;
261 //Phase 7 Pathaleons say 3 Sentence, every sentence need a subphase
262 case 7:
263 switch (PhaseSubphase)
264 {
265 //Subphase 1
266 case 0:
267 pathaleon->AI()->Talk(SAY_PATHALEON_CULATOR_IMAGE_2);
268 ++PhaseSubphase;
269 Phase_Timer = 12000;
270 break;
271 //Subphase 2
272 case 1:
273 pathaleon->AI()->Talk(SAY_PATHALEON_CULATOR_IMAGE_2_1);
274 ++PhaseSubphase;
275 Phase_Timer = 16000;
276 break;
277 //Subphase 3
278 case 2:
279 pathaleon->AI()->Talk(SAY_PATHALEON_CULATOR_IMAGE_2_2);
280 PhaseSubphase = 0;
281 ++Phase;
282 Phase_Timer = 10000;
283 break;
284 }
285 break;
286 //Phase 8 Dawnforge & Ardonis say
287 case 8:
288 Talk(SAY_COMMANDER_DAWNFORGE_4);
289 ardonis->AI()->Talk(SAY_ARCANIST_ARDONIS_2);
290 ++Phase;
291 Phase_Timer = 4000;
292 break;
293 //Phase 9 Pathaleons Despawn, Reset Dawnforge & Ardonis angle
294 case 9:
295 Turn_to_eachother();
296 //hide pathaleon, unit will despawn shortly
297 pathaleon->SetVisible(false);
298 PhaseSubphase = 0;
299 ++Phase;
300 Phase_Timer = 3000;
301 break;
302 //Phase 10 Dawnforge say
303 case 10:
304 Talk(SAY_COMMANDER_DAWNFORGE_5);
305 player->AreaExploredOrEventHappens(QUEST_INFO_GATHERING);
306 Reset();
307 break;
308 }
309 }
310 };

313class at_commander_dawnforge : public AreaTriggerScript
314{
315public:
316 at_commander_dawnforge() : AreaTriggerScript("at_commander_dawnforge") { }
317
318 bool OnTrigger(Player* player, const AreaTriggerEntry* /*at*/) override
319 {
320 //if player lost aura or not have at all, we should not try start event.
321 if (!player->HasAura(SPELL_SUNFURY_DISGUISE))
322 return false;
323
324 if (player->IsAlive() && player->GetQuestStatus(QUEST_INFO_GATHERING) == QUEST_STATUS_INCOMPLETE)
325 {
326 Creature* Dawnforge = player->FindNearestCreature(CreatureEntry[1], 30.0f);
327 if (!Dawnforge)
328 return false;
329
330 if (ENSURE_AI(npc_commander_dawnforge::npc_commander_dawnforgeAI, Dawnforge->AI())->CanStartEvent(player))
331 return true;
332 }
333 return false;
334 }

438 void Reset() override
439 {
440 Weak = false;
441 Materialize = false;
442 Drained = false;
443 WeakPercent = 25 + (rand32() % 16); // 25-40
444
445 PlayerGUID.Clear();
446
447 ManaBurnTimer = 5000 + (rand32() % 3 * 1000); // 5-8 sec cd
448
449 if (me->GetEntry() == NPC_DRAINED_PHASE_HUNTER_ENTRY)
450 me->UpdateEntry(NPC_PHASE_HUNTER_ENTRY);
451 }
452
453 void EnterCombat(Unit* who) override
454 {
455 if (who->GetTypeId() == TYPEID_PLAYER)
456 PlayerGUID = who->GetGUID();
457 }
458
459 //void SpellHit(Unit* /*caster*/, const SpellInfo* /*spell*/) override
460 //{
461 // DoCast(me, SPELL_DE_MATERIALIZE);
462 //}
463
464 void UpdateAI(uint32 diff) override
465 {
466 if (!Materialize)
467 {
468 DoCast(me, SPELL_MATERIALIZE);
469 Materialize = true;
470 }
471
472 if (me->HasAuraType(SPELL_AURA_MOD_DECREASE_SPEED) || me->HasUnitState(UNIT_STATE_ROOT)) // if the mob is rooted/slowed by spells eg.: Entangling Roots, Frost Nova, Hamstring, Crippling Poison, etc. => remove it
473 DoCast(me, SPELL_PHASE_SLIP);
474
475 if (!UpdateVictim())
476 return;
477
478 // some code to cast spell Mana Burn on random target which has mana
479 if (ManaBurnTimer <= diff)
480 {
481 std::list<HostileReference*> AggroList = me->getThreatManager().getThreatList();
482 std::list<Unit*> UnitsWithMana;
483
484 for (std::list<HostileReference*>::const_iterator itr = AggroList.begin(); itr != AggroList.end(); ++itr)
485 {
486 if (Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid()))
487 {
488 if (unit->GetCreateMana() > 0)
489 UnitsWithMana.push_back(unit);
490 }
491 }
492 if (!UnitsWithMana.empty())
493 {
494 DoCast(Trinity::Containers::SelectRandomContainerElement(UnitsWithMana), SPELL_MANA_BURN);
495 ManaBurnTimer = 8000 + (rand32() % 10 * 1000); // 8-18 sec cd
496 }
497 else
498 ManaBurnTimer = 3500;
499 } else ManaBurnTimer -= diff;
500
501 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID)) // start: support for quest 10190
502 {
503 if (!Weak && HealthBelowPct(WeakPercent)
504 && player->GetQuestStatus(QUEST_RECHARGING_THE_BATTERIES) == QUEST_STATUS_INCOMPLETE)
505 {
506 Talk(EMOTE_WEAK);
507 Weak = true;
508 }
509 if (Weak && !Drained && me->HasAura(SPELL_RECHARGING_BATTERY))
510 {
511 Drained = true;
512 int32 uHpPct = int32(me->GetHealthPct());
513
514 me->UpdateEntry(NPC_DRAINED_PHASE_HUNTER_ENTRY);
515
516 me->SetHealth(me->CountPctFromMaxHealth(uHpPct));
517 me->LowerPlayerDamageReq(me->GetMaxHealth() - me->GetHealth());
518 me->SetInCombatWith(player);
519 }
520 } // end: support for quest 10190
521
522 DoMeleeAttackIfReady();
523 }
524 };

653 void WaypointReached(uint32 waypointId) override
654 {
655 Player* player = GetPlayerForEscort();
656 if (!player)
657 return;
658
659 switch (waypointId)
660 {
661 case 7:
662 case 17:
663 case 29:
664 //Find Object and "work"

666 {
667 // take the GO -> animation
668 me->HandleEmoteCommand(EMOTE_STATE_LOOT);
669 SetEscortPaused(true);

671 }
672 break;
673 case 36: //return and quest_complete
674 player->CompleteQuest(QUEST_MARK_V_IS_ALIVE);
675 break;
676 }
677 }

681 if (Player* player = GetPlayerForEscort())
682 player->FailQuest(QUEST_MARK_V_IS_ALIVE);
683 }
684
685 void UpdateAI(uint32 uiDiff) override
686 {
687 npc_escortAI::UpdateAI(uiDiff);
688
689 if (bTake)
690 {
691 if (uiTakeTimer < uiDiff)
692 {
693 me->HandleEmoteCommand(EMOTE_STATE_NONE);
694 if (GameObject* go = GetClosestGameObjectWithEntry(me, GO_DRAENEI_MACHINE, INTERACTION_DISTANCE))
695 {
696 SetEscortPaused(false);
697 bTake=false;
698 uiTakeTimer = 3000;
699 go->Delete();
700 }
701 }
702 else
703 uiTakeTimer -= uiDiff;
704 }
705 DoMeleeAttackIfReady();
706 }
707 };

714 {
715 creature->setFaction(113);
716 pEscortAI->Start(false, false, player->GetGUID());
717 }
718 }
719 return true;
720 }

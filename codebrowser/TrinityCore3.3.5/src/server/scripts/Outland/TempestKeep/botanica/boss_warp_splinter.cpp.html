65class npc_warp_splinter_treant : public CreatureScript
66{
67 public:
68
69 npc_warp_splinter_treant()
70 : CreatureScript("npc_warp_splinter_treant")
71 {
72 }
73 struct npc_warp_splinter_treantAI : public ScriptedAI
74 {
75 npc_warp_splinter_treantAI(Creature* creature) : ScriptedAI(creature)
76 {
77 Initialize();
78 }
79
80 void Initialize()
81 {
82 check_Timer = 0;
83 }
84
85 ObjectGuid WarpGuid;
86 uint32 check_Timer;
87
88 void Reset() override
89 {
90 Initialize();
91 }
92
93 void EnterCombat(Unit* /*who*/) override { }
94
95 void MoveInLineOfSight(Unit* /*who*/) override { }
96
97
98 void UpdateAI(uint32 diff) override
99 {
100 if (!UpdateVictim() || !me->GetVictim())
101 {
102 if (WarpGuid && check_Timer <= diff)
103 {
104 if (Unit* Warp = ObjectAccessor::GetUnit(*me, WarpGuid))
105 {
106 if (me->IsWithinMeleeRange(Warp, 2.5f))
107 {
108 int32 CurrentHP_Treant = (int32)me->GetHealth();
109 Warp->CastCustomSpell(Warp, SPELL_HEAL_FATHER, &CurrentHP_Treant, 0, 0, true, 0, 0, me->GetGUID());
110 me->DealDamage(me, me->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
111 return;
112 }
113 me->GetMotionMaster()->MoveFollow(Warp, 0, 0);
114 }
115 check_Timer = 1000;
116 }
117 else
118 check_Timer -= diff;
119 return;
120 }
121
122 if (me->EnsureVictim()->GetGUID() != WarpGuid)
123 DoMeleeAttackIfReady();
124 }
125 };
126 CreatureAI* GetAI(Creature* creature) const override
127 {
128 return new npc_warp_splinter_treantAI(creature);
129 }
130};

135class boss_warp_splinter : public CreatureScript
136{
137 public:
138
139 boss_warp_splinter()
140 : CreatureScript("boss_warp_splinter")
141 {
142 }
143 struct boss_warp_splinterAI : public BossAI
144 {
145 boss_warp_splinterAI(Creature* creature) : BossAI(creature, DATA_WARP_SPLINTER)
146 {
147 Initialize();
148 Treant_Spawn_Pos_X = creature->GetPositionX();
149 Treant_Spawn_Pos_Y = creature->GetPositionY();
150 }
151
152 void Initialize()
153 {
154 War_Stomp_Timer = urand(25000, 40000);
155 Summon_Treants_Timer = 45000;
156 Arcane_Volley_Timer = urand(8000, 20000);
157 }
158
159 uint32 War_Stomp_Timer;
160 uint32 Summon_Treants_Timer;
161 uint32 Arcane_Volley_Timer;
162
163 float Treant_Spawn_Pos_X;
164 float Treant_Spawn_Pos_Y;
165
166 void Reset() override
167 {
168 Initialize();
169
170 me->SetSpeed(MOVE_RUN, 0.7f, true);
171 }
172
173 void EnterCombat(Unit* /*who*/) override
174 {
175 Talk(SAY_AGGRO);
176 }
177
178 void KilledUnit(Unit* /*victim*/) override
179 {
180 Talk(SAY_SLAY);
181 }
182
183 void JustDied(Unit* /*killer*/) override
184 {
185 Talk(SAY_DEATH);
186 }
187
188 void SummonTreants()
189 {
190 for (uint8 i = 0; i < 6; ++i)
191 {
192 float angle = (float(M_PI) / 3) * i;
193
194 float X = Treant_Spawn_Pos_X + TREANT_SPAWN_DIST * std::cos(angle);
195 float Y = Treant_Spawn_Pos_Y + TREANT_SPAWN_DIST * std::sin(angle);
196 float O = - me->GetAngle(X, Y);
197
198 if (Creature* pTreant = me->SummonCreature(CREATURE_TREANT, treant_pos[i][0], treant_pos[i][1], treant_pos[i][2], O, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 25000))
199 ENSURE_AI(npc_warp_splinter_treant::npc_warp_splinter_treantAI, pTreant->AI())->WarpGuid = me->GetGUID();
200 }
201 Talk(SAY_SUMMON);
202 }
203
204 void UpdateAI(uint32 diff) override
205 {
206 if (!UpdateVictim())
207 return;
208
209 //Check for War Stomp
210 if (War_Stomp_Timer <= diff)
211 {
212 DoCastVictim(WAR_STOMP);
213 War_Stomp_Timer = urand(25000, 40000);
214 }
215 else
216 War_Stomp_Timer -= diff;
217
218 //Check for Arcane Volley
219 if (Arcane_Volley_Timer <= diff)
220 {
221 DoCastVictim(ARCANE_VOLLEY);
222 Arcane_Volley_Timer = urand(20000, 35000);
223 }
224 else
225 Arcane_Volley_Timer -= diff;
226
227 //Check for Summon Treants
228 if (Summon_Treants_Timer <= diff)
229 {
230 SummonTreants();
231 Summon_Treants_Timer = 45000;
232 }
233 else
234 Summon_Treants_Timer -= diff;
235
236 DoMeleeAttackIfReady();
237 }
238 };
239
240 CreatureAI* GetAI(Creature* creature) const override
241 {
242 return new boss_warp_splinterAI(creature);
243 }
244};

74 struct npc_millhouse_manastormAI : public ScriptedAI
75 {
76 npc_millhouse_manastormAI(Creature* creature) : ScriptedAI(creature)
77 {
78 Initialize();
79 instance = creature->GetInstanceScript();
80 }
81
82 void Initialize()
83 {
84 EventProgress_Timer = 2000;
85 LowHp = false;
86 Init = false;
87 Phase = 1;
88
89 Pyroblast_Timer = 1000;
90 Fireball_Timer = 2500;
91 }
92
93 InstanceScript* instance;
94
95 uint32 EventProgress_Timer;
96 uint32 Phase;
97 bool Init;
98 bool LowHp;
99
100 uint32 Pyroblast_Timer;
101 uint32 Fireball_Timer;
102
103 void Reset() override
104 {
105 Initialize();
106
107 if (instance->GetData(DATA_WARDEN_2) == DONE)
108 Init = true;
109
110 if (instance->GetBossState(DATA_HARBINGER_SKYRISS) == DONE)
111 Talk(SAY_COMPLETE);
112 }
113
114 void AttackStart(Unit* who) override
115 {
116 if (me->Attack(who, true))
117 {
118 me->AddThreat(who, 0.0f);
119 me->SetInCombatWith(who);
120 who->SetInCombatWith(me);
121 me->GetMotionMaster()->MoveChase(who, 25.0f);
122 }
123 }
124
125 void KilledUnit(Unit* who) override
126 {
127 if (who->GetTypeId() == TYPEID_PLAYER)
128 Talk(SAY_KILL);
129 }
130
131 void JustDied(Unit* /*killer*/) override
132 {
133 Talk(SAY_DEATH);
134
135 /*for questId 10886 (heroic mode only)
136 if (instance->GetBossState(DATA_HARBINGER_SKYRISS) != DONE)
137 ->FailQuest();*/
138 }
139
140 void UpdateAI(uint32 diff) override
141 {
142 if (!Init)
143 {
144 if (EventProgress_Timer <= diff)
145 {
146 if (Phase < 8)
147 {
148 switch (Phase)
149 {
150 case 1:
151 Talk(SAY_INTRO_1);
152 EventProgress_Timer = 18000;
153 break;
154 case 2:
155 Talk(SAY_INTRO_2);
156 EventProgress_Timer = 18000;
157 break;
158 case 3:
159 Talk(SAY_WATER);
160 DoCast(me, SPELL_CONJURE_WATER);
161 EventProgress_Timer = 7000;
162 break;
163 case 4:
164 Talk(SAY_BUFFS);
165 DoCast(me, SPELL_ICE_ARMOR);
166 EventProgress_Timer = 7000;
167 break;
168 case 5:
169 Talk(SAY_DRINK);
170 DoCast(me, SPELL_ARCANE_INTELLECT);
171 EventProgress_Timer = 7000;
172 break;
173 case 6:
174 Talk(SAY_READY);
175 EventProgress_Timer = 6000;
176 break;
177 case 7:
178 instance->SetData(DATA_WARDEN_2, DONE);
179 Init = true;
180 break;
181 }
182 ++Phase;
183 }
184 }
185 else
186 EventProgress_Timer -= diff;
187 }
188
189 if (!UpdateVictim())
190 return;
191 if (!LowHp && HealthBelowPct(20))
192 {
193 Talk(SAY_LOWHP);
194 LowHp = true;
195 }
196
197 if (Pyroblast_Timer <= diff)
198 {
199 if (me->IsNonMeleeSpellCast(false))
200 return;
201
202 Talk(SAY_PYRO);
203
204 DoCastVictim(SPELL_PYROBLAST);
205 Pyroblast_Timer = 40000;
206 }
207 else
208 Pyroblast_Timer -=diff;
209
210 if (Fireball_Timer <= diff)
211 {
212 DoCastVictim(SPELL_FIREBALL);
213 Fireball_Timer = 4000;
214 }
215 else
216 Fireball_Timer -=diff;
217
218 DoMeleeAttackIfReady();
219 }
220 };
221
222 CreatureAI* GetAI(Creature* creature) const override
223 {
224 return GetArcatrazAI<npc_millhouse_manastormAI>(creature);
225 }

276 struct npc_warden_mellicharAI : public ScriptedAI

278 npc_warden_mellicharAI(Creature* creature) : ScriptedAI(creature)

280 Initialize();
281 instance = creature->GetInstanceScript();

283
284 void Initialize()
285 {
286 IsRunning = false;
287 CanSpawn = false;
288
289 EventProgress_Timer = 22000;
290 Phase = 1;
291 }
292
293 InstanceScript* instance;
294
295 bool IsRunning;
296 bool CanSpawn;
297
298 uint32 EventProgress_Timer;
299 uint32 Phase;
300
301 void Reset() override
302 {
303 Initialize();
304
305 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
306 DoCast(me, SPELL_TARGET_OMEGA);
307
308 instance->SetBossState(DATA_HARBINGER_SKYRISS, NOT_STARTED);
309 }
310
311 void AttackStart(Unit* /*who*/) override { }
312
313 void MoveInLineOfSight(Unit* who) override
314 {
315 if (IsRunning)
316 return;
317
318 if (!me->GetVictim() && me->CanCreatureAttack(who))
319 {
320 if (!me->CanFly() && me->GetDistanceZ(who) > CREATURE_Z_ATTACK_RANGE)
321 return;
322 if (who->GetTypeId() != TYPEID_PLAYER)
323 return;
324
325 float attackRadius = me->GetAttackDistance(who)/10;
326 if (me->IsWithinDistInMap(who, attackRadius) && me->IsWithinLOSInMap(who))
327 EnterCombat(who);
328 }
329 }
330
331 void EnterCombat(Unit* /*who*/) override
332 {
333 Talk(YELL_INTRO1);
334 DoCast(me, SPELL_BUBBLE_VISUAL);
335
336 instance->SetBossState(DATA_HARBINGER_SKYRISS, IN_PROGRESS);
337 instance->HandleGameObject(instance->GetGuidData(DATA_WARDENS_SHIELD), false);
338 IsRunning = true;
339 }
340
341 void JustSummoned(Creature* summon) override
342 {
343 DoZoneInCombat(summon);
344 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
345 summon->AI()->AttackStart(target);
346 }
347
348 bool CanProgress()
349 {
350 if (Phase == 7 && instance->GetData(DATA_WARDEN_4) == DONE)
351 return true;
352 if (Phase == 6 && instance->GetData(DATA_WARDEN_3) == DONE)
353 return true;
354 if (Phase == 5 && instance->GetData(DATA_WARDEN_2) == DONE)
355 return true;
356 if (Phase == 4)
357 return true;
358 if (Phase == 3 && instance->GetData(DATA_WARDEN_1) == DONE)
359 return true;
360 if (Phase == 2 && instance->GetBossState(DATA_HARBINGER_SKYRISS) == IN_PROGRESS)
361 return true;
362 if (Phase == 1 && instance->GetBossState(DATA_HARBINGER_SKYRISS) == IN_PROGRESS)
363 return true;
364
365 return false;
366 }
367
368 void DoPrepareForPhase()
369 {
370 me->InterruptNonMeleeSpells(true);
371 me->RemoveAurasByType(SPELL_AURA_DUMMY);
372
373 switch (Phase)
374 {
375 case 2:
376 DoCast(me, SPELL_TARGET_ALPHA);
377 instance->SetData(DATA_WARDEN_1, IN_PROGRESS);
378 instance->HandleGameObject(instance->GetGuidData(DATA_WARDENS_SHIELD), false);
379 break;
380 case 3:
381 DoCast(me, SPELL_TARGET_BETA);
382 instance->SetData(DATA_WARDEN_2, IN_PROGRESS);
383 break;
384 case 5:
385 DoCast(me, SPELL_TARGET_DELTA);
386 instance->SetData(DATA_WARDEN_3, IN_PROGRESS);
387 break;
388 case 6:
389 DoCast(me, SPELL_TARGET_GAMMA);
390 instance->SetData(DATA_WARDEN_4, IN_PROGRESS);
391 break;
392 case 7:
393 instance->SetData(DATA_WARDEN_5, IN_PROGRESS);
394 break;
395 }
396 CanSpawn = true;
397 }
398
399 void UpdateAI(uint32 diff) override
400 {
401 if (!IsRunning)
402 return;
403
404 if (EventProgress_Timer <= diff)
405 {
406 if (instance->GetBossState(DATA_HARBINGER_SKYRISS) == FAIL)
407 {
408 Reset();
409 return;
410 }
411
412 if (CanSpawn)
413 {
414 //continue beam omega pod, unless we are about to summon skyriss
415 if (Phase != 7)
416 DoCast(me, SPELL_TARGET_OMEGA);
417
418 switch (Phase)
419 {
420 case 2:
421 switch (urand(0, 1))
422 {
423 case 0:
424 me->SummonCreature(ENTRY_TRICKSTER, 478.326f, -148.505f, 42.56f, 3.19f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
425 break;
426 case 1:
427 me->SummonCreature(ENTRY_PH_HUNTER, 478.326f, -148.505f, 42.56f, 3.19f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
428 break;
429 }
430 break;
431 case 3:
432 me->SummonCreature(ENTRY_MILLHOUSE, 413.292f, -148.378f, 42.56f, 6.27f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
433 break;
434 case 4:
435 Talk(YELL_RELEASE2B);
436 break;
437 case 5:
438 switch (urand(0, 1))
439 {
440 case 0:
441 me->SummonCreature(ENTRY_AKKIRIS, 420.179f, -174.396f, 42.58f, 0.02f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
442 break;
443 case 1:
444 me->SummonCreature(ENTRY_SULFURON, 420.179f, -174.396f, 42.58f, 0.02f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
445 break;
446 }
447 break;
448 case 6:
449 switch (urand(0, 1))
450 {
451 case 0:
452 me->SummonCreature(ENTRY_TW_DRAK, 471.795f, -174.58f, 42.58f, 3.06f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
453 break;
454 case 1:
455 me->SummonCreature(ENTRY_BL_DRAK, 471.795f, -174.58f, 42.58f, 3.06f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
456 break;
457 }
458 break;
459 case 7:
460 me->SummonCreature(ENTRY_SKYRISS, 445.763f, -191.639f, 44.64f, 1.60f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
461 Talk(YELL_WELCOME);
462 break;
463 }
464 CanSpawn = false;
465 ++Phase;
466 }
467 if (CanProgress())
468 {
469 switch (Phase)
470 {
471 case 1:
472 Talk(YELL_INTRO2);
473 EventProgress_Timer = 10000;
474 ++Phase;
475 break;
476 case 2:
477 Talk(YELL_RELEASE1);
478 DoPrepareForPhase();
479 EventProgress_Timer = 7000;
480 break;
481 case 3:
482 Talk(YELL_RELEASE2A);
483 DoPrepareForPhase();
484 EventProgress_Timer = 10000;
485 break;
486 case 4:
487 DoPrepareForPhase();
488 EventProgress_Timer = 15000;
489 break;
490 case 5:
491 Talk(YELL_RELEASE3);
492 DoPrepareForPhase();
493 EventProgress_Timer = 15000;
494 break;
495 case 6:
496 Talk(YELL_RELEASE4);
497 DoPrepareForPhase();
498 EventProgress_Timer = 15000;
499 break;
500 case 7:
501 DoPrepareForPhase();
502 EventProgress_Timer = 15000;
503 break;
504 }
505 }
506 }
507 else
508 EventProgress_Timer -= diff;
509 }
510 };

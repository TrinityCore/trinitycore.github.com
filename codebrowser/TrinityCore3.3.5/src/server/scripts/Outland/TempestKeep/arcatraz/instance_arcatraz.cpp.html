18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "arcatraz.h"
21
22DoorData const doorData[] =
23{
24 { GO_CONTAINMENT_CORE_SECURITY_FIELD_ALPHA, DATA_SOCCOTHRATES, DOOR_TYPE_PASSAGE },
25 { GO_CONTAINMENT_CORE_SECURITY_FIELD_BETA, DATA_DALLIAH, DOOR_TYPE_PASSAGE },
26 { 0, 0, DOOR_TYPE_ROOM } // END
27};
28
29class instance_arcatraz : public InstanceMapScript
30{
31 public:
32 instance_arcatraz() : InstanceMapScript(ArcatrazScriptName, 552) { }
33
34 struct instance_arcatraz_InstanceMapScript : public InstanceScript
35 {
36 instance_arcatraz_InstanceMapScript(Map* map) : InstanceScript(map)
37 {
38 SetHeaders(DataHeader);
39 SetBossNumber(EncounterCount);
40 LoadDoorData(doorData);
41
42 ConversationState = NOT_STARTED;
43
44 memset(StasisPodStates, NOT_STARTED, 5 * sizeof(uint8));
45 }
46
47 void OnCreatureCreate(Creature* creature) override
48 {
49 switch (creature->GetEntry())
50 {
51 case NPC_DALLIAH:
52 DalliahGUID = creature->GetGUID();
53 break;
54 case NPC_SOCCOTHRATES:
55 SoccothratesGUID = creature->GetGUID();
56 break;
57 case NPC_MELLICHAR:
58 MellicharGUID = creature->GetGUID();
59 break;
60 default:
61 break;
62 }
63 }
64
65 void OnGameObjectCreate(GameObject* go) override
66 {
67 switch (go->GetEntry())
68 {
69 case GO_CONTAINMENT_CORE_SECURITY_FIELD_ALPHA:
70 case GO_CONTAINMENT_CORE_SECURITY_FIELD_BETA:
71 AddDoor(go, true);
72 break;
73 case GO_STASIS_POD_ALPHA:
74 StasisPodGUIDs[0] = go->GetGUID();
75 break;
76 case GO_STASIS_POD_BETA:
77 StasisPodGUIDs[1] = go->GetGUID();
78 break;
79 case GO_STASIS_POD_DELTA:
80 StasisPodGUIDs[2] = go->GetGUID();
81 break;
82 case GO_STASIS_POD_GAMMA:
83 StasisPodGUIDs[3] = go->GetGUID();
84 break;
85 case GO_STASIS_POD_OMEGA:
86 StasisPodGUIDs[4] = go->GetGUID();
87 break;
88 case GO_WARDENS_SHIELD:
89 WardensShieldGUID = go->GetGUID();
90 break;
91 default:
92 break;
93 }
94 }
95
96 void OnGameObjectRemove(GameObject* go) override
97 {
98 switch (go->GetEntry())
99 {
100 case GO_CONTAINMENT_CORE_SECURITY_FIELD_ALPHA:
101 case GO_CONTAINMENT_CORE_SECURITY_FIELD_BETA:
102 AddDoor(go, false);
103 break;
104 default:
105 break;
106 }
107 }
108
109 void SetData(uint32 type, uint32 data) override
110 {
111 switch (type)
112 {
113 case DATA_WARDEN_1:
114 case DATA_WARDEN_2:
115 case DATA_WARDEN_3:
116 case DATA_WARDEN_4:
117 case DATA_WARDEN_5:
118 if (data == IN_PROGRESS)
119 HandleGameObject(StasisPodGUIDs[type - DATA_WARDEN_1], true);
120 StasisPodStates[type - DATA_WARDEN_1] = uint8(data);
121 break;
122 case DATA_CONVERSATION:
123 ConversationState = uint8(data);
124 break;
125 default:
126 break;
127 }
128 }
129
130 uint32 GetData(uint32 type) const override
131 {
132 switch (type)
133 {
134 case DATA_WARDEN_1:
135 case DATA_WARDEN_2:
136 case DATA_WARDEN_3:
137 case DATA_WARDEN_4:
138 case DATA_WARDEN_5:
139 return StasisPodStates[type - DATA_WARDEN_1];
140 case DATA_CONVERSATION:
141 return ConversationState;
142 default:
143 break;
144 }
145 return 0;
146 }
147
148 ObjectGuid GetGuidData(uint32 data) const override
149 {
150 switch (data)
151 {
152 case DATA_DALLIAH:
153 return DalliahGUID;
154 case DATA_SOCCOTHRATES:
155 return SoccothratesGUID;
156 case DATA_MELLICHAR:
157 return MellicharGUID;
158 case DATA_WARDENS_SHIELD:
159 return WardensShieldGUID;
160 default:
161 break;
162 }
163 return ObjectGuid::Empty;
164 }
165
166 bool SetBossState(uint32 type, EncounterState state) override
167 {
168 if (!InstanceScript::SetBossState(type, state))
169 return false;
170
171 switch (type)
172 {
173 case DATA_HARBINGER_SKYRISS:
174 if (state == NOT_STARTED || state == FAIL)
175 {
176 SetData(DATA_WARDEN_1, NOT_STARTED);
177 SetData(DATA_WARDEN_2, NOT_STARTED);
178 SetData(DATA_WARDEN_3, NOT_STARTED);
179 SetData(DATA_WARDEN_4, NOT_STARTED);
180 SetData(DATA_WARDEN_5, NOT_STARTED);
181 }
182 break;
183 default:
184 break;
185 }
186 return true;
187 }
188
189 protected:
190 ObjectGuid DalliahGUID;
191 ObjectGuid SoccothratesGUID;
192 ObjectGuid StasisPodGUIDs[5];
193 ObjectGuid MellicharGUID;
194 ObjectGuid WardensShieldGUID;
195
196 uint8 ConversationState;
197 uint8 StasisPodStates[5];
198 };
199
200 InstanceScript* GetInstanceScript(InstanceMap* map) const override
201 {
202 return new instance_arcatraz_InstanceMapScript(map);
203 }
204};
205
206void AddSC_instance_arcatraz()
207{
208 new instance_arcatraz();
209}

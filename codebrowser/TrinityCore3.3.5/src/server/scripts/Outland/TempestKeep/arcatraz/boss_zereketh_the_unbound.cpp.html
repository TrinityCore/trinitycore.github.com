22enum Say
23{
24 SAY_AGGRO = 0,
25 SAY_SLAY = 1,
26 SAY_SHADOW_NOVA = 2,
27 SAY_DEATH = 3
28};
29
30enum Spells
31{
32 SPELL_VOID_ZONE = 36119,
33 SPELL_SHADOW_NOVA = 36127,
34 SPELL_SEED_OF_CORRUPTION = 36123
35};
36
37enum Events
38{
39 EVENT_VOID_ZONE = 1,
40 EVENT_SHADOW_NOVA = 2,
41 EVENT_SEED_OF_CORRUPTION = 3
42};
43
44class boss_zereketh_the_unbound : public CreatureScript
45{
46 public:
47 boss_zereketh_the_unbound() : CreatureScript("boss_zereketh_the_unbound") { }
48
49 struct boss_zereketh_the_unboundAI : public BossAI
50 {
51 boss_zereketh_the_unboundAI(Creature* creature) : BossAI(creature, DATA_ZEREKETH) { }
52
53 void Reset() override
54 {
55 _Reset();
56 }
57
58 void JustDied(Unit* /*killer*/) override
59 {
60 _JustDied();
61 Talk(SAY_DEATH);
62 }
63
64 void EnterCombat(Unit* /*who*/) override
65 {
66 _EnterCombat();
67 events.ScheduleEvent(EVENT_VOID_ZONE, urand (6000, 10000));
68 events.ScheduleEvent(EVENT_SHADOW_NOVA, urand (6000, 10000));
69 events.ScheduleEvent(EVENT_SEED_OF_CORRUPTION, urand(12000, 20000));
70 Talk(SAY_AGGRO);
71 }
72
73 void KilledUnit(Unit* /*victim*/) override
74 {
75 Talk(SAY_SLAY);
76 }
77
78 void UpdateAI(uint32 diff) override
79 {
80 if (!UpdateVictim())
81 return;
82
83 events.Update(diff);
84
85 if (me->HasUnitState(UNIT_STATE_CASTING))
86 return;
87
88 while (uint32 eventId = events.ExecuteEvent())
89 {
90 switch (eventId)
91 {
92 case EVENT_VOID_ZONE:
93 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true))
94 DoCast(target, SPELL_VOID_ZONE);
95 events.ScheduleEvent(EVENT_VOID_ZONE, urand (6000, 10000));
96 break;
97 case EVENT_SHADOW_NOVA:
98 DoCastVictim(SPELL_SHADOW_NOVA, true);
99 Talk(SAY_SHADOW_NOVA);
100 events.ScheduleEvent(EVENT_SHADOW_NOVA, urand (6000, 10000));
101 break;
102 case EVENT_SEED_OF_CORRUPTION:
103 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true))
104 DoCast(target, SPELL_SEED_OF_CORRUPTION);
105 events.ScheduleEvent(EVENT_SEED_OF_CORRUPTION, urand(12000, 20000));
106 break;
107 default:
108 break;
109 }
110 }
111
112 DoMeleeAttackIfReady();
113 }
114 };
115
116 CreatureAI* GetAI(Creature* creature) const override
117 {
118 return new boss_zereketh_the_unboundAI(creature);
119 }
120};
121
122void AddSC_boss_zereketh_the_unbound()
123{
124 new boss_zereketh_the_unbound();
125}

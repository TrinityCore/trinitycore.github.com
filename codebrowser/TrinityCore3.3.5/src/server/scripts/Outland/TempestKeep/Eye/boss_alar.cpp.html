31enum Spells

35 SPELL_REBIRTH = 34342, // Rebirth - beginning of second phase(after losing all health in phase 1)
36 SPELL_REBIRTH_2 = 35369, // Rebirth(another, without healing to full HP) - after Dive Bomb in phase 2
37 SPELL_MELT_ARMOR = 35410, // Melt Armor - every 60 sec in phase 2

166 void DamageTaken(Unit* /*killer*/, uint32 &damage) override
167 {
168 if (damage >= me->GetHealth() && Phase1)
169 {
170 damage = 0;
171 if (!WaitEvent)
172 {
173 WaitEvent = WE_DIE;
174 WaitTimer = 0;
175 me->SetHealth(0);
176 me->InterruptNonMeleeSpells(true);
177 me->RemoveAllAuras();
178 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
179 me->AttackStop();
180 me->SetTarget(ObjectGuid::Empty);

184 }
185 }

208 void UpdateAI(uint32 diff) override
209 {
210 if (!me->IsInCombat()) // sometimes IsInCombat but !incombat, faction bug?
211 return;
212
213 if (Berserk_Timer <= diff)
214 {
215 DoCast(me, SPELL_BERSERK, true);
216 Berserk_Timer = 60000;
217 }
218 else
219 Berserk_Timer -= diff;
220
221 if (ForceMove)
222 {
223 if (ForceTimer <= diff)
224 {
225 me->GetMotionMaster()->MovePoint(0, waypoint[cur_wp][0], waypoint[cur_wp][1], waypoint[cur_wp][2]);
226 ForceTimer = 5000;
227 }
228 else
229 ForceTimer -= diff;
230
231 }
232 if (WaitEvent)
233 {
234 if (WaitTimer)
235 {
236 if (WaitTimer <= diff)
237 {
238 if (AfterMoving)
239 {
240 me->GetMotionMaster()->MoveIdle();
241 AfterMoving = false;
242 }
243
244 switch (WaitEvent)
245 {
246 case WE_PLATFORM:
247 Platforms_Move_Timer = 30000 + rand32() % 5000;
248 break;
249 case WE_QUILL:
250 DoCast(me, SPELL_FLAME_QUILLS, true);
251 Platforms_Move_Timer = 1;
252 WaitTimer = 10000;
253 WaitEvent = WE_DUMMY;
254 return;
255 case WE_DIE:
256 ForceMove = false;
257 me->SetUInt32Value(UNIT_FIELD_BYTES_1, UNIT_STAND_STATE_DEAD);
258 WaitTimer = 5000;
259 WaitEvent = WE_REVIVE;
260 return;
261 case WE_REVIVE:
262 me->SetUInt32Value(UNIT_FIELD_BYTES_1, UNIT_STAND_STATE_STAND);
263 me->SetFullHealth();
264 me->SetSpeed(MOVE_RUN, DefaultMoveSpeedRate);
265 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
266 DoZoneInCombat();
267 DoCast(me, SPELL_REBIRTH, true);
268 MeltArmor_Timer = 60000;
269 Charge_Timer = 7000;
270 DiveBomb_Timer = 40000 + rand32() % 5000;
271 FlamePatch_Timer = 30000;
272 Phase1 = false;
273 break;
274 case WE_METEOR:
275 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FIRE, false);
276 DoCast(me, SPELL_DIVE_BOMB_VISUAL, false);
277 WaitEvent = WE_DIVE;
278 WaitTimer = 4000;
279 return;
280 case WE_DIVE:
281 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
282 {
283 me->RemoveAurasDueToSpell(SPELL_DIVE_BOMB_VISUAL);
284 DoCast(target, SPELL_DIVE_BOMB, true);
285 float dist = 3.0f;
286 if (me->IsWithinDist3d(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 5.0f))
287 dist = 5.0f;
288 WaitTimer = 1000 + uint32(floor(dist / 80 * 1000.0f));
289 me->SetPosition(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0.0f);
290 me->StopMoving();
291 WaitEvent = WE_LAND;
292 return;
293 }
294 else
295 {
296 EnterEvadeMode();
297 return;
298 }
299 case WE_LAND:
300 WaitEvent = WE_SUMMON;
301 WaitTimer = 2000;
302 return;
303 case WE_SUMMON:
304 for (uint8 i = 0; i < 2; ++i)
305 DoSpawnCreature(CREATURE_EMBER_OF_ALAR, 0, 0, 0, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 5000);
306 me->SetFloatValue(UNIT_FIELD_BOUNDINGRADIUS, 10);
307 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
308 me->SetDisplayId(me->GetNativeDisplayId());
309 DoCast(me, SPELL_REBIRTH_2, true);
310 break;
311 case WE_DUMMY:
312 default:
313 break;
314 }
315
316 WaitEvent = WE_NONE;
317 WaitTimer = 0;
318 }
319 else
320 WaitTimer -= diff;
321 }
322 return;
323 }
324
325 if (Phase1)
326 {
327 if (me->getThreatManager().getThreatList().empty())
328 {
329 EnterEvadeMode();
330 return;
331 }
332
333 if (Platforms_Move_Timer <= diff)
334 {
335 if (cur_wp == 4)
336 {
337 cur_wp = 0;
338 WaitEvent = WE_PLATFORM;
339 }
340 else
341 {
342 if (urand(0, 4)) // next platform
343 {
344 DoSpawnCreature(CREATURE_EMBER_OF_ALAR, 0, 0, 0, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 5000);
345 if (cur_wp == 3)
346 cur_wp = 0;
347 else
348 ++cur_wp;
349 WaitEvent = WE_PLATFORM;
350 }
351 else // flame quill
352 {
353 cur_wp = 4;
354 WaitEvent = WE_QUILL;
355 }
356 }
357 ForceMove = true;
358 ForceTimer = 5000;
359 me->GetMotionMaster()->MovePoint(0, waypoint[cur_wp][0], waypoint[cur_wp][1], waypoint[cur_wp][2]);
360 WaitTimer = 0;
361 return;
362 }
363 else
364 Platforms_Move_Timer -= diff;
365 }
366 else
367 {
368 if (Charge_Timer <= diff)
369 {
370 Unit* target= SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true);
371 if (target)
372 DoCast(target, SPELL_CHARGE);
373 Charge_Timer = 30000;
374 }
375 else
376 Charge_Timer -= diff;
377
378 if (MeltArmor_Timer <= diff)
379 {
380 DoCastVictim(SPELL_MELT_ARMOR);
381 MeltArmor_Timer = 60000;
382 }
383 else
384 MeltArmor_Timer -= diff;
385
386 if (DiveBomb_Timer <= diff)
387 {
388 me->AttackStop();
389 me->GetMotionMaster()->MovePoint(6, waypoint[4][0], waypoint[4][1], waypoint[4][2]);
390 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
391 me->SetFloatValue(UNIT_FIELD_BOUNDINGRADIUS, 50);
392 WaitEvent = WE_METEOR;
393 WaitTimer = 0;
394 DiveBomb_Timer = 40000 + rand32() % 5000;
395 return;
396 }
397 else
398 DiveBomb_Timer -= diff;
399
400 if (FlamePatch_Timer <= diff)
401 {
402 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
403 {
404 Creature* Summoned = me->SummonCreature(CREATURE_FLAME_PATCH_ALAR, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 120000);
405 if (Summoned)
406 {
407 Summoned->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
408 Summoned->SetObjectScale(Summoned->GetObjectScale() * 2.5f);
409 Summoned->SetDisplayId(11686);
410 Summoned->setFaction(me->getFaction());
411 Summoned->SetLevel(me->getLevel());
412 Summoned->CastSpell(Summoned, SPELL_FLAME_PATCH, false);
413 }
414 }
415 FlamePatch_Timer = 30000;
416 }
417 else
418 FlamePatch_Timer -= diff;
419 }
420
421 DoMeleeAttackIfReady();
422 }
423
424 void DoMeleeAttackIfReady()
425 {
426 if (me->isAttackReady() && !me->IsNonMeleeSpellCast(false))
427 {
428 if (me->IsWithinMeleeRange(me->GetVictim()))
429 {
430 me->AttackerStateUpdate(me->GetVictim());
431 me->resetAttackTimer();
432 }
433 else
434 {
435 Unit* target = NULL;
436 target = me->SelectNearestTargetInAttackDistance(5);
437 if (target)
438 AttackStart(target);
439 else
440 {
441 DoCast(me, SPELL_FLAME_BUFFET, true);
442 me->setAttackTimer(BASE_ATTACK, 1500);
443 }
444 }
445 }
446 }
447 };

485 DoZoneInCombat();
486 }
487
488 void EnterEvadeMode(EvadeReason /*why*/) override
489 {
490 me->setDeathState(JUST_DIED);
491 }
492
493 void DamageTaken(Unit* killer, uint32& damage) override
494 {
495 if (damage >= me->GetHealth() && killer != me && !toDie)
496 {
497 damage = 0;
498 DoCast(me, SPELL_EMBER_BLAST, true);
499 me->SetDisplayId(11686);
500 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
501 if (instance->GetBossState(DATA_ALAR) == IN_PROGRESS)
502 {
503 if (Unit* Alar = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_ALAR)))
504 {
505 int32 AlarHealth = int32(Alar->GetHealth()) - int32(Alar->CountPctFromMaxHealth(3));
506 if (AlarHealth > 0)
507 Alar->SetHealth(AlarHealth);
508 else
509 Alar->SetHealth(1);
510 }
511 }
512 toDie = true;
513 }
514 }
515
516 void UpdateAI(uint32 /*diff*/) override
517 {
518 if (!UpdateVictim())
519 return;
520
521 if (toDie)
522 {
523 me->DealDamage(me, me->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
524 //me->SetVisibility(VISIBILITY_OFF);
525 }
526
527 DoMeleeAttackIfReady();
528 }
529
530 };

69class boss_mechano_lord_capacitus : public CreatureScript
70{
71 public:
72 boss_mechano_lord_capacitus() : CreatureScript("boss_mechano_lord_capacitus") { }
73
74 struct boss_mechano_lord_capacitusAI : public BossAI
75 {
76 boss_mechano_lord_capacitusAI(Creature* creature) : BossAI(creature, DATA_MECHANOLORD_CAPACITUS) { }
77
78 void EnterCombat(Unit* /*who*/) override
79 {
80 _EnterCombat();
81 Talk(YELL_AGGRO);
82 events.ScheduleEvent(EVENT_HEADCRACK, 10 * IN_MILLISECONDS);
83 events.ScheduleEvent(EVENT_REFLECTIVE_DAMAGE_SHIELD, 15 * IN_MILLISECONDS);
84 events.ScheduleEvent(EVENT_SUMMON_NETHER_CHARGE, 10 * IN_MILLISECONDS);
85 events.ScheduleEvent(EVENT_BERSERK, 3 * MINUTE * IN_MILLISECONDS);
86
87 if (IsHeroic())
88 events.ScheduleEvent(EVENT_POSITIVE_SHIFT, 15 * IN_MILLISECONDS);
89 }
90
91 void KilledUnit(Unit* /*victim*/) override
92 {
93 Talk(YELL_KILL);
94 }
95
96 void JustDied(Unit* /*victim*/) override
97 {
98 _JustDied();
99 Talk(YELL_DEATH);
100 }
101
102 void UpdateAI(uint32 diff) override
103 {
104 if (!UpdateVictim())
105 return;
106
107 events.Update(diff);
108
109 if (me->HasUnitState(UNIT_STATE_CASTING))
110 return;
111
112 while (uint32 eventId = events.ExecuteEvent())
113 {
114 switch (eventId)
115 {
116 case EVENT_HEADCRACK:
117 DoCastVictim(SPELL_HEADCRACK);
118 events.ScheduleEvent(EVENT_HEADCRACK, 10 * IN_MILLISECONDS);
119 break;
120 case EVENT_REFLECTIVE_DAMAGE_SHIELD:
121 Talk(YELL_REFLECTIVE_DAMAGE_SHIELD);
122 DoCast(me, SPELL_REFLECTIVE_DAMAGE_SHIELD);
123 events.ScheduleEvent(EVENT_REFLECTIVE_MAGIE_SHIELD, 30 * IN_MILLISECONDS);
124 break;
125 case EVENT_REFLECTIVE_MAGIE_SHIELD:
126 Talk(YELL_REFLECTIVE_MAGIC_SHIELD);
127 DoCast(me, SPELL_REFLECTIVE_MAGIC_SHIELD);
128 events.ScheduleEvent(EVENT_REFLECTIVE_DAMAGE_SHIELD, 30 * IN_MILLISECONDS);
129 break;
130 case EVENT_POSITIVE_SHIFT:
131 DoCastAOE(SPELL_POLARITY_SHIFT);
132 events.ScheduleEvent(EVENT_POSITIVE_SHIFT, urand(45, 60) * IN_MILLISECONDS);
133 break;
134 case EVENT_SUMMON_NETHER_CHARGE:
135 {
136 Position pos = me->GetRandomNearPosition(5.0f);
137 me->SummonCreature(NPC_NETHER_CHARGE, pos, TEMPSUMMON_TIMED_DESPAWN, 18000);
138 events.ScheduleEvent(EVENT_SUMMON_NETHER_CHARGE, 10 * IN_MILLISECONDS);
139 break;
140 }
141 case EVENT_BERSERK:
142 DoCast(me, SPELL_BERSERK);
143 break;
144 default:
145 break;
146 }
147 }
148
149 DoMeleeAttackIfReady();
150 }
151 };
152
153 CreatureAI* GetAI(Creature* creature) const override
154 {
155 return new boss_mechano_lord_capacitusAI(creature);
156 }

159class spell_capacitus_polarity_charge : public SpellScriptLoader
160{
161 public:
162 spell_capacitus_polarity_charge() : SpellScriptLoader("spell_capacitus_polarity_charge") { }
163
164 class spell_capacitus_polarity_charge_SpellScript : public SpellScript
165 {
166 PrepareSpellScript(spell_capacitus_polarity_charge_SpellScript);
167
168 bool Validate(SpellInfo const* /*spell*/) override
169 {
170 if (!sSpellMgr->GetSpellInfo(SPELL_POSITIVE_CHARGE))
171 return false;
172 if (!sSpellMgr->GetSpellInfo(SPELL_POSITIVE_CHARGE_STACK))
173 return false;
174 if (!sSpellMgr->GetSpellInfo(SPELL_NEGATIVE_CHARGE))
175 return false;
176 if (!sSpellMgr->GetSpellInfo(SPELL_NEGATIVE_CHARGE_STACK))
177 return false;
178 return true;
179 }
180
181 void HandleTargets(std::list<WorldObject*>& targetList)
182 {
183 uint8 count = 0;
184 for (std::list<WorldObject*>::iterator ihit = targetList.begin(); ihit != targetList.end(); ++ihit)
185 if ((*ihit)->GetGUID() != GetCaster()->GetGUID())
186 if (Player* target = (*ihit)->ToPlayer())
187 if (target->HasAura(GetTriggeringSpell()->Id))
188 ++count;
189
190 if (count)
191 {
192 uint32 spellId = 0;
193
194 if (GetSpellInfo()->Id == SPELL_POSITIVE_CHARGE)
195 spellId = SPELL_POSITIVE_CHARGE_STACK;
196 else // if (GetSpellInfo()->Id == SPELL_NEGATIVE_CHARGE)
197 spellId = SPELL_NEGATIVE_CHARGE_STACK;
198
199 GetCaster()->SetAuraStack(spellId, GetCaster(), count);
200 }
201 }
202
203 void HandleDamage(SpellEffIndex /*effIndex*/)
204 {
205 if (!GetTriggeringSpell())
206 return;
207
208 Unit* target = GetHitUnit();
209
210 if (target->HasAura(GetTriggeringSpell()->Id))
211 SetHitDamage(0);
212 }
213
214 void Register() override
215 {
216 OnEffectHitTarget += SpellEffectFn(spell_capacitus_polarity_charge_SpellScript::HandleDamage, EFFECT_0, SPELL_EFFECT_SCHOOL_DAMAGE);
217 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_capacitus_polarity_charge_SpellScript::HandleTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ALLY);
218 }
219 };
220
221 SpellScript* GetSpellScript() const override
222 {
223 return new spell_capacitus_polarity_charge_SpellScript();
224 }

235
236 bool Validate(SpellInfo const* /*spell*/) override
237 {
238 if (!sSpellMgr->GetSpellInfo(SPELL_POSITIVE_POLARITY) || !sSpellMgr->GetSpellInfo(SPELL_NEGATIVE_POLARITY))
239 return false;
240 return true;
241 }
242
243 void HandleDummy(SpellEffIndex /* effIndex */)
244 {
245 Unit* target = GetHitUnit();
246 Unit* caster = GetCaster();
247
248 target->CastSpell(target, roll_chance_i(50) ? SPELL_POSITIVE_POLARITY : SPELL_NEGATIVE_POLARITY, true, NULL, NULL, caster->GetGUID());
249 }
250
251 void Register() override
252 {
253 OnEffectHitTarget += SpellEffectFn(spell_capacitus_polarity_shift_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
254 }
255 };

259 return new spell_capacitus_polarity_shift_SpellScript();

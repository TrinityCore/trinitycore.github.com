65class npcs_ashyen_and_keleth : public CreatureScript
66{
67public:
68 npcs_ashyen_and_keleth() : CreatureScript("npcs_ashyen_and_keleth") { }
69
70 bool OnGossipHello(Player* player, Creature* creature) override
71 {
72 if (player->GetReputationRank(942) > REP_NEUTRAL)
73 {
74 if (creature->GetEntry() == NPC_ASHYEN)
75 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_BLESS_ASH, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
76
77 if (creature->GetEntry() == NPC_KELETH)
78 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_BLESS_KEL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
79 }
80 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
81
82 return true;
83 }
84
85 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
86 {
87 player->PlayerTalkClass->ClearMenus();
88 if (action == GOSSIP_ACTION_INFO_DEF+1)
89 {
90 creature->setPowerType(POWER_MANA);
91 creature->SetMaxPower(POWER_MANA, 200); //set a "fake" mana value, we can't depend on database doing it in this case
92 creature->SetPower(POWER_MANA, 200);
93
94 if (creature->GetEntry() == NPC_ASHYEN) //check which Creature we are dealing with
95 {
96 uint32 spell = 0;
97 switch (player->GetReputationRank(942))
98 { //mark of lore
99 case REP_FRIENDLY:
100 spell = SPELL_BLESS_ASH_FRI;
101 break;
102 case REP_HONORED:
103 spell = SPELL_BLESS_ASH_HON;
104 break;
105 case REP_REVERED:
106 spell = SPELL_BLESS_ASH_REV;
107 break;
108 case REP_EXALTED:
109 spell = SPELL_BLESS_ASH_EXA;
110 break;
111 default:
112 break;
113 }
114
115 if (spell)
116 {
117 creature->CastSpell(player, spell, true);
118 }
119 }
120
121 if (creature->GetEntry() == NPC_KELETH)
122 {
123 uint32 spell = 0;
124 switch (player->GetReputationRank(942)) //mark of war
125 {
126 case REP_FRIENDLY:
127 spell = SPELL_BLESS_KEL_FRI;
128 break;
129 case REP_HONORED:
130 spell = SPELL_BLESS_KEL_HON;
131 break;
132 case REP_REVERED:
133 spell = SPELL_BLESS_KEL_REV;
134 break;
135 case REP_EXALTED:
136 spell = SPELL_BLESS_KEL_EXA;
137 break;
138 default:
139 break;
140 }
141
142 if (spell)
143 {
144 creature->CastSpell(player, spell, true);
145 }
146 }
147 player->CLOSE_GOSSIP_MENU();
148 player->TalkedToCreature(creature->GetEntry(), creature->GetGUID());
149 }
150 return true;
151 }
152};

167class npc_cooshcoosh : public CreatureScript
168{
169public:
170 npc_cooshcoosh() : CreatureScript("npc_cooshcoosh") { }
171
172 struct npc_cooshcooshAI : public ScriptedAI
173 {
174 npc_cooshcooshAI(Creature* creature) : ScriptedAI(creature)
175 {
176 m_uiNormFaction = creature->getFaction();
177 Initialize();
178 }
179
180 void Initialize()
181 {
182 LightningBolt_Timer = 2000;
183 }
184
185 uint32 m_uiNormFaction;
186 uint32 LightningBolt_Timer;
187
188 void Reset() override
189 {
190 Initialize();
191 if (me->getFaction() != m_uiNormFaction)
192 me->setFaction(m_uiNormFaction);
193 }
194
195 void EnterCombat(Unit* /*who*/) override { }
196
197 void UpdateAI(uint32 diff) override
198 {
199 if (!UpdateVictim())
200 return;
201
202 if (LightningBolt_Timer <= diff)
203 {
204 DoCastVictim(SPELL_LIGHTNING_BOLT);
205 LightningBolt_Timer = 5000;
206 } else LightningBolt_Timer -= diff;
207
208 DoMeleeAttackIfReady();
209 }
210 };
211
212 CreatureAI* GetAI(Creature* creature) const override
213 {
214 return new npc_cooshcooshAI(creature);
215 }
216
217 bool OnGossipHello(Player* player, Creature* creature) override
218 {
219 if (player->GetQuestStatus(QUEST_CRACK_SKULLS) == QUEST_STATUS_INCOMPLETE)
220 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_COOSH, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
221
222 player->SEND_GOSSIP_MENU(9441, creature->GetGUID());
223 return true;
224 }
225
226 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
227 {
228 player->PlayerTalkClass->ClearMenus();
229 if (action == GOSSIP_ACTION_INFO_DEF)
230 {
231 player->CLOSE_GOSSIP_MENU();
232 creature->setFaction(FACTION_HOSTILE_CO);
233 creature->AI()->AttackStart(player);
234 }
235 return true;
236 }
237};

247class npc_elder_kuruti : public CreatureScript
248{
249public:
250 npc_elder_kuruti() : CreatureScript("npc_elder_kuruti") { }
251
252 bool OnGossipHello(Player* player, Creature* creature) override
253 {
254 if (player->GetQuestStatus(9803) == QUEST_STATUS_INCOMPLETE)
255 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KUR1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
256
257 player->SEND_GOSSIP_MENU(9226, creature->GetGUID());
258
259 return true;
260 }
261
262 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
263 {
264 player->PlayerTalkClass->ClearMenus();
265 switch (action)
266 {
267 case GOSSIP_ACTION_INFO_DEF:
268 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KUR2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
269 player->SEND_GOSSIP_MENU(9227, creature->GetGUID());
270 break;
271 case GOSSIP_ACTION_INFO_DEF + 1:
272 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KUR3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
273 player->SEND_GOSSIP_MENU(9229, creature->GetGUID());
274 break;
275 case GOSSIP_ACTION_INFO_DEF + 2:
276 {
277 if (!player->HasItemCount(24573))
278 {
279 ItemPosCountVec dest;
280 uint32 itemId = 24573;
281 InventoryResult msg = player->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, itemId, 1, NULL);
282 if (msg == EQUIP_ERR_OK)
283 {
284 player->StoreNewItem(dest, itemId, true);
285 }
286 else
287 player->SendEquipError(msg, NULL, NULL, itemId);
288 }
289 player->SEND_GOSSIP_MENU(9231, creature->GetGUID());
290 break;
291 }
292 }
293 return true;
294 }

340class npc_kayra_longmane : public CreatureScript
341{
342public:
343 npc_kayra_longmane() : CreatureScript("npc_kayra_longmane") { }
344
345 struct npc_kayra_longmaneAI : public npc_escortAI
346 {
347 npc_kayra_longmaneAI(Creature* creature) : npc_escortAI(creature) { }
348
349 void Reset() override { }
350
351 void WaypointReached(uint32 waypointId) override
352 {
353 Player* player = GetPlayerForEscort();
354 if (!player)
355 return;
356
357 switch (waypointId)
358 {
359 case 4:
360 Talk(SAY_AMBUSH1, player);
361 DoSpawnCreature(NPC_SLAVEBINDER, -10.0f, -5.0f, 0.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
362 DoSpawnCreature(NPC_SLAVEBINDER, -8.0f, 5.0f, 0.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
363 break;
364 case 5:
365 Talk(SAY_PROGRESS, player);
366 SetRun();
367 break;
368 case 16:
369 Talk(SAY_AMBUSH2, player);
370 DoSpawnCreature(NPC_SLAVEBINDER, -10.0f, -5.0f, 0.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
371 DoSpawnCreature(NPC_SLAVEBINDER, -8.0f, 5.0f, 0.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
372 break;
373 case 17:
374 SetRun(false);
375 break;
376 case 25:
377 Talk(SAY_END, player);
378 player->GroupEventHappens(QUEST_ESCAPE_FROM, me);
379 break;
380 }
381 }
382 };
383
384 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
385 {
386 if (quest->GetQuestId() == QUEST_ESCAPE_FROM)
387 {
388 creature->AI()->Talk(SAY_START, player);
389
390 if (npc_escortAI* pEscortAI = CAST_AI(npc_kayra_longmane::npc_kayra_longmaneAI, creature->AI()))
391 pEscortAI->Start(false, false, player->GetGUID());
392 }
393 return true;
394 }
395
396 CreatureAI* GetAI(Creature* creature) const override
397 {
398 return new npc_kayra_longmaneAI(creature);
399 }

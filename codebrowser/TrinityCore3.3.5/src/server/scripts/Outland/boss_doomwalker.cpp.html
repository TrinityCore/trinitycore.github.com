22enum Texts
23{
24 SAY_AGGRO = 0,
25 SAY_EARTHQUAKE = 1,
26 SAY_OVERRUN = 2,
27 SAY_SLAY = 3,
28 SAY_DEATH = 4
29};
30
31enum Spells
32{
33 SPELL_EARTHQUAKE = 32686,
34 SPELL_SUNDER_ARMOR = 33661,
35 SPELL_CHAIN_LIGHTNING = 33665,
36 SPELL_OVERRUN = 32636,
37 SPELL_ENRAGE = 33653,
38 SPELL_MARK_DEATH = 37128,
39 SPELL_AURA_DEATH = 37131
40};
41
42enum Events
43{
44 EVENT_ENRAGE = 1,
45 EVENT_ARMOR = 2,
46 EVENT_CHAIN = 3,
47 EVENT_QUAKE = 4,
48 EVENT_OVERRUN = 5
49};
50
51class boss_doomwalker : public CreatureScript
52{
53 public:
54 boss_doomwalker() : CreatureScript("boss_doomwalker") { }
55
56 struct boss_doomwalkerAI : public ScriptedAI
57 {
58 boss_doomwalkerAI(Creature* creature) : ScriptedAI(creature)
59 {
60 Initialize();
61 }
62
63 void Initialize()
64 {
65 _inEnrage = false;
66 }
67
68 void Reset() override
69 {
70 _events.Reset();
71 _events.ScheduleEvent(EVENT_ENRAGE, 0);
72 _events.ScheduleEvent(EVENT_ARMOR, urand(5000, 13000));
73 _events.ScheduleEvent(EVENT_CHAIN, urand(10000, 30000));
74 _events.ScheduleEvent(EVENT_QUAKE, urand(25000, 35000));
75 _events.ScheduleEvent(EVENT_OVERRUN, urand(30000, 45000));
76 Initialize();
77 }
78
79 void KilledUnit(Unit* victim) override
80 {
81 victim->CastSpell(victim, SPELL_MARK_DEATH, 0);
82
83 if (urand(0, 4))
84 return;
85
86 Talk(SAY_SLAY);
87 }
88
89 void JustDied(Unit* /*killer*/) override
90 {
91 Talk(SAY_DEATH);
92 }
93
94 void EnterCombat(Unit* /*who*/) override
95 {
96 Talk(SAY_AGGRO);
97 }
98
99 void MoveInLineOfSight(Unit* who) override
100
101 {
102 if (who && who->GetTypeId() == TYPEID_PLAYER && me->IsValidAttackTarget(who))
103 if (who->HasAura(SPELL_MARK_DEATH))
104 who->CastSpell(who, SPELL_AURA_DEATH, 1);
105 }
106
107 void UpdateAI(uint32 diff) override
108 {
109 if (!UpdateVictim())
110 return;
111
112 _events.Update(diff);
113
114 if (me->HasUnitState(UNIT_STATE_CASTING))
115 return;
116
117 while (uint32 eventId = _events.ExecuteEvent())
118 {
119 switch (eventId)
120 {
121 case EVENT_ENRAGE:
122 if (!HealthAbovePct(20))
123 {
124 DoCast(me, SPELL_ENRAGE);
125 _events.ScheduleEvent(EVENT_ENRAGE, 6000);
126 _inEnrage = true;
127 }
128 break;
129 case EVENT_OVERRUN:
130 Talk(SAY_OVERRUN);
131 DoCastVictim(SPELL_OVERRUN);
132 _events.ScheduleEvent(EVENT_OVERRUN, urand(25000, 40000));
133 break;
134 case EVENT_QUAKE:
135 if (urand(0, 1))
136 return;
137
138 Talk(SAY_EARTHQUAKE);
139
140 //remove enrage before casting earthquake because enrage + earthquake = 16000dmg over 8sec and all dead
141 if (_inEnrage)
142 me->RemoveAurasDueToSpell(SPELL_ENRAGE);
143
144 DoCast(me, SPELL_EARTHQUAKE);
145 _events.ScheduleEvent(EVENT_QUAKE, urand(30000, 55000));
146 break;
147 case EVENT_CHAIN:
148 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
149 DoCast(target, SPELL_CHAIN_LIGHTNING);
150 _events.ScheduleEvent(EVENT_CHAIN, urand(7000, 27000));
151 break;
152 case EVENT_ARMOR:
153 DoCastVictim(SPELL_SUNDER_ARMOR);
154 _events.ScheduleEvent(EVENT_ARMOR, urand(10000, 25000));
155 break;
156 default:
157 break;
158 }
159 }
160 DoMeleeAttackIfReady();
161 }
162
163 private:
164 EventMap _events;
165 bool _inEnrage;
166 };
167
168 CreatureAI* GetAI(Creature* creature) const override
169 {
170 return new boss_doomwalkerAI(creature);
171 }
172};
173
174void AddSC_boss_doomwalker()
175{
176 new boss_doomwalker();
177}

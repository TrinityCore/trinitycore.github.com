61class boss_najentus : public CreatureScript
62{
63public:
64 boss_najentus() : CreatureScript("boss_najentus") { }
65
66 struct boss_najentusAI : public BossAI
67 {
68 boss_najentusAI(Creature* creature) : BossAI(creature, DATA_HIGH_WARLORD_NAJENTUS)
69 {
70 }
71
72 void Reset() override
73 {
74 _Reset();
75 SpineTargetGUID.Clear();
76 }
77
78 void KilledUnit(Unit* /*victim*/) override
79 {
80 Talk(SAY_SLAY);
81 events.DelayEvents(5000, GCD_YELL);
82 }
83
84 void JustDied(Unit* /*killer*/) override
85 {
86 _JustDied();
87 Talk(SAY_DEATH);
88 }
89
90 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
91 {
92 if (spell->Id == SPELL_HURL_SPINE && me->HasAura(SPELL_TIDAL_SHIELD))
93 {
94 me->RemoveAurasDueToSpell(SPELL_TIDAL_SHIELD);
95 DoCast(me, SPELL_TIDAL_BURST, true);
96 ResetTimer();
97 }
98 }
99
100 void EnterCombat(Unit* /*who*/) override
101 {
102 _EnterCombat();
103 Talk(SAY_AGGRO);
104 events.ScheduleEvent(EVENT_BERSERK, 480000, GCD_CAST);
105 events.ScheduleEvent(EVENT_YELL, 45000 + (rand32() % 76) * 1000, GCD_YELL);
106 ResetTimer();
107 }
108
109 bool RemoveImpalingSpine()
110 {
111 if (!SpineTargetGUID)
112 return false;
113
114 Unit* target = ObjectAccessor::GetUnit(*me, SpineTargetGUID);
115 if (target && target->HasAura(SPELL_IMPALING_SPINE))
116 target->RemoveAurasDueToSpell(SPELL_IMPALING_SPINE);
117 SpineTargetGUID.Clear();
118 return true;
119 }
120
121 void ResetTimer(uint32 inc = 0)
122 {
123 events.RescheduleEvent(EVENT_NEEDLE, 10000 + inc, GCD_CAST);
124 events.RescheduleEvent(EVENT_SPINE, 20000 + inc, GCD_CAST);
125 events.RescheduleEvent(EVENT_SHIELD, 60000 + inc);
126 }
127
128 void ExecuteEvent(uint32 eventId) override
129 {
130 switch (eventId)
131 {
132 case EVENT_SHIELD:
133 DoCast(me, SPELL_TIDAL_SHIELD, true);
134 ResetTimer(45000);
135 break;
136 case EVENT_BERSERK:
137 Talk(SAY_ENRAGE);
138 DoCast(me, SPELL_BERSERK, true);
139 events.DelayEvents(15000, GCD_YELL);
140 break;
141 case EVENT_SPINE:
142 {
143 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1);
144
145 if (!target)
146 target = me->GetVictim();
147
148 if (target)
149 {
150 DoCast(target, SPELL_IMPALING_SPINE, true);
151 SpineTargetGUID = target->GetGUID();
152 //must let target summon, otherwise you cannot click the spine
153 target->SummonGameObject(GO_NAJENTUS_SPINE, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), me->GetOrientation(), 0, 0, 0, 0, 30);
154 Talk(SAY_NEEDLE);
155 events.DelayEvents(1500, GCD_CAST);
156 events.DelayEvents(15000, GCD_YELL);
157 }
158 events.ScheduleEvent(EVENT_SPINE, 21000, GCD_CAST);
159 return;
160 }
161 case EVENT_NEEDLE:
162 {
163 //DoCast(me, SPELL_NEEDLE_SPINE, true);
164 std::list<Unit*> targets;
165 SelectTargetList(targets, 3, SELECT_TARGET_RANDOM, 80, true);
166 for (std::list<Unit*>::const_iterator i = targets.begin(); i != targets.end(); ++i)
167 DoCast(*i, 39835, true);
168 events.ScheduleEvent(EVENT_NEEDLE, urand(15000, 25000), GCD_CAST);
169 events.DelayEvents(1500, GCD_CAST);
170 return;
171 }
172 case EVENT_YELL:
173 Talk(SAY_SPECIAL);
174 events.ScheduleEvent(EVENT_YELL, urand(25000, 100000), GCD_YELL);
175 events.DelayEvents(15000, GCD_YELL);
176 break;
177 default:
178 break;
179 }
180 }
181
182 private:
183 ObjectGuid SpineTargetGUID;
184 };
185
186 CreatureAI* GetAI(Creature* creature) const override
187 {
188 return GetBlackTempleAI<boss_najentusAI>(creature);
189 }
190};

192class go_najentus_spine : public GameObjectScript
193{
194public:
195 go_najentus_spine() : GameObjectScript("go_najentus_spine") { }
196
197 bool OnGossipHello(Player* player, GameObject* go) override
198 {
199 if (InstanceScript* instance = go->GetInstanceScript())
200 if (Creature* Najentus = ObjectAccessor::GetCreature(*go, instance->GetGuidData(DATA_HIGH_WARLORD_NAJENTUS)))
201 if (ENSURE_AI(boss_najentus::boss_najentusAI, Najentus->AI())->RemoveImpalingSpine())
202 {
203 player->CastSpell(player, SPELL_CREATE_NAJENTUS_SPINE, true);
204 go->Delete();
205 }
206 return true;
207 }
208
209};

213 new boss_najentus();
214 new go_najentus_spine();

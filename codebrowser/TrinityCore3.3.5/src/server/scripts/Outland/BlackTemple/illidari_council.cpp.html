228 struct npc_illidari_councilAI : public ScriptedAI
229 {
230 npc_illidari_councilAI(Creature* creature) : ScriptedAI(creature)
231 {
232 Initialize();
233 instance = creature->GetInstanceScript();
234 }
235
236 void Initialize()
237 {
238 CheckTimer = 2000;
239 EndEventTimer = 0;
240
241 DeathCount = 0;
242 EventBegun = false;
243 }
244
245 InstanceScript* instance;
246
247 ObjectGuid Council[4];
248
249 uint32 CheckTimer;
250 uint32 EndEventTimer;
251
252 uint8 DeathCount;
253
254 bool EventBegun;
255
256 void Reset() override
257 {
258 Initialize();
259
260 Creature* pMember = NULL;
261 for (uint8 i = 0; i < 4; ++i)
262 {
263 pMember = ObjectAccessor::GetCreature((*me), Council[i]);
264 if (!pMember)
265 continue;
266
267 if (!pMember->IsAlive())
268 {
269 pMember->RemoveCorpse();
270 pMember->Respawn();
271 }
272 pMember->AI()->EnterEvadeMode();
273 }
274
275 instance->SetBossState(DATA_ILLIDARI_COUNCIL, NOT_STARTED);
276 if (Creature* VoiceTrigger = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BLOOD_ELF_COUNCIL_VOICE)))
277 VoiceTrigger->AI()->EnterEvadeMode();
278
279 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
280 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
281 me->SetDisplayId(11686);
282 }
283
284 void EnterCombat(Unit* /*who*/) override { }
285 void AttackStart(Unit* /*who*/) override { }
286 void MoveInLineOfSight(Unit* /*who*/) override { }
287
288
289 void StartEvent(Unit* target)
290 {
291 if (target && target->IsAlive())
292 {
293 Council[0] = instance->GetGuidData(DATA_GATHIOS_THE_SHATTERER);
294 Council[1] = instance->GetGuidData(DATA_HIGH_NETHERMANCER_ZEREVOR);
295 Council[2] = instance->GetGuidData(DATA_LADY_MALANDE);
296 Council[3] = instance->GetGuidData(DATA_VERAS_DARKSHADOW);
297
298 // Start the event for the Voice Trigger
299 if (Creature* VoiceTrigger = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BLOOD_ELF_COUNCIL_VOICE)))
300 {
301 ENSURE_AI(npc_blood_elf_council_voice_trigger::npc_blood_elf_council_voice_triggerAI, VoiceTrigger->AI())->LoadCouncilGUIDs();
302 ENSURE_AI(npc_blood_elf_council_voice_trigger::npc_blood_elf_council_voice_triggerAI, VoiceTrigger->AI())->EventStarted = true;
303 }
304
305 for (uint8 i = 0; i < 4; ++i)
306 {
307 if (Council[i])
308 {
309 if (Creature* member = ObjectAccessor::GetCreature(*me, Council[i]))
310 if (member->IsAlive())
311 member->AI()->AttackStart(target);
312 }
313 }
314
315 instance->SetBossState(DATA_ILLIDARI_COUNCIL, IN_PROGRESS);
316
317 EventBegun = true;
318 }
319 }
320
321 void UpdateAI(uint32 diff) override
322 {
323 if (!EventBegun)
324 return;
325
326 if (EndEventTimer)
327 {
328 if (EndEventTimer <= diff)
329 {
330 if (DeathCount > 3)
331 {
332 if (Creature* VoiceTrigger = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BLOOD_ELF_COUNCIL_VOICE)))
333 VoiceTrigger->DealDamage(VoiceTrigger, VoiceTrigger->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
334 instance->SetBossState(DATA_ILLIDARI_COUNCIL, DONE);
335 //me->SummonCreature(AKAMAID, 746.466980f, 304.394989f, 311.90208f, 6.272870f, TEMPSUMMON_DEAD_DESPAWN, 0);
336 me->DealDamage(me, me->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
337 return;
338 }
339
340 Creature* pMember = (ObjectAccessor::GetCreature(*me, Council[DeathCount]));
341 if (pMember && pMember->IsAlive())
342 pMember->DealDamage(pMember, pMember->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
343 ++DeathCount;
344 EndEventTimer = 1500;
345 } else EndEventTimer -= diff;
346 }
347
348 if (CheckTimer)
349 {
350 if (CheckTimer <= diff)
351 {
352 uint8 EvadeCheck = 0;
353 for (uint8 i = 0; i < 4; ++i)
354 {
355 if (Council[i])
356 {
357 if (Creature* Member = (ObjectAccessor::GetCreature((*me), Council[i])))
358 {
359 // This is the evade/death check.
360 if (Member->IsAlive() && !Member->GetVictim())
361 ++EvadeCheck; //If all members evade, we reset so that players can properly reset the event
362 else if (!Member->IsAlive()) // If even one member dies, kill the rest, set instance data, and kill self.
363 {
364 EndEventTimer = 1000;
365 CheckTimer = 0;
366 return;
367 }
368 }
369 }
370 }
371
372 if (EvadeCheck > 3)
373 Reset();
374
375 CheckTimer = 2000;
376 } else CheckTimer -= diff;
377 }
378
379 }
380 };

384struct boss_illidari_councilAI : public ScriptedAI
385{
386 boss_illidari_councilAI(Creature* creature) : ScriptedAI(creature)
387 {
388 instance = creature->GetInstanceScript();
389 LoadedGUIDs = false;
390 }
391
392 ObjectGuid Council[4];
393
394 InstanceScript* instance;
395
396 bool LoadedGUIDs;
397
398 void EnterCombat(Unit* who) override
399 {
400 if (Creature* controller = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ILLIDARI_COUNCIL)))
401 ENSURE_AI(npc_illidari_council::npc_illidari_councilAI, controller->AI())->StartEvent(who);
402 DoZoneInCombat();
403 // Load GUIDs on first aggro because the Creature guids are only set as the creatures are created in world-
404 // this means that for each creature, it will attempt to LoadGUIDs even though some of the other creatures are
405 // not in world, and thus have no GUID set in the instance data system. Putting it in aggro ensures that all the creatures
406 // have been loaded and have their GUIDs set in the instance data system.
407 if (!LoadedGUIDs)
408 LoadGUIDs();
409 }
410
411 void EnterEvadeMode(EvadeReason why) override
412 {
413 for (uint8 i = 0; i < 4; ++i)
414 {
415 if (Unit* unit = ObjectAccessor::GetUnit(*me, Council[i]))
416 if (unit != me && unit->GetVictim())
417 {
418 AttackStart(unit->GetVictim());
419 return;
420 }
421 }
422 ScriptedAI::EnterEvadeMode(why);
423 }
424
425 void DamageTaken(Unit* done_by, uint32 &damage) override
426 {
427 if (done_by == me)
428 return;
429
430 damage /= 4;
431 for (uint8 i = 0; i < 4; ++i)
432 {
433 if (Creature* unit = ObjectAccessor::GetCreature(*me, Council[i]))
434 if (unit != me && damage < unit->GetHealth())
435 {
436 unit->ModifyHealth(-int32(damage));
437 unit->LowerPlayerDamageReq(damage);
438 }
439 }
440 }
441
442 void LoadGUIDs()
443 {
444 Council[0] = instance->GetGuidData(DATA_LADY_MALANDE);
445 Council[1] = instance->GetGuidData(DATA_HIGH_NETHERMANCER_ZEREVOR);
446 Council[2] = instance->GetGuidData(DATA_GATHIOS_THE_SHATTERER);
447 Council[3] = instance->GetGuidData(DATA_VERAS_DARKSHADOW);
448
449 LoadedGUIDs = true;
450 }

872 }
873 } else VanishTimer -= diff;
874
875 DoMeleeAttackIfReady();
876 }

883 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);

886 DeadlyPoisonTimer += 6000;
887 VanishTimer = 90000;
888 AppearEnvenomTimer = 4000;
889 HasVanished = false;
890 } else VanishTimer -= diff;
891
892 if (AppearEnvenomTimer <= diff) // Appear 2 seconds before becoming attackable (Shifting out of vanish)
893 {
894 me->GetMotionMaster()->Clear();
895 me->GetMotionMaster()->MoveChase(me->GetVictim());
896 me->SetVisible(true);
897 AppearEnvenomTimer = 6000;
898 } else AppearEnvenomTimer -= diff;
899 }
900 }

138 struct boss_reliquary_of_soulsAI : public ScriptedAI
139 {
140 boss_reliquary_of_soulsAI(Creature* creature) : ScriptedAI(creature)
141 {
142 Initialize();
143 instance = creature->GetInstanceScript();
144 Counter = 0;
145 Timer = 0;

166 void Reset() override
167 {
168 instance->SetBossState(DATA_RELIQUARY_OF_SOULS, NOT_STARTED);
169
170 if (EssenceGUID)
171 {
172 if (Creature* essence = ObjectAccessor::GetCreature(*me, EssenceGUID))
173 essence->DespawnOrUnsummon();
174
175 EssenceGUID.Clear();
176 }
177
178 Initialize();
179
180 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
181 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_NONE);
182 me->RemoveAurasDueToSpell(SPELL_SUBMERGE);
183 }

185 void MoveInLineOfSight(Unit* who) override
186 {
187 if (!who)
188 return;
189
190 if (me->IsInCombat())
191 return;
192
193 if (who->GetTypeId() != TYPEID_PLAYER)
194 return;
195
196 if (me->GetDistance(who) > 50.0f)
197 return;
198
199 AttackStartNoMove(who);
200 }
201
202 void EnterCombat(Unit* who) override
203 {
204 me->AddThreat(who, 10000.0f);
205 DoZoneInCombat();
206 instance->SetBossState(DATA_RELIQUARY_OF_SOULS, IN_PROGRESS);
207
208 Phase = 1;
209 Counter = 0;
210 Timer = 0;
211 }

213 bool SummonSoul()
214 {
215 uint32 random = rand32() % 6;
216 float x = Coords[random].x;
217 float y = Coords[random].y;
218
219 Creature* Soul = me->SummonCreature(CREATURE_ENSLAVED_SOUL, x, y, me->GetPositionZ(), me->GetOrientation(), TEMPSUMMON_CORPSE_DESPAWN, 0);
220 if (!Soul)
221 return false;
222
223 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
224 {
225 ENSURE_AI(npc_enslaved_soul::npc_enslaved_soulAI, Soul->AI())->ReliquaryGUID = me->GetGUID();
226 Soul->AI()->AttackStart(target);
227 } else EnterEvadeMode();
228 return true;
229 }
230
231 void MergeThreatList(Creature* target)
232 {
233 if (!target)
234 return;
235
236 ThreatContainer::StorageType threatlist = target->getThreatManager().getThreatList();
237 for (ThreatContainer::StorageType::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
238 {
239 Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid());
240 if (unit)
241 {
242 DoModifyThreatPercent(unit, -100);
243 float threat = target->getThreatManager().getThreat(unit);
244 me->AddThreat(unit, threat); // This makes it so that the unit has the same amount of threat in Reliquary's threatlist as in the target creature's (One of the Essences).
245 }
246 }
247 }
248
249 void JustDied(Unit* /*killer*/) override
250 {
251 instance->SetBossState(DATA_RELIQUARY_OF_SOULS, DONE);
252 }
253
254 void UpdateAI(uint32 diff) override
255 {
256 if (!Phase)
257 return;
258
259 if (me->getThreatManager().getThreatList().empty()) // Reset if event is begun and we don't have a threatlist
260 {
261 EnterEvadeMode();
262 return;
263 }
264
265 Creature* Essence = NULL;
266 if (EssenceGUID)
267 {
268 Essence = ObjectAccessor::GetCreature(*me, EssenceGUID);
269 if (!Essence)
270 {
271 EnterEvadeMode();
272 return;
273 }
274 }
275
276 if (Timer <= diff)
277 {
278 switch (Counter)
279 {
280 case 0:
281 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_READY2H); // I R ANNNGRRRY!
282 DoStartNoMovement(me);
283 Timer = 3000;
284 break;
285 case 1:
286 Timer = 2800;
287 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_SUBMERGE); // Release the cube
288 DoCast(me, SPELL_SUBMERGE);
289 DoStartNoMovement(me);
290 break;
291 case 2:
292 Timer = 5000;
293 if (Creature* Summon = DoSpawnCreature(23417+Phase, 0, 0, 0, 0, TEMPSUMMON_DEAD_DESPAWN, 0))
294 {
295 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_SUBMERGED); // Ribs: open
296 Summon->AI()->AttackStart(SelectTarget(SELECT_TARGET_TOPAGGRO, 0));
297 EssenceGUID = Summon->GetGUID();
298 DoStartNoMovement(me);
299 } else EnterEvadeMode();
300 break;
301 case 3:
302 Timer = 1000;
303 if (Phase == 3)
304 {
305 if (Essence && !Essence->IsAlive())
306 DoCast(me, 7, true);
307 else return;
308 }
309 else
310 {
311 if (Essence && Essence->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE))
312 {
313 MergeThreatList(Essence);
314 Essence->RemoveAllAuras();
315 Essence->DeleteThreatList();
316 Essence->GetMotionMaster()->MoveFollow(me, 0.0f, 0.0f);
317 } else return;
318 }
319 break;
320 case 4:
321 Timer = 1500;
322 if (Essence)
323 {
324 if (Essence->IsWithinDistInMap(me, 10))
325 {
326 Essence->SetUInt32Value(UNIT_NPC_EMOTESTATE, 374); //rotate and disappear
327 Timer = 2000;
328 me->RemoveAurasDueToSpell(SPELL_SUBMERGE);
329 }
330 else
331 {
332 MergeThreatList(Essence);
333 Essence->RemoveAllAuras();
334 Essence->DeleteThreatList();
335 Essence->GetMotionMaster()->MoveFollow(me, 0, 0);
336 return;
337 }
338 }
339 break;
340 case 5:
341 if (Essence)
342 {
343 if (Phase == 1)
344 Essence->AI()->Talk(SUFF_SAY_AFTER);
345 else
346 Essence->AI()->Talk(DESI_SAY_AFTER);
347
348 Essence->DespawnOrUnsummon();
349 }
350
351 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
352 EssenceGUID.Clear();
353 SoulCount = 0;
354 SoulDeathCount = 0;
355 Timer = 3000;
356 break;
357 case 6:
358 if (SoulCount < NUMBER_ENSLAVED_SOUL)
359 {
360 if (SummonSoul())
361 ++SoulCount;
362 Timer = 500;
363 return;
364 }
365 break;
366 case 7:
367 if (SoulDeathCount >= SoulCount)
368 {
369 Counter = 1;
370 ++Phase;
371 Timer = 5000;
372 }
373 return;
374 default:
375 break;
376 }
377 ++Counter;
378 } else Timer -= diff;
379 }
380 };

418 uint32 AggroYellTimer;

421 uint32 SoulDrainTimer;
422 uint32 AuraTimer;

424 void Reset() override
425 {
426 Initialize();
427 }
428
429 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override
430 {
431 if (damage >= me->GetHealth())
432 {
433 damage = 0;
434 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
435 Talk(SUFF_SAY_RECAP);
436 me->SetReactState(REACT_PASSIVE);
437 }
438 }
439
440 void EnterCombat(Unit* /*who*/) override
441 {
442 if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE))
443 {
444 Talk(SUFF_SAY_FREED);
445 DoZoneInCombat();
446 DoCast(me, AURA_OF_SUFFERING, true); // linked aura need core support
447 DoCast(me, ESSENCE_OF_SUFFERING_PASSIVE, true);
448 DoCast(me, ESSENCE_OF_SUFFERING_PASSIVE2, true);
449 }
450 }
451
452 void KilledUnit(Unit* /*victim*/) override
453 {
454 Talk(SUFF_SAY_SLAY);
455 }
456
457 void CastFixate()
458 {
459 ThreatContainer::StorageType const &threatlist = me->getThreatManager().getThreatList();
460 if (threatlist.empty())
461 return; // No point continuing if empty threatlist.
462 std::list<Unit*> targets;
463 ThreatContainer::StorageType::const_iterator itr = threatlist.begin();
464 for (; itr != threatlist.end(); ++itr)
465 {
466 Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid());
467 if (unit && unit->IsAlive() && (unit->GetTypeId() == TYPEID_PLAYER)) // Only alive players
468 targets.push_back(unit);
469 }
470 if (targets.empty())
471 return; // No targets added for some reason. No point continuing.
472 targets.sort(Trinity::ObjectDistanceOrderPred(me)); // Sort players by distance.
473 targets.resize(1); // Only need closest target.
474 Unit* target = targets.front(); // Get the first target.
475 if (target)
476 target->CastSpell(me, SPELL_FIXATE_TAUNT, true);
477 DoResetThreat();
478 me->AddThreat(target, 1000000);
479 }
480
481 void UpdateAI(uint32 diff) override
482 {
483 if (me->IsInCombat())
484 {
485 //Supposed to be cast on nearest target
486 if (FixateTimer <= diff)
487 {
488 CastFixate();
489 FixateTimer = 5000;
490 if (!(rand32() % 16))
491 {
492 Talk(SUFF_SAY_AGGRO);
493 }
494 } else FixateTimer -= diff;
495 }
496
497 //Return since we have no target
498 if (!UpdateVictim())
499 return;
500
501 if (EnrageTimer <= diff)
502 {
503 DoCast(me, SPELL_ENRAGE);
504 EnrageTimer = 60000;
505 Talk(SUFF_EMOTE_ENRAGE);
506 } else EnrageTimer -= diff;
507
508 if (SoulDrainTimer <= diff)
509 {
510 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0), SPELL_SOUL_DRAIN);
511 SoulDrainTimer = 60000;
512 } else SoulDrainTimer -= diff;
513
514 DoMeleeAttackIfReady();
515 }
516 };

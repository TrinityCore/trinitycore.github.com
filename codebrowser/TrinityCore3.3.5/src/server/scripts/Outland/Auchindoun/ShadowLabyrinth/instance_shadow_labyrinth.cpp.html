18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "ScriptedCreature.h"
21#include "shadow_labyrinth.h"
22
23DoorData const doorData[] =
24{
25 { GO_REFECTORY_DOOR, DATA_BLACKHEART_THE_INCITER, DOOR_TYPE_PASSAGE },
26 { GO_SCREAMING_HALL_DOOR, DATA_GRANDMASTER_VORPIL, DOOR_TYPE_PASSAGE },
27 { 0, 0, DOOR_TYPE_ROOM } // END
28};
29
30class instance_shadow_labyrinth : public InstanceMapScript
31{
32 public:
33 instance_shadow_labyrinth() : InstanceMapScript(SLScriptName, 555) { }
34
35 struct instance_shadow_labyrinth_InstanceMapScript : public InstanceScript
36 {
37 instance_shadow_labyrinth_InstanceMapScript(Map* map) : InstanceScript(map)
38 {
39 SetHeaders(DataHeader);
40 SetBossNumber(EncounterCount);
41 LoadDoorData(doorData);
42
43 FelOverseerCount = 0;
44 }
45
46 void OnCreatureCreate(Creature* creature) override
47 {
48 switch (creature->GetEntry())
49 {
50 case NPC_AMBASSADOR_HELLMAW:
51 AmbassadorHellmawGUID = creature->GetGUID();
52 break;
53 case NPC_GRANDMASTER_VORPIL:
54 GrandmasterVorpilGUID = creature->GetGUID();
55 break;
56 case NPC_FEL_OVERSEER:
57 if (creature->IsAlive())
58 {
59 ++FelOverseerCount;
60 if (Creature* hellmaw = instance->GetCreature(AmbassadorHellmawGUID))
61 hellmaw->AI()->DoAction(ACTION_AMBASSADOR_HELLMAW_BANISH);
62 }
63 break;
64 default:
65 break;
66 }
67 }
68
69 void OnGameObjectCreate(GameObject* go) override
70 {
71 switch (go->GetEntry())
72 {
73 case GO_REFECTORY_DOOR:
74 case GO_SCREAMING_HALL_DOOR:
75 AddDoor(go, true);
76 break;
77 default:
78 break;
79 }
80 }
81
82 void OnGameObjectRemove(GameObject* go) override
83 {
84 switch (go->GetEntry())
85 {
86 case GO_REFECTORY_DOOR:
87 case GO_SCREAMING_HALL_DOOR:
88 AddDoor(go, false);
89 break;
90 default:
91 break;
92 }
93 }
94
95 void OnUnitDeath(Unit* unit) override
96 {
97 Creature* creature = unit->ToCreature();
98 if (!creature)
99 return;
100
101 if (creature->GetEntry() == NPC_FEL_OVERSEER)
102 {
103 if (FelOverseerCount)
104 --FelOverseerCount;
105
106 if (!FelOverseerCount)
107 if (Creature* hellmaw = instance->GetCreature(AmbassadorHellmawGUID))
108 hellmaw->AI()->DoAction(ACTION_AMBASSADOR_HELLMAW_INTRO);
109 }
110 }
111
112 uint32 GetData(uint32 type) const override
113 {
114 switch (type)
115 {
116 case DATA_FEL_OVERSEER:
117 return !FelOverseerCount ? 1 : 0;
118 default:
119 break;
120 }
121 return 0;
122 }
123
124 ObjectGuid GetGuidData(uint32 type) const override
125 {
126 switch (type)
127 {
128 case DATA_GRANDMASTER_VORPIL:
129 return GrandmasterVorpilGUID;
130 default:
131 break;
132 }
133 return ObjectGuid::Empty;
134 }
135
136 protected:
137 ObjectGuid AmbassadorHellmawGUID;
138 ObjectGuid GrandmasterVorpilGUID;
139 uint32 FelOverseerCount;
140 };
141
142 InstanceScript* GetInstanceScript(InstanceMap* map) const override
143 {
144 return new instance_shadow_labyrinth_InstanceMapScript(map);
145 }
146};
147
148void AddSC_instance_shadow_labyrinth()
149{
150 new instance_shadow_labyrinth();
151}

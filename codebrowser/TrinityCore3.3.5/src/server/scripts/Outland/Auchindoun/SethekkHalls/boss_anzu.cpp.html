63class boss_anzu : public CreatureScript
64{
65 public:
66 boss_anzu() : CreatureScript("boss_anzu") { }
67
68 struct boss_anzuAI : public BossAI
69 {
70 boss_anzuAI(Creature* creature) : BossAI(creature, DATA_ANZU)
71 {
72 Initialize();
73 }
74
75 void Initialize()
76 {
77 _under33Percent = false;
78 _under66Percent = false;
79 }
80
81 void Reset() override
82 {
83 //_Reset();
84 events.Reset();
85 Initialize();
86 }
87
88 void EnterCombat(Unit* /*who*/) override
89 {
90 _EnterCombat();
91 events.ScheduleEvent(EVENT_PARALYZING_SCREECH, 14000);
92 events.ScheduleEvent(EVENT_CYCLONE_OF_FEATHERS, 5000);
93 }
94
95 void JustDied(Unit* /*killer*/) override
96 {
97 _JustDied();
98 }
99
100 void DamageTaken(Unit* /*killer*/, uint32 &damage) override
101 {
102 if (me->HealthBelowPctDamaged(33, damage) && !_under33Percent)
103 {
104 _under33Percent = true;
105 Talk(SAY_SUMMON_BROOD);
106 events.ScheduleEvent(EVENT_SUMMON, 3000);
107 }
108
109 if (me->HealthBelowPctDamaged(66, damage) && !_under66Percent)
110 {
111 _under66Percent = true;
112 Talk(SAY_SUMMON_BROOD);
113 events.ScheduleEvent(EVENT_SUMMON, 3000);
114 }
115 }
116
117 void UpdateAI(uint32 diff) override
118 {
119 if (!UpdateVictim())
120 return;
121
122 events.Update(diff);
123
124 while (uint32 eventId = events.ExecuteEvent())
125 {
126 switch (eventId)
127 {
128 case EVENT_PARALYZING_SCREECH:
129 DoCastVictim(SPELL_PARALYZING_SCREECH);
130 events.ScheduleEvent(EVENT_PARALYZING_SCREECH, 26000);
131 break;
132 case EVENT_CYCLONE_OF_FEATHERS:
133 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
134 DoCast(target, SPELL_CYCLONE_OF_FEATHERS);
135 events.ScheduleEvent(EVENT_CYCLONE_OF_FEATHERS, 21000);
136 break;
137 case EVENT_SUMMON:
138 // TODO: Add pathing for Brood of Anzu
139 for (uint8 i = 0; i < 7; i++)
140 me->SummonCreature(NPC_BROOD_OF_ANZU, PosSummonBrood[i], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 46000);
141
142 DoCast(me, SPELL_BANISH_SELF);
143 events.ScheduleEvent(EVENT_SPELL_BOMB, 12000);
144 break;
145 case EVENT_SPELL_BOMB:
146 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
147 {
148 if (target->getPowerType() == POWER_MANA)
149 {
150 DoCast(target, SPELL_SPELL_BOMB);
151 Talk(SAY_SPELL_BOMB, target);
152 }
153 }
154 break;
155 default:
156 break;
157 }
158 }
159
160 DoMeleeAttackIfReady();
161 }
162
163 private:
164 bool _under33Percent;
165 bool _under66Percent;
166 };
167
168 CreatureAI* GetAI(Creature* creature) const override
169 {
170 return GetSethekkHallsAI<boss_anzuAI>(creature);
171 }
172};

55class npc_raliq_the_drunk : public CreatureScript
56{
57public:
58 npc_raliq_the_drunk() : CreatureScript("npc_raliq_the_drunk") { }
59
60 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
61 {
62 player->PlayerTalkClass->ClearMenus();
63 if (action == GOSSIP_ACTION_INFO_DEF+1)
64 {
65 player->CLOSE_GOSSIP_MENU();
66 creature->setFaction(FACTION_HOSTILE_RD);
67 creature->AI()->AttackStart(player);
68 }
69 return true;
70 }
71
72 bool OnGossipHello(Player* player, Creature* creature) override
73 {
74 if (player->GetQuestStatus(QUEST_CRACK_SKULLS) == QUEST_STATUS_INCOMPLETE)
75 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_VENDOR, GOSSIP_RALIQ, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
76
77 player->SEND_GOSSIP_MENU(9440, creature->GetGUID());
78 return true;
79 }
80
81 CreatureAI* GetAI(Creature* creature) const override
82 {
83 return new npc_raliq_the_drunkAI(creature);
84 }

86 struct npc_raliq_the_drunkAI : public ScriptedAI
87 {
88 npc_raliq_the_drunkAI(Creature* creature) : ScriptedAI(creature)
89 {
90 Initialize();
91 }
92
93 void Initialize()
94 {
95 Uppercut_Timer = 5000;
96 }
97
98 uint32 Uppercut_Timer;
99
100 void Reset() override
101 {
102 Initialize();
103 me->RestoreFaction();
104 }
105
106 void UpdateAI(uint32 diff) override
107 {
108 if (!UpdateVictim())
109 return;
110
111 if (Uppercut_Timer <= diff)
112 {
113 DoCastVictim(SPELL_UPPERCUT);
114 Uppercut_Timer = 15000;
115 } else Uppercut_Timer -= diff;
116
117 DoMeleeAttackIfReady();
118 }
119 };

212/*
213##################################################
214Shattrath City Flask Vendors provides flasks to people exalted with 3 fActions:
215Haldor the Compulsive
216Arcanist Xorith
217Both sell special flasks for use in Outlands 25man raids only,
218purchasable for one Mark of Illidari each
219Purchase requires exalted reputation with Scryers/Aldor, Cenarion Expedition and The Sha'tar
220##################################################
221*/
222
223class npc_shattrathflaskvendors : public CreatureScript
224{
225public:
226 npc_shattrathflaskvendors() : CreatureScript("npc_shattrathflaskvendors") { }
227
228 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
229 {
230 player->PlayerTalkClass->ClearMenus();
231 if (action == GOSSIP_ACTION_TRADE)
232 player->GetSession()->SendListInventory(creature->GetGUID());
233
234 return true;
235 }
236
237 bool OnGossipHello(Player* player, Creature* creature) override
238 {
239 if (creature->GetEntry() == 23484)
240 {
241 // Aldor vendor
242 if (creature->IsVendor() && (player->GetReputationRank(932) == REP_EXALTED) && (player->GetReputationRank(935) == REP_EXALTED) && (player->GetReputationRank(942) == REP_EXALTED))
243 {
244 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_VENDOR, GOSSIP_TEXT_BROWSE_GOODS, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRADE);
245 player->SEND_GOSSIP_MENU(11085, creature->GetGUID());
246 }
247 else
248 {
249 player->SEND_GOSSIP_MENU(11083, creature->GetGUID());
250 }
251 }
252
253 if (creature->GetEntry() == 23483)
254 {
255 // Scryers vendor
256 if (creature->IsVendor() && (player->GetReputationRank(934) == REP_EXALTED) && (player->GetReputationRank(935) == REP_EXALTED) && (player->GetReputationRank(942) == REP_EXALTED))
257 {
258 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_VENDOR, GOSSIP_TEXT_BROWSE_GOODS, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRADE);
259 player->SEND_GOSSIP_MENU(11085, creature->GetGUID());
260 }
261 else
262 {
263 player->SEND_GOSSIP_MENU(11084, creature->GetGUID());
264 }
265 }
266
267 return true;
268 }

425 void MoveInLineOfSight(Unit* who) override
426
427 {
428 if (HasEscortState(STATE_ESCORT_ESCORTING))
429 return;
430
431 Player* player = who->ToPlayer();
432 if (player && player->GetQuestStatus(10211) == QUEST_STATUS_INCOMPLETE)
433 {
434 float Radius = 10.0f;
435 if (me->IsWithinDistInMap(who, Radius))
436 {
437 Start(false, false, who->GetGUID());
438 }
439 }

66class npc_invis_infernal_caster : public CreatureScript
67{
68public:
69 npc_invis_infernal_caster() : CreatureScript("npc_invis_infernal_caster") { }
70
71 struct npc_invis_infernal_casterAI : public ScriptedAI
72 {
73 npc_invis_infernal_casterAI(Creature* creature) : ScriptedAI(creature)
74 {
75 ground = 0.f;
76 }
77
78 void Reset() override
79 {
80 ground = me->GetMap()->GetHeight(me->GetPositionX(), me->GetPositionY(), me->GetPositionZMinusOffset());
81 SummonInfernal();
82 events.ScheduleEvent(EVENT_CAST_SUMMON_INFERNAL, urand(1000, 3000));
83 }
84
85 void SetData(uint32 id, uint32 data) override
86 {
87 if (id == TYPE_INFERNAL && data == DATA_DIED)
88 SummonInfernal();
89 }
90
91 void SummonInfernal()
92 {
93 Creature* infernal = me->SummonCreature(NPC_INFERNAL_ATTACKER, me->GetPositionX(), me->GetPositionY(), ground + 0.05f, 0.0f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 60000);
94 infernalGUID = infernal->GetGUID();
95 }
96
97 void UpdateAI(uint32 diff) override
98 {
99 events.Update(diff);
100
101 while (uint32 eventId = events.ExecuteEvent())
102 {
103 switch (eventId)
104 {
105 case EVENT_CAST_SUMMON_INFERNAL:
106 {
107 if (Unit* infernal = ObjectAccessor::GetUnit(*me, infernalGUID))
108 if (infernal->GetDisplayId() == MODEL_INVISIBLE)
109 me->CastSpell(infernal, SPELL_SUMMON_INFERNAL, true);
110 events.ScheduleEvent(EVENT_CAST_SUMMON_INFERNAL, 12000);
111 break;
112 }
113 default:
114 break;
115 }
116 }
117 }
118
119 private:
120 EventMap events;
121 ObjectGuid infernalGUID;
122 float ground;
123 };
124
125 CreatureAI* GetAI(Creature* creature) const override
126 {
127 return new npc_invis_infernal_casterAI(creature);
128 }

218 struct npc_mature_netherwing_drakeAI : public ScriptedAI
219 {
220 npc_mature_netherwing_drakeAI(Creature* creature) : ScriptedAI(creature)
221 {
222 Initialize();
223 }
224
225 void Initialize()
226 {
227 uiPlayerGUID.Clear();
228
229 bCanEat = false;
230 bIsEating = false;
231
232 EatTimer = 5000;
233 CastTimer = 5000;
234 }
235
236 ObjectGuid uiPlayerGUID;
237
238 bool bCanEat;
239 bool bIsEating;
240
241 uint32 EatTimer;
242 uint32 CastTimer;
243
244 void Reset() override
245 {
246 Initialize();
247 }
248
249 void SpellHit(Unit* pCaster, SpellInfo const* spell) override
250 {
251 if (bCanEat || bIsEating)
252 return;
253
254 if (pCaster->GetTypeId() == TYPEID_PLAYER && spell->Id == SPELL_PLACE_CARCASS && !me->HasAura(SPELL_JUST_EATEN))
255 {
256 uiPlayerGUID = pCaster->GetGUID();
257 bCanEat = true;
258 }
259 }
260
261 void MovementInform(uint32 type, uint32 id) override
262 {
263 if (type != POINT_MOTION_TYPE)
264 return;
265
266 if (id == POINT_ID)
267 {
268 bIsEating = true;
269 EatTimer = 7000;
270 me->HandleEmoteCommand(EMOTE_ONESHOT_ATTACK_UNARMED);
271 }
272 }
273
274 void UpdateAI(uint32 diff) override
275 {
276 if (bCanEat || bIsEating)
277 {
278 if (EatTimer <= diff)
279 {
280 if (bCanEat && !bIsEating)
281 {
282 if (Unit* unit = ObjectAccessor::GetUnit(*me, uiPlayerGUID))
283 {
284 if (GameObject* go = unit->FindNearestGameObject(GO_CARCASS, 10))
285 {
286 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
287 me->GetMotionMaster()->MovementExpired();
288
289 me->GetMotionMaster()->MoveIdle();
290 me->StopMoving();
291
292 me->GetMotionMaster()->MovePoint(POINT_ID, go->GetPositionX(), go->GetPositionY(), go->GetPositionZ());
293 }
294 }
295 bCanEat = false;
296 }
297 else if (bIsEating)
298 {
299 DoCast(me, SPELL_JUST_EATEN);
300 Talk(SAY_JUST_EATEN);
301
302 if (Player* player = ObjectAccessor::GetPlayer(*me, uiPlayerGUID))
303 {
304 player->KilledMonsterCredit(NPC_EVENT_PINGER);
305
306 if (GameObject* go = player->FindNearestGameObject(GO_CARCASS, 10))
307 go->Delete();
308 }
309
310 Reset();
311 me->GetMotionMaster()->Clear();
312 }
313 }
314 else
315 EatTimer -= diff;
316
317 return;
318 }
319
320 if (!UpdateVictim())
321 return;
322
323 if (CastTimer <= diff)
324 {
325 DoCastVictim(SPELL_NETHER_BREATH);
326 CastTimer = 5000;
327 } else CastTimer -= diff;
328
329 DoMeleeAttackIfReady();
330 }
331 };

383 me->SetVisible(true);

386 void SpellHit(Unit* caster, const SpellInfo* spell) override
387 {
388 if (!caster)
389 return;
390
391 if (caster->GetTypeId() == TYPEID_PLAYER && spell->Id == SPELL_HIT_FORCE_OF_NELTHARAKU && !Tapped)
392 {
393 Tapped = true;
394 PlayerGUID = caster->GetGUID();
395
396 me->setFaction(FACTION_FRIENDLY);
397 DoCast(caster, SPELL_FORCE_OF_NELTHARAKU, true);
398
399 Unit* Dragonmaw = me->FindNearestCreature(NPC_DRAGONMAW_SUBJUGATOR, 50);
400 if (Dragonmaw)
401 {
402 me->AddThreat(Dragonmaw, 100000.0f);
403 AttackStart(Dragonmaw);

407 if (ref)
408 ref->removeReference();
409 }
410 }
411
412 void MovementInform(uint32 type, uint32 id) override
413 {
414 if (type != POINT_MOTION_TYPE)
415 return;
416
417 if (id == 1)
418 {

420 {
421 Unit* player = ObjectAccessor::GetUnit(*me, PlayerGUID);
422 if (player)
423 DoCast(player, SPELL_FORCE_OF_NELTHARAKU, true);
424
425 PlayerGUID.Clear();
426 }
427 me->SetVisible(false);
428 me->SetDisableGravity(false);
429 me->DealDamage(me, me->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
430 me->RemoveCorpse();
431 }
432 }

435 {
436 if (!UpdateVictim())
437 {

439 {
440 if (FlyTimer <= diff)
441 {

444 {
445 Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
446 if (player && player->GetQuestStatus(10854) == QUEST_STATUS_INCOMPLETE)
447 {
448 DoCast(player, SPELL_FORCE_OF_NELTHARAKU, true);

53class boss_mekgineer_steamrigger : public CreatureScript
54{
55public:
56 boss_mekgineer_steamrigger() : CreatureScript("boss_mekgineer_steamrigger") { }

63 struct boss_mekgineer_steamriggerAI : public ScriptedAI
64 {
65 boss_mekgineer_steamriggerAI(Creature* creature) : ScriptedAI(creature)
66 {
67 Initialize();
68 instance = creature->GetInstanceScript();
69 }
70
71 void Initialize()
72 {
73 Shrink_Timer = 20000;
74 Saw_Blade_Timer = 15000;
75 Electrified_Net_Timer = 10000;
76
77 Summon75 = false;
78 Summon50 = false;
79 Summon25 = false;
80 }
81
82 InstanceScript* instance;
83
84 uint32 Shrink_Timer;
85 uint32 Saw_Blade_Timer;
86 uint32 Electrified_Net_Timer;
87 bool Summon75;
88 bool Summon50;
89 bool Summon25;
90
91 void Reset() override
92 {
93 Initialize();
94
95 instance->SetBossState(DATA_MEKGINEER_STEAMRIGGER, NOT_STARTED);
96 }
97
98 void JustDied(Unit* /*killer*/) override
99 {
100 Talk(SAY_DEATH);
101
102 instance->SetBossState(DATA_MEKGINEER_STEAMRIGGER, DONE);
103 }
104
105 void KilledUnit(Unit* /*victim*/) override
106 {
107 Talk(SAY_SLAY);
108 }
109
110 void EnterCombat(Unit* /*who*/) override
111 {
112 Talk(SAY_AGGRO);
113
114 instance->SetBossState(DATA_MEKGINEER_STEAMRIGGER, IN_PROGRESS);
115 }
116
117 //no known summon spells exist
118 void SummonMechanichs()
119 {
120 Talk(SAY_MECHANICS);
121
122 DoSpawnCreature(NPC_STREAMRIGGER_MECHANIC, 5, 5, 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 240000);
123 DoSpawnCreature(NPC_STREAMRIGGER_MECHANIC, -5, 5, 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 240000);
124 DoSpawnCreature(NPC_STREAMRIGGER_MECHANIC, -5, -5, 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 240000);
125
126 if (rand32() % 2)
127 DoSpawnCreature(NPC_STREAMRIGGER_MECHANIC, 5, -7, 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 240000);
128 if (rand32() % 2)
129 DoSpawnCreature(NPC_STREAMRIGGER_MECHANIC, 7, -5, 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 240000);
130 }
131
132 void UpdateAI(uint32 diff) override
133 {
134 if (!UpdateVictim())
135 return;
136
137 if (Shrink_Timer <= diff)
138 {
139 DoCastVictim(SPELL_SUPER_SHRINK_RAY);
140 Shrink_Timer = 20000;
141 } else Shrink_Timer -= diff;
142
143 if (Saw_Blade_Timer <= diff)
144 {
145 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
146 DoCast(target, SPELL_SAW_BLADE);
147 else
148 DoCastVictim(SPELL_SAW_BLADE);
149
150 Saw_Blade_Timer = 15000;
151 } else Saw_Blade_Timer -= diff;
152
153 if (Electrified_Net_Timer <= diff)
154 {
155 DoCastVictim(SPELL_ELECTRIFIED_NET);
156 Electrified_Net_Timer = 10000;
157 }
158 else Electrified_Net_Timer -= diff;
159
160 if (!Summon75)
161 {
162 if (HealthBelowPct(75))
163 {
164 SummonMechanichs();
165 Summon75 = true;
166 }
167 }
168
169 if (!Summon50)
170 {
171 if (HealthBelowPct(50))
172 {
173 SummonMechanichs();
174 Summon50 = true;
175 }
176 }
177
178 if (!Summon25)
179 {
180 if (HealthBelowPct(25))
181 {
182 SummonMechanichs();
183 Summon25 = true;
184 }
185 }
186
187 DoMeleeAttackIfReady();
188 }
189 };
190
191};

243 if (me->IsWithinDistInMap(mekgineer, MAX_REPAIR_RANGE))
244 {
245 //are we already channeling? Doesn't work very well, find better check?

249 //me->GetMotionMaster()->MoveIdle();
250
251 DoCast(me, SPELL_REPAIR, true);
252 }
253 Repair_Timer = 5000;
254 }
255 else
256 {
257 //me->GetMotionMaster()->MovementExpired();
258 //me->GetMotionMaster()->MoveFollow(pMekgineer, 0, 0);
259 }
260 }
261 } else Repair_Timer = 5000;
262 } else Repair_Timer -= diff;

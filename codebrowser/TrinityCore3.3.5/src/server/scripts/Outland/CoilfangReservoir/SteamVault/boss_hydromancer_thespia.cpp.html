44class boss_hydromancer_thespia : public CreatureScript
45{
46 public:
47 boss_hydromancer_thespia() : CreatureScript("boss_hydromancer_thespia") { }
48
49 struct boss_thespiaAI : public BossAI
50 {
51 boss_thespiaAI(Creature* creature) : BossAI(creature, DATA_HYDROMANCER_THESPIA) { }
52
53 void Reset() override
54 {
55 _Reset();
56 }
57
58 void JustDied(Unit* /*killer*/) override
59 {
60 Talk(SAY_DEAD);
61 _JustDied();
62 }
63
64 void KilledUnit(Unit* who) override
65 {
66 if (who->GetTypeId() == TYPEID_PLAYER)
67 Talk(SAY_SLAY);
68 }
69
70 void EnterCombat(Unit* /*who*/) override
71 {
72 Talk(SAY_AGGRO);
73 _EnterCombat();
74
75 events.ScheduleEvent(EVENT_LIGHTNING_CLOUD, 15000);
76 events.ScheduleEvent(EVENT_LUNG_BURST, 7000);
77 events.ScheduleEvent(EVENT_ENVELOPING_WINDS, 9000);
78 }
79
80 void ExecuteEvent(uint32 eventId) override
81 {
82 switch (eventId)
83 {
84 case EVENT_LIGHTNING_CLOUD:
85 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30.0f, true))
86 DoCast(target, SPELL_LIGHTNING_CLOUD);
87 // cast twice in Heroic mode
88 if (IsHeroic())
89 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30.0f, true))
90 DoCast(target, SPELL_LIGHTNING_CLOUD);
91
92 events.ScheduleEvent(EVENT_LIGHTNING_CLOUD, urand(15000, 25000));
93 break;
94 case EVENT_LUNG_BURST:
95 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40.0f, true))
96 DoCast(target, SPELL_LUNG_BURST);
97 events.ScheduleEvent(EVENT_LUNG_BURST, urand(7000, 12000));
98 break;
99 case EVENT_ENVELOPING_WINDS:
100 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 35.0f, true))
101 DoCast(target, SPELL_ENVELOPING_WINDS);
102 // cast twice in Heroic mode
103 if (IsHeroic())
104 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 35.0f, true))
105 DoCast(target, SPELL_ENVELOPING_WINDS);
106
107 events.ScheduleEvent(EVENT_ENVELOPING_WINDS, urand(10000, 15000));
108 break;
109 default:
110 break;
111 }
112 }
113 };
114
115 CreatureAI* GetAI(Creature* creature) const override
116 {
117 return GetInstanceAI<boss_thespiaAI>(creature);
118 }
119};

127class npc_coilfang_waterelemental : public CreatureScript
128{
129 public:
130 npc_coilfang_waterelemental() : CreatureScript("npc_coilfang_waterelemental") { }
131
132 struct npc_coilfang_waterelementalAI : public ScriptedAI
133 {
134 npc_coilfang_waterelementalAI(Creature* creature) : ScriptedAI(creature) { }
135
136 void Reset() override
137 {
138 _events.Reset();
139 }
140
141 void EnterCombat(Unit* /*who*/) override
142 {
143 _events.ScheduleEvent(EVENT_WATER_BOLT_VOLLEY, urand(3000, 6000));
144 }
145
146 void UpdateAI(uint32 diff) override
147 {
148 if (!UpdateVictim())
149 return;
150
151 _events.Update(diff);
152
153 if (me->HasUnitState(UNIT_STATE_CASTING))
154 return;
155
156 while (uint32 eventId = _events.ExecuteEvent())
157 {
158 switch (eventId)
159 {
160 case EVENT_WATER_BOLT_VOLLEY:
161 DoCast(me, SPELL_WATER_BOLT_VOLLEY);
162 _events.ScheduleEvent(EVENT_WATER_BOLT_VOLLEY, urand(7000, 12000));
163 break;
164 default:
165 break;
166 }
167 }
168
169 DoMeleeAttackIfReady();
170 }
171
172 private:
173 EventMap _events;
174 };
175
176 CreatureAI* GetAI(Creature* creature) const override
177 {
178 return new npc_coilfang_waterelementalAI(creature);
179 }
180};

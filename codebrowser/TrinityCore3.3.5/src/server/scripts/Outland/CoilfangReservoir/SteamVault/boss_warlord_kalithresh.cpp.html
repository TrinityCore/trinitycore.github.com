30enum NagaDistiller
31{
32 SAY_INTRO = 0,
33 SAY_REGEN = 1,
34 SAY_AGGRO = 2,
35 SAY_SLAY = 3,
36 SAY_DEATH = 4,
37
38 SPELL_SPELL_REFLECTION = 31534,
39 SPELL_IMPALE = 39061,
40 SPELL_WARLORDS_RAGE = 37081,
41 SPELL_WARLORDS_RAGE_NAGA = 31543,
42
43 SPELL_WARLORDS_RAGE_PROC = 36453
44};

46class npc_naga_distiller : public CreatureScript
47{
48public:
49 npc_naga_distiller() : CreatureScript("npc_naga_distiller") { }
50
51 CreatureAI* GetAI(Creature* creature) const override
52 {
53 return GetInstanceAI<npc_naga_distillerAI>(creature);
54 }
55
56 struct npc_naga_distillerAI : public ScriptedAI
57 {
58 npc_naga_distillerAI(Creature* creature) : ScriptedAI(creature)
59 {
60 instance = creature->GetInstanceScript();
61 }
62
63 InstanceScript* instance;
64
65 void Reset() override
66 {
67 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
68 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
69
70 //hack, due to really weird spell behaviour :(
71 if (instance->GetData(DATA_DISTILLER) == IN_PROGRESS)
72 {
73 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
74 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
75 }
76 }
77
78 void EnterCombat(Unit* /*who*/) override { }
79
80 void StartRageGen(Unit* /*caster*/)
81 {
82 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
83 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
84
85 DoCast(me, SPELL_WARLORDS_RAGE_NAGA, true);
86
87 instance->SetData(DATA_DISTILLER, IN_PROGRESS);
88 }
89
90 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override
91 {
92 if (me->GetHealth() <= damage)
93 instance->SetData(DATA_DISTILLER, DONE);
94 }
95 };
96
97};

99class boss_warlord_kalithresh : public CreatureScript
100{
101public:
102 boss_warlord_kalithresh() : CreatureScript("boss_warlord_kalithresh") { }
103
104 CreatureAI* GetAI(Creature* creature) const override
105 {
106 return GetInstanceAI<boss_warlord_kalithreshAI>(creature);
107 }
108
109 struct boss_warlord_kalithreshAI : public ScriptedAI
110 {
111 boss_warlord_kalithreshAI(Creature* creature) : ScriptedAI(creature)
112 {
113 Initialize();
114 instance = creature->GetInstanceScript();
115 }
116
117 void Initialize()
118 {
119 Reflection_Timer = 10000;
120 Impale_Timer = 7000 + rand32() % 7000;
121 Rage_Timer = 45000;
122 CanRage = false;
123 }
124
125 InstanceScript* instance;
126
127 uint32 Reflection_Timer;
128 uint32 Impale_Timer;
129 uint32 Rage_Timer;
130 bool CanRage;
131
132 void Reset() override
133 {
134 Initialize();
135
136 instance->SetBossState(DATA_WARLORD_KALITHRESH, NOT_STARTED);
137 }
138
139 void EnterCombat(Unit* /*who*/) override
140 {
141 Talk(SAY_AGGRO);
142
143 instance->SetBossState(DATA_WARLORD_KALITHRESH, IN_PROGRESS);
144 }
145
146 void KilledUnit(Unit* /*victim*/) override
147 {
148 Talk(SAY_SLAY);
149 }
150
151 void SpellHit(Unit* /*caster*/, const SpellInfo* spell) override
152 {
153 //hack :(
154 if (spell->Id == SPELL_WARLORDS_RAGE_PROC)
155 if (instance->GetData(DATA_DISTILLER) == DONE)
156 me->RemoveAurasDueToSpell(SPELL_WARLORDS_RAGE_PROC);
157 }
158
159 void JustDied(Unit* /*killer*/) override
160 {
161 Talk(SAY_DEATH);
162
163 instance->SetBossState(DATA_WARLORD_KALITHRESH, DONE);
164 }
165
166 void UpdateAI(uint32 diff) override
167 {
168 if (!UpdateVictim())
169 return;
170
171 if (Rage_Timer <= diff)
172 {
173 if (Creature* distiller = me->FindNearestCreature(17954, 100.0f))
174 {
175 Talk(SAY_REGEN);
176 DoCast(me, SPELL_WARLORDS_RAGE);
177 ENSURE_AI(npc_naga_distiller::npc_naga_distillerAI, distiller->AI())->StartRageGen(me);
178 }
179 Rage_Timer = 3000 + rand32() % 15000;
180 } else Rage_Timer -= diff;
181
182 //Reflection_Timer
183 if (Reflection_Timer <= diff)
184 {
185 DoCast(me, SPELL_SPELL_REFLECTION);
186 Reflection_Timer = 15000 + rand32() % 10000;
187 } else Reflection_Timer -= diff;
188
189 //Impale_Timer
190 if (Impale_Timer <= diff)
191 {
192 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
193 DoCast(target, SPELL_IMPALE);
194
195 Impale_Timer = 7500 + rand32() % 5000;
196 } else Impale_Timer -= diff;
197
198 DoMeleeAttackIfReady();
199 }
200 };
201};

206 new boss_warlord_kalithresh();

85 boss_hydross_the_unstable() : CreatureScript("boss_hydross_the_unstable") { }

92 struct boss_hydross_the_unstableAI : public ScriptedAI
93 {
94 boss_hydross_the_unstableAI(Creature* creature) : ScriptedAI(creature), Summons(me)
95 {
96 Initialize();
97 instance = creature->GetInstanceScript();
98 }
99
100 void Initialize()
101 {
102 beams[0].Clear();
103 beams[1].Clear();
104 PosCheck_Timer = 2500;
105 MarkOfHydross_Timer = 15000;
106 MarkOfCorruption_Timer = 15000;
107 WaterTomb_Timer = 7000;
108 VileSludge_Timer = 7000;
109 MarkOfHydross_Count = 0;
110 MarkOfCorruption_Count = 0;
111 EnrageTimer = 600000;
112
113 CorruptedForm = false;
114 beam = false;
115 }
116
117 InstanceScript* instance;
118
119 ObjectGuid beams[2];
120 uint32 PosCheck_Timer;
121 uint32 MarkOfHydross_Timer;
122 uint32 MarkOfCorruption_Timer;
123 uint32 WaterTomb_Timer;
124 uint32 VileSludge_Timer;
125 uint32 MarkOfHydross_Count;
126 uint32 MarkOfCorruption_Count;
127 uint32 EnrageTimer;
128 bool CorruptedForm;
129 bool beam;
130 SummonList Summons;
131
132 void Reset() override
133 {
134 DeSummonBeams();
135 Initialize();
136
137 me->SetMeleeDamageSchool(SPELL_SCHOOL_FROST);
138 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FROST, true);
139 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, false);
140
141 me->SetDisplayId(MODEL_CLEAN);
142
143 instance->SetData(DATA_HYDROSSTHEUNSTABLEEVENT, NOT_STARTED);
144 Summons.DespawnAll();
145 }
146
147 void SummonBeams()
148 {
149 Creature* beamer = me->SummonCreature(ENTRY_BEAM_DUMMY, -258.333f, -356.34f, 22.0499f, 5.90835f, TEMPSUMMON_CORPSE_DESPAWN, 0);
150 if (beamer)
151 {
152 beamer->CastSpell(me, SPELL_BLUE_BEAM, true);
153 beamer->SetDisplayId(11686); //invisible
154 beamer->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
155 beams[0] = beamer->GetGUID();
156 }
157 beamer = me->SummonCreature(ENTRY_BEAM_DUMMY, -219.918f, -371.308f, 22.0042f, 2.73072f, TEMPSUMMON_CORPSE_DESPAWN, 0);
158 if (beamer)
159 {
160 beamer->CastSpell(me, SPELL_BLUE_BEAM, true);
161 beamer->SetDisplayId(11686); //invisible
162 beamer->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
163 beams[1] = beamer->GetGUID();
164 }
165 }
166 void DeSummonBeams()
167 {
168 for (uint8 i = 0; i < 2; ++i)
169 {
170 if (Creature* mob = ObjectAccessor::GetCreature(*me, beams[i]))
171 {
172 mob->setDeathState(DEAD);
173 mob->RemoveCorpse();
174 }
175 }
176 }
177 void EnterCombat(Unit* /*who*/) override
178 {
179 Talk(SAY_AGGRO);
180
181 instance->SetData(DATA_HYDROSSTHEUNSTABLEEVENT, IN_PROGRESS);
182 }
183
184 void KilledUnit(Unit* /*victim*/) override
185 {
186 Talk(CorruptedForm ? SAY_CORRUPT_SLAY : SAY_CLEAN_SLAY);
187 }
188
189 void JustSummoned(Creature* summoned) override
190 {
191 if (summoned->GetEntry() == ENTRY_PURE_SPAWN)
192 {
193 summoned->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FROST, true);
194 summoned->CastSpell(summoned, SPELL_ELEMENTAL_SPAWNIN, true);
195 Summons.Summon(summoned);
196 }
197 if (summoned->GetEntry() == ENTRY_TAINTED_SPAWN)
198 {
199 summoned->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, true);
200 summoned->CastSpell(summoned, SPELL_ELEMENTAL_SPAWNIN, true);
201 Summons.Summon(summoned);
202 }
203 }
204
205 void SummonedCreatureDespawn(Creature* summon) override
206 {
207 Summons.Despawn(summon);
208 }
209
210 void JustDied(Unit* /*killer*/) override
211 {
212 Talk(CorruptedForm ? SAY_CORRUPT_DEATH : SAY_CLEAN_DEATH);
213
214 instance->SetData(DATA_HYDROSSTHEUNSTABLEEVENT, DONE);
215 Summons.DespawnAll();
216 }

218 void UpdateAI(uint32 diff) override
219 {
220 if (!beam)
221 {
222 SummonBeams();
223 beam=true;
224 }
225 //Return since we have no target
226 if (!UpdateVictim())
227 return;
228
229 // corrupted form
230 if (CorruptedForm)
231 {
232 //MarkOfCorruption_Timer
233 if (MarkOfCorruption_Timer <= diff)
234 {
235 if (MarkOfCorruption_Count <= 5)
236 {
237 uint32 mark_spell = 0;
238
239 switch (MarkOfCorruption_Count)
240 {
241 case 0:
242 mark_spell = SPELL_MARK_OF_CORRUPTION1;
243 break;
244
245 case 1:
246 mark_spell = SPELL_MARK_OF_CORRUPTION2;
247 break;
248
249 case 2:
250 mark_spell = SPELL_MARK_OF_CORRUPTION3;
251 break;
252
253 case 3:
254 mark_spell = SPELL_MARK_OF_CORRUPTION4;
255 break;
256
257 case 4:
258 mark_spell = SPELL_MARK_OF_CORRUPTION5;
259 break;
260
261 case 5:
262 mark_spell = SPELL_MARK_OF_CORRUPTION6;
263 break;
264 }
265
266 DoCastVictim(mark_spell);
267
268 if (MarkOfCorruption_Count < 5)
269 ++MarkOfCorruption_Count;
270 }
271
272 MarkOfCorruption_Timer = 15000;
273 } else MarkOfCorruption_Timer -= diff;
274
275 //VileSludge_Timer
276 if (VileSludge_Timer <= diff)
277 {
278 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
279 DoCast(target, SPELL_VILE_SLUDGE);
280
281 VileSludge_Timer = 15000;
282 } else VileSludge_Timer -= diff;
283
284 //PosCheck_Timer
285 if (PosCheck_Timer <= diff)
286 {
287 if (me->IsWithinDist2d(HYDROSS_X, HYDROSS_Y, SWITCH_RADIUS))
288 {
289 // switch to clean form
290 me->SetDisplayId(MODEL_CLEAN);
291 CorruptedForm = false;
292 MarkOfHydross_Count = 0;
293
294 Talk(SAY_SWITCH_TO_CLEAN);
295 DoResetThreat();
296 SummonBeams();
297
298 // spawn 4 adds
299 DoSpawnCreature(ENTRY_PURE_SPAWN, SPAWN_X_DIFF1, SPAWN_Y_DIFF1, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
300 DoSpawnCreature(ENTRY_PURE_SPAWN, SPAWN_X_DIFF2, SPAWN_Y_DIFF2, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
301 DoSpawnCreature(ENTRY_PURE_SPAWN, SPAWN_X_DIFF3, SPAWN_Y_DIFF3, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
302 DoSpawnCreature(ENTRY_PURE_SPAWN, SPAWN_X_DIFF4, SPAWN_Y_DIFF4, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
303
304 me->SetMeleeDamageSchool(SPELL_SCHOOL_FROST);
305 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FROST, true);
306 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, false);
307 }
308
309 PosCheck_Timer = 2500;
310 } else PosCheck_Timer -=diff;
311 }
312 // clean form
313 else
314 {
315 //MarkOfHydross_Timer
316 if (MarkOfHydross_Timer <= diff)
317 {
318 if (MarkOfHydross_Count <= 5)
319 {
320 uint32 mark_spell = 0;
321
322 switch (MarkOfHydross_Count)
323 {
324 case 0:
325 mark_spell = SPELL_MARK_OF_HYDROSS1;
326 break;
327
328 case 1:
329 mark_spell = SPELL_MARK_OF_HYDROSS2;
330 break;
331
332 case 2:
333 mark_spell = SPELL_MARK_OF_HYDROSS3;
334 break;
335
336 case 3:
337 mark_spell = SPELL_MARK_OF_HYDROSS4;
338 break;
339
340 case 4:
341 mark_spell = SPELL_MARK_OF_HYDROSS5;
342 break;
343
344 case 5:
345 mark_spell = SPELL_MARK_OF_HYDROSS6;
346 break;
347 }
348
349 DoCastVictim(mark_spell);
350
351 if (MarkOfHydross_Count < 5)
352 ++MarkOfHydross_Count;
353 }
354
355 MarkOfHydross_Timer = 15000;
356 } else MarkOfHydross_Timer -= diff;
357
358 //WaterTomb_Timer
359 if (WaterTomb_Timer <= diff)
360 {
361 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true);
362 if (target)
363 DoCast(target, SPELL_WATER_TOMB);
364
365 WaterTomb_Timer = 7000;
366 } else WaterTomb_Timer -= diff;
367
368 //PosCheck_Timer
369 if (PosCheck_Timer <= diff)
370 {
371 if (!me->IsWithinDist2d(HYDROSS_X, HYDROSS_Y, SWITCH_RADIUS))
372 {
373 // switch to corrupted form
374 me->SetDisplayId(MODEL_CORRUPT);
375 MarkOfCorruption_Count = 0;
376 CorruptedForm = true;
377
378 Talk(SAY_SWITCH_TO_CORRUPT);
379 DoResetThreat();
380 DeSummonBeams();
381
382 // spawn 4 adds
383 DoSpawnCreature(ENTRY_TAINTED_SPAWN, SPAWN_X_DIFF1, SPAWN_Y_DIFF1, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
384 DoSpawnCreature(ENTRY_TAINTED_SPAWN, SPAWN_X_DIFF2, SPAWN_Y_DIFF2, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
385 DoSpawnCreature(ENTRY_TAINTED_SPAWN, SPAWN_X_DIFF3, SPAWN_Y_DIFF3, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
386 DoSpawnCreature(ENTRY_TAINTED_SPAWN, SPAWN_X_DIFF4, SPAWN_Y_DIFF4, 3, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
387
388 me->SetMeleeDamageSchool(SPELL_SCHOOL_NATURE);
389 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, true);
390 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FROST, false);
391 }
392
393 PosCheck_Timer = 2500;
394 } else PosCheck_Timer -=diff;
395 }
396
397 //EnrageTimer
398 if (EnrageTimer <= diff)
399 {
400 DoCast(me, SPELL_ENRAGE);
401 EnrageTimer = 60000;
402 } else EnrageTimer -= diff;
403
404 DoMeleeAttackIfReady();
405 }
406 };
407};

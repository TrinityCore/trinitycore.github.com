86 struct instance_serpentshrine_cavern_InstanceMapScript : public InstanceScript

88 instance_serpentshrine_cavern_InstanceMapScript(Map* map) : InstanceScript(map)
89 {
90 SetHeaders(DataHeader);
91 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
92
93 StrangePool = 0;
94 Water = WATERSTATE_FRENZY;
95
96 ShieldGeneratorDeactivated[0] = false;
97 ShieldGeneratorDeactivated[1] = false;
98 ShieldGeneratorDeactivated[2] = false;
99 ShieldGeneratorDeactivated[3] = false;
100 FishingTimer = 1000;
101 WaterCheckTimer = 500;
102 FrenzySpawnTimer = 2000;
103 DoSpawnFrenzy = false;
104 TrashCount = 0;
105 }
106
107 bool IsEncounterInProgress() const override
108 {
109 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
110 if (m_auiEncounter[i] == IN_PROGRESS)
111 return true;
112
113 return false;
114 }

116 void Update(uint32 diff) override
117 {
118 //Water checks
119 if (WaterCheckTimer <= diff)
120 {
121 if (TrashCount >= MIN_KILLS)
122 Water = WATERSTATE_SCALDING;
123 else
124 Water = WATERSTATE_FRENZY;
125
126 Map::PlayerList const &PlayerList = instance->GetPlayers();
127 if (PlayerList.isEmpty())
128 return;
129 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
130 {
131 if (Player* player = i->GetSource())
132 {
133 if (player->IsAlive() && /*i->GetSource()->GetPositionZ() <= -21.434931f*/player->IsInWater())
134 {
135 if (Water == WATERSTATE_SCALDING)
136 {
137
138 if (!player->HasAura(SPELL_SCALDINGWATER))
139 {
140 player->CastSpell(player, SPELL_SCALDINGWATER, true);
141 }
142 } else if (Water == WATERSTATE_FRENZY)
143 {
144 //spawn frenzy
145 if (DoSpawnFrenzy)
146 {
147 if (Creature* frenzy = player->SummonCreature(NPC_COILFANG_FRENZY, player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), player->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 2000))
148 {
149 frenzy->Attack(player, false);
150 frenzy->SetSwim(true);
151 frenzy->SetDisableGravity(true);
152 }
153 DoSpawnFrenzy = false;
154 }
155 }
156 }
157 if (!player->IsInWater())
158 player->RemoveAurasDueToSpell(SPELL_SCALDINGWATER);
159 }
160
161 }
162 WaterCheckTimer = 500;//remove stress from core
163 }
164 else
165 WaterCheckTimer -= diff;
166
167 if (FrenzySpawnTimer <= diff)
168 {
169 DoSpawnFrenzy = true;
170 FrenzySpawnTimer = 2000;
171 }
172 else
173 FrenzySpawnTimer -= diff;
174 }

267 void SetData(uint32 type, uint32 data) override
268 {
269 switch (type)
270 {
271 case DATA_STRANGE_POOL:
272 StrangePool = data;
273 break;
274 case DATA_CONTROL_CONSOLE:
275 if (data == DONE)
276 {
277 HandleGameObject(BridgePart[0], true);
278 HandleGameObject(BridgePart[1], true);
279 HandleGameObject(BridgePart[2], true);
280 }
281 break;
282 case DATA_TRASH:
283 if (data == 1 && TrashCount < MIN_KILLS)
284 ++TrashCount;//+1 died
285 SaveToDB();
286 break;
287 case DATA_WATER:
288 Water = data;
289 break;
290 case DATA_HYDROSSTHEUNSTABLEEVENT:
291 m_auiEncounter[0] = data;
292 break;
293 case DATA_LEOTHERASTHEBLINDEVENT:
294 m_auiEncounter[1] = data;
295 break;
296 case DATA_THELURKERBELOWEVENT:
297 m_auiEncounter[2] = data;
298 break;
299 case DATA_KARATHRESSEVENT:
300 m_auiEncounter[3] = data;
301 break;
302 case DATA_MOROGRIMTIDEWALKEREVENT:
303 m_auiEncounter[4] = data;
304 break;
305 //Lady Vashj
306 case DATA_LADYVASHJEVENT:
307 if (data == NOT_STARTED)
308 {
309 ShieldGeneratorDeactivated[0] = false;
310 ShieldGeneratorDeactivated[1] = false;
311 ShieldGeneratorDeactivated[2] = false;
312 ShieldGeneratorDeactivated[3] = false;
313 }
314 m_auiEncounter[5] = data;
315 break;
316 case DATA_SHIELDGENERATOR1:
317 ShieldGeneratorDeactivated[0] = data != 0;
318 break;
319 case DATA_SHIELDGENERATOR2:
320 ShieldGeneratorDeactivated[1] = data != 0;
321 break;
322 case DATA_SHIELDGENERATOR3:
323 ShieldGeneratorDeactivated[2] = data != 0;
324 break;
325 case DATA_SHIELDGENERATOR4:
326 ShieldGeneratorDeactivated[3] = data != 0;
327 break;
328 default:
329 break;
330 }
331
332 if (data == DONE)
333 SaveToDB();
334 }

336 uint32 GetData(uint32 type) const override
337 {
338 switch (type)
339 {
340 case DATA_HYDROSSTHEUNSTABLEEVENT:
341 return m_auiEncounter[0];
342 case DATA_LEOTHERASTHEBLINDEVENT:
343 return m_auiEncounter[1];
344 case DATA_THELURKERBELOWEVENT:
345 return m_auiEncounter[2];
346 case DATA_KARATHRESSEVENT:
347 return m_auiEncounter[3];
348 case DATA_MOROGRIMTIDEWALKEREVENT:
349 return m_auiEncounter[4];
350 //Lady Vashj
351 case DATA_LADYVASHJEVENT:
352 return m_auiEncounter[5];
353 case DATA_SHIELDGENERATOR1:
354 return ShieldGeneratorDeactivated[0];
355 case DATA_SHIELDGENERATOR2:
356 return ShieldGeneratorDeactivated[1];
357 case DATA_SHIELDGENERATOR3:
358 return ShieldGeneratorDeactivated[2];
359 case DATA_SHIELDGENERATOR4:
360 return ShieldGeneratorDeactivated[3];
361 case DATA_CANSTARTPHASE3:
362 if (ShieldGeneratorDeactivated[0] && ShieldGeneratorDeactivated[1] && ShieldGeneratorDeactivated[2] && ShieldGeneratorDeactivated[3])
363 return 1;
364 break;
365 case DATA_STRANGE_POOL:
366 return StrangePool;
367 case DATA_WATER:
368 return Water;
369 default:
370 break;
371 }
372
373 return 0;

376 std::string GetSaveData() override
377 {
378 OUT_SAVE_INST_DATA;
379 std::ostringstream stream;
380 stream << m_auiEncounter[0] << ' ' << m_auiEncounter[1] << ' ' << m_auiEncounter[2] << ' '
381 << m_auiEncounter[3] << ' ' << m_auiEncounter[4] << ' ' << m_auiEncounter[5] << ' ' << TrashCount;
382 OUT_SAVE_INST_DATA_COMPLETE;
383 return stream.str();

385
386 void Load(const char* in) override
387 {
388 if (!in)
389 {
390 OUT_LOAD_INST_DATA_FAIL;
391 return;
392 }
393
394 OUT_LOAD_INST_DATA(in);
395 std::istringstream stream(in);
396 stream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3]
397 >> m_auiEncounter[4] >> m_auiEncounter[5] >> TrashCount;
398
399 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
400 if (m_auiEncounter[i] == IN_PROGRESS) // Do not load an encounter as "In Progress" - reset it instead.
401 m_auiEncounter[i] = NOT_STARTED;
402
403 OUT_LOAD_INST_DATA_COMPLETE;
404 }

80 boss_the_lurker_below() : CreatureScript("boss_the_lurker_below") { }

87 struct boss_the_lurker_belowAI : public ScriptedAI
88 {
89 boss_the_lurker_belowAI(Creature* creature) : ScriptedAI(creature), Summons(me)
90 {
91 Initialize();
92 SetCombatMovement(false);
93 instance = creature->GetInstanceScript();
94 }
95
96 void Initialize()
97 {
98 SpoutAnimTimer = 1000;
99 RotTimer = 0;
100 WaterboltTimer = 15000; // give time to get in range when fight starts
101 SpoutTimer = 45000;
102 WhirlTimer = 18000; // after avery spout
103 PhaseTimer = 120000;
104 GeyserTimer = rand32() % 5000 + 15000;
105 CheckTimer = 15000; // give time to get in range when fight starts
106 WaitTimer = 60000; // never reached
107 WaitTimer2 = 60000; // never reached
108
109 Submerged = true; // will be false at combat start
110 Spawned = false;
111 InRange = false;
112 CanStartEvent = false;
113 }
114
115 InstanceScript* instance;
116 SummonList Summons;
117
118 bool Spawned;
119 bool Submerged;
120 bool InRange;
121 bool CanStartEvent;
122 uint32 RotTimer;
123 uint32 SpoutAnimTimer;
124 uint32 WaterboltTimer;
125 uint32 SpoutTimer;
126 uint32 WhirlTimer;
127 uint32 PhaseTimer;
128 uint32 GeyserTimer;
129 uint32 CheckTimer;
130 uint32 WaitTimer;
131 uint32 WaitTimer2;
132
133 bool CheckCanStart()//check if players fished
134 {
135 if (instance->GetData(DATA_STRANGE_POOL) == NOT_STARTED)
136 return false;
137 return true;
138 }
139
140 void Reset() override
141 {
142 me->SetSwim(true);
143 me->SetDisableGravity(true);
144 Initialize();
145
146 Summons.DespawnAll();
147
148 instance->SetData(DATA_THELURKERBELOWEVENT, NOT_STARTED);
149 instance->SetData(DATA_STRANGE_POOL, NOT_STARTED);
150 DoCast(me, SPELL_SUBMERGE); // submerge anim
151 me->SetVisible(false); // we start invis under water, submerged
152 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
153 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
154 }
155
156 void JustDied(Unit* /*killer*/) override
157 {
158 instance->SetData(DATA_THELURKERBELOWEVENT, DONE);
159 instance->SetData(DATA_STRANGE_POOL, IN_PROGRESS);
160
161 Summons.DespawnAll();
162 }
163
164 void EnterCombat(Unit* /*who*/) override
165 {
166 instance->SetData(DATA_THELURKERBELOWEVENT, IN_PROGRESS);
167 }
168
169 void MoveInLineOfSight(Unit* who) override
170
171 {
172 if (!CanStartEvent) // boss is invisible, don't attack
173 return;
174 if (!me->GetVictim() && who->IsValidAttackTarget(me))
175 {
176 float attackRadius = me->GetAttackDistance(who);
177 if (me->IsWithinDistInMap(who, attackRadius))
178 AttackStart(who);
179 }
180 }
181
182 void MovementInform(uint32 type, uint32 /*id*/) override
183 {
184 if (type == ROTATE_MOTION_TYPE)
185 me->SetReactState(REACT_AGGRESSIVE);
186 }
187
188 void UpdateAI(uint32 diff) override
189 {
190 if (!CanStartEvent) // boss is invisible, don't attack
191 {
192 if (CheckCanStart())
193 {
194 if (Submerged)
195 {
196 me->SetVisible(true);
197 Submerged = false;
198 WaitTimer2 = 500;
199 }
200
201 if (!Submerged && WaitTimer2 <= diff) // wait 500ms before emerge anim
202 {
203 me->RemoveAllAuras();
204 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
205 DoCast(me, SPELL_EMERGE, false);
206 WaitTimer2 = 60000; // never reached
207 WaitTimer = 3000;
208 }
209 else
210 WaitTimer2 -= diff;
211
212 if (WaitTimer <= diff) // wait 3secs for emerge anim, then attack
213 {
214 WaitTimer = 3000;
215 CanStartEvent = true; // fresh fished from pool
216 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
217 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
218 }
219 else
220 WaitTimer -= diff;
221 }
222 return;
223 }
224
225 if (me->getThreatManager().getThreatList().empty()) // check if should evade
226 {
227 if (me->IsInCombat())
228 EnterEvadeMode();
229 return;
230 }
231 if (!Submerged)
232 {
233 if (PhaseTimer <= diff)
234 {
235 me->InterruptNonMeleeSpells(false);
236 DoCast(me, SPELL_SUBMERGE);
237 PhaseTimer = 60000; // 60secs submerged
238 Submerged = true;
239 } else PhaseTimer-=diff;
240
241 if (SpoutTimer <= diff)
242 {
243 me->TextEmote(EMOTE_SPOUT, nullptr, true);
244 me->SetReactState(REACT_PASSIVE);
245 me->GetMotionMaster()->MoveRotate(20000, urand(0, 1) ? ROTATE_DIRECTION_LEFT : ROTATE_DIRECTION_RIGHT);
246 SpoutTimer = 45000;
247 WhirlTimer = 20000; // whirl directly after spout
248 RotTimer = 20000;
249 return;
250 } else SpoutTimer -= diff;
251
252 // Whirl directly after a Spout and at random times
253 if (WhirlTimer <= diff)
254 {
255 WhirlTimer = 18000;
256 DoCast(me, SPELL_WHIRL);
257 } else WhirlTimer -= diff;
258
259 if (CheckTimer <= diff)//check if there are players in melee range
260 {
261 InRange = false;
262 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
263 if (!PlayerList.isEmpty())
264 {
265 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
266 {
267 if (me->IsWithinMeleeRange(i->GetSource()))
268 InRange = true;
269 }
270 }
271 CheckTimer = 2000;
272 } else CheckTimer -= diff;
273
274 if (RotTimer)
275 {
276 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
277 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
278 {
279 if (i->GetSource() && i->GetSource()->IsAlive() && me->HasInArc(diff/20000.f*float(M_PI)*2.f, i->GetSource()) && me->IsWithinDist(i->GetSource(), SPOUT_DIST) && !i->GetSource()->IsInWater())
280 DoCast(i->GetSource(), SPELL_SPOUT, true); // only knock back players in arc, in 100yards, not in water
281 }
282
283 if (SpoutAnimTimer <= diff)
284 {
285 DoCast(me, SPELL_SPOUT_ANIM, true);
286 SpoutAnimTimer = 1000;
287 } else SpoutAnimTimer -= diff;
288
289 if (RotTimer <= diff)
290 {
291 RotTimer = 0;
292 } else RotTimer -= diff;
293 return;
294 }
295
296 if (GeyserTimer <= diff)
297 {
298 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1);
299 if (!target && me->GetVictim())
300 target = me->GetVictim();
301 if (target)
302 DoCast(target, SPELL_GEYSER, true);
303 GeyserTimer = rand32() % 5000 + 15000;
304 } else GeyserTimer -= diff;
305
306 if (!InRange) // if on players in melee range cast Waterbolt
307 {
308 if (WaterboltTimer <= diff)
309 {
310 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0);
311 if (!target && me->GetVictim())
312 target = me->GetVictim();
313 if (target)
314 DoCast(target, SPELL_WATERBOLT, true);
315 WaterboltTimer = 3000;
316 } else WaterboltTimer -= diff;
317 }
318
319 if (!UpdateVictim())
320 return;
321
322 DoMeleeAttackIfReady();
323
324 }
325 else // submerged
326 {
327 if (PhaseTimer <= diff)
328 {
329 Submerged = false;
330 me->InterruptNonMeleeSpells(false); // shouldn't be any
331 me->RemoveAllAuras();
332 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
333 me->RemoveFlag(UNIT_NPC_EMOTESTATE, EMOTE_STATE_SUBMERGED);
334 DoCast(me, SPELL_EMERGE, true);
335 Spawned = false;
336 SpoutTimer = 3000; // directly cast Spout after emerging!
337 PhaseTimer = 120000;
338 return;
339 } else PhaseTimer-=diff;
340
341 if (me->getThreatManager().getThreatList().empty()) // check if should evade
342 {
343 EnterEvadeMode();
344 return;
345 }
346
347 if (!me->IsInCombat())
348 DoZoneInCombat();
349
350 if (!Spawned)
351 {
352 me->SetUInt32Value(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
353 // spawn adds
354 for (uint8 i = 0; i < 9; ++i)
355 if (Creature* summoned = me->SummonCreature(i < 6 ? NPC_COILFANG_AMBUSHER : NPC_COILFANG_GUARDIAN, AddPos[i][0], AddPos[i][1], AddPos[i][2], 0, TEMPSUMMON_CORPSE_DESPAWN, 0))
356 Summons.Summon(summoned);
357 Spawned = true;
358 }
359 }
360 }
361 };
362};

374 struct npc_coilfang_ambusherAI : public ScriptedAI
375 {
376 npc_coilfang_ambusherAI(Creature* creature) : ScriptedAI(creature)
377 {
378 Initialize();
379 SetCombatMovement(false);
380 }
381
382 void Initialize()
383 {
384 MultiShotTimer = 10000;
385 ShootBowTimer = 4000;
386 }
387
388 uint32 MultiShotTimer;
389 uint32 ShootBowTimer;
390
391 void Reset() override
392 {
393 Initialize();
394 }
395
396 void MoveInLineOfSight(Unit* who) override
397
398 {
399 if (!who || me->GetVictim())
400 return;
401
402 if (who->isInAccessiblePlaceFor(me) && me->IsValidAttackTarget(who) && me->IsWithinDistInMap(who, 45))
403 AttackStart(who);
404 }
405
406 void UpdateAI(uint32 diff) override
407 {
408 if (MultiShotTimer <= diff)
409 {
410 if (me->GetVictim())
411 DoCastVictim(SPELL_SPREAD_SHOT, true);
412
413 MultiShotTimer = 10000 + rand32() % 10000;
414 ShootBowTimer += 1500; // add global cooldown
415 } else MultiShotTimer -= diff;
416
417 if (ShootBowTimer <= diff)
418 {
419 int bp0 = 1100;
420 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
421 me->CastCustomSpell(target, SPELL_SHOOT, &bp0, NULL, NULL, true);
422 ShootBowTimer = 4000 + rand32() % 5000;
423 MultiShotTimer += 1500; // add global cooldown
424 } else ShootBowTimer -= diff;
425 }
426 };

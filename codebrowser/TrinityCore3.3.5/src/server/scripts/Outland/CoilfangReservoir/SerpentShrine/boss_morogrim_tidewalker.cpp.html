87class boss_morogrim_tidewalker : public CreatureScript

89public:
90 boss_morogrim_tidewalker() : CreatureScript("boss_morogrim_tidewalker") { }

92 CreatureAI* GetAI(Creature* creature) const override

97 struct boss_morogrim_tidewalkerAI : public ScriptedAI
98 {
99 boss_morogrim_tidewalkerAI(Creature* creature) : ScriptedAI(creature)
100 {
101 Initialize();
102 instance = creature->GetInstanceScript();
103 Playercount = 0;
104 counter = 0;
105 }
106
107 void Initialize()
108 {
109 TidalWave_Timer = 10000;
110 WateryGrave_Timer = 30000;
111 Earthquake_Timer = 40000;
112 WateryGlobules_Timer = 0;
113 globulespell[0] = SPELL_SUMMON_WATER_GLOBULE_1;
114 globulespell[1] = SPELL_SUMMON_WATER_GLOBULE_2;
115 globulespell[2] = SPELL_SUMMON_WATER_GLOBULE_3;
116 globulespell[3] = SPELL_SUMMON_WATER_GLOBULE_4;
117
118 Earthquake = false;
119 Phase2 = false;
120 }
121
122 InstanceScript* instance;
123
124 uint32 TidalWave_Timer;
125 uint32 WateryGrave_Timer;
126 uint32 Earthquake_Timer;
127 uint32 WateryGlobules_Timer;
128 uint32 globulespell[4];
129 int8 Playercount;
130 int8 counter;
131
132 bool Earthquake;
133 bool Phase2;
134
135 void Reset() override
136 {
137 Initialize();
138
139 instance->SetData(DATA_MOROGRIMTIDEWALKEREVENT, NOT_STARTED);

141
142 void StartEvent()
143 {
144 Talk(SAY_AGGRO);
145
146 instance->SetData(DATA_MOROGRIMTIDEWALKEREVENT, IN_PROGRESS);
147 }
148
149 void KilledUnit(Unit* /*victim*/) override
150 {
151 Talk(SAY_SLAY);
152 }
153
154 void JustDied(Unit* /*killer*/) override
155 {
156 Talk(SAY_DEATH);
157
158 instance->SetData(DATA_MOROGRIMTIDEWALKEREVENT, DONE);
159 }
160
161 void EnterCombat(Unit* /*who*/) override
162 {
163 Playercount = me->GetMap()->GetPlayers().getSize();
164 StartEvent();
165 }
166
167 void ApplyWateryGrave(Unit* player, uint8 i)
168 {
169 switch (i)
170 {
171 case 0: player->CastSpell(player, SPELL_WATERY_GRAVE_1, true); break;
172 case 1: player->CastSpell(player, SPELL_WATERY_GRAVE_2, true); break;
173 case 2: player->CastSpell(player, SPELL_WATERY_GRAVE_3, true); break;
174 case 3: player->CastSpell(player, SPELL_WATERY_GRAVE_4, true); break;
175 }
176 }
177
178 void UpdateAI(uint32 diff) override
179 {
180 //Return since we have no target
181 if (!UpdateVictim())
182 return;
183
184 //Earthquake_Timer
185 if (Earthquake_Timer <= diff)
186 {
187 if (!Earthquake)
188 {
189 DoCastVictim(SPELL_EARTHQUAKE);
190 Earthquake = true;
191 Earthquake_Timer = 10000;
192 }
193 else
194 {
195 Talk(SAY_SUMMON);
196
197 for (uint8 i = 0; i < 10; ++i)
198 {
199 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
200 if (Creature* Murloc = me->SummonCreature(NPC_TIDEWALKER_LURKER, MurlocCords[i][0], MurlocCords[i][1], MurlocCords[i][2], MurlocCords[i][3], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 10000))
201 Murloc->AI()->AttackStart(target);
202 }
203 Talk(EMOTE_EARTHQUAKE);
204 Earthquake = false;
205 Earthquake_Timer = 40000 + rand32() % 5000;
206 }
207 } else Earthquake_Timer -= diff;
208
209 //TidalWave_Timer
210 if (TidalWave_Timer <= diff)
211 {
212 DoCastVictim(SPELL_TIDAL_WAVE);
213 TidalWave_Timer = 20000;
214 } else TidalWave_Timer -= diff;
215
216 if (!Phase2)
217 {
218 //WateryGrave_Timer
219 if (WateryGrave_Timer <= diff)
220 {
221 //Teleport 4 players under the waterfalls
222 GuidSet targets;
223 GuidSet::const_iterator itr;
224 for (uint8 i = 0; i < 4; ++i)
225 {
226 counter = 0;
227 Unit* target;
228 do
229 {
230 target = SelectTarget(SELECT_TARGET_RANDOM, 1, 50, true); //target players only
231 if (counter < Playercount)
232 break;
233 if (target)
234 itr = targets.find(target->GetGUID());
235 ++counter;
236 } while (itr != targets.end());
237
238 if (target)
239 {
240 targets.insert(target->GetGUID());
241 ApplyWateryGrave(target, i);
242 }
243 }
244
245 Talk(SAY_SUMMON_BUBL);
246
247 Talk(EMOTE_WATERY_GRAVE);
248 WateryGrave_Timer = 30000;
249 } else WateryGrave_Timer -= diff;
250
251 //Start Phase2
252 if (HealthBelowPct(25))
253 Phase2 = true;
254 }
255 else
256 {
257 //WateryGlobules_Timer
258 if (WateryGlobules_Timer <= diff)
259 {
260 GuidSet globules;
261 GuidSet::const_iterator itr;
262 for (uint8 g = 0; g < 4; g++) //one unit can't cast more than one spell per update, so some players have to cast for us XD
263 {
264 counter = 0;
265 Unit* pGlobuleTarget;
266 do
267 {
268 pGlobuleTarget = SelectTarget(SELECT_TARGET_RANDOM, 0, 50, true);
269 if (pGlobuleTarget)
270 itr = globules.find(pGlobuleTarget->GetGUID());
271 if (counter > Playercount)
272 break;
273 ++counter;
274 } while (itr != globules.end());
275 if (pGlobuleTarget)
276 {
277 globules.insert(pGlobuleTarget->GetGUID());
278 pGlobuleTarget->CastSpell(pGlobuleTarget, globulespell[g], true);
279 }
280 }
281 Talk(EMOTE_WATERY_GLOBULES);
282 WateryGlobules_Timer = 25000;
283 } else WateryGlobules_Timer -= diff;
284 }
285
286 DoMeleeAttackIfReady();
287 }
288 };

309 void Initialize()
310 {
311 Check_Timer = 1000;
312 }
313
314 uint32 Check_Timer;
315
316 void Reset() override
317 {
318 Initialize();
319
320 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
321 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
322 me->setFaction(14);
323 }
324
325 void EnterCombat(Unit* /*who*/) override { }
326
327 void MoveInLineOfSight(Unit* who) override
328
329 {
330 if (!who || me->GetVictim())
331 return;
332
333 if (me->CanCreatureAttack(who))
334 {
335 //no attack radius check - it attacks the first target that moves in his los
336 //who->RemoveSpellsCausingAura(SPELL_AURA_MOD_STEALTH);
337 AttackStart(who);
338 }
339 }
340
341 void UpdateAI(uint32 diff) override
342 {
343 //Return since we have no target
344 if (!UpdateVictim())
345 return;
346
347 if (Check_Timer <= diff)
348 {
349 if (me->IsWithinDistInMap(me->GetVictim(), 5))
350 {
351 DoCastVictim(SPELL_GLOBULE_EXPLOSION);
352
353 //despawn
354 me->DespawnOrUnsummon();
355 return;
356 }
357 Check_Timer = 500;
358 } else Check_Timer -= diff;
359
360 //do NOT deal any melee damage to the target.
361 }
362 };

26enum ShamanSpells
27{
28 SPELL_SHAMAN_ANGEREDEARTH = 36213,
29 SPELL_SHAMAN_FIREBLAST = 57984,
30 SPELL_SHAMAN_FIRENOVA = 12470,
31 SPELL_SHAMAN_FIRESHIELD = 13376
32};
33
34enum ShamanEvents
35{
36 // Earth Elemental
37 EVENT_SHAMAN_ANGEREDEARTH = 1,
38 // Fire Elemental
39 EVENT_SHAMAN_FIRENOVA = 1,
40 EVENT_SHAMAN_FIRESHIELD = 2,
41 EVENT_SHAMAN_FIREBLAST = 3
42};
43
44class npc_pet_shaman_earth_elemental : public CreatureScript
45{
46 public:
47 npc_pet_shaman_earth_elemental() : CreatureScript("npc_pet_shaman_earth_elemental") { }
48
49 struct npc_pet_shaman_earth_elementalAI : public ScriptedAI
50 {
51 npc_pet_shaman_earth_elementalAI(Creature* creature) : ScriptedAI(creature) { }
52
53
54 void Reset() override
55 {
56 _events.Reset();
57 _events.ScheduleEvent(EVENT_SHAMAN_ANGEREDEARTH, 0);
58 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, true);
59 }
60
61 void UpdateAI(uint32 diff) override
62 {
63 if (!UpdateVictim())
64 return;
65
66 _events.Update(diff);
67
68 if (_events.ExecuteEvent() == EVENT_SHAMAN_ANGEREDEARTH)
69 {
70 DoCastVictim(SPELL_SHAMAN_ANGEREDEARTH);
71 _events.ScheduleEvent(EVENT_SHAMAN_ANGEREDEARTH, urand(5000, 20000));
72 }
73
74 DoMeleeAttackIfReady();
75 }
76
77 private:
78 EventMap _events;
79 };
80
81 CreatureAI* GetAI(Creature* creature) const override
82 {
83 return new npc_pet_shaman_earth_elementalAI(creature);
84 }
85};

87class npc_pet_shaman_fire_elemental : public CreatureScript
88{
89 public:
90 npc_pet_shaman_fire_elemental() : CreatureScript("npc_pet_shaman_fire_elemental") { }
91
92 struct npc_pet_shaman_fire_elementalAI : public ScriptedAI
93 {
94 npc_pet_shaman_fire_elementalAI(Creature* creature) : ScriptedAI(creature) { }
95
96 void Reset() override
97 {
98 _events.Reset();
99 _events.ScheduleEvent(EVENT_SHAMAN_FIRENOVA, urand(5000, 20000));
100 _events.ScheduleEvent(EVENT_SHAMAN_FIREBLAST, urand(5000, 20000));
101 _events.ScheduleEvent(EVENT_SHAMAN_FIRESHIELD, 0);
102 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FIRE, true);
103 }
104
105 void UpdateAI(uint32 diff) override
106 {
107 if (!UpdateVictim())
108 return;
109
110 if (me->HasUnitState(UNIT_STATE_CASTING))
111 return;
112
113 _events.Update(diff);
114
115 while (uint32 eventId = _events.ExecuteEvent())
116 {
117 switch (eventId)
118 {
119 case EVENT_SHAMAN_FIRENOVA:
120 DoCastVictim(SPELL_SHAMAN_FIRENOVA);
121 _events.ScheduleEvent(EVENT_SHAMAN_FIRENOVA, urand(5000, 20000));
122 break;
123 case EVENT_SHAMAN_FIRESHIELD:
124 DoCastVictim(SPELL_SHAMAN_FIRESHIELD);
125 _events.ScheduleEvent(EVENT_SHAMAN_FIRESHIELD, 2000);
126 break;
127 case EVENT_SHAMAN_FIREBLAST:
128 DoCastVictim(SPELL_SHAMAN_FIREBLAST);
129 _events.ScheduleEvent(EVENT_SHAMAN_FIREBLAST, urand(5000, 20000));
130 break;
131 default:
132 break;
133 }
134 }
135
136 DoMeleeAttackIfReady();
137 }
138
139 private:
140 EventMap _events;
141 };
142
143 CreatureAI* GetAI(Creature* creature) const override
144 {
145 return new npc_pet_shaman_fire_elementalAI(creature);
146 }
147};

34enum EgbertMisc
35{
36 SPELL_EGBERT = 40669,
37 EVENT_RETURN = 3
38};

40class npc_pet_gen_egbert : public CreatureScript
41{
42public:
43 npc_pet_gen_egbert() : CreatureScript("npc_pet_gen_egbert") {}
44
45 struct npc_pet_gen_egbertAI : public NullCreatureAI
46 {
47 npc_pet_gen_egbertAI(Creature* creature) : NullCreatureAI(creature)
48 {
49 if (Unit* owner = me->GetCharmerOrOwner())
50 if (owner->GetMap()->GetEntry()->addon > 1)
51 me->SetCanFly(true);
52 }
53
54 void Reset() override
55 {
56 _events.Reset();
57 if (Unit* owner = me->GetCharmerOrOwner())
58 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, me->GetFollowAngle());
59 }
60
61 void EnterEvadeMode(EvadeReason why) override
62 {
63 if (!_EnterEvadeMode(why))
64 return;
65
66 Reset();
67 }
68
69 void UpdateAI(uint32 diff) override
70 {
71 _events.Update(diff);
72
73 if (Unit* owner = me->GetCharmerOrOwner())
74 {
75 if (!me->IsWithinDist(owner, 40.f))
76 {
77 me->RemoveAura(SPELL_EGBERT);
78 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, me->GetFollowAngle());
79 }
80 }
81
82 if (me->HasAura(SPELL_EGBERT))
83 _events.ScheduleEvent(EVENT_RETURN, urandms(5, 20));
84
85 while (uint32 eventId = _events.ExecuteEvent())
86 {
87 switch (eventId)
88 {
89 case EVENT_RETURN:
90 me->RemoveAura(SPELL_EGBERT);
91 break;
92 default:
93 break;
94 }
95 }
96 }
97 private:
98 EventMap _events;
99 };
100
101 CreatureAI* GetAI(Creature* creature) const
102 {
103 return new npc_pet_gen_egbertAI(creature);
104 }
105};

116class npc_pet_gen_pandaren_monk : public CreatureScript
117{
118public:
119 npc_pet_gen_pandaren_monk() : CreatureScript("npc_pet_gen_pandaren_monk") {}
120
121 struct npc_pet_gen_pandaren_monkAI : public NullCreatureAI
122 {
123 npc_pet_gen_pandaren_monkAI(Creature* creature) : NullCreatureAI(creature) { }
124
125 void Reset() override
126 {
127 _events.Reset();
128 _events.ScheduleEvent(EVENT_FOCUS, 1000);
129 }
130
131 void EnterEvadeMode(EvadeReason why) override
132 {
133 if (!_EnterEvadeMode(why))
134 return;
135
136 Reset();
137 }
138
139 void ReceiveEmote(Player* /*player*/, uint32 emote) override
140 {
141 me->InterruptSpell(CURRENT_CHANNELED_SPELL);
142 me->StopMoving();
143
144 switch (emote)
145 {
146 case TEXT_EMOTE_BOW:
147 _events.ScheduleEvent(EVENT_FOCUS, 1000);
148 break;
149 case TEXT_EMOTE_DRINK:
150 _events.ScheduleEvent(EVENT_DRINK, 1000);
151 break;
152 }
153 }
154
155 void UpdateAI(uint32 diff) override
156 {
157 _events.Update(diff);
158
159 if (Unit* owner = me->GetCharmerOrOwner())
160 if (!me->IsWithinDist(owner, 30.f))
161 me->InterruptSpell(CURRENT_CHANNELED_SPELL);
162
163 while (uint32 eventId = _events.ExecuteEvent())
164 {
165 switch (eventId)
166 {
167 case EVENT_FOCUS:
168 if (Unit* owner = me->GetCharmerOrOwner())
169 me->SetFacingToObject(owner);
170 _events.ScheduleEvent(EVENT_EMOTE, 1000);
171 break;
172 case EVENT_EMOTE:
173 me->HandleEmoteCommand(EMOTE_ONESHOT_BOW);
174 _events.ScheduleEvent(EVENT_FOLLOW, 1000);
175 break;
176 case EVENT_FOLLOW:
177 if (Unit* owner = me->GetCharmerOrOwner())
178 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
179 break;
180 case EVENT_DRINK:
181 me->CastSpell(me, SPELL_PANDAREN_MONK, false);
182 break;
183 default:
184 break;
185 }
186 }
187 }
188 private:
189 EventMap _events;
190 };
191
192 CreatureAI* GetAI(Creature* creature) const
193 {
194 return new npc_pet_gen_pandaren_monkAI(creature);
195 }

206class npc_pet_gen_mojo : public CreatureScript
207{
208 public:
209 npc_pet_gen_mojo() : CreatureScript("npc_pet_gen_mojo") { }
210
211 struct npc_pet_gen_mojoAI : public ScriptedAI
212 {
213 npc_pet_gen_mojoAI(Creature* creature) : ScriptedAI(creature)
214 {
215 }
216
217 void Reset() override
218 {
219 _victimGUID.Clear();
220
221 if (Unit* owner = me->GetOwner())
222 me->GetMotionMaster()->MoveFollow(owner, 0.0f, 0.0f);
223 }
224
225 void EnterCombat(Unit* /*who*/) override { }
226 void UpdateAI(uint32 /*diff*/) override { }
227
228 void ReceiveEmote(Player* player, uint32 emote) override
229 {
230 me->HandleEmoteCommand(emote);
231 Unit* owner = me->GetOwner();
232 if (emote != TEXT_EMOTE_KISS || !owner || owner->GetTypeId() != TYPEID_PLAYER ||
233 owner->ToPlayer()->GetTeam() != player->GetTeam())
234 {
235 return;
236 }
237
238 Talk(SAY_MOJO, player);
239
240 if (_victimGUID)
241 if (Player* victim = ObjectAccessor::GetPlayer(*me, _victimGUID))
242 victim->RemoveAura(SPELL_FEELING_FROGGY);
243
244 _victimGUID = player->GetGUID();
245
246 DoCast(player, SPELL_FEELING_FROGGY, true);
247 DoCast(me, SPELL_SEDUCTION_VISUAL, true);
248 me->GetMotionMaster()->MoveFollow(player, 0.0f, 0.0f);
249 }
250
251 private:
252 ObjectGuid _victimGUID;
253 };
254
255 CreatureAI* GetAI(Creature* creature) const override
256 {
257 return new npc_pet_gen_mojoAI(creature);
258 }
259};

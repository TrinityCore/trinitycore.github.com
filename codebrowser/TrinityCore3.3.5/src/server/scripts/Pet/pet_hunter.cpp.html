26enum HunterSpells
27{
28 SPELL_HUNTER_CRIPPLING_POISON = 30981, // Viper
29 SPELL_HUNTER_DEADLY_POISON_PASSIVE = 34657, // Venomous Snake
30 SPELL_HUNTER_MIND_NUMBING_POISON = 25810 // Viper
31};
32
33enum HunterCreatures
34{
35 NPC_HUNTER_VIPER = 19921
36};
37
38class npc_pet_hunter_snake_trap : public CreatureScript
39{
40 public:
41 npc_pet_hunter_snake_trap() : CreatureScript("npc_pet_hunter_snake_trap") { }
42
43 struct npc_pet_hunter_snake_trapAI : public ScriptedAI
44 {
45 npc_pet_hunter_snake_trapAI(Creature* creature) : ScriptedAI(creature)
46 {
47 Initialize();
48 }
49
50 void Initialize()
51 {
52 _spellTimer = 0;
53 _isViper = false;
54 }
55
56 void EnterCombat(Unit* /*who*/) override { }
57
58 void Reset() override
59 {
60 Initialize();
61
62 CreatureTemplate const* Info = me->GetCreatureTemplate();
63
64 _isViper = Info->Entry == NPC_HUNTER_VIPER ? true : false;
65
66 me->SetMaxHealth(uint32(107 * (me->getLevel() - 40) * 0.025f));
67 // Add delta to make them not all hit the same time
68 uint32 delta = (rand32() % 7) * 100;
69 me->SetAttackTime(BASE_ATTACK, Info->BaseAttackTime + delta);
70 //me->SetStatFloatValue(UNIT_FIELD_RANGED_ATTACK_POWER, float(Info->attackpower));
71
72 // Start attacking attacker of owner on first ai update after spawn - move in line of sight may choose better target
73 if (!me->GetVictim() && me->IsSummon())
74 if (Unit* Owner = me->ToTempSummon()->GetSummoner())
75 if (Owner->getAttackerForHelper())
76 AttackStart(Owner->getAttackerForHelper());
77
78 if (!_isViper)
79 DoCast(me, SPELL_HUNTER_DEADLY_POISON_PASSIVE, true);
80 }
81
82 // Redefined for random target selection:
83 void MoveInLineOfSight(Unit* who) override
84 {
85 if (!me->GetVictim() && me->CanCreatureAttack(who))
86 {
87 if (me->GetDistanceZ(who) > CREATURE_Z_ATTACK_RANGE)
88 return;
89
90 float attackRadius = me->GetAttackDistance(who);
91 if (me->IsWithinDistInMap(who, attackRadius) && me->IsWithinLOSInMap(who))
92 {
93 if (!(rand32() % 5))
94 {
95 me->setAttackTimer(BASE_ATTACK, (rand32() % 10) * 100);
96 _spellTimer = (rand32() % 10) * 100;
97 AttackStart(who);
98 }
99 }
100 }
101 }
102
103 void UpdateAI(uint32 diff) override
104 {
105 if (!UpdateVictim() || !me->GetVictim())
106 return;
107
108 if (me->EnsureVictim()->HasBreakableByDamageCrowdControlAura(me))
109 {
110 me->InterruptNonMeleeSpells(false);
111 return;
112 }
113
114 // Viper
115 if (_isViper)
116 {
117 if (_spellTimer <= diff)
118 {
119 if (urand(0, 2) == 0) // 33% chance to cast
120 DoCastVictim(RAND(SPELL_HUNTER_MIND_NUMBING_POISON, SPELL_HUNTER_CRIPPLING_POISON));
121
122 _spellTimer = 3000;
123 }
124 else
125 _spellTimer -= diff;
126 }
127
128 DoMeleeAttackIfReady();
129 }
130
131 private:
132 bool _isViper;
133 uint32 _spellTimer;
134 };
135
136 CreatureAI* GetAI(Creature* creature) const override
137 {
138 return new npc_pet_hunter_snake_trapAI(creature);
139 }
140};
141
142void AddSC_hunter_pet_scripts()
143{
144 new npc_pet_hunter_snake_trap();
145}

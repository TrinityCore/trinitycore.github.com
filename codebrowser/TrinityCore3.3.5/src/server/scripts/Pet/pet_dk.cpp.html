31enum DeathKnightSpells
32{
33 SPELL_DK_SUMMON_GARGOYLE_1 = 49206,
34 SPELL_DK_SUMMON_GARGOYLE_2 = 50514,
35 SPELL_DK_DISMISS_GARGOYLE = 50515,
36 SPELL_DK_SANCTUARY = 54661
37};
38
39class npc_pet_dk_ebon_gargoyle : public CreatureScript
40{
41 public:
42 npc_pet_dk_ebon_gargoyle() : CreatureScript("npc_pet_dk_ebon_gargoyle") { }
43
44 struct npc_pet_dk_ebon_gargoyleAI : CasterAI
45 {
46 npc_pet_dk_ebon_gargoyleAI(Creature* creature) : CasterAI(creature)
47 {
48 Initialize();
49 }
50
51 void Initialize()
52 {
53 // Not needed to be despawned now
54 _despawnTimer = 0;
55 }
56
57 void InitializeAI() override
58 {
59 Initialize();
60
61 CasterAI::InitializeAI();
62 ObjectGuid ownerGuid = me->GetOwnerGUID();
63 if (!ownerGuid)
64 return;
65
66 // Find victim of Summon Gargoyle spell
67 std::list<Unit*> targets;
68 Trinity::AnyUnfriendlyUnitInObjectRangeCheck u_check(me, me, 30.0f);
69 Trinity::UnitListSearcher<Trinity::AnyUnfriendlyUnitInObjectRangeCheck> searcher(me, targets, u_check);
70 me->VisitNearbyObject(30.0f, searcher);
71 for (std::list<Unit*>::const_iterator iter = targets.begin(); iter != targets.end(); ++iter)
72 if ((*iter)->HasAura(SPELL_DK_SUMMON_GARGOYLE_1, ownerGuid))
73 {
74 me->Attack((*iter), false);
75 break;
76 }
77 }
78
79 void JustDied(Unit* /*killer*/) override
80 {
81 // Stop Feeding Gargoyle when it dies
82 if (Unit* owner = me->GetOwner())
83 owner->RemoveAurasDueToSpell(SPELL_DK_SUMMON_GARGOYLE_2);
84 }
85
86 // Fly away when dismissed
87 void SpellHit(Unit* source, SpellInfo const* spell) override
88 {
89 if (spell->Id != SPELL_DK_DISMISS_GARGOYLE || !me->IsAlive())
90 return;
91
92 Unit* owner = me->GetOwner();
93 if (!owner || owner != source)
94 return;
95
96 // Stop Fighting
97 me->ApplyModFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE, true);
98
99 // Sanctuary
100 me->CastSpell(me, SPELL_DK_SANCTUARY, true);
101 me->SetReactState(REACT_PASSIVE);
102
103 //! HACK: Creature's can't have MOVEMENTFLAG_FLYING
104 // Fly Away
105 me->SetCanFly(true);
106 me->SetSpeed(MOVE_FLIGHT, 0.75f, true);
107 me->SetSpeed(MOVE_RUN, 0.75f, true);
108 float x = me->GetPositionX() + 20 * std::cos(me->GetOrientation());
109 float y = me->GetPositionY() + 20 * std::sin(me->GetOrientation());
110 float z = me->GetPositionZ() + 40;
111 me->GetMotionMaster()->Clear(false);
112 me->GetMotionMaster()->MovePoint(0, x, y, z);
113
114 // Despawn as soon as possible
115 _despawnTimer = 4 * IN_MILLISECONDS;
116 }
117
118 void UpdateAI(uint32 diff) override
119 {
120 if (_despawnTimer > 0)
121 {
122 if (_despawnTimer > diff)
123 _despawnTimer -= diff;
124 else
125 me->DespawnOrUnsummon();
126 return;
127 }
128
129 CasterAI::UpdateAI(diff);
130 }
131
132 private:
133 uint32 _despawnTimer;
134 };
135
136 CreatureAI* GetAI(Creature* creature) const override
137 {
138 return new npc_pet_dk_ebon_gargoyleAI(creature);
139 }
140};

142void AddSC_deathknight_pet_scripts()
143{
144 new npc_pet_dk_ebon_gargoyle();
145}

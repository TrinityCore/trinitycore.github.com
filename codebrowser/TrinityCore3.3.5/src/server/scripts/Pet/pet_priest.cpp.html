23#include "ScriptMgr.h"
24#include "ScriptedCreature.h"

28enum PriestSpells
29{
30 SPELL_PRIEST_GLYPH_OF_SHADOWFIEND = 58228,
31 SPELL_PRIEST_GLYPH_OF_SHADOWFIEND_MANA = 58227,
32 SPELL_PRIEST_LIGHTWELL_CHARGES = 59907
33};
34
35class npc_pet_pri_lightwell : public CreatureScript
36{
37 public:
38 npc_pet_pri_lightwell() : CreatureScript("npc_pet_pri_lightwell") { }
39
40 struct npc_pet_pri_lightwellAI : public PassiveAI
41 {
42 npc_pet_pri_lightwellAI(Creature* creature) : PassiveAI(creature)
43 {
44 DoCast(me, SPELL_PRIEST_LIGHTWELL_CHARGES, false);
45 }
46
47 void EnterEvadeMode(EvadeReason /*why*/) override
48 {
49 if (!me->IsAlive())
50 return;
51
52 me->DeleteThreatList();
53 me->CombatStop(true);
54 me->ResetPlayerDamageReq();
55 }
56 };
57
58 CreatureAI* GetAI(Creature* creature) const override
59 {
60 return new npc_pet_pri_lightwellAI(creature);
61 }
62};
63
64class npc_pet_pri_shadowfiend : public CreatureScript
65{
66 public:
67 npc_pet_pri_shadowfiend() : CreatureScript("npc_pet_pri_shadowfiend") { }
68
69 struct npc_pet_pri_shadowfiendAI : public PetAI
70 {
71 npc_pet_pri_shadowfiendAI(Creature* creature) : PetAI(creature) { }
72
73 void JustDied(Unit* /*killer*/) override
74 {
75 if (me->IsSummon())
76 if (Unit* owner = me->ToTempSummon()->GetSummoner())
77 if (owner->HasAura(SPELL_PRIEST_GLYPH_OF_SHADOWFIEND))
78 owner->CastSpell(owner, SPELL_PRIEST_GLYPH_OF_SHADOWFIEND_MANA, true);
79 }
80 };
81
82 CreatureAI* GetAI(Creature* creature) const override
83 {
84 return new npc_pet_pri_shadowfiendAI(creature);
85 }
86};
87
88void AddSC_priest_pet_scripts()
89{
90 new npc_pet_pri_lightwell();
91 new npc_pet_pri_shadowfiend();
92}

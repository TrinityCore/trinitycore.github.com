57 void Init()
58 {
59 Unit* owner = me->GetCharmerOrOwner();
60
61 std::list<Unit*> targets;
62 Trinity::AnyUnfriendlyUnitInObjectRangeCheck u_check(me, me, 30.0f);
63 Trinity::UnitListSearcher<Trinity::AnyUnfriendlyUnitInObjectRangeCheck> searcher(me, targets, u_check);
64 me->VisitNearbyObject(40.0f, searcher);
65
66 Unit* highestThreatUnit = nullptr;
67 float highestThreat = 0.0f;
68 Unit* nearestPlayer = nullptr;
69 for (std::list<Unit*>::const_iterator iter = targets.begin(); iter != targets.end(); ++iter)
70 {
71 // Consider only units without CC
72 if (!(*iter)->HasBreakableByDamageCrowdControlAura((*iter)))
73 {
74 // Take first found unit
75 if (!highestThreatUnit && (*iter)->GetTypeId() != TYPEID_PLAYER)
76 {
77 highestThreatUnit = (*iter);
78 continue;
79 }
80 if (!nearestPlayer && ((*iter)->GetTypeId() == TYPEID_PLAYER))
81 {
82 nearestPlayer = (*iter);
83 continue;
84 }
85 // else compare best fit unit with current unit
86 ThreatContainer::StorageType triggers = (*iter)->getThreatManager().getThreatList();
87 for (ThreatContainer::StorageType::const_iterator trig_citr = triggers.begin(); trig_citr != triggers.end(); ++trig_citr)
88 {
89 // Try to find threat referenced to owner
90 if ((*trig_citr)->getTarget() == owner)
91 {
92 // Check if best fit hostile unit hs lower threat than this current unit
93 if (highestThreat < (*trig_citr)->getThreat())
94 {
95 // If so, update best fit unit
96 highestThreat = (*trig_citr)->getThreat();
97 highestThreatUnit = (*iter);
98 break;
99 }
100 }
101 }
102 // In case no unit with threat was found so far, always check for nearest unit (only for players)
103 if ((*iter)->GetTypeId() == TYPEID_PLAYER)
104 {
105 // If this player is closer than the previous one, update it
106 if (me->GetDistance((*iter)->GetPosition()) < me->GetDistance(nearestPlayer->GetPosition()))
107 nearestPlayer = (*iter);
108 }
109 }
110 }
111 // Prioritize units with threat referenced to owner
112 if (highestThreat > 0.0f && highestThreatUnit)
113 me->Attack(highestThreatUnit, false);
114 // If there is no such target, try to attack nearest hostile unit if such exists
115 else if (nearestPlayer)
116 me->Attack(nearestPlayer, false);
117 }

119 bool IsInThreatList(Unit* target)
120 {
121 Unit* owner = me->GetCharmerOrOwner();
122
123 std::list<Unit*> targets;
124 Trinity::AnyUnfriendlyUnitInObjectRangeCheck u_check(me, me, 30.0f);
125 Trinity::UnitListSearcher<Trinity::AnyUnfriendlyUnitInObjectRangeCheck> searcher(me, targets, u_check);
126 me->VisitNearbyObject(40.0f, searcher);
127
128 for (std::list<Unit*>::const_iterator iter = targets.begin(); iter != targets.end(); ++iter)
129 {
130 if ((*iter) == target)
131 {
132 // Consider only units without CC
133 if (!(*iter)->HasBreakableByDamageCrowdControlAura((*iter)))
134 {
135 ThreatContainer::StorageType triggers = (*iter)->getThreatManager().getThreatList();
136 for (ThreatContainer::StorageType::const_iterator trig_citr = triggers.begin(); trig_citr != triggers.end(); ++trig_citr)
137 {
138 // Try to find threat referenced to owner
139 if ((*trig_citr)->getTarget() == owner)
140 return true;
141 }
142 }
143 }
144 }
145 return false;
146 }

148 void InitializeAI() override
149 {
150 CasterAI::InitializeAI();
151 Unit* owner = me->GetOwner();
152 if (!owner)
153 return;
154
155 // here mirror image casts on summoner spell (not present in client dbc) 49866
156 // here should be auras (not present in client dbc): 35657, 35658, 35659, 35660 selfcast by mirror images (stats related?)
157 // Clone Me!
158 owner->CastSpell(me, SPELL_MAGE_CLONE_ME, false);
159 }
160
161 void EnterCombat(Unit* who) override
162 {
163 if (me->GetVictim() && !me->GetVictim()->HasBreakableByDamageCrowdControlAura(me))
164 {
165 me->CastSpell(who, SPELL_MAGE_FIRE_BLAST, false);
166 events.ScheduleEvent(SPELL_MAGE_FROST_BOLT, TIMER_MIRROR_IMAGE_INIT);
167 events.ScheduleEvent(SPELL_MAGE_FIRE_BLAST, TIMER_MIRROR_IMAGE_FIRE_BLAST);
168 }
169 else
170 EnterEvadeMode(EVADE_REASON_OTHER);
171 }
172
173 void Reset() override
174 {
175 events.Reset();
176 }

178 void UpdateAI(uint32 diff) override
179 {
180 Unit* owner = me->GetCharmerOrOwner();
181 if (!owner)
182 return;
183
184 Unit* target = owner->getAttackerForHelper();
185
186 events.Update(diff);
187
188 // prevent CC interrupts by images
189 if (me->GetVictim() && me->EnsureVictim()->HasBreakableByDamageCrowdControlAura(me))
190 {
191 me->InterruptNonMeleeSpells(false);
192 return;
193 }
194
195 if (me->HasUnitState(UNIT_STATE_CASTING))
196 return;
197
198 // assign target if image doesnt have any or the target is not actual
199 if (!target || me->GetVictim() != target)
200 {
201 Unit* ownerTarget = nullptr;
202 if (Player* owner = me->GetCharmerOrOwner()->ToPlayer())
203 ownerTarget = owner->GetSelectedUnit();
204
205 // recognize which victim will be choosen
206 if (ownerTarget && ownerTarget->GetTypeId() == TYPEID_PLAYER)
207 {
208 if (!ownerTarget->HasBreakableByDamageCrowdControlAura(ownerTarget))
209 me->Attack(ownerTarget, false);
210 }
211 else if (ownerTarget && (ownerTarget->GetTypeId() != TYPEID_PLAYER) && IsInThreatList(ownerTarget))
212 {
213 if (!ownerTarget->HasBreakableByDamageCrowdControlAura(ownerTarget))
214 me->Attack(ownerTarget, false);
215 }
216 else
217 Init();
218 }
219
220 if (uint32 spellId = events.ExecuteEvent())
221 {
222 if (spellId == SPELL_MAGE_FROST_BOLT)
223 {
224 events.ScheduleEvent(SPELL_MAGE_FROST_BOLT, TIMER_MIRROR_IMAGE_FROST_BOLT);
225 DoCastVictim(spellId);
226 }
227 else if (spellId == SPELL_MAGE_FIRE_BLAST)
228 {
229 DoCastVictim(spellId);
230 events.ScheduleEvent(SPELL_MAGE_FIRE_BLAST, TIMER_MIRROR_IMAGE_FIRE_BLAST);
231 }
232 }
233 }

235 // Do not reload Creature templates on evade mode enter - prevent visual lost
236 void EnterEvadeMode(EvadeReason /*why*/) override
237 {
238 if (me->IsInEvadeMode() || !me->IsAlive())
239 return;
240
241 Unit* owner = me->GetCharmerOrOwner();
242
243 me->CombatStop(true);
244 if (owner && !me->HasUnitState(UNIT_STATE_FOLLOW))
245 {
246 me->GetMotionMaster()->Clear(false);
247 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, me->GetFollowAngle(), MOTION_SLOT_ACTIVE);
248 }
249 Init();
250 }
251 };

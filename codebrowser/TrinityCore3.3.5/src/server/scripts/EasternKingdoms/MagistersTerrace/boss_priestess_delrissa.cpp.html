166 Talk(SAY_AGGRO);

183 void InitializeLackeys()
184 {
185 //can be called if Creature are dead, so avoid
186 if (!me->IsAlive())
187 return;
188
189 uint8 j = 0;
190
191 //it's empty, so first time
192 if (LackeyEntryList.empty())
193 {
194 //pre-allocate size for speed
195 LackeyEntryList.resize((sizeof(m_auiAddEntries) / sizeof(uint32)));
196
197 //fill vector array with entries from Creature array
198 for (uint8 i = 0; i < LackeyEntryList.size(); ++i)
199 LackeyEntryList[i] = m_auiAddEntries[i];
200
201 //remove random entries
202 while (LackeyEntryList.size() > MAX_ACTIVE_LACKEY)
203 LackeyEntryList.erase(LackeyEntryList.begin() + rand32() % LackeyEntryList.size());
204
205 //summon all the remaining in vector
206 for (std::vector<uint32>::const_iterator itr = LackeyEntryList.begin(); itr != LackeyEntryList.end(); ++itr)
207 {
208 if (Creature* pAdd = me->SummonCreature((*itr), LackeyLocations[j][0], LackeyLocations[j][1], fZLocation, fOrientation, TEMPSUMMON_CORPSE_DESPAWN, 0))
209 m_auiLackeyGUID[j] = pAdd->GetGUID();
210
211 ++j;
212 }
213 }
214 else
215 {
216 for (std::vector<uint32>::const_iterator itr = LackeyEntryList.begin(); itr != LackeyEntryList.end(); ++itr)
217 {
218 Unit* pAdd = ObjectAccessor::GetUnit(*me, m_auiLackeyGUID[j]);
219
220 //object already removed, not exist
221 if (!pAdd)
222 {
223 pAdd = me->SummonCreature((*itr), LackeyLocations[j][0], LackeyLocations[j][1], fZLocation, fOrientation, TEMPSUMMON_CORPSE_DESPAWN, 0);
224 if (pAdd)
225 m_auiLackeyGUID[j] = pAdd->GetGUID();
226 }
227 ++j;
228 }
229 }
230 }

244 {

246
247 if (instance->GetData(DATA_DELRISSA_DEATH_COUNT) == MAX_ACTIVE_LACKEY)
248 instance->SetBossState(DATA_DELRISSA, DONE);
249 else
250 {

256 void UpdateAI(uint32 diff) override
257 {
258 if (!UpdateVictim())
259 return;
260
261 if (ResetTimer <= diff)
262 {
263 float x, y, z, o;
264 me->GetHomePosition(x, y, z, o);
265 if (me->GetPositionZ() >= z+10)
266 {
267 EnterEvadeMode();
268 return;
269 }
270 ResetTimer = 5000;
271 } else ResetTimer -= diff;
272
273 if (HealTimer <= diff)
274 {
275 uint32 health = me->GetHealth();
276 Unit* target = me;
277 for (uint8 i = 0; i < MAX_ACTIVE_LACKEY; ++i)
278 {
279 if (Unit* pAdd = ObjectAccessor::GetUnit(*me, m_auiLackeyGUID[i]))
280 {
281 if (pAdd->IsAlive() && pAdd->GetHealth() < health)
282 target = pAdd;
283 }
284 }
285
286 DoCast(target, SPELL_FLASH_HEAL);
287 HealTimer = 15000;
288 } else HealTimer -= diff;
289
290 if (RenewTimer <= diff)
291 {
292 Unit* target = me;
293
294 if (urand(0, 1))
295 if (Unit* pAdd = ObjectAccessor::GetUnit(*me, m_auiLackeyGUID[rand32() % MAX_ACTIVE_LACKEY]))
296 if (pAdd->IsAlive())
297 target = pAdd;
298
299 DoCast(target, SPELL_RENEW_NORMAL);
300 RenewTimer = 5000;
301 } else RenewTimer -= diff;
302
303 if (ShieldTimer <= diff)
304 {
305 Unit* target = me;
306
307 if (urand(0, 1))
308 if (Unit* pAdd = ObjectAccessor::GetUnit(*me, m_auiLackeyGUID[rand32() % MAX_ACTIVE_LACKEY]))
309 if (pAdd->IsAlive() && !pAdd->HasAura(SPELL_SHIELD))
310 target = pAdd;
311
312 DoCast(target, SPELL_SHIELD);
313 ShieldTimer = 7500;
314 } else ShieldTimer -= diff;
315
316 if (DispelTimer <= diff)
317 {
318 Unit* target = NULL;
319
320 if (urand(0, 1))
321 target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true);
322 else
323 {
324 if (urand(0, 1))
325 target = me;
326 else
327 if (Unit* pAdd = ObjectAccessor::GetUnit(*me, m_auiLackeyGUID[rand32() % MAX_ACTIVE_LACKEY]))
328 if (pAdd->IsAlive())
329 target = pAdd;
330 }
331
332 if (target)
333 DoCast(target, SPELL_DISPEL_MAGIC);
334
335 DispelTimer = 12000;
336 } else DispelTimer -= diff;
337
338 if (SWPainTimer <= diff)
339 {
340 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
341 DoCast(target, SPELL_SW_PAIN_NORMAL);
342
343 SWPainTimer = 10000;
344 } else SWPainTimer -= diff;
345
346 DoMeleeAttackIfReady();
347 }
348 };

356//all 8 possible lackey use this common
357struct boss_priestess_lackey_commonAI : public ScriptedAI
358{
359 boss_priestess_lackey_commonAI(Creature* creature) : ScriptedAI(creature)
360 {
361 Initialize();
362 instance = creature->GetInstanceScript();
363 AcquireGUIDs();
364 }
365
366 void Initialize()
367 {
368 UsedPotion = false;
369
370 // These guys does not follow normal threat system rules
371 // For later development, some alternative threat system should be made
372 // We do not know what this system is based upon, but one theory is class (healers=high threat, dps=medium, etc)
373 // We reset their threat frequently as an alternative until such a system exist
374 ResetThreatTimer = urand(5000, 20000);
375 }
376
377 InstanceScript* instance;
378
379 ObjectGuid m_auiLackeyGUIDs[MAX_ACTIVE_LACKEY];
380 uint32 ResetThreatTimer;
381
382 bool UsedPotion;
383
384 void Reset() override
385 {
386 Initialize();
387
388 // in case she is not alive and Reset was for some reason called, respawn her (most likely party wipe after killing her)
389 if (Creature* pDelrissa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DELRISSA)))
390 {
391 if (!pDelrissa->IsAlive())
392 pDelrissa->Respawn();
393 }
394 }
395
396 void EnterCombat(Unit* who) override
397 {
398 if (!who)
399 return;
400
401 for (uint8 i = 0; i < MAX_ACTIVE_LACKEY; ++i)
402 {
403 if (Unit* pAdd = ObjectAccessor::GetUnit(*me, m_auiLackeyGUIDs[i]))
404 {
405 if (!pAdd->GetVictim() && pAdd != me)
406 {
407 who->SetInCombatWith(pAdd);
408 pAdd->AddThreat(who, 0.0f);
409 }
410 }
411 }
412
413 if (Creature* pDelrissa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DELRISSA)))
414 {
415 if (pDelrissa->IsAlive() && !pDelrissa->GetVictim())
416 {
417 who->SetInCombatWith(pDelrissa);
418 pDelrissa->AddThreat(who, 0.0f);
419 }
420 }
421 }
422
423 void JustDied(Unit* /*killer*/) override
424 {
425 Creature* pDelrissa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DELRISSA));
426 uint32 uiLackeyDeathCount = instance->GetData(DATA_DELRISSA_DEATH_COUNT);
427
428 if (!pDelrissa)
429 return;
430
431 //should delrissa really yell if dead?
432 pDelrissa->AI()->Talk(LackeyDeath[uiLackeyDeathCount].id);
433
434 instance->SetData(DATA_DELRISSA_DEATH_COUNT, SPECIAL);
435
436 //increase local var, since we now may have four dead
437 ++uiLackeyDeathCount;
438
439 if (uiLackeyDeathCount == MAX_ACTIVE_LACKEY)
440 {
441 //time to make her lootable and complete event if she died before lackeys
442 if (!pDelrissa->IsAlive())
443 {
444 if (!pDelrissa->HasFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE))
445 pDelrissa->SetFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
446
447 instance->SetBossState(DATA_DELRISSA, DONE);
448 }
449 }
450 }
451
452 void KilledUnit(Unit* victim) override
453 {
454 if (Creature* Delrissa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DELRISSA)))
455 Delrissa->AI()->KilledUnit(victim);
456 }
457
458 void AcquireGUIDs()
459 {
460 if (Creature* Delrissa = (ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DELRISSA))))
461 {
462 for (uint8 i = 0; i < MAX_ACTIVE_LACKEY; ++i)
463 m_auiLackeyGUIDs[i] = ENSURE_AI(boss_priestess_delrissa::boss_priestess_delrissaAI, Delrissa->AI())->m_auiLackeyGUID[i];
464 }
465 }
466
467 void UpdateAI(uint32 diff) override
468 {
469 if (!UsedPotion && HealthBelowPct(25))
470 {
471 DoCast(me, SPELL_HEALING_POTION);
472 UsedPotion = true;
473 }
474
475 if (ResetThreatTimer <= diff)
476 {
477 DoResetThreat();
478 ResetThreatTimer = urand(5000, 20000);
479 } else ResetThreatTimer -= diff;
480 }
481};

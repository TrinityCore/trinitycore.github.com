66class npc_kalecgos : public CreatureScript
67{
68public:
69 npc_kalecgos() : CreatureScript("npc_kalecgos") { }
70
71 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
72 {
73 player->PlayerTalkClass->ClearMenus();
74 switch (action)
75 {
76 case GOSSIP_ACTION_INFO_DEF:
77 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KAEL_2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
78 player->SEND_GOSSIP_MENU(12500, creature->GetGUID());
79 break;
80 case GOSSIP_ACTION_INFO_DEF+1:
81 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KAEL_3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
82 player->SEND_GOSSIP_MENU(12502, creature->GetGUID());
83 break;
84 case GOSSIP_ACTION_INFO_DEF+2:
85 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KAEL_4, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
86 player->SEND_GOSSIP_MENU(12606, creature->GetGUID());
87 break;
88 case GOSSIP_ACTION_INFO_DEF+3:
89 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KAEL_5, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 4);
90 player->SEND_GOSSIP_MENU(12607, creature->GetGUID());
91 break;
92 case GOSSIP_ACTION_INFO_DEF+4:
93 player->SEND_GOSSIP_MENU(12608, creature->GetGUID());
94 break;
95 }
96
97 return true;
98 }
99
100 bool OnGossipHello(Player* player, Creature* creature) override
101 {
102 if (creature->IsQuestGiver())
103 player->PrepareQuestMenu(creature->GetGUID());
104
105 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_KAEL_1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
106 player->SEND_GOSSIP_MENU(12498, creature->GetGUID());
107
108 return true;
109 }

116 struct npc_kalecgosAI : public ScriptedAI
117 {
118 npc_kalecgosAI(Creature* creature) : ScriptedAI(creature) { }
119
120 void MovementInform(uint32 type, uint32 pointId) override
121 {
122 if (type != WAYPOINT_MOTION_TYPE)
123 return;
124
125 if (pointId == POINT_ID_PREPARE_LANDING)
126 {
127 me->HandleEmoteCommand(EMOTE_ONESHOT_LAND);
128 me->SetDisableGravity(false);
129 me->SetHover(false);
130 events.ScheduleEvent(EVENT_KALECGOS_LANDING, Seconds(2));
131 }
132 }
133
134 void UpdateAI(uint32 diff) override
135 {
136 events.Update(diff);
137
138 switch (events.ExecuteEvent())
139 {
140 case EVENT_KALECGOS_LANDING:
141 DoCastAOE(SPELL_CAMERA_SHAKE);
142 me->SetObjectScale(0.6f);
143 events.ScheduleEvent(EVENT_KALECGOS_TRANSFORM, Seconds(1));
144 break;
145 case EVENT_KALECGOS_TRANSFORM:
146 DoCast(me, SPELL_ORB_KILL_CREDIT, true);
147 DoCast(me, SPELL_TRANSFORM_VISUAL, false);
148 DoCast(me, SPELL_KALECGOS_TRANSFORM, false);
149 me->UpdateEntry(NPC_HUMAN_KALECGOS);
150 break;
151 default:
152 break;
153 }
154 }
155
156 private:
157 EventMap events;
158 };
159};

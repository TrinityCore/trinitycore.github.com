41Position const KalecgosSpawnPos = { 164.3747f, -397.1197f, 2.151798f, 1.66219f };
42
43class instance_magisters_terrace : public InstanceMapScript
44{
45 public:
46 instance_magisters_terrace() : InstanceMapScript("instance_magisters_terrace", 585) { }
47
48 struct instance_magisters_terrace_InstanceMapScript : public InstanceScript
49 {
50 instance_magisters_terrace_InstanceMapScript(Map* map) : InstanceScript(map)
51 {
52 SetHeaders(DataHeader);
53 SetBossNumber(EncounterCount);
54 LoadDoorData(doorData);
55
56 DelrissaDeathCount = 0;
57 }
58
59 uint32 GetData(uint32 type) const override
60 {
61 switch (type)
62 {
63 case DATA_DELRISSA_DEATH_COUNT:
64 return DelrissaDeathCount;
65 default:
66 break;
67 }
68 return 0;
69 }
70
71 void SetData(uint32 type, uint32 data) override
72 {
73 switch (type)
74 {
75 case DATA_DELRISSA_DEATH_COUNT:
76 if (data == SPECIAL)
77 ++DelrissaDeathCount;
78 else
79 DelrissaDeathCount = 0;
80 break;
81 case DATA_KAELTHAS_STATUES:
82 HandleGameObject(KaelStatue[0], data != 0);
83 HandleGameObject(KaelStatue[1], data != 0);
84 break;
85 default:
86 break;
87 }
88 }
89
90 void OnCreatureCreate(Creature* creature) override
91 {
92 switch (creature->GetEntry())
93 {
94 case NPC_SELIN:
95 SelinGUID = creature->GetGUID();
96 break;
97 case NPC_DELRISSA:
98 DelrissaGUID = creature->GetGUID();
99 break;
100 case NPC_KALECGOS:
101 case NPC_HUMAN_KALECGOS:
102 KalecgosGUID = creature->GetGUID();
103 break;
104 default:
105 break;
106 }
107 }
108
109 void OnGameObjectCreate(GameObject* go) override
110 {
111 switch (go->GetEntry())
112 {
113 case GO_VEXALLUS_DOOR:
114 case GO_SELIN_DOOR:
115 case GO_SELIN_ENCOUNTER_DOOR:
116 case GO_DELRISSA_DOOR:
117 case GO_KAEL_DOOR:
118 AddDoor(go, true);
119 break;
120 case GO_KAEL_STATUE_1:
121 KaelStatue[0] = go->GetGUID();
122 break;
123 case GO_KAEL_STATUE_2:
124 KaelStatue[1] = go->GetGUID();
125 break;
126 case GO_ESCAPE_ORB:
127 EscapeOrbGUID = go->GetGUID();
128 break;
129 default:
130 break;
131 }
132 }
133
134 void OnGameObjectRemove(GameObject* go) override
135 {
136 switch (go->GetEntry())
137 {
138 case GO_VEXALLUS_DOOR:
139 case GO_SELIN_DOOR:
140 case GO_SELIN_ENCOUNTER_DOOR:
141 case GO_DELRISSA_DOOR:
142 case GO_KAEL_DOOR:
143 AddDoor(go, false);
144 break;
145 default:
146 break;
147 }
148 }
149
150 void ProcessEvent(WorldObject* obj, uint32 eventId) override
151 {
152 if (eventId == EVENT_SPAWN_KALECGOS)
153 if (!ObjectAccessor::GetCreature(*obj, KalecgosGUID) && Events.Empty())
154 Events.ScheduleEvent(EVENT_SPAWN_KALECGOS, Minutes(1));
155 }
156
157 void Update(uint32 diff) override
158 {
159 Events.Update(diff);
160
161 if (Events.ExecuteEvent() == EVENT_SPAWN_KALECGOS)
162 if (Creature* kalecgos = instance->SummonCreature(NPC_KALECGOS, KalecgosSpawnPos))
163 {
164 kalecgos->GetMotionMaster()->MovePath(PATH_KALECGOS_FLIGHT, false);
165 kalecgos->AI()->Talk(SAY_KALECGOS_SPAWN);
166 }
167 }
168
169 bool SetBossState(uint32 type, EncounterState state) override
170 {
171 if (!InstanceScript::SetBossState(type, state))
172 return false;
173
174 switch (type)
175 {
176 case DATA_DELRISSA:
177 if (state == IN_PROGRESS)
178 DelrissaDeathCount = 0;
179 break;
180 default:
181 break;
182 }
183 return true;
184 }
185
186 ObjectGuid GetGuidData(uint32 type) const override
187 {
188 switch (type)
189 {
190 case DATA_SELIN:
191 return SelinGUID;
192 case DATA_DELRISSA:
193 return DelrissaGUID;
194 case DATA_KAEL_STATUE_LEFT:
195 return KaelStatue[0];
196 case DATA_KAEL_STATUE_RIGHT:
197 return KaelStatue[1];
198 case DATA_ESCAPE_ORB:
199 return EscapeOrbGUID;
200 default:
201 break;
202 }
203 return ObjectGuid::Empty;
204 }
205
206 protected:
207 EventMap Events;
208 ObjectGuid SelinGUID;
209 ObjectGuid DelrissaGUID;
210 ObjectGuid KaelStatue[2];
211 ObjectGuid EscapeOrbGUID;
212 ObjectGuid KalecgosGUID;
213 uint32 DelrissaDeathCount;
214 };
215
216 InstanceScript* GetInstanceScript(InstanceMap* map) const override
217 {
218 return new instance_magisters_terrace_InstanceMapScript(map);
219 }
220};
221
222void AddSC_instance_magisters_terrace()
223{
224 new instance_magisters_terrace();
225}

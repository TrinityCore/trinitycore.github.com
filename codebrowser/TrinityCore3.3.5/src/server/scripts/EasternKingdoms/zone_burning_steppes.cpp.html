39enum RaggedJohn
40{
41 QUEST_MOTHERS_MILK = 4866,
42 SPELL_MOTHERS_MILK = 16468,
43 SPELL_WICKED_MILKING = 16472
44};
45
46class npc_ragged_john : public CreatureScript
47{
48public:
49 npc_ragged_john() : CreatureScript("npc_ragged_john") { }
50
51 struct npc_ragged_johnAI : public ScriptedAI
52 {
53 npc_ragged_johnAI(Creature* creature) : ScriptedAI(creature) { }
54
55 void Reset() override { }
56
57 void MoveInLineOfSight(Unit* who) override
58 {
59 if (who->HasAura(SPELL_MOTHERS_MILK))
60 {
61 if (who->GetTypeId() == TYPEID_PLAYER && me->IsWithinDistInMap(who, 15) && who->isInAccessiblePlaceFor(me))
62 {
63 DoCast(who, SPELL_WICKED_MILKING);
64 if (Player* player = who->ToPlayer())
65 player->AreaExploredOrEventHappens(QUEST_MOTHERS_MILK);
66 }
67 }
68
69 ScriptedAI::MoveInLineOfSight(who);
70 }
71
72 void EnterCombat(Unit* /*who*/) override { }
73 };
74
75 bool OnGossipHello(Player* player, Creature* creature) override
76 {
77 if (creature->IsQuestGiver())
78 {
79 player->PrepareQuestMenu(creature->GetGUID());
80 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
81 }
82 return true;
83 }
84
85 CreatureAI* GetAI(Creature* creature) const override
86 {
87 return new npc_ragged_johnAI(creature);
88 }
89};
90
91void AddSC_burning_steppes()
92{
93 new npc_ragged_john();
94}

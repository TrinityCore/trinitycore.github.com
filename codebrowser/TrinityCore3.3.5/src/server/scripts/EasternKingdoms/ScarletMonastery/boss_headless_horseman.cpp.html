347 speaker->CastSpell(speaker, SPELL_HEAD_SPEAKS, false);

349 else laugh -= diff;
350 }
351 else
352 {
353 if (die)
354 {
355 if (wait <= diff)
356 {
357 die = false;
358 if (Unit* body = ObjectAccessor::GetUnit(*me, bodyGUID))
359 body->KillSelf();
360 me->KillSelf();
361 }
362 else wait -= diff;
363 }
364 }

536 Player* SelectRandomPlayer(float range = 0.0f, bool checkLoS = true)
537 {
538 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
539 if (PlayerList.isEmpty())
540 return NULL;
541
542 std::list<Player*> temp;
543 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
544 if ((me->IsWithinLOSInMap(i->GetSource()) || !checkLoS) && me->GetVictim() != i->GetSource() &&
545 me->IsWithinDistInMap(i->GetSource(), range) && i->GetSource()->IsAlive())
546 temp.push_back(i->GetSource());
547
548 if (!temp.empty())
549 {
550 std::list<Player*>::const_iterator j = temp.begin();
551 advance(j, rand32() % temp.size());
552 return (*j);
553 }
554 return NULL;
555 }

686 }

688 break;

696 burned = true;

714 SaySound(SAY_SPROUTING_PUMPKINS);

736 else cleave -= diff;
737 }
738 }
739 else
740 {

745 {
746 if (Phase > 1)
747 --Phase;

749 Phase = 1;

751 if (Head && Head->IsAlive())

758 }
759 else regen -= diff;
760
761 if (whirlwind <= diff)
762 {

764 if (urand(0, 1))
765 {

769 }
770 else

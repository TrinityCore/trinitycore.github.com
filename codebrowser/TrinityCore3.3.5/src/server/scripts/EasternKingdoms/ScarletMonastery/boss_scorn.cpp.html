18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "scarlet_monastery.h"
21
22enum Spells
23{
24 SPELL_LICHSLAP = 28873,
25 SPELL_FROSTBOLT_VOLLEY = 8398,
26 SPELL_MINDFLAY = 17313,
27 SPELL_FROSTNOVA = 15531
28};
29
30enum Events
31{
32 EVENT_LICH_SLAP = 1,
33 EVENT_FROSTBOLT_VOLLEY,
34 EVENT_MIND_FLAY,
35 EVENT_FROST_NOVA
36};
37
38class boss_scorn : public CreatureScript
39{
40 public:
41 boss_scorn() : CreatureScript("boss_scorn") { }
42
43 struct boss_scornAI : public BossAI
44 {
45 boss_scornAI(Creature* creature) : BossAI(creature, DATA_SCORN) { }
46
47 void Reset() override
48 {
49 _Reset();
50 }
51
52 void EnterCombat(Unit* /*who*/) override
53 {
54 _EnterCombat();
55 events.ScheduleEvent(EVENT_LICH_SLAP, 45000);
56 events.ScheduleEvent(EVENT_FROSTBOLT_VOLLEY, 30000);
57 events.ScheduleEvent(EVENT_MIND_FLAY, 30000);
58 events.ScheduleEvent(EVENT_FROST_NOVA, 30000);
59 }
60
61 void JustDied(Unit* /*killer*/) override
62 {
63 _JustDied();
64 }
65
66 void ExecuteEvent(uint32 eventId) override
67 {
68 switch (eventId)
69 {
70 case EVENT_LICH_SLAP:
71 DoCastVictim(SPELL_LICHSLAP);
72 events.ScheduleEvent(EVENT_LICH_SLAP, 45000);
73 break;
74 case EVENT_FROSTBOLT_VOLLEY:
75 DoCastVictim(SPELL_FROSTBOLT_VOLLEY);
76 events.ScheduleEvent(EVENT_FROSTBOLT_VOLLEY, 20000);
77 break;
78 case EVENT_MIND_FLAY:
79 DoCastVictim(SPELL_MINDFLAY);
80 events.ScheduleEvent(EVENT_MIND_FLAY, 20000);
81 break;
82 case EVENT_FROST_NOVA:
83 DoCastVictim(SPELL_FROSTNOVA);
84 events.ScheduleEvent(EVENT_FROST_NOVA, 15000);
85 break;
86 default:
87 break;
88 }
89 }
90 };
91
92 CreatureAI* GetAI(Creature* creature) const override
93 {
94 return GetScarletMonasteryAI<boss_scornAI>(creature);
95 }
96};
97
98void AddSC_boss_scorn()
99{
100 new boss_scorn();
101}

22enum Yells
23{
24 SAY_AGGRO = 0,
25 SAY_SPECIALAE = 1
26};
27
28enum Spells
29{
30 SPELL_SILENCE = 8988,
31 SPELL_ARCANE_EXPLOSION = 9433,
32 SPELL_DETONATION = 9435,
33 SPELL_ARCANE_BUBBLE = 9438,
34 SPELL_POLYMORPH = 13323
35};
36
37enum Events
38{
39 EVENT_SILENCE = 1,
40 EVENT_ARCANE_EXPLOSION = 2,
41 EVENT_ARCANE_BUBBLE = 3,
42 EVENT_POLYMORPH = 4
43};
44
45class boss_arcanist_doan : public CreatureScript
46{
47 public:
48 boss_arcanist_doan() : CreatureScript("boss_arcanist_doan") { }
49
50 struct boss_arcanist_doanAI : public BossAI
51 {
52 boss_arcanist_doanAI(Creature* creature) : BossAI(creature, DATA_ARCANIST_DOAN)
53 {
54 _healthAbove50Pct = true;
55 }
56
57 void Reset() override
58 {
59 _Reset();
60 _healthAbove50Pct = true;
61 }
62
63 void EnterCombat(Unit* /*who*/) override
64 {
65 _EnterCombat();
66 Talk(SAY_AGGRO);
67
68 events.ScheduleEvent(EVENT_SILENCE, 15 * IN_MILLISECONDS);
69 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, 3 * IN_MILLISECONDS);
70 events.ScheduleEvent(EVENT_POLYMORPH, 30 * IN_MILLISECONDS);
71 }
72
73 void JustDied(Unit* /*killer*/) override
74 {
75 _JustDied();
76 }
77
78 void UpdateAI(uint32 diff) override
79 {
80 if (!UpdateVictim())
81 return;
82
83 events.Update(diff);
84
85 if (me->HasUnitState(UNIT_STATE_CASTING))
86 return;
87
88 while (uint32 eventId = events.ExecuteEvent())
89 {
90 switch (eventId)
91 {
92 case EVENT_SILENCE:
93 DoCastVictim(SPELL_SILENCE);
94 events.ScheduleEvent(EVENT_SILENCE, urand(15, 20) * IN_MILLISECONDS);
95 break;
96 case EVENT_ARCANE_EXPLOSION:
97 DoCastVictim(SPELL_ARCANE_EXPLOSION);
98 events.ScheduleEvent(EVENT_SILENCE, 8 * IN_MILLISECONDS);
99 break;
100 case EVENT_POLYMORPH:
101 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 30.0f, true))
102 DoCast(target, SPELL_POLYMORPH);
103 events.ScheduleEvent(EVENT_POLYMORPH, 20 * IN_MILLISECONDS);
104 break;
105 default:
106 break;
107 }
108 }
109
110 if (_healthAbove50Pct && HealthBelowPct(50))
111 {
112 _healthAbove50Pct = false;
113 Talk(SAY_SPECIALAE);
114 DoCast(me, SPELL_ARCANE_BUBBLE);
115 DoCastAOE(SPELL_DETONATION);
116 }
117
118 DoMeleeAttackIfReady();
119 }
120
121 private:
122 bool _healthAbove50Pct;
123 };
124
125 CreatureAI* GetAI(Creature* creature) const override
126 {
127 return GetScarletMonasteryAI<boss_arcanist_doanAI>(creature);
128 }
129};
130
131void AddSC_boss_arcanist_doan()
132{
133 new boss_arcanist_doan();
134}

18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "scarlet_monastery.h"
21
22DoorData const doorData[] =
23{
24 { GO_HIGH_INQUISITORS_DOOR, DATA_MOGRAINE_AND_WHITE_EVENT, DOOR_TYPE_ROOM },
25 { 0, 0, DOOR_TYPE_ROOM } // END
26};
27
28class instance_scarlet_monastery : public InstanceMapScript
29{
30 public:
31 instance_scarlet_monastery() : InstanceMapScript(SMScriptName, 189) { }
32
33 struct instance_scarlet_monastery_InstanceMapScript : public InstanceScript
34 {
35 instance_scarlet_monastery_InstanceMapScript(Map* map) : InstanceScript(map)
36 {
37 SetHeaders(DataHeader);
38 SetBossNumber(EncounterCount);
39 LoadDoorData(doorData);
40
41 HorsemanAdds.clear();
42 }
43
44 void OnGameObjectCreate(GameObject* go) override
45 {
46 switch (go->GetEntry())
47 {
48 case GO_PUMPKIN_SHRINE:
49 PumpkinShrineGUID = go->GetGUID();
50 break;
51 case GO_HIGH_INQUISITORS_DOOR:
52 AddDoor(go, true);
53 break;
54 default:
55 break;
56 }
57 }
58
59 void OnGameObjectRemove(GameObject* go) override
60 {
61 switch (go->GetEntry())
62 {
63 case GO_HIGH_INQUISITORS_DOOR:
64 AddDoor(go, false);
65 break;
66 default:
67 break;
68 }
69 }
70
71 void OnCreatureCreate(Creature* creature) override
72 {
73 switch (creature->GetEntry())
74 {
75 case NPC_HORSEMAN:
76 HorsemanGUID = creature->GetGUID();
77 break;
78 case NPC_HEAD:
79 HeadGUID = creature->GetGUID();
80 break;
81 case NPC_PUMPKIN:
82 HorsemanAdds.insert(creature->GetGUID());
83 break;
84 case NPC_MOGRAINE:
85 MograineGUID = creature->GetGUID();
86 break;
87 case NPC_WHITEMANE:
88 WhitemaneGUID = creature->GetGUID();
89 break;
90 case NPC_VORREL:
91 VorrelGUID = creature->GetGUID();
92 break;
93 default:
94 break;
95 }
96 }
97
98 void SetData(uint32 type, uint32 /*data*/) override
99 {
100 switch (type)
101 {
102 case DATA_PUMPKIN_SHRINE:
103 HandleGameObject(PumpkinShrineGUID, false);
104 break;
105 default:
106 break;
107 }
108 }
109
110 bool SetBossState(uint32 type, EncounterState state) override
111 {
112 if (!InstanceScript::SetBossState(type, state))
113 return false;
114
115 switch (type)
116 {
117 case DATA_HORSEMAN_EVENT:
118 if (state == DONE)
119 {
120 for (ObjectGuid guid : HorsemanAdds)
121 {
122 Creature* add = instance->GetCreature(guid);
123 if (add && add->IsAlive())
124 add->KillSelf();
125 }
126 HorsemanAdds.clear();
127 HandleGameObject(PumpkinShrineGUID, false);
128 }
129 break;
130 default:
131 break;
132 }
133 return true;
134 }
135
136 ObjectGuid GetGuidData(uint32 type) const override
137 {
138 switch (type)
139 {
140 case DATA_MOGRAINE:
141 return MograineGUID;
142 case DATA_WHITEMANE:
143 return WhitemaneGUID;
144 case DATA_VORREL:
145 return VorrelGUID;
146 default:
147 break;
148 }
149 return ObjectGuid::Empty;
150 }
151
152 protected:
153 ObjectGuid PumpkinShrineGUID;
154 ObjectGuid HorsemanGUID;
155 ObjectGuid HeadGUID;
156 ObjectGuid MograineGUID;
157 ObjectGuid WhitemaneGUID;
158 ObjectGuid VorrelGUID;
159
160 GuidSet HorsemanAdds;
161 };
162
163 InstanceScript* GetInstanceScript(InstanceMap* map) const override
164 {
165 return new instance_scarlet_monastery_InstanceMapScript(map);
166 }
167};
168
169void AddSC_instance_scarlet_monastery()
170{
171 new instance_scarlet_monastery();
172}

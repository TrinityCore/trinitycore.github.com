45const Position OptionalSpawn[] =
46{
47 { -10960.981445f, -1940.138428f, 46.178097f, 4.12f }, // Hyakiss the Lurker
48 { -10945.769531f, -2040.153320f, 49.474438f, 0.077f }, // Shadikith the Glider
49 { -10899.903320f, -2085.573730f, 49.474449f, 1.38f } // Rokad the Ravager

52class instance_karazhan : public InstanceMapScript
53{
54public:
55 instance_karazhan() : InstanceMapScript("instance_karazhan", 532) { }
56
57 InstanceScript* GetInstanceScript(InstanceMap* map) const override
58 {
59 return new instance_karazhan_InstanceMapScript(map);
60 }
61
62 struct instance_karazhan_InstanceMapScript : public InstanceScript
63 {
64 instance_karazhan_InstanceMapScript(Map* map) : InstanceScript(map)
65 {
66 SetHeaders(DataHeader);
67 SetBossNumber(EncounterCount);
68
69 // 1 - OZ, 2 - HOOD, 3 - RAJ, this never gets altered.
70 OperaEvent = urand(EVENT_OZ, EVENT_RAJ);
71 OzDeathCount = 0;
72 OptionalBossCount = 0;
73 }
74
75 void OnCreatureCreate(Creature* creature) override
76 {
77 switch (creature->GetEntry())
78 {
79 case NPC_KILREK:
80 KilrekGUID = creature->GetGUID();
81 break;
82 case NPC_TERESTIAN_ILLHOOF:
83 TerestianGUID = creature->GetGUID();
84 break;
85 case NPC_MOROES:
86 MoroesGUID = creature->GetGUID();
87 break;
88 }
89 }
90
91 void OnUnitDeath(Unit* unit) override
92 {
93 Creature* creature = unit->ToCreature();
94 if (!creature)
95 return;
96
97 switch (creature->GetEntry())
98 {
99 case NPC_COLDMIST_WIDOW:
100 case NPC_COLDMIST_STALKER:
101 case NPC_SHADOWBAT:
102 case NPC_VAMPIRIC_SHADOWBAT:
103 case NPC_GREATER_SHADOWBAT:
104 case NPC_PHASE_HOUND:
105 case NPC_DREADBEAST:
106 case NPC_SHADOWBEAST:
107 if (GetBossState(DATA_OPTIONAL_BOSS) == TO_BE_DECIDED)
108 {
109 ++OptionalBossCount;
110 if (OptionalBossCount == OPTIONAL_BOSS_REQUIRED_DEATH_COUNT)
111 {
112 switch (urand(NPC_HYAKISS_THE_LURKER, NPC_ROKAD_THE_RAVAGER))
113 {
114 case NPC_HYAKISS_THE_LURKER:
115 instance->SummonCreature(NPC_HYAKISS_THE_LURKER, OptionalSpawn[0]);
116 break;
117 case NPC_SHADIKITH_THE_GLIDER:
118 instance->SummonCreature(NPC_SHADIKITH_THE_GLIDER, OptionalSpawn[1]);
119 break;
120 case NPC_ROKAD_THE_RAVAGER:
121 instance->SummonCreature(NPC_ROKAD_THE_RAVAGER, OptionalSpawn[2]);
122 break;
123 }
124 }
125 }
126 break;
127 default:
128 break;
129 }
130 }
131
132 void SetData(uint32 type, uint32 data) override
133 {
134 switch (type)
135 {
136 case DATA_OPERA_OZ_DEATHCOUNT:
137 if (data == SPECIAL)
138 ++OzDeathCount;
139 else if (data == IN_PROGRESS)
140 OzDeathCount = 0;
141 break;
142 }
143 }
144
145 bool SetBossState(uint32 type, EncounterState state) override
146 {
147 if (!InstanceScript::SetBossState(type, state))
148 return false;
149
150 switch (type)
151 {
152 case DATA_OPERA_PERFORMANCE:
153 if (state == DONE)
154 {
155 HandleGameObject(StageDoorLeftGUID, true);
156 HandleGameObject(StageDoorRightGUID, true);
157 if (GameObject* sideEntrance = instance->GetGameObject(SideEntranceDoor))
158 sideEntrance->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
159 UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, 16812, NULL);
160 }
161 break;
162 case DATA_CHESS:
163 if (state == DONE)
164 DoRespawnGameObject(DustCoveredChest, DAY);
165 break;
166 default:
167 break;
168 }
169
170 return true;
171 }
172
173 void SetGuidData(uint32 type, ObjectGuid data) override
174 {
175 if (type == DATA_IMAGE_OF_MEDIVH)
176 ImageGUID = data;
177 }
178
179 void OnGameObjectCreate(GameObject* go) override
180 {
181 switch (go->GetEntry())
182 {
183 case GO_STAGE_CURTAIN:
184 CurtainGUID = go->GetGUID();
185 break;
186 case GO_STAGE_DOOR_LEFT:
187 StageDoorLeftGUID = go->GetGUID();
188 if (GetBossState(DATA_OPERA_PERFORMANCE) == DONE)
189 go->SetGoState(GO_STATE_ACTIVE);
190 break;
191 case GO_STAGE_DOOR_RIGHT:
192 StageDoorRightGUID = go->GetGUID();
193 if (GetBossState(DATA_OPERA_PERFORMANCE) == DONE)
194 go->SetGoState(GO_STATE_ACTIVE);
195 break;
196 case GO_PRIVATE_LIBRARY_DOOR:
197 LibraryDoor = go->GetGUID();
198 break;
199 case GO_MASSIVE_DOOR:
200 MassiveDoor = go->GetGUID();
201 break;
202 case GO_GAMESMAN_HALL_DOOR:
203 GamesmansDoor = go->GetGUID();
204 break;
205 case GO_GAMESMAN_HALL_EXIT_DOOR:
206 GamesmansExitDoor = go->GetGUID();
207 break;
208 case GO_NETHERSPACE_DOOR:
209 NetherspaceDoor = go->GetGUID();
210 break;
211 case GO_MASTERS_TERRACE_DOOR:
212 MastersTerraceDoor[0] = go->GetGUID();
213 break;
214 case GO_MASTERS_TERRACE_DOOR2:
215 MastersTerraceDoor[1] = go->GetGUID();
216 break;
217 case GO_SIDE_ENTRANCE_DOOR:
218 SideEntranceDoor = go->GetGUID();
219 if (GetBossState(DATA_OPERA_PERFORMANCE) == DONE)
220 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
221 else
222 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
223 break;
224 case GO_DUST_COVERED_CHEST:
225 DustCoveredChest = go->GetGUID();
226 break;
227 }
228
229 switch (OperaEvent)
230 {
231 /// @todo Set Object visibilities for Opera based on performance
232 case EVENT_OZ:
233 break;
234
235 case EVENT_HOOD:
236 break;
237
238 case EVENT_RAJ:
239 break;
240 }
241 }

243 uint32 GetData(uint32 type) const override
244 {
245 switch (type)
246 {
247 case DATA_OPERA_PERFORMANCE:
248 return OperaEvent;
249 case DATA_OPERA_OZ_DEATHCOUNT:
250 return OzDeathCount;
251 }
252
253 return 0;
254 }
255
256 ObjectGuid GetGuidData(uint32 type) const override
257 {
258 switch (type)
259 {
260 case DATA_KILREK:
261 return KilrekGUID;
262 case DATA_TERESTIAN:
263 return TerestianGUID;
264 case DATA_MOROES:
265 return MoroesGUID;
266 case DATA_GO_STAGEDOORLEFT:
267 return StageDoorLeftGUID;
268 case DATA_GO_STAGEDOORRIGHT:
269 return StageDoorRightGUID;
270 case DATA_GO_CURTAINS:
271 return CurtainGUID;
272 case DATA_GO_LIBRARY_DOOR:
273 return LibraryDoor;
274 case DATA_GO_MASSIVE_DOOR:
275 return MassiveDoor;
276 case DATA_GO_SIDE_ENTRANCE_DOOR:
277 return SideEntranceDoor;
278 case DATA_GO_GAME_DOOR:
279 return GamesmansDoor;
280 case DATA_GO_GAME_EXIT_DOOR:
281 return GamesmansExitDoor;
282 case DATA_GO_NETHER_DOOR:
283 return NetherspaceDoor;
284 case DATA_MASTERS_TERRACE_DOOR_1:
285 return MastersTerraceDoor[0];
286 case DATA_MASTERS_TERRACE_DOOR_2:
287 return MastersTerraceDoor[1];
288 case DATA_IMAGE_OF_MEDIVH:
289 return ImageGUID;
290 }
291
292 return ObjectGuid::Empty;
293 }
294
295 private:
296 uint32 OperaEvent;
297 uint32 OzDeathCount;
298 uint32 OptionalBossCount;
299 ObjectGuid CurtainGUID;
300 ObjectGuid StageDoorLeftGUID;
301 ObjectGuid StageDoorRightGUID;
302 ObjectGuid KilrekGUID;
303 ObjectGuid TerestianGUID;
304 ObjectGuid MoroesGUID;
305 ObjectGuid LibraryDoor; // Door at Shade of Aran
306 ObjectGuid MassiveDoor; // Door at Netherspite
307 ObjectGuid SideEntranceDoor; // Side Entrance
308 ObjectGuid GamesmansDoor; // Door before Chess
309 ObjectGuid GamesmansExitDoor; // Door after Chess
310 ObjectGuid NetherspaceDoor; // Door at Malchezaar
311 ObjectGuid MastersTerraceDoor[2];
312 ObjectGuid ImageGUID;
313 ObjectGuid DustCoveredChest;
314 };
315};

36
37class item_defias_gunpowder : public ItemScript
38{
39public:
40 item_defias_gunpowder() : ItemScript("item_defias_gunpowder") { }
41
42 bool OnUse(Player* player, Item* item, SpellCastTargets const& targets) override
43 {
44 InstanceScript* instance = player->GetInstanceScript();
45
46 if (!instance)
47 {
48 player->GetSession()->SendNotification("Instance script not initialized");
49 return true;
50 }
51
52 if (instance->GetData(EVENT_STATE) != CANNON_NOT_USED)
53 return false;
54
55 if (targets.GetGOTarget() && targets.GetGOTarget()->GetEntry() == GO_DEFIAS_CANNON)
56 instance->SetData(EVENT_STATE, CANNON_GUNPOWDER_USED);
57
58 player->DestroyItemCount(item->GetEntry(), 1, true);
59 return true;
60 }
61};
62
63void AddSC_deadmines()
64{
65 new item_defias_gunpowder();
66}

32enum Sounds

34 SOUND_CANNONFIRE = 1400,

38enum Misc
39{
40 DATA_CANNON_BLAST_TIMER = 3000,
41 DATA_PIRATES_DELAY_TIMER = 1000,
42 DATA_SMITE_ALARM_DELAY_TIMER = 5000
43};

45class instance_deadmines : public InstanceMapScript
46{
47 public:
48 instance_deadmines()
49 : InstanceMapScript("instance_deadmines", 36)
50 {

52
53 struct instance_deadmines_InstanceMapScript : public InstanceScript
54 {
55 instance_deadmines_InstanceMapScript(Map* map) : InstanceScript(map)
56 {
57 SetHeaders(DataHeader);
58
59 State = CANNON_NOT_USED;
60 CannonBlast_Timer = 0;
61 PiratesDelay_Timer = 0;
62 SmiteAlarmDelay_Timer = 0;
63 }
64
65 ObjectGuid FactoryDoorGUID;
66 ObjectGuid IronCladDoorGUID;
67 ObjectGuid DefiasCannonGUID;
68 ObjectGuid DoorLeverGUID;
69 ObjectGuid DefiasPirate1GUID;
70 ObjectGuid DefiasPirate2GUID;
71 ObjectGuid DefiasCompanionGUID;
72 ObjectGuid MrSmiteGUID;
73
74 uint32 State;
75 uint32 CannonBlast_Timer;
76 uint32 PiratesDelay_Timer;
77 uint32 SmiteAlarmDelay_Timer;
78 ObjectGuid uiSmiteChestGUID;
79
80 virtual void Update(uint32 diff) override
81 {
82 if (!IronCladDoorGUID || !DefiasCannonGUID || !DoorLeverGUID)
83 return;
84
85 GameObject* pIronCladDoor = instance->GetGameObject(IronCladDoorGUID);
86 if (!pIronCladDoor)
87 return;
88
89 switch (State)
90 {
91 case CANNON_GUNPOWDER_USED:
92 CannonBlast_Timer = DATA_CANNON_BLAST_TIMER;
93 State = CANNON_BLAST_INITIATED;
94 break;
95 case CANNON_BLAST_INITIATED:
96 PiratesDelay_Timer = DATA_PIRATES_DELAY_TIMER;
97 SmiteAlarmDelay_Timer = DATA_SMITE_ALARM_DELAY_TIMER;
98 if (CannonBlast_Timer <= diff)
99 {
100 SummonCreatures();
101 ShootCannon();
102 BlastOutDoor();
103 LeverStucked();
104 instance->LoadGrid(-22.8f, -797.24f); // Loads Mr. Smite's grid.
105 if (Creature* smite = instance->GetCreature(MrSmiteGUID)) // goes off when door blows up
106 smite->AI()->Talk(SAY_ALARM1);
107 State = PIRATES_ATTACK;
108 } else CannonBlast_Timer -= diff;
109 break;
110 case PIRATES_ATTACK:
111 if (PiratesDelay_Timer <= diff)
112 {
113 MoveCreaturesInside();
114 State = SMITE_ALARMED;
115 } else PiratesDelay_Timer -= diff;
116 break;
117 case SMITE_ALARMED:
118 if (SmiteAlarmDelay_Timer <= diff)
119 {
120 if (Creature* smite = instance->GetCreature(MrSmiteGUID))
121 smite->AI()->Talk(SAY_ALARM2);
122 State = EVENT_DONE;
123 } else SmiteAlarmDelay_Timer -= diff;
124 break;
125 }
126 }
127
128 void SummonCreatures()
129 {
130 if (GameObject* pIronCladDoor = instance->GetGameObject(IronCladDoorGUID))
131 {
132 Creature* DefiasPirate1 = pIronCladDoor->SummonCreature(657, pIronCladDoor->GetPositionX() - 2, pIronCladDoor->GetPositionY()-7, pIronCladDoor->GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 3000);
133 Creature* DefiasPirate2 = pIronCladDoor->SummonCreature(657, pIronCladDoor->GetPositionX() + 3, pIronCladDoor->GetPositionY()-6, pIronCladDoor->GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 3000);
134 Creature* DefiasCompanion = pIronCladDoor->SummonCreature(3450, pIronCladDoor->GetPositionX() + 2, pIronCladDoor->GetPositionY()-6, pIronCladDoor->GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 3000);
135
136 DefiasPirate1GUID = DefiasPirate1->GetGUID();
137 DefiasPirate2GUID = DefiasPirate2->GetGUID();
138 DefiasCompanionGUID = DefiasCompanion->GetGUID();
139 }
140 }
141
142 void MoveCreaturesInside()
143 {
144 if (!DefiasPirate1GUID || !DefiasPirate2GUID || !DefiasCompanionGUID)
145 return;
146
147 Creature* pDefiasPirate1 = instance->GetCreature(DefiasPirate1GUID);
148 Creature* pDefiasPirate2 = instance->GetCreature(DefiasPirate2GUID);
149 Creature* pDefiasCompanion = instance->GetCreature(DefiasCompanionGUID);
150 if (!pDefiasPirate1 || !pDefiasPirate2 || !pDefiasCompanion)
151 return;
152
153 MoveCreatureInside(pDefiasPirate1);
154 MoveCreatureInside(pDefiasPirate2);
155 MoveCreatureInside(pDefiasCompanion);
156 }
157
158 void MoveCreatureInside(Creature* creature)
159 {
160 creature->SetWalk(false);
161 creature->GetMotionMaster()->MovePoint(0, -102.7f, -655.9f, creature->GetPositionZ());
162 }
163
164 void ShootCannon()
165 {
166 if (GameObject* pDefiasCannon = instance->GetGameObject(DefiasCannonGUID))
167 {
168 pDefiasCannon->SetGoState(GO_STATE_ACTIVE);
169 pDefiasCannon->PlayDirectSound(SOUND_CANNONFIRE);
170 }
171 }
172
173 void BlastOutDoor()
174 {
175 if (GameObject* pIronCladDoor = instance->GetGameObject(IronCladDoorGUID))
176 {
177 pIronCladDoor->SetGoState(GO_STATE_ACTIVE_ALTERNATIVE);
178 pIronCladDoor->PlayDirectSound(SOUND_DESTROYDOOR);
179 }
180 }
181
182 void LeverStucked()
183 {
184 if (GameObject* pDoorLever = instance->GetGameObject(DoorLeverGUID))
185 pDoorLever->SetUInt32Value(GAMEOBJECT_FLAGS, 4);
186 }
187
188 void OnCreatureCreate(Creature* creature) override
189 {
190 switch (creature->GetEntry())
191 {
192 case NPC_MR_SMITE:
193 MrSmiteGUID = creature->GetGUID();
194 break;
195 default:
196 break;
197 }
198 }
199
200 void OnGameObjectCreate(GameObject* go) override
201 {
202 switch (go->GetEntry())
203 {
204 case GO_FACTORY_DOOR: FactoryDoorGUID = go->GetGUID(); break;
205 case GO_IRONCLAD_DOOR: IronCladDoorGUID = go->GetGUID(); break;
206 case GO_DEFIAS_CANNON: DefiasCannonGUID = go->GetGUID(); break;
207 case GO_DOOR_LEVER: DoorLeverGUID = go->GetGUID(); break;
208 case GO_MR_SMITE_CHEST: uiSmiteChestGUID = go->GetGUID(); break;
209 }
210 }
211
212 void SetData(uint32 type, uint32 data) override
213 {
214 switch (type)
215 {
216 case EVENT_STATE:
217 if (DefiasCannonGUID && IronCladDoorGUID)
218 State=data;
219 break;
220 case EVENT_RHAHKZOR:
221 if (data == DONE)
222 if (GameObject* go = instance->GetGameObject(FactoryDoorGUID))
223 go->SetGoState(GO_STATE_ACTIVE);
224 break;
225 }
226 }
227
228 uint32 GetData(uint32 type) const override
229 {
230 switch (type)
231 {
232 case EVENT_STATE:
233 return State;
234 }
235
236 return 0;
237 }
238
239 ObjectGuid GetGuidData(uint32 data) const override
240 {
241 switch (data)
242 {
243 case DATA_SMITE_CHEST:
244 return uiSmiteChestGUID;
245 }
246
247 return ObjectGuid::Empty;
248 }
249 };
250
251 InstanceScript* GetInstanceScript(InstanceMap* map) const override
252 {
253 return new instance_deadmines_InstanceMapScript(map);
254 }
255};

59 struct boss_mr_smiteAI : public ScriptedAI

61 boss_mr_smiteAI(Creature* creature) : ScriptedAI(creature)

67 void Initialize()
68 {
69 uiTrashTimer = urand(5000, 9000);
70 uiSlamTimer = 9000;
71 uiNimbleReflexesTimer = urand(15500, 31600);
72
73 uiHealth = 0;
74
75 uiPhase = 0;
76 uiTimer = 0;
77
78 uiIsMoving = false;
79 }

81 InstanceScript* instance;

83 uint32 uiTrashTimer;
84 uint32 uiSlamTimer;
85 uint32 uiNimbleReflexesTimer;

87 uint8 uiHealth;

89 uint32 uiPhase;
90 uint32 uiTimer;

92 bool uiIsMoving;

94 void Reset() override
95 {
96 Initialize();
97
98 SetEquipmentSlots(false, EQUIP_SWORD, EQUIP_UNEQUIP, EQUIP_NO_CHANGE);
99 me->SetStandState(UNIT_STAND_STATE_STAND);
100 me->SetReactState(REACT_AGGRESSIVE);

105 }

107 bool bCheckChances()
108 {
109 uint32 uiChances = urand(0, 99);
110 if (uiChances <= 15)
111 return false;
112 else
113 return true;
114 }
115
116 void UpdateAI(uint32 uiDiff) override
117 {
118 if (!UpdateVictim())
119 return;
120
121 if (!uiIsMoving) // halt abilities in between phases
122 {
123 if (uiTrashTimer <= uiDiff)
124 {
125 if (bCheckChances())
126 DoCast(me, SPELL_TRASH);
127 uiTrashTimer = urand(6000, 15500);
128 }
129 else uiTrashTimer -= uiDiff;
130
131 if (uiSlamTimer <= uiDiff)
132 {
133 if (bCheckChances())
134 DoCastVictim(SPELL_SMITE_SLAM);
135 uiSlamTimer = 11000;
136 }
137 else uiSlamTimer -= uiDiff;
138
139 if (uiNimbleReflexesTimer <= uiDiff)
140 {
141 if (bCheckChances())
142 DoCast(me, SPELL_NIMBLE_REFLEXES);
143 uiNimbleReflexesTimer = urand(27300, 60100);
144 }
145 else uiNimbleReflexesTimer -= uiDiff;
146 }
147
148 if ((uiHealth == 0 && !HealthAbovePct(66)) || (uiHealth == 1 && !HealthAbovePct(33)))
149 {
150 ++uiHealth;
151 DoCastAOE(SPELL_SMITE_STOMP, false);
152 SetCombatMovement(false);
153 me->AttackStop();
154 me->InterruptNonMeleeSpells(false);
155 me->SetReactState(REACT_PASSIVE);
156 uiTimer = 2500;
157 uiPhase = 1;
158
159 switch (uiHealth)
160 {
161 case 1:
162 Talk(SAY_PHASE_1);
163 break;
164 case 2:
165 Talk(SAY_PHASE_2);
166 break;
167 }
168 }
169
170 if (uiPhase)
171 {
172 if (uiTimer <= uiDiff)
173 {
174 switch (uiPhase)
175 {
176 case 1:
177 {
178 if (uiIsMoving)
179 break;
180
181 if (GameObject* go = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_SMITE_CHEST)))
182 {
183 me->GetMotionMaster()->Clear();
184 me->GetMotionMaster()->MovePoint(1, go->GetPositionX() - 1.5f, go->GetPositionY() + 1.4f, go->GetPositionZ());
185 uiIsMoving = true;
186 }
187 break;
188 }
189 case 2:
190 if (uiHealth == 1)
191 SetEquipmentSlots(false, EQUIP_AXE, EQUIP_AXE, EQUIP_NO_CHANGE);
192 else
193 SetEquipmentSlots(false, EQUIP_MACE, EQUIP_UNEQUIP, EQUIP_NO_CHANGE);
194 uiTimer = 500;
195 uiPhase = 3;
196 break;
197 case 3:
198 me->SetStandState(UNIT_STAND_STATE_STAND);
199 uiTimer = 750;
200 uiPhase = 4;
201 break;
202 case 4:
203 me->SetReactState(REACT_AGGRESSIVE);
204 SetCombatMovement(true);
205 me->GetMotionMaster()->MoveChase(me->GetVictim(), me->m_CombatDistance);
206 uiIsMoving = false;
207 uiPhase = 0;
208 break;
209 }
210 } else uiTimer -= uiDiff;
211 }
212
213 DoMeleeAttackIfReady();
214 }
215
216 void MovementInform(uint32 uiType, uint32 /*uiId*/) override
217 {
218 if (uiType != POINT_MOTION_TYPE)
219 return;
220
221 me->SetFacingTo(5.47f);
222 me->SetStandState(UNIT_STAND_STATE_KNEEL);
223
224 uiTimer = 2000;
225 uiPhase = 2;
226 }
227 };
228};

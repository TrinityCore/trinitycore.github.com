39enum ProfessorPhizzlethorpe
40{
41 // Yells
42 SAY_PROGRESS_1 = 0,
43 SAY_PROGRESS_2 = 1,
44 SAY_PROGRESS_3 = 2,
45 EMOTE_PROGRESS_4 = 3,
46 SAY_AGGRO = 4,
47 SAY_PROGRESS_5 = 5,
48 SAY_PROGRESS_6 = 6,
49 SAY_PROGRESS_7 = 7,
50 EMOTE_PROGRESS_8 = 8,
51 SAY_PROGRESS_9 = 9,
52 // Quests
53 QUEST_SUNKEN_TREASURE = 665,
54 // Creatures

56 FACTION_SUNKEN_TREASURE = 113

58
59class npc_professor_phizzlethorpe : public CreatureScript
60{
61 public:
62 npc_professor_phizzlethorpe() : CreatureScript("npc_professor_phizzlethorpe") { }
63
64 struct npc_professor_phizzlethorpeAI : public npc_escortAI
65 {
66 npc_professor_phizzlethorpeAI(Creature* creature) : npc_escortAI(creature) { }
67
68 void WaypointReached(uint32 waypointId) override
69 {
70 Player* player = GetPlayerForEscort();
71 if (!player)
72 return;
73
74 switch (waypointId)
75 {
76 case 4:
77 Talk(SAY_PROGRESS_2, player);
78 break;
79 case 5:
80 Talk(SAY_PROGRESS_3, player);
81 break;
82 case 8:
83 Talk(EMOTE_PROGRESS_4);
84 break;
85 case 9:
86 me->SummonCreature(NPC_VENGEFUL_SURGE, -2052.96f, -2142.49f, 20.15f, 1.0f, TEMPSUMMON_CORPSE_DESPAWN, 0);
87 me->SummonCreature(NPC_VENGEFUL_SURGE, -2052.96f, -2142.49f, 20.15f, 1.0f, TEMPSUMMON_CORPSE_DESPAWN, 0);
88 break;
89 case 10:
90 Talk(SAY_PROGRESS_5, player);
91 break;
92 case 11:
93 Talk(SAY_PROGRESS_6, player);
94 SetRun();
95 break;
96 case 19:
97 Talk(SAY_PROGRESS_7, player);
98 break;
99 case 20:
100 Talk(EMOTE_PROGRESS_8);
101 Talk(SAY_PROGRESS_9, player);
102 player->GroupEventHappens(QUEST_SUNKEN_TREASURE, me);
103 break;
104 }
105 }
106
107 void JustSummoned(Creature* summoned) override
108 {
109 summoned->AI()->AttackStart(me);
110 }
111
112 void EnterCombat(Unit* /*who*/) override
113 {
114 Talk(SAY_AGGRO);
115 }
116
117 void sQuestAccept(Player* player, Quest const* quest) override
118 {
119 if (quest->GetQuestId() == QUEST_SUNKEN_TREASURE)
120 {
121 Talk(SAY_PROGRESS_1, player);
122 npc_escortAI::Start(false, false, player->GetGUID(), quest);
123 me->setFaction(FACTION_SUNKEN_TREASURE);
124 }
125 }
126
127 void UpdateAI(uint32 diff) override
128 {
129 npc_escortAI::UpdateAI(diff);
130 }
131 };
132
133 CreatureAI* GetAI(Creature* creature) const override
134 {
135 return new npc_professor_phizzlethorpeAI(creature);
136 }
137};

61class npc_deathstalker_erland : public CreatureScript
62{
63public:
64 npc_deathstalker_erland() : CreatureScript("npc_deathstalker_erland") { }
65
66 struct npc_deathstalker_erlandAI : public npc_escortAI
67 {
68 npc_deathstalker_erlandAI(Creature* creature) : npc_escortAI(creature) { }
69
70 void WaypointReached(uint32 waypointId) override
71 {
72 Player* player = GetPlayerForEscort();
73 if (!player)
74 return;
75
76 switch (waypointId)
77 {
78 case 1:
79 Talk(SAY_START, player);
80 break;
81 case 10:
82 Talk(SAY_PROGRESS);
83 break;
84 case 13:
85 Talk(SAY_LAST, player);
86 player->GroupEventHappens(QUEST_ESCORTING, me);
87 break;
88 case 15:
89 if (Creature* rane = me->FindNearestCreature(NPC_RANE, 20.0f))
90 rane->AI()->Talk(SAY_RANE);
91 break;
92 case 16:
93 Talk(SAY_RANE_ANSWER);
94 break;
95 case 17:
96 Talk(SAY_MOVE_QUINN);
97 break;
98 case 24:
99 Talk(SAY_QUINN);
100 break;
101 case 25:
102 if (Creature* quinn = me->FindNearestCreature(NPC_QUINN, 20.0f))
103 quinn->AI()->Talk(SAY_QUINN_ANSWER);
104 break;
105 case 26:
106 Talk(SAY_BYE);
107 break;
108 }
109 }
110
111 void Reset() override { }
112
113 void EnterCombat(Unit* who) override
114 {
115 Talk(SAY_AGGRO, who);
116 }
117 };
118
119 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
120 {
121 if (quest->GetQuestId() == QUEST_ESCORTING)
122 {
123 creature->AI()->Talk(SAY_QUESTACCEPT, player);
124
125 if (npc_escortAI* pEscortAI = CAST_AI(npc_deathstalker_erland::npc_deathstalker_erlandAI, creature->AI()))
126 pEscortAI->Start(true, false, player->GetGUID());
127 }
128
129 return true;
130 }
131
132 CreatureAI* GetAI(Creature* creature) const override
133 {
134 return new npc_deathstalker_erlandAI(creature);
135 }
136};

162#define WAIT_SECS 6000

164class pyrewood_ambush : public CreatureScript
165{
166public:
167 pyrewood_ambush() : CreatureScript("pyrewood_ambush") { }
168
169 bool OnQuestAccept(Player* player, Creature* creature, const Quest *quest) override
170 {
171 if (quest->GetQuestId() == QUEST_PYREWOOD_AMBUSH && !ENSURE_AI(pyrewood_ambush::pyrewood_ambushAI, creature->AI())->QuestInProgress)
172 {
173 ENSURE_AI(pyrewood_ambush::pyrewood_ambushAI, creature->AI())->QuestInProgress = true;
174 ENSURE_AI(pyrewood_ambush::pyrewood_ambushAI, creature->AI())->Phase = 0;
175 ENSURE_AI(pyrewood_ambush::pyrewood_ambushAI, creature->AI())->KillCount = 0;
176 ENSURE_AI(pyrewood_ambush::pyrewood_ambushAI, creature->AI())->PlayerGUID = player->GetGUID();
177 }
178
179 return true;
180 }
181
182 CreatureAI* GetAI(Creature* creature) const override
183 {
184 return new pyrewood_ambushAI(creature);
185 }
186
187 struct pyrewood_ambushAI : public ScriptedAI
188 {
189 pyrewood_ambushAI(Creature* creature) : ScriptedAI(creature), Summons(me)
190 {
191 Initialize();
192 WaitTimer = WAIT_SECS;
193 QuestInProgress = false;
194 }
195
196 void Initialize()
197 {
198 Phase = 0;
199 KillCount = 0;
200 PlayerGUID.Clear();
201 }
202
203 uint32 Phase;
204 int8 KillCount;
205 uint32 WaitTimer;
206 ObjectGuid PlayerGUID;
207 SummonList Summons;
208
209 bool QuestInProgress;
210
211 void Reset() override
212 {
213 WaitTimer = WAIT_SECS;
214
215 if (!QuestInProgress) //fix reset values (see UpdateVictim)
216 {
217 Initialize();
218 Summons.DespawnAll();
219 }
220 }
221
222 void EnterCombat(Unit* /*who*/) override { }
223
224 void JustSummoned(Creature* summoned) override
225 {
226 Summons.Summon(summoned);
227 ++KillCount;
228 }
229
230 void SummonedCreatureDespawn(Creature* summoned) override
231 {
232 Summons.Despawn(summoned);
233 --KillCount;
234 }
235
236 void SummonCreatureWithRandomTarget(uint32 creatureId, int position)
237 {
238 if (Creature* summoned = me->SummonCreature(creatureId, PyrewoodSpawnPoints[position][0], PyrewoodSpawnPoints[position][1], PyrewoodSpawnPoints[position][2], PyrewoodSpawnPoints[position][3], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000))
239 {
240 Unit* target = NULL;
241 if (PlayerGUID)
242 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
243 if (player->IsAlive() && RAND(0, 1))
244 target = player;
245
246 if (!target)
247 target = me;
248
249 summoned->setFaction(168);
250 summoned->AddThreat(target, 32.0f);
251 summoned->AI()->AttackStart(target);
252 }
253 }
254
255 void JustDied(Unit* /*killer*/) override
256 {
257 if (PlayerGUID)
258 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
259 if (player->GetQuestStatus(QUEST_PYREWOOD_AMBUSH) == QUEST_STATUS_INCOMPLETE)
260 player->FailQuest(QUEST_PYREWOOD_AMBUSH);
261 }
262
263 void UpdateAI(uint32 diff) override
264 {
265 //TC_LOG_INFO("scripts", "DEBUG: p(%i) k(%i) d(%u) W(%i)", Phase, KillCount, diff, WaitTimer);
266
267 if (!QuestInProgress)
268 return;
269
270 if (KillCount && Phase < 6)
271 {
272 if (!UpdateVictim()) //reset() on target Despawn...
273 return;
274
275 DoMeleeAttackIfReady();
276 return;
277 }
278
279 switch (Phase)
280 {
281 case 0:
282 if (WaitTimer == WAIT_SECS)
283 me->Say(NPCSAY_INIT, LANG_UNIVERSAL); //no blizzlike
284
285 if (WaitTimer <= diff)
286 {
287 WaitTimer -= diff;
288 return;
289 }
290 break;
291 case 1:
292 SummonCreatureWithRandomTarget(2060, 1);
293 break;
294 case 2:
295 SummonCreatureWithRandomTarget(2061, 2);
296 SummonCreatureWithRandomTarget(2062, 0);
297 break;
298 case 3:
299 SummonCreatureWithRandomTarget(2063, 1);
300 SummonCreatureWithRandomTarget(2064, 2);
301 SummonCreatureWithRandomTarget(2065, 0);
302 break;
303 case 4:
304 SummonCreatureWithRandomTarget(2066, 1);
305 SummonCreatureWithRandomTarget(2067, 0);
306 SummonCreatureWithRandomTarget(2068, 2);
307 break;
308 case 5: //end
309 if (PlayerGUID)
310 {
311 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
312 {
313 me->Say(NPCSAY_END, LANG_UNIVERSAL); //not blizzlike
314 player->GroupEventHappens(QUEST_PYREWOOD_AMBUSH, me);
315 }
316 }
317 QuestInProgress = false;
318 Reset();
319 break;
320 }
321 ++Phase; //prepare next phase
322 }
323 };

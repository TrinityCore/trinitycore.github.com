64class boss_baron_rivendare : public CreatureScript
65{
66public:
67 boss_baron_rivendare() : CreatureScript("boss_baron_rivendare") { }
68
69 CreatureAI* GetAI(Creature* creature) const override
70 {
71 return GetInstanceAI<boss_baron_rivendareAI>(creature);
72 }
73
74 struct boss_baron_rivendareAI : public ScriptedAI
75 {
76 boss_baron_rivendareAI(Creature* creature) : ScriptedAI(creature)
77 {
78 Initialize();
79 instance = me->GetInstanceScript();
80 }
81
82 void Initialize()
83 {
84 ShadowBolt_Timer = 5000;
85 Cleave_Timer = 8000;
86 MortalStrike_Timer = 12000;
87 // RaiseDead_Timer = 30000;
88 SummonSkeletons_Timer = 34000;
89 }
90
91 InstanceScript* instance;
92
93 uint32 ShadowBolt_Timer;
94 uint32 Cleave_Timer;
95 uint32 MortalStrike_Timer;
96 // uint32 RaiseDead_Timer;
97 uint32 SummonSkeletons_Timer;
98
99 void Reset() override
100 {
101 Initialize();
102 if (instance->GetData(TYPE_RAMSTEIN) == DONE)
103 instance->SetData(TYPE_BARON, NOT_STARTED);
104 }
105
106 void AttackStart(Unit* who) override
107 {
108 //can't use entercombat(), boss' dmg aura sets near players in combat, before entering the room's door
109 if (instance->GetData(TYPE_BARON) == NOT_STARTED)
110 instance->SetData(TYPE_BARON, IN_PROGRESS);
111 ScriptedAI::AttackStart(who);
112 }
113
114 void JustSummoned(Creature* summoned) override
115 {
116 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
117 summoned->AI()->AttackStart(target);
118 }
119
120 void JustDied(Unit* /*killer*/) override
121 {
122 instance->SetData(TYPE_BARON, DONE);
123 }
124
125 void UpdateAI(uint32 diff) override
126 {
127 if (!UpdateVictim())
128 return;
129
130 //ShadowBolt
131 if (ShadowBolt_Timer <= diff)
132 {
133 if (SelectTarget(SELECT_TARGET_RANDOM, 0))
134 DoCastVictim(SPELL_SHADOWBOLT);
135
136 ShadowBolt_Timer = 10000;
137 } else ShadowBolt_Timer -= diff;
138
139 //Cleave
140 if (Cleave_Timer <= diff)
141 {
142 DoCastVictim(SPELL_CLEAVE);
143 //13 seconds until we should cast this again
144 Cleave_Timer = 7000 + (rand32() % 10000);
145 } else Cleave_Timer -= diff;
146
147 //MortalStrike
148 if (MortalStrike_Timer <= diff)
149 {
150 DoCastVictim(SPELL_MORTALSTRIKE);
151 MortalStrike_Timer = 10000 + (rand32() % 15000);
152 } else MortalStrike_Timer -= diff;
153
154 //RaiseDead
155 // if (RaiseDead_Timer <= diff)
156 // {
157 // DoCast(me, SPELL_RAISEDEAD);
158 // RaiseDead_Timer = 45000;
159 // } else RaiseDead_Timer -= diff;
160
161 //SummonSkeletons
162 if (SummonSkeletons_Timer <= diff)
163 {
164 me->SummonCreature(11197, ADD_POS_1, TEMPSUMMON_TIMED_DESPAWN, 29000);
165 me->SummonCreature(11197, ADD_POS_2, TEMPSUMMON_TIMED_DESPAWN, 29000);
166 me->SummonCreature(11197, ADD_POS_3, TEMPSUMMON_TIMED_DESPAWN, 29000);
167 me->SummonCreature(11197, ADD_POS_4, TEMPSUMMON_TIMED_DESPAWN, 29000);
168 me->SummonCreature(11197, ADD_POS_5, TEMPSUMMON_TIMED_DESPAWN, 29000);
169 me->SummonCreature(11197, ADD_POS_6, TEMPSUMMON_TIMED_DESPAWN, 29000);
170
171 //34 seconds until we should cast this again
172 SummonSkeletons_Timer = 40000;
173 } else SummonSkeletons_Timer -= diff;
174
175 DoMeleeAttackIfReady();
176 }
177 };
178
179};

73{

82 struct boss_dathrohan_balnazzarAI : public ScriptedAI
83 {
84 boss_dathrohan_balnazzarAI(Creature* creature) : ScriptedAI(creature)
85 {
86 Initialize();
87 }
88
89 void Initialize()
90 {
91 m_uiCrusadersHammer_Timer = 8000;
92 m_uiCrusaderStrike_Timer = 12000;
93 m_uiMindBlast_Timer = 6000;
94 m_uiHolyStrike_Timer = 18000;
95 m_uiShadowShock_Timer = 4000;
96 m_uiPsychicScream_Timer = 16000;
97 m_uiDeepSleep_Timer = 20000;
98 m_uiMindControl_Timer = 10000;
99 m_bTransformed = false;
100 }
101
102 uint32 m_uiCrusadersHammer_Timer;
103 uint32 m_uiCrusaderStrike_Timer;
104 uint32 m_uiMindBlast_Timer;
105 uint32 m_uiHolyStrike_Timer;
106 uint32 m_uiShadowShock_Timer;
107 uint32 m_uiPsychicScream_Timer;
108 uint32 m_uiDeepSleep_Timer;
109 uint32 m_uiMindControl_Timer;
110 bool m_bTransformed;
111
112 void Reset() override
113 {
114 Initialize();
115
116 if (me->GetEntry() == NPC_BALNAZZAR)
117 me->UpdateEntry(NPC_DATHROHAN);
118 }
119
120 void JustDied(Unit* /*killer*/) override
121 {
122 static uint32 uiCount = sizeof(m_aSummonPoint)/sizeof(SummonDef);
123
124 for (uint8 i=0; i<uiCount; ++i)
125 me->SummonCreature(NPC_ZOMBIE,
126 m_aSummonPoint[i].m_fX, m_aSummonPoint[i].m_fY, m_aSummonPoint[i].m_fZ, m_aSummonPoint[i].m_fOrient,
127 TEMPSUMMON_TIMED_DESPAWN, HOUR*IN_MILLISECONDS);
128 }
129
130 void EnterCombat(Unit* /*who*/) override
131 {
132 }
133
134 void UpdateAI(uint32 uiDiff) override
135 {
136 if (!UpdateVictim())
137 return;
138
139 //START NOT TRANSFORMED
140 if (!m_bTransformed)
141 {
142 //MindBlast
143 if (m_uiMindBlast_Timer <= uiDiff)
144 {
145 DoCastVictim(SPELL_MINDBLAST);
146 m_uiMindBlast_Timer = urand(15000, 20000);
147 } else m_uiMindBlast_Timer -= uiDiff;
148
149 //CrusadersHammer
150 if (m_uiCrusadersHammer_Timer <= uiDiff)
151 {
152 DoCastVictim(SPELL_CRUSADERSHAMMER);
153 m_uiCrusadersHammer_Timer = 12000;
154 } else m_uiCrusadersHammer_Timer -= uiDiff;
155
156 //CrusaderStrike
157 if (m_uiCrusaderStrike_Timer <= uiDiff)
158 {
159 DoCastVictim(SPELL_CRUSADERSTRIKE);
160 m_uiCrusaderStrike_Timer = 15000;
161 } else m_uiCrusaderStrike_Timer -= uiDiff;
162
163 //HolyStrike
164 if (m_uiHolyStrike_Timer <= uiDiff)
165 {
166 DoCastVictim(SPELL_HOLYSTRIKE);
167 m_uiHolyStrike_Timer = 15000;
168 } else m_uiHolyStrike_Timer -= uiDiff;
169
170 //BalnazzarTransform
171 if (HealthBelowPct(40))
172 {
173 if (me->IsNonMeleeSpellCast(false))
174 me->InterruptNonMeleeSpells(false);
175
176 //restore hp, mana and stun
177 DoCast(me, SPELL_BALNAZZARTRANSFORM);
178 me->UpdateEntry(NPC_BALNAZZAR);
179 m_bTransformed = true;
180 }
181 }
182 else
183 {
184 //MindBlast
185 if (m_uiMindBlast_Timer <= uiDiff)
186 {
187 DoCastVictim(SPELL_MINDBLAST);
188 m_uiMindBlast_Timer = urand(15000, 20000);
189 } else m_uiMindBlast_Timer -= uiDiff;
190
191 //ShadowShock
192 if (m_uiShadowShock_Timer <= uiDiff)
193 {
194 DoCastVictim(SPELL_SHADOWSHOCK);
195 m_uiShadowShock_Timer = 11000;
196 } else m_uiShadowShock_Timer -= uiDiff;
197
198 //PsychicScream
199 if (m_uiPsychicScream_Timer <= uiDiff)
200 {
201 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
202 DoCast(target, SPELL_PSYCHICSCREAM);
203
204 m_uiPsychicScream_Timer = 20000;
205 } else m_uiPsychicScream_Timer -= uiDiff;
206
207 //DeepSleep
208 if (m_uiDeepSleep_Timer <= uiDiff)
209 {
210 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
211 DoCast(target, SPELL_SLEEP);
212
213 m_uiDeepSleep_Timer = 15000;
214 } else m_uiDeepSleep_Timer -= uiDiff;
215
216 //MindControl
217 if (m_uiMindControl_Timer <= uiDiff)
218 {
219 DoCastVictim(SPELL_MINDCONTROL);
220 m_uiMindControl_Timer = 15000;
221 } else m_uiMindControl_Timer -= uiDiff;
222 }
223
224 DoMeleeAttackIfReady();
225 }
226 };
227
228};

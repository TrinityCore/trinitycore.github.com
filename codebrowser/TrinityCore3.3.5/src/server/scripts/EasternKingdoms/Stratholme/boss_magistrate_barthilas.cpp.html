30enum Spells
31{
32 SPELL_DRAININGBLOW = 16793,
33 SPELL_CROWDPUMMEL = 10887,
34 SPELL_MIGHTYBLOW = 14099,
35 SPELL_FURIOUS_ANGER = 16791
36};

38enum Models
39{
40 MODEL_NORMAL = 10433,

44class boss_magistrate_barthilas : public CreatureScript
45{
46public:
47 boss_magistrate_barthilas() : CreatureScript("boss_magistrate_barthilas") { }
48
49 CreatureAI* GetAI(Creature* creature) const override
50 {
51 return new boss_magistrate_barthilasAI(creature);
52 }
53
54 struct boss_magistrate_barthilasAI : public ScriptedAI
55 {
56 boss_magistrate_barthilasAI(Creature* creature) : ScriptedAI(creature)
57 {
58 Initialize();
59 }
60
61 void Initialize()
62 {
63 DrainingBlow_Timer = 20000;
64 CrowdPummel_Timer = 15000;
65 MightyBlow_Timer = 10000;
66 FuriousAnger_Timer = 5000;
67 AngerCount = 0;
68 }
69
70 uint32 DrainingBlow_Timer;
71 uint32 CrowdPummel_Timer;
72 uint32 MightyBlow_Timer;
73 uint32 FuriousAnger_Timer;
74 uint32 AngerCount;
75
76 void Reset() override
77 {
78 Initialize();
79
80 if (me->IsAlive())
81 me->SetDisplayId(MODEL_NORMAL);
82 else
83 me->SetDisplayId(MODEL_HUMAN);
84 }
85
86 void MoveInLineOfSight(Unit* who) override
87
88 {
89 //nothing to see here yet
90
91 ScriptedAI::MoveInLineOfSight(who);
92 }
93
94 void JustDied(Unit* /*killer*/) override
95 {
96 me->SetDisplayId(MODEL_HUMAN);
97 }
98
99 void EnterCombat(Unit* /*who*/) override
100 {
101 }
102
103 void UpdateAI(uint32 diff) override
104 {
105 //Return since we have no target
106 if (!UpdateVictim())
107 return;
108
109 if (FuriousAnger_Timer <= diff)
110 {
111 FuriousAnger_Timer = 4000;
112 if (AngerCount > 25)
113 return;
114
115 ++AngerCount;
116 DoCast(me, SPELL_FURIOUS_ANGER, false);
117 } else FuriousAnger_Timer -= diff;
118
119 //DrainingBlow
120 if (DrainingBlow_Timer <= diff)
121 {
122 DoCastVictim(SPELL_DRAININGBLOW);
123 DrainingBlow_Timer = 15000;
124 } else DrainingBlow_Timer -= diff;
125
126 //CrowdPummel
127 if (CrowdPummel_Timer <= diff)
128 {
129 DoCastVictim(SPELL_CROWDPUMMEL);
130 CrowdPummel_Timer = 15000;
131 } else CrowdPummel_Timer -= diff;
132
133 //MightyBlow
134 if (MightyBlow_Timer <= diff)
135 {
136 DoCastVictim(SPELL_MIGHTYBLOW);
137 MightyBlow_Timer = 20000;
138 } else MightyBlow_Timer -= diff;
139
140 DoMeleeAttackIfReady();
141 }
142 };
143
144};
145
146void AddSC_boss_magistrate_barthilas()
147{
148 new boss_magistrate_barthilas();
149}

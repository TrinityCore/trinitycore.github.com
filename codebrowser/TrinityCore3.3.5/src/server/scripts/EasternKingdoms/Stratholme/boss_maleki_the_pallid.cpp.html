30enum Spells
31{
32 SPELL_FROSTBOLT = 17503,
33 SPELL_DRAINLIFE = 20743,
34 SPELL_DRAIN_MANA = 17243,
35 SPELL_ICETOMB = 16869
36};
37
38enum MalekiEvents
39{
40 EVENT_FROSTBOLT = 1,
41 EVENT_DRAINLIFE = 2,
42 EVENT_DRAIN_MANA = 3,
43 EVENT_ICETOMB = 4
44};
45
46class boss_maleki_the_pallid : public CreatureScript
47{
48public:
49 boss_maleki_the_pallid() : CreatureScript("boss_maleki_the_pallid") { }
50
51 struct boss_maleki_the_pallidAI : public ScriptedAI
52 {
53 boss_maleki_the_pallidAI(Creature* creature) : ScriptedAI(creature)
54 {
55 instance = me->GetInstanceScript();
56 }
57
58 void Reset() override
59 {
60 _events.Reset();
61 }
62
63 void EnterCombat(Unit* /*who*/) override
64 {
65 _events.ScheduleEvent(EVENT_FROSTBOLT, 1 * IN_MILLISECONDS);
66 _events.ScheduleEvent(EVENT_ICETOMB, 16 * IN_MILLISECONDS);
67 _events.ScheduleEvent(EVENT_DRAINLIFE, 31 * IN_MILLISECONDS);
68 }
69
70 void JustDied(Unit* /*killer*/) override
71 {
72 instance->SetData(TYPE_PALLID, IN_PROGRESS);
73 }
74
75 void UpdateAI(uint32 diff) override
76 {
77 //Return since we have no target
78 if (!UpdateVictim())
79 return;
80
81 _events.Update(diff);
82
83 if (me->HasUnitState(UNIT_STATE_CASTING))
84 return;
85
86 while (uint32 eventId = _events.ExecuteEvent())
87 {
88 switch (eventId)
89 {
90 case EVENT_FROSTBOLT:
91 if (rand32() % 90)
92 DoCastVictim(SPELL_FROSTBOLT);
93 _events.ScheduleEvent(EVENT_FROSTBOLT, 3.5 * IN_MILLISECONDS);
94 break;
95 case EVENT_ICETOMB:
96 if (rand32() % 65)
97 DoCastVictim(SPELL_ICETOMB);
98 _events.ScheduleEvent(EVENT_ICETOMB, 28 * IN_MILLISECONDS);
99 break;
100 case EVENT_DRAINLIFE:
101 if (rand32() % 55)
102 DoCastVictim(SPELL_DRAINLIFE);
103 _events.ScheduleEvent(EVENT_DRAINLIFE, 31 * IN_MILLISECONDS);
104 break;
105 default:
106 break;
107 }
108 }
109
110 DoMeleeAttackIfReady();
111 }
112
113 private:
114 EventMap _events;
115 InstanceScript* instance;
116 };
117
118 CreatureAI* GetAI(Creature* creature) const override
119 {
120 return GetInstanceAI<boss_maleki_the_pallidAI>(creature);
121 }
122};
123
124void AddSC_boss_maleki_the_pallid()
125{
126 new boss_maleki_the_pallid();
127}

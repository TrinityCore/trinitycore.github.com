39#define GOSSIP_MP "Open the gate please, i need to get to Searing Gorge"
40
41#define GOSSIP_MP1 "But i need to get there, now open the gate!"
42#define GOSSIP_MP2 "Ok, so what is this other way?"
43#define GOSSIP_MP3 "Doesn't matter, i'm invulnerable."
44#define GOSSIP_MP4 "Yes..."
45#define GOSSIP_MP5 "Ok, i'll try to remember that."
46#define GOSSIP_MP6 "A key? Ok!"
47
48class npc_mountaineer_pebblebitty : public CreatureScript
49{
50public:
51 npc_mountaineer_pebblebitty() : CreatureScript("npc_mountaineer_pebblebitty") { }
52
53 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
54 {
55 player->PlayerTalkClass->ClearMenus();
56 switch (action)
57 {
58 case GOSSIP_ACTION_INFO_DEF+1:
59 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_MP1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
60 player->SEND_GOSSIP_MENU(1833, creature->GetGUID());
61 break;
62 case GOSSIP_ACTION_INFO_DEF+2:
63 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_MP2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
64 player->SEND_GOSSIP_MENU(1834, creature->GetGUID());
65 break;
66 case GOSSIP_ACTION_INFO_DEF+3:
67 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_MP3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 4);
68 player->SEND_GOSSIP_MENU(1835, creature->GetGUID());
69 break;
70 case GOSSIP_ACTION_INFO_DEF+4:
71 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_MP4, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 5);
72 player->SEND_GOSSIP_MENU(1836, creature->GetGUID());
73 break;
74 case GOSSIP_ACTION_INFO_DEF+5:
75 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_MP5, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 6);
76 player->SEND_GOSSIP_MENU(1837, creature->GetGUID());
77 break;
78 case GOSSIP_ACTION_INFO_DEF+6:
79 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_MP6, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 7);
80 player->SEND_GOSSIP_MENU(1838, creature->GetGUID());
81 break;
82 case GOSSIP_ACTION_INFO_DEF+7:
83 player->CLOSE_GOSSIP_MENU();
84 break;
85 }
86 return true;
87 }
88
89 bool OnGossipHello(Player* player, Creature* creature) override
90 {
91 if (creature->IsQuestGiver())
92 player->PrepareQuestMenu(creature->GetGUID());
93
94 if (!player->GetQuestRewardStatus(3181) == 1)
95 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_MP, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
96
97 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
98
99 return true;
100 }
101};
102
103void AddSC_loch_modan()
104{
105 new npc_mountaineer_pebblebitty();
106}

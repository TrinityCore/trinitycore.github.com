18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20
21enum Emotes
22{
23 EMOTE_FRENZY = 0
24};
25
26enum Spells
27{
28 SPELL_FLAMESTRIKE = 18399,
29 SPELL_BLAST_WAVE = 16046,
30 SPELL_FIRE_SHIELD = 19626,
31 SPELL_FRENZY = 8269 // 28371
32};
33
34enum Events
35{
36 EVENT_FIRE_SHIELD = 1,
37 EVENT_BLAST_WAVE,
38 EVENT_FRENZY
39};
40
41class boss_vectus : public CreatureScript
42{
43public:
44 boss_vectus() : CreatureScript("boss_vectus") { }
45
46 struct boss_vectusAI : public ScriptedAI
47 {
48 boss_vectusAI(Creature* creature) : ScriptedAI(creature) { }
49
50 void Reset() override
51 {
52 events.Reset();
53 }
54
55 void EnterCombat(Unit* /*who*/) override
56 {
57 events.ScheduleEvent(EVENT_FIRE_SHIELD, 2000);
58 events.ScheduleEvent(EVENT_BLAST_WAVE, 14000);
59 }
60
61 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
62 {
63 if (me->HealthBelowPctDamaged(25, damage))
64 {
65 DoCast(me, SPELL_FRENZY);
66 Talk(EMOTE_FRENZY);
67 events.ScheduleEvent(EVENT_FRENZY, 24000);
68 }
69 }
70
71 void UpdateAI(uint32 diff) override
72 {
73 if (!UpdateVictim())
74 return;
75
76 events.Update(diff);
77
78 if (me->HasUnitState(UNIT_STATE_CASTING))
79 return;
80
81 while (uint32 eventId = events.ExecuteEvent())
82 {
83 switch (eventId)
84 {
85 case EVENT_FIRE_SHIELD:
86 DoCast(me, SPELL_FIRE_SHIELD);
87 events.ScheduleEvent(EVENT_FIRE_SHIELD, 90000);
88 break;
89 case EVENT_BLAST_WAVE:
90 DoCast(me, SPELL_BLAST_WAVE);
91 events.ScheduleEvent(EVENT_BLAST_WAVE, 12000);
92 break;
93 case EVENT_FRENZY:
94 DoCast(me, SPELL_FRENZY);
95 Talk(EMOTE_FRENZY);
96 events.ScheduleEvent(EVENT_FRENZY, 24000);
97 break;
98 default:
99 break;
100 }
101 }
102
103 DoMeleeAttackIfReady();
104 }
105
106 private:
107 EventMap events;
108 };
109
110 CreatureAI* GetAI(Creature* creature) const override
111 {
112 return new boss_vectusAI(creature);
113 }
114};
115
116void AddSC_boss_vectus()
117{
118 new boss_vectus();
119}

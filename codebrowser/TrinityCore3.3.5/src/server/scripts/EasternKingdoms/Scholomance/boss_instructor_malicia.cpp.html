29enum Spells
30{
31 SPELL_CALLOFGRAVES = 17831,
32 SPELL_CORRUPTION = 11672,
33 SPELL_FLASHHEAL = 10917,
34 SPELL_RENEW = 10929,
35 SPELL_HEALINGTOUCH = 9889
36};
37
38enum Events
39{
40 EVENT_CALLOFGRAVES = 1,
41 EVENT_CORRUPTION = 2,
42 EVENT_FLASHHEAL = 3,
43 EVENT_RENEW = 4,
44 EVENT_HEALINGTOUCH = 5
45};
46
47class boss_instructor_malicia : public CreatureScript
48{
49 public: boss_instructor_malicia() : CreatureScript("boss_instructor_malicia") { }
50
51 struct boss_instructormaliciaAI : public BossAI
52 {
53 boss_instructormaliciaAI(Creature* creature) : BossAI(creature, DATA_INSTRUCTORMALICIA)
54 {
55 Initialize();
56 }
57
58 void Initialize()
59 {
60 FlashCounter = 0;
61 TouchCounter = 0;
62 }
63
64 uint32 FlashCounter;
65 uint32 TouchCounter;
66
67 void Reset() override
68 {
69 _Reset();
70 Initialize();
71 }
72
73 void EnterCombat(Unit* /*who*/) override
74 {
75 _EnterCombat();
76 events.ScheduleEvent(EVENT_CALLOFGRAVES, 4000);
77 events.ScheduleEvent(EVENT_CORRUPTION, 8000);
78 events.ScheduleEvent(EVENT_RENEW, 32000);
79 events.ScheduleEvent(EVENT_FLASHHEAL, 38000);
80 events.ScheduleEvent(EVENT_HEALINGTOUCH, 45000);
81 }
82
83 void UpdateAI(uint32 diff) override
84 {
85 if (!UpdateVictim())
86 return;
87
88 events.Update(diff);
89
90 if (me->HasUnitState(UNIT_STATE_CASTING))
91 return;
92
93 while (uint32 eventId = events.ExecuteEvent())
94 {
95 switch (eventId)
96 {
97 case EVENT_CALLOFGRAVES:
98 DoCastVictim(SPELL_CALLOFGRAVES, true);
99 events.ScheduleEvent(EVENT_CALLOFGRAVES, 65000);
100 break;
101 case EVENT_CORRUPTION:
102 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_CORRUPTION, true);
103 events.ScheduleEvent(EVENT_CORRUPTION, 24000);
104 break;
105 case EVENT_RENEW:
106 DoCast(me, SPELL_RENEW);
107 events.ScheduleEvent(EVENT_RENEW, 10000);
108 break;
109 case EVENT_FLASHHEAL:
110 //5 Flashheals will be cast
111 DoCast(me, SPELL_FLASHHEAL);
112 if (FlashCounter < 2)
113 {
114 events.ScheduleEvent(EVENT_FLASHHEAL, 5000);
115 ++FlashCounter;
116 }
117 else
118 {
119 FlashCounter=0;
120 events.ScheduleEvent(EVENT_FLASHHEAL, 30000);
121 }
122 break;
123 case EVENT_HEALINGTOUCH:
124 //3 Healing Touch will be cast
125 DoCast(me, SPELL_HEALINGTOUCH);
126 if (TouchCounter < 2)
127 {
128 events.ScheduleEvent(EVENT_HEALINGTOUCH, 5500);
129 ++TouchCounter;
130 }
131 else
132 {
133 TouchCounter=0;
134 events.ScheduleEvent(EVENT_HEALINGTOUCH, 30000);
135 }
136 break;
137 default:
138 break;
139 }
140 }
141
142 DoMeleeAttackIfReady();
143 }
144 };
145
146 CreatureAI* GetAI(Creature* creature) const override
147 {
148 return new boss_instructormaliciaAI(creature);
149 }
150
151};

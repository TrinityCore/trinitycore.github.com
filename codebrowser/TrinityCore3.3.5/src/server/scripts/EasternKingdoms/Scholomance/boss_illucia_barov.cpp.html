25#include "ScriptMgr.h"

29enum Spells
30{
31 SPELL_CURSEOFAGONY = 34794,
32 SPELL_SHADOWSHOCK = 34799,
33 SPELL_SILENCE = 34803,
34 SPELL_FEAR = 34803
35};
36
37enum Events
38{
39 EVENT_CURSEOFAGONY = 1,
40 EVENT_SHADOWSHOCK = 2,
41 EVENT_SILENCE = 3,
42 EVENT_FEAR = 4
43};
44
45class boss_illucia_barov : public CreatureScript
46{
47 public: boss_illucia_barov() : CreatureScript("boss_illucia_barov") { }
48
49 struct boss_illuciabarovAI : public BossAI
50 {
51 boss_illuciabarovAI(Creature* creature) : BossAI(creature, DATA_LADYILLUCIABAROV) { }
52
53 void EnterCombat(Unit* /*who*/) override
54 {
55 _EnterCombat();
56 events.ScheduleEvent(EVENT_CURSEOFAGONY, 18000);
57 events.ScheduleEvent(EVENT_SHADOWSHOCK, 9000);
58 events.ScheduleEvent(EVENT_SILENCE, 5000);
59 events.ScheduleEvent(EVENT_FEAR, 30000);
60 }
61
62 void UpdateAI(uint32 diff) override
63 {
64 if (!UpdateVictim())
65 return;
66
67 events.Update(diff);
68
69 if (me->HasUnitState(UNIT_STATE_CASTING))
70 return;
71
72 while (uint32 eventId = events.ExecuteEvent())
73 {
74 switch (eventId)
75 {
76 case EVENT_CURSEOFAGONY:
77 DoCastVictim(SPELL_CURSEOFAGONY, true);
78 events.ScheduleEvent(EVENT_CURSEOFAGONY, 30000);
79 break;
80 case EVENT_SHADOWSHOCK:
81 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_SHADOWSHOCK, true);
82 events.ScheduleEvent(EVENT_SHADOWSHOCK, 12000);
83 break;
84 case EVENT_SILENCE:
85 DoCastVictim(SPELL_SILENCE, true);
86 events.ScheduleEvent(EVENT_SILENCE, 14000);
87 break;
88 case EVENT_FEAR:
89 DoCastVictim(SPELL_FEAR, true);
90 events.ScheduleEvent(EVENT_FEAR, 30000);
91 break;
92 default:
93 break;
94 }
95 }
96
97 DoMeleeAttackIfReady();
98 }
99 };
100
101 CreatureAI* GetAI(Creature* creature) const override
102 {
103 return new boss_illuciabarovAI(creature);
104 }
105};
106
107void AddSC_boss_illuciabarov()
108{
109 new boss_illucia_barov();
110}

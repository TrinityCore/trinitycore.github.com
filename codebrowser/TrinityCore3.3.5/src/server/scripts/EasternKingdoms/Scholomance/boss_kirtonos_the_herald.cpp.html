73class boss_kirtonos_the_herald : public CreatureScript
74{
75 public: boss_kirtonos_the_herald() : CreatureScript("boss_kirtonos_the_herald") { }
76
77 struct boss_kirtonos_the_heraldAI : public BossAI
78 {
79 boss_kirtonos_the_heraldAI(Creature* creature) : BossAI(creature, DATA_KIRTONOS) { }
80
81 void Reset() override
82 {
83 _Reset();
84 }
85
86 void EnterCombat(Unit* /*who*/) override
87 {
88 events.ScheduleEvent(EVENT_SWOOP, urand(8000, 8000));
89 events.ScheduleEvent(EVENT_WING_FLAP, urand(15000, 15000));
90 events.ScheduleEvent(EVENT_PIERCE_ARMOR, urand(18000, 18000));
91 events.ScheduleEvent(EVENT_DISARM, urand(22000, 22000));
92 events.ScheduleEvent(EVENT_SHADOW_BOLT, urand(42000, 42000));
93 events.ScheduleEvent(EVENT_CURSE_OF_TONGUES, urand(53000, 53000));
94 events.ScheduleEvent(EVENT_DOMINATE_MIND, urand(34000, 48000));
95 events.ScheduleEvent(EVENT_KIRTONOS_TRANSFORM, urand(20000, 20000));
96 _EnterCombat();
97 }
98
99 void JustDied(Unit* /*killer*/) override
100 {
101 if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GATE_KIRTONOS)))
102 gate->SetGoState(GO_STATE_ACTIVE);
103 if (GameObject* brazier = me->GetMap()->GetGameObject(instance->GetGuidData(GO_BRAZIER_OF_THE_HERALD)))
104 {
105 brazier->ResetDoorOrButton();
106 brazier->SetGoState(GO_STATE_READY);
107 }
108 _JustDied();
109 }
110
111 void EnterEvadeMode(EvadeReason /*why*/) override
112 {
113 if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GATE_KIRTONOS)))
114 gate->SetGoState(GO_STATE_ACTIVE);
115 if (GameObject* brazier = me->GetMap()->GetGameObject(instance->GetGuidData(GO_BRAZIER_OF_THE_HERALD)))
116 {
117 brazier->ResetDoorOrButton();
118 brazier->SetGoState(GO_STATE_READY);
119 }
120 me->DespawnOrUnsummon(5000);
121 }
122
123 void IsSummonedBy(Unit* /*summoner*/) override
124 {
125 events.ScheduleEvent(INTRO_1, 500);
126 me->SetDisableGravity(true);
127 me->SetReactState(REACT_PASSIVE);
128 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
129 Talk(EMOTE_SUMMONED);
130 }
131
132 void JustSummoned(Creature* summon) override
133 {
134 BossAI::JustSummoned(summon);
135 }
136
137 void MovementInform(uint32 type, uint32 id) override
138 {
139 if (type == WAYPOINT_MOTION_TYPE && id == POINT_KIRTONOS_LAND)
140 {
141 events.ScheduleEvent(INTRO_2, 1500);
142 }
143 }
144
145 void UpdateAI(uint32 diff) override
146 {
147 events.Update(diff);
148
149 if (!UpdateVictim())
150 {
151 while (uint32 eventId = events.ExecuteEvent())
152 {
153 switch (eventId)
154 {
155 case INTRO_1:
156 me->GetMotionMaster()->MovePath(KIRTONOS_PATH, false);
157 break;
158 case INTRO_2:
159 me->GetMotionMaster()->MovePoint(0, PosMove[0]);
160 events.ScheduleEvent(INTRO_3, 1000);
161 break;
162 case INTRO_3:
163 if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GATE_KIRTONOS)))
164 gate->SetGoState(GO_STATE_READY);
165 me->SetFacingTo(0.01745329f);
166 events.ScheduleEvent(INTRO_4, 3000);
167 break;
168 case INTRO_4:
169 if (GameObject* brazier = me->GetMap()->GetGameObject(instance->GetGuidData(GO_BRAZIER_OF_THE_HERALD)))
170 brazier->SetGoState(GO_STATE_READY);
171 me->SetWalk(true);
172 me->SetDisableGravity(false);
173 DoCast(me, SPELL_KIRTONOS_TRANSFORM);
174 me->SetCanFly(false);
175 events.ScheduleEvent(INTRO_5, 1000);
176 break;
177 case INTRO_5:
178 me->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
179 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_KIRTONOS_STAFF));
180 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
181 me->SetReactState(REACT_AGGRESSIVE);
182 events.ScheduleEvent(INTRO_6, 5000);
183 break;
184 case INTRO_6:
185 me->GetMotionMaster()->MovePoint(0, PosMove[1]);
186 break;
187 default:
188 break;
189 }
190 }
191
192 return;
193 }
194
195 if (me->HasUnitState(UNIT_STATE_CASTING))
196 return;
197
198 while (uint32 eventId = events.ExecuteEvent())
199 {
200 switch (eventId)
201 {
202 case EVENT_SWOOP:
203 DoCast(me, SPELL_SWOOP);
204 events.ScheduleEvent(EVENT_SWOOP, 15000);
205 break;
206 case EVENT_WING_FLAP:
207 DoCast(me, SPELL_WING_FLAP);
208 events.ScheduleEvent(EVENT_WING_FLAP, 13000);
209 break;
210 case EVENT_PIERCE_ARMOR:
211 DoCastVictim(SPELL_PIERCE_ARMOR, true);
212 events.ScheduleEvent(EVENT_PIERCE_ARMOR, 12000);
213 break;
214 case EVENT_DISARM:
215 DoCastVictim(SPELL_DISARM, true);
216 events.ScheduleEvent(EVENT_DISARM, 11000);
217 break;
218 case EVENT_SHADOW_BOLT:
219 DoCastVictim(SPELL_SHADOW_BOLT, true);
220 events.ScheduleEvent(EVENT_SHADOW_BOLT, 42000);
221 break;
222 case EVENT_CURSE_OF_TONGUES:
223 DoCastVictim(SPELL_CURSE_OF_TONGUES, true);
224 events.ScheduleEvent(EVENT_CURSE_OF_TONGUES, 35000);
225 break;
226 case EVENT_DOMINATE_MIND:
227 DoCastVictim(SPELL_DOMINATE_MIND, true);
228 events.ScheduleEvent(EVENT_DOMINATE_MIND, urand(44000, 48000));
229 break;
230 case EVENT_KIRTONOS_TRANSFORM:
231 if (me->HasAura(SPELL_KIRTONOS_TRANSFORM))
232 {
233 me->RemoveAura(SPELL_KIRTONOS_TRANSFORM);
234 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(0));
235 me->SetCanFly(false);
236 }
237 else
238 {
239 DoCast(me, SPELL_KIRTONOS_TRANSFORM);
240 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_KIRTONOS_STAFF));
241 me->SetCanFly(true);
242 }
243 events.ScheduleEvent(EVENT_KIRTONOS_TRANSFORM, urand(16000, 18000));
244 break;
245 default:
246 break;
247 }
248 }
249
250 DoMeleeAttackIfReady();
251 }
252 };
253
254 CreatureAI* GetAI(Creature* creature) const override
255 {
256 return GetInstanceAI<boss_kirtonos_the_heraldAI>(creature);
257 }

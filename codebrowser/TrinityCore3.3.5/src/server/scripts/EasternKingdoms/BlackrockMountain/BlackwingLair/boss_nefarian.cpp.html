179 void Reset() override
180 {
181 Initialize();
182
183 if (me->GetMapId() == 469)
184 {
185 if (!me->FindNearestCreature(NPC_NEFARIAN, 1000.0f, true))
186 _Reset();
187
188 me->SetVisible(true);
189 me->SetPhaseMask(1, true);
190 me->SetUInt32Value(UNIT_NPC_FLAGS, 1);
191 me->setFaction(35);
192 me->SetStandState(UNIT_STAND_STATE_SIT_HIGH_CHAIR);
193 me->RemoveAura(SPELL_NEFARIANS_BARRIER);
194 }
195 }
196
197 void JustReachedHome() override
198 {
199 Reset();
200 }
201
202 void BeginEvent(Player* target)
203 {
204 _EnterCombat();
205
206 Talk(SAY_GAMESBEGIN_2);
207
208 me->setFaction(103);
209 me->SetUInt32Value(UNIT_NPC_FLAGS, 0);
210 DoCast(me, SPELL_NEFARIANS_BARRIER);
211 me->SetStandState(UNIT_STAND_STATE_STAND);
212 AttackStart(target);
213 events.ScheduleEvent(EVENT_SHADOW_BOLT, urand(3000, 10000));
214 events.ScheduleEvent(EVENT_FEAR, urand(10000, 20000));
215 //events.ScheduleEvent(EVENT_MIND_CONTROL, urand(30000, 35000));
216 events.ScheduleEvent(EVENT_SPAWN_ADD, 10000);
217 }
218
219 void SummonedCreatureDies(Creature* summon, Unit* /*killer*/) override

223 summon->UpdateEntry(NPC_BONE_CONSTRUCT);
224 summon->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
225 summon->SetReactState(REACT_PASSIVE);
226 summon->SetStandState(UNIT_STAND_STATE_DEAD);

244 void UpdateAI(uint32 diff) override
245 {
246 if (!UpdateVictim())
247 {
248 events.Update(diff);
249
250 while (uint32 eventId = events.ExecuteEvent())
251 {
252 switch (eventId)
253 {
254 case EVENT_PATH_2:
255 me->GetMotionMaster()->MovePath(NEFARIUS_PATH_2, false);
256 events.ScheduleEvent(EVENT_CHAOS_1, 7000);
257 break;
258 case EVENT_CHAOS_1:
259 if (Creature* gyth = me->FindNearestCreature(NPC_GYTH, 75.0f, true))
260 {
261 me->SetFacingToObject(gyth);
262 Talk(SAY_CHAOS_SPELL);
263 }
264 events.ScheduleEvent(EVENT_CHAOS_2, 2000);
265 break;
266 case EVENT_CHAOS_2:
267 DoCast(SPELL_CHROMATIC_CHAOS);
268 me->SetFacingTo(1.570796f);
269 break;
270 case EVENT_SUCCESS_1:
271 if (Unit* player = me->SelectNearestPlayer(60.0f))
272 {
273 me->SetInFront(player);
274 Talk(SAY_SUCCESS);
275 if (GameObject* portcullis1 = me->FindNearestGameObject(GO_PORTCULLIS_ACTIVE, 65.0f))
276 portcullis1->SetGoState(GO_STATE_ACTIVE);
277 if (GameObject* portcullis2 = me->FindNearestGameObject(GO_PORTCULLIS_TOBOSSROOMS, 80.0f))
278 portcullis2->SetGoState(GO_STATE_ACTIVE);
279 }
280 events.ScheduleEvent(EVENT_SUCCESS_2, 4000);
281 break;
282 case EVENT_SUCCESS_2:
283 DoCast(me, SPELL_VAELASTRASZZ_SPAWN);
284 me->DespawnOrUnsummon(1000);
285 break;
286 case EVENT_PATH_3:
287 me->GetMotionMaster()->MovePath(NEFARIUS_PATH_3, false);
288 break;
289 default:
290 break;
291 }
292 }
293 return;
294 }
295
296 // Only do this if we haven't spawned nefarian yet
297 if (UpdateVictim() && SpawnedAdds <= 42)
298 {
299 events.Update(diff);
300
301 if (me->HasUnitState(UNIT_STATE_CASTING))
302 return;
303
304 while (uint32 eventId = events.ExecuteEvent())
305 {
306 switch (eventId)
307 {
308 case EVENT_SHADOW_BOLT:
309 switch (urand(0, 1))
310 {
311 case 0:
312 DoCastVictim(SPELL_SHADOWBOLT_VOLLEY);
313 break;
314 case 1:
315 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true))
316 DoCast(target, SPELL_SHADOWBOLT);
317 break;
318 }
319 DoResetThreat();
320 events.ScheduleEvent(EVENT_SHADOW_BOLT, urand(3000, 10000));
321 break;
322 case EVENT_FEAR:
323 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true))
324 DoCast(target, SPELL_FEAR);
325 events.ScheduleEvent(EVENT_FEAR, urand(10000, 20000));
326 break;
327 case EVENT_MIND_CONTROL:
328 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true))
329 DoCast(target, SPELL_SHADOW_COMMAND);
330 events.ScheduleEvent(EVENT_MIND_CONTROL, urand(30000, 35000));
331 break;
332 case EVENT_SPAWN_ADD:
333 for (uint8 i=0; i<2; ++i)
334 {
335 uint32 CreatureID;
336 if (urand(0, 2) == 0)
337 CreatureID = NPC_CHROMATIC_DRAKANOID;
338 else
339 CreatureID = Entry[urand(0, 4)];
340 if (Creature* dragon = me->SummonCreature(CreatureID, DrakeSpawnLoc[i]))
341 {
342 dragon->setFaction(103);
343 dragon->AI()->AttackStart(me->GetVictim());
344 }
345
346 if (++SpawnedAdds >= 42)
347 {
348 if (Creature* nefarian = me->SummonCreature(NPC_NEFARIAN, NefarianLoc[0]))
349 {
350 nefarian->setActive(true);
351 nefarian->SetCanFly(true);
352 nefarian->SetDisableGravity(true);
353 nefarian->CastSpell((Unit*)NULL, SPELL_SHADOWFLAME_INITIAL);
354 nefarian->GetMotionMaster()->MovePoint(1, NefarianLoc[1]);
355 }
356 events.CancelEvent(EVENT_MIND_CONTROL);
357 events.CancelEvent(EVENT_FEAR);
358 events.CancelEvent(EVENT_SHADOW_BOLT);
359 me->SetVisible(false);
360 return;
361 }
362 }
363 events.ScheduleEvent(EVENT_SPAWN_ADD, 4000);
364 break;
365 }
366 }
367 }
368 }

390class boss_nefarian : public CreatureScript
391{
392public:
393 boss_nefarian() : CreatureScript("boss_nefarian") { }
394
395 struct boss_nefarianAI : public BossAI
396 {
397 boss_nefarianAI(Creature* creature) : BossAI(creature, DATA_NEFARIAN)
398 {
399 Initialize();
400 }
401
402 void Initialize()
403 {
404 Phase3 = false;
405 canDespawn = false;
406 DespawnTimer = 30000;
407 }
408
409 void Reset() override
410 {
411 Initialize();
412 }
413
414 void JustReachedHome() override
415 {
416 canDespawn = true;
417 }
418
419 void EnterCombat(Unit* /*who*/) override
420 {
421 events.ScheduleEvent(EVENT_SHADOWFLAME, 12000);
422 events.ScheduleEvent(EVENT_FEAR, urand(25000, 35000));
423 events.ScheduleEvent(EVENT_VEILOFSHADOW, urand(25000, 35000));
424 events.ScheduleEvent(EVENT_CLEAVE, 7000);
425 //events.ScheduleEvent(EVENT_TAILLASH, 10000);
426 events.ScheduleEvent(EVENT_CLASSCALL, urand(30000, 35000));
427 Talk(SAY_RANDOM);
428 }
429
430 void JustDied(Unit* /*Killer*/) override
431 {
432 _JustDied();
433 Talk(SAY_DEATH);
434 }
435
436 void KilledUnit(Unit* victim) override
437 {
438 if (rand32() % 5)
439 return;
440
441 Talk(SAY_SLAY, victim);
442 }
443
444 void MovementInform(uint32 type, uint32 id) override
445 {
446 if (type != POINT_MOTION_TYPE)
447 return;
448
449 if (id == 1)
450 {
451 me->SetInCombatWithZone();
452 if (me->GetVictim())
453 AttackStart(me->GetVictim());
454 }
455 }
456
457 void UpdateAI(uint32 diff) override
458 {
459 if (canDespawn && DespawnTimer <= diff)
460 {
461 instance->SetBossState(DATA_NEFARIAN, FAIL);
462
463 std::list<Creature*> constructList;
464 me->GetCreatureListWithEntryInGrid(constructList, NPC_BONE_CONSTRUCT, 500.0f);
465 for (std::list<Creature*>::const_iterator itr = constructList.begin(); itr != constructList.end(); ++itr)
466 (*itr)->DespawnOrUnsummon();
467
468 } else DespawnTimer -= diff;
469
470 if (!UpdateVictim())
471 return;
472
473 if (canDespawn)
474 canDespawn = false;
475
476 events.Update(diff);
477
478 if (me->HasUnitState(UNIT_STATE_CASTING))
479 return;
480
481 while (uint32 eventId = events.ExecuteEvent())
482 {
483 switch (eventId)
484 {
485 case EVENT_SHADOWFLAME:
486 DoCastVictim(SPELL_SHADOWFLAME);
487 events.ScheduleEvent(EVENT_SHADOWFLAME, 12000);
488 break;
489 case EVENT_FEAR:
490 DoCastVictim(SPELL_BELLOWINGROAR);
491 events.ScheduleEvent(EVENT_FEAR, urand(25000, 35000));
492 break;
493 case EVENT_VEILOFSHADOW:
494 DoCastVictim(SPELL_VEILOFSHADOW);
495 events.ScheduleEvent(EVENT_VEILOFSHADOW, urand(25000, 35000));
496 break;
497 case EVENT_CLEAVE:
498 DoCastVictim(SPELL_CLEAVE);
499 events.ScheduleEvent(EVENT_CLEAVE, 7000);
500 break;
501 case EVENT_TAILLASH:
502 // Cast NYI since we need a better check for behind target
503 DoCastVictim(SPELL_TAILLASH);
504 events.ScheduleEvent(EVENT_TAILLASH, 10000);
505 break;
506 case EVENT_CLASSCALL:
507 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
508 switch (target->getClass())
509 {
510 case CLASS_MAGE:
511 Talk(SAY_MAGE);
512 DoCast(me, SPELL_MAGE);
513 break;
514 case CLASS_WARRIOR:
515 Talk(SAY_WARRIOR);
516 DoCast(me, SPELL_WARRIOR);
517 break;
518 case CLASS_DRUID:
519 Talk(SAY_DRUID);
520 DoCast(target, SPELL_DRUID);
521 break;
522 case CLASS_PRIEST:
523 Talk(SAY_PRIEST);
524 DoCast(me, SPELL_PRIEST);
525 break;
526 case CLASS_PALADIN:
527 Talk(SAY_PALADIN);
528 DoCast(me, SPELL_PALADIN);
529 break;
530 case CLASS_SHAMAN:
531 Talk(SAY_SHAMAN);
532 DoCast(me, SPELL_SHAMAN);
533 break;
534 case CLASS_WARLOCK:
535 Talk(SAY_WARLOCK);
536 DoCast(me, SPELL_WARLOCK);
537 break;
538 case CLASS_HUNTER:
539 Talk(SAY_HUNTER);
540 DoCast(me, SPELL_HUNTER);
541 break;
542 case CLASS_ROGUE:
543 Talk(SAY_ROGUE);
544 DoCast(me, SPELL_ROGUE);
545 break;
546 case CLASS_DEATH_KNIGHT:
547 Talk(SAY_DEATH_KNIGHT);
548 DoCast(me, SPELL_DEATH_KNIGHT);
549 break;
550 default:
551 break;
552 }
553 events.ScheduleEvent(EVENT_CLASSCALL, urand(30000, 35000));
554 break;
555 }
556 }
557
558 // Phase3 begins when health below 20 pct
559 if (!Phase3 && HealthBelowPct(20))
560 {
561 std::list<Creature*> constructList;
562 me->GetCreatureListWithEntryInGrid(constructList, NPC_BONE_CONSTRUCT, 500.0f);
563 for (std::list<Creature*>::const_iterator itr = constructList.begin(); itr != constructList.end(); ++itr)
564 if ((*itr) && !(*itr)->IsAlive())
565 {
566 (*itr)->Respawn();
567 (*itr)->SetInCombatWithZone();
568 (*itr)->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
569 (*itr)->SetReactState(REACT_AGGRESSIVE);
570 (*itr)->SetStandState(UNIT_STAND_STATE_STAND);
571 }
572
573 Phase3 = true;
574 Talk(SAY_RAISE_SKELETONS);
575 }
576
577 DoMeleeAttackIfReady();
578 }
579
580 private:
581 bool canDespawn;
582 uint32 DespawnTimer;
583 bool Phase3;
584
585 };

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20
21enum Spells
22{
23 SPELL_SHADOWWORDPAIN = 10894,
24 SPELL_MANABURN = 10876,
25 SPELL_PSYCHICSCREAM = 8122,
26 SPELL_SHADOWSHIELD = 22417
27};
28
29enum Events
30{
31 EVENT_SHADOW_WORD_PAIN = 1,
32 EVENT_MANABURN = 2,
33 EVENT_PSYCHIC_SCREAM = 3,
34 EVENT_SHADOWSHIELD = 4
35};
36
37class boss_high_interrogator_gerstahn : public CreatureScript
38{
39 public:
40 boss_high_interrogator_gerstahn() : CreatureScript("boss_high_interrogator_gerstahn") { }
41
42 struct boss_high_interrogator_gerstahnAI : public ScriptedAI
43 {
44 boss_high_interrogator_gerstahnAI(Creature* creature) : ScriptedAI(creature) { }
45
46 void Reset() override
47 {
48 _events.Reset();
49 }
50
51 void EnterCombat(Unit* /*who*/) override
52 {
53 _events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, 4000);
54 _events.ScheduleEvent(EVENT_MANABURN, 14000);
55 _events.ScheduleEvent(EVENT_PSYCHIC_SCREAM, 32000);
56 _events.ScheduleEvent(EVENT_SHADOWSHIELD, 8000);
57 }
58
59 void UpdateAI(uint32 diff) override
60 {
61 if (!UpdateVictim())
62 return;
63
64 _events.Update(diff);
65
66 while (uint32 eventId = _events.ExecuteEvent())
67 {
68 switch (eventId)
69 {
70 case EVENT_SHADOW_WORD_PAIN:
71 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
72 DoCast(target, SPELL_SHADOWWORDPAIN);
73 _events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, 7000);
74 break;
75 case EVENT_PSYCHIC_SCREAM:
76 DoCastVictim(SPELL_PSYCHICSCREAM);
77 _events.ScheduleEvent(EVENT_PSYCHIC_SCREAM, 30000);
78 break;
79 case EVENT_MANABURN:
80 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
81 DoCast(target, SPELL_MANABURN);
82 _events.ScheduleEvent(EVENT_MANABURN, 10000);
83 break;
84 case EVENT_SHADOWSHIELD:
85 DoCast(me, SPELL_SHADOWSHIELD);
86 _events.ScheduleEvent(EVENT_SHADOWSHIELD, 25000);
87 break;
88 default:
89 break;
90 }
91 }
92
93 DoMeleeAttackIfReady();
94 }
95
96 private:
97 EventMap _events;
98 };
99
100 CreatureAI* GetAI(Creature* creature) const override
101 {
102 return new boss_high_interrogator_gerstahnAI(creature);
103 }
104};
105
106void AddSC_boss_high_interrogator_gerstahn()
107{
108 new boss_high_interrogator_gerstahn();
109}

18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20#include "blackrock_depths.h"
21
22enum Spells
23{
24 SPELL_FIERYBURST = 13900,
25 SPELL_WARSTOMP = 24375
26};
27
28enum Events
29{
30 EVENT_FIERY_BURST = 1,
31 EVENT_WARSTOMP = 2
32};
33
34enum Phases
35{
36 PHASE_ONE = 1,
37 PHASE_TWO = 2
38};
39
40class boss_magmus : public CreatureScript
41{
42 public:
43 boss_magmus() : CreatureScript("boss_magmus") { }
44
45 struct boss_magmusAI : public ScriptedAI
46 {
47 boss_magmusAI(Creature* creature) : ScriptedAI(creature) { }
48
49 void Reset() override
50 {
51 _events.Reset();
52 }
53
54 void EnterCombat(Unit* /*who*/) override
55 {
56 _events.SetPhase(PHASE_ONE);
57 _events.ScheduleEvent(EVENT_FIERY_BURST, 5000);
58 }
59
60 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
61 {
62 if (me->HealthBelowPctDamaged(50, damage) && _events.IsInPhase(PHASE_ONE))
63 {
64 _events.SetPhase(PHASE_TWO);
65 _events.ScheduleEvent(EVENT_WARSTOMP, 0, 0, PHASE_TWO);
66 }
67 }
68
69 void UpdateAI(uint32 diff) override
70 {
71 if (!UpdateVictim())
72 return;
73
74 _events.Update(diff);
75
76 while (uint32 eventId = _events.ExecuteEvent())
77 {
78 switch (eventId)
79 {
80 case EVENT_FIERY_BURST:
81 DoCastVictim(SPELL_FIERYBURST);
82 _events.ScheduleEvent(EVENT_FIERY_BURST, 6000);
83 break;
84 case EVENT_WARSTOMP:
85 DoCastVictim(SPELL_WARSTOMP);
86 _events.ScheduleEvent(EVENT_WARSTOMP, 8000, 0, PHASE_TWO);
87 break;
88 default:
89 break;
90 }
91 }
92
93 DoMeleeAttackIfReady();
94 }
95
96 void JustDied(Unit* /*killer*/) override
97 {
98 if (InstanceScript* instance = me->GetInstanceScript())
99 instance->HandleGameObject(instance->GetGuidData(DATA_THRONE_DOOR), true);
100 }
101
102 private:
103 EventMap _events;
104 };
105
106 CreatureAI* GetAI(Creature* creature) const override
107 {
108 return new boss_magmusAI(creature);
109 }
110};
111
112void AddSC_boss_magmus()
113{
114 new boss_magmus();
115}

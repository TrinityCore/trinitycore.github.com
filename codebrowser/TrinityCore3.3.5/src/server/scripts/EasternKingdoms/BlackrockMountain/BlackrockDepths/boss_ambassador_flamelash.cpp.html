18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20
21enum Spells
22{
23 SPELL_FIREBLAST = 15573
24};
25
26enum Events
27{
28 EVENT_FIREBLAST = 1,
29 EVENT_SUMMON_SPIRITS = 2
30};
31
32class boss_ambassador_flamelash : public CreatureScript
33{
34 public:
35 boss_ambassador_flamelash() : CreatureScript("boss_ambassador_flamelash") { }
36
37 struct boss_ambassador_flamelashAI : public ScriptedAI
38 {
39 boss_ambassador_flamelashAI(Creature* creature) : ScriptedAI(creature) { }
40
41 void Reset() override
42 {
43 _events.Reset();
44 }
45
46 void EnterCombat(Unit* /*who*/) override
47 {
48 _events.ScheduleEvent(EVENT_FIREBLAST, 2000);
49 _events.ScheduleEvent(EVENT_SUMMON_SPIRITS, 24000);
50 }
51
52 void SummonSpirit(Unit* victim)
53 {
54 if (Creature* spirit = DoSpawnCreature(9178, frand(-9, 9), frand(-9, 9), 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 60000))
55 spirit->AI()->AttackStart(victim);
56 }
57
58 void UpdateAI(uint32 diff) override
59 {
60 if (!UpdateVictim())
61 return;
62
63 _events.Update(diff);
64
65 while (uint32 eventId = _events.ExecuteEvent())
66 {
67 switch (eventId)
68 {
69 case EVENT_FIREBLAST:
70 DoCastVictim(SPELL_FIREBLAST);
71 _events.ScheduleEvent(EVENT_FIREBLAST, 7000);
72 break;
73 case EVENT_SUMMON_SPIRITS:
74 for (uint32 i = 0; i < 4; ++i)
75 SummonSpirit(me->GetVictim());
76 _events.ScheduleEvent(EVENT_SUMMON_SPIRITS, 30000);
77 break;
78 default:
79 break;
80 }
81 }
82
83 DoMeleeAttackIfReady();
84 }
85
86 private:
87 EventMap _events;
88 };
89
90 CreatureAI* GetAI(Creature* creature) const override
91 {
92 return new boss_ambassador_flamelashAI(creature);
93 }
94};
95
96void AddSC_boss_ambassador_flamelash()
97{
98 new boss_ambassador_flamelash();
99}

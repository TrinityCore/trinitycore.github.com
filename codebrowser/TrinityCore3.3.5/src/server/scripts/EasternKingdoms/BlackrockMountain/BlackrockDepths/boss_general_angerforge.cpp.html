18#include "ScriptMgr.h"
19#include "ScriptedCreature.h"
20
21enum Spells
22{
23 SPELL_MIGHTYBLOW = 14099,
24 SPELL_HAMSTRING = 9080,
25 SPELL_CLEAVE = 20691
26};
27
28enum Events
29{
30 EVENT_MIGHTYBLOW = 1,
31 EVENT_HAMSTRING = 2,
32 EVENT_CLEAVE = 3,
33 EVENT_MEDIC = 4,
34 EVENT_ADDS = 5
35};
36
37enum Phases
38{
39 PHASE_ONE = 1,
40 PHASE_TWO = 2
41};
42
43class boss_general_angerforge : public CreatureScript
44{
45 public:
46 boss_general_angerforge() : CreatureScript("boss_general_angerforge") { }
47
48 struct boss_general_angerforgeAI : public ScriptedAI
49 {
50 boss_general_angerforgeAI(Creature* creature) : ScriptedAI(creature) { }
51
52 void Reset() override
53 {
54 _events.Reset();
55 }
56
57 void EnterCombat(Unit* /*who*/) override
58 {
59 _events.SetPhase(PHASE_ONE);
60 _events.ScheduleEvent(EVENT_MIGHTYBLOW, 8000);
61 _events.ScheduleEvent(EVENT_HAMSTRING, 12000);
62 _events.ScheduleEvent(EVENT_CLEAVE, 16000);
63 }
64
65 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
66 {
67 if (me->HealthBelowPctDamaged(20, damage) && _events.IsInPhase(PHASE_ONE))
68 {
69 _events.SetPhase(PHASE_TWO);
70 _events.ScheduleEvent(EVENT_MEDIC, 0, 0, PHASE_TWO);
71 _events.ScheduleEvent(EVENT_ADDS, 0, 0, PHASE_TWO);
72 }
73 }
74
75 void SummonAdd(Unit* victim)
76 {
77 if (Creature* SummonedAdd = DoSpawnCreature(8901, float(irand(-14, 14)), float(irand(-14, 14)), 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 120000))
78 SummonedAdd->AI()->AttackStart(victim);
79 }
80
81 void SummonMedic(Unit* victim)
82 {
83 if (Creature* SummonedMedic = DoSpawnCreature(8894, float(irand(-9, 9)), float(irand(-9, 9)), 0, 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 120000))
84 SummonedMedic->AI()->AttackStart(victim);
85 }
86
87 void UpdateAI(uint32 diff) override
88 {
89 if (!UpdateVictim())
90 return;
91
92 _events.Update(diff);
93
94 while (uint32 eventId = _events.ExecuteEvent())
95 {
96 switch (eventId)
97 {
98 case EVENT_MIGHTYBLOW:
99 DoCastVictim(SPELL_MIGHTYBLOW);
100 _events.ScheduleEvent(EVENT_MIGHTYBLOW, 18000);
101 break;
102 case EVENT_HAMSTRING:
103 DoCastVictim(SPELL_HAMSTRING);
104 _events.ScheduleEvent(EVENT_HAMSTRING, 15000);
105 break;
106 case EVENT_CLEAVE:
107 DoCastVictim(SPELL_CLEAVE);
108 _events.ScheduleEvent(EVENT_CLEAVE, 9000);
109 break;
110 case EVENT_MEDIC:
111 for (uint8 i = 0; i < 2; ++i)
112 SummonMedic(me->GetVictim());
113 break;
114 case EVENT_ADDS:
115 for (uint8 i = 0; i < 3; ++i)
116 SummonAdd(me->GetVictim());
117 _events.ScheduleEvent(EVENT_ADDS, 25000, 0, PHASE_TWO);
118 break;
119 default:
120 break;
121 }
122 }
123
124 DoMeleeAttackIfReady();
125 }
126
127 private:
128 EventMap _events;
129 };
130
131 CreatureAI* GetAI(Creature* creature) const override
132 {
133 return new boss_general_angerforgeAI(creature);
134 }
135};
136
137void AddSC_boss_general_angerforge()
138{
139 new boss_general_angerforge();
140}

52class boss_gloomrel : public CreatureScript
53{
54public:
55 boss_gloomrel() : CreatureScript("boss_gloomrel") { }
56
57 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
58 {
59 player->PlayerTalkClass->ClearMenus();
60 switch (action)
61 {
62 case GOSSIP_ACTION_INFO_DEF+1:
63 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_TEACH_2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 11);
64 player->SEND_GOSSIP_MENU(2606, creature->GetGUID());
65 break;
66 case GOSSIP_ACTION_INFO_DEF+11:
67 player->CLOSE_GOSSIP_MENU();
68 player->CastSpell(player, SPELL_LEARN_SMELT, false);
69 break;
70 case GOSSIP_ACTION_INFO_DEF+2:
71 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_TEACH_3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 22);
72 player->SEND_GOSSIP_MENU(2604, creature->GetGUID());
73 break;
74 case GOSSIP_ACTION_INFO_DEF+22:
75 player->CLOSE_GOSSIP_MENU();
76 if (InstanceScript* instance = creature->GetInstanceScript())
77 {
78 //are 5 minutes expected? go template may have data to despawn when used at quest
79 instance->DoRespawnGameObject(instance->GetGuidData(DATA_GO_CHALICE), MINUTE*5);
80 }
81 break;
82 }
83 return true;
84 }
85
86 bool OnGossipHello(Player* player, Creature* creature) override
87 {
88 if (player->GetQuestRewardStatus(QUEST_SPECTRAL_CHALICE) == 1 && player->GetSkillValue(SKILL_MINING) >= DATA_SKILLPOINT_MIN && !player->HasSpell(SPELL_SMELT_DARK_IRON))
89 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_TEACH_1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
90
91 if (player->GetQuestRewardStatus(QUEST_SPECTRAL_CHALICE) == 0 && player->GetSkillValue(SKILL_MINING) >= DATA_SKILLPOINT_MIN)
92 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_TRIBUTE, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
93
94 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
95 return true;
96 }

120class boss_doomrel : public CreatureScript
121{
122 public:
123 boss_doomrel() : CreatureScript("boss_doomrel") { }
124
125 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
126 {
127 player->PlayerTalkClass->ClearMenus();
128 switch (action)
129 {
130 case GOSSIP_ACTION_INFO_DEF+1:
131 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_SELECT_DOOMREL, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
132 player->SEND_GOSSIP_MENU(2605, creature->GetGUID());
133 break;
134 case GOSSIP_ACTION_INFO_DEF+2:
135 player->CLOSE_GOSSIP_MENU();
136 //start event here
137 creature->setFaction(FACTION_HOSTILE);
138 creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
139 creature->AI()->AttackStart(player);
140 InstanceScript* instance = creature->GetInstanceScript();
141 if (instance)
142 instance->SetGuidData(DATA_EVENSTARTER, player->GetGUID());
143 break;
144 }
145 return true;
146 }
147
148 bool OnGossipHello(Player* player, Creature* creature) override
149 {
150 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_ITEM_CHALLENGE, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
151 player->SEND_GOSSIP_MENU(2601, creature->GetGUID());
152
153 return true;
154 }

156 struct boss_doomrelAI : public ScriptedAI
157 {
158 boss_doomrelAI(Creature* creature) : ScriptedAI(creature)
159 {
160 Initialize();
161 _instance = creature->GetInstanceScript();
162 }
163
164 void Initialize()
165 {
166 _voidwalkers = false;
167 }
168
169 void Reset() override
170 {
171 Initialize();
172
173 me->setFaction(FACTION_FRIEND);
174
175 // was set before event start, so set again
176 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
177
178 if (_instance->GetData(DATA_GHOSTKILL) >= 7)
179 me->SetUInt32Value(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_NONE);
180 else
181 me->SetUInt32Value(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
182 }
183
184 void EnterCombat(Unit* /*who*/) override
185 {
186 _events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, 10000);
187 _events.ScheduleEvent(EVENT_IMMOLATE, 18000);
188 _events.ScheduleEvent(EVENT_CURSE_OF_WEAKNESS, 5000);
189 _events.ScheduleEvent(EVENT_DEMONARMOR, 16000);
190 }
191
192 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
193 {
194 if (!_voidwalkers && !HealthAbovePct(50))
195 {
196 DoCastVictim(SPELL_SUMMON_VOIDWALKERS, true);
197 _voidwalkers = true;
198 }
199 }
200
201 void EnterEvadeMode(EvadeReason why) override
202 {
203 ScriptedAI::EnterEvadeMode(why);
204
205 _instance->SetGuidData(DATA_EVENSTARTER, ObjectGuid::Empty);
206 }
207
208 void JustDied(Unit* /*killer*/) override
209 {
210 _instance->SetData(DATA_GHOSTKILL, 1);
211 }
212
213 void UpdateAI(uint32 diff) override
214 {
215 if (!UpdateVictim())
216 return;
217
218 _events.Update(diff);
219
220 while (uint32 eventId = _events.ExecuteEvent())
221 {
222 switch (eventId)
223 {
224 case EVENT_SHADOW_BOLT_VOLLEY:
225 DoCastVictim(SPELL_SHADOWBOLTVOLLEY);
226 _events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, 12000);
227 break;
228 case EVENT_IMMOLATE:
229 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
230 DoCast(target, SPELL_IMMOLATE);
231 _events.ScheduleEvent(EVENT_IMMOLATE, 25000);
232 break;
233 case EVENT_CURSE_OF_WEAKNESS:
234 DoCastVictim(SPELL_CURSEOFWEAKNESS);
235 _events.ScheduleEvent(EVENT_CURSE_OF_WEAKNESS, 45000);
236 break;
237 case EVENT_DEMONARMOR:
238 DoCast(me, SPELL_DEMONARMOR);
239 _events.ScheduleEvent(EVENT_DEMONARMOR, 300000);
240 break;
241 default:
242 break;
243 }
244 }
245
246 DoMeleeAttackIfReady();
247 }
248
249 private:
250 InstanceScript* _instance;
251 EventMap _events;
252 bool _voidwalkers;
253 };
254
255 CreatureAI* GetAI(Creature* creature) const override
256 {
257 return GetInstanceAI<boss_doomrelAI>(creature);
258 }
259};

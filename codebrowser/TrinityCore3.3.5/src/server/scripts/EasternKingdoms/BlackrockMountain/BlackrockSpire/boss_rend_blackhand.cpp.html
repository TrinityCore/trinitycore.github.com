160class boss_rend_blackhand : public CreatureScript
161{
162public:
163 boss_rend_blackhand() : CreatureScript("boss_rend_blackhand") { }
164
165 struct boss_rend_blackhandAI : public BossAI
166 {
167 boss_rend_blackhandAI(Creature* creature) : BossAI(creature, DATA_WARCHIEF_REND_BLACKHAND)
168 {
169 gythEvent = false;
170 victorGUID.Clear();
171 portcullisGUID.Clear();
172 }
173
174 void Reset() override
175 {
176 _Reset();
177 gythEvent = false;
178 victorGUID.Clear();
179 portcullisGUID.Clear();
180 }
181
182 void EnterCombat(Unit* /*who*/) override
183 {
184 _EnterCombat();
185 events.ScheduleEvent(EVENT_WHIRLWIND, urand(13000, 15000));
186 events.ScheduleEvent(EVENT_CLEAVE, urand(15000, 17000));
187 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(17000, 19000));
188 }
189
190 void JustDied(Unit* /*killer*/) override
191 {
192 _JustDied();
193 if (Creature* victor = me->FindNearestCreature(NPC_LORD_VICTOR_NEFARIUS, 75.0f, true))
194 victor->AI()->SetData(1, 2);
195 }
196
197 void SetData(uint32 type, uint32 data) override
198 {
199 if (type == AREATRIGGER && data == AREATRIGGER_BLACKROCK_STADIUM)
200 {
201 if (!gythEvent)
202 {
203 gythEvent = true;
204
205 if (Creature* victor = me->FindNearestCreature(NPC_LORD_VICTOR_NEFARIUS, 5.0f, true))
206 victorGUID = victor->GetGUID();
207
208 if (GameObject* portcullis = me->FindNearestGameObject(GO_DR_PORTCULLIS, 50.0f))
209 portcullisGUID = portcullis->GetGUID();
210
211 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);
212 events.ScheduleEvent(EVENT_START_1, 1000);
213 }
214 }
215 }
216
217 void MovementInform(uint32 type, uint32 id) override
218 {
219 if (type == WAYPOINT_MOTION_TYPE)
220 {
221 switch (id)
222 {
223 case 5:
224 events.ScheduleEvent(EVENT_TELEPORT_1, 2000);
225 break;
226 case 11:
227 if (Creature* gyth = me->FindNearestCreature(NPC_GYTH, 10.0f, true))
228 gyth->AI()->SetData(1, 1);
229 me->DespawnOrUnsummon(1000);
230 break;
231 }
232 }
233 }
234
235 void UpdateAI(uint32 diff) override
236 {
237 if (gythEvent)
238 {
239 events.Update(diff);
240
241 while (uint32 eventId = events.ExecuteEvent())
242 {
243 switch (eventId)
244 {
245 case EVENT_START_1:
246 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
247 victor->AI()->Talk(SAY_NEFARIUS_0);
248 events.ScheduleEvent(EVENT_START_2, 4000);
249 break;
250 case EVENT_START_2:
251 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);
252 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
253 victor->HandleEmoteCommand(EMOTE_ONESHOT_POINT);
254 events.ScheduleEvent(EVENT_START_3, 4000);
255 break;
256 case EVENT_START_3:
257 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
258 victor->AI()->Talk(SAY_NEFARIUS_1);
259 events.ScheduleEvent(EVENT_WAVE_1, 2000);
260 events.ScheduleEvent(EVENT_TURN_TO_REND, 4000);
261 events.ScheduleEvent(EVENT_WAVES_TEXT_1, 20000);
262 break;
263 case EVENT_TURN_TO_REND:
264 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
265 {
266 victor->SetFacingToObject(me);
267 victor->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
268 }
269 break;
270 case EVENT_TURN_TO_PLAYER:
271 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
272 if (Unit* player = victor->SelectNearestPlayer(60.0f))
273 victor->SetFacingToObject(player);
274 break;
275 case EVENT_TURN_TO_FACING_1:
276 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
277 victor->SetFacingTo(1.518436f);
278 break;
279 case EVENT_TURN_TO_FACING_2:
280 me->SetFacingTo(1.658063f);
281 break;
282 case EVENT_TURN_TO_FACING_3:
283 me->SetFacingTo(1.500983f);
284 break;
285 case EVENT_WAVES_EMOTE_1:
286 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
287 victor->HandleEmoteCommand(EMOTE_ONESHOT_QUESTION);
288 break;
289 case EVENT_WAVES_EMOTE_2:
290 me->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
291 break;
292 case EVENT_WAVES_TEXT_1:
293 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);
294 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
295 victor->AI()->Talk(SAY_NEFARIUS_2);
296 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
297 events.ScheduleEvent(EVENT_TURN_TO_FACING_1, 4000);
298 events.ScheduleEvent(EVENT_WAVES_EMOTE_1, 5000);
299 events.ScheduleEvent(EVENT_WAVE_2, 2000);
300 events.ScheduleEvent(EVENT_WAVES_TEXT_2, 20000);
301 break;
302 case EVENT_WAVES_TEXT_2:
303 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);
304 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
305 victor->AI()->Talk(SAY_NEFARIUS_3);
306 events.ScheduleEvent(EVENT_TURN_TO_FACING_1, 4000);
307 events.ScheduleEvent(EVENT_WAVE_3, 2000);
308 events.ScheduleEvent(EVENT_WAVES_TEXT_3, 20000);
309 break;
310 case EVENT_WAVES_TEXT_3:
311 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);
312 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
313 victor->AI()->Talk(SAY_NEFARIUS_4);
314 events.ScheduleEvent(EVENT_TURN_TO_FACING_1, 4000);
315 events.ScheduleEvent(EVENT_WAVE_4, 2000);
316 events.ScheduleEvent(EVENT_WAVES_TEXT_4, 20000);
317 break;
318 case EVENT_WAVES_TEXT_4:
319 Talk(SAY_BLACKHAND_1);
320 events.ScheduleEvent(EVENT_WAVES_EMOTE_2, 4000);
321 events.ScheduleEvent(EVENT_TURN_TO_FACING_3, 8000);
322 events.ScheduleEvent(EVENT_WAVE_5, 2000);
323 events.ScheduleEvent(EVENT_WAVES_TEXT_5, 20000);
324 break;
325 case EVENT_WAVES_TEXT_5:
326 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);
327 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
328 victor->AI()->Talk(SAY_NEFARIUS_5);
329 events.ScheduleEvent(EVENT_TURN_TO_FACING_1, 4000);
330 events.ScheduleEvent(EVENT_WAVE_6, 2000);
331 events.ScheduleEvent(EVENT_WAVES_COMPLETE_TEXT_1, 20000);
332 break;
333 case EVENT_WAVES_COMPLETE_TEXT_1:
334 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);
335 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
336 victor->AI()->Talk(SAY_NEFARIUS_6);
337 events.ScheduleEvent(EVENT_TURN_TO_FACING_1, 4000);
338 events.ScheduleEvent(EVENT_WAVES_COMPLETE_TEXT_2, 13000);
339 break;
340 case EVENT_WAVES_COMPLETE_TEXT_2:
341 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
342 victor->AI()->Talk(SAY_NEFARIUS_7);
343 Talk(SAY_BLACKHAND_2);
344 events.ScheduleEvent(EVENT_PATH_REND, 1000);
345 events.ScheduleEvent(EVENT_WAVES_COMPLETE_TEXT_3, 4000);
346 break;
347 case EVENT_WAVES_COMPLETE_TEXT_3:
348 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
349 victor->AI()->Talk(SAY_NEFARIUS_8);
350 events.ScheduleEvent(EVENT_PATH_NEFARIUS, 1000);
351 events.ScheduleEvent(EVENT_PATH_REND, 1000);
352 break;
353 case EVENT_PATH_NEFARIUS:
354 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
355 victor->GetMotionMaster()->MovePath(NEFARIUS_PATH_1, true);
356 break;
357 case EVENT_PATH_REND:
358 me->GetMotionMaster()->MovePath(REND_PATH_1, false);
359 break;
360 case EVENT_TELEPORT_1:
361 me->NearTeleportTo(194.2993f, -474.0814f, 121.4505f, -0.01225555f);
362 events.ScheduleEvent(EVENT_TELEPORT_2, 50000);
363 break;
364 case EVENT_TELEPORT_2:
365 me->NearTeleportTo(216.485f, -434.93f, 110.888f, -0.01225555f);
366 me->SummonCreature(NPC_GYTH, 211.762f, -397.5885f, 111.1817f, 4.747295f);
367 break;
368 case EVENT_WAVE_1:
369 if (GameObject* portcullis = me->GetMap()->GetGameObject(portcullisGUID))
370 portcullis->UseDoorOrButton();
371 // move wave
372 break;
373 case EVENT_WAVE_2:
374 // spawn wave
375 if (GameObject* portcullis = me->GetMap()->GetGameObject(portcullisGUID))
376 portcullis->UseDoorOrButton();
377 // move wave
378 break;
379 case EVENT_WAVE_3:
380 // spawn wave
381 if (GameObject* portcullis = me->GetMap()->GetGameObject(portcullisGUID))
382 portcullis->UseDoorOrButton();
383 // move wave
384 break;
385 case EVENT_WAVE_4:
386 // spawn wave
387 if (GameObject* portcullis = me->GetMap()->GetGameObject(portcullisGUID))
388 portcullis->UseDoorOrButton();
389 // move wave
390 break;
391 case EVENT_WAVE_5:
392 // spawn wave
393 if (GameObject* portcullis = me->GetMap()->GetGameObject(portcullisGUID))
394 portcullis->UseDoorOrButton();
395 // move wave
396 break;
397 case EVENT_WAVE_6:
398 // spawn wave
399 if (GameObject* portcullis = me->GetMap()->GetGameObject(portcullisGUID))
400 portcullis->UseDoorOrButton();
401 // move wave
402 break;
403 default:
404 break;
405 }
406 }
407 }
408
409 if (!UpdateVictim())
410 return;
411
412 events.Update(diff);
413
414 if (me->HasUnitState(UNIT_STATE_CASTING))
415 return;
416
417 while (uint32 eventId = events.ExecuteEvent())
418 {
419 switch (eventId)
420 {
421 case EVENT_WHIRLWIND:
422 DoCast(SPELL_WHIRLWIND);
423 events.ScheduleEvent(EVENT_WHIRLWIND, urand(13000, 18000));
424 break;
425 case EVENT_CLEAVE:
426 DoCastVictim(SPELL_CLEAVE);
427 events.ScheduleEvent(EVENT_CLEAVE, urand(10000, 14000));
428 break;
429 case EVENT_MORTAL_STRIKE:
430 DoCastVictim(SPELL_MORTAL_STRIKE);
431 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(14000, 16000));
432 break;
433 }
434 }
435 DoMeleeAttackIfReady();
436 }
437
438 private:
439 bool gythEvent;
440 ObjectGuid victorGUID;
441 ObjectGuid portcullisGUID;
442 };

19#include "ScriptMgr.h"
20#include "ScriptedCreature.h"
21#include "blackrock_spire.h"
22
23enum Spells
24{
25 SPELL_FLAMEBREAK = 16785,
26 SPELL_IMMOLATE = 20294,
27 SPELL_TERRIFYINGROAR = 14100,
28};
29
30enum Events
31{
32 EVENT_FLAME_BREAK = 1,
33 EVENT_IMMOLATE = 2,
34 EVENT_TERRIFYING_ROAR = 3,
35};
36
37class boss_the_beast : public CreatureScript
38{
39public:
40 boss_the_beast() : CreatureScript("boss_the_beast") { }
41
42 CreatureAI* GetAI(Creature* creature) const override
43 {
44 return new boss_thebeastAI(creature);
45 }
46
47 struct boss_thebeastAI : public BossAI
48 {
49 boss_thebeastAI(Creature* creature) : BossAI(creature, DATA_THE_BEAST) { }
50
51 void Reset() override
52 {
53 _Reset();
54 }
55
56 void EnterCombat(Unit* /*who*/) override
57 {
58 _EnterCombat();
59 events.ScheduleEvent(EVENT_FLAME_BREAK, 12 * IN_MILLISECONDS);
60 events.ScheduleEvent(EVENT_IMMOLATE, 3 * IN_MILLISECONDS);
61 events.ScheduleEvent(EVENT_TERRIFYING_ROAR, 23 * IN_MILLISECONDS);
62 }
63
64 void JustDied(Unit* /*killer*/) override
65 {
66 _JustDied();
67 }
68
69 void UpdateAI(uint32 diff) override
70 {
71 if (!UpdateVictim())
72 return;
73
74 events.Update(diff);
75
76 if (me->HasUnitState(UNIT_STATE_CASTING))
77 return;
78
79 while (uint32 eventId = events.ExecuteEvent())
80 {
81 switch (eventId)
82 {
83 case EVENT_FLAME_BREAK:
84 DoCastVictim(SPELL_FLAMEBREAK);
85 events.ScheduleEvent(EVENT_FLAME_BREAK, 10 * IN_MILLISECONDS);
86 break;
87 case EVENT_IMMOLATE:
88 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
89 DoCast(target, SPELL_IMMOLATE);
90 events.ScheduleEvent(EVENT_IMMOLATE, 8 * IN_MILLISECONDS);
91 break;
92 case EVENT_TERRIFYING_ROAR:
93 DoCastVictim(SPELL_TERRIFYINGROAR);
94 events.ScheduleEvent(EVENT_TERRIFYING_ROAR, 20 * IN_MILLISECONDS);
95 break;
96 }
97 }
98 DoMeleeAttackIfReady();
99 }
100 };
101};
102
103void AddSC_boss_thebeast()
104{
105 new boss_the_beast();
106}

23enum Spells
24{
25 SPELL_SNAPKICK = 15618,
26 SPELL_CLEAVE = 15284,
27 SPELL_UPPERCUT = 10966,
28 SPELL_MORTALSTRIKE = 16856,
29 SPELL_PUMMEL = 15615,
30 SPELL_THROWAXE = 16075
31};
32
33enum Events
34{
35 EVENT_SNAP_KICK = 1,
36 EVENT_CLEAVE = 2,
37 EVENT_UPPERCUT = 3,
38 EVENT_MORTAL_STRIKE = 4,
39 EVENT_PUMMEL = 5,
40 EVENT_THROW_AXE = 6
41};
42
43class boss_warmaster_voone : public CreatureScript
44{
45public:
46 boss_warmaster_voone() : CreatureScript("boss_warmaster_voone") { }
47
48 struct boss_warmastervooneAI : public BossAI
49 {
50 boss_warmastervooneAI(Creature* creature) : BossAI(creature, DATA_WARMASTER_VOONE) { }
51
52 void Reset() override
53 {
54 _Reset();
55 }
56
57 void EnterCombat(Unit* /*who*/) override
58 {
59 _EnterCombat();
60 events.ScheduleEvent(EVENT_SNAP_KICK, 8 * IN_MILLISECONDS);
61 events.ScheduleEvent(EVENT_CLEAVE, 14 * IN_MILLISECONDS);
62 events.ScheduleEvent(EVENT_UPPERCUT, 20 * IN_MILLISECONDS);
63 events.ScheduleEvent(EVENT_MORTAL_STRIKE, 12 * IN_MILLISECONDS);
64 events.ScheduleEvent(EVENT_PUMMEL, 32 * IN_MILLISECONDS);
65 events.ScheduleEvent(EVENT_THROW_AXE, 1 * IN_MILLISECONDS);
66 }
67
68 void JustDied(Unit* /*killer*/) override
69 {
70 _JustDied();
71 }
72
73 void UpdateAI(uint32 diff) override
74 {
75 if (!UpdateVictim())
76 return;
77
78 events.Update(diff);
79
80 if (me->HasUnitState(UNIT_STATE_CASTING))
81 return;
82
83 while (uint32 eventId = events.ExecuteEvent())
84 {
85 switch (eventId)
86 {
87 case EVENT_SNAP_KICK:
88 DoCastVictim(SPELL_SNAPKICK);
89 events.ScheduleEvent(EVENT_SNAP_KICK, 6 * IN_MILLISECONDS);
90 break;
91 case EVENT_CLEAVE:
92 DoCastVictim(SPELL_CLEAVE);
93 events.ScheduleEvent(EVENT_CLEAVE, 12 * IN_MILLISECONDS);
94 break;
95 case EVENT_UPPERCUT:
96 DoCastVictim(SPELL_UPPERCUT);
97 events.ScheduleEvent(EVENT_UPPERCUT, 14 * IN_MILLISECONDS);
98 break;
99 case EVENT_MORTAL_STRIKE:
100 DoCastVictim(SPELL_MORTALSTRIKE);
101 events.ScheduleEvent(EVENT_MORTAL_STRIKE, 10 * IN_MILLISECONDS);
102 break;
103 case EVENT_PUMMEL:
104 DoCastVictim(SPELL_PUMMEL);
105 events.ScheduleEvent(EVENT_MORTAL_STRIKE, 16 * IN_MILLISECONDS);
106 break;
107 case EVENT_THROW_AXE:
108 DoCastVictim(SPELL_THROWAXE);
109 events.ScheduleEvent(EVENT_THROW_AXE, 8 * IN_MILLISECONDS);
110 break;
111 }
112 }
113 DoMeleeAttackIfReady();
114 }
115 };
116
117 CreatureAI* GetAI(Creature* creature) const override
118 {
119 return new boss_warmastervooneAI(creature);
120 }
121};
122
123void AddSC_boss_warmastervoone()
124{
125 new boss_warmaster_voone();
126}

31enum Emotes

36enum Spells

44enum Events
45{
46 EVENT_INFERNO = 1,
47 EVENT_IGNITE_MANA = 2,
48 EVENT_LIVING_BOMB = 3,
49};
50
51class boss_baron_geddon : public CreatureScript
52{
53 public:
54 boss_baron_geddon() : CreatureScript("boss_baron_geddon") { }
55
56 struct boss_baron_geddonAI : public BossAI
57 {
58 boss_baron_geddonAI(Creature* creature) : BossAI(creature, BOSS_BARON_GEDDON)
59 {
60 }
61
62 void EnterCombat(Unit* victim) override
63 {
64 BossAI::EnterCombat(victim);
65 events.ScheduleEvent(EVENT_INFERNO, 45000);
66 events.ScheduleEvent(EVENT_IGNITE_MANA, 30000);
67 events.ScheduleEvent(EVENT_LIVING_BOMB, 35000);
68 }
69
70 void UpdateAI(uint32 diff) override
71 {
72 if (!UpdateVictim())
73 return;
74
75 events.Update(diff);
76
77 // If we are <2% hp cast Armageddon
78 if (!HealthAbovePct(2))
79 {
80 me->InterruptNonMeleeSpells(true);
81 DoCast(me, SPELL_ARMAGEDDON);
82 Talk(EMOTE_SERVICE);
83 return;
84 }
85
86 if (me->HasUnitState(UNIT_STATE_CASTING))
87 return;
88
89 while (uint32 eventId = events.ExecuteEvent())
90 {
91 switch (eventId)
92 {
93 case EVENT_INFERNO:
94 DoCast(me, SPELL_INFERNO);
95 events.ScheduleEvent(EVENT_INFERNO, 45000);
96 break;
97 case EVENT_IGNITE_MANA:
98 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_IGNITE_MANA))
99 DoCast(target, SPELL_IGNITE_MANA);
100 events.ScheduleEvent(EVENT_IGNITE_MANA, 30000);
101 break;
102 case EVENT_LIVING_BOMB:
103 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
104 DoCast(target, SPELL_LIVING_BOMB);
105 events.ScheduleEvent(EVENT_LIVING_BOMB, 35000);
106 break;
107 default:
108 break;
109 }
110 }
111
112 DoMeleeAttackIfReady();
113 }
114 };
115
116 CreatureAI* GetAI(Creature* creature) const override
117 {
118 return new boss_baron_geddonAI(creature);
119 }
120};

124 new boss_baron_geddon();
125}

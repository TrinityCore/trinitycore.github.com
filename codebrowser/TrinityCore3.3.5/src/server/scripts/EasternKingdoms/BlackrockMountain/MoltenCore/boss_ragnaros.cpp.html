81 struct boss_ragnarosAI : public BossAI
82 {
83 boss_ragnarosAI(Creature* creature) : BossAI(creature, BOSS_RAGNAROS)
84 {
85 Initialize();
86 _introState = 0;
87 me->SetReactState(REACT_PASSIVE);
88 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
89 SetCombatMovement(false);
90 }
91
92 void Initialize()
93 {
94 _emergeTimer = 90000;
95 _hasYelledMagmaBurst = false;
96 _hasSubmergedOnce = false;
97 _isBanished = false;
98 }
99
100 void Reset() override
101 {
102 BossAI::Reset();
103 Initialize();
104 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
105 }
106
107 void EnterCombat(Unit* victim) override
108 {
109 BossAI::EnterCombat(victim);
110 events.ScheduleEvent(EVENT_ERUPTION, 15000);
111 events.ScheduleEvent(EVENT_WRATH_OF_RAGNAROS, 30000);
112 events.ScheduleEvent(EVENT_HAND_OF_RAGNAROS, 25000);
113 events.ScheduleEvent(EVENT_LAVA_BURST, 10000);
114 events.ScheduleEvent(EVENT_ELEMENTAL_FIRE, 3000);
115 events.ScheduleEvent(EVENT_MAGMA_BLAST, 2000);
116 events.ScheduleEvent(EVENT_SUBMERGE, 180000);
117 }
118
119 void KilledUnit(Unit* /*victim*/) override
120 {
121 if (urand(0, 99) < 25)
122 Talk(SAY_KILL);
123 }
124
125 void UpdateAI(uint32 diff) override
126 {
127 if (_introState != 2)
128 {
129 if (!_introState)
130 {
131 me->HandleEmoteCommand(EMOTE_ONESHOT_EMERGE);
132 events.ScheduleEvent(EVENT_INTRO_1, 4000);
133 events.ScheduleEvent(EVENT_INTRO_2, 23000);
134 events.ScheduleEvent(EVENT_INTRO_3, 42000);
135 events.ScheduleEvent(EVENT_INTRO_4, 43000);
136 events.ScheduleEvent(EVENT_INTRO_5, 53000);
137 _introState = 1;
138 }
139
140 events.Update(diff);
141
142 while (uint32 eventId = events.ExecuteEvent())
143 {
144 switch (eventId)
145 {
146 case EVENT_INTRO_1:
147 Talk(SAY_ARRIVAL1_RAG);
148 break;
149 case EVENT_INTRO_2:
150 Talk(SAY_ARRIVAL3_RAG);
151 break;
152 case EVENT_INTRO_3:
153 me->HandleEmoteCommand(EMOTE_ONESHOT_ATTACK1H);
154 break;
155 case EVENT_INTRO_4:
156 Talk(SAY_ARRIVAL5_RAG);
157 if (Creature* executus = ObjectAccessor::GetCreature(*me, instance->GetGuidData(BOSS_MAJORDOMO_EXECUTUS)))
158 me->Kill(executus);
159 break;
160 case EVENT_INTRO_5:
161 me->SetReactState(REACT_AGGRESSIVE);
162 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
163 _introState = 2;
164 break;
165 default:
166 break;
167 }
168 }
169 }
170 else
171 {
172 if (_isBanished && ((_emergeTimer <= diff) || (instance->GetData(DATA_RAGNAROS_ADDS)) > 8))
173 {
174 //Become unbanished again
175 me->SetReactState(REACT_AGGRESSIVE);
176 me->setFaction(14);
177 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
178 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
179 me->HandleEmoteCommand(EMOTE_ONESHOT_EMERGE);
180 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
181 AttackStart(target);
182 instance->SetData(DATA_RAGNAROS_ADDS, 0);
183
184 //DoCast(me, SPELL_RAGEMERGE); //"phase spells" didnt worked correctly so Ive commented them and wrote solution witch doesnt need core support
185 _isBanished = false;
186 }
187 else if (_isBanished)
188 {
189 _emergeTimer -= diff;
190 //Do nothing while banished
191 return;
192 }
193
194 //Return since we have no target
195 if (!UpdateVictim())
196 return;
197
198 events.Update(diff);
199
200 while (uint32 eventId = events.ExecuteEvent())
201 {
202 switch (eventId)
203 {
204 case EVENT_ERUPTION:
205 DoCastVictim(SPELL_ERRUPTION);
206 events.ScheduleEvent(EVENT_ERUPTION, urand(20000, 45000));
207 break;
208 case EVENT_WRATH_OF_RAGNAROS:
209 DoCastVictim(SPELL_WRATH_OF_RAGNAROS);
210 if (urand(0, 1))
211 Talk(SAY_WRATH);
212 events.ScheduleEvent(EVENT_WRATH_OF_RAGNAROS, 25000);
213 break;
214 case EVENT_HAND_OF_RAGNAROS:
215 DoCast(me, SPELL_HAND_OF_RAGNAROS);
216 if (urand(0, 1))
217 Talk(SAY_HAND);
218 events.ScheduleEvent(EVENT_HAND_OF_RAGNAROS, 20000);
219 break;
220 case EVENT_LAVA_BURST:
221 DoCastVictim(SPELL_LAVA_BURST);
222 events.ScheduleEvent(EVENT_LAVA_BURST, 10000);
223 break;
224 case EVENT_ELEMENTAL_FIRE:
225 DoCastVictim(SPELL_ELEMENTAL_FIRE);
226 events.ScheduleEvent(EVENT_ELEMENTAL_FIRE, urand(10000, 14000));
227 break;
228 case EVENT_MAGMA_BLAST:
229 if (!me->IsWithinMeleeRange(me->GetVictim()))
230 {
231 DoCastVictim(SPELL_MAGMA_BLAST);
232 if (!_hasYelledMagmaBurst)
233 {
234 //Say our dialog
235 Talk(SAY_MAGMABURST);
236 _hasYelledMagmaBurst = true;
237 }
238 }
239 events.ScheduleEvent(EVENT_MAGMA_BLAST, 2500);
240 break;
241 case EVENT_SUBMERGE:
242 {
243 if (!_isBanished)
244 {
245 //Creature spawning and ragnaros becomming unattackable
246 //is not very well supported in the core //no it really isnt
247 //so added normaly spawning and banish workaround and attack again after 90 secs.
248 me->AttackStop();
249 DoResetThreat();
250 me->SetReactState(REACT_PASSIVE);
251 me->InterruptNonMeleeSpells(false);
252 //Root self
253 //DoCast(me, 23973);
254 me->setFaction(35);
255 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
256 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_SUBMERGED);
257 me->HandleEmoteCommand(EMOTE_ONESHOT_SUBMERGE);
258 instance->SetData(DATA_RAGNAROS_ADDS, 0);
259
260 if (!_hasSubmergedOnce)
261 {
262 Talk(SAY_REINFORCEMENTS1);
263
264 // summon 8 elementals
265 for (uint8 i = 0; i < 8; ++i)
266 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
267 if (Creature* summoned = me->SummonCreature(12143, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0.0f, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 900000))
268 summoned->AI()->AttackStart(target);
269
270 _hasSubmergedOnce = true;
271 _isBanished = true;
272 //DoCast(me, SPELL_RAGSUBMERGE);
273 _emergeTimer = 90000;
274
275 }
276 else
277 {
278 Talk(SAY_REINFORCEMENTS2);
279
280 for (uint8 i = 0; i < 8; ++i)
281 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
282 if (Creature* summoned = me->SummonCreature(12143, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0.0f, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 900000))
283 summoned->AI()->AttackStart(target);
284
285 _isBanished = true;
286 //DoCast(me, SPELL_RAGSUBMERGE);
287 _emergeTimer = 90000;
288 }
289 }
290 events.ScheduleEvent(EVENT_SUBMERGE, 180000);
291 break;
292 }
293 default:
294 break;
295 }
296 }
297
298 DoMeleeAttackIfReady();
299 }
300 }
301
302 private:
303 uint32 _emergeTimer;
304 uint8 _introState;
305 bool _hasYelledMagmaBurst;
306 bool _hasSubmergedOnce;
307 bool _isBanished;
308 };

312 return GetInstanceAI<boss_ragnarosAI>(creature);
313 }

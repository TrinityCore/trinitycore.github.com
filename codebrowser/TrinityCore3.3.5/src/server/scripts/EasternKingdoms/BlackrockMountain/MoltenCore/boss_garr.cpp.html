31enum Spells
32{
33 // Garr
34 SPELL_ANTIMAGIC_PULSE = 19492,
35 SPELL_MAGMA_SHACKLES = 19496,
36 SPELL_ENRAGE = 19516,
37
38 // Adds
39 SPELL_ERUPTION = 19497,
40 SPELL_IMMOLATE = 20294,
41};
42
43enum Events
44{
45 EVENT_ANTIMAGIC_PULSE = 1,
46 EVENT_MAGMA_SHACKLES = 2,
47};
48
49class boss_garr : public CreatureScript
50{
51 public:
52 boss_garr() : CreatureScript("boss_garr") { }
53
54 struct boss_garrAI : public BossAI
55 {
56 boss_garrAI(Creature* creature) : BossAI(creature, BOSS_GARR)
57 {
58 }
59
60 void EnterCombat(Unit* victim) override
61 {
62 BossAI::EnterCombat(victim);
63 events.ScheduleEvent(EVENT_ANTIMAGIC_PULSE, 25000);
64 events.ScheduleEvent(EVENT_MAGMA_SHACKLES, 15000);
65 }
66
67 void UpdateAI(uint32 diff) override
68 {
69 if (!UpdateVictim())
70 return;
71
72 events.Update(diff);
73
74 if (me->HasUnitState(UNIT_STATE_CASTING))
75 return;
76
77 while (uint32 eventId = events.ExecuteEvent())
78 {
79 switch (eventId)
80 {
81 case EVENT_ANTIMAGIC_PULSE:
82 DoCast(me, SPELL_ANTIMAGIC_PULSE);
83 events.ScheduleEvent(EVENT_ANTIMAGIC_PULSE, urand(10000, 15000));
84 break;
85 case EVENT_MAGMA_SHACKLES:
86 DoCast(me, SPELL_MAGMA_SHACKLES);
87 events.ScheduleEvent(EVENT_MAGMA_SHACKLES, urand(8000, 12000));
88 break;
89 default:
90 break;
91 }
92 }
93
94 DoMeleeAttackIfReady();
95 }
96 };
97
98 CreatureAI* GetAI(Creature* creature) const override
99 {
100 return new boss_garrAI(creature);
101 }
102};

104class npc_firesworn : public CreatureScript
105{
106 public:
107 npc_firesworn() : CreatureScript("npc_firesworn") { }
108
109 struct npc_fireswornAI : public ScriptedAI
110 {
111 npc_fireswornAI(Creature* creature) : ScriptedAI(creature)
112 {
113 Initialize();
114 }
115
116 void Initialize()
117 {
118 immolateTimer = 4000; //These times are probably wrong
119 }
120
121 uint32 immolateTimer;
122
123 void Reset() override
124 {
125 Initialize();
126 }
127
128 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
129 {
130 uint32 const health10pct = me->CountPctFromMaxHealth(10);
131 uint32 health = me->GetHealth();
132 if (int32(health) - int32(damage) < int32(health10pct))
133 {
134 damage = 0;
135 DoCastVictim(SPELL_ERUPTION);
136 me->DespawnOrUnsummon();
137 }
138 }
139
140 void UpdateAI(uint32 diff) override
141 {
142 if (!UpdateVictim())
143 return;
144
145 if (immolateTimer <= diff)
146 {
147 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
148 DoCast(target, SPELL_IMMOLATE);
149 immolateTimer = urand(5000, 10000);
150 }
151 else
152 immolateTimer -= diff;
153
154 DoMeleeAttackIfReady();
155 }
156 };
157
158 CreatureAI* GetAI(Creature* creature) const override
159 {
160 return new npc_fireswornAI(creature);
161 }
162};

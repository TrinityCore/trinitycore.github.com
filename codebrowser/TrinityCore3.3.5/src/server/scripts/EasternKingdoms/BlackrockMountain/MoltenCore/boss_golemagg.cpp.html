55class boss_golemagg : public CreatureScript
56{
57 public:
58 boss_golemagg() : CreatureScript("boss_golemagg") { }
59
60 struct boss_golemaggAI : public BossAI
61 {
62 boss_golemaggAI(Creature* creature) : BossAI(creature, BOSS_GOLEMAGG_THE_INCINERATOR)
63 {
64 }
65
66 void Reset() override
67 {
68 BossAI::Reset();
69 DoCast(me, SPELL_MAGMASPLASH, true);
70 }
71
72 void EnterCombat(Unit* victim) override
73 {
74 BossAI::EnterCombat(victim);
75 events.ScheduleEvent(EVENT_PYROBLAST, 7000);
76 }
77
78 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
79 {
80 if (!HealthBelowPct(10) || me->HasAura(SPELL_ENRAGE))
81 return;
82
83 DoCast(me, SPELL_ENRAGE, true);
84 events.ScheduleEvent(EVENT_EARTHQUAKE, 3000);
85 }
86
87 void UpdateAI(uint32 diff) override
88 {
89 if (!UpdateVictim())
90 return;
91
92 events.Update(diff);
93
94 if (me->HasUnitState(UNIT_STATE_CASTING))
95 return;
96
97 while (uint32 eventId = events.ExecuteEvent())
98 {
99 switch (eventId)
100 {
101 case EVENT_PYROBLAST:
102 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
103 DoCast(target, SPELL_PYROBLAST);
104 events.ScheduleEvent(EVENT_PYROBLAST, 7000);
105 break;
106 case EVENT_EARTHQUAKE:
107 DoCastVictim(SPELL_EARTHQUAKE);
108 events.ScheduleEvent(EVENT_EARTHQUAKE, 3000);
109 break;
110 default:
111 break;
112 }
113 }
114
115 DoMeleeAttackIfReady();
116 }
117 };
118
119 CreatureAI* GetAI(Creature* creature) const override
120 {
121 return new boss_golemaggAI(creature);
122 }
123};

125class npc_core_rager : public CreatureScript
126{
127 public:
128 npc_core_rager() : CreatureScript("npc_core_rager") { }
129
130 struct npc_core_ragerAI : public ScriptedAI
131 {
132 npc_core_ragerAI(Creature* creature) : ScriptedAI(creature)
133 {
134 Initialize();
135 instance = creature->GetInstanceScript();
136 }
137
138 void Initialize()
139 {
140 mangleTimer = 7 * IN_MILLISECONDS; // These times are probably wrong
141 }
142
143 void Reset() override
144 {
145 Initialize();
146 }
147
148 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
149 {
150 if (HealthAbovePct(50) || !instance)
151 return;
152
153 if (Creature* pGolemagg = ObjectAccessor::GetCreature(*me, instance->GetGuidData(BOSS_GOLEMAGG_THE_INCINERATOR)))
154 {
155 if (pGolemagg->IsAlive())
156 {
157 me->AddAura(SPELL_GOLEMAGG_TRUST, me);
158 Talk(EMOTE_LOWHP);
159 me->SetFullHealth();
160 }
161 }
162 }
163
164 void UpdateAI(uint32 diff) override
165 {
166 if (!UpdateVictim())
167 return;
168
169 // Mangle
170 if (mangleTimer <= diff)
171 {
172 DoCastVictim(SPELL_MANGLE);
173 mangleTimer = 10*IN_MILLISECONDS;
174 }
175 else
176 mangleTimer -= diff;
177
178 DoMeleeAttackIfReady();
179 }
180
181 private:
182 InstanceScript* instance;
183 uint32 mangleTimer;
184 };
185
186 CreatureAI* GetAI(Creature* creature) const override
187 {
188 return GetInstanceAI<npc_core_ragerAI>(creature);
189 }
190};

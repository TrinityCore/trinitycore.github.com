29enum CorporalKeeshan
30{
31 QUEST_MISSING_IN_ACTION = 219,
32 SAY_CORPORAL_1 = 0,
33 SAY_CORPORAL_2 = 1,
34 SAY_CORPORAL_3 = 2,
35 SAY_CORPORAL_4 = 3,
36 SAY_CORPORAL_5 = 4,
37 SPELL_MOCKING_BLOW = 21008,
38 SPELL_SHIELD_BASH = 11972
39};
40
41class npc_corporal_keeshan : public CreatureScript
42{
43public:
44 npc_corporal_keeshan() : CreatureScript("npc_corporal_keeshan") { }
45
46 struct npc_corporal_keeshanAI : public npc_escortAI
47 {
48 npc_corporal_keeshanAI(Creature* creature) : npc_escortAI(creature)
49 {
50 Initialize();
51 }
52
53 void Initialize()
54 {
55 timer = 0;
56 phase = 0;
57 mockingBlowTimer = 5000;
58 shieldBashTimer = 8000;
59 }
60
61 void Reset() override
62 {
63 Initialize();
64 }
65
66 void sQuestAccept(Player* player, Quest const* quest) override
67 {
68 if (quest->GetQuestId() == QUEST_MISSING_IN_ACTION)
69 {
70 Talk(SAY_CORPORAL_1, player);
71 me->setFaction(250);
72 npc_escortAI::Start(true, false, player->GetGUID(), quest);
73 }
74 }
75
76 void WaypointReached(uint32 waypointId) override
77 {
78 Player* player = GetPlayerForEscort();
79 if (!player)
80 return;
81
82 if (waypointId >= 65)
83 me->SetWalk(false);
84
85 switch (waypointId)
86 {
87 case 39:
88 SetEscortPaused(true);
89 timer = 2000;
90 phase = 1;
91 break;
92 case 65:
93 me->SetWalk(false);
94 break;
95 case 115:
96 player->GroupEventHappens(QUEST_MISSING_IN_ACTION, me);
97 timer = 2000;
98 phase = 4;
99 break;
100 }
101 }
102
103 void UpdateAI(uint32 diff) override
104 {
105 if (HasEscortState(STATE_ESCORT_NONE))
106 return;
107
108 npc_escortAI::UpdateAI(diff);
109
110 if (phase)
111 {
112 if (timer <= diff)
113 {
114 switch (phase)
115 {
116 case 1:
117 me->SetStandState(UNIT_STAND_STATE_SIT);
118 timer = 1000;
119 phase = 2;
120 break;
121 case 2:
122 Talk(SAY_CORPORAL_2);
123 timer = 15000;
124 phase = 3;
125 break;
126 case 3:
127 Talk(SAY_CORPORAL_3);
128 me->SetStandState(UNIT_STAND_STATE_STAND);
129 SetEscortPaused(false);
130 timer = 0;
131 phase = 0;
132 break;
133 case 4:
134 Talk(SAY_CORPORAL_4);
135 timer = 2500;
136 phase = 5;
137 break;
138 case 5:
139 Talk(SAY_CORPORAL_5);
140 timer = 0;
141 phase = 0;
142 break;
143 }
144 } else timer -= diff;
145 }
146
147 if (!UpdateVictim())
148 return;
149
150 if (mockingBlowTimer <= diff)
151 {
152 DoCastVictim(SPELL_MOCKING_BLOW);
153 mockingBlowTimer = 5000;
154 } else mockingBlowTimer -= diff;
155
156 if (shieldBashTimer <= diff)
157 {
158 DoCastVictim(SPELL_MOCKING_BLOW);
159 shieldBashTimer = 8000;
160 } else shieldBashTimer -= diff;
161
162 DoMeleeAttackIfReady();
163 }
164
165 private:
166 uint32 phase;
167 uint32 timer;
168 uint32 mockingBlowTimer;
169 uint32 shieldBashTimer;
170 };
171
172 CreatureAI* GetAI(Creature* creature) const override
173 {
174 return new npc_corporal_keeshanAI(creature);
175 }
176};
177
178void AddSC_redridge_mountains()
179{
180 new npc_corporal_keeshan();
181}

48 struct instance_sunwell_plateau_InstanceMapScript : public InstanceScript
49 {
50 instance_sunwell_plateau_InstanceMapScript(Map* map) : InstanceScript(map)
51 {
52 SetHeaders(DataHeader);
53 SetBossNumber(EncounterCount);
54 LoadDoorData(doorData);
55
56 SpectralPlayers = 0;
57 SpectralRealmTimer = 5000;
58 }
59
60 Player const* GetPlayerInMap() const
61 {
62 Map::PlayerList const& players = instance->GetPlayers();
63
64 if (!players.isEmpty())
65 {
66 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
67 {
68 Player* player = itr->GetSource();
69 if (player && !player->HasAura(45839))
70 return player;
71 }
72 }
73 else
74 TC_LOG_DEBUG("scripts", "Instance Sunwell Plateau: GetPlayerInMap, but PlayerList is empty!");
75
76 return NULL;
77 }

127 void OnGameObjectCreate(GameObject* go) override
128 {
129 switch (go->GetEntry())
130 {
131 case GO_FORCE_FIELD:
132 case GO_BOSS_COLLISION_1:
133 case GO_BOSS_COLLISION_2:
134 case GO_FIRE_BARRIER:
135 case GO_MURUS_GATE_1:
136 case GO_MURUS_GATE_2:
137 AddDoor(go, true);
138 break;
139 default:
140 break;
141 }
142 }
143
144 void OnGameObjectRemove(GameObject* go) override
145 {
146 switch (go->GetEntry())
147 {
148 case GO_FIRE_BARRIER:
149 case GO_MURUS_GATE_1:
150 case GO_MURUS_GATE_2:
151 case GO_BOSS_COLLISION_1:
152 case GO_BOSS_COLLISION_2:
153 case GO_FORCE_FIELD:
154 AddDoor(go, false);
155 break;
156 default:
157 break;
158 }
159 }

161 ObjectGuid GetGuidData(uint32 id) const override
162 {
163 switch (id)
164 {
165 case DATA_KALECGOS_DRAGON:
166 return KalecgosDragonGUID;
167 case DATA_KALECGOS_HUMAN:
168 return KalecgosHumanGUID;
169 case DATA_SATHROVARR:
170 return SathrovarrGUID;
171 case DATA_BRUTALLUS:
172 return BrutallusGUID;
173 case DATA_MADRIGOSA:
174 return MadrigosaGUID;
175 case DATA_FELMYST:
176 return FelmystGUID;
177 case DATA_ALYTHESS:
178 return AlythessGUID;
179 case DATA_SACROLASH:
180 return SacrolashGUID;
181 case DATA_MURU:
182 return MuruGUID;
183 case DATA_KILJAEDEN:
184 return KilJaedenGUID;
185 case DATA_KILJAEDEN_CONTROLLER:
186 return KilJaedenControllerGUID;
187 case DATA_ANVEENA:
188 return AnveenaGUID;
189 case DATA_KALECGOS_KJ:
190 return KalecgosKjGUID;
191 case DATA_PLAYER_GUID:
192 {
193 Player const* target = GetPlayerInMap();
194 return target ? target->GetGUID() : ObjectGuid::Empty;
195 }
196 default:
197 break;
198 }
199 return ObjectGuid::Empty;
200 }

195 void UpdateAI(uint32 diff) override
196 {
197 if (TalkTimer)
198 {
199 if (!TalkSequence)
200 {
201 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE + UNIT_FLAG_NOT_SELECTABLE);
202 me->InterruptNonMeleeSpells(true);
203 me->RemoveAllAuras();
204 me->DeleteThreatList();
205 me->CombatStop();
206 ++TalkSequence;
207 }
208 if (TalkTimer <= diff)
209 {
210 if (isFriendly)
211 GoodEnding();
212 else
213 BadEnding();
214 ++TalkSequence;
215 } else TalkTimer -= diff;
216 }
217 else
218 {
219 if (bJustReset)
220 {
221 if (ResetTimer <= diff)
222 {
223 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
224 me->SetDisableGravity(false);
225 me->SetVisible(true);
226 me->SetStandState(UNIT_STAND_STATE_SLEEP);
227 ResetTimer = 10000;
228 bJustReset = false;
229 } else ResetTimer -= diff;
230 return;
231 }
232
233 if (!UpdateVictim())
234 return;
235
236 if (CheckTimer <= diff)
237 {
238 if (me->GetDistance(CENTER_X, CENTER_Y, DRAGON_REALM_Z) >= 75)
239 {
240 EnterEvadeMode(EVADE_REASON_BOUNDARY);
241 return;
242 }
243 if (HealthBelowPct(10) && !isEnraged)
244 {
245 if (Creature* Sath = ObjectAccessor::GetCreature(*me, SathGUID))
246 Sath->AI()->DoAction(DO_ENRAGE);
247 DoAction(DO_ENRAGE);
248 }
249 if (!isBanished && HealthBelowPct(1))
250 {
251 if (Creature* Sath = ObjectAccessor::GetCreature(*me, SathGUID))
252 {
253 if (Sath->HasAura(SPELL_BANISH))
254 {
255 Sath->DealDamage(Sath, Sath->GetHealth());
256 return;
257 }
258 else
259 DoAction(DO_BANISH);
260 }
261 else
262 {
263 TC_LOG_ERROR("scripts", "Didn't find Shathrowar. Kalecgos event reseted.");
264 EnterEvadeMode(EVADE_REASON_OTHER);
265 return;
266 }
267 }
268 CheckTimer = 1000;
269 } else CheckTimer -= diff;
270
271 if (ArcaneBuffetTimer <= diff)
272 {
273 DoCastAOE(SPELL_ARCANE_BUFFET);
274 ArcaneBuffetTimer = 8000;
275 } else ArcaneBuffetTimer -= diff;
276
277 if (FrostBreathTimer <= diff)
278 {
279 DoCastAOE(SPELL_FROST_BREATH);
280 FrostBreathTimer = 15000;
281 } else FrostBreathTimer -= diff;
282
283 if (TailLashTimer <= diff)
284 {
285 DoCastAOE(SPELL_TAIL_LASH);
286 TailLashTimer = 15000;
287 } else TailLashTimer -= diff;
288
289 if (WildMagicTimer <= diff)
290 {
291 DoCastAOE(WildMagic[rand32() % 6]);
292 WildMagicTimer = 20000;
293 } else WildMagicTimer -= diff;
294
295 if (SpectralBlastTimer <= diff)
296 {
297 ThreatContainer::StorageType const& m_threatlist = me->getThreatManager().getThreatList();
298 std::list<Unit*> targetList;
299 for (ThreatContainer::StorageType::const_iterator itr = m_threatlist.begin(); itr!= m_threatlist.end(); ++itr)
300 {
301 Unit* target = (*itr)->getTarget();
302 if (target
303 && target->GetTypeId() == TYPEID_PLAYER
304 && (!target->GetVictim() || target->GetGUID() != me->EnsureVictim()->GetGUID())
305 && target->GetPositionZ() > me->GetPositionZ() - 5
306 && !target->HasAura(AURA_SPECTRAL_EXHAUSTION))
307 {
308 targetList.push_back(target);
309 }
310 }
311 if (targetList.empty())
312 {
313 SpectralBlastTimer = 1000;
314 return;
315 }
316
317 std::list<Unit*>::const_iterator i = targetList.begin();
318 advance(i, rand32() % targetList.size());
319 if ((*i))
320 {
321 (*i)->CastSpell((*i), SPELL_SPECTRAL_BLAST, true);
322 SpectralBlastTimer = 20000 + rand32() % 5000;
323 } else SpectralBlastTimer = 1000;
324 } else SpectralBlastTimer -= diff;
325
326 DoMeleeAttackIfReady();
327 }

718 void UpdateAI(uint32 diff) override
719 {
720 if (!me->HasAura(AURA_SPECTRAL_INVISIBILITY))
721 me->CastSpell(me, AURA_SPECTRAL_INVISIBILITY, true);
722
723 if (!UpdateVictim())
724 return;
725
726 if (CheckTimer <= diff)
727 {
728 Creature* Kalec = ObjectAccessor::GetCreature(*me, KalecGUID);
729 if (!Kalec || !Kalec->IsAlive())
730 {
731 if (Creature* Kalecgos = ObjectAccessor::GetCreature(*me, KalecgosGUID))
732 Kalecgos->AI()->EnterEvadeMode();
733 return;
734 }
735
736 if (HealthBelowPct(10) && !isEnraged)
737 {
738 if (Creature* Kalecgos = ObjectAccessor::GetCreature(*me, KalecgosGUID))
739 Kalecgos->AI()->DoAction(DO_ENRAGE);
740 DoAction(DO_ENRAGE);
741 }
742
743 Creature* Kalecgos = ObjectAccessor::GetCreature(*me, KalecgosGUID);
744 if (Kalecgos && !Kalecgos->IsInCombat())
745 {
746 EnterEvadeMode();
747 return;
748 }
749
750 if (!isBanished && HealthBelowPct(1))
751 {
752 if (Kalecgos)
753 {
754 if (Kalecgos->HasAura(SPELL_BANISH))
755 {
756 me->DealDamage(me, me->GetHealth());
757 return;
758 }
759 DoAction(DO_BANISH);
760 }
761 else
762 {
763 me->TextEmote(EMOTE_UNABLE_TO_FIND);
764 EnterEvadeMode();
765 return;
766 }
767 }
768 CheckTimer = 1000;
769 } else CheckTimer -= diff;
770
771 if (ResetThreat <= diff)
772 {
773 ThreatContainer::StorageType threatlist = me->getThreatManager().getThreatList();
774 for (ThreatContainer::StorageType::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
775 {
776 if (Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid()))
777 if (unit->GetPositionZ() > me->GetPositionZ() + 5)
778 me->getThreatManager().modifyThreatPercent(unit, -100);
779 }
780 ResetThreat = 1000;
781 } else ResetThreat -= diff;
782
783 if (ShadowBoltTimer <= diff)
784 {
785 if (!(rand32() % 5))
786 Talk(SAY_SATH_SPELL1);
787 DoCast(me, SPELL_SHADOW_BOLT);
788 ShadowBoltTimer = 7000 + (rand32() % 3000);
789 } else ShadowBoltTimer -= diff;
790
791 if (AgonyCurseTimer <= diff)
792 {
793 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1);
794 if (!target)
795 target = me->GetVictim();
796 DoCast(target, SPELL_AGONY_CURSE);
797 AgonyCurseTimer = 20000;
798 } else AgonyCurseTimer -= diff;
799

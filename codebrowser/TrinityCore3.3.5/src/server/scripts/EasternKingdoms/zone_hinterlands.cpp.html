43 SAY_OOX_AGGRO = 1,

45 SAY_OOX_AMBUSH_REPLY = 3,
46 SAY_OOX_END = 4,
47 QUEST_RESQUE_OOX_09 = 836,
48 NPC_MARAUDING_OWL = 7808,

54class npc_oox09hl : public CreatureScript
55{
56public:
57 npc_oox09hl() : CreatureScript("npc_oox09hl") { }
58
59 struct npc_oox09hlAI : public npc_escortAI
60 {
61 npc_oox09hlAI(Creature* creature) : npc_escortAI(creature) { }
62
63 void Reset() override { }
64
65 void EnterCombat(Unit* who) override
66 {
67 if (who->GetEntry() == NPC_MARAUDING_OWL || who->GetEntry() == NPC_VILE_AMBUSHER)
68 return;
69
70 Talk(SAY_OOX_AGGRO);
71 }
72
73 void JustSummoned(Creature* summoned) override
74 {
75 summoned->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
76 }
77
78 void sQuestAccept(Player* player, Quest const* quest) override
79 {
80 if (quest->GetQuestId() == QUEST_RESQUE_OOX_09)
81 {
82 me->SetStandState(UNIT_STAND_STATE_STAND);
83 me->setFaction(player->GetTeam() == ALLIANCE ? FACTION_ESCORTEE_A : FACTION_ESCORTEE_H);
84 Talk(SAY_OOX_START, player);
85 npc_escortAI::Start(false, false, player->GetGUID(), quest);
86 }
87 }
88
89 void WaypointReached(uint32 waypointId) override
90 {
91 switch (waypointId)
92 {
93 case 26:
94 Talk(SAY_OOX_AMBUSH);
95 break;
96 case 43:
97 Talk(SAY_OOX_AMBUSH);
98 break;
99 case 64:
100 Talk(SAY_OOX_END);
101 if (Player* player = GetPlayerForEscort())
102 player->GroupEventHappens(QUEST_RESQUE_OOX_09, me);
103 break;
104 }
105 }
106
107 void WaypointStart(uint32 pointId) override
108 {
109 switch (pointId)
110 {
111 case 27:
112 for (uint8 i = 0; i < 3; ++i)
113 {
114 const Position src = {147.927444f, -3851.513428f, 130.893f, 0};
115 Position dst = me->GetRandomPoint(src, 7.0f);
116 DoSummon(NPC_MARAUDING_OWL, dst, 25000, TEMPSUMMON_CORPSE_TIMED_DESPAWN);
117 }
118 break;
119 case 44:
120 for (uint8 i = 0; i < 3; ++i)
121 {
122 const Position src = {-141.151581f, -4291.213867f, 120.130f, 0};
123 Position dst = me->GetRandomPoint(src, 7.0f);
124 me->SummonCreature(NPC_VILE_AMBUSHER, dst, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 25000);
125 }
126 break;
127 }
128 }
129 };
130
131 CreatureAI* GetAI(Creature* creature) const override
132 {
133 return new npc_oox09hlAI(creature);
134 }

167class npc_rinji : public CreatureScript
168{
169public:
170 npc_rinji() : CreatureScript("npc_rinji") { }
171
172 struct npc_rinjiAI : public npc_escortAI
173 {
174 npc_rinjiAI(Creature* creature) : npc_escortAI(creature)
175 {
176 Initialize();
177 _IsByOutrunner = false;
178 spawnId = 0;
179 }
180
181 void Initialize()
182 {
183 postEventCount = 0;
184 postEventTimer = 3000;
185 }
186
187 void Reset() override
188 {
189 Initialize();
190 }
191
192 void JustRespawned() override
193 {
194 _IsByOutrunner = false;
195 spawnId = 0;
196
197 npc_escortAI::JustRespawned();
198 }
199
200 void EnterCombat(Unit* who) override
201 {
202 if (HasEscortState(STATE_ESCORT_ESCORTING))
203 {
204 if (who->GetEntry() == NPC_OUTRUNNER && !_IsByOutrunner)
205 {
206 if (Creature* talker = who->ToCreature())
207 talker->AI()->Talk(SAY_RIN_BY_OUTRUNNER);
208 _IsByOutrunner = true;
209 }
210
211 if (rand32() % 4)
212 return;
213
214 //only if attacked and escorter is not in combat?
215 Talk(SAY_RIN_HELP);
216 }
217 }
218
219 void DoSpawnAmbush(bool _first)
220 {
221 if (!_first)
222 spawnId = 1;
223
224 me->SummonCreature(NPC_RANGER, AmbushSpawn[spawnId], TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 60000);
225
226 for (int i = 0; i < 2; ++i)
227 {
228 me->SummonCreature(NPC_OUTRUNNER, AmbushSpawn[spawnId], TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 60000);
229 }
230 }
231
232 void JustSummoned(Creature* summoned) override
233 {
234 summoned->SetWalk(false);
235 summoned->GetMotionMaster()->MovePoint(0, AmbushMoveTo[spawnId]);
236 }
237
238 void sQuestAccept(Player* player, Quest const* quest) override
239 {
240 if (quest->GetQuestId() == QUEST_RINJI_TRAPPED)
241 {
242 if (GameObject* go = me->FindNearestGameObject(GO_RINJI_CAGE, INTERACTION_DISTANCE))
243 go->UseDoorOrButton();
244
245 npc_escortAI::Start(false, false, player->GetGUID(), quest);
246 }
247 }
248
249 void WaypointReached(uint32 waypointId) override
250 {
251 Player* player = GetPlayerForEscort();
252 if (!player)
253 return;
254
255 switch (waypointId)
256 {
257 case 1:
258 Talk(SAY_RIN_FREE, player);
259 break;
260 case 7:
261 DoSpawnAmbush(true);
262 break;
263 case 13:
264 DoSpawnAmbush(false);
265 break;
266 case 17:
267 Talk(SAY_RIN_COMPLETE, player);
268 player->GroupEventHappens(QUEST_RINJI_TRAPPED, me);
269 SetRun();
270 postEventCount = 1;
271 break;
272 }
273 }
274
275 void UpdateEscortAI(uint32 diff) override
276 {
277 //Check if we have a current target
278 if (!UpdateVictim())
279 {
280 if (HasEscortState(STATE_ESCORT_ESCORTING) && postEventCount)
281 {
282 if (postEventTimer <= diff)
283 {
284 postEventTimer = 3000;
285
286 if (Player* player = GetPlayerForEscort())
287 {
288 switch (postEventCount)
289 {
290 case 1:
291 Talk(SAY_RIN_PROGRESS_1, player);
292 ++postEventCount;
293 break;
294 case 2:
295 Talk(SAY_RIN_PROGRESS_2, player);
296 postEventCount = 0;
297 break;
298 }
299 }
300 else
301 {
302 me->DespawnOrUnsummon();
303 return;
304 }
305 }
306 else
307 postEventTimer -= diff;
308 }
309 return;
310 }
311 DoMeleeAttackIfReady();
312 }
313
314 private:
315 uint32 postEventCount;
316 uint32 postEventTimer;
317 int spawnId;
318 bool _IsByOutrunner;
319 };
320
321 CreatureAI* GetAI(Creature* creature) const override
322 {
323 return new npc_rinjiAI(creature);
324 }
325};

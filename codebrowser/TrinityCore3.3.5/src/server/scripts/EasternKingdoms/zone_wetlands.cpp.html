40enum TapokeSlim
41{
42 QUEST_MISSING_DIPLO_PT11 = 1249,
43 FACTION_ENEMY = 168,
44 SPELL_STEALTH = 1785,
45 SPELL_CALL_FRIENDS = 16457, //summons 1x friend
46 NPC_SLIMS_FRIEND = 4971,
47 NPC_TAPOKE_SLIM_JAHN = 4962
48};
49
50class npc_tapoke_slim_jahn : public CreatureScript
51{
52public:
53 npc_tapoke_slim_jahn() : CreatureScript("npc_tapoke_slim_jahn") { }
54
55 CreatureAI* GetAI(Creature* creature) const override
56 {
57 return new npc_tapoke_slim_jahnAI(creature);
58 }
59
60 struct npc_tapoke_slim_jahnAI : public npc_escortAI
61 {
62 npc_tapoke_slim_jahnAI(Creature* creature) : npc_escortAI(creature)
63 {
64 Initialize();
65 }
66
67 void Initialize()
68 {
69 IsFriendSummoned = false;
70 }
71
72 bool IsFriendSummoned;
73
74 void Reset() override
75 {
76 if (!HasEscortState(STATE_ESCORT_ESCORTING))
77 Initialize();
78 }
79
80 void WaypointReached(uint32 waypointId) override
81 {
82 switch (waypointId)
83 {
84 case 2:
85 if (me->HasStealthAura())
86 me->RemoveAurasByType(SPELL_AURA_MOD_STEALTH);
87 SetRun();
88 me->setFaction(FACTION_ENEMY);
89 break;
90 }
91 }
92
93 void EnterCombat(Unit* /*who*/) override
94 {
95 if (HasEscortState(STATE_ESCORT_ESCORTING) && !IsFriendSummoned && GetPlayerForEscort())
96 {
97 for (uint8 i = 0; i < 3; ++i)
98 DoCast(me, SPELL_CALL_FRIENDS, true);
99
100 IsFriendSummoned = true;
101 }
102 }
103
104 void JustSummoned(Creature* summoned) override
105 {
106 if (Player* player = GetPlayerForEscort())
107 summoned->AI()->AttackStart(player);
108 }
109
110 void AttackedBy(Unit* pAttacker) override
111 {
112 if (me->GetVictim())
113 return;
114
115 if (me->IsFriendlyTo(pAttacker))
116 return;
117
118 AttackStart(pAttacker);
119 }
120
121 void DamageTaken(Unit* /*pDoneBy*/, uint32& uiDamage) override
122 {
123 if (HealthBelowPct(20))
124 {
125 if (Player* player = GetPlayerForEscort())
126 {
127 player->GroupEventHappens(QUEST_MISSING_DIPLO_PT11, me);
128
129 uiDamage = 0;
130
131 me->RestoreFaction();
132 me->RemoveAllAuras();
133 me->DeleteThreatList();
134 me->CombatStop(true);
135
136 SetRun(false);
137 }
138 }
139 }
140 };
141};

147class npc_mikhail : public CreatureScript
148{
149public:
150 npc_mikhail() : CreatureScript("npc_mikhail") { }
151
152 bool OnQuestAccept(Player* player, Creature* creature, const Quest* quest) override
153 {
154 if (quest->GetQuestId() == QUEST_MISSING_DIPLO_PT11)
155 {
156 Creature* pSlim = creature->FindNearestCreature(NPC_TAPOKE_SLIM_JAHN, 25.0f);
157
158 if (!pSlim)
159 return false;
160
161 if (!pSlim->HasStealthAura())
162 pSlim->CastSpell(pSlim, SPELL_STEALTH, true);
163
164 if (npc_tapoke_slim_jahn::npc_tapoke_slim_jahnAI* pEscortAI = CAST_AI(npc_tapoke_slim_jahn::npc_tapoke_slim_jahnAI, pSlim->AI()))
165 pEscortAI->Start(false, false, player->GetGUID(), quest);
166 }
167 return false;
168 }
169};

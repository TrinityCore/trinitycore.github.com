94class npc_lady_sylvanas_windrunner : public CreatureScript

97 npc_lady_sylvanas_windrunner() : CreatureScript("npc_lady_sylvanas_windrunner") { }
98
99 bool OnQuestReward(Player* player, Creature* creature, const Quest *_Quest, uint32 /*slot*/) override
100 {
101 if (_Quest->GetQuestId() == QUEST_JOURNEY_TO_UNDERCITY)
102 creature->AI()->SetGUID(player->GetGUID(), GUID_EVENT_INVOKER);
103
104 return true;
105 }
106
107 struct npc_lady_sylvanas_windrunnerAI : public ScriptedAI
108 {
109 npc_lady_sylvanas_windrunnerAI(Creature* creature) : ScriptedAI(creature)
110 {
111 Initialize();
112 }
113
114 void Initialize()
115 {
116 LamentEvent = false;
117 targetGUID.Clear();
118 playerGUID.Clear();
119 }
120
121 void Reset() override
122 {
123 Initialize();
124 _events.Reset();
125 }
126
127 void EnterCombat(Unit* /*who*/) override
128 {
129 _events.ScheduleEvent(EVENT_FADE, 30000);
130 _events.ScheduleEvent(EVENT_SUMMON_SKELETON, 20000);
131 _events.ScheduleEvent(EVENT_BLACK_ARROW, 15000);
132 _events.ScheduleEvent(EVENT_SHOOT, 8000);
133 _events.ScheduleEvent(EVENT_MULTI_SHOT, 10000);
134 }
135
136 void SetGUID(ObjectGuid guid, int32 type) override
137 {
138 if (type == GUID_EVENT_INVOKER)
139 {
140 Talk(EMOTE_LAMENT);
141 DoPlaySoundToSet(me, SOUND_CREDIT);
142 DoCast(me, SPELL_SYLVANAS_CAST, false);
143 playerGUID = guid;
144 LamentEvent = true;
145
146 for (uint8 i = 0; i < 4; ++i)
147 me->SummonCreature(NPC_HIGHBORNE_LAMENTER, HighborneLoc[i][0], HighborneLoc[i][1], HIGHBORNE_LOC_Y, HighborneLoc[i][2], TEMPSUMMON_TIMED_DESPAWN, 160000);
148
149 _events.ScheduleEvent(EVENT_LAMENT_OF_THE_HIGHBORN, 2000);
150 _events.ScheduleEvent(EVENT_SUNSORROW_WHISPER, 10000);
151 }
152 }
153
154 void JustSummoned(Creature* summoned) override
155 {
156 if (summoned->GetEntry() == NPC_HIGHBORNE_BUNNY)
157 {
158 if (Creature* target = ObjectAccessor::GetCreature(*summoned, targetGUID))
159 {
160 target->GetMotionMaster()->MoveJump(target->GetPositionX(), target->GetPositionY(), me->GetPositionZ() + 15.0f, me->GetOrientation(), 0);
161 target->SetPosition(target->GetPositionX(), target->GetPositionY(), me->GetPositionZ()+15.0f, 0.0f);
162 summoned->CastSpell(target, SPELL_RIBBON_OF_SOULS, false);
163 }
164
165 summoned->SetDisableGravity(true);
166 targetGUID = summoned->GetGUID();
167 }
168 }
169
170 void UpdateAI(uint32 diff) override
171 {
172 if (!UpdateVictim() && !LamentEvent)
173 return;
174
175 _events.Update(diff);
176
177 if (me->HasUnitState(UNIT_STATE_CASTING))
178 return;
179
180 while (uint32 eventId = _events.ExecuteEvent())
181 {
182 switch (eventId)
183 {
184 case EVENT_FADE:
185 DoCast(me, SPELL_FADE);
186 // add a blink to simulate a stealthed movement and reappearing elsewhere
187 DoCast(me, SPELL_FADE_BLINK);
188 // if the victim is out of melee range she cast multi shot
189 if (Unit* victim = me->GetVictim())
190 if (me->GetDistance(victim) > 10.0f)
191 DoCast(victim, SPELL_MULTI_SHOT);
192 _events.ScheduleEvent(EVENT_FADE, urand(30000, 35000));
193 break;
194 case EVENT_SUMMON_SKELETON:
195 DoCast(me, SPELL_SUMMON_SKELETON);
196 _events.ScheduleEvent(EVENT_SUMMON_SKELETON, urand(20000, 30000));
197 break;
198 case EVENT_BLACK_ARROW:
199 if (Unit* victim = me->GetVictim())
200 DoCast(victim, SPELL_BLACK_ARROW);
201 _events.ScheduleEvent(EVENT_BLACK_ARROW, urand(15000, 20000));
202 break;
203 case EVENT_SHOOT:
204 if (Unit* victim = me->GetVictim())
205 DoCast(victim, SPELL_SHOT);
206 _events.ScheduleEvent(EVENT_SHOOT, urand(8000, 10000));
207 break;
208 case EVENT_MULTI_SHOT:
209 if (Unit* victim = me->GetVictim())
210 DoCast(victim, SPELL_MULTI_SHOT);
211 _events.ScheduleEvent(EVENT_MULTI_SHOT, urand(10000, 13000));
212 break;
213 case EVENT_LAMENT_OF_THE_HIGHBORN:
214 if (!me->HasAura(SPELL_SYLVANAS_CAST))
215 {
216 Talk(SAY_LAMENT_END);
217 Talk(EMOTE_LAMENT_END);
218 LamentEvent = false;
219 me->HandleEmoteCommand(EMOTE_ONESHOT_KNEEL);
220 Reset();
221 }
222 else
223 {
224 DoSummon(NPC_HIGHBORNE_BUNNY, me, 10.0f, 3000, TEMPSUMMON_TIMED_DESPAWN);
225 _events.ScheduleEvent(EVENT_LAMENT_OF_THE_HIGHBORN, 2000);
226 }
227 break;
228 case EVENT_SUNSORROW_WHISPER:
229 if (Creature* ambassador = me->FindNearestCreature(NPC_AMBASSADOR_SUNSORROW, 20.0f))
230 if (Player* player = ObjectAccessor::GetPlayer(*me, playerGUID))
231 ambassador->AI()->Talk(SAY_SUNSORROW_WHISPER, player);
232 break;
233 default:
234 break;
235 }
236 }
237
238 DoMeleeAttackIfReady();
239 }
240
241 private:
242 EventMap _events;
243 bool LamentEvent;
244 ObjectGuid targetGUID;
245 ObjectGuid playerGUID;
246 };

248 CreatureAI* GetAI(Creature* creature) const override
249 {
250 return new npc_lady_sylvanas_windrunnerAI(creature);
251 }

268 struct npc_highborne_lamenterAI : public ScriptedAI
269 {
270 npc_highborne_lamenterAI(Creature* creature) : ScriptedAI(creature)
271 {
272 Initialize();
273 }
274
275 void Initialize()
276 {
277 EventMoveTimer = 10000;
278 EventCastTimer = 17500;
279 EventMove = true;
280 EventCast = true;
281 }
282
283 uint32 EventMoveTimer;
284 uint32 EventCastTimer;
285 bool EventMove;
286 bool EventCast;
287
288 void Reset() override
289 {
290 Initialize();
291 }
292
293 void EnterCombat(Unit* /*who*/) override { }
294
295 void UpdateAI(uint32 diff) override
296 {
297 if (EventMove)
298 {
299 if (EventMoveTimer <= diff)
300 {
301 me->SetDisableGravity(true);
302 me->MonsterMoveWithSpeed(me->GetPositionX(), me->GetPositionY(), HIGHBORNE_LOC_Y_NEW, me->GetDistance(me->GetPositionX(), me->GetPositionY(), HIGHBORNE_LOC_Y_NEW) / (5000 * 0.001f));
303 me->SetPosition(me->GetPositionX(), me->GetPositionY(), HIGHBORNE_LOC_Y_NEW, me->GetOrientation());
304 EventMove = false;
305 } else EventMoveTimer -= diff;
306 }
307 if (EventCast)
308 {
309 if (EventCastTimer <= diff)
310 {
311 DoCast(me, SPELL_HIGHBORNE_AURA);
312 EventCast = false;
313 } else EventCastTimer -= diff;
314 }
315 }
316 };
317};

248 struct boss_hex_lord_malacrassAI : public ScriptedAI
249 {
250 boss_hex_lord_malacrassAI(Creature* creature) : ScriptedAI(creature)
251 {
252 Initialize();
253 instance = creature->GetInstanceScript();
254 SelectAddEntry();
255 for (uint8 i = 0; i < 4; ++i)
256 AddGUID[i].Clear();
257 PlayerGUID.Clear();
258 PlayerClass = CLASS_NONE;
259 }
260
261 void Initialize()
262 {
263 SpiritBolts_Timer = 20000;
264 DrainPower_Timer = 60000;
265 SiphonSoul_Timer = 100000;
266 PlayerAbility_Timer = 99999;
267 CheckAddState_Timer = 5000;
268 ResetTimer = 5000;
269 }

271 InstanceScript* instance;

273 ObjectGuid AddGUID[4];

278 uint32 SpiritBolts_Timer;

282 uint32 CheckAddState_Timer;

287 void Reset() override
288 {
289 instance->SetData(DATA_HEXLORDEVENT, NOT_STARTED);
290
291 Initialize();
292
293 SpawnAdds();
294
295 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID, 46916);
296 me->SetByteValue(UNIT_FIELD_BYTES_2, 0, SHEATH_STATE_MELEE);

298
299 void EnterCombat(Unit* /*who*/) override
300 {
301 instance->SetData(DATA_HEXLORDEVENT, IN_PROGRESS);
302
303 DoZoneInCombat();
304 Talk(YELL_AGGRO);
305
306 for (uint8 i = 0; i < 4; ++i)
307 {
308 Creature* creature = ObjectAccessor::GetCreature(*me, AddGUID[i]);
309 if (creature && creature->IsAlive())
310 creature->AI()->AttackStart(me->GetVictim());
311 else
312 {
313 EnterEvadeMode();
314 break;
315 }
316 }
317 }
318
319 void KilledUnit(Unit* /*victim*/) override
320 {
321 switch (urand(0, 1))
322 {
323 case 0:
324 Talk(YELL_KILL_ONE);
325 break;
326 case 1:
327 Talk(YELL_KILL_TWO);
328 break;
329 }
330 }
331
332 void JustDied(Unit* /*killer*/) override
333 {
334 instance->SetData(DATA_HEXLORDEVENT, DONE);
335
336 Talk(YELL_DEATH);
337
338 for (uint8 i = 0; i < 4; ++i)
339 {
340 Unit* Temp = ObjectAccessor::GetUnit(*me, AddGUID[i]);
341 if (Temp && Temp->IsAlive())
342 Temp->DealDamage(Temp, Temp->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
343 }
344 }
345
346 void SelectAddEntry()
347 {
348 std::vector<uint32> AddList;
349
350 for (uint8 i = 0; i < 8; ++i)
351 AddList.push_back(AddEntryList[i]);
352
353 while (AddList.size() > 4)
354 AddList.erase(AddList.begin() + rand32() % AddList.size());
355
356 uint8 i = 0;
357 for (std::vector<uint32>::const_iterator itr = AddList.begin(); itr != AddList.end(); ++itr, ++i)
358 AddEntry[i] = *itr;
359 }
360
361 void SpawnAdds()
362 {
363 for (uint8 i = 0; i < 4; ++i)
364 {
365 Creature* creature = (ObjectAccessor::GetCreature((*me), AddGUID[i]));
366 if (!creature || !creature->IsAlive())
367 {
368 if (creature) creature->setDeathState(DEAD);
369 creature = me->SummonCreature(AddEntry[i], Pos_X[i], POS_Y, POS_Z, ORIENT, TEMPSUMMON_DEAD_DESPAWN, 0);
370 if (creature) AddGUID[i] = creature->GetGUID();
371 }
372 else
373 {
374 creature->AI()->EnterEvadeMode();
375 creature->SetPosition(Pos_X[i], POS_Y, POS_Z, ORIENT);
376 creature->StopMoving();
377 }
378 }
379 }
380
381 void UpdateAI(uint32 diff) override
382 {
383 if (!UpdateVictim())
384 return;
385
386 if (ResetTimer <= diff)
387 {
388 if (me->IsWithinDist3d(119.223f, 1035.45f, 29.4481f, 10))
389 {
390 EnterEvadeMode();
391 return;
392 }
393 ResetTimer = 5000;
394 } else ResetTimer -= diff;
395
396 if (CheckAddState_Timer <= diff)
397 {
398 for (uint8 i = 0; i < 4; ++i)
399 if (Creature* temp = ObjectAccessor::GetCreature(*me, AddGUID[i]))
400 if (temp->IsAlive() && !temp->GetVictim())
401 temp->AI()->AttackStart(me->GetVictim());
402
403 CheckAddState_Timer = 5000;
404 } else CheckAddState_Timer -= diff;
405
406 if (DrainPower_Timer <= diff)
407 {
408 DoCast(me, SPELL_DRAIN_POWER, true);
409 Talk(YELL_DRAIN_POWER);
410 DrainPower_Timer = urand(40000, 55000); // must cast in 60 sec, or buff/debuff will disappear
411 } else DrainPower_Timer -= diff;
412
413 if (SpiritBolts_Timer <= diff)
414 {
415 if (DrainPower_Timer < 12000) // channel 10 sec
416 SpiritBolts_Timer = 13000; // cast drain power first
417 else
418 {
419 DoCast(me, SPELL_SPIRIT_BOLTS, false);
420 Talk(YELL_SPIRIT_BOLTS);
421 SpiritBolts_Timer = 40000;
422 SiphonSoul_Timer = 10000; // ready to drain
423 PlayerAbility_Timer = 99999;
424 }
425 } else SpiritBolts_Timer -= diff;
426
427 if (SiphonSoul_Timer <= diff)
428 {
429 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 70, true);
430 Unit* trigger = DoSpawnCreature(NPC_TEMP_TRIGGER, 0, 0, 0, 0, TEMPSUMMON_TIMED_DESPAWN, 30000);
431 if (!target || !trigger)
432 {
433 EnterEvadeMode();
434 return;
435 }
436 else
437 {
438 trigger->SetDisplayId(11686);
439 trigger->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
440 trigger->CastSpell(target, SPELL_SIPHON_SOUL, true);
441 trigger->GetMotionMaster()->MoveChase(me);
442
443 //DoCast(target, SPELL_SIPHON_SOUL, true);
444 //me->SetUInt64Value(UNIT_FIELD_CHANNEL_OBJECT, target->GetGUID());
445 //me->SetUInt32Value(UNIT_CHANNEL_SPELL, SPELL_SIPHON_SOUL);
446
447 PlayerGUID = target->GetGUID();
448 PlayerAbility_Timer = urand(8000, 10000);
449 PlayerClass = target->getClass() - 1;
450
451 if (PlayerClass == CLASS_DRUID-1)
452 PlayerClass = CLASS_DRUID;
453 else if (PlayerClass == CLASS_PRIEST-1 && target->HasSpell(15473))
454 PlayerClass = CLASS_PRIEST; // shadow priest
455
456 SiphonSoul_Timer = 99999; // buff lasts 30 sec
457 }
458 } else SiphonSoul_Timer -= diff;
459
460 if (PlayerAbility_Timer <= diff)
461 {
462 //Unit* target = ObjectAccessor::GetUnit(*me, PlayerGUID);
463 //if (target && target->IsAlive())
464 //{
465 UseAbility();
466 PlayerAbility_Timer = urand(8000, 10000);
467 //}
468 } else PlayerAbility_Timer -= diff;
469
470 DoMeleeAttackIfReady();
471 }
472
473 void UseAbility()
474 {
475 uint8 random = urand(0, 2);
476 Unit* target = NULL;
477 switch (PlayerAbility[PlayerClass][random].target)
478 {
479 case ABILITY_TARGET_SELF:
480 target = me;
481 break;
482 case ABILITY_TARGET_VICTIM:
483 target = me->GetVictim();
484 break;
485 case ABILITY_TARGET_ENEMY:
486 default:
487 target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true);
488 break;
489 case ABILITY_TARGET_HEAL:
490 target = DoSelectLowestHpFriendly(50, 0);
491 break;
492 case ABILITY_TARGET_BUFF:
493 {
494 std::list<Creature*> templist = DoFindFriendlyMissingBuff(50, PlayerAbility[PlayerClass][random].spell);
495 if (!templist.empty())
496 target = *(templist.begin());
497 }
498 break;
499 }
500 if (target)
501 DoCast(target, PlayerAbility[PlayerClass][random].spell, false);
502 }
503 };

505 CreatureAI* GetAI(Creature* creature) const override
506 {
507 return GetInstanceAI<boss_hex_lord_malacrassAI>(creature);
508 }

160 void SummonHostage(uint8 num)
161 {
162 if (!QuestMinute)
163 return;
164
165 Map::PlayerList const &PlayerList = instance->GetPlayers();
166 if (PlayerList.isEmpty())
167 return;
168
169 Map::PlayerList::const_iterator i = PlayerList.begin();
170 if (Player* i_pl = i->GetSource())
171 {
172 if (Unit* Hostage = i_pl->SummonCreature(HostageInfo[num].npc, HostageInfo[num].x, HostageInfo[num].y, HostageInfo[num].z, HostageInfo[num].o, TEMPSUMMON_DEAD_DESPAWN, 0))
173 {
174 Hostage->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
175 Hostage->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
176 }
177 }
178 }

200 void Load(const char* load) override
201 {
202 if (!load)
203 return;
204
205 std::istringstream ss(load);
206 //TC_LOG_ERROR("scripts", "Zul'aman loaded, %s.", ss.str().c_str());
207 char dataHead; // S
208 uint16 data1, data2, data3;
209 ss >> dataHead >> data1 >> data2 >> data3;
210 //TC_LOG_ERROR("scripts", "Zul'aman loaded, %d %d %d.", data1, data2, data3);
211 if (dataHead == 'S')
212 {
213 BossKilled = data1;
214 ChestLooted = data2;
215 QuestMinute = data3;
216 } else TC_LOG_ERROR("scripts", "Zul'aman: corrupted save data.");
217 }

219 void SetData(uint32 type, uint32 data) override
220 {
221 switch (type)
222 {
223 case DATA_GONGEVENT:
224 m_auiEncounter[DATA_GONGEVENT] = data;
225 if (data == IN_PROGRESS)
226 SaveToDB();
227 else if (data == DONE)
228 QuestMinute = 21;
229 break;
230 case DATA_NALORAKKEVENT:
231 m_auiEncounter[DATA_NALORAKKEVENT] = data;
232 if (data == DONE)
233 {
234 if (QuestMinute)
235 {
236 QuestMinute += 15;
237 DoUpdateWorldState(WORLDSTATE_TIME_TO_SACRIFICE, QuestMinute);
238 }
239 SummonHostage(0);
240 }
241 break;
242 case DATA_AKILZONEVENT:
243 m_auiEncounter[DATA_AKILZONEVENT] = data;
244 HandleGameObject(AkilzonDoorGUID, data != IN_PROGRESS);
245 if (data == DONE)
246 {
247 if (QuestMinute)
248 {
249 QuestMinute += 10;
250 DoUpdateWorldState(WORLDSTATE_TIME_TO_SACRIFICE, QuestMinute);
251 }
252 SummonHostage(1);
253 }
254 break;
255 case DATA_JANALAIEVENT:
256 m_auiEncounter[DATA_JANALAIEVENT] = data;
257 if (data == DONE)
258 SummonHostage(2);
259 break;
260 case DATA_HALAZZIEVENT:
261 m_auiEncounter[DATA_HALAZZIEVENT] = data;
262 HandleGameObject(HalazziDoorGUID, data != IN_PROGRESS);
263 if (data == DONE) SummonHostage(3);
264 break;
265 case DATA_HEXLORDEVENT:
266 m_auiEncounter[DATA_HEXLORDEVENT] = data;
267 if (data == IN_PROGRESS)
268 HandleGameObject(HexLordGateGUID, false);
269 else if (data == NOT_STARTED)
270 CheckInstanceStatus();
271 break;
272 case DATA_ZULJINEVENT:
273 m_auiEncounter[DATA_ZULJINEVENT] = data;
274 HandleGameObject(ZulJinDoorGUID, data != IN_PROGRESS);
275 break;
276 case DATA_CHESTLOOTED:
277 ++ChestLooted;
278 SaveToDB();
279 break;
280 case TYPE_RAND_VENDOR_1:
281 RandVendor[0] = data;
282 break;
283 case TYPE_RAND_VENDOR_2:
284 RandVendor[1] = data;
285 break;
286 }
287
288 if (data == DONE)
289 {
290 ++BossKilled;
291 if (QuestMinute && BossKilled >= DATA_HALAZZIEVENT)
292 {
293 QuestMinute = 0;
294 DoUpdateWorldState(WORLDSTATE_SHOW_TIMER, 0);
295 }
296 CheckInstanceStatus();
297 SaveToDB();
298 }
299 }

306 case DATA_NALORAKKEVENT: return m_auiEncounter[DATA_NALORAKKEVENT];
307 case DATA_AKILZONEVENT: return m_auiEncounter[DATA_AKILZONEVENT];
308 case DATA_JANALAIEVENT: return m_auiEncounter[DATA_JANALAIEVENT];
309 case DATA_HALAZZIEVENT: return m_auiEncounter[DATA_HALAZZIEVENT];
310 case DATA_HEXLORDEVENT: return m_auiEncounter[DATA_HEXLORDEVENT];
311 case DATA_ZULJINEVENT: return m_auiEncounter[DATA_ZULJINEVENT];
312 case DATA_CHESTLOOTED: return ChestLooted;
313 case TYPE_RAND_VENDOR_1: return RandVendor[0];
314 case TYPE_RAND_VENDOR_2: return RandVendor[1];
315 default: return 0;
316 }

318
319 void Update(uint32 diff) override
320 {
321 if (QuestMinute)
322 {
323 if (QuestTimer <= diff)
324 {
325 QuestMinute--;
326 SaveToDB();
327 QuestTimer += 60000;
328 if (QuestMinute)
329 {
330 DoUpdateWorldState(WORLDSTATE_SHOW_TIMER, 1);
331 DoUpdateWorldState(WORLDSTATE_TIME_TO_SACRIFICE, QuestMinute);
332 } else DoUpdateWorldState(WORLDSTATE_SHOW_TIMER, 0);
333 }
334 QuestTimer -= diff;
335 }
336 }

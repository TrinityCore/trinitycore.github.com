158 void SendAttacker(Unit* target)
159 {
160 std::list<Creature*> templist;
161 float x, y, z;
162 me->GetPosition(x, y, z);
163
164 {
165 CellCoord pair(Trinity::ComputeCellCoord(x, y));
166 Cell cell(pair);
167 cell.SetNoCreate();
168
169 Trinity::AllFriendlyCreaturesInGrid check(me);
170 Trinity::CreatureListSearcher<Trinity::AllFriendlyCreaturesInGrid> searcher(me, templist, check);
171
172 TypeContainerVisitor<Trinity::CreatureListSearcher<Trinity::AllFriendlyCreaturesInGrid>, GridTypeMapContainer> cSearcher(searcher);
173
174 cell.Visit(pair, cSearcher, *(me->GetMap()), *me, me->GetGridActivationRange());
175 }
176
177 if (templist.empty())
178 return;
179
180 for (std::list<Creature*>::const_iterator i = templist.begin(); i != templist.end(); ++i)
181 {
182 if ((*i) && me->IsWithinDistInMap((*i), 25))
183 {
184 (*i)->SetNoCallAssistance(true);
185 (*i)->AI()->AttackStart(target);
186 }
187 }
188 }

196 void MoveInLineOfSight(Unit* who) override
197
198 {
199 if (!MoveEvent)
200 {
201 ScriptedAI::MoveInLineOfSight(who);
202 }
203 else
204 {
205 if (me->IsHostileTo(who))
206 {
207 if (!inMove)
208 {
209 switch (MovePhase)
210 {
211 case 0:
212 if (me->IsWithinDistInMap(who, 50))
213 {
214 Talk(YELL_NALORAKK_WAVE1);
215
216 (*me).GetMotionMaster()->MovePoint(1, NalorakkWay[1][0], NalorakkWay[1][1], NalorakkWay[1][2]);
217 MovePhase ++;
218 inMove = true;
219
220 SendAttacker(who);
221 }
222 break;
223 case 2:
224 if (me->IsWithinDistInMap(who, 40))
225 {
226 Talk(YELL_NALORAKK_WAVE2);
227
228 (*me).GetMotionMaster()->MovePoint(3, NalorakkWay[3][0], NalorakkWay[3][1], NalorakkWay[3][2]);
229 MovePhase ++;
230 inMove = true;
231
232 SendAttacker(who);
233 }
234 break;
235 case 5:
236 if (me->IsWithinDistInMap(who, 40))
237 {
238 Talk(YELL_NALORAKK_WAVE3);
239
240 (*me).GetMotionMaster()->MovePoint(6, NalorakkWay[6][0], NalorakkWay[6][1], NalorakkWay[6][2]);
241 MovePhase ++;
242 inMove = true;
243
244 SendAttacker(who);
245 }
246 break;
247 case 7:
248 if (me->IsWithinDistInMap(who, 50))
249 {
250 SendAttacker(who);
251
252 Talk(YELL_NALORAKK_WAVE4);
253
254 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
255 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
256
257 MoveEvent = false;
258 }
259 break;
260 }
261 }
262 }

309 case 2:

312 return;
313 case 1:
314 case 3:
315 case 4:

334 void UpdateAI(uint32 diff) override
335 {
336 if (waitTimer && inMove)
337 {
338 if (waitTimer <= diff)
339 {
340 (*me).GetMotionMaster()->MovementExpired();
341 (*me).GetMotionMaster()->MovePoint(MovePhase, NalorakkWay[MovePhase][0], NalorakkWay[MovePhase][1], NalorakkWay[MovePhase][2]);
342 waitTimer = 0;
343 } else waitTimer -= diff;
344 }
345
346 if (!UpdateVictim())
347 return;
348
349 if (Berserk_Timer <= diff)
350 {
351 DoCast(me, SPELL_BERSERK, true);
352 Talk(YELL_BERSERK);
353 Berserk_Timer = 600000;
354 } else Berserk_Timer -= diff;
355
356 if (ShapeShift_Timer <= diff)
357 {
358 if (inBearForm)
359 {
360 // me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, 5122);
361 Talk(YELL_SHIFTEDTOTROLL);
362 me->RemoveAurasDueToSpell(SPELL_BEARFORM);
363 Surge_Timer = urand(15000, 20000);
364 BrutalSwipe_Timer = urand(7000, 12000);
365 Mangle_Timer = urand(10000, 15000);
366 ShapeShift_Timer = urand(45000, 50000);
367 inBearForm = false;
368 }
369 else
370 {
371 // me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, 0);
372 Talk(YELL_SHIFTEDTOBEAR);
373 DoCast(me, SPELL_BEARFORM, true);
374 LaceratingSlash_Timer = 2000; // dur 18s
375 RendFlesh_Timer = 3000; // dur 5s
376 DeafeningRoar_Timer = urand(5000, 10000); // dur 2s
377 ShapeShift_Timer = urand(20000, 25000); // dur 30s
378 inBearForm = true;
379 }
380 } else ShapeShift_Timer -= diff;
381
382 if (!inBearForm)
383 {
384 if (BrutalSwipe_Timer <= diff)
385 {
386 DoCastVictim(SPELL_BRUTALSWIPE);
387 BrutalSwipe_Timer = urand(7000, 12000);
388 } else BrutalSwipe_Timer -= diff;
389
390 if (Mangle_Timer <= diff)
391 {
392 if (me->GetVictim() && !me->EnsureVictim()->HasAura(SPELL_MANGLEEFFECT))
393 {
394 DoCastVictim(SPELL_MANGLE);
395 Mangle_Timer = 1000;
396 }
397 else Mangle_Timer = urand(10000, 15000);
398 } else Mangle_Timer -= diff;
399
400 if (Surge_Timer <= diff)
401 {
402 Talk(YELL_SURGE);
403 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 45, true);
404 if (target)
405 DoCast(target, SPELL_SURGE);
406 Surge_Timer = urand(15000, 20000);
407 } else Surge_Timer -= diff;
408 }
409 else
410 {
411 if (LaceratingSlash_Timer <= diff)
412 {
413 DoCastVictim(SPELL_LACERATINGSLASH);
414 LaceratingSlash_Timer = urand(18000, 23000);
415 } else LaceratingSlash_Timer -= diff;
416
417 if (RendFlesh_Timer <= diff)
418 {
419 DoCastVictim(SPELL_RENDFLESH);
420 RendFlesh_Timer = urand(5000, 10000);
421 } else RendFlesh_Timer -= diff;
422
423 if (DeafeningRoar_Timer <= diff)
424 {
425 DoCastVictim(SPELL_DEAFENINGROAR);
426 DeafeningRoar_Timer = urand(15000, 20000);
427 } else DeafeningRoar_Timer -= diff;
428 }
429
430 DoMeleeAttackIfReady();
431 }

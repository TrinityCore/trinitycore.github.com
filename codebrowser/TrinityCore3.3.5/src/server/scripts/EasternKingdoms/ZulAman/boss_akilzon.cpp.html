141 void SetWeather(uint32 weather, float grade)
142 {
143 Map* map = me->GetMap();
144 if (!map->IsDungeon())

153 void HandleStormSequence(Unit* Cloud) // 1: begin, 2-9: tick, 10: end
154 {
155 if (StormCount < 10 && StormCount > 1)
156 {
157 // deal damage
158 int32 bp0 = 800;
159 for (uint8 i = 2; i < StormCount; ++i)
160 bp0 *= 2;
161
162 CellCoord p(Trinity::ComputeCellCoord(me->GetPositionX(), me->GetPositionY()));
163 Cell cell(p);
164 cell.SetNoCreate();
165
166 std::list<Unit*> tempUnitMap;
167
168 {
169 Trinity::AnyAoETargetUnitInObjectRangeCheck u_check(me, me, SIZE_OF_GRIDS);
170 Trinity::UnitListSearcher<Trinity::AnyAoETargetUnitInObjectRangeCheck> searcher(me, tempUnitMap, u_check);
171
172 TypeContainerVisitor<Trinity::UnitListSearcher<Trinity::AnyAoETargetUnitInObjectRangeCheck>, WorldTypeMapContainer > world_unit_searcher(searcher);
173 TypeContainerVisitor<Trinity::UnitListSearcher<Trinity::AnyAoETargetUnitInObjectRangeCheck>, GridTypeMapContainer > grid_unit_searcher(searcher);
174
175 cell.Visit(p, world_unit_searcher, *me->GetMap(), *me, SIZE_OF_GRIDS);
176 cell.Visit(p, grid_unit_searcher, *me->GetMap(), *me, SIZE_OF_GRIDS);
177 }
178
179 // deal damage
180 for (std::list<Unit*>::const_iterator i = tempUnitMap.begin(); i != tempUnitMap.end(); ++i)
181 {
182 if (Unit* target = (*i))
183 {
184 if (Cloud && !Cloud->IsWithinDist(target, 6, false))
185 Cloud->CastCustomSpell(target, SPELL_ZAP, &bp0, NULL, NULL, true, 0, 0, me->GetGUID());
186 }
187 }
188
189 // visual
190 float x, y, z;
191 z = me->GetPositionZ();
192 uint8 maxCount = 5 + rand32() % 5;
193 for (uint8 i = 0; i < maxCount; ++i)
194 {
195 x = 343.0f + rand32() % 60;
196 y = 1380.0f + rand32() % 60;
197 if (Unit* trigger = me->SummonTrigger(x, y, z, 0, 2000))
198 {
199 trigger->setFaction(35);
200 trigger->SetMaxHealth(100000);
201 trigger->SetHealth(100000);
202 trigger->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
203 if (Cloud)
204 Cloud->CastCustomSpell(trigger, /*43661*/SPELL_ZAP, &bp0, NULL, NULL, true, 0, 0, Cloud->GetGUID());
205 }
206 }
207 }
208
209 ++StormCount;
210
211 if (StormCount > 10)
212 {
213 StormCount = 0; // finish
214 events.ScheduleEvent(EVENT_SUMMON_EAGLES, 5000);
215 me->InterruptNonMeleeSpells(false);
216 CloudGUID.Clear();
217 if (Cloud)
218 Cloud->DealDamage(Cloud, Cloud->GetHealth(), NULL, DIRECT_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
219 SetWeather(WEATHER_STATE_FINE, 0.0f);
220 isRaining = false;
221 }
222 events.ScheduleEvent(EVENT_STORM_SEQUENCE, 1000);
223 }

225 void UpdateAI(uint32 diff) override
226 {
227 if (!UpdateVictim())
228 return;
229
230 events.Update(diff);
231
232 while (uint32 eventId = events.ExecuteEvent())
233 {
234 switch (eventId)
235 {
236 case EVENT_STATIC_DISRUPTION:
237 {
238 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1);
239 if (!target)
240 target = me->GetVictim();
241 if (target)
242 {
243 TargetGUID = target->GetGUID();
244 DoCast(target, SPELL_STATIC_DISRUPTION, false);
245 me->SetInFront(me->GetVictim());
246 }
247 /*if (float dist = me->IsWithinDist3d(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 5.0f) dist = 5.0f;
248 SDisruptAOEVisual_Timer = 1000 + floor(dist / 30 * 1000.0f);*/
249 events.ScheduleEvent(EVENT_STATIC_DISRUPTION, urand(10000, 18000));
250 break;
251 }
252 case EVENT_GUST_OF_WIND:
253 {
254 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1);
255 if (!target)
256 target = me->GetVictim();
257 if (target)
258 DoCast(target, SPELL_GUST_OF_WIND);
259 events.ScheduleEvent(EVENT_GUST_OF_WIND, urand(20000, 30000));
260 break;
261 }
262 case EVENT_CALL_LIGHTNING:
263 DoCastVictim(SPELL_CALL_LIGHTNING);
264 events.ScheduleEvent(EVENT_CALL_LIGHTNING, urand(12000, 17000)); // totaly random timer. can't find any info on this
265 break;
266 case EVENT_ELECTRICAL_STORM:
267 {
268 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50, true);
269 if (!target)
270 {
271 EnterEvadeMode();
272 return;
273 }
274 target->CastSpell(target, 44007, true); // cloud visual
275 DoCast(target, SPELL_ELECTRICAL_STORM, false); // storm cyclon + visual
276 float x, y, z;
277 target->GetPosition(x, y, z);
278 /// @todo: fix it in correct way, that causes player to can fly until logout
279 /*
280 if (target)
281 {
282 target->SetDisableGravity(true);
283 target->MonsterMoveWithSpeed(x, y, me->GetPositionZ()+15, 0);
284 }
285 */
286
287 Unit* Cloud = me->SummonTrigger(x, y, me->GetPositionZ()+16, 0, 15000);
288 if (Cloud)
289 {
290 CloudGUID = Cloud->GetGUID();
291 Cloud->SetDisableGravity(true);
292 Cloud->StopMoving();
293 Cloud->SetObjectScale(1.0f);
294 Cloud->setFaction(35);
295 Cloud->SetMaxHealth(9999999);
296 Cloud->SetHealth(9999999);
297 Cloud->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
298 }
299 StormCount = 1;
300 events.ScheduleEvent(EVENT_ELECTRICAL_STORM, 60000); // 60 seconds(bosskillers)
301 events.ScheduleEvent(EVENT_RAIN, urand(47000, 52000));
302 break;
303 }
304 case EVENT_RAIN:
305 if (!isRaining)
306 {
307 SetWeather(WEATHER_STATE_HEAVY_RAIN, 0.9999f);
308 isRaining = true;
309 }
310 else
311 events.ScheduleEvent(EVENT_RAIN, 1000);
312 break;
313 case EVENT_STORM_SEQUENCE:
314 {
315 Unit* target = ObjectAccessor::GetUnit(*me, CloudGUID);
316 if (!target || !target->IsAlive())
317 {
318 EnterEvadeMode();
319 return;
320 }
321 else if (Unit* Cyclone = ObjectAccessor::GetUnit(*me, CycloneGUID))
322 Cyclone->CastSpell(target, SPELL_SAND_STORM, true); // keep casting or...
323 HandleStormSequence(target);
324 break;
325 }
326 case EVENT_SUMMON_EAGLES:
327 Talk(SAY_SUMMON);
328
329 float x, y, z;
330 me->GetPosition(x, y, z);
331
332 for (uint8 i = 0; i < 8; ++i)
333 {
334 Unit* bird = ObjectAccessor::GetUnit(*me, BirdGUIDs[i]);
335 if (!bird) //they despawned on die
336 {
337 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
338 {
339 x = target->GetPositionX() + irand(-10, 10);
340 y = target->GetPositionY() + irand(-10, 10);
341 z = target->GetPositionZ() + urand(16, 20);
342 if (z > 95)
343 z = 95.0f - urand(0, 5);
344 }
345 Creature* creature = me->SummonCreature(NPC_SOARING_EAGLE, x, y, z, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
346 if (creature)
347 {
348 creature->AddThreat(me->GetVictim(), 1.0f);
349 creature->AI()->AttackStart(me->GetVictim());
350 BirdGUIDs[i] = creature->GetGUID();
351 }
352 }
353 }
354 break;
355 case EVENT_ENRAGE:
356 Talk(SAY_ENRAGE);
357 DoCast(me, SPELL_BERSERK, true);
358 events.ScheduleEvent(EVENT_ENRAGE, 600000);
359 break;
360 default:
361 break;
362 }
363 }
364
365 DoMeleeAttackIfReady();
366 }

395 void Initialize()
396 {
397 EagleSwoop_Timer = urand(5000, 10000);
398 arrived = true;

400 }

402 uint32 EagleSwoop_Timer;
403 bool arrived;
404 ObjectGuid TargetGUID;
405
406 void Reset() override
407 {
408 Initialize();
409 me->SetDisableGravity(true);
410 }
411
412 void EnterCombat(Unit* /*who*/) override
413 {
414 DoZoneInCombat();
415 }
416
417 void MoveInLineOfSight(Unit* /*who*/) override { }
418
419
420 void MovementInform(uint32, uint32) override
421 {
422 arrived = true;
423 if (TargetGUID)
424 {
425 if (Unit* target = ObjectAccessor::GetUnit(*me, TargetGUID))
426 DoCast(target, SPELL_EAGLE_SWOOP, true);
427 TargetGUID.Clear();
428 me->SetSpeed(MOVE_RUN, 1.2f);
429 EagleSwoop_Timer = urand(5000, 10000);
430 }
431 }
432
433 void UpdateAI(uint32 diff) override
434 {
435 if (EagleSwoop_Timer <= diff)
436 EagleSwoop_Timer = 0;
437 else
438 EagleSwoop_Timer -= diff;
439
440 if (arrived)
441 {
442 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
443 {
444 float x, y, z;
445 if (EagleSwoop_Timer)
446 {
447 x = target->GetPositionX() + irand(-10, 10);
448 y = target->GetPositionY() + irand(-10, 10);
449 z = target->GetPositionZ() + urand(10, 15);
450 if (z > 95)
451 z = 95.0f - urand(0, 5);
452 }
453 else
454 {
455 target->GetContactPoint(me, x, y, z);
456 z += 2;
457 me->SetSpeed(MOVE_RUN, 5.0f);
458 TargetGUID = target->GetGUID();
459 }
460 me->GetMotionMaster()->MovePoint(0, x, y, z);
461 arrived = false;
462 }
463 }
464 }
465 };

470 }

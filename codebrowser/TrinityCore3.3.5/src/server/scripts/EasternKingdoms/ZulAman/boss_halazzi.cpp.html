108 void Reset() override
109 {
110 instance->SetData(DATA_HALAZZIEVENT, NOT_STARTED);

128 void JustSummoned(Creature* summon) override
129 {
130 summon->AI()->AttackStart(me->GetVictim());

136 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override

138 if (damage >= me->GetHealth() && Phase != PHASE_ENRAGE)

142 void SpellHit(Unit*, const SpellInfo* spell) override

144 if (spell->Id == SPELL_TRANSFORM_SPLIT2)

149 {
150 if (Phase != PHASE_MERGE)
151 ScriptedAI::AttackStart(who);
152 }
153
154 void EnterPhase(PhaseHalazzi NextPhase)
155 {
156 switch (NextPhase)
157 {
158 case PHASE_LYNX:
159 case PHASE_ENRAGE:
160 if (Phase == PHASE_MERGE)
161 {
162 DoCast(me, SPELL_TRANSFORM_MERGE, true);
163 me->Attack(me->GetVictim(), true);
164 me->GetMotionMaster()->MoveChase(me->GetVictim());
165 }
166 if (Creature* Lynx = ObjectAccessor::GetCreature(*me, LynxGUID))
167 Lynx->DisappearAndDie();
168 me->SetMaxHealth(600000);
169 me->SetHealth(600000 - 150000 * TransformCount);
170 FrenzyTimer = 16000;
171 SaberlashTimer = 20000;
172 ShockTimer = 10000;
173 TotemTimer = 12000;
174 break;
175 case PHASE_SPLIT:
176 Talk(SAY_SPLIT);
177 DoCast(me, SPELL_TRANSFORM_SPLIT, true);
178 break;
179 case PHASE_HUMAN:
180 //DoCast(me, SPELL_SUMMON_LYNX, true);
181 DoSpawnCreature(NPC_SPIRIT_LYNX, 5, 5, 0, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
182 me->SetMaxHealth(400000);
183 me->SetHealth(400000);
184 ShockTimer = 10000;
185 TotemTimer = 12000;
186 break;
187 case PHASE_MERGE:
188 if (Unit* pLynx = ObjectAccessor::GetUnit(*me, LynxGUID))
189 {
190 Talk(SAY_MERGE);
191 pLynx->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
192 pLynx->GetMotionMaster()->Clear();
193 pLynx->GetMotionMaster()->MoveFollow(me, 0, 0);
194 me->GetMotionMaster()->Clear();
195 me->GetMotionMaster()->MoveFollow(pLynx, 0, 0);
196 ++TransformCount;
197 }
198 break;
199 default:
200 break;
201 }
202 Phase = NextPhase;
203 }
204
205 void UpdateAI(uint32 diff) override
206 {
207 if (!UpdateVictim())
208 return;
209
210 if (BerserkTimer <= diff)
211 {
212 Talk(SAY_BERSERK);
213 DoCast(me, SPELL_BERSERK, true);
214 BerserkTimer = 60000;
215 } else BerserkTimer -= diff;
216
217 if (Phase == PHASE_LYNX || Phase == PHASE_ENRAGE)
218 {
219 if (SaberlashTimer <= diff)
220 {
221 // A tank with more than 490 defense skills should receive no critical hit
222 //DoCast(me, 41296, true);
223 DoCastVictim(SPELL_SABER_LASH, true);
224 //me->RemoveAurasDueToSpell(41296);
225 SaberlashTimer = 30000;
226 } else SaberlashTimer -= diff;
227
228 if (FrenzyTimer <= diff)
229 {
230 DoCast(me, SPELL_FRENZY);
231 FrenzyTimer = urand(10000, 15000);
232 } else FrenzyTimer -= diff;
233
234 if (Phase == PHASE_LYNX)
235 {
236 if (CheckTimer <= diff)
237 {
238 if (HealthBelowPct(25 * (3 - TransformCount)))
239 EnterPhase(PHASE_SPLIT);
240 CheckTimer = 1000;
241 } else CheckTimer -= diff;
242 }
243 }
244
245 if (Phase == PHASE_HUMAN || Phase == PHASE_ENRAGE)
246 {
247 if (TotemTimer <= diff)
248 {
249 DoCast(me, SPELL_SUMMON_TOTEM);
250 TotemTimer = 20000;
251 } else TotemTimer -= diff;
252
253 if (ShockTimer <= diff)
254 {
255 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
256 {
257 if (target->IsNonMeleeSpellCast(false))
258 DoCast(target, SPELL_EARTHSHOCK);
259 else
260 DoCast(target, SPELL_FLAMESHOCK);
261 ShockTimer = urand(10000, 15000);
262 }
263 } else ShockTimer -= diff;
264
265 if (Phase == PHASE_HUMAN)
266 {
267 if (CheckTimer <= diff)
268 {
269 if (!HealthAbovePct(20) /*HealthBelowPct(10)*/)
270 EnterPhase(PHASE_MERGE);
271 else
272 {
273 Unit* Lynx = ObjectAccessor::GetUnit(*me, LynxGUID);
274 if (Lynx && !Lynx->HealthAbovePct(20) /*Lynx->HealthBelowPct(10)*/)
275 EnterPhase(PHASE_MERGE);
276 }
277 CheckTimer = 1000;
278 } else CheckTimer -= diff;
279 }
280 }
281
282 if (Phase == PHASE_MERGE)
283 {
284 if (CheckTimer <= diff)
285 {
286 Unit* Lynx = ObjectAccessor::GetUnit(*me, LynxGUID);
287 if (Lynx)
288 {
289 Lynx->GetMotionMaster()->MoveFollow(me, 0, 0);
290 me->GetMotionMaster()->MoveFollow(Lynx, 0, 0);
291 if (me->IsWithinDistInMap(Lynx, 6.0f))
292 {
293 if (TransformCount < 3)
294 EnterPhase(PHASE_LYNX);
295 else
296 EnterPhase(PHASE_ENRAGE);
297 }
298 }
299 CheckTimer = 1000;
300 } else CheckTimer -= diff;
301 }
302
303 DoMeleeAttackIfReady();
304 }

318 }

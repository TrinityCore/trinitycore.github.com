75 void DoSpawnRandom()
76 {
77 uint32 cEntry = 0;
78 switch (rand32() % 10)
79 {
80 case 0: cEntry = 24397; break; //Mannuth
81 case 1: cEntry = 24403; break; //Deez
82 case 2: cEntry = 24404; break; //Galathryn
83 case 3: cEntry = 24405; break; //Adarrah
84 case 4: cEntry = 24406; break; //Fudgerick
85 case 5: cEntry = 24407; break; //Darwen
86 case 6: cEntry = 24445; break; //Mitzi
87 case 7: cEntry = 24448; break; //Christian
88 case 8: cEntry = 24453; break; //Brennan
89 case 9: cEntry = 24455; break; //Hollee
90 }
91
92 if (!instance->GetData(TYPE_RAND_VENDOR_1))
93 if (rand32() % 10 == 1) cEntry = 24408; //Gunter
94 if (!instance->GetData(TYPE_RAND_VENDOR_2))
95 if (rand32() % 10 == 1) cEntry = 24409; //Kyren
96
97 if (cEntry) me->UpdateEntry(cEntry);
98
99 if (cEntry == 24408) instance->SetData(TYPE_RAND_VENDOR_1, DONE);
100 if (cEntry == 24409) instance->SetData(TYPE_RAND_VENDOR_2, DONE);
101 }
102
103 void SpellHit(Unit* caster, const SpellInfo* spell) override
104 {
105 if (spell->Id == SPELL_REMOVE_AMANI_CURSE && caster->GetTypeId() == TYPEID_PLAYER && me->GetEntry() == NPC_FOREST_FROG)
106 {
107 //increase or decrease chance of mojo?
108 if (rand32() % 99 == 50) DoCast(caster, SPELL_PUSH_MOJO, true);
109 else DoSpawnRandom();
110 }
111 }
112 };
113
114 CreatureAI* GetAI(Creature* creature) const override
115 {
116 return GetInstanceAI<npc_forest_frogAI>(creature);
117 }

126static uint32 HostageEntry[] = {23790, 23999, 24024, 24001};
127static uint32 ChestEntry[] = {186648, 187021, 186672, 186667};

129class npc_zulaman_hostage : public CreatureScript
130{
131 public:
132 npc_zulaman_hostage() : CreatureScript("npc_zulaman_hostage") { }
133
134 bool OnGossipHello(Player* player, Creature* creature) override
135 {
136 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HOSTAGE1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
137 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
138 return true;
139 }
140
141 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
142 {
143 player->PlayerTalkClass->ClearMenus();
144
145 if (action == GOSSIP_ACTION_INFO_DEF + 1)
146 player->CLOSE_GOSSIP_MENU();
147
148 if (!creature->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP))
149 return true;
150
151 creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
152
153 InstanceScript* instance = creature->GetInstanceScript();
154 if (instance)
155 {
156 //uint8 progress = instance->GetData(DATA_CHESTLOOTED);
157 instance->SetData(DATA_CHESTLOOTED, 0);
158 float x, y, z;
159 creature->GetPosition(x, y, z);
160 uint32 entry = creature->GetEntry();
161 for (uint8 i = 0; i < 4; ++i)
162 {
163 if (HostageEntry[i] == entry)
164 {
165 creature->SummonGameObject(ChestEntry[i], x-2, y, z, 0, 0, 0, 0, 0, 0);
166 break;
167 }
168 }
169 }
170 return true;
171 }

243 struct npc_harrison_jonesAI : public ScriptedAI
244 {
245 npc_harrison_jonesAI(Creature* creature) : ScriptedAI(creature)
246 {
247 Initialize();
248 instance = creature->GetInstanceScript();
249 }
250
251 void Initialize()
252 {
253 _gongEvent = 0;
254 _gongTimer = 0;
255 uiTargetGUID = 0;
256 }
257
258 InstanceScript* instance;
259
260 uint8 _gongEvent;
261 uint32 _gongTimer;
262 uint64 uiTargetGUID;
263
264 void Reset() override
265 {
266 Initialize();
267 }
268
269 void EnterCombat(Unit* /*who*/) override { }
270
271 void sGossipSelect(Player* player, uint32 menuId, uint32 gossipListId) override
272 {
273 if (me->GetCreatureTemplate()->GossipMenuId == menuId && !gossipListId)
274 {
275 player->CLOSE_GOSSIP_MENU();
276 me->SetFacingToObject(player);
277 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
278 Talk(SAY_HARRISON_0);
279 _gongEvent = GONG_EVENT_1;
280 _gongTimer = 4000;
281 }
282 }
283
284 void SpellHit(Unit*, const SpellInfo* spell) override
285 {
286 if (spell->Id == SPELL_COSMETIC_SPEAR_THROW)
287 {
288 me->RemoveAllAuras();
289 me->SetEntry(NPC_HARRISON_JONES_2);
290 me->SetDisplayId(MODEL_HARRISON_JONES_2);
291 me->SetTarget(ObjectGuid::Empty);
292 me->SetByteValue(UNIT_FIELD_BYTES_1, 0, UNIT_STAND_STATE_DEAD);
293 me->SetFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_DEAD);
294 instance->SetData(DATA_GONGEVENT, DONE);
295 }
296 }
297
298 void UpdateAI(uint32 diff) override
299 {
300 if (_gongEvent)
301 {
302 if (_gongTimer <= diff)
303 {
304 switch (_gongEvent)
305 {
306 case GONG_EVENT_1:
307 me->GetMotionMaster()->MovePath(HARRISON_MOVE_1, false);
308 _gongEvent = GONG_EVENT_2;
309 _gongTimer = 12000;
310 break;
311 case GONG_EVENT_2:
312 me->SetFacingTo(6.235659f);
313 Talk(SAY_HARRISON_1);
314 DoCast(me, SPELL_BANGING_THE_GONG);
315 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_MACE));
316 me->SetSheath(SHEATH_STATE_MELEE);
317 _gongEvent = GONG_EVENT_3;
318 _gongTimer = 4000;
319 break;
320 case GONG_EVENT_3:
321 if (GameObject* gong = me->GetMap()->GetGameObject(instance->GetGuidData(GO_STRANGE_GONG)))
322 gong->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
323 _gongEvent = GONG_EVENT_4;
324 _gongTimer = 105000;
325 break;
326 case GONG_EVENT_4:
327 me->RemoveAura(SPELL_BANGING_THE_GONG);
328 if (GameObject* gong = me->GetMap()->GetGameObject(instance->GetGuidData(GO_STRANGE_GONG)))
329 gong->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
330
331 // trigger or gong will need to be scripted to set IN_PROGRESS after enough hits.
332 // This is temp workaround.
333 instance->SetData(DATA_GONGEVENT, IN_PROGRESS); // to be removed.
334
335 if (instance->GetData(DATA_GONGEVENT) == IN_PROGRESS)
336 {
337 // Players are Now Saved to instance at SPECIAL (Player should be notified?)
338 me->GetMotionMaster()->MovePath(HARRISON_MOVE_2, false);
339 _gongEvent = GONG_EVENT_5;
340 _gongTimer = 5000;
341 }
342 else
343 {
344 _gongTimer = 1000;
345 _gongEvent = GONG_EVENT_9;
346 }
347 break;
348 case GONG_EVENT_5:
349 me->SetEntry(NPC_HARRISON_JONES_1);
350 me->SetDisplayId(MODEL_HARRISON_JONES_1);
351 Talk(SAY_HARRISON_2);
352 _gongTimer = 12000;
353 _gongEvent = GONG_EVENT_6;
354 break;
355 case GONG_EVENT_6:
356 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_USE_STANDING);
357 Talk(SAY_HARRISON_3);
358 _gongTimer = 7000;
359 _gongEvent = GONG_EVENT_7;
360 break;
361 case GONG_EVENT_7:
362 if (!uiTargetGUID)
363 {
364 std::list<Creature*> targetList;
365 GetCreatureListWithEntryInGrid(targetList, me, NPC_AMANISHI_GUARDIAN, 26.0f);
366 if (!targetList.empty())
367 {
368 for (std::list<Creature*>::const_iterator itr = targetList.begin(); itr != targetList.end(); ++itr)
369 {
370 if (Creature* ptarget = *itr)
371 {
372 if (ptarget->GetPositionX() > 120)
373 {
374 ptarget->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_SPEAR));
375 ptarget->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
376 ptarget->SetReactState(REACT_PASSIVE);
377 ptarget->AI()->SetData(0, 1);
378 }
379 else
380 {
381 ptarget->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
382 ptarget->SetReactState(REACT_PASSIVE);
383 ptarget->AI()->SetData(0, 2);
384 }
385 }
386 }
387 }
388 }
389
390 if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_MASSIVE_GATE)))
391 gate->SetGoState(GO_STATE_ACTIVE);
392 _gongTimer = 2000;
393 _gongEvent = GONG_EVENT_8;
394 break;
395 case GONG_EVENT_8:
396 DoCast(me, SPELL_STEALTH);
397 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(0));
398 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_NONE);
399 me->GetMotionMaster()->MovePath(HARRISON_MOVE_3, false);
400 _gongTimer = 1000;
401 _gongEvent = 0;
402 break;
403 case GONG_EVENT_9:
404 me->GetMotionMaster()->MovePoint(0, 120.687f, 1674.0f, 42.0217f);
405 _gongTimer = 12000;
406 _gongEvent = GONG_EVENT_10;
407 break;
408 case GONG_EVENT_10:
409 me->SetFacingTo(1.59044f);
410 _gongEvent = 11;
411 _gongTimer = 6000;
412 break;
413 case GONG_EVENT_11:
414 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
415
416 instance->SetData(DATA_GONGEVENT, NOT_STARTED);
417 _gongEvent = 0;
418 _gongTimer = 1000;
419 break;
420 }
421 }
422 else
423 _gongTimer -= diff;
424 }
425 }
426 };

202 void FireWall()
203 {
204 uint8 WallNum;
205 Creature* wall = NULL;
206 for (uint8 i = 0; i < 4; ++i)
207 {
208 if (i == 0 || i == 2)
209 WallNum = 3;
210 else
211 WallNum = 2;
212
213 for (uint8 j = 0; j < WallNum; j++)
214 {
215 if (WallNum == 3)
216 wall = me->SummonCreature(NPC_FIRE_BOMB, FireWallCoords[i][0], FireWallCoords[i][1]+5*(j-1), FireWallCoords[i][2], FireWallCoords[i][3], TEMPSUMMON_TIMED_DESPAWN, 15000);
217 else
218 wall = me->SummonCreature(NPC_FIRE_BOMB, FireWallCoords[i][0]-2+4*j, FireWallCoords[i][1], FireWallCoords[i][2], FireWallCoords[i][3], TEMPSUMMON_TIMED_DESPAWN, 15000);
219 if (wall) wall->CastSpell(wall, SPELL_FIRE_WALL, true);
220 }

239 bool HatchAllEggs(uint32 action) //1: reset, 2: isHatching all
240 {
241 std::list<Creature*> templist;
242 float x, y, z;
243 me->GetPosition(x, y, z);
244
245 {
246 CellCoord pair(Trinity::ComputeCellCoord(x, y));
247 Cell cell(pair);
248 cell.SetNoCreate();
249
250 Trinity::AllCreaturesOfEntryInRange check(me, NPC_EGG, 100);
251 Trinity::CreatureListSearcher<Trinity::AllCreaturesOfEntryInRange> searcher(me, templist, check);
252
253 TypeContainerVisitor<Trinity::CreatureListSearcher<Trinity::AllCreaturesOfEntryInRange>, GridTypeMapContainer> cSearcher(searcher);
254
255 cell.Visit(pair, cSearcher, *me->GetMap(), *me, me->GetGridActivationRange());
256 }
257
258 //TC_LOG_ERROR("scripts", "Eggs %d at middle", templist.size());
259 if (templist.empty())
260 return false;
261
262 for (std::list<Creature*>::const_iterator i = templist.begin(); i != templist.end(); ++i)
263 {
264 if (action == 1)
265 (*i)->SetDisplayId(10056);
266 else if (action == 2 &&(*i)->GetDisplayId() != 11686)
267 (*i)->CastSpell(*i, SPELL_HATCH_EGG, false);
268 }
269 return true;
270 }

297 void HandleBombSequence()
298 {
299 if (BombCount < 40)
300 {
301 if (Unit* FireBomb = ObjectAccessor::GetUnit(*me, FireBombGUIDs[BombCount]))
302 {
303 FireBomb->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
304 DoCast(FireBomb, SPELL_FIRE_BOMB_THROW, true);
305 FireBomb->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
306 }
307 ++BombCount;
308 if (BombCount == 40)
309 {
310 BombSequenceTimer = 5000;
311 } else BombSequenceTimer = 100;
312 }
313 else
314 {
315 Boom();
316 isBombing = false;
317 BombTimer = urand(20000, 40000);
318 me->RemoveAurasDueToSpell(SPELL_FIRE_BOMB_CHANNEL);
319 if (EnrageTimer <= 10000)
320 EnrageTimer = 0;
321 else
322 EnrageTimer -= 10000;
323 }
324 }

326 void UpdateAI(uint32 diff) override
327 {
328 if (isFlameBreathing)
329 {
330 if (!me->IsNonMeleeSpellCast(false))
331 isFlameBreathing = false;
332 else
333 return;
334 }
335
336 if (isBombing)
337 {
338 if (BombSequenceTimer <= diff)
339 HandleBombSequence();
340 else
341 BombSequenceTimer -= diff;
342 return;
343 }
344
345 if (!UpdateVictim())
346 return;
347
348 //enrage if under 25% hp before 5 min.
349 if (!enraged && HealthBelowPct(25))
350 EnrageTimer = 0;
351
352 if (EnrageTimer <= diff)
353 {
354 if (!enraged)
355 {
356 DoCast(me, SPELL_ENRAGE, true);
357 enraged = true;
358 EnrageTimer = 300000;
359 }
360 else
361 {
362 Talk(SAY_BERSERK);
363 DoCast(me, SPELL_BERSERK, true);
364 EnrageTimer = 300000;
365 }
366 } else EnrageTimer -= diff;
367
368 if (BombTimer <= diff)
369 {
370 Talk(SAY_FIRE_BOMBS);
371
372 me->AttackStop();
373 me->GetMotionMaster()->Clear();
374 DoTeleportTo(JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2]);
375 me->StopMoving();
376 DoCast(me, SPELL_FIRE_BOMB_CHANNEL, false);
377 //DoTeleportPlayer(me, JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2], 0);
378 //DoCast(me, SPELL_TELE_TO_CENTER, true);
379
380 FireWall();
381 SpawnBombs();
382 isBombing = true;
383 BombSequenceTimer = 100;
384
385 //Teleport every Player into the middle
386 Map* map = me->GetMap();
387 if (!map->IsDungeon())
388 return;
389
390 Map::PlayerList const &PlayerList = map->GetPlayers();
391 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
392 if (Player* i_pl = i->GetSource())
393 if (i_pl->IsAlive())
394 DoTeleportPlayer(i_pl, JanalainPos[0][0] - 5 + rand32() % 10, JanalainPos[0][1] - 5 + rand32() % 10, JanalainPos[0][2], 0);
395 //DoCast(Temp, SPELL_SUMMON_PLAYERS, true) // core bug, spell does not work if too far
396 return;
397 } else BombTimer -= diff;
398
399 if (!noeggs)
400 {
401 if (HealthBelowPct(35))
402 {
403 Talk(SAY_ALL_EGGS);
404
405 me->AttackStop();
406 me->GetMotionMaster()->Clear();
407 DoTeleportTo(JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2]);
408 me->StopMoving();
409 DoCast(me, SPELL_HATCH_ALL, false);
410 HatchAllEggs(2);
411 noeggs = true;
412 }
413 else if (HatcherTimer <= diff)
414 {
415 if (HatchAllEggs(0))
416 {
417 Talk(SAY_SUMMON_HATCHER);
418 me->SummonCreature(NPC_AMANI_HATCHER, hatcherway[0][0][0], hatcherway[0][0][1], hatcherway[0][0][2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
419 me->SummonCreature(NPC_AMANI_HATCHER, hatcherway[1][0][0], hatcherway[1][0][1], hatcherway[1][0][2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
420 HatcherTimer = 90000;
421 }
422 else
423 noeggs = true;
424 } else HatcherTimer -= diff;
425 }
426
427 DoMeleeAttackIfReady();
428
429 if (FireBreathTimer <= diff)
430 {
431 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
432 {
433 me->AttackStop();
434 me->GetMotionMaster()->Clear();
435 DoCast(target, SPELL_FLAME_BREATH, false);
436 me->StopMoving();
437 isFlameBreathing = true;
438 }
439 FireBreathTimer = 8000;
440 } else FireBreathTimer -= diff;
441 }
442 };

578 void UpdateAI(uint32 diff) override
579 {
580 if (!instance || !(instance->GetData(DATA_JANALAIEVENT) == IN_PROGRESS))
581 {
582 me->DisappearAndDie();
583 return;
584 }
585
586 if (!isHatching)
587 {
588 if (WaitTimer)
589 {
590 me->GetMotionMaster()->Clear();
591 me->GetMotionMaster()->MovePoint(0, hatcherway[side][waypoint][0], hatcherway[side][waypoint][1], hatcherway[side][waypoint][2]);
592 ++waypoint;
593 WaitTimer = 0;
594 }
595 }
596 else
597 {
598 if (WaitTimer <= diff)
599 {
600 if (HatchEggs(HatchNum))
601 {
602 ++HatchNum;
603 WaitTimer = 10000;
604 }
605 else if (!hasChangedSide)
606 {
607 side = side ? 0 : 1;
608 isHatching = false;
609 waypoint = 3;
610 WaitTimer = 1;
611 hasChangedSide = true;
612 }
613 else
614 me->DisappearAndDie();
615
616 } else WaitTimer -= diff;
617 }
618 }

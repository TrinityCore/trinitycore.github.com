252 void DoMeleeAttackIfReady()
253 {
254 if (!me->IsNonMeleeSpellCast(false))
255 {
256 if (me->isAttackReady() && me->IsWithinMeleeRange(me->GetVictim()))
257 {
258 if (Phase == 1 && !Overpower_Timer)
259 {
260 uint32 health = me->EnsureVictim()->GetHealth();
261 me->AttackerStateUpdate(me->GetVictim());
262 if (me->GetVictim() && health == me->EnsureVictim()->GetHealth())
263 {
264 DoCastVictim(SPELL_OVERPOWER, false);
265 Overpower_Timer = 5000;
266 }
267 } else me->AttackerStateUpdate(me->GetVictim());
268 me->resetAttackTimer();
269 }
270 }
271 }

273 void SpawnAdds()
274 {
275 Creature* creature = NULL;
276 for (uint8 i = 0; i < 4; ++i)
277 {
278 creature = me->SummonCreature(SpiritInfo[i].entry, SpiritInfo[i].x, SpiritInfo[i].y, SpiritInfo[i].z, SpiritInfo[i].orient, TEMPSUMMON_DEAD_DESPAWN, 0);
279 if (creature)
280 {
281 creature->CastSpell(creature, SPELL_SPIRIT_AURA, true);
282 creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
283 creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
284 SpiritGUID[i] = creature->GetGUID();
285 }
286 }
287 }

300 }
301 SpiritGUID[i].Clear();
302 }

315 void EnterPhase(uint32 NextPhase)
316 {
317 switch (NextPhase)
318 {
319 case 0:
320 break;
321 case 1:
322 case 2:
323 case 3:
324 case 4:
325 DoTeleportTo(CENTER_X, CENTER_Y, CENTER_Z, 100);
326 DoResetThreat();
327 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID, 0);
328 me->RemoveAurasDueToSpell(Transform[Phase].unaura);
329 DoCast(me, Transform[Phase].spell);
330 Talk(Transform[Phase].text);
331 if (Phase > 0)
332 {
333 if (Unit* Temp = ObjectAccessor::GetUnit(*me, SpiritGUID[Phase - 1]))
334 Temp->SetUInt32Value(UNIT_FIELD_BYTES_1, UNIT_STAND_STATE_DEAD);
335 }
336 if (Unit* Temp = ObjectAccessor::GetUnit(*me, SpiritGUID[NextPhase - 1]))
337 Temp->CastSpell(me, SPELL_SIPHON_SOUL, false); // should m cast on temp
338 if (NextPhase == 2)
339 {
340 me->GetMotionMaster()->Clear();
341 DoCast(me, SPELL_ENERGY_STORM, true); // enemy aura
342 for (uint8 i = 0; i < 4; ++i)
343 {
344 Creature* Vortex = DoSpawnCreature(CREATURE_FEATHER_VORTEX, 0, 0, 0, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
345 if (Vortex)
346 {
347 Vortex->CastSpell(Vortex, SPELL_CYCLONE_PASSIVE, true);
348 Vortex->CastSpell(Vortex, SPELL_CYCLONE_VISUAL, true);
349 Vortex->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
350 Vortex->SetSpeed(MOVE_RUN, 1.0f);
351 Vortex->AI()->AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0));
352 DoZoneInCombat(Vortex);
353 }
354 }
355 }
356 else
357 AttackStart(me->GetVictim());
358 if (NextPhase == 3)
359 {
360 me->RemoveAurasDueToSpell(SPELL_ENERGY_STORM);
361 Summons.DespawnEntry(CREATURE_FEATHER_VORTEX);
362 me->GetMotionMaster()->MoveChase(me->GetVictim());
363 }
364 break;
365 default:
366 break;
367 }
368 Phase = NextPhase;
369 }

371 void UpdateAI(uint32 diff) override
372 {
373 if (!TankGUID)
374 {
375 if (!UpdateVictim())
376 return;
377
378 if (me->GetHealth() < health_20 * (4 - Phase))
379 EnterPhase(Phase + 1);
380 }
381
382 if (Berserk_Timer <= diff)
383 {
384 DoCast(me, SPELL_BERSERK, true);
385 Talk(YELL_BERSERK);
386 Berserk_Timer = 60000;
387 } else Berserk_Timer -= diff;
388
389 switch (Phase)
390 {
391 case 0:
392 if (Intro_Timer)
393 {
394 if (Intro_Timer <= diff)
395 {
396 Talk(YELL_AGGRO);
397 Intro_Timer = 0;
398 } else Intro_Timer -= diff;
399 }
400
401 if (Whirlwind_Timer <= diff)
402 {
403 DoCast(me, SPELL_WHIRLWIND);
404 Whirlwind_Timer = urand(15000, 20000);
405 } else Whirlwind_Timer -= diff;
406
407 if (Grievous_Throw_Timer <= diff)
408 {
409 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
410 DoCast(target, SPELL_GRIEVOUS_THROW, false);
411 Grievous_Throw_Timer = 10000;
412 } else Grievous_Throw_Timer -= diff;
413 break;
414
415 case 1:
416 if (Creeping_Paralysis_Timer <= diff)
417 {
418 DoCast(me, SPELL_CREEPING_PARALYSIS);
419 Creeping_Paralysis_Timer = 20000;
420 } else Creeping_Paralysis_Timer -= diff;
421
422 if (Overpower_Timer <= diff)
423 {
424 // implemented in DoMeleeAttackIfReady()
425 Overpower_Timer = 0;
426 } else Overpower_Timer -= diff;
427 break;
428
429 case 2:
430 return;
431
432 case 3:
433 if (Claw_Rage_Timer <= diff)
434 {
435 if (!TankGUID)
436 {
437 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
438 {
439 if (me->GetVictim())
440 TankGUID = me->EnsureVictim()->GetGUID();
441 me->SetSpeed(MOVE_RUN, 5.0f);
442 AttackStart(target); // change victim
443 Claw_Rage_Timer = 0;
444 Claw_Loop_Timer = 500;
445 Claw_Counter = 0;
446 }
447 }
448 else if (!Claw_Rage_Timer) // do not do this when Lynx_Rush
449 {
450 if (Claw_Loop_Timer <= diff)
451 {
452 Unit* target = me->GetVictim();
453 if (!target || !target->isTargetableForAttack()) target = ObjectAccessor::GetUnit(*me, TankGUID);
454 if (!target || !target->isTargetableForAttack()) target = SelectTarget(SELECT_TARGET_RANDOM, 0);
455 if (target)
456 {
457 AttackStart(target);
458 if (me->IsWithinMeleeRange(target))
459 {
460 DoCast(target, SPELL_CLAW_RAGE_DAMAGE, true);
461 ++Claw_Counter;
462 if (Claw_Counter == 12)
463 {
464 Claw_Rage_Timer = urand(15000, 20000);
465 me->SetSpeed(MOVE_RUN, 1.2f);
466 AttackStart(ObjectAccessor::GetUnit(*me, TankGUID));
467 TankGUID.Clear();
468 return;
469 }
470 else
471 Claw_Loop_Timer = 500;
472 }
473 }
474 else
475 {
476 EnterEvadeMode(); // if (target)
477 return;
478 }
479 } else Claw_Loop_Timer -= diff;
480 } //if (TankGUID)
481 } else Claw_Rage_Timer -= diff;
482
483 if (Lynx_Rush_Timer <= diff)
484 {
485 if (!TankGUID)
486 {
487 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
488 {
489 TankGUID = me->EnsureVictim()->GetGUID();
490 me->SetSpeed(MOVE_RUN, 5.0f);
491 AttackStart(target); // change victim
492 Lynx_Rush_Timer = 0;
493 Claw_Counter = 0;
494 }
495 }
496 else if (!Lynx_Rush_Timer)
497 {
498 Unit* target = me->GetVictim();
499 if (!target || !target->isTargetableForAttack())
500 {
501 target = SelectTarget(SELECT_TARGET_RANDOM, 0);
502 AttackStart(target);
503 }
504 if (target)
505 {
506 if (me->IsWithinMeleeRange(target))
507 {
508 DoCast(target, SPELL_LYNX_RUSH_DAMAGE, true);
509 ++Claw_Counter;
510 if (Claw_Counter == 9)
511 {
512 Lynx_Rush_Timer = urand(15000, 20000);
513 me->SetSpeed(MOVE_RUN, 1.2f);
514 AttackStart(ObjectAccessor::GetUnit(*me, TankGUID));
515 TankGUID.Clear();
516 }
517 else
518 AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0));
519 }
520 }
521 else
522 {
523 EnterEvadeMode(); // if (target)
524 return;
525 }
526 } //if (TankGUID)
527 } else Lynx_Rush_Timer -= diff;
528
529 break;
530 case 4:
531 if (Flame_Whirl_Timer <= diff)
532 {
533 DoCast(me, SPELL_FLAME_WHIRL);
534 Flame_Whirl_Timer = 12000;
535 }Flame_Whirl_Timer -= diff;
536
537 if (Pillar_Of_Fire_Timer <= diff)
538 {
539 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
540 DoCast(target, SPELL_SUMMON_PILLAR);
541 Pillar_Of_Fire_Timer = 10000;
542 } else Pillar_Of_Fire_Timer -= diff;
543
544 if (Flame_Breath_Timer <= diff)
545 {
546 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
547 me->SetInFront(target);
548 DoCast(me, SPELL_FLAME_BREATH);
549 Flame_Breath_Timer = 10000;
550 } else Flame_Breath_Timer -= diff;
551 break;
552
553 default:
554 break;
555 }
556
557 if (!TankGUID)
558 DoMeleeAttackIfReady();
559 }
560 };

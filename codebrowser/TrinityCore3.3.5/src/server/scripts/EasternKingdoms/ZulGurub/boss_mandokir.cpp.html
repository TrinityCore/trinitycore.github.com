108class boss_mandokir : public CreatureScript
109{
110 public:
111 boss_mandokir() : CreatureScript("boss_mandokir") { }
112
113 struct boss_mandokirAI : public BossAI
114 {
115 boss_mandokirAI(Creature* creature) : BossAI(creature, DATA_MANDOKIR)
116 {
117 Initialize();
118 }
119
120 void Initialize()
121 {
122 killCount = 0;
123 }
124
125 void Reset() override
126 {
127 if (me->GetPositionZ() > 140.0f)
128 {
129 _Reset();
130 Initialize();
131 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
132 events.ScheduleEvent(EVENT_CHECK_START, 1000);
133 if (Creature* speaker = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_VILEBRANCH_SPEAKER)))
134 if (!speaker->IsAlive())
135 speaker->Respawn(true);
136 }
137 summons.DespawnAll();
138 me->Mount(MODEL_OHGAN_MOUNT);
139 }
140
141 void JustDied(Unit* /*killer*/) override
142 {
143 // Do not want to unsummon Ohgan
144 for (int i = 0; i < CHAINED_SPIRT_COUNT; ++i)
145 if (Creature* unsummon = ObjectAccessor::GetCreature(*me, chainedSpirtGUIDs[i]))
146 unsummon->DespawnOrUnsummon();
147 instance->SetBossState(DATA_MANDOKIR, DONE);
148 instance->SaveToDB();
149 }
150
151 void EnterCombat(Unit* /*who*/) override
152 {
153 _EnterCombat();
154 events.ScheduleEvent(EVENT_OVERPOWER, urand(7000, 9000));
155 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(12000, 18000));
156 events.ScheduleEvent(EVENT_WHIRLWIND, urand(24000, 30000));
157 events.ScheduleEvent(EVENT_CHECK_OHGAN, 1000);
158 events.ScheduleEvent(EVENT_WATCH_PLAYER, urand(13000, 15000));
159 events.ScheduleEvent(EVENT_CHARGE_PLAYER, urand(33000, 38000));
160 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
161 Talk(SAY_AGGRO);
162 me->Dismount();
163 // Summon Ohgan (Spell missing) TEMP HACK
164 me->SummonCreature(NPC_OHGAN, me->GetPositionX()-3, me->GetPositionY(), me->GetPositionZ(), me->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 35000);
165 // Summon Chained Spirits
166 for (int i = 0; i < CHAINED_SPIRT_COUNT; ++i)
167 {
168 Creature* chainedSpirt = me->SummonCreature(NPC_CHAINED_SPIRT, PosSummonChainedSpirits[i], TEMPSUMMON_CORPSE_DESPAWN);
169 chainedSpirtGUIDs[i] = chainedSpirt->GetGUID();
170 }
171 DoZoneInCombat();
172 }
173
174 void KilledUnit(Unit* victim) override
175 {
176 if (victim->GetTypeId() != TYPEID_PLAYER)
177 return;
178
179 if (++killCount == 3)
180 {
181 Talk(SAY_DING_KILL);
182 if (Creature* jindo = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_JINDO)))
183 if (jindo->IsAlive())
184 jindo->AI()->Talk(SAY_GRATS_JINDO);
185 DoCast(me, SPELL_LEVEL_UP, true);
186 killCount = 0;
187 }
188 }
189
190 void MovementInform(uint32 type, uint32 id) override
191 {
192 if (type == WAYPOINT_MOTION_TYPE)
193 {
194 me->SetWalk(false);
195 if (id == POINT_MANDOKIR_END)
196 {
197 me->SetHomePosition(PosMandokir[0]);
198 instance->SetBossState(DATA_MANDOKIR, NOT_STARTED);
199 me->DespawnOrUnsummon(6000); // No idea how to respawn on wipe.
200 }
201 }
202 }
203
204 void UpdateAI(uint32 diff) override
205 {
206 events.Update(diff);
207
208 if (!UpdateVictim())
209 {
210 if (instance->GetBossState(DATA_MANDOKIR) == NOT_STARTED || instance->GetBossState(DATA_MANDOKIR) == SPECIAL)
211 {
212 while (uint32 eventId = events.ExecuteEvent())
213 {
214 switch (eventId)
215 {
216 case EVENT_CHECK_START:
217 if (instance->GetBossState(DATA_MANDOKIR) == SPECIAL)
218 {
219 me->GetMotionMaster()->MovePoint(0, PosMandokir[1].m_positionX, PosMandokir[1].m_positionY, PosMandokir[1].m_positionZ);
220 events.ScheduleEvent(EVENT_STARTED, 6000);
221 }
222 else
223 events.ScheduleEvent(EVENT_CHECK_START, 1000);
224 break;
225 case EVENT_STARTED:
226 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
227 me->GetMotionMaster()->MovePath(PATH_MANDOKIR, false);
228 break;
229 default:
230 break;
231 }
232 }
233 }
234 return;
235 }
236
237 if (me->HasUnitState(UNIT_STATE_CASTING))
238 return;
239
240 while (uint32 eventId = events.ExecuteEvent())
241 {
242 switch (eventId)
243 {
244 case EVENT_OVERPOWER:
245 DoCastVictim(SPELL_OVERPOWER, true);
246 events.ScheduleEvent(EVENT_OVERPOWER, urand(6000, 12000));
247 break;
248 case EVENT_MORTAL_STRIKE:
249 if (me->GetVictim() && me->EnsureVictim()->HealthBelowPct(50))
250 DoCastVictim(SPELL_MORTAL_STRIKE, true);
251 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(12000, 18000));
252 break;
253 case EVENT_WHIRLWIND:
254 DoCast(me, SPELL_WHIRLWIND);
255 events.ScheduleEvent(EVENT_WHIRLWIND, urand(22000, 26000));
256 break;
257 case EVENT_CHECK_OHGAN:
258 if (instance->GetBossState(DATA_OHGAN) == DONE)
259 {
260 DoCast(me, SPELL_FRENZY);
261 Talk(SAY_OHGAN_DEAD);
262 }
263 else
264 events.ScheduleEvent(EVENT_CHECK_OHGAN, 1000);
265 break;
266 case EVENT_WATCH_PLAYER:
267 if (Unit* player = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
268 {
269 DoCast(player, SPELL_WATCH);
270 Talk(SAY_WATCH, player);
271 }
272 events.ScheduleEvent(EVENT_WATCH_PLAYER, urand(12000, 15000));
273 break;
274 case EVENT_CHARGE_PLAYER:
275 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true), SPELL_CHARGE);
276 events.ScheduleEvent(EVENT_CHARGE_PLAYER, urand(22000, 30000));
277 break;
278 default:
279 break;
280 }
281 }
282
283 DoMeleeAttackIfReady();
284 }
285
286 private:
287 uint8 killCount;
288 ObjectGuid chainedSpirtGUIDs[CHAINED_SPIRT_COUNT];
289 };
290
291 CreatureAI* GetAI(Creature* creature) const override
292 {
293 return GetZulGurubAI<boss_mandokirAI>(creature);
294 }

438 {
439 if (Unit* caster = GetCaster())
440 if (Unit* target = GetTarget())
441 if (GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_EXPIRE && GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_DEATH)
442 caster->CastSpell(target, SPELL_WATCH_CHARGE);
443 }
444

446 {
447 OnEffectRemove += AuraEffectRemoveFn(spell_threatening_gaze_AuraScript::OnRemove, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
448 }
449 };

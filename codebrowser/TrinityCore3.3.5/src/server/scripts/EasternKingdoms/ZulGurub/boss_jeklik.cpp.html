79class boss_jeklik : public CreatureScript
80{
81 public: boss_jeklik() : CreatureScript("boss_jeklik") { }
82
83 struct boss_jeklikAI : public BossAI
84 {
85 boss_jeklikAI(Creature* creature) : BossAI(creature, DATA_JEKLIK) { }
86
87 void Reset() override
88 {
89 _Reset();
90 }
91
92 void JustDied(Unit* /*killer*/) override
93 {
94 _JustDied();
95 Talk(SAY_DEATH);
96 }
97
98 void EnterCombat(Unit* /*who*/) override
99 {
100 _EnterCombat();
101 Talk(SAY_AGGRO);
102 events.SetPhase(PHASE_ONE);
103
104 events.ScheduleEvent(EVENT_CHARGE_JEKLIK, 20000, 0, PHASE_ONE);
105 events.ScheduleEvent(EVENT_SONIC_BURST, 8000, 0, PHASE_ONE);
106 events.ScheduleEvent(EVENT_SCREECH, 13000, 0, PHASE_ONE);
107 events.ScheduleEvent(EVENT_SPAWN_BATS, 60000, 0, PHASE_ONE);
108
109 me->SetCanFly(true);
110 DoCast(me, SPELL_BAT_FORM);
111 }
112
113 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
114 {
115 if (events.IsInPhase(PHASE_ONE) && !HealthAbovePct(50))
116 {
117 me->RemoveAurasDueToSpell(SPELL_BAT_FORM);
118 me->SetCanFly(false);
119 DoResetThreat();
120 events.SetPhase(PHASE_TWO);
121 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, 6000, 0, PHASE_TWO);
122 events.ScheduleEvent(EVENT_MIND_FLAY, 11000, 0, PHASE_TWO);
123 events.ScheduleEvent(EVENT_CHAIN_MIND_FLAY, 26000, 0, PHASE_TWO);
124 events.ScheduleEvent(EVENT_GREATER_HEAL, 50000, 0, PHASE_TWO);
125 events.ScheduleEvent(EVENT_SPAWN_FLYING_BATS, 10000, 0, PHASE_TWO);
126 return;
127 }
128 }
129
130 void UpdateAI(uint32 diff) override
131 {
132 if (!UpdateVictim())
133 return;
134
135 events.Update(diff);
136
137 if (me->HasUnitState(UNIT_STATE_CASTING))
138 return;
139
140 while (uint32 eventId = events.ExecuteEvent())
141 {
142 switch (eventId)
143 {
144 case EVENT_CHARGE_JEKLIK:
145 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
146 {
147 DoCast(target, SPELL_CHARGE);
148 AttackStart(target);
149 }
150 events.ScheduleEvent(EVENT_CHARGE_JEKLIK, urand(15000, 30000), 0, PHASE_ONE);
151 break;
152 case EVENT_SONIC_BURST:
153 DoCastVictim(SPELL_SONICBURST);
154 events.ScheduleEvent(EVENT_SONIC_BURST, urand(8000, 13000), 0, PHASE_ONE);
155 break;
156 case EVENT_SCREECH:
157 DoCastVictim(SPELL_SCREECH);
158 events.ScheduleEvent(EVENT_SCREECH, urand(18000, 26000), 0, PHASE_ONE);
159 break;
160 case EVENT_SPAWN_BATS:
161 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
162 for (uint8 i = 0; i < 6; ++i)
163 if (Creature* bat = me->SummonCreature(NPC_BLOODSEEKER_BAT, SpawnBat[i], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
164 bat->AI()->AttackStart(target);
165 events.ScheduleEvent(EVENT_SPAWN_BATS, 60000, 0, PHASE_ONE);
166 break;
167 case EVENT_SHADOW_WORD_PAIN:
168 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
169 DoCast(target, SPELL_SHADOW_WORD_PAIN);
170 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(12000, 18000), 0, PHASE_TWO);
171 break;
172 case EVENT_MIND_FLAY:
173 DoCastVictim(SPELL_MIND_FLAY);
174 events.ScheduleEvent(EVENT_MIND_FLAY, 16000, 0, PHASE_TWO);
175 break;
176 case EVENT_CHAIN_MIND_FLAY:
177 me->InterruptNonMeleeSpells(false);
178 DoCastVictim(SPELL_CHAIN_MIND_FLAY);
179 events.ScheduleEvent(EVENT_CHAIN_MIND_FLAY, urand(15000, 30000), 0, PHASE_TWO);
180 break;
181 case EVENT_GREATER_HEAL:
182 me->InterruptNonMeleeSpells(false);
183 DoCast(me, SPELL_GREATERHEAL);
184 events.ScheduleEvent(EVENT_GREATER_HEAL, urand(25000, 35000), 0, PHASE_TWO);
185 break;
186 case EVENT_SPAWN_FLYING_BATS:
187 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
188 if (Creature* flyingBat = me->SummonCreature(NPC_FRENZIED_BAT, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ() + 15.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
189 flyingBat->AI()->AttackStart(target);
190 events.ScheduleEvent(EVENT_SPAWN_FLYING_BATS, urand(10000, 15000), 0, PHASE_TWO);
191 break;
192 default:
193 break;
194 }
195 }
196
197 DoMeleeAttackIfReady();
198 }
199 };
200
201 CreatureAI* GetAI(Creature* creature) const override
202 {
203 return GetZulGurubAI<boss_jeklikAI>(creature);
204 }

208class npc_batrider : public CreatureScript
209{
210 public:
211 npc_batrider() : CreatureScript("npc_batrider") { }
212
213 struct npc_batriderAI : public ScriptedAI
214 {
215 npc_batriderAI(Creature* creature) : ScriptedAI(creature)
216 {
217 Initialize();
218 }
219
220 void Initialize()
221 {
222 Bomb_Timer = 2000;
223 }
224
225 uint32 Bomb_Timer;
226
227 void Reset() override
228 {
229 Initialize();
230 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
231 }
232
233 void EnterCombat(Unit* /*who*/) override { }
234
235 void UpdateAI(uint32 diff) override
236 {
237 if (!UpdateVictim())
238 return;
239
240 if (Bomb_Timer <= diff)
241 {
242 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
243 {
244 DoCast(target, SPELL_BOMB);
245 Bomb_Timer = 5000;
246 }
247 }
248 else
249 Bomb_Timer -= diff;
250
251 DoMeleeAttackIfReady();
252 }
253 };
254
255 CreatureAI* GetAI(Creature* creature) const override
256 {
257 return new npc_batriderAI(creature);
258 }
259};

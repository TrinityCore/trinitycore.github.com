72class boss_marli : public CreatureScript
73{
74 public: boss_marli() : CreatureScript("boss_marli") { }
75
76 struct boss_marliAI : public BossAI
77 {
78 boss_marliAI(Creature* creature) : BossAI(creature, DATA_MARLI) { }
79
80 void Reset() override
81 {
82 if (events.IsInPhase(PHASE_THREE))
83 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, false); // hack
84 _Reset();
85 }
86
87 void JustDied(Unit* /*killer*/) override
88 {
89 _JustDied();
90 Talk(SAY_DEATH);
91 }
92
93 void EnterCombat(Unit* /*who*/) override
94 {
95 _EnterCombat();
96 events.ScheduleEvent(EVENT_SPAWN_START_SPIDERS, 1000, 0, PHASE_ONE);
97 Talk(SAY_AGGRO);
98 }
99
100 void UpdateAI(uint32 diff) override
101 {
102 if (!UpdateVictim())
103 return;
104
105 events.Update(diff);
106
107 if (me->HasUnitState(UNIT_STATE_CASTING))
108 return;
109
110 while (uint32 eventId = events.ExecuteEvent())
111 {
112 switch (eventId)
113 {
114 case EVENT_SPAWN_START_SPIDERS:
115 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
116 {
117 Talk(SAY_SPIDER_SPAWN);
118 for (uint8 i = 0; i < 4; ++i)
119 if (Creature* spider = me->SummonCreature(NPC_SPIDER, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
120 spider->AI()->AttackStart(target);
121 }
122 events.ScheduleEvent(EVENT_ASPECT_OF_MARLI, 12000, 0, PHASE_TWO);
123 events.ScheduleEvent(EVENT_TRANSFORM, 45000, 0, PHASE_TWO);
124 events.ScheduleEvent(EVENT_POISON_VOLLEY, 15000);
125 events.ScheduleEvent(EVENT_SPAWN_SPIDER, 30000);
126 events.ScheduleEvent(EVENT_TRANSFORM, 45000, 0, PHASE_TWO);
127 events.SetPhase(PHASE_TWO);
128 break;
129 case EVENT_POISON_VOLLEY:
130 DoCastVictim(SPELL_POISON_VOLLEY, true);
131 events.ScheduleEvent(EVENT_POISON_VOLLEY, urand(10000, 20000));
132 break;
133 case EVENT_ASPECT_OF_MARLI:
134 DoCastVictim(SPELL_ASPECT_OF_MARLI, true);
135 events.ScheduleEvent(EVENT_ASPECT_OF_MARLI, urand(13000, 18000), 0, PHASE_TWO);
136 break;
137 case EVENT_SPAWN_SPIDER:
138 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
139 if (Creature* spider = me->SummonCreature(NPC_SPIDER, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
140 spider->AI()->AttackStart(target);
141 events.ScheduleEvent(EVENT_SPAWN_SPIDER, urand(12000, 17000));
142 break;
143 case EVENT_TRANSFORM:
144 {
145 Talk(SAY_TRANSFORM);
146 DoCast(me, SPELL_SPIDER_FORM); // SPELL_AURA_TRANSFORM
147 /*
148 CreatureTemplate const* cinfo = me->GetCreatureTemplate();
149 me->SetBaseWeaponDamage(BASE_ATTACK, MINDAMAGE, (cinfo->mindmg +((cinfo->mindmg/100) * 35)));
150 me->SetBaseWeaponDamage(BASE_ATTACK, MAXDAMAGE, (cinfo->maxdmg +((cinfo->maxdmg/100) * 35)));
151 me->UpdateDamagePhysical(BASE_ATTACK);
152 */
153 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, true); // hack
154 DoCastVictim(SPELL_ENVOLWINGWEB);
155 if (DoGetThreat(me->GetVictim()))
156 DoModifyThreatPercent(me->GetVictim(), -100);
157 events.ScheduleEvent(EVENT_CHARGE_PLAYER, 1500, 0, PHASE_THREE);
158 events.ScheduleEvent(EVENT_TRANSFORM_BACK, 25000, 0, PHASE_THREE);
159 events.SetPhase(PHASE_THREE);
160 break;
161 }
162 case EVENT_CHARGE_PLAYER:
163 {
164 Unit* target = NULL;
165 int i = 0;
166 while (i++ < 3) // max 3 tries to get a random target with power_mana
167 {
168 target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true); // not aggro leader
169 if (target && target->getPowerType() == POWER_MANA)
170 break;
171 }
172 if (target)
173 {
174 DoCast(target, SPELL_CHARGE);
175 AttackStart(target);
176 }
177 events.ScheduleEvent(EVENT_CHARGE_PLAYER, 8000, 0, PHASE_THREE);
178 break;
179 }
180 case EVENT_TRANSFORM_BACK:
181 {
182 me->RemoveAura(SPELL_SPIDER_FORM);
183 /*
184 CreatureTemplate const* cinfo = me->GetCreatureTemplate();
185 me->SetBaseWeaponDamage(BASE_ATTACK, MINDAMAGE, (cinfo->mindmg +((cinfo->mindmg/100) * 1)));
186 me->SetBaseWeaponDamage(BASE_ATTACK, MAXDAMAGE, (cinfo->maxdmg +((cinfo->maxdmg/100) * 1)));
187 me->UpdateDamagePhysical(BASE_ATTACK);
188 */
189 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, false); // hack
190 events.ScheduleEvent(EVENT_ASPECT_OF_MARLI, 12000, 0, PHASE_TWO);
191 events.ScheduleEvent(EVENT_TRANSFORM, 45000, 0, PHASE_TWO);
192 events.ScheduleEvent(EVENT_POISON_VOLLEY, 15000);
193 events.ScheduleEvent(EVENT_SPAWN_SPIDER, 30000);
194 events.ScheduleEvent(EVENT_TRANSFORM, urand(35000, 60000), 0, PHASE_TWO);
195 events.SetPhase(PHASE_TWO);
196 break;
197 }
198 default:
199 break;
200 }
201 }
202
203 DoMeleeAttackIfReady();
204 }
205 };

207 CreatureAI* GetAI(Creature* creature) const override
208 {
209 return new boss_marliAI(creature);
210 }

222 Initialize();
223 }
224
225 void Initialize()
226 {
227 LevelUp_Timer = 3000;
228 }
229
230 uint32 LevelUp_Timer;
231
232 void Reset() override
233 {
234 Initialize();
235 }
236
237 void EnterCombat(Unit* /*who*/) override { }
238
239 void UpdateAI(uint32 diff) override
240 {
241 //Return since we have no target
242 if (!UpdateVictim())
243 return;
244
245 //LevelUp_Timer
246 if (LevelUp_Timer <= diff)
247 {
248 DoCast(me, SPELL_LEVELUP);
249 LevelUp_Timer = 3000;
250 } else LevelUp_Timer -= diff;
251
252 DoMeleeAttackIfReady();
253 }
254 };
255
256 CreatureAI* GetAI(Creature* creature) const override
257 {
258 return new npc_spawn_of_marliAI(creature);
259 }

30enum Spells
31{
32 SPELL_MANABURN = 26046,
33 SPELL_SLEEP = 24664
34};
35
36enum Events
37{
38 EVENT_MANABURN = 1,
39 EVENT_SLEEP = 2,
40 EVENT_ILLUSIONS = 3
41};
42
43class boss_hazzarah : public CreatureScript
44{
45 public:
46 boss_hazzarah() : CreatureScript("boss_hazzarah") { }
47
48 struct boss_hazzarahAI : public BossAI
49 {
50 boss_hazzarahAI(Creature* creature) : BossAI(creature, DATA_EDGE_OF_MADNESS) { }
51
52 void Reset() override
53 {
54 _Reset();
55 }
56
57 void JustDied(Unit* /*killer*/) override
58 {
59 _JustDied();
60 }
61
62 void EnterCombat(Unit* /*who*/) override
63 {
64 _EnterCombat();
65 events.ScheduleEvent(EVENT_MANABURN, urand(4000, 10000));
66 events.ScheduleEvent(EVENT_SLEEP, urand(10000, 18000));
67 events.ScheduleEvent(EVENT_ILLUSIONS, urand(10000, 18000));
68 }
69
70 void UpdateAI(uint32 diff) override
71 {
72 if (!UpdateVictim())
73 return;
74
75 events.Update(diff);
76
77 if (me->HasUnitState(UNIT_STATE_CASTING))
78 return;
79
80 while (uint32 eventId = events.ExecuteEvent())
81 {
82 switch (eventId)
83 {
84 case EVENT_MANABURN:
85 DoCastVictim(SPELL_MANABURN, true);
86 events.ScheduleEvent(EVENT_MANABURN, urand(8000, 16000));
87 break;
88 case EVENT_SLEEP:
89 DoCastVictim(SPELL_SLEEP, true);
90 events.ScheduleEvent(EVENT_SLEEP, urand(12000, 20000));
91 break;
92 case EVENT_ILLUSIONS:
93 // We will summon 3 illusions that will spawn on a random gamer and attack this gamer
94 // We will just use one model for the beginning
95 for (uint8 i = 0; i < 3; ++i)
96 {
97 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
98 if (Creature* Illusion = me->SummonCreature(NPC_NIGHTMARE_ILLUSION, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 30000))
99 Illusion->AI()->AttackStart(target);
100 }
101 events.ScheduleEvent(EVENT_ILLUSIONS, urand(15000, 25000));
102 break;
103 default:
104 break;
105 }
106 }
107
108 DoMeleeAttackIfReady();
109 }
110 };
111
112 CreatureAI* GetAI(Creature* creature) const override
113 {
114 return new boss_hazzarahAI(creature);
115 }
116};
117
118void AddSC_boss_hazzarah()
119{
120 new boss_hazzarah();
121}

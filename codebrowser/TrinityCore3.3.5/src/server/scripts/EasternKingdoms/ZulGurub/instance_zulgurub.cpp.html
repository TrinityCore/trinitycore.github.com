30DoorData const doorData[] =
31{
32 { GO_FORCEFIELD, DATA_ARLOKK, DOOR_TYPE_ROOM },
33 { 0, 0, DOOR_TYPE_ROOM } // END
34};
35
36class instance_zulgurub : public InstanceMapScript
37{
38 public: instance_zulgurub(): InstanceMapScript(ZGScriptName, 309) { }
39
40 struct instance_zulgurub_InstanceMapScript : public InstanceScript
41 {
42 instance_zulgurub_InstanceMapScript(Map* map) : InstanceScript(map)
43 {
44 SetHeaders(DataHeader);
45 SetBossNumber(EncounterCount);
46 LoadDoorData(doorData);
47 }
48
49 bool IsEncounterInProgress() const override
50 {
51 // not active in Zul'Gurub
52 return false;
53 }
54
55 void OnCreatureCreate(Creature* creature) override
56 {
57 switch (creature->GetEntry())
58 {
59 case NPC_ZEALOT_LORKHAN:
60 _zealotLorkhanGUID = creature->GetGUID();
61 break;
62 case NPC_ZEALOT_ZATH:
63 _zealotZathGUID = creature->GetGUID();
64 break;
65 case NPC_HIGH_PRIEST_THEKAL:
66 _highPriestTekalGUID = creature->GetGUID();
67 break;
68 case NPC_JINDO_THE_HEXXER:
69 _jindoTheHexxerGUID = creature->GetGUID();
70 break;
71 case NPC_VILEBRANCH_SPEAKER:
72 _vilebranchSpeakerGUID = creature->GetGUID();
73 break;
74 case NPC_ARLOKK:
75 _arlokkGUID = creature->GetGUID();
76 break;
77 }
78 }
79
80 void OnGameObjectCreate(GameObject* go) override
81 {
82 switch (go->GetEntry())
83 {
84 case GO_FORCEFIELD:
85 AddDoor(go, true);
86 break;
87 case GO_GONG_OF_BETHEKK:
88 _goGongOfBethekkGUID = go->GetGUID();
89 if (GetBossState(DATA_ARLOKK) == DONE)
90 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
91 else
92 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
93 break;
94 default:
95 break;
96 }
97 }
98
99 void OnGameObjectRemove(GameObject* go) override
100 {
101 switch (go->GetEntry())
102 {
103 case GO_FORCEFIELD:
104 AddDoor(go, false);
105 break;
106 default:
107 break;
108 }
109 }
110
111 ObjectGuid GetGuidData(uint32 uiData) const override
112 {
113 switch (uiData)
114 {
115 case DATA_LORKHAN:
116 return _zealotLorkhanGUID;
117 break;
118 case DATA_ZATH:
119 return _zealotZathGUID;
120 break;
121 case DATA_THEKAL:
122 return _highPriestTekalGUID;
123 break;
124 case DATA_JINDO:
125 return _jindoTheHexxerGUID;
126 break;
127 case NPC_ARLOKK:
128 return _arlokkGUID;
129 break;
130 case GO_GONG_OF_BETHEKK:
131 return _goGongOfBethekkGUID;
132 break;
133 }
134 return ObjectGuid::Empty;
135 }
136
137 private:
138 //If all High Priest bosses were killed. Lorkhan, Zath and Ohgan are added too.
139 //Storing Lorkhan, Zath and Thekal because we need to cast on them later. Jindo is needed for healfunction too.
140
141 ObjectGuid _zealotLorkhanGUID;
142 ObjectGuid _zealotZathGUID;
143 ObjectGuid _highPriestTekalGUID;
144 ObjectGuid _jindoTheHexxerGUID;
145 ObjectGuid _vilebranchSpeakerGUID;
146 ObjectGuid _arlokkGUID;
147 ObjectGuid _goGongOfBethekkGUID;
148 };
149
150 InstanceScript* GetInstanceScript(InstanceMap* map) const override
151 {
152 return new instance_zulgurub_InstanceMapScript(map);
153 }
154};
155
156void AddSC_instance_zulgurub()
157{
158 new instance_zulgurub();
159}

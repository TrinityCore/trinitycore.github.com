86class boss_venoxis : public CreatureScript
87{
88 public:
89 boss_venoxis() : CreatureScript("boss_venoxis") { }
90
91 struct boss_venoxisAI : public BossAI
92 {
93 boss_venoxisAI(Creature* creature) : BossAI(creature, DATA_VENOXIS)
94 {
95 Initialize();
96 }
97
98 void Initialize()
99 {
100 _inMeleeRange = 0;
101 _transformed = false;
102 _frenzied = false;
103 }
104
105 void Reset() override
106 {
107 _Reset();
108 // remove all spells and auras from previous attempts
109 me->RemoveAllAuras();
110 me->SetReactState(REACT_PASSIVE);
111 // set some internally used variables to their defaults
112 Initialize();
113 events.SetPhase(PHASE_ONE);
114 }
115
116 void JustDied(Unit* /*killer*/) override
117 {
118 _JustDied();
119 Talk(SAY_VENOXIS_DEATH);
120 me->RemoveAllAuras();
121 }
122
123 void EnterCombat(Unit* /*who*/) override
124 {
125 _EnterCombat();
126 me->SetReactState(REACT_AGGRESSIVE);
127 // Always running events
128 events.ScheduleEvent(EVENT_THRASH, 5000);
129 // Phase one events (regular form)
130 events.ScheduleEvent(EVENT_HOLY_NOVA, 5000, 0, PHASE_ONE);
131 events.ScheduleEvent(EVENT_DISPEL_MAGIC, 35000, 0, PHASE_ONE);
132 events.ScheduleEvent(EVENT_HOLY_FIRE, 10000, 0, PHASE_ONE);
133 events.ScheduleEvent(EVENT_RENEW, 30000, 0, PHASE_ONE);
134 events.ScheduleEvent(EVENT_HOLY_WRATH, 60000, 0, PHASE_ONE);
135
136 events.SetPhase(PHASE_ONE);
137
138 // Set zone in combat
139 DoZoneInCombat();
140 }
141
142 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
143 {
144 // check if venoxis is ready to transform
145 if (!_transformed && !HealthAbovePct(50))
146 {
147 _transformed = true;
148 // schedule the event that changes our phase
149 events.ScheduleEvent(EVENT_TRANSFORM, 100);
150 }
151 // we're losing health, bad, go frenzy
152 else if (!_frenzied && !HealthAbovePct(20))
153 {
154 _frenzied = true;
155 events.ScheduleEvent(EVENT_FRENZY, 100);
156 }
157 }
158
159 void UpdateAI(uint32 diff) override
160 {
161 if (!UpdateVictim())
162 return;
163
164 events.Update(diff);
165
166 // return back to main code if we're still casting
167 if (me->HasUnitState(UNIT_STATE_CASTING))
168 return;
169
170 while (uint32 eventId = events.ExecuteEvent())
171 {
172 switch (eventId)
173 {
174 // thrash is available in all phases
175 case EVENT_THRASH:
176 DoCast(me, SPELL_THRASH, true);
177 events.ScheduleEvent(EVENT_THRASH, urand(10000, 20000));
178 break;
179 // troll form spells and Actions (first part)
180 case EVENT_DISPEL_MAGIC:
181 DoCast(me, SPELL_DISPEL_MAGIC);
182 events.ScheduleEvent(EVENT_DISPEL_MAGIC, urand(15000, 20000), 0, PHASE_ONE);
183 break;
184 case EVENT_RENEW:
185 DoCast(me, SPELL_RENEW);
186 events.ScheduleEvent(EVENT_RENEW, urand(25000, 30000), 0, PHASE_ONE);
187 break;
188 case EVENT_HOLY_NOVA:
189 _inMeleeRange = 0;
190
191 for (uint8 i = 0; i < 10; ++i)
192 {
193 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, i))
194 // check if target is within melee-distance
195 if (me->IsWithinMeleeRange(target))
196 ++_inMeleeRange;
197 }
198
199 // trigger spellcast only if we have 3 or more targets to affect
200 if (_inMeleeRange >= 3)
201 DoCastVictim(SPELL_HOLY_NOVA);
202
203 events.ScheduleEvent(EVENT_HOLY_NOVA, urand(45000, 75000), 0, PHASE_ONE);
204 break;
205 case EVENT_HOLY_FIRE:
206 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
207 DoCast(target, SPELL_HOLY_FIRE);
208 events.ScheduleEvent(EVENT_HOLY_FIRE, urand(45000, 60000), 0, PHASE_ONE);
209 break;
210 case EVENT_HOLY_WRATH:
211 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
212 DoCast(target, SPELL_HOLY_WRATH);
213 events.ScheduleEvent(EVENT_HOLY_WRATH, urand(45000, 60000), 0, PHASE_ONE);
214 break;
215
216 //
217 // snake form spells and Actions
218 //
219
220 case EVENT_VENOM_SPIT:
221 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
222 DoCast(target, SPELL_VENOM_SPIT);
223 events.ScheduleEvent(EVENT_VENOM_SPIT, urand(5000, 15000), 0, PHASE_TWO);
224 break;
225 case EVENT_POISON_CLOUD:
226 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
227 DoCast(target, SPELL_POISON_CLOUD);
228 events.ScheduleEvent(EVENT_POISON_CLOUD, urand(15000, 20000), 0, PHASE_TWO);
229 break;
230 case EVENT_PARASITIC_SERPENT:
231 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
232 DoCast(target, SPELL_SUMMON_PARASITIC_SERPENT);
233 events.ScheduleEvent(EVENT_PARASITIC_SERPENT, 15000, 0, PHASE_TWO);
234 break;
235 case EVENT_FRENZY:
236 // frenzy at 20% health
237 DoCast(me, SPELL_FRENZY, true);
238 break;
239
240 //
241 // shape and phase-changing
242 //
243
244 case EVENT_TRANSFORM:
245 // shapeshift at 50% health
246 DoCast(me, SPELL_VENOXIS_TRANSFORM);
247 Talk(SAY_VENOXIS_TRANSFORM);
248 DoResetThreat();
249
250 // phase two events (snakeform)
251 events.ScheduleEvent(EVENT_VENOM_SPIT, 5000, 0, PHASE_TWO);
252 events.ScheduleEvent(EVENT_POISON_CLOUD, 10000, 0, PHASE_TWO);
253 events.ScheduleEvent(EVENT_PARASITIC_SERPENT, 30000, 0, PHASE_TWO);
254
255 // transformed, start phase two
256 events.SetPhase(PHASE_TWO);
257
258 break;
259 default:
260 break;
261 }
262 }
263
264 DoMeleeAttackIfReady();
265 }
266
267 private:
268 uint8 _inMeleeRange;
269 bool _transformed;
270 bool _frenzied;
271 };
272
273 CreatureAI* GetAI(Creature* creature) const override
274 {
275 return new boss_venoxisAI(creature);
276 }
277};

67class boss_hakkar : public CreatureScript
68{
69 public:
70 boss_hakkar() : CreatureScript("boss_hakkar") { }
71
72 struct boss_hakkarAI : public BossAI
73 {
74 boss_hakkarAI(Creature* creature) : BossAI(creature, DATA_HAKKAR) { }
75
76 void Reset() override
77 {
78 _Reset();
79 }
80
81 void JustDied(Unit* /*killer*/) override
82 {
83 _JustDied();
84 }
85
86 void EnterCombat(Unit* /*who*/) override
87 {
88 _EnterCombat();
89 events.ScheduleEvent(EVENT_BLOOD_SIPHON, 90000);
90 events.ScheduleEvent(EVENT_CORRUPTED_BLOOD, 25000);
91 events.ScheduleEvent(EVENT_CAUSE_INSANITY, 17000);
92 events.ScheduleEvent(EVENT_WILL_OF_HAKKAR, 17000);
93 events.ScheduleEvent(EVENT_ENRAGE, 600000);
94 if (instance->GetBossState(DATA_JEKLIK) != DONE)
95 events.ScheduleEvent(EVENT_ASPECT_OF_JEKLIK, 4000);
96 if (instance->GetBossState(DATA_VENOXIS) != DONE)
97 events.ScheduleEvent(EVENT_ASPECT_OF_VENOXIS, 7000);
98 if (instance->GetBossState(DATA_MARLI) != DONE)
99 events.ScheduleEvent(EVENT_ASPECT_OF_MARLI, 12000);
100 if (instance->GetBossState(DATA_THEKAL) != DONE)
101 events.ScheduleEvent(EVENT_ASPECT_OF_THEKAL, 8000);
102 if (instance->GetBossState(DATA_ARLOKK) != DONE)
103 events.ScheduleEvent(EVENT_ASPECT_OF_ARLOKK, 18000);
104 Talk(SAY_AGGRO);
105 }
106
107 void UpdateAI(uint32 diff) override
108 {
109 if (!UpdateVictim())
110 return;
111
112 events.Update(diff);
113
114 if (me->HasUnitState(UNIT_STATE_CASTING))
115 return;
116
117 while (uint32 eventId = events.ExecuteEvent())
118 {
119 switch (eventId)
120 {
121 case EVENT_BLOOD_SIPHON:
122 DoCastVictim(SPELL_BLOOD_SIPHON, true);
123 events.ScheduleEvent(EVENT_BLOOD_SIPHON, 90000);
124 break;
125 case EVENT_CORRUPTED_BLOOD:
126 DoCastVictim(SPELL_CORRUPTED_BLOOD, true);
127 events.ScheduleEvent(EVENT_CORRUPTED_BLOOD, urand(30000, 45000));
128 break;
129 case EVENT_CAUSE_INSANITY:
130 // DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_CAUSE_INSANITY);
131 // events.ScheduleEvent(EVENT_CAUSE_INSANITY, urand(35000, 45000));
132 break;
133 case EVENT_WILL_OF_HAKKAR:
134 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_WILL_OF_HAKKAR);
135 events.ScheduleEvent(EVENT_WILL_OF_HAKKAR, urand(25000, 35000));
136 break;
137 case EVENT_ENRAGE:
138 if (!me->HasAura(SPELL_ENRAGE))
139 DoCast(me, SPELL_ENRAGE);
140 events.ScheduleEvent(EVENT_ENRAGE, 90000);
141 break;
142 case EVENT_ASPECT_OF_JEKLIK:
143 DoCastVictim(SPELL_ASPECT_OF_JEKLIK, true);
144 events.ScheduleEvent(EVENT_ASPECT_OF_JEKLIK, urand(10000, 14000));
145 break;
146 case EVENT_ASPECT_OF_VENOXIS:
147 DoCastVictim(SPELL_ASPECT_OF_VENOXIS, true);
148 events.ScheduleEvent(EVENT_ASPECT_OF_VENOXIS, 8000);
149 break;
150 case EVENT_ASPECT_OF_MARLI:
151 DoCastVictim(SPELL_ASPECT_OF_MARLI, true);
152 events.ScheduleEvent(EVENT_ASPECT_OF_MARLI, 10000);
153 break;
154 case EVENT_ASPECT_OF_THEKAL:
155 DoCastVictim(SPELL_ASPECT_OF_THEKAL, true);
156 events.ScheduleEvent(EVENT_ASPECT_OF_THEKAL, 15000);
157 break;
158 case EVENT_ASPECT_OF_ARLOKK:
159 DoCastVictim(SPELL_ASPECT_OF_ARLOKK, true);
160 events.ScheduleEvent(EVENT_ASPECT_OF_ARLOKK, urand(10000, 15000));
161 break;
162 default:
163 break;
164 }
165 }
166
167 DoMeleeAttackIfReady();
168 }
169 };
170
171 CreatureAI* GetAI(Creature* creature) const override
172 {
173 return GetInstanceAI<boss_hakkarAI>(creature);
174 }
175};

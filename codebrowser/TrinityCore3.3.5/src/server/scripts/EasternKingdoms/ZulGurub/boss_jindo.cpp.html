68class boss_jindo : public CreatureScript
69{
70 public:
71 boss_jindo() : CreatureScript("boss_jindo") { }
72
73 struct boss_jindoAI : public BossAI
74 {
75 boss_jindoAI(Creature* creature) : BossAI(creature, DATA_JINDO) { }
76
77 void Reset() override
78 {
79 _Reset();
80 }
81
82 void JustDied(Unit* /*killer*/) override
83 {
84 _JustDied();
85 }
86
87 void EnterCombat(Unit* /*who*/) override
88 {
89 _EnterCombat();
90 events.ScheduleEvent(EVENT_BRAIN_WASH_TOTEM, 20000);
91 events.ScheduleEvent(EVENT_POWERFULL_HEALING_WARD, 16000);
92 events.ScheduleEvent(EVENT_HEX, 8000);
93 events.ScheduleEvent(EVENT_DELUSIONS_OF_JINDO, 10000);
94 events.ScheduleEvent(EVENT_TELEPORT, 5000);
95 Talk(SAY_AGGRO);
96 }
97
98 void UpdateAI(uint32 diff) override
99 {
100 if (!UpdateVictim())
101 return;
102
103 events.Update(diff);
104
105 if (me->HasUnitState(UNIT_STATE_CASTING))
106 return;
107
108 while (uint32 eventId = events.ExecuteEvent())
109 {
110 switch (eventId)
111 {
112 case EVENT_BRAIN_WASH_TOTEM:
113 DoCast(me, SPELL_BRAIN_WASH_TOTEM);
114 events.ScheduleEvent(EVENT_BRAIN_WASH_TOTEM, urand(18000, 26000));
115 break;
116 case EVENT_POWERFULL_HEALING_WARD:
117 DoCast(me, SPELL_POWERFULL_HEALING_WARD);
118 events.ScheduleEvent(EVENT_POWERFULL_HEALING_WARD, urand(14000, 20000));
119 break;
120 case EVENT_HEX:
121 if (Unit* target = me->GetVictim())
122 {
123 DoCast(target, SPELL_HEX, true);
124 if (DoGetThreat(target))
125 DoModifyThreatPercent(target, -80);
126 }
127 events.ScheduleEvent(EVENT_HEX, urand(12000, 20000));
128 break;
129 case EVENT_DELUSIONS_OF_JINDO:
130 // Casting the delusion curse with a shade so shade will attack the same target with the curse.
131 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
132 {
133 DoCast(target, SPELL_SHADE_OF_JINDO, true);
134 DoCast(target, SPELL_DELUSIONS_OF_JINDO);
135 }
136 events.ScheduleEvent(EVENT_DELUSIONS_OF_JINDO, urand(4000, 12000));
137 break;
138 case EVENT_TELEPORT:
139 // Teleports a random player and spawns 9 Sacrificed Trolls to attack player
140 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
141 {
142 DoTeleportPlayer(target, TeleportLoc.GetPositionX(), TeleportLoc.GetPositionY(), TeleportLoc.GetPositionZ(), TeleportLoc.GetOrientation());
143 if (DoGetThreat(me->GetVictim()))
144 DoModifyThreatPercent(target, -100);
145
146 // Summon a formation of trolls
147 for (uint8 i = 0; i < 10; ++i)
148 if (Creature* SacrificedTroll = me->SummonCreature(NPC_SACRIFICED_TROLL, Formation[i].GetPositionX(), Formation[i].GetPositionY(), Formation[i].GetPositionZ(), Formation[i].GetOrientation(), TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
149 SacrificedTroll->AI()->AttackStart(target);
150 }
151 events.ScheduleEvent(EVENT_TELEPORT, urand(15000, 23000));
152 break;
153 default:
154 break;
155 }
156 }
157
158 DoMeleeAttackIfReady();
159 }
160 };
161
162 CreatureAI* GetAI(Creature* creature) const override
163 {
164 return new boss_jindoAI(creature);
165 }

169class npc_healing_ward : public CreatureScript
170{
171 public:
172 npc_healing_ward() : CreatureScript("npc_healing_ward") { }
173
174 struct npc_healing_wardAI : public ScriptedAI
175 {
176 npc_healing_wardAI(Creature* creature) : ScriptedAI(creature)
177 {
178 Initialize();
179 instance = creature->GetInstanceScript();
180 }
181
182 void Initialize()
183 {
184 Heal_Timer = 2000;
185 }
186
187 uint32 Heal_Timer;
188 InstanceScript* instance;
189
190 void Reset() override
191 {
192 Initialize();
193 }
194
195 void EnterCombat(Unit* /*who*/) override { }
196
197 void UpdateAI(uint32 diff) override
198 {
199 // Heal_Timer
200 if (Heal_Timer <= diff)
201 {
202 if (Unit* jindo = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_JINDO)))
203 DoCast(jindo, SPELL_HEAL);
204 Heal_Timer = 3000;
205 } else Heal_Timer -= diff;
206
207 DoMeleeAttackIfReady();
208 }
209 };
210
211 CreatureAI* GetAI(Creature* creature) const override
212 {
213 return GetInstanceAI<npc_healing_wardAI>(creature);
214 }
215};

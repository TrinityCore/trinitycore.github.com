79class boss_thekal : public CreatureScript

81 public:
82 boss_thekal() : CreatureScript("boss_thekal") { }

84 struct boss_thekalAI : public BossAI
85 {
86 boss_thekalAI(Creature* creature) : BossAI(creature, DATA_THEKAL)
87 {
88 Initialize();
89 }
90
91 void Initialize()
92 {
93 Enraged = false;
94 WasDead = false;
95 }
96
97 bool Enraged;
98 bool WasDead;
99
100 void Reset() override
101 {
102 if (events.IsInPhase(PHASE_TWO))
103 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, false); // hack
104 _Reset();
105 Initialize();
106 }
107
108 void JustDied(Unit* /*killer*/) override
109 {
110 _JustDied();
111 Talk(SAY_DEATH);
112 }
113
114 void EnterCombat(Unit* /*who*/) override
115 {
116 _EnterCombat();
117 events.ScheduleEvent(EVENT_MORTALCLEAVE, 4000, 0, PHASE_ONE); // Phase 1
118 events.ScheduleEvent(EVENT_SILENCE, 9000, 0, PHASE_ONE); // Phase 1
119 events.ScheduleEvent(EVENT_CHECK_TIMER, 10000, 0, PHASE_ONE); // Phase 1
120 events.ScheduleEvent(EVENT_RESURRECT_TIMER, 10000, 0, PHASE_ONE); // Phase 1
121 Talk(SAY_AGGRO);
122 }
123
124 void JustReachedHome() override
125 {
126 instance->SetBossState(DATA_THEKAL, NOT_STARTED);
127 }
128
129 void UpdateAI(uint32 diff) override
130 {
131 if (!UpdateVictim())
132 return;
133
134 events.Update(diff);
135
136 if (me->HasUnitState(UNIT_STATE_CASTING))
137 return;
138
139 while (uint32 eventId = events.ExecuteEvent())
140 {
141 switch (eventId)
142 {
143 case EVENT_MORTALCLEAVE:
144 DoCastVictim(SPELL_MORTALCLEAVE, true);
145 events.ScheduleEvent(EVENT_MORTALCLEAVE, urand(15000, 20000), 0, PHASE_ONE);
146 break;
147 case EVENT_SILENCE:
148 DoCastVictim(SPELL_SILENCE, true);
149 events.ScheduleEvent(EVENT_SILENCE, urand(20000, 25000), 0, PHASE_ONE);
150 break;
151 case EVENT_RESURRECT_TIMER:
152 //Thekal will transform to Tiger if he died and was not resurrected after 10 seconds.
153 if (WasDead)
154 {
155 DoCast(me, SPELL_TIGER_FORM); // SPELL_AURA_TRANSFORM
156 me->SetObjectScale(2.00f);
157 me->SetStandState(UNIT_STAND_STATE_STAND);
158 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
159 /*
160 const CreatureTemplate* cinfo = me->GetCreatureTemplate();
161 me->SetBaseWeaponDamage(BASE_ATTACK, MINDAMAGE, (cinfo->mindmg +((cinfo->mindmg/100) * 40)));
162 me->SetBaseWeaponDamage(BASE_ATTACK, MAXDAMAGE, (cinfo->maxdmg +((cinfo->maxdmg/100) * 40)));
163 me->UpdateDamagePhysical(BASE_ATTACK);
164 */
165 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 40.0f, true); // hack
166 DoResetThreat();
167 events.ScheduleEvent(EVENT_FRENZY, 30000, 0, PHASE_TWO); // Phase 2
168 events.ScheduleEvent(EVENT_FORCEPUNCH, 4000, 0, PHASE_TWO); // Phase 2
169 events.ScheduleEvent(EVENT_SPELL_CHARGE, 12000, 0, PHASE_TWO); // Phase 2
170 events.ScheduleEvent(EVENT_ENRAGE, 32000, 0, PHASE_TWO); // Phase 2
171 events.ScheduleEvent(EVENT_SUMMONTIGERS, 25000, 0, PHASE_TWO); // Phase 2
172 events.SetPhase(PHASE_TWO);
173 }
174 events.ScheduleEvent(EVENT_RESURRECT_TIMER, 10000, 0, PHASE_ONE);
175 break;
176 case EVENT_CHECK_TIMER:
177 //Check_Timer for the death of LorKhan and Zath.
178 if (!WasDead)
179 {
180 if (instance->GetBossState(DATA_LORKHAN) == SPECIAL)
181 {
182 //Resurrect LorKhan
183 if (Unit* pLorKhan = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_LORKHAN)))
184 {
185 pLorKhan->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
186 pLorKhan->setFaction(14);
187 pLorKhan->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
188 pLorKhan->SetFullHealth();
189 instance->SetData(DATA_LORKHAN, DONE);
190 }
191 }
192
193 if (instance->GetBossState(DATA_ZATH) == SPECIAL)
194 {
195 //Resurrect Zath
196 if (Unit* pZath = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_ZATH)))
197 {
198 pZath->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
199 pZath->setFaction(14);
200 pZath->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
201 pZath->SetFullHealth();
202 instance->SetBossState(DATA_ZATH, DONE);
203 }
204 }
205 }
206 events.ScheduleEvent(EVENT_CHECK_TIMER, 5000, 0, PHASE_ONE);
207 break;
208 case EVENT_FRENZY:
209 DoCast(me, SPELL_FRENZY);
210 events.ScheduleEvent(EVENT_FRENZY, 30000, 0, PHASE_TWO);
211 break;
212 case EVENT_FORCEPUNCH:
213 DoCastVictim(SPELL_FORCEPUNCH, true);
214 events.ScheduleEvent(EVENT_FORCEPUNCH, urand(16000, 21000), 0, PHASE_TWO);
215 break;
216 case EVENT_CHARGE:
217 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
218 {
219 DoCast(target, SPELL_CHARGE);
220 DoResetThreat();
221 AttackStart(target);
222 }
223 events.ScheduleEvent(EVENT_CHARGE, urand(15000, 22000), 0, PHASE_TWO);
224 break;
225 case EVENT_ENRAGE:
226 if (HealthBelowPct(11) && !Enraged)
227 {
228 DoCast(me, SPELL_ENRAGE);
229 Enraged = true;
230 }
231 events.ScheduleEvent(EVENT_ENRAGE, 30000);
232 break;
233 case EVENT_SUMMONTIGERS:
234 DoCastVictim(SPELL_SUMMONTIGERS, true);
235 events.ScheduleEvent(EVENT_SUMMONTIGERS, urand(10000, 14000), 0, PHASE_TWO);
236 break;
237 default:
238 break;
239 }
240
241 if (me->IsFullHealth() && WasDead)
242 WasDead = false;
243
244 if ((events.IsInPhase(PHASE_ONE)) && !WasDead && !HealthAbovePct(5))
245 {
246 me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE_PERCENT);
247 me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE);
248 me->RemoveAurasByType(SPELL_AURA_PERIODIC_LEECH);
249 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
250 me->SetStandState(UNIT_STAND_STATE_SLEEP);
251 me->AttackStop();
252 instance->SetBossState(DATA_THEKAL, SPECIAL);
253 WasDead=true;
254 }
255 }
256 DoMeleeAttackIfReady();
257 }
258 };
259
260 CreatureAI* GetAI(Creature* creature) const override
261 {
262 return GetInstanceAI<boss_thekalAI>(creature);
263 }

279 void Initialize()
280 {
281 Shield_Timer = 1000;

286

288 }
289
290 uint32 Shield_Timer;
291 uint32 BloodLust_Timer;
292 uint32 GreaterHeal_Timer;
293 uint32 Disarm_Timer;
294 uint32 Check_Timer;
295
296 bool FakeDeath;
297
298 InstanceScript* instance;
299
300 void Reset() override
301 {
302 Initialize();
303
304 instance->SetBossState(DATA_LORKHAN, NOT_STARTED);
305
306 me->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
307 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
308 }
309
310 void EnterCombat(Unit* /*who*/) override
311 {
312 }
313
314 void UpdateAI(uint32 diff) override
315 {
316 if (!UpdateVictim())
317 return;
318
319 //Shield_Timer
320 if (Shield_Timer <= diff)
321 {
322 DoCast(me, SPELL_SHIELD);
323 Shield_Timer = 61000;
324 } else Shield_Timer -= diff;
325
326 //BloodLust_Timer
327 if (BloodLust_Timer <= diff)
328 {
329 DoCast(me, SPELL_BLOODLUST);
330 BloodLust_Timer = 20000 + rand32() % 8000;
331 } else BloodLust_Timer -= diff;
332
333 //Casting Greaterheal to Thekal or Zath if they are in meele range.
334 if (GreaterHeal_Timer <= diff)
335 {
336 Unit* pThekal = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_THEKAL));
337 Unit* pZath = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_ZATH));
338
339 if (!pThekal || !pZath)
340 return;
341
342 switch (urand(0, 1))
343 {
344 case 0:
345 if (me->IsWithinMeleeRange(pThekal))
346 DoCast(pThekal, SPELL_GREATERHEAL);
347 break;
348 case 1:
349 if (me->IsWithinMeleeRange(pZath))
350 DoCast(pZath, SPELL_GREATERHEAL);
351 break;
352 }
353
354 GreaterHeal_Timer = 15000 + rand32() % 5000;
355 } else GreaterHeal_Timer -= diff;
356
357 //Disarm_Timer
358 if (Disarm_Timer <= diff)
359 {
360 DoCastVictim(SPELL_DISARM);
361 Disarm_Timer = 15000 + rand32() % 10000;
362 } else Disarm_Timer -= diff;
363
364 //Check_Timer for the death of LorKhan and Zath.
365 if (!FakeDeath && Check_Timer <= diff)
366 {
367 if (instance->GetBossState(DATA_THEKAL) == SPECIAL)
368 {
369 //Resurrect Thekal
370 if (Unit* pThekal = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_THEKAL)))
371 {
372 pThekal->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
373 pThekal->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
374 pThekal->setFaction(14);
375 pThekal->SetFullHealth();
376 }
377 }
378
379 if (instance->GetBossState(DATA_ZATH) == SPECIAL)
380 {
381 //Resurrect Zath
382 if (Unit* pZath = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_ZATH)))
383 {
384 pZath->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
385 pZath->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
386 pZath->setFaction(14);
387 pZath->SetFullHealth();
388 }
389 }
390
391 Check_Timer = 5000;
392 } else Check_Timer -= diff;
393
394 if (!HealthAbovePct(5))
395 {
396 me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE_PERCENT);
397 me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE);
398 me->RemoveAurasByType(SPELL_AURA_PERIODIC_LEECH);
399 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
400 me->SetStandState(UNIT_STAND_STATE_SLEEP);
401 me->setFaction(35);
402 me->AttackStop();
403
404 instance->SetBossState(DATA_LORKHAN, SPECIAL);
405
406 FakeDeath = true;
407 }
408
409 DoMeleeAttackIfReady();
410 }
411 };

415 return GetInstanceAI<npc_zealot_lorkhanAI>(creature);

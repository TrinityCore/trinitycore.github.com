93 struct boss_arlokkAI : public BossAI
94 {
95 boss_arlokkAI(Creature* creature) : BossAI(creature, DATA_ARLOKK)
96 {
97 Initialize();
98 }
99
100 void Initialize()
101 {
102 _summonCountA = 0;
103 _summonCountB = 0;
104 }
105
106 void Reset() override
107 {
108 if (events.IsInPhase(PHASE_TWO))
109 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, false); // hack
110 _Reset();
111 Initialize();
112 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_DAGGER));
113 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, uint32(WEAPON_DAGGER));
114 me->SetWalk(false);
115 me->GetMotionMaster()->MovePoint(0, PosMoveOnSpawn[0]);
116 }
117
118 void JustDied(Unit* /*killer*/) override
119 {
120 _JustDied();
121 Talk(SAY_DEATH);
122 }
123
124 void EnterCombat(Unit* /*who*/) override
125 {
126 _EnterCombat();
127 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(7000, 9000), 0, PHASE_ONE);
128 events.ScheduleEvent(EVENT_GOUGE, urand(12000, 15000), 0, PHASE_ONE);
129 events.ScheduleEvent(EVENT_SUMMON_PROWLERS, 6000, 0, PHASE_ALL);
130 events.ScheduleEvent(EVENT_MARK_OF_ARLOKK, urand(9000, 11000), 0, PHASE_ALL);
131 events.ScheduleEvent(EVENT_TRANSFORM, urand(15000, 20000), 0, PHASE_ONE);
132 Talk(SAY_AGGRO);
133
134 // Sets up list of Panther spawners to cast on
135 std::list<Creature*> triggerList;
136 GetCreatureListWithEntryInGrid(triggerList, me, NPC_PANTHER_TRIGGER, 100.0f);
137 if (!triggerList.empty())
138 {
139 uint8 sideA = 0;
140 uint8 sideB = 0;
141 for (std::list<Creature*>::const_iterator itr = triggerList.begin(); itr != triggerList.end(); ++itr)
142 {
143 if (Creature* trigger = *itr)
144 {
145 if (trigger->GetPositionY() < -1625.0f)
146 {
147 _triggersSideAGUID[sideA] = trigger->GetGUID();
148 ++sideA;
149 }
150 else
151 {
152 _triggersSideBGUID[sideB] = trigger->GetGUID();
153 ++sideB;
154 }
155 }
156 }
157 }
158 }
159
160 void EnterEvadeMode(EvadeReason why) override
161 {
162 BossAI::EnterEvadeMode(why);
163 if (GameObject* object = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(GO_GONG_OF_BETHEKK)))
164 object->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
165 me->DespawnOrUnsummon(4000);
166 }
167
168 void SetData(uint32 id, uint32 /*value*/) override
169 {
170 if (id == 1)
171 --_summonCountA;
172 else if (id == 2)
173 --_summonCountB;
174 }
175
176 void UpdateAI(uint32 diff) override
177 {
178 if (!UpdateVictim())
179 return;
180
181 events.Update(diff);
182
183 if (me->HasUnitState(UNIT_STATE_CASTING))
184 return;
185
186 while (uint32 eventId = events.ExecuteEvent())
187 {
188 switch (eventId)
189 {
190 case EVENT_SHADOW_WORD_PAIN:
191 DoCastVictim(SPELL_SHADOW_WORD_PAIN, true);
192 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(5000, 7000), 0, PHASE_ONE);
193 break;
194 case EVENT_GOUGE:
195 DoCastVictim(SPELL_GOUGE, true);
196 break;
197 case EVENT_SUMMON_PROWLERS:
198 if (_summonCountA < MAX_PROWLERS_PER_SIDE)
199 {
200 if (Unit* trigger = ObjectAccessor::GetUnit(*me, _triggersSideAGUID[urand(0, 4)]))
201 {
202 trigger->CastSpell(trigger, SPELL_SUMMON_PROWLER);
203 ++_summonCountA;
204 }
205 }
206 if (_summonCountB < MAX_PROWLERS_PER_SIDE)
207 {
208 if (Unit* trigger = ObjectAccessor::GetUnit(*me, _triggersSideBGUID[urand(0, 4)]))
209 {
210 trigger->CastSpell(trigger, SPELL_SUMMON_PROWLER);
211 ++_summonCountB;
212 }
213 }
214 events.ScheduleEvent(EVENT_SUMMON_PROWLERS, 6000, 0, PHASE_ALL);
215 break;
216 case EVENT_MARK_OF_ARLOKK:
217 {
218 Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, urand(1, 3), 0.0f, false, -SPELL_MARK_OF_ARLOKK);
219 if (!target)
220 target = me->GetVictim();
221 if (target)
222 {
223 DoCast(target, SPELL_MARK_OF_ARLOKK, true);
224 Talk(SAY_FEAST_PROWLER, target);
225 }
226 events.ScheduleEvent(EVENT_MARK_OF_ARLOKK, urand(120000, 130000));
227 break;
228 }
229 case EVENT_TRANSFORM:
230 {
231 DoCast(me, SPELL_PANTHER_TRANSFORM); // SPELL_AURA_TRANSFORM
232 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(EQUIP_UNEQUIP));
233 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, uint32(EQUIP_UNEQUIP));
234 /*
235 const CreatureTemplate* cinfo = me->GetCreatureTemplate();
236 me->SetBaseWeaponDamage(BASE_ATTACK, MINDAMAGE, (cinfo->mindmg +((cinfo->mindmg/100) * 35)));
237 me->SetBaseWeaponDamage(BASE_ATTACK, MAXDAMAGE, (cinfo->maxdmg +((cinfo->maxdmg/100) * 35)));
238 me->UpdateDamagePhysical(BASE_ATTACK);
239 */
240 me->AttackStop();
241 DoResetThreat();
242 me->SetReactState(REACT_PASSIVE);
243 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
244 DoCast(me, SPELL_VANISH_VISUAL);
245 DoCast(me, SPELL_VANISH);
246 events.ScheduleEvent(EVENT_VANISH, 1000, 0, PHASE_ONE);
247 break;
248 }
249 case EVENT_VANISH:
250 DoCast(me, SPELL_SUPER_INVIS);
251 me->SetWalk(false);
252 me->GetMotionMaster()->MovePoint(0, frand(-11551.0f, -11508.0f), frand(-1638.0f, -1617.0f), me->GetPositionZ());
253 events.ScheduleEvent(EVENT_VANISH_2, 9000, 0, PHASE_ONE);
254 break;
255 case EVENT_VANISH_2:
256 DoCast(me, SPELL_VANISH);
257 DoCast(me, SPELL_SUPER_INVIS);
258 events.ScheduleEvent(EVENT_VISIBLE, urand(7000, 10000), 0, PHASE_ONE);
259 break;
260 case EVENT_VISIBLE:
261 me->SetReactState(REACT_AGGRESSIVE);
262 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
263 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
264 AttackStart(target);
265 me->RemoveAura(SPELL_SUPER_INVIS);
266 me->RemoveAura(SPELL_VANISH);
267 events.ScheduleEvent(EVENT_RAVAGE, urand(10000, 14000), 0, PHASE_TWO);
268 events.ScheduleEvent(EVENT_TRANSFORM_BACK, urand(15000, 18000), 0, PHASE_TWO);
269 events.SetPhase(PHASE_TWO);
270 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, true); // hack
271 break;
272 case EVENT_RAVAGE:
273 DoCastVictim(SPELL_RAVAGE, true);
274 events.ScheduleEvent(EVENT_RAVAGE, urand(10000, 14000), 0, PHASE_TWO);
275 break;
276 case EVENT_TRANSFORM_BACK:
277 {
278 me->RemoveAura(SPELL_PANTHER_TRANSFORM); // SPELL_AURA_TRANSFORM
279 DoCast(me, SPELL_VANISH_VISUAL);
280 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_DAGGER));
281 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, uint32(WEAPON_DAGGER));
282 /*
283 const CreatureTemplate* cinfo = me->GetCreatureTemplate();
284 me->SetBaseWeaponDamage(BASE_ATTACK, MINDAMAGE, (cinfo->mindmg));
285 me->SetBaseWeaponDamage(BASE_ATTACK, MAXDAMAGE, (cinfo->maxdmg));
286 me->UpdateDamagePhysical(BASE_ATTACK);
287 */
288 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, false); // hack
289 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(4000, 7000), 0, PHASE_ONE);
290 events.ScheduleEvent(EVENT_GOUGE, urand(12000, 15000), 0, PHASE_ONE);
291 events.ScheduleEvent(EVENT_TRANSFORM, urand(16000, 20000), 0, PHASE_ONE);
292 events.SetPhase(PHASE_ONE);
293 break;
294 }
295 default:
296 break;
297 }
298 }
299
300 DoMeleeAttackIfReady();
301 }
302
303 private:
304 uint8 _summonCountA;
305 uint8 _summonCountB;
306 ObjectGuid _triggersSideAGUID[5];
307 ObjectGuid _triggersSideBGUID[5];
308 };

312 return GetZulGurubAI<boss_arlokkAI>(creature);
313 }

337class npc_zulian_prowler : public CreatureScript
338{
339 public: npc_zulian_prowler() : CreatureScript("npc_zulian_prowler") { }
340
341 struct npc_zulian_prowlerAI : public ScriptedAI
342 {
343 npc_zulian_prowlerAI(Creature* creature) : ScriptedAI(creature), _instance(creature->GetInstanceScript())
344 {
345 _sideData = 0;
346 }
347
348 void Reset() override
349 {
350 if (me->GetPositionY() < -1625.0f)
351 _sideData = 1;
352 else
353 _sideData = 2;
354
355 DoCast(me, SPELL_SNEAK_RANK_1_1);
356 DoCast(me, SPELL_SNEAK_RANK_1_2);
357
358 if (Unit* arlokk = ObjectAccessor::GetUnit(*me, _instance->GetGuidData(NPC_ARLOKK)))
359 me->GetMotionMaster()->MovePoint(0, arlokk->GetPositionX(), arlokk->GetPositionY(), arlokk->GetPositionZ());
360 _events.ScheduleEvent(EVENT_ATTACK, 6000);
361 }
362
363 void EnterCombat(Unit* /*who*/) override
364 {
365 me->GetMotionMaster()->Clear(false);
366 me->RemoveAura(SPELL_SNEAK_RANK_1_1);
367 me->RemoveAura(SPELL_SNEAK_RANK_1_2);
368 }
369
370 void SpellHit(Unit* caster, SpellInfo const* spell) override
371 {
372 if (spell->Id == SPELL_MARK_OF_ARLOKK_TRIGGER) // Should only hit if line of sight
373 me->Attack(caster, true);
374 }
375
376 void JustDied(Unit* /*killer*/) override
377 {
378 if (Unit* arlokk = ObjectAccessor::GetUnit(*me, _instance->GetGuidData(NPC_ARLOKK)))
379 {
380 if (arlokk->IsAlive())
381 arlokk->GetAI()->SetData(_sideData, 0);
382 }
383 me->DespawnOrUnsummon(4000);
384 }
385
386 void UpdateAI(uint32 diff) override
387 {
388 if (UpdateVictim())
389 {
390 DoMeleeAttackIfReady();
391 return;
392 }
393
394 _events.Update(diff);
395
396 while (uint32 eventId = _events.ExecuteEvent())
397 {
398 switch (eventId)
399 {
400 case EVENT_ATTACK:
401 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0.0f, 100, false))
402 me->Attack(target, true);
403 break;
404 default:
405 break;
406 }
407 }
408 }
409
410 private:
411 int32 _sideData;
412 EventMap _events;
413 InstanceScript* _instance;
414 };
415
416 CreatureAI* GetAI(Creature* creature) const override
417 {
418 return GetZulGurubAI<npc_zulian_prowlerAI>(creature);
419 }
420};

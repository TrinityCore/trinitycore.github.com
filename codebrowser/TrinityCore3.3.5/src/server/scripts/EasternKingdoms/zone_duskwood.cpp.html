47/*######
48# boss_twilight_corrupter
49######*/
50
51class boss_twilight_corrupter : public CreatureScript
52{
53public:
54 boss_twilight_corrupter() : CreatureScript("boss_twilight_corrupter") { }
55
56 struct boss_twilight_corrupterAI : public ScriptedAI
57 {
58 boss_twilight_corrupterAI(Creature* creature) : ScriptedAI(creature)
59 {
60 Initialize();
61 }
62
63 void Initialize()
64 {
65 KillCount = 0;
66 }
67
68 void Reset() override
69 {
70 _events.Reset();
71 Initialize();
72 }
73
74 void EnterCombat(Unit* /*who*/) override
75 {
76 Talk(YELL_TWILIGHT_CORRUPTOR_AGGRO);
77 _events.ScheduleEvent(EVENT_SOUL_CORRUPTION, 15000);
78 _events.ScheduleEvent(EVENT_CREATURE_OF_NIGHTMARE, 30000);
79 }
80
81 void KilledUnit(Unit* victim) override
82 {
83 if (victim->GetTypeId() == TYPEID_PLAYER)
84 {
85 ++KillCount;
86 Talk(YELL_TWILIGHT_CORRUPTOR_KILL, victim);
87
88 if (KillCount == 3)
89 {
90 DoCast(me, SPELL_LEVEL_UP, true);
91 KillCount = 0;
92 }
93 }
94 }
95
96 void UpdateAI(uint32 diff) override
97 {
98 if (!UpdateVictim())
99 return;
100
101 _events.Update(diff);
102
103 if (me->HasUnitState(UNIT_STATE_CASTING))
104 return;
105
106 while (uint32 eventId = _events.ExecuteEvent())
107 {
108 switch (eventId)
109 {
110 case EVENT_SOUL_CORRUPTION:
111 DoCastAOE(SPELL_SOUL_CORRUPTION);
112 _events.ScheduleEvent(EVENT_SOUL_CORRUPTION, urand(15000, 19000));
113 break;
114 case EVENT_CREATURE_OF_NIGHTMARE:
115 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
116 DoCast(target, SPELL_CREATURE_OF_NIGHTMARE);
117 _events.ScheduleEvent(EVENT_CREATURE_OF_NIGHTMARE, 45000);
118 break;
119 default:
120 break;
121 }
122 }
123
124 DoMeleeAttackIfReady();
125 }
126
127 private:
128 EventMap _events;
129 uint8 KillCount;
130 };
131
132 CreatureAI* GetAI(Creature* creature) const override
133 {
134 return new boss_twilight_corrupterAI(creature);
135 }
136};

142Position const TwillightCorrupter = { -10328.16f, -489.57f, 49.95f, 0.0f };
143
144class at_twilight_grove : public AreaTriggerScript
145{
146 public:
147 at_twilight_grove() : AreaTriggerScript("at_twilight_grove") { }
148
149 bool OnTrigger(Player* player, const AreaTriggerEntry* /*at*/) override
150 {
151 if (player->GetQuestStatus(QUEST_NIGHTMARES_CORRUPTION) == QUEST_STATUS_INCOMPLETE)
152 if (!player->FindNearestCreature(NPC_TWILIGHT_CORRUPTER, 500.0f, true))
153 if (Creature* corrupter = player->SummonCreature(NPC_TWILIGHT_CORRUPTER, TwillightCorrupter, TEMPSUMMON_MANUAL_DESPAWN, 60000))
154 corrupter->AI()->Talk(YELL_TWILIGHT_CORRUPTOR_RESPAWN, player);
155
156 return false;
157 };
158};

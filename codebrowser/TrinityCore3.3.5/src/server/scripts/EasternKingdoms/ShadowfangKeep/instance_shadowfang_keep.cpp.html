69class instance_shadowfang_keep : public InstanceMapScript
70{
71public:
72 instance_shadowfang_keep() : InstanceMapScript("instance_shadowfang_keep", 33) { }
73
74 InstanceScript* GetInstanceScript(InstanceMap* map) const override
75 {
76 return new instance_shadowfang_keep_InstanceMapScript(map);
77 }
78
79 struct instance_shadowfang_keep_InstanceMapScript : public InstanceScript
80 {
81 instance_shadowfang_keep_InstanceMapScript(Map* map) : InstanceScript(map)
82 {
83 SetHeaders(DataHeader);
84 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
85
86 uiPhase = 0;
87 uiTimer = 0;
88 }
89
90 uint32 m_auiEncounter[MAX_ENCOUNTER];
91 std::string str_data;
92
93 ObjectGuid uiAshGUID;
94 ObjectGuid uiAdaGUID;
95 ObjectGuid uiArchmageArugalGUID;
96
97 ObjectGuid DoorCourtyardGUID;
98 ObjectGuid DoorSorcererGUID;
99 ObjectGuid DoorArugalGUID;
100
101 uint8 uiPhase;
102 uint16 uiTimer;
103
104 void OnCreatureCreate(Creature* creature) override
105 {
106 switch (creature->GetEntry())
107 {
108 case NPC_ASH: uiAshGUID = creature->GetGUID(); break;
109 case NPC_ADA: uiAdaGUID = creature->GetGUID(); break;
110 case NPC_ARCHMAGE_ARUGAL: uiArchmageArugalGUID = creature->GetGUID(); break;
111 }
112 }
113
114 void OnGameObjectCreate(GameObject* go) override
115 {
116 switch (go->GetEntry())
117 {
118 case GO_COURTYARD_DOOR:
119 DoorCourtyardGUID = go->GetGUID();
120 if (m_auiEncounter[0] == DONE)
121 HandleGameObject(ObjectGuid::Empty, true, go);
122 break;
123 case GO_SORCERER_DOOR:
124 DoorSorcererGUID = go->GetGUID();
125 if (m_auiEncounter[2] == DONE)
126 HandleGameObject(ObjectGuid::Empty, true, go);
127 break;
128 case GO_ARUGAL_DOOR:
129 DoorArugalGUID = go->GetGUID();
130 if (m_auiEncounter[3] == DONE)
131 HandleGameObject(ObjectGuid::Empty, true, go);
132 break;
133 }
134 }
135
136 void DoSpeech()
137 {
138 Creature* pAda = instance->GetCreature(uiAdaGUID);
139 Creature* pAsh = instance->GetCreature(uiAshGUID);
140
141 if (pAda && pAda->IsAlive() && pAsh && pAsh->IsAlive())
142 {
143 pAda->AI()->Talk(SAY_BOSS_DIE_AD);
144 pAsh->AI()->Talk(SAY_BOSS_DIE_AS);
145 }
146 }
147
148 void SetData(uint32 type, uint32 data) override
149 {
150 switch (type)
151 {
152 case TYPE_FREE_NPC:
153 if (data == DONE)
154 DoUseDoorOrButton(DoorCourtyardGUID);
155 m_auiEncounter[0] = data;
156 break;
157 case TYPE_RETHILGORE:
158 if (data == DONE)
159 DoSpeech();
160 m_auiEncounter[1] = data;
161 break;
162 case TYPE_FENRUS:
163 switch (data)
164 {
165 case DONE:
166 uiTimer = 1000;
167 uiPhase = 1;
168 break;
169 case 7:
170 DoUseDoorOrButton(DoorSorcererGUID);
171 break;
172 }
173 m_auiEncounter[2] = data;
174 break;
175 case TYPE_NANDOS:
176 if (data == DONE)
177 DoUseDoorOrButton(DoorArugalGUID);
178 m_auiEncounter[3] = data;
179 break;
180 }
181
182 if (data == DONE)
183 {
184 OUT_SAVE_INST_DATA;
185
186 std::ostringstream saveStream;
187 saveStream << m_auiEncounter[0] << ' ' << m_auiEncounter[1] << ' ' << m_auiEncounter[2] << ' ' << m_auiEncounter[3];
188
189 str_data = saveStream.str();
190
191 SaveToDB();
192 OUT_SAVE_INST_DATA_COMPLETE;
193 }
194 }
195
196 uint32 GetData(uint32 type) const override
197 {
198 switch (type)
199 {
200 case TYPE_FREE_NPC:
201 return m_auiEncounter[0];
202 case TYPE_RETHILGORE:
203 return m_auiEncounter[1];
204 case TYPE_FENRUS:
205 return m_auiEncounter[2];
206 case TYPE_NANDOS:
207 return m_auiEncounter[3];
208 }
209 return 0;
210 }
211
212 std::string GetSaveData() override
213 {
214 return str_data;
215 }
216
217 void Load(const char* in) override
218 {
219 if (!in)
220 {
221 OUT_LOAD_INST_DATA_FAIL;
222 return;
223 }
224
225 OUT_LOAD_INST_DATA(in);
226
227 std::istringstream loadStream(in);
228 loadStream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3];
229
230 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
231 {
232 if (m_auiEncounter[i] == IN_PROGRESS)
233 m_auiEncounter[i] = NOT_STARTED;
234 }
235
236 OUT_LOAD_INST_DATA_COMPLETE;
237 }
238
239 void Update(uint32 uiDiff) override
240 {
241 if (GetData(TYPE_FENRUS) != DONE)
242 return;
243
244 Creature* pArchmage = instance->GetCreature(uiArchmageArugalGUID);
245
246 if (!pArchmage || !pArchmage->IsAlive())
247 return;
248
249 if (uiPhase)
250 {
251 if (uiTimer <= uiDiff)
252 {
253 switch (uiPhase)
254 {
255 case 1:
256 {
257 Creature* summon = pArchmage->SummonCreature(pArchmage->GetEntry(), SpawnLocation[4], TEMPSUMMON_TIMED_DESPAWN, 10000);
258 summon->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
259 summon->SetReactState(REACT_DEFENSIVE);
260 summon->CastSpell(summon, SPELL_ASHCROMBE_TELEPORT, true);
261 summon->AI()->Talk(SAY_ARCHMAGE);
262 uiTimer = 2000;
263 uiPhase = 2;
264 break;
265 }
266 case 2:
267 {
268 pArchmage->SummonCreature(NPC_ARUGAL_VOIDWALKER, SpawnLocation[0], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 60000);
269 pArchmage->SummonCreature(NPC_ARUGAL_VOIDWALKER, SpawnLocation[1], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 60000);
270 pArchmage->SummonCreature(NPC_ARUGAL_VOIDWALKER, SpawnLocation[2], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 60000);
271 pArchmage->SummonCreature(NPC_ARUGAL_VOIDWALKER, SpawnLocation[3], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 60000);
272 uiPhase = 0;
273 break;
274 }
275
276 }
277 } else uiTimer -= uiDiff;
278 }
279 }
280 };

66class npc_shadowfang_prisoner : public CreatureScript
67{
68public:
69 npc_shadowfang_prisoner() : CreatureScript("npc_shadowfang_prisoner") { }
70
71 CreatureAI* GetAI(Creature* creature) const override
72 {
73 return GetInstanceAI<npc_shadowfang_prisonerAI>(creature);
74 }
75
76 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
77 {
78 player->PlayerTalkClass->ClearMenus();
79 if (action == GOSSIP_ACTION_INFO_DEF+1)
80 {
81 player->CLOSE_GOSSIP_MENU();
82
83 if (npc_escortAI* pEscortAI = CAST_AI(npc_shadowfang_prisoner::npc_shadowfang_prisonerAI, creature->AI()))
84 pEscortAI->Start(false, false);
85 }
86 return true;
87 }
88
89 bool OnGossipHello(Player* player, Creature* creature) override
90 {
91 InstanceScript* instance = creature->GetInstanceScript();
92
93 if (instance && instance->GetData(TYPE_FREE_NPC) != DONE && instance->GetData(TYPE_RETHILGORE) == DONE)
94 player->ADD_GOSSIP_ITEM_DB(Player::GetDefaultGossipMenuForSource(creature), 0, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
95
96 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
97
98 return true;
99 }

101 struct npc_shadowfang_prisonerAI : public npc_escortAI
102 {
103 npc_shadowfang_prisonerAI(Creature* creature) : npc_escortAI(creature)
104 {
105 instance = creature->GetInstanceScript();
106 }
107
108 InstanceScript* instance;
109
110 void WaypointReached(uint32 waypointId) override
111 {
112 switch (waypointId)
113 {
114 case 0:
115 if (me->GetEntry() == NPC_ASH)
116 Talk(SAY_FREE_AS);
117 else
118 Talk(SAY_FREE_AD);
119 break;
120 case 10:
121 if (me->GetEntry() == NPC_ASH)
122 Talk(SAY_OPEN_DOOR_AS);
123 else
124 Talk(SAY_OPEN_DOOR_AD);
125 break;
126 case 11:
127 if (me->GetEntry() == NPC_ASH)
128 DoCast(me, SPELL_UNLOCK);
129 break;
130 case 12:
131 if (me->GetEntry() == NPC_ASH)
132 Talk(SAY_POST_DOOR_AS);
133 else
134 Talk(SAY_POST1_DOOR_AD);
135
136 instance->SetData(TYPE_FREE_NPC, DONE);
137 break;
138 case 13:
139 if (me->GetEntry() != NPC_ASH)
140 Talk(SAY_POST2_DOOR_AD);
141 break;
142 }
143 }
144
145 void Reset() override { }
146 void EnterCombat(Unit* /*who*/) override { }
147 };

161 struct npc_arugal_voidwalkerAI : public ScriptedAI
162 {
163 npc_arugal_voidwalkerAI(Creature* creature) : ScriptedAI(creature)
164 {
165 Initialize();
166 instance = creature->GetInstanceScript();
167 }
168
169 void Initialize()
170 {
171 uiDarkOffering = urand(200, 1000);
172 }
173
174 InstanceScript* instance;
175
176 uint32 uiDarkOffering;
177
178 void Reset() override
179 {
180 Initialize();
181 }
182
183 void UpdateAI(uint32 uiDiff) override
184 {
185 if (!UpdateVictim())
186 return;
187
188 if (uiDarkOffering <= uiDiff)
189 {
190 if (Creature* pFriend = me->FindNearestCreature(me->GetEntry(), 25.0f, true))
191 DoCast(pFriend, SPELL_DARK_OFFERING);
192 else
193 DoCast(me, SPELL_DARK_OFFERING);
194 uiDarkOffering = urand(4400, 12500);
195 } else uiDarkOffering -= uiDiff;
196
197 DoMeleeAttackIfReady();
198 }
199
200 void JustDied(Unit* /*killer*/) override
201 {
202 instance->SetData(TYPE_FENRUS, instance->GetData(TYPE_FENRUS) + 1);
203 }
204 };

208class spell_shadowfang_keep_haunting_spirits : public SpellScriptLoader
209{
210 public:
211 spell_shadowfang_keep_haunting_spirits() : SpellScriptLoader("spell_shadowfang_keep_haunting_spirits") { }

213 class spell_shadowfang_keep_haunting_spirits_AuraScript : public AuraScript
214 {
215 PrepareAuraScript(spell_shadowfang_keep_haunting_spirits_AuraScript);
216
217 void CalcPeriodic(AuraEffect const* /*aurEff*/, bool& isPeriodic, int32& amplitude)
218 {
219 isPeriodic = true;
220 amplitude = (irand(0, 60) + 30) * IN_MILLISECONDS;
221 }
222
223 void HandleDummyTick(AuraEffect const* aurEff)
224 {
225 GetTarget()->CastSpell((Unit*)NULL, aurEff->GetAmount(), true);
226 }
227
228 void HandleUpdatePeriodic(AuraEffect* aurEff)
229 {
230 aurEff->CalculatePeriodic(GetCaster());
231 }
232
233 void Register() override
234 {
235 DoEffectCalcPeriodic += AuraEffectCalcPeriodicFn(spell_shadowfang_keep_haunting_spirits_AuraScript::CalcPeriodic, EFFECT_0, SPELL_AURA_DUMMY);
236 OnEffectPeriodic += AuraEffectPeriodicFn(spell_shadowfang_keep_haunting_spirits_AuraScript::HandleDummyTick, EFFECT_0, SPELL_AURA_DUMMY);
237 OnEffectUpdatePeriodic += AuraEffectUpdatePeriodicFn(spell_shadowfang_keep_haunting_spirits_AuraScript::HandleUpdatePeriodic, EFFECT_0, SPELL_AURA_DUMMY);
238 }
239 };
240
241 AuraScript* GetAuraScript() const override
242 {
243 return new spell_shadowfang_keep_haunting_spirits_AuraScript();
244 }

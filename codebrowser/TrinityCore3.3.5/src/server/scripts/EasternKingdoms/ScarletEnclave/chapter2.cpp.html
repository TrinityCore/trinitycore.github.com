77 void SpellHit(Unit* caster, const SpellInfo* spell) override
78 {
79 if (spell->Id == SPELL_PERSUASIVE_STRIKE && caster->GetTypeId() == TYPEID_PLAYER && me->IsAlive() && !speechCounter)
80 {
81 if (Player* player = caster->ToPlayer())
82 {
83 if (player->GetQuestStatus(QUEST_HOW_TO_WIN_FRIENDS) == QUEST_STATUS_INCOMPLETE)
84 {
85 playerGUID = player->GetGUID();
86 speechTimer = 1000;
87 speechCounter = 1;
88 me->setFaction(player->getFaction());
89 me->CombatStop(true);
90 me->GetMotionMaster()->MoveIdle();
91 me->SetReactState(REACT_PASSIVE);
92 DoCastAOE(SPELL_THREAT_PULSE, true);
93
94 sCreatureTextMgr->SendChat(me, SAY_PERSUADE_RAND, NULL, CHAT_MSG_ADDON, LANG_ADDON, TEXT_RANGE_NORMAL, 0, TEAM_OTHER, false, player);
95 Talk(SAY_CRUSADER);
96 }
97 }
98 }
99 }

101 void UpdateAI(uint32 diff) override
102 {
103 if (speechCounter)
104 {
105 if (speechTimer <= diff)
106 {
107 Player* player = ObjectAccessor::GetPlayer(*me, playerGUID);
108 if (!player)
109 {
110 EnterEvadeMode();
111 return;
112 }
113
114 switch (speechCounter)
115 {
116 case 1:
117 Talk(SAY_PERSUADED1);
118 speechTimer = 8000;
119 break;
120
121 case 2:
122 Talk(SAY_PERSUADED2);
123 speechTimer = 8000;
124 break;
125
126 case 3:
127 Talk(SAY_PERSUADED3);
128 speechTimer = 8000;
129 break;
130
131 case 4:
132 Talk(SAY_PERSUADED4);
133 speechTimer = 8000;
134 break;
135
136 case 5:
137 sCreatureTextMgr->SendChat(me, SAY_PERSUADED5, NULL, CHAT_MSG_ADDON, LANG_ADDON, TEXT_RANGE_NORMAL, 0, TEAM_OTHER, false, player);
138 speechTimer = 8000;
139 break;
140
141 case 6:
142 Talk(SAY_PERSUADED6);
143 player->Kill(me);
144 speechCounter = 0;
145 player->GroupEventHappens(QUEST_HOW_TO_WIN_FRIENDS, me);
146 return;
147 }
148
149 ++speechCounter;
150 DoCastAOE(SPELL_THREAT_PULSE, true);
151
152 } else
153 speechTimer -= diff;
154
155 return;
156 }
157
158 if (!UpdateVictim())
159 return;
160
161 DoMeleeAttackIfReady();
162 }

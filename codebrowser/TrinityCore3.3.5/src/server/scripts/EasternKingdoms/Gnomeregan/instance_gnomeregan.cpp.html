18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "gnomeregan.h"
21#include "Player.h"
22
23#define MAX_ENCOUNTER 1
24
25class instance_gnomeregan : public InstanceMapScript
26{
27public:
28 instance_gnomeregan() : InstanceMapScript("instance_gnomeregan", 90) { }
29
30 InstanceScript* GetInstanceScript(InstanceMap* map) const override
31 {
32 return new instance_gnomeregan_InstanceMapScript(map);
33 }
34
35 struct instance_gnomeregan_InstanceMapScript : public InstanceScript
36 {
37 instance_gnomeregan_InstanceMapScript(Map* map) : InstanceScript(map)
38 {
39 SetHeaders(DataHeader);
40 memset(&m_auiEncounter, 0, sizeof(m_auiEncounter));
41 }
42
43 uint32 m_auiEncounter[MAX_ENCOUNTER];
44
45 ObjectGuid uiCaveInLeftGUID;
46 ObjectGuid uiCaveInRightGUID;
47
48 ObjectGuid uiBastmasterEmiShortfuseGUID;
49
50 void Load(const char* in) override
51 {
52 if (!in)
53 {
54 OUT_LOAD_INST_DATA_FAIL;
55 return;
56 }
57
58 OUT_LOAD_INST_DATA(in);
59
60 std::istringstream loadStream(in);
61 loadStream >> m_auiEncounter[0];
62
63 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
64 {
65 if (m_auiEncounter[i] == IN_PROGRESS)
66 m_auiEncounter[i] = NOT_STARTED;
67 }
68
69 OUT_LOAD_INST_DATA_COMPLETE;
70 }
71
72 void OnCreatureCreate(Creature* creature) override
73 {
74 switch (creature->GetEntry())
75 {
76 case NPC_BLASTMASTER_EMI_SHORTFUSE: uiBastmasterEmiShortfuseGUID = creature->GetGUID(); break;
77 }
78 }
79
80 void OnGameObjectCreate(GameObject* go) override
81 {
82 switch (go->GetEntry())
83 {
84 case GO_CAVE_IN_LEFT:
85 uiCaveInLeftGUID = go->GetGUID();
86 if (m_auiEncounter[0] == DONE || m_auiEncounter[0] == NOT_STARTED)
87 HandleGameObject(ObjectGuid::Empty, false, go);
88 break;
89 case GO_CAVE_IN_RIGHT:
90 uiCaveInRightGUID = go->GetGUID();
91 if (m_auiEncounter[0] == DONE || m_auiEncounter[0] == NOT_STARTED)
92 HandleGameObject(ObjectGuid::Empty, false, go);
93 break;
94 }
95 }
96
97 void SetData(uint32 uiType, uint32 uiData) override
98 {
99 switch (uiType)
100 {
101 case TYPE_EVENT:
102 m_auiEncounter[0] = uiData;
103 if (uiData == DONE)
104 SaveToDB();
105 break;
106 }
107 }
108
109 uint32 GetData(uint32 uiType) const override
110 {
111 switch (uiType)
112 {
113 case TYPE_EVENT: return m_auiEncounter[0];
114 }
115 return 0;
116 }
117
118 ObjectGuid GetGuidData(uint32 uiType) const override
119 {
120 switch (uiType)
121 {
122 case DATA_GO_CAVE_IN_LEFT: return uiCaveInLeftGUID;
123 case DATA_GO_CAVE_IN_RIGHT: return uiCaveInRightGUID;
124 case DATA_NPC_BASTMASTER_EMI_SHORTFUSE: return uiBastmasterEmiShortfuseGUID;
125 }
126
127 return ObjectGuid::Empty;
128 }
129 };
130
131};
132
133void AddSC_instance_gnomeregan()
134{
135 new instance_gnomeregan();
136}

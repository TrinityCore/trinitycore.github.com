40enum Calvin
41{
42 SAY_COMPLETE = 0,
43 SPELL_DRINK = 2639, // possibly not correct spell (but iconId is correct)

45 FACTION_HOSTILE = 168

47
48class npc_calvin_montague : public CreatureScript
49{
50public:
51 npc_calvin_montague() : CreatureScript("npc_calvin_montague") { }
52
53 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
54 {
55 if (quest->GetQuestId() == QUEST_590)
56 {
57 creature->setFaction(FACTION_HOSTILE);
58 creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
59 ENSURE_AI(npc_calvin_montague::npc_calvin_montagueAI, creature->AI())->AttackStart(player);
60 }
61 return true;
62 }
63
64 CreatureAI* GetAI(Creature* creature) const override
65 {
66 return new npc_calvin_montagueAI(creature);
67 }
68
69 struct npc_calvin_montagueAI : public ScriptedAI
70 {
71 npc_calvin_montagueAI(Creature* creature) : ScriptedAI(creature)
72 {
73 Initialize();
74 }
75
76 void Initialize()
77 {
78 m_uiPhase = 0;
79 m_uiPhaseTimer = 5000;
80 m_uiPlayerGUID.Clear();
81 }
82
83 uint32 m_uiPhase;
84 uint32 m_uiPhaseTimer;
85 ObjectGuid m_uiPlayerGUID;
86
87 void Reset() override
88 {
89 Initialize();
90
91 me->RestoreFaction();
92
93 if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC))
94 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
95 }
96
97 void EnterCombat(Unit* /*who*/) override { }
98
99 void AttackedBy(Unit* pAttacker) override
100 {
101 if (me->GetVictim() || me->IsFriendlyTo(pAttacker))
102 return;
103
104 AttackStart(pAttacker);
105 }
106
107 void DamageTaken(Unit* pDoneBy, uint32 &uiDamage) override
108 {
109 if (uiDamage > me->GetHealth() || me->HealthBelowPctDamaged(15, uiDamage))
110 {
111 uiDamage = 0;
112
113 me->RestoreFaction();
114 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
115 me->CombatStop(true);
116
117 m_uiPhase = 1;
118
119 if (pDoneBy->GetTypeId() == TYPEID_PLAYER)
120 m_uiPlayerGUID = pDoneBy->GetGUID();
121 }
122 }
123
124 void UpdateAI(uint32 diff) override
125 {
126 if (m_uiPhase)
127 {
128 if (m_uiPhaseTimer <= diff)
129 m_uiPhaseTimer = 7500;
130 else
131 {
132 m_uiPhaseTimer -= diff;
133 return;
134 }
135
136 switch (m_uiPhase)
137 {
138 case 1:
139 Talk(SAY_COMPLETE);
140 ++m_uiPhase;
141 break;
142 case 2:
143 if (Player* player = ObjectAccessor::GetPlayer(*me, m_uiPlayerGUID))
144 player->AreaExploredOrEventHappens(QUEST_590);
145
146 DoCast(me, SPELL_DRINK, true);
147 ++m_uiPhase;
148 break;
149 case 3:
150 EnterEvadeMode();
151 break;
152 }
153
154 return;
155 }
156
157 if (!UpdateVictim())
158 return;
159
160 DoMeleeAttackIfReady();
161 }
162 };
163};

170enum Mausoleum
171{
172 QUEST_ULAG = 1819,
173 NPC_ULAG = 6390,
174 GO_TRIGGER = 104593,
175 GO_DOOR = 176594
176};
177
178class go_mausoleum_door : public GameObjectScript
179{
180public:
181 go_mausoleum_door() : GameObjectScript("go_mausoleum_door") { }
182
183 bool OnGossipHello(Player* player, GameObject* /*go*/) override
184 {
185 if (player->GetQuestStatus(QUEST_ULAG) != QUEST_STATUS_INCOMPLETE)
186 return false;
187
188 if (!player->FindNearestCreature(NPC_ULAG, 50.0f))
189 if (GameObject* pTrigger = player->FindNearestGameObject(GO_TRIGGER, 30.0f))
190 {
191 pTrigger->SetGoState(GO_STATE_READY);
192 player->SummonCreature(NPC_ULAG, 2390.26f, 336.47f, 40.01f, 2.26f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 300000);
193 return false;
194 }
195
196 return false;
197 }
198};
199
200class go_mausoleum_trigger : public GameObjectScript
201{
202public:
203 go_mausoleum_trigger() : GameObjectScript("go_mausoleum_trigger") { }
204
205 bool OnGossipHello(Player* player, GameObject* go) override
206 {
207 if (player->GetQuestStatus(QUEST_ULAG) != QUEST_STATUS_INCOMPLETE)
208 return false;
209
210 if (GameObject* pDoor = player->FindNearestGameObject(GO_DOOR, 30.0f))
211 {
212 go->SetGoState(GO_STATE_ACTIVE);
213 pDoor->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_INTERACT_COND);
214 return true;
215 }
216
217 return false;
218 }
219};

50class npc_bartleby : public CreatureScript
51{
52public:
53 npc_bartleby() : CreatureScript("npc_bartleby") { }
54
55 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
56 {
57 if (quest->GetQuestId() == QUEST_BEAT)
58 {
59 creature->setFaction(FACTION_ENEMY);
60 creature->AI()->AttackStart(player);
61 }
62 return true;
63 }
64
65 CreatureAI* GetAI(Creature* creature) const override
66 {
67 return new npc_bartlebyAI(creature);
68 }
69
70 struct npc_bartlebyAI : public ScriptedAI
71 {
72 npc_bartlebyAI(Creature* creature) : ScriptedAI(creature)
73 {
74 m_uiNormalFaction = creature->getFaction();
75 }
76
77 uint32 m_uiNormalFaction;
78
79 void Reset() override
80 {
81 if (me->getFaction() != m_uiNormalFaction)
82 me->setFaction(m_uiNormalFaction);
83 }
84
85 void AttackedBy(Unit* pAttacker) override
86 {
87 if (me->GetVictim())
88 return;
89
90 if (me->IsFriendlyTo(pAttacker))
91 return;
92
93 AttackStart(pAttacker);
94 }
95
96 void DamageTaken(Unit* pDoneBy, uint32 &uiDamage) override
97 {
98 if (uiDamage > me->GetHealth() || me->HealthBelowPctDamaged(15, uiDamage))
99 {
100 //Take 0 damage
101 uiDamage = 0;
102
103 if (Player* player = pDoneBy->ToPlayer())
104 player->AreaExploredOrEventHappens(QUEST_BEAT);
105 EnterEvadeMode();
106 }
107 }
108 };
109};

132class npc_lord_gregor_lescovar : public CreatureScript
133{
134public:
135 npc_lord_gregor_lescovar() : CreatureScript("npc_lord_gregor_lescovar") { }
136
137 CreatureAI* GetAI(Creature* creature) const override
138 {
139 return new npc_lord_gregor_lescovarAI(creature);
140 }
141
142 struct npc_lord_gregor_lescovarAI : public npc_escortAI
143 {
144 npc_lord_gregor_lescovarAI(Creature* creature) : npc_escortAI(creature)
145 {
146 Initialize();
147 }
148
149 void Initialize()
150 {
151 uiTimer = 0;
152 uiPhase = 0;
153
154 MarzonGUID.Clear();
155 }
156
157 uint32 uiTimer;
158 uint32 uiPhase;
159
160 ObjectGuid MarzonGUID;
161
162 void Reset() override
163 {
164 Initialize();
165 }
166
167 void EnterEvadeMode(EvadeReason /*why*/) override
168 {
169 me->DisappearAndDie();
170
171 if (Creature* pMarzon = ObjectAccessor::GetCreature(*me, MarzonGUID))
172 {
173 if (pMarzon->IsAlive())
174 pMarzon->DisappearAndDie();
175 }
176 }
177
178 void EnterCombat(Unit* who) override
179 {
180 if (Creature* pMarzon = ObjectAccessor::GetCreature(*me, MarzonGUID))
181 {
182 if (pMarzon->IsAlive() && !pMarzon->IsInCombat())
183 pMarzon->AI()->AttackStart(who);
184 }
185 }
186
187 void WaypointReached(uint32 waypointId) override
188 {
189 switch (waypointId)
190 {
191 case 14:
192 SetEscortPaused(true);
193 Talk(SAY_LESCOVAR_2);
194 uiTimer = 3000;
195 uiPhase = 1;
196 break;
197 case 16:
198 SetEscortPaused(true);
199 if (Creature* pMarzon = me->SummonCreature(NPC_MARZON_BLADE, -8411.360352f, 480.069733f, 123.760895f, 4.941504f, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 1000))
200 {
201 pMarzon->GetMotionMaster()->MovePoint(0, -8408.000977f, 468.611450f, 123.759903f);
202 MarzonGUID = pMarzon->GetGUID();
203 }
204 uiTimer = 2000;
205 uiPhase = 4;
206 break;
207 }
208 }
209 //TO-DO: We don't have movemaps, also we can't make 2 npcs walks to one point propperly (and we can not use escort ai, because they are 2 different spawns and with same entry), because of it we make them, disappear.
210 void DoGuardsDisappearAndDie()
211 {
212 std::list<Creature*> GuardList;
213 me->GetCreatureListWithEntryInGrid(GuardList, NPC_STORMWIND_ROYAL, 8.0f);
214 if (!GuardList.empty())
215 {
216 for (std::list<Creature*>::const_iterator itr = GuardList.begin(); itr != GuardList.end(); ++itr)
217 {
218 if (Creature* pGuard = *itr)
219 pGuard->DisappearAndDie();
220 }
221 }
222 }
223
224 void UpdateAI(uint32 uiDiff) override
225 {
226 if (uiPhase)
227 {
228 if (uiTimer <= uiDiff)
229 {
230 switch (uiPhase)
231 {
232 case 1:
233 if (Creature* pGuard = me->FindNearestCreature(NPC_STORMWIND_ROYAL, 8.0f, true))
234 pGuard->AI()->Talk(SAY_GUARD_2);
235 uiTimer = 3000;
236 uiPhase = 2;
237 break;
238 case 2:
239 DoGuardsDisappearAndDie();
240 uiTimer = 2000;
241 uiPhase = 3;
242 break;
243 case 3:
244 SetEscortPaused(false);
245 uiTimer = 0;
246 uiPhase = 0;
247 break;
248 case 4:
249 Talk(SAY_LESCOVAR_3);
250 uiTimer = 0;
251 uiPhase = 0;
252 break;
253 case 5:
254 if (Creature* pMarzon = ObjectAccessor::GetCreature(*me, MarzonGUID))
255 pMarzon->AI()->Talk(SAY_MARZON_1);
256 uiTimer = 3000;
257 uiPhase = 6;
258 break;
259 case 6:
260 Talk(SAY_LESCOVAR_4);
261 if (Player* player = GetPlayerForEscort())
262 player->AreaExploredOrEventHappens(QUEST_THE_ATTACK);
263 uiTimer = 2000;
264 uiPhase = 7;
265 break;
266 case 7:
267 if (Creature* pTyrion = me->FindNearestCreature(NPC_TYRION, 20.0f, true))
268 pTyrion->AI()->Talk(SAY_TYRION_2);
269 if (Creature* pMarzon = ObjectAccessor::GetCreature(*me, MarzonGUID))
270 pMarzon->setFaction(14);
271 me->setFaction(14);
272 uiTimer = 0;
273 uiPhase = 0;
274 break;
275 }
276 } else uiTimer -= uiDiff;
277 }
278 npc_escortAI::UpdateAI(uiDiff);
279
280 if (!UpdateVictim())
281 return;
282
283 DoMeleeAttackIfReady();
284 }
285 };
286};

292class npc_marzon_silent_blade : public CreatureScript
293{
294public:
295 npc_marzon_silent_blade() : CreatureScript("npc_marzon_silent_blade") { }
296
297 CreatureAI* GetAI(Creature* creature) const override
298 {
299 return new npc_marzon_silent_bladeAI(creature);
300 }
301
302 struct npc_marzon_silent_bladeAI : public ScriptedAI
303 {
304 npc_marzon_silent_bladeAI(Creature* creature) : ScriptedAI(creature)
305 {
306 me->SetWalk(true);
307 }
308
309 void Reset() override
310 {
311 me->RestoreFaction();
312 }
313
314 void EnterCombat(Unit* who) override
315 {
316 Talk(SAY_MARZON_2);
317
318 if (me->IsSummon())
319 {
320 if (Unit* summoner = me->ToTempSummon()->GetSummoner())
321 {
322 if (summoner->GetTypeId() == TYPEID_UNIT && summoner->IsAlive() && !summoner->IsInCombat())
323 summoner->ToCreature()->AI()->AttackStart(who);
324 }
325 }
326 }
327
328 void EnterEvadeMode(EvadeReason /*why*/) override
329 {
330 me->DisappearAndDie();
331
332 if (me->IsSummon())
333 {
334 if (Unit* summoner = me->ToTempSummon()->GetSummoner())
335 {
336 if (summoner->GetTypeId() == TYPEID_UNIT && summoner->IsAlive())
337 summoner->ToCreature()->DisappearAndDie();
338 }
339 }
340 }
341
342 void MovementInform(uint32 uiType, uint32 /*uiId*/) override
343 {
344 if (uiType != POINT_MOTION_TYPE)
345 return;
346
347 if (me->IsSummon())
348 {
349 Unit* summoner = me->ToTempSummon()->GetSummoner();
350 if (summoner && summoner->GetTypeId() == TYPEID_UNIT && summoner->IsAIEnabled)
351 {
352 npc_lord_gregor_lescovar::npc_lord_gregor_lescovarAI* ai =
353 CAST_AI(npc_lord_gregor_lescovar::npc_lord_gregor_lescovarAI, summoner->GetAI());
354 if (ai)
355 {
356 ai->uiTimer = 2000;
357 ai->uiPhase = 5;
358 }
359 //me->ChangeOrient(0.0f, summoner);
360 }
361 }
362 }
363
364 void UpdateAI(uint32 /*diff*/) override
365 {
366 if (!UpdateVictim())
367 return;
368
369 DoMeleeAttackIfReady();
370 }
371 };

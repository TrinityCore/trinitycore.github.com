21enum Yells
22{
23 YELL_AGGRO = 0,
24 YELL_EVADE = 1,
25 //YELL_RESPAWN1 = -1810010, // Missing in database
26 //YELL_RESPAWN2 = -1810011, // Missing in database
27 YELL_RANDOM = 2,
28 YELL_SPELL = 3,

31enum Spells
32{
33 SPELL_AVATAR = 19135,
34 SPELL_THUNDERCLAP = 15588,
35 SPELL_STORMBOLT = 20685 // not sure
36};
37
38class boss_vanndar : public CreatureScript
39{
40public:
41 boss_vanndar() : CreatureScript("boss_vanndar") { }
42
43 struct boss_vanndarAI : public ScriptedAI
44 {
45 boss_vanndarAI(Creature* creature) : ScriptedAI(creature)
46 {
47 Initialize();
48 }
49
50 void Initialize()
51 {
52 AvatarTimer = 3 * IN_MILLISECONDS;
53 ThunderclapTimer = 4 * IN_MILLISECONDS;
54 StormboltTimer = 6 * IN_MILLISECONDS;
55 ResetTimer = 5 * IN_MILLISECONDS;
56 YellTimer = urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS);
57 }
58
59 uint32 AvatarTimer;
60 uint32 ThunderclapTimer;
61 uint32 StormboltTimer;
62 uint32 ResetTimer;
63 uint32 YellTimer;
64
65 void Reset() override
66 {
67 Initialize();
68 }
69
70 void EnterCombat(Unit* /*who*/) override
71 {
72 Talk(YELL_AGGRO);
73 }
74
75 void UpdateAI(uint32 diff) override
76 {
77 if (!UpdateVictim())
78 return;
79
80 if (AvatarTimer <= diff)
81 {
82 DoCastVictim(SPELL_AVATAR);
83 AvatarTimer = urand(15 * IN_MILLISECONDS, 20 * IN_MILLISECONDS);
84 } else AvatarTimer -= diff;
85
86 if (ThunderclapTimer <= diff)
87 {
88 DoCastVictim(SPELL_THUNDERCLAP);
89 ThunderclapTimer = urand(5 * IN_MILLISECONDS, 15 * IN_MILLISECONDS);
90 } else ThunderclapTimer -= diff;
91
92 if (StormboltTimer <= diff)
93 {
94 DoCastVictim(SPELL_STORMBOLT);
95 StormboltTimer = urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS);
96 } else StormboltTimer -= diff;
97
98 if (YellTimer <= diff)
99 {
100 Talk(YELL_RANDOM);
101 YellTimer = urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS); //20 to 30 seconds
102 } else YellTimer -= diff;
103
104 // check if creature is not outside of building
105 if (ResetTimer <= diff)
106 {
107 if (me->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 50)
108 {
109 EnterEvadeMode();
110 Talk(YELL_EVADE);
111 }
112 ResetTimer = 5 * IN_MILLISECONDS;
113 } else ResetTimer -= diff;
114
115 DoMeleeAttackIfReady();
116 }
117 };
118
119 CreatureAI* GetAI(Creature* creature) const override
120 {
121 return new boss_vanndarAI(creature);
122 }
123};
124
125void AddSC_boss_vanndar()
126{
127 new boss_vanndar;
128}

104 static bool HandlePvPstatsCommand(ChatHandler * handler, char const* /*args*/)
105 {
106 if (sWorld->getBoolConfig(CONFIG_BATTLEGROUND_STORE_STATISTICS_ENABLE))
107 {
108 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_PVPSTATS_FACTIONS_OVERALL);
109 PreparedQueryResult result = CharacterDatabase.Query(stmt);
110
111 if (result)
112 {
113 Field* fields = result->Fetch();
114 uint32 horde_victories = fields[1].GetUInt32();
115
116 if (!(result->NextRow()))
117 return false;
118
119 fields = result->Fetch();
120 uint32 alliance_victories = fields[1].GetUInt32();
121
122 handler->PSendSysMessage(LANG_PVPSTATS, alliance_victories, horde_victories);
123 }
124 else
125 return false;
126 }
127 else
128 handler->PSendSysMessage(LANG_PVPSTATS_DISABLED);
129
130 return true;
131 }

133 static bool HandleDevCommand(ChatHandler* handler, char const* args)
134 {
135 if (!*args)
136 {
137 if (handler->GetSession()->GetPlayer()->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_DEVELOPER))
138 handler->GetSession()->SendNotification(LANG_DEV_ON);
139 else
140 handler->GetSession()->SendNotification(LANG_DEV_OFF);

143

145

147 {
148 handler->GetSession()->GetPlayer()->SetFlag(PLAYER_FLAGS, PLAYER_FLAGS_DEVELOPER);
149 handler->GetSession()->SendNotification(LANG_DEV_ON);

154 {
155 handler->GetSession()->GetPlayer()->RemoveFlag(PLAYER_FLAGS, PLAYER_FLAGS_DEVELOPER);

157 return true;
158 }
159
160 handler->SendSysMessage(LANG_USE_BOL);
161 handler->SetSentErrorMessage(true);
162 return false;

165 static bool HandleGPSCommand(ChatHandler* handler, char const* args)
166 {
167 WorldObject* object = NULL;
168 if (*args)
169 {
170 ObjectGuid guid = handler->extractGuidFromLink((char*)args);
171 if (guid)
172 object = (WorldObject*)ObjectAccessor::GetObjectByTypeMask(*handler->GetSession()->GetPlayer(), guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
173
174 if (!object)
175 {
176 handler->SendSysMessage(LANG_PLAYER_NOT_FOUND);
177 handler->SetSentErrorMessage(true);
178 return false;
179 }
180 }
181 else
182 {
183 object = handler->getSelectedUnit();
184
185 if (!object)
186 {
187 handler->SendSysMessage(LANG_SELECT_CHAR_OR_CREATURE);
188 handler->SetSentErrorMessage(true);
189 return false;
190 }
191 }
192
193 CellCoord cellCoord = Trinity::ComputeCellCoord(object->GetPositionX(), object->GetPositionY());
194 Cell cell(cellCoord);
195
196 uint32 zoneId, areaId;
197 object->GetZoneAndAreaId(zoneId, areaId);
198 uint32 mapId = object->GetMapId();
199
200 MapEntry const* mapEntry = sMapStore.LookupEntry(mapId);
201 AreaTableEntry const* zoneEntry = sAreaTableStore.LookupEntry(zoneId);
202 AreaTableEntry const* areaEntry = sAreaTableStore.LookupEntry(areaId);
203
204 float zoneX = object->GetPositionX();
205 float zoneY = object->GetPositionY();
206
207 Map2ZoneCoordinates(zoneX, zoneY, zoneId);
208
209 Map const* map = object->GetMap();
210 float groundZ = map->GetHeight(object->GetPhaseMask(), object->GetPositionX(), object->GetPositionY(), MAX_HEIGHT);
211 float floorZ = map->GetHeight(object->GetPhaseMask(), object->GetPositionX(), object->GetPositionY(), object->GetPositionZ());
212
213 GridCoord gridCoord = Trinity::ComputeGridCoord(object->GetPositionX(), object->GetPositionY());
214
215 int gridX = (MAX_NUMBER_OF_GRIDS - 1) - gridCoord.x_coord;
216 int gridY = (MAX_NUMBER_OF_GRIDS - 1) - gridCoord.y_coord;
217
218 uint32 haveMap = Map::ExistMap(mapId, gridX, gridY) ? 1 : 0;
219 uint32 haveVMap = Map::ExistVMap(mapId, gridX, gridY) ? 1 : 0;
220 uint32 haveMMap = (DisableMgr::IsPathfindingEnabled(mapId) && MMAP::MMapFactory::createOrGetMMapManager()->GetNavMesh(handler->GetSession()->GetPlayer()->GetMapId())) ? 1 : 0;
221
222 if (haveVMap)
223 {
224 if (map->IsOutdoors(object->GetPositionX(), object->GetPositionY(), object->GetPositionZ()))
225 handler->PSendSysMessage(LANG_GPS_POSITION_OUTDOORS);
226 else
227 handler->PSendSysMessage(LANG_GPS_POSITION_INDOORS);
228 }
229 else
230 handler->PSendSysMessage(LANG_GPS_NO_VMAP);
231
232 char const* unknown = handler->GetTrinityString(LANG_UNKNOWN);
233
234 handler->PSendSysMessage(LANG_MAP_POSITION,
235 mapId, (mapEntry ? mapEntry->name[handler->GetSessionDbcLocale()] : unknown),
236 zoneId, (zoneEntry ? zoneEntry->area_name[handler->GetSessionDbcLocale()] : unknown),
237 areaId, (areaEntry ? areaEntry->area_name[handler->GetSessionDbcLocale()] : unknown),
238 object->GetPhaseMask(),
239 object->GetPositionX(), object->GetPositionY(), object->GetPositionZ(), object->GetOrientation());
240 handler->PSendSysMessage(LANG_GRID_POSITION,
241 cell.GridX(), cell.GridY(), cell.CellX(), cell.CellY(), object->GetInstanceId(),
242 zoneX, zoneY, groundZ, floorZ, haveMap, haveVMap, haveMMap);
243
244 LiquidData liquidStatus;
245 ZLiquidStatus status = map->getLiquidStatus(object->GetPositionX(), object->GetPositionY(), object->GetPositionZ(), MAP_ALL_LIQUIDS, &liquidStatus);
246
247 if (status)
248 handler->PSendSysMessage(LANG_LIQUID_STATUS, liquidStatus.level, liquidStatus.depth_level, liquidStatus.entry, liquidStatus.type_flags, status);
249
250 return true;
251 }

298 // Teleport to Player
299 static bool HandleAppearCommand(ChatHandler* handler, char const* args)
300 {
301 Player* target;
302 ObjectGuid targetGuid;
303 std::string targetName;
304 if (!handler->extractPlayerTarget((char*)args, &target, &targetGuid, &targetName))
305 return false;
306
307 Player* _player = handler->GetSession()->GetPlayer();
308 if (target == _player || targetGuid == _player->GetGUID())
309 {
310 handler->SendSysMessage(LANG_CANT_TELEPORT_SELF);
311 handler->SetSentErrorMessage(true);
312 return false;
313 }
314
315 if (target)
316 {
317 // check online security
318 if (handler->HasLowerSecurity(target, ObjectGuid::Empty))
319 return false;
320
321 std::string chrNameLink = handler->playerLink(targetName);
322
323 Map* map = target->GetMap();
324 if (map->IsBattlegroundOrArena())
325 {
326 // only allow if gm mode is on
327 if (!_player->IsGameMaster())
328 {
329 handler->PSendSysMessage(LANG_CANNOT_GO_TO_BG_GM, chrNameLink.c_str());
330 handler->SetSentErrorMessage(true);
331 return false;
332 }
333 // if both players are in different bgs
334 else if (_player->GetBattlegroundId() && _player->GetBattlegroundId() != target->GetBattlegroundId())
335 _player->LeaveBattleground(false); // Note: should be changed so _player gets no Deserter debuff
336
337 // all's well, set bg id
338 // when porting out from the bg, it will be reset to 0
339 _player->SetBattlegroundId(target->GetBattlegroundId(), target->GetBattlegroundTypeId());
340 // remember current position as entry point for return at bg end teleportation
341 if (!_player->GetMap()->IsBattlegroundOrArena())
342 _player->SetBattlegroundEntryPoint();
343 }
344 else if (map->IsDungeon())
345 {
346 // we have to go to instance, and can go to player only if:
347 // 1) we are in his group (either as leader or as member)
348 // 2) we are not bound to any group and have GM mode on
349 if (_player->GetGroup())
350 {
351 // we are in group, we can go only if we are in the player group
352 if (_player->GetGroup() != target->GetGroup())
353 {
354 handler->PSendSysMessage(LANG_CANNOT_GO_TO_INST_PARTY, chrNameLink.c_str());
355 handler->SetSentErrorMessage(true);
356 return false;
357 }
358 }
359 else
360 {
361 // we are not in group, let's verify our GM mode
362 if (!_player->IsGameMaster())
363 {
364 handler->PSendSysMessage(LANG_CANNOT_GO_TO_INST_GM, chrNameLink.c_str());
365 handler->SetSentErrorMessage(true);
366 return false;
367 }
368 }
369
370 // if the player or the player's group is bound to another instance
371 // the player will not be bound to another one
372 InstancePlayerBind* bind = _player->GetBoundInstance(target->GetMapId(), target->GetDifficulty(map->IsRaid()));
373 if (!bind)
374 {
375 Group* group = _player->GetGroup();
376 // if no bind exists, create a solo bind
377 InstanceGroupBind* gBind = group ? group->GetBoundInstance(target) : NULL; // if no bind exists, create a solo bind
378 if (!gBind)
379 if (InstanceSave* save = sInstanceSaveMgr->GetInstanceSave(target->GetInstanceId()))
380 _player->BindToInstance(save, !save->CanReset());
381 }
382
383 if (map->IsRaid())
384 _player->SetRaidDifficulty(target->GetRaidDifficulty());
385 else
386 _player->SetDungeonDifficulty(target->GetDungeonDifficulty());
387 }
388
389 handler->PSendSysMessage(LANG_APPEARING_AT, chrNameLink.c_str());
390
391 // stop flight if need
392 if (_player->IsInFlight())
393 {
394 _player->GetMotionMaster()->MovementExpired();
395 _player->CleanupAfterTaxiFlight();
396 }
397 // save only in non-flight case
398 else
399 _player->SaveRecallPosition();
400
401 // to point to see at target with same orientation
402 float x, y, z;
403 target->GetContactPoint(_player, x, y, z);
404
405 _player->TeleportTo(target->GetMapId(), x, y, z, _player->GetAngle(target), TELE_TO_GM_MODE);
406 _player->SetPhaseMask(target->GetPhaseMask(), true);
407 }
408 else
409 {
410 // check offline security
411 if (handler->HasLowerSecurity(NULL, targetGuid))
412 return false;
413
414 std::string nameLink = handler->playerLink(targetName);
415
416 handler->PSendSysMessage(LANG_APPEARING_AT, nameLink.c_str());
417
418 // to point where player stay (if loaded)
419 float x, y, z, o;
420 uint32 map;
421 bool in_flight;
422 if (!Player::LoadPositionFromDB(map, x, y, z, o, in_flight, targetGuid))
423 return false;
424
425 // stop flight if need
426 if (_player->IsInFlight())
427 {
428 _player->GetMotionMaster()->MovementExpired();
429 _player->CleanupAfterTaxiFlight();
430 }
431 // save only in non-flight case
432 else
433 _player->SaveRecallPosition();
434
435 _player->TeleportTo(map, x, y, z, _player->GetOrientation());
436 }
437
438 return true;
439 }

441 static bool HandleSummonCommand(ChatHandler* handler, char const* args)
442 {
443 Player* target;
444 ObjectGuid targetGuid;
445 std::string targetName;
446 if (!handler->extractPlayerTarget((char*)args, &target, &targetGuid, &targetName))
447 return false;
448
449 Player* _player = handler->GetSession()->GetPlayer();
450 if (target == _player || targetGuid == _player->GetGUID())
451 {
452 handler->PSendSysMessage(LANG_CANT_TELEPORT_SELF);
453 handler->SetSentErrorMessage(true);
454 return false;
455 }
456
457 if (target)
458 {
459 std::string nameLink = handler->playerLink(targetName);
460 // check online security
461 if (handler->HasLowerSecurity(target, ObjectGuid::Empty))
462 return false;
463
464 if (target->IsBeingTeleported())
465 {
466 handler->PSendSysMessage(LANG_IS_TELEPORTED, nameLink.c_str());
467 handler->SetSentErrorMessage(true);
468 return false;
469 }
470
471 Map* map = handler->GetSession()->GetPlayer()->GetMap();
472
473 if (map->IsBattlegroundOrArena())
474 {
475 // only allow if gm mode is on
476 if (!_player->IsGameMaster())
477 {
478 handler->PSendSysMessage(LANG_CANNOT_GO_TO_BG_GM, nameLink.c_str());
479 handler->SetSentErrorMessage(true);
480 return false;
481 }
482 // if both players are in different bgs
483 else if (target->GetBattlegroundId() && handler->GetSession()->GetPlayer()->GetBattlegroundId() != target->GetBattlegroundId())
484 target->LeaveBattleground(false); // Note: should be changed so target gets no Deserter debuff
485
486 // all's well, set bg id
487 // when porting out from the bg, it will be reset to 0
488 target->SetBattlegroundId(handler->GetSession()->GetPlayer()->GetBattlegroundId(), handler->GetSession()->GetPlayer()->GetBattlegroundTypeId());
489 // remember current position as entry point for return at bg end teleportation
490 if (!target->GetMap()->IsBattlegroundOrArena())
491 target->SetBattlegroundEntryPoint();
492 }
493 else if (map->IsDungeon())
494 {
495 Map* destMap = target->GetMap();
496
497 if (destMap->Instanceable() && destMap->GetInstanceId() != map->GetInstanceId())
498 target->UnbindInstance(map->GetInstanceId(), target->GetDungeonDifficulty(), true);
499
500 // we are in an instance, and can only summon players in our group with us as leader
501 if (!handler->GetSession()->GetPlayer()->GetGroup() || !target->GetGroup() ||
502 (target->GetGroup()->GetLeaderGUID() != handler->GetSession()->GetPlayer()->GetGUID()) ||
503 (handler->GetSession()->GetPlayer()->GetGroup()->GetLeaderGUID() != handler->GetSession()->GetPlayer()->GetGUID()))
504 // the last check is a bit excessive, but let it be, just in case
505 {
506 handler->PSendSysMessage(LANG_CANNOT_SUMMON_TO_INST, nameLink.c_str());
507 handler->SetSentErrorMessage(true);
508 return false;
509 }
510 }
511
512 handler->PSendSysMessage(LANG_SUMMONING, nameLink.c_str(), "");
513 if (handler->needReportToTarget(target))
514 ChatHandler(target->GetSession()).PSendSysMessage(LANG_SUMMONED_BY, handler->playerLink(_player->GetName()).c_str());
515
516 // stop flight if need
517 if (target->IsInFlight())
518 {
519 target->GetMotionMaster()->MovementExpired();
520 target->CleanupAfterTaxiFlight();
521 }
522 // save only in non-flight case
523 else
524 target->SaveRecallPosition();
525
526 // before GM
527 float x, y, z;
528 handler->GetSession()->GetPlayer()->GetClosePoint(x, y, z, target->GetObjectSize());
529 target->TeleportTo(handler->GetSession()->GetPlayer()->GetMapId(), x, y, z, target->GetOrientation());
530 target->SetPhaseMask(handler->GetSession()->GetPlayer()->GetPhaseMask(), true);
531 }
532 else
533 {
534 // check offline security
535 if (handler->HasLowerSecurity(NULL, targetGuid))
536 return false;
537
538 std::string nameLink = handler->playerLink(targetName);
539
540 handler->PSendSysMessage(LANG_SUMMONING, nameLink.c_str(), handler->GetTrinityString(LANG_OFFLINE));
541
542 // in point where GM stay
543 SQLTransaction dummy;
544 Player::SavePositionInDB(WorldLocation(handler->GetSession()->GetPlayer()->GetMapId(),
545 handler->GetSession()->GetPlayer()->GetPositionX(),
546 handler->GetSession()->GetPlayer()->GetPositionY(),
547 handler->GetSession()->GetPlayer()->GetPositionZ(),
548 handler->GetSession()->GetPlayer()->GetOrientation()),
549 handler->GetSession()->GetPlayer()->GetZoneId(),
550 targetGuid, dummy);
551 }
552
553 return true;
554 }

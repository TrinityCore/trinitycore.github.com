73 static bool HandleLearnCommand(ChatHandler* handler, char const* args)
74 {
75 Player* targetPlayer = handler->getSelectedPlayerOrSelf();
76
77 if (!targetPlayer)
78 {
79 handler->SendSysMessage(LANG_PLAYER_NOT_FOUND);
80 handler->SetSentErrorMessage(true);
81 return false;
82 }
83
84 // number or [name] Shift-click form |color|Hspell:spell_id|h[name]|h|r or Htalent form
85 uint32 spell = handler->extractSpellIdFromLink((char*)args);
86 if (!spell || !sSpellMgr->GetSpellInfo(spell))
87 return false;
88
89 char const* all = strtok(NULL, " ");
90 bool allRanks = all ? (strncmp(all, "all", strlen(all)) == 0) : false;
91
92 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spell);
93 if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo, handler->GetSession()->GetPlayer()))
94 {
95 handler->PSendSysMessage(LANG_COMMAND_SPELL_BROKEN, spell);
96 handler->SetSentErrorMessage(true);
97 return false;
98 }
99
100 if (!allRanks && targetPlayer->HasSpell(spell))
101 {
102 if (targetPlayer == handler->GetSession()->GetPlayer())
103 handler->SendSysMessage(LANG_YOU_KNOWN_SPELL);
104 else
105 handler->PSendSysMessage(LANG_TARGET_KNOWN_SPELL, handler->GetNameLink(targetPlayer).c_str());
106 handler->SetSentErrorMessage(true);
107 return false;
108 }
109
110 if (allRanks)
111 targetPlayer->LearnSpellHighestRank(spell);
112 else
113 targetPlayer->LearnSpell(spell, false);
114
115 if (GetTalentSpellCost(spellInfo->GetFirstRankSpell()->Id))
116 targetPlayer->SendTalentsInfoData(false);
117
118 return true;
119 }

121 static bool HandleLearnAllGMCommand(ChatHandler* handler, char const* /*args*/)
122 {
123 for (uint32 i = 0; i < sSpellMgr->GetSpellInfoStoreSize(); ++i)
124 {
125 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(i);
126 if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo, handler->GetSession()->GetPlayer(), false))
127 continue;
128
129 if (!spellInfo->IsAbilityOfSkillType(SKILL_INTERNAL))
130 continue;
131
132 handler->GetSession()->GetPlayer()->LearnSpell(i, false);
133 }
134
135 handler->SendSysMessage(LANG_LEARNING_GM_SKILLS);
136 return true;
137 }

146 static bool HandleLearnAllMySpellsCommand(ChatHandler* handler, char const* /*args*/)
147 {
148 ChrClassesEntry const* classEntry = sChrClassesStore.LookupEntry(handler->GetSession()->GetPlayer()->getClass());
149 if (!classEntry)
150 return true;
151 uint32 family = classEntry->spellfamily;
152
153 for (uint32 i = 0; i < sSkillLineAbilityStore.GetNumRows(); ++i)
154 {
155 SkillLineAbilityEntry const* entry = sSkillLineAbilityStore.LookupEntry(i);
156 if (!entry)
157 continue;
158
159 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(entry->spellId);
160 if (!spellInfo)
161 continue;
162
163 // skip server-side/triggered spells
164 if (spellInfo->SpellLevel == 0)
165 continue;
166
167 // skip wrong class/race skills
168 if (!handler->GetSession()->GetPlayer()->IsSpellFitByClassAndRace(spellInfo->Id))
169 continue;
170
171 // skip other spell families
172 if (spellInfo->SpellFamilyName != family)
173 continue;
174
175 // skip spells with first rank learned as talent (and all talents then also)
176 if (GetTalentSpellCost(spellInfo->GetFirstRankSpell()->Id) > 0)
177 continue;
178
179 // skip broken spells
180 if (!SpellMgr::IsSpellValid(spellInfo, handler->GetSession()->GetPlayer(), false))
181 continue;
182
183 handler->GetSession()->GetPlayer()->LearnSpell(spellInfo->Id, false);
184 }
185
186 handler->SendSysMessage(LANG_COMMAND_LEARN_CLASS_SPELLS);
187 return true;
188 }

190 static bool HandleLearnAllMyTalentsCommand(ChatHandler* handler, char const* /*args*/)
191 {
192 Player* player = handler->GetSession()->GetPlayer();
193 uint32 classMask = player->getClassMask();
194
195 for (uint32 i = 0; i < sTalentStore.GetNumRows(); ++i)
196 {
197 TalentEntry const* talentInfo = sTalentStore.LookupEntry(i);
198 if (!talentInfo)
199 continue;
200
201 TalentTabEntry const* talentTabInfo = sTalentTabStore.LookupEntry(talentInfo->TalentTab);
202 if (!talentTabInfo)
203 continue;
204
205 if ((classMask & talentTabInfo->ClassMask) == 0)
206 continue;
207
208 // search highest talent rank
209 uint32 spellId = 0;
210 for (int8 rank = MAX_TALENT_RANK - 1; rank >= 0; --rank)
211 {
212 if (talentInfo->RankID[rank] != 0)
213 {
214 spellId = talentInfo->RankID[rank];
215 break;
216 }
217 }
218
219 if (!spellId) // ??? none spells in talent
220 continue;
221
222 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
223 if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo, handler->GetSession()->GetPlayer(), false))
224 continue;
225
226 // learn highest rank of talent and learn all non-talent spell ranks (recursive by tree)
227 player->LearnSpellHighestRank(spellId);
228 player->AddTalent(spellId, player->GetActiveSpec(), true);
229 }
230
231 player->SetFreeTalentPoints(0);
232
233 handler->SendSysMessage(LANG_COMMAND_LEARN_CLASS_TALENTS);
234 return true;
235 }

237 static bool HandleLearnAllMyPetTalentsCommand(ChatHandler* handler, char const* /*args*/)
238 {
239 Player* player = handler->GetSession()->GetPlayer();
240
241 Pet* pet = player->GetPet();
242 if (!pet)
243 {
244 handler->SendSysMessage(LANG_NO_PET_FOUND);
245 handler->SetSentErrorMessage(true);
246 return false;
247 }
248
249 CreatureTemplate const* creatureInfo = pet->GetCreatureTemplate();
250 if (!creatureInfo)
251 {
252 handler->SendSysMessage(LANG_WRONG_PET_TYPE);
253 handler->SetSentErrorMessage(true);
254 return false;
255 }
256
257 CreatureFamilyEntry const* petFamily = sCreatureFamilyStore.LookupEntry(creatureInfo->family);
258 if (!petFamily)
259 {
260 handler->SendSysMessage(LANG_WRONG_PET_TYPE);
261 handler->SetSentErrorMessage(true);
262 return false;
263 }
264
265 if (petFamily->petTalentType < 0) // not hunter pet
266 {
267 handler->SendSysMessage(LANG_WRONG_PET_TYPE);
268 handler->SetSentErrorMessage(true);
269 return false;
270 }
271
272 for (uint32 i = 0; i < sTalentStore.GetNumRows(); ++i)
273 {
274 TalentEntry const* talentInfo = sTalentStore.LookupEntry(i);
275 if (!talentInfo)
276 continue;
277
278 TalentTabEntry const* talentTabInfo = sTalentTabStore.LookupEntry(talentInfo->TalentTab);
279 if (!talentTabInfo)
280 continue;
281
282 // prevent learn talent for different family (cheating)
283 if (((1 << petFamily->petTalentType) & talentTabInfo->petTalentMask) == 0)
284 continue;
285
286 // search highest talent rank
287 uint32 spellId = 0;
288
289 for (int8 rank = MAX_TALENT_RANK-1; rank >= 0; --rank)
290 {
291 if (talentInfo->RankID[rank] != 0)
292 {
293 spellId = talentInfo->RankID[rank];
294 break;
295 }
296 }
297
298 if (!spellId) // ??? none spells in talent
299 continue;
300
301 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
302 if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo, handler->GetSession()->GetPlayer(), false))
303 continue;
304
305 // learn highest rank of talent and learn all non-talent spell ranks (recursive by tree)
306 pet->learnSpellHighRank(spellId);
307 }
308
309 pet->SetFreeTalentPoints(0);
310
311 handler->SendSysMessage(LANG_COMMAND_LEARN_PET_TALENTS);
312 return true;
313 }

339 static bool HandleLearnAllCraftsCommand(ChatHandler* handler, char const* args)
340 {
341 Player* target;
342 if (!handler->extractPlayerTarget((char*)args, &target))
343 return false;
344
345 for (uint32 i = 0; i < sSkillLineStore.GetNumRows(); ++i)
346 {
347 SkillLineEntry const* skillInfo = sSkillLineStore.LookupEntry(i);
348 if (!skillInfo)
349 continue;
350
351 if ((skillInfo->categoryId == SKILL_CATEGORY_PROFESSION || skillInfo->categoryId == SKILL_CATEGORY_SECONDARY) &&
352 skillInfo->canLink) // only prof. with recipes have
353 {
354 HandleLearnSkillRecipesHelper(target, skillInfo->id);
355 }
356 }
357
358 handler->SendSysMessage(LANG_COMMAND_LEARN_ALL_CRAFT);
359 return true;
360 }

362 static bool HandleLearnAllRecipesCommand(ChatHandler* handler, char const* args)
363 {
364 // Learns all recipes of specified profession and sets skill to max
365 // Example: .learn all_recipes enchanting
366
367 Player* target = handler->getSelectedPlayer();
368 if (!target)
369 {
370 handler->SendSysMessage(LANG_PLAYER_NOT_FOUND);
371 return false;
372 }
373
374 if (!*args)
375 return false;
376
377 std::wstring namePart;
378
379 if (!Utf8toWStr(args, namePart))
380 return false;
381
382 // converting string that we try to find to lower case
383 wstrToLower(namePart);
384
385 std::string name;
386
387 SkillLineEntry const* targetSkillInfo = NULL;
388 for (uint32 i = 1; i < sSkillLineStore.GetNumRows(); ++i)
389 {
390 SkillLineEntry const* skillInfo = sSkillLineStore.LookupEntry(i);
391 if (!skillInfo)
392 continue;
393
394 if ((skillInfo->categoryId != SKILL_CATEGORY_PROFESSION &&
395 skillInfo->categoryId != SKILL_CATEGORY_SECONDARY) ||
396 !skillInfo->canLink) // only prof with recipes have set
397 continue;
398
399 int locale = handler->GetSessionDbcLocale();
400 name = skillInfo->name[locale];
401 if (name.empty())
402 continue;
403
404 if (!Utf8FitTo(name, namePart))
405 {
406 locale = 0;
407 for (; locale < TOTAL_LOCALES; ++locale)
408 {
409 if (locale == handler->GetSessionDbcLocale())
410 continue;
411
412 name = skillInfo->name[locale];
413 if (name.empty())
414 continue;
415
416 if (Utf8FitTo(name, namePart))
417 break;
418 }
419 }
420
421 if (locale < TOTAL_LOCALES)
422 {
423 targetSkillInfo = skillInfo;
424 break;
425 }
426 }
427
428 if (!targetSkillInfo)
429 return false;
430
431 HandleLearnSkillRecipesHelper(target, targetSkillInfo->id);
432
433 uint16 maxLevel = target->GetPureMaxSkillValue(targetSkillInfo->id);
434 target->SetSkill(targetSkillInfo->id, target->GetSkillStep(targetSkillInfo->id), maxLevel, maxLevel);
435 handler->PSendSysMessage(LANG_COMMAND_LEARN_ALL_RECIPES, name.c_str());
436 return true;
437 }

439 static void HandleLearnSkillRecipesHelper(Player* player, uint32 skillId)
440 {
441 uint32 classmask = player->getClassMask();
442
443 for (uint32 j = 0; j < sSkillLineAbilityStore.GetNumRows(); ++j)
444 {
445 SkillLineAbilityEntry const* skillLine = sSkillLineAbilityStore.LookupEntry(j);
446 if (!skillLine)
447 continue;
448
449 // wrong skill
450 if (skillLine->skillId != skillId)
451 continue;
452
453 // not high rank
454 if (skillLine->forward_spellid)
455 continue;
456
457 // skip racial skills
458 if (skillLine->racemask != 0)
459 continue;
460
461 // skip wrong class skills
462 if (skillLine->classmask && (skillLine->classmask & classmask) == 0)
463 continue;
464
465 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(skillLine->spellId);
466 if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo, player, false))
467 continue;
468
469 player->LearnSpell(skillLine->spellId, false);
470 }
471 }

473 static bool HandleUnLearnCommand(ChatHandler* handler, char const* args)
474 {
475 if (!*args)
476 return false;
477
478 // number or [name] Shift-click form |color|Hspell:spell_id|h[name]|h|r
479 uint32 spellId = handler->extractSpellIdFromLink((char*)args);
480 if (!spellId)
481 return false;
482
483 char const* allStr = strtok(NULL, " ");
484 bool allRanks = allStr ? (strncmp(allStr, "all", strlen(allStr)) == 0) : false;
485
486 Player* target = handler->getSelectedPlayer();
487 if (!target)
488 {
489 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
490 handler->SetSentErrorMessage(true);
491 return false;
492 }
493
494 if (allRanks)
495 spellId = sSpellMgr->GetFirstSpellInChain(spellId);
496
497 if (target->HasSpell(spellId))
498 target->RemoveSpell(spellId, false, !allRanks);
499 else
500 handler->SendSysMessage(LANG_FORGET_SPELL);
501
502 if (GetTalentSpellCost(spellId))
503 target->SendTalentsInfoData(false);
504
505 return true;
506 }

31enum Spells
32{
33 LFG_SPELL_DUNGEON_DESERTER = 71041,
34 BG_SPELL_DESERTER = 26013

37class deserter_commandscript : public CommandScript
38{
39public:
40 deserter_commandscript() : CommandScript("deserter_commandscript") { }

42 /**
43 * @brief Returns the command structure for the system.
44 */
45
46 std::vector<ChatCommand> GetCommands() const override
47 {
48 static std::vector<ChatCommand> deserterInstanceCommandTable =
49 {
50 { "add", rbac::RBAC_PERM_COMMAND_DESERTER_INSTANCE_ADD, false, &HandleDeserterInstanceAdd, "" },
51 { "remove", rbac::RBAC_PERM_COMMAND_DESERTER_INSTANCE_REMOVE, false, &HandleDeserterInstanceRemove, "" },
52 };
53 static std::vector<ChatCommand> deserterBGCommandTable =
54 {
55 { "add", rbac::RBAC_PERM_COMMAND_DESERTER_BG_ADD, false, &HandleDeserterBGAdd, "" },
56 { "remove", rbac::RBAC_PERM_COMMAND_DESERTER_BG_REMOVE, false, &HandleDeserterBGRemove, "" },
57 };
58
59 static std::vector<ChatCommand> deserterCommandTable =
60 {
61 { "instance", rbac::RBAC_PERM_COMMAND_DESERTER_INSTANCE, false, NULL, "", deserterInstanceCommandTable },
62 { "bg", rbac::RBAC_PERM_COMMAND_DESERTER_BG, false, NULL, "", deserterBGCommandTable },
63 };
64 static std::vector<ChatCommand> commandTable =
65 {
66 { "deserter", rbac::RBAC_PERM_COMMAND_DESERTER, false, NULL, "", deserterCommandTable },

71 /**
72 * @brief Applies the Deserter Debuff to a player
73 *
74 * This function applies a Deserter Debuff of the given type (Instance or BG) to the
75 * selected player, with the provided duration in seconds.
76 *
77 * @param handler The ChatHandler, passed by the system.
78 * @param args The provided duration in seconds.
79 * @param isInstance provided by the relaying functions, so we don't have
80 * to write that much code :)
81 *
82 * @return true if everything was correct, false if an error occured.
83 *
84 * Example Usage:
85 * @code
86 * .deserter instance add 3600 (one hour)
87 * -or-
88 * .deserter bg add 3600 (one hour)
89 * @endcode
90 */
91 static bool HandleDeserterAdd(ChatHandler* handler, char const* args, bool isInstance)
92 {
93 if (!*args)
94 return false;
95
96 Player* player = handler->getSelectedPlayer();
97 if (!player)
98 {
99 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
100 handler->SetSentErrorMessage(true);
101 return false;
102 }
103 char* timeStr = strtok((char*)args, " ");
104 if (!timeStr)
105 {
106 handler->SendSysMessage(LANG_BAD_VALUE);
107 handler->SetSentErrorMessage(true);
108 return false;
109 }
110 uint32 time = atoi(timeStr);
111
112 if (!time)
113 {
114 handler->SendSysMessage(LANG_BAD_VALUE);
115 handler->SetSentErrorMessage(true);
116 return false;
117 }
118
119 Aura* aura = player->AddAura(isInstance ? LFG_SPELL_DUNGEON_DESERTER : BG_SPELL_DESERTER, player);
120
121 if (!aura)
122 {
123 handler->SendSysMessage(LANG_BAD_VALUE);
124 handler->SetSentErrorMessage(true);
125 return false;
126 }
127 aura->SetDuration(time * IN_MILLISECONDS);
128
129 return true;
130 }
131
132 /**
133 * @brief Removes the Deserter Debuff from a player
134 *
135 * This function removes a Deserter Debuff of the given type (Instance or BG) from the
136 * selected player.
137 *
138 * @param handler The ChatHandler, passed by the system.
139 * @param args Should be nothing.
140 * @param isInstance provided by the relaying functions, so we don't have
141 * to write that much code :)
142 *
143 * @return true if everything was correct, false if an error occured.
144 *
145 * Example Usage:
146 * @code
147 * .deserter instance remove
148 * -or-
149 * .deserter bg remove
150 * @endcode
151 */
152 static bool HandleDeserterRemove(ChatHandler* handler, char const* /*args*/, bool isInstance)
153 {
154 Player* player = handler->getSelectedPlayer();
155 if (!player)
156 {
157 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
158 handler->SetSentErrorMessage(true);
159 return false;
160 }
161
162 player->RemoveAura(isInstance ? LFG_SPELL_DUNGEON_DESERTER : BG_SPELL_DESERTER);
163
164 return true;
165 }
166
167 /// @sa HandleDeserterAdd()
168 static bool HandleDeserterInstanceAdd(ChatHandler* handler, char const* args)
169 {
170 return HandleDeserterAdd(handler, args, true);
171 }
172
173 /// @sa HandleDeserterAdd()
174 static bool HandleDeserterBGAdd(ChatHandler* handler, char const* args)
175 {
176 return HandleDeserterAdd(handler, args, false);
177 }
178
179 /// @sa HandleDeserterRemove()
180 static bool HandleDeserterInstanceRemove(ChatHandler* handler, char const* args)
181 {
182 return HandleDeserterRemove(handler, args, true);
183 }
184
185 /// @sa HandleDeserterRemove()
186 static bool HandleDeserterBGRemove(ChatHandler* handler, char const* args)
187 {
188 return HandleDeserterRemove(handler, args, false);
189 }
190};

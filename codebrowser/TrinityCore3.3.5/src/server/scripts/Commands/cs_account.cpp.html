74 static bool HandleAccountAddonCommand(ChatHandler* handler, char const* args)
75 {
76 if (!*args)
77 {
78 handler->SendSysMessage(LANG_CMD_SYNTAX);
79 handler->SetSentErrorMessage(true);
80 return false;
81 }
82
83 char* exp = strtok((char*)args, " ");
84
85 uint32 accountId = handler->GetSession()->GetAccountId();
86
87 int expansion = atoi(exp); //get int anyway (0 if error)
88 if (expansion < 0 || uint8(expansion) > sWorld->getIntConfig(CONFIG_EXPANSION))
89 {
90 handler->SendSysMessage(LANG_IMPROPER_VALUE);
91 handler->SetSentErrorMessage(true);
92 return false;
93 }
94
95 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_EXPANSION);
96
97 stmt->setUInt8(0, uint8(expansion));
98 stmt->setUInt32(1, accountId);
99
100 LoginDatabase.Execute(stmt);
101
102 handler->PSendSysMessage(LANG_ACCOUNT_ADDON, expansion);
103 return true;
104 }

106 /// Create an account
107 static bool HandleAccountCreateCommand(ChatHandler* handler, char const* args)
108 {
109 if (!*args)
110 return false;
111
112 std::string email;
113
114 ///- %Parse the command line arguments
115 char* accountName = strtok((char*)args, " ");
116 char* password = strtok(NULL, " ");
117 char* possibleEmail = strtok(NULL, " ' ");
118 if (possibleEmail)
119 email = possibleEmail;
120
121 if (!accountName || !password)
122 return false;
123
124 AccountOpResult result = sAccountMgr->CreateAccount(std::string(accountName), std::string(password), email);
125 switch (result)
126 {
127 case AOR_OK:
128 handler->PSendSysMessage(LANG_ACCOUNT_CREATED, accountName);
129 if (handler->GetSession())
130 {
131 TC_LOG_INFO("entities.player.character", "Account: %d (IP: %s) Character:[%s] (GUID: %u) created Account %s (Email: '%s')",
132 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
133 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter(),
134 accountName, email.c_str());
135 }
136 break;
137 case AOR_NAME_TOO_LONG:
138 handler->SendSysMessage(LANG_ACCOUNT_TOO_LONG);
139 handler->SetSentErrorMessage(true);
140 return false;
141 case AOR_NAME_ALREADY_EXIST:
142 handler->SendSysMessage(LANG_ACCOUNT_ALREADY_EXIST);
143 handler->SetSentErrorMessage(true);
144 return false;
145 case AOR_DB_INTERNAL_ERROR:
146 handler->PSendSysMessage(LANG_ACCOUNT_NOT_CREATED_SQL_ERROR, accountName);
147 handler->SetSentErrorMessage(true);
148 return false;
149 default:
150 handler->PSendSysMessage(LANG_ACCOUNT_NOT_CREATED, accountName);
151 handler->SetSentErrorMessage(true);
152 return false;
153 }
154
155 return true;
156 }

158 /// Delete a user account and all associated characters in this realm
159 /// @todo This function has to be enhanced to respect the login/realm split (delete char, delete account chars in realm then delete account)
160 static bool HandleAccountDeleteCommand(ChatHandler* handler, char const* args)
161 {
162 if (!*args)
163 return false;
164
165 ///- Get the account name from the command line
166 char* account = strtok((char*)args, " ");
167 if (!account)
168 return false;
169
170 std::string accountName = account;
171 if (!AccountMgr::normalizeString(accountName))
172 {
173 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
174 handler->SetSentErrorMessage(true);
175 return false;
176 }
177
178 uint32 accountId = AccountMgr::GetId(accountName);
179 if (!accountId)
180 {
181 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
182 handler->SetSentErrorMessage(true);
183 return false;
184 }
185
186 /// Commands not recommended call from chat, but support anyway
187 /// can delete only for account with less security
188 /// This is also reject self apply in fact
189 if (handler->HasLowerSecurityAccount(NULL, accountId, true))
190 return false;
191
192 AccountOpResult result = AccountMgr::DeleteAccount(accountId);
193 switch (result)
194 {
195 case AOR_OK:
196 handler->PSendSysMessage(LANG_ACCOUNT_DELETED, accountName.c_str());
197 break;
198 case AOR_NAME_NOT_EXIST:
199 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
200 handler->SetSentErrorMessage(true);
201 return false;
202 case AOR_DB_INTERNAL_ERROR:
203 handler->PSendSysMessage(LANG_ACCOUNT_NOT_DELETED_SQL_ERROR, accountName.c_str());
204 handler->SetSentErrorMessage(true);
205 return false;
206 default:
207 handler->PSendSysMessage(LANG_ACCOUNT_NOT_DELETED, accountName.c_str());
208 handler->SetSentErrorMessage(true);
209 return false;
210 }
211
212 return true;

215 /// Display info on users currently in the realm
216 static bool HandleAccountOnlineListCommand(ChatHandler* handler, char const* /*args*/)
217 {
218 ///- Get the list of accounts ID logged to the realm
219 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHARACTER_ONLINE);
220
221 PreparedQueryResult result = CharacterDatabase.Query(stmt);
222
223 if (!result)
224 {
225 handler->SendSysMessage(LANG_ACCOUNT_LIST_EMPTY);
226 return true;
227 }
228
229 ///- Display the list of account/characters online
230 handler->SendSysMessage(LANG_ACCOUNT_LIST_BAR_HEADER);
231 handler->SendSysMessage(LANG_ACCOUNT_LIST_HEADER);
232 handler->SendSysMessage(LANG_ACCOUNT_LIST_BAR);
233
234 ///- Cycle through accounts
235 do
236 {
237 Field* fieldsDB = result->Fetch();
238 std::string name = fieldsDB[0].GetString();
239 uint32 account = fieldsDB[1].GetUInt32();
240
241 ///- Get the username, last IP and GM level of each account
242 // No SQL injection. account is uint32.
243 stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_ACCOUNT_INFO);
244 stmt->setUInt32(0, account);
245 PreparedQueryResult resultLogin = LoginDatabase.Query(stmt);
246
247 if (resultLogin)
248 {
249 Field* fieldsLogin = resultLogin->Fetch();
250 handler->PSendSysMessage(LANG_ACCOUNT_LIST_LINE,
251 fieldsLogin[0].GetCString(), name.c_str(), fieldsLogin[1].GetCString(),
252 fieldsDB[2].GetUInt16(), fieldsDB[3].GetUInt16(), fieldsLogin[3].GetUInt8(),
253 fieldsLogin[2].GetUInt8());
254 }
255 else
256 handler->PSendSysMessage(LANG_ACCOUNT_LIST_ERROR, name.c_str());
257 }
258 while (result->NextRow());
259
260 handler->SendSysMessage(LANG_ACCOUNT_LIST_BAR);
261 return true;
262 }

270 return false;
271 }
272 std::string param = (char*)args;
273
274 if (!param.empty())
275 {
276 if (param == "on")
277 {
278 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_LOGON_COUNTRY);
279 uint32 ip = inet_addr(handler->GetSession()->GetRemoteAddress().c_str());
280 EndianConvertReverse(ip);
281 stmt->setUInt32(0, ip);
282 PreparedQueryResult result = LoginDatabase.Query(stmt);
283 if (result)
284 {
285 Field* fields = result->Fetch();
286 std::string country = fields[0].GetString();
287 stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_ACCOUNT_LOCK_CONTRY);
288 stmt->setString(0, country);
289 stmt->setUInt32(1, handler->GetSession()->GetAccountId());
290 LoginDatabase.Execute(stmt);
291 handler->PSendSysMessage(LANG_COMMAND_ACCLOCKLOCKED);
292 }
293 else
294 {
295 handler->PSendSysMessage("[IP2NATION] Table empty");
296 TC_LOG_DEBUG("server.authserver", "[IP2NATION] Table empty");
297 }
298 }
299 else if (param == "off")
300 {
301 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_ACCOUNT_LOCK_CONTRY);
302 stmt->setString(0, "00");
303 stmt->setUInt32(1, handler->GetSession()->GetAccountId());
304 LoginDatabase.Execute(stmt);
305 handler->PSendSysMessage(LANG_COMMAND_ACCLOCKUNLOCKED);

309 handler->SendSysMessage(LANG_USE_BOL);
310 handler->SetSentErrorMessage(true);

351 static bool HandleAccountEmailCommand(ChatHandler* handler, char const* args)
352 {
353 if (!*args)
354 {
355 handler->SendSysMessage(LANG_CMD_SYNTAX);
356 handler->SetSentErrorMessage(true);
357 return false;
358 }
359
360 char* oldEmail = strtok((char*)args, " ");
361 char* password = strtok(NULL, " ");
362 char* email = strtok(NULL, " ");
363 char* emailConfirmation = strtok(NULL, " ");
364
365 if (!oldEmail || !password || !email || !emailConfirmation)
366 {
367 handler->SendSysMessage(LANG_CMD_SYNTAX);
368 handler->SetSentErrorMessage(true);
369 return false;
370 }
371
372 if (!AccountMgr::CheckEmail(handler->GetSession()->GetAccountId(), std::string(oldEmail)))
373 {
374 handler->SendSysMessage(LANG_COMMAND_WRONGEMAIL);
375 sScriptMgr->OnFailedEmailChange(handler->GetSession()->GetAccountId());
376 handler->SetSentErrorMessage(true);
377 TC_LOG_INFO("entities.player.character", "Account: %u (IP: %s) Character:[%s] (GUID: %u) Tried to change email, but the provided email [%s] is not equal to registration email [%s].",
378 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
379 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter(),
380 email, oldEmail);
381 return false;
382 }
383
384 if (!AccountMgr::CheckPassword(handler->GetSession()->GetAccountId(), std::string(password)))
385 {
386 handler->SendSysMessage(LANG_COMMAND_WRONGOLDPASSWORD);
387 sScriptMgr->OnFailedEmailChange(handler->GetSession()->GetAccountId());
388 handler->SetSentErrorMessage(true);
389 TC_LOG_INFO("entities.player.character", "Account: %u (IP: %s) Character:[%s] (GUID: %u) Tried to change email, but the provided password is wrong.",
390 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
391 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter());
392 return false;
393 }
394
395 if (strcmp(email, oldEmail) == 0)
396 {
397 handler->SendSysMessage(LANG_OLD_EMAIL_IS_NEW_EMAIL);
398 sScriptMgr->OnFailedEmailChange(handler->GetSession()->GetAccountId());
399 handler->SetSentErrorMessage(true);
400 return false;
401 }
402
403 if (strcmp(email, emailConfirmation) != 0)
404 {
405 handler->SendSysMessage(LANG_NEW_EMAILS_NOT_MATCH);
406 sScriptMgr->OnFailedEmailChange(handler->GetSession()->GetAccountId());
407 handler->SetSentErrorMessage(true);
408 TC_LOG_INFO("entities.player.character", "Account: %u (IP: %s) Character:[%s] (GUID: %u) Tried to change email, but the provided password is wrong.",
409 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
410 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter());
411 return false;
412 }
413
414
415 AccountOpResult result = AccountMgr::ChangeEmail(handler->GetSession()->GetAccountId(), std::string(email));
416 switch (result)
417 {
418 case AOR_OK:
419 handler->SendSysMessage(LANG_COMMAND_EMAIL);
420 sScriptMgr->OnEmailChange(handler->GetSession()->GetAccountId());
421 TC_LOG_INFO("entities.player.character", "Account: %u (IP: %s) Character:[%s] (GUID: %u) Changed Email from [%s] to [%s].",
422 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
423 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter(),
424 oldEmail, email);
425 break;
426 case AOR_EMAIL_TOO_LONG:
427 handler->SendSysMessage(LANG_EMAIL_TOO_LONG);
428 sScriptMgr->OnFailedEmailChange(handler->GetSession()->GetAccountId());
429 handler->SetSentErrorMessage(true);
430 return false;
431 default:
432 handler->SendSysMessage(LANG_COMMAND_NOTCHANGEEMAIL);
433 handler->SetSentErrorMessage(true);
434 return false;
435 }
436
437 return true;
438 }

440 static bool HandleAccountPasswordCommand(ChatHandler* handler, char const* args)
441 {
442 // If no args are given at all, we can return false right away.
443 if (!*args)
444 {
445 handler->SendSysMessage(LANG_CMD_SYNTAX);
446 handler->SetSentErrorMessage(true);
447 return false;
448 }
449
450 // First, we check config. What security type (sec type) is it ? Depending on it, the command branches out
451 uint32 pwConfig = sWorld->getIntConfig(CONFIG_ACC_PASSCHANGESEC); // 0 - PW_NONE, 1 - PW_EMAIL, 2 - PW_RBAC
452
453 // Command is supposed to be: .account password [$oldpassword] [$newpassword] [$newpasswordconfirmation] [$emailconfirmation]
454 char* oldPassword = strtok((char*)args, " "); // This extracts [$oldpassword]
455 char* newPassword = strtok(NULL, " "); // This extracts [$newpassword]
456 char* passwordConfirmation = strtok(NULL, " "); // This extracts [$newpasswordconfirmation]
457 char const* emailConfirmation = strtok(NULL, " "); // This defines the emailConfirmation variable, which is optional depending on sec type.
458 if (!emailConfirmation) // This extracts [$emailconfirmation]. If it doesn't exist, however...
459 emailConfirmation = ""; // ... it's simply "" for emailConfirmation.
460
461 //Is any of those variables missing for any reason ? We return false.
462 if (!oldPassword || !newPassword || !passwordConfirmation)
463 {
464 handler->SendSysMessage(LANG_CMD_SYNTAX);
465 handler->SetSentErrorMessage(true);
466 return false;
467 }
468
469 // We compare the old, saved password to the entered old password - no chance for the unauthorized.
470 if (!AccountMgr::CheckPassword(handler->GetSession()->GetAccountId(), std::string(oldPassword)))
471 {
472 handler->SendSysMessage(LANG_COMMAND_WRONGOLDPASSWORD);
473 sScriptMgr->OnFailedPasswordChange(handler->GetSession()->GetAccountId());
474 handler->SetSentErrorMessage(true);
475 TC_LOG_INFO("entities.player.character", "Account: %u (IP: %s) Character:[%s] (GUID: %u) Tried to change password, but the provided old password is wrong.",
476 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
477 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter());
478 return false;
479 }
480
481 // This compares the old, current email to the entered email - however, only...
482 if ((pwConfig == PW_EMAIL || (pwConfig == PW_RBAC && handler->HasPermission(rbac::RBAC_PERM_EMAIL_CONFIRM_FOR_PASS_CHANGE))) // ...if either PW_EMAIL or PW_RBAC with the Permission is active...
483 && !AccountMgr::CheckEmail(handler->GetSession()->GetAccountId(), std::string(emailConfirmation))) // ... and returns false if the comparison fails.
484 {
485 handler->SendSysMessage(LANG_COMMAND_WRONGEMAIL);
486 sScriptMgr->OnFailedPasswordChange(handler->GetSession()->GetAccountId());
487 handler->SetSentErrorMessage(true);
488 TC_LOG_INFO("entities.player.character", "Account: %u (IP: %s) Character:[%s] (GUID: %u) Tried to change password, but the entered email [%s] is wrong.",
489 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
490 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter(),
491 emailConfirmation);
492 return false;
493 }
494
495 // Making sure that newly entered password is correctly entered.
496 if (strcmp(newPassword, passwordConfirmation) != 0)
497 {
498 handler->SendSysMessage(LANG_NEW_PASSWORDS_NOT_MATCH);
499 sScriptMgr->OnFailedPasswordChange(handler->GetSession()->GetAccountId());
500 handler->SetSentErrorMessage(true);
501 return false;
502 }
503
504 // Changes password and prints result.
505 AccountOpResult result = AccountMgr::ChangePassword(handler->GetSession()->GetAccountId(), std::string(newPassword));
506 switch (result)
507 {
508 case AOR_OK:
509 handler->SendSysMessage(LANG_COMMAND_PASSWORD);
510 sScriptMgr->OnPasswordChange(handler->GetSession()->GetAccountId());
511 TC_LOG_INFO("entities.player.character", "Account: %u (IP: %s) Character:[%s] (GUID: %u) Changed Password.",
512 handler->GetSession()->GetAccountId(), handler->GetSession()->GetRemoteAddress().c_str(),
513 handler->GetSession()->GetPlayer()->GetName().c_str(), handler->GetSession()->GetPlayer()->GetGUID().GetCounter());
514 break;
515 case AOR_PASS_TOO_LONG:
516 handler->SendSysMessage(LANG_PASSWORD_TOO_LONG);
517 sScriptMgr->OnFailedPasswordChange(handler->GetSession()->GetAccountId());
518 handler->SetSentErrorMessage(true);
519 return false;
520 default:
521 handler->SendSysMessage(LANG_COMMAND_NOTCHANGEPASSWORD);
522 handler->SetSentErrorMessage(true);
523 return false;
524 }
525
526 return true;
527 }

529 static bool HandleAccountCommand(ChatHandler* handler, char const* /*args*/)
530 {
531 // GM Level
532 AccountTypes gmLevel = handler->GetSession()->GetSecurity();
533 handler->PSendSysMessage(LANG_ACCOUNT_LEVEL, uint32(gmLevel));
534
535 // Security level required
536 bool hasRBAC = (handler->HasPermission(rbac::RBAC_PERM_EMAIL_CONFIRM_FOR_PASS_CHANGE) ? true : false);
537 uint32 pwConfig = sWorld->getIntConfig(CONFIG_ACC_PASSCHANGESEC); // 0 - PW_NONE, 1 - PW_EMAIL, 2 - PW_RBAC
538
539 handler->PSendSysMessage(LANG_ACCOUNT_SEC_TYPE, (pwConfig == PW_NONE ? "Lowest level: No Email input required." :
540 pwConfig == PW_EMAIL ? "Highest level: Email input required." :
541 pwConfig == PW_RBAC ? "Special level: Your account may require email input depending on settings. That is the case if another lien is printed." :
542 "Unknown security level: Notify technician for details."));
543
544 // RBAC required display - is not displayed for console
545 if (pwConfig == PW_RBAC && handler->GetSession() && hasRBAC)
546 handler->PSendSysMessage(LANG_RBAC_EMAIL_REQUIRED);
547
548 // Email display if sufficient rights
549 if (handler->HasPermission(rbac::RBAC_PERM_MAY_CHECK_OWN_EMAIL))
550 {
551 std::string emailoutput;
552 uint32 accountId = handler->GetSession()->GetAccountId();
553
554 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_GET_EMAIL_BY_ID);
555 stmt->setUInt32(0, accountId);
556 PreparedQueryResult result = LoginDatabase.Query(stmt);
557
558 if (result)
559 {
560 emailoutput = (*result)[0].GetString();
561 handler->PSendSysMessage(LANG_COMMAND_EMAIL_OUTPUT, emailoutput.c_str());
562 }
563 }
564
565 return true;
566 }

568 /// Set/Unset the expansion level for an account
569 static bool HandleAccountSetAddonCommand(ChatHandler* handler, char const* args)
570 {
571 ///- Get the command line arguments
572 char* account = strtok((char*)args, " ");
573 char* exp = strtok(NULL, " ");
574
575 if (!account)
576 return false;
577
578 std::string accountName;
579 uint32 accountId;
580
581 if (!exp)
582 {
583 Player* player = handler->getSelectedPlayer();
584 if (!player)
585 return false;
586
587 accountId = player->GetSession()->GetAccountId();
588 AccountMgr::GetName(accountId, accountName);
589 exp = account;
590 }
591 else
592 {
593 ///- Convert Account name to Upper Format
594 accountName = account;
595 if (!AccountMgr::normalizeString(accountName))
596 {
597 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
598 handler->SetSentErrorMessage(true);
599 return false;
600 }
601
602 accountId = AccountMgr::GetId(accountName);
603 if (!accountId)
604 {
605 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
606 handler->SetSentErrorMessage(true);
607 return false;
608 }
609 }
610
611 // Let set addon state only for lesser (strong) security level
612 // or to self account
613 if (handler->GetSession() && handler->GetSession()->GetAccountId() != accountId &&
614 handler->HasLowerSecurityAccount(NULL, accountId, true))
615 return false;
616
617 int expansion = atoi(exp); //get int anyway (0 if error)
618 if (expansion < 0 || uint8(expansion) > sWorld->getIntConfig(CONFIG_EXPANSION))
619 return false;
620
621 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_EXPANSION);
622
623 stmt->setUInt8(0, expansion);
624 stmt->setUInt32(1, accountId);
625
626 LoginDatabase.Execute(stmt);
627
628 handler->PSendSysMessage(LANG_ACCOUNT_SETADDON, accountName.c_str(), accountId, expansion);
629 return true;
630 }

632 static bool HandleAccountSetGmLevelCommand(ChatHandler* handler, char const* args)
633 {
634 if (!*args)
635 {
636 handler->SendSysMessage(LANG_CMD_SYNTAX);
637 handler->SetSentErrorMessage(true);
638 return false;
639 }
640
641 std::string targetAccountName;
642 uint32 targetAccountId = 0;
643 uint32 targetSecurity = 0;
644 uint32 gm = 0;
645 char* arg1 = strtok((char*)args, " ");
646 char* arg2 = strtok(NULL, " ");
647 char* arg3 = strtok(NULL, " ");
648 bool isAccountNameGiven = true;
649
650 if (!arg3)
651 {
652 if (!handler->getSelectedPlayer())
653 return false;
654 isAccountNameGiven = false;
655 }
656
657 // Check for second parameter
658 if (!isAccountNameGiven && !arg2)
659 return false;
660
661 // Check for account
662 if (isAccountNameGiven)
663 {
664 targetAccountName = arg1;
665 if (!AccountMgr::normalizeString(targetAccountName) || !AccountMgr::GetId(targetAccountName))
666 {
667 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, targetAccountName.c_str());
668 handler->SetSentErrorMessage(true);
669 return false;
670 }
671 }
672
673 // Check for invalid specified GM level.
674 gm = (isAccountNameGiven) ? atoi(arg2) : atoi(arg1);
675 if (gm > SEC_CONSOLE)
676 {
677 handler->SendSysMessage(LANG_BAD_VALUE);
678 handler->SetSentErrorMessage(true);
679 return false;
680 }
681
682 // handler->getSession() == NULL only for console
683 targetAccountId = (isAccountNameGiven) ? AccountMgr::GetId(targetAccountName) : handler->getSelectedPlayer()->GetSession()->GetAccountId();
684 int32 gmRealmID = (isAccountNameGiven) ? atoi(arg3) : atoi(arg2);
685 uint32 playerSecurity;
686 if (handler->GetSession())
687 playerSecurity = AccountMgr::GetSecurity(handler->GetSession()->GetAccountId(), gmRealmID);
688 else
689 playerSecurity = SEC_CONSOLE;
690
691 // can set security level only for target with less security and to less security that we have
692 // This also restricts setting handler's own security.
693 targetSecurity = AccountMgr::GetSecurity(targetAccountId, gmRealmID);
694 if (targetSecurity >= playerSecurity || gm >= playerSecurity)
695 {
696 handler->SendSysMessage(LANG_YOURS_SECURITY_IS_LOW);
697 handler->SetSentErrorMessage(true);
698 return false;
699 }
700
701 // Check and abort if the target gm has a higher rank on one of the realms and the new realm is -1
702 if (gmRealmID == -1 && !AccountMgr::IsConsoleAccount(playerSecurity))
703 {
704 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_ACCOUNT_ACCESS_GMLEVEL_TEST);
705
706 stmt->setUInt32(0, targetAccountId);
707 stmt->setUInt8(1, uint8(gm));
708
709 PreparedQueryResult result = LoginDatabase.Query(stmt);
710
711 if (result)
712 {
713 handler->SendSysMessage(LANG_YOURS_SECURITY_IS_LOW);
714 handler->SetSentErrorMessage(true);
715 return false;
716 }
717 }
718
719 // Check if provided realmID has a negative value other than -1
720 if (gmRealmID < -1)
721 {
722 handler->SendSysMessage(LANG_INVALID_REALMID);
723 handler->SetSentErrorMessage(true);
724 return false;
725 }
726
727 rbac::RBACData* rbac = isAccountNameGiven ? NULL : handler->getSelectedPlayer()->GetSession()->GetRBACData();
728 sAccountMgr->UpdateAccountAccess(rbac, targetAccountId, uint8(gm), gmRealmID);
729
730 handler->PSendSysMessage(LANG_YOU_CHANGE_SECURITY, targetAccountName.c_str(), gm);
731 return true;
732 }

734 /// Set password for account
735 static bool HandleAccountSetPasswordCommand(ChatHandler* handler, char const* args)
736 {
737 if (!*args)
738 {
739 handler->SendSysMessage(LANG_CMD_SYNTAX);
740 handler->SetSentErrorMessage(true);
741 return false;
742 }
743
744 ///- Get the command line arguments
745 char* account = strtok((char*)args, " ");
746 char* password = strtok(NULL, " ");
747 char* passwordConfirmation = strtok(NULL, " ");
748
749 if (!account || !password || !passwordConfirmation)
750 return false;
751
752 std::string accountName = account;
753 if (!AccountMgr::normalizeString(accountName))
754 {
755 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
756 handler->SetSentErrorMessage(true);
757 return false;
758 }
759
760 uint32 targetAccountId = AccountMgr::GetId(accountName);
761 if (!targetAccountId)
762 {
763 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
764 handler->SetSentErrorMessage(true);
765 return false;
766 }
767
768 /// can set password only for target with less security
769 /// This also restricts setting handler's own password
770 if (handler->HasLowerSecurityAccount(NULL, targetAccountId, true))
771 return false;
772
773 if (strcmp(password, passwordConfirmation) != 0)
774 {
775 handler->SendSysMessage(LANG_NEW_PASSWORDS_NOT_MATCH);
776 handler->SetSentErrorMessage(true);
777 return false;
778 }
779
780 AccountOpResult result = AccountMgr::ChangePassword(targetAccountId, password);
781
782 switch (result)
783 {
784 case AOR_OK:
785 handler->SendSysMessage(LANG_COMMAND_PASSWORD);
786 break;
787 case AOR_NAME_NOT_EXIST:
788 handler->PSendSysMessage(LANG_ACCOUNT_NOT_EXIST, accountName.c_str());
789 handler->SetSentErrorMessage(true);
790 return false;
791 case AOR_PASS_TOO_LONG:
792 handler->SendSysMessage(LANG_PASSWORD_TOO_LONG);
793 handler->SetSentErrorMessage(true);
794 return false;
795 default:
796 handler->SendSysMessage(LANG_COMMAND_NOTCHANGEPASSWORD);
797 handler->SetSentErrorMessage(true);
798 return false;
799 }
800 return true;

36 std::vector<ChatCommand> GetCommands() const override
37 {
38 static std::vector<ChatCommand> questCommandTable =
39 {
40 { "add", rbac::RBAC_PERM_COMMAND_QUEST_ADD, false, &HandleQuestAdd, "" },
41 { "complete", rbac::RBAC_PERM_COMMAND_QUEST_COMPLETE, false, &HandleQuestComplete, "" },
42 { "remove", rbac::RBAC_PERM_COMMAND_QUEST_REMOVE, false, &HandleQuestRemove, "" },
43 { "reward", rbac::RBAC_PERM_COMMAND_QUEST_REWARD, false, &HandleQuestReward, "" },
44 };
45 static std::vector<ChatCommand> commandTable =
46 {
47 { "quest", rbac::RBAC_PERM_COMMAND_QUEST, false, NULL, "", questCommandTable },
48 };
49 return commandTable;
50 }

52 static bool HandleQuestAdd(ChatHandler* handler, const char* args)
53 {
54 Player* player = handler->getSelectedPlayerOrSelf();
55 if (!player)
56 {
57 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
58 handler->SetSentErrorMessage(true);
59 return false;
60 }
61
62 // .addquest #entry'
63 // number or [name] Shift-click form |color|Hquest:quest_id:quest_level|h[name]|h|r
64 char* cId = handler->extractKeyFromLink((char*)args, "Hquest");
65 if (!cId)
66 return false;
67
68 uint32 entry = atoul(cId);
69
70 Quest const* quest = sObjectMgr->GetQuestTemplate(entry);
71
72 if (!quest)
73 {
74 handler->PSendSysMessage(LANG_COMMAND_QUEST_NOTFOUND, entry);
75 handler->SetSentErrorMessage(true);
76 return false;
77 }
78
79 // check item starting quest (it can work incorrectly if added without item in inventory)
80 ItemTemplateContainer const* itc = sObjectMgr->GetItemTemplateStore();
81 ItemTemplateContainer::const_iterator result = find_if (itc->begin(), itc->end(), Finder<uint32, ItemTemplate>(entry, &ItemTemplate::StartQuest));
82
83 if (result != itc->end())
84 {
85 handler->PSendSysMessage(LANG_COMMAND_QUEST_STARTFROMITEM, entry, result->second.ItemId);
86 handler->SetSentErrorMessage(true);
87 return false;
88 }
89
90 // ok, normal (creature/GO starting) quest
91 if (player->CanAddQuest(quest, true))
92 player->AddQuestAndCheckCompletion(quest, NULL);
93
94 return true;
95 }

97 static bool HandleQuestRemove(ChatHandler* handler, const char* args)
98 {
99 Player* player = handler->getSelectedPlayer();
100 if (!player)
101 {
102 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
103 handler->SetSentErrorMessage(true);
104 return false;
105 }
106
107 // .removequest #entry'
108 // number or [name] Shift-click form |color|Hquest:quest_id:quest_level|h[name]|h|r
109 char* cId = handler->extractKeyFromLink((char*)args, "Hquest");
110 if (!cId)
111 return false;
112
113 uint32 entry = atoul(cId);
114
115 Quest const* quest = sObjectMgr->GetQuestTemplate(entry);
116
117 if (!quest)
118 {
119 handler->PSendSysMessage(LANG_COMMAND_QUEST_NOTFOUND, entry);
120 handler->SetSentErrorMessage(true);
121 return false;
122 }
123
124 // remove all quest entries for 'entry' from quest log
125 for (uint8 slot = 0; slot < MAX_QUEST_LOG_SIZE; ++slot)
126 {
127 uint32 logQuest = player->GetQuestSlotQuestId(slot);
128 if (logQuest == entry)
129 {
130 player->SetQuestSlot(slot, 0);
131
132 // we ignore unequippable quest items in this case, its' still be equipped
133 player->TakeQuestSourceItem(logQuest, false);
134
135 if (quest->HasFlag(QUEST_FLAGS_FLAGS_PVP))
136 {
137 player->pvpInfo.IsHostile = player->pvpInfo.IsInHostileArea || player->HasPvPForcingQuest();
138 player->UpdatePvPState();
139 }
140 }
141 }
142
143 player->RemoveActiveQuest(entry, false);
144 player->RemoveRewardedQuest(entry);
145
146 handler->SendSysMessage(LANG_COMMAND_QUEST_REMOVED);
147 return true;
148 }

150 static bool HandleQuestComplete(ChatHandler* handler, const char* args)
151 {
152 Player* player = handler->getSelectedPlayerOrSelf();
153 if (!player)
154 {
155 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
156 handler->SetSentErrorMessage(true);
157 return false;
158 }
159
160 // .quest complete #entry
161 // number or [name] Shift-click form |color|Hquest:quest_id:quest_level|h[name]|h|r
162 char* cId = handler->extractKeyFromLink((char*)args, "Hquest");
163 if (!cId)
164 return false;
165
166 uint32 entry = atoul(cId);
167
168 Quest const* quest = sObjectMgr->GetQuestTemplate(entry);
169
170 // If player doesn't have the quest
171 if (!quest || player->GetQuestStatus(entry) == QUEST_STATUS_NONE)
172 {
173 handler->PSendSysMessage(LANG_COMMAND_QUEST_NOTFOUND, entry);
174 handler->SetSentErrorMessage(true);
175 return false;
176 }
177
178 // Add quest items for quests that require items
179 for (uint8 x = 0; x < QUEST_ITEM_OBJECTIVES_COUNT; ++x)
180 {
181 uint32 id = quest->RequiredItemId[x];
182 uint32 count = quest->RequiredItemCount[x];
183 if (!id || !count)
184 continue;
185
186 uint32 curItemCount = player->GetItemCount(id, true);
187
188 ItemPosCountVec dest;
189 uint8 msg = player->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, id, count-curItemCount);
190 if (msg == EQUIP_ERR_OK)
191 {
192 Item* item = player->StoreNewItem(dest, id, true);
193 player->SendNewItem(item, count-curItemCount, true, false);
194 }
195 }
196
197 // All creature/GO slain/cast (not required, but otherwise it will display "Creature slain 0/10")
198 for (uint8 i = 0; i < QUEST_OBJECTIVES_COUNT; ++i)
199 {
200 int32 creature = quest->RequiredNpcOrGo[i];
201 uint32 creatureCount = quest->RequiredNpcOrGoCount[i];
202
203 if (creature > 0)
204 {
205 if (CreatureTemplate const* creatureInfo = sObjectMgr->GetCreatureTemplate(creature))
206 for (uint16 z = 0; z < creatureCount; ++z)
207 player->KilledMonster(creatureInfo, ObjectGuid::Empty);
208 }
209 else if (creature < 0)
210 for (uint16 z = 0; z < creatureCount; ++z)
211 player->KillCreditGO(creature);
212 }
213
214 // If the quest requires reputation to complete
215 if (uint32 repFaction = quest->GetRepObjectiveFaction())
216 {
217 uint32 repValue = quest->GetRepObjectiveValue();
218 uint32 curRep = player->GetReputationMgr().GetReputation(repFaction);
219 if (curRep < repValue)
220 if (FactionEntry const* factionEntry = sFactionStore.LookupEntry(repFaction))
221 player->GetReputationMgr().SetReputation(factionEntry, repValue);
222 }
223
224 // If the quest requires a SECOND reputation to complete
225 if (uint32 repFaction = quest->GetRepObjectiveFaction2())
226 {
227 uint32 repValue2 = quest->GetRepObjectiveValue2();
228 uint32 curRep = player->GetReputationMgr().GetReputation(repFaction);
229 if (curRep < repValue2)
230 if (FactionEntry const* factionEntry = sFactionStore.LookupEntry(repFaction))
231 player->GetReputationMgr().SetReputation(factionEntry, repValue2);
232 }
233
234 // If the quest requires money
235 int32 ReqOrRewMoney = quest->GetRewOrReqMoney();
236 if (ReqOrRewMoney < 0)
237 player->ModifyMoney(-ReqOrRewMoney);
238
239 if (sWorld->getBoolConfig(CONFIG_QUEST_ENABLE_QUEST_TRACKER)) // check if Quest Tracker is enabled
240 {
241 // prepare Quest Tracker datas
242 PreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_QUEST_TRACK_GM_COMPLETE);
243 stmt->setUInt32(0, quest->GetQuestId());
244 stmt->setUInt32(1, player->GetGUID().GetCounter());
245
246 // add to Quest Tracker
247 CharacterDatabase.Execute(stmt);
248 }
249
250 player->CompleteQuest(entry);
251 return true;
252 }

254 static bool HandleQuestReward(ChatHandler* handler, char const* args)
255 {
256 Player* player = handler->getSelectedPlayer();
257 if (!player)
258 {
259 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
260 handler->SetSentErrorMessage(true);
261 return false;
262 }
263
264 // .quest reward #entry
265 // number or [name] Shift-click form |color|Hquest:quest_id:quest_level|h[name]|h|r
266 char* cId = handler->extractKeyFromLink((char*)args, "Hquest");
267 if (!cId)
268 return false;
269
270 uint32 entry = atoul(cId);
271
272 Quest const* quest = sObjectMgr->GetQuestTemplate(entry);
273
274 // If player doesn't have the quest
275 if (!quest || player->GetQuestStatus(entry) != QUEST_STATUS_COMPLETE)
276 {
277 handler->PSendSysMessage(LANG_COMMAND_QUEST_NOTFOUND, entry);
278 handler->SetSentErrorMessage(true);
279 return false;
280 }
281
282 player->RewardQuest(quest, 0, player);
283 return true;
284 }
285};

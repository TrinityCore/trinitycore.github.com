57 static bool HandleGodModeCheatCommand(ChatHandler* handler, const char* args)
58 {
59 if (!handler->GetSession() && !handler->GetSession()->GetPlayer())
60 return false;
61
62 std::string argstr = (char*)args;
63
64 if (!*args)
65 argstr = (handler->GetSession()->GetPlayer()->GetCommandStatus(CHEAT_GOD)) ? "off" : "on";
66
67 if (argstr == "off")
68 {
69 handler->GetSession()->GetPlayer()->SetCommandStatusOff(CHEAT_GOD);
70 handler->SendSysMessage("Godmode is OFF. You can take damage.");
71 return true;
72 }
73 else if (argstr == "on")
74 {
75 handler->GetSession()->GetPlayer()->SetCommandStatusOn(CHEAT_GOD);
76 handler->SendSysMessage("Godmode is ON. You won't take damage.");
77 return true;
78 }
79
80 return false;
81 }

161 static bool HandleCheatStatusCommand(ChatHandler* handler, const char* /*args*/)
162 {
163 Player* player = handler->GetSession()->GetPlayer();
164
165 const char* enabled = "ON";
166 const char* disabled = "OFF";
167
168 handler->SendSysMessage(LANG_COMMAND_CHEAT_STATUS);
169 handler->PSendSysMessage(LANG_COMMAND_CHEAT_GOD, player->GetCommandStatus(CHEAT_GOD) ? enabled : disabled);
170 handler->PSendSysMessage(LANG_COMMAND_CHEAT_CD, player->GetCommandStatus(CHEAT_COOLDOWN) ? enabled : disabled);
171 handler->PSendSysMessage(LANG_COMMAND_CHEAT_CT, player->GetCommandStatus(CHEAT_CASTTIME) ? enabled : disabled);
172 handler->PSendSysMessage(LANG_COMMAND_CHEAT_POWER, player->GetCommandStatus(CHEAT_POWER) ? enabled : disabled);
173 handler->PSendSysMessage(LANG_COMMAND_CHEAT_WW, player->GetCommandStatus(CHEAT_WATERWALK) ? enabled : disabled);
174 handler->PSendSysMessage(LANG_COMMAND_CHEAT_TAXINODES, player->isTaxiCheater() ? enabled : disabled);
175
176 return true;
177 }

207 static bool HandleTaxiCheatCommand(ChatHandler* handler, const char* args)
208 {
209 std::string argstr = (char*)args;
210 Player* chr = handler->getSelectedPlayer();
211
212 if (!chr)
213 chr = handler->GetSession()->GetPlayer();
214 else if (handler->HasLowerSecurity(chr, ObjectGuid::Empty)) // check online security
215 return false;
216
217 if (!*args)
218 {
219 argstr = (chr->isTaxiCheater()) ? "off" : "on";
220 }
221
222 if (argstr == "off")
223 {
224 chr->SetTaxiCheater(false);
225 handler->PSendSysMessage(LANG_YOU_REMOVE_TAXIS, handler->GetNameLink(chr).c_str());
226 if (handler->needReportToTarget(chr))
227 ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_TAXIS_REMOVED, handler->GetNameLink().c_str());
228 return true;
229 }
230 else if (argstr == "on")
231 {
232 chr->SetTaxiCheater(true);
233 handler->PSendSysMessage(LANG_YOU_GIVE_TAXIS, handler->GetNameLink(chr).c_str());
234 if (handler->needReportToTarget(chr))
235 ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_TAXIS_ADDED, handler->GetNameLink().c_str());
236 return true;
237 }
238
239 handler->SendSysMessage(LANG_USE_BOL);
240 handler->SetSentErrorMessage(true);
241 return false;
242 }

244 static bool HandleExploreCheatCommand(ChatHandler* handler, const char *args)
245 {
246 if (!*args)
247 return false;
248
249 // std::int flag = (char*)args;
250 int flag = atoi((char*)args);
251
252 Player* chr = handler->getSelectedPlayer();
253 if (!chr)
254 {
255 handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
256 handler->SetSentErrorMessage(true);
257 return false;
258 }
259
260 if (flag != 0)
261 {
262 handler->PSendSysMessage(LANG_YOU_SET_EXPLORE_ALL, handler->GetNameLink(chr).c_str());
263 if (handler->needReportToTarget(chr))
264 ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_EXPLORE_SET_ALL, handler->GetNameLink().c_str());
265 }
266 else
267 {
268 handler->PSendSysMessage(LANG_YOU_SET_EXPLORE_NOTHING, handler->GetNameLink(chr).c_str());
269 if (handler->needReportToTarget(chr))
270 ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_EXPLORE_SET_NOTHING, handler->GetNameLink().c_str());
271 }
272
273 for (uint8 i = 0; i < PLAYER_EXPLORED_ZONES_SIZE; ++i)
274 {
275 if (flag != 0)
276 handler->GetSession()->GetPlayer()->SetFlag(PLAYER_EXPLORED_ZONES_1+i, 0xFFFFFFFF);
277 else
278 handler->GetSession()->GetPlayer()->SetFlag(PLAYER_EXPLORED_ZONES_1+i, 0);
279 }
280
281 return true;
282 }

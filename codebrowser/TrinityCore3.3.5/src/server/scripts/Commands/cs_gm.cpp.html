57 // Enables or disables hiding of the staff badge
58 static bool HandleGMChatCommand(ChatHandler* handler, char const* args)
59 {
60 if (WorldSession* session = handler->GetSession())
61 {
62 if (!*args)
63 {
64 if (session->HasPermission(rbac::RBAC_PERM_CHAT_USE_STAFF_BADGE) && session->GetPlayer()->isGMChat())
65 session->SendNotification(LANG_GM_CHAT_ON);
66 else
67 session->SendNotification(LANG_GM_CHAT_OFF);
68 return true;
69 }
70
71 std::string param = (char*)args;
72
73 if (param == "on")
74 {
75 session->GetPlayer()->SetGMChat(true);
76 session->SendNotification(LANG_GM_CHAT_ON);
77 return true;
78 }
79
80 if (param == "off")
81 {
82 session->GetPlayer()->SetGMChat(false);
83 session->SendNotification(LANG_GM_CHAT_OFF);
84 return true;
85 }
86 }
87
88 handler->SendSysMessage(LANG_USE_BOL);
89 handler->SetSentErrorMessage(true);
90 return false;
91 }

93 static bool HandleGMFlyCommand(ChatHandler* handler, char const* args)
94 {
95 if (!*args)
96 return false;
97
98 Player* target = handler->getSelectedPlayer();
99 if (!target)
100 target = handler->GetSession()->GetPlayer();
101
102 WorldPacket data(12);
103 if (strncmp(args, "on", 3) == 0)
104 data.SetOpcode(SMSG_MOVE_SET_CAN_FLY);
105 else if (strncmp(args, "off", 4) == 0)
106 data.SetOpcode(SMSG_MOVE_UNSET_CAN_FLY);
107 else
108 {
109 handler->SendSysMessage(LANG_USE_BOL);
110 return false;
111 }
112 data << target->GetPackGUID();
113 data << uint32(0); // unknown
114 target->SendMessageToSet(&data, true);
115 handler->PSendSysMessage(LANG_COMMAND_FLYMODE_STATUS, handler->GetNameLink(target).c_str(), args);
116 return true;
117 }

119 static bool HandleGMListIngameCommand(ChatHandler* handler, char const* /*args*/)
120 {
121 bool first = true;
122 bool footer = false;
123
124 boost::shared_lock<boost::shared_mutex> lock(*HashMapHolder<Player>::GetLock());
125 HashMapHolder<Player>::MapType const& m = ObjectAccessor::GetPlayers();
126 for (HashMapHolder<Player>::MapType::const_iterator itr = m.begin(); itr != m.end(); ++itr)
127 {
128 AccountTypes itrSec = itr->second->GetSession()->GetSecurity();
129 if ((itr->second->IsGameMaster() ||
130 (itr->second->GetSession()->HasPermission(rbac::RBAC_PERM_COMMANDS_APPEAR_IN_GM_LIST) &&
131 itrSec <= AccountTypes(sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_GM_LIST)))) &&
132 (!handler->GetSession() || itr->second->IsVisibleGloballyFor(handler->GetSession()->GetPlayer())))
133 {
134 if (first)
135 {
136 first = false;
137 footer = true;
138 handler->SendSysMessage(LANG_GMS_ON_SRV);
139 handler->SendSysMessage("========================");
140 }
141 std::string const& name = itr->second->GetName();
142 uint8 size = name.size();
143 uint8 security = itrSec;
144 uint8 max = ((16 - size) / 2);
145 uint8 max2 = max;
146 if ((max + max2 + size) == 16)
147 max2 = max - 1;
148 if (handler->GetSession())
149 handler->PSendSysMessage("| %s GMLevel %u", name.c_str(), security);
150 else
151 handler->PSendSysMessage("|%*s%s%*s| %u |", max, " ", name.c_str(), max2, " ", security);
152 }
153 }
154 if (footer)
155 handler->SendSysMessage("========================");
156 if (first)
157 handler->SendSysMessage(LANG_GMS_NOT_LOGGED);
158 return true;
159 }

161 /// Display the list of GMs
162 static bool HandleGMListFullCommand(ChatHandler* handler, char const* /*args*/)
163 {
164 ///- Get the accounts with GM Level >0
165 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_GM_ACCOUNTS);
166 stmt->setUInt8(0, uint8(SEC_MODERATOR));
167 stmt->setInt32(1, int32(realm.Id.Realm));
168 PreparedQueryResult result = LoginDatabase.Query(stmt);
169
170 if (result)
171 {
172 handler->SendSysMessage(LANG_GMLIST);
173 handler->SendSysMessage("========================");
174 ///- Cycle through them. Display username and GM level
175 do
176 {
177 Field* fields = result->Fetch();
178 char const* name = fields[0].GetCString();
179 uint8 security = fields[1].GetUInt8();
180 uint8 max = (16 - strlen(name)) / 2;
181 uint8 max2 = max;
182 if ((max + max2 + strlen(name)) == 16)
183 max2 = max - 1;
184 if (handler->GetSession())
185 handler->PSendSysMessage("| %s GMLevel %u", name, security);
186 else
187 handler->PSendSysMessage("|%*s%s%*s| %u |", max, " ", name, max2, " ", security);
188 } while (result->NextRow());
189 handler->SendSysMessage("========================");
190 }
191 else
192 handler->PSendSysMessage(LANG_GMLIST_EMPTY);
193 return true;
194 }

196 //Enable\Disable Invisible mode
197 static bool HandleGMVisibleCommand(ChatHandler* handler, char const* args)
198 {
199 Player* _player = handler->GetSession()->GetPlayer();
200
201 if (!*args)
202 {
203 handler->PSendSysMessage(LANG_YOU_ARE, _player->isGMVisible() ? handler->GetTrinityString(LANG_VISIBLE) : handler->GetTrinityString(LANG_INVISIBLE));
204 return true;
205 }
206
207 const uint32 VISUAL_AURA = 37800;
208 std::string param = (char*)args;
209
210 if (param == "on")
211 {
212 if (_player->HasAura(VISUAL_AURA))
213 _player->RemoveAurasDueToSpell(VISUAL_AURA);
214
215 _player->SetGMVisible(true);
216 _player->UpdateObjectVisibility();
217 handler->GetSession()->SendNotification(LANG_INVISIBLE_VISIBLE);
218 return true;
219 }
220
221 if (param == "off")
222 {
223 _player->AddAura(VISUAL_AURA, _player);
224 _player->SetGMVisible(false);
225 _player->UpdateObjectVisibility();
226 handler->GetSession()->SendNotification(LANG_INVISIBLE_INVISIBLE);
227 return true;
228 }
229
230 handler->SendSysMessage(LANG_USE_BOL);
231 handler->SetSentErrorMessage(true);
232 return false;
233 }

235 //Enable\Disable GM Mode
236 static bool HandleGMCommand(ChatHandler* handler, char const* args)
237 {
238 Player* _player = handler->GetSession()->GetPlayer();
239
240 if (!*args)
241 {
242 handler->GetSession()->SendNotification(_player->IsGameMaster() ? LANG_GM_ON : LANG_GM_OFF);
243 return true;
244 }
245
246 std::string param = (char*)args;
247
248 if (param == "on")
249 {
250 _player->SetGameMaster(true);
251 handler->GetSession()->SendNotification(LANG_GM_ON);
252 _player->UpdateTriggerVisibility();
253#ifdef _DEBUG_VMAPS
254 VMAP::IVMapManager* vMapManager = VMAP::VMapFactory::createOrGetVMapManager();
255 vMapManager->processCommand("stoplog");
256#endif
257 return true;
258 }
259
260 if (param == "off")
261 {
262 _player->SetGameMaster(false);
263 handler->GetSession()->SendNotification(LANG_GM_OFF);
264 _player->UpdateTriggerVisibility();
265#ifdef _DEBUG_VMAPS
266 VMAP::IVMapManager* vMapManager = VMAP::VMapFactory::createOrGetVMapManager();
267 vMapManager->processCommand("startlog");
268#endif
269 return true;
270 }
271
272 handler->SendSysMessage(LANG_USE_BOL);
273 handler->SetSentErrorMessage(true);
274 return false;
275 }

62 /** \brief Teleport the GM to the specified creature
63 *
64 * .gocreature <GUID> --> TP using creature.guid
65 * .gocreature azuregos --> TP player to the mob with this name
66 * Warning: If there is more than one mob with this name
67 * you will be teleported to the first one that is found.
68 * .gocreature id 6109 --> TP player to the mob, that has this creature_template.entry
69 * Warning: If there is more than one mob with this "id"
70 * you will be teleported to the first one that is found.
71 */
72 //teleport to creature
73 static bool HandleGoCreatureCommand(ChatHandler* handler, char const* args)
74 {
75 if (!*args)
76 return false;
77
78 Player* player = handler->GetSession()->GetPlayer();
79
80 // "id" or number or [name] Shift-click form |color|Hcreature_entry:creature_id|h[name]|h|r
81 char* param1 = handler->extractKeyFromLink((char*)args, "Hcreature");
82 if (!param1)
83 return false;
84
85 std::ostringstream whereClause;
86
87 // User wants to teleport to the NPC's template entry
88 if (strcmp(param1, "id") == 0)
89 {
90 // Get the "creature_template.entry"
91 // number or [name] Shift-click form |color|Hcreature_entry:creature_id|h[name]|h|r
92 char* tail = strtok(NULL, "");
93 if (!tail)
94 return false;
95 char* id = handler->extractKeyFromLink(tail, "Hcreature_entry");
96 if (!id)
97 return false;
98
99 int32 entry = atoi(id);
100 if (!entry)
101 return false;
102
103 whereClause << "WHERE id = '" << entry << '\'';
104 }
105 else
106 {
107 int32 guid = atoi(param1);
108
109 // Number is invalid - maybe the user specified the mob's name
110 if (!guid)
111 {
112 std::string name = param1;
113 WorldDatabase.EscapeString(name);
114 whereClause << ", creature_template WHERE creature.id = creature_template.entry AND creature_template.name " _LIKE_" '" << name << '\'';
115 }
116 else
117 whereClause << "WHERE guid = '" << guid << '\'';
118 }
119
120 QueryResult result = WorldDatabase.PQuery("SELECT position_x, position_y, position_z, orientation, map, guid, id FROM creature %s", whereClause.str().c_str());
121 if (!result)
122 {
123 handler->SendSysMessage(LANG_COMMAND_GOCREATNOTFOUND);
124 handler->SetSentErrorMessage(true);
125 return false;
126 }
127 if (result->GetRowCount() > 1)
128 handler->SendSysMessage(LANG_COMMAND_GOCREATMULTIPLE);
129
130 Field* fields = result->Fetch();
131 float x = fields[0].GetFloat();
132 float y = fields[1].GetFloat();
133 float z = fields[2].GetFloat();
134 float o = fields[3].GetFloat();
135 uint32 mapId = fields[4].GetUInt16();
136
137 Transport* transport = NULL;
138
139 if (!MapManager::IsValidMapCoord(mapId, x, y, z, o) || sObjectMgr->IsTransportMap(mapId))
140 {
141 handler->PSendSysMessage(LANG_INVALID_TARGET_COORD, x, y, mapId);
142 handler->SetSentErrorMessage(true);
143 return false;
144 }
145
146 // stop flight if need
147 if (player->IsInFlight())
148 {
149 player->GetMotionMaster()->MovementExpired();
150 player->CleanupAfterTaxiFlight();
151 }
152 // save only in non-flight case
153 else
154 player->SaveRecallPosition();
155
156 if (player->TeleportTo(mapId, x, y, z, o))
157 {
158 if (transport)
159 transport->AddPassenger(player);
160 }
161 return true;
162 }

164 static bool HandleGoGraveyardCommand(ChatHandler* handler, char const* args)
165 {
166 Player* player = handler->GetSession()->GetPlayer();
167
168 if (!*args)
169 return false;
170
171 char* gyId = strtok((char*)args, " ");
172 if (!gyId)
173 return false;
174

176

178 return false;
179
180 WorldSafeLocsEntry const* gy = sWorldSafeLocsStore.LookupEntry(graveyardId);
181 if (!gy)
182 {

186 }
187

189 {
190 handler->PSendSysMessage(LANG_INVALID_TARGET_COORD, gy->x, gy->y, gy->map_id);

192 return false;
193 }
194
195 // stop flight if need
196 if (player->IsInFlight())
197 {
198 player->GetMotionMaster()->MovementExpired();
199 player->CleanupAfterTaxiFlight();
200 }
201 // save only in non-flight case
202 else
203 player->SaveRecallPosition();
204
205 player->TeleportTo(gy->map_id, gy->x, gy->y, gy->z, player->GetOrientation());
206 return true;

209 //teleport to grid
210 static bool HandleGoGridCommand(ChatHandler* handler, char const* args)
211 {
212 if (!*args)
213 return false;
214
215 Player* player = handler->GetSession()->GetPlayer();
216
217 char* gridX = strtok((char*)args, " ");
218 char* gridY = strtok(NULL, " ");
219 char* id = strtok(NULL, " ");
220
221 if (!gridX || !gridY)
222 return false;
223
224 uint32 mapId = id ? (uint32)atoi(id) : player->GetMapId();
225
226 // center of grid
227 float x = ((float)atof(gridX) - CENTER_GRID_ID + 0.5f) * SIZE_OF_GRIDS;
228 float y = ((float)atof(gridY) - CENTER_GRID_ID + 0.5f) * SIZE_OF_GRIDS;
229
230 if (!MapManager::IsValidMapCoord(mapId, x, y))
231 {
232 handler->PSendSysMessage(LANG_INVALID_TARGET_COORD, x, y, mapId);
233 handler->SetSentErrorMessage(true);
234 return false;
235 }
236
237 // stop flight if need
238 if (player->IsInFlight())
239 {
240 player->GetMotionMaster()->MovementExpired();
241 player->CleanupAfterTaxiFlight();
242 }
243 // save only in non-flight case
244 else
245 player->SaveRecallPosition();
246
247 Map const* map = sMapMgr->CreateBaseMap(mapId);
248 float z = std::max(map->GetHeight(x, y, MAX_HEIGHT), map->GetWaterLevel(x, y));
249
250 player->TeleportTo(mapId, x, y, z, player->GetOrientation());
251 return true;
252 }

401 //teleport at coordinates
402 static bool HandleGoZoneXYCommand(ChatHandler* handler, char const* args)
403 {
404 if (!*args)
405 return false;
406
407 Player* player = handler->GetSession()->GetPlayer();
408
409 char* zoneX = strtok((char*)args, " ");
410 char* zoneY = strtok(NULL, " ");
411 char* tail = strtok(NULL, "");
412
413 char* id = handler->extractKeyFromLink(tail, "Harea"); // string or [name] Shift-click form |color|Harea:area_id|h[name]|h|r
414
415 if (!zoneX || !zoneY)
416 return false;
417
418 float x = (float)atof(zoneX);
419 float y = (float)atof(zoneY);
420
421 // prevent accept wrong numeric args
422 if ((x == 0.0f && *zoneX != '0') || (y == 0.0f && *zoneY != '0'))
423 return false;
424
425 uint32 areaId = id ? (uint32)atoi(id) : player->GetZoneId();
426
427 AreaTableEntry const* areaEntry = sAreaTableStore.LookupEntry(areaId);
428
429 if (x < 0 || x > 100 || y < 0 || y > 100 || !areaEntry)
430 {
431 handler->PSendSysMessage(LANG_INVALID_ZONE_COORD, x, y, areaId);
432 handler->SetSentErrorMessage(true);
433 return false;
434 }
435
436 // update to parent zone if exist (client map show only zones without parents)
437 AreaTableEntry const* zoneEntry = areaEntry->zone ? sAreaTableStore.LookupEntry(areaEntry->zone) : areaEntry;
438 ASSERT(zoneEntry);
439
440 Map const* map = sMapMgr->CreateBaseMap(zoneEntry->mapid);
441
442 if (map->Instanceable())
443 {
444 handler->PSendSysMessage(LANG_INVALID_ZONE_MAP, areaEntry->ID, areaEntry->area_name[handler->GetSessionDbcLocale()], map->GetId(), map->GetMapName());
445 handler->SetSentErrorMessage(true);
446 return false;
447 }
448
449 Zone2MapCoordinates(x, y, zoneEntry->ID);
450
451 if (!MapManager::IsValidMapCoord(zoneEntry->mapid, x, y))
452 {
453 handler->PSendSysMessage(LANG_INVALID_TARGET_COORD, x, y, zoneEntry->mapid);
454 handler->SetSentErrorMessage(true);
455 return false;
456 }
457
458 // stop flight if need
459 if (player->IsInFlight())
460 {
461 player->GetMotionMaster()->MovementExpired();
462 player->CleanupAfterTaxiFlight();
463 }
464 // save only in non-flight case
465 else
466 player->SaveRecallPosition();
467
468 float z = std::max(map->GetHeight(x, y, MAX_HEIGHT), map->GetWaterLevel(x, y));
469
470 player->TeleportTo(zoneEntry->mapid, x, y, z, player->GetOrientation());
471 return true;
472 }

474 //teleport at coordinates, including Z and orientation
475 static bool HandleGoXYZCommand(ChatHandler* handler, char const* args)
476 {
477 if (!*args)
478 return false;
479
480 Player* player = handler->GetSession()->GetPlayer();
481
482 char* goX = strtok((char*)args, " ");
483 char* goY = strtok(NULL, " ");
484 char* goZ = strtok(NULL, " ");
485 char* id = strtok(NULL, " ");
486 char* port = strtok(NULL, " ");
487
488 if (!goX || !goY)
489 return false;
490
491 float x = (float)atof(goX);
492 float y = (float)atof(goY);
493 float z;
494 float ort = port ? (float)atof(port) : player->GetOrientation();
495 uint32 mapId = id ? (uint32)atoi(id) : player->GetMapId();
496
497 if (goZ)
498 {
499 z = (float)atof(goZ);
500 if (!MapManager::IsValidMapCoord(mapId, x, y, z))
501 {
502 handler->PSendSysMessage(LANG_INVALID_TARGET_COORD, x, y, mapId);
503 handler->SetSentErrorMessage(true);
504 return false;
505 }
506 }
507 else
508 {
509 if (!MapManager::IsValidMapCoord(mapId, x, y))
510 {
511 handler->PSendSysMessage(LANG_INVALID_TARGET_COORD, x, y, mapId);
512 handler->SetSentErrorMessage(true);
513 return false;
514 }
515 Map const* map = sMapMgr->CreateBaseMap(mapId);
516 z = std::max(map->GetHeight(x, y, MAX_HEIGHT), map->GetWaterLevel(x, y));
517 }
518
519 // stop flight if need
520 if (player->IsInFlight())
521 {
522 player->GetMotionMaster()->MovementExpired();
523 player->CleanupAfterTaxiFlight();
524 }
525 // save only in non-flight case
526 else
527 player->SaveRecallPosition();
528
529 player->TeleportTo(mapId, x, y, z, ort);
530 return true;
531 }

533 static bool HandleGoTicketCommand(ChatHandler* handler, char const* args)
534 {
535 if (!*args)
536 return false;
537
538 char* id = strtok((char*)args, " ");
539 if (!id)
540 return false;
541
542 uint32 ticketId = atoi(id);
543 if (!ticketId)
544 return false;
545
546 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
547 if (!ticket)
548 {
549 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
550 return true;
551 }
552
553 Player* player = handler->GetSession()->GetPlayer();
554 if (player->IsInFlight())
555 {
556 player->GetMotionMaster()->MovementExpired();
557 player->CleanupAfterTaxiFlight();
558 }
559 else
560 player->SaveRecallPosition();
561
562 ticket->TeleportTo(player);
563 return true;
564 }

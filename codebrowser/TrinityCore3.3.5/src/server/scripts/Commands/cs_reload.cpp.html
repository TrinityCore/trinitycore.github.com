417 static bool HandleReloadCreatureTemplateCommand(ChatHandler* handler, const char* args)
418 {
419 if (!*args)
420 return false;
421
422 Tokenizer entries(std::string(args), ' ');
423
424 for (Tokenizer::const_iterator itr = entries.begin(); itr != entries.end(); ++itr)
425 {
426 uint32 entry = uint32(atoi(*itr));
427
428 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_CREATURE_TEMPLATE);
429 stmt->setUInt32(0, entry);
430 PreparedQueryResult result = WorldDatabase.Query(stmt);
431
432 if (!result)
433 {
434 handler->PSendSysMessage(LANG_COMMAND_CREATURETEMPLATE_NOTFOUND, entry);
435 continue;
436 }
437
438 CreatureTemplate const* cInfo = sObjectMgr->GetCreatureTemplate(entry);
439 if (!cInfo)
440 {
441 handler->PSendSysMessage(LANG_COMMAND_CREATURESTORAGE_NOTFOUND, entry);
442 continue;
443 }
444
445 TC_LOG_INFO("misc", "Reloading creature template entry %u", entry);
446
447 Field* fields = result->Fetch();
448 sObjectMgr->LoadCreatureTemplate(fields);
449 sObjectMgr->CheckCreatureTemplate(cInfo);
450 }
451
452 handler->SendGlobalGMSysMessage("Creature template reloaded.");
453 return true;
454 }

235 //add spawn of creature
236 static bool HandleNpcAddCommand(ChatHandler* handler, char const* args)
237 {
238 if (!*args)
239 return false;
240
241 char* charID = handler->extractKeyFromLink((char*)args, "Hcreature_entry");
242 if (!charID)
243 return false;
244
245 uint32 id = atoi(charID);
246 if (!sObjectMgr->GetCreatureTemplate(id))
247 return false;
248
249 Player* chr = handler->GetSession()->GetPlayer();
250 float x = chr->GetPositionX();
251 float y = chr->GetPositionY();
252 float z = chr->GetPositionZ();
253 float o = chr->GetOrientation();
254 Map* map = chr->GetMap();
255
256 if (Transport* trans = chr->GetTransport())
257 {
258 ObjectGuid::LowType guid = map->GenerateLowGuid<HighGuid::Unit>();
259 CreatureData& data = sObjectMgr->NewOrExistCreatureData(guid);
260 data.id = id;
261 data.phaseMask = chr->GetPhaseMaskForSpawn();
262 data.posX = chr->GetTransOffsetX();
263 data.posY = chr->GetTransOffsetY();
264 data.posZ = chr->GetTransOffsetZ();
265 data.orientation = chr->GetTransOffsetO();
266
267 Creature* creature = trans->CreateNPCPassenger(guid, &data);
268
269 creature->SaveToDB(trans->GetGOInfo()->moTransport.mapID, 1 << map->GetSpawnMode(), chr->GetPhaseMaskForSpawn());
270
271 sObjectMgr->AddCreatureToGrid(guid, &data);
272 return true;
273 }
274
275 Creature* creature = new Creature();
276 if (!creature->Create(map->GenerateLowGuid<HighGuid::Unit>(), map, chr->GetPhaseMaskForSpawn(), id, x, y, z, o))
277 {
278 delete creature;
279 return false;
280 }
281
282 creature->SaveToDB(map->GetId(), (1 << map->GetSpawnMode()), chr->GetPhaseMaskForSpawn());
283
284 ObjectGuid::LowType db_guid = creature->GetSpawnId();
285
286 // To call _LoadGoods(); _LoadQuests(); CreateTrainerSpells()
287 // current "creature" variable is deleted and created fresh new, otherwise old values might trigger asserts or cause undefined behavior
288 creature->CleanupsBeforeDelete();
289 delete creature;
290 creature = new Creature();
291 if (!creature->LoadCreatureFromDB(db_guid, map))
292 {
293 delete creature;
294 return false;
295 }
296
297 sObjectMgr->AddCreatureToGrid(db_guid, sObjectMgr->GetCreatureData(db_guid));
298 return true;
299 }

301 //add item in vendorlist
302 static bool HandleNpcAddVendorItemCommand(ChatHandler* handler, char const* args)
303 {
304 if (!*args)
305 return false;
306
307 char* pitem = handler->extractKeyFromLink((char*)args, "Hitem");
308 if (!pitem)
309 {
310 handler->SendSysMessage(LANG_COMMAND_NEEDITEMSEND);
311 handler->SetSentErrorMessage(true);
312 return false;
313 }
314
315 int32 item_int = atol(pitem);
316 if (item_int <= 0)
317 return false;
318
319 uint32 itemId = item_int;
320
321 char* fmaxcount = strtok(NULL, " "); //add maxcount, default: 0
322 uint32 maxcount = 0;
323 if (fmaxcount)
324 maxcount = atoul(fmaxcount);
325
326 char* fincrtime = strtok(NULL, " "); //add incrtime, default: 0
327 uint32 incrtime = 0;
328 if (fincrtime)
329 incrtime = atoul(fincrtime);
330
331 char* fextendedcost = strtok(NULL, " "); //add ExtendedCost, default: 0
332 uint32 extendedcost = fextendedcost ? atoul(fextendedcost) : 0;
333 Creature* vendor = handler->getSelectedCreature();
334 if (!vendor)
335 {
336 handler->SendSysMessage(LANG_SELECT_CREATURE);
337 handler->SetSentErrorMessage(true);
338 return false;
339 }
340
341 uint32 vendor_entry = vendor->GetEntry();
342
343 if (!sObjectMgr->IsVendorItemValid(vendor_entry, itemId, maxcount, incrtime, extendedcost, handler->GetSession()->GetPlayer()))
344 {
345 handler->SetSentErrorMessage(true);
346 return false;
347 }
348
349 sObjectMgr->AddVendorItem(vendor_entry, itemId, maxcount, incrtime, extendedcost);
350
351 ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(itemId);
352
353 handler->PSendSysMessage(LANG_ITEM_ADDED_TO_LIST, itemId, itemTemplate->Name1.c_str(), maxcount, incrtime, extendedcost);
354 return true;
355 }

357 //add move for creature
358 static bool HandleNpcAddMoveCommand(ChatHandler* handler, char const* args)
359 {
360 if (!*args)
361 return false;
362
363 char* guidStr = strtok((char*)args, " ");
364 char* waitStr = strtok((char*)NULL, " ");
365
366 ObjectGuid::LowType lowGuid = atoi((char*)guidStr);
367
368 // attempt check creature existence by DB data
369 CreatureData const* data = sObjectMgr->GetCreatureData(lowGuid);
370 if (!data)
371 {
372 handler->PSendSysMessage(LANG_COMMAND_CREATGUIDNOTFOUND, lowGuid);
373 handler->SetSentErrorMessage(true);
374 return false;
375 }
376
377 int wait = waitStr ? atoi(waitStr) : 0;
378
379 if (wait < 0)
380 wait = 0;
381
382 // Update movement type
383 PreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_UPD_CREATURE_MOVEMENT_TYPE);
384
385 stmt->setUInt8(0, uint8(WAYPOINT_MOTION_TYPE));
386 stmt->setUInt32(1, lowGuid);
387
388 WorldDatabase.Execute(stmt);
389
390 handler->SendSysMessage(LANG_WAYPOINT_ADDED);
391
392 return true;
393 }

434 //change level of creature or pet
435 static bool HandleNpcSetLevelCommand(ChatHandler* handler, char const* args)
436 {
437 if (!*args)
438 return false;
439
440 uint8 lvl = (uint8) atoi((char*)args);
441 if (lvl < 1 || lvl > sWorld->getIntConfig(CONFIG_MAX_PLAYER_LEVEL) + 3)
442 {
443 handler->SendSysMessage(LANG_BAD_VALUE);
444 handler->SetSentErrorMessage(true);
445 return false;
446 }
447
448 Creature* creature = handler->getSelectedCreature();
449 if (!creature)
450 {
451 handler->SendSysMessage(LANG_SELECT_CREATURE);
452 handler->SetSentErrorMessage(true);
453 return false;
454 }
455
456 if (creature->IsPet())
457 {
458 if (((Pet*)creature)->getPetType() == HUNTER_PET)
459 {
460 creature->SetUInt32Value(UNIT_FIELD_PETNEXTLEVELEXP, sObjectMgr->GetXPForLevel(lvl)/4);
461 creature->SetUInt32Value(UNIT_FIELD_PETEXPERIENCE, 0);
462 }
463 ((Pet*)creature)->GivePetLevel(lvl);
464 }
465 else
466 {
467 creature->SetMaxHealth(100 + 30*lvl);
468 creature->SetHealth(100 + 30*lvl);
469 creature->SetLevel(lvl);
470 creature->SaveToDB();
471 }
472
473 return true;
474 }

476 static bool HandleNpcDeleteCommand(ChatHandler* handler, char const* args)
477 {
478 Creature* unit = NULL;
479
480 if (*args)
481 {
482 // number or [name] Shift-click form |color|Hcreature:creature_guid|h[name]|h|r
483 char* cId = handler->extractKeyFromLink((char*)args, "Hcreature");
484 if (!cId)
485 return false;
486
487 ObjectGuid::LowType lowguid = atoi(cId);
488 if (!lowguid)
489 return false;
490
491 if (CreatureData const* cr_data = sObjectMgr->GetCreatureData(lowguid))
492 unit = handler->GetSession()->GetPlayer()->GetMap()->GetCreature(ObjectGuid(HighGuid::Unit, cr_data->id, lowguid));
493 }
494 else
495 unit = handler->getSelectedCreature();
496
497 if (!unit || unit->IsPet() || unit->IsTotem())
498 {
499 handler->SendSysMessage(LANG_SELECT_CREATURE);
500 handler->SetSentErrorMessage(true);
501 return false;
502 }
503
504 // Delete the creature
505 unit->CombatStop();
506 unit->DeleteFromDB();
507 unit->AddObjectToRemoveList();
508
509 handler->SendSysMessage(LANG_COMMAND_DELCREATMESSAGE);
510
511 return true;
512 }

514 //del item from vendor list
515 static bool HandleNpcDeleteVendorItemCommand(ChatHandler* handler, char const* args)
516 {
517 if (!*args)
518 return false;
519
520 Creature* vendor = handler->getSelectedCreature();
521 if (!vendor || !vendor->IsVendor())
522 {
523 handler->SendSysMessage(LANG_COMMAND_VENDORSELECTION);
524 handler->SetSentErrorMessage(true);
525 return false;
526 }
527
528 char* pitem = handler->extractKeyFromLink((char*)args, "Hitem");
529 if (!pitem)
530 {
531 handler->SendSysMessage(LANG_COMMAND_NEEDITEMSEND);
532 handler->SetSentErrorMessage(true);
533 return false;
534 }
535 uint32 itemId = atoul(pitem);
536
537 if (!sObjectMgr->RemoveVendorItem(vendor->GetEntry(), itemId))
538 {
539 handler->PSendSysMessage(LANG_ITEM_NOT_IN_LIST, itemId);
540 handler->SetSentErrorMessage(true);
541 return false;
542 }
543
544 ItemTemplate const* itemTemplate = sObjectMgr->GetItemTemplate(itemId);
545
546 handler->PSendSysMessage(LANG_ITEM_DELETED_FROM_LIST, itemId, itemTemplate->Name1.c_str());
547 return true;
548 }

904 /**HandleNpcSetMoveTypeCommand
905 * Set the movement type for an NPC.<br/>
906 * <br/>
907 * Valid movement types are:
908 * <ul>
909 * <li> stay - NPC wont move </li>
910 * <li> random - NPC will move randomly according to the spawndist </li>
911 * <li> way - NPC will move with given waypoints set </li>
912 * </ul>
913 * additional parameter: NODEL - so no waypoints are deleted, if you
914 * change the movement type
915 */
916 static bool HandleNpcSetMoveTypeCommand(ChatHandler* handler, char const* args)
917 {
918 if (!*args)
919 return false;
920
921 // 3 arguments:
922 // GUID (optional - you can also select the creature)
923 // stay|random|way (determines the kind of movement)
924 // NODEL (optional - tells the system NOT to delete any waypoints)
925 // this is very handy if you want to do waypoints, that are
926 // later switched on/off according to special events (like escort
927 // quests, etc)
928 char* guid_str = strtok((char*)args, " ");
929 char* type_str = strtok((char*)NULL, " ");
930 char* dontdel_str = strtok((char*)NULL, " ");
931
932 bool doNotDelete = false;
933
934 if (!guid_str)
935 return false;
936
937 ObjectGuid::LowType lowguid = 0;
938 Creature* creature = NULL;
939
940 if (dontdel_str)
941 {
942 //TC_LOG_ERROR("misc", "DEBUG: All 3 params are set");
943
944 // All 3 params are set
945 // GUID
946 // type
947 // doNotDEL
948 if (stricmp(dontdel_str, "NODEL") == 0)
949 {
950 //TC_LOG_ERROR("misc", "DEBUG: doNotDelete = true;");
951 doNotDelete = true;
952 }
953 }
954 else
955 {
956 // Only 2 params - but maybe NODEL is set
957 if (type_str)
958 {
959 TC_LOG_ERROR("misc", "DEBUG: Only 2 params ");
960 if (stricmp(type_str, "NODEL") == 0)
961 {
962 //TC_LOG_ERROR("misc", "DEBUG: type_str, NODEL ");
963 doNotDelete = true;
964 type_str = NULL;
965 }
966 }
967 }
968
969 if (!type_str) // case .setmovetype $move_type (with selected creature)
970 {
971 type_str = guid_str;
972 creature = handler->getSelectedCreature();
973 if (!creature || creature->IsPet())
974 return false;
975 lowguid = creature->GetSpawnId();
976 }
977 else // case .setmovetype #creature_guid $move_type (with selected creature)
978 {
979 lowguid = atoi((char*)guid_str);
980
981 /* impossible without entry
982 if (lowguid)
983 creature = ObjectAccessor::GetCreature(*handler->GetSession()->GetPlayer(), MAKE_GUID(lowguid, HighGuid::Unit));
984 */
985
986 // attempt check creature existence by DB data
987 if (!creature)
988 {
989 CreatureData const* data = sObjectMgr->GetCreatureData(lowguid);
990 if (!data)
991 {
992 handler->PSendSysMessage(LANG_COMMAND_CREATGUIDNOTFOUND, lowguid);
993 handler->SetSentErrorMessage(true);
994 return false;
995 }
996 }
997 else
998 {
999 lowguid = creature->GetSpawnId();
1000 }
1001 }
1002
1003 // now lowguid is low guid really existed creature
1004 // and creature point (maybe) to this creature or NULL
1005
1006 MovementGeneratorType move_type;
1007
1008 std::string type = type_str;
1009
1010 if (type == "stay")
1011 move_type = IDLE_MOTION_TYPE;
1012 else if (type == "random")
1013 move_type = RANDOM_MOTION_TYPE;
1014 else if (type == "way")
1015 move_type = WAYPOINT_MOTION_TYPE;
1016 else
1017 return false;
1018
1019 // update movement type
1020 //if (doNotDelete == false)
1021 // WaypointMgr.DeletePath(lowguid);
1022
1023 if (creature)
1024 {
1025 // update movement type
1026 if (doNotDelete == false)
1027 creature->LoadPath(0);
1028
1029 creature->SetDefaultMovementType(move_type);
1030 creature->GetMotionMaster()->Initialize();
1031 if (creature->IsAlive()) // dead creature will reset movement generator at respawn
1032 {
1033 creature->setDeathState(JUST_DIED);
1034 creature->Respawn();
1035 }
1036 creature->SaveToDB();
1037 }
1038 if (doNotDelete == false)
1039 {
1040 handler->PSendSysMessage(LANG_MOVE_TYPE_SET, type_str);
1041 }
1042 else
1043 {
1044 handler->PSendSysMessage(LANG_MOVE_TYPE_SET_NODEL, type_str);
1045 }
1046
1047 return true;
1048 }

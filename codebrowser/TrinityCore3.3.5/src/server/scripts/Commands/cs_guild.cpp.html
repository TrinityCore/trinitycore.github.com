56 /** \brief GM command level 3 - Create a guild.
57 *
58 * This command allows a GM (level 3) to create a guild.
59 *
60 * The "args" parameter contains the name of the guild leader
61 * and then the name of the guild.
62 *
63 */
64 static bool HandleGuildCreateCommand(ChatHandler* handler, char const* args)
65 {
66 if (!*args)
67 return false;
68
69 // if not guild name only (in "") then player name
70 Player* target;
71 if (!handler->extractPlayerTarget(*args != '"' ? (char*)args : NULL, &target))
72 return false;
73
74 char* tailStr = *args != '"' ? strtok(NULL, "") : (char*)args;
75 if (!tailStr)
76 return false;
77
78 char* guildStr = handler->extractQuotedArg(tailStr);
79 if (!guildStr)
80 return false;
81
82 std::string guildName = guildStr;
83
84 if (target->GetGuildId())
85 {
86 handler->SendSysMessage(LANG_PLAYER_IN_GUILD);
87 return true;
88 }
89
90 Guild* guild = new Guild;
91 if (!guild->Create(target, guildName))
92 {
93 delete guild;
94 handler->SendSysMessage(LANG_GUILD_NOT_CREATED);
95 handler->SetSentErrorMessage(true);
96 return false;
97 }
98
99 sGuildMgr->AddGuild(guild);
100
101 return true;
102 }

104 static bool HandleGuildDeleteCommand(ChatHandler* handler, char const* args)
105 {
106 if (!*args)
107 return false;
108
109 char* guildStr = handler->extractQuotedArg((char*)args);
110 if (!guildStr)
111 return false;
112
113 std::string guildName = guildStr;
114
115 Guild* targetGuild = sGuildMgr->GetGuildByName(guildName);
116 if (!targetGuild)
117 return false;
118
119 targetGuild->Disband();
120 delete targetGuild;
121
122 return true;
123 }

125 static bool HandleGuildInviteCommand(ChatHandler* handler, char const* args)
126 {
127 if (!*args)
128 return false;
129
130 // if not guild name only (in "") then player name
131 ObjectGuid targetGuid;
132 if (!handler->extractPlayerTarget(*args != '"' ? (char*)args : NULL, NULL, &targetGuid))
133 return false;
134
135 char* tailStr = *args != '"' ? strtok(NULL, "") : (char*)args;
136 if (!tailStr)
137 return false;
138
139 char* guildStr = handler->extractQuotedArg(tailStr);
140 if (!guildStr)
141 return false;
142
143 std::string guildName = guildStr;
144 Guild* targetGuild = sGuildMgr->GetGuildByName(guildName);
145 if (!targetGuild)
146 return false;
147
148 // player's guild membership checked in AddMember before add
149 return targetGuild->AddMember(targetGuid);
150 }

152 static bool HandleGuildUninviteCommand(ChatHandler* handler, char const* args)
153 {
154 Player* target;
155 ObjectGuid targetGuid;
156 if (!handler->extractPlayerTarget((char*)args, &target, &targetGuid))
157 return false;
158
159 ObjectGuid::LowType guildId = target ? target->GetGuildId() : Player::GetGuildIdFromDB(targetGuid);
160 if (!guildId)
161 return false;
162
163 Guild* targetGuild = sGuildMgr->GetGuildById(guildId);
164 if (!targetGuild)
165 return false;
166
167 targetGuild->DeleteMember(targetGuid, false, true, true);
168 return true;
169 }

171 static bool HandleGuildRankCommand(ChatHandler* handler, char const* args)
172 {
173 char* nameStr;
174 char* rankStr;
175 handler->extractOptFirstArg((char*)args, &nameStr, &rankStr);
176 if (!rankStr)
177 return false;
178
179 Player* target;
180 ObjectGuid targetGuid;
181 std::string target_name;
182 if (!handler->extractPlayerTarget(nameStr, &target, &targetGuid, &target_name))
183 return false;
184
185 ObjectGuid::LowType guildId = target ? target->GetGuildId() : Player::GetGuildIdFromDB(targetGuid);
186 if (!guildId)
187 return false;
188
189 Guild* targetGuild = sGuildMgr->GetGuildById(guildId);
190 if (!targetGuild)
191 return false;
192
193 uint8 newRank = uint8(atoi(rankStr));
194 return targetGuild->ChangeMemberRank(targetGuid, newRank);
195 }

197 static bool HandleGuildRenameCommand(ChatHandler* handler, char const* _args)
198 {
199 if (!*_args)
200 return false;
201
202 char *args = (char *)_args;
203
204 char const* oldGuildStr = handler->extractQuotedArg(args);
205 if (!oldGuildStr)
206 {
207 handler->SendSysMessage(LANG_BAD_VALUE);
208 handler->SetSentErrorMessage(true);
209 return false;
210 }
211
212 char const* newGuildStr = handler->extractQuotedArg(strtok(NULL, ""));
213 if (!newGuildStr)
214 {
215 handler->SendSysMessage(LANG_INSERT_GUILD_NAME);
216 handler->SetSentErrorMessage(true);
217 return false;
218 }
219
220 Guild* guild = sGuildMgr->GetGuildByName(oldGuildStr);
221 if (!guild)
222 {
223 handler->PSendSysMessage(LANG_COMMAND_COULDNOTFIND, oldGuildStr);
224 handler->SetSentErrorMessage(true);
225 return false;
226 }
227
228 if (sGuildMgr->GetGuildByName(newGuildStr))
229 {
230 handler->PSendSysMessage(LANG_GUILD_RENAME_ALREADY_EXISTS, newGuildStr);
231 handler->SetSentErrorMessage(true);
232 return false;
233 }
234
235 if (!guild->SetName(newGuildStr))
236 {
237 handler->SendSysMessage(LANG_BAD_VALUE);
238 handler->SetSentErrorMessage(true);
239 return false;
240 }
241
242 handler->PSendSysMessage(LANG_GUILD_RENAME_DONE, oldGuildStr, newGuildStr);
243 return true;
244 }

246 static bool HandleGuildInfoCommand(ChatHandler* handler, char const* args)
247 {
248 Guild* guild = nullptr;
249
250 if (args && args[0] != '\0')
251 {
252 if (isNumeric(args))
253 {
254 uint32 guildId = uint32(atoi(args));
255 guild = sGuildMgr->GetGuildById(guildId);
256 }
257 else
258 {
259 std::string guildName = args;
260 guild = sGuildMgr->GetGuildByName(guildName);
261 }
262 }
263 else if (Player* target = handler->getSelectedPlayerOrSelf())
264 guild = target->GetGuild();
265
266 if (!guild)
267 return false;
268
269 // Display Guild Information
270 handler->PSendSysMessage(LANG_GUILD_INFO_NAME, guild->GetName().c_str(), guild->GetId()); // Guild Id + Name
271
272 std::string guildMasterName;
273 if (sObjectMgr->GetPlayerNameByGUID(guild->GetLeaderGUID(), guildMasterName))
274 handler->PSendSysMessage(LANG_GUILD_INFO_GUILD_MASTER, guildMasterName.c_str(), guild->GetLeaderGUID().GetCounter()); // Guild Master
275
276 // Format creation date
277 char createdDateStr[20];
278 time_t createdDate = guild->GetCreatedDate();
279 tm localTm;
280 strftime(createdDateStr, 20, "%Y-%m-%d %H:%M:%S", localtime_r(&createdDate, &localTm));
281
282 handler->PSendSysMessage(LANG_GUILD_INFO_CREATION_DATE, createdDateStr); // Creation Date
283 handler->PSendSysMessage(LANG_GUILD_INFO_MEMBER_COUNT, guild->GetMemberCount()); // Number of Members
284 handler->PSendSysMessage(LANG_GUILD_INFO_BANK_GOLD, guild->GetBankMoney() / 100 / 100); // Bank Gold (in gold coins)
285 handler->PSendSysMessage(LANG_GUILD_INFO_MOTD, guild->GetMOTD().c_str()); // Message of the Day
286 handler->PSendSysMessage(LANG_GUILD_INFO_EXTRA_INFO, guild->GetInfo().c_str()); // Extra Information
287 return true;
288 }

72 static bool HandleGMTicketAssignToCommand(ChatHandler* handler, char const* args)
73 {
74 if (!*args)
75 return false;
76
77 char* ticketIdStr = strtok((char*)args, " ");
78 uint32 ticketId = atoi(ticketIdStr);
79
80 char* targetStr = strtok(NULL, " ");
81 if (!targetStr)
82 return false;
83
84 std::string target(targetStr);
85 if (!normalizePlayerName(target))
86 return false;
87
88 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
89 if (!ticket || ticket->IsClosed())
90 {
91 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
92 return true;
93 }
94
95 ObjectGuid targetGuid = sObjectMgr->GetPlayerGUIDByName(target);
96 uint32 accountId = sObjectMgr->GetPlayerAccountIdByGUID(targetGuid);
97 // Target must exist and have administrative rights
98 if (!AccountMgr::HasPermission(accountId, rbac::RBAC_PERM_COMMANDS_BE_ASSIGNED_TICKET, realm.Id.Realm))
99 {
100 handler->SendSysMessage(LANG_COMMAND_TICKETASSIGNERROR_A);
101 return true;
102 }
103
104 // If already assigned, leave
105 if (ticket->IsAssignedTo(targetGuid))
106 {
107 handler->PSendSysMessage(LANG_COMMAND_TICKETASSIGNERROR_B, ticket->GetId());
108 return true;
109 }
110
111 // If assigned to different player other than current, leave
112 //! Console can override though
113 Player* player = handler->GetSession() ? handler->GetSession()->GetPlayer() : nullptr;
114 if (player && ticket->IsAssignedNotTo(player->GetGUID()))
115 {
116 handler->PSendSysMessage(LANG_COMMAND_TICKETALREADYASSIGNED, ticket->GetId(), target.c_str());
117 return true;
118 }
119
120 // Assign ticket
121 SQLTransaction trans = SQLTransaction(NULL);
122 ticket->SetAssignedTo(targetGuid, AccountMgr::IsAdminAccount(AccountMgr::GetSecurity(accountId, realm.Id.Realm)));
123 ticket->SaveToDB(trans);
124 sTicketMgr->UpdateLastChange();
125
126 std::string msg = ticket->FormatMessageString(*handler, NULL, target.c_str(), NULL, NULL, NULL);
127 handler->SendGlobalGMSysMessage(msg.c_str());
128 return true;
129 }

131 static bool HandleGMTicketCloseByIdCommand(ChatHandler* handler, char const* args)
132 {
133 if (!*args)
134 return false;
135
136 uint32 ticketId = atoi(args);
137 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
138 if (!ticket || ticket->IsClosed() || ticket->IsCompleted())
139 {
140 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
141 return true;
142 }
143
144 // Ticket should be assigned to the player who tries to close it.
145 // Console can override though
146 Player* player = handler->GetSession() ? handler->GetSession()->GetPlayer() : nullptr;
147 if (player && ticket->IsAssignedNotTo(player->GetGUID()))
148 {
149 handler->PSendSysMessage(LANG_COMMAND_TICKETCANNOTCLOSE, ticket->GetId());
150 return true;
151 }
152
153 sTicketMgr->ResolveAndCloseTicket(ticket->GetId(), player ? player->GetGUID() : ObjectGuid(uint64(0)));
154 sTicketMgr->UpdateLastChange();
155
156 std::string msg = ticket->FormatMessageString(*handler, player ? player->GetName().c_str() : "Console", NULL, NULL, NULL, NULL);
157 handler->SendGlobalGMSysMessage(msg.c_str());
158
159 // Inform player, who submitted this ticket, that it is closed
160 if (Player* submitter = ticket->GetPlayer())
161 {
162 WorldPacket data(SMSG_GMTICKET_DELETETICKET, 4);
163 data << uint32(GMTICKET_RESPONSE_TICKET_DELETED);
164 submitter->GetSession()->SendPacket(&data);
165 }
166 return true;
167 }

169 static bool HandleGMTicketCommentCommand(ChatHandler* handler, char const* args)
170 {
171 if (!*args)
172 return false;
173
174 char* ticketIdStr = strtok((char*)args, " ");
175 uint32 ticketId = atoi(ticketIdStr);
176
177 char* comment = strtok(NULL, "\n");
178 if (!comment)
179 return false;
180
181 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
182 if (!ticket || ticket->IsClosed())
183 {
184 handler->PSendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
185 return true;
186 }
187
188 // Cannot comment ticket assigned to someone else
189 //! Console excluded
190 Player* player = handler->GetSession() ? handler->GetSession()->GetPlayer() : nullptr;
191 if (player && ticket->IsAssignedNotTo(player->GetGUID()))
192 {
193 handler->PSendSysMessage(LANG_COMMAND_TICKETALREADYASSIGNED, ticket->GetId());
194 return true;
195 }
196
197 SQLTransaction trans = SQLTransaction(NULL);
198 ticket->SetComment(comment);
199 ticket->SaveToDB(trans);
200 sTicketMgr->UpdateLastChange();
201
202 std::string msg = ticket->FormatMessageString(*handler, NULL, ticket->GetAssignedToName().c_str(), NULL, NULL, NULL);
203 msg += handler->PGetParseString(LANG_COMMAND_TICKETLISTADDCOMMENT, player ? player->GetName().c_str() : "Console", comment);
204 handler->SendGlobalGMSysMessage(msg.c_str());
205
206 return true;
207 }

215 static bool HandleGMTicketCompleteCommand(ChatHandler* handler, char const* args)
216 {
217 if (!*args)
218 return false;
219
220 char* ticketIdStr = strtok((char*)args, " ");
221 uint32 ticketId = atoi(ticketIdStr);
222
223 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
224 if (!ticket || ticket->IsClosed() || ticket->IsCompleted())
225 {
226 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
227 return true;
228 }
229
230 char* response = strtok(NULL, "\n");
231 if (response)
232 {
233 // Cannot add response to ticket, assigned to someone else
234 //! Console excluded
235 Player* player = handler->GetSession() ? handler->GetSession()->GetPlayer() : nullptr;
236 if (player && ticket->IsAssignedNotTo(player->GetGUID()))
237 {
238 handler->PSendSysMessage(LANG_COMMAND_TICKETALREADYASSIGNED, ticket->GetId());
239 return true;
240 }
241
242 ticket->AppendResponse(response);
243 }
244
245 if (Player* player = ticket->GetPlayer())
246 ticket->SendResponse(player->GetSession());
247
248 Player* gm = handler->GetSession() ? handler->GetSession()->GetPlayer() : nullptr;
249
250 SQLTransaction trans = SQLTransaction(NULL);
251 ticket->SetResolvedBy(gm ? gm->GetGUID() : ObjectGuid(uint64(0)));
252 ticket->SetCompleted();
253 ticket->SaveToDB(trans);
254
255 std::string msg = ticket->FormatMessageString(*handler, NULL, NULL,
256 NULL, NULL, handler->GetSession() ? handler->GetSession()->GetPlayer()->GetName().c_str() : "Console");
257 handler->SendGlobalGMSysMessage(msg.c_str());
258 sTicketMgr->UpdateLastChange();
259 return true;
260 }

262 static bool HandleGMTicketDeleteByIdCommand(ChatHandler* handler, char const* args)
263 {
264 if (!*args)
265 return false;
266
267 uint32 ticketId = atoi(args);
268 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
269 if (!ticket)
270 {
271 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
272 return true;
273 }
274
275 if (!ticket->IsClosed())
276 {
277 handler->SendSysMessage(LANG_COMMAND_TICKETCLOSEFIRST);
278 return true;
279 }
280
281 std::string msg = ticket->FormatMessageString(*handler, NULL, NULL, NULL, handler->GetSession() ? handler->GetSession()->GetPlayer()->GetName().c_str() : "Console", NULL);
282 handler->SendGlobalGMSysMessage(msg.c_str());
283
284 sTicketMgr->RemoveTicket(ticket->GetId());
285 sTicketMgr->UpdateLastChange();
286
287 if (Player* player = ticket->GetPlayer())
288 {
289 // Force abandon ticket
290 WorldPacket data(SMSG_GMTICKET_DELETETICKET, 4);
291 data << uint32(GMTICKET_RESPONSE_TICKET_DELETED);
292 player->GetSession()->SendPacket(&data);
293 }
294
295 return true;
296 }

298 static bool HandleGMTicketEscalateCommand(ChatHandler* handler, char const* args)
299 {
300 if (!*args)
301 return false;
302
303 uint32 ticketId = atoi(args);
304 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
305 if (!ticket || ticket->IsClosed() || ticket->IsCompleted() || ticket->GetEscalatedStatus() != TICKET_UNASSIGNED)
306 {
307 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
308 return true;
309 }
310
311 ticket->SetEscalatedStatus(TICKET_IN_ESCALATION_QUEUE);
312
313 if (Player* player = ticket->GetPlayer())
314 sTicketMgr->SendTicket(player->GetSession(), ticket);
315
316 sTicketMgr->UpdateLastChange();
317 return true;
318 }

362 static bool HandleGMTicketUnAssignCommand(ChatHandler* handler, char const* args)
363 {
364 if (!*args)
365 return false;
366
367 uint32 ticketId = atoi(args);
368 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
369 if (!ticket || ticket->IsClosed())
370 {
371 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
372 return true;
373 }
374 // Ticket must be assigned
375 if (!ticket->IsAssigned())
376 {
377 handler->PSendSysMessage(LANG_COMMAND_TICKETNOTASSIGNED, ticket->GetId());
378 return true;
379 }
380
381 // Get security level of player, whom this ticket is assigned to
382 uint32 security = SEC_PLAYER;
383 Player* assignedPlayer = ticket->GetAssignedPlayer();
384 if (assignedPlayer)
385 security = assignedPlayer->GetSession()->GetSecurity();
386 else
387 {
388 ObjectGuid guid = ticket->GetAssignedToGUID();
389 uint32 accountId = sObjectMgr->GetPlayerAccountIdByGUID(guid);
390 security = AccountMgr::GetSecurity(accountId, realm.Id.Realm);
391 }
392
393 // Check security
394 //! If no m_session present it means we're issuing this command from the console
395 uint32 mySecurity = handler->GetSession() ? handler->GetSession()->GetSecurity() : SEC_CONSOLE;
396 if (security > mySecurity)
397 {
398 handler->SendSysMessage(LANG_COMMAND_TICKETUNASSIGNSECURITY);
399 return true;
400 }
401
402 std::string assignedTo = ticket->GetAssignedToName(); // copy assignedto name because we need it after the ticket has been unnassigned
403 SQLTransaction trans = SQLTransaction(NULL);
404 ticket->SetUnassigned();
405 ticket->SaveToDB(trans);
406 sTicketMgr->UpdateLastChange();
407
408 std::string msg = ticket->FormatMessageString(*handler, NULL, assignedTo.c_str(),
409 handler->GetSession() ? handler->GetSession()->GetPlayer()->GetName().c_str() : "Console", NULL, NULL);
410 handler->SendGlobalGMSysMessage(msg.c_str());
411
412 return true;
413 }

436 static bool HandleGMTicketGetByNameCommand(ChatHandler* handler, char const* args)
437 {
438 if (!*args)
439 return false;
440
441 std::string name(args);
442 if (!normalizePlayerName(name))
443 return false;
444
445 // Detect target's GUID
446 ObjectGuid guid;
447 if (Player* player = ObjectAccessor::FindPlayerByName(name))
448 guid = player->GetGUID();
449 else
450 guid = sObjectMgr->GetPlayerGUIDByName(name);
451
452 // Target must exist
453 if (!guid)
454 {
455 handler->SendSysMessage(LANG_NO_PLAYERS_FOUND);
456 return true;
457 }
458
459 // Ticket must exist
460 GmTicket* ticket = sTicketMgr->GetTicketByPlayer(guid);
461 if (!ticket)
462 {
463 handler->SendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
464 return true;
465 }
466
467 SQLTransaction trans = SQLTransaction(NULL);
468 ticket->SetViewed();
469 ticket->SaveToDB(trans);
470
471 handler->SendSysMessage(ticket->FormatMessageString(*handler, true).c_str());
472 return true;
473 }

475 static bool _HandleGMTicketResponseAppendCommand(char const* args, bool newLine, ChatHandler* handler)
476 {
477 if (!*args)
478 return false;
479
480 char* ticketIdStr = strtok((char*)args, " ");
481 uint32 ticketId = atoi(ticketIdStr);
482
483 char* response = strtok(NULL, "\n");
484 if (!response)
485 return false;
486
487 GmTicket* ticket = sTicketMgr->GetTicket(ticketId);
488 if (!ticket || ticket->IsClosed())
489 {
490 handler->PSendSysMessage(LANG_COMMAND_TICKETNOTEXIST);
491 return true;
492 }
493
494 // Cannot add response to ticket, assigned to someone else
495 //! Console excluded
496 Player* player = handler->GetSession() ? handler->GetSession()->GetPlayer() : nullptr;
497 if (player && ticket->IsAssignedNotTo(player->GetGUID()))
498 {
499 handler->PSendSysMessage(LANG_COMMAND_TICKETALREADYASSIGNED, ticket->GetId());
500 return true;
501 }
502
503 SQLTransaction trans = SQLTransaction(NULL);
504 ticket->AppendResponse(response);
505 if (newLine)
506 ticket->AppendResponse("\n");
507 ticket->SaveToDB(trans);
508
509 return true;
510 }

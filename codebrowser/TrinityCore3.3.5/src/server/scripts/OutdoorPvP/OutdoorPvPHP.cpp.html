65bool OutdoorPvPHP::SetupOutdoorPvP()
66{
67 m_AllianceTowersControlled = 0;
68 m_HordeTowersControlled = 0;
69 // add the zones affected by the pvp buff
70 for (int i = 0; i < OutdoorPvPHPBuffZonesNum; ++i)
71 RegisterZone(OutdoorPvPHPBuffZones[i]);
72
73 AddCapturePoint(new OPvPCapturePointHP(this, HP_TOWER_BROKEN_HILL));
74
75 AddCapturePoint(new OPvPCapturePointHP(this, HP_TOWER_OVERLOOK));
76
77 AddCapturePoint(new OPvPCapturePointHP(this, HP_TOWER_STADIUM));
78
79 return true;
80}

82void OutdoorPvPHP::HandlePlayerEnterZone(Player* player, uint32 zone)
83{
84 // add buffs
85 if (player->GetTeam() == ALLIANCE)
86 {
87 if (m_AllianceTowersControlled >=3)
88 player->CastSpell(player, AllianceBuff, true);
89 }
90 else
91 {
92 if (m_HordeTowersControlled >=3)
93 player->CastSpell(player, HordeBuff, true);
94 }
95 OutdoorPvP::HandlePlayerEnterZone(player, zone);
96}

109 OutdoorPvP::HandlePlayerLeaveZone(player, zone);

112bool OutdoorPvPHP::Update(uint32 diff)
113{
114 bool changed = OutdoorPvP::Update(diff);
115 if (changed)
116 {
117 if (m_AllianceTowersControlled == 3)
118 TeamApplyBuff(TEAM_ALLIANCE, AllianceBuff, HordeBuff);
119 else if (m_HordeTowersControlled == 3)
120 TeamApplyBuff(TEAM_HORDE, HordeBuff, AllianceBuff);
121 else
122 {
123 TeamCastSpell(TEAM_ALLIANCE, -AllianceBuff);
124 TeamCastSpell(TEAM_HORDE, -HordeBuff);
125 }
126 SendUpdateWorldState(HP_UI_TOWER_COUNT_A, m_AllianceTowersControlled);
127 SendUpdateWorldState(HP_UI_TOWER_COUNT_H, m_HordeTowersControlled);
128 }
129 return changed;
130}

132void OutdoorPvPHP::SendRemoveWorldStates(Player* player)
133{
134 player->SendUpdateWorldState(HP_UI_TOWER_DISPLAY_A, 0);
135 player->SendUpdateWorldState(HP_UI_TOWER_DISPLAY_H, 0);
136 player->SendUpdateWorldState(HP_UI_TOWER_COUNT_H, 0);
137 player->SendUpdateWorldState(HP_UI_TOWER_COUNT_A, 0);
138
139 for (int i = 0; i < HP_TOWER_NUM; ++i)
140 {
141 player->SendUpdateWorldState(HP_MAP_N[i], 0);
142 player->SendUpdateWorldState(HP_MAP_A[i], 0);
143 player->SendUpdateWorldState(HP_MAP_H[i], 0);
144 }
145}

147void OutdoorPvPHP::FillInitialWorldStates(WorldPacket &data)
148{
149 data << uint32(HP_UI_TOWER_DISPLAY_A) << uint32(1);
150 data << uint32(HP_UI_TOWER_DISPLAY_H) << uint32(1);

158void OPvPCapturePointHP::ChangeState()
159{
160 uint32 field = 0;
161 switch (m_OldState)
162 {
163 case OBJECTIVESTATE_NEUTRAL:
164 field = HP_MAP_N[m_TowerType];
165 break;
166 case OBJECTIVESTATE_ALLIANCE:
167 field = HP_MAP_A[m_TowerType];
168 if (uint32 alliance_towers = ((OutdoorPvPHP*)m_PvP)->GetAllianceTowersControlled())
169 ((OutdoorPvPHP*)m_PvP)->SetAllianceTowersControlled(--alliance_towers);
170 break;
171 case OBJECTIVESTATE_HORDE:
172 field = HP_MAP_H[m_TowerType];
173 if (uint32 horde_towers = ((OutdoorPvPHP*)m_PvP)->GetHordeTowersControlled())
174 ((OutdoorPvPHP*)m_PvP)->SetHordeTowersControlled(--horde_towers);
175 break;
176 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
177 field = HP_MAP_N[m_TowerType];
178 break;
179 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
180 field = HP_MAP_N[m_TowerType];
181 break;
182 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
183 field = HP_MAP_A[m_TowerType];
184 break;
185 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
186 field = HP_MAP_H[m_TowerType];
187 break;
188 }
189
190 // send world state update
191 if (field)
192 {
193 m_PvP->SendUpdateWorldState(field, 0);
194 field = 0;
195 }
196 uint32 artkit = 21;
197 uint32 artkit2 = HP_TowerArtKit_N[m_TowerType];
198 switch (m_State)
199 {
200 case OBJECTIVESTATE_NEUTRAL:
201 field = HP_MAP_N[m_TowerType];
202 break;
203 case OBJECTIVESTATE_ALLIANCE:
204 {
205 field = HP_MAP_A[m_TowerType];
206 artkit = 2;
207 artkit2 = HP_TowerArtKit_A[m_TowerType];
208 uint32 alliance_towers = ((OutdoorPvPHP*)m_PvP)->GetAllianceTowersControlled();
209 if (alliance_towers < 3)
210 ((OutdoorPvPHP*)m_PvP)->SetAllianceTowersControlled(++alliance_towers);
211 m_PvP->SendDefenseMessage(OutdoorPvPHPBuffZones[0], HP_LANG_CAPTURE_A[m_TowerType]);
212 break;
213 }
214 case OBJECTIVESTATE_HORDE:
215 {
216 field = HP_MAP_H[m_TowerType];
217 artkit = 1;
218 artkit2 = HP_TowerArtKit_H[m_TowerType];
219 uint32 horde_towers = ((OutdoorPvPHP*)m_PvP)->GetHordeTowersControlled();
220 if (horde_towers < 3)
221 ((OutdoorPvPHP*)m_PvP)->SetHordeTowersControlled(++horde_towers);
222 m_PvP->SendDefenseMessage(OutdoorPvPHPBuffZones[0], HP_LANG_CAPTURE_H[m_TowerType]);
223 break;
224 }
225 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
226 field = HP_MAP_N[m_TowerType];
227 break;
228 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
229 field = HP_MAP_N[m_TowerType];
230 break;
231 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
232 field = HP_MAP_A[m_TowerType];
233 artkit = 2;
234 artkit2 = HP_TowerArtKit_A[m_TowerType];
235 break;
236 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
237 field = HP_MAP_H[m_TowerType];
238 artkit = 1;
239 artkit2 = HP_TowerArtKit_H[m_TowerType];
240 break;
241 }
242
243 Map* map = sMapMgr->FindMap(530, 0);
244 auto bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
245 for (auto itr = bounds.first; itr != bounds.second; ++itr)
246 itr->second->SetGoArtKit(artkit);
247
248 bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_Objects[m_TowerType]);
249 for (auto itr = bounds.first; itr != bounds.second; ++itr)
250 itr->second->SetGoArtKit(artkit2);
251
252 // send world state update
253 if (field)
254 m_PvP->SendUpdateWorldState(field, 1);
255
256 // complete quest objective
257 if (m_State == OBJECTIVESTATE_ALLIANCE || m_State == OBJECTIVESTATE_HORDE)
258 SendObjectiveComplete(HP_CREDITMARKER[m_TowerType], ObjectGuid::Empty);
259}

261void OPvPCapturePointHP::FillInitialWorldStates(WorldPacket &data)
262{
263 switch (m_State)
264 {
265 case OBJECTIVESTATE_ALLIANCE:
266 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
267 data << uint32(HP_MAP_N[m_TowerType]) << uint32(0);
268 data << uint32(HP_MAP_A[m_TowerType]) << uint32(1);
269 data << uint32(HP_MAP_H[m_TowerType]) << uint32(0);
270 break;
271 case OBJECTIVESTATE_HORDE:
272 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
273 data << uint32(HP_MAP_N[m_TowerType]) << uint32(0);
274 data << uint32(HP_MAP_A[m_TowerType]) << uint32(0);
275 data << uint32(HP_MAP_H[m_TowerType]) << uint32(1);
276 break;
277 case OBJECTIVESTATE_NEUTRAL:
278 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
279 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
280 default:
281 data << uint32(HP_MAP_N[m_TowerType]) << uint32(1);
282 data << uint32(HP_MAP_A[m_TowerType]) << uint32(0);
283 data << uint32(HP_MAP_H[m_TowerType]) << uint32(0);
284 break;
285 }
286}

288void OutdoorPvPHP::HandleKillImpl(Player* player, Unit* killed)
289{
290 if (killed->GetTypeId() != TYPEID_PLAYER)
291 return;
292
293 if (player->GetTeam() == ALLIANCE && killed->ToPlayer()->GetTeam() != ALLIANCE)
294 player->CastSpell(player, AlliancePlayerKillReward, true);
295 else if (player->GetTeam() == HORDE && killed->ToPlayer()->GetTeam() != HORDE)
296 player->CastSpell(player, HordePlayerKillReward, true);
297}

55void OPvPCapturePointZM_Beacon::ChangeState()
56{
57 // if changing from controlling alliance to horde
58 if (m_OldState == OBJECTIVESTATE_ALLIANCE)
59 {
60 if (uint32 alliance_towers = ((OutdoorPvPZM*)m_PvP)->GetAllianceTowersControlled())
61 ((OutdoorPvPZM*)m_PvP)->SetAllianceTowersControlled(--alliance_towers);
62 }
63 // if changing from controlling horde to alliance
64 else if (m_OldState == OBJECTIVESTATE_HORDE)
65 {
66 if (uint32 horde_towers = ((OutdoorPvPZM*)m_PvP)->GetHordeTowersControlled())
67 ((OutdoorPvPZM*)m_PvP)->SetHordeTowersControlled(--horde_towers);
68 }
69
70 switch (m_State)
71 {
72 case OBJECTIVESTATE_ALLIANCE:
73 {
74 m_TowerState = ZM_TOWERSTATE_A;
75 uint32 alliance_towers = ((OutdoorPvPZM*)m_PvP)->GetAllianceTowersControlled();
76 if (alliance_towers < ZM_NUM_BEACONS)
77 ((OutdoorPvPZM*)m_PvP)->SetAllianceTowersControlled(++alliance_towers);
78 m_PvP->SendDefenseMessage(ZM_GRAVEYARD_ZONE, ZMBeaconCaptureA[m_TowerType]);
79 break;
80 }
81 case OBJECTIVESTATE_HORDE:
82 {
83 m_TowerState = ZM_TOWERSTATE_H;
84 uint32 horde_towers = ((OutdoorPvPZM*)m_PvP)->GetHordeTowersControlled();
85 if (horde_towers < ZM_NUM_BEACONS)
86 ((OutdoorPvPZM*)m_PvP)->SetHordeTowersControlled(++horde_towers);
87 m_PvP->SendDefenseMessage(ZM_GRAVEYARD_ZONE, ZMBeaconCaptureH[m_TowerType]);
88 break;
89 }
90 case OBJECTIVESTATE_NEUTRAL:
91 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
92 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
93 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
94 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
95 m_TowerState = ZM_TOWERSTATE_N;
96 break;
97 }
98
99 UpdateTowerState();
100}

102bool OutdoorPvPZM::Update(uint32 diff)
103{
104 bool changed = OutdoorPvP::Update(diff);
105 if (changed)
106 {
107 if (m_AllianceTowersControlled == ZM_NUM_BEACONS)
108 m_GraveYard->SetBeaconState(ALLIANCE);
109 else if (m_HordeTowersControlled == ZM_NUM_BEACONS)
110 m_GraveYard->SetBeaconState(HORDE);
111 else
112 m_GraveYard->SetBeaconState(0);
113 }
114 return changed;
115}

117void OutdoorPvPZM::HandlePlayerEnterZone(Player* player, uint32 zone)
118{
119 if (player->GetTeam() == ALLIANCE)
120 {
121 if (m_GraveYard->GetGraveYardState() & ZM_GRAVEYARD_A)
122 player->CastSpell(player, ZM_CAPTURE_BUFF, true);
123 }
124 else
125 {
126 if (m_GraveYard->GetGraveYardState() & ZM_GRAVEYARD_H)
127 player->CastSpell(player, ZM_CAPTURE_BUFF, true);
128 }
129 OutdoorPvP::HandlePlayerEnterZone(player, zone);
130}

150bool OutdoorPvPZM::SetupOutdoorPvP()
151{
152 m_AllianceTowersControlled = 0;
153 m_HordeTowersControlled = 0;
154
155 SetMapFromZone(OutdoorPvPZMBuffZones[0]);
156
157 // add the zones affected by the pvp buff
158 for (uint8 i = 0; i < OutdoorPvPZMBuffZonesNum; ++i)
159 RegisterZone(OutdoorPvPZMBuffZones[i]);
160
161 AddCapturePoint(new OPvPCapturePointZM_Beacon(this, ZM_BEACON_WEST));
162 AddCapturePoint(new OPvPCapturePointZM_Beacon(this, ZM_BEACON_EAST));
163 m_GraveYard = new OPvPCapturePointZM_GraveYard(this);
164 AddCapturePoint(m_GraveYard); // though the update function isn't used, the handleusego is!
165
166 return true;
167}

169void OutdoorPvPZM::HandleKillImpl(Player* player, Unit* killed)
170{
171 if (killed->GetTypeId() != TYPEID_PLAYER)
172 return;
173
174 if (player->GetTeam() == ALLIANCE && killed->ToPlayer()->GetTeam() != ALLIANCE)
175 player->CastSpell(player, ZM_AlliancePlayerKillReward, true);
176 else if (player->GetTeam() == HORDE && killed->ToPlayer()->GetTeam() != HORDE)
177 player->CastSpell(player, ZM_HordePlayerKillReward, true);

187int32 OPvPCapturePointZM_GraveYard::HandleOpenGo(Player* player, GameObject* go)
188{
189 int32 retval = OPvPCapturePoint::HandleOpenGo(player, go);
190 if (retval >= 0)
191 {
192 if (player->HasAura(ZM_BATTLE_STANDARD_A) && m_GraveYardState != ZM_GRAVEYARD_A)
193 {
194 m_GraveYardState = ZM_GRAVEYARD_A;
195 DelObject(0); // only one gotype is used in the whole outdoor pvp, no need to call it a constant
196 AddObject(0, ZM_Banner_A.entry, ZM_Banner_A.map, ZM_Banner_A.x, ZM_Banner_A.y, ZM_Banner_A.z, ZM_Banner_A.o, ZM_Banner_A.rot0, ZM_Banner_A.rot1, ZM_Banner_A.rot2, ZM_Banner_A.rot3);
197 sObjectMgr->RemoveGraveYardLink(ZM_GRAVEYARD_ID, ZM_GRAVEYARD_ZONE, HORDE); // rem gy
198 sObjectMgr->AddGraveYardLink(ZM_GRAVEYARD_ID, ZM_GRAVEYARD_ZONE, ALLIANCE, false); // add gy
199 m_PvP->TeamApplyBuff(TEAM_ALLIANCE, ZM_CAPTURE_BUFF);
200 player->RemoveAurasDueToSpell(ZM_BATTLE_STANDARD_A);
201 m_PvP->SendDefenseMessage(ZM_GRAVEYARD_ZONE, TEXT_TWIN_SPIRE_RUINS_TAKEN_ALLIANCE);
202 }
203 else if (player->HasAura(ZM_BATTLE_STANDARD_H) && m_GraveYardState != ZM_GRAVEYARD_H)
204 {
205 m_GraveYardState = ZM_GRAVEYARD_H;
206 DelObject(0); // only one gotype is used in the whole outdoor pvp, no need to call it a constant
207 AddObject(0, ZM_Banner_H.entry, ZM_Banner_H.map, ZM_Banner_H.x, ZM_Banner_H.y, ZM_Banner_H.z, ZM_Banner_H.o, ZM_Banner_H.rot0, ZM_Banner_H.rot1, ZM_Banner_H.rot2, ZM_Banner_H.rot3);
208 sObjectMgr->RemoveGraveYardLink(ZM_GRAVEYARD_ID, ZM_GRAVEYARD_ZONE, ALLIANCE); // rem gy
209 sObjectMgr->AddGraveYardLink(ZM_GRAVEYARD_ID, ZM_GRAVEYARD_ZONE, HORDE, false); // add gy
210 m_PvP->TeamApplyBuff(TEAM_HORDE, ZM_CAPTURE_BUFF);
211 player->RemoveAurasDueToSpell(ZM_BATTLE_STANDARD_H);
212 m_PvP->SendDefenseMessage(ZM_GRAVEYARD_ZONE, TEXT_TWIN_SPIRE_RUINS_TAKEN_HORDE);
213 }
214 UpdateTowerState();
215 }
216 return retval;
217}

256void OPvPCapturePointZM_GraveYard::SetBeaconState(uint32 controlling_faction)
257{
258 // nothing to do here
259 if (m_BothControllingFaction == controlling_faction)
260 return;
261 m_BothControllingFaction = controlling_faction;
262
263 switch (controlling_faction)
264 {
265 case ALLIANCE:
266 // if ally already controls the gy and taken back both beacons, return, nothing to do for us
267 if (m_GraveYardState & ZM_GRAVEYARD_A)
268 return;
269 // ally doesn't control the gy, but controls the side beacons -> add gossip option, add neutral banner
270 break;
271 case HORDE:
272 // if horde already controls the gy and taken back both beacons, return, nothing to do for us
273 if (m_GraveYardState & ZM_GRAVEYARD_H)
274 return;
275 // horde doesn't control the gy, but controls the side beacons -> add gossip option, add neutral banner
276 break;
277 default:
278 // if the graveyard is not neutral, then leave it that way
279 // if the graveyard is neutral, then we have to dispel the buff from the flag carrier
280 if (m_GraveYardState & ZM_GRAVEYARD_N)
281 {
282 // gy was neutral, thus neutral banner was spawned, it is possible that someone was taking the flag to the gy
283 if (m_FlagCarrierGUID)
284 {
285 // remove flag from carrier, reset flag carrier guid
286 Player* p = ObjectAccessor::FindPlayer(m_FlagCarrierGUID);
287 if (p)
288 {
289 p->RemoveAurasDueToSpell(ZM_BATTLE_STANDARD_A);
290 p->RemoveAurasDueToSpell(ZM_BATTLE_STANDARD_H);
291 }
292 m_FlagCarrierGUID.Clear();
293 }
294 }
295 break;
296 }
297 // send worldstateupdate
298 UpdateTowerState();
299}

301bool OPvPCapturePointZM_GraveYard::CanTalkTo(Player* player, Creature* c, GossipMenuItems const& /*gso*/)
302{
303 std::map<uint32, uint32>::iterator itr = m_CreatureTypes.find(c->GetSpawnId());
304
305 if (itr != m_CreatureTypes.end())
306 {
307 if (itr->second == ZM_ALLIANCE_FIELD_SCOUT && player->GetTeam() == ALLIANCE && m_BothControllingFaction == ALLIANCE && !m_FlagCarrierGUID && m_GraveYardState != ZM_GRAVEYARD_A)
308 return true;
309 else if (itr->second == ZM_HORDE_FIELD_SCOUT && player->GetTeam() == HORDE && m_BothControllingFaction == HORDE && !m_FlagCarrierGUID && m_GraveYardState != ZM_GRAVEYARD_H)
310 return true;
311 }
312 return false;
313}

315bool OPvPCapturePointZM_GraveYard::HandleGossipOption(Player* player, Creature* creature, uint32 /*gossipid*/)
316{
317 std::map<uint32, uint32>::iterator itr = m_CreatureTypes.find(creature->GetSpawnId());
318
319 if (itr != m_CreatureTypes.end())
320 {
321 // if the flag is already taken, then return
322 if (m_FlagCarrierGUID)
323 return true;
324 if (itr->second == ZM_ALLIANCE_FIELD_SCOUT)
325 {
326 creature->CastSpell(player, ZM_BATTLE_STANDARD_A, true);
327 m_FlagCarrierGUID = player->GetGUID();
328 }
329 else if (itr->second == ZM_HORDE_FIELD_SCOUT)
330 {
331 creature->CastSpell(player, ZM_BATTLE_STANDARD_H, true);
332 m_FlagCarrierGUID = player->GetGUID();
333 }
334 UpdateTowerState();
335 player->PlayerTalkClass->SendCloseGossip();
336 return true;
337 }
338 return false;
339}

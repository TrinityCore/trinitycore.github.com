37void OPvPCapturePointEP_EWT::ChangeState()
38{
39 // if changing from controlling alliance to horde or vice versa
40 if (m_OldState == OBJECTIVESTATE_ALLIANCE || m_OldState == OBJECTIVESTATE_HORDE)
41 ((OutdoorPvPEP*)m_PvP)->SetControlledState(EP_EWT, 0);
42
43 uint32 artkit = 21;
44
45 switch (m_State)
46 {
47 case OBJECTIVESTATE_ALLIANCE:
48 m_TowerState = EP_TS_A;
49 artkit = 2;
50 SummonSupportUnitAtNorthpassTower(ALLIANCE);
51 ((OutdoorPvPEP*)m_PvP)->SetControlledState(EP_EWT, ALLIANCE);
52 m_PvP->SendDefenseMessage(EP_GraveYardZone, TEXT_EASTWALL_TOWER_TAKEN_ALLIANCE);
53 break;
54 case OBJECTIVESTATE_HORDE:
55 m_TowerState = EP_TS_H;
56 artkit = 1;
57 SummonSupportUnitAtNorthpassTower(HORDE);
58 ((OutdoorPvPEP*)m_PvP)->SetControlledState(EP_EWT, HORDE);
59 m_PvP->SendDefenseMessage(EP_GraveYardZone, TEXT_EASTWALL_TOWER_TAKEN_HORDE);
60 break;
61 case OBJECTIVESTATE_NEUTRAL:
62 m_TowerState = EP_TS_N;
63 break;
64 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
65 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
66 m_TowerState = EP_TS_N_A;
67 break;
68 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
69 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
70 m_TowerState = EP_TS_N_H;
71 break;
72 }
73
74 Map* map = sMapMgr->FindMap(0, 0);
75 auto bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
76 for (auto itr = bounds.first; itr != bounds.second; ++itr)
77 itr->second->SetGoArtKit(artkit);
78
79 bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_Objects[EP_EWT_FLAGS]);
80 for (auto itr = bounds.first; itr != bounds.second; ++itr)
81 itr->second->SetGoArtKit(artkit);
82
83 UpdateTowerState();
84
85 // complete quest objective
86 if (m_TowerState == EP_TS_A || m_TowerState == EP_TS_H)
87 SendObjectiveComplete(EP_EWT_CM, ObjectGuid::Empty);
88}

108void OPvPCapturePointEP_EWT::SummonSupportUnitAtNorthpassTower(uint32 team)
109{
110 if (m_UnitsSummonedSide != team)
111 {
112 m_UnitsSummonedSide = team;
113 creature_type const* ct = nullptr;
114 if (team == ALLIANCE)
115 ct = EP_EWT_Summons_A;
116 else
117 ct = EP_EWT_Summons_H;
118
119 for (uint8 i = 0; i < EP_EWT_NUM_CREATURES; ++i)
120 {
121 DelCreature(i);
122 AddCreature(i, ct[i].entry, ct[i].map, ct[i].x, ct[i].y, ct[i].z, ct[i].o, OutdoorPvP::GetTeamIdByTeam(team), 1000000);
123 }
124 }

482bool OutdoorPvPEP::Update(uint32 diff)
483{
484 if (OutdoorPvP::Update(diff))
485 {
486 m_AllianceTowersControlled = 0;
487 m_HordeTowersControlled = 0;
488 for (int i = 0; i < EP_TOWER_NUM; ++i)
489 {
490 if (EP_Controls[i] == ALLIANCE)
491 ++m_AllianceTowersControlled;
492 else if (EP_Controls[i] == HORDE)

494 SendUpdateWorldState(EP_UI_TOWER_COUNT_A, m_AllianceTowersControlled);
495 SendUpdateWorldState(EP_UI_TOWER_COUNT_H, m_HordeTowersControlled);
496 BuffTeams();
497 }
498 return true;
499 }
500 return false;

503void OutdoorPvPEP::HandlePlayerEnterZone(Player* player, uint32 zone)
504{

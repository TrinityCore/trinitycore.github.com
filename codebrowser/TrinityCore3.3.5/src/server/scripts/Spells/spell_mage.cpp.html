236 void Absorb(AuraEffect* aurEff, DamageInfo& dmgInfo, uint32& absorbAmount)
237 {
238 Unit* target = GetTarget();
239 if (AuraEffect* talentAurEff = target->GetAuraEffectOfRankedSpell(SPELL_MAGE_FROST_WARDING_R1, EFFECT_0))
240 {
241 int32 chance = talentAurEff->GetSpellInfo()->Effects[EFFECT_1].CalcValue(); // SPELL_EFFECT_DUMMY with NO_TARGET
242
243 if (roll_chance_i(chance))
244 {
245 int32 bp = dmgInfo.GetDamage();
246 dmgInfo.AbsorbDamage(bp);
247 target->CastCustomSpell(target, SPELL_MAGE_FROST_WARDING_TRIGGERED, &bp, NULL, NULL, true, NULL, aurEff);
248 absorbAmount = 0;
249 PreventDefaultAction();
250 }
251 }
252 }
253

255 {
256 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_mage_fire_frost_ward_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);

322{
323 public:
324 spell_mage_ice_barrier() : SpellScriptLoader("spell_mage_ice_barrier") { }
325
326 class spell_mage_ice_barrier_AuraScript : public spell_mage_incanters_absorbtion_base_AuraScript
327 {
328 PrepareAuraScript(spell_mage_ice_barrier_AuraScript);
329
330 bool Validate(SpellInfo const* /*spellInfo*/) override
331 {
332 if (!sSpellMgr->GetSpellInfo(SPELL_MAGE_SHATTERED_BARRIER))
333 return false;
334 return true;
335 }
336
337 void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool& canBeRecalculated)
338 {
339 canBeRecalculated = false;
340 if (Unit* caster = GetCaster())
341 {
342 // +80.68% from sp bonus
343 float bonus = 0.8068f;
344
345 bonus *= caster->SpellBaseHealingBonusDone(GetSpellInfo()->GetSchoolMask());
346
347 // Glyph of Ice Barrier: its weird having a SPELLMOD_ALL_EFFECTS here but its blizzards doing :)
348 // Glyph of Ice Barrier is only applied at the spell damage bonus because it was already applied to the base value in CalculateSpellDamage
349 bonus = caster->ApplyEffectModifiers(GetSpellInfo(), aurEff->GetEffIndex(), bonus);
350
351 bonus *= caster->CalculateLevelPenalty(GetSpellInfo());
352
353 amount += int32(bonus);
354 }
355 }
356
357 void AfterRemove(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
358 {
359 // Shattered Barrier
360 // Procs only if removed by damage.
361 if (aurEff->GetAmount() <= 0)
362 if (Unit* caster = GetCaster())
363 if (AuraEffect* dummy = caster->GetDummyAuraEffect(SPELLFAMILY_MAGE, SPELL_ICON_MAGE_SHATTERED_BARRIER, EFFECT_0))
364 if (roll_chance_i(dummy->GetSpellInfo()->ProcChance))
365 caster->CastSpell(GetTarget(), SPELL_MAGE_SHATTERED_BARRIER, true, nullptr, aurEff);
366 }
367
368 void Register() override
369 {
370 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_mage_ice_barrier_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);
371 AfterEffectAbsorb += AuraEffectAbsorbFn(spell_mage_ice_barrier_AuraScript::Trigger, EFFECT_0);
372 AfterEffectRemove += AuraEffectRemoveFn(spell_mage_ice_barrier_AuraScript::AfterRemove, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB, AURA_EFFECT_HANDLE_REAL);

57class PowerCheck
58{
59 public:
60 explicit PowerCheck(Powers const power) : _power(power) { }
61
62 bool operator()(WorldObject* obj) const
63 {
64 if (Unit* target = obj->ToUnit())
65 return target->getPowerType() != _power;
66
67 return true;
68 }
69
70 private:
71 Powers const _power;
72};
73
74class RaidCheck
75{
76 public:
77 explicit RaidCheck(Unit const* caster) : _caster(caster) { }
78
79 bool operator()(WorldObject* obj) const
80 {
81 if (Unit* target = obj->ToUnit())
82 return !_caster->IsInRaidWith(target);
83
84 return true;
85 }
86
87 private:
88 Unit const* _caster;
89};

151 {
152 return eventInfo.GetProcTarget() != nullptr;
153 }

155 void HandleProc(AuraEffect const* aurEff, ProcEventInfo& eventInfo)
156 {
157 PreventDefaultAction();
158
159 int32 absorb = CalculatePct(int32(eventInfo.GetHealInfo()->GetHeal()), aurEff->GetAmount());
160
161 // Multiple effects stack, so let's try to find this aura.
162 if (AuraEffect const* aegis = eventInfo.GetProcTarget()->GetAuraEffect(SPELL_PRIEST_DIVINE_AEGIS, EFFECT_0))
163 absorb += aegis->GetAmount();
164
165 absorb = std::min(absorb, eventInfo.GetProcTarget()->getLevel() * 125);
166
167 GetTarget()->CastCustomSpell(SPELL_PRIEST_DIVINE_AEGIS, SPELLVALUE_BASE_POINT0, absorb, eventInfo.GetProcTarget(), true, NULL, aurEff);
168 }
169
170 void Register() override
171 {
172 DoCheckProc += AuraCheckProcFn(spell_pri_divine_aegis_AuraScript::CheckProc);
173 OnEffectProc += AuraEffectProcFn(spell_pri_divine_aegis_AuraScript::HandleProc, EFFECT_0, SPELL_AURA_DUMMY);
174 }
175 };

177 AuraScript* GetAuraScript() const override

264 PrepareAuraScript(spell_pri_guardian_spirit_AuraScript);
265
266 public:
267 spell_pri_guardian_spirit_AuraScript()
268 {
269 healPct = 0;
270 }
271
272 private:
273 uint32 healPct;
274
275 bool Validate(SpellInfo const* /*spellInfo*/) override
276 {
277 if (!sSpellMgr->GetSpellInfo(SPELL_PRIEST_GUARDIAN_SPIRIT_HEAL))
278 return false;
279 return true;
280 }
281
282 bool Load() override
283 {
284 healPct = GetSpellInfo()->Effects[EFFECT_1].CalcValue();
285 return true;
286 }
287
288 void CalculateAmount(AuraEffect const* /*aurEff*/, int32 & amount, bool & /*canBeRecalculated*/)
289 {
290 // Set absorbtion amount to unlimited
291 amount = -1;
292 }
293
294 void Absorb(AuraEffect* /*aurEff*/, DamageInfo & dmgInfo, uint32 & absorbAmount)
295 {
296 Unit* target = GetTarget();
297 if (dmgInfo.GetDamage() < target->GetHealth())
298 return;
299
300 int32 healAmount = int32(target->CountPctFromMaxHealth(healPct));
301 // remove the aura now, we don't want 40% healing bonus
302 Remove(AURA_REMOVE_BY_ENEMY_SPELL);
303 target->CastCustomSpell(target, SPELL_PRIEST_GUARDIAN_SPIRIT_HEAL, &healAmount, NULL, NULL, true);
304 absorbAmount = dmgInfo.GetDamage();
305 }
306
307 void Register() override
308 {
309 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pri_guardian_spirit_AuraScript::CalculateAmount, EFFECT_1, SPELL_AURA_SCHOOL_ABSORB);
310 OnEffectAbsorb += AuraEffectAbsorbFn(spell_pri_guardian_spirit_AuraScript::Absorb, EFFECT_1);
311 }
312 };
313
314 AuraScript* GetAuraScript() const override
315 {
316 return new spell_pri_guardian_spirit_AuraScript();

566class spell_pri_penance : public SpellScriptLoader
567{
568 public:
569 spell_pri_penance() : SpellScriptLoader("spell_pri_penance") { }
570
571 class spell_pri_penance_SpellScript : public SpellScript
572 {
573 PrepareSpellScript(spell_pri_penance_SpellScript);
574
575 bool Load() override
576 {
577 return GetCaster()->GetTypeId() == TYPEID_PLAYER;
578 }
579
580 bool Validate(SpellInfo const* spellInfo) override
581 {
582 SpellInfo const* firstRankSpellInfo = sSpellMgr->GetSpellInfo(SPELL_PRIEST_PENANCE_R1);
583 if (!firstRankSpellInfo)
584 return false;
585
586 // can't use other spell than this penance due to spell_ranks dependency
587 if (!spellInfo->IsRankOf(firstRankSpellInfo))
588 return false;
589
590 uint8 rank = spellInfo->GetRank();
591 if (!sSpellMgr->GetSpellWithRank(SPELL_PRIEST_PENANCE_R1_DAMAGE, rank, true))
592 return false;
593 if (!sSpellMgr->GetSpellWithRank(SPELL_PRIEST_PENANCE_R1_HEAL, rank, true))
594 return false;
595
596 return true;
597 }
598
599 void HandleDummy(SpellEffIndex /*effIndex*/)
600 {
601 Unit* caster = GetCaster();
602 if (Unit* unitTarget = GetHitUnit())
603 {
604 if (!unitTarget->IsAlive())
605 return;
606
607 uint8 rank = GetSpellInfo()->GetRank();
608
609 if (caster->IsFriendlyTo(unitTarget))
610 caster->CastSpell(unitTarget, sSpellMgr->GetSpellWithRank(SPELL_PRIEST_PENANCE_R1_HEAL, rank), false);
611 else
612 caster->CastSpell(unitTarget, sSpellMgr->GetSpellWithRank(SPELL_PRIEST_PENANCE_R1_DAMAGE, rank), false);
613 }
614 }
615
616 SpellCastResult CheckCast()
617 {
618 Player* caster = GetCaster()->ToPlayer();
619 if (Unit* target = GetExplTargetUnit())
620 if (!caster->IsFriendlyTo(target))
621 {
622 if (!caster->IsValidAttackTarget(target))
623 return SPELL_FAILED_BAD_TARGETS;
624 if (!caster->isInFront(target))
625 return SPELL_FAILED_UNIT_NOT_INFRONT;
626 }
627 return SPELL_CAST_OK;
628 }
629
630 void Register() override
631 {
632 OnEffectHitTarget += SpellEffectFn(spell_pri_penance_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
633 OnCheckCast += SpellCheckCastFn(spell_pri_penance_SpellScript::CheckCast);
634 }
635 };

637 SpellScript* GetSpellScript() const override
638 {
639 return new spell_pri_penance_SpellScript;

643// -17 - Power Word: Shield
644class spell_pri_power_word_shield : public SpellScriptLoader
645{
646 public:
647 spell_pri_power_word_shield() : SpellScriptLoader("spell_pri_power_word_shield") { }
648
649 class spell_pri_power_word_shield_AuraScript : public AuraScript
650 {
651 PrepareAuraScript(spell_pri_power_word_shield_AuraScript);
652
653 bool Validate(SpellInfo const* /*spellInfo*/) override
654 {
655 if (!sSpellMgr->GetSpellInfo(SPELL_PRIEST_REFLECTIVE_SHIELD_TRIGGERED))
656 return false;
657 if (!sSpellMgr->GetSpellInfo(SPELL_PRIEST_REFLECTIVE_SHIELD_R1))
658 return false;
659 return true;
660 }
661
662 void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool& canBeRecalculated)
663 {
664 canBeRecalculated = false;
665 if (Unit* caster = GetCaster())
666 {
667 // +80.68% from sp bonus
668 float bonus = 0.8068f;
669
670 // Borrowed Time
671 if (AuraEffect const* borrowedTime = caster->GetDummyAuraEffect(SPELLFAMILY_PRIEST, PRIEST_ICON_ID_BORROWED_TIME, EFFECT_1))
672 bonus += CalculatePct(1.0f, borrowedTime->GetAmount());
673
674 bonus *= caster->SpellBaseHealingBonusDone(GetSpellInfo()->GetSchoolMask());
675
676 // Improved PW: Shield: its weird having a SPELLMOD_ALL_EFFECTS here but its blizzards doing :)
677 // Improved PW: Shield is only applied at the spell healing bonus because it was already applied to the base value in CalculateSpellDamage
678 bonus = caster->ApplyEffectModifiers(GetSpellInfo(), aurEff->GetEffIndex(), bonus);
679 bonus *= caster->CalculateLevelPenalty(GetSpellInfo());
680
681 amount += int32(bonus);
682
683 // Twin Disciplines
684 if (AuraEffect const* twinDisciplines = caster->GetAuraEffect(SPELL_AURA_ADD_PCT_MODIFIER, SPELLFAMILY_PRIEST, 0x400000, 0, 0, GetCasterGUID()))
685 AddPct(amount, twinDisciplines->GetAmount());
686
687 // Focused Power
688 amount *= caster->GetTotalAuraMultiplier(SPELL_AURA_MOD_HEALING_DONE_PERCENT);
689 }
690 }
691
692 void ReflectDamage(AuraEffect* aurEff, DamageInfo& dmgInfo, uint32& absorbAmount)
693 {
694 Unit* target = GetTarget();
695 if (dmgInfo.GetAttacker() == target)
696 return;
697
698 if (AuraEffect* talentAurEff = target->GetAuraEffectOfRankedSpell(SPELL_PRIEST_REFLECTIVE_SHIELD_R1, EFFECT_0))
699 {
700 int32 bp = CalculatePct(absorbAmount, talentAurEff->GetAmount());
701 target->CastCustomSpell(dmgInfo.GetAttacker(), SPELL_PRIEST_REFLECTIVE_SHIELD_TRIGGERED, &bp, NULL, NULL, true, NULL, aurEff);
702 }
703 }

706 {
707 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pri_power_word_shield_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);
708 AfterEffectAbsorb += AuraEffectAbsorbFn(spell_pri_power_word_shield_AuraScript::ReflectDamage, EFFECT_0);

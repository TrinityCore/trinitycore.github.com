387class CorpseExplosionCheck
388{
389public:
390 explicit CorpseExplosionCheck(ObjectGuid casterGUID, bool allowGhoul) : _casterGUID(casterGUID),
391 _allowGhoul(allowGhoul) { }
392
393 bool operator()(WorldObject* obj) const
394 {
395 if (Unit* target = obj->ToUnit())
396 {
397 if ((target->isDead() || (_allowGhoul && target->GetEntry() == NPC_DK_GHOUL && target->GetOwnerGUID() == _casterGUID))
398 && !(target->GetCreatureTypeMask() & CREATURE_TYPEMASK_MECHANICAL_OR_ELEMENTAL)
399 && target->GetDisplayId() == target->GetNativeDisplayId())
400 return false;
401 }
402
403 return true;
404 }
405
406private:
407 ObjectGuid _casterGUID;
408 bool _allowGhoul;
409};

411// 49158 - Corpse Explosion (51325, 51326, 51327, 51328)
412class spell_dk_corpse_explosion : public SpellScriptLoader
413{
414 public:
415 spell_dk_corpse_explosion() : SpellScriptLoader("spell_dk_corpse_explosion") { }
416
417 class spell_dk_corpse_explosion_SpellScript : public SpellScript
418 {
419 PrepareSpellScript(spell_dk_corpse_explosion_SpellScript);
420
421 public:
422 spell_dk_corpse_explosion_SpellScript()
423 {
424 _target = nullptr;
425 }
426
427 private:
428 bool Validate(SpellInfo const* spellInfo) override
429 {
430 if (!sSpellMgr->GetSpellInfo(SPELL_DK_CORPSE_EXPLOSION_TRIGGERED)
431 || !sSpellMgr->GetSpellInfo(SPELL_DK_GHOUL_EXPLODE)
432 || !sSpellMgr->GetSpellInfo(SPELL_DK_CORPSE_EXPLOSION_VISUAL)
433 || !sSpellMgr->GetSpellInfo(spellInfo->Effects[EFFECT_1].CalcValue()))
434 return false;
435 return true;
436 }
437
438 void CheckTarget(WorldObject*& target)
439 {
440 if (CorpseExplosionCheck(GetCaster()->GetGUID(), true)(target))
441 target = NULL;
442
443 _target = target;
444 }
445
446 void CheckTargets(std::list<WorldObject*>& targets)
447 {
448 WorldObject* target = _target;
449 if (!target)
450 {
451 targets.remove_if(CorpseExplosionCheck(GetCaster()->GetGUID(), false));
452 if (targets.empty())
453 {
454 FinishCast(SPELL_FAILED_CANT_DO_THAT_RIGHT_NOW);
455 return;
456 }
457 target = Trinity::Containers::SelectRandomContainerElement(targets);
458 targets.clear();
459 targets.push_back(target);
460 }
461 else
462 targets.clear();
463 }
464
465 void HandleDamage(SpellEffIndex effIndex, Unit* target)
466 {
467 if (effIndex == EFFECT_0)
468 GetCaster()->CastCustomSpell(GetSpellInfo()->Effects[EFFECT_1].CalcValue(), SPELLVALUE_BASE_POINT0, GetEffectValue(), target, true);
469 else if (effIndex == EFFECT_1)
470 GetCaster()->CastCustomSpell(GetEffectValue(), SPELLVALUE_BASE_POINT0, GetSpell()->CalculateDamage(EFFECT_0, NULL), target, true);
471 }
472
473 void HandleCorpseExplosion(SpellEffIndex effIndex)
474 {
475 if (Unit* unitTarget = GetHitUnit())
476 {
477 if (unitTarget->IsAlive()) // Living ghoul as a target
478 {
479 unitTarget->CastSpell(unitTarget, SPELL_DK_GHOUL_EXPLODE, false);
480 // Corpse Explosion (Suicide) and Set corpse look handled in SpellScript of SPELL_DK_GHOUL_EXPLODE
481 }
482 else // Some corpse
483 {
484 HandleDamage(effIndex, unitTarget);
485 // Corpse Explosion (Suicide)
486 unitTarget->CastSpell(unitTarget, SPELL_DK_CORPSE_EXPLOSION_TRIGGERED, true);
487 // Set corpse look
488 GetCaster()->CastSpell(unitTarget, SPELL_DK_CORPSE_EXPLOSION_VISUAL, true);
489 }
490 }
491 }
492
493 void Register() override
494 {
495 OnObjectTargetSelect += SpellObjectTargetSelectFn(spell_dk_corpse_explosion_SpellScript::CheckTarget, EFFECT_0, TARGET_UNIT_TARGET_ANY);
496 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_dk_corpse_explosion_SpellScript::CheckTargets, EFFECT_1, TARGET_UNIT_DEST_AREA_ENTRY);
497 OnEffectHitTarget += SpellEffectFn(spell_dk_corpse_explosion_SpellScript::HandleCorpseExplosion, EFFECT_0, SPELL_EFFECT_DUMMY);
498 OnEffectHitTarget += SpellEffectFn(spell_dk_corpse_explosion_SpellScript::HandleCorpseExplosion, EFFECT_1, SPELL_EFFECT_DUMMY);
499 }
500
501 private:
502 WorldObject* _target;
503 };
504
505 SpellScript* GetSpellScript() const override
506 {
507 return new spell_dk_corpse_explosion_SpellScript();
508 }

99// 31850 - Ardent Defender
100class spell_pal_ardent_defender : public SpellScriptLoader
101{
102 public:
103 spell_pal_ardent_defender() : SpellScriptLoader("spell_pal_ardent_defender") { }
104
105 class spell_pal_ardent_defender_AuraScript : public AuraScript
106 {
107 PrepareAuraScript(spell_pal_ardent_defender_AuraScript);
108
109 public:
110 spell_pal_ardent_defender_AuraScript()
111 {
112 absorbPct = 0;
113 healPct = 0;
114 }
115
116 private:
117 uint32 absorbPct, healPct;
118
119 enum Spell
120 {
121 PAL_SPELL_ARDENT_DEFENDER_HEAL = 66235,
122 };
123
124 bool Load() override
125 {
126 healPct = GetSpellInfo()->Effects[EFFECT_1].CalcValue();
127 absorbPct = GetSpellInfo()->Effects[EFFECT_0].CalcValue();
128 return GetUnitOwner()->GetTypeId() == TYPEID_PLAYER;
129 }
130
131 void CalculateAmount(AuraEffect const* /*aurEff*/, int32 & amount, bool & /*canBeRecalculated*/)
132 {
133 // Set absorbtion amount to unlimited
134 amount = -1;
135 }
136
137 void Absorb(AuraEffect* aurEff, DamageInfo & dmgInfo, uint32 & absorbAmount)
138 {
139 Unit* victim = GetTarget();
140 int32 remainingHealth = victim->GetHealth() - dmgInfo.GetDamage();
141 uint32 allowedHealth = victim->CountPctFromMaxHealth(35);
142 // If damage kills us
143 if (remainingHealth <= 0 && !victim->GetSpellHistory()->HasCooldown(PAL_SPELL_ARDENT_DEFENDER_HEAL))
144 {
145 // Cast healing spell, completely avoid damage
146 absorbAmount = dmgInfo.GetDamage();
147
148 uint32 defenseSkillValue = victim->GetDefenseSkillValue();
149 // Max heal when defense skill denies critical hits from raid bosses
150 // Formula: max defense at level + 140 (raiting from gear)
151 uint32 reqDefForMaxHeal = victim->getLevel() * 5 + 140;
152 float pctFromDefense = (defenseSkillValue >= reqDefForMaxHeal)
153 ? 1.0f
154 : float(defenseSkillValue) / float(reqDefForMaxHeal);
155
156 int32 healAmount = int32(victim->CountPctFromMaxHealth(uint32(healPct * pctFromDefense)));
157 victim->CastCustomSpell(victim, PAL_SPELL_ARDENT_DEFENDER_HEAL, &healAmount, NULL, NULL, true, NULL, aurEff);
158 victim->GetSpellHistory()->AddCooldown(PAL_SPELL_ARDENT_DEFENDER_HEAL, 0, std::chrono::minutes(2));
159 }
160 else if (remainingHealth < int32(allowedHealth))
161 {
162 // Reduce damage that brings us under 35% (or full damage if we are already under 35%) by x%
163 uint32 damageToReduce = (victim->GetHealth() < allowedHealth)
164 ? dmgInfo.GetDamage()
165 : allowedHealth - remainingHealth;
166 absorbAmount = CalculatePct(damageToReduce, absorbPct);
167 }
168 }
169
170 void Register() override
171 {
172 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pal_ardent_defender_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);
173 OnEffectAbsorb += AuraEffectAbsorbFn(spell_pal_ardent_defender_AuraScript::Absorb, EFFECT_0);
174 }
175 };
176
177 AuraScript* GetAuraScript() const override
178 {
179 return new spell_pal_ardent_defender_AuraScript();
180 }
181};

418// 64205 - Divine Sacrifice
419class spell_pal_divine_sacrifice : public SpellScriptLoader
420{
421 public:
422 spell_pal_divine_sacrifice() : SpellScriptLoader("spell_pal_divine_sacrifice") { }
423
424 class spell_pal_divine_sacrifice_AuraScript : public AuraScript
425 {
426 PrepareAuraScript(spell_pal_divine_sacrifice_AuraScript);
427
428 uint32 groupSize, minHpPct;
429 int32 remainingAmount;
430
431 bool Load() override
432 {
433 if (Unit* caster = GetCaster())
434 {
435 if (caster->GetTypeId() == TYPEID_PLAYER)
436 {
437 if (caster->ToPlayer()->GetGroup())
438 groupSize = caster->ToPlayer()->GetGroup()->GetMembersCount();
439 else
440 groupSize = 1;
441 }
442 else
443 return false;
444
445 remainingAmount = (caster->CountPctFromMaxHealth(GetSpellInfo()->Effects[EFFECT_2].CalcValue(caster)) * groupSize);
446 minHpPct = GetSpellInfo()->Effects[EFFECT_1].CalcValue(caster);
447 return true;
448 }
449 return false;
450 }
451
452 void Split(AuraEffect* /*aurEff*/, DamageInfo & /*dmgInfo*/, uint32 & splitAmount)
453 {
454 remainingAmount -= splitAmount;
455 // break when absorbed everything it could, or if the casters hp drops below 20%
456 if (Unit* caster = GetCaster())
457 if (remainingAmount <= 0 || (caster->GetHealthPct() < minHpPct))
458 caster->RemoveAura(SPELL_PALADIN_DIVINE_SACRIFICE);
459 }
460
461 void Register() override
462 {
463 OnEffectSplit += AuraEffectSplitFn(spell_pal_divine_sacrifice_AuraScript::Split, EFFECT_0);
464 }
465 };
466
467 AuraScript* GetAuraScript() const override
468 {
469 return new spell_pal_divine_sacrifice_AuraScript();
470 }

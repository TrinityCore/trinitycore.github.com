113 PrepareAuraScript(spell_rog_cheat_death_AuraScript);
114
115 public:
116 spell_rog_cheat_death_AuraScript()
117 {
118 absorbChance = 0;
119 }
120
121 private:

123
124 bool Validate(SpellInfo const* /*spellInfo*/) override
125 {
126 if (!sSpellMgr->GetSpellInfo(SPELL_ROGUE_CHEAT_DEATH_COOLDOWN))
127 return false;
128 return true;
129 }
130
131 bool Load() override
132 {
133 absorbChance = GetSpellInfo()->Effects[EFFECT_0].CalcValue();
134 return GetUnitOwner()->GetTypeId() == TYPEID_PLAYER;
135 }
136
137 void CalculateAmount(AuraEffect const* /*aurEff*/, int32 & amount, bool & /*canBeRecalculated*/)
138 {
139 // Set absorbtion amount to unlimited
140 amount = -1;
141 }
142
143 void Absorb(AuraEffect* /*aurEff*/, DamageInfo & dmgInfo, uint32 & absorbAmount)
144 {
145 Player* target = GetTarget()->ToPlayer();
146 if (dmgInfo.GetDamage() < target->GetHealth() || target->GetSpellHistory()->HasCooldown(SPELL_ROGUE_CHEAT_DEATH_COOLDOWN) || !roll_chance_i(absorbChance))
147 return;
148
149 target->CastSpell(target, SPELL_ROGUE_CHEAT_DEATH_COOLDOWN, true);
150 target->GetSpellHistory()->AddCooldown(SPELL_ROGUE_CHEAT_DEATH_COOLDOWN, 0, std::chrono::minutes(1));
151
152 uint32 health10 = target->CountPctFromMaxHealth(10);
153
154 // hp > 10% - absorb hp till 10%
155 if (target->GetHealth() > health10)
156 absorbAmount = dmgInfo.GetDamage() - target->GetHealth() + health10;
157 // hp lower than 10% - absorb everything
158 else
159 absorbAmount = dmgInfo.GetDamage();
160 }
161
162 void Register() override
163 {
164 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_rog_cheat_death_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);
165 OnEffectAbsorb += AuraEffectAbsorbFn(spell_rog_cheat_death_AuraScript::Absorb, EFFECT_0);
166 }
167 };
168
169 AuraScript* GetAuraScript() const override
170 {
171 return new spell_rog_cheat_death_AuraScript();
172 }

175// -2818 - Deadly Poison
176class spell_rog_deadly_poison : public SpellScriptLoader
177{
178 public:
179 spell_rog_deadly_poison() : SpellScriptLoader("spell_rog_deadly_poison") { }
180
181 class spell_rog_deadly_poison_SpellScript : public SpellScript
182 {
183 PrepareSpellScript(spell_rog_deadly_poison_SpellScript);
184
185 public:
186 spell_rog_deadly_poison_SpellScript()
187 {
188 _stackAmount = 0;
189 }
190
191 private:
192 bool Load() override
193 {
194 // at this point CastItem must already be initialized
195 return GetCaster()->GetTypeId() == TYPEID_PLAYER && GetCastItem();
196 }
197
198 void HandleBeforeHit()
199 {
200 if (Unit* target = GetHitUnit())
201 // Deadly Poison
202 if (AuraEffect const* aurEff = target->GetAuraEffect(SPELL_AURA_PERIODIC_DAMAGE, SPELLFAMILY_ROGUE, 0x10000, 0x80000, 0, GetCaster()->GetGUID()))
203 _stackAmount = aurEff->GetBase()->GetStackAmount();
204 }
205
206 void HandleAfterHit()
207 {
208 if (_stackAmount < 5)
209 return;
210
211 Player* player = GetCaster()->ToPlayer();
212
213 if (Unit* target = GetHitUnit())
214 {
215
216 Item* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_MAINHAND);
217
218 if (item == GetCastItem())
219 item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_OFFHAND);
220
221 if (!item)
222 return;
223
224 // item combat enchantments
225 for (uint8 slot = 0; slot < MAX_ENCHANTMENT_SLOT; ++slot)
226 {
227 SpellItemEnchantmentEntry const* enchant = sSpellItemEnchantmentStore.LookupEntry(item->GetEnchantmentId(EnchantmentSlot(slot)));
228 if (!enchant)
229 continue;
230
231 for (uint8 s = 0; s < 3; ++s)
232 {
233 if (enchant->type[s] != ITEM_ENCHANTMENT_TYPE_COMBAT_SPELL)
234 continue;
235
236 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(enchant->spellid[s]);
237 if (!spellInfo)
238 {
239 TC_LOG_ERROR("spells", "Player::CastItemCombatSpell Enchant %i, player (Name: %s, GUID: %u) cast unknown spell %i", enchant->ID, player->GetName().c_str(), player->GetGUID().GetCounter(), enchant->spellid[s]);
240 continue;
241 }
242
243 // Proc only rogue poisons
244 if (spellInfo->SpellFamilyName != SPELLFAMILY_ROGUE || spellInfo->Dispel != DISPEL_POISON)
245 continue;
246
247 // Do not reproc deadly
248 if (spellInfo->SpellFamilyFlags.IsEqual(0x10000, 0x80000, 0))
249 continue;
250
251 if (spellInfo->IsPositive())
252 player->CastSpell(player, enchant->spellid[s], true, item);
253 else
254 player->CastSpell(target, enchant->spellid[s], true, item);
255 }
256 }
257 }
258 }
259
260 void Register() override
261 {
262 BeforeHit += SpellHitFn(spell_rog_deadly_poison_SpellScript::HandleBeforeHit);
263 AfterHit += SpellHitFn(spell_rog_deadly_poison_SpellScript::HandleAfterHit);
264 }
265
266 uint8 _stackAmount;
267 };
268
269 SpellScript* GetSpellScript() const override
270 {
271 return new spell_rog_deadly_poison_SpellScript();
272 }

543 {

545
546 float const attackpowerPerCombo[6] =
547 {
548 0.0f,
549 0.015f, // 1 point: ${($m1 + $b1*1 + 0.015 * $AP) * 4} damage over 8 secs

551 0.03f, // 3 points: ${($m1 + $b1*3 + 0.03 * $AP) * 6} damage over 12 secs
552 0.03428571f, // 4 points: ${($m1 + $b1*4 + 0.03428571 * $AP) * 7} damage over 14 secs
553 0.0375f // 5 points: ${($m1 + $b1*5 + 0.0375 * $AP) * 8} damage over 16 secs
554 };
555
556 uint8 cp = caster->ToPlayer()->GetComboPoints();
557 if (cp > 5)
558 cp = 5;
559
560 amount += int32(caster->GetTotalAttackPowerValue(BASE_ATTACK) * attackpowerPerCombo[cp]);
561 }
562 }
563
564 void Register() override
565 {
566 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_rog_rupture_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE);

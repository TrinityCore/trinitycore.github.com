117// 6201 - Create Healthstone (and ranks)
118class spell_warl_create_healthstone : public SpellScriptLoader
119{
120 public:
121 spell_warl_create_healthstone() : SpellScriptLoader("spell_warl_create_healthstone") { }
122
123 class spell_warl_create_healthstone_SpellScript : public SpellScript
124 {
125 PrepareSpellScript(spell_warl_create_healthstone_SpellScript);
126
127 static uint32 const iTypes[8][3];
128
129 bool Validate(SpellInfo const* /*spellInfo*/) override
130 {
131 if (!sSpellMgr->GetSpellInfo(SPELL_WARLOCK_IMPROVED_HEALTHSTONE_R1) || !sSpellMgr->GetSpellInfo(SPELL_WARLOCK_IMPROVED_HEALTHSTONE_R2))
132 return false;
133 return true;
134 }
135
136 SpellCastResult CheckCast()
137 {
138 if (Player* caster = GetCaster()->ToPlayer())
139 {
140 uint8 spellRank = GetSpellInfo()->GetRank();
141 ItemPosCountVec dest;
142 InventoryResult msg = caster->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, iTypes[spellRank - 1][0], 1, NULL);
143 if (msg != EQUIP_ERR_OK)
144 return SPELL_FAILED_TOO_MANY_OF_ITEM;
145 }
146 return SPELL_CAST_OK;
147 }
148
149 void HandleScriptEffect(SpellEffIndex effIndex)
150 {
151 if (Unit* unitTarget = GetHitUnit())
152 {
153 uint32 rank = 0;
154 // Improved Healthstone
155 if (AuraEffect const* aurEff = unitTarget->GetDummyAuraEffect(SPELLFAMILY_WARLOCK, 284, 0))
156 {
157 switch (aurEff->GetId())
158 {
159 case SPELL_WARLOCK_IMPROVED_HEALTHSTONE_R1:
160 rank = 1;
161 break;
162 case SPELL_WARLOCK_IMPROVED_HEALTHSTONE_R2:
163 rank = 2;
164 break;
165 default:
166 TC_LOG_ERROR("spells", "Unknown rank of Improved Healthstone id: %d", aurEff->GetId());
167 break;
168 }
169 }
170 uint8 spellRank = GetSpellInfo()->GetRank();
171 if (spellRank > 0 && spellRank <= 8)
172 CreateItem(effIndex, iTypes[spellRank - 1][rank]);
173 }
174 }
175
176 void Register() override
177 {
178 OnEffectHitTarget += SpellEffectFn(spell_warl_create_healthstone_SpellScript::HandleScriptEffect, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
179 OnCheckCast += SpellCheckCastFn(spell_warl_create_healthstone_SpellScript::CheckCast);
180 }
181 };
182
183 SpellScript* GetSpellScript() const override
184 {
185 return new spell_warl_create_healthstone_SpellScript();
186 }
187};

247// 48018 - Demonic Circle: Summon
248class spell_warl_demonic_circle_summon : public SpellScriptLoader
249{
250 public:
251 spell_warl_demonic_circle_summon() : SpellScriptLoader("spell_warl_demonic_circle_summon") { }
252
253 class spell_warl_demonic_circle_summon_AuraScript : public AuraScript
254 {
255 PrepareAuraScript(spell_warl_demonic_circle_summon_AuraScript);
256
257 void HandleRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes mode)
258 {
259 // If effect is removed by expire remove the summoned demonic circle too.
260 if (!(mode & AURA_EFFECT_HANDLE_REAPPLY))
261 GetTarget()->RemoveGameObject(GetId(), true);
262
263 GetTarget()->RemoveAura(SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST);
264 }
265
266 void HandleDummyTick(AuraEffect const* /*aurEff*/)
267 {
268 if (GameObject* circle = GetTarget()->GetGameObject(GetId()))
269 {
270 // Here we check if player is in demonic circle teleport range, if so add
271 // WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST; allowing him to cast the WARLOCK_DEMONIC_CIRCLE_TELEPORT.
272 // If not in range remove the WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST.
273
274 SpellInfo const* spellInfo = sSpellMgr->EnsureSpellInfo(SPELL_WARLOCK_DEMONIC_CIRCLE_TELEPORT);
275
276 if (GetTarget()->IsWithinDist(circle, spellInfo->GetMaxRange(true)))
277 {
278 if (!GetTarget()->HasAura(SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST))
279 GetTarget()->CastSpell(GetTarget(), SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST, true);
280 }
281 else
282 GetTarget()->RemoveAura(SPELL_WARLOCK_DEMONIC_CIRCLE_ALLOW_CAST);
283 }
284 }
285
286 void Register() override
287 {
288 OnEffectRemove += AuraEffectApplyFn(spell_warl_demonic_circle_summon_AuraScript::HandleRemove, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY, AURA_EFFECT_HANDLE_REAL_OR_REAPPLY_MASK);
289 OnEffectPeriodic += AuraEffectPeriodicFn(spell_warl_demonic_circle_summon_AuraScript::HandleDummyTick, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
290 }
291 };
292
293 AuraScript* GetAuraScript() const override
294 {
295 return new spell_warl_demonic_circle_summon_AuraScript();
296 }

342
343 bool Validate(SpellInfo const* /*spellInfo*/) override
344 {
345 if (!sSpellMgr->GetSpellInfo(SPELL_WARLOCK_DEMONIC_EMPOWERMENT_SUCCUBUS) || !sSpellMgr->GetSpellInfo(SPELL_WARLOCK_DEMONIC_EMPOWERMENT_VOIDWALKER) || !sSpellMgr->GetSpellInfo(SPELL_WARLOCK_DEMONIC_EMPOWERMENT_FELGUARD) || !sSpellMgr->GetSpellInfo(SPELL_WARLOCK_DEMONIC_EMPOWERMENT_FELHUNTER) || !sSpellMgr->GetSpellInfo(SPELL_WARLOCK_DEMONIC_EMPOWERMENT_IMP))

352 if (Creature* targetCreature = GetHitCreature())
353 {
354 if (targetCreature->IsPet())
355 {
356 CreatureTemplate const* ci = sObjectMgr->GetCreatureTemplate(targetCreature->GetEntry());
357 ASSERT(ci);
358 switch (ci->family)
359 {
360 case CREATURE_FAMILY_SUCCUBUS:
361 targetCreature->CastSpell(targetCreature, SPELL_WARLOCK_DEMONIC_EMPOWERMENT_SUCCUBUS, true);
362 break;
363 case CREATURE_FAMILY_VOIDWALKER:
364 {
365 SpellInfo const* spellInfo = sSpellMgr->EnsureSpellInfo(SPELL_WARLOCK_DEMONIC_EMPOWERMENT_VOIDWALKER);
366 int32 hp = int32(targetCreature->CountPctFromMaxHealth(GetCaster()->CalculateSpellDamage(targetCreature, spellInfo, 0)));
367 targetCreature->CastCustomSpell(targetCreature, SPELL_WARLOCK_DEMONIC_EMPOWERMENT_VOIDWALKER, &hp, NULL, NULL, true);
368 //unitTarget->CastSpell(unitTarget, 54441, true);
369 break;
370 }
371 case CREATURE_FAMILY_FELGUARD:
372 targetCreature->CastSpell(targetCreature, SPELL_WARLOCK_DEMONIC_EMPOWERMENT_FELGUARD, true);
373 break;
374 case CREATURE_FAMILY_FELHUNTER:
375 targetCreature->CastSpell(targetCreature, SPELL_WARLOCK_DEMONIC_EMPOWERMENT_FELHUNTER, true);
376 break;
377 case CREATURE_FAMILY_IMP:
378 targetCreature->CastSpell(targetCreature, SPELL_WARLOCK_DEMONIC_EMPOWERMENT_IMP, true);
379 break;
380 }
381 }
382 }

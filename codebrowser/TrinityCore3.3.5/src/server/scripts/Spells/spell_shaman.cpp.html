75 {
76 PrepareSpellScript(spell_sha_ancestral_awakening_proc_SpellScript);
77
78 bool Validate(SpellInfo const* /*spellInfo*/) override
79 {
80 if (!sSpellMgr->GetSpellInfo(SPELL_SHAMAN_ANCESTRAL_AWAKENING_PROC))
81 return false;
82 return true;
83 }
84
85 void FilterTargets(std::list<WorldObject*>& targets)
86 {
87 if (targets.size() < 2)
88 return;
89
90 targets.sort(Trinity::HealthPctOrderPred());

93 targets.clear();
94 targets.push_back(target);
95 }

102 }
103
104 void Register() override
105 {
106 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_sha_ancestral_awakening_proc_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_CASTER_AREA_RAID);

298// -974 - Earth Shield
299class spell_sha_earth_shield : public SpellScriptLoader
300{
301 public:
302 spell_sha_earth_shield() : SpellScriptLoader("spell_sha_earth_shield") { }
303
304 class spell_sha_earth_shield_AuraScript : public AuraScript
305 {
306 PrepareAuraScript(spell_sha_earth_shield_AuraScript);
307
308 bool Validate(SpellInfo const* /*spellInfo*/) override
309 {
310 if (!sSpellMgr->GetSpellInfo(SPELL_SHAMAN_EARTH_SHIELD_HEAL))
311 return false;
312 if (!sSpellMgr->GetSpellInfo(SPELL_SHAMAN_GLYPH_OF_EARTH_SHIELD))
313 return false;
314 return true;
315 }
316
317 void CalculateAmount(AuraEffect const* /*aurEff*/, int32& amount, bool & /*canBeRecalculated*/)
318 {
319 if (Unit* caster = GetCaster())
320 {
321 amount = caster->SpellHealingBonusDone(GetUnitOwner(), GetSpellInfo(), amount, HEAL);
322 amount = GetUnitOwner()->SpellHealingBonusTaken(caster, GetSpellInfo(), amount, HEAL);
323
324 //! WORKAROUND
325 // If target is affected by healing reduction, modifier is guaranteed to be negative
326 // value (e.g. -50). To revert the effect, multiply amount with reciprocal of relative value:
327 // (100 / ((-1) * modifier)) * 100 = (-1) * 100 * 100 / modifier = -10000 / modifier
328 if (int32 modifier = GetUnitOwner()->GetMaxNegativeAuraModifier(SPELL_AURA_MOD_HEALING_PCT))
329 ApplyPct(amount, -10000.0f / float(modifier));
330
331 // Glyph of Earth Shield
332 //! WORKAROUND
333 //! this glyph is a proc
334 if (AuraEffect* glyph = caster->GetAuraEffect(SPELL_SHAMAN_GLYPH_OF_EARTH_SHIELD, EFFECT_0))
335 AddPct(amount, glyph->GetAmount());
336 }
337 }
338
339 bool CheckProc(ProcEventInfo& /*eventInfo*/)
340 {
341 //! HACK due to currenct proc system implementation
342 if (Player* player = GetTarget()->ToPlayer())
343 if (player->GetSpellHistory()->HasCooldown(SPELL_SHAMAN_EARTH_SHIELD_HEAL))
344 return false;
345 return true;
346 }
347
348 void HandleProc(AuraEffect const* aurEff, ProcEventInfo& /*eventInfo*/)
349 {
350 PreventDefaultAction();
351
352 GetTarget()->CastCustomSpell(SPELL_SHAMAN_EARTH_SHIELD_HEAL, SPELLVALUE_BASE_POINT0, aurEff->GetAmount(), GetTarget(), true, NULL, aurEff, GetCasterGUID());
353
354 /// @hack: due to currenct proc system implementation
355 if (Player* player = GetTarget()->ToPlayer())
356 player->GetSpellHistory()->AddCooldown(SPELL_SHAMAN_EARTH_SHIELD_HEAL, 0, std::chrono::seconds(3));
357 }
358
359 void Register() override
360 {
361 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_sha_earth_shield_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_DUMMY);
362 DoCheckProc += AuraCheckProcFn(spell_sha_earth_shield_AuraScript::CheckProc);
363 OnEffectProc += AuraEffectProcFn(spell_sha_earth_shield_AuraScript::HandleProc, EFFECT_0, SPELL_AURA_DUMMY);
364 }
365 };
366
367 AuraScript* GetAuraScript() const override
368 {
369 return new spell_sha_earth_shield_AuraScript();
370 }

583 {
584 PrepareSpellScript(spell_sha_healing_stream_totem_SpellScript);
585
586 bool Validate(SpellInfo const* /*spellInfo*/) override
587 {
588 if (!sSpellMgr->GetSpellInfo(SPELL_SHAMAN_GLYPH_OF_HEALING_STREAM_TOTEM) || !sSpellMgr->GetSpellInfo(SPELL_SHAMAN_TOTEM_HEALING_STREAM_HEAL))

592
593 void HandleDummy(SpellEffIndex /*effIndex*/)
594 {
595 int32 damage = GetEffectValue();
596 SpellInfo const* triggeringSpell = GetTriggeringSpell();
597 if (Unit* target = GetHitUnit())
598 if (Unit* caster = GetCaster())
599 {
600 if (Unit* owner = caster->GetOwner())
601 {
602 if (triggeringSpell)
603 damage = int32(owner->SpellHealingBonusDone(target, triggeringSpell, damage, HEAL));
604
605 // Restorative Totems
606 if (AuraEffect* dummy = owner->GetAuraEffect(SPELL_AURA_DUMMY, SPELLFAMILY_SHAMAN, SHAMAN_ICON_ID_RESTORATIVE_TOTEMS, 1))
607 AddPct(damage, dummy->GetAmount());
608
609 // Glyph of Healing Stream Totem
610 if (AuraEffect const* aurEff = owner->GetAuraEffect(SPELL_SHAMAN_GLYPH_OF_HEALING_STREAM_TOTEM, EFFECT_0))
611 AddPct(damage, aurEff->GetAmount());
612

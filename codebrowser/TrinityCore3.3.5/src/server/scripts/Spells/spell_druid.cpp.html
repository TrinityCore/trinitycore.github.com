54// 1178 - Bear Form (Passive)
55// 9635 - Dire Bear Form (Passive)
56class spell_dru_bear_form_passive : public SpellScriptLoader
57{
58 public:
59 spell_dru_bear_form_passive() : SpellScriptLoader("spell_dru_bear_form_passive") { }
60
61 class spell_dru_bear_form_passive_AuraScript : public AuraScript
62 {
63 PrepareAuraScript(spell_dru_bear_form_passive_AuraScript);
64
65 bool Validate(SpellInfo const* /*spellInfo*/) override
66 {
67 if (!sSpellMgr->GetSpellInfo(SPELL_DRUID_ENRAGE)
68 || !sSpellMgr->GetSpellInfo(SPELL_DRUID_ITEM_T10_FERAL_4P_BONUS))
69 return false;
70 return true;
71 }
72
73 void CalculateAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
74 {
75 if (!GetUnitOwner()->HasAura(SPELL_DRUID_ENRAGE) || GetUnitOwner()->HasAura(SPELL_DRUID_ITEM_T10_FERAL_4P_BONUS))
76 return;
77
78 int32 mod = 0;
79 switch (GetId())
80 {
81 case SPELL_DRUID_BEAR_FORM_PASSIVE:
82 mod = -27;
83 break;
84 case SPELL_DRUID_DIRE_BEAR_FORM_PASSIVE:
85 mod = -16;
86 break;
87 default:
88 return;
89 }
90 amount += mod;
91 }
92
93 void Register() override
94 {
95 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_dru_bear_form_passive_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_MOD_BASE_RESISTANCE_PCT);
96 }
97 };
98
99 AuraScript* GetAuraScript() const override
100 {
101 return new spell_dru_bear_form_passive_AuraScript();

134// 5229 - Enrage
135class spell_dru_enrage : public SpellScriptLoader
136{
137 public:
138 spell_dru_enrage() : SpellScriptLoader("spell_dru_enrage") { }
139
140 class spell_dru_enrage_AuraScript : public AuraScript
141 {
142 PrepareAuraScript(spell_dru_enrage_AuraScript);
143
144 bool Validate(SpellInfo const* /*spellInfo*/) override
145 {
146 if (!sSpellMgr->GetSpellInfo(SPELL_DRUID_KING_OF_THE_JUNGLE)
147 || !sSpellMgr->GetSpellInfo(SPELL_DRUID_ENRAGE_MOD_DAMAGE)
148 || !sSpellMgr->GetSpellInfo(SPELL_DRUID_ENRAGED_DEFENSE)
149 || !sSpellMgr->GetSpellInfo(SPELL_DRUID_ITEM_T10_FERAL_4P_BONUS))
150 return false;
151 return true;
152 }
153
154 void RecalculateBaseArmor()
155 {
156 Unit::AuraEffectList const& auras = GetTarget()->GetAuraEffectsByType(SPELL_AURA_MOD_BASE_RESISTANCE_PCT);
157 for (Unit::AuraEffectList::const_iterator i = auras.begin(); i != auras.end(); ++i)
158 {
159 SpellInfo const* spellInfo = (*i)->GetSpellInfo();
160 // Dire- / Bear Form (Passive)
161 if (spellInfo->SpellFamilyName == SPELLFAMILY_DRUID && spellInfo->SpellFamilyFlags.HasFlag(0x0, 0x0, 0x2))
162 (*i)->RecalculateAmount();
163 }
164 }
165
166 void HandleApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
167 {
168 Unit* target = GetTarget();
169 if (AuraEffect const* aurEff = target->GetAuraEffectOfRankedSpell(SPELL_DRUID_KING_OF_THE_JUNGLE, EFFECT_0))
170 target->CastCustomSpell(SPELL_DRUID_ENRAGE_MOD_DAMAGE, SPELLVALUE_BASE_POINT0, aurEff->GetAmount(), target, true);
171
172 // Item - Druid T10 Feral 4P Bonus
173 if (target->HasAura(SPELL_DRUID_ITEM_T10_FERAL_4P_BONUS))
174 target->CastSpell(target, SPELL_DRUID_ENRAGED_DEFENSE, true);
175
176 RecalculateBaseArmor();
177 }
178
179 void HandleRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
180 {
181 GetTarget()->RemoveAurasDueToSpell(SPELL_DRUID_ENRAGE_MOD_DAMAGE);
182 GetTarget()->RemoveAurasDueToSpell(SPELL_DRUID_ENRAGED_DEFENSE);
183
184 RecalculateBaseArmor();
185 }
186
187 void Register() override
188 {
189 AfterEffectApply += AuraEffectApplyFn(spell_dru_enrage_AuraScript::HandleApply, EFFECT_0, SPELL_AURA_PERIODIC_ENERGIZE, AURA_EFFECT_HANDLE_REAL);
190 AfterEffectRemove += AuraEffectRemoveFn(spell_dru_enrage_AuraScript::HandleRemove, EFFECT_0, SPELL_AURA_PERIODIC_ENERGIZE, AURA_EFFECT_HANDLE_REAL);
191 }
192 };
193
194 AuraScript* GetAuraScript() const override
195 {
196 return new spell_dru_enrage_AuraScript();
197 }

200// 54846 - Glyph of Starfire
201class spell_dru_glyph_of_starfire : public SpellScriptLoader
202{
203 public:
204 spell_dru_glyph_of_starfire() : SpellScriptLoader("spell_dru_glyph_of_starfire") { }
205
206 class spell_dru_glyph_of_starfire_SpellScript : public SpellScript
207 {
208 PrepareSpellScript(spell_dru_glyph_of_starfire_SpellScript);
209
210 bool Validate(SpellInfo const* /*spellInfo*/) override
211 {
212 if (!sSpellMgr->GetSpellInfo(SPELL_DRUID_INCREASED_MOONFIRE_DURATION) || !sSpellMgr->GetSpellInfo(SPELL_DRUID_NATURES_SPLENDOR))
213 return false;
214 return true;
215 }
216
217 void HandleScriptEffect(SpellEffIndex /*effIndex*/)
218 {
219 Unit* caster = GetCaster();
220 if (Unit* unitTarget = GetHitUnit())
221 if (AuraEffect const* aurEff = unitTarget->GetAuraEffect(SPELL_AURA_PERIODIC_DAMAGE, SPELLFAMILY_DRUID, 0x00000002, 0, 0, caster->GetGUID()))
222 {
223 Aura* aura = aurEff->GetBase();
224
225 uint32 countMin = aura->GetMaxDuration();
226 uint32 countMax = aura->GetSpellInfo()->GetMaxDuration() + 9000;
227 if (caster->HasAura(SPELL_DRUID_INCREASED_MOONFIRE_DURATION))
228 countMax += 3000;
229 if (caster->HasAura(SPELL_DRUID_NATURES_SPLENDOR))
230 countMax += 3000;
231
232 if (countMin < countMax)
233 {
234 aura->SetDuration(uint32(aura->GetDuration() + 3000));
235 aura->SetMaxDuration(countMin + 3000);
236 }
237 }
238 }
239
240 void Register() override
241 {
242 OnEffectHitTarget += SpellEffectFn(spell_dru_glyph_of_starfire_SpellScript::HandleScriptEffect, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
243 }
244 };
245
246 SpellScript* GetSpellScript() const override
247 {
248 return new spell_dru_glyph_of_starfire_SpellScript();
249 }

347// -33763 - Lifebloom
348class spell_dru_lifebloom : public SpellScriptLoader
349{
350 public:
351 spell_dru_lifebloom() : SpellScriptLoader("spell_dru_lifebloom") { }
352
353 class spell_dru_lifebloom_AuraScript : public AuraScript
354 {
355 PrepareAuraScript(spell_dru_lifebloom_AuraScript);
356
357 bool Validate(SpellInfo const* /*spell*/) override
358 {
359 if (!sSpellMgr->GetSpellInfo(SPELL_DRUID_LIFEBLOOM_FINAL_HEAL))
360 return false;
361 if (!sSpellMgr->GetSpellInfo(SPELL_DRUID_LIFEBLOOM_ENERGIZE))
362 return false;
363 return true;
364 }
365
366 void AfterRemove(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
367 {
368 // Final heal only on duration end
369 if (GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_EXPIRE)
370 return;
371
372 // final heal
373 int32 stack = GetStackAmount();
374 int32 healAmount = aurEff->GetAmount();
375 if (Unit* caster = GetCaster())
376 {
377 healAmount = caster->SpellHealingBonusDone(GetTarget(), GetSpellInfo(), healAmount, HEAL, stack);
378 healAmount = GetTarget()->SpellHealingBonusTaken(caster, GetSpellInfo(), healAmount, HEAL, stack);
379
380 GetTarget()->CastCustomSpell(GetTarget(), SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, aurEff, GetCasterGUID());
381
382 // restore mana
383 int32 returnMana = CalculatePct(caster->GetCreateMana(), GetSpellInfo()->ManaCostPercentage) * stack / 2;
384 caster->CastCustomSpell(caster, SPELL_DRUID_LIFEBLOOM_ENERGIZE, &returnMana, NULL, NULL, true, NULL, aurEff, GetCasterGUID());
385 return;
386 }
387
388 GetTarget()->CastCustomSpell(GetTarget(), SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, aurEff, GetCasterGUID());
389 }
390
391 void HandleDispel(DispelInfo* dispelInfo)
392 {
393 if (Unit* target = GetUnitOwner())
394 {
395 if (AuraEffect const* aurEff = GetEffect(EFFECT_1))
396 {
397 // final heal
398 int32 healAmount = aurEff->GetAmount();
399 if (Unit* caster = GetCaster())
400 {
401 healAmount = caster->SpellHealingBonusDone(target, GetSpellInfo(), healAmount, HEAL, dispelInfo->GetRemovedCharges());
402 healAmount = target->SpellHealingBonusTaken(caster, GetSpellInfo(), healAmount, HEAL, dispelInfo->GetRemovedCharges());
403 target->CastCustomSpell(target, SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, NULL, GetCasterGUID());
404
405 // restore mana
406 int32 returnMana = CalculatePct(caster->GetCreateMana(), GetSpellInfo()->ManaCostPercentage) * dispelInfo->GetRemovedCharges() / 2;
407 caster->CastCustomSpell(caster, SPELL_DRUID_LIFEBLOOM_ENERGIZE, &returnMana, NULL, NULL, true, NULL, NULL, GetCasterGUID());
408 return;
409 }
410
411 target->CastCustomSpell(target, SPELL_DRUID_LIFEBLOOM_FINAL_HEAL, &healAmount, NULL, NULL, true, NULL, NULL, GetCasterGUID());
412 }
413 }
414 }
415
416 void Register() override
417 {
418 AfterEffectRemove += AuraEffectRemoveFn(spell_dru_lifebloom_AuraScript::AfterRemove, EFFECT_1, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
419 AfterDispel += AuraDispelFn(spell_dru_lifebloom_AuraScript::HandleDispel);
420 }
421 };
422
423 AuraScript* GetAuraScript() const override
424 {
425 return new spell_dru_lifebloom_AuraScript();
426 }

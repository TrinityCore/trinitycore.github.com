74// 23881 - Bloodthirst
75class spell_warr_bloodthirst : public SpellScriptLoader
76{
77 public:
78 spell_warr_bloodthirst() : SpellScriptLoader("spell_warr_bloodthirst") { }
79
80 class spell_warr_bloodthirst_SpellScript : public SpellScript
81 {
82 PrepareSpellScript(spell_warr_bloodthirst_SpellScript);
83
84 void HandleDamage(SpellEffIndex /*effIndex*/)
85 {
86 int32 damage = GetEffectValue();
87 ApplyPct(damage, GetCaster()->GetTotalAttackPowerValue(BASE_ATTACK));
88
89 if (Unit* target = GetHitUnit())
90 {
91 damage = GetCaster()->SpellDamageBonusDone(target, GetSpellInfo(), uint32(damage), SPELL_DIRECT_DAMAGE);
92 damage = target->SpellDamageBonusTaken(GetCaster(), GetSpellInfo(), uint32(damage), SPELL_DIRECT_DAMAGE);
93 }
94 SetHitDamage(damage);
95 }
96
97 void HandleDummy(SpellEffIndex /*effIndex*/)
98 {
99 int32 damage = GetEffectValue();
100 GetCaster()->CastCustomSpell(GetCaster(), SPELL_WARRIOR_BLOODTHIRST, &damage, NULL, NULL, true, NULL);
101 }
102
103 void Register() override
104 {
105 OnEffectHitTarget += SpellEffectFn(spell_warr_bloodthirst_SpellScript::HandleDamage, EFFECT_0, SPELL_EFFECT_SCHOOL_DAMAGE);
106 OnEffectHit += SpellEffectFn(spell_warr_bloodthirst_SpellScript::HandleDummy, EFFECT_1, SPELL_EFFECT_DUMMY);
107 }
108 };
109
110 SpellScript* GetSpellScript() const override
111 {
112 return new spell_warr_bloodthirst_SpellScript();

249// -12162 - Deep Wounds
250class spell_warr_deep_wounds : public SpellScriptLoader
251{
252 public:
253 spell_warr_deep_wounds() : SpellScriptLoader("spell_warr_deep_wounds") { }
254
255 class spell_warr_deep_wounds_SpellScript : public SpellScript
256 {
257 PrepareSpellScript(spell_warr_deep_wounds_SpellScript);
258
259 bool Validate(SpellInfo const* /*spellInfo*/) override
260 {
261 if (!sSpellMgr->GetSpellInfo(SPELL_WARRIOR_DEEP_WOUNDS_RANK_1) || !sSpellMgr->GetSpellInfo(SPELL_WARRIOR_DEEP_WOUNDS_RANK_2) || !sSpellMgr->GetSpellInfo(SPELL_WARRIOR_DEEP_WOUNDS_RANK_3))
262 return false;
263 return true;
264 }
265
266 void HandleDummy(SpellEffIndex /*effIndex*/)
267 {
268 int32 damage = GetEffectValue();
269 Unit* caster = GetCaster();
270 if (Unit* target = GetHitUnit())
271 {
272 ApplyPct(damage, 16 * GetSpellInfo()->GetRank());
273
274 SpellInfo const* spellInfo = sSpellMgr->EnsureSpellInfo(SPELL_WARRIOR_DEEP_WOUNDS_PERIODIC);
275 uint32 ticks = uint32(spellInfo->GetDuration()) / spellInfo->Effects[EFFECT_0].Amplitude;
276
277 // Add remaining ticks to damage done
278 if (AuraEffect const* aurEff = target->GetAuraEffect(SPELL_WARRIOR_DEEP_WOUNDS_PERIODIC, EFFECT_0, caster->GetGUID()))
279 damage += (aurEff->GetAmount() + aurEff->GetBonusAmount()) * aurEff->GetDonePct() * int32(ticks - aurEff->GetTickNumber());
280
281 damage /= int32(ticks);
282
283 caster->CastCustomSpell(target, SPELL_WARRIOR_DEEP_WOUNDS_PERIODIC, &damage, NULL, NULL, true);
284 }
285 }
286
287 void Register() override
288 {
289 OnEffectHitTarget += SpellEffectFn(spell_warr_deep_wounds_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
290 }
291 };

299// -5308 - Execute
300class spell_warr_execute : public SpellScriptLoader
301{
302 public:
303 spell_warr_execute() : SpellScriptLoader("spell_warr_execute") { }
304
305 class spell_warr_execute_SpellScript : public SpellScript
306 {
307 PrepareSpellScript(spell_warr_execute_SpellScript);
308
309 bool Validate(SpellInfo const* /*spellInfo*/) override
310 {
311 if (!sSpellMgr->GetSpellInfo(SPELL_WARRIOR_EXECUTE) || !sSpellMgr->GetSpellInfo(SPELL_WARRIOR_GLYPH_OF_EXECUTION))
312 return false;
313 return true;
314 }
315
316 void HandleEffect(SpellEffIndex effIndex)
317 {
318 Unit* caster = GetCaster();
319 if (Unit* target = GetHitUnit())
320 {
321 SpellInfo const* spellInfo = GetSpellInfo();
322 int32 rageUsed = std::min<int32>(300 - spellInfo->CalcPowerCost(caster, SpellSchoolMask(spellInfo->SchoolMask)), caster->GetPower(POWER_RAGE));
323 int32 newRage = std::max<int32>(0, caster->GetPower(POWER_RAGE) - rageUsed);
324
325 // Sudden Death rage save
326 if (AuraEffect* aurEff = caster->GetAuraEffect(SPELL_AURA_PROC_TRIGGER_SPELL, SPELLFAMILY_GENERIC, WARRIOR_ICON_ID_SUDDEN_DEATH, EFFECT_0))
327 {
328 int32 ragesave = aurEff->GetSpellInfo()->Effects[EFFECT_1].CalcValue() * 10;
329 newRage = std::max(newRage, ragesave);
330 }
331
332 caster->SetPower(POWER_RAGE, uint32(newRage));
333 // Glyph of Execution bonus
334 if (AuraEffect* aurEff = caster->GetAuraEffect(SPELL_WARRIOR_GLYPH_OF_EXECUTION, EFFECT_0))
335 rageUsed += aurEff->GetAmount() * 10;
336
337
338 int32 bp = GetEffectValue() + int32(rageUsed * spellInfo->Effects[effIndex].DamageMultiplier + caster->GetTotalAttackPowerValue(BASE_ATTACK) * 0.2f);
339 caster->CastCustomSpell(target, SPELL_WARRIOR_EXECUTE, &bp, NULL, NULL, true, NULL, NULL, GetOriginalCaster()->GetGUID());
340 }
341 }
342
343 void Register() override
344 {
345 OnEffectHitTarget += SpellEffectFn(spell_warr_execute_SpellScript::HandleEffect, EFFECT_0, SPELL_EFFECT_DUMMY);
346 }
347 };
348
349 SpellScript* GetSpellScript() const override
350 {
351 return new spell_warr_execute_SpellScript();
352 }

521// -772 - Rend
522class spell_warr_rend : public SpellScriptLoader
523{
524 public:
525 spell_warr_rend() : SpellScriptLoader("spell_warr_rend") { }
526
527 class spell_warr_rend_AuraScript : public AuraScript
528 {
529 PrepareAuraScript(spell_warr_rend_AuraScript);
530
531 void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool& canBeRecalculated)
532 {
533 if (Unit* caster = GetCaster())
534 {
535 canBeRecalculated = false;
536
537 // $0.2 * (($MWB + $mwb) / 2 + $AP / 14 * $MWS) bonus per tick
538 float ap = caster->GetTotalAttackPowerValue(BASE_ATTACK);
539 int32 mws = caster->GetAttackTime(BASE_ATTACK);
540 float mwbMin = caster->GetWeaponDamageRange(BASE_ATTACK, MINDAMAGE);
541 float mwbMax = caster->GetWeaponDamageRange(BASE_ATTACK, MAXDAMAGE);
542 float mwb = ((mwbMin + mwbMax) / 2 + ap * mws / 14000) * 0.2f;
543 amount += int32(caster->ApplyEffectModifiers(GetSpellInfo(), aurEff->GetEffIndex(), mwb));
544
545 // "If used while your target is above 75% health, Rend does 35% more damage."
546 // as for 3.1.3 only ranks above 9 (wrong tooltip?)
547 if (GetSpellInfo()->GetRank() >= 9)
548 {
549 if (GetUnitOwner()->HasAuraState(AURA_STATE_HEALTH_ABOVE_75_PERCENT, GetSpellInfo(), caster))
550 AddPct(amount, GetSpellInfo()->Effects[EFFECT_2].CalcValue(caster));
551 }
552 }
553 }
554
555 void Register() override
556 {
557 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_warr_rend_AuraScript::CalculateAmount, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE);
558 }
559 };
560
561 AuraScript* GetAuraScript() const override
562 {
563 return new spell_warr_rend_AuraScript();

594 }
595

772 Unit* target = GetTarget();
773 target->CastSpell(target, SPELL_GEN_DAMAGE_REDUCTION_AURA, true);
774
775 if (Unit* caster = GetCaster())
776 target->CastSpell(caster, SPELL_WARRIOR_VIGILANCE_REDIRECT_THREAT, true);
777 }

779 void HandleAfterApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
780 {
781 //! WORKAROUND
782 //! this glyph is a proc
783 if (Unit* caster = GetCaster())
784 {
785 if (AuraEffect const* glyph = caster->GetAuraEffect(SPELL_WARRIOR_GLYPH_OF_VIGILANCE, EFFECT_0))
786 GetTarget()->ModifyRedirectThreat(glyph->GetAmount());
787 }
788 }
789
790 void HandleRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
791 {
792 Unit* target = GetTarget();
793 if (target->HasAura(SPELL_GEN_DAMAGE_REDUCTION_AURA) &&
794 !(target->HasAura(SPELL_PALADIN_BLESSING_OF_SANCTUARY) ||
795 target->HasAura(SPELL_PALADIN_GREATER_BLESSING_OF_SANCTUARY) ||
796 target->HasAura(SPELL_PRIEST_RENEWED_HOPE)))
797 {
798 target->RemoveAurasDueToSpell(SPELL_GEN_DAMAGE_REDUCTION_AURA);
799 }
800
801 target->ResetRedirectThreat();
802 }

808 }

819 AfterEffectApply += AuraEffectApplyFn(spell_warr_vigilance_AuraScript::HandleAfterApply, EFFECT_0, SPELL_AURA_PROC_TRIGGER_SPELL, AURA_EFFECT_HANDLE_REAL_OR_REAPPLY_MASK);

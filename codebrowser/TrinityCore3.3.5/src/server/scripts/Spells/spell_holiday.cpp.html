45class spell_love_is_in_the_air_romantic_picnic : public SpellScriptLoader
46{
47 public:
48 spell_love_is_in_the_air_romantic_picnic() : SpellScriptLoader("spell_love_is_in_the_air_romantic_picnic") { }
49
50 class spell_love_is_in_the_air_romantic_picnic_AuraScript : public AuraScript
51 {
52 PrepareAuraScript(spell_love_is_in_the_air_romantic_picnic_AuraScript);
53
54 void OnApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
55 {
56 Unit* target = GetTarget();
57 target->SetStandState(UNIT_STAND_STATE_SIT);
58 target->CastSpell(target, SPELL_MEAL_PERIODIC, false);
59 }
60
61 void OnPeriodic(AuraEffect const* /*aurEff*/)
62 {
63 // Every 5 seconds
64 Unit* target = GetTarget();
65 Unit* caster = GetCaster();
66
67 // If our player is no longer sit, remove all auras
68 if (target->getStandState() != UNIT_STAND_STATE_SIT)
69 {
70 target->RemoveAura(SPELL_ROMANTIC_PICNIC_ACHIEV);
71 target->RemoveAura(GetAura());
72 return;
73 }
74
75 target->CastSpell(target, SPELL_BASKET_CHECK, false); // unknown use, it targets Romantic Basket
76 target->CastSpell(target, RAND(SPELL_MEAL_EAT_VISUAL, SPELL_DRINK_VISUAL), false);
77
78 bool foundSomeone = false;
79 // For nearby players, check if they have the same aura. If so, cast Romantic Picnic (45123)
80 // required by achievement and "hearts" visual
81 std::list<Player*> playerList;
82 Trinity::AnyPlayerInObjectRangeCheck checker(target, INTERACTION_DISTANCE*2);
83 Trinity::PlayerListSearcher<Trinity::AnyPlayerInObjectRangeCheck> searcher(target, playerList, checker);
84 target->VisitNearbyWorldObject(INTERACTION_DISTANCE*2, searcher);
85 for (std::list<Player*>::const_iterator itr = playerList.begin(); itr != playerList.end(); ++itr)
86 {
87 if ((*itr) != target && (*itr)->HasAura(GetId())) // && (*itr)->getStandState() == UNIT_STAND_STATE_SIT)
88 {
89 if (caster)
90 {
91 caster->CastSpell(*itr, SPELL_ROMANTIC_PICNIC_ACHIEV, true);
92 caster->CastSpell(target, SPELL_ROMANTIC_PICNIC_ACHIEV, true);
93 }
94 foundSomeone = true;
95 // break;
96 }
97 }
98
99 if (!foundSomeone && target->HasAura(SPELL_ROMANTIC_PICNIC_ACHIEV))
100 target->RemoveAura(SPELL_ROMANTIC_PICNIC_ACHIEV);
101 }
102
103 void Register() override
104 {
105 AfterEffectApply += AuraEffectApplyFn(spell_love_is_in_the_air_romantic_picnic_AuraScript::OnApply, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY, AURA_EFFECT_HANDLE_REAL);
106 OnEffectPeriodic += AuraEffectPeriodicFn(spell_love_is_in_the_air_romantic_picnic_AuraScript::OnPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
107 }
108 };
109
110 AuraScript* GetAuraScript() const override
111 {
112 return new spell_love_is_in_the_air_romantic_picnic_AuraScript();
113 }

156 switch (urand(0, 5))
157 {
158 case 1:
159 spellId = gender ? SPELL_LEPER_GNOME_COSTUME_FEMALE : SPELL_LEPER_GNOME_COSTUME_MALE;
160 break;
161 case 2:

164 case 3:

166 break;
167 case 4:
168 spellId = gender ? SPELL_NINJA_COSTUME_FEMALE : SPELL_NINJA_COSTUME_MALE;
169 break;
170 case 5:
171 spellId = SPELL_SKELETON_COSTUME;
172 break;
173 default:
174 break;
175 }
176
177 caster->CastSpell(target, spellId, true);
178 }
179 }

594
595 void HandleScript(SpellEffIndex effIndex)
596 {
597 PreventHitDefaultEffect(effIndex);
598 Player* target = GetHitPlayer();
599 if (!target)
600 return;

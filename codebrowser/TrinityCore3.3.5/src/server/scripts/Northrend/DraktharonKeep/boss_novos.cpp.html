76class boss_novos : public CreatureScript
77{
78public:
79 boss_novos() : CreatureScript("boss_novos") { }
80
81 struct boss_novosAI : public BossAI
82 {
83 boss_novosAI(Creature* creature) : BossAI(creature, DATA_NOVOS)
84 {
85 Initialize();
86 _bubbled = false;
87 }
88
89 void Initialize()
90 {
91 _ohNovos = true;
92 _crystalHandlerCount = 0;
93 }
94
95 void Reset() override
96 {
97 _Reset();
98
99 Initialize();
100 SetCrystalsStatus(false);
101 SetSummonerStatus(false);
102 SetBubbled(false);
103 }
104
105 void EnterCombat(Unit* /* victim */) override
106 {
107 _EnterCombat();
108 Talk(SAY_AGGRO);
109
110 SetCrystalsStatus(true);
111 SetSummonerStatus(true);
112 SetBubbled(true);
113 }
114
115 void AttackStart(Unit* target) override
116 {
117 if (!target)
118 return;
119
120 if (me->Attack(target, true))
121 DoStartNoMovement(target);
122 }
123
124 void KilledUnit(Unit* who) override
125 {
126 if (who->GetTypeId() == TYPEID_PLAYER)
127 Talk(SAY_KILL);
128 }
129
130 void JustDied(Unit* /*killer*/) override
131 {
132 _JustDied();
133 Talk(SAY_DEATH);
134 }
135
136 void UpdateAI(uint32 diff) override
137 {
138 if (!UpdateVictim() || _bubbled)
139 return;
140
141 events.Update(diff);
142
143 if (me->HasUnitState(UNIT_STATE_CASTING))
144 return;
145
146 if (uint32 eventId = events.ExecuteEvent())
147 {
148 switch (eventId)
149 {
150 case EVENT_SUMMON_MINIONS:
151 DoCast(SPELL_SUMMON_MINIONS);
152 events.ScheduleEvent(EVENT_SUMMON_MINIONS, 15000);
153 break;
154 case EVENT_ATTACK:
155 if (Unit* victim = SelectTarget(SELECT_TARGET_RANDOM))
156 DoCast(victim, RAND(SPELL_ARCANE_BLAST, SPELL_BLIZZARD, SPELL_FROSTBOLT, SPELL_WRATH_OF_MISERY));
157 events.ScheduleEvent(EVENT_ATTACK, 3000);
158 break;
159 default:
160 break;
161 }
162 }
163 }
164
165 void DoAction(int32 action) override
166 {
167 if (action == ACTION_CRYSTAL_HANDLER_DIED)
168 CrystalHandlerDied();
169 }
170
171 void MoveInLineOfSight(Unit* who) override
172 {
173 BossAI::MoveInLineOfSight(who);
174
175 if (!_ohNovos || !who || who->GetTypeId() != TYPEID_UNIT || who->GetPositionY() > MAX_Y_COORD_OH_NOVOS)
176 return;
177
178 uint32 entry = who->GetEntry();
179 if (entry == NPC_HULKING_CORPSE || entry == NPC_RISEN_SHADOWCASTER || entry == NPC_FETID_TROLL_CORPSE)
180 _ohNovos = false;
181 }
182
183 uint32 GetData(uint32 type) const override
184 {
185 return type == DATA_NOVOS_ACHIEV && _ohNovos ? 1 : 0;
186 }
187
188 void JustSummoned(Creature* summon) override
189 {
190 summons.Summon(summon);
191 }
192
193 private:
194 void SetBubbled(bool state)
195 {
196 _bubbled = state;
197 if (!state)
198 {
199 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
200 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
201 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC))
202 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
203 if (me->HasUnitState(UNIT_STATE_CASTING))
204 me->CastStop();
205 }
206 else
207 {
208 if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
209 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
210 if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC))
211 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
212 DoCast(SPELL_ARCANE_FIELD);
213 }
214 }
215
216 void SetSummonerStatus(bool active)
217 {
218 for (uint8 i = 0; i < 4; i++)
219 if (ObjectGuid guid = instance->GetGuidData(summoners[i].data))
220 if (Creature* crystalChannelTarget = ObjectAccessor::GetCreature(*me, guid))
221 {
222 if (active)
223 crystalChannelTarget->AI()->SetData(summoners[i].spell, summoners[i].timer);
224 else
225 crystalChannelTarget->AI()->Reset();
226 }
227 }
228
229 void SetCrystalsStatus(bool active)
230 {
231 for (uint8 i = 0; i < 4; i++)
232 if (ObjectGuid guid = instance->GetGuidData(DATA_NOVOS_CRYSTAL_1 + i))
233 if (GameObject* crystal = ObjectAccessor::GetGameObject(*me, guid))
234 SetCrystalStatus(crystal, active);
235 }
236
237 void SetCrystalStatus(GameObject* crystal, bool active)
238 {
239 crystal->SetGoState(active ? GO_STATE_ACTIVE : GO_STATE_READY);
240 if (Creature* crystalChannelTarget = crystal->FindNearestCreature(NPC_CRYSTAL_CHANNEL_TARGET, 5.0f))
241 {
242 if (active)
243 crystalChannelTarget->CastSpell((Unit*)NULL, SPELL_BEAM_CHANNEL);
244 else if (crystalChannelTarget->HasUnitState(UNIT_STATE_CASTING))
245 crystalChannelTarget->CastStop();
246 }
247 }
248
249 void CrystalHandlerDied()
250 {
251 for (uint8 i = 0; i < 4; i++)
252 if (ObjectGuid guid = instance->GetGuidData(DATA_NOVOS_CRYSTAL_1 + i))
253 if (GameObject* crystal = ObjectAccessor::GetGameObject(*me, guid))
254 if (crystal->GetGoState() == GO_STATE_ACTIVE)
255 {
256 SetCrystalStatus(crystal, false);
257 break;
258 }
259
260 if (++_crystalHandlerCount >= 4)
261 {
262 Talk(SAY_ARCANE_FIELD);
263 SetSummonerStatus(false);
264 SetBubbled(false);
265 events.ScheduleEvent(EVENT_ATTACK, 3000);
266 if (IsHeroic())
267 events.ScheduleEvent(EVENT_SUMMON_MINIONS, 15000);
268 }
269 else if (ObjectGuid guid = instance->GetGuidData(DATA_NOVOS_SUMMONER_4))
270 if (Creature* crystalChannelTarget = ObjectAccessor::GetCreature(*me, guid))
271 crystalChannelTarget->AI()->SetData(SPELL_SUMMON_CRYSTAL_HANDLER, 15000);
272 }
273
274 uint8 _crystalHandlerCount;
275 bool _ohNovos;
276 bool _bubbled;
277 };
278
279 CreatureAI* GetAI(Creature* creature) const override
280 {
281 return GetDrakTharonKeepAI<boss_novosAI>(creature);
282 }

285class npc_crystal_channel_target : public CreatureScript
286{
287public:
288 npc_crystal_channel_target() : CreatureScript("npc_crystal_channel_target") { }
289
290 struct npc_crystal_channel_targetAI : public ScriptedAI
291 {
292 npc_crystal_channel_targetAI(Creature* creature) : ScriptedAI(creature)
293 {
294 Initialize();
295 }
296
297 void Initialize()
298 {
299 _spell = 0;
300 _timer = 0;
301 _temp = 0;
302 }
303
304 void Reset() override
305 {
306 Initialize();
307 }
308
309 void UpdateAI(uint32 diff) override
310 {
311 if (_spell)
312 {
313 if (_temp <= diff)
314 {
315 DoCast(_spell);
316 _temp = _timer;
317 }
318 else
319 _temp -= diff;
320 }
321 }
322
323 void SetData(uint32 id, uint32 value) override
324 {
325 _spell = id;
326 _timer = value;
327 _temp = value;
328 }
329
330 void JustSummoned(Creature* summon) override
331 {
332 if (InstanceScript* instance = me->GetInstanceScript())
333 if (ObjectGuid guid = instance->GetGuidData(DATA_NOVOS))
334 if (Creature* novos = ObjectAccessor::GetCreature(*me, guid))
335 novos->AI()->JustSummoned(summon);
336
337 if (summon)
338 summon->GetMotionMaster()->MovePath(summon->GetEntry() * 100, false);
339
340 if (_spell == SPELL_SUMMON_CRYSTAL_HANDLER)
341 Reset();
342 }
343
344 private:
345 uint32 _spell;
346 uint32 _timer;
347 uint32 _temp;
348 };
349
350 CreatureAI* GetAI(Creature* creature) const override
351 {
352 return GetDrakTharonKeepAI<npc_crystal_channel_targetAI>(creature);
353 }

374 PrepareSpellScript(spell_novos_summon_minions_SpellScript);
375
376 bool Validate(SpellInfo const* /*spellInfo*/) override
377 {
378 if (!sSpellMgr->GetSpellInfo(SPELL_SUMMON_COPY_OF_MINIONS))
379 return false;
380 return true;
381 }
382
383 void HandleScript(SpellEffIndex /*effIndex*/)
384 {
385 for (uint8 i = 0; i < 2; ++i)
386 GetCaster()->CastSpell((Unit*)NULL, SPELL_SUMMON_COPY_OF_MINIONS, true);
387 }
388
389 void Register() override
390 {
391 OnEffectHitTarget += SpellEffectFn(spell_novos_summon_minions_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
392 }
393 };

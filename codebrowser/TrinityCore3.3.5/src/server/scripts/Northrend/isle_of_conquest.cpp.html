30class npc_four_car_garage : public CreatureScript
31{
32 public:
33 npc_four_car_garage() : CreatureScript("npc_four_car_garage") { }
34
35 struct npc_four_car_garageAI : public NullCreatureAI
36 {
37 npc_four_car_garageAI(Creature* creature) : NullCreatureAI(creature) { }
38
39 void PassengerBoarded(Unit* who, int8 /*seatId*/, bool apply) override
40 {
41 if (apply)
42 {
43 uint32 spellId = 0;
44
45 switch (me->GetEntry())
46 {
47 case NPC_DEMOLISHER:
48 spellId = SPELL_DRIVING_CREDIT_DEMOLISHER;
49 break;
50 case NPC_GLAIVE_THROWER_A:
51 case NPC_GLAIVE_THROWER_H:
52 spellId = SPELL_DRIVING_CREDIT_GLAIVE;
53 break;
54 case NPC_SIEGE_ENGINE_H:
55 case NPC_SIEGE_ENGINE_A:
56 spellId = SPELL_DRIVING_CREDIT_SIEGE;
57 break;
58 case NPC_CATAPULT:
59 spellId = SPELL_DRIVING_CREDIT_CATAPULT;
60 break;
61 default:
62 return;
63 }
64
65 me->CastSpell(who, spellId, true);
66 }
67 }
68 };
69
70 CreatureAI* GetAI(Creature* creature) const override
71 {
72 return new npc_four_car_garageAI(creature);
73 }
74};

87class npc_ioc_gunship_captain : public CreatureScript
88{
89 public:
90 npc_ioc_gunship_captain() : CreatureScript("npc_ioc_gunship_captain") { }
91
92 struct npc_ioc_gunship_captainAI : public ScriptedAI
93 {
94 npc_ioc_gunship_captainAI(Creature* creature) : ScriptedAI(creature) { }
95
96 void DoAction(int32 action) override
97 {
98 if (action == ACTION_GUNSHIP_READY)
99 {
100 DoCast(me, SPELL_SIMPLE_TELEPORT);
101 _events.ScheduleEvent(EVENT_TALK, 3000);
102 }
103 }
104
105 void UpdateAI(uint32 diff) override
106 {
107 _events.Update(diff);
108 while (uint32 eventId = _events.ExecuteEvent())
109 {
110 switch (eventId)
111 {
112 case EVENT_TALK:
113 _events.ScheduleEvent(EVENT_DESPAWN, 1000);
114 Talk(SAY_ONBOARD);
115 DoCast(me, SPELL_TELEPORT_VISUAL_ONLY);
116 break;
117 case EVENT_DESPAWN:
118 if (me->GetMap()->ToBattlegroundMap())
119 if (Battleground* bgIoC = me->GetMap()->ToBattlegroundMap()->GetBG())
120 bgIoC->DelCreature(BG_IC_NPC_GUNSHIP_CAPTAIN_1);
121 break;
122 default:
123 break;
124 }
125 }
126 }
127
128 private:
129 EventMap _events;
130 };
131
132 CreatureAI* GetAI(Creature* creature) const override
133 {
134 return new npc_ioc_gunship_captainAI(creature);
135 }
136};

138class spell_ioc_gunship_portal : public SpellScriptLoader
139{
140 public:
141 spell_ioc_gunship_portal() : SpellScriptLoader("spell_ioc_gunship_portal") { }
142
143 class spell_ioc_gunship_portal_SpellScript : public SpellScript
144 {
145 PrepareSpellScript(spell_ioc_gunship_portal_SpellScript);
146
147 bool Load() override
148 {
149 return GetCaster()->GetTypeId() == TYPEID_PLAYER;
150 }
151
152 void HandleScript(SpellEffIndex /*effIndex*/)
153 {
154 Player* caster = GetCaster()->ToPlayer();
155 /*
156 * HACK: GetWorldLocation() returns real position and not transportposition.
157 * ServertoClient: SMSG_MOVE_TELEPORT (0x0B39)
158 * counter: 45
159 * Tranpsort Guid: Full: xxxx Type: MOTransport Low: xxx
160 * Transport Position X: 0 Y: 0 Z: 0 O: 0
161 * Position: X: 7.305609 Y: -0.095246 Z: 34.51022 O: 0
162 */
163 caster->TeleportTo(GetHitCreature()->GetWorldLocation(), TELE_TO_NOT_LEAVE_TRANSPORT);
164 }
165
166 void Register() override
167 {
168 OnEffectHitTarget += SpellEffectFn(spell_ioc_gunship_portal_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
169 }
170 };
171
172 SpellScript* GetSpellScript() const override
173 {
174 return new spell_ioc_gunship_portal_SpellScript();
175 }

206class StartLaunchEvent : public BasicEvent
207{
208 public:
209 StartLaunchEvent(Position const& pos, ObjectGuid::LowType lowGuid) : _pos(pos), _lowGuid(lowGuid)
210 {
211 }
212
213 bool Execute(uint64 /*time*/, uint32 /*diff*/)
214 {
215 Player* player = sObjectMgr->GetPlayerByLowGUID(_lowGuid);
216 if (!player || !player->GetVehicle())
217 return true;
218
219 player->AddAura(SPELL_LAUNCH_NO_FALLING_DAMAGE, player); // prevents falling damage
220 float speedZ = 10.0f;
221 float dist = player->GetExactDist2d(&_pos);
222
223 player->ExitVehicle();
224 player->GetMotionMaster()->MoveJump(_pos, dist, speedZ, EVENT_JUMP, true);
225 return true;
226 }
227
228 private:
229 Position _pos;
230 ObjectGuid::LowType _lowGuid;
231};

244 if (!GetCaster()->ToCreature() || !GetExplTargetDest())
245 return;
246
247 GetCaster()->ToCreature()->m_Events.AddEvent(new StartLaunchEvent(*GetExplTargetDest(), GetHitPlayer()->GetGUID().GetCounter()), GetCaster()->ToCreature()->m_Events.CalculateTime(2500));
248 }
249
250 void Register() override
251 {
252 AfterHit += SpellHitFn(spell_ioc_launch_SpellScript::Launch);
253 }
254 };

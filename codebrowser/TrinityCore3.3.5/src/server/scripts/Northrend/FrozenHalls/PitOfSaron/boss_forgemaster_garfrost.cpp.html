78class boss_garfrost : public CreatureScript
79{
80 public:
81 boss_garfrost() : CreatureScript("boss_garfrost") { }
82
83 struct boss_garfrostAI : public BossAI
84 {
85 boss_garfrostAI(Creature* creature) : BossAI(creature, DATA_GARFROST)
86 {
87 Initialize();
88 }
89
90 void Initialize()
91 {
92 _permafrostStack = 0;
93 }
94
95 void Reset() override
96 {
97 _Reset();
98 events.SetPhase(PHASE_ONE);
99 SetEquipmentSlots(true);
100 Initialize();
101 }
102
103 void EnterCombat(Unit* /*who*/) override
104 {
105 _EnterCombat();
106 Talk(SAY_AGGRO);
107 DoCast(me, SPELL_PERMAFROST);
108 me->CallForHelp(70.0f);
109 events.ScheduleEvent(EVENT_THROW_SARONITE, 7000);
110 }
111
112 void KilledUnit(Unit* victim) override
113 {
114 if (victim->GetTypeId() == TYPEID_PLAYER)
115 Talk(SAY_SLAY);
116 }
117
118 void JustDied(Unit* /*killer*/) override
119 {
120 _JustDied();
121 Talk(SAY_DEATH);
122 me->RemoveAllGameObjects();
123
124 if (Creature* tyrannus = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_TYRANNUS)))
125 tyrannus->AI()->Talk(SAY_TYRANNUS_DEATH);
126 }
127
128 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
129 {
130 if (events.IsInPhase(PHASE_ONE) && !HealthAbovePct(66))
131 {
132 events.SetPhase(PHASE_TWO);
133 Talk(SAY_PHASE2);
134 events.DelayEvents(8000);
135 DoCast(me, SPELL_THUNDERING_STOMP);
136 events.ScheduleEvent(EVENT_FORGE_JUMP, 1500);
137 return;
138 }
139
140 if (events.IsInPhase(PHASE_TWO) && !HealthAbovePct(33))
141 {
142 events.SetPhase(PHASE_THREE);
143 Talk(SAY_PHASE3);
144 events.DelayEvents(8000);
145 DoCast(me, SPELL_THUNDERING_STOMP);
146 events.ScheduleEvent(EVENT_FORGE_JUMP, 1500);
147 return;
148 }
149 }
150
151 void MovementInform(uint32 type, uint32 id) override
152 {
153 if (type != EFFECT_MOTION_TYPE || id != POINT_FORGE)
154 return;
155
156 if (events.IsInPhase(PHASE_TWO))
157 {
158 DoCast(me, SPELL_FORGE_BLADE);
159 SetEquipmentSlots(false, EQUIP_ID_SWORD);
160 }
161 if (events.IsInPhase(PHASE_THREE))
162 {
163 me->RemoveAurasDueToSpell(SPELL_FORGE_BLADE_HELPER);
164 DoCast(me, SPELL_FORGE_MACE);
165 SetEquipmentSlots(false, EQUIP_ID_MACE);
166 }
167 events.ScheduleEvent(EVENT_RESUME_ATTACK, 5000);
168 }
169
170 void SpellHitTarget(Unit* target, const SpellInfo* spell) override
171 {
172 if (spell->Id == SPELL_PERMAFROST_HELPER)
173 {
174 if (Aura* aura = target->GetAura(SPELL_PERMAFROST_HELPER))
175 _permafrostStack = std::max<uint32>(_permafrostStack, aura->GetStackAmount());
176 }
177 }
178
179 uint32 GetData(uint32 /*type*/) const override
180 {
181 return _permafrostStack;
182 }
183
184 void UpdateAI(uint32 diff) override
185 {
186 if (!UpdateVictim())
187 return;
188
189 events.Update(diff);
190
191 if (me->HasUnitState(UNIT_STATE_CASTING))
192 return;
193
194 while (uint32 eventId = events.ExecuteEvent())
195 {
196 switch (eventId)
197 {
198 case EVENT_THROW_SARONITE:
199 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
200 {
201 Talk(SAY_THROW_SARONITE, target);
202 DoCast(target, SPELL_THROW_SARONITE);
203 }
204 events.ScheduleEvent(EVENT_THROW_SARONITE, urand(12500, 20000));
205 break;
206 case EVENT_CHILLING_WAVE:
207 DoCast(me, SPELL_CHILLING_WAVE);
208 events.ScheduleEvent(EVENT_CHILLING_WAVE, 40000, 0, PHASE_TWO);
209 break;
210 case EVENT_DEEP_FREEZE:
211 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
212 {
213 Talk(SAY_CAST_DEEP_FREEZE, target);
214 DoCast(target, SPELL_DEEP_FREEZE);
215 }
216 events.ScheduleEvent(EVENT_DEEP_FREEZE, 35000, 0, PHASE_THREE);
217 break;
218 case EVENT_FORGE_JUMP:
219 me->AttackStop();
220 if (events.IsInPhase(PHASE_TWO))
221 me->GetMotionMaster()->MoveJump(northForgePos, 25.0f, 15.0f, POINT_FORGE);
222 else if (events.IsInPhase(PHASE_THREE))
223 me->GetMotionMaster()->MoveJump(southForgePos, 25.0f, 15.0f, POINT_FORGE);
224 break;
225 case EVENT_RESUME_ATTACK:
226 if (events.IsInPhase(PHASE_TWO))
227 events.ScheduleEvent(EVENT_CHILLING_WAVE, 5000, 0, PHASE_TWO);
228 else if (events.IsInPhase(PHASE_THREE))
229 events.ScheduleEvent(EVENT_DEEP_FREEZE, 10000, 0, PHASE_THREE);
230 AttackStart(me->GetVictim());
231 break;
232 default:
233 break;
234 }
235 }
236
237 DoMeleeAttackIfReady();
238 }
239
240 private:
241 uint32 _permafrostStack;
242 };
243
244 CreatureAI* GetAI(Creature* creature) const override
245 {
246 return GetPitOfSaronAI<boss_garfrostAI>(creature);
247 }

250class spell_garfrost_permafrost : public SpellScriptLoader
251{
252 public:
253 spell_garfrost_permafrost() : SpellScriptLoader("spell_garfrost_permafrost") { }
254
255 class spell_garfrost_permafrost_SpellScript : public SpellScript
256 {
257 PrepareSpellScript(spell_garfrost_permafrost_SpellScript);
258
259 public:
260 spell_garfrost_permafrost_SpellScript()
261 {
262 prevented = false;
263 }
264
265 private:
266 void PreventHitByLoS()
267 {
268 if (Unit* target = GetHitUnit())
269 {
270 Unit* caster = GetCaster();
271 //Temporary Line of Sight Check
272 std::list<GameObject*> blockList;
273 caster->GetGameObjectListWithEntryInGrid(blockList, GO_SARONITE_ROCK, 100.0f);
274 if (!blockList.empty())
275 {
276 for (std::list<GameObject*>::const_iterator itr = blockList.begin(); itr != blockList.end(); ++itr)
277 {
278 if (!(*itr)->IsInvisibleDueToDespawn())
279 {
280 if ((*itr)->IsInBetween(caster, target, 4.0f))
281 {
282 prevented = true;
283 target->ApplySpellImmune(GetSpellInfo()->Id, IMMUNITY_ID, GetSpellInfo()->Id, true);
284 PreventHitDefaultEffect(EFFECT_0);
285 PreventHitDefaultEffect(EFFECT_1);
286 PreventHitDefaultEffect(EFFECT_2);
287 PreventHitDamage();
288 break;
289 }
290 }
291 }
292 }
293 }
294 }
295
296 void RestoreImmunity()
297 {
298 if (Unit* target = GetHitUnit())
299 {
300 target->ApplySpellImmune(GetSpellInfo()->Id, IMMUNITY_ID, GetSpellInfo()->Id, false);
301 if (prevented)
302 PreventHitAura();
303 }
304 }
305
306 void Register() override
307 {
308 BeforeHit += SpellHitFn(spell_garfrost_permafrost_SpellScript::PreventHitByLoS);
309 AfterHit += SpellHitFn(spell_garfrost_permafrost_SpellScript::RestoreImmunity);
310 }
311
312 bool prevented;
313 };
314
315 SpellScript* GetSpellScript() const override
316 {
317 return new spell_garfrost_permafrost_SpellScript();
318 }

321class achievement_doesnt_go_to_eleven : public AchievementCriteriaScript
322{
323 public:
324 achievement_doesnt_go_to_eleven() : AchievementCriteriaScript("achievement_doesnt_go_to_eleven") { }
325
326 bool OnCheck(Player* /*source*/, Unit* target) override
327 {
328 if (target)
329 if (Creature* garfrost = target->ToCreature())
330 if (garfrost->AI()->GetData(ACHIEV_DOESNT_GO_TO_ELEVEN) <= 10)
331 return true;
332
333 return false;
334 }

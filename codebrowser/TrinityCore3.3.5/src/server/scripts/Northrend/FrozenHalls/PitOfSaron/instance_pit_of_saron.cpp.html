23// positions for Martin Victus (37591) and Gorkun Ironskull (37592)
24Position const SlaveLeaderPos = {689.7158f, -104.8736f, 513.7360f, 0.0f};
25// position for Jaina and Sylvanas
26Position const EventLeaderPos2 = {1054.368f, 107.14620f, 628.4467f, 0.0f};
27
28DoorData const Doors[] =
29{
30 {GO_ICE_WALL, DATA_GARFROST, DOOR_TYPE_PASSAGE },
31 {GO_ICE_WALL, DATA_ICK, DOOR_TYPE_PASSAGE },
32 {GO_HALLS_OF_REFLECTION_PORTCULLIS, DATA_TYRANNUS, DOOR_TYPE_PASSAGE },
33};

35class instance_pit_of_saron : public InstanceMapScript
36{
37 public:
38 instance_pit_of_saron() : InstanceMapScript(PoSScriptName, 658) { }
39
40 struct instance_pit_of_saron_InstanceScript : public InstanceScript
41 {
42 instance_pit_of_saron_InstanceScript(Map* map) : InstanceScript(map)
43 {
44 SetHeaders(DataHeader);
45 SetBossNumber(EncounterCount);
46 LoadDoorData(Doors);
47 _teamInInstance = 0;
48 }
49
50 void OnPlayerEnter(Player* player) override
51 {
52 if (!_teamInInstance)
53 _teamInInstance = player->GetTeam();
54 }
55
56 void OnCreatureCreate(Creature* creature) override
57 {
58 if (!_teamInInstance)
59 {
60 Map::PlayerList const& players = instance->GetPlayers();
61 if (!players.isEmpty())
62 if (Player* player = players.begin()->GetSource())
63 _teamInInstance = player->GetTeam();
64 }
65
66 switch (creature->GetEntry())
67 {
68 case NPC_GARFROST:
69 _garfrostGUID = creature->GetGUID();
70 break;
71 case NPC_KRICK:
72 _krickGUID = creature->GetGUID();
73 break;
74 case NPC_ICK:
75 _ickGUID = creature->GetGUID();
76 break;
77 case NPC_TYRANNUS:
78 _tyrannusGUID = creature->GetGUID();
79 break;
80 case NPC_RIMEFANG:
81 _rimefangGUID = creature->GetGUID();
82 break;
83 case NPC_TYRANNUS_EVENTS:
84 _tyrannusEventGUID = creature->GetGUID();
85 break;
86 case NPC_SYLVANAS_PART1:
87 if (_teamInInstance == ALLIANCE)
88 creature->UpdateEntry(NPC_JAINA_PART1);
89 _jainaOrSylvanas1GUID = creature->GetGUID();
90 break;
91 case NPC_SYLVANAS_PART2:
92 if (_teamInInstance == ALLIANCE)
93 creature->UpdateEntry(NPC_JAINA_PART2);
94 _jainaOrSylvanas2GUID = creature->GetGUID();
95 break;
96 case NPC_KILARA:
97 if (_teamInInstance == ALLIANCE)
98 creature->UpdateEntry(NPC_ELANDRA);
99 break;
100 case NPC_KORALEN:
101 if (_teamInInstance == ALLIANCE)
102 creature->UpdateEntry(NPC_KORLAEN);
103 break;
104 case NPC_CHAMPION_1_HORDE:
105 if (_teamInInstance == ALLIANCE)
106 creature->UpdateEntry(NPC_CHAMPION_1_ALLIANCE);
107 break;
108 case NPC_CHAMPION_2_HORDE:
109 if (_teamInInstance == ALLIANCE)
110 creature->UpdateEntry(NPC_CHAMPION_2_ALLIANCE);
111 break;
112 case NPC_CHAMPION_3_HORDE: // No 3rd set for Alliance?
113 if (_teamInInstance == ALLIANCE)
114 creature->UpdateEntry(NPC_CHAMPION_2_ALLIANCE);
115 break;
116 case NPC_HORDE_SLAVE_1:
117 if (_teamInInstance == ALLIANCE)
118 creature->UpdateEntry(NPC_ALLIANCE_SLAVE_1);
119 break;
120 case NPC_HORDE_SLAVE_2:
121 if (_teamInInstance == ALLIANCE)
122 creature->UpdateEntry(NPC_ALLIANCE_SLAVE_2);
123 break;
124 case NPC_HORDE_SLAVE_3:
125 if (_teamInInstance == ALLIANCE)
126 creature->UpdateEntry(NPC_ALLIANCE_SLAVE_3);
127 break;
128 case NPC_HORDE_SLAVE_4:
129 if (_teamInInstance == ALLIANCE)
130 creature->UpdateEntry(NPC_ALLIANCE_SLAVE_4);
131 break;
132 case NPC_FREED_SLAVE_1_HORDE:
133 if (_teamInInstance == ALLIANCE)
134 creature->UpdateEntry(NPC_FREED_SLAVE_1_ALLIANCE);
135 break;
136 case NPC_FREED_SLAVE_2_HORDE:
137 if (_teamInInstance == ALLIANCE)
138 creature->UpdateEntry(NPC_FREED_SLAVE_2_ALLIANCE);
139 break;
140 case NPC_FREED_SLAVE_3_HORDE:
141 if (_teamInInstance == ALLIANCE)
142 creature->UpdateEntry(NPC_FREED_SLAVE_3_ALLIANCE);
143 break;
144 case NPC_RESCUED_SLAVE_HORDE:
145 if (_teamInInstance == ALLIANCE)
146 creature->UpdateEntry(NPC_RESCUED_SLAVE_ALLIANCE);
147 break;
148 case NPC_MARTIN_VICTUS_1:
149 if (_teamInInstance == ALLIANCE)
150 creature->UpdateEntry(NPC_MARTIN_VICTUS_1);
151 break;
152 case NPC_MARTIN_VICTUS_2:
153 if (_teamInInstance == ALLIANCE)
154 creature->UpdateEntry(NPC_MARTIN_VICTUS_2);
155 break;
156 default:
157 break;
158 }
159 }
160
161 void OnGameObjectCreate(GameObject* go) override
162 {
163 switch (go->GetEntry())
164 {
165 case GO_ICE_WALL:
166 case GO_HALLS_OF_REFLECTION_PORTCULLIS:
167 AddDoor(go, true);
168 break;
169 }
170 }
171
172 void OnGameObjectRemove(GameObject* go) override
173 {
174 switch (go->GetEntry())
175 {
176 case GO_ICE_WALL:
177 case GO_HALLS_OF_REFLECTION_PORTCULLIS:
178 AddDoor(go, false);
179 break;
180 }
181 }
182
183 bool SetBossState(uint32 type, EncounterState state) override
184 {
185 if (!InstanceScript::SetBossState(type, state))
186 return false;
187
188 switch (type)
189 {
190 case DATA_GARFROST:
191 if (state == DONE)
192 {
193 if (Creature* summoner = instance->GetCreature(_garfrostGUID))
194 {
195 if (_teamInInstance == ALLIANCE)
196 summoner->SummonCreature(NPC_MARTIN_VICTUS_1, SlaveLeaderPos, TEMPSUMMON_MANUAL_DESPAWN);
197 else
198 summoner->SummonCreature(NPC_GORKUN_IRONSKULL_2, SlaveLeaderPos, TEMPSUMMON_MANUAL_DESPAWN);
199 }
200 }
201 break;
202 case DATA_TYRANNUS:
203 if (state == DONE)
204 {
205 if (Creature* summoner = instance->GetCreature(_tyrannusGUID))
206 {
207 if (_teamInInstance == ALLIANCE)
208 summoner->SummonCreature(NPC_JAINA_PART2, EventLeaderPos2, TEMPSUMMON_MANUAL_DESPAWN);
209 else
210 summoner->SummonCreature(NPC_SYLVANAS_PART2, EventLeaderPos2, TEMPSUMMON_MANUAL_DESPAWN);
211 }
212 }
213 break;
214 default:
215 break;
216 }
217
218 return true;
219 }
220
221 uint32 GetData(uint32 type) const override
222 {
223 switch (type)
224 {
225 case DATA_TEAM_IN_INSTANCE:
226 return _teamInInstance;
227 default:
228 break;
229 }
230
231 return 0;
232 }
233
234 ObjectGuid GetGuidData(uint32 type) const override
235 {
236 switch (type)
237 {
238 case DATA_GARFROST:
239 return _garfrostGUID;
240 case DATA_KRICK:
241 return _krickGUID;
242 case DATA_ICK:
243 return _ickGUID;
244 case DATA_TYRANNUS:
245 return _tyrannusGUID;
246 case DATA_RIMEFANG:
247 return _rimefangGUID;
248 case DATA_TYRANNUS_EVENT:
249 return _tyrannusEventGUID;
250 case DATA_JAINA_SYLVANAS_1:
251 return _jainaOrSylvanas1GUID;
252 case DATA_JAINA_SYLVANAS_2:
253 return _jainaOrSylvanas2GUID;
254 default:
255 break;
256 }
257
258 return ObjectGuid::Empty;
259 }
260
261 private:
262 ObjectGuid _garfrostGUID;
263 ObjectGuid _krickGUID;
264 ObjectGuid _ickGUID;
265 ObjectGuid _tyrannusGUID;
266 ObjectGuid _rimefangGUID;
267
268 ObjectGuid _tyrannusEventGUID;
269 ObjectGuid _jainaOrSylvanas1GUID;
270 ObjectGuid _jainaOrSylvanas2GUID;
271
272 uint32 _teamInInstance;
273 };

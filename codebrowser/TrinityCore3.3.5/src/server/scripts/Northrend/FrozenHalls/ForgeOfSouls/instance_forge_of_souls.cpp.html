18#include "ScriptMgr.h"
19#include "InstanceScript.h"
20#include "forge_of_souls.h"
21#include "Player.h"
22
23class instance_forge_of_souls : public InstanceMapScript
24{
25 public:
26 instance_forge_of_souls() : InstanceMapScript(FoSScriptName, 632) { }
27
28 struct instance_forge_of_souls_InstanceScript : public InstanceScript
29 {
30 instance_forge_of_souls_InstanceScript(Map* map) : InstanceScript(map)
31 {
32 SetHeaders(DataHeader);
33 SetBossNumber(EncounterCount);
34
35 teamInInstance = 0;
36 }
37
38 void OnPlayerEnter(Player* player) override
39 {
40 if (!teamInInstance)
41 teamInInstance = player->GetTeam();
42 }
43
44 void OnCreatureCreate(Creature* creature) override
45 {
46 if (!teamInInstance)
47 {
48 Map::PlayerList const& players = instance->GetPlayers();
49 if (!players.isEmpty())
50 if (Player* player = players.begin()->GetSource())
51 teamInInstance = player->GetTeam();
52 }
53
54 switch (creature->GetEntry())
55 {
56 case NPC_BRONJAHM:
57 bronjahm = creature->GetGUID();
58 break;
59 case NPC_DEVOURER:
60 devourerOfSouls = creature->GetGUID();
61 break;
62 case NPC_SYLVANAS_PART1:
63 if (teamInInstance == ALLIANCE)
64 creature->UpdateEntry(NPC_JAINA_PART1);
65 break;
66 case NPC_LORALEN:
67 if (teamInInstance == ALLIANCE)
68 creature->UpdateEntry(NPC_ELANDRA);
69 break;
70 case NPC_KALIRA:
71 if (teamInInstance == ALLIANCE)
72 creature->UpdateEntry(NPC_KORELN);
73 break;
74 }
75 }
76
77 uint32 GetData(uint32 type) const override
78 {
79 switch (type)
80 {
81 case DATA_TEAM_IN_INSTANCE:
82 return teamInInstance;
83 default:
84 break;
85 }
86
87 return 0;
88 }
89
90 ObjectGuid GetGuidData(uint32 type) const override
91 {
92 switch (type)
93 {
94 case DATA_BRONJAHM:
95 return bronjahm;
96 case DATA_DEVOURER_OF_SOULS:
97 return devourerOfSouls;
98 default:
99 break;
100 }
101
102 return ObjectGuid::Empty;
103 }
104
105 private:
106 ObjectGuid bronjahm;
107 ObjectGuid devourerOfSouls;
108
109 uint32 teamInInstance;
110 };
111
112 InstanceScript* GetInstanceScript(InstanceMap* map) const override
113 {
114 return new instance_forge_of_souls_InstanceScript(map);
115 }
116};
117
118void AddSC_instance_forge_of_souls()
119{
120 new instance_forge_of_souls();
121}

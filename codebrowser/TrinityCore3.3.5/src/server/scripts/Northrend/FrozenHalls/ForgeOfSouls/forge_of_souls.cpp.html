74class npc_sylvanas_fos : public CreatureScript
75{
76public:
77 npc_sylvanas_fos() : CreatureScript("npc_sylvanas_fos") { }
78
79 struct npc_sylvanas_fosAI : public ScriptedAI
80 {
81 npc_sylvanas_fosAI(Creature* creature) : ScriptedAI(creature)
82 {
83 Initialize();
84 instance = me->GetInstanceScript();
85 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
86 }
87
88 void Initialize()
89 {
90 phase = PHASE_NORMAL;
91 }
92
93 InstanceScript* instance;
94
95 EventMap events;
96 Phase phase;
97
98 void Reset() override
99 {
100 events.Reset();
101 Initialize();
102 }
103
104 void DoAction(int32 actionId) override
105 {
106 switch (actionId)
107 {
108 case ACTION_INTRO:
109 {
110 phase = PHASE_INTRO;
111 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
112 events.Reset();
113 events.ScheduleEvent(EVENT_INTRO_1, 1000);
114 }
115 }
116 }
117
118 void UpdateAI(uint32 diff) override
119 {
120 if (phase == PHASE_INTRO)
121 {
122 events.Update(diff);
123 switch (events.ExecuteEvent())
124 {
125 case EVENT_INTRO_1:
126 Talk(SAY_SYLVANAS_INTRO_1);
127 events.ScheduleEvent(EVENT_INTRO_2, 11500);
128 break;
129
130 case EVENT_INTRO_2:
131 Talk(SAY_SYLVANAS_INTRO_2);
132 events.ScheduleEvent(EVENT_INTRO_3, 10500);
133 break;
134
135 case EVENT_INTRO_3:
136 Talk(SAY_SYLVANAS_INTRO_3);
137 events.ScheduleEvent(EVENT_INTRO_4, 9500);
138 break;
139
140 case EVENT_INTRO_4:
141 Talk(SAY_SYLVANAS_INTRO_4);
142 events.ScheduleEvent(EVENT_INTRO_5, 10500);
143 break;
144
145 case EVENT_INTRO_5:
146 Talk(SAY_SYLVANAS_INTRO_5);
147 events.ScheduleEvent(EVENT_INTRO_6, 9500);
148 break;
149
150 case EVENT_INTRO_6:
151 Talk(SAY_SYLVANAS_INTRO_6);
152 // End of Intro
153 phase = PHASE_NORMAL;
154 break;
155 }
156 }
157
158 //Return since we have no target
159 if (!UpdateVictim())
160 return;
161
162 events.Update(diff);
163 DoMeleeAttackIfReady();
164 }
165 };
166
167 bool OnGossipHello(Player* player, Creature* creature) override
168 {
169 if (creature->IsQuestGiver())
170 player->PrepareQuestMenu(creature->GetGUID());
171
172 if (creature->GetEntry() == NPC_JAINA_PART1)
173 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_JAINA_ITEM, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
174 else
175 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_SYLVANAS_ITEM, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
176
177 player->SEND_GOSSIP_MENU(DEFAULT_GOSSIP_MESSAGE, creature->GetGUID());
178 return true;
179 }
180
181 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
182 {
183 player->PlayerTalkClass->ClearMenus();
184 switch (action)
185 {
186 case GOSSIP_ACTION_INFO_DEF+1:
187 player->CLOSE_GOSSIP_MENU();
188
189 if (creature->AI())
190 creature->AI()->DoAction(ACTION_INTRO);
191 break;
192 }
193
194 return true;
195 }
196
197 CreatureAI* GetAI(Creature* creature) const override
198 {
199 return GetInstanceAI<npc_sylvanas_fosAI>(creature);
200 }
201};

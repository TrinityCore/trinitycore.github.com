367 void sGossipSelect(Player* player, uint32 /*menuId*/, uint32 gossipListId) override
368 {
369 player->PlayerTalkClass->ClearMenus();
370
371 switch (gossipListId)
372 {
373 case 0:
374 player->PlayerTalkClass->SendCloseGossip();
375 _events.ScheduleEvent(EVENT_START_INTRO, 1000);
376 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP | UNIT_NPC_FLAG_QUESTGIVER);
377 break;
378 case 1:
379 player->PlayerTalkClass->SendCloseGossip();
380 _events.ScheduleEvent(EVENT_SKIP_INTRO, 1000);
381 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP | UNIT_NPC_FLAG_QUESTGIVER);
382 break;
383 default:
384 break;
385 }
386 }

389 {
390 _events.Reset();
391
392 _utherGUID.Clear();
393 _lichkingGUID.Clear();
394

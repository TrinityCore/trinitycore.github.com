195// Base class for FALRIC and MARWYN
196struct boss_horAI : BossAI
197{
198 boss_horAI(Creature* creature, uint32 bossId) : BossAI(creature, bossId) { }
199
200 void Reset() override
201 {
202 _Reset();
203 me->SetVisible(false);
204 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
205 me->SetReactState(REACT_PASSIVE);
206 if (instance->GetData(DATA_WAVE_COUNT) != NOT_STARTED)
207 instance->ProcessEvent(NULL, EVENT_DO_WIPE);
208 }
209
210 void DoAction(int32 actionId) override
211 {
212 switch (actionId)
213 {
214 case ACTION_ENTER_COMBAT: // called by InstanceScript when boss shall enter in combat.
215 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
216 me->SetReactState(REACT_AGGRESSIVE);
217 DoZoneInCombat(me, 150.0f);
218 break;
219 default:
220 break;
221 }
222 }
223
224 void JustSummoned(Creature* summon) override
225 {
226 summons.Summon(summon);
227 }
228};

230class GameObjectDeleteDelayEvent : public BasicEvent
231{
232 public:
233 GameObjectDeleteDelayEvent(Unit* owner, ObjectGuid gameObjectGUID) : _owner(owner), _gameObjectGUID(gameObjectGUID) { }
234
235 void DeleteGameObject()
236 {
237 if (GameObject* go = ObjectAccessor::GetGameObject(*_owner, _gameObjectGUID))
238 go->Delete();
239 }
240
241 bool Execute(uint64 /*execTime*/, uint32 /*diff*/) override
242 {
243 DeleteGameObject();
244 return true;
245 }
246
247 void Abort(uint64 /*execTime*/) override
248 {
249 DeleteGameObject();
250 }
251
252 private:
253 Unit* _owner;
254 ObjectGuid _gameObjectGUID;
255};

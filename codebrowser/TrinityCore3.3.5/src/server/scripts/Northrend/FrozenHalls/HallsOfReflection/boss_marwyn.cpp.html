50class boss_marwyn : public CreatureScript
51{
52 public:
53 boss_marwyn() : CreatureScript("boss_marwyn") { }
54
55 struct boss_marwynAI : public boss_horAI
56 {
57 boss_marwynAI(Creature* creature) : boss_horAI(creature, DATA_MARWYN) { }
58
59 void Reset() override
60 {
61 boss_horAI::Reset();
62 }
63
64 void EnterCombat(Unit* /*who*/) override
65 {
66 Talk(SAY_AGGRO);
67 DoZoneInCombat();
68 instance->SetBossState(DATA_MARWYN, IN_PROGRESS);
69
70 events.ScheduleEvent(EVENT_OBLITERATE, urand(8000, 13000));
71 events.ScheduleEvent(EVENT_WELL_OF_CORRUPTION, 13000);
72 events.ScheduleEvent(EVENT_CORRUPTED_FLESH, 20000);
73 events.ScheduleEvent(EVENT_SHARED_SUFFERING, urand(14000, 15000));
74 }
75
76 void JustDied(Unit* /*killer*/) override
77 {
78 Talk(SAY_DEATH);
79 events.Reset();
80 instance->SetBossState(DATA_MARWYN, DONE);
81 }
82
83 void KilledUnit(Unit* who) override
84 {
85 if (who->GetTypeId() == TYPEID_PLAYER)
86 Talk(SAY_SLAY);
87 }
88
89 void UpdateAI(uint32 diff) override
90 {
91 if (!UpdateVictim())
92 return;
93
94 events.Update(diff);
95
96 if (me->HasUnitState(UNIT_STATE_CASTING))
97 return;
98
99 switch (events.ExecuteEvent())
100 {
101 case EVENT_OBLITERATE:
102 DoCastVictim(SPELL_OBLITERATE);
103 events.ScheduleEvent(EVENT_OBLITERATE, urand(8000, 13000));
104 break;
105 case EVENT_WELL_OF_CORRUPTION:
106 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
107 DoCast(target, SPELL_WELL_OF_CORRUPTION);
108 events.ScheduleEvent(EVENT_WELL_OF_CORRUPTION, 13000);
109 break;
110 case EVENT_CORRUPTED_FLESH:
111 Talk(SAY_CORRUPTED_FLESH);
112 DoCastAOE(SPELL_CORRUPTED_FLESH);
113 events.ScheduleEvent(EVENT_CORRUPTED_FLESH, 20000);
114 break;
115 case EVENT_SHARED_SUFFERING:
116 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
117 DoCast(target, SPELL_SHARED_SUFFERING);
118 events.ScheduleEvent(EVENT_SHARED_SUFFERING, urand(14000, 15000));
119 break;
120 default:
121 break;
122 }
123
124 DoMeleeAttackIfReady();
125 }
126 };
127
128 CreatureAI* GetAI(Creature* creature) const override
129 {
130 return GetHallsOfReflectionAI<boss_marwynAI>(creature);
131 }
132};

134// 72368, 72369 - Shared Suffering
135class spell_marwyn_shared_suffering : public SpellScriptLoader
136{
137 public:
138 spell_marwyn_shared_suffering() : SpellScriptLoader("spell_marwyn_shared_suffering") { }
139
140 class spell_marwyn_shared_suffering_AuraScript : public AuraScript
141 {
142 PrepareAuraScript(spell_marwyn_shared_suffering_AuraScript);
143
144 void HandleEffectRemove(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
145 {
146 if (GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_ENEMY_SPELL)
147 return;
148
149 if (Unit* caster = GetCaster())
150 {
151 int32 remainingDamage = aurEff->GetAmount() * (aurEff->GetTotalTicks() - aurEff->GetTickNumber());
152 if (remainingDamage > 0)
153 caster->CastCustomSpell(SPELL_SHARED_SUFFERING_DISPEL, SPELLVALUE_BASE_POINT1, remainingDamage, GetTarget(), TRIGGERED_FULL_MASK);
154 }
155 }
156
157 void Register() override
158 {
159 AfterEffectRemove += AuraEffectRemoveFn(spell_marwyn_shared_suffering_AuraScript::HandleEffectRemove, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE, AURA_EFFECT_HANDLE_REAL);
160 }
161 };
162
163 AuraScript* GetAuraScript() const override
164 {
165 return new spell_marwyn_shared_suffering_AuraScript();
166 }
167};

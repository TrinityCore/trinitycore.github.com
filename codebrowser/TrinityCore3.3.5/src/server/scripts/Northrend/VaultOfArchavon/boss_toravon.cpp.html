57class boss_toravon : public CreatureScript
58{
59 public:
60 boss_toravon() : CreatureScript("boss_toravon") { }
61
62 struct boss_toravonAI : public BossAI
63 {
64 boss_toravonAI(Creature* creature) : BossAI(creature, DATA_TORAVON)
65 {
66 }
67
68 void EnterCombat(Unit* /*who*/) override
69 {
70 DoCast(me, SPELL_FROZEN_MALLET);
71
72 events.ScheduleEvent(EVENT_FROZEN_ORB, 11000);
73 events.ScheduleEvent(EVENT_WHITEOUT, 13000);
74 events.ScheduleEvent(EVENT_FREEZING_GROUND, 15000);
75
76 _EnterCombat();
77 }
78
79 void UpdateAI(uint32 diff) override
80 {
81 if (!UpdateVictim())
82 return;
83
84 events.Update(diff);
85
86 if (me->HasUnitState(UNIT_STATE_CASTING))
87 return;
88
89 while (uint32 eventId = events.ExecuteEvent())
90 {
91 switch (eventId)
92 {
93 case EVENT_FROZEN_ORB:
94 me->CastCustomSpell(SPELL_FROZEN_ORB, SPELLVALUE_MAX_TARGETS, 1, me);
95 events.ScheduleEvent(EVENT_FROZEN_ORB, 38000);
96 break;
97 case EVENT_WHITEOUT:
98 DoCast(me, SPELL_WHITEOUT);
99 events.ScheduleEvent(EVENT_WHITEOUT, 38000);
100 break;
101 case EVENT_FREEZING_GROUND:
102 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
103 DoCast(target, SPELL_FREEZING_GROUND);
104 events.ScheduleEvent(EVENT_FREEZING_GROUND, 20000);
105 break;
106 default:
107 break;
108 }
109 }
110
111 DoMeleeAttackIfReady();
112 }
113 };
114
115 CreatureAI* GetAI(Creature* creature) const override
116 {
117 return new boss_toravonAI(creature);
118 }
119};

135 events.Reset();
136 }
137
138 void EnterCombat(Unit* /*who*/) override
139 {
140 DoZoneInCombat();
141
142 DoCast(me, SPELL_FROZEN_MALLET_2);
143
144 events.ScheduleEvent(EVENT_FROST_BLAST, 5000);
145 }
146
147 void UpdateAI(uint32 diff) override
148 {
149 if (!UpdateVictim())
150 return;
151
152 events.Update(diff);
153
154 if (me->HasUnitState(UNIT_STATE_CASTING))
155 return;
156
157 if (events.ExecuteEvent() == EVENT_FROST_BLAST)
158 {
159 DoCastVictim(SPELL_FROST_BLAST);
160 events.ScheduleEvent(EVENT_FROST_BLAST, 20000);
161 }
162
163 DoMeleeAttackIfReady();
164 }
165
166 private:
167 EventMap events;
168 };
169
170 CreatureAI* GetAI(Creature* creature) const override
171 {
172 return new npc_frost_warderAI(creature);
173 }

179class npc_frozen_orb : public CreatureScript
180{
181public:
182 npc_frozen_orb() : CreatureScript("npc_frozen_orb") { }
183
184 struct npc_frozen_orbAI : public ScriptedAI
185 {
186 npc_frozen_orbAI(Creature* creature) : ScriptedAI(creature)
187 {
188 Initialize();
189 }
190
191 void Initialize()
192 {
193 done = false;
194 killTimer = 60000; // if after this time there is no victim -> destroy!
195 }
196
197 void Reset() override
198 {
199 Initialize();
200 }
201
202 void EnterCombat(Unit* /*who*/) override
203 {
204 DoZoneInCombat();
205 }
206
207 void UpdateAI(uint32 diff) override
208 {
209 if (!done)
210 {
211 DoCast(me, SPELL_FROZEN_ORB_AURA, true);
212 DoCast(me, SPELL_FROZEN_ORB_DMG, true);
213 done = true;
214 }
215
216 if (killTimer <= diff)
217 {
218 if (!UpdateVictim())
219 me->DespawnOrUnsummon();
220 killTimer = 10000;
221 }
222 else
223 killTimer -= diff;
224 }
225
226 private:
227 uint32 killTimer;
228 bool done;
229 };
230
231 CreatureAI* GetAI(Creature* creature) const override
232 {
233 return new npc_frozen_orbAI(creature);
234 }
235};

240class npc_frozen_orb_stalker : public CreatureScript
241{
242 public:
243 npc_frozen_orb_stalker() : CreatureScript("npc_frozen_orb_stalker") { }
244
245 struct npc_frozen_orb_stalkerAI : public ScriptedAI
246 {
247 npc_frozen_orb_stalkerAI(Creature* creature) : ScriptedAI(creature)
248 {
249 creature->SetVisible(false);
250 creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE);
251 creature->SetReactState(REACT_PASSIVE);
252
253 instance = creature->GetInstanceScript();
254 spawned = false;
255
256 SetCombatMovement(false);
257 }
258
259 void UpdateAI(uint32 /*diff*/) override
260 {
261 if (spawned)
262 return;
263
264 spawned = true;
265 Unit* toravon = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_TORAVON));
266 if (!toravon)
267 return;
268
269 uint8 num_orbs = RAID_MODE(1, 3);
270 for (uint8 i = 0; i < num_orbs; ++i)
271 {
272 Position pos;
273 me->GetNearPoint(toravon, pos.m_positionX, pos.m_positionY, pos.m_positionZ, 0.0f, 10.0f, 0.0f);
274 me->SetPosition(pos);
275 DoCast(me, SPELL_FROZEN_ORB_SUMMON);
276 }
277 }
278
279 private:
280 InstanceScript* instance;
281 bool spawned;
282 };
283
284 CreatureAI* GetAI(Creature* creature) const override
285 {
286 return GetInstanceAI<npc_frozen_orb_stalkerAI>(creature);
287 }
288};

25{
26 // Koralon
27 EVENT_BURNING_BREATH = 1,
28 EVENT_BURNING_FURY = 2,
29 EVENT_FLAME_CINDER = 3,
30 EVENT_METEOR_FISTS = 4,

32 // Flame Warder
33 EVENT_FW_LAVA_BIRST = 5,
34 EVENT_FW_METEOR_FISTS = 6

53class boss_koralon : public CreatureScript
54{
55 public:
56 boss_koralon() : CreatureScript("boss_koralon") { }
57
58 struct boss_koralonAI : public BossAI
59 {
60 boss_koralonAI(Creature* creature) : BossAI(creature, DATA_KORALON)
61 {
62 }
63
64 void EnterCombat(Unit* /*who*/) override
65 {
66 DoCast(me, SPELL_BURNING_FURY);
67
68 events.ScheduleEvent(EVENT_BURNING_FURY, 20000); /// @todo check timer
69 events.ScheduleEvent(EVENT_BURNING_BREATH, 15000); // 1st after 15sec, then every 45sec
70 events.ScheduleEvent(EVENT_METEOR_FISTS, 75000); // 1st after 75sec, then every 45sec
71 events.ScheduleEvent(EVENT_FLAME_CINDER, 30000); /// @todo check timer
72
73 _EnterCombat();
74 }
75
76 void UpdateAI(uint32 diff) override
77 {
78 if (!UpdateVictim())
79 return;
80
81 events.Update(diff);
82
83 if (me->HasUnitState(UNIT_STATE_CASTING))
84 return;
85
86 while (uint32 eventId = events.ExecuteEvent())
87 {
88 switch (eventId)
89 {
90 case EVENT_BURNING_FURY:
91 DoCast(me, SPELL_BURNING_FURY);
92 events.ScheduleEvent(EVENT_BURNING_FURY, 20000);
93 break;
94 case EVENT_BURNING_BREATH:
95 DoCast(me, SPELL_BURNING_BREATH);
96 events.ScheduleEvent(EVENT_BURNING_BREATH, 45000);
97 break;
98 case EVENT_METEOR_FISTS:
99 DoCast(me, SPELL_METEOR_FISTS);
100 events.ScheduleEvent(EVENT_METEOR_FISTS, 45000);
101 break;
102 case EVENT_FLAME_CINDER:
103 DoCast(me, SPELL_FLAME_CINDER_A);
104 events.ScheduleEvent(EVENT_FLAME_CINDER, 30000);
105 break;
106 default:
107 break;
108 }
109 }
110
111 DoMeleeAttackIfReady();
112 }
113 };
114
115 CreatureAI* GetAI(Creature* creature) const override
116 {
117 return new boss_koralonAI(creature);
118 }
119};

124class npc_flame_warder : public CreatureScript
125{
126 public:
127 npc_flame_warder() : CreatureScript("npc_flame_warder") { }
128
129 struct npc_flame_warderAI : public ScriptedAI
130 {
131 npc_flame_warderAI(Creature* creature) : ScriptedAI(creature)
132 {
133 }
134
135 void Reset() override
136 {
137 events.Reset();
138 }
139
140 void EnterCombat(Unit* /*who*/) override
141 {
142 DoZoneInCombat();
143
144 events.ScheduleEvent(EVENT_FW_LAVA_BIRST, 5000);
145 events.ScheduleEvent(EVENT_FW_METEOR_FISTS, 10000);
146 }
147
148 void UpdateAI(uint32 diff) override
149 {
150 if (!UpdateVictim())
151 return;
152
153 events.Update(diff);
154
155 while (uint32 eventId = events.ExecuteEvent())
156 {
157 switch (eventId)
158 {
159 case EVENT_FW_LAVA_BIRST:
160 DoCastVictim(SPELL_FW_LAVA_BIRST);
161 events.ScheduleEvent(EVENT_FW_LAVA_BIRST, 15000);
162 break;
163 case EVENT_FW_METEOR_FISTS:
164 DoCast(me, SPELL_FW_METEOR_FISTS);
165 events.ScheduleEvent(EVENT_FW_METEOR_FISTS, 20000);
166 break;
167 default:
168 break;
169 }
170 }
171
172 DoMeleeAttackIfReady();
173 }
174
175 private:
176 EventMap events;
177 };
178
179 CreatureAI* GetAI(Creature* creature) const override
180 {
181 return new npc_flame_warderAI(creature);
182 }
183};

219class spell_koralon_meteor_fists_damage : public SpellScriptLoader
220{
221 public:
222 spell_koralon_meteor_fists_damage() : SpellScriptLoader("spell_koralon_meteor_fists_damage") { }
223
224 class spell_koralon_meteor_fists_damage_SpellScript : public SpellScript
225 {
226 PrepareSpellScript(spell_koralon_meteor_fists_damage_SpellScript);
227
228 public:
229 spell_koralon_meteor_fists_damage_SpellScript()
230 {
231 _chainTargets = 0;
232 }
233
234 private:
235 void FilterTargets(std::list<WorldObject*>& targets)
236 {
237 _chainTargets = targets.size();
238 }
239
240 void CalculateSplitDamage()
241 {
242 if (_chainTargets)
243 SetHitDamage(GetHitDamage() / (_chainTargets + 1));
244 }
245
246 void Register() override
247 {
248 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_koralon_meteor_fists_damage_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_TARGET_ENEMY);
249 OnHit += SpellHitFn(spell_koralon_meteor_fists_damage_SpellScript::CalculateSplitDamage);
250 }
251
252 private:
253 uint8 _chainTargets;
254 };
255
256 SpellScript* GetSpellScript() const override
257 {
258 return new spell_koralon_meteor_fists_damage_SpellScript();
259 }

29class instance_eye_of_eternity : public InstanceMapScript

31public:
32 instance_eye_of_eternity() : InstanceMapScript("instance_eye_of_eternity", 616) { }

39 struct instance_eye_of_eternity_InstanceMapScript : public InstanceScript
40 {
41 instance_eye_of_eternity_InstanceMapScript(Map* map) : InstanceScript(map)
42 {
43 SetHeaders(DataHeader);
44 SetBossNumber(MAX_ENCOUNTER);
45 LoadBossBoundaries(boundaries);
46 }
47
48 void OnPlayerEnter(Player* player) override
49 {
50 if (GetBossState(DATA_MALYGOS_EVENT) == DONE)
51 player->CastSpell(player, SPELL_SUMMOM_RED_DRAGON_BUDDY, true);
52 }
53
54 bool SetBossState(uint32 type, EncounterState state) override
55 {
56 if (!InstanceScript::SetBossState(type, state))
57 return false;
58
59 if (type == DATA_MALYGOS_EVENT)
60 {
61 if (state == FAIL)
62 {
63 for (GuidList::const_iterator itr_trigger = portalTriggers.begin(); itr_trigger != portalTriggers.end(); ++itr_trigger)
64 {
65 if (Creature* trigger = instance->GetCreature(*itr_trigger))
66 {
67 // just in case
68 trigger->RemoveAllAuras();
69 trigger->AI()->Reset();
70 }
71 }
72
73 SpawnGameObject(GO_EXIT_PORTAL, exitPortalPosition);
74
75 if (GameObject* platform = instance->GetGameObject(platformGUID))
76 platform->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_DESTROYED);
77 }
78 else if (state == DONE)
79 SpawnGameObject(GO_EXIT_PORTAL, exitPortalPosition);
80 }
81 return true;
82 }
83
84 /// @todo this should be handled in map, maybe add a summon function in map
85 // There is no other way afaik...
86 void SpawnGameObject(uint32 entry, Position const& pos)
87 {
88 GameObject* go = new GameObject();
89 if (!go->Create(instance->GenerateLowGuid<HighGuid::GameObject>(), entry, instance,
90 PHASEMASK_NORMAL, pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ(), pos.GetOrientation(),
91 0, 0, 0, 0, 120, GO_STATE_READY))
92 {
93 delete go;
94 return;
95 }
96
97 instance->AddToMap(go);
98 }
99
100 void OnGameObjectCreate(GameObject* go) override
101 {
102 switch (go->GetEntry())
103 {
104 case GO_NEXUS_RAID_PLATFORM:
105 platformGUID = go->GetGUID();
106 break;
107 case GO_FOCUSING_IRIS_10:
108 case GO_FOCUSING_IRIS_25:
109 irisGUID = go->GetGUID();
110 focusingIrisPosition = go->GetPosition();
111 break;
112 case GO_EXIT_PORTAL:
113 exitPortalGUID = go->GetGUID();
114 exitPortalPosition = go->GetPosition();
115 break;
116 case GO_HEART_OF_MAGIC_10:
117 case GO_HEART_OF_MAGIC_25:
118 heartOfMagicGUID = go->GetGUID();
119 break;
120 default:
121 break;
122 }
123 }
124
125 void OnCreatureCreate(Creature* creature) override
126 {
127 switch (creature->GetEntry())
128 {
129 case NPC_VORTEX_TRIGGER:
130 vortexTriggers.push_back(creature->GetGUID());
131 break;
132 case NPC_MALYGOS:
133 malygosGUID = creature->GetGUID();
134 break;
135 case NPC_PORTAL_TRIGGER:
136 portalTriggers.push_back(creature->GetGUID());
137 break;
138 case NPC_ALEXSTRASZA_BUNNY:
139 alexstraszaBunnyGUID = creature->GetGUID();
140 break;
141 case NPC_ALEXSTRASZAS_GIFT:
142 giftBoxBunnyGUID = creature->GetGUID();
143 break;
144 }
145 }
146
147 void OnUnitDeath(Unit* unit) override
148 {
149 if (unit->GetTypeId() != TYPEID_PLAYER)
150 return;
151
152 // Player continues to be moving after death no matter if spline will be cleared along with all movements,
153 // so on next world tick was all about delay if box will pop or not (when new movement will be registered)
154 // since in EoE you never stop falling. However root at this precise* moment works,
155 // it will get cleared on release. If by any chance some lag happen "Reload()" and "RepopMe()" works,
156 // last test I made now gave me 50/0 of this bug so I can't do more about it.
157 unit->SetControlled(true, UNIT_STATE_ROOT);
158 }
159
160 void ProcessEvent(WorldObject* /*obj*/, uint32 eventId) override
161 {
162 if (eventId == EVENT_FOCUSING_IRIS)
163 {
164 if (Creature* alexstraszaBunny = instance->GetCreature(alexstraszaBunnyGUID))
165 alexstraszaBunny->CastSpell(alexstraszaBunny, SPELL_IRIS_OPENED);
166
167 if (GameObject* iris = instance->GetGameObject(irisGUID))
168 iris->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_IN_USE);
169
170 if (Creature* malygos = instance->GetCreature(malygosGUID))
171 malygos->AI()->DoAction(0); // ACTION_LAND_ENCOUNTER_START
172
173 if (GameObject* exitPortal = instance->GetGameObject(exitPortalGUID))
174 exitPortal->Delete();
175 }
176 }
177
178 void VortexHandling()
179 {
180 if (Creature* malygos = instance->GetCreature(malygosGUID))
181 {
182 ThreatContainer::StorageType const& threatList = malygos->getThreatManager().getThreatList();
183 for (GuidList::const_iterator itr_vortex = vortexTriggers.begin(); itr_vortex != vortexTriggers.end(); ++itr_vortex)
184 {
185 if (threatList.empty())
186 return;
187
188 uint8 counter = 0;
189 if (Creature* trigger = instance->GetCreature(*itr_vortex))
190 {
191 // each trigger have to cast the spell to 5 players.
192 for (ThreatContainer::StorageType::const_iterator itr = threatList.begin(); itr != threatList.end(); ++itr)
193 {
194 if (counter >= 5)
195 break;
196
197 if (Unit* target = (*itr)->getTarget())
198 {
199 Player* player = target->ToPlayer();
200
201 if (!player || player->IsGameMaster() || player->HasAura(SPELL_VORTEX_4))
202 continue;
203
204 player->CastSpell(trigger, SPELL_VORTEX_4, true);
205 counter++;
206 }
207 }
208 }
209 }
210 }
211 }
212
213 void PowerSparksHandling()
214 {
215 bool next = (lastPortalGUID == portalTriggers.back() || !lastPortalGUID ? true : false);
216
217 for (GuidList::const_iterator itr_trigger = portalTriggers.begin(); itr_trigger != portalTriggers.end(); ++itr_trigger)
218 {
219 if (next)
220 {
221 if (Creature* trigger = instance->GetCreature(*itr_trigger))
222 {
223 lastPortalGUID = trigger->GetGUID();
224 trigger->CastSpell(trigger, SPELL_PORTAL_OPENED, true);
225 return;
226 }
227 }
228
229 if (*itr_trigger == lastPortalGUID)
230 next = true;
231 }
232 }
233
234 void SetData(uint32 data, uint32 /*value*/) override
235 {
236 switch (data)
237 {
238 case DATA_VORTEX_HANDLING:
239 VortexHandling();
240 break;
241 case DATA_POWER_SPARKS_HANDLING:
242 PowerSparksHandling();
243 break;
244 case DATA_RESPAWN_IRIS:
245 SpawnGameObject(instance->GetDifficulty() == RAID_DIFFICULTY_10MAN_NORMAL ? GO_FOCUSING_IRIS_10 : GO_FOCUSING_IRIS_25, focusingIrisPosition);
246 break;
247 }
248 }

250 ObjectGuid GetGuidData(uint32 data) const override
251 {
252 switch (data)
253 {
254 case DATA_TRIGGER:
255 return vortexTriggers.front();
256 case DATA_MALYGOS:
257 return malygosGUID;
258 case DATA_PLATFORM:
259 return platformGUID;
260 case DATA_ALEXSTRASZA_BUNNY_GUID:
261 return alexstraszaBunnyGUID;
262 case DATA_HEART_OF_MAGIC_GUID:
263 return heartOfMagicGUID;
264 case DATA_FOCUSING_IRIS_GUID:
265 return irisGUID;
266 case DATA_GIFT_BOX_BUNNY_GUID:
267 return giftBoxBunnyGUID;
268 }
269
270 return ObjectGuid::Empty;
271 }

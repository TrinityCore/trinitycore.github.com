62class boss_keristrasza : public CreatureScript
63{
64 public:
65 boss_keristrasza() : CreatureScript("boss_keristrasza") { }
66
67 struct boss_keristraszaAI : public BossAI
68 {
69 boss_keristraszaAI(Creature* creature) : BossAI(creature, DATA_KERISTRASZA)
70 {
71 Initialize();
72 }
73
74 void Initialize()
75 {
76 _enrage = false;
77 _intenseCold = true;
78 }
79
80 void Reset() override
81 {
82 Initialize();
83 _intenseColdList.clear();
84
85 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED);
86
87 RemovePrison(CheckContainmentSpheres());
88 _Reset();
89 }
90
91 void EnterCombat(Unit* /*who*/) override
92 {
93 Talk(SAY_AGGRO);
94 DoCastAOE(SPELL_INTENSE_COLD);
95 _EnterCombat();
96
97 events.ScheduleEvent(EVENT_CRYSTAL_FIRE_BREATH, 14000);
98 events.ScheduleEvent(EVENT_CRYSTAL_CHAINS_CRYSTALIZE, DUNGEON_MODE(30000, 11000));
99 events.ScheduleEvent(EVENT_TAIL_SWEEP, 5000);
100 }
101
102 void JustDied(Unit* /*killer*/) override
103 {
104 Talk(SAY_DEATH);
105 _JustDied();
106 }
107
108 void KilledUnit(Unit* who) override
109 {
110 if (who->GetTypeId() == TYPEID_PLAYER)
111 Talk(SAY_SLAY);
112 }
113
114 bool CheckContainmentSpheres(bool remove_prison = false)
115 {
116 ContainmentSphereGUIDs[0] = instance->GetGuidData(ANOMALUS_CONTAINMET_SPHERE);
117 ContainmentSphereGUIDs[1] = instance->GetGuidData(ORMOROKS_CONTAINMET_SPHERE);
118 ContainmentSphereGUIDs[2] = instance->GetGuidData(TELESTRAS_CONTAINMET_SPHERE);
119
120 for (uint8 i = 0; i < DATA_CONTAINMENT_SPHERES; ++i)
121 {
122 GameObject* ContainmentSphere = ObjectAccessor::GetGameObject(*me, ContainmentSphereGUIDs[i]);
123 if (!ContainmentSphere)
124 return false;
125 if (ContainmentSphere->GetGoState() != GO_STATE_ACTIVE)
126 return false;
127 }
128 if (remove_prison)
129 RemovePrison(true);
130 return true;
131 }
132
133 void RemovePrison(bool remove)
134 {
135 if (remove)
136 {
137 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
138 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
139 if (me->HasAura(SPELL_FROZEN_PRISON))
140 me->RemoveAurasDueToSpell(SPELL_FROZEN_PRISON);
141 }
142 else
143 {
144 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
145 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
146 DoCast(me, SPELL_FROZEN_PRISON, false);
147 }
148 }
149
150 void SetGUID(ObjectGuid guid, int32 id/* = 0 */) override
151 {
152 if (id == DATA_INTENSE_COLD)
153 _intenseColdList.push_back(guid);
154 }
155
156 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
157 {
158 if (!_enrage && me->HealthBelowPctDamaged(25, damage))
159 {
160 Talk(SAY_ENRAGE);
161 Talk(SAY_FRENZY);
162 DoCast(me, SPELL_ENRAGE);
163 _enrage = true;
164 }
165 }
166
167 void UpdateAI(uint32 diff) override
168 {
169 if (!UpdateVictim())
170 return;
171
172 events.Update(diff);
173
174 if (me->HasUnitState(UNIT_STATE_CASTING))
175 return;
176
177 while (uint32 eventId = events.ExecuteEvent())
178 {
179 switch (eventId)
180 {
181 case EVENT_CRYSTAL_FIRE_BREATH:
182 DoCastVictim(SPELL_CRYSTALFIRE_BREATH);
183 events.ScheduleEvent(EVENT_CRYSTAL_FIRE_BREATH, 14000);
184 break;
185 case EVENT_CRYSTAL_CHAINS_CRYSTALIZE:
186 DoCast(me, SPELL_TAIL_SWEEP);
187 events.ScheduleEvent(EVENT_CRYSTAL_CHAINS_CRYSTALIZE, 5000);
188 break;
189 case EVENT_TAIL_SWEEP:
190 Talk(SAY_CRYSTAL_NOVA);
191 if (IsHeroic())
192 DoCast(me, SPELL_CRYSTALIZE);
193 else if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
194 DoCast(target, SPELL_CRYSTAL_CHAINS);
195 events.ScheduleEvent(EVENT_TAIL_SWEEP, DUNGEON_MODE(30000, 11000));
196 break;
197 default:
198 break;
199 }
200 }
201
202 DoMeleeAttackIfReady();
203 }
204
205 private:
206 bool _intenseCold;
207 bool _enrage;
208 ObjectGuid ContainmentSphereGUIDs[DATA_CONTAINMENT_SPHERES];
209 public:
210 GuidList _intenseColdList;
211 };
212
213 CreatureAI* GetAI(Creature* creature) const override
214 {
215 return GetInstanceAI<boss_keristraszaAI>(creature);
216 }

219class containment_sphere : public GameObjectScript
220{
221public:
222 containment_sphere() : GameObjectScript("containment_sphere") { }
223
224 bool OnGossipHello(Player* /*player*/, GameObject* go) override
225 {
226 InstanceScript* instance = go->GetInstanceScript();
227
228 Creature* pKeristrasza = ObjectAccessor::GetCreature(*go, instance->GetGuidData(DATA_KERISTRASZA));
229 if (pKeristrasza && pKeristrasza->IsAlive())
230 {
231 // maybe these are hacks :(
232 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
233 go->SetGoState(GO_STATE_ACTIVE);
234
235 ENSURE_AI(boss_keristrasza::boss_keristraszaAI, pKeristrasza->AI())->CheckContainmentSpheres(true);
236 }
237 return true;
238 }

242class spell_intense_cold : public SpellScriptLoader
243{
244 public:
245 spell_intense_cold() : SpellScriptLoader("spell_intense_cold") { }
246
247 class spell_intense_cold_AuraScript : public AuraScript
248 {
249 PrepareAuraScript(spell_intense_cold_AuraScript);
250
251 void HandlePeriodicTick(AuraEffect const* aurEff)
252 {
253 if (aurEff->GetBase()->GetStackAmount() < 2)
254 return;
255 Unit* caster = GetCaster();
256 /// @todo the caster should be boss but not the player
257 if (!caster || !caster->GetAI())
258 return;
259 caster->GetAI()->SetGUID(GetTarget()->GetGUID(), DATA_INTENSE_COLD);
260 }
261
262 void Register() override
263 {
264 OnEffectPeriodic += AuraEffectPeriodicFn(spell_intense_cold_AuraScript::HandlePeriodicTick, EFFECT_1, SPELL_AURA_PERIODIC_DAMAGE);
265 }
266 };
267
268 AuraScript* GetAuraScript() const override
269 {
270 return new spell_intense_cold_AuraScript();
271 }
272};

274class achievement_intense_cold : public AchievementCriteriaScript
275{
276 public:
277 achievement_intense_cold() : AchievementCriteriaScript("achievement_intense_cold")
278 {
279 }
280
281 bool OnCheck(Player* player, Unit* target) override
282 {
283 if (!target)
284 return false;
285
286 GuidList _intenseColdList = ENSURE_AI(boss_keristrasza::boss_keristraszaAI, target->ToCreature()->AI())->_intenseColdList;
287 if (!_intenseColdList.empty())
288 for (GuidList::iterator itr = _intenseColdList.begin(); itr != _intenseColdList.end(); ++itr)
289 if (player->GetGUID() == *itr)
290 return false;
291
292 return true;
293 }
294};

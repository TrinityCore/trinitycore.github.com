19#include "ScriptMgr.h"
20#include "InstanceScript.h"
21#include "Player.h"
22#include "nexus.h"
23
24enum Factions
25{
26 FACTION_HOSTILE_FOR_ALL = 16
27};
28
29class instance_nexus : public InstanceMapScript
30{
31 public:
32 instance_nexus() : InstanceMapScript(NexusScriptName, 576) { }
33
34 struct instance_nexus_InstanceMapScript : public InstanceScript
35 {
36 instance_nexus_InstanceMapScript(Map* map) : InstanceScript(map)
37 {
38 SetHeaders(DataHeader);
39 SetBossNumber(EncounterCount);
40 _teamInInstance = 0;
41 }
42
43 void OnPlayerEnter(Player* player) override
44 {
45 if (!_teamInInstance)
46 _teamInInstance = player->GetTeam();
47 }
48
49 void OnCreatureCreate(Creature* creature) override
50 {
51 switch (creature->GetEntry())
52 {
53 case NPC_ANOMALUS:
54 AnomalusGUID = creature->GetGUID();
55 break;
56 case NPC_KERISTRASZA:
57 KeristraszaGUID = creature->GetGUID();
58 break;
59 // Alliance npcs are spawned by default, if you are alliance, you will fight against horde npcs.
60 case NPC_ALLIANCE_BERSERKER:
61 if (ServerAllowsTwoSideGroups())
62 creature->setFaction(FACTION_HOSTILE_FOR_ALL);
63 if (_teamInInstance == ALLIANCE)
64 creature->UpdateEntry(NPC_HORDE_BERSERKER);
65 break;
66 case NPC_ALLIANCE_RANGER:
67 if (ServerAllowsTwoSideGroups())
68 creature->setFaction(FACTION_HOSTILE_FOR_ALL);
69 if (_teamInInstance == ALLIANCE)
70 creature->UpdateEntry(NPC_HORDE_RANGER);
71 break;
72 case NPC_ALLIANCE_CLERIC:
73 if (ServerAllowsTwoSideGroups())
74 creature->setFaction(FACTION_HOSTILE_FOR_ALL);
75 if (_teamInInstance == ALLIANCE)
76 creature->UpdateEntry(NPC_HORDE_CLERIC);
77 break;
78 case NPC_ALLIANCE_COMMANDER:
79 if (ServerAllowsTwoSideGroups())
80 creature->setFaction(FACTION_HOSTILE_FOR_ALL);
81 if (_teamInInstance == ALLIANCE)
82 creature->UpdateEntry(NPC_HORDE_COMMANDER);
83 break;
84 case NPC_COMMANDER_STOUTBEARD:
85 if (ServerAllowsTwoSideGroups())
86 creature->setFaction(FACTION_HOSTILE_FOR_ALL);
87 if (_teamInInstance == ALLIANCE)
88 creature->UpdateEntry(NPC_COMMANDER_KOLURG);
89 break;
90 default:
91 break;
92 }
93 }
94
95 void OnGameObjectCreate(GameObject* go) override
96 {
97 switch (go->GetEntry())
98 {
99 case GO_ANOMALUS_CONTAINMET_SPHERE:
100 AnomalusContainmentSphere = go->GetGUID();
101 if (GetBossState(DATA_ANOMALUS) == DONE)
102 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
103 break;
104 case GO_ORMOROKS_CONTAINMET_SPHERE:
105 OrmoroksContainmentSphere = go->GetGUID();
106 if (GetBossState(DATA_ORMOROK) == DONE)
107 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
108 break;
109 case GO_TELESTRAS_CONTAINMET_SPHERE:
110 TelestrasContainmentSphere = go->GetGUID();
111 if (GetBossState(DATA_MAGUS_TELESTRA) == DONE)
112 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
113 break;
114 default:
115 break;
116 }
117 }
118
119 bool SetBossState(uint32 type, EncounterState state) override
120 {
121 if (!InstanceScript::SetBossState(type, state))
122 return false;
123
124 switch (type)
125 {
126 case DATA_MAGUS_TELESTRA:
127 if (state == DONE)
128 {
129 if (GameObject* sphere = instance->GetGameObject(TelestrasContainmentSphere))
130 sphere->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
131 }
132 break;
133 case DATA_ANOMALUS:
134 if (state == DONE)
135 {
136 if (GameObject* sphere = instance->GetGameObject(AnomalusContainmentSphere))
137 sphere->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
138 }
139 break;
140 case DATA_ORMOROK:
141 if (state == DONE)
142 {
143 if (GameObject* sphere = instance->GetGameObject(OrmoroksContainmentSphere))
144 sphere->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
145 }
146 break;
147 default:
148 break;
149 }
150
151 return true;
152 }
153
154 ObjectGuid GetGuidData(uint32 type) const override
155 {
156 switch (type)
157 {
158 case DATA_ANOMALUS:
159 return AnomalusGUID;
160 case DATA_KERISTRASZA:
161 return KeristraszaGUID;
162 case ANOMALUS_CONTAINMET_SPHERE:
163 return AnomalusContainmentSphere;
164 case ORMOROKS_CONTAINMET_SPHERE:
165 return OrmoroksContainmentSphere;
166 case TELESTRAS_CONTAINMET_SPHERE:
167 return TelestrasContainmentSphere;
168 default:
169 break;
170 }
171
172 return ObjectGuid::Empty;
173 }
174
175 private:
176 ObjectGuid AnomalusGUID;
177 ObjectGuid KeristraszaGUID;
178 ObjectGuid AnomalusContainmentSphere;
179 ObjectGuid OrmoroksContainmentSphere;
180 ObjectGuid TelestrasContainmentSphere;
181 uint32 _teamInInstance;
182 };
183
184 InstanceScript* GetInstanceScript(InstanceMap* map) const override
185 {
186 return new instance_nexus_InstanceMapScript(map);
187 }
188};
189
190void AddSC_instance_nexus()
191{
192 new instance_nexus();
193}

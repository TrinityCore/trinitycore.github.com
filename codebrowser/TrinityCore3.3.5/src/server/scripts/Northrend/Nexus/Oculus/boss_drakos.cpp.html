53class boss_drakos : public CreatureScript
54{
55 public:
56 boss_drakos() : CreatureScript("boss_drakos") { }
57
58 struct boss_drakosAI : public BossAI
59 {
60 boss_drakosAI(Creature* creature) : BossAI(creature, DATA_DRAKOS)
61 {
62 Initialize();
63 }
64
65 void Initialize()
66 {
67 postPull = false;
68 }
69
70 void Reset() override
71 {
72 _Reset();
73
74 events.ScheduleEvent(EVENT_MAGIC_PULL, 15000);
75 events.ScheduleEvent(EVENT_STOMP, 17000);
76 events.ScheduleEvent(EVENT_BOMB_SUMMON, 2000);
77
78 Initialize();
79 }
80
81 void EnterCombat(Unit* /*who*/) override
82 {
83 _EnterCombat();
84 Talk(SAY_AGGRO);
85 }
86
87 void UpdateAI(uint32 diff) override
88 {
89 if (!UpdateVictim())
90 return;
91
92 events.Update(diff);
93
94 if (me->HasUnitState(UNIT_STATE_CASTING))
95 return;
96
97 while (uint32 eventId = events.ExecuteEvent())
98 {
99 switch (eventId)
100 {
101 case EVENT_BOMB_SUMMON:
102 {
103 for (uint8 i = 0; i <= (postPull ? 3 : 0); i++)
104 {
105 Position position = me->GetRandomNearPosition(frand(0.0f, 10.0f));
106 me->SummonCreature(NPC_UNSTABLE_SPHERE, position);
107 }
108 }
109 events.ScheduleEvent(EVENT_BOMB_SUMMON, 2000);
110 break;
111 case EVENT_MAGIC_PULL:
112 DoCast(SPELL_MAGIC_PULL);
113 postPull = true;
114 events.ScheduleEvent(EVENT_MAGIC_PULL, 15000);
115 break;
116 case EVENT_STOMP:
117 Talk(SAY_STOMP);
118 DoCast(SPELL_THUNDERING_STOMP);
119 events.ScheduleEvent(EVENT_STOMP, 17000);
120 break;
121 default:
122 break;
123 }
124 }
125
126 DoMeleeAttackIfReady();
127 }
128
129 void JustDied(Unit* /*killer*/) override
130 {
131 _JustDied();
132
133 Talk(SAY_DEATH);
134
135 // start achievement timer (kill Eregos within 20 min)
136 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMED_START_EVENT);
137 }
138
139 void KilledUnit(Unit* /*victim*/) override
140 {
141 Talk(SAY_KILL);
142 }
143
144 private:
145 bool postPull;
146 };
147
148 CreatureAI* GetAI(Creature* creature) const override
149 {
150 return GetOculusAI<boss_drakosAI>(creature);
151 }
152};

154class npc_unstable_sphere : public CreatureScript
155{
156 public:
157 npc_unstable_sphere() : CreatureScript("npc_unstable_sphere") { }
158
159 struct npc_unstable_sphereAI : public ScriptedAI
160 {
161 npc_unstable_sphereAI(Creature* creature) : ScriptedAI(creature)
162 {
163 Initialize();
164 }
165
166 void Initialize()
167 {
168 pulseTimer = 3000;
169 }
170
171 void Reset() override
172 {
173 me->SetReactState(REACT_PASSIVE);
174 me->GetMotionMaster()->MoveRandom(40.0f);
175
176 me->AddAura(SPELL_UNSTABLE_SPHERE_PASSIVE, me);
177 me->AddAura(SPELL_UNSTABLE_SPHERE_TIMER, me);
178
179 Initialize();
180
181 me->DespawnOrUnsummon(19000);
182 }
183
184 void UpdateAI(uint32 diff) override
185 {
186 if (pulseTimer <= diff)
187 {
188 DoCast(SPELL_UNSTABLE_SPHERE_PULSE);
189 pulseTimer = 3 * IN_MILLISECONDS;
190 }
191 else
192 pulseTimer -= diff;
193 }
194
195 private:
196 uint32 pulseTimer;
197 };
198
199 CreatureAI* GetAI(Creature* creature) const override
200 {
201 return new npc_unstable_sphereAI(creature);
202 }
203};

89class boss_urom : public CreatureScript
90{
91 public:
92 boss_urom() : CreatureScript("boss_urom") { }
93
94 struct boss_uromAI : public BossAI
95 {
96 boss_uromAI(Creature* creature) : BossAI(creature, DATA_UROM)
97 {
98 Initialize();
99 platform = 0;
100
101 for (uint8 i = 0; i < 3; ++i)
102 group[i] = i;
103
104 std::random_shuffle(group, group + 3);
105 }
106
107 void Initialize()
108 {
109 x = 0.0f;
110 y = 0.0f;
111 canCast = false;
112 canGoBack = false;
113
114 teleportTimer = urand(30000, 35000);
115 arcaneExplosionTimer = 9000;
116 castArcaneExplosionTimer = 2000;
117 frostBombTimer = urand(5000, 8000);
118 timeBombTimer = urand(20000, 25000);
119 }
120
121 void Reset() override
122 {
123 me->CastSpell(me, SPELL_EVOCATE);
124
125 _Reset();
126
127 me->GetMotionMaster()->MoveIdle();
128 }
129
130 void EnterCombat(Unit* /*who*/) override
131 {
132 _EnterCombat();
133
134 StartAttack();
135 }
136
137 void AttackStart(Unit* who) override
138 {
139 if (!who)
140 return;
141
142 if (me->GetPositionZ() > 518.63f)
143 DoStartNoMovement(who);
144 else
145 BossAI::AttackStart(who);
146 }
147
148 void SetPosition(uint8 i)
149 {
150 switch (i)
151 {
152 case 0:
153 x = me->GetPositionX() + 4;
154 y = me->GetPositionY() - 4;
155 break;
156 case 1:
157 x = me->GetPositionX() + 4;
158 y = me->GetPositionY() + 4;
159 break;
160 case 2:
161 x = me->GetPositionX() - 4;
162 y = me->GetPositionY() + 4;
163 break;
164 case 3:
165 x = me->GetPositionX() - 4;
166 y = me->GetPositionY() - 4;
167 break;
168 default:
169 break;
170 }
171 }
172
173 void StartAttack()
174 {
175 if (platform > 2)
176 {
177 Talk(SAY_AGGRO);
178 return;
179 }
180
181 // summon guards and jump to next platform
182 for (uint8 i = 0; i < 4; ++i)
183 {
184 SetPosition(i);
185 me->SummonCreature(Group[group[platform]].entry[i], x, y, me->GetPositionZ(), me->GetOrientation());
186 }
187
188 Talk(platform);
189 DoCast(TeleportSpells[platform]);
190
191 ++platform;
192 }
193
194 void KilledUnit(Unit* who) override
195 {
196 if (who->GetTypeId() == TYPEID_PLAYER)
197 Talk(SAY_PLAYER_KILL);
198 }
199
200 void UpdateAI(uint32 diff) override
201 {
202 if (!UpdateVictim())
203 return;
204
205 if (platform < 3)
206 return;
207
208 events.Update(diff);
209
210 if (teleportTimer <= diff)
211 {
212 me->InterruptNonMeleeSpells(false);
213 me->GetMotionMaster()->MoveIdle();
214 DoCast(SPELL_TELEPORT);
215 teleportTimer = urand(30000, 35000);
216 }
217 else
218 teleportTimer -= diff;
219
220 if (canCast && !me->FindCurrentSpellBySpellId(SPELL_EMPOWERED_ARCANE_EXPLOSION))
221 {
222 if (castArcaneExplosionTimer <= diff)
223 {
224 canCast = false;
225 canGoBack = true;
226 DoCastAOE(SPELL_EMPOWERED_ARCANE_EXPLOSION);
227 castArcaneExplosionTimer = 2000;
228 }
229 else
230 castArcaneExplosionTimer -= diff;
231 }
232
233 if (canGoBack)
234 {
235 if (arcaneExplosionTimer <= diff)
236 {
237 if (me->GetVictim())
238 {
239 Position pos = me->EnsureVictim()->GetPosition();
240
241 me->NearTeleportTo(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ(), pos.GetOrientation());
242 me->GetMotionMaster()->MoveChase(me->GetVictim());
243 }
244 me->SetWalk(true);
245
246 Talk(EMOTE_ARCANE_EXPLOSION);
247 Talk(SAY_ARCANE_EXPLOSION);
248
249 canCast = false;
250 canGoBack = false;
251 arcaneExplosionTimer = 9000;
252 }
253 else
254 arcaneExplosionTimer -= diff;
255 }
256
257 if (!me->IsNonMeleeSpellCast(false, true, true))
258 {
259 if (frostBombTimer <= diff)
260 {
261 DoCastVictim(SPELL_FROSTBOMB);
262 frostBombTimer = urand(5000, 8000);
263 }
264 else
265 frostBombTimer -= diff;
266
267 if (timeBombTimer <= diff)
268 {
269 if (Unit* unit = SelectTarget(SELECT_TARGET_RANDOM))
270 DoCast(unit, SPELL_TIME_BOMB);
271
272 timeBombTimer = urand(20000, 25000);
273 }
274 else
275 timeBombTimer -= diff;
276 }
277
278 DoMeleeAttackIfReady();
279 }

288 void LeaveCombat()
289 {

291 me->CombatStop(false);
292 me->DeleteThreatList();
293 }
294
295 void SpellHit(Unit* /*caster*/, SpellInfo const* spellInfo) override
296 {
297 switch (spellInfo->Id)
298 {
299 case SPELL_SUMMON_MENAGERIE:
300 me->SetHomePosition(968.66f, 1042.53f, 527.32f, 0.077f);
301 LeaveCombat();
302 me->CastSpell(me, SPELL_EVOCATE);
303 break;
304 case SPELL_SUMMON_MENAGERIE_2:
305 me->SetHomePosition(1164.02f, 1170.85f, 527.321f, 3.66f);
306 LeaveCombat();
307 me->CastSpell(me, SPELL_EVOCATE);
308 break;
309 case SPELL_SUMMON_MENAGERIE_3:
310 me->SetHomePosition(1118.31f, 1080.377f, 508.361f, 4.25f);
311 LeaveCombat();
312 me->CastSpell(me, SPELL_EVOCATE);
313 break;
314 case SPELL_TELEPORT:
315 //! Unconfirmed, previous below
316 me->SetDisableGravity(true);
317 //me->AddUnitMovementFlag(MOVEMENTFLAG_CAN_FLY); // with out it the npc will fall down while is casting
318 canCast = true;
319 break;
320 default:
321 break;
322 }
323 }

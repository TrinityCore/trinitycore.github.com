86class boss_general_zarithrian : public CreatureScript
87{
88 public:
89 boss_general_zarithrian() : CreatureScript("boss_general_zarithrian") { }
90
91 struct boss_general_zarithrianAI : public BossAI
92 {
93 boss_general_zarithrianAI(Creature* creature) : BossAI(creature, DATA_GENERAL_ZARITHRIAN)
94 {
95 }
96
97 void Reset() override
98 {
99 _Reset();
100 if (instance->GetBossState(DATA_SAVIANA_RAGEFIRE) == DONE && instance->GetBossState(DATA_BALTHARUS_THE_WARBORN) == DONE)
101 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
102 }
103
104 void EnterCombat(Unit* /*who*/) override
105 {
106 _EnterCombat();
107 Talk(SAY_AGGRO);
108 events.Reset();
109 events.ScheduleEvent(EVENT_CLEAVE, 15000);
110 events.ScheduleEvent(EVENT_INTIDMDATING_ROAR, 42000);
111 events.ScheduleEvent(EVENT_SUMMON_ADDS, 40000);
112 }
113
114 void JustReachedHome() override
115 {
116 _JustReachedHome();
117 instance->SetBossState(DATA_GENERAL_ZARITHRIAN, FAIL);
118 }
119
120 // Override to not set adds in combat yet.
121 void JustSummoned(Creature* summon) override
122 {
123 summons.Summon(summon);
124 }
125
126 void JustDied(Unit* /*killer*/) override
127 {
128 _JustDied();
129 Talk(SAY_DEATH);
130 }
131
132 void KilledUnit(Unit* victim) override
133 {
134 if (victim->GetTypeId() == TYPEID_PLAYER)
135 Talk(SAY_KILL);
136 }
137
138 bool CanAIAttack(Unit const* target) const override
139 {
140 return (instance->GetBossState(DATA_SAVIANA_RAGEFIRE) == DONE && instance->GetBossState(DATA_BALTHARUS_THE_WARBORN) == DONE && BossAI::CanAIAttack(target));
141 }
142
143 void UpdateAI(uint32 diff) override
144 {
145 if (!UpdateVictim())
146 return;
147
148 events.Update(diff);
149
150 if (me->HasUnitState(UNIT_STATE_CASTING))
151 return;
152
153 while (uint32 eventId = events.ExecuteEvent())
154 {
155 switch (eventId)
156 {
157 case EVENT_SUMMON_ADDS:
158 {
159 if (Creature* stalker1 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ZARITHRIAN_SPAWN_STALKER_1)))
160 stalker1->CastSpell(stalker1, SPELL_SUMMON_FLAMECALLER, false);
161
162 if (Creature* stalker2 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ZARITHRIAN_SPAWN_STALKER_2)))
163 stalker2->CastSpell(stalker2, SPELL_SUMMON_FLAMECALLER, false);
164
165 Talk(SAY_ADDS);
166 events.ScheduleEvent(EVENT_SUMMON_ADDS, 42000);
167 break;
168 }
169 case EVENT_INTIDMDATING_ROAR:
170 DoCast(me, SPELL_INTIMIDATING_ROAR, true);
171 events.ScheduleEvent(EVENT_INTIDMDATING_ROAR, 42000);
172 case EVENT_CLEAVE:
173 DoCastVictim(SPELL_CLEAVE_ARMOR);
174 events.ScheduleEvent(EVENT_CLEAVE, 15000);
175 break;
176 default:
177 break;
178 }
179 }
180
181 DoMeleeAttackIfReady();
182 }
183 };
184
185 CreatureAI* GetAI(Creature* creature) const override
186 {
187 return GetRubySanctumAI<boss_general_zarithrianAI>(creature);
188 }

191class npc_onyx_flamecaller : public CreatureScript
192{
193 public:
194 npc_onyx_flamecaller() : CreatureScript("npc_onyx_flamecaller") { }
195
196 struct npc_onyx_flamecallerAI : public npc_escortAI
197 {
198 npc_onyx_flamecallerAI(Creature* creature) : npc_escortAI(creature), _instance(creature->GetInstanceScript())
199 {
200 Initialize();
201 npc_escortAI::SetDespawnAtEnd(false);
202 }
203
204 void Initialize()
205 {
206 _lavaGoutCount = 0;
207 }
208
209 void Reset() override
210 {
211 Initialize();
212 me->setActive(true);
213 AddWaypoints();
214 Start(true, true);
215 }
216
217 void EnterCombat(Unit* /*who*/) override
218 {
219 _events.Reset();
220 _events.ScheduleEvent(EVENT_BLAST_NOVA, urand(20000, 30000));
221 _events.ScheduleEvent(EVENT_LAVA_GOUT, 5000);
222 }
223
224 void EnterEvadeMode(EvadeReason /*why*/) override
225 {
226 // Prevent EvadeMode
227 }
228
229 void IsSummonedBy(Unit* /*summoner*/) override
230 {
231 // Let Zarithrian count as summoner.
232 if (Creature* zarithrian = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_GENERAL_ZARITHRIAN)))
233 zarithrian->AI()->JustSummoned(me);
234 }
235
236 void WaypointReached(uint32 waypointId) override
237 {
238 if (waypointId == MAX_PATH_FLAMECALLER_WAYPOINTS - 1 || waypointId == MAX_PATH_FLAMECALLER_WAYPOINTS * 2 - 1)
239 {
240 DoZoneInCombat();
241 SetEscortPaused(true);
242 }
243 }
244
245 void AddWaypoints()
246 {
247 if (me->GetPositionY() < 500.0f)
248 {
249 for (uint8 i = 0; i < MAX_PATH_FLAMECALLER_WAYPOINTS; i++)
250 AddWaypoint(i, FlamecallerWaypoints[i].GetPositionX(), FlamecallerWaypoints[i].GetPositionY(), FlamecallerWaypoints[i].GetPositionZ());
251 }
252 else
253 {
254 for (uint8 i = 0, j = MAX_PATH_FLAMECALLER_WAYPOINTS; j < MAX_PATH_FLAMECALLER_WAYPOINTS*2; j++, i++)
255 AddWaypoint(i, FlamecallerWaypoints[j].GetPositionX(), FlamecallerWaypoints[j].GetPositionY(), FlamecallerWaypoints[j].GetPositionZ());
256 }
257 }
258
259 void UpdateEscortAI(uint32 diff) override
260 {
261 if (!UpdateVictim())
262 return;
263
264 _events.Update(diff);
265
266 if (me->HasUnitState(UNIT_STATE_CASTING))
267 return;
268
269 while (uint32 eventId = _events.ExecuteEvent())
270 {
271 switch (eventId)
272 {
273 case EVENT_BLAST_NOVA:
274 DoCastAOE(SPELL_BLAST_NOVA);
275 _events.ScheduleEvent(EVENT_BLAST_NOVA, urand(20000, 30000));
276 break;
277 case EVENT_LAVA_GOUT:
278 if (_lavaGoutCount >= 3)
279 {
280 _lavaGoutCount = 0;
281 _events.ScheduleEvent(EVENT_LAVA_GOUT, 8000);
282 break;
283 }
284 DoCastVictim(SPELL_LAVA_GOUT);
285 _lavaGoutCount++;
286 _events.ScheduleEvent(EVENT_LAVA_GOUT, 1500);
287 break;
288 default:
289 break;
290 }
291 }
292
293 DoMeleeAttackIfReady();
294 }
295 private:
296 EventMap _events;
297 InstanceScript* _instance;
298 uint8 _lavaGoutCount;
299 };
300
301 CreatureAI* GetAI(Creature* creature) const override
302 {
303 return GetRubySanctumAI<npc_onyx_flamecallerAI>(creature);
304 }
305};

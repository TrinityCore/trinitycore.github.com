67class boss_baltharus_the_warborn : public CreatureScript
68{
69 public:
70 boss_baltharus_the_warborn() : CreatureScript("boss_baltharus_the_warborn") { }
71
72 struct boss_baltharus_the_warbornAI : public BossAI
73 {
74 boss_baltharus_the_warbornAI(Creature* creature) : BossAI(creature, DATA_BALTHARUS_THE_WARBORN)
75 {
76 Initialize();
77 _introDone = false;
78 }
79
80 void Initialize()
81 {
82 _cloneCount = RAID_MODE<uint8>(1, 2, 2, 2);
83 }
84
85 void Reset() override
86 {
87 _Reset();
88 events.SetPhase(PHASE_INTRO);
89 events.ScheduleEvent(EVENT_OOC_CHANNEL, 0, 0, PHASE_INTRO);
90 Initialize();
91 instance->SetData(DATA_BALTHARUS_SHARED_HEALTH, me->GetMaxHealth());
92 }
93
94 void DoAction(int32 action) override
95 {
96 switch (action)
97 {
98 case ACTION_INTRO_BALTHARUS:
99 if (_introDone)
100 return;
101 _introDone = true;
102 me->setActive(true);
103 events.ScheduleEvent(EVENT_INTRO_TALK, 7000, 0, PHASE_INTRO);
104 break;
105 case ACTION_CLONE:
106 {
107 DoCast(me, SPELL_CLEAR_DEBUFFS);
108 DoCast(me, SPELL_CLONE);
109 DoCast(me, SPELL_REPELLING_WAVE);
110 Talk(SAY_CLONE);
111 --_cloneCount;
112 break;
113 }
114 default:
115 break;
116 }
117 }
118
119 void EnterCombat(Unit* /*who*/) override
120 {
121 me->InterruptNonMeleeSpells(false);
122 _EnterCombat();
123 events.Reset();
124 events.SetPhase(PHASE_COMBAT);
125 events.ScheduleEvent(EVENT_CLEAVE, 11000, 0, PHASE_COMBAT);
126 events.ScheduleEvent(EVENT_ENERVATING_BRAND, 13000, 0, PHASE_COMBAT);
127 events.ScheduleEvent(EVENT_BLADE_TEMPEST, 15000, 0, PHASE_COMBAT);
128 Talk(SAY_AGGRO);
129 }
130
131 void JustDied(Unit* /*killer*/) override
132 {
133 _JustDied();
134 Talk(SAY_DEATH);
135 if (Creature* xerestrasza = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_XERESTRASZA)))
136 xerestrasza->AI()->DoAction(ACTION_BALTHARUS_DEATH);
137 }
138
139 void KilledUnit(Unit* victim) override
140 {
141 if (victim->GetTypeId() == TYPEID_PLAYER)
142 Talk(SAY_KILL);
143 }
144
145 void JustSummoned(Creature* summon) override
146 {
147 summons.Summon(summon);
148 summon->SetHealth(me->GetHealth());
149 summon->CastSpell(summon, SPELL_SPAWN_EFFECT, true);
150 }
151
152 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
153 {
154 if (GetDifficulty() == RAID_DIFFICULTY_10MAN_NORMAL)
155 {
156 if (me->HealthBelowPctDamaged(50, damage) && _cloneCount == 1)
157 DoAction(ACTION_CLONE);
158 }
159 else
160 {
161 if (me->HealthBelowPctDamaged(66, damage) && _cloneCount == 2)
162 DoAction(ACTION_CLONE);
163 else if (me->HealthBelowPctDamaged(33, damage) && _cloneCount == 1)
164 DoAction(ACTION_CLONE);
165 }
166
167 if (me->GetHealth() > damage)
168 instance->SetData(DATA_BALTHARUS_SHARED_HEALTH, me->GetHealth() - damage);
169 }
170
171 void UpdateAI(uint32 diff) override
172 {
173 bool introPhase = events.IsInPhase(PHASE_INTRO);
174 if (!UpdateVictim() && !introPhase)
175 return;
176
177 if (!introPhase)
178 me->SetHealth(instance->GetData(DATA_BALTHARUS_SHARED_HEALTH));
179
180 events.Update(diff);
181
182 if (me->HasUnitState(UNIT_STATE_CASTING) && !introPhase)
183 return;
184
185 while (uint32 eventId = events.ExecuteEvent())
186 {
187 switch (eventId)
188 {
189 case EVENT_INTRO_TALK:
190 Talk(SAY_BALTHARUS_INTRO);
191 break;
192 case EVENT_OOC_CHANNEL:
193 if (Creature* channelTarget = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_CRYSTAL_CHANNEL_TARGET)))
194 DoCast(channelTarget, SPELL_BARRIER_CHANNEL);
195 events.ScheduleEvent(EVENT_OOC_CHANNEL, 7000, 0, PHASE_INTRO);
196 break;
197 case EVENT_CLEAVE:
198 DoCastVictim(SPELL_CLEAVE);
199 events.ScheduleEvent(EVENT_CLEAVE, 24000, 0, PHASE_COMBAT);
200 break;
201 case EVENT_BLADE_TEMPEST:
202 DoCast(me, SPELL_BLADE_TEMPEST);
203 events.ScheduleEvent(EVENT_BLADE_TEMPEST, 24000, 0, PHASE_COMBAT);
204 break;
205 case EVENT_ENERVATING_BRAND:
206 for (uint8 i = 0; i < RAID_MODE<uint8>(4, 8, 8, 10); i++)
207 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
208 DoCast(target, SPELL_ENERVATING_BRAND);
209 events.ScheduleEvent(EVENT_ENERVATING_BRAND, 26000, 0, PHASE_COMBAT);
210 break;
211 default:
212 break;
213 }
214 }
215
216 DoMeleeAttackIfReady();
217 }
218
219 private:
220 uint8 _cloneCount;
221 bool _introDone;
222 };
223
224 CreatureAI* GetAI(Creature* creature) const override
225 {
226 return GetRubySanctumAI<boss_baltharus_the_warbornAI>(creature);
227 }
228};

230class npc_baltharus_the_warborn_clone : public CreatureScript
231{
232 public:
233 npc_baltharus_the_warborn_clone() : CreatureScript("npc_baltharus_the_warborn_clone") { }
234
235 struct npc_baltharus_the_warborn_cloneAI : public ScriptedAI
236 {
237 npc_baltharus_the_warborn_cloneAI(Creature* creature) : ScriptedAI(creature),
238 _instance(creature->GetInstanceScript())
239 {
240 }
241
242 void EnterCombat(Unit* /*who*/) override
243 {
244 DoZoneInCombat();
245 _events.Reset();
246 _events.ScheduleEvent(EVENT_CLEAVE, urand(5000, 10000));
247 _events.ScheduleEvent(EVENT_BLADE_TEMPEST, urand(18000, 25000));
248 _events.ScheduleEvent(EVENT_ENERVATING_BRAND, urand(10000, 15000));
249 }
250
251 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
252 {
253 // Setting DATA_BALTHARUS_SHARED_HEALTH to 0 when killed would bug the boss.
254 if (me->GetHealth() > damage)
255 _instance->SetData(DATA_BALTHARUS_SHARED_HEALTH, me->GetHealth() - damage);
256 }
257
258 void JustDied(Unit* killer) override
259 {
260 // This is here because DamageTaken wont trigger if the damage is deadly.
261 if (Creature* baltharus = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_BALTHARUS_THE_WARBORN)))
262 killer->Kill(baltharus);
263 }
264
265 void UpdateAI(uint32 diff) override
266 {
267 if (!UpdateVictim())
268 return;
269
270 me->SetHealth(_instance->GetData(DATA_BALTHARUS_SHARED_HEALTH));
271
272 _events.Update(diff);
273
274 if (me->HasUnitState(UNIT_STATE_CASTING))
275 return;
276
277 while (uint32 eventId = _events.ExecuteEvent())
278 {
279 switch (eventId)
280 {
281 case EVENT_CLEAVE:
282 DoCastVictim(SPELL_CLEAVE);
283 _events.ScheduleEvent(EVENT_CLEAVE, 24000);
284 break;
285 case EVENT_BLADE_TEMPEST:
286 DoCastVictim(SPELL_BLADE_TEMPEST);
287 _events.ScheduleEvent(EVENT_BLADE_TEMPEST, 24000);
288 break;
289 case EVENT_ENERVATING_BRAND:
290 for (uint8 i = 0; i < RAID_MODE<uint8>(4, 8, 8, 10); i++)
291 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
292 DoCast(target, SPELL_ENERVATING_BRAND);
293 _events.ScheduleEvent(EVENT_ENERVATING_BRAND, 26000);
294 break;
295 default:
296 break;
297 }
298 }
299
300 DoMeleeAttackIfReady();
301 }
302
303 private:
304 EventMap _events;
305 InstanceScript* _instance;
306 };
307
308 CreatureAI* GetAI(Creature* creature) const override
309 {
310 return GetRubySanctumAI<npc_baltharus_the_warborn_cloneAI>(creature);
311 }

315{
316 public:
317 spell_baltharus_enervating_brand_trigger() : SpellScriptLoader("spell_baltharus_enervating_brand_trigger") { }
318
319 class spell_baltharus_enervating_brand_trigger_SpellScript : public SpellScript
320 {
321 PrepareSpellScript(spell_baltharus_enervating_brand_trigger_SpellScript);
322
323 void CheckDistance()
324 {
325 if (Unit* caster = GetOriginalCaster())
326 {
327 if (Unit* target = GetHitUnit())
328 target->CastSpell(caster, SPELL_SIPHONED_MIGHT, true);
329 }
330 }
331

333 {
334 OnHit += SpellHitFn(spell_baltharus_enervating_brand_trigger_SpellScript::CheckDistance);
335 }
336 };

340 return new spell_baltharus_enervating_brand_trigger_SpellScript();
341 }

48Position const xerestraszaMovePos = {3151.236f, 379.8733f, 86.31996f, 0.0f};

50class npc_xerestrasza : public CreatureScript
51{
52 public:
53 npc_xerestrasza() : CreatureScript("npc_xerestrasza") { }
54
55 struct npc_xerestraszaAI : public ScriptedAI
56 {
57 npc_xerestraszaAI(Creature* creature) : ScriptedAI(creature)
58 {
59 _isIntro = true;
60 _introDone = false;
61 }
62
63 void Reset() override
64 {
65 _events.Reset();
66 me->RemoveFlag(UNIT_NPC_FLAGS, GOSSIP_OPTION_QUESTGIVER);
67 }
68
69 void DoAction(int32 action) override
70 {
71 if (action == ACTION_BALTHARUS_DEATH)
72 {
73 me->setActive(true);
74 _isIntro = false;
75
76 Talk(SAY_XERESTRASZA_EVENT);
77 me->SetWalk(true);
78 me->GetMotionMaster()->MovePoint(0, xerestraszaMovePos);
79
80 _events.ScheduleEvent(EVENT_XERESTRASZA_EVENT_1, 16000);
81 _events.ScheduleEvent(EVENT_XERESTRASZA_EVENT_2, 25000);
82 _events.ScheduleEvent(EVENT_XERESTRASZA_EVENT_3, 32000);
83 _events.ScheduleEvent(EVENT_XERESTRASZA_EVENT_4, 42000);
84 _events.ScheduleEvent(EVENT_XERESTRASZA_EVENT_5, 51000);
85 _events.ScheduleEvent(EVENT_XERESTRASZA_EVENT_6, 61000);
86 _events.ScheduleEvent(EVENT_XERESTRASZA_EVENT_7, 69000);
87 }
88 else if (action == ACTION_INTRO_BALTHARUS && !_introDone)
89 {
90 _introDone = true;
91 Talk(SAY_XERESTRASZA_INTRO);
92 }
93 }
94
95 void UpdateAI(uint32 diff) override
96 {
97 if (_isIntro)
98 return;
99
100 _events.Update(diff);
101
102 while (uint32 eventId = _events.ExecuteEvent())
103 {
104 switch (eventId)
105 {
106 case EVENT_XERESTRASZA_EVENT_1:
107 Talk(SAY_XERESTRASZA_EVENT_1);
108 break;
109 case EVENT_XERESTRASZA_EVENT_2:
110 Talk(SAY_XERESTRASZA_EVENT_2);
111 break;
112 case EVENT_XERESTRASZA_EVENT_3:
113 Talk(SAY_XERESTRASZA_EVENT_3);
114 break;
115 case EVENT_XERESTRASZA_EVENT_4:
116 Talk(SAY_XERESTRASZA_EVENT_4);
117 break;
118 case EVENT_XERESTRASZA_EVENT_5:
119 Talk(SAY_XERESTRASZA_EVENT_5);
120 break;
121 case EVENT_XERESTRASZA_EVENT_6:
122 Talk(SAY_XERESTRASZA_EVENT_6);
123 break;
124 case EVENT_XERESTRASZA_EVENT_7:
125 me->SetFlag(UNIT_NPC_FLAGS, GOSSIP_OPTION_QUESTGIVER);
126 Talk(SAY_XERESTRASZA_EVENT_7);
127 me->setActive(false);
128 break;
129 default:
130 break;
131 }
132 }
133 }
134
135 private:
136 EventMap _events;
137 bool _isIntro;
138 bool _introDone;
139 };
140
141 CreatureAI* GetAI(Creature* creature) const override
142 {
143 return GetRubySanctumAI<npc_xerestraszaAI>(creature);
144 }
145};

147class at_baltharus_plateau : public AreaTriggerScript
148{
149 public:
150 at_baltharus_plateau() : AreaTriggerScript("at_baltharus_plateau") { }
151
152 bool OnTrigger(Player* player, AreaTriggerEntry const* /*areaTrigger*/) override
153 {
154 // Only trigger once
155 if (InstanceScript* instance = player->GetInstanceScript())
156 {
157 if (Creature* xerestrasza = ObjectAccessor::GetCreature(*player, instance->GetGuidData(DATA_XERESTRASZA)))
158 xerestrasza->AI()->DoAction(ACTION_INTRO_BALTHARUS);
159
160 if (Creature* baltharus = ObjectAccessor::GetCreature(*player, instance->GetGuidData(DATA_BALTHARUS_THE_WARBORN)))
161 baltharus->AI()->DoAction(ACTION_INTRO_BALTHARUS);
162 }
163
164 return true;
165 }
166};

168void AddSC_ruby_sanctum()

170 new npc_xerestrasza();
171 new at_baltharus_plateau();
172}

159// to control each dragons common abilities
160struct dummy_dragonAI : public ScriptedAI
161{
162 dummy_dragonAI(Creature* creature) : ScriptedAI(creature)
163 {
164 Initialize();
165 instance = creature->GetInstanceScript();
166 }
167
168 void Initialize()
169 {
170 waypointId = 0;
171 portalRespawnTime = 30000;
172 _canMoveFree = false;
173 _canLoot = true;
174 }
175
176 void Reset() override
177 {
178 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
179 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
180
181 events.Reset();
182 Initialize();
183 }
184
185 void EnterCombat(Unit* /*who*/) override
186 {
187 Talk(SAY_AGGRO);
188 DoZoneInCombat();
189
190 events.ScheduleEvent(EVENT_SHADOW_FISSURE, 5000);
191 events.ScheduleEvent(EVENT_SHADOW_BREATH, 20000);
192 }
193
194 void SetData(uint32 type, uint32 value) override
195 {
196 if (type == DATA_CAN_LOOT)
197 _canLoot = value != 0;
198 }
199
200 void MovementInform(uint32 type, uint32 pointId) override
201 {
202 if (!instance || type != POINT_MOTION_TYPE)
203 return;
204
205 // debug_log("dummy_dragonAI: %s reached point %u", me->GetName(), uiPointId);
206
207 // if healers messed up the raid and we was already initialized
208 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
209 {
210 EnterEvadeMode();
211 return;
212 }
213
214 // this is end, if we reach this, don't do much
215 if (pointId == POINT_ID_LAND)
216 {
217 me->GetMotionMaster()->Clear();
218 me->SetInCombatWithZone();
219 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0, true))
220 {
221 me->AddThreat(target, 1.0f);
222 me->Attack(target, true);
223 me->GetMotionMaster()->MoveChase(target);
224 }
225
226 _canMoveFree = false;
227 return;
228 }
229
230 // increase
231 waypointId = pointId + 1;
232
233 // if we have reached a point bigger or equal to count, it mean we must reset to point 0
234 if (waypointId >= MAX_WAYPOINT)
235 {
236 if (!_canMoveFree)
237 _canMoveFree = true;
238
239 waypointId = 0;
240 }
241
242 events.ScheduleEvent(EVENT_FREE_MOVEMENT, 500);
243 }
244
245 // "opens" the portal and does the "opening" whisper
246 void OpenPortal()
247 {
248 // there are 4 portal spawn locations, each are expected to be spawned with negative spawntimesecs in database
249
250 // using a grid search here seem to be more efficient than caching all four guids
251 // in instance script and calculate range to each.
252 GameObject* portal = me->FindNearestGameObject(GO_TWILIGHT_PORTAL, 50.0f);
253
254 switch (me->GetEntry())
255 {
256 case NPC_TENEBRON:
257 {
258 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
259 {
260 for (uint32 i = 0; i < 6; ++i)
261 me->SummonCreature(NPC_TWILIGHT_EGG, TwilightEggs[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000);
262 }
263 else
264 {
265 for (uint32 i = 0; i < 6; ++i)
266 me->SummonCreature(NPC_SARTHARION_TWILIGHT_EGG, TwilightEggsSarth[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000);
267 }
268 break;
269 }
270 case NPC_SHADRON:
271 {
272 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
273 me->SummonCreature(NPC_ACOLYTE_OF_SHADRON, AcolyteofShadron, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 28000);
274 else
275 me->SummonCreature(NPC_ACOLYTE_OF_SHADRON, AcolyteofShadron2, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 28000);
276
277 break;
278 }
279 case NPC_VESPERON:
280 {
281 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
282 {
283 if (Creature* acolyte = me->SummonCreature(NPC_ACOLYTE_OF_VESPERON, AcolyteofVesperon, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000))
284 {
285 me->InterruptNonMeleeSpells(true);
286 acolyte->InterruptNonMeleeSpells(true);
287 me->CastSpell(me, 32747, false);
288 }
289 }
290 else
291 {
292 if (Creature* acolyte = me->SummonCreature(NPC_ACOLYTE_OF_VESPERON, AcolyteofVesperon2, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000))
293 {
294 me->InterruptNonMeleeSpells(true);
295 acolyte->InterruptNonMeleeSpells(true);
296 me->CastSpell(me, 32747, false);
297 }
298 }
299
300 break;
301 }
302 }
303
304 Talk(WHISPER_OPEN_PORTAL);
305 Talk(WHISPER_OPENED_PORTAL);
306
307 // By using SetRespawnTime() we will actually "spawn" the object with our defined time.
308 // Once time is up, portal will disappear again.
309 if (portal && !portal->isSpawned())
310 portal->SetRespawnTime(portalRespawnTime);
311
312 // Unclear what are expected to happen if one drake has a portal open already
313 // Refresh respawnTime so time again are set to 30secs?
314 }
315
316 void KilledUnit(Unit* who) override
317 {
318 if (who->GetTypeId() == TYPEID_PLAYER)
319 Talk(SAY_SLAY);
320 }
321
322 void JustDied(Unit* /*killer*/) override
323 {
324 if (!_canLoot)
325 me->SetLootRecipient(NULL);
326
327 uint32 spellId = 0;
328
329 switch (me->GetEntry())
330 {
331 case NPC_TENEBRON:
332 spellId = SPELL_POWER_OF_TENEBRON;
333 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
334 instance->SetBossState(DATA_TENEBRON, DONE);
335 break;
336 case NPC_SHADRON:
337 spellId = SPELL_POWER_OF_SHADRON;
338 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
339 instance->SetBossState(DATA_SHADRON, DONE);
340 if (Creature* acolyte = me->FindNearestCreature(NPC_ACOLYTE_OF_SHADRON, 100.0f))
341 acolyte->KillSelf();
342 break;
343 case NPC_VESPERON:
344 spellId = SPELL_POWER_OF_VESPERON;
345 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
346 instance->SetBossState(DATA_VESPERON, DONE);
347 if (Creature* acolyte = me->FindNearestCreature(NPC_ACOLYTE_OF_VESPERON, 100.0f))
348 acolyte->KillSelf();
349 break;
350 }
351
352 Talk(SAY_DEATH);
353 me->RemoveAurasDueToSpell(spellId);
354
355 instance->DoRemoveAurasDueToSpellOnPlayers(spellId);
356
357 // not if solo mini-boss fight
358 if (instance->GetBossState(DATA_SARTHARION) != IN_PROGRESS)
359 return;
360
361 // Twilight Revenge to main boss
362 if (Unit* sartharion = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_SARTHARION)))
363 if (sartharion->IsAlive())
364 {
365 sartharion->RemoveAurasDueToSpell(spellId);
366 DoCast(sartharion, SPELL_TWILIGHT_REVENGE, true);
367 }
368 }
369
370 void UpdateAI(uint32 diff) override
371 {
372 events.Update(diff);
373
374 if (events.ExecuteEvent() == EVENT_FREE_MOVEMENT)
375 {
376 if (_canMoveFree && waypointId < MAX_WAYPOINT)
377 me->GetMotionMaster()->MovePoint(waypointId, dragonCommon[waypointId]);
378 }
379 }
380
381 void ExecuteEvent(uint32 eventId)
382 {
383 switch (eventId)
384 {
385 case EVENT_SHADOW_FISSURE:
386 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
387 DoCast(target, SPELL_SHADOW_FISSURE);
388 events.ScheduleEvent(eventId, urand(15000, 20000));
389 break;
390 case EVENT_SHADOW_BREATH:
391 Talk(SAY_BREATH);
392 DoCastVictim(SPELL_SHADOW_BREATH);
393 events.ScheduleEvent(eventId, urand(20000, 25000));
394 break;
395 default:
396 break;
397 }
398 }
399
400 protected:
401 InstanceScript* instance;
402 EventMap events;
403 uint32 waypointId;
404 int32 portalRespawnTime;
405 bool _canMoveFree;
406 bool _canLoot;
407};

457 }

460 }

618class npc_acolyte_of_shadron : public CreatureScript
619{
620 public:
621 npc_acolyte_of_shadron() : CreatureScript("npc_acolyte_of_shadron") { }
622
623 struct npc_acolyte_of_shadronAI : public ScriptedAI
624 {
625 npc_acolyte_of_shadronAI(Creature* creature) : ScriptedAI(creature)
626 {
627 instance = creature->GetInstanceScript();
628 }
629
630 void Reset() override
631 {
632 // Despawn the NPC automatically after 28 seconds
633 me->DespawnOrUnsummon(28000);
634
635 //if not solo fight, buff main boss, else place debuff on mini-boss. both spells TARGET_SCRIPT
636 if (instance->GetBossState(DATA_SARTHARION) == IN_PROGRESS)
637 {
638 if (Creature* sartharion = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SARTHARION)))
639 sartharion->AddAura(SPELL_GIFT_OF_TWILIGTH_SAR, sartharion);
640 }
641 else
642 {
643 if (Creature* shadron = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SHADRON)))
644 shadron->AddAura(SPELL_GIFT_OF_TWILIGTH_SHA, shadron);
645 }
646
647 me->AddAura(SPELL_TWILIGHT_SHIFT_ENTER, me);
648 }
649
650 void JustDied(Unit* /*killer*/) override
651 {
652 if (ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SHADRON)))
653 instance->SetBossState(DATA_PORTAL_OPEN, NOT_STARTED);
654
655 Map::PlayerList const& PlayerList = me->GetMap()->GetPlayers();
656
657 if (PlayerList.isEmpty())
658 return;
659
660 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
661 {
662 if (i->GetSource()->IsAlive() && i->GetSource()->HasAura(SPELL_TWILIGHT_SHIFT) && !i->GetSource()->GetVictim())
663 {
664 i->GetSource()->CastSpell(i->GetSource(), SPELL_TWILIGHT_SHIFT_REMOVAL_ALL, true);
665 i->GetSource()->CastSpell(i->GetSource(), SPELL_TWILIGHT_RESIDUE, true);
666 i->GetSource()->RemoveAurasDueToSpell(SPELL_TWILIGHT_SHIFT);
667 i->GetSource()->RemoveAurasDueToSpell(SPELL_TWILIGHT_SHIFT_ENTER);
668 }
669 }
670
671 // not solo fight, so main boss has debuff
672 if (Creature* debuffTarget = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SARTHARION)))
673 if (debuffTarget->IsAlive() && debuffTarget->HasAura(SPELL_GIFT_OF_TWILIGTH_SAR))
674 debuffTarget->RemoveAurasDueToSpell(SPELL_GIFT_OF_TWILIGTH_SAR);
675
676 // event not in progress, then solo fight and must remove debuff mini-boss

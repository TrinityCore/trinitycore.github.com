75 void MoveInLineOfSight(Unit* who) override
76 {
77 if (!who || !who->IsInWorld() || who->GetZoneId() != 4395)
78 return;
79
80 if (!me->IsWithinDist(who, 65.0f, false))
81 return;
82
83 Player* player = who->GetCharmerOrOwnerPlayerOrPlayerItself();
84
85 if (!player || player->IsGameMaster() || player->IsBeingTeleported() ||
86 // If player has Disguise aura for quest A Meeting With The Magister or An Audience With The Arcanist, do not teleport it away but let it pass
87 player->HasAura(SPELL_SUNREAVER_DISGUISE_FEMALE) || player->HasAura(SPELL_SUNREAVER_DISGUISE_MALE) ||
88 player->HasAura(SPELL_SILVER_COVENANT_DISGUISE_FEMALE) || player->HasAura(SPELL_SILVER_COVENANT_DISGUISE_MALE))
89 return;
90
91 switch (me->GetEntry())
92 {
93 case NPC_SILVER_COVENANT_GUARDIAN_MAGE:
94 if (player->GetTeam() == HORDE) // Horde unit found in Alliance area
95 {
96 if (GetClosestCreatureWithEntry(me, NPC_APPLEBOUGH_A, 32.0f))
97 {
98 if (me->isInBackInMap(who, 12.0f)) // In my line of sight, "outdoors", and behind me
99 DoCast(who, SPELL_TRESPASSER_A); // Teleport the Horde unit out
100 }
101 else // In my line of sight, and "indoors"
102 DoCast(who, SPELL_TRESPASSER_A); // Teleport the Horde unit out
103 }
104 break;
105 case NPC_SUNREAVER_GUARDIAN_MAGE:
106 if (player->GetTeam() == ALLIANCE) // Alliance unit found in Horde area
107 {
108 if (GetClosestCreatureWithEntry(me, NPC_SWEETBERRY_H, 32.0f))
109 {
110 if (me->isInBackInMap(who, 12.0f)) // In my line of sight, "outdoors", and behind me
111 DoCast(who, SPELL_TRESPASSER_H); // Teleport the Alliance unit out
112 }
113 else // In my line of sight, and "indoors"
114 DoCast(who, SPELL_TRESPASSER_H); // Teleport the Alliance unit out
115 }
116 break;
117 }
118 me->SetOrientation(me->GetHomePosition().GetOrientation());
119 return;
120 }

122 void UpdateAI(uint32 /*diff*/) override { }
123 };

158 me->setActive(true);

161 void Reset() override
162 {
163 me->SetVisible(false);
164 events.ScheduleEvent(EVENT_SELECT_TARGET, IN_MILLISECONDS);
165 }
166
167 Player* SelectTargetInDalaran()
168 {
169 std::list<Player*> PlayerInDalaranList;
170 PlayerInDalaranList.clear();
171
172 Map::PlayerList const &players = me->GetMap()->GetPlayers();
173 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
174 if (Player* player = itr->GetSource()->ToPlayer())
175 if (player->GetZoneId() == ZONE_DALARAN && !player->IsFlying() && !player->IsMounted() && !player->IsGameMaster())
176 PlayerInDalaranList.push_back(player);
177
178 if (PlayerInDalaranList.empty())
179 return NULL;
180 return Trinity::Containers::SelectRandomContainerElement(PlayerInDalaranList);
181 }
182
183 void SendMailToPlayer(Player* player)
184 {
185 SQLTransaction trans = CharacterDatabase.BeginTransaction();
186 int16 deliverDelay = irand(MAIL_DELIVER_DELAY_MIN, MAIL_DELIVER_DELAY_MAX);
187 MailDraft(MAIL_MINIGOB_ENTRY, true).SendMailTo(trans, MailReceiver(player), MailSender(MAIL_CREATURE, me->GetEntry()), MAIL_CHECK_MASK_NONE, deliverDelay);
188 CharacterDatabase.CommitTransaction(trans);
189 }
190
191 void UpdateAI(uint32 diff) override
192 {
193 events.Update(diff);
194
195 while (uint32 eventId = events.ExecuteEvent())
196 {
197 switch (eventId)
198 {
199 case EVENT_SELECT_TARGET:
200 me->SetVisible(true);
201 DoCast(me, SPELL_TELEPORT_VISUAL);
202 if (Player* player = SelectTargetInDalaran())
203 {
204 me->NearTeleportTo(player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), 0.0f);
205 DoCast(player, SPELL_MANABONKED);
206 SendMailToPlayer(player);
207 }
208 events.ScheduleEvent(EVENT_BLINK, 3*IN_MILLISECONDS);
209 break;
210 case EVENT_BLINK:
211 {
212 DoCast(me, SPELL_IMPROVED_BLINK);
213 Position pos = me->GetRandomNearPosition(frand(15, 40));
214 me->GetMotionMaster()->MovePoint(0, pos.m_positionX, pos.m_positionY, pos.m_positionZ);
215 events.ScheduleEvent(EVENT_DESPAWN, 3 * IN_MILLISECONDS);
216 events.ScheduleEvent(EVENT_DESPAWN_VISUAL, 2.5*IN_MILLISECONDS);
217 break;
218 }
219 case EVENT_DESPAWN_VISUAL:
220 DoCast(me, SPELL_TELEPORT_VISUAL);
221 break;
222 case EVENT_DESPAWN:
223 me->DespawnOrUnsummon();
224 break;
225 default:
226 break;
227 }
228 }
229 }
230
231 private:
232 EventMap events;
233 };

238 }

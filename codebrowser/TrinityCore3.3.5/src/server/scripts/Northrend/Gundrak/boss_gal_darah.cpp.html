75class boss_gal_darah : public CreatureScript
76{
77 public:
78 boss_gal_darah() : CreatureScript("boss_gal_darah") { }
79
80 struct boss_gal_darahAI : public BossAI
81 {
82 boss_gal_darahAI(Creature* creature) : BossAI(creature, DATA_GAL_DARAH)
83 {
84 Initialize();
85 }
86
87 void Initialize()
88 {
89 _phaseCounter = 0;
90 }
91
92 void InitializeAI() override
93 {
94 BossAI::InitializeAI();
95 DoCastAOE(SPELL_HEARTH_BEAM_VISUAL, true);
96 }
97
98 void Reset() override
99 {
100 Initialize();
101 _Reset();
102 impaledPlayers.clear();
103 }
104
105 void JustReachedHome() override
106 {
107 _JustReachedHome();
108 DoCastAOE(SPELL_HEARTH_BEAM_VISUAL, true);
109 }
110
111 void EnterCombat(Unit* /*who*/) override
112 {
113 _EnterCombat();
114 Talk(SAY_AGGRO);
115
116 SetPhase(PHASE_TROLL);
117 }
118
119 void SetPhase(CombatPhase phase)
120 {
121 events.SetPhase(phase);
122 switch (phase)
123 {
124 case PHASE_TROLL:
125 events.ScheduleEvent(EVENT_STAMPEDE, 10 * IN_MILLISECONDS, 0, PHASE_TROLL);
126 events.ScheduleEvent(EVENT_WHIRLING_SLASH, 21 * IN_MILLISECONDS, 0, PHASE_TROLL);
127 break;
128 case PHASE_RHINO:
129 events.ScheduleEvent(EVENT_STOMP, 25 * IN_MILLISECONDS, 0, PHASE_RHINO);
130 events.ScheduleEvent(EVENT_IMPALING_CHARGE, 21 * IN_MILLISECONDS, 0, PHASE_RHINO);
131 events.ScheduleEvent(EVENT_ENRAGE, 15 * IN_MILLISECONDS, 0, PHASE_RHINO);
132 events.ScheduleEvent(EVENT_PUNCTURE, 10 * IN_MILLISECONDS, 0, PHASE_RHINO);
133 break;
134 }
135 }
136
137 void SetGUID(ObjectGuid guid, int32 type /*= 0*/) override
138 {
139 if (type == DATA_SHARE_THE_LOVE)
140 {
141 if (Unit* target = ObjectAccessor::GetUnit(*me, guid))
142 Talk(EMOTE_IMPALE, target);
143 impaledPlayers.insert(guid);
144 }
145 }
146
147 uint32 GetData(uint32 type) const override
148 {
149 if (type == DATA_SHARE_THE_LOVE)
150 return impaledPlayers.size();
151
152 return 0;
153 }
154
155 void JustDied(Unit* /*killer*/) override
156 {
157 _JustDied();
158 Talk(SAY_DEATH);
159 }
160
161 void KilledUnit(Unit* victim) override
162 {
163 if (victim->GetTypeId() == TYPEID_PLAYER)
164 Talk(SAY_SLAY);
165 }
166
167 void SpellHit(Unit* /*caster*/, SpellInfo const* spellInfo) override
168 {
169 if (spellInfo->Id == SPELL_TRANSFORM_BACK)
170 me->RemoveAurasDueToSpell(SPELL_TRANSFORM_RHINO);
171 }
172
173 void ExecuteEvent(uint32 eventId) override
174 {
175 switch (eventId)
176 {
177 case EVENT_IMPALING_CHARGE:
178 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 60.0f, true))
179 DoCast(target, SPELL_IMPALING_CHARGE);
180 if (++_phaseCounter >= 2)
181 events.ScheduleEvent(EVENT_TRANSFORM, 5 * IN_MILLISECONDS);
182 events.ScheduleEvent(eventId, 31 * IN_MILLISECONDS, 0, PHASE_RHINO);
183 break;
184 case EVENT_STOMP:
185 DoCastAOE(SPELL_STOMP);
186 events.ScheduleEvent(eventId, 20 * IN_MILLISECONDS, 0, PHASE_RHINO);
187 break;
188 case EVENT_PUNCTURE:
189 DoCastVictim(SPELL_PUNCTURE);
190 events.ScheduleEvent(eventId, 8 * IN_MILLISECONDS, 0, PHASE_RHINO);
191 break;
192 case EVENT_STAMPEDE:
193 Talk(SAY_SUMMON_RHINO);
194 DoCast(me, SPELL_STAMPEDE);
195 events.ScheduleEvent(eventId, 15 * IN_MILLISECONDS, 0, PHASE_TROLL);
196 break;
197 case EVENT_WHIRLING_SLASH:
198 DoCastVictim(SPELL_WHIRLING_SLASH);
199 if (++_phaseCounter >= 2)
200 events.ScheduleEvent(EVENT_TRANSFORM, 5 * IN_MILLISECONDS);
201 events.ScheduleEvent(eventId, 21 * IN_MILLISECONDS, 0, PHASE_TROLL);
202 break;
203 case EVENT_ENRAGE:
204 DoCast(me, SPELL_ENRAGE);
205 events.ScheduleEvent(eventId, 20 * IN_MILLISECONDS, 0, PHASE_RHINO);
206 break;
207 case EVENT_TRANSFORM:
208 if (events.IsInPhase(PHASE_TROLL))
209 {
210 Talk(SAY_TRANSFORM_1);
211 DoCast(me, SPELL_TRANSFORM_RHINO);
212 SetPhase(PHASE_RHINO);
213 }
214 else if (events.IsInPhase(PHASE_RHINO))
215 {
216 Talk(SAY_TRANSFORM_2);
217 DoCast(me, SPELL_TRANSFORM_BACK);
218 SetPhase(PHASE_TROLL);
219 }
220 _phaseCounter = 0;
221 break;
222 default:
223 break;
224 }
225 }
226
227 private:
228 std::set<uint64> impaledPlayers;
229 uint8 _phaseCounter;
230 };
231
232 CreatureAI* GetAI(Creature* creature) const override
233 {
234 return GetGundrakAI<boss_gal_darahAI>(creature);
235 }
236};

238// 54956, 59827 - Impaling Charge
239class spell_gal_darah_impaling_charge : public SpellScriptLoader
240{
241 public:
242 spell_gal_darah_impaling_charge() : SpellScriptLoader("spell_gal_darah_impaling_charge") { }
243
244 class spell_gal_darah_impaling_charge_SpellScript : public SpellScript
245 {
246 PrepareSpellScript(spell_gal_darah_impaling_charge_SpellScript);
247
248 bool Validate(SpellInfo const* /*spellInfo*/) override
249 {
250 if (!sSpellMgr->GetSpellInfo(SPELL_IMPALING_CHARGE_CONTROL_VEHICLE))
251 return false;
252 return true;
253 }
254
255 bool Load() override
256 {
257 return GetCaster()->GetVehicleKit() && GetCaster()->GetEntry() == NPC_GAL_DARAH;
258 }
259
260 void HandleScript(SpellEffIndex /*effIndex*/)
261 {
262 if (Unit* target = GetHitUnit())
263 {
264 Unit* caster = GetCaster();
265 target->CastSpell(caster, SPELL_IMPALING_CHARGE_CONTROL_VEHICLE, true);
266 caster->ToCreature()->AI()->SetGUID(target->GetGUID(), DATA_SHARE_THE_LOVE);
267 }
268 }
269
270 void Register() override
271 {
272 OnEffectHitTarget += SpellEffectFn(spell_gal_darah_impaling_charge_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_CHARGE);
273 }
274 };
275
276 SpellScript* GetSpellScript() const override
277 {
278 return new spell_gal_darah_impaling_charge_SpellScript();
279 }

285 achievement_share_the_love() : AchievementCriteriaScript("achievement_share_the_love") { }

287 bool OnCheck(Player* /*player*/, Unit* target) override
288 {
289 if (!target)
290 return false;
291
292 if (Creature* GalDarah = target->ToCreature())
293 if (GalDarah->AI()->GetData(DATA_SHARE_THE_LOVE) >= 5)
294 return true;
295
296 return false;
297 }

83class boss_drakkari_colossus : public CreatureScript
84{
85 public:
86 boss_drakkari_colossus() : CreatureScript("boss_drakkari_colossus") { }
87
88 struct boss_drakkari_colossusAI : public BossAI
89 {
90 boss_drakkari_colossusAI(Creature* creature) : BossAI(creature, DATA_DRAKKARI_COLOSSUS)
91 {
92 Initialize();
93 me->SetReactState(REACT_PASSIVE);
94 introDone = false;
95 }
96
97 void Initialize()
98 {
99 phase = COLOSSUS_PHASE_NORMAL;
100 }
101
102 void Reset() override
103 {
104 _Reset();
105
106 if (GetData(DATA_INTRO_DONE))
107 {
108 me->SetReactState(REACT_AGGRESSIVE);
109 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
110 me->RemoveAura(SPELL_FREEZE_ANIM);
111 }
112
113 events.ScheduleEvent(EVENT_MIGHTY_BLOW, urand(10000, 30000));
114
115 Initialize();
116 }
117
118 void EnterCombat(Unit* /*who*/) override
119 {
120 _EnterCombat();
121 me->RemoveAura(SPELL_FREEZE_ANIM);
122 }
123
124 void JustDied(Unit* /*killer*/) override
125 {
126 _JustDied();
127 }
128
129 void DoAction(int32 action) override
130 {
131 switch (action)
132 {
133 case ACTION_SUMMON_ELEMENTAL:
134 DoCast(SPELL_EMERGE);
135 break;
136 case ACTION_FREEZE_COLOSSUS:
137 me->GetMotionMaster()->Clear();
138 me->GetMotionMaster()->MoveIdle();
139
140 me->SetReactState(REACT_PASSIVE);
141 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
142 DoCast(me, SPELL_FREEZE_ANIM);
143 break;
144 case ACTION_UNFREEZE_COLOSSUS:
145
146 if (me->GetReactState() == REACT_AGGRESSIVE)
147 return;
148
149 me->SetReactState(REACT_AGGRESSIVE);
150 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
151 me->RemoveAura(SPELL_FREEZE_ANIM);
152
153 me->SetInCombatWithZone();
154
155 if (me->GetVictim())
156 me->GetMotionMaster()->MoveChase(me->GetVictim(), 0, 0);
157
158 break;
159 }
160 }
161
162 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
163 {
164 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC))
165 damage = 0;
166
167 if (phase == COLOSSUS_PHASE_NORMAL ||
168 phase == COLOSSUS_PHASE_FIRST_ELEMENTAL_SUMMON)
169 {
170 if (HealthBelowPct(phase == COLOSSUS_PHASE_NORMAL ? 50 : 5))
171 {
172 damage = 0;
173 phase = (phase == COLOSSUS_PHASE_NORMAL ? COLOSSUS_PHASE_FIRST_ELEMENTAL_SUMMON : COLOSSUS_PHASE_SECOND_ELEMENTAL_SUMMON);
174 DoAction(ACTION_FREEZE_COLOSSUS);
175 DoAction(ACTION_SUMMON_ELEMENTAL);
176 }
177 }
178 }
179
180 uint32 GetData(uint32 data) const override
181 {
182 if (data == DATA_COLOSSUS_PHASE)
183 return phase;
184 else if (data == DATA_INTRO_DONE)
185 return introDone;
186
187 return 0;
188 }
189
190 void SetData(uint32 type, uint32 data) override
191 {
192 if (type == DATA_INTRO_DONE)
193 introDone = data != 0;
194 }
195
196 void UpdateAI(uint32 diff) override
197 {
198 if (!UpdateVictim())
199 return;
200
201 events.Update(diff);
202
203 if (me->HasUnitState(UNIT_STATE_CASTING))
204 return;
205
206 while (uint32 eventId = events.ExecuteEvent())
207 {
208 switch (eventId)
209 {
210 case EVENT_MIGHTY_BLOW:
211 DoCastVictim(SPELL_MIGHTY_BLOW);
212 events.ScheduleEvent(EVENT_MIGHTY_BLOW, urand(5000, 15000));
213 break;
214 }
215 }
216
217 if (me->GetReactState() == REACT_AGGRESSIVE)
218 DoMeleeAttackIfReady();
219 }
220
221 void JustSummoned(Creature* summon) override
222 {
223 summon->SetInCombatWithZone();
224
225 if (phase == COLOSSUS_PHASE_SECOND_ELEMENTAL_SUMMON)
226 summon->SetHealth(summon->GetMaxHealth() / 2);
227 }
228
229 private:
230 uint8 phase;
231 bool introDone;
232 };
233
234 CreatureAI* GetAI(Creature* creature) const override
235 {
236 return GetGundrakAI<boss_drakkari_colossusAI>(creature);
237 }
238};

240class boss_drakkari_elemental : public CreatureScript
241{
242 public:
243 boss_drakkari_elemental() : CreatureScript("boss_drakkari_elemental") { }
244
245 struct boss_drakkari_elementalAI : public ScriptedAI
246 {
247 boss_drakkari_elementalAI(Creature* creature) : ScriptedAI(creature)
248 {
249 DoCast(me, SPELL_ELEMENTAL_SPAWN_EFFECT);
250 instance = creature->GetInstanceScript();
251 }
252
253 void Reset() override
254 {
255 events.Reset();
256 events.ScheduleEvent(EVENT_SURGE, urand(5000, 15000));
257
258 me->AddAura(SPELL_MOJO_VOLLEY, me);
259 }
260
261 void JustDied(Unit* killer) override
262 {
263 Talk(EMOTE_ACTIVATE_ALTAR);
264
265 if (Creature* colossus = instance->GetCreature(DATA_DRAKKARI_COLOSSUS))
266 killer->Kill(colossus);
267 }
268
269 void UpdateAI(uint32 diff) override
270 {
271 if (!UpdateVictim())
272 return;
273
274 events.Update(diff);
275
276 if (me->HasUnitState(UNIT_STATE_CASTING))
277 return;
278
279 while (uint32 eventId = events.ExecuteEvent())
280 {
281 switch (eventId)
282 {
283 case EVENT_SURGE:
284 DoCast(SPELL_SURGE_VISUAL);
285 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
286 DoCast(target, SPELL_SURGE);
287 events.ScheduleEvent(EVENT_SURGE, urand(5000, 15000));
288 break;
289 }
290 }
291
292 DoMeleeAttackIfReady();
293 }
294
295 void DoAction(int32 action) override
296 {
297 switch (action)
298 {
299 case ACTION_RETURN_TO_COLOSSUS:
300 Talk(EMOTE_MOJO);
301 DoCast(SPELL_SURGE_VISUAL);
302 if (Creature* colossus = instance->GetCreature(DATA_DRAKKARI_COLOSSUS))
303 // what if the elemental is more than 80 yards from drakkari colossus ?
304 DoCast(colossus, SPELL_MERGE, true);
305 break;
306 }
307 }
308
309 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
310 {
311 if (HealthBelowPct(50))
312 {
313 if (Creature* colossus = instance->GetCreature(DATA_DRAKKARI_COLOSSUS))
314 {
315 if (colossus->AI()->GetData(DATA_COLOSSUS_PHASE) == COLOSSUS_PHASE_FIRST_ELEMENTAL_SUMMON)
316 {
317 damage = 0;
318
319 // to prevent spell spaming
320 if (me->HasUnitState(UNIT_STATE_CHARGING))
321 return;
322
323 // not sure about this, the idea of this code is to prevent bug the elemental
324 // if it is not in a acceptable distance to cast the charge spell.
325 /*if (me->GetDistance(colossus) > 80.0f)
326 {
327 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == POINT_MOTION_TYPE)
328 return;
329
330 me->GetMotionMaster()->MovePoint(0, colossus->GetPositionX(), colossus->GetPositionY(), colossus->GetPositionZ());
331 return;
332 }*/
333 DoAction(ACTION_RETURN_TO_COLOSSUS);
334 }
335 }
336 }
337 }
338
339 void EnterEvadeMode(EvadeReason /*why*/) override
340 {
341 me->DespawnOrUnsummon();
342 }
343
344 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
345 {
346 if (spell->Id == SPELL_MERGE)
347 {
348 if (Creature* colossus = target->ToCreature())
349 {
350 colossus->AI()->DoAction(ACTION_UNFREEZE_COLOSSUS);
351 me->DespawnOrUnsummon();
352 }
353 }
354 }
355
356 private:
357 EventMap events;
358 InstanceScript* instance;
359 };
360
361 CreatureAI* GetAI(Creature* creature) const override
362 {
363 return GetGundrakAI<boss_drakkari_elementalAI>(creature);
364 }

367class npc_living_mojo : public CreatureScript
368{
369public:
370 npc_living_mojo() : CreatureScript("npc_living_mojo") { }

372 CreatureAI* GetAI(Creature* creature) const override
373 {
374 return GetGundrakAI<npc_living_mojoAI>(creature);
375 }
376
377 struct npc_living_mojoAI : public ScriptedAI
378 {
379 npc_living_mojoAI(Creature* creature) : ScriptedAI(creature)
380 {
381 Initialize();
382 instance = creature->GetInstanceScript();
383 }
384
385 void Initialize()
386 {
387 mojoWaveTimer = 2 * IN_MILLISECONDS;
388 mojoPuddleTimer = 7 * IN_MILLISECONDS;
389 }
390
391 void Reset() override
392 {
393 Initialize();
394 }
395
396 void MoveMojos(Creature* boss)
397 {
398 std::list<Creature*> mojosList;
399 boss->GetCreatureListWithEntryInGrid(mojosList, me->GetEntry(), 12.0f);
400 if (!mojosList.empty())
401 {
402 for (std::list<Creature*>::const_iterator itr = mojosList.begin(); itr != mojosList.end(); ++itr)
403 {
404 if (Creature* mojo = *itr)
405 mojo->GetMotionMaster()->MovePoint(1, boss->GetHomePosition().GetPositionX(), boss->GetHomePosition().GetPositionY(), boss->GetHomePosition().GetPositionZ());
406 }
407 }
408 }
409
410 void MovementInform(uint32 type, uint32 id) override
411 {
412 if (type != POINT_MOTION_TYPE)
413 return;
414
415 if (id == 1)
416 {
417 if (Creature* colossus = instance->GetCreature(DATA_DRAKKARI_COLOSSUS))
418 {
419 colossus->AI()->DoAction(ACTION_UNFREEZE_COLOSSUS);
420 if (!colossus->AI()->GetData(DATA_INTRO_DONE))
421 colossus->AI()->SetData(DATA_INTRO_DONE, true);
422 colossus->SetInCombatWithZone();
423 me->DespawnOrUnsummon();
424 }
425 }
426 }
427
428 void AttackStart(Unit* attacker) override
429 {
430 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == POINT_MOTION_TYPE)
431 return;
432
433 // we do this checks to see if the creature is one of the creatures that sorround the boss
434 if (Creature* colossus = instance->GetCreature(DATA_DRAKKARI_COLOSSUS))
435 {
436 Position homePosition = me->GetHomePosition();
437
438 float distance = homePosition.GetExactDist(&colossus->GetHomePosition());
439
440 if (distance < 12.0f)
441 {
442 MoveMojos(colossus);
443 me->SetReactState(REACT_PASSIVE);
444 }
445 else
446 ScriptedAI::AttackStart(attacker);
447 }
448 }
449
450 void UpdateAI(uint32 diff) override
451 {
452 //Return since we have no target
453 if (!UpdateVictim())
454 return;
455
456 if (mojoWaveTimer <= diff)
457 {
458 DoCastVictim(SPELL_MOJO_WAVE);
459 mojoWaveTimer = 15*IN_MILLISECONDS;
460 } else mojoWaveTimer -= diff;
461
462 if (mojoPuddleTimer <= diff)
463 {
464 DoCastVictim(SPELL_MOJO_PUDDLE);
465 mojoPuddleTimer = 18*IN_MILLISECONDS;
466 } else mojoPuddleTimer -= diff;
467
468 DoMeleeAttackIfReady();
469 }
470
471 private:
472 InstanceScript* instance;
473 uint32 mojoWaveTimer;
474 uint32 mojoPuddleTimer;
475 };
476};

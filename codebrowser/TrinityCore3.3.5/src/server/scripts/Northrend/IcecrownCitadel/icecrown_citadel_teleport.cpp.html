26static std::vector<uint32> const TeleportSpells =
27{
28 LIGHT_S_HAMMER_TELEPORT, // 0
29 ORATORY_OF_THE_DAMNED_TELEPORT, // 1
30 0, // 2
31 RAMPART_OF_SKULLS_TELEPORT, // 3
32 DEATHBRINGER_S_RISE_TELEPORT, // 4
33 UPPER_SPIRE_TELEPORT, // 5
34 SINDRAGOSA_S_LAIR_TELEPORT // 6
35};

37class icecrown_citadel_teleport : public GameObjectScript
38{
39 static_assert(DATA_UPPERSPIRE_TELE_ACT == 41, "icecrown_citadel.h DATA_UPPERSPIRE_TELE_ACT set to value != 41, gossip condition of the teleporters won't work as intended.");
40
41 public:
42 icecrown_citadel_teleport() : GameObjectScript("icecrown_citadel_teleport") { }
43
44 struct icecrown_citadel_teleportAI : public GameObjectAI
45 {
46 icecrown_citadel_teleportAI(GameObject* go) : GameObjectAI(go)
47 {
48 }
49
50 bool GossipSelect(Player* player, uint32 /*menuId*/, uint32 gossipListId) override
51 {
52 if (gossipListId >= TeleportSpells.size())
53 return false;
54
55 player->PlayerTalkClass->ClearMenus();
56 player->CLOSE_GOSSIP_MENU();
57 SpellInfo const* spell = sSpellMgr->GetSpellInfo(TeleportSpells[gossipListId]);
58 if (!spell)
59 return false;
60
61 if (player->IsInCombat())
62 {
63 Spell::SendCastResult(player, spell, 0, SPELL_FAILED_AFFECTING_COMBAT);
64 return true;
65 }
66
67 player->CastSpell(player, spell, true);
68 return true;
69 }
70 };
71
72 GameObjectAI* GetAI(GameObject* go) const override
73 {
74 return GetIcecrownCitadelAI<icecrown_citadel_teleportAI>(go);
75 }
76};

78class at_frozen_throne_teleport : public AreaTriggerScript
79{
80 public:
81 at_frozen_throne_teleport() : AreaTriggerScript("at_frozen_throne_teleport") { }
82
83 bool OnTrigger(Player* player, AreaTriggerEntry const* /*areaTrigger*/) override
84 {
85 if (player->IsInCombat())
86 {
87 if (SpellInfo const* spell = sSpellMgr->GetSpellInfo(FROZEN_THRONE_TELEPORT))
88 Spell::SendCastResult(player, spell, 0, SPELL_FAILED_AFFECTING_COMBAT);
89 return true;
90 }
91
92 if (InstanceScript* instance = player->GetInstanceScript())
93 if (instance->GetBossState(DATA_PROFESSOR_PUTRICIDE) == DONE &&
94 instance->GetBossState(DATA_BLOOD_QUEEN_LANA_THEL) == DONE &&
95 instance->GetBossState(DATA_SINDRAGOSA) == DONE &&
96 instance->GetBossState(DATA_THE_LICH_KING) != IN_PROGRESS)
97 player->CastSpell(player, FROZEN_THRONE_TELEPORT, true);
98
99 return true;
100 }
101};

103void AddSC_icecrown_citadel_teleport()
104{
105 new icecrown_citadel_teleport();
106 new at_frozen_throne_teleport();
107}

101class BoneSpikeTargetSelector : public std::unary_function<Unit*, bool>
102{
103 public:
104 BoneSpikeTargetSelector(UnitAI* ai) : _ai(ai) { }
105
106 bool operator()(Unit* unit) const
107 {
108 if (unit->GetTypeId() != TYPEID_PLAYER)
109 return false;
110
111 if (unit->HasAura(SPELL_IMPALED))
112 return false;
113
114 // Check if it is one of the tanks soaking Bone Slice
115 for (uint32 i = 0; i < MAX_BONE_SPIKE_IMMUNE; ++i)
116 if (unit->GetGUID() == _ai->GetGUID(DATA_SPIKE_IMMUNE + i))
117 return false;
118
119 return true;
120 }
121
122 private:
123 UnitAI* _ai;
124};

131 struct boss_lord_marrowgarAI : public BossAI
132 {
133 boss_lord_marrowgarAI(Creature* creature) : BossAI(creature, DATA_LORD_MARROWGAR)
134 {
135 _boneStormDuration = RAID_MODE<uint32>(20000, 30000, 20000, 30000);
136 _baseSpeed = creature->GetSpeedRate(MOVE_RUN);
137 _coldflameLastPos.Relocate(creature);
138 _introDone = false;
139 _boneSlice = false;
140 }
141
142 void Reset() override
143 {
144 _Reset();
145 me->SetSpeed(MOVE_RUN, _baseSpeed, true);
146 me->RemoveAurasDueToSpell(SPELL_BONE_STORM);
147 me->RemoveAurasDueToSpell(SPELL_BERSERK);
148 events.ScheduleEvent(EVENT_ENABLE_BONE_SLICE, 10000);
149 events.ScheduleEvent(EVENT_BONE_SPIKE_GRAVEYARD, 15000, EVENT_GROUP_SPECIAL);
150 events.ScheduleEvent(EVENT_COLDFLAME, 5000, EVENT_GROUP_SPECIAL);
151 events.ScheduleEvent(EVENT_WARN_BONE_STORM, urand(45000, 50000));
152 events.ScheduleEvent(EVENT_ENRAGE, 600000);
153 _introDone = false;
154 _boneSlice = false;
155 _boneSpikeImmune.clear();
156 }
157
158 void EnterCombat(Unit* /*who*/) override
159 {
160 Talk(SAY_AGGRO);
161
162 me->setActive(true);
163 DoZoneInCombat();
164 instance->SetBossState(DATA_LORD_MARROWGAR, IN_PROGRESS);
165 }
166
167 void JustDied(Unit* /*killer*/) override
168 {
169 Talk(SAY_DEATH);
170
171 _JustDied();
172 }
173
174 void JustReachedHome() override
175 {
176 _JustReachedHome();
177 instance->SetBossState(DATA_LORD_MARROWGAR, FAIL);
178 instance->SetData(DATA_BONED_ACHIEVEMENT, uint32(true)); // reset
179 }
180
181 void KilledUnit(Unit* victim) override
182 {
183 if (victim->GetTypeId() == TYPEID_PLAYER)
184 Talk(SAY_KILL);
185 }
186
187 void UpdateAI(uint32 diff) override
188 {
189 if (!UpdateVictim())
190 return;
191
192 events.Update(diff);
193
194 if (me->HasUnitState(UNIT_STATE_CASTING))
195 return;
196
197 while (uint32 eventId = events.ExecuteEvent())
198 {
199 switch (eventId)
200 {
201 case EVENT_BONE_SPIKE_GRAVEYARD:
202 if (IsHeroic() || !me->HasAura(SPELL_BONE_STORM))
203 DoCast(me, SPELL_BONE_SPIKE_GRAVEYARD);
204 events.ScheduleEvent(EVENT_BONE_SPIKE_GRAVEYARD, urand(15000, 20000), EVENT_GROUP_SPECIAL);
205 break;
206 case EVENT_COLDFLAME:
207 _coldflameLastPos.Relocate(me);
208 _coldflameTarget.Clear();
209 if (!me->HasAura(SPELL_BONE_STORM))
210 DoCastAOE(SPELL_COLDFLAME_NORMAL);
211 else
212 DoCast(me, SPELL_COLDFLAME_BONE_STORM);
213 events.ScheduleEvent(EVENT_COLDFLAME, 5000, EVENT_GROUP_SPECIAL);
214 break;
215 case EVENT_WARN_BONE_STORM:
216 _boneSlice = false;
217 Talk(EMOTE_BONE_STORM);
218 me->FinishSpell(CURRENT_MELEE_SPELL, false);
219 DoCast(me, SPELL_BONE_STORM);
220 events.DelayEvents(3000, EVENT_GROUP_SPECIAL);
221 events.ScheduleEvent(EVENT_BONE_STORM_BEGIN, 3050);
222 events.ScheduleEvent(EVENT_WARN_BONE_STORM, urand(90000, 95000));
223 break;
224 case EVENT_BONE_STORM_BEGIN:
225 if (Aura* pStorm = me->GetAura(SPELL_BONE_STORM))
226 pStorm->SetDuration(int32(_boneStormDuration));
227 me->SetSpeed(MOVE_RUN, _baseSpeed*3.0f, true);
228 Talk(SAY_BONE_STORM);
229 events.ScheduleEvent(EVENT_BONE_STORM_END, _boneStormDuration+1);
230 // no break here
231 case EVENT_BONE_STORM_MOVE:
232 {
233 events.ScheduleEvent(EVENT_BONE_STORM_MOVE, _boneStormDuration/3);
234 Unit* unit = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me));
235 if (!unit)
236 unit = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true);
237 if (unit)
238 me->GetMotionMaster()->MovePoint(POINT_TARGET_BONESTORM_PLAYER, *unit);
239 break;
240 }
241 case EVENT_BONE_STORM_END:
242 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == POINT_MOTION_TYPE)
243 me->GetMotionMaster()->MovementExpired();
244 me->GetMotionMaster()->MoveChase(me->GetVictim());
245 me->SetSpeed(MOVE_RUN, _baseSpeed, true);
246 events.CancelEvent(EVENT_BONE_STORM_MOVE);
247 events.ScheduleEvent(EVENT_ENABLE_BONE_SLICE, 10000);
248 if (!IsHeroic())
249 events.RescheduleEvent(EVENT_BONE_SPIKE_GRAVEYARD, 15000, EVENT_GROUP_SPECIAL);
250 break;
251 case EVENT_ENABLE_BONE_SLICE:
252 _boneSlice = true;
253 break;
254 case EVENT_ENRAGE:
255 DoCast(me, SPELL_BERSERK, true);
256 Talk(SAY_BERSERK);
257 break;
258 }
259 }
260
261 // We should not melee attack when storming
262 if (me->HasAura(SPELL_BONE_STORM))
263 return;
264
265 // 10 seconds since encounter start Bone Slice replaces melee attacks
266 if (_boneSlice && !me->GetCurrentSpell(CURRENT_MELEE_SPELL))
267 DoCastVictim(SPELL_BONE_SLICE);
268
269 DoMeleeAttackIfReady();
270 }
271
272 void MovementInform(uint32 type, uint32 id) override
273 {
274 if (type != POINT_MOTION_TYPE || id != POINT_TARGET_BONESTORM_PLAYER)
275 return;
276
277 // lock movement
278 me->GetMotionMaster()->MoveIdle();
279 }
280
281 Position const* GetLastColdflamePosition() const
282 {
283 return &_coldflameLastPos;
284 }
285
286 ObjectGuid GetGUID(int32 type /*= 0 */) const override
287 {
288 switch (type)
289 {
290 case DATA_COLDFLAME_GUID:
291 return _coldflameTarget;
292 case DATA_SPIKE_IMMUNE + 0:
293 case DATA_SPIKE_IMMUNE + 1:
294 case DATA_SPIKE_IMMUNE + 2:
295 {
296 uint32 index = uint32(type - DATA_SPIKE_IMMUNE);
297 if (index < _boneSpikeImmune.size())
298 return _boneSpikeImmune[index];
299
300 break;
301 }
302 }
303
304 return ObjectGuid::Empty;
305 }
306
307 void SetGUID(ObjectGuid guid, int32 type /*= 0 */) override
308 {
309 switch (type)
310 {
311 case DATA_COLDFLAME_GUID:
312 _coldflameTarget = guid;
313 break;
314 case DATA_SPIKE_IMMUNE:
315 _boneSpikeImmune.push_back(guid);
316 break;
317 }
318 }
319
320 void DoAction(int32 action) override
321 {
322 switch (action)
323 {
324 case ACTION_CLEAR_SPIKE_IMMUNITIES:
325 _boneSpikeImmune.clear();
326 break;
327 case ACTION_TALK_ENTER_ZONE:
328 if (!_introDone)
329 {
330 Talk(SAY_ENTER_ZONE);
331 _introDone = true;
332 }
333 break;
334 default:
335 break;
336 }
337 }
338
339 private:
340 Position _coldflameLastPos;
341 GuidVector _boneSpikeImmune;
342 ObjectGuid _coldflameTarget;
343 uint32 _boneStormDuration;
344 float _baseSpeed;
345 bool _introDone;
346 bool _boneSlice;
347 };
348
349 CreatureAI* GetAI(Creature* creature) const override
350 {
351 return GetIcecrownCitadelAI<boss_lord_marrowgarAI>(creature);
352 }

367
368 void IsSummonedBy(Unit* owner) override
369 {
370 if (owner->GetTypeId() != TYPEID_UNIT)
371 return;
372
373 Position pos;
374 if (MarrowgarAI* marrowgarAI = CAST_AI(MarrowgarAI, owner->GetAI()))
375 pos.Relocate(marrowgarAI->GetLastColdflamePosition());
376 else
377 pos.Relocate(owner);
378
379 if (owner->HasAura(SPELL_BONE_STORM))
380 {
381 float ang = Position::NormalizeOrientation(pos.GetAngle(me));
382 me->SetOrientation(ang);
383 owner->GetNearPoint2D(pos.m_positionX, pos.m_positionY, 5.0f - owner->GetObjectSize(), ang);
384 }
385 else
386 {
387 Player* target = ObjectAccessor::GetPlayer(*owner, owner->GetAI()->GetGUID(DATA_COLDFLAME_GUID));
388 if (!target)
389 {
390 me->DespawnOrUnsummon();
391 return;
392 }
393
394 float ang = Position::NormalizeOrientation(pos.GetAngle(target));
395 me->SetOrientation(ang);
396 owner->GetNearPoint2D(pos.m_positionX, pos.m_positionY, 15.0f - owner->GetObjectSize(), ang);
397 }
398
399 me->NearTeleportTo(pos.GetPositionX(), pos.GetPositionY(), me->GetPositionZ(), me->GetOrientation());
400 DoCast(SPELL_COLDFLAME_SUMMON);
401 _events.ScheduleEvent(EVENT_COLDFLAME_TRIGGER, 500);
402 }
403
404 void UpdateAI(uint32 diff) override
405 {
406 _events.Update(diff);
407
408 if (_events.ExecuteEvent() == EVENT_COLDFLAME_TRIGGER)
409 {
410 Position newPos = me->GetNearPosition(5.0f, 0.0f);
411 me->NearTeleportTo(newPos.GetPositionX(), newPos.GetPositionY(), me->GetPositionZ(), me->GetOrientation());
412 DoCast(SPELL_COLDFLAME_SUMMON);
413 _events.ScheduleEvent(EVENT_COLDFLAME_TRIGGER, 500);
414 }
415 }
416
417 private:
418 EventMap _events;
419 };

423 return GetIcecrownCitadelAI<npc_coldflameAI>(creature);

637 PreventHitDefaultEffect(effIndex);
638 if (Creature* marrowgar = GetCaster()->ToCreature())

641 uint8 boneSpikeCount = uint8(GetCaster()->GetMap()->GetSpawnMode() & 1 ? 3 : 1);

650 {
651 Unit* target = *itr;
652 target->CastSpell(target, BoneSpikeSummonId[i], true);
653 }
654
655 marrowgarAI->Talk(SAY_BONESPIKE);
656 }
657 }

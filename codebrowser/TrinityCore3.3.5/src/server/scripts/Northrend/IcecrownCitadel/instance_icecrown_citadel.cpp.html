347 // Weekly quest spawn prevention
348 uint32 GetCreatureEntry(ObjectGuid::LowType /*guidLow*/, CreatureData const* data) override
349 {
350 uint32 entry = data->id;
351 switch (entry)
352 {
353 case NPC_INFILTRATOR_MINCHAR:
354 case NPC_KOR_KRON_LIEUTENANT:
355 case NPC_ALCHEMIST_ADRIANNA:
356 case NPC_ALRIN_THE_AGILE:
357 case NPC_INFILTRATOR_MINCHAR_BQ:
358 case NPC_MINCHAR_BEAM_STALKER:
359 case NPC_VALITHRIA_DREAMWALKER_QUEST:
360 {
361 for (uint8 questIndex = 0; questIndex < WeeklyNPCs; ++questIndex)
362 {
363 if (WeeklyQuestData[questIndex].creatureEntry == entry)
364 {
365 uint8 diffIndex = uint8(instance->GetSpawnMode() & 1);
366 if (!sPoolMgr->IsSpawnedObject<Quest>(WeeklyQuestData[questIndex].questId[diffIndex]))
367 return 0;
368 break;
369 }
370 }
371 break;
372 }
373 case NPC_HORDE_GUNSHIP_CANNON:
374 case NPC_ORGRIMS_HAMMER_CREW:
375 case NPC_SKY_REAVER_KORM_BLACKSCAR:
376 if (TeamInInstance == ALLIANCE)
377 return 0;
378 break;
379 case NPC_ALLIANCE_GUNSHIP_CANNON:
380 case NPC_SKYBREAKER_DECKHAND:
381 case NPC_HIGH_CAPTAIN_JUSTIN_BARTLETT:
382 if (TeamInInstance == HORDE)
383 return 0;
384 break;
385 case NPC_ZAFOD_BOOMBOX:
386 if (GameObjectTemplate const* go = sObjectMgr->GetGameObjectTemplate(GO_THE_SKYBREAKER_A))
387 if ((TeamInInstance == ALLIANCE && data->mapid == go->moTransport.mapID) ||
388 (TeamInInstance == HORDE && data->mapid != go->moTransport.mapID))
389 return entry;
390 return 0;
391 case NPC_IGB_MURADIN_BRONZEBEARD:
392 if ((TeamInInstance == ALLIANCE && data->posX > 10.0f) ||
393 (TeamInInstance == HORDE && data->posX < 10.0f))
394 return entry;
395 return 0;
396 default:
397 break;
398 }
399
400 return entry;
401 }

428 void OnUnitDeath(Unit* unit) override
429 {
430 Creature* creature = unit->ToCreature();
431 if (!creature)
432 return;
433
434 switch (creature->GetEntry())
435 {
436 case NPC_YMIRJAR_BATTLE_MAIDEN:
437 case NPC_YMIRJAR_DEATHBRINGER:
438 case NPC_YMIRJAR_FROSTBINDER:
439 case NPC_YMIRJAR_HUNTRESS:
440 case NPC_YMIRJAR_WARLORD:
441 if (Creature* crok = instance->GetCreature(CrokScourgebaneGUID))
442 crok->AI()->SetGUID(creature->GetGUID(), ACTION_VRYKUL_DEATH);
443 break;
444 case NPC_FROSTWING_WHELP:
445 if (FrostwyrmGUIDs.empty())
446 return;
447
448 if (creature->AI()->GetData(1/*DATA_FROSTWYRM_OWNER*/) == DATA_SPINESTALKER)
449 {
450 SpinestalkerTrash.erase(creature->GetSpawnId());
451 if (SpinestalkerTrash.empty())
452 if (Creature* spinestalk = instance->GetCreature(SpinestalkerGUID))
453 spinestalk->AI()->DoAction(ACTION_START_FROSTWYRM);
454 }
455 else
456 {
457 RimefangTrash.erase(creature->GetSpawnId());
458 if (RimefangTrash.empty())
459 if (Creature* spinestalk = instance->GetCreature(RimefangGUID))
460 spinestalk->AI()->DoAction(ACTION_START_FROSTWYRM);
461 }
462 break;
463 case NPC_RIMEFANG:
464 case NPC_SPINESTALKER:
465 {
466 if (instance->IsHeroic() && !HeroicAttempts)
467 return;
468
469 if (GetBossState(DATA_SINDRAGOSA) == DONE)
470 return;
471
472 FrostwyrmGUIDs.erase(creature->GetSpawnId());
473 if (FrostwyrmGUIDs.empty())
474 {
475 instance->LoadGrid(SindragosaSpawnPos.GetPositionX(), SindragosaSpawnPos.GetPositionY());
476 if (Creature* boss = instance->SummonCreature(NPC_SINDRAGOSA, SindragosaSpawnPos))
477 boss->AI()->DoAction(ACTION_START_FROSTWYRM);
478 }
479 break;
480 }
481 default:
482 break;
483 }

820 switch (type)
821 {
822 case DATA_LORD_MARROWGAR:
823 {
824 if (state == DONE)
825 {

834 {

860 Events.ScheduleEvent(EVENT_RESPAWN_GUNSHIP, 30000);

863 switch (state)
864 {
865 case DONE:
866 {
867 if (GameObject* loot = instance->GetGameObject(DeathbringersCacheGUID))
868 {
869 if (Creature* deathbringer = instance->GetCreature(DeathbringerSaurfangGUID))
870 loot->SetLootRecipient(deathbringer->GetLootRecipient());
871 loot->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED | GO_FLAG_NOT_SELECTABLE | GO_FLAG_NODESPAWN);
872 }
873
874 if (GameObject* teleporter = instance->GetGameObject(TeleporterUpperSpireGUID))
875 SetTeleporterState(teleporter, true);
876
877 if (GameObject* teleporter = instance->GetGameObject(TeleporterDeathBringerGUID))
878 SetTeleporterState(teleporter, true);
879 break;
880 }
881 case NOT_STARTED:
882 {
883 if (GameObject* teleporter = instance->GetGameObject(TeleporterDeathBringerGUID))
884 SetTeleporterState(teleporter, true);
885 break;
886 }
887 case IN_PROGRESS:
888 {
889 if (GameObject* teleporter = instance->GetGameObject(TeleporterDeathBringerGUID))
890 SetTeleporterState(teleporter, false);
891 break;
892 }

894 break;
895 }

1022 }
1023 default:
1024 break;
1025 }

1027 return true;

1099 SaveToDB();

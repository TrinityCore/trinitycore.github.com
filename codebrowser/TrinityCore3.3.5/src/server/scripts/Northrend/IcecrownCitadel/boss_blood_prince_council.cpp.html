205 void EnterCombat(Unit* who) override
206 {
207 if (instance->GetBossState(DATA_BLOOD_PRINCE_COUNCIL) == IN_PROGRESS)
208 return;
209
210 if (!instance->CheckRequiredBosses(DATA_BLOOD_PRINCE_COUNCIL, who->ToPlayer()))
211 {
212 EnterEvadeMode();
213 instance->DoCastSpellOnPlayers(LIGHT_S_HAMMER_TELEPORT);

219 DoCast(me, SPELL_INVOCATION_OF_BLOOD_VALANAR);
220
221 if (Creature* keleseth = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PRINCE_KELESETH_GUID)))
222 {
223 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, keleseth);
224 DoZoneInCombat(keleseth);
225 }
226
227 if (Creature* taldaram = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PRINCE_TALDARAM_GUID)))
228 {
229 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, taldaram);
230 DoZoneInCombat(taldaram);
231 }
232
233 if (Creature* valanar = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PRINCE_VALANAR_GUID)))
234 {
235 instance->SendEncounterUnit(ENCOUNTER_FRAME_ENGAGE, valanar);
236 DoZoneInCombat(valanar);
237 }
238
239 events.ScheduleEvent(EVENT_INVOCATION_OF_BLOOD, 46500);
240
241 _invocationOrder[0] = InvocationData(instance->GetGuidData(DATA_PRINCE_VALANAR_GUID), SPELL_INVOCATION_OF_BLOOD_VALANAR, EMOTE_VALANAR_INVOCATION, 71070);
242 if (urand(0, 1))
243 {
244 _invocationOrder[1] = InvocationData(instance->GetGuidData(DATA_PRINCE_TALDARAM_GUID), SPELL_INVOCATION_OF_BLOOD_TALDARAM, EMOTE_TALDARAM_INVOCATION, 71081);
245 _invocationOrder[2] = InvocationData(instance->GetGuidData(DATA_PRINCE_KELESETH_GUID), SPELL_INVOCATION_OF_BLOOD_KELESETH, EMOTE_KELESETH_INVOCATION, 71080);
246 }
247 else
248 {
249 _invocationOrder[1] = InvocationData(instance->GetGuidData(DATA_PRINCE_KELESETH_GUID), SPELL_INVOCATION_OF_BLOOD_KELESETH, EMOTE_KELESETH_INVOCATION, 71080);
250 _invocationOrder[2] = InvocationData(instance->GetGuidData(DATA_PRINCE_TALDARAM_GUID), SPELL_INVOCATION_OF_BLOOD_TALDARAM, EMOTE_TALDARAM_INVOCATION, 71081);

261 void JustReachedHome() override
262 {
263 _resetCounter = 0;

265 keleseth->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);

271 valanar->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);

273
274 void JustDied(Unit* killer) override
275 {
276 _JustDied();
277 // kill all prices
278 for (uint8 i = 0; i < 3; ++i)
279 {
280 if (++_invocationStage == 3)
281 _invocationStage = 0;
282
283 if (Creature* prince = ObjectAccessor::GetCreature(*me, _invocationOrder[_invocationStage].guid))
284 {
285 // make sure looting is allowed
286 if (me->IsDamageEnoughForLootingAndReward())
287 prince->LowerPlayerDamageReq(prince->GetMaxHealth());
288 killer->Kill(prince);
289 }
290 }
291 }
292
293 void UpdateAI(uint32 diff) override
294 {
295 if (!UpdateVictim())
296 return;
297
298 events.Update(diff);
299
300 if (me->HasUnitState(UNIT_STATE_CASTING))
301 return;
302
303 while (uint32 eventId = events.ExecuteEvent())
304 {
305 switch (eventId)
306 {
307 case EVENT_INVOCATION_OF_BLOOD:
308 {
309 Creature* oldPrince = ObjectAccessor::GetCreature(*me, _invocationOrder[_invocationStage].guid);
310 if (++_invocationStage == 3)
311 _invocationStage = 0;
312
313 Creature* newPrince = ObjectAccessor::GetCreature(*me, _invocationOrder[_invocationStage].guid);
314 if (oldPrince)
315 {
316 oldPrince->AI()->DoAction(ACTION_REMOVE_INVOCATION);
317 if (newPrince)
318 oldPrince->CastSpell(newPrince, _invocationOrder[_invocationStage].visualSpell, true);
319 }
320
321 if (newPrince)
322 {
323 newPrince->SetHealth(me->GetHealth());
324 newPrince->AI()->Talk(uint8(_invocationOrder[_invocationStage].textId));
325 }
326
327 DoCast(me, _invocationOrder[_invocationStage].spellId);
328 events.ScheduleEvent(EVENT_INVOCATION_OF_BLOOD, 46500);
329 break;
330 }
331 default:
332 break;
333 }
334 }
335 }
336
337 private:
338 struct InvocationData
339 {
340 ObjectGuid guid;
341 uint32 spellId;
342 uint32 textId;
343 uint32 visualSpell;
344
345 InvocationData(ObjectGuid _guid, uint32 _spellId, uint32 _textId, uint32 _visualSpell)
346 {
347 guid = _guid;
348 spellId = _spellId;
349 textId = _textId;
350 visualSpell = _visualSpell;
351 }
352
353 InvocationData() : spellId(0), textId(0), visualSpell(0) { }
354 } _invocationOrder[3];
355
356 uint32 _invocationStage;
357 uint32 _resetCounter;
358 };

360 CreatureAI* GetAI(Creature* creature) const override

362 return GetIcecrownCitadelAI<boss_blood_council_controllerAI>(creature);

484 void DoAction(int32 action) override
485 {
486 switch (action)
487 {
488 case ACTION_STAND_UP:
489 me->RemoveAurasDueToSpell(SPELL_FEIGN_DEATH);
490 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_IMMUNE_TO_PC);
491 me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_DEAD);
492 me->RemoveFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
493 me->ForceValuesUpdateAtIndex(UNIT_NPC_FLAGS); // was in sniff. don't ask why
494 me->m_Events.AddEvent(new StandUpEvent(*me), me->m_Events.CalculateTime(1000));
495 break;
496 case ACTION_CAST_INVOCATION:
497 Talk(SAY_KELESETH_INVOCATION);
498 DoCast(me, SPELL_INVOCATION_VISUAL_ACTIVE, true);
499 _isEmpowered = true;
500 break;
501 case ACTION_REMOVE_INVOCATION:
502 me->SetHealth(_spawnHealth);
503 me->RemoveAurasDueToSpell(SPELL_INVOCATION_VISUAL_ACTIVE);
504 me->RemoveAurasDueToSpell(SPELL_INVOCATION_OF_BLOOD_KELESETH);
505 _isEmpowered = false;
506 break;
507 default:
508 break;
509 }

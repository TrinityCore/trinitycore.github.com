252 struct boss_deathbringer_saurfangAI : public BossAI
253 {
254 boss_deathbringer_saurfangAI(Creature* creature) : BossAI(creature, DATA_DEATHBRINGER_SAURFANG)
255 {
256 Initialize();
257 ASSERT(creature->GetVehicleKit()); // we dont actually use it, just check if exists
258 _introDone = false;
259 _fallenChampionCastCount = 0;
260 }
261
262 void Initialize()
263 {
264 _frenzied = false;
265 _dead = false;
266 }
267
268 void Reset() override
269 {
270 _Reset();
271 events.SetPhase(PHASE_COMBAT);
272 Initialize();
273 me->SetPower(POWER_ENERGY, 0);
274 DoCast(me, SPELL_ZERO_POWER, true);
275 DoCast(me, SPELL_BLOOD_LINK, true);
276 DoCast(me, SPELL_BLOOD_POWER, true);
277 DoCast(me, SPELL_MARK_OF_THE_FALLEN_CHAMPION_S, true);
278 DoCast(me, SPELL_RUNE_OF_BLOOD_S, true);
279 me->RemoveAurasDueToSpell(SPELL_BERSERK);
280 me->RemoveAurasDueToSpell(SPELL_FRENZY);
281 }
282
283 void EnterCombat(Unit* who) override
284 {
285 if (_dead)
286 return;
287
288 if (!instance->CheckRequiredBosses(DATA_DEATHBRINGER_SAURFANG, who->ToPlayer()))
289 {
290 EnterEvadeMode(EVADE_REASON_OTHER);
291 instance->DoCastSpellOnPlayers(LIGHT_S_HAMMER_TELEPORT);
292 return;
293 }
294
295 // oh just screw intro, enter combat - no exploits please
296 me->setActive(true);
297 DoZoneInCombat();
298
299 events.Reset();
300 events.SetPhase(PHASE_COMBAT);
301 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
302 if (!_introDone)
303 {
304 DoCast(me, SPELL_GRIP_OF_AGONY);
305 if (Creature* creature = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SAURFANG_EVENT_NPC)))
306 creature->AI()->DoAction(ACTION_INTERRUPT_INTRO);
307 }
308
309 _introDone = true;
310
311 Talk(SAY_AGGRO);
312 events.ScheduleEvent(EVENT_SUMMON_BLOOD_BEAST, 30000, 0, PHASE_COMBAT);
313 events.ScheduleEvent(EVENT_BERSERK, IsHeroic() ? 360000 : 480000, 0, PHASE_COMBAT);
314 events.ScheduleEvent(EVENT_BOILING_BLOOD, 15500, 0, PHASE_COMBAT);
315 events.ScheduleEvent(EVENT_BLOOD_NOVA, 17000, 0, PHASE_COMBAT);
316 events.ScheduleEvent(EVENT_RUNE_OF_BLOOD, 20000, 0, PHASE_COMBAT);
317
318 _fallenChampionCastCount = 0;
319 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_MARK_OF_THE_FALLEN_CHAMPION);
320 instance->SetBossState(DATA_DEATHBRINGER_SAURFANG, IN_PROGRESS);
321 }

355
356 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
357 {
358 if (damage >= me->GetHealth())
359 damage = me->GetHealth() - 1;
360
361 if (!_frenzied && HealthBelowPct(31)) // AT 30%, not below
362 {
363 _frenzied = true;
364 DoCast(me, SPELL_FRENZY);
365 Talk(SAY_FRENZY);
366 }
367
368 if (!_dead && me->GetHealth() < FightWonValue)
369 {
370 _dead = true;
371 _JustDied();
372 _EnterEvadeMode();
373 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
374
375 DoCastAOE(SPELL_REMOVE_MARKS_OF_THE_FALLEN_CHAMPION);
376 DoCast(me, SPELL_ACHIEVEMENT, true);
377 Talk(SAY_DEATH);
378
379 //instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_MARK_OF_THE_FALLEN_CHAMPION);
380 DoCast(me, SPELL_PERMANENT_FEIGN_DEATH);
381 if (Creature* creature = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SAURFANG_EVENT_NPC)))
382 creature->AI()->DoAction(ACTION_START_OUTRO);
383 }
384 }

386 void JustSummoned(Creature* summon) override
387 {
388 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
389 {
390 if (target->GetTransport())
391 {
392 summon->DespawnOrUnsummon(1);
393 EnterEvadeMode(EVADE_REASON_OTHER);
394 return;
395 }
396
397 summon->AI()->AttackStart(target);
398 }
399
400 summon->CastSpell(summon, SPELL_BLOOD_LINK_BEAST, true);
401 summon->CastSpell(summon, SPELL_RESISTANT_SKIN, true);
402 summons.Summon(summon);
403 DoZoneInCombat(summon);
404 }

419 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
420 {
421 if (target->GetTransport())
422 {
423 EnterEvadeMode(EVADE_REASON_OTHER);
424 return;
425 }
426
427 switch (spell->Id)
428 {
429 case SPELL_MARK_OF_THE_FALLEN_CHAMPION:
430 Talk(SAY_MARK_OF_THE_FALLEN_CHAMPION);
431 break;
432 case 72255: // Mark of the Fallen Champion, triggered id
433 case 72444:
434 case 72445:
435 case 72446:
436 if (me->GetPower(POWER_ENERGY) != me->GetMaxPower(POWER_ENERGY))
437 target->CastCustomSpell(SPELL_BLOOD_LINK_DUMMY, SPELLVALUE_BASE_POINT0, 1, me, true);
438 break;
439 default:
440 break;
441 }
442 }

444 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
445 {

450
451 void UpdateAI(uint32 diff) override
452 {
453 if (!UpdateVictim() && !(events.IsInPhase(PHASE_INTRO_A) || events.IsInPhase(PHASE_INTRO_H)))
454 return;
455
456 events.Update(diff);
457
458 if (me->HasUnitState(UNIT_STATE_CASTING))
459 return;
460
461 while (uint32 eventId = events.ExecuteEvent())
462 {
463 switch (eventId)
464 {
465 case EVENT_INTRO_ALLIANCE_2:
466 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
467 me->setFaction(FACTION_SCOURGE);
468 Talk(SAY_INTRO_ALLIANCE_2);
469 break;
470 case EVENT_INTRO_ALLIANCE_3:
471 Talk(SAY_INTRO_ALLIANCE_3);
472 break;
473 case EVENT_INTRO_ALLIANCE_6:
474 Talk(SAY_INTRO_ALLIANCE_6);
475 Talk(SAY_INTRO_ALLIANCE_7);
476 DoCast(me, SPELL_GRIP_OF_AGONY);
477 break;
478 case EVENT_INTRO_HORDE_2:
479 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
480 me->setFaction(FACTION_SCOURGE);
481 Talk(SAY_INTRO_HORDE_2);
482 break;
483 case EVENT_INTRO_HORDE_4:
484 Talk(SAY_INTRO_HORDE_4);
485 break;
486 case EVENT_INTRO_HORDE_9:
487 DoCast(me, SPELL_GRIP_OF_AGONY);
488 Talk(SAY_INTRO_HORDE_9);
489 break;
490 case EVENT_INTRO_FINISH:
491 events.SetPhase(PHASE_COMBAT);
492 _introDone = true;
493 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
494 break;
495 case EVENT_SUMMON_BLOOD_BEAST:
496 for (uint32 i10 = 0; i10 < 2; ++i10)
497 DoCast(me, SPELL_SUMMON_BLOOD_BEAST+i10);
498 if (Is25ManRaid())
499 for (uint32 i25 = 0; i25 < 3; ++i25)
500 DoCast(me, SPELL_SUMMON_BLOOD_BEAST_25_MAN+i25);
501 Talk(SAY_BLOOD_BEASTS);
502 events.ScheduleEvent(EVENT_SUMMON_BLOOD_BEAST, 40000, 0, PHASE_COMBAT);
503 if (IsHeroic())
504 events.ScheduleEvent(EVENT_SCENT_OF_BLOOD, 10000, 0, PHASE_COMBAT);
505 break;
506 case EVENT_BLOOD_NOVA:
507 DoCastAOE(SPELL_BLOOD_NOVA_TRIGGER);
508 events.ScheduleEvent(EVENT_BLOOD_NOVA, urand(20000, 25000), 0, PHASE_COMBAT);
509 break;
510 case EVENT_RUNE_OF_BLOOD:
511 DoCastVictim(SPELL_RUNE_OF_BLOOD);
512 events.ScheduleEvent(EVENT_RUNE_OF_BLOOD, urand(20000, 25000), 0, PHASE_COMBAT);
513 break;
514 case EVENT_BOILING_BLOOD:
515 DoCast(me, SPELL_BOILING_BLOOD);
516 events.ScheduleEvent(EVENT_BOILING_BLOOD, urand(15000, 20000), 0, PHASE_COMBAT);
517 break;
518 case EVENT_BERSERK:
519 DoCast(me, SPELL_BERSERK);
520 Talk(SAY_BERSERK);
521 break;
522 case EVENT_SCENT_OF_BLOOD:
523 if (!summons.empty())
524 {
525 Talk(EMOTE_SCENT_OF_BLOOD);
526 DoCastAOE(SPELL_SCENT_OF_BLOOD);
527 }
528 break;
529 default:
530 break;
531 }
532 }
533
534 DoMeleeAttackIfReady();
535 }

546 // intro setup
547 void DoAction(int32 action) override
548 {
549 switch (action)
550 {
551 case PHASE_INTRO_A:
552 case PHASE_INTRO_H:
553 {
554 // controls what events will execute
555 events.SetPhase(uint32(action));
556
557 me->SetHomePosition(deathbringerPos.GetPositionX(), deathbringerPos.GetPositionY(), deathbringerPos.GetPositionZ(), me->GetOrientation());
558 me->GetMotionMaster()->MovePoint(POINT_SAURFANG, deathbringerPos.GetPositionX(), deathbringerPos.GetPositionY(), deathbringerPos.GetPositionZ());
559
560 events.ScheduleEvent(EVENT_INTRO_ALLIANCE_2, 2500, 0, PHASE_INTRO_A);
561 events.ScheduleEvent(EVENT_INTRO_ALLIANCE_3, 20000, 0, PHASE_INTRO_A);
562
563 events.ScheduleEvent(EVENT_INTRO_HORDE_2, 5000, 0, PHASE_INTRO_H);
564 break;
565 }
566 case ACTION_CONTINUE_INTRO:
567 {
568 if (_introDone)
569 return;
570
571 events.ScheduleEvent(EVENT_INTRO_ALLIANCE_6, 6500+500, 0, PHASE_INTRO_A);
572 events.ScheduleEvent(EVENT_INTRO_FINISH, 8000, 0, PHASE_INTRO_A);
573
574 events.ScheduleEvent(EVENT_INTRO_HORDE_4, 6500, 0, PHASE_INTRO_H);
575 events.ScheduleEvent(EVENT_INTRO_HORDE_9, 46700+1000+500, 0, PHASE_INTRO_H);
576 events.ScheduleEvent(EVENT_INTRO_FINISH, 46700+1000+8000, 0, PHASE_INTRO_H);
577 break;
578 }
579 case ACTION_MARK_OF_THE_FALLEN_CHAMPION:
580 {
581 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true, -SPELL_MARK_OF_THE_FALLEN_CHAMPION))
582 {
583 ++_fallenChampionCastCount;
584 DoCast(target, SPELL_MARK_OF_THE_FALLEN_CHAMPION);
585 me->SetPower(POWER_ENERGY, 0);
586 if (Aura* bloodPower = me->GetAura(SPELL_BLOOD_POWER))
587 bloodPower->RecalculateAmountOfEffects();
588 }
589 break;
590 }
591 default:
592 break;
593 }
594 }

288class FrostwingVrykulSearcher
289{
290 public:
291 FrostwingVrykulSearcher(Creature const* source, float range) : _source(source), _range(range) { }
292
293 bool operator()(Unit* unit)
294 {
295 if (!unit->IsAlive())
296 return false;
297
298 switch (unit->GetEntry())
299 {
300 case NPC_YMIRJAR_BATTLE_MAIDEN:
301 case NPC_YMIRJAR_DEATHBRINGER:
302 case NPC_YMIRJAR_FROSTBINDER:
303 case NPC_YMIRJAR_HUNTRESS:
304 case NPC_YMIRJAR_WARLORD:
305 break;
306 default:
307 return false;
308 }
309
310 if (!unit->IsWithinDist(_source, _range, false))
311 return false;
312
313 return true;
314 }
315
316 private:
317 Creature const* _source;
318 float _range;

326 switch (creature->GetOriginalEntry())

353

355 creature->SetPosition(data->posX, data->posY, data->posZ, data->orientation);
356 creature->DespawnOrUnsummon();
357
358 creature->SetCorpseDelay(corpseDelay);
359 creature->SetRespawnDelay(respawnDelay);

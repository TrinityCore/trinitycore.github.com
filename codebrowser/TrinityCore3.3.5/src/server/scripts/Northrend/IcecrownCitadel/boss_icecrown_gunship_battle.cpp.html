374class PassengerController
375{
376public:
377 PassengerController()
378 {
379 ResetSlots(HORDE);
380 }
381
382 void SetTransport(Transport* transport) { _transport = transport; }
383
384 void ResetSlots(uint32 team)
385 {
386 _transport = NULL;
387 for (uint32 i = 0; i < MAX_SLOTS; ++i)
388 _controlledSlots[i].Clear();
389
390 memset(_respawnCooldowns, 0, sizeof(time_t)* MAX_SLOTS);
391 _spawnPoint = team == HORDE ? &OrgrimsHammerAddsSpawnPos : &SkybreakerAddsSpawnPos;
392 _slotInfo = team == HORDE ? OrgrimsHammerSlotInfo : SkybreakerSlotInfo;
393 }
394
395 bool SummonCreatures(PassengerSlots first, PassengerSlots last)
396 {
397 if (!_transport)
398 return false;
399
400 bool summoned = false;
401 time_t now = time(NULL);
402 for (int32 i = first; i <= last; ++i)
403 {
404 if (_respawnCooldowns[i] > now)
405 continue;
406
407 if (_controlledSlots[i])
408 {
409 Creature* current = ObjectAccessor::GetCreature(*_transport, _controlledSlots[i]);
410 if (current && current->IsAlive())
411 continue;
412 }
413
414 if (Creature* passenger = _transport->SummonPassenger(_slotInfo[i].Entry, SelectSpawnPoint(), TEMPSUMMON_CORPSE_TIMED_DESPAWN, NULL, 15000))
415 {
416 _controlledSlots[i] = passenger->GetGUID();
417 _respawnCooldowns[i] = time_t(0);
418 passenger->AI()->SetData(ACTION_SET_SLOT, i);
419 summoned = true;
420 }
421 }
422
423 return summoned;
424 }
425
426 void ClearSlot(PassengerSlots slot)
427 {
428 _controlledSlots[slot].Clear();
429 _respawnCooldowns[slot] = time(NULL) + _slotInfo[slot].Cooldown;
430 }
431
432 bool SlotsNeedRefill(PassengerSlots first, PassengerSlots last) const
433 {
434 for (int32 i = first; i <= last; ++i)
435 if (!_controlledSlots[i])
436 return true;
437
438 return false;
439 }
440
441private:
442 Position SelectSpawnPoint() const
443 {
444 Position newPos;
445 float angle = frand(float(-M_PI) * 0.5f, float(M_PI) * 0.5f);
446 newPos.m_positionX = _spawnPoint->GetPositionX() + 2.0f * std::cos(angle);
447 newPos.m_positionY = _spawnPoint->GetPositionY() + 2.0f * std::sin(angle);
448 newPos.m_positionZ = _spawnPoint->GetPositionZ();
449 newPos.SetOrientation(_spawnPoint->GetOrientation());
450 return newPos;
451 }
452
453 Transport* _transport;
454 ObjectGuid _controlledSlots[MAX_SLOTS];
455 time_t _respawnCooldowns[MAX_SLOTS];
456 Position const* _spawnPoint;
457 SlotInfo const* _slotInfo;
458};

250 {
251 switch (action)

253 case ACTION_START_INTRO:

262 events.ScheduleEvent(EVENT_INTRO_5, 39500, 0, PHASE_INTRO);
263 events.ScheduleEvent(EVENT_INTRO_6, 48500, 0, PHASE_INTRO);
264 events.ScheduleEvent(EVENT_INTRO_7, 58000, 0, PHASE_INTRO);

266 break;
267 default:
268 break;
269 }

272 void AttackStart(Unit* victim) override
273 {
274 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))

280
281 void EnterCombat(Unit* who) override
282 {
283 if (!instance->CheckRequiredBosses(DATA_LADY_DEATHWHISPER, who->ToPlayer()))
284 {
285 EnterEvadeMode();
286 instance->DoCastSpellOnPlayers(LIGHT_S_HAMMER_TELEPORT);
287 return;
288 }
289
290 me->setActive(true);
291 DoZoneInCombat();
292
293 events.Reset();
294 events.SetPhase(PHASE_ONE);
295 // phase-independent events
296 events.ScheduleEvent(EVENT_BERSERK, 600000);
297 events.ScheduleEvent(EVENT_DEATH_AND_DECAY, 10000);
298 // phase one only
299 events.ScheduleEvent(EVENT_P1_SUMMON_WAVE, 5000, 0, PHASE_ONE);
300 events.ScheduleEvent(EVENT_P1_SHADOW_BOLT, urand(5500, 6000), 0, PHASE_ONE);
301 events.ScheduleEvent(EVENT_P1_EMPOWER_CULTIST, urand(20000, 30000), 0, PHASE_ONE);
302 if (GetDifficulty() != RAID_DIFFICULTY_10MAN_NORMAL)
303 events.ScheduleEvent(EVENT_DOMINATE_MIND_H, 27000);
304
305 Talk(SAY_AGGRO);
306 DoStartNoMovement(who);
307 me->RemoveAurasDueToSpell(SPELL_SHADOW_CHANNELING);
308 DoCast(me, SPELL_MANA_BARRIER, true);
309
310 instance->SetBossState(DATA_LADY_DEATHWHISPER, IN_PROGRESS);
311 }

313 void JustDied(Unit* killer) override
314 {
315 Talk(SAY_DEATH);
316
317 std::set<uint32> livingAddEntries;
318 // Full House achievement
319 for (SummonList::iterator itr = summons.begin(); itr != summons.end(); ++itr)
320 if (Unit* unit = ObjectAccessor::GetUnit(*me, *itr))
321 if (unit->IsAlive() && unit->GetEntry() != NPC_VENGEFUL_SHADE)
322 livingAddEntries.insert(unit->GetEntry());
323
324 if (livingAddEntries.size() >= 5)
325 instance->DoUpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_FULL_HOUSE, 0, me);
326
327 if (Creature* darnavan = ObjectAccessor::GetCreature(*me, _darnavanGUID))
328 {
329 if (darnavan->IsAlive())
330 {
331 darnavan->setFaction(35);
332 darnavan->CombatStop(true);
333 darnavan->GetMotionMaster()->MoveIdle();
334 darnavan->SetReactState(REACT_PASSIVE);
335 darnavan->m_Events.AddEvent(new DaranavanMoveEvent(*darnavan), darnavan->m_Events.CalculateTime(10000));
336 darnavan->AI()->Talk(SAY_DARNAVAN_RESCUED);
337 if (Player* owner = killer->GetCharmerOrOwnerPlayerOrPlayerItself())
338 {
339 if (Group* group = owner->GetGroup())
340 {
341 for (GroupReference* itr = group->GetFirstMember(); itr != NULL; itr = itr->next())
342 if (Player* member = itr->GetSource())
343 member->KilledMonsterCredit(NPC_DARNAVAN_CREDIT);
344 }
345 else
346 owner->KilledMonsterCredit(NPC_DARNAVAN_CREDIT);
347 }
348 }
349 }
350
351 _JustDied();
352 }

373 void DamageTaken(Unit* /*damageDealer*/, uint32& damage) override
374 {
375 // phase transition
376 if (events.IsInPhase(PHASE_ONE) && damage > me->GetPower(POWER_MANA))
377 {
378 Talk(SAY_PHASE_2);
379 Talk(EMOTE_PHASE_2);
380 DoStartMovement(me->GetVictim());
381 damage -= me->GetPower(POWER_MANA);
382 me->SetPower(POWER_MANA, 0);
383 me->RemoveAurasDueToSpell(SPELL_MANA_BARRIER);
384 events.SetPhase(PHASE_TWO);
385 events.ScheduleEvent(EVENT_P2_FROSTBOLT, urand(10000, 12000), 0, PHASE_TWO);
386 events.ScheduleEvent(EVENT_P2_FROSTBOLT_VOLLEY, urand(19000, 21000), 0, PHASE_TWO);
387 events.ScheduleEvent(EVENT_P2_TOUCH_OF_INSIGNIFICANCE, urand(6000, 9000), 0, PHASE_TWO);
388 events.ScheduleEvent(EVENT_P2_SUMMON_SHADE, urand(12000, 15000), 0, PHASE_TWO);
389 // on heroic mode Lady Deathwhisper is immune to taunt effects in phase 2 and continues summoning adds
390 if (IsHeroic())
391 {
392 me->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TAUNT, true);
393 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_ATTACK_ME, true);
394 events.ScheduleEvent(EVENT_P2_SUMMON_WAVE, 45000, 0, PHASE_TWO);
395 }
396 }

399 void JustSummoned(Creature* summon) override
400 {
401 if (summon->GetEntry() == NPC_DARNAVAN)
402 _darnavanGUID = summon->GetGUID();
403 else
404 summons.Summon(summon);
405
406 Unit* target = NULL;
407 if (summon->GetEntry() == NPC_VENGEFUL_SHADE)
408 {
409 target = ObjectAccessor::GetUnit(*me, _nextVengefulShadeTargetGUID); // Vengeful Shade
410 _nextVengefulShadeTargetGUID.Clear();
411 }
412 else
413 target = SelectTarget(SELECT_TARGET_RANDOM); // Wave adds
414
415 summon->AI()->AttackStart(target); // CAN be NULL
416 if (summon->GetEntry() == NPC_REANIMATED_FANATIC)
417 summon->CastSpell(summon, SPELL_FANATIC_S_DETERMINATION, true);
418 else if (summon->GetEntry() == NPC_REANIMATED_ADHERENT)
419 summon->CastSpell(summon, SPELL_ADHERENT_S_DETERMINATION, true);
420 }

422 void UpdateAI(uint32 diff) override
423 {
424 if (!UpdateVictim() && !events.IsInPhase(PHASE_INTRO))
425 return;
426
427 events.Update(diff);
428
429 if (me->HasUnitState(UNIT_STATE_CASTING) && !events.IsInPhase(PHASE_INTRO))
430 return;
431
432 while (uint32 eventId = events.ExecuteEvent())
433 {
434 switch (eventId)
435 {
436 case EVENT_INTRO_2:
437 Talk(SAY_INTRO_2);
438 break;
439 case EVENT_INTRO_3:
440 Talk(SAY_INTRO_3);
441 break;
442 case EVENT_INTRO_4:
443 Talk(SAY_INTRO_4);
444 break;
445 case EVENT_INTRO_5:
446 Talk(SAY_INTRO_5);
447 break;
448 case EVENT_INTRO_6:
449 Talk(SAY_INTRO_6);
450 break;
451 case EVENT_INTRO_7:
452 Talk(SAY_INTRO_7);
453 break;
454 case EVENT_DEATH_AND_DECAY:
455 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
456 DoCast(target, SPELL_DEATH_AND_DECAY);
457 events.ScheduleEvent(EVENT_DEATH_AND_DECAY, urand(22000, 30000));
458 break;
459 case EVENT_DOMINATE_MIND_H:
460 Talk(SAY_DOMINATE_MIND);
461 for (uint8 i = 0; i < _dominateMindCount; i++)
462 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true, -SPELL_DOMINATE_MIND_H))
463 DoCast(target, SPELL_DOMINATE_MIND_H);
464 events.ScheduleEvent(EVENT_DOMINATE_MIND_H, urand(40000, 45000));
465 break;
466 case EVENT_P1_SUMMON_WAVE:
467 SummonWaveP1();
468 events.ScheduleEvent(EVENT_P1_SUMMON_WAVE, IsHeroic() ? 45000 : 60000, 0, PHASE_ONE);
469 break;
470 case EVENT_P1_SHADOW_BOLT:
471 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
472 DoCast(target, SPELL_SHADOW_BOLT);
473 events.ScheduleEvent(EVENT_P1_SHADOW_BOLT, urand(5000, 8000), 0, PHASE_ONE);
474 break;
475 case EVENT_P1_REANIMATE_CULTIST:
476 ReanimateCultist();
477 break;
478 case EVENT_P1_EMPOWER_CULTIST:
479 EmpowerCultist();
480 events.ScheduleEvent(EVENT_P1_EMPOWER_CULTIST, urand(18000, 25000));
481 break;
482 case EVENT_P2_FROSTBOLT:
483 DoCastVictim(SPELL_FROSTBOLT);
484 events.ScheduleEvent(EVENT_P2_FROSTBOLT, urand(10000, 11000), 0, PHASE_TWO);
485 break;
486 case EVENT_P2_FROSTBOLT_VOLLEY:
487 DoCastAOE(SPELL_FROSTBOLT_VOLLEY);
488 events.ScheduleEvent(EVENT_P2_FROSTBOLT_VOLLEY, urand(13000, 15000), 0, PHASE_TWO);
489 break;
490 case EVENT_P2_TOUCH_OF_INSIGNIFICANCE:
491 DoCastVictim(SPELL_TOUCH_OF_INSIGNIFICANCE);
492 events.ScheduleEvent(EVENT_P2_TOUCH_OF_INSIGNIFICANCE, urand(9000, 13000), 0, PHASE_TWO);
493 break;
494 case EVENT_P2_SUMMON_SHADE:
495 if (Unit* shadeTarget = SelectTarget(SELECT_TARGET_RANDOM, 1))
496 {
497 _nextVengefulShadeTargetGUID = shadeTarget->GetGUID();
498 DoCast(shadeTarget, SPELL_SUMMON_SHADE);
499 }
500 events.ScheduleEvent(EVENT_P2_SUMMON_SHADE, urand(18000, 23000), 0, PHASE_TWO);
501 break;
502 case EVENT_P2_SUMMON_WAVE:
503 SummonWaveP2();
504 events.ScheduleEvent(EVENT_P2_SUMMON_WAVE, 45000, 0, PHASE_TWO);
505 break;
506 case EVENT_BERSERK:
507 DoCast(me, SPELL_BERSERK);
508 Talk(SAY_BERSERK);
509 break;
510 }
511 }
512
513 // We should not melee attack when barrier is up
514 if (me->HasAura(SPELL_MANA_BARRIER))
515 return;
516
517 DoMeleeAttackIfReady();
518 }

520 // summoning function for first phase
521 void SummonWaveP1()
522 {
523 uint8 addIndex = _waveCounter & 1;
524 uint8 addIndexOther = uint8(addIndex ^ 1);
525
526 // Summon first add, replace it with Darnavan if weekly quest is active
527 if (_waveCounter || !sPoolMgr->IsSpawnedObject<Quest>(QUEST_DEPROGRAMMING))
528 Summon(SummonEntries[addIndex], SummonPositions[addIndex * 3]);
529 else
530 Summon(NPC_DARNAVAN, SummonPositions[addIndex * 3]);
531
532 Summon(SummonEntries[addIndexOther], SummonPositions[addIndex * 3 + 1]);
533 Summon(SummonEntries[addIndex], SummonPositions[addIndex * 3 + 2]);
534 if (Is25ManRaid())
535 {
536 Summon(SummonEntries[addIndexOther], SummonPositions[addIndexOther * 3]);
537 Summon(SummonEntries[addIndex], SummonPositions[addIndexOther * 3 + 1]);
538 Summon(SummonEntries[addIndexOther], SummonPositions[addIndexOther * 3 + 2]);
539 Summon(SummonEntries[urand(0, 1)], SummonPositions[6]);
540 }
541
542 ++_waveCounter;
543 }
544
545 // summoning function for second phase
546 void SummonWaveP2()
547 {
548 if (Is25ManRaid())
549 {
550 uint8 addIndex = _waveCounter & 1;
551 Summon(SummonEntries[addIndex], SummonPositions[addIndex * 3]);
552 Summon(SummonEntries[addIndex ^ 1], SummonPositions[addIndex * 3 + 1]);
553 Summon(SummonEntries[addIndex], SummonPositions[addIndex * 3+ 2]);
554 }
555 else
556 Summon(SummonEntries[urand(0, 1)], SummonPositions[6]);
557
558 ++_waveCounter;
559 }

592 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
593 {
594 if (spell->Id == SPELL_DARK_MARTYRDOM_T)
595 {
596 Position pos = target->GetPosition();
597 if (target->GetEntry() == NPC_CULT_FANATIC)
598 me->SummonCreature(NPC_REANIMATED_FANATIC, pos, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
599 else
600 me->SummonCreature(NPC_REANIMATED_ADHERENT, pos, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);

615 if (cre->IsAlive() && (cre->GetEntry() == NPC_CULT_FANATIC || cre->GetEntry() == NPC_CULT_ADHERENT))
616 temp.push_back(cre);
617
618 // noone to empower
619 if (temp.empty())
620 return;
621
622 // select random cultist
623 Creature* cultist = Trinity::Containers::SelectRandomContainerElement(temp);
624 DoCast(cultist, cultist->GetEntry() == NPC_CULT_FANATIC ? SPELL_DARK_TRANSFORMATION_T : SPELL_DARK_EMPOWERMENT_T, true);
625 Talk(uint8(cultist->GetEntry() == NPC_CULT_FANATIC ? SAY_DARK_TRANSFORMATION : SAY_DARK_EMPOWERMENT));
626 }

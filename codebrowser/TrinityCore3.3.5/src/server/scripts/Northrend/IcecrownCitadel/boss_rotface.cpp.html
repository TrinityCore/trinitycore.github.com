92class boss_rotface : public CreatureScript
93{
94 public:
95 boss_rotface() : CreatureScript("boss_rotface") { }
96
97 struct boss_rotfaceAI : public BossAI
98 {
99 boss_rotfaceAI(Creature* creature) : BossAI(creature, DATA_ROTFACE)
100 {
101 infectionStage = 0;
102 infectionCooldown = 14000;
103 }
104
105 void Reset() override
106 {
107 _Reset();
108 events.ScheduleEvent(EVENT_SLIME_SPRAY, 20000);
109 events.ScheduleEvent(EVENT_HASTEN_INFECTIONS, 90000);
110 events.ScheduleEvent(EVENT_MUTATED_INFECTION, 14000);
111 if (IsHeroic())
112 events.ScheduleEvent(EVENT_VILE_GAS, urand(22000, 27000));
113
114 infectionStage = 0;
115 infectionCooldown = 14000;
116 }
117
118 void EnterCombat(Unit* who) override
119 {
120 if (!instance->CheckRequiredBosses(DATA_ROTFACE, who->ToPlayer()))
121 {
122 EnterEvadeMode(EVADE_REASON_OTHER);
123 instance->DoCastSpellOnPlayers(LIGHT_S_HAMMER_TELEPORT);
124 return;
125 }
126
127 me->setActive(true);
128 Talk(SAY_AGGRO);
129 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
130 professor->AI()->DoAction(ACTION_ROTFACE_COMBAT);
131
132 DoZoneInCombat();
133 DoCast(me, SPELL_GREEN_ABOMINATION_HITTIN__YA_PROC, true);
134 }
135
136 void JustDied(Unit* /*killer*/) override
137 {
138 instance->DoRemoveAurasDueToSpellOnPlayers(MUTATED_INFECTION);
139 _JustDied();
140 Talk(SAY_DEATH);
141 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
142 professor->AI()->DoAction(ACTION_ROTFACE_DEATH);
143 }
144
145 void JustReachedHome() override
146 {
147 _JustReachedHome();
148 instance->SetBossState(DATA_ROTFACE, FAIL);
149 instance->SetData(DATA_OOZE_DANCE_ACHIEVEMENT, uint32(true)); // reset
150 }
151
152 void KilledUnit(Unit* victim) override
153 {
154 if (victim->GetTypeId() == TYPEID_PLAYER)
155 Talk(SAY_KILL);
156 }
157
158 void EnterEvadeMode(EvadeReason why) override
159 {
160 ScriptedAI::EnterEvadeMode(why);
161 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
162 professor->AI()->EnterEvadeMode();
163 }
164
165 void SpellHitTarget(Unit* /*target*/, SpellInfo const* spell) override
166 {
167 if (spell->Id == SPELL_SLIME_SPRAY)
168 Talk(SAY_SLIME_SPRAY);
169 }
170
171 void JustSummoned(Creature* summon) override
172 {
173 if (summon->GetEntry() == NPC_VILE_GAS_STALKER)
174 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
175 professor->CastSpell(summon, SPELL_VILE_GAS_H, true);
176
177 summons.Summon(summon);
178 }
179
180 void UpdateAI(uint32 diff) override
181 {
182 if (!UpdateVictim())
183 return;
184
185 events.Update(diff);
186
187 if (me->HasUnitState(UNIT_STATE_CASTING))
188 return;
189
190 while (uint32 eventId = events.ExecuteEvent())
191 {
192 switch (eventId)
193 {
194 case EVENT_SLIME_SPRAY:
195 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
196 {
197 DoSummon(NPC_OOZE_SPRAY_STALKER, *target, 8000, TEMPSUMMON_TIMED_DESPAWN);
198 Talk(EMOTE_SLIME_SPRAY);
199 DoCast(me, SPELL_SLIME_SPRAY);
200 }
201 events.ScheduleEvent(EVENT_SLIME_SPRAY, 20000);
202 break;
203 case EVENT_HASTEN_INFECTIONS:
204 if (infectionStage++ < 4)
205 {
206 infectionCooldown -= 2000;
207 events.ScheduleEvent(EVENT_HASTEN_INFECTIONS, 90000);
208 }
209 break;
210 case EVENT_MUTATED_INFECTION:
211 DoCastAOE(SPELL_MUTATED_INFECTION);
212 events.ScheduleEvent(EVENT_MUTATED_INFECTION, infectionCooldown);
213 break;
214 case EVENT_VILE_GAS:
215 DoCastAOE(SPELL_VILE_GAS_TRIGGER);
216 events.ScheduleEvent(EVENT_VILE_GAS, urand(30000, 35000));
217 break;
218 default:
219 break;
220 }
221 }
222
223 DoMeleeAttackIfReady();
224 }
225
226 private:
227 uint32 infectionCooldown;
228 uint32 infectionStage;
229 };
230
231 CreatureAI* GetAI(Creature* creature) const override
232 {
233 return GetIcecrownCitadelAI<boss_rotfaceAI>(creature);
234 }

453 class spell_rotface_ooze_flood_SpellScript : public SpellScript
454 {
455 PrepareSpellScript(spell_rotface_ooze_flood_SpellScript);
456
457 void HandleScript(SpellEffIndex /*effIndex*/)
458 {
459 if (!GetHitUnit())
460 return;
461
462 std::list<Creature*> triggers;
463 GetHitUnit()->GetCreatureListWithEntryInGrid(triggers, GetHitUnit()->GetEntry(), 12.5f);
464
465 if (triggers.empty())
466 return;
467
468 triggers.sort(Trinity::ObjectDistanceOrderPred(GetHitUnit()));
469 GetHitUnit()->CastSpell(triggers.back(), uint32(GetEffectValue()), false, NULL, NULL, GetOriginalCaster() ? GetOriginalCaster()->GetGUID() : ObjectGuid::Empty);
470 }
471
472 void FilterTargets(std::list<WorldObject*>& targets)
473 {
474 // get 2 targets except 2 nearest
475 targets.sort(Trinity::ObjectDistanceOrderPred(GetCaster()));
476
477 // .resize() runs pop_back();
478 if (targets.size() > 5)
479 targets.resize(5);
480
481 while (targets.size() > 2)
482 targets.pop_front();
483 }
484
485 void Register() override
486 {
487 OnEffectHitTarget += SpellEffectFn(spell_rotface_ooze_flood_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
488 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_rotface_ooze_flood_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENTRY);
489 }
490 };
491
492 SpellScript* GetSpellScript() const override
493 {
494 return new spell_rotface_ooze_flood_SpellScript();
495 }

812 class spell_rotface_vile_gas_trigger_SpellScript : public SpellScript
813 {
814 PrepareSpellScript(spell_rotface_vile_gas_trigger_SpellScript);
815
816 void FilterTargets(std::list<WorldObject*>& targets)
817 {
818 targets.sort(Trinity::ObjectDistanceOrderPred(GetCaster()));
819 if (targets.empty())
820 return;
821
822 std::list<WorldObject*> ranged, melee;
823 std::list<WorldObject*>::iterator itr = targets.begin();
824 while (itr != targets.end() && (*itr)->GetDistance(GetCaster()) < 5.0f)
825 {
826 melee.push_back((*itr)->ToUnit());
827 ++itr;
828 }
829
830 while (itr != targets.end())
831 {
832 ranged.push_back((*itr)->ToUnit());
833 ++itr;
834 }
835
836 uint32 minTargets = GetCaster()->GetMap()->Is25ManRaid() ? 8 : 3;
837 while (ranged.size() < minTargets)
838 {
839 if (melee.empty())
840 break;
841
842 WorldObject* target = Trinity::Containers::SelectRandomContainerElement(melee);
843 ranged.push_back(target);
844 melee.remove(target);
845 }
846
847 if (!ranged.empty())
848 Trinity::Containers::RandomResizeList(ranged, GetCaster()->GetMap()->Is25ManRaid() ? 3 : 1);
849
850 targets.swap(ranged);
851 }
852
853 void HandleDummy(SpellEffIndex effIndex)
854 {
855 PreventHitDefaultEffect(effIndex);
856 GetCaster()->CastSpell(GetHitUnit(), SPELL_VILE_GAS_TRIGGER_SUMMON);
857 }
858
859 void Register()
860 {
861 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_rotface_vile_gas_trigger_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
862 OnEffectHitTarget += SpellEffectFn(spell_rotface_vile_gas_trigger_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
863 }
864 };

866 SpellScript* GetSpellScript() const
867 {
868 return new spell_rotface_vile_gas_trigger_SpellScript();

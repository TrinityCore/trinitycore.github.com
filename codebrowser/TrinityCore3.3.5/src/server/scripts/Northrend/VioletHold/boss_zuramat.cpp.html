30 SPELL_ZURAMAT_ADD_DUMMY = 54351,
31 SPELL_SUMMON_VOID_SENTRY_BALL = 58650
32};

50class boss_zuramat : public CreatureScript
51{
52 public:
53 boss_zuramat() : CreatureScript("boss_zuramat") { }
54
55 struct boss_zuramatAI : public BossAI
56 {
57 boss_zuramatAI(Creature* creature) : BossAI(creature, DATA_ZURAMAT)
58 {
59 Initialize();
60 }
61
62 void Initialize()
63 {
64 _voidDance = true;
65 }
66
67 void Reset() override
68 {
69 BossAI::Reset();
70 Initialize();
71 }
72
73 void EnterCombat(Unit* who) override
74 {
75 BossAI::EnterCombat(who);
76 Talk(SAY_AGGRO);
77 }
78
79 void JustReachedHome() override
80 {
81 BossAI::JustReachedHome();
82 instance->SetData(DATA_HANDLE_CELLS, DATA_ZURAMAT);
83 }
84
85 void SummonedCreatureDies(Creature* summon, Unit* /*who*/) override
86 {
87 if (summon->GetEntry() == NPC_VOID_SENTRY)
88 _voidDance = false;
89 }
90
91 void SummonedCreatureDespawn(Creature* summon) override
92 {
93 if (summon->GetEntry() == NPC_VOID_SENTRY)
94 summon->AI()->DoAction(ACTION_DESPAWN_VOID_SENTRY_BALL);
95 BossAI::SummonedCreatureDespawn(summon);
96 }
97
98 uint32 GetData(uint32 type) const override
99 {
100 if (type == DATA_VOID_DANCE)
101 return _voidDance ? 1 : 0;
102
103 return 0;
104 }
105
106 void JustDied(Unit* killer) override
107 {
108 BossAI::JustDied(killer);
109 Talk(SAY_DEATH);
110 }
111
112 void KilledUnit(Unit* victim) override
113 {
114 if (victim->GetTypeId() == TYPEID_PLAYER)
115 Talk(SAY_SLAY);
116 }
117
118 void UpdateAI(uint32 diff) override
119 {
120 if (!UpdateVictim())
121 return;
122
123 scheduler.Update(diff,
124 std::bind(&BossAI::DoMeleeAttackIfReady, this));
125 }
126
127 void ScheduleTasks() override
128 {
129 scheduler.Schedule(Seconds(4), [this](TaskContext task)
130 {
131 DoCast(me, SPELL_SUMMON_VOID_SENTRY);
132 task.Repeat(Seconds(7), Seconds(10));
133 });
134
135 scheduler.Schedule(Seconds(9), [this](TaskContext task)
136 {
137 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 60.0f, true))
138 DoCast(target, SPELL_VOID_SHIFT);
139 task.Repeat(Seconds(15));
140 });
141
142 scheduler.Schedule(Seconds(18), Seconds(20), [this](TaskContext task)
143 {
144 DoCast(me, SPELL_SHROUD_OF_DARKNESS);
145 task.Repeat(Seconds(18), Seconds(20));
146 });
147 }
148
149 private:
150 bool _voidDance;
151 };
152
153 CreatureAI* GetAI(Creature* creature) const override
154 {
155 return GetVioletHoldAI<boss_zuramatAI>(creature);
156 }
157};

159class npc_void_sentry : public CreatureScript
160{
161 public:
162 npc_void_sentry() : CreatureScript("npc_void_sentry") { }
163
164 struct npc_void_sentryAI : public ScriptedAI
165 {
166 npc_void_sentryAI(Creature* creature) : ScriptedAI(creature), _summons(creature)
167 {
168 me->SetReactState(REACT_PASSIVE);
169 }
170
171 void IsSummonedBy(Unit* /*summoner*/) override
172 {
173 me->CastSpell(me, SPELL_SUMMON_VOID_SENTRY_BALL, true);
174 }
175
176 void JustSummoned(Creature* summon) override
177 {
178 _summons.Summon(summon);
179 summon->SetReactState(REACT_PASSIVE);
180 }
181
182 void SummonedCreatureDespawn(Creature* summon) override
183 {
184 _summons.Despawn(summon);
185 }
186
187 void DoAction(int32 actionId) override
188 {
189 if (actionId == ACTION_DESPAWN_VOID_SENTRY_BALL)
190 _summons.DespawnAll();
191 }
192
193 void JustDied(Unit* /*killer*/) override
194 {
195 DoAction(ACTION_DESPAWN_VOID_SENTRY_BALL);
196 }
197
198 private:
199 SummonList _summons;
200 };
201
202 CreatureAI* GetAI(Creature* creature) const override
203 {
204 return GetVioletHoldAI<npc_void_sentryAI>(creature);
205 }
206};

208class achievement_void_dance : public AchievementCriteriaScript
209{
210 public:
211 achievement_void_dance() : AchievementCriteriaScript("achievement_void_dance") { }
212
213 bool OnCheck(Player* /*player*/, Unit* target) override
214 {
215 if (!target)
216 return false;
217
218 if (Creature* Zuramat = target->ToCreature())
219 if (Zuramat->AI()->GetData(DATA_VOID_DANCE))
220 return true;
221
222 return false;
223 }
224};

226void AddSC_boss_zuramat()
227{
228 new boss_zuramat();
229 new npc_void_sentry();
230 new achievement_void_dance();
231}

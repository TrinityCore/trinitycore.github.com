72class boss_ichoron : public CreatureScript
73{
74 public:
75 boss_ichoron() : CreatureScript("boss_ichoron") { }
76
77 struct boss_ichoronAI : public BossAI
78 {
79 boss_ichoronAI(Creature* creature) : BossAI(creature, DATA_ICHORON)
80 {
81 Initialize();
82
83 /// for some reason ichoron can't walk back to it's water basin on evade
84 me->AddUnitState(UNIT_STATE_IGNORE_PATHFINDING);
85 }
86
87 void Initialize()
88 {
89 _isFrenzy = false;
90 _dehydration = true;
91 }
92
93 void Reset() override
94 {
95 Initialize();
96 BossAI::Reset();
97
98 DoCast(me, SPELL_THREAT_PROC, true);
99 }
100
101 void EnterCombat(Unit* who) override
102 {
103 BossAI::EnterCombat(who);
104 Talk(SAY_AGGRO);
105 }
106
107 void JustReachedHome() override
108 {
109 BossAI::JustReachedHome();
110 instance->SetData(DATA_HANDLE_CELLS, DATA_ICHORON);
111 }
112
113 void DoAction(int32 actionId) override
114 {
115 switch (actionId)
116 {
117 case ACTION_WATER_GLOBULE_HIT:
118 if (!me->IsAlive())
119 break;
120
121 me->ModifyHealth(int32(me->CountPctFromMaxHealth(3)));
122 _dehydration = false;
123 break;
124 case ACTION_PROTECTIVE_BUBBLE_SHATTERED:
125 {
126 Talk(SAY_SHATTER);
127 Talk(EMOTE_SHATTER);
128
129 DoCastAOE(SPELL_SPLATTER, true);
130 DoCastAOE(SPELL_BURST, true);
131 DoCast(me, SPELL_DRAINED, true);
132
133 uint32 damage = me->CountPctFromMaxHealth(30);
134 me->LowerPlayerDamageReq(damage);
135 me->ModifyHealth(-std::min<int32>(damage, me->GetHealth() - 1));
136
137 scheduler.DelayAll(Seconds(15));
138 break;
139 }
140 case ACTION_DRAINED:
141 if (HealthAbovePct(30))
142 {
143 Talk(SAY_BUBBLE);
144 DoCast(me, SPELL_PROTECTIVE_BUBBLE, true);
145 }
146 break;
147 default:
148 break;
149 }
150 }
151
152 uint32 GetData(uint32 type) const override
153 {
154 if (type == DATA_DEHYDRATION)
155 return _dehydration ? 1 : 0;
156 return 0;
157 }
158
159 void KilledUnit(Unit* victim) override
160 {
161 if (victim->GetTypeId() == TYPEID_PLAYER)
162 Talk(SAY_SLAY);
163 }
164
165 void JustDied(Unit* killer) override
166 {
167 BossAI::JustDied(killer);
168 Talk(SAY_DEATH);
169 }
170
171 void JustSummoned(Creature* summon) override
172 {
173 summons.Summon(summon);
174
175 if (summon->GetEntry() == NPC_ICHOR_GLOBULE)
176 DoCast(summon, SPELL_WATER_GLOBULE_VISUAL);
177 }
178
179 void SummonedCreatureDespawn(Creature* summon) override
180 {
181 BossAI::SummonedCreatureDespawn(summon);
182
183 if (summons.empty())
184 me->RemoveAurasDueToSpell(SPELL_DRAINED, ObjectGuid::Empty, 0, AURA_REMOVE_BY_EXPIRE);
185 }
186
187 void UpdateAI(uint32 diff) override
188 {
189 if (!UpdateVictim())
190 return;
191
192 if (!_isFrenzy && HealthBelowPct(25) && !me->HasAura(SPELL_DRAINED))
193 {
194 Talk(SAY_ENRAGE);
195 DoCast(me, SPELL_FRENZY, true);
196 _isFrenzy = true;
197 }
198
199 scheduler.Update(diff,
200 std::bind(&BossAI::DoMeleeAttackIfReady, this));
201 }
202
203 void ScheduleTasks() override
204 {
205 scheduler.Async([this]
206 {
207 DoCast(me, SPELL_SHRINK);
208 DoCast(me, SPELL_PROTECTIVE_BUBBLE);
209 });
210
211 scheduler.Schedule(Seconds(10), Seconds(15), [this](TaskContext task)
212 {
213 DoCastAOE(SPELL_WATER_BOLT_VOLLEY);
214 task.Repeat(Seconds(10), Seconds(15));
215 });
216
217 scheduler.Schedule(Seconds(6), Seconds(9), [this](TaskContext task)
218 {
219 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f))
220 DoCast(target, SPELL_WATER_BLAST);
221 task.Repeat(Seconds(6), Seconds(9));
222 });
223 }
224
225 private:
226 bool _isFrenzy;
227 bool _dehydration;
228 };
229
230 CreatureAI* GetAI(Creature* creature) const override
231 {
232 return GetVioletHoldAI<boss_ichoronAI>(creature);
233 }
234};

236class npc_ichor_globule : public CreatureScript
237{
238 public:
239 npc_ichor_globule() : CreatureScript("npc_ichor_globule") { }
240
241 struct npc_ichor_globuleAI : public ScriptedAI
242 {
243 npc_ichor_globuleAI(Creature* creature) : ScriptedAI(creature)
244 {
245 _instance = creature->GetInstanceScript();
246 creature->SetReactState(REACT_PASSIVE);
247 }
248
249 void SpellHit(Unit* caster, SpellInfo const* spellInfo) override
250 {
251 if (spellInfo->Id == SPELL_WATER_GLOBULE_VISUAL)
252 {
253 DoCast(me, SPELL_WATER_GLOBULE_TRANSFORM);
254 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
255 me->GetMotionMaster()->MoveFollow(caster, 0.0f, 0.0f);
256 }
257 }
258
259 void MovementInform(uint32 type, uint32 id) override
260 {
261 if (type != FOLLOW_MOTION_TYPE)
262 return;
263
264 if (_instance->GetObjectGuid(DATA_ICHORON).GetCounter() != id)
265 return;
266
267 me->CastSpell(me, SPELL_MERGE);
268 me->DespawnOrUnsummon(1);
269 }
270
271 // on retail spell casted on a creature's death are not casted after death but keeping mob at 1 health, casting it and then letting the mob die.
272 // this feature should be still implemented
273 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
274 {
275 if (damage >= me->GetHealth())
276 DoCastAOE(SPELL_SPLASH);
277 }
278
279 void UpdateAI(uint32 /*diff*/) override { }
280
281 private:
282 InstanceScript* _instance;
283 };
284
285 CreatureAI* GetAI(Creature* creature) const override
286 {
287 return GetVioletHoldAI<npc_ichor_globuleAI>(creature);
288 }

292class spell_ichoron_drained : public SpellScriptLoader
293{
294 public:
295 spell_ichoron_drained() : SpellScriptLoader("spell_ichoron_drained") { }
296
297 class spell_ichoron_drained_AuraScript : public AuraScript
298 {
299 PrepareAuraScript(spell_ichoron_drained_AuraScript);
300
301 bool Load() override
302 {
303 return GetOwner()->GetEntry() == NPC_ICHORON || GetOwner()->GetEntry() == NPC_DUMMY_ICHORON;
304 }
305
306 void HandleApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
307 {
308 GetTarget()->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_UNK_31);
309 GetTarget()->SetFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
310 }
311
312 void HandleRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
313 {
314 GetTarget()->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_UNK_31);
315 GetTarget()->RemoveFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
316
317 if (GetTargetApplication()->GetRemoveMode() == AURA_REMOVE_BY_EXPIRE)
318 if (GetTarget()->IsAIEnabled)
319 GetTarget()->GetAI()->DoAction(ACTION_DRAINED);
320 }
321
322 void Register() override
323 {
324 AfterEffectApply += AuraEffectApplyFn(spell_ichoron_drained_AuraScript::HandleApply, EFFECT_0, SPELL_AURA_MOD_STUN, AURA_EFFECT_HANDLE_REAL);
325 AfterEffectRemove += AuraEffectRemoveFn(spell_ichoron_drained_AuraScript::HandleRemove, EFFECT_0, SPELL_AURA_MOD_STUN, AURA_EFFECT_HANDLE_REAL);
326 }
327 };
328
329 AuraScript* GetAuraScript() const override
330 {
331 return new spell_ichoron_drained_AuraScript();
332 }

342 {
343 PrepareSpellScript(spell_ichoron_merge_SpellScript);
344
345 bool Validate(SpellInfo const* /*spellInfo*/) override
346 {
347 if (!sSpellMgr->GetSpellInfo(SPELL_SHRINK))
348 return false;
349 return true;
350 }
351
352 void HandleScript(SpellEffIndex /*effIndex*/)
353 {
354 if (Creature* target = GetHitCreature())
355 {
356 if (Aura* aura = target->GetAura(SPELL_SHRINK))
357 aura->ModStackAmount(-1);
358
359 target->AI()->DoAction(ACTION_WATER_GLOBULE_HIT);
360 }
361 }
362
363 void Register() override
364 {
365 OnEffectHitTarget += SpellEffectFn(spell_ichoron_merge_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
366 }
367 };
368
369 SpellScript* GetSpellScript() const override
370 {
371 return new spell_ichoron_merge_SpellScript();
372 }

411class spell_ichoron_splatter : public SpellScriptLoader
412{
413 public:
414 spell_ichoron_splatter() : SpellScriptLoader("spell_ichoron_splatter") { }
415
416 class spell_ichoron_splatter_AuraScript : public AuraScript
417 {
418 PrepareAuraScript(spell_ichoron_splatter_AuraScript);
419
420 bool Validate(SpellInfo const* /*spellInfo*/) override
421 {
422 if (!sSpellMgr->GetSpellInfo(SPELL_WATER_GLOBULE_SUMMON_1)
423 || !sSpellMgr->GetSpellInfo(SPELL_WATER_GLOBULE_SUMMON_2)
424 || !sSpellMgr->GetSpellInfo(SPELL_WATER_GLOBULE_SUMMON_3)
425 || !sSpellMgr->GetSpellInfo(SPELL_WATER_GLOBULE_SUMMON_4)
426 || !sSpellMgr->GetSpellInfo(SPELL_WATER_GLOBULE_SUMMON_5)
427 || !sSpellMgr->GetSpellInfo(SPELL_SHRINK))
428 return false;
429 return true;
430 }
431
432 void PeriodicTick(AuraEffect const* /*aurEff*/)
433 {
434 PreventDefaultAction();
435 GetTarget()->CastSpell(GetTarget(), RAND(SPELL_WATER_GLOBULE_SUMMON_1, SPELL_WATER_GLOBULE_SUMMON_2, SPELL_WATER_GLOBULE_SUMMON_3, SPELL_WATER_GLOBULE_SUMMON_4, SPELL_WATER_GLOBULE_SUMMON_5), true);
436 }
437
438 void HandleRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
439 {
440 if (GetTargetApplication()->GetRemoveMode() == AURA_REMOVE_BY_EXPIRE)
441 if (Aura* aura = GetTarget()->GetAura(SPELL_SHRINK))
442 aura->ModStackAmount(10);
443 }
444
445 void Register() override
446 {
447 OnEffectPeriodic += AuraEffectPeriodicFn(spell_ichoron_splatter_AuraScript::PeriodicTick, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
448 AfterEffectRemove += AuraEffectRemoveFn(spell_ichoron_splatter_AuraScript::HandleRemove, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY, AURA_EFFECT_HANDLE_REAL);
449 }
450 };
451
452 AuraScript* GetAuraScript() const override
453 {
454 return new spell_ichoron_splatter_AuraScript();
455 }

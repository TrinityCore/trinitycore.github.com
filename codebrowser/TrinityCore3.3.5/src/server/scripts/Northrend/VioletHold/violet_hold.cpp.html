320 bool OnGossipHello(Player* player, Creature* creature) override
321 {
322 // override default gossip
323 if (InstanceScript* instance = creature->GetInstanceScript())
324 {
325 switch (instance->GetData(DATA_MAIN_EVENT_STATE))
326 {
327 case IN_PROGRESS:
328 player->PrepareGossipMenu(creature, GOSSIP_MENU_SEND_ME_IN, true);
329 player->SendPreparedGossip(creature);
330 return true;
331 case DONE:
332 return true; // NYI
333 case NOT_STARTED:
334 case FAIL:
335 default:
336 break;
337 }
338 }
339
340 // load default gossip
341 return false;
342 }

344 struct npc_sinclariAI : public ScriptedAI
345 {
346 npc_sinclariAI(Creature* creature) : ScriptedAI(creature), _summons(creature)
347 {
348 _instance = creature->GetInstanceScript();
349 }
350
351 void Reset() override
352 {
353 _summons.DespawnAll();
354 for (uint8 i = 0; i < PortalIntroCount; ++i)
355 if (Creature* summon = me->SummonCreature(NPC_TELEPORTATION_PORTAL_INTRO, PortalIntroPositions[i], TEMPSUMMON_MANUAL_DESPAWN))
356 summon->AI()->SetData(DATA_PORTAL_LOCATION, i);
357
358 me->SetVisible(true);
359 me->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
360
361 std::list<Creature*> guardList;
362 me->GetCreatureListWithEntryInGrid(guardList, NPC_VIOLET_HOLD_GUARD, 100.0f);
363 for (Creature* guard : guardList)
364 {
365 guard->Respawn(true);
366 guard->SetVisible(true);
367 guard->SetReactState(REACT_AGGRESSIVE);
368 guard->AI()->EnterEvadeMode();
369 }
370 }
371
372 void sGossipSelect(Player* player, uint32 menuId, uint32 gossipListId) override
373 {
374 if (menuId == GOSSIP_MENU_START_ENCOUNTER && gossipListId == 0)
375 {
376 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
377 _instance->SetData(DATA_MAIN_EVENT_STATE, SPECIAL);
378 ScheduleIntro();
379 player->PlayerTalkClass->SendCloseGossip();
380 }
381 else if (menuId == GOSSIP_MENU_SEND_ME_IN && gossipListId == 0)
382 {
383 me->CastSpell(player, SPELL_TELEPORT_PLAYER, true);
384 player->PlayerTalkClass->SendCloseGossip();
385 }
386 }
387
388 void DoAction(int32 actionId) override
389 {
390 if (actionId == ACTION_SINCLARI_OUTRO)
391 {
392 me->SetVisible(true);
393 ScheduleOutro();
394 }
395 }
396
397 void UpdateAI(uint32 diff) override
398 {
399 _scheduler.Update(diff);
400
401 if (!UpdateVictim())
402 return;
403
404 DoMeleeAttackIfReady();
405 }

407 void ScheduleIntro()
408 {
409 _scheduler.Schedule(Seconds(1), [this](TaskContext task)
410 {
411 switch (task.GetRepeatCounter())
412 {
413 case 0:
414 me->SetWalk(true);
415 me->GetMotionMaster()->MovePoint(0, SinclariPositions[0]);
416 task.Repeat(Seconds(1));
417 break;
418 case 1:
419 me->HandleEmoteCommand(EMOTE_ONESHOT_USE_STANDING);
420 me->GetMap()->SummonCreature(NPC_DEFENSE_SYSTEM, DefenseSystemLocation);
421 task.Repeat(Seconds(3));
422 break;
423 case 2:
424 me->SetFacingTo(SinclariPositions[0].GetOrientation());
425 Talk(SAY_SINCLARI_INTRO_1);
426
427 task.Schedule(Seconds(1), [this](TaskContext /*task*/)
428 {
429 std::list<Creature*> guardList;
430 me->GetCreatureListWithEntryInGrid(guardList, NPC_VIOLET_HOLD_GUARD, 100.0f);
431 for (Creature* guard : guardList)
432 {
433 guard->SetReactState(REACT_PASSIVE);
434 guard->SetWalk(false);
435 guard->GetMotionMaster()->MovePoint(0, GuardsMovePosition);
436 }
437 });
438
439 task.Repeat(Seconds(2));
440 break;
441 case 3:
442 me->GetMotionMaster()->MovePoint(0, SinclariPositions[1]);
443 _summons.DespawnAll();
444 task.Repeat(Seconds(5));
445 break;
446 case 4:
447 me->SetFacingTo(SinclariPositions[1].GetOrientation());
448
449 task.Schedule(Seconds(1), [this](TaskContext /*task*/)
450 {
451 std::list<Creature*> guardList;
452 me->GetCreatureListWithEntryInGrid(guardList, NPC_VIOLET_HOLD_GUARD, 100.0f);
453 for (Creature* guard : guardList)
454 guard->SetVisible(false);
455 });
456
457 task.Repeat(Seconds(6));
458 break;
459 case 5:
460 Talk(SAY_SINCLARI_INTRO_2);
461 task.Repeat(Seconds(4));
462 break;
463 case 6:
464 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK_NO_SHEATHE);
465 task.Repeat(Seconds(1));
466 break;
467 case 7:
468 if (GameObject* mainDoor = _instance->GetGameObject(DATA_MAIN_DOOR))
469 {
470 mainDoor->SetGoState(GO_STATE_READY);
471 mainDoor->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
472 }
473 task.Repeat(Seconds(5));
474 break;
475 case 8:
476 me->SetVisible(false);
477 task.Repeat(Seconds(1));
478 break;
479 case 9:
480 _instance->SetData(DATA_MAIN_EVENT_STATE, IN_PROGRESS);
481 // [1] GUID: Full: 0xF1300077C202E6DD Type: Creature Entry: 30658 Low: 190173
482 break;
483 default:
484 break;
485 }
486 });
487 }

533 struct npc_azure_saboteurAI : public ScriptedAI
534 {
535 npc_azure_saboteurAI(Creature* creature) : ScriptedAI(creature)
536 {
537 _instance = creature->GetInstanceScript();
538
539 if (_instance->GetData(DATA_WAVE_COUNT) == 6)
540 _bossId = _instance->GetData(DATA_1ST_BOSS);
541 else
542 _bossId = _instance->GetData(DATA_2ND_BOSS);
543 }
544
545 void StartMovement()
546 {
547 uint32 pathSize = 0;
548 G3D::Vector3 const* path = nullptr;
549
550 switch (_bossId)
551 {
552 case DATA_MORAGG:
553 pathSize = SaboteurMoraggPathSize;
554 path = SaboteurMoraggPath;
555 break;
556 case DATA_EREKEM:
557 pathSize = SaboteurErekemPathSize;
558 path = SaboteurErekemPath;
559 break;
560 case DATA_ICHORON:
561 pathSize = SaboteurIchoronPathSize;
562 path = SaboteurIchoronPath;
563 break;
564 case DATA_LAVANTHOR:
565 pathSize = SaboteurLavanthorPathSize;
566 path = SaboteurLavanthorPath;
567 break;
568 case DATA_XEVOZZ:
569 pathSize = SaboteurXevozzPathSize;
570 path = SaboteurXevozzPath;
571 break;
572 case DATA_ZURAMAT:
573 pathSize = SaboteurZuramatPathSize;
574 path = SaboteurZuramatPath;
575 break;
576 }
577
578 if (path)
579 me->GetMotionMaster()->MoveSmoothPath(POINT_INTRO, path, pathSize, false);
580 }
581
582 void Reset() override
583 {
584 _scheduler.CancelAll();
585 _scheduler.Schedule(Seconds(2), [this](TaskContext /*task*/)
586 {
587 StartMovement();
588 });
589 }
590
591 void MovementInform(uint32 type, uint32 pointId) override
592 {
593 if (type == EFFECT_MOTION_TYPE && pointId == POINT_INTRO)
594 {
595 _scheduler.Schedule(Seconds(0), [this](TaskContext task)
596 {
597 me->CastSpell(me, SPELL_SHIELD_DISRUPTION, false);
598
599 if (task.GetRepeatCounter() < 2)
600 task.Repeat(Seconds(1));
601 else
602 {
603 task.Schedule(Seconds(2), [this](TaskContext /*task*/)
604 {
605 _instance->SetData(DATA_START_BOSS_ENCOUNTER, 1);
606 me->CastSpell(me, SPELL_TELEPORT_VISUAL, false);
607 me->DespawnOrUnsummon(1000);
608 });
609 }
610 });
611 }
612 }
613
614 void UpdateAI(uint32 diff) override
615 {
616 _scheduler.Update(diff);
617 }

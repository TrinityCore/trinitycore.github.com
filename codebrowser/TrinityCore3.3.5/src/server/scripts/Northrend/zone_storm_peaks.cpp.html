52 void WaypointReached(uint32 waypointId) override
53 {
54 Player* player = GetPlayerForEscort();
55 if (!player)
56 return;
57
58 switch (waypointId)
59 {
60 case 26:
61 Talk(SAY_END_WP_REACHED, player);
62 break;
63 case 27:
64 player->GroupEventHappens(QUEST_BITTER_DEPARTURE, me);
65 break;

67 }
68
69 void EnterCombat(Unit* /*who*/) override { }
70
71 void Reset() override { }
72
73 void JustDied(Unit* /*killer*/) override
74 {
75 Player* player = GetPlayerForEscort();
76 if (HasEscortState(STATE_ESCORT_ESCORTING) && player)
77 player->FailQuest(QUEST_BITTER_DEPARTURE);
78 }
79
80 void UpdateAI(uint32 uiDiff) override
81 {
82 npc_escortAI::UpdateAI(uiDiff);
83 if (!UpdateVictim())
84 return;
85 DoMeleeAttackIfReady();
86 }
87
88 void sGossipSelect(Player* player, uint32 menuId, uint32 gossipListId) override
89 {
90 if (menuId == GOSSIP_ID && gossipListId == GOSSIP_OPTION_ID)
91 {
92 player->CLOSE_GOSSIP_MENU();
93 me->setFaction(113);
94 Start(true, true, player->GetGUID());
95 }
96 }
97 };
98
99 CreatureAI* GetAI(Creature* creature) const override
100 {
101 return new npc_injured_goblinAI(creature);
102 }
103
104 bool OnQuestAccept(Player* /*player*/, Creature* creature, Quest const* quest) override
105 {
106 if (quest->GetQuestId() == QUEST_BITTER_DEPARTURE)
107 creature->AI()->Talk(SAY_QUEST_ACCEPT);
108
109 return false;

256class npc_freed_protodrake : public CreatureScript
257{
258public:
259 npc_freed_protodrake() : CreatureScript("npc_freed_protodrake") { }
260
261 struct npc_freed_protodrakeAI : public VehicleAI
262 {
263 npc_freed_protodrakeAI(Creature* creature) : VehicleAI(creature) { }
264
265 EventMap events;
266
267 void Reset() override
268 {
269 events.ScheduleEvent(EVENT_CHECK_AREA, 5000);
270 }
271
272 void MovementInform(uint32 type, uint32 id) override
273 {
274 if (type != WAYPOINT_MOTION_TYPE)
275 return;
276
277 if (id == 15)
278 // drake reached village
279 events.ScheduleEvent(EVENT_REACHED_HOME, 2000);
280 }
281
282 void UpdateAI(uint32 diff) override
283 {
284 events.Update(diff);
285
286 switch (events.ExecuteEvent())
287 {
288 case EVENT_CHECK_AREA:
289 if (me->GetAreaId() == AREA_VALLEY_OF_ANCIENT_WINTERS)
290 {
291 if (Vehicle* vehicle = me->GetVehicleKit())
292 if (Unit* passenger = vehicle->GetPassenger(0))
293 {
294 Talk(TEXT_EMOTE, passenger);
295 me->GetMotionMaster()->MovePath(NPC_DRAKE, false);
296 }
297 }
298 else
299 events.ScheduleEvent(EVENT_CHECK_AREA, 5000);
300 break;
301 case EVENT_REACHED_HOME:
302 if (Vehicle* vehicle = me->GetVehicleKit())
303 if (Unit* player = vehicle->GetPassenger(0))
304 if (player->GetTypeId() == TYPEID_PLAYER)
305 {
306 // for each prisoner on drake, give credit
307 for (uint8 i = 1; i < 4; ++i)
308 if (Unit* prisoner = me->GetVehicleKit()->GetPassenger(i))
309 {
310 if (prisoner->GetTypeId() != TYPEID_UNIT)
311 return;
312 prisoner->CastSpell(player, SPELL_KILL_CREDIT_PRISONER, true);
313 prisoner->CastSpell(prisoner, SPELL_SUMMON_LIBERATED, true);
314 prisoner->ExitVehicle();
315 }
316 me->CastSpell(me, SPELL_KILL_CREDIT_DRAKE, true);
317 player->ExitVehicle();
318 }
319 break;
320 }
321 }
322 };
323
324 CreatureAI* GetAI(Creature* creature) const override
325 {
326 return new npc_freed_protodrakeAI(creature);
327 }

382 {

385
386 void PassengerBoarded(Unit* who, int8 /*seat*/, bool apply) override
387 {
388 if (apply)
389 return;
390
391 if (who->GetEntry() == NPC_HYLDSMEET_DRAKERIDER)
392 _accessoryRespawnTimer = 5 * MINUTE * IN_MILLISECONDS;
393 }
394
395 void UpdateAI(uint32 diff) override
396 {
397 //! We need to manually reinstall accessories because the vehicle itself is friendly to players,
398 //! so EnterEvadeMode is never triggered. The accessory on the other hand is hostile and killable.
399 Vehicle* _vehicleKit = me->GetVehicleKit();
400 if (_accessoryRespawnTimer && _accessoryRespawnTimer <= diff && _vehicleKit)
401 {
402 _vehicleKit->InstallAllAccessories(true);
403 _accessoryRespawnTimer = 0;
404 }
405 else
406 _accessoryRespawnTimer -= diff;
407 }
408
409 private:
410 uint32 _accessoryRespawnTimer;
411 };
412
413 CreatureAI* GetAI(Creature* creature) const override
414 {
415 return new npc_hyldsmeet_protodrakeAI(creature);
416 }

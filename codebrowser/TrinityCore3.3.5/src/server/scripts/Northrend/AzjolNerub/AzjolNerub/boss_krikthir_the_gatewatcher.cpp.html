83class boss_krik_thir : public CreatureScript
84{
85 public:
86 boss_krik_thir() : CreatureScript("boss_krik_thir") { }
87
88 struct boss_krik_thirAI : public BossAI
89 {
90 boss_krik_thirAI(Creature* creature) : BossAI(creature, DATA_KRIKTHIR_THE_GATEWATCHER) { }
91
92 void Reset() override
93 {
94 _Reset();
95 }
96
97 void EnterCombat(Unit* /*who*/) override
98 {
99 Talk(SAY_AGGRO);
100 _EnterCombat();
101 Summon();
102
103 events.ScheduleEvent(EVENT_SUMMON, 15000);
104 events.ScheduleEvent(EVENT_MIND_FLAY, 15000);
105 events.ScheduleEvent(EVENT_CURSE_FATIGUE, 12000);
106 }
107
108 void Summon()
109 {
110 for (uint8 i = 0; i < 8; i++)
111 {
112 me->SummonCreature(NPC_SKITTERING_SWARMER, SpawnPoint[i], TEMPSUMMON_TIMED_DESPAWN, 25000);
113 uint32 summon_entry = (i == 4 || i == 5 || i == 6) ? NPC_SKITTERING_INFECTIOR : NPC_SKITTERING_SWARMER;
114 me->SummonCreature(summon_entry, SpawnPoint[i], TEMPSUMMON_TIMED_DESPAWN, 25000);
115 }
116 }
117
118 void UpdateAI(uint32 diff) override
119 {
120 if (!UpdateVictim())
121 return;
122
123 events.Update(diff);
124
125 if (me->HasUnitState(UNIT_STATE_CASTING))
126 return;
127
128 while (uint32 eventId = events.ExecuteEvent())
129 {
130 switch (eventId)
131 {
132 case EVENT_SUMMON:
133 Summon();
134 events.ScheduleEvent(EVENT_SUMMON, 15000);
135 break;
136 case EVENT_MIND_FLAY:
137 DoCastVictim(SPELL_MIND_FLAY);
138 events.ScheduleEvent(EVENT_MIND_FLAY, 15000);
139 break;
140 case EVENT_CURSE_FATIGUE:
141 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
142 DoCast(target, SPELL_CURSE_OF_FATIGUE);
143 events.ScheduleEvent(EVENT_CURSE_FATIGUE, 10000);
144 break;
145 default:
146 break;
147 }
148 }
149
150 if (!me->HasAura(SPELL_FRENZY) && HealthBelowPct(10))
151 DoCast(me, SPELL_FRENZY, true);
152
153 DoMeleeAttackIfReady();
154 }
155
156 void JustDied(Unit* /*killer*/) override
157 {
158 Talk(SAY_DEATH);
159 _JustDied();
160 }
161
162 void KilledUnit(Unit* victim) override
163 {
164 if (victim->GetTypeId() != TYPEID_PLAYER)
165 return;
166
167 Talk(SAY_SLAY);
168 }
169
170 void JustSummoned(Creature* summoned) override
171 {
172 summoned->GetMotionMaster()->MovePoint(0, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
173 }
174 };
175
176 CreatureAI* GetAI(Creature* creature) const override
177 {
178 return GetAzjolNerubAI<boss_krik_thirAI>(creature);
179 }

645class achievement_watch_him_die : public AchievementCriteriaScript
646{
647 public:
648 achievement_watch_him_die() : AchievementCriteriaScript("achievement_watch_him_die") { }
649
650 bool OnCheck(Player* /*player*/, Unit* target) override
651 {
652 if (!target)
653 return false;
654
655 InstanceScript* instance = target->GetInstanceScript();
656 if (!instance)
657 return false;
658
659 for (uint8 n = 0; n < 3; ++n)
660 {
661 if (Creature* watcher = instance->GetCreature(DATA_WATCHER_GASHRA + n))
662 if (!watcher->IsAlive())
663 return false;
664 }
665
666 return true;
667 }

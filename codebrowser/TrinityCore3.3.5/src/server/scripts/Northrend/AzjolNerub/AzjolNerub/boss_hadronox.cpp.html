18/*
19* Comment: No Waves atm and the doors spells are crazy...
20*
21* When your group enters the main room (the one after the bridge), you will notice a group of 3 Nerubians.
22* When you engage them, 2 more groups like this one spawn behind the first one - it is important to pull the first group back,
23* so you don't aggro all 3. Hadronox will be under you, fighting Nerubians.
24*
25* This is the timed gauntlet - waves of non-elite spiders
26* will spawn from the 3 doors located a little above the main room, and will then head down to fight Hadronox. After clearing the
27* main room, it is recommended to just stay in it, kill the occasional non-elites that will attack you instead of the boss, and wait for
28* Hadronox to make his way to you. When Hadronox enters the main room, she will web the doors, and no more non-elites will spawn.
29*/

36{
37 SPELL_ACID_CLOUD = 53400, // Victim
38 SPELL_LEECH_POISON = 53030, // Victim
39 SPELL_PIERCE_ARMOR = 53418, // Victim
40 SPELL_WEB_GRAB = 57731, // Victim
41 SPELL_WEB_FRONT_DOORS = 53177, // Self
42 SPELL_WEB_SIDE_DOORS = 53185, // Self
43 H_SPELL_ACID_CLOUD = 59419,
44 H_SPELL_LEECH_POISON = 59417,
45 H_SPELL_WEB_GRAB = 59421
46};
47
48class boss_hadronox : public CreatureScript
49{
50public:
51 boss_hadronox() : CreatureScript("boss_hadronox") { }
52
53 struct boss_hadronoxAI : public ScriptedAI
54 {
55 boss_hadronoxAI(Creature* creature) : ScriptedAI(creature)
56 {
57 Initialize();
58 instance = creature->GetInstanceScript();
59 fMaxDistance = 50.0f;
60 bFirstTime = true;
61 }
62
63 void Initialize()
64 {
65 uiAcidTimer = urand(10 * IN_MILLISECONDS, 14 * IN_MILLISECONDS);
66 uiLeechTimer = urand(3 * IN_MILLISECONDS, 9 * IN_MILLISECONDS);
67 uiPierceTimer = urand(1 * IN_MILLISECONDS, 3 * IN_MILLISECONDS);
68 uiGrabTimer = urand(15 * IN_MILLISECONDS, 19 * IN_MILLISECONDS);
69 uiDoorsTimer = urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS);
70 uiCheckDistanceTimer = 2 * IN_MILLISECONDS;
71 }
72
73 InstanceScript* instance;
74
75 uint32 uiAcidTimer;
76 uint32 uiLeechTimer;
77 uint32 uiPierceTimer;
78 uint32 uiGrabTimer;
79 uint32 uiDoorsTimer;
80 uint32 uiCheckDistanceTimer;
81
82 bool bFirstTime;
83
84 float fMaxDistance;
85
86 void Reset() override
87 {
88 me->SetFloatValue(UNIT_FIELD_BOUNDINGRADIUS, 9.0f);
89 me->SetFloatValue(UNIT_FIELD_COMBATREACH, 9.0f);
90
91 Initialize();
92
93 if (instance->GetBossState(DATA_HADRONOX) != DONE && !bFirstTime)
94 instance->SetBossState(DATA_HADRONOX, FAIL);
95
96 bFirstTime = false;
97 }
98
99 //when Hadronox kills any enemy (that includes a party member) she will regain 10% of her HP if the target had Leech Poison on
100 void KilledUnit(Unit* Victim) override
101 {
102 // not sure if this aura check is correct, I think it is though
103 if (!Victim || !Victim->HasAura(DUNGEON_MODE(SPELL_LEECH_POISON, H_SPELL_LEECH_POISON)) || !me->IsAlive())
104 return;
105
106 me->ModifyHealth(int32(me->CountPctFromMaxHealth(10)));
107 }
108
109 void JustDied(Unit* /*killer*/) override
110 {
111 instance->SetBossState(DATA_HADRONOX, DONE);
112 }
113
114 void EnterCombat(Unit* /*who*/) override
115 {
116 instance->SetBossState(DATA_HADRONOX, IN_PROGRESS);
117 me->SetInCombatWithZone();
118 }
119
120 void CheckDistance(float dist, const uint32 uiDiff)
121 {
122 if (!me->IsInCombat())
123 return;
124
125 float x=0.0f, y=0.0f, z=0.0f;
126 me->GetRespawnPosition(x, y, z);
127
128 if (uiCheckDistanceTimer <= uiDiff)
129 uiCheckDistanceTimer = 5*IN_MILLISECONDS;
130 else
131 {
132 uiCheckDistanceTimer -= uiDiff;
133 return;
134 }
135 if (me->IsInEvadeMode() || !me->GetVictim())
136 return;
137 if (me->GetDistance(x, y, z) > dist)
138 EnterEvadeMode();
139 }
140
141 void UpdateAI(uint32 diff) override
142 {
143 //Return since we have no target
144 if (!UpdateVictim())
145 return;
146
147 // Without he comes up through the air to players on the bridge after krikthir if players crossing this bridge!
148 CheckDistance(fMaxDistance, diff);
149
150 if (me->HasAura(SPELL_WEB_FRONT_DOORS) || me->HasAura(SPELL_WEB_SIDE_DOORS))
151 {
152 if (IsCombatMovementAllowed())
153 SetCombatMovement(false);
154 }
155 else if (!IsCombatMovementAllowed())
156 SetCombatMovement(true);
157
158 if (uiPierceTimer <= diff)
159 {
160 DoCastVictim(SPELL_PIERCE_ARMOR);
161 uiPierceTimer = 8*IN_MILLISECONDS;
162 } else uiPierceTimer -= diff;
163
164 if (uiAcidTimer <= diff)
165 {
166 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
167 DoCast(target, SPELL_ACID_CLOUD);
168
169 uiAcidTimer = urand(20*IN_MILLISECONDS, 30*IN_MILLISECONDS);
170 } else uiAcidTimer -= diff;
171
172 if (uiLeechTimer <= diff)
173 {
174 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
175 DoCast(target, SPELL_LEECH_POISON);
176
177 uiLeechTimer = urand(11*IN_MILLISECONDS, 14*IN_MILLISECONDS);
178 } else uiLeechTimer -= diff;
179
180 if (uiGrabTimer <= diff)
181 {
182 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0)) // Draws all players (and attacking Mobs) to itself.
183 DoCast(target, SPELL_WEB_GRAB);
184
185 uiGrabTimer = urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS);
186 } else uiGrabTimer -= diff;
187
188 if (uiDoorsTimer <= diff)
189 {
190 uiDoorsTimer = urand(30*IN_MILLISECONDS, 60*IN_MILLISECONDS);
191 } else uiDoorsTimer -= diff;
192
193 DoMeleeAttackIfReady();
194 }
195 };
196
197 CreatureAI* GetAI(Creature* creature) const override
198 {
199 return GetInstanceAI<boss_hadronoxAI>(creature);
200 }
201};

205 new boss_hadronox();

57class boss_elder_nadox : public CreatureScript
58{
59 public:
60 boss_elder_nadox() : CreatureScript("boss_elder_nadox") { }
61
62 struct boss_elder_nadoxAI : public BossAI
63 {
64 boss_elder_nadoxAI(Creature* creature) : BossAI(creature, DATA_ELDER_NADOX)
65 {
66 Initialize();
67 }
68
69 void Initialize()
70 {
71 GuardianSummoned = false;
72 GuardianDied = false;
73 }
74
75 void Reset() override
76 {
77 _Reset();
78 Initialize();
79 }
80
81 void EnterCombat(Unit* /*who*/) override
82 {
83 _EnterCombat();
84 Talk(SAY_AGGRO);
85
86 events.ScheduleEvent(EVENT_PLAGUE, 13 * IN_MILLISECONDS);
87 events.ScheduleEvent(EVENT_SUMMON_SWARMER, 10 * IN_MILLISECONDS);
88
89 if (IsHeroic())
90 {
91 events.ScheduleEvent(EVENT_RAGE, 12 * IN_MILLISECONDS);
92 events.ScheduleEvent(EVENT_CHECK_ENRAGE, 5 * IN_MILLISECONDS);
93 }
94 }
95
96 void SummonedCreatureDies(Creature* summon, Unit* /*killer*/) override
97 {
98 if (summon->GetEntry() == NPC_AHNKAHAR_GUARDIAN)
99 GuardianDied = true;
100 }
101
102 uint32 GetData(uint32 type) const override
103 {
104 if (type == DATA_RESPECT_YOUR_ELDERS)
105 return !GuardianDied ? 1 : 0;
106
107 return 0;
108 }
109
110 void KilledUnit(Unit* who) override
111 {
112 if (who->GetTypeId() == TYPEID_PLAYER)
113 Talk(SAY_SLAY);
114 }
115
116 void JustDied(Unit* /*killer*/) override
117 {
118 _JustDied();
119 Talk(SAY_DEATH);
120 }
121
122 void UpdateAI(uint32 diff) override
123 {
124 if (!UpdateVictim())
125 return;
126
127 events.Update(diff);
128
129 while (uint32 eventId = events.ExecuteEvent())
130 {
131 switch (eventId)
132 {
133 case EVENT_PLAGUE:
134 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_BROOD_PLAGUE, true);
135 events.ScheduleEvent(EVENT_PLAGUE, 15 * IN_MILLISECONDS);
136 break;
137 case EVENT_RAGE:
138 DoCast(H_SPELL_BROOD_RAGE);
139 events.ScheduleEvent(EVENT_RAGE, urand(10 * IN_MILLISECONDS, 50 * IN_MILLISECONDS));
140 break;
141 case EVENT_SUMMON_SWARMER:
142 /// @todo: summoned by egg
143 DoCast(me, SPELL_SUMMON_SWARMERS);
144 if (urand(1, 3) == 3) // 33% chance of dialog
145 Talk(SAY_EGG_SAC);
146 events.ScheduleEvent(EVENT_SUMMON_SWARMER, 10 * IN_MILLISECONDS);
147 break;
148 case EVENT_CHECK_ENRAGE:
149 if (me->HasAura(SPELL_ENRAGE))
150 return;
151 if (me->GetPositionZ() < 24.0f)
152 DoCast(me, SPELL_ENRAGE, true);
153 events.ScheduleEvent(EVENT_CHECK_ENRAGE, 5 * IN_MILLISECONDS);
154 break;
155 default:
156 break;
157 }
158 }
159
160 if (!GuardianSummoned && me->HealthBelowPct(50))
161 {
162 /// @todo: summoned by egg
163 Talk(EMOTE_HATCHES, me);
164 DoCast(me, SPELL_SUMMON_SWARM_GUARD);
165 GuardianSummoned = true;
166 }
167
168 DoMeleeAttackIfReady();
169 }
170
171 private:
172 bool GuardianSummoned;
173 bool GuardianDied;
174 };
175
176 CreatureAI* GetAI(Creature* creature) const override
177 {
178 return GetAhnKahetAI<boss_elder_nadoxAI>(creature);
179 }
180};

182class npc_ahnkahar_nerubian : public CreatureScript
183{
184 public:
185 npc_ahnkahar_nerubian() : CreatureScript("npc_ahnkahar_nerubian") { }
186
187 struct npc_ahnkahar_nerubianAI : public ScriptedAI
188 {
189 npc_ahnkahar_nerubianAI(Creature* creature) : ScriptedAI(creature) { }
190
191 void Reset() override
192 {
193 _events.Reset();
194 _events.ScheduleEvent(EVENT_SPRINT, 13 * IN_MILLISECONDS);
195 }
196
197 void UpdateAI(uint32 diff) override
198 {
199 if (!UpdateVictim())
200 return;
201
202 _events.Update(diff);
203
204 if (me->HasUnitState(UNIT_STATE_CASTING))
205 return;
206
207 while (uint32 eventId = _events.ExecuteEvent())
208 {
209 switch (eventId)
210 {
211 case EVENT_SPRINT:
212 DoCast(me, SPELL_SPRINT);
213 _events.ScheduleEvent(EVENT_SPRINT, 20 * IN_MILLISECONDS);
214 break;
215 }
216 }
217
218 DoMeleeAttackIfReady();
219 }
220
221 private:
222 EventMap _events;
223 };
224
225 CreatureAI* GetAI(Creature* creature) const override
226 {
227 return new npc_ahnkahar_nerubianAI(creature);
228 }

231// 56159 - Swarm
232class spell_ahn_kahet_swarm : public SpellScriptLoader
233{
234 public:
235 spell_ahn_kahet_swarm() : SpellScriptLoader("spell_ahn_kahet_swarm") { }
236
237 class spell_ahn_kahet_swarm_SpellScript : public SpellScript
238 {
239 PrepareSpellScript(spell_ahn_kahet_swarm_SpellScript);
240
241 public:
242 spell_ahn_kahet_swarm_SpellScript()
243 {
244 _targetCount = 0;
245 }
246
247 private:
248 bool Validate(SpellInfo const* /*spellInfo*/) override
249 {
250 if (!sSpellMgr->GetSpellInfo(SPELL_SWARM_BUFF))
251 return false;
252 return true;
253 }
254
255 void CountTargets(std::list<WorldObject*>& targets)
256 {
257 _targetCount = targets.size();
258 }
259
260 void HandleDummy(SpellEffIndex /*effIndex*/)
261 {
262 if (_targetCount)
263 {
264 if (Aura* aura = GetCaster()->GetAura(SPELL_SWARM_BUFF))
265 {
266 aura->SetStackAmount(_targetCount);
267 aura->RefreshDuration();
268 }
269 else
270 GetCaster()->CastCustomSpell(SPELL_SWARM_BUFF, SPELLVALUE_AURA_STACK, _targetCount, GetCaster(), TRIGGERED_FULL_MASK);
271 }
272 else
273 GetCaster()->RemoveAurasDueToSpell(SPELL_SWARM_BUFF);
274 }
275
276 void Register() override
277 {
278 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_ahn_kahet_swarm_SpellScript::CountTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ALLY);
279 OnEffectHit += SpellEffectFn(spell_ahn_kahet_swarm_SpellScript::HandleDummy, EFFECT_0, SPELL_EFFECT_DUMMY);
280 }
281
282 uint32 _targetCount;
283 };
284
285 SpellScript* GetSpellScript() const override
286 {
287 return new spell_ahn_kahet_swarm_SpellScript();
288 }

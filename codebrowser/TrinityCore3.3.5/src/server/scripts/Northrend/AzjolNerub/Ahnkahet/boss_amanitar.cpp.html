51class boss_amanitar : public CreatureScript
52{
53 public:
54 boss_amanitar() : CreatureScript("boss_amanitar") { }
55
56 struct boss_amanitarAI : public BossAI
57 {
58 boss_amanitarAI(Creature* creature) : BossAI(creature, DATA_AMANITAR) { }
59
60 void Reset() override
61 {
62 _Reset();
63 me->SetMeleeDamageSchool(SPELL_SCHOOL_NATURE);
64 me->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, true);
65 }
66
67 void EnterCombat(Unit* /*who*/) override
68 {
69 _EnterCombat();
70
71 events.ScheduleEvent(EVENT_ROOT, urand(5, 9) * IN_MILLISECONDS);
72 events.ScheduleEvent(EVENT_BASH, urand(10, 14) * IN_MILLISECONDS);
73 events.ScheduleEvent(EVENT_BOLT, urand(15, 20) * IN_MILLISECONDS);
74 events.ScheduleEvent(EVENT_MINI, urand(12, 18) * IN_MILLISECONDS);
75 events.ScheduleEvent(EVENT_SPAWN, 5 * IN_MILLISECONDS);
76 }
77
78 void JustDied(Unit* /*killer*/) override
79 {
80 _JustDied();
81 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_MINI);
82 }
83
84 void SpawnAdds()
85 {
86 uint8 u = 0;
87
88 for (uint8 i = 0; i < 30; ++i)
89 {
90 Position pos = me->GetRandomNearPosition(30.0f);
91 pos.m_positionZ = me->GetMap()->GetHeight(pos.GetPositionX(), pos.GetPositionY(), MAX_HEIGHT) + 2.0f;
92
93 if (Creature* trigger = me->SummonCreature(NPC_TRIGGER, pos))
94 {
95 Creature* temp1 = trigger->FindNearestCreature(NPC_HEALTHY_MUSHROOM, 4.0f, true);
96 Creature* temp2 = trigger->FindNearestCreature(NPC_POISONOUS_MUSHROOM, 4.0f, true);
97 if (temp1 || temp2)
98 {
99 trigger->DisappearAndDie();
100 }
101 else
102 {
103 u = 1 - u;
104 trigger->DisappearAndDie();
105 me->SummonCreature(u > 0 ? NPC_POISONOUS_MUSHROOM : NPC_HEALTHY_MUSHROOM, pos, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 60 * IN_MILLISECONDS);
106 }
107 }
108 }
109 }
110
111 void UpdateAI(uint32 diff) override
112 {
113 if (!UpdateVictim())
114 return;
115
116 events.Update(diff);
117
118 if (me->HasUnitState(UNIT_STATE_CASTING))
119 return;
120
121 while (uint32 eventId = events.ExecuteEvent())
122 {
123 switch (eventId)
124 {
125 case EVENT_SPAWN:
126 SpawnAdds();
127 events.ScheduleEvent(EVENT_SPAWN, 20 * IN_MILLISECONDS);
128 break;
129 case EVENT_MINI:
130 DoCast(SPELL_MINI);
131 events.ScheduleEvent(EVENT_MINI, urand(25, 30) * IN_MILLISECONDS);
132 break;
133 case EVENT_ROOT:
134 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_ENTANGLING_ROOTS, true);
135 events.ScheduleEvent(EVENT_ROOT, urand(10, 15) * IN_MILLISECONDS);
136 break;
137 case EVENT_BASH:
138 DoCastVictim(SPELL_BASH);
139 events.ScheduleEvent(EVENT_BASH, urand(7, 12) * IN_MILLISECONDS);
140 break;
141 case EVENT_BOLT:
142 DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true), SPELL_VENOM_BOLT_VOLLEY, true);
143 events.ScheduleEvent(EVENT_BOLT, urand(18, 22) * IN_MILLISECONDS);
144 break;
145 default:
146 break;
147 }
148 }
149 DoMeleeAttackIfReady();
150 }
151 };
152
153 CreatureAI* GetAI(Creature* creature) const override
154 {
155 return GetAhnKahetAI<boss_amanitarAI>(creature);
156 }
157};

159class npc_amanitar_mushrooms : public CreatureScript
160{
161public:
162 npc_amanitar_mushrooms() : CreatureScript("npc_amanitar_mushrooms") { }
163
164 struct npc_amanitar_mushroomsAI : public ScriptedAI
165 {
166 npc_amanitar_mushroomsAI(Creature* creature) : ScriptedAI(creature) { }
167
168 EventMap events;
169
170 void Reset() override
171 {
172 events.Reset();
173 events.ScheduleEvent(EVENT_AURA, 1 * IN_MILLISECONDS);
174
175 me->SetDisplayId(me->GetCreatureTemplate()->Modelid2);
176 DoCast(SPELL_PUTRID_MUSHROOM);
177
178 if (me->GetEntry() == NPC_POISONOUS_MUSHROOM)
179 DoCast(SPELL_POISONOUS_MUSHROOM_VISUAL_AURA);
180 else
181 DoCast(SPELL_POWER_MUSHROOM_VISUAL_AURA);
182 }
183
184 void DamageTaken(Unit* /*attacker*/, uint32 &damage) override
185 {
186 if (damage >= me->GetHealth() && me->GetEntry() == NPC_HEALTHY_MUSHROOM)
187 DoCast(me, SPELL_HEALTHY_MUSHROOM_POTENT_FUNGUS, true);
188 }
189
190 void EnterCombat(Unit* /*who*/) override { }
191 void AttackStart(Unit* /*victim*/) override { }
192
193 void UpdateAI(uint32 diff) override
194 {
195 if (!UpdateVictim())
196 return;
197
198 events.Update(diff);
199
200 if (me->HasUnitState(UNIT_STATE_CASTING))
201 return;
202
203 while (uint32 eventId = events.ExecuteEvent())
204 {
205 switch (eventId)
206 {
207 case EVENT_AURA:
208 if (me->GetEntry() == NPC_POISONOUS_MUSHROOM)
209 {
210 DoCast(me, SPELL_POISONOUS_MUSHROOM_VISUAL_AREA, true);
211 DoCast(me, SPELL_POISONOUS_MUSHROOM_POISON_CLOUD);
212 }
213 events.ScheduleEvent(EVENT_AURA, 7 * IN_MILLISECONDS);
214 break;
215 default:
216 break;
217 }
218 }
219 }
220 };
221
222 CreatureAI* GetAI(Creature* creature) const override
223 {
224 return new npc_amanitar_mushroomsAI(creature);
225 }
226};

231 new npc_amanitar_mushrooms();

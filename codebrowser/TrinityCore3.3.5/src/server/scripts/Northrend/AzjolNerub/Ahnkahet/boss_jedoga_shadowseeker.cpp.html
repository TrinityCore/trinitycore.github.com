119 {

164 void MoveInLineOfSight(Unit* who) override
165 {
166 if (!instance || !who || (who->GetTypeId() == TYPEID_UNIT && who->GetEntry() == NPC_JEDOGA_CONTROLLER))
167 return;
168
169 if (!bPreDone && who->GetTypeId() == TYPEID_PLAYER && me->GetDistance(who) < 100.0f)
170 {
171 Talk(TEXT_PREACHING);
172 bPreDone = true;
173 }
174
175 if (instance->GetBossState(DATA_JEDOGA_SHADOWSEEKER) != IN_PROGRESS || !bOnGround)
176 return;
177
178 if (!me->GetVictim() && me->CanCreatureAttack(who))
179 {
180 float attackRadius = me->GetAttackDistance(who);
181 if (me->IsWithinDistInMap(who, attackRadius) && me->IsWithinLOSInMap(who))
182 {
183 if (!me->GetVictim())
184 {
185 who->RemoveAurasByType(SPELL_AURA_MOD_STEALTH);
186 AttackStart(who);
187 }
188 else
189 {
190 who->SetInCombatWith(me);
191 me->AddThreat(who, 0.0f);
192 }
193 }
194 }
195 }
196
197 void MoveDown()
198 {
199 bOpFerokFail = false;
200
201 instance->SetData(DATA_JEDOGA_TRIGGER_SWITCH, 0);
202 me->GetMotionMaster()->MovePoint(1, JedogaPosition[1]);
203 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, false);
204 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_MAGIC, false);
205 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE + UNIT_FLAG_NON_ATTACKABLE);
206
207 me->RemoveAurasDueToSpell(SPELL_SPHERE_VISUAL);
208
209 bOnGround = true;
210
211 if (UpdateVictim())
212 {
213 AttackStart(me->GetVictim());
214 me->GetMotionMaster()->MoveChase(me->GetVictim());
215 }
216 else
217 {
218 if (Unit* target = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_PL_JEDOGA_TARGET)))
219 {
220 AttackStart(target);
221 instance->SetData(DATA_JEDOGA_RESET_INITIANDS, 0);
222 if (instance->GetBossState(DATA_JEDOGA_SHADOWSEEKER) != IN_PROGRESS)
223 EnterCombat(target);
224 }
225 else if (!me->IsInCombat())
226 EnterEvadeMode();
227 }
228 }
229
230 void MoveUp()
231 {
232 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, true);
233 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_MAGIC, true);
234 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE + UNIT_FLAG_NON_ATTACKABLE);
235
236 me->AttackStop();
237 me->RemoveAllAuras();
238 me->LoadCreaturesAddon();
239 me->GetMotionMaster()->MovePoint(0, JedogaPosition[0]);
240
241 instance->SetData(DATA_JEDOGA_TRIGGER_SWITCH, 1);
242 if (instance->GetBossState(DATA_JEDOGA_SHADOWSEEKER) == IN_PROGRESS)
243 OpferRufen();
244
245 bOnGround = false;
246 uiOpFerTimer = urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS);
247 }
248
249 void OpferRufen()
250 {
251 ObjectGuid opfer = instance->GetGuidData(DATA_ADD_JEDOGA_INITIAND);
252
253 if (opfer)
254 {
255 Talk(TEXT_SACRIFICE_1);
256 instance->SetGuidData(DATA_ADD_JEDOGA_OPFER, opfer);
257 }
258 else
259 bCanDown = true;
260 }
261
262 void Opfern()
263 {
264 Talk(TEXT_SACRIFICE_2);
265
266 me->InterruptNonMeleeSpells(false);
267 DoCast(me, SPELL_GIFT_OF_THE_HERALD, false);
268
269 bOpFerok = false;
270 bCanDown = true;
271 }
272
273 void UpdateAI(uint32 diff) override
274 {
275 if (instance->GetBossState(DATA_JEDOGA_SHADOWSEEKER) != IN_PROGRESS && instance->GetData(DATA_ALL_INITIAND_DEAD))
276 MoveDown();
277
278 if (bOpFerok && !bOnGround && !bCanDown) Opfern();
279
280 if (bOpFerokFail && !bOnGround && !bCanDown)
281 bCanDown = true;
282
283 if (bCanDown)
284 {
285 MoveDown();
286 bCanDown = false;
287 }
288
289 if (bOnGround)
290 {
291 if (!UpdateVictim())
292 return;
293
294 if (uiCycloneTimer <= diff)
295 {
296 DoCast(me, SPELL_CYCLONE_STRIKE, false);
297 uiCycloneTimer = urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS);
298 } else uiCycloneTimer -= diff;
299
300 if (uiBoltTimer <= diff)
301 {
302 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
303 me->CastSpell(target, SPELL_LIGHTNING_BOLT, false);
304
305 uiBoltTimer = urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS);
306 } else uiBoltTimer -= diff;
307
308 if (uiThunderTimer <= diff)
309 {
310 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
311 me->CastSpell(target, SPELL_THUNDERSHOCK, false);
312
313 uiThunderTimer = urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS);
314 } else uiThunderTimer -= diff;
315
316 if (uiOpFerTimer <= diff)
317 MoveUp();
318 else
319 uiOpFerTimer -= diff;
320
321 DoMeleeAttackIfReady();
322 }
323 }
324 };
325
326 CreatureAI* GetAI(Creature* creature) const override
327 {
328 return GetInstanceAI<boss_jedoga_shadowseekerAI>(creature);
329 }

357 void Reset() override
358 {
359 Initialize();
360

362 {
363 me->RemoveAurasDueToSpell(SPELL_SPHERE_VISUAL);

372 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_MAGIC, true);

374 }

377 void JustDied(Unit* killer) override
378 {
379 if (!killer || !instance)
380 return;
381
382 if (bWalking)
383 {
384 if (Creature* boss = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_JEDOGA_SHADOWSEEKER)))
385 {
386 if (!ENSURE_AI(boss_jedoga_shadowseeker::boss_jedoga_shadowseekerAI, boss->AI())->bOpFerok)
387 ENSURE_AI(boss_jedoga_shadowseeker::boss_jedoga_shadowseekerAI, boss->AI())->bOpFerokFail = true;
388
389 if (killer->GetTypeId() == TYPEID_PLAYER)
390 boss->AI()->DoAction(ACTION_INITIAND_KILLED);
391 }
392
393 instance->SetGuidData(DATA_ADD_JEDOGA_OPFER, ObjectGuid::Empty);
394
395 bWalking = false;
396 }
397 if (killer->GetTypeId() == TYPEID_PLAYER)
398 instance->SetGuidData(DATA_PL_JEDOGA_TARGET, killer->GetGUID());
399 }

422 void MovementInform(uint32 uiType, uint32 uiPointId) override
423 {
424 if (uiType != POINT_MOTION_TYPE || !instance)
425 return;
426
427 switch (uiPointId)
428 {
429 case 1:
430 {
431 Creature* boss = me->GetMap()->GetCreature(instance->GetGuidData(DATA_JEDOGA_SHADOWSEEKER));
432 if (boss)
433 {
434 ENSURE_AI(boss_jedoga_shadowseeker::boss_jedoga_shadowseekerAI, boss->AI())->bOpFerok = true;
435 ENSURE_AI(boss_jedoga_shadowseeker::boss_jedoga_shadowseekerAI, boss->AI())->bOpFerokFail = false;
436 me->KillSelf();
437 }
438 }
439 break;
440 }
441 }
442
443 void UpdateAI(uint32 diff) override
444 {
445 if (bCheckTimer <= diff)
446 {
447 if (me->GetGUID() == instance->GetGuidData(DATA_ADD_JEDOGA_OPFER) && !bWalking)
448 {
449 me->RemoveAurasDueToSpell(SPELL_SPHERE_VISUAL);
450 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, false);
451 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_MAGIC, false);
452 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE + UNIT_FLAG_NON_ATTACKABLE);
453
454 float distance = me->GetDistance(JedogaPosition[1]);
455
456 if (distance < 9.0f)
457 me->SetSpeed(MOVE_WALK, 0.5f, true);
458 else if (distance < 15.0f)
459 me->SetSpeed(MOVE_WALK, 0.75f, true);
460 else if (distance < 20.0f)
461 me->SetSpeed(MOVE_WALK, 1.0f, true);
462
463 me->GetMotionMaster()->Clear(false);
464 me->GetMotionMaster()->MovePoint(1, JedogaPosition[1]);
465 bWalking = true;
466 }
467 if (!bWalking)
468 {
469 if (instance->GetBossState(DATA_JEDOGA_SHADOWSEEKER) != IN_PROGRESS && me->HasAura(SPELL_SPHERE_VISUAL))
470 {
471 me->RemoveAurasDueToSpell(SPELL_SPHERE_VISUAL);
472 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, false);
473 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_MAGIC, false);
474 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE + UNIT_FLAG_NON_ATTACKABLE);
475 }
476 if (instance->GetBossState(DATA_JEDOGA_SHADOWSEEKER) == IN_PROGRESS && !me->HasAura(SPELL_SPHERE_VISUAL))
477 {
478 DoCast(me, SPELL_SPHERE_VISUAL, false);
479 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_NORMAL, true);
480 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_MAGIC, true);
481 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE + UNIT_FLAG_NON_ATTACKABLE);
482 }
483 }
484 bCheckTimer = 2*IN_MILLISECONDS;
485 } else bCheckTimer -= diff;
486
487 //Return since we have no target
488 if (!UpdateVictim())
489 return;
490
491 DoMeleeAttackIfReady();
492 }
493 };

560 {
561 DoCast(me, SPELL_BEAM_VISUAL_JEDOGAS_AUFSEHER_2, false);
562 bCast2 = true;

565 {
566 me->InterruptNonMeleeSpells(true);
567 bCast2 = false;
568 }
569 if (!bRemoved2 && instance->GetBossState(DATA_JEDOGA_SHADOWSEEKER) == DONE)
570 {
571 me->InterruptNonMeleeSpells(true);
572 bRemoved2 = true;
573 }
574 }
575 }

57class boss_volazj : public CreatureScript

60 boss_volazj() : CreatureScript("boss_volazj") { }

62 struct boss_volazjAI : public ScriptedAI
63 {
64 boss_volazjAI(Creature* creature) : ScriptedAI(creature), Summons(me)
65 {
66 Initialize();
67 instance = creature->GetInstanceScript();
68 }
69
70 void Initialize()
71 {
72 uiMindFlayTimer = 8 * IN_MILLISECONDS;
73 uiShadowBoltVolleyTimer = 5 * IN_MILLISECONDS;
74 uiShiverTimer = 15 * IN_MILLISECONDS;
75 // Used for Insanity handling
76 insanityHandled = 0;
77 }
78
79 InstanceScript* instance;
80
81 uint32 uiMindFlayTimer;
82 uint32 uiShadowBoltVolleyTimer;
83 uint32 uiShiverTimer;
84 uint32 insanityHandled;
85 SummonList Summons;
86
87 // returns the percentage of health after taking the given damage.
88 uint32 GetHealthPct(uint32 damage)
89 {
90 if (damage > me->GetHealth())
91 return 0;
92 return 100*(me->GetHealth()-damage)/me->GetMaxHealth();
93 }
94
95 void DamageTaken(Unit* /*pAttacker*/, uint32 &damage) override
96 {
97 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE))
98 damage = 0;
99
100 if ((GetHealthPct(0) >= 66 && GetHealthPct(damage) < 66)||
101 (GetHealthPct(0) >= 33 && GetHealthPct(damage) < 33))
102 {
103 me->InterruptNonMeleeSpells(false);
104 DoCast(me, SPELL_INSANITY, false);
105 }
106 }
107
108 void SpellHitTarget(Unit* target, const SpellInfo* spell) override
109 {
110 if (spell->Id == SPELL_INSANITY)
111 {
112 // Not good target or too many players
113 if (target->GetTypeId() != TYPEID_PLAYER || insanityHandled > 4)
114 return;
115 // First target - start channel visual and set self as unnattackable
116 if (!insanityHandled)
117 {
118 // Channel visual
119 DoCast(me, INSANITY_VISUAL, true);
120 // Unattackable
121 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
122 me->SetControlled(true, UNIT_STATE_STUNNED);
123 }
124 // phase mask
125 target->CastSpell(target, SPELL_INSANITY_TARGET+insanityHandled, true);
126 // summon twisted party members for this target
127 Map::PlayerList const &players = me->GetMap()->GetPlayers();
128 for (Map::PlayerList::const_iterator i = players.begin(); i != players.end(); ++i)
129 {
130 Player* player = i->GetSource();
131 if (!player || !player->IsAlive())
132 continue;
133 // Summon clone
134 if (Unit* summon = me->SummonCreature(NPC_TWISTED_VISAGE, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation(), TEMPSUMMON_CORPSE_DESPAWN, 0))
135 {
136 // clone
137 player->CastSpell(summon, SPELL_CLONE_PLAYER, true);
138 // set phase
139 summon->SetPhaseMask((1<<(4+insanityHandled)), true);
140 }
141 }
142 ++insanityHandled;
143 }
144 }
145
146 void ResetPlayersPhaseMask()
147 {
148 Map::PlayerList const &players = me->GetMap()->GetPlayers();
149 for (Map::PlayerList::const_iterator i = players.begin(); i != players.end(); ++i)
150 {
151 Player* player = i->GetSource();
152 player->RemoveAurasDueToSpell(GetSpellForPhaseMask(player->GetPhaseMask()));
153 }
154 }
155
156 void Reset() override
157 {
158 Initialize();
159
160 instance->SetBossState(DATA_HERALD_VOLAZJ, NOT_STARTED);
161 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_QUICK_DEMISE_START_EVENT);
162
163 // Visible for all players in insanity
164 me->SetPhaseMask((1|16|32|64|128|256), true);
165
166 ResetPlayersPhaseMask();
167
168 // Cleanup
169 Summons.DespawnAll();
170 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
171 me->SetControlled(false, UNIT_STATE_STUNNED);
172 }

176 Talk(SAY_AGGRO);

180 }

184 Summons.Summon(summon);

187 uint32 GetSpellForPhaseMask(uint32 phase)
188 {
189 uint32 spell = 0;
190 switch (phase)
191 {
192 case 16:
193 spell = SPELL_INSANITY_PHASING_1;
194 break;
195 case 32:
196 spell = SPELL_INSANITY_PHASING_2;
197 break;
198 case 64:
199 spell = SPELL_INSANITY_PHASING_3;
200 break;
201 case 128:
202 spell = SPELL_INSANITY_PHASING_4;
203 break;
204 case 256:
205 spell = SPELL_INSANITY_PHASING_5;
206 break;
207 }
208 return spell;
209 }
210
211 void SummonedCreatureDespawn(Creature* summon) override
212 {
213 uint32 phase = summon->GetPhaseMask();
214 uint32 nextPhase = 0;
215 Summons.Despawn(summon);
216
217 // Check if all summons in this phase killed
218 for (SummonList::const_iterator iter = Summons.begin(); iter != Summons.end(); ++iter)
219 {
220 if (Creature* visage = ObjectAccessor::GetCreature(*me, *iter))
221 {
222 // Not all are dead
223 if (phase == visage->GetPhaseMask())
224 return;
225 else
226 nextPhase = visage->GetPhaseMask();
227 }
228 }
229
230 // Roll Insanity
231 uint32 spell = GetSpellForPhaseMask(phase);
232 uint32 spell2 = GetSpellForPhaseMask(nextPhase);
233 Map::PlayerList const &PlayerList = me->GetMap()->GetPlayers();
234 if (!PlayerList.isEmpty())
235 {
236 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
237 {
238 if (Player* player = i->GetSource())
239 {
240 if (player->HasAura(spell))
241 {
242 player->RemoveAurasDueToSpell(spell);
243 if (spell2) // if there is still some different mask cast spell for it
244 player->CastSpell(player, spell2, true);
245 }
246 }
247 }
248 }
249 }
250
251 void UpdateAI(uint32 diff) override
252 {
253 //Return since we have no target
254 if (!UpdateVictim())
255 return;
256
257 if (insanityHandled)
258 {
259 if (!Summons.empty())
260 return;
261
262 insanityHandled = 0;
263 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
264 me->SetControlled(false, UNIT_STATE_STUNNED);
265 me->RemoveAurasDueToSpell(INSANITY_VISUAL);
266 }
267
268 if (uiMindFlayTimer <= diff)
269 {
270 DoCastVictim(SPELL_MIND_FLAY);
271 uiMindFlayTimer = 20*IN_MILLISECONDS;
272 } else uiMindFlayTimer -= diff;
273
274 if (uiShadowBoltVolleyTimer <= diff)
275 {
276 DoCastVictim(SPELL_SHADOW_BOLT_VOLLEY);
277 uiShadowBoltVolleyTimer = 5*IN_MILLISECONDS;
278 } else uiShadowBoltVolleyTimer -= diff;
279
280 if (uiShiverTimer <= diff)
281 {
282 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
283 DoCast(target, SPELL_SHIVER);
284 uiShiverTimer = 15*IN_MILLISECONDS;
285 } else uiShiverTimer -= diff;
286
287 DoMeleeAttackIfReady();
288 }
289

291 {
292 Talk(SAY_DEATH);
293

297 ResetPlayersPhaseMask();
298 }

301 {

303 Talk(SAY_SLAY);
304 }

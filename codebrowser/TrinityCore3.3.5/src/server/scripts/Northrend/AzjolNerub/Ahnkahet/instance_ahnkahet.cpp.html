23DoorData const doorData[] =
24{
25 { GO_PRINCE_TALDARAM_GATE, DATA_PRINCE_TALDARAM, DOOR_TYPE_PASSAGE },
26 { 0, 0, DOOR_TYPE_ROOM } // END

29class instance_ahnkahet : public InstanceMapScript
30{
31 public:
32 instance_ahnkahet() : InstanceMapScript(AhnKahetScriptName, 619) { }
33
34 struct instance_ahnkahet_InstanceScript : public InstanceScript
35 {
36 instance_ahnkahet_InstanceScript(Map* map) : InstanceScript(map)
37 {
38 SetHeaders(DataHeader);
39 SetBossNumber(EncounterCount);
40 LoadDoorData(doorData);
41
42 SwitchTrigger = 0;
43
44 SpheresState[0] = 0;
45 SpheresState[1] = 0;
46 }
47
48 void OnCreatureCreate(Creature* creature) override
49 {
50 switch (creature->GetEntry())
51 {
52 case NPC_ELDER_NADOX:
53 ElderNadoxGUID = creature->GetGUID();
54 break;
55 case NPC_PRINCE_TALDARAM:
56 PrinceTaldaramGUID = creature->GetGUID();
57 break;
58 case NPC_JEDOGA_SHADOWSEEKER:
59 JedogaShadowseekerGUID = creature->GetGUID();
60 break;
61 case NPC_AMANITAR:
62 AmanitarGUID = creature->GetGUID();
63 break;
64 case NPC_HERALD_VOLAZJ:
65 HeraldVolazjGUID = creature->GetGUID();
66 break;
67 case NPC_INITIAND:
68 InitiandGUIDs.insert(creature->GetGUID());
69 break;
70 default:
71 break;
72 }
73 }
74
75 void OnGameObjectCreate(GameObject* go) override
76 {
77 switch (go->GetEntry())
78 {
79 case GO_PRINCE_TALDARAM_PLATFORM:
80 PrinceTaldaramPlatformGUID = go->GetGUID();
81 if (GetBossState(DATA_PRINCE_TALDARAM) == DONE)
82 HandleGameObject(ObjectGuid::Empty, true, go);
83 break;
84 case GO_SPHERE_1:
85 if (SpheresState[0])
86 {
87 go->SetGoState(GO_STATE_ACTIVE);
88 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
89 }
90 else
91 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
92 break;
93 case GO_SPHERE_2:
94 if (SpheresState[1])
95 {
96 go->SetGoState(GO_STATE_ACTIVE);
97 go->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
98 }
99 else
100 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
101 break;
102 case GO_PRINCE_TALDARAM_GATE:
103 AddDoor(go, true);
104 break;
105 default:
106 break;
107 }
108 }
109
110 void OnGameObjectRemove(GameObject* go) override
111 {
112 switch (go->GetEntry())
113 {
114 case GO_PRINCE_TALDARAM_GATE:
115 AddDoor(go, false);
116 break;
117 default:
118 break;
119 }
120 }
121
122 void SetData(uint32 type, uint32 data) override
123 {
124 switch (type)
125 {
126 case DATA_SPHERE_1:
127 case DATA_SPHERE_2:
128 SpheresState[type - DATA_SPHERE_1] = data;
129 break;
130 case DATA_JEDOGA_TRIGGER_SWITCH:
131 SwitchTrigger = data;
132 break;
133 case DATA_JEDOGA_RESET_INITIANDS:
134 for (ObjectGuid guid : InitiandGUIDs)
135 {
136 if (Creature* creature = instance->GetCreature(guid))
137 {
138 creature->Respawn();
139 if (!creature->IsInEvadeMode())
140 creature->AI()->EnterEvadeMode();
141 }
142 }
143 break;
144 default:
145 break;
146 }
147 }
148
149 uint32 GetData(uint32 type) const override
150 {
151 switch (type)
152 {
153 case DATA_SPHERE_1:
154 case DATA_SPHERE_2:
155 return SpheresState[type - DATA_SPHERE_1];
156 case DATA_ALL_INITIAND_DEAD:
157 for (ObjectGuid guid : InitiandGUIDs)
158 {
159 Creature* cr = instance->GetCreature(guid);
160 if (!cr || cr->IsAlive())
161 return 0;
162 }
163 return 1;
164 case DATA_JEDOGA_TRIGGER_SWITCH:
165 return SwitchTrigger;
166 default:
167 break;
168 }
169 return 0;
170 }
171
172 void SetGuidData(uint32 type, ObjectGuid data) override
173 {
174 switch (type)
175 {
176 case DATA_ADD_JEDOGA_OPFER:
177 JedogaSacrifices = data;
178 break;
179 case DATA_PL_JEDOGA_TARGET:
180 JedogaTarget = data;
181 break;
182 default:
183 break;
184 }
185 }
186
187 ObjectGuid GetGuidData(uint32 type) const override
188 {
189 switch (type)
190 {
191 case DATA_ELDER_NADOX:
192 return ElderNadoxGUID;
193 case DATA_PRINCE_TALDARAM:
194 return PrinceTaldaramGUID;
195 case DATA_JEDOGA_SHADOWSEEKER:
196 return JedogaShadowseekerGUID;
197 case DATA_AMANITAR:
198 return AmanitarGUID;
199 case DATA_HERALD_VOLAZJ:
200 return HeraldVolazjGUID;
201 case DATA_PRINCE_TALDARAM_PLATFORM:
202 return PrinceTaldaramPlatformGUID;
203 case DATA_ADD_JEDOGA_INITIAND:
204 {
205 GuidVector vInitiands;
206 vInitiands.reserve(InitiandGUIDs.size());
207 for (ObjectGuid guid : InitiandGUIDs)
208 {
209 Creature* cr = instance->GetCreature(guid);
210 if (cr && cr->IsAlive())
211 vInitiands.push_back(guid);
212 }
213 if (vInitiands.empty())
214 return ObjectGuid::Empty;
215
216 return Trinity::Containers::SelectRandomContainerElement(vInitiands);
217 }
218 case DATA_ADD_JEDOGA_OPFER:
219 return JedogaSacrifices;
220 case DATA_PL_JEDOGA_TARGET:
221 return JedogaTarget;
222 default:
223 break;
224 }
225 return ObjectGuid::Empty;
226 }
227
228 bool SetBossState(uint32 type, EncounterState state) override
229 {
230 if (!InstanceScript::SetBossState(type, state))
231 return false;
232
233 switch (type)
234 {
235 case DATA_JEDOGA_SHADOWSEEKER:
236 if (state == DONE)
237 {
238 for (ObjectGuid guid : InitiandGUIDs)
239 if (Creature* cr = instance->GetCreature(guid))
240 cr->DespawnOrUnsummon();
241 }
242 break;
243 default:
244 break;
245 }
246 return true;
247 }
248
249 void WriteSaveDataMore(std::ostringstream& data) override
250 {
251 data << SpheresState[0] << ' ' << SpheresState[1];
252 }
253
254 void ReadSaveDataMore(std::istringstream& data) override
255 {
256 data >> SpheresState[0];
257 data >> SpheresState[1];
258 }

260 protected:
261 ObjectGuid ElderNadoxGUID;
262 ObjectGuid PrinceTaldaramGUID;
263 ObjectGuid JedogaShadowseekerGUID;
264 ObjectGuid AmanitarGUID;
265 ObjectGuid HeraldVolazjGUID;
266
267 ObjectGuid PrinceTaldaramPlatformGUID;
268 ObjectGuid JedogaSacrifices;
269 ObjectGuid JedogaTarget;
270
271 GuidSet InitiandGUIDs;
272
273 uint32 SpheresState[2];
274 uint8 SwitchTrigger;
275 };
276
277 InstanceScript* GetInstanceScript(InstanceMap* map) const override
278 {
279 return new instance_ahnkahet_InstanceScript(map);
280 }

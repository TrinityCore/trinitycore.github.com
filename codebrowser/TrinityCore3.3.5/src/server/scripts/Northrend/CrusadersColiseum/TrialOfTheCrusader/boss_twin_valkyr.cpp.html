170 ShieldSpellId = 0;

176 void Reset() override
177 {
178 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
179 me->SetReactState(REACT_PASSIVE);
180 me->ModifyAuraState(AuraState, true);

183 me->SetFlying(true); */

185 summons.DespawnAll();
186 }

191
192 summons.DespawnAll();

197 {
198 if (uiType != POINT_MOTION_TYPE)

203 case 1:
204 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
205 me->SetReactState(REACT_AGGRESSIVE);
206 break;
207 default:
208 break;
209 }

213 {
214 if (who->GetTypeId() == TYPEID_PLAYER)
215 Talk(SAY_KILL_PLAYER);
216 }

218 void SummonedCreatureDespawn(Creature* summoned) override
219 {
220 switch (summoned->GetEntry())
221 {
222 case NPC_LIGHT_ESSENCE:
223 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_LIGHT_ESSENCE_HELPER);
224 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_POWERING_UP_HELPER);
225 break;
226 case NPC_DARK_ESSENCE:
227 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_DARK_ESSENCE_HELPER);
228 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_POWERING_UP_HELPER);
229 break;
230 case NPC_BULLET_CONTROLLER:
231 me->m_Events.AddEvent(new OrbsDespawner(me), me->m_Events.CalculateTime(100));
232 break;
233 default:
234 break;
235 }
236 summons.Despawn(summoned);
237 }
238
239 void JustDied(Unit* /*killer*/) override
240 {
241 Talk(SAY_DEATH);
242 if (Creature* pSister = GetSister())
243 {
244 if (!pSister->IsAlive())
245 {
246 me->SetFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
247 pSister->SetFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
248 _JustDied();
249 }
250 else
251 {
252 me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
253 instance->SetBossState(BOSS_VALKIRIES, SPECIAL);
254 }
255 }
256 summons.DespawnAll();
257 }

262 return ObjectAccessor::GetCreature((*me), instance->GetGuidData(SisterNpcId));
263 }

266 {
267 me->SetInCombatWithZone();

269 {
270 me->AddAura(MyEmphatySpellId, pSister);
271 pSister->SetInCombatWithZone();
272 }
273 instance->SetBossState(BOSS_VALKIRIES, IN_PROGRESS);
274
275 Talk(SAY_AGGRO);
276 DoCast(me, SurgeSpellId);
277
278 events.ScheduleEvent(EVENT_TWIN_SPIKE, 20 * IN_MILLISECONDS);
279 events.ScheduleEvent(EVENT_BERSERK, IsHeroic() ? 6 * MINUTE*IN_MILLISECONDS : 10 * MINUTE*IN_MILLISECONDS);
280 if (IsHeroic())
281 events.ScheduleEvent(EVENT_TOUCH, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
282 }
283

285 {
286 switch (action)
287 {
288 case ACTION_VORTEX:
289 Talk(EMOTE_VORTEX);
290 DoCastAOE(VortexSpellId);
291 break;
292 case ACTION_PACT:
293 Talk(EMOTE_TWIN_PACT);

296 sister->CastSpell(sister, SPELL_POWER_TWINS, false);
297 DoCast(me, ShieldSpellId);
298 DoCast(me, TwinPactSpellId);
299 break;
300 default:
301 break;
302 }
303 }

308 me->SetCanDualWield(mode);
309 }

311 void ExecuteEvent(uint32 eventId) override
312 {
313 switch (eventId)
314 {
315 case EVENT_TWIN_SPIKE:
316 DoCastVictim(SpikeSpellId);
317 events.ScheduleEvent(EVENT_TWIN_SPIKE, 20 * IN_MILLISECONDS);

319 case EVENT_TOUCH:

322 events.ScheduleEvent(EVENT_TOUCH, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
323 break;
324 case EVENT_BERSERK:
325 DoCast(me, SPELL_BERSERK);

327 break;
328 default:
329 break;
330 }
331 }

358
359 void Reset() override
360 {
361 SetEquipmentSlots(false, EQUIP_MAIN_1, EQUIP_UNEQUIP, EQUIP_NO_CHANGE);
362 Weapon = EQUIP_MAIN_1;
363 AuraState = AURA_STATE_UNKNOWN22;
364 SisterNpcId = NPC_DARKBANE;
365 MyEmphatySpellId = SPELL_TWIN_EMPATHY_DARK;
366 OtherEssenceSpellId = SPELL_DARK_ESSENCE_HELPER;
367 SurgeSpellId = SPELL_LIGHT_SURGE;
368 VortexSpellId = SPELL_LIGHT_VORTEX;
369 ShieldSpellId = SPELL_LIGHT_SHIELD;
370 TwinPactSpellId = SPELL_LIGHT_TWIN_PACT;
371 TouchSpellId = SPELL_LIGHT_TOUCH;
372 SpikeSpellId = SPELL_LIGHT_TWIN_SPIKE;
373
374 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, EVENT_START_TWINS_FIGHT);
375 boss_twin_baseAI::Reset();
376 }
377
378 void ExecuteEvent(uint32 eventId) override
379 {
380 if (eventId == EVENT_SPECIAL_ABILITY)
381 {
382 if (CurrentStage == MAX_STAGES)
383 GenerateStageSequence();
384
385 switch (Stage[CurrentStage])
386 {
387 case STAGE_DARK_VORTEX:
388 if (Creature* sister = GetSister())
389 sister->AI()->DoAction(ACTION_VORTEX);
390 break;
391 case STAGE_DARK_PACT:
392 if (Creature* sister = GetSister())
393 sister->AI()->DoAction(ACTION_PACT);
394 break;
395 case STAGE_LIGHT_VORTEX:
396 DoAction(ACTION_VORTEX);
397 break;
398 case STAGE_LIGHT_PACT:
399 DoAction(ACTION_PACT);
400 break;
401 default:
402 break;
403 }
404 ++CurrentStage;
405 events.ScheduleEvent(EVENT_SPECIAL_ABILITY, 45 * IN_MILLISECONDS);
406 }
407 else
408 boss_twin_baseAI::ExecuteEvent(eventId);
409 }
410
411 void EnterCombat(Unit* who) override
412 {
413 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, EVENT_START_TWINS_FIGHT);
414 events.ScheduleEvent(EVENT_SPECIAL_ABILITY, 45 * IN_MILLISECONDS);
415 me->SummonCreature(NPC_BULLET_CONTROLLER, ToCCommonLoc[1].GetPositionX(), ToCCommonLoc[1].GetPositionY(), ToCCommonLoc[1].GetPositionZ(), 0.0f, TEMPSUMMON_MANUAL_DESPAWN);
416 boss_twin_baseAI::EnterCombat(who);
417 }
418
419 void EnterEvadeMode(EvadeReason why) override
420 {
421 instance->DoUseDoorOrButton(instance->GetGuidData(GO_MAIN_GATE_DOOR));
422 boss_twin_baseAI::EnterEvadeMode(why);
423 }
424
425 void JustReachedHome() override
426 {
427 instance->DoUseDoorOrButton(instance->GetGuidData(GO_MAIN_GATE_DOOR));
428
429 boss_twin_baseAI::JustReachedHome();
430 }
431
432 void GenerateStageSequence()
433 {
434 CurrentStage = 0;
435
436 // Initialize and clean up.
437 for (int i = 0; i < MAX_STAGES; ++i)
438 Stage[i] = i;
439
440 // Allocate an unique random stage to each position in the array.
441 for (int i = 0; i < MAX_STAGES - 1; ++i)
442 {
443 int random = i + (rand32() % (MAX_STAGES - i));
444 int temp = Stage[i];
445 Stage[i] = Stage[random];
446 Stage[random] = temp;
447 }
448 }
449 private:
450 uint8 Stage[4];
451 uint8 CurrentStage;
452 };
453
454 CreatureAI* GetAI(Creature* creature) const override
455 {
456 return GetInstanceAI<boss_fjolaAI>(creature);
457 }

535struct npc_unleashed_ballAI : public ScriptedAI
536{
537 npc_unleashed_ballAI(Creature* creature) : ScriptedAI(creature)
538 {
539 Initialize();
540 }
541
542 void Initialize()
543 {
544 RangeCheckTimer = 0.5*IN_MILLISECONDS;
545 }
546
547 void MoveToNextPoint()
548 {
549 float x0 = ToCCommonLoc[1].GetPositionX(), y0 = ToCCommonLoc[1].GetPositionY(), r = 47.0f;
550 float y = y0;
551 float x = frand(x0 - r, x0 + r);
552 float sq = std::pow(r, 2.f) - std::pow(x - x0, 2.f);
553 float rt = std::sqrt(std::fabs(sq));
554 if (urand(0, 1))
555 y = y0 + rt;
556 else
557 y = y0 - rt;
558 me->GetMotionMaster()->MovePoint(0, x, y, me->GetPositionZ());
559 }
560
561 void Reset() override
562 {
563 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
564 me->SetReactState(REACT_PASSIVE);
565 me->SetDisableGravity(true);
566 me->SetCanFly(true);
567 SetCombatMovement(false);
568 MoveToNextPoint();
569 Initialize();
570 }
571
572 void MovementInform(uint32 uiType, uint32 uiId) override
573 {
574 if (uiType != POINT_MOTION_TYPE)
575 return;
576
577 switch (uiId)
578 {
579 case 0:
580 if (urand(0, 3) == 0)
581 MoveToNextPoint();
582 else
583 me->DisappearAndDie();
584 break;
585 default:
586 break;
587 }
588 }
589
590 protected:
591 uint32 RangeCheckTimer;
592};

687class spell_powering_up : public SpellScriptLoader
688{
689 public:
690 spell_powering_up() : SpellScriptLoader("spell_powering_up") { }
691
692 class spell_powering_up_SpellScript : public SpellScript
693 {
694 PrepareSpellScript(spell_powering_up_SpellScript);
695
696 public:
697 spell_powering_up_SpellScript()
698 {
699 spellId = 0;
700 poweringUp = 0;
701 }
702
703 private:
704 uint32 spellId;
705 uint32 poweringUp;
706
707 bool Load() override
708 {
709 spellId = sSpellMgr->GetSpellIdForDifficulty(SPELL_SURGE_OF_SPEED, GetCaster());
710 if (!sSpellMgr->GetSpellInfo(spellId))
711 return false;
712
713 poweringUp = sSpellMgr->GetSpellIdForDifficulty(SPELL_POWERING_UP, GetCaster());
714 if (!sSpellMgr->GetSpellInfo(poweringUp))
715 return false;
716
717 return true;
718 }
719
720 void HandleScriptEffect(SpellEffIndex /*effIndex*/)
721 {
722 if (Unit* target = GetHitUnit())
723 {
724 if (Aura* pAura = target->GetAura(poweringUp))
725 {
726 if (pAura->GetStackAmount() >= 100)
727 {
728 if (target->GetDummyAuraEffect(SPELLFAMILY_GENERIC, 2206, EFFECT_1))
729 target->CastSpell(target, SPELL_EMPOWERED_DARK, true);
730
731 if (target->GetDummyAuraEffect(SPELLFAMILY_GENERIC, 2845, EFFECT_1))
732 target->CastSpell(target, SPELL_EMPOWERED_LIGHT, true);
733
734 target->RemoveAurasDueToSpell(poweringUp);

770 bool Load() override
771 {
772 spellId = sSpellMgr->GetSpellIdForDifficulty(SPELL_SURGE_OF_SPEED, GetCaster());
773 if (!sSpellMgr->GetSpellInfo(spellId))
774 return false;
775 return true;
776 }

778 void Absorb(AuraEffect* /*aurEff*/, DamageInfo & dmgInfo, uint32 & /*absorbAmount*/)
779 {
780 if (Unit* owner = GetUnitOwner())
781 {
782 if (dmgInfo.GetSpellInfo())
783 {
784 if (uint32 poweringUp = sSpellMgr->GetSpellIdForDifficulty(SPELL_POWERING_UP, owner))
785 {
786 if (urand(0, 99) < 5)
787 GetTarget()->CastSpell(GetTarget(), spellId, true);
788
789 // Twin Vortex part
790 uint32 lightVortex = sSpellMgr->GetSpellIdForDifficulty(SPELL_LIGHT_VORTEX_DAMAGE, owner);
791 uint32 darkVortex = sSpellMgr->GetSpellIdForDifficulty(SPELL_DARK_VORTEX_DAMAGE, owner);
792 int32 stacksCount = dmgInfo.GetSpellInfo()->Effects[EFFECT_0].CalcValue() / 1000 - 1;
793
794 if (lightVortex && darkVortex && stacksCount)
795 {
796 if (dmgInfo.GetSpellInfo()->Id == darkVortex || dmgInfo.GetSpellInfo()->Id == lightVortex)
797 {
798 Aura* pAura = owner->GetAura(poweringUp);
799 if (pAura)
800 {
801 pAura->ModStackAmount(stacksCount);
802 owner->CastSpell(owner, poweringUp, true);
803 }
804 else
805 {
806 owner->CastSpell(owner, poweringUp, true);
807 if (Aura* pTemp = owner->GetAura(poweringUp))
808 pTemp->ModStackAmount(stacksCount);
809 }
810 }
811 }
812
813 // Picking floating balls
814 uint32 unleashedDark = sSpellMgr->GetSpellIdForDifficulty(SPELL_UNLEASHED_DARK, owner);
815 uint32 unleashedLight = sSpellMgr->GetSpellIdForDifficulty(SPELL_UNLEASHED_LIGHT, owner);
816
817 if (unleashedDark && unleashedLight)
818 {
819 if (dmgInfo.GetSpellInfo()->Id == unleashedDark || dmgInfo.GetSpellInfo()->Id == unleashedLight)
820 {
821 // need to do the things in this order, else players might have 100 charges of Powering Up without anything happening
822 Aura* pAura = owner->GetAura(poweringUp);
823 if (pAura)
824 {
825 // 2 lines together add the correct amount of buff stacks
826 pAura->ModStackAmount(stacksCount);
827 owner->CastSpell(owner, poweringUp, true);
828 }
829 else
830 {
831 owner->CastSpell(owner, poweringUp, true);
832 if (Aura* pTemp = owner->GetAura(poweringUp))
833 pTemp->ModStackAmount(stacksCount);
834 }
835 }
836 }
837 }
838 }
839 }
840 }

844 OnEffectAbsorb += AuraEffectAbsorbFn(spell_valkyr_essences_AuraScript::Absorb, EFFECT_0);

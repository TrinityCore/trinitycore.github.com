93class boss_jaraxxus : public CreatureScript
94{
95 public:
96 boss_jaraxxus() : CreatureScript("boss_jaraxxus") { }
97
98 struct boss_jaraxxusAI : public BossAI
99 {
100 boss_jaraxxusAI(Creature* creature) : BossAI(creature, BOSS_JARAXXUS) { }
101
102 void Reset() override
103 {
104 _Reset();
105 events.ScheduleEvent(EVENT_FEL_FIREBALL, 5*IN_MILLISECONDS);
106 events.ScheduleEvent(EVENT_FEL_LIGHTNING, urand(10*IN_MILLISECONDS, 15*IN_MILLISECONDS));
107 events.ScheduleEvent(EVENT_INCINERATE_FLESH, urand(20*IN_MILLISECONDS, 25*IN_MILLISECONDS));
108 events.ScheduleEvent(EVENT_NETHER_POWER, 40*IN_MILLISECONDS);
109 events.ScheduleEvent(EVENT_LEGION_FLAME, 30*IN_MILLISECONDS);
110 events.ScheduleEvent(EVENT_SUMMONO_NETHER_PORTAL, 20*IN_MILLISECONDS);
111 events.ScheduleEvent(EVENT_SUMMON_INFERNAL_ERUPTION, 80*IN_MILLISECONDS);
112 }
113
114 void JustReachedHome() override
115 {
116 _JustReachedHome();
117 instance->SetBossState(BOSS_JARAXXUS, FAIL);
118 DoCast(me, SPELL_JARAXXUS_CHAINS);
119 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
120 }
121
122 void KilledUnit(Unit* who) override
123 {
124 if (who->GetTypeId() == TYPEID_PLAYER)
125 Talk(SAY_KILL_PLAYER);
126 }
127
128 void JustDied(Unit* /*killer*/) override
129 {
130 _JustDied();
131 Talk(SAY_DEATH);
132 }
133
134 void EnterCombat(Unit* /*who*/) override
135 {
136 _EnterCombat();
137 Talk(SAY_AGGRO);
138 }
139
140 void UpdateAI(uint32 diff) override
141 {
142 if (!UpdateVictim())
143 return;
144
145 events.Update(diff);
146
147 if (me->HasUnitState(UNIT_STATE_CASTING))
148 return;
149
150 while (uint32 eventId = events.ExecuteEvent())
151 {
152 switch (eventId)
153 {
154 case EVENT_FEL_FIREBALL:
155 DoCastVictim(SPELL_FEL_FIREBALL);
156 events.ScheduleEvent(EVENT_FEL_FIREBALL, urand(10*IN_MILLISECONDS, 15*IN_MILLISECONDS));
157 return;
158 case EVENT_FEL_LIGHTNING:
159 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_LORD_HITTIN))
160 DoCast(target, SPELL_FEL_LIGHTING);
161 events.ScheduleEvent(EVENT_FEL_LIGHTNING, urand(10*IN_MILLISECONDS, 15*IN_MILLISECONDS));
162 return;
163 case EVENT_INCINERATE_FLESH:
164 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true, -SPELL_LORD_HITTIN))
165 {
166 Talk(EMOTE_INCINERATE, target);
167 Talk(SAY_INCINERATE);
168 DoCast(target, SPELL_INCINERATE_FLESH);
169 }
170 events.ScheduleEvent(EVENT_INCINERATE_FLESH, urand(20*IN_MILLISECONDS, 25*IN_MILLISECONDS));
171 return;
172 case EVENT_NETHER_POWER:
173 me->CastCustomSpell(SPELL_NETHER_POWER, SPELLVALUE_AURA_STACK, RAID_MODE<uint32>(5, 10, 5, 10), me, true);
174 events.ScheduleEvent(EVENT_NETHER_POWER, 40*IN_MILLISECONDS);
175 return;
176 case EVENT_LEGION_FLAME:
177 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true, -SPELL_LORD_HITTIN))
178 {
179 Talk(EMOTE_LEGION_FLAME, target);
180 DoCast(target, SPELL_LEGION_FLAME);
181 }
182 events.ScheduleEvent(EVENT_LEGION_FLAME, 30*IN_MILLISECONDS);
183 return;
184 case EVENT_SUMMONO_NETHER_PORTAL:
185 Talk(EMOTE_NETHER_PORTAL);
186 Talk(SAY_MISTRESS_OF_PAIN);
187 DoCast(SPELL_NETHER_PORTAL);
188 events.ScheduleEvent(EVENT_SUMMONO_NETHER_PORTAL, 2*MINUTE*IN_MILLISECONDS);
189 return;
190 case EVENT_SUMMON_INFERNAL_ERUPTION:
191 Talk(EMOTE_INFERNAL_ERUPTION);
192 Talk(SAY_INFERNAL_ERUPTION);
193 DoCast(SPELL_INFERNAL_ERUPTION);
194 events.ScheduleEvent(EVENT_SUMMON_INFERNAL_ERUPTION, 2*MINUTE*IN_MILLISECONDS);
195 return;
196 }
197 }
198
199 DoMeleeAttackIfReady();
200 }
201 };
202
203 CreatureAI* GetAI(Creature* creature) const override
204 {
205 return GetInstanceAI<boss_jaraxxusAI>(creature);
206 }

311

313 {
314 _felStreakTimer = 30 * IN_MILLISECONDS;
315 }
316
317 void Reset() override
318 {
319 Initialize();
320 me->SetInCombatWithZone();
321 }
322
323 void UpdateAI(uint32 diff) override
324 {
325 if (_instance->GetBossState(BOSS_JARAXXUS) != IN_PROGRESS)
326 {
327 me->DespawnOrUnsummon();
328 return;
329 }
330
331 if (!UpdateVictim())
332 return;
333
334 if (_felStreakTimer <= diff)
335 {
336 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
337 DoCast(target, SPELL_FEL_STREAK);
338 _felStreakTimer = 30*IN_MILLISECONDS;
339 }
340 else
341 _felStreakTimer -= diff;
342
343 DoMeleeAttackIfReady();
344 }
345 private:
346 uint32 _felStreakTimer;
347 InstanceScript* _instance;
348 };
349
350 CreatureAI* GetAI(Creature* creature) const override
351 {
352 return GetInstanceAI<npc_fel_infernalAI>(creature);

431 _instance->SetData(DATA_MISTRESS_OF_PAIN_COUNT, DECREASE);
432 }
433
434 void UpdateAI(uint32 diff) override
435 {
436 if (_instance->GetBossState(BOSS_JARAXXUS) != IN_PROGRESS)
437 {
438 me->DespawnOrUnsummon();
439 return;
440 }
441
442 if (!UpdateVictim())
443 return;
444
445 _events.Update(diff);
446
447 if (me->HasUnitState(UNIT_STATE_CASTING))
448 return;
449
450 while (uint32 eventId = _events.ExecuteEvent())
451 {
452 switch (eventId)
453 {
454 case EVENT_SHIVAN_SLASH:
455 DoCastVictim(SPELL_SHIVAN_SLASH);
456 _events.ScheduleEvent(EVENT_SHIVAN_SLASH, 30*IN_MILLISECONDS);
457 return;
458 case EVENT_SPINNING_STRIKE:
459 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
460 DoCast(target, SPELL_SPINNING_STRIKE);
461 _events.ScheduleEvent(EVENT_SPINNING_STRIKE, 30*IN_MILLISECONDS);
462 return;
463 case EVENT_MISTRESS_KISS:
464 DoCast(me, SPELL_MISTRESS_KISS);
465 _events.ScheduleEvent(EVENT_MISTRESS_KISS, 30*IN_MILLISECONDS);
466 return;
467 default:
468 break;
469 }
470 }
471
472 DoMeleeAttackIfReady();
473 }
474 private:
475 InstanceScript* _instance;
476 EventMap _events;
477 };

502 void HandleDummyTick(AuraEffect const* /*aurEff*/)
503 {
504 Unit* caster = GetCaster();
505 Unit* target = GetTarget();
506 if (caster && target)
507 {
508 if (target->HasUnitState(UNIT_STATE_CASTING))
509 {
510 caster->CastSpell(target, SPELL_MISTRESS_KISS_DAMAGE_SILENCE, true);
511 target->RemoveAurasDueToSpell(GetSpellInfo()->Id);
512 }
513 }
514 }
515
516 void Register() override
517 {
518 OnEffectPeriodic += AuraEffectPeriodicFn(spell_mistress_kiss_AuraScript::HandleDummyTick, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);

552 void FilterTargets(std::list<WorldObject*>& targets)
553 {
554 // get a list of players with mana
555 targets.remove_if(MistressKissTargetSelector());
556 if (targets.empty())
557 return;
558
559 WorldObject* target = Trinity::Containers::SelectRandomContainerElement(targets);
560 targets.clear();
561 targets.push_back(target);
562 }
563
564 void HandleScript(SpellEffIndex /*effIndex*/)
565 {
566 GetCaster()->CastSpell(GetHitUnit(), uint32(GetEffectValue()), true);
567 }
568

570 {

572 OnEffectHitTarget += SpellEffectFn(spell_mistress_kiss_area_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
573 }

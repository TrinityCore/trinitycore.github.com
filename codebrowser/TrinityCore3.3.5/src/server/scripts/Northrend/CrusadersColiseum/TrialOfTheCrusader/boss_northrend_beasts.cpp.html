152class boss_gormok : public CreatureScript
153{
154 public:
155 boss_gormok() : CreatureScript("boss_gormok") { }
156
157 struct boss_gormokAI : public BossAI
158 {
159 boss_gormokAI(Creature* creature) : BossAI(creature, BOSS_BEASTS) { }
160
161 void Reset() override
162 {
163 events.ScheduleEvent(EVENT_IMPALE, urand(8*IN_MILLISECONDS, 10*IN_MILLISECONDS));
164 events.ScheduleEvent(EVENT_STAGGERING_STOMP, 15*IN_MILLISECONDS);
165 events.ScheduleEvent(EVENT_THROW, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS));
166
167 summons.DespawnAll();
168 }
169
170 void EnterEvadeMode(EvadeReason why) override
171 {
172 instance->DoUseDoorOrButton(instance->GetGuidData(GO_MAIN_GATE_DOOR));
173 ScriptedAI::EnterEvadeMode(why);
174 }
175
176 void MovementInform(uint32 type, uint32 pointId) override
177 {
178 if (type != POINT_MOTION_TYPE)
179 return;
180
181 switch (pointId)
182 {
183 case 0:
184 instance->DoUseDoorOrButton(instance->GetGuidData(GO_MAIN_GATE_DOOR));
185 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
186 me->SetReactState(REACT_AGGRESSIVE);
187 me->SetInCombatWithZone();
188 break;
189 default:
190 break;
191 }
192 }
193
194 void JustDied(Unit* /*killer*/) override
195 {
196 instance->SetData(TYPE_NORTHREND_BEASTS, GORMOK_DONE);
197 }
198
199 void JustReachedHome() override
200 {
201 instance->DoUseDoorOrButton(instance->GetGuidData(GO_MAIN_GATE_DOOR));
202 instance->SetData(TYPE_NORTHREND_BEASTS, FAIL);
203
204 me->DespawnOrUnsummon();
205 }
206
207 void EnterCombat(Unit* /*who*/) override
208 {
209 _EnterCombat();
210 me->SetInCombatWithZone();
211 instance->SetData(TYPE_NORTHREND_BEASTS, GORMOK_IN_PROGRESS);
212
213 for (uint8 i = 0; i < MAX_SNOBOLDS; i++)
214 {
215 if (Creature* pSnobold = DoSpawnCreature(NPC_SNOBOLD_VASSAL, 0, 0, 0, 0, TEMPSUMMON_CORPSE_DESPAWN, 0))
216 {
217 pSnobold->EnterVehicle(me, i);
218 pSnobold->SetInCombatWithZone();
219 pSnobold->AI()->DoAction(ACTION_ENABLE_FIRE_BOMB);
220 }
221 }
222 }
223
224 void DamageTaken(Unit* /*who*/, uint32& damage) override
225 {
226 // despawn the remaining passengers on death
227 if (damage >= me->GetHealth())
228 for (uint8 i = 0; i < MAX_SNOBOLDS; ++i)
229 if (Unit* pSnobold = me->GetVehicleKit()->GetPassenger(i))
230 pSnobold->ToCreature()->DespawnOrUnsummon();
231 }
232
233 void UpdateAI(uint32 diff) override
234 {
235 if (!UpdateVictim())
236 return;
237
238 events.Update(diff);
239
240 if (me->HasUnitState(UNIT_STATE_CASTING))
241 return;
242
243 while (uint32 eventId = events.ExecuteEvent())
244 {
245 switch (eventId)
246 {
247 case EVENT_IMPALE:
248 DoCastVictim(SPELL_IMPALE);
249 events.ScheduleEvent(EVENT_IMPALE, urand(8*IN_MILLISECONDS, 10*IN_MILLISECONDS));
250 return;
251 case EVENT_STAGGERING_STOMP:
252 DoCastVictim(SPELL_STAGGERING_STOMP);
253 events.ScheduleEvent(EVENT_STAGGERING_STOMP, 15*IN_MILLISECONDS);
254 return;
255 case EVENT_THROW:
256 for (uint8 i = 0; i < MAX_SNOBOLDS; ++i)
257 {
258 if (Unit* pSnobold = me->GetVehicleKit()->GetPassenger(i))
259 {
260 pSnobold->ExitVehicle();
261 pSnobold->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
262 pSnobold->ToCreature()->SetReactState(REACT_AGGRESSIVE);
263 pSnobold->ToCreature()->AI()->DoAction(ACTION_DISABLE_FIRE_BOMB);
264 pSnobold->CastSpell(me, SPELL_RISING_ANGER, true);
265 Talk(EMOTE_SNOBOLLED);
266 break;
267 }
268 }
269 events.ScheduleEvent(EVENT_THROW, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS));
270 return;
271 default:
272 return;
273 }
274 }
275
276 DoMeleeAttackIfReady();
277 }
278 };
279
280 CreatureAI* GetAI(Creature* creature) const override
281 {
282 return GetInstanceAI<boss_gormokAI>(creature);
283 }

286class npc_snobold_vassal : public CreatureScript
287{
288 public:
289 npc_snobold_vassal() : CreatureScript("npc_snobold_vassal") { }
290
291 struct npc_snobold_vassalAI : public ScriptedAI
292 {
293 npc_snobold_vassalAI(Creature* creature) : ScriptedAI(creature)
294 {
295 _targetDied = false;
296 _instance = creature->GetInstanceScript();
297 _instance->SetData(DATA_SNOBOLD_COUNT, INCREASE);
298 }
299
300 void Reset() override
301 {
302 _events.ScheduleEvent(EVENT_BATTER, 5*IN_MILLISECONDS);
303 _events.ScheduleEvent(EVENT_HEAD_CRACK, 25*IN_MILLISECONDS);
304
305 _targetGUID.Clear();
306 _targetDied = false;
307
308 //Workaround for Snobold
309 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
310 }
311
312 void EnterCombat(Unit* who) override
313 {
314 _targetGUID = who->GetGUID();
315 me->TauntApply(who);
316 DoCast(who, SPELL_SNOBOLLED);
317 }
318
319 void DamageTaken(Unit* pDoneBy, uint32 &uiDamage) override
320 {
321 if (pDoneBy->GetGUID() == _targetGUID)
322 uiDamage = 0;
323 }
324
325 void MovementInform(uint32 type, uint32 pointId) override
326 {
327 if (type != POINT_MOTION_TYPE)
328 return;
329
330 switch (pointId)
331 {
332 case 0:
333 if (_targetDied)
334 me->DespawnOrUnsummon();
335 break;
336 default:
337 break;
338 }
339 }
340
341 void JustDied(Unit* /*killer*/) override
342 {
343 if (Unit* target = ObjectAccessor::GetPlayer(*me, _targetGUID))
344 if (target->IsAlive())
345 target->RemoveAurasDueToSpell(SPELL_SNOBOLLED);
346 _instance->SetData(DATA_SNOBOLD_COUNT, DECREASE);
347 }
348
349 void DoAction(int32 action) override
350 {
351 switch (action)
352 {
353 case ACTION_ENABLE_FIRE_BOMB:
354 _events.ScheduleEvent(EVENT_FIRE_BOMB, urand(5*IN_MILLISECONDS, 30*IN_MILLISECONDS));
355 break;
356 case ACTION_DISABLE_FIRE_BOMB:
357 _events.CancelEvent(EVENT_FIRE_BOMB);
358 break;
359 default:
360 break;
361 }
362 }
363
364 void UpdateAI(uint32 diff) override
365 {
366 if (!UpdateVictim() || _targetDied)
367 return;
368
369 if (Unit* target = ObjectAccessor::GetPlayer(*me, _targetGUID))
370 {
371 if (!target->IsAlive())
372 {
373 Unit* gormok = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(NPC_GORMOK));
374 if (gormok && gormok->IsAlive())
375 {
376 SetCombatMovement(false);
377 _targetDied = true;
378
379 // looping through Gormoks seats
380 for (uint8 i = 0; i < MAX_SNOBOLDS; i++)
381 {
382 if (!gormok->GetVehicleKit()->GetPassenger(i))
383 {
384 me->EnterVehicle(gormok, i);
385 DoAction(ACTION_ENABLE_FIRE_BOMB);
386 break;
387 }
388 }
389 }
390 else if (Unit* target2 = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
391 {
392 _targetGUID = target2->GetGUID();
393 me->GetMotionMaster()->MoveJump(*target2, 15.0f, 15.0f);
394 }
395 }
396 }
397
398 _events.Update(diff);
399
400 if (me->HasUnitState(UNIT_STATE_CASTING))
401 return;
402
403 while (uint32 eventId = _events.ExecuteEvent())
404 {
405 switch (eventId)
406 {
407 case EVENT_FIRE_BOMB:
408 if (me->GetVehicleBase())
409 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, -me->GetVehicleBase()->GetCombatReach(), true))
410 me->CastSpell(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), SPELL_FIRE_BOMB, true);
411 _events.ScheduleEvent(EVENT_FIRE_BOMB, 20*IN_MILLISECONDS);
412 return;
413 case EVENT_HEAD_CRACK:
414 // commented out while SPELL_SNOBOLLED gets fixed
415 //if (Unit* target = ObjectAccessor::GetPlayer(*me, m_uiTargetGUID))
416 DoCastVictim(SPELL_HEAD_CRACK);
417 _events.ScheduleEvent(EVENT_HEAD_CRACK, 30*IN_MILLISECONDS);
418 return;
419 case EVENT_BATTER:
420 // commented out while SPELL_SNOBOLLED gets fixed
421 //if (Unit* target = ObjectAccessor::GetPlayer(*me, m_uiTargetGUID))
422 DoCastVictim(SPELL_BATTER);
423 _events.ScheduleEvent(EVENT_BATTER, 10*IN_MILLISECONDS);
424 return;
425 default:
426 return;
427 }
428 }
429
430 // do melee attack only when not on Gormoks back
431 if (!me->GetVehicleBase())
432 DoMeleeAttackIfReady();
433 }
434 private:
435 EventMap _events;
436 InstanceScript* _instance;
437 ObjectGuid _targetGUID;
438 bool _targetDied;
439 };
440
441 CreatureAI* GetAI(Creature* creature) const override
442 {
443 return GetInstanceAI<npc_snobold_vassalAI>(creature);
444 }
445};

483struct boss_jormungarAI : public BossAI
484{
485 boss_jormungarAI(Creature* creature) : BossAI(creature, BOSS_BEASTS)
486 {
487 OtherWormEntry = 0;
488 ModelStationary = 0;
489 ModelMobile = 0;
490
491 BiteSpell = 0;
492 SpewSpell = 0;
493 SpitSpell = 0;
494 SpraySpell = 0;
495
496 Phase = PHASE_MOBILE;
497 Enraged = false;
498 WasMobile = false;
499 }
500
501 void Reset() override
502 {
503 Enraged = false;
504
505 events.ScheduleEvent(EVENT_SPIT, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
506 events.ScheduleEvent(EVENT_SPRAY, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
507 events.ScheduleEvent(EVENT_SWEEP, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
508 events.ScheduleEvent(EVENT_BITE, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_MOBILE);
509 events.ScheduleEvent(EVENT_SPEW, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_MOBILE);
510 events.ScheduleEvent(EVENT_SLIME_POOL, 15*IN_MILLISECONDS, 0, PHASE_MOBILE);
511 }
512
513 void JustDied(Unit* /*killer*/) override
514 {
515 if (Creature* otherWorm = ObjectAccessor::GetCreature(*me, instance->GetGuidData(OtherWormEntry)))
516 {
517 if (!otherWorm->IsAlive())
518 {
519 instance->SetData(TYPE_NORTHREND_BEASTS, SNAKES_DONE);
520
521 me->DespawnOrUnsummon();
522 otherWorm->DespawnOrUnsummon();
523 }
524 else
525 instance->SetData(TYPE_NORTHREND_BEASTS, SNAKES_SPECIAL);
526 }
527 }
528
529 void JustReachedHome() override
530 {
531 // prevent losing 2 attempts at once on heroics
532 if (instance->GetData(TYPE_NORTHREND_BEASTS) != FAIL)
533 instance->SetData(TYPE_NORTHREND_BEASTS, FAIL);
534
535 me->DespawnOrUnsummon();
536 }
537
538 void EnterCombat(Unit* /*who*/) override
539 {
540 _EnterCombat();
541 me->SetInCombatWithZone();
542 instance->SetData(TYPE_NORTHREND_BEASTS, SNAKES_IN_PROGRESS);
543 }
544
545 void UpdateAI(uint32 diff) override
546 {
547 if (!UpdateVictim())
548 return;
549
550 if (!Enraged && instance->GetData(TYPE_NORTHREND_BEASTS) == SNAKES_SPECIAL)
551 {
552 me->RemoveAurasDueToSpell(SPELL_SUBMERGE_0);
553 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
554 DoCast(SPELL_ENRAGE);
555 Enraged = true;
556 Talk(EMOTE_ENRAGE);
557 }
558
559 events.Update(diff);
560
561 if (me->HasUnitState(UNIT_STATE_CASTING))
562 return;
563
564 while (uint32 eventId = events.ExecuteEvent())
565 {
566 switch (eventId)
567 {
568 case EVENT_EMERGE:
569 Emerge();
570 return;
571 case EVENT_SUBMERGE:
572 Submerge();
573 return;
574 case EVENT_BITE:
575 DoCastVictim(BiteSpell);
576 events.ScheduleEvent(EVENT_BITE, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_MOBILE);
577 return;
578 case EVENT_SPEW:
579 DoCastAOE(SpewSpell);
580 events.ScheduleEvent(EVENT_SPEW, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_MOBILE);
581 return;
582 case EVENT_SLIME_POOL:
583 DoCast(me, SUMMON_SLIME_POOL);
584 events.ScheduleEvent(EVENT_SLIME_POOL, 30*IN_MILLISECONDS, 0, PHASE_MOBILE);
585 return;
586 case EVENT_SUMMON_ACIDMAW:
587 if (Creature* acidmaw = me->SummonCreature(NPC_ACIDMAW, ToCCommonLoc[9].GetPositionX(), ToCCommonLoc[9].GetPositionY(), ToCCommonLoc[9].GetPositionZ(), 5, TEMPSUMMON_MANUAL_DESPAWN))
588 {
589 acidmaw->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
590 acidmaw->SetReactState(REACT_AGGRESSIVE);
591 acidmaw->SetInCombatWithZone();
592 acidmaw->CastSpell(acidmaw, SPELL_EMERGE_0);
593 }
594 return;
595 case EVENT_SPRAY:
596 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
597 DoCast(target, SpraySpell);
598 events.ScheduleEvent(EVENT_SPRAY, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
599 return;
600 case EVENT_SWEEP:
601 DoCastAOE(SPELL_SWEEP_0);
602 events.ScheduleEvent(EVENT_SWEEP, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
603 return;
604 default:
605 return;
606 }
607 }
608 if (events.IsInPhase(PHASE_MOBILE))
609 DoMeleeAttackIfReady();
610 if (events.IsInPhase(PHASE_STATIONARY))
611 DoSpellAttackIfReady(SpitSpell);
612 }
613
614 void Submerge()
615 {
616 DoCast(me, SPELL_SUBMERGE_0);
617 me->RemoveAurasDueToSpell(SPELL_EMERGE_0);
618 me->SetInCombatWithZone();
619 events.SetPhase(PHASE_SUBMERGED);
620 events.ScheduleEvent(EVENT_EMERGE, 5*IN_MILLISECONDS, 0, PHASE_SUBMERGED);
621 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
622 me->GetMotionMaster()->MovePoint(0, ToCCommonLoc[1].GetPositionX()+ frand(-40.0f, 40.0f), ToCCommonLoc[1].GetPositionY() + frand(-40.0f, 40.0f), ToCCommonLoc[1].GetPositionZ());
623 WasMobile = !WasMobile;
624 }
625
626 void Emerge()
627 {
628 DoCast(me, SPELL_EMERGE_0);
629 me->SetDisplayId(ModelMobile);
630 me->RemoveAurasDueToSpell(SPELL_SUBMERGE_0);
631 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
632
633 // if the worm was mobile before submerging, make him stationary now
634 if (WasMobile)
635 {
636 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
637 SetCombatMovement(false);
638 me->SetDisplayId(ModelStationary);
639 events.SetPhase(PHASE_STATIONARY);
640 events.ScheduleEvent(EVENT_SUBMERGE, 45*IN_MILLISECONDS, 0, PHASE_STATIONARY);
641 events.ScheduleEvent(EVENT_SPIT, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
642 events.ScheduleEvent(EVENT_SPRAY, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
643 events.ScheduleEvent(EVENT_SWEEP, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_STATIONARY);
644 }
645 else
646 {
647 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
648 SetCombatMovement(true);
649 me->GetMotionMaster()->MoveChase(me->GetVictim());
650 me->SetDisplayId(ModelMobile);
651 events.SetPhase(PHASE_MOBILE);
652 events.ScheduleEvent(EVENT_SUBMERGE, 45*IN_MILLISECONDS, 0, PHASE_MOBILE);
653 events.ScheduleEvent(EVENT_BITE, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_MOBILE);
654 events.ScheduleEvent(EVENT_SPEW, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS), 0, PHASE_MOBILE);
655 events.ScheduleEvent(EVENT_SLIME_POOL, 15*IN_MILLISECONDS, 0, PHASE_MOBILE);
656 }
657 }
658
659 protected:
660 uint32 OtherWormEntry;
661 uint32 ModelStationary;
662 uint32 ModelMobile;
663
664 uint32 BiteSpell;
665 uint32 SpewSpell;
666 uint32 SpitSpell;
667 uint32 SpraySpell;
668
669 Phases Phase;
670 bool Enraged;
671 bool WasMobile;
672};

902 _stage = 4;

904 _stage = 6;
905 }

907 break;
908 case 1: // Finish trample
909 _movementFinish = true;
910 break;
911 case 2:
912 instance->DoUseDoorOrButton(instance->GetGuidData(GO_MAIN_GATE_DOOR));
913 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);

953 void UpdateAI(uint32 diff) override
954 {
955 if (!UpdateVictim())
956 return;
957
958 events.Update(diff);
959
960 if (me->HasUnitState(UNIT_STATE_CASTING))
961 return;
962
963 switch (_stage)
964 {
965 case 0:
966 {
967 while (uint32 eventId = events.ExecuteEvent())
968 {
969 switch (eventId)
970 {
971 case EVENT_FEROCIOUS_BUTT:
972 DoCastVictim(SPELL_FEROCIOUS_BUTT);
973 events.ScheduleEvent(EVENT_FEROCIOUS_BUTT, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS));
974 return;
975 case EVENT_ARCTIC_BREATH:
976 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
977 DoCast(target, SPELL_ARCTIC_BREATH);
978 return;
979 case EVENT_WHIRL:
980 DoCastAOE(SPELL_WHIRL);
981 events.ScheduleEvent(EVENT_WHIRL, urand(15*IN_MILLISECONDS, 30*IN_MILLISECONDS));
982 return;
983 case EVENT_MASSIVE_CRASH:
984 me->GetMotionMaster()->MoveJump(ToCCommonLoc[1], 20.0f, 20.0f, 0); // 1: Middle of the room
985 SetCombatMovement(false);
986 me->AttackStop();
987 _stage = 7; //Invalid (Do nothing more than move)
988 return;
989 default:
990 break;
991 }
992 }
993 DoMeleeAttackIfReady();
994 break;
995 }
996 case 1:
997 DoCastAOE(SPELL_MASSIVE_CRASH);
998 me->StopMoving();
999 me->AttackStop();
1000 _stage = 2;
1001 break;
1002 case 2:
1003 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
1004 {
1005 me->StopMoving();
1006 me->AttackStop();
1007 _trampleTargetGUID = target->GetGUID();
1008 me->SetTarget(_trampleTargetGUID);
1009 _trampleCast = false;
1010 SetCombatMovement(false);
1011 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE);
1012 me->GetMotionMaster()->Clear();
1013 me->GetMotionMaster()->MoveIdle();
1014 events.ScheduleEvent(EVENT_TRAMPLE, 4*IN_MILLISECONDS);
1015 _stage = 3;
1016 }
1017 else
1018 _stage = 6;
1019 break;
1020 case 3:
1021 while (uint32 eventId = events.ExecuteEvent())
1022 {
1023 switch (eventId)
1024 {
1025 case EVENT_TRAMPLE:
1026 {
1027 if (Unit* target = ObjectAccessor::GetPlayer(*me, _trampleTargetGUID))
1028 {
1029 me->StopMoving();
1030 me->AttackStop();
1031 _trampleCast = false;
1032 _movementStarted = true;
1033 _trampleTargetX = target->GetPositionX();
1034 _trampleTargetY = target->GetPositionY();
1035 _trampleTargetZ = target->GetPositionZ();
1036 // 2: Hop Backwards
1037 me->GetMotionMaster()->MoveJump(2*me->GetPositionX() - _trampleTargetX, 2*me->GetPositionY() - _trampleTargetY, me->GetPositionZ(), me->GetOrientation(), 30.0f, 20.0f, 0);
1038 _stage = 7; //Invalid (Do nothing more than move)
1039 }
1040 else
1041 _stage = 6;
1042 break;
1043 }
1044 default:
1045 break;
1046 }
1047 }
1048 break;
1049 case 4:
1050 me->StopMoving();
1051 me->AttackStop();
1052 if (Player* target = ObjectAccessor::GetPlayer(*me, _trampleTargetGUID))
1053 Talk(EMOTE_TRAMPLE_START, target);
1054 me->GetMotionMaster()->MoveCharge(_trampleTargetX, _trampleTargetY, _trampleTargetZ, 42, 1);
1055 me->SetTarget(ObjectGuid::Empty);
1056 _stage = 5;
1057 break;
1058 case 5:
1059 if (_movementFinish)
1060 {
1061 DoCastAOE(SPELL_TRAMPLE);
1062 _movementFinish = false;
1063 _stage = 6;
1064 return;
1065 }
1066 if (events.ExecuteEvent() == EVENT_TRAMPLE)
1067 {
1068 Map::PlayerList const &lPlayers = me->GetMap()->GetPlayers();
1069 for (Map::PlayerList::const_iterator itr = lPlayers.begin(); itr != lPlayers.end(); ++itr)
1070 {
1071 if (Unit* player = itr->GetSource())
1072 {
1073 if (player->IsAlive() && player->IsWithinDistInMap(me, 6.0f))
1074 {
1075 DoCastAOE(SPELL_TRAMPLE);
1076 events.ScheduleEvent(EVENT_TRAMPLE, 4*IN_MILLISECONDS);
1077 break;
1078 }
1079 }
1080 }
1081 }
1082 break;
1083 case 6:
1084 if (!_trampleCast)
1085 {
1086 DoCast(me, SPELL_STAGGERED_DAZE);
1087 Talk(EMOTE_TRAMPLE_CRASH);
1088 }
1089 else
1090 {
1091 DoCast(me, SPELL_FROTHING_RAGE, true);
1092 Talk(EMOTE_TRAMPLE_FAIL);
1093 }
1094 _movementStarted = false;
1095 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE);
1096 SetCombatMovement(true);
1097 me->GetMotionMaster()->MovementExpired();
1098 me->GetMotionMaster()->Clear();
1099 me->GetMotionMaster()->MoveChase(me->GetVictim());
1100 AttackStart(me->GetVictim());
1101 events.ScheduleEvent(EVENT_MASSIVE_CRASH, 40*IN_MILLISECONDS);
1102 events.ScheduleEvent(EVENT_ARCTIC_BREATH, urand(15*IN_MILLISECONDS, 25*IN_MILLISECONDS));
1103 _stage = 0;
1104 break;
1105 default:
1106 break;
1107 }
1108 }
1109
1110 private:
1111 float _trampleTargetX, _trampleTargetY, _trampleTargetZ;
1112 ObjectGuid _trampleTargetGUID;
1113 bool _movementStarted;
1114 bool _movementFinish;
1115 bool _trampleCast;
1116 uint8 _stage;
1117 };

112class OrientationCheck : public std::unary_function<Unit*, bool>
113{
114 public:
115 explicit OrientationCheck(Unit* _caster) : caster(_caster) { }
116 bool operator()(WorldObject* object)
117 {
118 return !object->isInFront(caster, 2.5f) || !object->IsWithinDist(caster, 40.0f);
119 }
120
121 private:
122 Unit* caster;
123};

151class boss_eadric : public CreatureScript
152{
153public:
154 boss_eadric() : CreatureScript("boss_eadric") { }
155 struct boss_eadricAI : public ScriptedAI
156 {
157 boss_eadricAI(Creature* creature) : ScriptedAI(creature)
158 {
159 Initialize();
160 instance = creature->GetInstanceScript();
161 creature->SetReactState(REACT_PASSIVE);
162 creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
163 }
164
165 void Initialize()
166 {
167 uiVenganceTimer = 10000;
168 uiRadianceTimer = 16000;
169 uiHammerJusticeTimer = 25000;
170 uiResetTimer = 5000;
171
172 bDone = false;
173 }
174
175 InstanceScript* instance;
176
177 uint32 uiVenganceTimer;
178 uint32 uiRadianceTimer;
179 uint32 uiHammerJusticeTimer;
180 uint32 uiResetTimer;
181
182 bool bDone;
183
184 void Reset() override
185 {
186 Initialize();
187 }
188
189 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override
190 {
191 if (damage >= me->GetHealth())
192 {
193 damage = 0;
194 EnterEvadeMode();
195 me->setFaction(35);
196 bDone = true;
197 }
198 }
199
200 void MovementInform(uint32 MovementType, uint32 /*Data*/) override
201 {
202 if (MovementType != POINT_MOTION_TYPE)
203 return;
204
205 instance->SetData(BOSS_ARGENT_CHALLENGE_E, DONE);
206
207 me->DisappearAndDie();
208 }
209
210 void UpdateAI(uint32 uiDiff) override
211 {
212 if (bDone && uiResetTimer <= uiDiff)
213 {
214 me->GetMotionMaster()->MovePoint(0, 746.87f, 665.87f, 411.75f);
215 bDone = false;
216 } else uiResetTimer -= uiDiff;
217
218 if (!UpdateVictim())
219 return;
220
221 if (uiHammerJusticeTimer <= uiDiff)
222 {
223 me->InterruptNonMeleeSpells(true);
224
225 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 250, true))
226 {
227 if (target && target->IsAlive())
228 {
229 DoCast(target, SPELL_HAMMER_JUSTICE);
230 DoCast(target, SPELL_HAMMER_RIGHTEOUS);
231 }
232 }
233 uiHammerJusticeTimer = 25000;
234 } else uiHammerJusticeTimer -= uiDiff;
235
236 if (uiVenganceTimer <= uiDiff)
237 {
238 DoCast(me, SPELL_VENGEANCE);
239
240 uiVenganceTimer = 10000;
241 } else uiVenganceTimer -= uiDiff;
242
243 if (uiRadianceTimer <= uiDiff)
244 {
245 DoCastAOE(SPELL_RADIANCE);
246
247 uiRadianceTimer = 16000;
248 } else uiRadianceTimer -= uiDiff;
249
250 DoMeleeAttackIfReady();
251 }
252 };
253
254 CreatureAI* GetAI(Creature* creature) const override
255 {
256 return GetInstanceAI<boss_eadricAI>(creature);
257 }

260class boss_paletress : public CreatureScript
261{
262public:
263 boss_paletress() : CreatureScript("boss_paletress") { }
264
265 struct boss_paletressAI : public ScriptedAI
266 {
267 boss_paletressAI(Creature* creature) : ScriptedAI(creature)
268 {
269 Initialize();
270 instance = creature->GetInstanceScript();
271
272 creature->SetReactState(REACT_PASSIVE);
273 creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
274 creature->RestoreFaction();
275 }
276
277 void Initialize()
278 {
279 uiHolyFireTimer = urand(9000, 12000);
280 uiHolySmiteTimer = urand(5000, 7000);
281 uiRenewTimer = urand(2000, 5000);
282
283 uiResetTimer = 7000;
284
285 bHealth = false;
286 bDone = false;
287 }
288
289 InstanceScript* instance;
290 ObjectGuid MemoryGUID;
291
292 bool bHealth;
293 bool bDone;
294
295 uint32 uiHolyFireTimer;
296 uint32 uiHolySmiteTimer;
297 uint32 uiRenewTimer;
298 uint32 uiResetTimer;
299
300 void Reset() override
301 {
302 me->RemoveAllAuras();
303
304 Initialize();
305
306 if (Creature* pMemory = ObjectAccessor::GetCreature(*me, MemoryGUID))
307 if (pMemory->IsAlive())
308 pMemory->RemoveFromWorld();
309 }
310
311 void SetData(uint32 uiId, uint32 /*uiValue*/) override
312 {
313 if (uiId == 1)
314 me->RemoveAura(SPELL_SHIELD);
315 }
316
317 void DamageTaken(Unit* /*done_by*/, uint32 &damage) override
318 {
319 if (damage >= me->GetHealth())
320 {
321 damage = 0;
322 EnterEvadeMode();
323 me->setFaction(35);
324 bDone = true;
325 }
326 }
327
328 void MovementInform(uint32 MovementType, uint32 Point) override
329 {
330 if (MovementType != POINT_MOTION_TYPE || Point != 0)
331 return;
332
333 instance->SetData(BOSS_ARGENT_CHALLENGE_P, DONE);
334
335 me->DisappearAndDie();
336 }
337
338 void UpdateAI(uint32 uiDiff) override
339 {
340 if (bDone && uiResetTimer <= uiDiff)
341 {
342 me->GetMotionMaster()->MovePoint(0, 746.87f, 665.87f, 411.75f);
343 bDone = false;
344 } else uiResetTimer -= uiDiff;
345
346 if (!UpdateVictim())
347 return;
348
349 if (uiHolyFireTimer <= uiDiff)
350 {
351 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 250, true))
352 {
353 if (target && target->IsAlive())
354 DoCast(target, SPELL_HOLY_FIRE);
355 }
356 if (me->HasAura(SPELL_SHIELD))
357 uiHolyFireTimer = 13000;
358 else
359 uiHolyFireTimer = urand(9000, 12000);
360 } else uiHolyFireTimer -= uiDiff;
361
362 if (uiHolySmiteTimer <= uiDiff)
363 {
364 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 250, true))
365 {
366 if (target && target->IsAlive())
367 DoCast(target, SPELL_SMITE);
368 }
369 if (me->HasAura(SPELL_SHIELD))
370 uiHolySmiteTimer = 9000;
371 else
372 uiHolySmiteTimer = urand(5000, 7000);
373 } else uiHolySmiteTimer -= uiDiff;
374
375 if (me->HasAura(SPELL_SHIELD))
376 {
377 if (uiRenewTimer <= uiDiff)
378 {
379 me->InterruptNonMeleeSpells(true);
380 uint8 uiTarget = urand(0, 1);
381 switch (uiTarget)
382 {
383 case 0:
384 DoCast(me, SPELL_RENEW);
385 break;
386 case 1:
387 if (Creature* pMemory = ObjectAccessor::GetCreature(*me, MemoryGUID))
388 if (pMemory->IsAlive())
389 DoCast(pMemory, SPELL_RENEW);
390 break;
391 }
392 uiRenewTimer = urand(15000, 17000);
393 } else uiRenewTimer -= uiDiff;
394 }
395
396 if (!bHealth && !HealthAbovePct(25))
397 {
398 me->InterruptNonMeleeSpells(true);
399 DoCastAOE(SPELL_HOLY_NOVA, false);
400 DoCast(me, SPELL_SHIELD);
401 DoCastAOE(SPELL_SUMMON_MEMORY, false);
402 DoCastAOE(SPELL_CONFESS, false);
403
404 bHealth = true;
405 }
406
407 DoMeleeAttackIfReady();
408 }
409
410 void JustSummoned(Creature* summon) override
411 {
412 MemoryGUID = summon->GetGUID();
413 }
414 };
415
416 CreatureAI* GetAI(Creature* creature) const override
417 {
418 return GetInstanceAI<boss_paletressAI>(creature);
419 }

642class spell_paletress_summon_memory : public SpellScriptLoader
643{
644 public:
645 spell_paletress_summon_memory() : SpellScriptLoader("spell_paletress_summon_memory") { }
646
647 class spell_paletress_summon_memory_SpellScript : public SpellScript
648 {
649 PrepareSpellScript(spell_paletress_summon_memory_SpellScript);
650
651 bool Validate(SpellInfo const* /*spellInfo*/) override
652 {
653 for (uint8 i = 0; i < 25; ++i)
654 if (!sSpellMgr->GetSpellInfo(memorySpellId[i]))
655 return false;
656 return true;
657 }
658
659 void FilterTargets(std::list<WorldObject*>& targets)
660 {
661 if (targets.empty())
662 return;
663
664 WorldObject* target = Trinity::Containers::SelectRandomContainerElement(targets);
665 targets.clear();
666 targets.push_back(target);
667 }
668
669 void HandleScript(SpellEffIndex /*effIndex*/)
670 {
671 GetHitUnit()->CastSpell(GetHitUnit(), memorySpellId[urand(0, 24)], true, NULL, NULL, GetCaster()->GetGUID());
672 }
673
674 void Register() override
675 {
676 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_paletress_summon_memory_SpellScript::FilterTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
677 OnEffectHitTarget += SpellEffectFn(spell_paletress_summon_memory_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
678 }
679 };
680
681 SpellScript* GetSpellScript() const override
682 {
683 return new spell_paletress_summon_memory_SpellScript();
684 }

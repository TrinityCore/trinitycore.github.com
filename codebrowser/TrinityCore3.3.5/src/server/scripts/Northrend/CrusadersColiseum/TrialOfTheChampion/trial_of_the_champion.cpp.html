123 void SetData(uint32 uiType, uint32 /*uiData*/) override
124 {
125 switch (uiType)
126 {
127 case DATA_START:
128 DoSummonGrandChampion(uiFirstBoss);
129 NextStep(10000, false, 1);
130 break;
131 case DATA_IN_POSITION: //movement done.
132 me->GetMotionMaster()->MovePoint(1, 735.81f, 661.92f, 412.39f);
133 if (GameObject* go = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_MAIN_GATE)))
134 instance->HandleGameObject(go->GetGUID(), false);
135 NextStep(10000, false, 3);
136 break;
137 case DATA_LESSER_CHAMPIONS_DEFEATED:
138 {
139 ++uiLesserChampions;
140 GuidList TempList;
141 if (uiLesserChampions == 3 || uiLesserChampions == 6)
142 {
143 switch (uiLesserChampions)
144 {
145 case 3:
146 TempList = Champion2List;
147 break;
148 case 6:
149 TempList = Champion3List;
150 break;
151 }
152
153 for (GuidList::const_iterator itr = TempList.begin(); itr != TempList.end(); ++itr)
154 if (Creature* summon = ObjectAccessor::GetCreature(*me, *itr))
155 AggroAllPlayers(summon);
156 }else if (uiLesserChampions == 9)
157 StartGrandChampionsAttack();
158
159 break;
160 }
161 }
162 }

164 void StartGrandChampionsAttack()
165 {
166 Creature* pGrandChampion1 = ObjectAccessor::GetCreature(*me, uiVehicle1GUID);
167 Creature* pGrandChampion2 = ObjectAccessor::GetCreature(*me, uiVehicle2GUID);
168 Creature* pGrandChampion3 = ObjectAccessor::GetCreature(*me, uiVehicle3GUID);
169
170 if (pGrandChampion1 && pGrandChampion2 && pGrandChampion3)
171 {
172 AggroAllPlayers(pGrandChampion1);
173 AggroAllPlayers(pGrandChampion2);
174 AggroAllPlayers(pGrandChampion3);
175 }

187 void DoSummonGrandChampion(uint32 uiBoss)
188 {
189 ++uiSummonTimes;
190 uint32 VEHICLE_TO_SUMMON1 = 0;
191 uint32 VEHICLE_TO_SUMMON2 = 0;
192 switch (uiBoss)
193 {
194 case 0:
195 VEHICLE_TO_SUMMON1 = VEHICLE_MOKRA_SKILLCRUSHER_MOUNT;
196 VEHICLE_TO_SUMMON2 = VEHICLE_ORGRIMMAR_WOLF;
197 break;
198 case 1:
199 VEHICLE_TO_SUMMON1 = VEHICLE_ERESSEA_DAWNSINGER_MOUNT;
200 VEHICLE_TO_SUMMON2 = VEHICLE_SILVERMOON_HAWKSTRIDER;
201 break;
202 case 2:
203 VEHICLE_TO_SUMMON1 = VEHICLE_RUNOK_WILDMANE_MOUNT;
204 VEHICLE_TO_SUMMON2 = VEHICLE_THUNDER_BLUFF_KODO;
205 break;
206 case 3:
207 VEHICLE_TO_SUMMON1 = VEHICLE_ZUL_TORE_MOUNT;
208 VEHICLE_TO_SUMMON2 = VEHICLE_DARKSPEAR_RAPTOR;
209 break;
210 case 4:
211 VEHICLE_TO_SUMMON1 = VEHICLE_DEATHSTALKER_VESCERI_MOUNT;
212 VEHICLE_TO_SUMMON2 = VEHICLE_FORSAKE_WARHORSE;
213 break;
214 default:
215 return;
216 }
217
218 if (Creature* pBoss = me->SummonCreature(VEHICLE_TO_SUMMON1, SpawnPosition))
219 {
220 switch (uiSummonTimes)
221 {
222 case 1:
223 {
224 uiVehicle1GUID = pBoss->GetGUID();
225 ObjectGuid uiGrandChampionBoss1;
226 if (Vehicle* pVehicle = pBoss->GetVehicleKit())
227 if (Unit* unit = pVehicle->GetPassenger(0))
228 uiGrandChampionBoss1 = unit->GetGUID();
229 instance->SetGuidData(DATA_GRAND_CHAMPION_VEHICLE_1, uiVehicle1GUID);
230 instance->SetGuidData(DATA_GRAND_CHAMPION_1, uiGrandChampionBoss1);
231 pBoss->AI()->SetData(1, 0);
232 break;
233 }
234 case 2:
235 {
236 uiVehicle2GUID = pBoss->GetGUID();
237 ObjectGuid uiGrandChampionBoss2;
238 if (Vehicle* pVehicle = pBoss->GetVehicleKit())
239 if (Unit* unit = pVehicle->GetPassenger(0))
240 uiGrandChampionBoss2 = unit->GetGUID();
241 instance->SetGuidData(DATA_GRAND_CHAMPION_VEHICLE_2, uiVehicle2GUID);
242 instance->SetGuidData(DATA_GRAND_CHAMPION_2, uiGrandChampionBoss2);
243 pBoss->AI()->SetData(2, 0);
244 break;
245 }
246 case 3:
247 {
248 uiVehicle3GUID = pBoss->GetGUID();
249 ObjectGuid uiGrandChampionBoss3;
250 if (Vehicle* pVehicle = pBoss->GetVehicleKit())
251 if (Unit* unit = pVehicle->GetPassenger(0))
252 uiGrandChampionBoss3 = unit->GetGUID();
253 instance->SetGuidData(DATA_GRAND_CHAMPION_VEHICLE_3, uiVehicle3GUID);
254 instance->SetGuidData(DATA_GRAND_CHAMPION_3, uiGrandChampionBoss3);
255 pBoss->AI()->SetData(3, 0);
256 break;
257 }
258 default:
259 return;
260 }
261
262 for (uint8 i = 0; i < 3; ++i)
263 {
264 if (Creature* pAdd = me->SummonCreature(VEHICLE_TO_SUMMON2, SpawnPosition, TEMPSUMMON_CORPSE_DESPAWN))
265 {
266 switch (uiSummonTimes)
267 {
268 case 1:
269 Champion1List.push_back(pAdd->GetGUID());
270 break;
271 case 2:
272 Champion2List.push_back(pAdd->GetGUID());
273 break;
274 case 3:
275 Champion3List.push_back(pAdd->GetGUID());
276 break;
277 }
278
279 switch (i)
280 {
281 case 0:
282 pAdd->GetMotionMaster()->MoveFollow(pBoss, 2.0f, float(M_PI));
283 break;
284 case 1:
285 pAdd->GetMotionMaster()->MoveFollow(pBoss, 2.0f, float(M_PI) / 2);
286 break;
287 case 2:
288 pAdd->GetMotionMaster()->MoveFollow(pBoss, 2.0f, float(M_PI) / 2 + float(M_PI));
289 break;
290 }
291 }
292
293 }
294 }

297 void DoStartArgentChampionEncounter()
298 {
299 me->GetMotionMaster()->MovePoint(1, 735.81f, 661.92f, 412.39f);
300
301 if (me->SummonCreature(uiArgentChampion, SpawnPosition))
302 {
303 for (uint8 i = 0; i < 3; ++i)
304 {
305 if (Creature* pTrash = me->SummonCreature(NPC_ARGENT_LIGHWIELDER, SpawnPosition))
306 pTrash->AI()->SetData(i, 0);
307 if (Creature* pTrash = me->SummonCreature(NPC_ARGENT_MONK, SpawnPosition))
308 pTrash->AI()->SetData(i, 0);
309 if (Creature* pTrash = me->SummonCreature(NPC_PRIESTESS, SpawnPosition))
310 pTrash->AI()->SetData(i, 0);
311 }
312 }
313 }

315 void SetGrandChampionsForEncounter()
316 {
317 uiFirstBoss = urand(0, 4);
318
319 while (uiSecondBoss == uiFirstBoss || uiThirdBoss == uiFirstBoss || uiThirdBoss == uiSecondBoss)
320 {
321 uiSecondBoss = urand(0, 4);
322 uiThirdBoss = urand(0, 4);
323 }
324 }

326 void SetArgentChampion()
327 {
328 uint8 uiTempBoss = urand(0, 1);
329
330 switch (uiTempBoss)
331 {
332 case 0:
333 uiArgentChampion = NPC_EADRIC;
334 break;

337 break;
338 }

341 void StartEncounter()
342 {
343 me->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
344
345 if (instance->GetData(BOSS_BLACK_KNIGHT) == NOT_STARTED)
346 {
347 if (instance->GetData(BOSS_ARGENT_CHALLENGE_E) == NOT_STARTED && instance->GetData(BOSS_ARGENT_CHALLENGE_P) == NOT_STARTED)
348 {
349 if (instance->GetData(BOSS_GRAND_CHAMPIONS) == NOT_STARTED)
350 SetData(DATA_START, 0);
351
352 if (instance->GetData(BOSS_GRAND_CHAMPIONS) == DONE)
353 DoStartArgentChampionEncounter();
354 }
355
356 if ((instance->GetData(BOSS_GRAND_CHAMPIONS) == DONE &&
357 instance->GetData(BOSS_ARGENT_CHALLENGE_E) == DONE) ||
358 instance->GetData(BOSS_ARGENT_CHALLENGE_P) == DONE)
359 me->SummonCreature(VEHICLE_BLACK_KNIGHT, 769.834f, 651.915f, 447.035f, 0);
360 }
361 }

363 void AggroAllPlayers(Creature* temp)
364 {
365 Map::PlayerList const &PlList = me->GetMap()->GetPlayers();
366
367 if (PlList.isEmpty())
368 return;
369
370 for (Map::PlayerList::const_iterator i = PlList.begin(); i != PlList.end(); ++i)
371 {
372 if (Player* player = i->GetSource())
373 {
374 if (player->IsGameMaster())
375 continue;
376
377 if (player->IsAlive())
378 {
379 temp->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
380 temp->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
381 temp->SetReactState(REACT_AGGRESSIVE);
382 temp->SetInCombatWith(player);
383 player->SetInCombatWith(temp);
384 temp->AddThreat(player, 0.0f);
385 }
386 }
387 }

390 void UpdateAI(uint32 uiDiff) override
391 {
392 ScriptedAI::UpdateAI(uiDiff);
393
394 if (uiTimer <= uiDiff)
395 {
396 switch (uiPhase)
397 {
398 case 1:
399 DoSummonGrandChampion(uiSecondBoss);
400 NextStep(10000, true);
401 break;
402 case 2:
403 DoSummonGrandChampion(uiThirdBoss);
404 NextStep(0, false);
405 break;
406 case 3:
407 if (!Champion1List.empty())
408 {
409 for (GuidList::const_iterator itr = Champion1List.begin(); itr != Champion1List.end(); ++itr)
410 if (Creature* summon = ObjectAccessor::GetCreature(*me, *itr))
411 AggroAllPlayers(summon);
412 NextStep(0, false);
413 }
414 break;
415 }
416 } else uiTimer -= uiDiff;
417
418 if (!UpdateVictim())
419 return;
420 }

456 bool OnGossipHello(Player* player, Creature* creature) override
457 {
458 InstanceScript* instance = creature->GetInstanceScript();
459
460 if (instance &&
461 ((instance->GetData(BOSS_GRAND_CHAMPIONS) == DONE &&
462 instance->GetData(BOSS_BLACK_KNIGHT) == DONE &&
463 instance->GetData(BOSS_ARGENT_CHALLENGE_E) == DONE) ||
464 instance->GetData(BOSS_ARGENT_CHALLENGE_P) == DONE))
465 return false;
466
467 if (instance &&
468 instance->GetData(BOSS_GRAND_CHAMPIONS) == NOT_STARTED &&
469 instance->GetData(BOSS_ARGENT_CHALLENGE_E) == NOT_STARTED &&
470 instance->GetData(BOSS_ARGENT_CHALLENGE_P) == NOT_STARTED &&
471 instance->GetData(BOSS_BLACK_KNIGHT) == NOT_STARTED)
472 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_START_EVENT1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
473 else if (instance)
474 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_START_EVENT2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF+1);
475
476 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
477
478 return true;
479 }
480
481 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
482 {
483 player->PlayerTalkClass->ClearMenus();
484 if (action == GOSSIP_ACTION_INFO_DEF+1)
485 {
486 player->CLOSE_GOSSIP_MENU();
487 ENSURE_AI(npc_announcer_toc5::npc_announcer_toc5AI, creature->AI())->StartEncounter();
488 }
489
490 return true;
491 }

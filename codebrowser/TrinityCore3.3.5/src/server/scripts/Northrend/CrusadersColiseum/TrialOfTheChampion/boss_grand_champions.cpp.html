96void AggroAllPlayers(Creature* temp)
97{
98 Map::PlayerList const &PlList = temp->GetMap()->GetPlayers();
99
100 if (PlList.isEmpty())
101 return;
102
103 for (Map::PlayerList::const_iterator i = PlList.begin(); i != PlList.end(); ++i)
104 {
105 if (Player* player = i->GetSource())
106 {
107 if (player->IsGameMaster())
108 continue;
109
110 if (player->IsAlive())
111 {
112 temp->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_IMMUNE_TO_PC);
113 temp->SetReactState(REACT_AGGRESSIVE);
114 temp->SetInCombatWith(player);
115 player->SetInCombatWith(temp);
116 temp->AddThreat(player, 0.0f);
117 }
118 }
119 }

122bool GrandChampionsOutVehicle(Creature* me)
123{
124 InstanceScript* instance = me->GetInstanceScript();
125
126 if (!instance)
127 return false;
128
129 Creature* pGrandChampion1 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_GRAND_CHAMPION_1));
130 Creature* pGrandChampion2 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_GRAND_CHAMPION_2));
131 Creature* pGrandChampion3 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_GRAND_CHAMPION_3));
132
133 if (pGrandChampion1 && pGrandChampion2 && pGrandChampion3)
134 {
135 if (!pGrandChampion1->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT) &&
136 !pGrandChampion2->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT) &&
137 !pGrandChampion3->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT))
138 return true;
139 }
140
141 return false;
142}

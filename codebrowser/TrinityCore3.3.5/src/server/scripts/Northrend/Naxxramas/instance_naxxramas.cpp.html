103// from P2 teleport spell stored target
104float const HeiganPos[2] = { 2793.86f, -3707.38f };
105float const HeiganEruptionSlope[3] =
106{
107 (-3703.303223f - HeiganPos[1]) / (2777.494141f - HeiganPos[0]), // between right center and far right
108 (-3696.948242f - HeiganPos[1]) / (2785.624268f - HeiganPos[0]), // between left and right halves
109 (-3691.880615f - HeiganPos[1]) / (2790.280029f - HeiganPos[0]) // between far left and left center
110};
111
112// 0 H x
113// 1 ^
114// 2 |
115// 3 y<--o
116inline uint32 GetEruptionSection(float x, float y)
117{
118 y -= HeiganPos[1];
119 if (y < 1.0f)
120 return 0;
121
122 x -= HeiganPos[0];
123 if (x > -1.0f)
124 return 3;
125
126 float slope = y/x;
127 for (uint32 i = 0; i < 3; ++i)
128 if (slope > HeiganEruptionSlope[i])
129 return i;
130 return 3;
131}

224 void OnGameObjectCreate(GameObject* go) override
225 {
226 if (go->GetGOInfo()->displayId == 6785 || go->GetGOInfo()->displayId == 1287)
227 {

239 case GO_HORSEMEN_CHEST_HERO:

241 break;

245 case GO_KELTHUZAD_PORTAL02:
246 PortalsGUID[1] = go->GetGUID();
247 break;

249 PortalsGUID[2] = go->GetGUID();

298 sapphiron->AI()->DoAction(DATA_SAPPHIRON_BIRTH);
299 return;
300 }
301 break;
302 default:
303 break;
304 }
305
306 InstanceScript::OnGameObjectRemove(go);

309 void OnUnitDeath(Unit* unit) override
310 {
311 if (unit->GetTypeId() == TYPEID_PLAYER && IsEncounterInProgress())
312 {
313 playerDied = 1;
314 SaveToDB();
315 }
316
317 if (Creature* creature = unit->ToCreature())
318 if (creature->GetEntry() == NPC_BIGGLESWORTH)
319 {
320 // Loads Kel'Thuzad's grid. We need this as he must be active in order for his texts to work.
321 instance->LoadGrid(3749.67f, -5114.06f);
322 if (Creature* kelthuzad = instance->GetCreature(KelthuzadGUID))
323 kelthuzad->AI()->Talk(SAY_KELTHUZAD_CAT_DIED);
324 }

421 bool SetBossState(uint32 id, EncounterState state) override
422 {
423 if (!InstanceScript::SetBossState(id, state))
424 return false;
425
426 switch (id)
427 {
428 case BOSS_MAEXXNA:
429 if (state == DONE)
430 {
431 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_ARACHNID))
432 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
433
434 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
435 }
436 break;
437 case BOSS_LOATHEB:
438 if (state == DONE)
439 {
440 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_PLAGUE))
441 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
442
443 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
444 }
445 break;
446 case BOSS_THADDIUS:
447 if (state == DONE)
448 {
449 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_CONSTRUCT))
450 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
451
452 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
453 }
454 break;
455 case BOSS_GOTHIK:
456 if (state == DONE)
457 events.ScheduleEvent(EVENT_DIALOGUE_GOTHIK_KORTHAZZ, 10000);
458 break;
459 case BOSS_HORSEMEN:
460 if (state == DONE)
461 {
462 if (GameObject* horsemenChest = instance->GetGameObject(HorsemenChestGUID))
463 {
464 horsemenChest->SetRespawnTime(horsemenChest->GetRespawnDelay());
465 horsemenChest->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
466 }
467
468 if (GameObject* teleporter = GetGameObject(DATA_NAXX_PORTAL_MILITARY))
469 teleporter->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
470
471 events.ScheduleEvent(EVENT_KELTHUZAD_WING_TAUNT, 6000);
472 }
473 break;
474 case BOSS_SAPPHIRON:
475 if (state == DONE)
476 events.ScheduleEvent(EVENT_DIALOGUE_SAPPHIRON_KELTHUZAD, 6000);
477 break;
478 default:
479 break;
480 }
481
482 return true;
483 }

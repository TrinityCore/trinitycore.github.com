253// predicate function to select not charmed target
254struct NotCharmedTargetSelector : public std::unary_function<Unit*, bool>
255{
256 NotCharmedTargetSelector() { }
257
258 bool operator()(Unit const* target) const
259 {
260 return !target->IsCharmed();
261 }

269 struct boss_kelthuzadAI : public BossAI
270 {
271 boss_kelthuzadAI(Creature* creature) : BossAI(creature, BOSS_KELTHUZAD), spawns(creature)

277 void Initialize()
278 {
279 nGuardiansOfIcecrownCount = 0;
280
281 nAbomination = 0;
282 nWeaver = 0;

285 uint32 uiFaction;

287 uint8 nGuardiansOfIcecrownCount;

291 std::map<ObjectGuid, float> chained;
292
293 SummonList spawns; // adds spawn by the trigger. kept in separated list (i.e. not in summons)

295 void ResetPlayerScale()
296 {
297 std::map<ObjectGuid, float>::const_iterator itr;
298 for (itr = chained.begin(); itr != chained.end(); ++itr)
299 {
300 if (Player* charmed = ObjectAccessor::GetPlayer(*me, itr->first))
301 charmed->SetObjectScale(itr->second);
302 }
303
304 chained.clear();
305 }

307 void Reset() override
308 {
309 _Reset();
310
311 me->setFaction(35);
312 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE | UNIT_FLAG_NOT_SELECTABLE);
313
314 ResetPlayerScale();
315 spawns.DespawnAll();
316
317 instance->SetData(DATA_ABOMINATION_KILLED, 0);
318
319 if (GameObject* trigger = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_KELTHUZAD_TRIGGER)))
320 {
321 trigger->ResetDoorOrButton();
322 trigger->SetPhaseMask(1, true);
323 }
324
325 for (uint8 i = 0; i <= 3; ++i)
326 {
327 if (GameObject* portal = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_KELTHUZAD_PORTAL01 + i)))
328 if (!((portal->getLootState() == GO_READY) || (portal->getLootState() == GO_NOT_READY)))
329 portal->ResetDoorOrButton();
330 }
331
332 Initialize();
333 }

348 void EnterCombat(Unit* /*who*/) override
349 {
350 me->setFaction(uiFaction);
351 _EnterCombat();
352 for (uint8 i = 0; i <= 3; ++i)
353 {
354 if (GameObject* portal = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_KELTHUZAD_PORTAL01 + i)))
355 portal->ResetDoorOrButton();
356 }
357 DoCast(me, SPELL_KELTHUZAD_CHANNEL, false);
358 Talk(SAY_SUMMON_MINIONS);
359 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE | UNIT_FLAG_NOT_SELECTABLE);
360 me->SetFloatValue(UNIT_FIELD_COMBATREACH, 4);
361 me->SetFloatValue(UNIT_FIELD_BOUNDINGRADIUS, 4);
362 events.SetPhase(PHASE_ONE);
363 events.ScheduleEvent(EVENT_TRIGGER, 5000);
364 events.ScheduleEvent(EVENT_WASTE, 15000);
365 events.ScheduleEvent(EVENT_ABOMIN, 30000);
366 events.ScheduleEvent(EVENT_WEAVER, 50000);
367 events.ScheduleEvent(EVENT_PHASE, 228000);
368 }
369
370 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
371 {
372 if (events.IsInPhase(PHASE_TWO) && me->HealthBelowPctDamaged(45, damage))
373 {
374 Talk(SAY_REQUEST_AID);
375 events.SetPhase(PHASE_THREE);
376 events.ScheduleEvent(EVENT_ANSWER_REQUEST, 4000);
377
378 for (uint8 i = 0; i <= 3; ++i)
379 {
380 if (GameObject* portal = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_KELTHUZAD_PORTAL01 + i)))
381 if (portal->getLootState() == GO_READY)
382 portal->UseDoorOrButton();
383 }
384 }
385 }
386
387 void UpdateAI(uint32 diff) override
388 {
389 if (!UpdateVictim())
390 return;
391
392 events.Update(diff);
393
394 if (events.IsInPhase(PHASE_ONE))
395 {
396 while (uint32 eventId = events.ExecuteEvent())
397 {
398 switch (eventId)
399 {
400 case EVENT_WASTE:
401 DoSummon(NPC_WASTE, Pos[RAND(0, 3, 6, 9)]);
402 events.Repeat(2000, 5000);
403 break;
404 case EVENT_ABOMIN:
405 if (nAbomination < 8)
406 {
407 DoSummon(NPC_ABOMINATION, Pos[RAND(1, 4, 7, 10)]);
408 nAbomination++;
409 events.Repeat(20000);
410 }
411 break;
412 case EVENT_WEAVER:
413 if (nWeaver < 8)
414 {
415 DoSummon(NPC_WEAVER, Pos[RAND(0, 3, 6, 9)]);
416 nWeaver++;
417 events.Repeat(25000);
418 }
419 break;
420 case EVENT_TRIGGER:
421 if (GameObject* trigger = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_KELTHUZAD_TRIGGER)))
422 trigger->SetPhaseMask(2, true);
423 break;
424 case EVENT_PHASE:
425 events.Reset();
426 Talk(SAY_AGGRO);
427 Talk(EMOTE_PHASE_TWO);
428 spawns.DespawnAll();
429 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE | UNIT_FLAG_NOT_SELECTABLE);
430 me->CastStop();
431
432 DoStartMovement(me->GetVictim());
433 events.ScheduleEvent(EVENT_BOLT, urand(5000, 10000));
434 events.ScheduleEvent(EVENT_NOVA, 15000);
435 events.ScheduleEvent(EVENT_DETONATE, urand(30000, 40000));
436 events.ScheduleEvent(EVENT_FISSURE, urand(10000, 30000));
437 events.ScheduleEvent(EVENT_BLAST, urand(60000, 120000));
438 if (GetDifficulty() == RAID_DIFFICULTY_25MAN_NORMAL)
439 events.ScheduleEvent(EVENT_CHAIN, urand(30000, 60000));
440 events.SetPhase(PHASE_TWO);
441 break;
442 default:
443 break;
444 }
445 }
446 }
447 else
448 {
449 if (me->HasUnitState(UNIT_STATE_CASTING))
450 return;
451
452 if (uint32 eventId = events.ExecuteEvent())
453 {
454 switch (eventId)
455 {
456 case EVENT_BOLT:
457 DoCastVictim(SPELL_FROST_BOLT);
458 events.Repeat(5000, 10000);
459 break;
460 case EVENT_NOVA:
461 DoCastAOE(SPELL_FROST_BOLT_AOE);
462 events.Repeat(15000, 30000);
463 break;
464 case EVENT_CHAIN:
465 {
466 uint32 count = urand(1, 3);
467 for (uint8 i = 1; i <= count; i++)
468 {
469 Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 200, true);
470 if (target && !target->IsCharmed() && (chained.find(target->GetGUID()) == chained.end()))
471 {
472 DoCast(target, SPELL_CHAINS_OF_KELTHUZAD);
473 float scale = target->GetObjectScale();
474 chained.insert(std::make_pair(target->GetGUID(), scale));
475 target->SetObjectScale(scale * 2);
476 events.ScheduleEvent(EVENT_CHAINED_SPELL, 2000); //core has 2000ms to set unit flag charm
477 }
478 }
479 if (!chained.empty())
480 Talk(SAY_CHAIN);
481 events.Repeat(100000, 180000);
482 break;
483 }
484 case EVENT_CHAINED_SPELL:
485 {
486 std::map<ObjectGuid, float>::iterator itr;
487 for (itr = chained.begin(); itr != chained.end();)
488 {
489 if (Unit* player = ObjectAccessor::GetPlayer(*me, itr->first))
490 {
491 if (!player->IsCharmed())
492 {
493 player->SetObjectScale(itr->second);
494 std::map<ObjectGuid, float>::iterator next = itr;
495 ++next;
496 chained.erase(itr);
497 itr = next;
498 continue;
499 }
500
501 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0, NotCharmedTargetSelector()))
502 {
503 switch (player->getClass())
504 {
505 case CLASS_DRUID:
506 if (urand(0, 1))
507 player->CastSpell(target, SPELL_MOONFIRE, false);
508 else
509 player->CastSpell(me, SPELL_LIFEBLOOM, false);
510 break;
511 case CLASS_HUNTER:
512 player->CastSpell(target, RAND(SPELL_MULTI_SHOT, SPELL_VOLLEY), false);
513 break;
514 case CLASS_MAGE:
515 player->CastSpell(target, RAND(SPELL_FROST_FIREBOLT, SPELL_ARCANE_MISSILES), false);
516 break;
517 case CLASS_WARLOCK:
518 player->CastSpell(target, RAND(SPELL_CURSE_OF_AGONY, SPELL_SHADOW_BOLT), true);
519 break;
520 case CLASS_WARRIOR:
521 player->CastSpell(target, RAND(SPELL_BLADESTORM, SPELL_CLEAVE), false);
522 break;
523 case CLASS_PALADIN:
524 if (urand(0, 1))
525 player->CastSpell(target, SPELL_HAMMER_OF_JUSTICE, false);
526 else
527 player->CastSpell(me, SPELL_HOLY_SHOCK, false);
528 break;
529 case CLASS_PRIEST:
530 if (urand(0, 1))
531 player->CastSpell(target, SPELL_VAMPIRIC_TOUCH, false);
532 else
533 player->CastSpell(me, SPELL_RENEW, false);
534 break;
535 case CLASS_SHAMAN:
536 if (urand(0, 1))
537 player->CastSpell(target, SPELL_EARTH_SHOCK, false);
538 else
539 player->CastSpell(me, SPELL_HEALING_WAVE, false);
540 break;
541 case CLASS_ROGUE:
542 player->CastSpell(target, RAND(SPELL_HEMORRHAGE, SPELL_MUTILATE), false);
543 break;
544 case CLASS_DEATH_KNIGHT:
545 if (urand(0, 1))
546 player->CastSpell(target, SPELL_PLAGUE_STRIKE, true);
547 else
548 player->CastSpell(target, SPELL_HOWLING_BLAST, true);
549 break;
550 }
551 }
552 }
553 ++itr;
554 }
555
556 if (!chained.empty())
557 events.Repeat(5000);
558
559 break;
560 }
561 case EVENT_DETONATE:
562 {
563 std::vector<Unit*> unitList;
564 ThreatContainer::StorageType const &threatList = me->getThreatManager().getThreatList();
565 for (ThreatContainer::StorageType::const_iterator itr = threatList.begin(); itr != threatList.end(); ++itr)
566 {
567 Unit* const target = (*itr)->getTarget();
568
569 if (target->GetTypeId() == TYPEID_PLAYER
570 && target->getPowerType() == POWER_MANA
571 && target->GetPower(POWER_MANA))
572 {
573 unitList.push_back(target);
574 }
575 }
576
577 if (!unitList.empty())
578 {
579 std::vector<Unit*>::const_iterator itr = unitList.begin();
580 advance(itr, rand32() % unitList.size());
581 DoCast(*itr, SPELL_MANA_DETONATION);
582 Talk(SAY_SPECIAL);
583 }
584
585 events.Repeat(20000, 50000);
586 break;
587 }
588 case EVENT_FISSURE:
589 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
590 DoCast(target, SPELL_SHADOW_FISURE);
591 events.Repeat(10000, 45000);
592 break;
593 case EVENT_BLAST:
594 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, RAID_MODE(1, 0), 0, true))
595 DoCast(target, SPELL_FROST_BLAST);
596 if (rand32() % 2)
597 Talk(SAY_FROST_BLAST);
598 events.Repeat(30000, 90000);
599 break;
600 case EVENT_ANSWER_REQUEST:
601 if (Creature* lichKing = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_LICH_KING)))
602 lichKing->AI()->Talk(SAY_ANSWER_REQUEST);
603 events.ScheduleEvent(EVENT_SUMMON_GUARDIANS, 5000);
604 break;
605 case EVENT_SUMMON_GUARDIANS:

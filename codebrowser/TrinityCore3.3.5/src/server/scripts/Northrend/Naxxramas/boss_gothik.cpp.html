129/* Room side checking logic */
130static AreaBoundary* const livingSide = new RectangleBoundary(2633.84f, 2750.49f, -3434.0f, -3360.78f);
131static AreaBoundary* const deadSide = new RectangleBoundary(2633.84f, 2750.49f, -3360.78f, -3285.0f);
132enum Side
133{
134 SIDE_NONE = 0,
135 SIDE_LIVING,
136 SIDE_DEAD
137};
138inline static Side GetSide(Position const* who)
139{
140 if (livingSide->IsWithinBoundary(who))
141 return SIDE_LIVING;
142 if (deadSide->IsWithinBoundary(who))
143 return SIDE_DEAD;
144 return SIDE_NONE;
145}
146inline static bool IsOnSameSide(Position const* who, Position const* other)
147{
148 return (GetSide(who) == GetSide(other));
149}
150static Player* FindEligibleTarget(Creature const* me, bool isGateOpen)
151{
152 Map::PlayerList const& players = me->GetMap()->GetPlayers();
153 for (Map::PlayerList::const_iterator it = players.begin(); it != players.end(); ++it)
154 {
155 Player* player = it->GetSource();
156 if (player && (isGateOpen || IsOnSameSide(me, player)) && me->CanSeeOrDetect(player) && me->IsValidAttackTarget(player) && player->isInAccessiblePlaceFor(me))
157 {
158 return player;
159 }
160 }
161
162 return nullptr;
163}

302 boss_gothikAI(Creature* creature) : BossAI(creature, BOSS_GOTHIK)

306
307 void Initialize()
308 {
309 _waveCount = 0;
310 _gateCanOpen = false;
311 _gateIsOpen = true;
312 _lastTeleportDead = false;
313 }
314
315 void Reset() override
316 {
317 me->SetReactState(REACT_PASSIVE);
318 instance->SetData(DATA_GOTHIK_GATE, GO_STATE_ACTIVE);
319 _Reset();
320 Initialize();
321 }
322
323 void EnterCombat(Unit* /*who*/) override
324 {
325 _EnterCombat();
326 events.SetPhase(PHASE_ONE);
327 events.ScheduleEvent(EVENT_SUMMON, 25 * IN_MILLISECONDS, 0, PHASE_ONE);
328 events.ScheduleEvent(EVENT_DOORS_UNLOCK, 205 * IN_MILLISECONDS, 0, PHASE_ONE);
329 events.ScheduleEvent(EVENT_PHASE_TWO, 270 * IN_MILLISECONDS, 0, PHASE_ONE);
330 Talk(SAY_INTRO_1);
331 events.ScheduleEvent(EVENT_INTRO_2, 4 * IN_MILLISECONDS);
332 events.ScheduleEvent(EVENT_INTRO_3, 9 * IN_MILLISECONDS);
333 events.ScheduleEvent(EVENT_INTRO_4, 14 * IN_MILLISECONDS);
334 instance->SetData(DATA_GOTHIK_GATE, GO_STATE_READY);
335 _gateIsOpen = false;
336 }

338 void JustSummoned(Creature* summon) override
339 {
340 summons.Summon(summon);
341 if (me->IsInCombat())
342 {
343 summon->AI()->DoAction(_gateIsOpen ? ACTION_GATE_OPENED : ACTION_ACQUIRE_TARGET);
344 summon->SetCombatPulseDelay(5);

346 else

368
369 void OpenGate()
370 {
371 if (_gateIsOpen)
372 return;
373 instance->SetData(DATA_GOTHIK_GATE, GO_STATE_ACTIVE);
374 Talk(EMOTE_GATE_OPENED);
375 _gateIsOpen = true;
376
377 for (ObjectGuid summonGuid : summons)
378 {
379 if (Creature* summon = ObjectAccessor::GetCreature(*me, summonGuid))
380 summon->AI()->DoAction(ACTION_GATE_OPENED);
381 if (summons.empty()) // ACTION_GATE_OPENED may cause an evade, despawning summons and invalidating our iterator
382 break;
383 }
384 }
385
386 void DamageTaken(Unit* /*who*/, uint32& damage) override
387 {
388 if (!events.IsInPhase(PHASE_TWO))
389 damage = 0;
390 }
391
392 void DoAction(int32 action) override
393 {
394 switch (action)
395 {
396 case ACTION_MINION_EVADE:
397 if (_gateIsOpen || me->getThreatManager().isThreatListEmpty())
398 return EnterEvadeMode(EVADE_REASON_NO_HOSTILES);
399 if (_gateCanOpen)
400 OpenGate();
401 break;
402 }
403 }
404
405 void EnterEvadeMode(EvadeReason why) override
406 {
407 BossAI::EnterEvadeMode(why);
408 Position const& home = me->GetHomePosition();
409 me->NearTeleportTo(home.GetPositionX(), home.GetPositionY(), home.GetPositionZ(), home.GetOrientation());
410 }
411
412 void UpdateAI(uint32 diff) override
413 {
414 if (!UpdateVictim())
415 return;
416
417 if (me->HasReactState(REACT_AGGRESSIVE) && !_gateIsOpen && !IsOnSameSide(me, me->GetVictim()))
418 {
419 // NBD: this should only happen in practice if there is nobody left alive on our side (we should open gate)
420 // thus we only do a cursory check to make sure (edge cases?)
421 if (Player* newTarget = FindEligibleTarget(me, _gateIsOpen))
422 {
423 me->getThreatManager().resetAllAggro();
424 me->AddThreat(newTarget, 1.0f);
425 AttackStart(newTarget);
426 }
427 else
428 OpenGate();
429 }
430
431 events.Update(diff);
432
433 if (!_gateIsOpen && HealthBelowPct(30) && events.IsInPhase(PHASE_TWO))
434 OpenGate();
435
436 while (uint32 eventId = events.ExecuteEvent())
437 {
438 switch (eventId)
439 {
440 case EVENT_SUMMON:
441 {
442 if (RAID_MODE(waves10,waves25).size() <= _waveCount) // bounds check
443 {
444 TC_LOG_INFO("scripts", "GothikAI: Wave count %d is out of range for difficulty %d.", _waveCount, GetDifficulty());
445 break;
446 }
447
448 std::list<Creature*> triggers;
449 me->GetCreatureListWithEntryInGrid(triggers, NPC_TRIGGER, 150.0f);
450 for (GothikWaveEntry entry : RAID_MODE(waves10, waves25)[_waveCount].first)
451 for (uint8 i = 0; i < entry.second; ++i)
452 {
453 // GUID layout is as follows:
454 // CGUID+4: center (back of platform) - primary rider spawn
455 // CGUID+5: north (back of platform) - primary knight spawn
456 // CGUID+6: center (front of platform) - second spawn
457 // CGUID+7: south (front of platform) - primary trainee spawn
458 uint32 targetDBGuid;
459 switch (entry.first)
460 {
461 case NPC_LIVE_RIDER: // only spawns from center (back) > north
462 targetDBGuid = (CGUID_TRIGGER + 4) + (i % 2);
463 break;
464 case NPC_LIVE_KNIGHT: // spawns north > center (front) > south
465 targetDBGuid = (CGUID_TRIGGER + 5) + (i % 3);
466 break;
467 case NPC_LIVE_TRAINEE: // spawns south > center (front) > north
468 targetDBGuid = (CGUID_TRIGGER + 7) - (i % 3);
469 break;
470 default:
471 targetDBGuid = 0;
472 }
473
474 for (Creature* trigger : triggers)
475 if (trigger && trigger->GetSpawnId() == targetDBGuid)
476 {
477 DoSummon(entry.first, trigger, 1.0f, 15 * IN_MILLISECONDS, TEMPSUMMON_CORPSE_TIMED_DESPAWN);
478 break;
479 }
480 }
481
482 if (uint8 timeToNext = RAID_MODE(waves10, waves25)[_waveCount].second)
483 events.ScheduleEvent(EVENT_SUMMON, timeToNext * IN_MILLISECONDS, 0, PHASE_ONE);
484
485 ++_waveCount;
486 break;
487 }
488 case EVENT_DOORS_UNLOCK:
489 _gateCanOpen = true;
490 for (ObjectGuid summonGuid : summons)
491 if (Creature* summon = ObjectAccessor::GetCreature(*me, summonGuid))
492 if (summon->IsAlive() && (!summon->IsInCombat() || summon->IsInEvadeMode()))
493 {
494 OpenGate();
495 break;
496 }
497 break;
498 case EVENT_PHASE_TWO:
499 events.SetPhase(PHASE_TWO);
500 events.ScheduleEvent(EVENT_TELEPORT, 20 * IN_MILLISECONDS, 0, PHASE_TWO);
501 events.ScheduleEvent(EVENT_HARVEST, 15 * IN_MILLISECONDS, 0, PHASE_TWO);
502 events.ScheduleEvent(EVENT_RESUME_ATTACK, 2 * IN_MILLISECONDS, 0, PHASE_TWO);
503 Talk(SAY_PHASE_TWO);
504 Talk(EMOTE_PHASE_TWO);
505 me->SetReactState(REACT_PASSIVE);
506 me->getThreatManager().resetAllAggro();
507 DoCastAOE(SPELL_TELEPORT_LIVE);
508 break;
509 case EVENT_TELEPORT:
510 if (!HealthBelowPct(30))
511 {
512 me->CastStop();
513 me->AttackStop();
514 me->StopMoving();
515 me->SetReactState(REACT_PASSIVE);
516 me->getThreatManager().resetAllAggro();
517 DoCastAOE(_lastTeleportDead ? SPELL_TELEPORT_LIVE : SPELL_TELEPORT_DEAD);
518 _lastTeleportDead = !_lastTeleportDead;
519
520 events.CancelEvent(EVENT_BOLT);
521 events.ScheduleEvent(EVENT_TELEPORT, 20 * IN_MILLISECONDS, 0, PHASE_TWO);
522 events.ScheduleEvent(EVENT_RESUME_ATTACK, 2 * IN_MILLISECONDS, 0, PHASE_TWO);
523 }
524 break;
525
526 case EVENT_HARVEST:
527 DoCastAOE(SPELL_HARVEST_SOUL, true); // triggered allows this to go "through" shadow bolt
528 events.ScheduleEvent(EVENT_HARVEST, 15 * IN_MILLISECONDS, 0, PHASE_TWO);
529 break;
530 case EVENT_RESUME_ATTACK:
531 me->SetReactState(REACT_AGGRESSIVE);
532 events.ScheduleEvent(EVENT_BOLT, 0, 0, PHASE_TWO);
533 // return to the start of this method so victim side etc is re-evaluated
534 return UpdateAI(0u); // tail recursion for efficiency
535 case EVENT_BOLT:
536 DoCastVictim(SPELL_SHADOW_BOLT);
537 events.ScheduleEvent(EVENT_BOLT, 1 * IN_MILLISECONDS, 0, PHASE_TWO);
538 break;
539 case EVENT_INTRO_2:
540 Talk(SAY_INTRO_2);
541 break;
542 case EVENT_INTRO_3:
543 Talk(SAY_INTRO_3);
544 break;
545 case EVENT_INTRO_4:
546 Talk(SAY_INTRO_4);
547 break;
548 }
549 }
550 }
551
552 private:
553 uint32 _waveCount;
554 bool _gateCanOpen;
555 bool _gateIsOpen;
556 bool _lastTeleportDead;
557 };
558
559 CreatureAI* GetAI(Creature* creature) const override
560 {
561 return GetInstanceAI<boss_gothikAI>(creature);
562 }

565struct npc_gothik_minion_baseAI : public ScriptedAI
566{
567 public:
568 npc_gothik_minion_baseAI(Creature* creature, uint32 deathNotify=0) : ScriptedAI(creature), _deathNotify(deathNotify), _gateIsOpen(false) { }
569
570 void JustDied(Unit* /*killer*/) override
571 {
572 if (_deathNotify)
573 DoCastAOE(_deathNotify, true);
574 }
575
576 inline bool isOnSameSide(Unit const* who) const
577 {
578 return IsOnSameSide(me, who);
579 }
580
581 void DamageTaken(Unit* attacker, uint32 &damage) override
582 { // do not allow minions to take damage before the gate is opened
583 if (!_gateIsOpen && !isOnSameSide(attacker))
584 damage = 0;
585 }
586
587 void DoAction(int32 action) override
588 {
589 switch (action)
590 {
591 case ACTION_GATE_OPENED:
592 _gateIsOpen = true;
593 // intentional missing break
594 case ACTION_ACQUIRE_TARGET:
595 if (Player* target = FindEligibleTarget(me, _gateIsOpen))
596 {
597 me->AddThreat(target, 1.0f);
598 AttackStart(target);
599 }
600 else
601 EnterEvadeMode(EVADE_REASON_NO_HOSTILES);
602 break;
603 }
604 }
605
606 void EnterEvadeMode(EvadeReason why) override
607 {
608 ScriptedAI::EnterEvadeMode(why);
609
610 if (InstanceScript* instance = me->GetInstanceScript())
611 if (Creature* gothik = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_GOTHIK)))
612 gothik->AI()->DoAction(ACTION_MINION_EVADE);
613 }
614
615 void UpdateAI(uint32 diff) override
616 {
617 if (!UpdateVictim())
618 return;
619
620 if (!_gateIsOpen && !isOnSameSide(me->GetVictim()))
621 { // reset threat, then try to find someone on same side as us to attack
622 if (Player* newTarget = FindEligibleTarget(me, _gateIsOpen))
623 {
624 me->RemoveAurasByType(SPELL_AURA_MOD_TAUNT);
625 me->getThreatManager().resetAllAggro();
626 me->AddThreat(newTarget, 1.0f);
627 AttackStart(newTarget);
628 }
629 else
630 EnterEvadeMode(EVADE_REASON_NO_HOSTILES);
631 }
632
633 _UpdateAI(diff);
634 }
635
636 virtual void _UpdateAI(uint32 diff) { ScriptedAI::UpdateAI(diff); };
637
638 private:
639 uint32 _deathNotify;
640 bool _gateIsOpen;
641};

789class npc_gothik_minion_spectralrider : public CreatureScript
790{
791 public:
792 npc_gothik_minion_spectralrider() : CreatureScript("npc_gothik_minion_spectralrider") { }
793
794 struct npc_gothik_minion_spectralriderAI : public npc_gothik_minion_baseAI
795 {
796 npc_gothik_minion_spectralriderAI(Creature* creature) : npc_gothik_minion_baseAI(creature), _frenzyTimer(urandms(2,5)), _drainTimer(urandms(8,12)) { }
797
798 void _UpdateAI(uint32 diff)
799 {
800 if (diff < _frenzyTimer)
801 _frenzyTimer -= diff;
802 else if (me->HasUnitState(UNIT_STATE_CASTING))
803 _frenzyTimer = 0;
804 else
805 { // target priority: knight > other rider > horse > gothik
806 std::list<Creature*> potentialTargets = DoFindFriendlyMissingBuff(30.0, SPELLHELPER_UNHOLY_FRENZY);
807 Creature *knightTarget = nullptr, *riderTarget = nullptr, *horseTarget = nullptr, *gothikTarget = nullptr;
808 for (Creature* pTarget : potentialTargets)
809 {
810 switch (pTarget->GetEntry())
811 {
812 case NPC_DEAD_KNIGHT:
813 knightTarget = pTarget;
814 break;
815 case NPC_DEAD_RIDER:
816 riderTarget = pTarget;
817 break;
818 case NPC_DEAD_HORSE:
819 horseTarget = pTarget;
820 break;
821 case NPC_GOTHIK:
822 gothikTarget = pTarget;
823 break;
824 }
825 if (knightTarget)
826 break;
827 }
828 Creature* target = knightTarget ? knightTarget : riderTarget ? riderTarget : horseTarget ? horseTarget : gothikTarget ? gothikTarget : nullptr;
829 if (target)
830 DoCast(target, SPELL_UNHOLY_FRENZY);
831 _frenzyTimer = 20 * IN_MILLISECONDS;
832 }
833
834 if (diff < _drainTimer)
835 _drainTimer -= diff;
836 else
837 {
838 DoCastVictim(SPELL_DRAIN_LIFE);
839 _drainTimer = urandms(10,15);
840 }
841
842 if (!me->HasUnitState(UNIT_STATE_CASTING))
843 DoMeleeAttackIfReady();
844 }
845 uint32 _frenzyTimer, _drainTimer;
846 };
847
848 CreatureAI* GetAI(Creature* creature) const override
849 {
850 return GetInstanceAI<npc_gothik_minion_spectralriderAI>(creature);
851 }

938 if (Creature* target = SelectRandomSkullPile())
939 DoCast(target, SPELL_SKULLS_RIDER, true);
940 break;
941 case SPELL_SKULLS_TRAINEE:
942 DoSummon(NPC_DEAD_TRAINEE, me, 0.0f, 15 * IN_MILLISECONDS, TEMPSUMMON_CORPSE_TIMED_DESPAWN);

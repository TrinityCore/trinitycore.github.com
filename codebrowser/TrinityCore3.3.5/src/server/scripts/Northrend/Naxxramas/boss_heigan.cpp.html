63class boss_heigan : public CreatureScript
64{
65public:
66 boss_heigan() : CreatureScript("boss_heigan") { }
67
68 CreatureAI* GetAI(Creature* creature) const override
69 {
70 return GetInstanceAI<boss_heiganAI>(creature);
71 }
72
73 struct boss_heiganAI : public BossAI
74 {
75 boss_heiganAI(Creature* creature) : BossAI(creature, BOSS_HEIGAN), eruptSection(0), eruptDirection(false), safetyDance(false) { }
76
77 void Reset() override
78 {
79 me->SetReactState(REACT_AGGRESSIVE);
80 _Reset();
81 }
82
83 void KilledUnit(Unit* who) override
84 {
85 Talk(SAY_SLAY);
86
87 if (who->GetTypeId() == TYPEID_PLAYER)
88 safetyDance = false;
89 }
90
91 uint32 GetData(uint32 type) const override
92 {
93 return (type == DATA_SAFETY_DANCE && safetyDance) ? 1u : 0u;
94 }
95
96 void JustDied(Unit* /*killer*/) override
97 {
98 _JustDied();
99 Talk(SAY_DEATH);
100 }
101
102 void EnterCombat(Unit* /*who*/) override
103 {
104 _EnterCombat();
105 Talk(SAY_AGGRO);
106
107 eruptSection = 3;
108 events.ScheduleEvent(EVENT_DISRUPT, urand(15 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, PHASE_FIGHT);
109 events.ScheduleEvent(EVENT_FEVER, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, PHASE_FIGHT);
110 events.ScheduleEvent(EVENT_DANCE, 90 * IN_MILLISECONDS, 0, PHASE_FIGHT);
111 events.ScheduleEvent(EVENT_ERUPT, 15 * IN_MILLISECONDS, 0, PHASE_FIGHT);
112
113 safetyDance = true;
114 }
115
116 void UpdateAI(uint32 diff) override
117 {
118 if (!UpdateVictim())
119 return;
120
121 events.Update(diff);
122
123 while (uint32 eventId = events.ExecuteEvent())
124 {
125 switch (eventId)
126 {
127 case EVENT_DISRUPT:
128 DoCastAOE(SPELL_SPELL_DISRUPTION);
129 events.ScheduleEvent(EVENT_DISRUPT, 11 * IN_MILLISECONDS);
130 break;
131 case EVENT_FEVER:
132 DoCastAOE(SPELL_DECREPIT_FEVER);
133 events.ScheduleEvent(EVENT_FEVER, urand(20 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
134 break;
135 case EVENT_DANCE:
136 events.SetPhase(PHASE_DANCE);
137 Talk(SAY_TAUNT);
138 Talk(EMOTE_DANCE);
139 eruptSection = 3;
140 me->SetReactState(REACT_PASSIVE);
141 me->AttackStop();
142 me->StopMoving();
143 DoCast(SPELL_TELEPORT_SELF);
144 DoCastAOE(SPELL_PLAGUE_CLOUD);
145 events.ScheduleEvent(EVENT_DANCE_END, 45 * IN_MILLISECONDS, 0, PHASE_DANCE);
146 events.ScheduleEvent(EVENT_ERUPT, 10 * IN_MILLISECONDS);
147 break;
148 case EVENT_DANCE_END:
149 events.SetPhase(PHASE_FIGHT);
150 Talk(EMOTE_DANCE_END);
151 eruptSection = 3;
152 events.ScheduleEvent(EVENT_DISRUPT, urand(10, 25) * IN_MILLISECONDS, 0, PHASE_FIGHT);
153 events.ScheduleEvent(EVENT_FEVER, urand(15, 20) * IN_MILLISECONDS, 0, PHASE_FIGHT);
154 events.ScheduleEvent(EVENT_DANCE, 90 * IN_MILLISECONDS, 0, PHASE_FIGHT);
155 events.ScheduleEvent(EVENT_ERUPT, 15 * IN_MILLISECONDS, 0, PHASE_FIGHT);
156 me->CastStop();
157 me->SetReactState(REACT_AGGRESSIVE);
158 DoZoneInCombat();
159 break;
160 case EVENT_ERUPT:
161 instance->SetData(DATA_HEIGAN_ERUPT, eruptSection);
162 TeleportCheaters();
163
164 if (eruptSection == 0)
165 eruptDirection = true;
166 else if (eruptSection == 3)
167 eruptDirection = false;
168
169 eruptDirection ? ++eruptSection : --eruptSection;
170
171 if (events.IsInPhase(PHASE_DANCE))
172 events.ScheduleEvent(EVENT_ERUPT, 3 * IN_MILLISECONDS, 0, PHASE_DANCE);
173 else
174 events.ScheduleEvent(EVENT_ERUPT, 10 * IN_MILLISECONDS, 0, PHASE_FIGHT);
175 break;
176 }
177 }
178
179 DoMeleeAttackIfReady();
180 }
181
182 private:
183 uint32 eruptSection;
184 bool eruptDirection;
185
186 bool safetyDance; // is achievement still possible? (= no player deaths yet)
187 };
188
189};

191class spell_heigan_eruption : public SpellScriptLoader
192{
193 public:
194 spell_heigan_eruption() : SpellScriptLoader("spell_heigan_eruption") { }
195
196 class spell_heigan_eruption_SpellScript : public SpellScript
197 {
198 PrepareSpellScript(spell_heigan_eruption_SpellScript);
199
200 void HandleScript(SpellEffIndex /*eff*/)
201 {
202 Unit* caster = GetCaster();
203 if (!caster || !GetHitPlayer())
204 return;
205
206 if (GetHitDamage() >= int32(GetHitPlayer()->GetHealth()))
207 if (InstanceScript* instance = caster->GetInstanceScript())
208 if (Creature* Heigan = ObjectAccessor::GetCreature(*caster, instance->GetGuidData(DATA_HEIGAN)))
209 Heigan->AI()->KilledUnit(GetHitPlayer());
210 }
211
212 void Register() override
213 {
214 OnEffectHitTarget += SpellEffectFn(spell_heigan_eruption_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCHOOL_DAMAGE);
215 }
216 };
217
218 SpellScript* GetSpellScript() const override
219 {
220 return new spell_heigan_eruption_SpellScript();
221 }
222};

224class achievement_safety_dance : public AchievementCriteriaScript
225{
226 public:
227 achievement_safety_dance() : AchievementCriteriaScript("achievement_safety_dance") { }
228
229 bool OnCheck(Player* /*player*/, Unit* target) override
230 {
231 if (!target)
232 return false;
233
234 if (Creature* Heigan = target->ToCreature())
235 if (Heigan->AI()->GetData(DATA_SAFETY_DANCE))
236 return true;
237
238 return false;
239 }
240};

92 struct boss_sapphironAI : public BossAI
93 {
94 boss_sapphironAI(Creature* creature) :
95 BossAI(creature, BOSS_SAPPHIRON), _iceboltCount(0), _map(me->GetMap())

100 void Initialize()
101 {
102 _phase = PHASE_NULL;
103
104 _canTheHundredClub = true;
105 }
106
107 void InitializeAI() override
108 {
109 _canTheHundredClub = true;
110
111 float x, y, z;
112 me->GetPosition(x, y, z);
113 me->SummonGameObject(GO_BIRTH, x, y, z, 0, 0, 0, 0, 0, 0);
114 me->SetVisible(false);
115 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
116 me->SetReactState(REACT_PASSIVE);
117
118 BossAI::InitializeAI();
119 }
120
121 void Reset() override
122 {
123 _Reset();
124
125 if (_phase == PHASE_FLIGHT)
126 {
127 ClearIceBlock();
128 me->SetReactState(REACT_AGGRESSIVE);
129 if (me->IsHovering())
130 {
131 me->HandleEmoteCommand(EMOTE_ONESHOT_LAND);
132 me->SetHover(false);
133 }
134 me->SetDisableGravity(false);
135 }
136
137 Initialize();
138 }
139
140 void EnterCombat(Unit* /*who*/) override
141 {
142 _EnterCombat();
143
144 me->CastSpell(me, SPELL_FROST_AURA, true);
145
146 DoCast(me, SPELL_CHECK_RESISTS);
147 events.ScheduleEvent(EVENT_CHECK_RESISTS, 30 * IN_MILLISECONDS);
148 events.ScheduleEvent(EVENT_BERSERK, 15 * MINUTE * IN_MILLISECONDS);
149 EnterPhaseGround();
150 }
151
152 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
153 {
154 switch(spell->Id)
155 {
156 case SPELL_ICEBOLT:
157 {
158 IceBlockMap::iterator itr = _iceblocks.find(target->GetGUID());
159 if (itr != _iceblocks.end() && !itr->second)
160 {
161 if (GameObject* iceblock = me->SummonGameObject(GO_ICEBLOCK, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, 0, 0, 0, 0, 25))
162 itr->second = iceblock->GetGUID();
163 }
164 break;
165 }
166 case SPELL_CHECK_RESISTS:
167 if (target && target->GetResistance(SPELL_SCHOOL_FROST) > MAX_FROST_RESISTANCE)
168 _canTheHundredClub = false;
169 break;
170 }
171 }
172
173 void JustDied(Unit* /*killer*/) override
174 {
175 _JustDied();
176 me->CastSpell(me, SPELL_DIES, true);
177 }
178
179 void MovementInform(uint32 /*type*/, uint32 id) override
180 {
181 if (id == 1)
182 events.ScheduleEvent(EVENT_LIFTOFF, 0);
183 }
184
185 void DoAction(int32 param) override
186 {
187 if (param == DATA_SAPPHIRON_BIRTH)
188 {
189 _phase = PHASE_BIRTH;
190 events.ScheduleEvent(EVENT_BIRTH, 23 * IN_MILLISECONDS);
191 }
192 }
193
194 void EnterPhaseGround()
195 {
196 _phase = PHASE_GROUND;
197 me->SetReactState(REACT_AGGRESSIVE);
198 events.SetPhase(PHASE_GROUND);
199 events.ScheduleEvent(EVENT_CLEAVE, urand(5, 15) * IN_MILLISECONDS, 0, PHASE_GROUND);
200 events.ScheduleEvent(EVENT_TAIL, urand(5, 15) * IN_MILLISECONDS, 0, PHASE_GROUND);
201 events.ScheduleEvent(EVENT_DRAIN, 24 * IN_MILLISECONDS, 0, PHASE_GROUND);
202 events.ScheduleEvent(EVENT_BLIZZARD, urand(5, 10) * IN_MILLISECONDS, 0, PHASE_GROUND);
203 events.ScheduleEvent(EVENT_FLIGHT, 45 * IN_MILLISECONDS);
204 }
205
206 void ClearIceBlock()
207 {
208 for (IceBlockMap::const_iterator itr = _iceblocks.begin(); itr != _iceblocks.end(); ++itr)
209 {
210 if (Player* player = ObjectAccessor::GetPlayer(*me, itr->first))
211 player->RemoveAura(SPELL_ICEBOLT);
212
213 if (GameObject* go = ObjectAccessor::GetGameObject(*me, itr->second))
214 go->Delete();
215 }
216 _iceblocks.clear();
217 }
218
219 uint32 GetData(uint32 data) const override
220 {
221 if (data == DATA_THE_HUNDRED_CLUB)
222 return _canTheHundredClub;
223
224 return 0;
225 }

227 void UpdateAI(uint32 diff) override
228 {
229 if (!_phase)
230 return;
231
232 events.Update(diff);
233
234 if (_phase != PHASE_BIRTH && !UpdateVictim())
235 return;
236
237 if (_phase == PHASE_GROUND)
238 {
239 while (uint32 eventId = events.ExecuteEvent())
240 {
241 switch (eventId)
242 {
243 case EVENT_CHECK_RESISTS:
244 DoCast(me, SPELL_CHECK_RESISTS);
245 events.ScheduleEvent(EVENT_CHECK_RESISTS, 30 * IN_MILLISECONDS);
246 return;
247 case EVENT_BERSERK:
248 Talk(EMOTE_ENRAGE);
249 DoCast(me, SPELL_BERSERK);
250 return;
251 case EVENT_CLEAVE:
252 DoCastVictim(SPELL_CLEAVE);
253 events.ScheduleEvent(EVENT_CLEAVE, urand(5, 15) * IN_MILLISECONDS, 0, PHASE_GROUND);
254 return;
255 case EVENT_TAIL:
256 DoCastAOE(SPELL_TAIL_SWEEP);
257 events.ScheduleEvent(EVENT_TAIL, urand(5, 15) * IN_MILLISECONDS, 0, PHASE_GROUND);
258 return;
259 case EVENT_DRAIN:
260 DoCastAOE(SPELL_LIFE_DRAIN);
261 events.ScheduleEvent(EVENT_DRAIN, 24 * IN_MILLISECONDS, 0, PHASE_GROUND);
262 return;
263 case EVENT_BLIZZARD:
264 {
265 if (Creature* summon = DoSummon(NPC_BLIZZARD, me, 0.0f, urand(25, 30) * IN_MILLISECONDS, TEMPSUMMON_TIMED_DESPAWN))
266 summon->GetMotionMaster()->MoveRandom(40);
267 events.ScheduleEvent(EVENT_BLIZZARD, RAID_MODE(20, 7) * IN_MILLISECONDS, 0, PHASE_GROUND);
268 break;
269 }
270 case EVENT_FLIGHT:
271 if (HealthAbovePct(10))
272 {
273 _phase = PHASE_FLIGHT;
274 events.SetPhase(PHASE_FLIGHT);
275 me->SetReactState(REACT_PASSIVE);
276 me->AttackStop();
277 float x, y, z, o;
278 me->GetHomePosition(x, y, z, o);
279 me->GetMotionMaster()->MovePoint(1, x, y, z);
280 return;
281 }
282 break;
283 }
284 }
285
286 DoMeleeAttackIfReady();
287 }
288 else
289 {
290 if (uint32 eventId = events.ExecuteEvent())
291 {
292 switch (eventId)
293 {
294 case EVENT_CHECK_RESISTS:
295 DoCast(me, SPELL_CHECK_RESISTS);
296 events.ScheduleEvent(EVENT_CHECK_RESISTS, 30 * IN_MILLISECONDS);
297 return;
298 case EVENT_LIFTOFF:
299 Talk(EMOTE_AIR_PHASE);
300 me->SetDisableGravity(true);
301 me->SetHover(true);
302 events.ScheduleEvent(EVENT_ICEBOLT, 1500);
303 _iceboltCount = RAID_MODE(2, 3);
304 return;
305 case EVENT_ICEBOLT:
306 {
307 std::vector<Unit*> targets;
308 std::list<HostileReference*>::const_iterator i = me->getThreatManager().getThreatList().begin();
309 for (; i != me->getThreatManager().getThreatList().end(); ++i)
310 if ((*i)->getTarget()->GetTypeId() == TYPEID_PLAYER && !(*i)->getTarget()->HasAura(SPELL_ICEBOLT))
311 targets.push_back((*i)->getTarget());
312
313 if (targets.empty())
314 _iceboltCount = 0;
315 else
316 {
317 std::vector<Unit*>::const_iterator itr = targets.begin();
318 advance(itr, rand32() % targets.size());
319 _iceblocks.insert(std::make_pair((*itr)->GetGUID(), ObjectGuid::Empty));
320 DoCast(*itr, SPELL_ICEBOLT);
321 --_iceboltCount;
322 }
323
324 if (_iceboltCount)
325 events.ScheduleEvent(EVENT_ICEBOLT, 1 * IN_MILLISECONDS);
326 else
327 events.ScheduleEvent(EVENT_BREATH, 1 * IN_MILLISECONDS);
328 return;
329 }
330 case EVENT_BREATH:
331 {
332 Talk(EMOTE_BREATH);
333 DoCastAOE(SPELL_FROST_MISSILE);
334 events.ScheduleEvent(EVENT_EXPLOSION, 8 * IN_MILLISECONDS);
335 return;
336 }
337 case EVENT_EXPLOSION:
338 CastExplosion();
339 ClearIceBlock();
340 events.ScheduleEvent(EVENT_LAND, 3 * IN_MILLISECONDS);
341 return;
342 case EVENT_LAND:
343 me->HandleEmoteCommand(EMOTE_ONESHOT_LAND);
344 Talk(EMOTE_GROUND_PHASE);
345 me->SetHover(false);
346 me->SetDisableGravity(false);
347 events.ScheduleEvent(EVENT_GROUND, 1500);
348 return;
349 case EVENT_GROUND:
350 EnterPhaseGround();
351 return;
352 case EVENT_BIRTH:
353 me->SetVisible(true);
354 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
355 me->SetReactState(REACT_AGGRESSIVE);
356 return;
357 }
358 }
359 }
360 }

362 void CastExplosion()
363 {
364 DoZoneInCombat(); // make sure everyone is in threatlist
365 std::vector<Unit*> targets;
366 std::list<HostileReference*>::const_iterator i = me->getThreatManager().getThreatList().begin();
367 for (; i != me->getThreatManager().getThreatList().end(); ++i)
368 {
369 Unit* target = (*i)->getTarget();
370 if (target->GetTypeId() != TYPEID_PLAYER)
371 continue;
372
373 if (target->HasAura(SPELL_ICEBOLT))
374 {
375 target->ApplySpellImmune(0, IMMUNITY_ID, SPELL_FROST_EXPLOSION, true);
376 targets.push_back(target);
377 continue;
378 }
379
380 for (IceBlockMap::const_iterator itr = _iceblocks.begin(); itr != _iceblocks.end(); ++itr)
381 {
382 if (GameObject* go = ObjectAccessor::GetGameObject(*me, itr->second))
383 {
384 if (go->IsInBetween(me, target, 2.0f)
385 && me->GetExactDist2d(target->GetPositionX(), target->GetPositionY()) - me->GetExactDist2d(go->GetPositionX(), go->GetPositionY()) < 5.0f)
386 {
387 target->ApplySpellImmune(0, IMMUNITY_ID, SPELL_FROST_EXPLOSION, true);
388 targets.push_back(target);
389 break;
390 }
391 }
392 }
393 }
394
395 me->CastSpell(me, SPELL_FROST_EXPLOSION, true);
396
397 for (std::vector<Unit*>::const_iterator itr = targets.begin(); itr != targets.end(); ++itr)
398 (*itr)->ApplySpellImmune(0, IMMUNITY_ID, SPELL_FROST_EXPLOSION, false);
399 }
400
401 private:
402 Phases _phase;
403 uint32 _iceboltCount;
404 IceBlockMap _iceblocks;
405 bool _canTheHundredClub;
406 Map* _map;
407 };

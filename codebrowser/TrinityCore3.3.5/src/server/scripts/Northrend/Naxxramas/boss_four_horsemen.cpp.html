109struct boss_four_horsemen_baseAI : public BossAI
110{
111 public:
112 Creature* getHorsemanHandle(Horseman who) const
113 {
114 if (_which == who)
115 return me;
116 else
117 return ObjectAccessor::GetCreature(*me, instance->GetGuidData(who));
118 }
119 boss_four_horsemen_baseAI(Creature* creature, Horseman which, Position const* initialPath) :
120 BossAI(creature, BOSS_HORSEMEN), _which(which), _initialPath(initialPath), _myMovementFinished(false), _nextMovement(0), _timeDied(0), _ourMovementFinished(false)
121 {
122 if (!me->IsAlive() && instance->GetBossState(BOSS_HORSEMEN) != DONE)
123 me->SetRespawnTime(10);
124 }
125
126 uint32 GetData(uint32 type) const override
127 {
128 switch (type)
129 {
130 case DATA_MOVEMENT_FINISHED:
131 return _myMovementFinished ? 1 : 0;
132 case DATA_DEATH_TIME:
133 return _timeDied;
134 case DATA_HORSEMEN_IS_TIMED_KILL:
135 {
136 uint32 minTime = 0, maxTime = 0;
137 for (Horseman boss : horsemen)
138 if (Creature* cBoss = getHorsemanHandle(boss))
139 {
140 uint32 deathTime = cBoss->AI()->GetData(DATA_DEATH_TIME);
141 if (!deathTime)
142 {
143 TC_LOG_WARN("scripts", "FourHorsemenAI: Checking for achievement credit but horseman %s is reporting not dead", cBoss->GetName().c_str());
144 return 0;
145 }
146 if (!minTime || deathTime < minTime)
147 minTime = deathTime;
148 if (!maxTime || deathTime > maxTime)
149 maxTime = deathTime;
150 }
151 else
152 {
153 TC_LOG_WARN("scripts", "FourHorsemenAI: Checking for achievement credit but horseman with id %u is not present", uint32(boss));
154 return 0;
155 }
156 return (getMSTimeDiff(minTime, maxTime) <= 15 * IN_MILLISECONDS) ? 1 : 0;
157 }
158 default:
159 return 0;
160 }
161 }
162
163 void DoAction(int32 action) override
164 {
165 switch (action)
166 {
167 case ACTION_BEGIN_MOVEMENT:
168 me->GetMotionMaster()->MovePoint(1, _initialPath[0], true);
169 break;
170 case ACTION_BEGIN_FIGHTING:
171 if (_ourMovementFinished)
172 break;
173 me->SetCombatPulseDelay(5);
174 BeginFighting();
175 _ourMovementFinished = true;
176 break;
177 }
178 }
179
180 void CheckIsMovementFinished()
181 {
182 for (Horseman boss : horsemen)
183 {
184 if (Creature* cBoss = getHorsemanHandle(boss))
185 {
186 if (cBoss->IsAlive() && !cBoss->AI()->GetData(DATA_MOVEMENT_FINISHED))
187 return;
188 }
189 else
190 {
191 TC_LOG_WARN("scripts", "FourHorsemenAI: Checking if movement is finished but horseman with id %u is not present", uint32(boss));
192 ResetEncounter();
193 return;
194 }
195 }
196
197 for (Horseman boss : horsemen)
198 if (Creature* cBoss = getHorsemanHandle(boss))
199 cBoss->AI()->DoAction(ACTION_BEGIN_FIGHTING);
200 }
201
202 void BeginEncounter()
203 {
204 if (instance->GetBossState(BOSS_HORSEMEN) == IN_PROGRESS)
205 return;
206 if (!instance->CheckRequiredBosses(BOSS_HORSEMEN))
207 {
208 ResetEncounter();
209 return;
210 }
211 instance->SetBossState(BOSS_HORSEMEN, IN_PROGRESS);
212 Map::PlayerList const &players = me->GetMap()->GetPlayers();
213 if (players.isEmpty()) // sanity check
214 ResetEncounter();
215
216 for (Horseman boss : horsemen)
217 {
218 if (Creature* cBoss = getHorsemanHandle(boss))
219 {
220 if (!cBoss->IsAlive())
221 {
222 ResetEncounter();
223 return;
224 }
225 cBoss->SetReactState(REACT_PASSIVE);
226 cBoss->AttackStop(); // clear initial target that was set on enter combat
227 cBoss->setActive(true);
228
229 for (Map::PlayerList::const_iterator it = players.begin(); it != players.end(); ++it)
230 {
231 if (Player* player = it->GetSource())
232 {
233 if (player->IsGameMaster())
234 continue;
235
236 if (player->IsAlive())
237 {
238 cBoss->AddThreat(player, 0.0f);
239 cBoss->SetInCombatWith(player);
240 player->SetInCombatWith(cBoss);
241 }
242 }
243 }
244
245 /* Why do the Four Horsemen run to opposite corners of the room when engaged? *
246 * They saw all the mobs leading up to them being AoE'd down and made a judgment call. */
247 cBoss->AI()->DoAction(ACTION_BEGIN_MOVEMENT);
248 }
249 else
250 {
251 TC_LOG_WARN("scripts", "FourHorsemenAI: Encounter starting but horseman with id %u is not present", uint32(boss));
252 ResetEncounter();
253 return;
254 }
255 }
256 }
257
258 void ResetEncounter()
259 {
260 if (instance->GetBossState(BOSS_HORSEMEN) == NOT_STARTED || instance->GetBossState(BOSS_HORSEMEN) == DONE)
261 return;
262 instance->SetBossState(BOSS_HORSEMEN, NOT_STARTED);
263 for (Horseman boss : horsemen)
264 {
265 if (Creature* cBoss = getHorsemanHandle(boss))
266 {
267 cBoss->DespawnOrUnsummon(0);
268 cBoss->SetRespawnTime(15);
269 }
270 else
271 {
272 TC_LOG_WARN("scripts", "FourHorsemenAI: Encounter resetting but horseman with id %u is not present", uint32(boss));
273 }
274 }
275 }
276
277 void EncounterCleared()
278 {
279 if (instance->GetBossState(BOSS_HORSEMEN) == DONE)
280 return;
281 instance->SetBossState(BOSS_HORSEMEN, DONE);
282 //instance->DoUpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_ENCOUNTER_CREDIT);
283 DoCastAOE(SPELL_ENCOUNTER_CREDIT, true);
284 }
285
286 void EnterCombat(Unit* /*who*/) override
287 {
288 if (instance->GetBossState(BOSS_HORSEMEN) != NOT_STARTED) // another horseman already did it
289 return;
290 Talk(SAY_AGGRO);
291 BeginEncounter();
292 }
293
294 void EnterEvadeMode(EvadeReason /*why*/) override
295 {
296 ResetEncounter();
297 }
298
299 void Reset() override
300 {
301 if (!me->IsAlive())
302 return;
303 _myMovementFinished = false;
304 _nextMovement = 0;
305 _timeDied = 0;
306 _ourMovementFinished = false;
307 me->SetReactState(REACT_AGGRESSIVE);
308 SetCombatMovement(false);
309 me->SetCombatPulseDelay(0);
310 me->ResetLootMode();
311 events.Reset();
312 summons.DespawnAll();
313 }
314
315 void KilledUnit(Unit* victim) override
316 {
317 if (victim->GetTypeId() == TYPEID_PLAYER)
318 Talk(SAY_SLAY, victim);
319 }
320
321 void JustDied(Unit* /*killer*/) override
322 {
323 if (instance->GetBossState(BOSS_HORSEMEN) != IN_PROGRESS) // necessary in case a horseman gets one-shot
324 {
325 BeginEncounter();
326 return;
327 }
328
329 Talk(SAY_DEATH);
330 _timeDied = getMSTime();
331 for (Horseman boss : horsemen)
332 {
333 if (Creature* cBoss = getHorsemanHandle(boss))
334 {
335 if (cBoss->IsAlive())
336 {
337 // in case a horseman dies while moving (unlikely but possible especially in non-335 branch)
338 CheckIsMovementFinished();
339 return;
340 }
341 }
342 else
343 {
344 TC_LOG_WARN("scripts", "FourHorsemenAI: %s died but horseman with id %u is not present", me->GetName().c_str(), uint32(boss));
345 ResetEncounter();
346 }
347 }
348
349 EncounterCleared();
350 }
351
352 void MovementInform(uint32 type, uint32 i) override
353 {
354 if (type != POINT_MOTION_TYPE)
355 return;
356 if (i < 3)
357 _nextMovement = i; // delay to next updateai to prevent it from instantly expiring
358 else
359 {
360 _myMovementFinished = true;
361 CheckIsMovementFinished();
362 }
363 }
364
365 void UpdateAI(uint32 diff) override
366 {
367 if (_nextMovement)
368 {
369 me->GetMotionMaster()->MovePoint(_nextMovement + 1, _initialPath[_nextMovement], true);
370 _nextMovement = 0;
371 }
372 _UpdateAI(diff);
373 }
374
375 virtual void BeginFighting() = 0;
376 virtual void _UpdateAI(uint32 /*diff*/) = 0;
377
378 private:
379 const Horseman _which;
380 const Position* _initialPath;
381 bool _myMovementFinished;
382 uint8 _nextMovement;
383 uint32 _timeDied;
384 protected:
385 bool _ourMovementFinished;
386};

388class boss_four_horsemen_baron : public CreatureScript
389{
390 public:
391 boss_four_horsemen_baron() : CreatureScript("boss_four_horsemen_baron") { }
392
393 struct boss_four_horsemen_baronAI : public boss_four_horsemen_baseAI
394 {
395 boss_four_horsemen_baronAI(Creature* creature) : boss_four_horsemen_baseAI(creature, BARON, baronPath) { }
396 void BeginFighting() override
397 {
398 SetCombatMovement(true);
399 me->SetReactState(REACT_AGGRESSIVE);
400 ThreatManager& threat = me->getThreatManager();
401 if (threat.isThreatListEmpty())
402 {
403 if (Unit* nearest = me->SelectNearestPlayer(5000.0f))
404 {
405 me->AddThreat(nearest, 1.0f);
406 AttackStart(nearest);
407 }
408 else
409 ResetEncounter();
410 }
411 else
412 AttackStart(threat.getHostilTarget());
413
414 events.ScheduleEvent(EVENT_BERSERK, 10 * MINUTE * IN_MILLISECONDS);
415 events.ScheduleEvent(EVENT_MARK, 24 * IN_MILLISECONDS);
416 events.ScheduleEvent(EVENT_UNHOLYSHADOW, urandms(3,7));
417 }
418
419 void _UpdateAI(uint32 diff) override
420 {
421 if (!_ourMovementFinished || !UpdateVictim())
422 return;
423 events.Update(diff);
424
425 while (uint32 eventId = events.ExecuteEvent())
426 {
427 switch (eventId)
428 {
429 case EVENT_BERSERK:
430 DoCastAOE(SPELL_BERSERK, true);
431 break;
432 case EVENT_MARK:
433 DoCastAOE(SPELL_BARON_MARK, true);
434 events.ScheduleEvent(EVENT_MARK, 12 * IN_MILLISECONDS);
435 break;
436 case EVENT_UNHOLYSHADOW:
437 DoCastVictim(SPELL_UNHOLY_SHADOW);
438 events.ScheduleEvent(EVENT_UNHOLYSHADOW, urandms(10,30));
439 break;
440 }
441 }
442
443 if (me->HasUnitState(UNIT_STATE_CASTING))
444 return;
445 DoMeleeAttackIfReady();
446 }
447
448 void SpellHitTarget(Unit* /*target*/, SpellInfo const* spell) override
449 {
450 if (spell->Id == SPELL_UNHOLY_SHADOW)
451 Talk(SAY_SPECIAL);
452 }
453 };
454
455 CreatureAI* GetAI(Creature* creature) const override
456 {
457 return GetInstanceAI<boss_four_horsemen_baronAI>(creature);
458 }

700 class spell_four_horsemen_mark_AuraScript : public AuraScript
701 {
702 PrepareAuraScript(spell_four_horsemen_mark_AuraScript);
703
704 void OnApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
705 {
706 if (Unit* caster = GetCaster())
707 {
708 int32 damage;
709 switch (GetStackAmount())
710 {
711 case 1:
712 damage = 0;
713 break;
714 case 2:
715 damage = 500;
716 break;
717 case 3:
718 damage = 1000;
719 break;
720 case 4:
721 damage = 1500;
722 break;
723 case 5:
724 damage = 4000;
725 break;
726 case 6:
727 damage = 12000;
728 break;
729 default:
730 damage = 20000 + 1000 * (GetStackAmount() - 7);
731 break;
732 }
733 if (damage)
734 caster->CastCustomSpell(SPELL_MARK_DAMAGE, SPELLVALUE_BASE_POINT0, damage, GetTarget());
735 }
736 }
737
738 void Register() override
739 {
740 AfterEffectApply += AuraEffectApplyFn(spell_four_horsemen_mark_AuraScript::OnApply, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL_OR_REAPPLY_MASK);
741 }
742 };
743
744 AuraScript* GetAuraScript() const override
745 {
746 return new spell_four_horsemen_mark_AuraScript();
747 }

22enum Phases
23{
24 PHASE_NONE,
25 PHASE_GROUND,
26 PHASE_BALCONY
27};
28
29enum Events
30{
31 EVENT_CURSE = 1, // curse of the plaguebringer
32 EVENT_BLINK, // blink (25m only)
33 EVENT_WARRIOR, // summon warriors during ground phase
34 EVENT_BALCONY, // become untargetable and begin balcony phase
35 EVENT_BALCONY_TELEPORT, // actually teleport to balcony, this is slightly delayed
36 EVENT_WAVE, // spawn wave during balcony phase
37 EVENT_GROUND, // end balcony phase and teleport to ground
38 EVENT_GROUND_ATTACKABLE // become attackable and aggressive again at start of ground phase, once again slightly delayed to prevent motionmaster weirdness
39};
40
41enum Talk
42{
43 SAY_AGGRO = 0,
44 SAY_SUMMON = 1,
45 SAY_SLAY = 2,
46 SAY_DEATH = 3,
47
48 EMOTE_SUMMON = 4, // ground phase
49 EMOTE_SUMMON_WAVE = 5, // balcony phase
50 EMOTE_TELEPORT_1 = 6, // ground to balcony
51 EMOTE_TELEPORT_2 = 7 // balcony to ground
52};
53
54enum Spells
55{
56 SPELL_CURSE = 29213, // 25-man: 54835
57 SPELL_CRIPPLE = 29212, // 25-man: 54814
58
59 SPELL_TELEPORT = 29216, // ground to balcony
60 SPELL_TELEPORT_BACK = 29231 // balcony to ground
61};
62
63enum Adds
64{
65 N_WARRIOR_SPELLS = 3,
66 N_CHAMPION_SPELLS = 6,
67 N_GUARDIAN_SPELLS = 3
68};
69const uint32 SummonWarriorSpells[N_WARRIOR_SPELLS] = { 29247, 29248, 29249 };
70const uint32 SummonChampionSpells[N_CHAMPION_SPELLS] = { 29238, 29255, 29257, 29258, 29262, 29267 };
71const uint32 SummonGuardianSpells[N_GUARDIAN_SPELLS] = { 29239, 29256, 29268 };
72
73#define SPELL_BLINK RAND(29208, 29209, 29210, 29211)
74
75class boss_noth : public CreatureScript
76{
77public:
78 boss_noth() : CreatureScript("boss_noth") { }
79
80 struct boss_nothAI : public BossAI
81 {
82 boss_nothAI(Creature* creature) : BossAI(creature, BOSS_NOTH), balconyCount(0), justBlinked(false)
83 {
84 std::copy(SummonWarriorSpells, SummonWarriorSpells + N_WARRIOR_SPELLS, _SummonWarriorSpells);
85 std::copy(SummonChampionSpells, SummonChampionSpells + N_CHAMPION_SPELLS, _SummonChampionSpells);
86 std::copy(SummonGuardianSpells, SummonGuardianSpells + N_GUARDIAN_SPELLS, _SummonGuardianSpells);
87
88 events.SetPhase(PHASE_NONE);
89 }
90
91 void EnterEvadeMode(EvadeReason /*why*/) override
92 {
93 Reset(); // teleport back first
94 _EnterEvadeMode();
95 }
96
97 void Reset() override
98 {
99 if (!me->IsAlive())
100 return;
101
102 // in case we reset during balcony phase
103 if (events.IsInPhase(PHASE_BALCONY))
104 {
105 DoCastAOE(SPELL_TELEPORT_BACK);
106 me->SetReactState(REACT_AGGRESSIVE);
107 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
108 }
109
110 balconyCount = 0;
111 events.SetPhase(PHASE_NONE);
112 justBlinked = false;
113
114 _Reset();
115 }
116
117 void EnterCombat(Unit* /*who*/) override
118 {
119 _EnterCombat();
120 Talk(SAY_AGGRO);
121 EnterPhaseGround();
122 }
123
124 void EnterPhaseGround()
125 {
126 events.SetPhase(PHASE_GROUND);
127
128 DoZoneInCombat();
129
130 if (me->getThreatManager().isThreatListEmpty())
131 Reset();
132 else
133 {
134 uint8 secondsGround;
135 switch (balconyCount)
136 {
137 case 0:
138 secondsGround = 90;
139 break;
140 case 1:
141 secondsGround = 110;
142 break;
143 case 2:
144 default:
145 secondsGround = 180;
146 }
147 events.ScheduleEvent(EVENT_GROUND_ATTACKABLE, 2 * IN_MILLISECONDS, 0, PHASE_GROUND);
148 events.ScheduleEvent(EVENT_BALCONY, secondsGround * IN_MILLISECONDS, 0, PHASE_GROUND);
149 events.ScheduleEvent(EVENT_CURSE, urand(10,25) * IN_MILLISECONDS, 0, PHASE_GROUND);
150 events.ScheduleEvent(EVENT_WARRIOR, urand(20,30) * IN_MILLISECONDS, 0, PHASE_GROUND);
151 if (GetDifficulty() == RAID_DIFFICULTY_25MAN_NORMAL)
152 events.ScheduleEvent(EVENT_BLINK, urand(20,30) * IN_MILLISECONDS, 0, PHASE_GROUND);
153 }
154 }
155
156 void KilledUnit(Unit* victim) override
157 {
158 if (victim->GetTypeId() == TYPEID_PLAYER)
159 Talk(SAY_SLAY);
160 }

164 summons.Summon(summon);

167 }

169 void JustDied(Unit* /*killer*/) override
170 {
171 _JustDied();
172 Talk(SAY_DEATH);
173 }
174
175 void DamageTaken(Unit* /*who*/, uint32& damage) override // prevent noth from somehow dying in the balcony phase
176 {
177 if (!events.IsInPhase(PHASE_BALCONY))
178 return;
179 if (damage < me->GetHealth())
180 return;
181
182 me->SetHealth(1u);
183 damage = 0u;
184 }
185
186 void HandleSummon(uint32* spellsList, const uint8 nSpells, uint8 num)
187 { // this ensures we do not spawn two mobs using the same spell (<=> in the same position) if we can help it
188 while (num)
189 for (uint8 it = 0; it < nSpells && num; ++it)
190 {
191 num--;
192 uint8 selected = urand(it, nSpells - 1);
193 DoCastAOE(spellsList[selected]);
194 if (selected != it) // shuffle the selected into the part of the array that is no longer being searched
195 std::swap(spellsList[selected], spellsList[it]);
196 }
197 }
198
199 void CastSummon(uint8 nWarrior, uint8 nChampion, uint8 nGuardian)
200 {
201 HandleSummon(_SummonWarriorSpells, N_WARRIOR_SPELLS, nWarrior);
202 HandleSummon(_SummonChampionSpells, N_CHAMPION_SPELLS, nChampion);
203 HandleSummon(_SummonGuardianSpells, N_GUARDIAN_SPELLS, nGuardian);
204 }
205
206 void UpdateAI(uint32 diff) override
207 {
208 if (!UpdateVictim())
209 return;
210
211 events.Update(diff);
212
213 if (me->HasUnitState(UNIT_STATE_CASTING))
214 return;
215
216 while (uint32 eventId = events.ExecuteEvent())
217 {
218 switch (eventId)
219 {
220 case EVENT_CURSE:
221 {
222 DoCastAOE(SPELL_CURSE);
223 events.ScheduleEvent(EVENT_CURSE, urand(50, 70) * IN_MILLISECONDS, 0, PHASE_GROUND);
224 break;
225 }
226 case EVENT_WARRIOR:
227 Talk(SAY_SUMMON);
228 Talk(EMOTE_SUMMON);
229
230 CastSummon(RAID_MODE(2, 3), 0, 0);
231
232 events.ScheduleEvent(EVENT_WARRIOR, 40 * IN_MILLISECONDS, 0, PHASE_GROUND);
233 break;
234 case EVENT_BLINK:
235 DoCastAOE(SPELL_CRIPPLE, true);
236 DoCastAOE(SPELL_BLINK);
237 DoResetThreat();
238 justBlinked = true;
239
240 events.ScheduleEvent(EVENT_BLINK, 40000, 0, PHASE_GROUND);
241 break;
242 case EVENT_BALCONY:
243 events.SetPhase(PHASE_BALCONY);
244 me->SetReactState(REACT_PASSIVE);
245 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
246 me->AttackStop();
247 me->StopMoving();
248 me->RemoveAllAuras();
249
250 events.ScheduleEvent(EVENT_BALCONY_TELEPORT, 3 * IN_MILLISECONDS, 0, PHASE_BALCONY);
251 events.ScheduleEvent(EVENT_WAVE, urand(5 * IN_MILLISECONDS, 8 * IN_MILLISECONDS), 0, PHASE_BALCONY);
252
253 uint8 secondsBalcony;
254 switch (balconyCount)
255 {
256 case 0:
257 secondsBalcony = 70;
258 break;
259 case 1:
260 secondsBalcony = 97;
261 break;
262 case 2:
263 default:
264 secondsBalcony = 120;
265 break;
266 }
267 events.ScheduleEvent(EVENT_GROUND, secondsBalcony * IN_MILLISECONDS, 0, PHASE_BALCONY);
268 break;
269 case EVENT_BALCONY_TELEPORT:
270 Talk(EMOTE_TELEPORT_1);
271 DoCastAOE(SPELL_TELEPORT);
272 break;
273 case EVENT_WAVE:
274 Talk(EMOTE_SUMMON_WAVE);
275 switch (balconyCount)
276 {
277 case 0:
278 CastSummon(0, RAID_MODE(2, 4), 0);
279 break;
280 case 1:
281 CastSummon(0, RAID_MODE(1, 2), RAID_MODE(1, 2));
282 break;
283 case 2:
284 CastSummon(0, 0, RAID_MODE(2, 4));
285 break;
286 default:
287 CastSummon(0, RAID_MODE(5, 10), RAID_MODE(5, 10));
288 break;
289 }
290 events.ScheduleEvent(EVENT_WAVE, urand(30, 45) * IN_MILLISECONDS, 0, PHASE_BALCONY);
291 break;
292 case EVENT_GROUND:
293 ++balconyCount;
294
295 DoCastAOE(SPELL_TELEPORT_BACK);
296 Talk(EMOTE_TELEPORT_2);
297
298 EnterPhaseGround();
299 break;
300 case EVENT_GROUND_ATTACKABLE:
301 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_NOT_SELECTABLE);
302 me->SetReactState(REACT_AGGRESSIVE);
303 break;
304 }
305 }
306
307 if (events.IsInPhase(PHASE_GROUND))
308 {
309 /* workaround for movechase breaking after blinking
310 without this noth would just stand there unless his current target moves */
311 if (justBlinked && me->GetVictim() && !me->IsWithinMeleeRange(me->EnsureVictim()))
312 {
313 me->GetMotionMaster()->Clear();
314 me->GetMotionMaster()->MoveChase(me->EnsureVictim());
315 justBlinked = false;
316 }
317 else
318 DoMeleeAttackIfReady();
319 }
320 }
321
322 private:
323 uint32 balconyCount;
324
325 bool justBlinked;
326
327 uint32 _SummonWarriorSpells[N_WARRIOR_SPELLS];
328 uint32 _SummonChampionSpells[N_CHAMPION_SPELLS];
329 uint32 _SummonGuardianSpells[N_GUARDIAN_SPELLS];
330 };
331
332 CreatureAI* GetAI(Creature* creature) const override
333 {
334 return GetInstanceAI<boss_nothAI>(creature);
335 }
336};

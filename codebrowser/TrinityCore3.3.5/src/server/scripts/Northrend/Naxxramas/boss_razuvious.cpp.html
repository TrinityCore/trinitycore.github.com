56class boss_razuvious : public CreatureScript
57{
58public:
59 boss_razuvious() : CreatureScript("boss_razuvious") { }
60
61 CreatureAI* GetAI(Creature* creature) const override
62 {
63 return new boss_razuviousAI(creature);
64 }
65
66 struct boss_razuviousAI : public BossAI
67 {
68 boss_razuviousAI(Creature* creature) : BossAI(creature, BOSS_RAZUVIOUS) { }
69
70 void SummonAdds()
71 {
72 me->SummonCreatureGroup(SUMMON_GROUP_10MAN);
73 if (Is25ManRaid())
74 me->SummonCreatureGroup(SUMMON_GROUP_25MAN);
75 }
76
77 void InitializeAI() override
78 {
79 if (!me->isDead())
80 {
81 Reset();
82 SummonAdds();
83 }
84 }
85
86 void JustReachedHome() override
87 {
88 _JustReachedHome();
89 SummonAdds();
90 me->GetMotionMaster()->Initialize();
91 }
92
93 void KilledUnit(Unit* victim) override
94 {
95 if (victim->GetTypeId() == TYPEID_PLAYER || (victim->GetTypeId() == TYPEID_UNIT && victim->GetEntry() == NPC_DK_UNDERSTUDY))
96 Talk(SAY_SLAY);
97 }
98
99 void SpellHit(Unit* caster, SpellInfo const* spell) override
100 {
101 if (spell->Id == SPELL_UNDERSTUDY_TAUNT)
102 Talk(SAY_TAUNTED, caster);
103 }
104
105 void JustDied(Unit* /*killer*/) override
106 {
107 Talk(SAY_DEATH);
108 DoCastAOE(SPELL_HOPELESS, true);
109
110 events.Reset();
111 instance->SetBossState(BOSS_RAZUVIOUS, DONE);
112 }
113
114 void EnterCombat(Unit* /*who*/) override
115 {
116 _EnterCombat();
117 me->StopMoving();
118 summons.DoZoneInCombat();
119 Talk(SAY_AGGRO);
120 events.ScheduleEvent(EVENT_ATTACK, 7 * IN_MILLISECONDS);
121 events.ScheduleEvent(EVENT_STRIKE, 21 * IN_MILLISECONDS);
122 events.ScheduleEvent(EVENT_SHOUT, 16 * IN_MILLISECONDS);
123 events.ScheduleEvent(EVENT_KNIFE, 10 * IN_MILLISECONDS);
124 }
125
126 void UpdateAI(uint32 diff) override
127 {
128 if (!UpdateVictim())
129 return;
130
131 events.Update(diff);
132
133 while (uint32 eventId = events.ExecuteEvent())
134 {
135 switch (eventId)
136 {
137 case EVENT_ATTACK:
138 SetCombatMovement(true);
139 if (Unit* victim = me->GetVictim())
140 me->GetMotionMaster()->MoveChase(victim);
141 break;
142 case EVENT_STRIKE:
143 DoCastVictim(SPELL_UNBALANCING_STRIKE);
144 events.ScheduleEvent(EVENT_STRIKE, 6 * IN_MILLISECONDS);
145 return;
146 case EVENT_SHOUT:
147 DoCastAOE(SPELL_DISRUPTING_SHOUT);
148 events.ScheduleEvent(EVENT_SHOUT, 16 * IN_MILLISECONDS);
149 return;
150 case EVENT_KNIFE:
151 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f))
152 DoCast(target, SPELL_JAGGED_KNIFE);
153 events.ScheduleEvent(EVENT_KNIFE, urandms(10,15));
154 return;
155 }
156 }
157
158 DoMeleeAttackIfReady();
159 }
160
161 void Reset() override
162 {
163 SetCombatMovement(false);
164 _Reset();
165 }
166 };
167
168};

170class npc_dk_understudy : public CreatureScript
171{
172 public:
173 npc_dk_understudy() : CreatureScript("npc_dk_understudy") { }
174
175 struct npc_dk_understudyAI : public ScriptedAI
176 {
177 npc_dk_understudyAI(Creature* creature) : ScriptedAI(creature), _instance(creature->GetInstanceScript()), bloodStrikeTimer(0) { }
178
179 void EnterCombat(Unit* /*who*/) override
180 {
181 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_NONE);
182 if (Creature* razuvious = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_RAZUVIOUS)))
183 razuvious->AI()->DoZoneInCombat(nullptr, 250.0f);
184 }
185
186 void JustReachedHome() override
187 {
188 if (_instance->GetBossState(BOSS_RAZUVIOUS) == DONE)
189 me->DespawnOrUnsummon();
190 else
191 ScriptedAI::JustReachedHome();
192 }
193
194 void UpdateAI(uint32 diff) override
195 {
196 if (!me->isPossessedByPlayer() && !UpdateVictim())
197 return;
198
199 if (!me->isPossessedByPlayer())
200 {
201 if (diff < bloodStrikeTimer)
202 bloodStrikeTimer -= diff;
203 else
204 DoCastVictim(SPELL_UNDERSTUDY_BLOOD_STRIKE);
205 }
206
207 DoMeleeAttackIfReady();
208 }
209
210 void OnCharmed(bool apply) override
211 {
212 ScriptedAI::OnCharmed(apply);
213 if (apply)
214 {
215 if (!me->IsInCombat())
216 EnterCombat(nullptr);
217 me->StopMoving();
218 me->SetReactState(REACT_PASSIVE);
219 _charmer = me->GetCharmerGUID();
220 }
221 else
222 {
223 me->SetReactState(REACT_AGGRESSIVE);
224 if (Unit* charmer = ObjectAccessor::GetUnit(*me, _charmer))
225 me->AddThreat(charmer, 100000.0f);
226 DoZoneInCombat(nullptr, 250.0f);
227 }
228 }
229 private:
230 InstanceScript* const _instance;
231 ObjectGuid _charmer;
232 uint32 bloodStrikeTimer;
233 };
234
235 CreatureAI* GetAI(Creature* creature) const override
236 {
237 return GetInstanceAI<npc_dk_understudyAI>(creature);
238 }
239};

244 new npc_dk_understudy();

67class boss_faerlina : public CreatureScript
68{
69 public:
70 boss_faerlina() : CreatureScript("boss_faerlina") { }
71
72 struct boss_faerlinaAI : public BossAI
73 {
74 boss_faerlinaAI(Creature* creature) : BossAI(creature, BOSS_FAERLINA), _frenzyDispels(0) { }
75
76 void SummonAdds()
77 {
78 me->SummonCreatureGroup(SUMMON_GROUP_WORSHIPPERS);
79 if (Is25ManRaid())
80 me->SummonCreatureGroup(SUMMON_GROUP_FOLLOWERS);
81 }
82
83 void InitializeAI() override
84 {
85 if (!me->isDead())
86 {
87 Reset();
88 SummonAdds();
89 }
90 }
91
92 void JustReachedHome() override
93 {
94 _JustReachedHome();
95 SummonAdds();
96 }
97
98 void EnterCombat(Unit* /*who*/) override
99 {
100 _EnterCombat();
101 Talk(SAY_AGGRO);
102 summons.DoZoneInCombat();
103 events.ScheduleEvent(EVENT_POISON, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
104 events.ScheduleEvent(EVENT_FIRE, urand(6 * IN_MILLISECONDS, 18 * IN_MILLISECONDS));
105 events.ScheduleEvent(EVENT_FRENZY, urand(60 * IN_MILLISECONDS, 80 * IN_MILLISECONDS));
106 }
107
108 void Reset() override
109 {
110 _Reset();
111 _frenzyDispels = 0;
112 }
113
114 void KilledUnit(Unit* /*victim*/) override
115 {
116 if (!urand(0, 2))
117 Talk(SAY_SLAY);
118 }
119
120 void JustDied(Unit* /*killer*/) override
121 {
122 _JustDied();
123 Talk(SAY_DEATH);
124 }
125
126 void SpellHit(Unit* caster, SpellInfo const* spell) override
127 {
128 if (spell->Id == SPELL_WIDOWS_EMBRACE_HELPER)
129 {
130 ++_frenzyDispels;
131 Talk(EMOTE_WIDOW_EMBRACE, caster);
132 me->Kill(caster);
133 }
134 }
135
136 uint32 GetData(uint32 type) const override
137 {
138 if (type == DATA_FRENZY_DISPELS)
139 return _frenzyDispels;
140
141 return 0;
142 }
143
144 void UpdateAI(uint32 diff) override
145 {
146 if (!UpdateVictim())
147 return;
148
149 events.Update(diff);
150
151 if (me->HasUnitState(UNIT_STATE_CASTING))
152 return;
153
154 while (uint32 eventId = events.ExecuteEvent())
155 {
156 switch (eventId)
157 {
158 case EVENT_POISON:
159 if (!me->HasAura(SPELL_WIDOWS_EMBRACE_HELPER))
160 DoCastAOE(SPELL_POISON_BOLT_VOLLEY);
161 events.ScheduleEvent(EVENT_POISON, urand(8000, 15000));
162 break;
163 case EVENT_FIRE:
164 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
165 DoCast(target, SPELL_RAIN_OF_FIRE);
166 events.ScheduleEvent(EVENT_FIRE, urand(6000, 18000));
167 break;
168 case EVENT_FRENZY:
169 if (Aura* widowsEmbrace = me->GetAura(SPELL_WIDOWS_EMBRACE_HELPER))
170 events.ScheduleEvent(EVENT_FRENZY, widowsEmbrace->GetDuration()+1 * IN_MILLISECONDS);
171 else
172 {
173 DoCast(SPELL_FRENZY);
174 Talk(EMOTE_FRENZY);
175 events.ScheduleEvent(EVENT_FRENZY, urand(60 * IN_MILLISECONDS, 80 * IN_MILLISECONDS));
176 }
177 break;
178 }
179 }
180
181 DoMeleeAttackIfReady();
182 }
183
184 private:
185 uint32 _frenzyDispels;
186 };
187
188 CreatureAI* GetAI(Creature* creature) const override
189 {
190 return new boss_faerlinaAI(creature);
191 }
192};

194class npc_faerlina_add : public CreatureScript
195{
196 public:
197 npc_faerlina_add() : CreatureScript("npc_faerlina_add") { }
198
199 struct npc_faerlina_addAI : public ScriptedAI
200 {
201 npc_faerlina_addAI(Creature* creature) : ScriptedAI(creature),
202 _instance(creature->GetInstanceScript())
203 {
204 }
205
206 void Reset() override
207 {
208 if (!Is25ManRaid()) {
209 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_BIND, true);
210 me->ApplySpellImmune(0, IMMUNITY_MECHANIC, MECHANIC_CHARM, true);
211 }
212 }
213
214 void EnterCombat(Unit* /*who*/) override
215 {
216 if (Creature* faerlina = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_FAERLINA)))
217 faerlina->AI()->DoZoneInCombat(nullptr, 250.0f);
218 }
219
220 void JustDied(Unit* /*killer*/) override
221 {
222 if (!Is25ManRaid())
223 if (Creature* faerlina = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(DATA_FAERLINA)))
224 DoCast(faerlina, SPELL_WIDOWS_EMBRACE);
225 }
226
227 void UpdateAI(uint32 /*diff*/) override
228 {
229 if (!UpdateVictim())
230 return;
231 if (me->HasUnitState(UNIT_STATE_CASTING))
232 return;
233
234 DoCastVictim(SPELL_ADD_FIREBALL);
235 DoMeleeAttackIfReady(); // this will only happen if the fireball cast fails for some reason
236 }
237
238 private:
239 InstanceScript* const _instance;
240 };
241
242 CreatureAI* GetAI(Creature* creature) const override
243 {
244 return GetInstanceAI<npc_faerlina_addAI>(creature);
245 }

259class at_faerlina_entrance : public AreaTriggerScript
260{
261 public:
262 at_faerlina_entrance() : AreaTriggerScript("at_faerlina_entrance") { }
263
264 bool OnTrigger(Player* player, AreaTriggerEntry const* /*areaTrigger*/) override
265 {
266 InstanceScript* instance = player->GetInstanceScript();
267 if (!instance || instance->GetData(DATA_HAD_FAERLINA_GREET) || instance->GetBossState(BOSS_FAERLINA) != NOT_STARTED)
268 return true;
269
270 if (Creature* faerlina = ObjectAccessor::GetCreature(*player, instance->GetGuidData(DATA_FAERLINA)))
271 faerlina->AI()->Talk(SAY_GREET);
272 instance->SetData(DATA_HAD_FAERLINA_GREET, 1u);
273
274 return true;
275 }
276};

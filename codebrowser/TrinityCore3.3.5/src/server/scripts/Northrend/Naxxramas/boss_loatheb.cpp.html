57class boss_loatheb : public CreatureScript
58{
59 public:
60 boss_loatheb() : CreatureScript("boss_loatheb") { }
61
62 struct boss_loathebAI : public BossAI
63 {
64 boss_loathebAI(Creature* creature) : BossAI(creature, BOSS_LOATHEB)
65 {
66 Initialize();
67 }
68
69 void Initialize()
70 {
71 _doomCounter = 0;
72 _sporeLoserData = true;
73 }
74
75 void Reset() override
76 {
77 _Reset();
78 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_FUNGAL_CREEP);
79 Initialize();
80 }
81
82 void EnterCombat(Unit* /*who*/) override
83 {
84 _EnterCombat();
85 events.ScheduleEvent(EVENT_NECROTIC_AURA, 17 * IN_MILLISECONDS);
86 events.ScheduleEvent(EVENT_DEATHBLOOM, 5 * IN_MILLISECONDS);
87 events.ScheduleEvent(EVENT_SPORE, RAID_MODE(36,18) * IN_MILLISECONDS);
88 events.ScheduleEvent(EVENT_INEVITABLE_DOOM, 2 * MINUTE * IN_MILLISECONDS);
89 }
90
91 void SummonedCreatureDies(Creature* summon, Unit* /*killer*/) override
92 {
93 _sporeLoserData = false;
94 summon->CastSpell(summon,SPELL_FUNGAL_CREEP,true);
95 }
96
97 void SummonedCreatureDespawn(Creature* summon) override
98 {
99 summons.Despawn(summon);
100 if (summon->IsAlive())
101 summon->CastSpell(summon,SPELL_FUNGAL_CREEP,true);
102 }
103
104 uint32 GetData(uint32 id) const override
105 {
106 return (_sporeLoserData && id == DATA_ACHIEVEMENT_SPORE_LOSER) ? 1u : 0u;
107 }
108
109 void UpdateAI(uint32 diff) override
110 {
111 if (!UpdateVictim())
112 return;
113
114 events.Update(diff);
115
116 while (uint32 eventId = events.ExecuteEvent())
117 {
118 switch (eventId)
119 {
120 case EVENT_NECROTIC_AURA:
121 DoCastAOE(SPELL_NECROTIC_AURA);
122 DoCast(me, SPELL_WARN_NECROTIC_AURA);
123 events.ScheduleEvent(EVENT_NECROTIC_AURA, 20 * IN_MILLISECONDS);
124 events.ScheduleEvent(EVENT_NECROTIC_AURA_FADING, 14 * IN_MILLISECONDS);
125 break;
126 case EVENT_DEATHBLOOM:
127 DoCastAOE(SPELL_DEATHBLOOM);
128 events.ScheduleEvent(EVENT_DEATHBLOOM, 30 * IN_MILLISECONDS);
129 break;
130 case EVENT_INEVITABLE_DOOM:
131 _doomCounter++;
132 DoCastAOE(SPELL_INEVITABLE_DOOM);
133 if (_doomCounter > 6)
134 {
135 if (_doomCounter & 1) // odd
136 events.ScheduleEvent(EVENT_INEVITABLE_DOOM, 14 * IN_MILLISECONDS);
137 else
138 events.ScheduleEvent(EVENT_INEVITABLE_DOOM, 17 * IN_MILLISECONDS);
139 }
140 else
141 events.ScheduleEvent(EVENT_INEVITABLE_DOOM, 30 * IN_MILLISECONDS);
142 break;
143 case EVENT_SPORE:
144 DoCast(me, SPELL_SUMMON_SPORE, false);
145 events.ScheduleEvent(EVENT_SPORE, RAID_MODE(36,18) * IN_MILLISECONDS);
146 break;
147 case EVENT_NECROTIC_AURA_FADING:
148 Talk(SAY_NECROTIC_AURA_FADING);
149 break;
150 default:
151 break;
152 }
153 }
154
155 DoMeleeAttackIfReady();
156 }
157
158 private:
159 bool _sporeLoserData;
160 uint8 _doomCounter;
161 };
162
163 CreatureAI* GetAI(Creature* creature) const override
164 {
165 return new boss_loathebAI(creature);
166 }
167};

182class spell_loatheb_necrotic_aura_warning : public SpellScriptLoader

186
187 class spell_loatheb_necrotic_aura_warning_AuraScript : public AuraScript
188 {
189 PrepareAuraScript(spell_loatheb_necrotic_aura_warning_AuraScript);
190
191 bool Validate(SpellInfo const* /*spell*/) override
192 {
193 if (!sSpellStore.LookupEntry(SPELL_WARN_NECROTIC_AURA))
194 return false;
195 return true;
196 }
197
198 void HandleEffectApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
199 {
200 if (GetTarget()->IsAIEnabled)
201 ENSURE_AI(LoathebAI, GetTarget()->GetAI())->Talk(SAY_NECROTIC_AURA_APPLIED);
202 }
203
204 void HandleEffectRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
205 {
206 if (GetTarget()->IsAIEnabled)
207 ENSURE_AI(LoathebAI, GetTarget()->GetAI())->Talk(SAY_NECROTIC_AURA_REMOVED);
208 }
209
210 void Register() override
211 {
212 AfterEffectApply += AuraEffectApplyFn(spell_loatheb_necrotic_aura_warning_AuraScript::HandleEffectApply, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
213 AfterEffectRemove += AuraEffectRemoveFn(spell_loatheb_necrotic_aura_warning_AuraScript::HandleEffectRemove, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL);
214 }
215 };
216
217 AuraScript* GetAuraScript() const override
218 {
219 return new spell_loatheb_necrotic_aura_warning_AuraScript();
220 }

224{
225 public:
226 spell_loatheb_deathbloom() : SpellScriptLoader("spell_loatheb_deathbloom") { }
227
228 class spell_loatheb_deathbloom_AuraScript : public AuraScript
229 {
230 PrepareAuraScript(spell_loatheb_deathbloom_AuraScript);
231
232 bool Validate(SpellInfo const* /*spell*/) override
233 {
234 if (!sSpellMgr->GetSpellInfo(SPELL_DEATHBLOOM_FINAL_DAMAGE))
235 return false;
236 return true;
237 }
238
239 void AfterRemove(AuraEffect const* eff, AuraEffectHandleModes /*mode*/)
240 {
241 if (GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_EXPIRE)
242 return;
243
244 GetTarget()->CastSpell(nullptr, SPELL_DEATHBLOOM_FINAL_DAMAGE, true, nullptr, eff, GetCasterGUID());
245 }
246
247 void Register() override
248 {
249 AfterEffectRemove += AuraEffectRemoveFn(spell_loatheb_deathbloom_AuraScript::AfterRemove, EFFECT_0, SPELL_AURA_PERIODIC_DAMAGE, AURA_EFFECT_HANDLE_REAL);
250 }
251 };
252
253 AuraScript* GetAuraScript() const override
254 {
255 return new spell_loatheb_deathbloom_AuraScript();
256 }

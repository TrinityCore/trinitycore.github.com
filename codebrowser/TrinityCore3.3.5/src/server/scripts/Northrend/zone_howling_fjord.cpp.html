52class npc_apothecary_hanes : public CreatureScript
53{
54public:
55 npc_apothecary_hanes() : CreatureScript("npc_apothecary_hanes") { }
56
57 bool OnQuestAccept(Player* player, Creature* creature, Quest const* quest) override
58 {
59 if (quest->GetQuestId() == QUEST_TRAIL_OF_FIRE)
60 {
61 switch (player->GetTeam())
62 {
63 case ALLIANCE:
64 creature->setFaction(FACTION_ESCORTEE_A);
65 break;
66 case HORDE:
67 creature->setFaction(FACTION_ESCORTEE_H);
68 break;
69 }
70 ENSURE_AI(npc_escortAI, (creature->AI()))->Start(true, false, player->GetGUID());
71 }
72 return true;
73 }

75 struct npc_Apothecary_HanesAI : public npc_escortAI
76 {
77 npc_Apothecary_HanesAI(Creature* creature) : npc_escortAI(creature)
78 {
79 Initialize();
80 }
81
82 void Initialize()
83 {
84 PotTimer = 10000; //10 sec cooldown on potion
85 }
86
87 uint32 PotTimer;
88
89 void Reset() override
90 {
91 SetDespawnAtFar(false);
92 Initialize();
93 }
94
95 void JustDied(Unit* /*killer*/) override
96 {
97 if (Player* player = GetPlayerForEscort())
98 player->FailQuest(QUEST_TRAIL_OF_FIRE);
99 }
100
101 void UpdateEscortAI(uint32 diff) override
102 {
103 if (HealthBelowPct(75))
104 {
105 if (PotTimer <= diff)
106 {
107 DoCast(me, SPELL_HEALING_POTION, true);
108 PotTimer = 10000;
109 } else PotTimer -= diff;
110 }
111 if (GetAttack() && UpdateVictim())
112 DoMeleeAttackIfReady();
113 }
114
115 void WaypointReached(uint32 waypointId) override
116 {
117 Player* player = GetPlayerForEscort();
118 if (!player)
119 return;
120
121 switch (waypointId)
122 {
123 case 1:
124 me->SetReactState(REACT_AGGRESSIVE);
125 SetRun(true);
126 break;
127 case 23:
128 player->GroupEventHappens(QUEST_TRAIL_OF_FIRE, me);
129 me->DespawnOrUnsummon();
130 break;
131 case 5:
132 if (Unit* Trigger = me->FindNearestCreature(NPC_HANES_FIRE_TRIGGER, 10.0f))
133 Trigger->CastSpell(Trigger, SPELL_COSMETIC_LOW_POLY_FIRE, false);
134 SetRun(false);
135 break;
136 case 6:
137 if (Unit* Trigger = me->FindNearestCreature(NPC_HANES_FIRE_TRIGGER, 10.0f))
138 Trigger->CastSpell(Trigger, SPELL_COSMETIC_LOW_POLY_FIRE, false);
139 SetRun(true);
140 break;
141 case 8:
142 if (Unit* Trigger = me->FindNearestCreature(NPC_HANES_FIRE_TRIGGER, 10.0f))
143 Trigger->CastSpell(Trigger, SPELL_COSMETIC_LOW_POLY_FIRE, false);
144 SetRun(false);
145 break;
146 case 9:
147 if (Unit* Trigger = me->FindNearestCreature(NPC_HANES_FIRE_TRIGGER, 10.0f))
148 Trigger->CastSpell(Trigger, SPELL_COSMETIC_LOW_POLY_FIRE, false);
149 break;
150 case 10:
151 SetRun(true);
152 break;
153 case 13:
154 SetRun(false);
155 break;
156 case 14:
157 if (Unit* Trigger = me->FindNearestCreature(NPC_HANES_FIRE_TRIGGER, 10.0f))
158 Trigger->CastSpell(Trigger, SPELL_COSMETIC_LOW_POLY_FIRE, false);
159 SetRun(true);
160 break;
161 }
162 }
163 };
164
165 CreatureAI* GetAI(Creature* creature) const override
166 {
167 return new npc_Apothecary_HanesAI(creature);
168 }

189 void Reset() override
190 {
191 ObjectGuid summonerGUID;
192
193 if (me->IsSummon())
194 if (Unit* summoner = me->ToTempSummon()->GetSummoner())
195 if (summoner->GetTypeId() == TYPEID_PLAYER)
196 summonerGUID = summoner->GetGUID();
197
198 if (!summonerGUID)
199 return;
200
201 me->SetWalk(true);
202 Start(false, false, summonerGUID);
203 }
204
205 void WaypointReached(uint32 waypointId) override
206 {
207 if (waypointId != 26)
208 return;
209
210 me->DespawnOrUnsummon();
211 }
212 };

239{
240public:
241 npc_razael_and_lyana() : CreatureScript("npc_razael_and_lyana") { }
242
243 bool OnGossipHello(Player* player, Creature* creature) override
244 {
245 if (creature->IsQuestGiver())
246 player->PrepareQuestMenu(creature->GetGUID());
247
248 if (player->GetQuestStatus(QUEST_REPORTS_FROM_THE_FIELD) == QUEST_STATUS_INCOMPLETE)
249 switch (creature->GetEntry())
250 {
251 case NPC_RAZAEL:
252 if (!player->GetReqKillOrCastCurrentCount(QUEST_REPORTS_FROM_THE_FIELD, NPC_RAZAEL))
253 {
254 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_RAZAEL_REPORT, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
255 player->SEND_GOSSIP_MENU(GOSSIP_TEXTID_RAZAEL1, creature->GetGUID());
256 return true;
257 }
258 break;
259 case NPC_LYANA:
260 if (!player->GetReqKillOrCastCurrentCount(QUEST_REPORTS_FROM_THE_FIELD, NPC_LYANA))
261 {
262 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_LYANA_REPORT, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
263 player->SEND_GOSSIP_MENU(GOSSIP_TEXTID_LYANA1, creature->GetGUID());
264 return true;
265 }
266 break;
267 }
268 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
269 return true;
270 }
271
272 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
273 {
274 player->PlayerTalkClass->ClearMenus();
275 switch (action)
276 {
277 case GOSSIP_ACTION_INFO_DEF + 1:
278 player->SEND_GOSSIP_MENU(GOSSIP_TEXTID_RAZAEL2, creature->GetGUID());
279 player->TalkedToCreature(NPC_RAZAEL, creature->GetGUID());
280 break;
281 case GOSSIP_ACTION_INFO_DEF + 2:
282 player->SEND_GOSSIP_MENU(GOSSIP_TEXTID_LYANA2, creature->GetGUID());
283 player->TalkedToCreature(NPC_LYANA, creature->GetGUID());
284 break;
285 }
286 return true;
287 }

327 /// @todo make prisoners help (unclear if summoned or using npc's from surrounding cages (summon inside small cages?))
328 struct npc_daegarnAI : public ScriptedAI
329 {
330 npc_daegarnAI(Creature* creature) : ScriptedAI(creature)
331 {
332 Initialize();
333 }
334
335 void Initialize()
336 {
337 bEventInProgress = false;
338 uiPlayerGUID.Clear();
339 }
340
341 bool bEventInProgress;
342 ObjectGuid uiPlayerGUID;
343
344 void Reset() override
345 {
346 Initialize();
347 }
348
349 void StartEvent(ObjectGuid uiGUID)
350 {
351 if (bEventInProgress)
352 return;
353
354 uiPlayerGUID = uiGUID;
355
356 SummonGladiator(NPC_FIRJUS);
357 }
358
359 void JustSummoned(Creature* summon) override
360 {
361 if (Player* player = ObjectAccessor::GetPlayer(*me, uiPlayerGUID))
362 {
363 if (player->IsAlive())
364 {
365 summon->SetWalk(false);
366 summon->GetMotionMaster()->MovePoint(0, afCenter[0], afCenter[1], afCenter[2]);
367 summon->AI()->AttackStart(player);
368 return;
369 }
370 }
371
372 Reset();
373 }
374
375 void SummonGladiator(uint32 uiEntry)
376 {
377 me->SummonCreature(uiEntry, afSummon[0], afSummon[1], afSummon[2], 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30*IN_MILLISECONDS);
378 }
379
380 void SummonedCreatureDies(Creature* summoned, Unit* /*killer*/) override
381 {
382 uint32 uiEntry = 0;
383
384 // will eventually reset the event if something goes wrong
385 switch (summoned->GetEntry())
386 {
387 case NPC_FIRJUS: uiEntry = NPC_JLARBORN; break;
388 case NPC_JLARBORN: uiEntry = NPC_YOROS; break;
389 case NPC_YOROS: uiEntry = NPC_OLUF; break;
390 case NPC_OLUF: Reset(); return;
391 }
392
393 SummonGladiator(uiEntry);
394 }
395 };

397 CreatureAI* GetAI(Creature* creature) const override
398 {
399 return new npc_daegarnAI(creature);

408class npc_mindless_abomination : public CreatureScript
409{
410public:
411 npc_mindless_abomination() : CreatureScript("npc_mindless_abomination") { }
412
413 struct npc_mindless_abominationAI : public ScriptedAI
414 {
415 npc_mindless_abominationAI(Creature* creature) : ScriptedAI(creature) { }
416
417 void Reset() override
418 {
419 events.ScheduleEvent(EVENT_CHECK_CHARMED, 1000);
420 }
421
422 void UpdateAI(uint32 diff) override
423 {
424 events.Update(diff);
425
426 while (uint32 eventId = events.ExecuteEvent())
427 {
428 switch (eventId)
429 {
430 case EVENT_CHECK_CHARMED:
431 if (!me->IsCharmedOwnedByPlayerOrPlayer())
432 me->DespawnOrUnsummon();
433 else
434 events.ScheduleEvent(EVENT_CHECK_CHARMED, 1000);
435 break;
436 }
437 }
438 }
439
440 private:
441 EventMap events;
442 };
443
444 CreatureAI* GetAI(Creature* creature) const override
445 {
446 return new npc_mindless_abominationAI(creature);
447 }

91 boss_volkhan() : CreatureScript("boss_volkhan") { }

93 struct boss_volkhanAI : public BossAI
94 {
95 boss_volkhanAI(Creature* creature) : BossAI(creature, DATA_VOLKHAN)
96 {
97 Initialize();
98 }
99
100 void Initialize()
101 {
102 m_bIsStriking = false;
103 m_bHasTemper = false;
104 m_bCanShatterGolem = false;
105 m_uiDelay_Timer = 1000;
106 m_uiSummonPhase = 0;
107 GolemsShattered = 0;
108
109 m_uiHealthAmountModifier = 1;
110 }
111
112 void Reset() override
113 {
114 Initialize();
115 _Reset();
116 DespawnGolem();
117 m_lGolemGUIDList.clear();
118 events.SetPhase(PHASE_INTRO);
119 events.ScheduleEvent(EVENT_FORGE_CAST, 2 * IN_MILLISECONDS, 0, PHASE_INTRO);
120 }
121
122 void EnterCombat(Unit* /*who*/) override
123 {
124 Talk(SAY_AGGRO);
125 events.SetPhase(PHASE_NORMAL);
126 events.ScheduleEvent(EVENT_PAUSE, 3.5 * IN_MILLISECONDS, 0, PHASE_NORMAL);
127 events.ScheduleEvent(EVENT_SHATTERING_STOMP, 0 * IN_MILLISECONDS, 0, PHASE_NORMAL);
128 events.ScheduleEvent(EVENT_SHATTER, 5 * IN_MILLISECONDS, 0, PHASE_NORMAL);
129 _EnterCombat();
130 }
131
132 void AttackStart(Unit* who) override
133 {
134 if (me->Attack(who, true))
135 {
136 me->AddThreat(who, 0.0f);
137 me->SetInCombatWith(who);
138 who->SetInCombatWith(me);
139
140 if (!m_bHasTemper)
141 me->GetMotionMaster()->MoveChase(who);
142 }
143 }
144
145 void JustDied(Unit* /*killer*/) override
146 {
147 Talk(SAY_DEATH);
148 DespawnGolem();
149
150 _JustDied();
151 }
152
153 void KilledUnit(Unit* who) override
154 {
155 if (who->GetTypeId() == TYPEID_PLAYER)
156 Talk(SAY_SLAY);
157 }
158
159 void DespawnGolem()
160 {
161 if (m_lGolemGUIDList.empty())
162 return;
163
164 for (ObjectGuid guid : m_lGolemGUIDList)
165 {
166 if (Creature* temp = ObjectAccessor::GetCreature(*me, guid))
167 if (temp->IsAlive())
168 temp->DespawnOrUnsummon();
169 }
170
171 m_lGolemGUIDList.clear();
172 }
173
174 void ShatterGolem()
175 {
176 if (m_lGolemGUIDList.empty())
177 return;
178
179 for (ObjectGuid guid : m_lGolemGUIDList)
180 {
181 if (Creature* temp = ObjectAccessor::GetCreature(*me, guid))
182 {
183 // Only shatter brittle golems
184 if (temp->IsAlive() && temp->GetEntry() == NPC_BRITTLE_GOLEM)
185 {
186 temp->CastSpell(temp, SPELL_SHATTER, false);
187 GolemsShattered += 1;
188 }
189 }
190 }
191 }
192
193 void JustSummoned(Creature* summoned) override
194 {
195 if (summoned->GetEntry() == NPC_MOLTEN_GOLEM)
196 {
197 m_lGolemGUIDList.push_back(summoned->GetGUID());
198
199 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
200 summoned->GetMotionMaster()->MoveFollow(target, 0.0f, 0.0f);
201
202 // Why healing when just summoned?
203 summoned->CastSpell(summoned, SPELL_HEAT, false, NULL, NULL, me->GetGUID());
204 }
205 }
206
207 void JustReachedHome() override
208 {
209 if (m_uiSummonPhase == 2)
210 {
211 me->SetOrientation(2.29f);
212 m_uiSummonPhase = 3;
213 }
214 }
215
216 uint32 GetData(uint32 data) const override
217 {
218 if (data == DATA_SHATTER_RESISTANT)
219 return GolemsShattered;
220
221 return 0;
222 }
223
224 void UpdateAI(uint32 diff) override
225 {
226 // Return since we have no target and are in CombatPhase
227 if (events.IsInPhase(PHASE_NORMAL) && !UpdateVictim())
228 return;
229
230 events.Update(diff);
231
232 if (me->HasUnitState(UNIT_STATE_CASTING))
233 return;
234
235 while (uint32 eventId = events.ExecuteEvent())
236 {
237 switch (eventId)
238 {
239 case EVENT_PAUSE:
240 if (m_bIsStriking)
241 {
242 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() != CHASE_MOTION_TYPE)
243 if (me->GetVictim())
244 me->GetMotionMaster()->MoveChase(me->GetVictim());
245
246 m_bHasTemper = false;
247 m_bIsStriking = false;
248 events.ScheduleEvent(EVENT_PAUSE, 3.5 * IN_MILLISECONDS, 0, PHASE_NORMAL);
249 }
250 break;
251 case EVENT_SHATTERING_STOMP:
252 if (!m_bHasTemper && m_uiHealthAmountModifier >= 3)
253 {
254 // Should he stomp even if he has no brittle golem to shatter?
255 Talk(SAY_STOMP);
256
257 DoCast(me, SPELL_SHATTERING_STOMP);
258
259 Talk(EMOTE_SHATTER);
260 events.ScheduleEvent(EVENT_SHATTERING_STOMP, 30 * IN_MILLISECONDS, 0, PHASE_NORMAL);
261 m_bCanShatterGolem = true;
262 }
263 break;
264 case EVENT_SHATTER:
265 if (m_bCanShatterGolem)
266 {
267 ShatterGolem();
268 events.ScheduleEvent(EVENT_SHATTER, 3 * IN_MILLISECONDS, 0, PHASE_NORMAL);
269 m_bCanShatterGolem = false;
270 }
271 break;
272 case EVENT_FORGE_CAST:
273 DoCast(me, SPELL_FORGE_VISUAL);
274 events.ScheduleEvent(EVENT_FORGE_CAST, 15 * IN_MILLISECONDS, 0, PHASE_INTRO);
275 break;
276 default:
277 break;
278 }
279 }
280
281 // Health check
282 if (!m_bCanShatterGolem && me->HealthBelowPct(100 - 20 * m_uiHealthAmountModifier))
283 {
284 ++m_uiHealthAmountModifier;
285
286 if (me->IsNonMeleeSpellCast(false))
287 me->InterruptNonMeleeSpells(false);
288
289 Talk(SAY_FORGE);
290
291 m_bHasTemper = true;
292
293 m_uiSummonPhase = 1;
294 }
295
296 switch (m_uiSummonPhase)
297 {
298 case 1:
299 // 1 - Start run to Anvil
300 Talk(EMOTE_TO_ANVIL);
301 me->GetMotionMaster()->MoveTargetedHome();
302 m_uiSummonPhase = 2; // Set Next Phase
303 break;
304 case 2:
305 // 2 - Check if reached Anvil
306 // This is handled in: void JustReachedHome() override
307 break;
308 case 3:
309 // 3 - Cast Temper on the Anvil
310 if (Unit* target = GetClosestCreatureWithEntry(me, NPC_VOLKHAN_ANVIL, 1000.0f, true))
311 {
312 me->SetOrientation(2.29f);
313 DoCast(target, SPELL_TEMPER, false);
314 DoCast(target, SPELL_TEMPER_DUMMY, false);
315 }
316 m_uiDelay_Timer = 1000; // Delay 2 seconds before next phase can begin
317 m_uiSummonPhase = 4; // Set Next Phase
318 break;
319 case 4:
320 // 4 - Wait for delay to expire
321 if (m_uiDelay_Timer <= diff)
322 {
323 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0))
324 {
325 me->SetReactState(REACT_AGGRESSIVE);
326 me->SetInCombatWith(target);
327 me->GetMotionMaster()->MoveFollow(target, 0.0f, 0.0f);
328 }
329 m_uiSummonPhase = 5;
330 }
331 else
332 m_uiDelay_Timer -= diff;
333 break;
334 case 5:
335 // 5 - Spawn the Golems
336 if (Creature* creatureTarget = GetClosestCreatureWithEntry(me, NPC_VOLKHAN_ANVIL, 1000.0f, true))
337 for (uint8 i = 0; i < MAX_GOLEM; ++i)
338 me->CastSpell(creatureTarget, SPELL_SUMMON_MOLTEN_GOLEM, true);
339
340 m_bIsStriking = true;
341 m_uiSummonPhase = 0; // Reset back to Phase 0 for next time
342 break;
343 }
344
345 DoMeleeAttackIfReady();
346 }
347
348 private:
349 GuidList m_lGolemGUIDList;
350 uint32 m_uiHealthAmountModifier;
351 uint8 GolemsShattered;
352 uint32 m_uiDelay_Timer;
353 uint32 m_uiSummonPhase;
354
355 bool m_bHasTemper;
356 bool m_bIsStriking;
357 bool m_bCanShatterGolem;
358 };

362 return GetInstanceAI<boss_volkhanAI>(creature);
363 }

371class npc_molten_golem : public CreatureScript
372{
373public:
374 npc_molten_golem() : CreatureScript("npc_molten_golem") { }

376 CreatureAI* GetAI(Creature* creature) const override
377 {
378 return new npc_molten_golemAI(creature);
379 }
380
381 struct npc_molten_golemAI : public ScriptedAI
382 {
383 npc_molten_golemAI(Creature* creature) : ScriptedAI(creature)
384 {
385 Initialize();
386 }
387
388 void Initialize()
389 {
390 m_bIsFrozen = false;
391 events.ScheduleEvent(EVENT_BLAST, 20 * IN_MILLISECONDS);
392 events.ScheduleEvent(EVENT_IMMOLATION, 5 * IN_MILLISECONDS);
393 }
394
395 bool m_bIsFrozen;
396
397 void Reset() override
398 {
399 Initialize();
400 }
401
402 void AttackStart(Unit* who) override
403 {
404 if (me->Attack(who, true))
405 {
406 me->AddThreat(who, 0.0f);
407 me->SetInCombatWith(who);
408 who->SetInCombatWith(me);
409
410 if (!m_bIsFrozen)
411 me->GetMotionMaster()->MoveChase(who);
412 }
413 }
414
415 void DamageTaken(Unit* /*pDoneBy*/, uint32 &uiDamage) override
416 {
417 if (uiDamage > me->GetHealth())
418 {
419 me->UpdateEntry(NPC_BRITTLE_GOLEM);
420 me->SetHealth(1);
421 uiDamage = 0;
422 me->RemoveAllAuras();
423 me->AttackStop();
424 // me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED); //Set in DB
425 // me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE); //Set in DB
426 if (me->IsNonMeleeSpellCast(false))
427 me->InterruptNonMeleeSpells(false);
428 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
429 me->GetMotionMaster()->MovementExpired();
430 m_bIsFrozen = true;
431 }
432 }
433
434 void SpellHit(Unit* /*pCaster*/, const SpellInfo* pSpell) override
435 {
436 // This is the dummy effect of the spells
437 if (pSpell->Id == sSpellMgr->GetSpellIdForDifficulty(SPELL_SHATTER, me))
438 if (me->GetEntry() == NPC_BRITTLE_GOLEM)
439 me->DespawnOrUnsummon();
440 }
441
442 void UpdateAI(uint32 diff) override
443 {
444 // Return since we have no target or if we are frozen
445 if (!UpdateVictim() || m_bIsFrozen)
446 return;
447
448 events.Update(diff);
449
450 if (me->HasUnitState(UNIT_STATE_CASTING))
451 return;
452
453 while (uint32 eventId = events.ExecuteEvent())
454 {
455 switch (eventId)
456 {
457 case EVENT_BLAST:
458 DoCast(me, SPELL_BLAST_WAVE);
459 events.ScheduleEvent(EVENT_BLAST, 20 * IN_MILLISECONDS);
460 break;
461 case EVENT_IMMOLATION:
462 DoCastVictim(SPELL_IMMOLATION_STRIKE);
463 events.ScheduleEvent(EVENT_BLAST, 5 * IN_MILLISECONDS);
464 break;
465 default:
466 break;
467 }
468 }
469
470 DoMeleeAttackIfReady();
471 }
472
473 private:
474 EventMap events;
475 };
476};

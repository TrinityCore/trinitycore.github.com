175class npc_flash_freeze : public CreatureScript
176{
177 public:
178 npc_flash_freeze() : CreatureScript("npc_flash_freeze") { }
179
180 struct npc_flash_freezeAI : public ScriptedAI
181 {
182 npc_flash_freezeAI(Creature* creature) : ScriptedAI(creature)
183 {
184 Initialize();
185 instance = me->GetInstanceScript();
186 me->SetDisplayId(me->GetCreatureTemplate()->Modelid2);
187 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE | UNIT_FLAG_STUNNED | UNIT_FLAG_PACIFIED);
188 }
189
190 void Initialize()
191 {
192 targetGUID.Clear();
193 checkDespawnTimer = 1000;
194 }
195
196 InstanceScript* instance;
197
198 ObjectGuid targetGUID;
199 uint32 checkDespawnTimer;
200
201 void Reset() override
202 {
203 Initialize();
204 }
205
206 void UpdateAI(uint32 diff) override
207 {
208 if (!UpdateVictim()
209 || !me->GetVictim()
210 || me->EnsureVictim()->HasAura(SPELL_BLOCK_OF_ICE)
211 || me->EnsureVictim()->HasAura(SPELL_FLASH_FREEZE_HELPER))
212 return;
213
214 if (me->EnsureVictim()->GetGUID() != targetGUID || instance->GetBossState(BOSS_HODIR) != IN_PROGRESS)
215 me->DespawnOrUnsummon();
216
217 if (checkDespawnTimer <= diff)
218 {
219 if (Unit* target = ObjectAccessor::GetUnit(*me, targetGUID))
220 if (!target->IsAlive())
221 me->DisappearAndDie();
222 checkDespawnTimer = 2500;
223 }
224 else
225 checkDespawnTimer -= diff;
226 }
227
228 void IsSummonedBy(Unit* summoner) override
229 {
230 targetGUID = summoner->GetGUID();
231 me->SetInCombatWith(summoner);
232 me->AddThreat(summoner, 250.0f);
233 if (Unit* target = ObjectAccessor::GetUnit(*me, targetGUID))
234 {
235 DoCast(target, SPELL_BLOCK_OF_ICE, true);
236 // Prevents to have Ice Block on other place than target is
237 me->NearTeleportTo(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), target->GetOrientation());
238 if (target->GetTypeId() == TYPEID_PLAYER)
239 if (Creature* Hodir = ObjectAccessor::GetCreature(*me, instance->GetGuidData(BOSS_HODIR)))
240 Hodir->AI()->DoAction(ACTION_CHEESE_THE_FREEZE);
241 }
242 }
243 };
244
245 CreatureAI* GetAI(Creature* creature) const override
246 {
247 return GetInstanceAI<npc_flash_freezeAI>(creature);
248 }

311class boss_hodir : public CreatureScript
312{
313 public:
314 boss_hodir() : CreatureScript("boss_hodir") { }
315
316 struct boss_hodirAI : public BossAI
317 {
318 boss_hodirAI(Creature* creature) : BossAI(creature, BOSS_HODIR)
319 {
320 Initialize();
321 me->SetReactState(REACT_PASSIVE);
322 }
323
324 void Initialize()
325 {
326 gettingColdInHereTimer = 0;
327 gettingColdInHere = false;
328 cheeseTheFreeze = false;
329 iHaveTheCoolestFriends = false;
330 iCouldSayThatThisCacheWasRare = false;
331 }
332
333 uint32 gettingColdInHereTimer;
334
335 bool gettingColdInHere;
336 bool cheeseTheFreeze;
337 bool iHaveTheCoolestFriends;
338 bool iCouldSayThatThisCacheWasRare;
339
340 void Reset() override
341 {
342 Initialize();
343
344 _Reset();
345 me->SetReactState(REACT_PASSIVE);
346
347 for (uint8 n = 0; n < FRIENDS_COUNT; ++n)
348 if (Creature* FrozenHelper = me->SummonCreature(Entry[n], SummonPositions[n], TEMPSUMMON_MANUAL_DESPAWN))
349 FrozenHelper->CastSpell(FrozenHelper, SPELL_SUMMON_FLASH_FREEZE_HELPER, true);
350 }
351
352 void EnterCombat(Unit* /*who*/) override
353 {
354 _EnterCombat();
355 Talk(SAY_AGGRO);
356 DoCast(me, SPELL_BITING_COLD, true);
357
358 gettingColdInHereTimer = 1000;
359 gettingColdInHere = true;
360 cheeseTheFreeze = true;
361 iHaveTheCoolestFriends = true;
362 iCouldSayThatThisCacheWasRare = true;
363
364 events.ScheduleEvent(EVENT_ICICLE, 2000);
365 events.ScheduleEvent(EVENT_FREEZE, 25000);
366 events.ScheduleEvent(EVENT_BLOWS, urand(60000, 65000));
367 events.ScheduleEvent(EVENT_FLASH_FREEZE, 45000);
368 events.ScheduleEvent(EVENT_RARE_CACHE, 180000);
369 events.ScheduleEvent(EVENT_BERSERK, 480000);
370 }
371
372 void KilledUnit(Unit* who) override
373 {
374 if (who->GetTypeId() == TYPEID_PLAYER)
375 Talk(SAY_SLAY);
376 }
377
378 void DamageTaken(Unit* /*who*/, uint32& damage) override
379 {
380 if (damage >= me->GetHealth())
381 {
382 damage = 0;
383 Talk(SAY_DEATH);
384 if (iCouldSayThatThisCacheWasRare)
385 instance->SetData(DATA_HODIR_RARE_CACHE, 1);
386
387 me->RemoveAllAuras();
388 me->RemoveAllAttackers();
389 me->AttackStop();
390 me->SetReactState(REACT_PASSIVE);
391 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE);
392 me->InterruptNonMeleeSpells(true);
393 me->StopMoving();
394 me->GetMotionMaster()->Clear();
395 me->GetMotionMaster()->MoveIdle();
396 me->SetControlled(true, UNIT_STATE_STUNNED);
397 me->CombatStop(true);
398
399 DoCastAOE(SPELL_KILL_CREDIT, true); /// need to be cast before changing boss faction
400 /// spell will target enemies only
401 me->setFaction(35);
402 me->DespawnOrUnsummon(10000);
403
404 _JustDied();
405 }
406 }
407
408 void UpdateAI(uint32 diff) override
409 {
410 if (!UpdateVictim())
411 return;
412
413 events.Update(diff);
414
415 if (me->HasUnitState(UNIT_STATE_CASTING))
416 return;
417
418 while (uint32 eventId = events.ExecuteEvent())
419 {
420 switch (eventId)
421 {
422 case EVENT_FREEZE:
423 DoCastAOE(SPELL_FREEZE);
424 events.ScheduleEvent(EVENT_FREEZE, urand(30000, 45000));
425 break;
426 case EVENT_ICICLE:
427 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
428 DoCast(target, SPELL_ICICLE);
429 events.ScheduleEvent(EVENT_ICICLE, RAID_MODE(5500, 3500));
430 break;
431 case EVENT_FLASH_FREEZE:
432 Talk(SAY_FLASH_FREEZE);
433 Talk(EMOTE_FREEZE);
434 for (uint8 n = 0; n < RAID_MODE(2, 3); ++n)
435 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
436 target->CastSpell(target, SPELL_ICICLE_SNOWDRIFT, true);
437 DoCast(SPELL_FLASH_FREEZE);
438 events.ScheduleEvent(EVENT_FLASH_FREEZE_EFFECT, 500);
439 break;
440 case EVENT_FLASH_FREEZE_EFFECT:
441 {
442 std::list<Creature*> IcicleSnowdriftList;
443 GetCreatureListWithEntryInGrid(IcicleSnowdriftList, me, NPC_SNOWPACKED_ICICLE, 100.0f);
444 for (std::list<Creature*>::iterator itr = IcicleSnowdriftList.begin(); itr != IcicleSnowdriftList.end(); ++itr)
445 (*itr)->CastSpell(me, SPELL_FLASH_FREEZE_VISUAL, true);
446 FlashFreeze();
447 events.CancelEvent(EVENT_FLASH_FREEZE_EFFECT);
448 events.ScheduleEvent(EVENT_FLASH_FREEZE, urand(25000, 35000));
449 break;
450 }
451 case EVENT_BLOWS:
452 Talk(SAY_STALACTITE);
453 Talk(EMOTE_BLOWS);
454 DoCast(me, SPELL_FROZEN_BLOWS);
455 events.ScheduleEvent(EVENT_BLOWS, urand(60000, 65000));
456 break;
457 case EVENT_RARE_CACHE:
458 Talk(SAY_HARD_MODE_FAILED);
459 iCouldSayThatThisCacheWasRare = false;
460 instance->SetData(DATA_HODIR_RARE_CACHE, 0);
461 events.CancelEvent(EVENT_RARE_CACHE);
462 break;
463 case EVENT_BERSERK:
464 Talk(SAY_BERSERK);
465 DoCast(me, SPELL_BERSERK, true);
466 events.CancelEvent(EVENT_BERSERK);
467 break;
468 }
469 }
470
471 if (gettingColdInHereTimer <= diff && gettingColdInHere)
472 {
473 std::list<HostileReference*> ThreatList = me->getThreatManager().getThreatList();
474 for (std::list<HostileReference*>::const_iterator itr = ThreatList.begin(); itr != ThreatList.end(); ++itr)
475 if (Unit* target = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid()))
476 if (Aura* BitingColdAura = target->GetAura(SPELL_BITING_COLD_TRIGGERED))
477 if ((target->GetTypeId() == TYPEID_PLAYER) && (BitingColdAura->GetStackAmount() > 2))
478 SetData(DATA_GETTING_COLD_IN_HERE, 0);
479 gettingColdInHereTimer = 1000;
480 }
481 else
482 gettingColdInHereTimer -= diff;
483
484 DoMeleeAttackIfReady();
485 }
486
487 void DoAction(int32 action) override
488 {
489 switch (action)
490 {
491 case ACTION_I_HAVE_THE_COOLEST_FRIENDS:
492 iHaveTheCoolestFriends = false;
493 break;
494 case ACTION_CHEESE_THE_FREEZE:
495 cheeseTheFreeze = false;
496 break;
497 default:
498 break;
499 }
500 }
501
502 void FlashFreeze()
503 {
504 std::list<Unit*> TargetList;
505 Trinity::AnyUnfriendlyUnitInObjectRangeCheck checker(me, me, 100.0f);
506 Trinity::UnitListSearcher<Trinity::AnyUnfriendlyUnitInObjectRangeCheck> searcher(me, TargetList, checker);
507 me->VisitNearbyObject(100.0f, searcher);
508 for (std::list<Unit*>::iterator itr = TargetList.begin(); itr != TargetList.end(); ++itr)
509 {
510 Unit* target = *itr;
511 if (!target || !target->IsAlive() || GetClosestCreatureWithEntry(target, NPC_SNOWPACKED_ICICLE, 5.0f))
512 continue;
513
514 if (target->HasAura(SPELL_FLASH_FREEZE_HELPER) || target->HasAura(SPELL_BLOCK_OF_ICE))
515 {
516 me->CastSpell(target, SPELL_FLASH_FREEZE_KILL, true);
517 continue;
518 }
519
520 target->CastSpell(target, SPELL_SUMMON_BLOCK_OF_ICE, true);
521 }
522 }
523 };
524
525 CreatureAI* GetAI(Creature* creature) const override
526 {
527 return GetUlduarAI<boss_hodirAI>(creature);
528 };

643 void Reset() override
644 {
645 events.Reset();
646 events.ScheduleEvent(EVENT_HEAL, urand(4000, 8000));
647 events.ScheduleEvent(EVENT_DISPEL_MAGIC, urand(15000, 20000));
648 }
649
650 void UpdateAI(uint32 diff) override
651 {
652 if (!UpdateVictim() || me->HasUnitState(UNIT_STATE_STUNNED) || me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED))
653 return;
654
655 events.Update(diff);
656
657 if (me->HasUnitState(UNIT_STATE_CASTING))
658 return;
659
660 if (HealthBelowPct(30))
661 DoCast(me, SPELL_GREATER_HEAL);
662
663 while (uint32 eventId = events.ExecuteEvent())
664 {
665 switch (eventId)
666 {
667 case EVENT_HEAL:
668 DoCastAOE(SPELL_GREATER_HEAL);
669 events.ScheduleEvent(EVENT_HEAL, urand(7500, 10000));
670 break;
671 case EVENT_DISPEL_MAGIC:
672 {
673 std::list<Unit*> TargetList;
674 Trinity::AnyFriendlyUnitInObjectRangeCheck checker(me, me, 30.0f);
675 Trinity::UnitListSearcher<Trinity::AnyFriendlyUnitInObjectRangeCheck> searcher(me, TargetList, checker);
676 me->VisitNearbyObject(30.0f, searcher);
677 for (std::list<Unit*>::iterator itr = TargetList.begin(); itr != TargetList.end(); ++itr)
678 if ((*itr)->HasAura(SPELL_FREEZE))
679 DoCast(*itr, SPELL_DISPEL_MAGIC, true);
680 events.ScheduleEvent(EVENT_DISPEL_MAGIC, urand(15000, 20000));
681 break;
682 }
683 default:
684 break;
685 }
686 }
687
688 DoSpellAttackIfReady(SPELL_SMITE);
689 }
690
691 void JustDied(Unit* /*killer*/) override

947{
948 public:
949 spell_biting_cold() : SpellScriptLoader("spell_biting_cold") { }
950
951 class spell_biting_cold_AuraScript : public AuraScript
952 {
953 PrepareAuraScript(spell_biting_cold_AuraScript);
954
955 void HandleEffectPeriodic(AuraEffect const* /*aurEff*/)
956 {
957 Unit* target = GetTarget();
958 bool found = false;
959
960 for (TargetList::iterator itr = listOfTargets.begin(); itr != listOfTargets.end(); ++itr)
961 {
962 if (itr->first != target->GetGUID())
963 continue;
964
965 if (itr->second >= 4)
966 {
967 target->CastSpell(target, SPELL_BITING_COLD_TRIGGERED, true);
968 itr->second = 1;
969 }
970 else
971 {
972 if (target->isMoving())
973 itr->second = 1;
974 else
975 itr->second++;
976 }
977
978 found = true;
979 break;
980 }
981
982 if (!found)
983 listOfTargets.push_back(std::make_pair(target->GetGUID(), 1));
984 }
985
986 void Register() override
987 {
988 OnEffectPeriodic += AuraEffectPeriodicFn(spell_biting_cold_AuraScript::HandleEffectPeriodic, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
989 }
990
991 private:
992 typedef std::list<std::pair<ObjectGuid, uint8>> TargetList;
993 TargetList listOfTargets;
994 };
995
996 AuraScript* GetAuraScript() const override
997 {
998 return new spell_biting_cold_AuraScript();
999 }

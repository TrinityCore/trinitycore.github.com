447class PlayerOrPetCheck
448{
449 public:
450 bool operator()(WorldObject* object) const
451 {
452 if (object->GetTypeId() == TYPEID_PLAYER)
453 return false;
454
455 if (Creature* creature = object->ToCreature())
456 return !creature->IsPet();
457
458 return true;
459 }

106 boss_general_vezaxAI(Creature* creature) : BossAI(creature, BOSS_VEZAX)

111 void Initialize()
112 {
113 shadowDodger = true;
114 smellSaronite = true;
115 animusDead = false;
116 vaporCount = 0;
117 }
118
119 bool shadowDodger;
120 bool smellSaronite; // HardMode
121 bool animusDead; // Check against getting a HardMode achievement before killing Saronite Animus
122 uint8 vaporCount;
123
124 void Reset() override
125 {
126 _Reset();
127
128 Initialize();
129 }
130
131 void EnterCombat(Unit* /*who*/) override
132 {
133 _EnterCombat();
134
135 Talk(SAY_AGGRO);
136 DoCast(me, SPELL_AURA_OF_DESPAIR);
137 CheckShamanisticRage();
138
139 events.ScheduleEvent(EVENT_SHADOW_CRASH, urand(8000, 10000));
140 events.ScheduleEvent(EVENT_SEARING_FLAMES, 12000);
141 events.ScheduleEvent(EVENT_MARK_OF_THE_FACELESS, urand(35000, 40000));
142 events.ScheduleEvent(EVENT_SARONITE_VAPORS, 30000);
143 events.ScheduleEvent(EVENT_SURGE_OF_DARKNESS, 60000);
144 events.ScheduleEvent(EVENT_BERSERK, 600000);
145 }
146
147 void UpdateAI(uint32 diff) override
148 {
149 if (!UpdateVictim())
150 return;
151
152 events.Update(diff);
153
154 if (me->HasUnitState(UNIT_STATE_CASTING))
155 return;
156
157 while (uint32 eventId = events.ExecuteEvent())
158 {
159 switch (eventId)
160 {
161 case EVENT_SHADOW_CRASH:
162 {
163 Unit* target = CheckPlayersInRange(RAID_MODE<uint8>(4, 9), 15.0f, 50.0f);
164 if (!target)
165 target = SelectTarget(SELECT_TARGET_RANDOM, 0, 150.0f, true);
166 if (target)
167 DoCast(target, SPELL_SHADOW_CRASH);
168 events.ScheduleEvent(EVENT_SHADOW_CRASH, urand(8000, 12000));
169 break;
170 }
171 case EVENT_SEARING_FLAMES:
172 DoCastAOE(SPELL_SEARING_FLAMES);
173 events.ScheduleEvent(EVENT_SEARING_FLAMES, urand(14000, 17500));
174 break;
175 case EVENT_MARK_OF_THE_FACELESS:
176 {
177 Unit* target = CheckPlayersInRange(RAID_MODE<uint8>(4, 9), 15.0f, 50.0f);
178 if (!target)
179 target = SelectTarget(SELECT_TARGET_RANDOM, 0, 150.0f, true);
180 if (target)
181 DoCast(target, SPELL_MARK_OF_THE_FACELESS);
182 events.ScheduleEvent(EVENT_MARK_OF_THE_FACELESS, urand(35000, 45000));
183 break;
184 }
185 case EVENT_SURGE_OF_DARKNESS:
186 Talk(EMOTE_SURGE_OF_DARKNESS);
187 Talk(SAY_SURGE_OF_DARKNESS);
188 DoCast(me, SPELL_SURGE_OF_DARKNESS);
189 events.ScheduleEvent(EVENT_SURGE_OF_DARKNESS, urand(50000, 70000));
190 break;
191 case EVENT_SARONITE_VAPORS:
192 DoCast(SPELL_SUMMON_SARONITE_VAPORS);
193 events.ScheduleEvent(EVENT_SARONITE_VAPORS, urand(30000, 35000));
194 if (++vaporCount == 6 && smellSaronite)
195 {
196 Talk(SAY_HARDMODE);
197 Talk(EMOTE_BARRIER);
198 summons.DespawnAll();
199 DoCast(me, SPELL_SARONITE_BARRIER);
200 DoCast(SPELL_SUMMON_SARONITE_ANIMUS);
201 me->AddLootMode(LOOT_MODE_HARD_MODE_1);
202 events.CancelEvent(EVENT_SARONITE_VAPORS);
203 events.CancelEvent(EVENT_SEARING_FLAMES);
204 }
205 break;
206 case EVENT_BERSERK:
207 Talk(SAY_BERSERK);
208 DoCast(me, SPELL_BERSERK);
209 break;
210 }
211 }
212
213 DoMeleeAttackIfReady();
214 }
215
216 void SpellHitTarget(Unit* who, SpellInfo const* spell) override
217 {
218 if (who && who->GetTypeId() == TYPEID_PLAYER && spell->Id == SPELL_SHADOW_CRASH_HIT)
219 shadowDodger = false;
220 }
221
222 void KilledUnit(Unit* who) override
223 {
224 if (who->GetTypeId() == TYPEID_PLAYER)
225 Talk(SAY_SLAY);
226 }
227
228 void JustDied(Unit* /*killer*/) override
229 {
230 _JustDied();
231 Talk(SAY_DEATH);
232 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_AURA_OF_DESPAIR);
233 }
234
235 void CheckShamanisticRage()
236 {
237 // If Shaman has Shamanistic Rage and use it during the fight, it will cast Corrupted Rage on him
238 Map::PlayerList const& Players = me->GetMap()->GetPlayers();
239 for (Map::PlayerList::const_iterator itr = Players.begin(); itr != Players.end(); ++itr)
240 if (Player* player = itr->GetSource())
241 if (player->HasSpell(SPELL_SHAMANTIC_RAGE))
242 player->CastSpell(player, SPELL_CORRUPTED_RAGE, false);
243 }
244
245 uint32 GetData(uint32 type) const override
246 {
247 switch (type)
248 {
249 case DATA_SHADOWDODGER:
250 return shadowDodger ? 1 : 0;
251 case DATA_SMELL_SARONITE:
252 return smellSaronite ? 1 : 0;
253 }
254
255 return 0;
256 }
257
258 void DoAction(int32 action) override
259 {
260 switch (action)
261 {
262 case ACTION_VAPORS_DIE:
263 smellSaronite = false;
264 break;
265 case ACTION_ANIMUS_DIE:
266 me->RemoveAurasDueToSpell(SPELL_SARONITE_BARRIER);
267 events.ScheduleEvent(EVENT_SEARING_FLAMES, urand(7000, 12000));
268 animusDead = true;
269 break;
270 }
271 }
272
273 /* Player Range Check
274 Purpose: If there are playersMin people within rangeMin, rangeMax: return a random players in that range.
275 If not, return NULL and allow other target selection
276 */
277 Unit* CheckPlayersInRange(uint8 playersMin, float rangeMin, float rangeMax)
278 {
279 std::list<Player*> PlayerList;
280 Map::PlayerList const& Players = me->GetMap()->GetPlayers();
281 for (Map::PlayerList::const_iterator itr = Players.begin(); itr != Players.end(); ++itr)
282 {
283 if (Player* player = itr->GetSource())
284 {
285 float distance = player->GetDistance(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
286 if (rangeMin > distance || distance > rangeMax)
287 continue;
288
289 PlayerList.push_back(player);
290 }
291 }
292
293 if (PlayerList.empty())
294 return NULL;
295
296 size_t size = PlayerList.size();
297 if (size < playersMin)
298 return NULL;
299
300 return Trinity::Containers::SelectRandomContainerElement(PlayerList);
301 }
302 };

304 CreatureAI* GetAI(Creature* creature) const override
305 {
306 return GetUlduarAI<boss_general_vezaxAI>(creature);
307 }

384 me->SetReactState(REACT_PASSIVE);
385 }
386
387 void Reset() override
388 {
389 events.Reset();
390 events.ScheduleEvent(EVENT_RANDOM_MOVE, urand(5000, 7500));
391 }
392
393 void UpdateAI(uint32 diff) override
394 {
395 events.Update(diff);
396
397 while (uint32 eventId = events.ExecuteEvent())
398 {
399 switch (eventId)
400 {
401 case EVENT_RANDOM_MOVE:
402 me->GetMotionMaster()->MoveRandom(30.0f);
403 events.ScheduleEvent(EVENT_RANDOM_MOVE, urand(5000, 7500));
404 break;
405 default:
406 break;
407 }
408 }
409 }
410
411 void DamageTaken(Unit* /*who*/, uint32& damage) override
412 {
413 // This can't be on JustDied. In 63322 dummy handler caster needs to be this NPC
414 // if caster == target then damage mods will increase the damage taken
415 if (damage >= me->GetHealth())
416 {
417 damage = 0;
418 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_DISABLE_MOVE);
419 me->SetStandState(UNIT_STAND_STATE_DEAD);
420 me->SetHealth(me->GetMaxHealth());
421 me->RemoveAllAuras();
422 DoCast(me, SPELL_SARONITE_VAPORS);
423 me->DespawnOrUnsummon(30000);
424
425 if (Creature* Vezax = ObjectAccessor::GetCreature(*me, instance->GetGuidData(BOSS_VEZAX)))
426 Vezax->AI()->DoAction(ACTION_VAPORS_DIE);
427 }
428 }
429
430 private:
431 InstanceScript* instance;
432 EventMap events;
433 };

513 bool Validate(SpellInfo const* /*spell*/) override
514 {
515 if (!sSpellMgr->GetSpellInfo(SPELL_SARONITE_VAPORS_ENERGIZE) || !sSpellMgr->GetSpellInfo(SPELL_SARONITE_VAPORS_DAMAGE))
516 return false;

518 }
519
520 void HandleEffectApply(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
521 {
522 if (Unit* caster = GetCaster())
523 {
524 int32 mana = int32(aurEff->GetAmount() * std::pow(2.0f, GetStackAmount())); // mana restore - bp * 2^stackamount
525 int32 damage = mana * 2;
526 caster->CastCustomSpell(GetTarget(), SPELL_SARONITE_VAPORS_ENERGIZE, &mana, NULL, NULL, true);
527 caster->CastCustomSpell(GetTarget(), SPELL_SARONITE_VAPORS_DAMAGE, &damage, NULL, NULL, true);
528 }
529 }
530
531 void Register() override
532 {
533 AfterEffectApply += AuraEffectApplyFn(spell_general_vezax_saronite_vapors_AuraScript::HandleEffectApply, EFFECT_0, SPELL_AURA_DUMMY, AURA_EFFECT_HANDLE_REAL_OR_REAPPLY_MASK);
534 }

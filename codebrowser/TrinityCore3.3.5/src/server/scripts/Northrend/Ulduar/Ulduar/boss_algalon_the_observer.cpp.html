324 {
325 if (victim->GetTypeId() == TYPEID_PLAYER)
326 {
327 _fedOnTears = true;
328 if (!_hasYelled)

337 void DoAction(int32 action) override
338 {
339 switch (action)
340 {
341 case ACTION_START_INTRO:
342 {
343 me->SetFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_INSTANTLY_APPEAR_MODEL);
344 me->SetDisableGravity(true);
345 DoCast(me, SPELL_ARRIVAL, true);
346 DoCast(me, SPELL_RIDE_THE_LIGHTNING, true);
347 me->GetMotionMaster()->MovePoint(POINT_ALGALON_LAND, AlgalonLandPos);
348 me->SetHomePosition(AlgalonLandPos);
349 Movement::MoveSplineInit init(me);
350 init.MoveTo(AlgalonLandPos.GetPositionX(), AlgalonLandPos.GetPositionY(), AlgalonLandPos.GetPositionZ(), false);
351 init.SetOrientationFixed(true);
352 init.Launch();
353 events.Reset();
354 events.SetPhase(PHASE_ROLE_PLAY);
355 events.ScheduleEvent(EVENT_INTRO_1, 5000, 0, PHASE_ROLE_PLAY);
356 events.ScheduleEvent(EVENT_INTRO_2, 15000, 0, PHASE_ROLE_PLAY);
357 events.ScheduleEvent(EVENT_INTRO_3, 23000, 0, PHASE_ROLE_PLAY);
358 events.ScheduleEvent(EVENT_INTRO_FINISH, 36000, 0, PHASE_ROLE_PLAY);
359 break;
360 }
361 case ACTION_ASCEND:
362 events.SetPhase(PHASE_BIG_BANG);
363 events.CancelEvent(EVENT_RESUME_UPDATING);
364 events.ScheduleEvent(EVENT_ASCEND_TO_THE_HEAVENS, 1500);
365 break;
366 case EVENT_DESPAWN_ALGALON:
367 events.Reset();
368 events.SetPhase(PHASE_ROLE_PLAY);
369 if (me->IsInCombat())
370 events.ScheduleEvent(EVENT_ASCEND_TO_THE_HEAVENS, 1);
371 events.ScheduleEvent(EVENT_DESPAWN_ALGALON_1, 5000);
372 events.ScheduleEvent(EVENT_DESPAWN_ALGALON_2, 17000);
373 events.ScheduleEvent(EVENT_DESPAWN_ALGALON_3, 26000);
374 me->DespawnOrUnsummon(34000);
375 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_IMMUNE_TO_NPC);
376 break;
377 case ACTION_INIT_ALGALON:
378 _firstPull = false;
379 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
380 break;
381 }

406 if (Creature* brann = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_BRANN_BRONZEBEARD_ALG)))
407 brann->AI()->DoAction(ACTION_FINISH_INTRO);
408
409 me->setActive(true);
410 DoZoneInCombat();
411 introDelay = 26000;
412 summons.DespawnEntry(NPC_AZEROTH);
413 instance->SetData(EVENT_DESPAWN_ALGALON, 0);
414 events.ScheduleEvent(EVENT_START_COMBAT, 18000);
415 }
416
417 events.ScheduleEvent(EVENT_INTRO_TIMER_DONE, introDelay);
418 events.ScheduleEvent(EVENT_QUANTUM_STRIKE, 3500 + introDelay);

424

426 me->GetCreatureListWithEntryInGrid(stalkers, NPC_ALGALON_STALKER, 200.0f);

428 (*itr)->m_Events.KillAllEvents(true);
429 }

454 {
455 summons.Summon(summon);
456 switch (summon->GetEntry())
457 {
458 case NPC_AZEROTH:
459 DoCastAOE(SPELL_REORIGINATION, true);

461 case NPC_COLLAPSING_STAR:
462 summon->SetReactState(REACT_PASSIVE);

464 summon->CastSpell(summon, SPELL_COLLAPSE, TRIGGERED_FULL_MASK);
465 break;
466 case NPC_BLACK_HOLE:
467 summon->SetReactState(REACT_PASSIVE);
468 summon->CastSpell((Unit*)NULL, SPELL_BLACK_HOLE_TRIGGER, TRIGGERED_FULL_MASK);

470 summon->CastSpell((Unit*)NULL, SPELL_BLACK_HOLE_EXPLOSION);
471 summon->CastSpell(summon, SPELL_SUMMON_VOID_ZONE_VISUAL, TRIGGERED_FULL_MASK);
472 break;
473 case NPC_ALGALON_VOID_ZONE_VISUAL_STALKER:
474 summon->CastSpell(summon, SPELL_VOID_ZONE_VISUAL, TRIGGERED_FULL_MASK);
475 break;
476 case NPC_ALGALON_STALKER_ASTEROID_TARGET_01:
477 summon->CastSpell(summon, SPELL_COSMIC_SMASH_VISUAL_STATE, TRIGGERED_FULL_MASK);
478 break;
479 case NPC_ALGALON_STALKER_ASTEROID_TARGET_02:
480 summon->m_Events.AddEvent(new CosmicSmashDamageEvent(summon), summon->m_Events.CalculateTime(3250));
481 break;
482 case NPC_WORM_HOLE:

487 case NPC_UNLEASHED_DARK_MATTER:
488 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me)))

491 break;
492 }

503 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
504 {
505 if (_fightWon)
506 {
507 damage = 0;
508 return;
509 }
510
511 if (!_phaseTwo && me->HealthBelowPctDamaged(20, damage))
512 {
513 _phaseTwo = true;
514 Talk(SAY_ALGALON_PHASE_TWO);
515 summons.DespawnEntry(NPC_LIVING_CONSTELLATION);
516 summons.DespawnEntry(NPC_COLLAPSING_STAR);
517 summons.DespawnEntry(NPC_BLACK_HOLE);
518 summons.DespawnEntry(NPC_ALGALON_VOID_ZONE_VISUAL_STALKER);
519 events.CancelEvent(EVENT_SUMMON_COLLAPSING_STAR);
520 std::list<Creature*> stalkers;
521 me->GetCreatureListWithEntryInGrid(stalkers, NPC_ALGALON_STALKER, 200.0f);
522 for (std::list<Creature*>::iterator itr = stalkers.begin(); itr != stalkers.end(); ++itr)
523 (*itr)->m_Events.KillAllEvents(true);
524 for (uint32 i = 0; i < COLLAPSING_STAR_COUNT; ++i)
525 if (Creature* wormHole = DoSummon(NPC_WORM_HOLE, CollapsingStarPos[i], TEMPSUMMON_MANUAL_DESPAWN))
526 wormHole->m_Events.AddEvent(new SummonUnleashedDarkMatter(wormHole), wormHole->m_Events.CalculateTime(i >= 2 ? 8000 : 6000));
527 }
528 else if ((int32(me->GetHealth()) - int32(damage)) < CalculatePct<int32>(int32(me->GetMaxHealth()), 2.5f) && !_fightWon)
529 {
530 _fightWon = true;
531 damage = 0;
532 me->SetReactState(REACT_PASSIVE);
533 me->AttackStop();
534 me->setFaction(35);
535 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
536 DoCast(me, SPELL_SELF_STUN);
537 events.Reset();
538 summons.DespawnAll();
539 events.SetPhase(PHASE_ROLE_PLAY);
540 events.ScheduleEvent(EVENT_OUTRO_START, 1500);
541 events.ScheduleEvent(EVENT_OUTRO_1, 7200);
542 events.ScheduleEvent(EVENT_OUTRO_2, 8700);
543 }

546 void UpdateAI(uint32 diff) override
547 {
548 if (!(events.IsInPhase(PHASE_ROLE_PLAY) || events.IsInPhase(PHASE_BIG_BANG)) && !UpdateVictim())
549 return;
550
551 events.Update(diff);
552
553 if (!events.IsInPhase(PHASE_ROLE_PLAY))
554 if (me->HasUnitState(UNIT_STATE_CASTING))
555 return;
556
557 while (uint32 eventId = events.ExecuteEvent())
558 {
559 switch (eventId)
560 {
561 case EVENT_INTRO_1:
562 me->RemoveAurasDueToSpell(SPELL_RIDE_THE_LIGHTNING);
563 Talk(SAY_ALGALON_INTRO_1);
564 break;
565 case EVENT_INTRO_2:
566 DoCastAOE(SPELL_SUMMON_AZEROTH, true);
567 Talk(SAY_ALGALON_INTRO_2);
568 break;
569 case EVENT_INTRO_3:
570 Talk(SAY_ALGALON_INTRO_3);
571 break;
572 case EVENT_INTRO_FINISH:
573 events.Reset();
574 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
575 break;
576 case EVENT_START_COMBAT:
577 instance->SetBossState(BOSS_ALGALON, IN_PROGRESS);
578 break;
579 case EVENT_INTRO_TIMER_DONE:
580 {
581 events.SetPhase(PHASE_NORMAL);
582 me->SetSheath(SHEATH_STATE_MELEE);
583 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_IMMUNE_TO_NPC);
584 me->SetReactState(REACT_DEFENSIVE);
585 DoCastAOE(SPELL_SUPERMASSIVE_FAIL, true);
586 //! Workaround for Creature::_IsTargetAcceptable returning false
587 //! for creatures that start combat in REACT_PASSIVE and UNIT_FLAG_NOT_SELECTABLE

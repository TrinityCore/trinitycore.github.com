195 struct boss_xt002_AI : public BossAI
196 {
197 boss_xt002_AI(Creature* creature) : BossAI(creature, BOSS_XT002)
198 {
199 Initialize();
200 _transferHealth = 0;
201 }
202
203 void Initialize()
204 {
205 _healthRecovered = false;
206 _gravityBombCasualty = false;
207 _hardMode = false;
208
209 _phase = 1;
210 _heartExposed = 0;
211 }
212
213 void Reset() override
214 {
215 _Reset();
216
217 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
218 me->SetReactState(REACT_AGGRESSIVE);
219 DoCast(me, SPELL_STAND);
220
221 Initialize();
222
223 instance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_MUST_DECONSTRUCT_FASTER);
224 }
225
226 void EnterCombat(Unit* /*who*/) override
227 {
228 Talk(SAY_AGGRO);
229 _EnterCombat();
230
231 events.ScheduleEvent(EVENT_ENRAGE, TIMER_ENRAGE);
232 events.ScheduleEvent(EVENT_GRAVITY_BOMB, TIMER_GRAVITY_BOMB);
233 events.ScheduleEvent(EVENT_SEARING_LIGHT, TIMER_SEARING_LIGHT);
234 //Tantrum is cast a bit slower the first time.
235 events.ScheduleEvent(EVENT_TYMPANIC_TANTRUM, urand(TIMER_TYMPANIC_TANTRUM_MIN, TIMER_TYMPANIC_TANTRUM_MAX) * 2);
236
237 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_MUST_DECONSTRUCT_FASTER);
238 }
239
240 void DoAction(int32 action) override
241 {
242 switch (action)
243 {
244 case ACTION_ENTER_HARD_MODE:
245 events.ScheduleEvent(EVENT_ENTER_HARD_MODE, 1);
246 break;
247 }
248 }
249
250 void KilledUnit(Unit* who) override
251 {
252 if (who->GetTypeId() == TYPEID_PLAYER)
253 Talk(SAY_SLAY);
254 }
255
256 void JustDied(Unit* /*killer*/) override
257 {
258 Talk(SAY_DEATH);
259 _JustDied();
260 }
261
262 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
263 {
264 if (!_hardMode && _phase == 1 && !HealthAbovePct(100 - 25 * (_heartExposed+1)))
265 ExposeHeart();
266 }
267
268 void UpdateAI(uint32 diff) override
269 {
270 if (!UpdateVictim())
271 return;
272
273 events.Update(diff);
274
275 if (me->HasUnitState(UNIT_STATE_CASTING))
276 return;
277
278 while (uint32 eventId = events.ExecuteEvent())
279 {
280 switch (eventId)
281 {
282 case EVENT_SEARING_LIGHT:
283 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
284 DoCast(target, SPELL_SEARING_LIGHT);
285
286 events.ScheduleEvent(EVENT_SEARING_LIGHT, TIMER_SEARING_LIGHT);
287 break;
288 case EVENT_GRAVITY_BOMB:
289 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
290 DoCast(target, SPELL_GRAVITY_BOMB);
291
292 events.ScheduleEvent(EVENT_GRAVITY_BOMB, TIMER_GRAVITY_BOMB);
293 break;
294 case EVENT_TYMPANIC_TANTRUM:
295 Talk(SAY_TYMPANIC_TANTRUM);
296 Talk(EMOTE_TYMPANIC_TANTRUM);
297 DoCast(SPELL_TYMPANIC_TANTRUM);
298 events.ScheduleEvent(EVENT_TYMPANIC_TANTRUM, urand(TIMER_TYMPANIC_TANTRUM_MIN, TIMER_TYMPANIC_TANTRUM_MAX));
299 break;
300 case EVENT_DISPOSE_HEART:
301 SetPhaseOne();
302 break;
303 case EVENT_ENRAGE:
304 Talk(SAY_BERSERK);
305 DoCast(me, SPELL_ENRAGE);
306 break;
307 case EVENT_ENTER_HARD_MODE:
308 me->SetFullHealth();
309 DoCast(me, SPELL_HEARTBREAK, true);
310 me->AddLootMode(LOOT_MODE_HARD_MODE_1);
311 _hardMode = true;
312 SetPhaseOne();
313 break;
314 }
315 }
316
317 if (_phase == 1)
318 DoMeleeAttackIfReady();
319 }
320
321 void PassengerBoarded(Unit* who, int8 /*seatId*/, bool apply) override
322 {
323 if (apply && who->GetEntry() == NPC_XS013_SCRAPBOT)
324 {
325 // Need this so we can properly determine when to expose heart again in damagetaken hook
326 if (me->GetHealthPct() > (25 * (4 - _heartExposed)))
327 ++_heartExposed;
328
329 Talk(EMOTE_SCRAPBOT);
330 _healthRecovered = true;
331 }
332 }
333
334 uint32 GetData(uint32 type) const override
335 {
336 switch (type)
337 {
338 case DATA_HARD_MODE:
339 return _hardMode ? 1 : 0;
340 case DATA_HEALTH_RECOVERED:
341 return _healthRecovered ? 1 : 0;
342 case DATA_GRAVITY_BOMB_CASUALTY:
343 return _gravityBombCasualty ? 1 : 0;
344 }
345
346 return 0;
347 }
348
349 void SetData(uint32 type, uint32 data) override
350 {
351 switch (type)
352 {
353 case DATA_TRANSFERED_HEALTH:
354 _transferHealth = data;
355 break;
356 case DATA_GRAVITY_BOMB_CASUALTY:
357 _gravityBombCasualty = (data > 0) ? true : false;
358 break;
359 }
360 }
361
362 void ExposeHeart()
363 {
364 Talk(SAY_HEART_OPENED);
365 Talk(EMOTE_HEART_OPENED);
366
367 DoCast(me, SPELL_SUBMERGE); // WIll make creature untargetable
368 me->AttackStop();
369 me->SetReactState(REACT_PASSIVE);
370
371 Unit* heart = me->GetVehicleKit() ? me->GetVehicleKit()->GetPassenger(HEART_VEHICLE_SEAT_NORMAL) : NULL;
372 if (heart)
373 {
374 heart->CastSpell(heart, SPELL_HEART_OVERLOAD, false);
375 heart->CastSpell(me, SPELL_HEART_LIGHTNING_TETHER, false);
376 heart->CastSpell(heart, SPELL_HEART_HEAL_TO_FULL, true);
377 heart->CastSpell(heart, SPELL_EXPOSED_HEART, false); // Channeled
378 heart->ChangeSeat(HEART_VEHICLE_SEAT_EXPOSED, true);
379 heart->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
380 heart->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNK_29);
381 }
382
383 events.CancelEvent(EVENT_SEARING_LIGHT);
384 events.CancelEvent(EVENT_GRAVITY_BOMB);
385 events.CancelEvent(EVENT_TYMPANIC_TANTRUM);
386
387 // Start "end of phase 2 timer"
388 events.ScheduleEvent(EVENT_DISPOSE_HEART, TIMER_HEART_PHASE);
389
390 // Phase 2 has officially started
391 _phase = 2;
392 _heartExposed++;
393 }
394
395 void SetPhaseOne()
396 {
397 Talk(SAY_HEART_CLOSED);
398 Talk(EMOTE_HEART_CLOSED);
399
400 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
401 me->SetReactState(REACT_AGGRESSIVE);
402 DoCast(me, SPELL_STAND);
403
404 _phase = 1;
405
406 events.RescheduleEvent(EVENT_SEARING_LIGHT, TIMER_SEARING_LIGHT / 2);
407 events.RescheduleEvent(EVENT_GRAVITY_BOMB, TIMER_GRAVITY_BOMB);
408 events.RescheduleEvent(EVENT_TYMPANIC_TANTRUM, urand(TIMER_TYMPANIC_TANTRUM_MIN, TIMER_TYMPANIC_TANTRUM_MAX));
409
410 Unit* heart = me->GetVehicleKit() ? me->GetVehicleKit()->GetPassenger(HEART_VEHICLE_SEAT_EXPOSED) : NULL;
411 if (!heart)
412 return;
413
414 heart->ChangeSeat(HEART_VEHICLE_SEAT_NORMAL, false);
415 heart->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
416 heart->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNK_29);
417 heart->RemoveAurasDueToSpell(SPELL_EXPOSED_HEART);
418
419 if (!_hardMode)
420 {
421 if (!_transferHealth)
422 _transferHealth = (heart->GetMaxHealth() - heart->GetHealth());
423
424 me->ModifyHealth(-((int32)_transferHealth));
425 }
426 }
427
428 private:
429 // Achievement related
430 bool _healthRecovered; // Did a scrapbot recover XT-002's health during the encounter?
431 bool _hardMode; // Are we in hard mode? Or: was the heart killed during phase 2?
432 bool _gravityBombCasualty; // Did someone die because of Gravity Bomb damage?
433
434 uint8 _phase;
435 uint8 _heartExposed;
436 uint32 _transferHealth;
437 };
438};

628/*-------------------------------------------------------

631 *

633class BoomEvent : public BasicEvent
634{
635 public:
636 BoomEvent(Creature* me) : _me(me)
637 {
638 }
639
640 bool Execute(uint64 /*time*/, uint32 /*diff*/)
641 {
642 // This hack is here because we suspect our implementation of spell effect execution on targets
643 // is done in the wrong order. We suspect that EFFECT_0 needs to be applied on all targets,
644 // then EFFECT_1, etc - instead of applying each effect on target1, then target2, etc.
645 // The above situation causes the visual for this spell to be bugged, so we remove the instakill
646 // effect and implement a script hack for that.
647
648 _me->CastSpell(_me, SPELL_BOOM, false);
649 return true;
650 }
651
652 private:
653 Creature* _me;
654};

671 _instance = creature->GetInstanceScript();
672 }

674 void Initialize()
675 {
676 _boomed = false;
677 }
678
679 void Reset() override
680 {
681 Initialize();
682
683 DoCast(SPELL_AURA_BOOMBOT); // For achievement
684
685 // HACK/workaround:
686 // these values aren't confirmed - lack of data - and the values in DB are incorrect
687 // these values are needed for correct damage of Boom spell
688 me->SetFloatValue(UNIT_FIELD_MINDAMAGE, 15000.0f);
689 me->SetFloatValue(UNIT_FIELD_MAXDAMAGE, 18000.0f);
690
691 /// @todo proper waypoints?
692 if (Creature* pXT002 = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(BOSS_XT002)))
693 me->GetMotionMaster()->MoveFollow(pXT002, 0.0f, 0.0f);
694 }
695
696 void DamageTaken(Unit* /*who*/, uint32& damage) override
697 {
698 if (damage >= (me->GetHealth() - me->GetMaxHealth() * 0.5f) && !_boomed)
699 {
700 _boomed = true; // Prevent recursive calls
701
702 WorldPacket data(SMSG_SPELLINSTAKILLLOG, 8+8+4);
703 data << uint64(me->GetGUID());
704 data << uint64(me->GetGUID());
705 data << uint32(SPELL_BOOM);
706 me->SendMessageToSet(&data, false);
707
708 me->DealDamage(me, me->GetHealth(), NULL, NODAMAGE, SPELL_SCHOOL_MASK_NORMAL, NULL, false);
709
710 damage = 0;
711
712 // Visual only seems to work if the instant kill event is delayed or the spell itself is delayed
713 // Casting done from player and caster source has the same targetinfo flags,
714 // so that can't be the issue
715 // See BoomEvent class
716 // Schedule 1s delayed
717 me->m_Events.AddEvent(new BoomEvent(me), me->m_Events.CalculateTime(1*IN_MILLISECONDS));
718 }
719 }
720
721 void UpdateAI(uint32 /*diff*/) override
722 {
723 if (!UpdateVictim())
724 return;
725
726 // No melee attack
727 }
728
729 private:
730 InstanceScript* _instance;

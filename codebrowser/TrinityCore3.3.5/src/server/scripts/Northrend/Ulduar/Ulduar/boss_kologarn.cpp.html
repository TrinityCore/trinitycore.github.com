101 struct boss_kologarnAI : public BossAI
102 {
103 boss_kologarnAI(Creature* creature) : BossAI(creature, BOSS_KOLOGARN),
104 left(false), right(false)
105 {
106 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
107 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE);
108
109 DoCast(SPELL_KOLOGARN_REDUCE_PARRY);
110 SetCombatMovement(false);
111 Reset();
112 }
113
114 bool left, right;
115 ObjectGuid eyebeamTarget;
116
117 void EnterCombat(Unit* /*who*/) override
118 {
119 Talk(SAY_AGGRO);
120
121 events.ScheduleEvent(EVENT_MELEE_CHECK, 6000);
122 events.ScheduleEvent(EVENT_SMASH, 5000);
123 events.ScheduleEvent(EVENT_SWEEP, 19000);
124 events.ScheduleEvent(EVENT_STONE_GRIP, 25000);
125 events.ScheduleEvent(EVENT_FOCUSED_EYEBEAM, 21000);
126 events.ScheduleEvent(EVENT_ENRAGE, 600000);
127
128 if (Vehicle* vehicle = me->GetVehicleKit())
129 for (uint8 i = 0; i < 2; ++i)
130 if (Unit* arm = vehicle->GetPassenger(i))
131 arm->ToCreature()->SetInCombatWithZone();
132
133 _EnterCombat();
134 }
135
136 void Reset() override
137 {
138 _Reset();
139 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
140 eyebeamTarget.Clear();
141 }
142
143 void JustDied(Unit* /*killer*/) override
144 {
145 Talk(SAY_DEATH);
146 DoCast(SPELL_KOLOGARN_PACIFY);
147 me->GetMotionMaster()->MoveTargetedHome();
148 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
149 me->SetCorpseDelay(604800); // Prevent corpse from despawning.
150 _JustDied();
151 }
152
153 void KilledUnit(Unit* who) override
154 {
155 if (who->GetTypeId() == TYPEID_PLAYER)
156 Talk(SAY_SLAY);
157 }
158
159 void PassengerBoarded(Unit* who, int8 /*seatId*/, bool apply) override
160 {
161 bool isEncounterInProgress = instance->GetBossState(BOSS_KOLOGARN) == IN_PROGRESS;
162 if (who->GetEntry() == NPC_LEFT_ARM)
163 {
164 left = apply;
165 if (!apply && isEncounterInProgress)
166 {
167 Talk(SAY_LEFT_ARM_GONE);
168 events.ScheduleEvent(EVENT_RESPAWN_LEFT_ARM, 40000);
169 }
170 }
171
172 else if (who->GetEntry() == NPC_RIGHT_ARM)
173 {
174 right = apply;
175 if (!apply && isEncounterInProgress)
176 {
177 Talk(SAY_RIGHT_ARM_GONE);
178 events.ScheduleEvent(EVENT_RESPAWN_RIGHT_ARM, 40000);
179 }
180 }
181
182 if (!isEncounterInProgress)
183 return;
184
185 if (!apply)
186 {
187 who->CastSpell(me, SPELL_ARM_DEAD_DAMAGE, true);
188
189 if (Creature* rubbleStalker = who->FindNearestCreature(NPC_RUBBLE_STALKER, 70.0f))
190 {
191 rubbleStalker->CastSpell(rubbleStalker, SPELL_FALLING_RUBBLE, true);
192 rubbleStalker->CastSpell(rubbleStalker, SPELL_SUMMON_RUBBLE, true);
193 who->ToCreature()->DespawnOrUnsummon();
194 }
195
196 if (!right && !left)
197 events.ScheduleEvent(EVENT_STONE_SHOUT, 5000);
198
199 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, CRITERIA_DISARMED);
200 }
201 else
202 {
203 events.CancelEvent(EVENT_STONE_SHOUT);
204 who->ToCreature()->SetInCombatWithZone();
205 }
206 }
207
208 void JustSummoned(Creature* summon) override
209 {
210 switch (summon->GetEntry())
211 {
212 case NPC_FOCUSED_EYEBEAM:
213 summon->CastSpell(me, SPELL_FOCUSED_EYEBEAM_VISUAL_LEFT, true);
214 break;
215 case NPC_FOCUSED_EYEBEAM_RIGHT:
216 summon->CastSpell(me, SPELL_FOCUSED_EYEBEAM_VISUAL_RIGHT, true);
217 break;
218 case NPC_RUBBLE:
219 summons.Summon(summon);
220 // absence of break intended
221 default:
222 return;
223 }
224
225 summon->CastSpell(summon, SPELL_FOCUSED_EYEBEAM_PERIODIC, true);
226 summon->CastSpell(summon, SPELL_FOCUSED_EYEBEAM_VISUAL, true);
227 summon->SetReactState(REACT_PASSIVE);
228 // One of the above spells is a channeled spell, we need to clear this unit state for MoveChase to work
229 summon->ClearUnitState(UNIT_STATE_CASTING);
230
231 // Victim gets 67351
232 if (eyebeamTarget)
233 {
234 if (Unit* target = ObjectAccessor::GetUnit(*summon, eyebeamTarget))
235 {
236 summon->Attack(target, false);
237 summon->GetMotionMaster()->MoveChase(target);
238 }
239 }
240 }
241
242 void UpdateAI(uint32 diff) override
243 {
244 if (!UpdateVictim())
245 return;
246
247 events.Update(diff);
248
249 if (me->HasUnitState(UNIT_STATE_CASTING))
250 return;
251
252 while (uint32 eventId = events.ExecuteEvent())
253 {
254 switch (eventId)
255 {
256 case EVENT_MELEE_CHECK:
257 if (!me->IsWithinMeleeRange(me->GetVictim()))
258 DoCast(SPELL_PETRIFY_BREATH);
259 events.ScheduleEvent(EVENT_MELEE_CHECK, 1 * IN_MILLISECONDS);
260 break;
261 case EVENT_SWEEP:
262 if (left)
263 DoCast(me->FindNearestCreature(NPC_ARM_SWEEP_STALKER, 500.0f, true), SPELL_ARM_SWEEP, true);
264 events.ScheduleEvent(EVENT_SWEEP, 25 * IN_MILLISECONDS);
265 break;
266 case EVENT_SMASH:
267 if (left && right)
268 DoCastVictim(SPELL_TWO_ARM_SMASH);
269 else if (left || right)
270 DoCastVictim(SPELL_ONE_ARM_SMASH);
271 events.ScheduleEvent(EVENT_SMASH, 15 * IN_MILLISECONDS);
272 break;
273 case EVENT_STONE_SHOUT:
274 DoCast(SPELL_STONE_SHOUT);
275 events.ScheduleEvent(EVENT_STONE_SHOUT, 2 * IN_MILLISECONDS);
276 break;
277 case EVENT_ENRAGE:
278 DoCast(SPELL_BERSERK);
279 Talk(SAY_BERSERK);
280 break;
281 case EVENT_RESPAWN_LEFT_ARM:
282 case EVENT_RESPAWN_RIGHT_ARM:
283 {
284 if (Vehicle* vehicle = me->GetVehicleKit())
285 {
286 int8 seat = eventId == EVENT_RESPAWN_LEFT_ARM ? 0 : 1;
287 uint32 entry = eventId == EVENT_RESPAWN_LEFT_ARM ? NPC_LEFT_ARM : NPC_RIGHT_ARM;
288 vehicle->InstallAccessory(entry, seat, true, TEMPSUMMON_MANUAL_DESPAWN, 0);
289 }
290 break;
291 }
292 case EVENT_STONE_GRIP:
293 {
294 if (right)
295 {
296 DoCast(SPELL_STONE_GRIP);
297 Talk(SAY_GRAB_PLAYER);
298 Talk(EMOTE_STONE_GRIP);
299 }
300 events.ScheduleEvent(EVENT_STONE_GRIP, 25 * IN_MILLISECONDS);
301 break;
302 }
303 case EVENT_FOCUSED_EYEBEAM:
304 if (Unit* eyebeamTargetUnit = SelectTarget(SELECT_TARGET_FARTHEST, 0, 0, true))
305 {
306 eyebeamTarget = eyebeamTargetUnit->GetGUID();
307 DoCast(me, SPELL_SUMMON_FOCUSED_EYEBEAM, true);
308 }
309 events.ScheduleEvent(EVENT_FOCUSED_EYEBEAM, urand(15, 35) * IN_MILLISECONDS);
310 break;
311 }
312 }
313
314 DoMeleeAttackIfReady();
315 }
316 };

357// predicate function to select non main tank target
358class StoneGripTargetSelector : public std::unary_function<Unit*, bool>
359{
360 public:
361 StoneGripTargetSelector(Creature* me, Unit const* victim) : _me(me), _victim(victim) { }
362
363 bool operator()(WorldObject* target)
364 {
365 if (target == _victim && _me->getThreatManager().getThreatList().size() > 1)
366 return true;
367
368 if (target->GetTypeId() != TYPEID_PLAYER)
369 return true;
370
371 return false;
372 }
373
374 Creature* _me;
375 Unit const* _victim;

393
394 void FilterTargetsInitial(std::list<WorldObject*>& unitList)
395 {
396 // Remove "main tank" and non-player targets
397 unitList.remove_if(StoneGripTargetSelector(GetCaster()->ToCreature(), GetCaster()->GetVictim()));
398 // Maximum affected targets per difficulty mode
399 uint32 maxTargets = 1;
400 if (GetSpellInfo()->Id == 63981)
401 maxTargets = 3;
402
403 // Return a random amount of targets based on maxTargets
404 while (maxTargets < unitList.size())
405 {
406 std::list<WorldObject*>::iterator itr = unitList.begin();
407 advance(itr, urand(0, unitList.size()-1));
408 unitList.erase(itr);
409 }
410
411 // For subsequent effects
412 _unitList = unitList;
413 }
414
415 void FillTargetsSubsequential(std::list<WorldObject*>& unitList)
416 {
417 unitList = _unitList;
418 }
419
420 void Register() override
421 {
422 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_ulduar_stone_grip_cast_target_SpellScript::FilterTargetsInitial, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
423 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_ulduar_stone_grip_cast_target_SpellScript::FillTargetsSubsequential, EFFECT_1, TARGET_UNIT_SRC_AREA_ENEMY);
424 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_ulduar_stone_grip_cast_target_SpellScript::FillTargetsSubsequential, EFFECT_2, TARGET_UNIT_SRC_AREA_ENEMY);
425 }
426
427 private:
428 // Shared between effects
429 std::list<WorldObject*> _unitList;
430 };

526 //! What we do here is remove all harmful aura's related and teleport to safe spot.
527 void OnRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
528 {
529 if (GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_ENEMY_SPELL)
530 return;
531
532 if (!GetOwner()->ToCreature())
533 return;
534
535 uint32 rubbleStalkerEntry = (GetOwner()->GetMap()->GetDifficulty() == DUNGEON_DIFFICULTY_NORMAL ? 33809 : 33942);
536 Creature* rubbleStalker = GetOwner()->FindNearestCreature(rubbleStalkerEntry, 200.0f, true);
537 if (rubbleStalker)
538 rubbleStalker->CastSpell(rubbleStalker, SPELL_STONE_GRIP_CANCEL, true);
539 }
540

542 {
543 AfterEffectRemove += AuraEffectRemoveFn(spell_ulduar_stone_grip_absorb_AuraScript::OnRemove, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB, AURA_EFFECT_HANDLE_REAL);

561
562 void OnRemoveStun(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
563 {
564 if (Player* owner = GetOwner()->ToPlayer())
565 owner->RemoveAurasDueToSpell(aurEff->GetAmount());
566 }
567
568 void OnRemoveVehicle(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
569 {
570 PreventDefaultAction();
571 Unit* caster = GetCaster();
572 if (!caster)
573 return;
574
575 Position exitPosition;
576 exitPosition.m_positionX = 1750.0f;
577 exitPosition.m_positionY = -7.5f + frand(-3.0f, 3.0f);
578 exitPosition.m_positionZ = 457.9322f;
579
580 // Remove pending passengers before exiting vehicle - might cause an Uninstall
581 GetTarget()->GetVehicleKit()->RemovePendingEventsForPassenger(caster);
582 caster->_ExitVehicle(&exitPosition);
583 caster->RemoveAurasDueToSpell(GetId());
584
585 // Temporarily relocate player to vehicle exit dest serverside to send proper fall movement
586 // beats me why blizzard sends these 2 spline packets one after another instantly
587 Position oldPos = caster->GetPosition();
588 caster->Relocate(exitPosition);
589 caster->GetMotionMaster()->MoveFall();
590 caster->Relocate(oldPos);
591 }
592
593 void Register() override
594 {
595 OnEffectRemove += AuraEffectRemoveFn(spell_ulduar_stone_grip_AuraScript::OnRemoveVehicle, EFFECT_0, SPELL_AURA_CONTROL_VEHICLE, AURA_EFFECT_HANDLE_REAL);
596 AfterEffectRemove += AuraEffectRemoveFn(spell_ulduar_stone_grip_AuraScript::OnRemoveStun, EFFECT_2, SPELL_AURA_MOD_STUN, AURA_EFFECT_HANDLE_REAL);
597 }

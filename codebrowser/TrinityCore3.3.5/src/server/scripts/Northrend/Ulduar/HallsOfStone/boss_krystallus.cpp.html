51class boss_krystallus : public CreatureScript
52{
53 public:
54 boss_krystallus() : CreatureScript("boss_krystallus") { }
55
56 struct boss_krystallusAI : public BossAI
57 {
58 boss_krystallusAI(Creature* creature) : BossAI(creature, DATA_KRYSTALLUS) { }
59
60 void Reset() override
61 {
62 _Reset();
63 }
64
65 void EnterCombat(Unit* /*who*/) override
66 {
67 Talk(SAY_AGGRO);
68 _EnterCombat();
69
70 events.ScheduleEvent(EVENT_BOULDER_TOSS, urand(3000, 9000));
71 events.ScheduleEvent(EVENT_GROUND_SLAM, urand(15000, 18000));
72 events.ScheduleEvent(EVENT_STOMP, urand(20000, 29000));
73 if (IsHeroic())
74 events.ScheduleEvent(EVENT_GROUND_SPIKE, urand(9000, 14000));
75 }
76
77 void UpdateAI(uint32 diff) override
78 {
79 // Return since we have no target
80 if (!UpdateVictim())
81 return;
82
83 events.Update(diff);
84
85 if (me->HasUnitState(UNIT_STATE_CASTING))
86 return;
87
88 while (uint32 eventId = events.ExecuteEvent())
89 {
90 switch (eventId)
91 {
92 case EVENT_BOULDER_TOSS:
93 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true))
94 DoCast(target, SPELL_BOULDER_TOSS);
95 events.ScheduleEvent(EVENT_BOULDER_TOSS, urand(9000, 15000));
96 break;
97 case EVENT_GROUND_SPIKE:
98 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
99 DoCast(target, SPELL_GROUND_SPIKE);
100 events.ScheduleEvent(EVENT_GROUND_SPIKE, urand(12000, 17000));
101 break;
102 case EVENT_GROUND_SLAM:
103 DoCast(me, SPELL_GROUND_SLAM);
104 events.ScheduleEvent(EVENT_SHATTER, 10000);
105 events.ScheduleEvent(EVENT_GROUND_SLAM, urand(15000, 18000));
106 break;
107 case EVENT_STOMP:
108 DoCast(me, SPELL_STOMP);
109 events.ScheduleEvent(EVENT_STOMP, urand(20000, 29000));
110 break;
111 case EVENT_SHATTER:
112 DoCast(me, SPELL_SHATTER);
113 break;
114 default:
115 break;
116 }
117 }
118
119 DoMeleeAttackIfReady();
120 }
121
122 void JustDied(Unit* /*killer*/) override
123 {
124 Talk(SAY_DEATH);
125 _JustDied();
126 }
127
128 void KilledUnit(Unit* victim) override
129 {
130 if (victim->GetTypeId() == TYPEID_PLAYER)
131 Talk(SAY_KILL);
132 }
133 };
134
135 CreatureAI* GetAI(Creature* creature) const override
136 {
137 return GetHallsOfStoneAI<boss_krystallusAI>(creature);
138 }
139};

141class spell_krystallus_shatter : public SpellScriptLoader
142{
143 public:
144 spell_krystallus_shatter() : SpellScriptLoader("spell_krystallus_shatter") { }
145
146 class spell_krystallus_shatter_SpellScript : public SpellScript
147 {
148 PrepareSpellScript(spell_krystallus_shatter_SpellScript);
149
150 void HandleScript(SpellEffIndex /*effIndex*/)
151 {
152 if (Unit* target = GetHitUnit())
153 {
154 target->RemoveAurasDueToSpell(SPELL_STONED);
155 target->CastSpell((Unit*)NULL, SPELL_SHATTER_EFFECT, true);
156 }
157 }
158
159 void Register() override
160 {
161 OnEffectHitTarget += SpellEffectFn(spell_krystallus_shatter_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
162 }
163 };
164
165 SpellScript* GetSpellScript() const override
166 {
167 return new spell_krystallus_shatter_SpellScript();
168 }
169};

171class spell_krystallus_shatter_effect : public SpellScriptLoader
172{
173 public:
174 spell_krystallus_shatter_effect() : SpellScriptLoader("spell_krystallus_shatter_effect") { }
175
176 class spell_krystallus_shatter_effect_SpellScript : public SpellScript
177 {
178 PrepareSpellScript(spell_krystallus_shatter_effect_SpellScript);
179
180 void CalculateDamage()
181 {
182 if (!GetHitUnit())
183 return;
184
185 float radius = GetSpellInfo()->Effects[EFFECT_0].CalcRadius(GetCaster());
186 if (!radius)
187 return;
188
189 float distance = GetCaster()->GetDistance2d(GetHitUnit());
190 if (distance > 1.0f)
191 SetHitDamage(int32(GetHitDamage() * ((radius - distance) / radius)));
192 }
193
194 void Register() override
195 {
196 OnHit += SpellHitFn(spell_krystallus_shatter_effect_SpellScript::CalculateDamage);
197 }
198 };
199
200 SpellScript* GetSpellScript() const override
201 {
202 return new spell_krystallus_shatter_effect_SpellScript();
203 }
204};

91 struct boss_palehoofAI : public BossAI
92 {
93 boss_palehoofAI(Creature* creature) : BossAI(creature, DATA_GORTOK_PALEHOOF)
94 {
95 Initialize();
96 }
97
98 void Initialize()
99 {
100 /// There is a good reason to store them like this, we are going to shuffle the order.
101 for (uint32 i = PHASE_FRENZIED_WORGEN; i < PHASE_GORTOK_PALEHOOF; ++i)
102 Sequence[i] = Phase(i);
103
104 /// This ensures a random order and only executes each phase once.
105 std::random_shuffle(Sequence, Sequence + PHASE_GORTOK_PALEHOOF);
106
107 uiArcingSmashTimer = 15000;
108 uiImpaleTimer = 12000;
109 uiWhiteringRoarTimer = 10000;
110
111 AddCount = 0;
112
113 currentPhase = PHASE_NONE;
114 }
115
116 uint32 uiArcingSmashTimer;
117 uint32 uiImpaleTimer;
118 uint32 uiWhiteringRoarTimer;
119 Phase currentPhase;
120 uint8 AddCount;
121 Phase Sequence[4];
122
123 void Reset() override
124 {
125 _Reset();
126
127 Initialize();
128
129 me->GetMotionMaster()->MoveTargetedHome();
130
131 for (uint8 i = DATA_FRENZIED_WORGEN; i <= DATA_FEROCIOUS_RHINO; ++i)
132 if (Creature* temp = ObjectAccessor::GetCreature(*me, instance->GetGuidData(i)))
133 if (!temp->IsAlive())
134 temp->Respawn();
135
136 if (GameObject* go = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_GORTOK_PALEHOOF_SPHERE)))
137 {
138 go->SetGoState(GO_STATE_READY);
139 go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
140 }
141 }
142
143 void EnterCombat(Unit* /*who*/) override
144 {
145 Talk(SAY_AGGRO);
146 }
147
148 void AttackStart(Unit* who) override
149 {
150 if (!who)
151 return;
152
153 if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
154 return;
155
156 if (me->Attack(who, true))
157 {
158 me->AddThreat(who, 0.0f);
159 me->SetInCombatWith(who);
160 who->SetInCombatWith(me);
161 DoStartMovement(who);
162 }
163 }
164
165 void UpdateAI(uint32 diff) override
166 {
167 if (currentPhase != PHASE_GORTOK_PALEHOOF)
168 return;
169
170 if (!UpdateVictim())
171 return;
172
173 if (uiArcingSmashTimer <= diff)
174 {
175 DoCast(me, SPELL_ARCING_SMASH);
176 uiArcingSmashTimer = urand(13000, 17000);
177 } else uiArcingSmashTimer -= diff;
178
179 if (uiImpaleTimer <= diff)
180 {
181 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
182 DoCast(target, SPELL_IMPALE);
183 uiImpaleTimer = urand(8000, 12000);
184 } else uiImpaleTimer -= diff;
185
186 if (uiWhiteringRoarTimer <= diff)
187 {
188 DoCast(me, SPELL_WITHERING_ROAR);
189 uiWhiteringRoarTimer = urand(8000, 12000);
190 } else uiWhiteringRoarTimer -= diff;
191
192 DoMeleeAttackIfReady();
193 }
194
195 void JustDied(Unit* /*killer*/) override
196 {
197 _JustDied();
198 //Talk(SAY_DEATH);
199 }
200
201 void KilledUnit(Unit* who) override
202 {
203 if (who->GetTypeId() == TYPEID_PLAYER)
204 Talk(SAY_SLAY);
205 }
206
207 void DoAction(int32 actionId) override
208 {
209 if (actionId != ACTION_NEXT_PHASE)
210 return;
211
212 if (currentPhase == PHASE_NONE)
213 {
214 instance->SetBossState(DATA_GORTOK_PALEHOOF, IN_PROGRESS);
215
216 if (Creature* orb = me->SummonCreature(NPC_STASIS_CONTROLLER, moveLocs[5], TEMPSUMMON_CORPSE_DESPAWN))
217 orb->CastSpell(me, SPELL_ORB_VISUAL, true);
218 }
219
220 Phase move = PHASE_NONE;
221 if (AddCount >= DUNGEON_MODE(2, 4))
222 move = PHASE_GORTOK_PALEHOOF;
223 else
224 move = Sequence[AddCount++];
225
226 // send orb to summon spot
227 if (Creature* orb = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_GORTOK_ORB)))
228 if (orb->IsAlive())
229 orb->GetMotionMaster()->MovePoint(move, moveLocs[move]);
230
231 currentPhase = move;
232 }
233
234 void JustReachedHome() override
235 {
236 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE|UNIT_FLAG_NOT_ATTACKABLE_1|UNIT_FLAG_IMMUNE_TO_PC);
237 me->SetStandState(UNIT_STAND_STATE_STAND);
238 DoCast(me, SPELL_FREEZE);
239 }
240 };
241
242 CreatureAI* GetAI(Creature* creature) const override
243 {
244 return GetUtgardePinnacleAI<boss_palehoofAI>(creature);
245 }

742
743 void Reset() override
744 {
745 Initialize();
746 //! HACK: Creature's can't have MOVEMENTFLAG_FLYING
747 me->AddUnitMovementFlag(MOVEMENTFLAG_FLYING);
748 me->RemoveAurasDueToSpell(SPELL_ORB_VISUAL);
749 me->SetSpeed(MOVE_FLIGHT, 0.5f);
750 }
751
752 void UpdateAI(uint32 diff) override
753 {
754 if (currentPhase == PHASE_NONE)
755 return;
756
757 if (SummonTimer <= diff)
758 {
759 uint8 nextBossId = 0;
760 switch (currentPhase)
761 {
762 case PHASE_FRENZIED_WORGEN:
763 nextBossId = DATA_FRENZIED_WORGEN;
764 break;
765 case PHASE_RAVENOUS_FURLBORG:
766 nextBossId = DATA_RAVENOUS_FURBOLG;
767 break;
768 case PHASE_MASSIVE_JORMUNGAR:
769 nextBossId = DATA_MASSIVE_JORMUNGAR;
770 break;
771 case PHASE_FEROCIOUS_RHINO:
772 nextBossId = DATA_FEROCIOUS_RHINO;
773 break;
774 case PHASE_GORTOK_PALEHOOF:
775 nextBossId = DATA_GORTOK_PALEHOOF;
776 break;
777 default:
778 return;
779 }
780
781 if (Creature* nextBoss = ObjectAccessor::GetCreature(*me, instance->GetGuidData(nextBossId)))
782 {
783 nextBoss->RemoveAurasDueToSpell(SPELL_FREEZE);

787 nextBoss->Attack(nextBoss->SelectNearestTarget(100), true);
788 }
789 currentPhase = PHASE_NONE;

791 if (nextBossId == DATA_GORTOK_PALEHOOF)
792 me->DespawnOrUnsummon();
793 }
794 else
795 SummonTimer -= diff;
796 }

803 uint8 nextBossId = 0;
804 switch (id)
805 {
806 case PHASE_FRENZIED_WORGEN:

808 break;
809 case PHASE_RAVENOUS_FURLBORG:
810 nextBossId = DATA_RAVENOUS_FURBOLG;
811 break;
812 case PHASE_MASSIVE_JORMUNGAR:
813 nextBossId = DATA_MASSIVE_JORMUNGAR;
814 break;

817 break;
818 case PHASE_GORTOK_PALEHOOF:
819 nextBossId = DATA_GORTOK_PALEHOOF;
820 break;
821 default:
822 return;
823 }
824
825 if (Creature* nextBoss = ObjectAccessor::GetCreature(*me, instance->GetGuidData(nextBossId)))
826 DoCast(nextBoss, SPELL_ORB_CHANNEL, false);
827
828 currentPhase = Phase(id);
829 SummonTimer = 5000;
830 }

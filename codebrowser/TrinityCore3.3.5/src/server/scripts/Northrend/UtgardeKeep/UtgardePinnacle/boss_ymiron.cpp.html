99class boss_ymiron : public CreatureScript
100{
101public:
102 boss_ymiron() : CreatureScript("boss_ymiron") { }
103
104 struct boss_ymironAI : public BossAI
105 {
106 boss_ymironAI(Creature* creature) : BossAI(creature, DATA_KING_YMIRON)
107 {
108 Initialize();
109 // This ensures a random sequence of ancestors. Not sure if the order should change on reset or not, reason why this is left out of Initialize().
110 for (int i = 0; i < 4; ++i)
111 ActiveOrder[i] = i;
112 for (int i = 0; i < 3; ++i)
113 {
114 int r = i + (rand32() % (4 - i));
115 int temp = ActiveOrder[i];
116 ActiveOrder[i] = ActiveOrder[r];
117 ActiveOrder[r] = temp;
118 }
119 }
120
121 void Initialize()
122 {
123 kingsBane = true;
124 ActivedNumber = 0;
125 HealthAmountModifier = 1;
126 HealthAmountMultipler = DUNGEON_MODE(20, 25);
127 ActiveAncestorGUID.Clear();
128 SpiritFountGUID.Clear();
129 }
130
131 void Reset() override
132 {
133 _Reset();
134 Initialize();
135 me->SetReactState(REACT_AGGRESSIVE);
136 }
137
138 void EnterCombat(Unit* /*who*/) override
139 {
140 _EnterCombat();
141 Talk(SAY_AGGRO);
142 events.ScheduleEvent(EVENT_BANE, urand(18000, 23000), EVENT_GROUP_BASE_SPELLS);
143 events.ScheduleEvent(EVENT_FETID_ROT, urand(8000, 13000), EVENT_GROUP_BASE_SPELLS);
144 events.ScheduleEvent(EVENT_DARK_SLASH, urand(28000, 33000), EVENT_GROUP_BASE_SPELLS);
145 events.ScheduleEvent(EVENT_ANCESTORS_VENGEANCE, DUNGEON_MODE(60000, 45000), EVENT_GROUP_BASE_SPELLS);
146 }
147
148 void SpellHitTarget(Unit* who, SpellInfo const* spell) override
149 {
150 if (who && who->GetTypeId() == TYPEID_PLAYER && spell->Id == SPELL_BANE_HIT)
151 kingsBane = false;
152 }
153
154 uint32 GetData(uint32 type) const override
155 {
156 if (type == DATA_KINGS_BANE)
157 return kingsBane ? 1 : 0;
158
159 return 0;
160 }
161
162 void MovementInform(uint32 type, uint32 pointId) override
163 {
164 if (type != POINT_MOTION_TYPE)
165 return;
166
167 if (pointId == POINT_BOAT) // Check might not be needed.
168 {
169 Talk(ActiveBoat[ActiveOrder[ActivedNumber]].say);
170 if (Creature* ancestor = me->SummonCreature(ActiveBoat[ActiveOrder[ActivedNumber]].npc, ActiveBoat[ActiveOrder[ActivedNumber]].SpawnX, ActiveBoat[ActiveOrder[ActivedNumber]].SpawnY, ActiveBoat[ActiveOrder[ActivedNumber]].SpawnZ, ActiveBoat[ActiveOrder[ActivedNumber]].SpawnO, TEMPSUMMON_CORPSE_DESPAWN, 0))
171 {
172 DoCast(ancestor, SPELL_CHANNEL_YMIRON_TO_SPIRIT);
173 ancestor->CastSpell(me, SPELL_CHANNEL_SPIRIT_TO_YMIRON, true);
174 ancestor->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
175 ancestor->SetDisableGravity(true);
176 ActiveAncestorGUID = ancestor->GetGUID();
177 }
178 events.ScheduleEvent(EVENT_RESUME_COMBAT, 5000);
179 }
180 }
181
182 void JustSummoned(Creature* summon) override
183 {
184 switch (summon->GetEntry())
185 {
186 case NPC_SPIRIT_FOUNT:
187 summon->CastSpell(summon, SPELL_SPIRIT_FOUNT, true);
188 summon->SetDisplayId(11686);
189 SpiritFountGUID = summon->GetGUID();
190 break;
191 case NPC_AVENGING_SPIRIT:
192 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
193 {
194 summon->AddThreat(target, 0.0f);
195 summon->AI()->AttackStart(target);
196 }
197 break;
198 default:
199 break;
200 }
201
202 summons.Summon(summon);
203 }
204
205 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
206 {
207 if (me->HealthBelowPctDamaged(100 - HealthAmountMultipler * HealthAmountModifier, damage) && !(damage >= me->GetHealth()))
208 {
209 uint8 Order = HealthAmountModifier - 1;
210 ++HealthAmountModifier;
211
212 me->InterruptNonMeleeSpells(true);
213 DoCast(me, SPELL_SCREAMS_OF_THE_DEAD);
214
215 me->AttackStop();
216 me->SetReactState(REACT_PASSIVE);
217 me->GetMotionMaster()->MovePoint(POINT_BOAT, ActiveBoat[ActiveOrder[Order]].MoveX, ActiveBoat[ActiveOrder[Order]].MoveY, ActiveBoat[ActiveOrder[Order]].MoveZ);
218
219 DespawnBoatGhosts(ActiveAncestorGUID);
220 DespawnBoatGhosts(SpiritFountGUID);
221
222 events.CancelEvent(ActiveBoat[ActiveOrder[ActivedNumber]].event); // Cancels the event started on the previous transition.
223 events.DelayEvents(10000, EVENT_GROUP_BASE_SPELLS);
224
225 ActivedNumber = Order;
226 }
227 }
228
229 void ExecuteEvent(uint32 eventId) override
230 {
231 switch (eventId)
232 {
233 case EVENT_BANE:
234 DoCast(SPELL_BANE);
235 events.ScheduleEvent(EVENT_BANE, urand(20000, 25000));
236 break;
237 case EVENT_FETID_ROT:
238 DoCastVictim(SPELL_FETID_ROT);
239 events.ScheduleEvent(EVENT_FETID_ROT, urand(10000, 15000));
240 break;
241 case EVENT_DARK_SLASH:
242 DoCastVictim(SPELL_DARK_SLASH);
243 events.ScheduleEvent(EVENT_DARK_SLASH, urand(30000, 35000));
244 break;
245 case EVENT_ANCESTORS_VENGEANCE:
246 DoCast(me, SPELL_ANCESTORS_VENGEANCE);
247 events.ScheduleEvent(EVENT_ANCESTORS_VENGEANCE, DUNGEON_MODE(urand(60000, 65000), urand(45000, 50000)));
248 break;
249 case EVENT_RESUME_COMBAT:
250 me->SetReactState(REACT_AGGRESSIVE);
251 events.ScheduleEvent(ActiveBoat[ActiveOrder[ActivedNumber]].event, 5000);
252 break;
253 case EVENT_BJORN_SPIRIT_FOUNT:
254 DoCast(SPELL_SUMMON_SPIRIT_FOUNT);
255 break;
256 case EVENT_HALDOR_SPIRIT_STRIKE:
257 DoCastVictim(SPELL_SPIRIT_STRIKE);
258 events.ScheduleEvent(EVENT_HALDOR_SPIRIT_STRIKE, 5000);
259 break;
260 case EVENT_RANULF_SPIRIT_BURST:
261 DoCast(me, SPELL_SPIRIT_BURST);
262 events.ScheduleEvent(EVENT_RANULF_SPIRIT_BURST, 10000);
263 break;
264 case EVENT_TORGYN_SUMMON_AVENGING_SPIRITS:
265 for (uint8 i = 0; i < 4; ++i)
266 DoCast(SPELL_SUMMON_AVENGING_SPIRIT);
267 events.ScheduleEvent(EVENT_TORGYN_SUMMON_AVENGING_SPIRITS, 15000);
268 break;
269 default:
270 break;
271 }
272 }
273
274 void JustDied(Unit* /*killer*/) override
275 {
276 _JustDied();
277 Talk(SAY_DEATH);
278 }
279
280 void KilledUnit(Unit* who) override
281 {
282 if (who->GetTypeId() == TYPEID_PLAYER)
283 Talk(SAY_SLAY);
284 }

286 void DespawnBoatGhosts(ObjectGuid& CreatureGUID)
287 {
288 // @todo: fire visual after ancestor despawns.
289 if (CreatureGUID)
290 if (Creature* temp = ObjectAccessor::GetCreature(*me, CreatureGUID))
291 temp->DisappearAndDie();
292
293 CreatureGUID.Clear();
294 }
295
296 private:
297 bool kingsBane; // Achievement King's Bane
298 uint8 ActiveOrder[4];
299 uint8 ActivedNumber;
300 uint32 HealthAmountModifier;
301 uint32 HealthAmountMultipler;
302 ObjectGuid ActiveAncestorGUID;
303 ObjectGuid SpiritFountGUID;
304 };

307 {

309 }

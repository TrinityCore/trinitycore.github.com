136 struct boss_svalaAI : public BossAI
137 {
138 boss_svalaAI(Creature* creature) : BossAI(creature, DATA_SVALA_SORROWGRAVE)
139 {
140 Initialize();
141 _introCompleted = false;
142 }
143
144 void Initialize()
145 {
146 _arthasGUID.Clear();
147 _sacrificed = false;
148 }
149
150 void Reset() override
151 {
152 _Reset();
153
154 SetCombatMovement(true);
155
156 if (_introCompleted)
157 events.SetPhase(NORMAL);
158 else
159 events.SetPhase(IDLE);
160
161 me->SetDisableGravity(events.IsInPhase(NORMAL));
162
163 Initialize();
164
165 instance->SetGuidData(DATA_SACRIFICED_PLAYER, ObjectGuid::Empty);
166 }
167
168 void EnterCombat(Unit* /*who*/) override
169 {
170 _EnterCombat();
171 Talk(SAY_AGGRO);
172 }
173
174 void JustSummoned(Creature* summon) override
175 {
176 if (summon->GetEntry() == NPC_RITUAL_CHANNELER)
177 summon->CastSpell(summon, SPELL_SUMMONED_VIS, true);
178 summons.Summon(summon);
179 }
180
181 void MoveInLineOfSight(Unit* who) override
182 {
183 if (!who)
184 return;
185
186 if (events.IsInPhase(IDLE) && me->IsValidAttackTarget(who) && me->IsWithinDistInMap(who, 40))
187 {
188 events.SetPhase(INTRO);
189 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
190
191 if (GameObject* mirror = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_UTGARDE_MIRROR)))
192 mirror->SetGoState(GO_STATE_READY);
193
194 if (Creature* arthas = me->SummonCreature(NPC_ARTHAS, ArthasPos, TEMPSUMMON_MANUAL_DESPAWN))
195 {
196 arthas->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
197 _arthasGUID = arthas->GetGUID();
198 }
199 events.ScheduleEvent(EVENT_INTRO_SVALA_TALK_0, 1 * IN_MILLISECONDS, 0, INTRO);
200 }
201 }
202
203 void KilledUnit(Unit* who) override
204 {
205 if (who->GetTypeId() == TYPEID_PLAYER)
206 Talk(SAY_SLAY);
207 }
208
209 void JustDied(Unit* /*killer*/) override
210 {
211 if (events.IsInPhase(SACRIFICING))
212 SetEquipmentSlots(false, EQUIP_UNEQUIP, EQUIP_NO_CHANGE, EQUIP_NO_CHANGE);
213 me->HandleEmoteCommand(EMOTE_ONESHOT_FLYDEATH);
214 _JustDied();
215 Talk(SAY_DEATH);
216 }
217
218 void SpellHitTarget(Unit* /*target*/, SpellInfo const* spellInfo) override
219 {
220 if (spellInfo->Id == SPELL_RITUAL_STRIKE_EFF_1 && !events.IsInPhase(NORMAL) && !events.IsInPhase(SVALADEAD))
221 {
222 events.SetPhase(NORMAL);
223 events.ScheduleEvent(EVENT_SINISTER_STRIKE, 7 * IN_MILLISECONDS, 0, NORMAL);
224 events.ScheduleEvent(EVENT_CALL_FLAMES, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, NORMAL);
225 SetCombatMovement(true);
226
227 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 300.0f, true))
228 me->GetMotionMaster()->MoveChase(target);
229 }
230 }
231
232 void UpdateAI(uint32 diff) override
233 {
234 if (events.IsInPhase(IDLE))
235 return;
236
237 if (events.IsInPhase(NORMAL) && !UpdateVictim())
238 return;
239
240 events.Update(diff);
241
242 if (!_sacrificed && HealthBelowPct(50))
243 {
244 _sacrificed = true;
245 events.SetPhase(SACRIFICING);
246 events.ScheduleEvent(EVENT_RITUAL_PREPARATION, 0, 0, SACRIFICING);
247 }
248
249 if (events.IsInPhase(NORMAL))
250 DoMeleeAttackIfReady();
251
252 while (uint32 eventId = events.ExecuteEvent())
253 {
254 switch (eventId)
255 {
256 case EVENT_INTRO_SVALA_TALK_0:
257 Talk(SAY_SVALA_INTRO_0);
258 events.ScheduleEvent(EVENT_INTRO_ARTHAS_TALK_0, 8.1 * IN_MILLISECONDS, 0, INTRO);
259 break;
260 case EVENT_INTRO_ARTHAS_TALK_0:
261 if (Creature* arthas = ObjectAccessor::GetCreature(*me, _arthasGUID))
262 arthas->AI()->Talk(SAY_DIALOG_OF_ARTHAS_1);
263 events.ScheduleEvent(EVENT_INTRO_TRANSFORM_0, 10 * IN_MILLISECONDS, 0, INTRO);
264 break;
265 case EVENT_INTRO_TRANSFORM_0:
266 {
267 if (Creature* arthas = ObjectAccessor::GetCreature(*me, _arthasGUID))
268 arthas->CastSpell(me, SPELL_TRANSFORMING_CHANNEL, true);
269 Position pos;
270 pos.Relocate(me);
271 pos.m_positionZ += 8.0f;
272 me->GetMotionMaster()->MoveTakeoff(0, pos);
273 // spectators flee event
274 std::list<Creature*> lspectatorList;
275 GetCreatureListWithEntryInGrid(lspectatorList, me, NPC_SPECTATOR, 100.0f);
276 for (std::list<Creature*>::iterator itr = lspectatorList.begin(); itr != lspectatorList.end(); ++itr)
277 {
278 if ((*itr)->IsAlive())
279 {
280 (*itr)->SetStandState(UNIT_STAND_STATE_STAND);
281 (*itr)->SetWalk(false);
282 (*itr)->GetMotionMaster()->MovePoint(1, spectatorWP[0]);
283 }
284 }
285 events.ScheduleEvent(EVENT_INTRO_TRANSFORM_1, 4.2 * IN_MILLISECONDS, 0, INTRO);
286 break;
287 }
288 case EVENT_INTRO_TRANSFORM_1:
289 me->CastSpell(me, SPELL_SVALA_TRANSFORMING1, false);
290 events.ScheduleEvent(EVENT_INTRO_TRANSFORM_2, 6.2 * IN_MILLISECONDS, 0, INTRO);
291 break;
292 case EVENT_INTRO_TRANSFORM_2:
293 me->CastSpell(me, SPELL_SVALA_TRANSFORMING2, false);
294 if (Creature* arthas = ObjectAccessor::GetCreature(*me, _arthasGUID))
295 {
296 arthas->InterruptNonMeleeSpells(true);
297 me->SetFacingToObject(arthas);
298 }
299 me->RemoveAllAuras();
300 me->UpdateEntry(NPC_SVALA_SORROWGRAVE);
301 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
302 events.ScheduleEvent(EVENT_INTRO_SVALA_TALK_1, 10 * IN_MILLISECONDS, 0, INTRO);
303 break;
304 case EVENT_INTRO_SVALA_TALK_1:
305 Talk(SAY_SVALA_INTRO_1);
306 events.ScheduleEvent(EVENT_INTRO_ARTHAS_TALK_1, 3.2 * IN_MILLISECONDS, 0, INTRO);
307 break;
308 case EVENT_INTRO_ARTHAS_TALK_1:
309 if (Creature* arthas = ObjectAccessor::GetCreature(*me, _arthasGUID))
310 arthas->AI()->Talk(SAY_DIALOG_OF_ARTHAS_2);
311 events.ScheduleEvent(EVENT_INTRO_SVALA_TALK_2, 7.2 * IN_MILLISECONDS, 0, INTRO);
312 break;
313 case EVENT_INTRO_SVALA_TALK_2:
314 Talk(SAY_SVALA_INTRO_2);
315 me->SetFacingTo(1.58f);
316 if (Creature* arthas = ObjectAccessor::GetCreature(*me, _arthasGUID))
317 arthas->SetVisible(false);
318 events.ScheduleEvent(EVENT_INTRO_RELOCATE_SVALA, 13.8 * IN_MILLISECONDS, 0, INTRO);
319 break;
320 case EVENT_INTRO_RELOCATE_SVALA:
321 {
322 Position pos;
323 pos.Relocate(me);
324 pos.m_positionX = me->GetHomePosition().GetPositionX();
325 pos.m_positionY = me->GetHomePosition().GetPositionY();
326 pos.m_positionZ = 90.6065f;
327 me->GetMotionMaster()->MoveLand(0, pos);
328 me->SetDisableGravity(false, true);
329 me->SetHover(true);
330 events.ScheduleEvent(EVENT_INTRO_DESPAWN_ARTHAS, 3 * IN_MILLISECONDS, 0, INTRO);
331 break;
332 }
333 case EVENT_INTRO_DESPAWN_ARTHAS:
334 if (GameObject* mirror = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_UTGARDE_MIRROR)))
335 mirror->SetGoState(GO_STATE_ACTIVE);
336 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
337 if (Creature* arthas = ObjectAccessor::GetCreature(*me, _arthasGUID))
338 arthas->DespawnOrUnsummon();
339 _arthasGUID.Clear();
340 events.SetPhase(NORMAL);
341 _introCompleted = true;
342 events.ScheduleEvent(EVENT_SINISTER_STRIKE, 7 * IN_MILLISECONDS, 0, NORMAL);
343 events.ScheduleEvent(EVENT_CALL_FLAMES, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, NORMAL);
344 break;
345 case EVENT_SINISTER_STRIKE:
346 DoCastVictim(SPELL_SINSTER_STRIKE);
347 events.ScheduleEvent(EVENT_SINISTER_STRIKE, urand(5 * IN_MILLISECONDS, 9 * IN_MILLISECONDS), 0, NORMAL);
348 break;
349 case EVENT_CALL_FLAMES:
350 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
351 DoCast(target, SPELL_CALL_FLAMES);
352 events.ScheduleEvent(EVENT_CALL_FLAMES, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS), 0, NORMAL);
353 break;
354 case EVENT_RITUAL_PREPARATION:
355 if (Unit* sacrificeTarget = SelectTarget(SELECT_TARGET_RANDOM, 0, 80.0f, true))
356 {
357 instance->SetGuidData(DATA_SACRIFICED_PLAYER, sacrificeTarget->GetGUID());
358 Talk(SAY_SACRIFICE_PLAYER);
359 DoCast(sacrificeTarget, SPELL_RITUAL_PREPARATION);
360 SetCombatMovement(false);
361 DoCast(me, SPELL_RITUAL_OF_THE_SWORD);
362 }
363 events.ScheduleEvent(EVENT_SPAWN_RITUAL_CHANNELERS, 1 * IN_MILLISECONDS, 0, SACRIFICING);
364 break;
365 case EVENT_SPAWN_RITUAL_CHANNELERS:
366 DoCast(me, SPELL_RITUAL_CHANNELER_1, true);
367 DoCast(me, SPELL_RITUAL_CHANNELER_2, true);
368 DoCast(me, SPELL_RITUAL_CHANNELER_3, true);
369 events.ScheduleEvent(EVENT_RITUAL_STRIKE, 2 * IN_MILLISECONDS, 0, SACRIFICING);
370 break;
371 case EVENT_RITUAL_STRIKE:
372 me->StopMoving();
373 me->GetMotionMaster()->MoveIdle();
374 me->InterruptNonMeleeSpells(true);
375 DoCast(me, SPELL_RITUAL_STRIKE_TRIGGER, true);
376 events.ScheduleEvent(EVENT_RITUAL_DISARM, 200, 0, SACRIFICING);
377 break;
378 case EVENT_RITUAL_DISARM:
379 DoCast(me, SPELL_RITUAL_DISARM);
380 break;
381 default:
382 break;
383 }
384 }
385 }
386
387 private:
388 ObjectGuid _arthasGUID;
389 bool _sacrificed;
390 bool _introCompleted;
391 };

393 CreatureAI* GetAI(Creature* creature) const override

395 return GetUtgardePinnacleAI<boss_svalaAI>(creature);
396 }

523class npc_scourge_hulk : public CreatureScript
524{
525 public:
526 npc_scourge_hulk() : CreatureScript("npc_scourge_hulk") { }
527
528 struct npc_scourge_hulkAI : public ScriptedAI
529 {
530 npc_scourge_hulkAI(Creature* creature) : ScriptedAI(creature)
531 {
532 Initialize();
533 }
534
535 void Initialize()
536 {
537 mightyBlow = urand(4000, 9000);
538 volatileInfection = urand(10000, 14000);
539 killedByRitualStrike = false;
540 }
541
542 uint32 mightyBlow;
543 uint32 volatileInfection;
544
545 void Reset() override
546 {
547 Initialize();
548 }
549
550 uint32 GetData(uint32 type) const override
551 {
552 return type == DATA_INCREDIBLE_HULK ? killedByRitualStrike : 0;
553 }
554
555 void DamageTaken(Unit* attacker, uint32 &damage) override
556 {
557 if (damage >= me->GetHealth() && attacker->GetEntry() == NPC_SVALA_SORROWGRAVE)
558 killedByRitualStrike = true;
559 }
560
561 void UpdateAI(uint32 diff) override
562 {
563 if (!UpdateVictim())
564 return;
565
566 if (mightyBlow <= diff)
567 {
568 if (Unit* victim = me->GetVictim())
569 if (!victim->HasUnitState(UNIT_STATE_STUNNED)) // Prevent knocking back a ritual player
570 DoCast(victim, SPELL_MIGHTY_BLOW);
571 mightyBlow = urand(12000, 17000);
572 }
573 else
574 mightyBlow -= diff;
575
576 if (volatileInfection <= diff)
577 {
578 DoCastVictim(SPELL_VOLATILE_INFECTION);
579 volatileInfection = urand(13000, 17000);
580 }
581 else
582 volatileInfection -= diff;
583
584 DoMeleeAttackIfReady();
585 }
586
587 private:
588 bool killedByRitualStrike;
589 };
590
591 CreatureAI* GetAI(Creature* creature) const override
592 {
593 return GetUtgardePinnacleAI<npc_scourge_hulkAI>(creature);
594 }

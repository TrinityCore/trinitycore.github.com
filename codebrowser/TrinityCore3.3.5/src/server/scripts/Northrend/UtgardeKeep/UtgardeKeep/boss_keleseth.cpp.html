111class boss_keleseth : public CreatureScript
112{
113 public:
114 boss_keleseth() : CreatureScript("boss_keleseth") { }
115
116 struct boss_kelesethAI : public BossAI
117 {
118 boss_kelesethAI(Creature* creature) : BossAI(creature, DATA_PRINCE_KELESETH)
119 {
120 Initialize();
121 }
122
123 void Initialize()
124 {
125 onTheRocks = true;
126 }
127
128 void Reset() override
129 {
130 _Reset();
131 events.ScheduleEvent(EVENT_SHADOWBOLT, urand(2, 3)*IN_MILLISECONDS);
132 events.ScheduleEvent(EVENT_FROST_TOMB, urand(14, 19)*IN_MILLISECONDS);
133 events.ScheduleEvent(EVENT_SUMMON_SKELETONS, 6*IN_MILLISECONDS);
134
135 Initialize();
136 }
137
138 void EnterCombat(Unit* who) override
139 {
140 _EnterCombat();
141 Talk(SAY_START_COMBAT);
142
143 if (!who)
144 return;
145
146 std::list<Creature*> guards;
147 me->GetCreatureListWithEntryInGrid(guards, NPC_RUNEMAGE, 60.0f);
148 me->GetCreatureListWithEntryInGrid(guards, NPC_STRATEGIST, 60.0f);
149 if (!guards.empty())
150 {
151 for (std::list<Creature*>::iterator itr = guards.begin(); itr != guards.end(); ++itr)
152 {
153 if ((*itr)->IsAlive() && (*itr)->IsWithinLOSInMap(me))
154 (*itr)->AI()->AttackStart(who);
155 }
156 }
157 }
158
159 void JustDied(Unit* /*killer*/) override
160 {
161 _JustDied();
162 Talk(SAY_DEATH);
163 }
164
165 void SetData(uint32 data, uint32 value) override
166 {
167 if (data == DATA_ON_THE_ROCKS)
168 onTheRocks = value != 0;
169 }
170
171 uint32 GetData(uint32 data) const override
172 {
173 if (data == DATA_ON_THE_ROCKS)
174 return onTheRocks;
175
176 return 0;
177 }
178
179 void UpdateAI(uint32 diff) override
180 {
181 if (!UpdateVictim())
182 return;
183
184 events.Update(diff);
185
186 if (me->HasUnitState(UNIT_STATE_CASTING))
187 return;
188
189 while (uint32 eventId = events.ExecuteEvent())
190 {
191 switch (eventId)
192 {
193 case EVENT_SUMMON_SKELETONS:
194 Talk(SAY_SUMMON_SKELETONS);
195 SummonSkeletons();
196 break;
197 case EVENT_SHADOWBOLT:
198 DoCastVictim(SPELL_SHADOWBOLT);
199 events.ScheduleEvent(EVENT_SHADOWBOLT, urand(2, 3) * IN_MILLISECONDS);
200 break;
201 case EVENT_FROST_TOMB:
202 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true, -SPELL_FROST_TOMB))
203 {
204 Talk(SAY_FROST_TOMB);
205 Talk(SAY_FROST_TOMB_EMOTE, target);
206
207 DoCast(target, SPELL_FROST_TOMB_STUN, true);
208 // checked from sniffs - the player casts the spell
209 target->CastSpell(target, SPELL_FROST_TOMB_SUMMON, true);
210 }
211 events.ScheduleEvent(EVENT_FROST_TOMB, urand(14, 19) * IN_MILLISECONDS);
212 break;
213 default:
214 break;
215 }
216 }
217
218 DoMeleeAttackIfReady();
219 }
220
221 void SummonSkeletons()
222 {
223 // I could not found any spell cast for this
224 for (uint8 i = 0; i < 4; ++i)
225 me->SummonCreature(NPC_SKELETON, SkeletonSpawnPoint[0][0], SkeletonSpawnPoint[0][1], SKELETONSPAWN_Z, 0);
226 }
227
228 private:
229 bool onTheRocks;
230 };
231
232 CreatureAI* GetAI(Creature* creature) const override
233 {
234 return GetUtgardeKeepAI<boss_kelesethAI>(creature);
235 }

238class npc_vrykul_skeleton : public CreatureScript
239{
240 public:
241 npc_vrykul_skeleton() : CreatureScript("npc_vrykul_skeleton") { }
242
243 struct npc_vrykul_skeletonAI : public ScriptedAI
244 {
245 npc_vrykul_skeletonAI(Creature* creature) : ScriptedAI(creature) { }
246
247 void Reset() override
248 {
249 events.Reset();
250 events.ScheduleEvent(EVENT_DECREPIFY, urand(4, 6) * IN_MILLISECONDS);
251 }
252
253 void DamageTaken(Unit* /*doneBy*/, uint32& damage) override
254 {
255 if (damage >= me->GetHealth())
256 {
257 damage = 0;
258
259 // There are some issues with pets
260 // they will still attack. I would say it is a PetAI bug
261 if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE))
262 {
263 // from sniffs
264 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
265 me->SetFlag(UNIT_FIELD_BYTES_1, UNIT_STAND_STATE_DEAD);
266
267 events.Reset();
268 events.ScheduleEvent(EVENT_RESURRECT, urand(18, 22) * IN_MILLISECONDS);
269
270 me->GetMotionMaster()->MovementExpired(false);
271 me->GetMotionMaster()->MoveIdle();
272 }
273 }
274 }
275
276 void UpdateAI(uint32 diff) override
277 {
278 if (!UpdateVictim())
279 return;
280
281 events.Update(diff);
282
283 if (me->HasUnitState(UNIT_STATE_CASTING))
284 return;
285
286 while (uint32 eventId = events.ExecuteEvent())
287 {
288 switch (eventId)
289 {
290 case EVENT_DECREPIFY:
291 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_DECREPIFY))
292 DoCast(target, SPELL_DECREPIFY);
293 events.ScheduleEvent(EVENT_DECREPIFY, urand(1, 5)*IN_MILLISECONDS);
294 break;
295 case EVENT_RESURRECT:
296 events.ScheduleEvent(EVENT_FULL_HEAL, 1 * IN_MILLISECONDS);
297 events.ScheduleEvent(EVENT_SHADOW_FISSURE, 1 * IN_MILLISECONDS);
298 break;
299 case EVENT_FULL_HEAL:
300 DoCast(me, SPELL_FULL_HEAL, true);
301 break;
302 case EVENT_SHADOW_FISSURE:
303 DoCast(me, SPELL_SHADOW_FISSURE, true);
304 DoCastAOE(SPELL_BONE_ARMOR, true);
305 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
306 me->RemoveFlag(UNIT_FIELD_BYTES_1, UNIT_STAND_STATE_DEAD);
307 me->GetMotionMaster()->MoveChase(me->GetVictim());
308 events.ScheduleEvent(EVENT_DECREPIFY, urand(4, 6) * IN_MILLISECONDS);
309 break;
310 default:
311 break;
312 }
313 }
314
315 if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE))
316 DoMeleeAttackIfReady();
317 }
318
319 private:
320 EventMap events;
321 };
322
323 CreatureAI* GetAI(Creature* creature) const override
324 {
325 return GetUtgardeKeepAI<npc_vrykul_skeletonAI>(creature);
326 }

329class spell_frost_tomb : public SpellScriptLoader
330{
331 public:
332 spell_frost_tomb() : SpellScriptLoader("spell_frost_tomb") { }
333
334 class spell_frost_tomb_AuraScript : public AuraScript
335 {
336 PrepareAuraScript(spell_frost_tomb_AuraScript);
337
338 void OnRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
339 {
340 if (GetTargetApplication()->GetRemoveMode() != AURA_REMOVE_BY_DEATH)
341 if (Unit* caster = GetCaster())
342 if (caster->IsAlive())
343 if (Creature* creature = caster->ToCreature())
344 creature->DespawnOrUnsummon(1000);
345 }
346
347 void Register() override
348 {
349 AfterEffectRemove += AuraEffectRemoveFn(spell_frost_tomb_AuraScript::OnRemove, EFFECT_0, SPELL_AURA_MOD_STUN, AURA_EFFECT_HANDLE_REAL);
350 }
351 };
352
353 AuraScript* GetAuraScript() const override
354 {
355 return new spell_frost_tomb_AuraScript();
356 }

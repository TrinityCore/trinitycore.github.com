29enum Texts
30{
31 // Texts are common for both bosses and their ghosts.
32 SAY_AGGRO = 0,
33 SAY_DEATH = 1, // Said once both bosses are dead.
34 SAY_DIED_FIRST = 2, // Said by the first boss that dies.
35 SAY_KILL = 3,
36 SAY_DEATH_RESPONSE = 4 // Said by the boss alive after the first one dies.
37
38};

40enum Spells
41{
42 // Spells of Skarvald and his Ghost
43 SPELL_CHARGE = 43651,
44 SPELL_STONE_STRIKE = 48583,
45 SPELL_ENRAGE = 48193,
46 SPELL_SUMMON_SKARVALD_GHOST = 48613,
47
48 // Spells of Dalronn and his Ghost
49 SPELL_SHADOW_BOLT = 43649,
50 SPELL_SUMMON_SKELETONS = 52611,

55enum Events
56{
57 // Skarvald the Constructor
58 EVENT_SKARVALD_CHARGE = 1,
59 EVENT_STONE_STRIKE,
60
61 // Dalronn the Controller
62 EVENT_SHADOW_BOLT,
63 EVENT_DEBILITATE,
64 EVENT_SUMMON_SKELETONS,
65 EVENT_DELAYED_AGGRO_SAY, // Dalronn's SAY_AGGRO is delayed so it doesn't overlap Skarvald's one.
66

68 // Delays SAY_DEATH_RESPONSE so it doesn't overlap with the SAY_DIED_FIRST from the boss that has just died.

72enum Actions

74 ACTION_OTHER_JUST_DIED = 1,
75 ACTION_DESPAWN_SUMMONS = 2 // Only needed to clear off the ghosts when the second boss dies.

78class SkarvaldChargePredicate
79{
80 public:
81 SkarvaldChargePredicate(Unit* unit) : _me(unit) { }
82
83 bool operator() (WorldObject* target) const
84 {
85 return target->GetDistance2d(_me) >= 5.0f && target->GetDistance2d(_me) <= 30.0f;
86 }
87
88 private:
89 Unit* _me;
90};

92struct generic_boss_controllerAI : public BossAI
93{
94 generic_boss_controllerAI(Creature* creature) : BossAI(creature, DATA_SKARVALD_DALRONN)
95 {
96 OtherBossData = 0;
97 IsInGhostForm = me->GetEntry() == NPC_SKARVALD_GHOST || me->GetEntry() == NPC_DALRONN_GHOST;
98 }
99
100 void Reset() override
101 {
102 if (IsInGhostForm)
103 {
104 // Call this here since ghosts aren't set in combat as they spawn.
105 DoZoneInCombat(me, 50.0f);
106 }
107 else
108 _Reset();
109 }
110
111 void EnterCombat(Unit* /*who*/) override
112 {
113 if (!IsInGhostForm)
114 _EnterCombat();
115 }
116
117 void JustDied(Unit* /*killer*/) override
118 {
119 if (Creature* otherBoss = ObjectAccessor::GetCreature(*me, instance->GetGuidData(OtherBossData)))
120 {
121 if (otherBoss->IsAlive())
122 {
123 Talk(SAY_DIED_FIRST);
124 me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
125 otherBoss->AI()->DoAction(ACTION_OTHER_JUST_DIED);
126 DoCast(me, OtherBossData == DATA_DALRONN ? SPELL_SUMMON_SKARVALD_GHOST : SPELL_SUMMON_DALRONN_GHOST, true);
127 }
128 else
129 {
130 Talk(SAY_DEATH);
131 otherBoss->AI()->DoAction(ACTION_DESPAWN_SUMMONS);
132 _JustDied();
133 }
134 }
135 }
136
137 void DoAction(int32 actionId) override
138 {
139 switch (actionId)
140 {
141 case ACTION_OTHER_JUST_DIED:
142 events.ScheduleEvent(EVENT_DEATH_RESPONSE, 2000);
143 break;
144 case ACTION_DESPAWN_SUMMONS:
145 summons.DespawnAll();
146 break;
147 default:
148 break;
149 }
150 }
151
152 void ExecuteEvent(uint32 eventId) override
153 {
154 if (eventId == EVENT_DEATH_RESPONSE)
155 Talk(SAY_DEATH_RESPONSE);
156 }
157
158 void KilledUnit(Unit* who) override
159 {
160 if (!IsInGhostForm && who->GetTypeId() == TYPEID_PLAYER)
161 Talk(SAY_KILL);
162 }
163
164 protected:
165 uint32 OtherBossData;
166 bool IsInGhostForm;
167};

169class boss_skarvald_the_constructor : public CreatureScript
170{
171 public:
172 boss_skarvald_the_constructor() : CreatureScript("boss_skarvald_the_constructor") { }
173
174 struct boss_skarvald_the_constructorAI : public generic_boss_controllerAI
175 {
176 boss_skarvald_the_constructorAI(Creature* creature) : generic_boss_controllerAI(creature)
177 {
178 OtherBossData = DATA_DALRONN;
179 Enraged = false;
180 }
181
182 void Reset() override
183 {
184 Enraged = false;
185 generic_boss_controllerAI::Reset();
186 }
187
188 void EnterCombat(Unit* who) override
189 {
190 generic_boss_controllerAI::EnterCombat(who);
191
192 if (!IsInGhostForm)
193 Talk(SAY_AGGRO);
194
195 events.ScheduleEvent(EVENT_SKARVALD_CHARGE, 5000);
196 events.ScheduleEvent(EVENT_STONE_STRIKE, 10000);
197 }
198
199 void ExecuteEvent(uint32 eventId) override
200 {
201 switch (eventId)
202 {
203 case EVENT_SKARVALD_CHARGE:
204 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, SkarvaldChargePredicate(me)))
205 DoCast(target, SPELL_CHARGE);
206 events.ScheduleEvent(EVENT_CHARGE, urand(5000, 10000));
207 break;
208 case EVENT_STONE_STRIKE:
209 DoCastVictim(SPELL_STONE_STRIKE);
210 events.ScheduleEvent(EVENT_STONE_STRIKE, urand(5000, 10000));
211 break;
212 default:
213 generic_boss_controllerAI::ExecuteEvent(eventId);
214 break;
215 }
216 }
217
218 void DamageTaken(Unit* /*attacker*/, uint32& damage) override
219 {
220 if (!Enraged && !IsInGhostForm && me->HealthBelowPctDamaged(15, damage))
221 {
222 Enraged = true;
223 DoCast(me, SPELL_ENRAGE);
224 }
225 }
226 private:
227 bool Enraged;
228 };
229
230 CreatureAI* GetAI(Creature* creature) const override
231 {
232 return GetUtgardeKeepAI<boss_skarvald_the_constructorAI>(creature);
233 }
234};

236class boss_dalronn_the_controller : public CreatureScript
237{
238 public:
239 boss_dalronn_the_controller() : CreatureScript("boss_dalronn_the_controller") { }
240
241 struct boss_dalronn_the_controllerAI : public generic_boss_controllerAI
242 {
243 boss_dalronn_the_controllerAI(Creature* creature) : generic_boss_controllerAI(creature)
244 {
245 OtherBossData = DATA_SKARVALD;
246 }
247
248 void EnterCombat(Unit* who) override
249 {
250 generic_boss_controllerAI::EnterCombat(who);
251
252 events.ScheduleEvent(EVENT_SHADOW_BOLT, 1000);
253 events.ScheduleEvent(EVENT_DEBILITATE, 5000);
254
255 if (!IsInGhostForm)
256 events.ScheduleEvent(EVENT_DELAYED_AGGRO_SAY, 5000);
257
258 if (IsHeroic())
259 events.ScheduleEvent(EVENT_SUMMON_SKELETONS, 10000);
260 }
261
262 void ExecuteEvent(uint32 eventId) override
263 {
264 switch (eventId)
265 {
266 case EVENT_SHADOW_BOLT:
267 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f, true))
268 DoCast(target, SPELL_SHADOW_BOLT);
269 events.ScheduleEvent(EVENT_SHADOW_BOLT, 2100); //give a 100ms pause to try cast other spells
270 break;
271 case EVENT_DEBILITATE:
272 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true))
273 DoCast(target, SPELL_DEBILITATE);
274 events.ScheduleEvent(EVENT_DEBILITATE, urand(5000, 10000));
275 break;
276 case EVENT_SUMMON_SKELETONS:
277 DoCast(me, SPELL_SUMMON_SKELETONS);
278 events.ScheduleEvent(EVENT_SUMMON_SKELETONS, urand(10000, 30000));
279 break;
280 case EVENT_DELAYED_AGGRO_SAY:
281 Talk(SAY_AGGRO);
282 break;
283 default:
284 generic_boss_controllerAI::ExecuteEvent(eventId);
285 break;
286 }
287 }
288 };
289
290 CreatureAI* GetAI(Creature* creature) const override
291 {
292 return GetUtgardeKeepAI<boss_dalronn_the_controllerAI>(creature);
293 }

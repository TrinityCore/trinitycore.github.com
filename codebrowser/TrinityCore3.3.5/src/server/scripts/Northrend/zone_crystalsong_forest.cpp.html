34enum Spells

39enum NPCs
40{
41 NPC_TRANSITUS_SHIELD_DUMMY = 27306,
42 NPC_WARMAGE_SARINA = 32369,
43 NPC_WARMAGE_HALISTER = 32371,
44 NPC_WARMAGE_ILSUDRIA = 32372
45};
46
47class npc_warmage_violetstand : public CreatureScript
48{
49public:
50 npc_warmage_violetstand() : CreatureScript("npc_warmage_violetstand") { }
51
52 struct npc_warmage_violetstandAI : public ScriptedAI
53 {
54 npc_warmage_violetstandAI(Creature* creature) : ScriptedAI(creature)
55 {
56 SetCombatMovement(false);
57 }
58
59 ObjectGuid targetGUID;
60
61 void Reset() override
62 {
63 targetGUID.Clear();
64 }
65
66 void UpdateAI(uint32 /*diff*/) override
67 {
68 if (me->IsNonMeleeSpellCast(false))
69 return;
70
71 if (me->GetEntry() == NPC_WARMAGE_SARINA)
72 {
73 if (!targetGUID)
74 {
75 std::list<Creature*> orbList;
76 GetCreatureListWithEntryInGrid(orbList, me, NPC_TRANSITUS_SHIELD_DUMMY, 32.0f);
77 if (!orbList.empty())
78 {
79 for (Creature* orb : orbList)
80 {
81 if (orb->GetPositionY() < 1000)
82 {
83 targetGUID = orb->GetGUID();
84 break;
85 }
86 }
87 }
88 }
89 }
90 else
91 {
92 if (!targetGUID)
93 if (Creature* pOrb = GetClosestCreatureWithEntry(me, NPC_TRANSITUS_SHIELD_DUMMY, 32.0f))
94 targetGUID = pOrb->GetGUID();
95
96 }
97
98 if (Creature* pOrb = ObjectAccessor::GetCreature(*me, targetGUID))
99 DoCast(pOrb, SPELL_TRANSITUS_SHIELD_BEAM);
100
101 }
102 };
103
104 CreatureAI* GetAI(Creature* creature) const override
105 {
106 return new npc_warmage_violetstandAI(creature);
107 }
108};
109
110void AddSC_crystalsong_forest()
111{
112 new npc_warmage_violetstand;
113}

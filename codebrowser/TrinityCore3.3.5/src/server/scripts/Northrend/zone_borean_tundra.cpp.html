69 npc_sinkhole_kill_creditAI(Creature* creature) : ScriptedAI(creature)
70 {
71 Initialize();
72 }
73
74 void Initialize()
75 {
76 phaseTimer = 500;
77 phase = 0;
78 casterGuid.Clear();
79 }
80
81 uint32 phaseTimer;
82 uint8 phase;
83 ObjectGuid casterGuid;
84
85 void Reset() override
86 {
87 Initialize();
88 }
89
90 void SpellHit(Unit* caster, const SpellInfo* spell) override
91 {
92 if (phase || spell->Id != SPELL_SET_CART)
93 return;
94
95 Player* player = caster->ToPlayer();
96 if (player && player->GetQuestStatus(QUEST_PLUG_THE_SINKHOLES) == QUEST_STATUS_INCOMPLETE)
97 {
98 phase = 1;
99 casterGuid = caster->GetGUID();
100 }
101 }
102
103 void EnterCombat(Unit* /*who*/) override { }
104
105 void UpdateAI(uint32 diff) override
106 {
107 if (!phase)
108 return;
109
110 if (phaseTimer <= diff)
111 {
112 switch (phase)
113 {
114 case 1:
115 DoCast(me, SPELL_EXPLODE_CART, true);
116 DoCast(me, SPELL_SUMMON_CART, true);
117 if (GameObject* cart = me->FindNearestGameObject(GO_EXPLOSIVES_CART, 3.0f))
118 cart->SetUInt32Value(GAMEOBJECT_FACTION, 14);
119 phaseTimer = 3000;
120 phase = 2;
121 break;
122 case 2:
123 if (GameObject* cart = me->FindNearestGameObject(GO_EXPLOSIVES_CART, 3.0f))
124 cart->UseDoorOrButton();
125 DoCast(me, SPELL_EXPLODE_CART, true);
126 phaseTimer = 3000;
127 phase = 3;
128 break;
129 case 3:
130 DoCast(me, SPELL_EXPLODE_CART, true);
131 phaseTimer = 2000;
132 phase = 5; // @fixme: phase 4 is missing...
133 break;
134 case 5:
135 DoCast(me, SPELL_SUMMON_WORM, true);
136 if (Unit* worm = me->FindNearestCreature(NPC_SCOURGED_BURROWER, 3.0f))
137 {
138 worm->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
139 worm->HandleEmoteCommand(EMOTE_ONESHOT_EMERGE);
140 }
141 phaseTimer = 1000;
142 phase = 6;
143 break;
144 case 6:
145 DoCast(me, SPELL_EXPLODE_CART, true);
146 if (Unit* worm = me->FindNearestCreature(NPC_SCOURGED_BURROWER, 3.0f))
147 {
148 me->Kill(worm);
149 worm->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
150 }
151 phaseTimer = 2000;
152 phase = 7;
153 break;
154 case 7:
155 DoCast(me, SPELL_EXPLODE_CART, true);
156 if (Player* caster = ObjectAccessor::GetPlayer(*me, casterGuid))
157 caster->KilledMonster(me->GetCreatureTemplate(), me->GetGUID());
158 phaseTimer = 5000;
159 phase = 8;
160 break;
161 case 8:
162 EnterEvadeMode();
163 break;
164 }
165 } else phaseTimer -= diff;
166
167 }
168
169 };
170
171 CreatureAI* GetAI(Creature* creature) const override
172 {
173 return new npc_sinkhole_kill_creditAI(creature);
174 }

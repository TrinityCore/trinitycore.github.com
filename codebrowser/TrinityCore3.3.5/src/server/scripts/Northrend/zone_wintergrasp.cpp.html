113class npc_wg_demolisher_engineer : public CreatureScript
114{
115 public:
116 npc_wg_demolisher_engineer() : CreatureScript("npc_wg_demolisher_engineer") { }
117
118 bool OnGossipHello(Player* player, Creature* creature) override
119 {
120 if (creature->IsQuestGiver())
121 player->PrepareQuestMenu(creature->GetGUID());
122
123 if (CanBuild(creature))
124 {
125 if (player->HasAura(SPELL_CORPORAL))
126 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HELLO_DEMO1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
127 else if (player->HasAura(SPELL_LIEUTENANT))
128 {
129 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HELLO_DEMO1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
130 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HELLO_DEMO2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
131 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HELLO_DEMO3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
132 }
133 }
134 else
135 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_HELLO_DEMO4, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 9);
136
137 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
138 return true;
139 }
140
141 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
142 {
143 player->CLOSE_GOSSIP_MENU();
144
145 if (CanBuild(creature))
146 {
147 switch (action - GOSSIP_ACTION_INFO_DEF)
148 {
149 case 0:
150 creature->CastSpell(player, SPELL_BUILD_CATAPULT_FORCE, true);
151 break;
152 case 1:
153 creature->CastSpell(player, SPELL_BUILD_DEMOLISHER_FORCE, true);
154 break;
155 case 2:
156 creature->CastSpell(player, player->GetTeamId() == TEAM_ALLIANCE ? SPELL_BUILD_SIEGE_VEHICLE_FORCE_ALLIANCE : SPELL_BUILD_SIEGE_VEHICLE_FORCE_HORDE, true);
157 break;
158 }
159 if (Creature* controlArms = creature->FindNearestCreature(NPC_WINTERGRASP_CONTROL_ARMS, 30.0f, true))
160 creature->CastSpell(controlArms, SPELL_ACTIVATE_CONTROL_ARMS, true);
161 }
162 return true;
163 }
164
165 private:
166 bool CanBuild(Creature* creature)
167 {
168 Battlefield* wintergrasp = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG);
169 if (!wintergrasp)
170 return false;
171
172 switch (creature->GetEntry())
173 {
174 case NPC_GOBLIN_MECHANIC:
175 return (wintergrasp->GetData(BATTLEFIELD_WG_DATA_MAX_VEHICLE_H) > wintergrasp->GetData(BATTLEFIELD_WG_DATA_VEHICLE_H));
176 case NPC_GNOMISH_ENGINEER:
177 return (wintergrasp->GetData(BATTLEFIELD_WG_DATA_MAX_VEHICLE_A) > wintergrasp->GetData(BATTLEFIELD_WG_DATA_VEHICLE_A));
178 default:
179 return false;
180 }
181 }

189 bool OnGossipHello(Player* player, Creature* creature) override

192 player->PrepareQuestMenu(creature->GetGUID());
193
194 Battlefield* wintergrasp = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG);

198 GraveyardVect graveyard = wintergrasp->GetGraveyardVector();

200 if (graveyard[i]->GetControlTeamId() == player->GetTeamId())
201 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, player->GetSession()->GetTrinityString(((BfGraveyardWG*)graveyard[i])->GetTextId()), GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + i);

204 return true;
205 }
206
207 bool OnGossipSelect(Player* player, Creature* /*creature*/, uint32 /*sender*/, uint32 action) override
208 {
209 player->CLOSE_GOSSIP_MENU();
210
211 Battlefield* wintergrasp = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG);
212 if (wintergrasp)
213 {
214 GraveyardVect gy = wintergrasp->GetGraveyardVector();
215 for (uint8 i = 0; i < gy.size(); i++)
216 if (action - GOSSIP_ACTION_INFO_DEF == i && gy[i]->GetControlTeamId() == player->GetTeamId())
217 if (WorldSafeLocsEntry const* safeLoc = sWorldSafeLocsStore.LookupEntry(gy[i]->GetGraveyardId()))
218 player->TeleportTo(safeLoc->map_id, safeLoc->x, safeLoc->y, safeLoc->z, 0);
219 }
220 return true;
221 }

250 struct npc_wg_queueAI : public ScriptedAI

253 {
254 FrostArmor_Timer = 0;
255 }
256
257 uint32 FrostArmor_Timer;
258
259 void Reset() override
260 {
261 FrostArmor_Timer = 0;
262 }
263
264 void EnterCombat(Unit* /*who*/) override { }
265
266 void UpdateAI(uint32 diff) override
267 {
268 if (FrostArmor_Timer <= diff)
269 {

276 }

280 {

282 }
283
284 bool OnGossipHello(Player* player, Creature* creature) override
285 {
286 if (creature->IsQuestGiver())
287 player->PrepareQuestMenu(creature->GetGUID());
288
289 Battlefield* wintergrasp = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG);
290 if (!wintergrasp)
291 return true;
292
293 if (wintergrasp->IsWarTime())
294 {
295 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, player->GetSession()->GetTrinityString(WG_NPCQUEUE_TEXTOPTION_JOIN), GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
296 player->SEND_GOSSIP_MENU(wintergrasp->GetDefenderTeam() ? WG_NPCQUEUE_TEXT_H_WAR : WG_NPCQUEUE_TEXT_A_WAR, creature->GetGUID());
297 }
298 else
299 {
300 uint32 timer = wintergrasp->GetTimer() / 1000;
301 player->SendUpdateWorldState(4354, time(NULL) + timer);
302 if (timer < 15 * MINUTE)
303 {
304 player->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, player->GetSession()->GetTrinityString(WG_NPCQUEUE_TEXTOPTION_JOIN), GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
305 player->SEND_GOSSIP_MENU(wintergrasp->GetDefenderTeam() ? WG_NPCQUEUE_TEXT_H_QUEUE : WG_NPCQUEUE_TEXT_A_QUEUE, creature->GetGUID());
306 }
307 else
308 player->SEND_GOSSIP_MENU(wintergrasp->GetDefenderTeam() ? WG_NPCQUEUE_TEXT_H_NOWAR : WG_NPCQUEUE_TEXT_A_NOWAR, creature->GetGUID());
309 }
310 return true;
311 }
312
313 bool OnGossipSelect(Player* player, Creature* /*creature*/, uint32 /*sender*/, uint32 /*action*/) override
314 {
315 player->CLOSE_GOSSIP_MENU();
316
317 Battlefield* wintergrasp = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG);
318 if (!wintergrasp)
319 return true;
320
321 if (wintergrasp->IsWarTime())
322 wintergrasp->InvitePlayerToWar(player);
323 else
324 {
325 uint32 timer = wintergrasp->GetTimer() / 1000;
326 if (timer < 15 * MINUTE)
327 wintergrasp->InvitePlayerToQueue(player);
328 }
329 return true;
330 }

343 {
344 if (_checkTimer <= diff)
345 {
346 if (Battlefield* wg = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG))
347 // Tabulation madness in the hole!
348 for (uint8 i = 0; i < MAX_WINTERGRASP_VEHICLES; i++)
349 if (Creature* vehicleCreature = go->FindNearestCreature(vehiclesList[i], 3.0f, true))
350 if (!vehicleCreature->HasAura(SPELL_VEHICLE_TELEPORT) && vehicleCreature->getFaction() == WintergraspFaction[wg->GetDefenderTeam()])
351 if (Creature* teleportTrigger = vehicleCreature->FindNearestCreature(NPC_WORLD_TRIGGER_LARGE_AOI_NOT_IMMUNE_PC_NPC, 100.0f, true))
352 teleportTrigger->CastSpell(vehicleCreature, SPELL_VEHICLE_TELEPORT, true);
353
354 _checkTimer = 1000;
355 }
356 else _checkTimer -= diff;
357 }
358
359 private:
360 uint32 _checkTimer;
361 };
362

364 {
365 return new go_wg_vehicle_teleporterAI(go);

371 public:
372 npc_wg_quest_giver() : CreatureScript("npc_wg_quest_giver") { }

374 bool OnGossipHello(Player* player, Creature* creature) override
375 {
376 Battlefield* wintergrasp = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG);
377 if (!wintergrasp)
378 return true;
379
380 if (creature->IsVendor())
381 {
382 player->ADD_GOSSIP_ITEM_DB(Player::GetDefaultGossipMenuForSource(creature), 0, GOSSIP_SENDER_MAIN, GOSSIP_OPTION_VENDOR);
383 player->PlayerTalkClass->GetGossipMenu().AddGossipMenuItemData(0, 0, 0);
384 }
385
386 /// @todo: move this to conditions or something else
387
388 // Player::PrepareQuestMenu(guid)
389 if (creature->IsQuestGiver())
390 {
391 QuestRelationBounds objectQR = sObjectMgr->GetCreatureQuestRelationBounds(creature->GetEntry());
392 QuestRelationBounds objectQIR = sObjectMgr->GetCreatureQuestInvolvedRelationBounds(creature->GetEntry());
393
394 QuestMenu& qm = player->PlayerTalkClass->GetQuestMenu();
395 qm.ClearMenu();
396
397 for (QuestRelations::const_iterator i = objectQIR.first; i != objectQIR.second; ++i)
398 {
399 uint32 questId = i->second;
400 QuestStatus status = player->GetQuestStatus(questId);
401 if (status == QUEST_STATUS_COMPLETE)
402 qm.AddMenuItem(questId, 4);
403 else if (status == QUEST_STATUS_INCOMPLETE)
404 qm.AddMenuItem(questId, 4);
405 //else if (status == QUEST_STATUS_AVAILABLE)
406 // qm.AddMenuItem(quest_id, 2);
407 }
408
409 for (QuestRelations::const_iterator i = objectQR.first; i != objectQR.second; ++i)
410 {
411 uint32 questId = i->second;
412 Quest const* quest = sObjectMgr->GetQuestTemplate(questId);
413 if (!quest)
414 continue;
415
416 if (!player->CanTakeQuest(quest, false))
417 continue;
418
419 switch (questId)
420 {
421 // Horde attacker
422 case QUEST_BONES_AND_ARROWS_HORDE_ATT:
423 case QUEST_JINXING_THE_WALLS_HORDE_ATT:
424 case QUEST_SLAY_THEM_ALL_HORDE_ATT:
425 case QUEST_FUELING_THE_DEMOLISHERS_HORDE_ATT:
426 case QUEST_HEALING_WITH_ROSES_HORDE_ATT:
427 case QUEST_DEFEND_THE_SIEGE_HORDE_ATT:
428 if (wintergrasp->GetAttackerTeam() != TEAM_HORDE)
429 continue;
430 break;
431 // Horde defender
432 case QUEST_BONES_AND_ARROWS_HORDE_DEF:
433 case QUEST_WARDING_THE_WALLS_HORDE_DEF:
434 case QUEST_SLAY_THEM_ALL_HORDE_DEF:
435 case QUEST_FUELING_THE_DEMOLISHERS_HORDE_DEF:
436 case QUEST_HEALING_WITH_ROSES_HORDE_DEF:
437 case QUEST_TOPPLING_THE_TOWERS_HORDE_DEF:
438 case QUEST_STOP_THE_SIEGE_HORDE_DEF:
439 if (wintergrasp->GetDefenderTeam() != TEAM_HORDE)
440 continue;
441 break;
442 // Alliance attacker
443 case QUEST_BONES_AND_ARROWS_ALLIANCE_ATT:
444 case QUEST_WARDING_THE_WARRIORS_ALLIANCE_ATT:
445 case QUEST_NO_MERCY_FOR_THE_MERCILESS_ALLIANCE_ATT:
446 case QUEST_DEFEND_THE_SIEGE_ALLIANCE_ATT:
447 case QUEST_A_RARE_HERB_ALLIANCE_ATT:
448 if (wintergrasp->GetAttackerTeam() != TEAM_ALLIANCE)
449 continue;
450 break;
451 // Alliance defender
452 case QUEST_BONES_AND_ARROWS_ALLIANCE_DEF:
453 case QUEST_WARDING_THE_WARRIORS_ALLIANCE_DEF:
454 case QUEST_NO_MERCY_FOR_THE_MERCILESS_ALLIANCE_DEF:
455 case QUEST_SHOUTHERN_SABOTAGE_ALLIANCE_DEF:
456 case QUEST_STOP_THE_SIEGE_ALLIANCE_DEF:
457 case QUEST_A_RARE_HERB_ALLIANCE_DEF:
458 if (wintergrasp->GetDefenderTeam() != TEAM_ALLIANCE)
459 continue;
460 break;
461 default:
462 break;
463 }
464
465 if (quest->IsAutoComplete())
466 qm.AddMenuItem(questId, 4);
467 else if (player->GetQuestStatus(questId) == QUEST_STATUS_NONE)
468 qm.AddMenuItem(questId, 2);
469 }
470 }
471
472 player->SEND_GOSSIP_MENU(player->GetGossipTextId(creature), creature->GetGUID());
473 return true;
474 }

396
397 void MoveInLineOfSight(Unit* who) override
398 {
399 if (!phase && who && who->GetDistance2d(me) < 10.0f)
400 if (Player* player = who->ToPlayer())
401 if (player->GetQuestStatus(QUEST_THE_MIGHTY_HEMET_NESINGWARY) == QUEST_STATUS_INCOMPLETE)
402 {
403 playerGUID = player->GetGUID();
404 if (Aura* orphanOut = player->GetAura(SPELL_ORPHAN_OUT))
405 if (orphanOut->GetCaster() && orphanOut->GetCaster()->GetEntry() == ORPHAN_WOLVAR)
406 {
407 orphanGUID = orphanOut->GetCaster()->GetGUID();
408 phase = 1;
409 }
410 }

412
413 void proceedCwEvent(const uint32 diff)
414 {
415 if (timer <= diff)
416 {
417 Player* player = ObjectAccessor::GetPlayer(*me, playerGUID);
418 Creature* orphan = ObjectAccessor::GetCreature(*me, orphanGUID);
419
420 if (!orphan || !player)
421 {
422 Reset();
423 return;
424 }
425
426 switch (phase)
427 {
428 case 1:
429 orphan->GetMotionMaster()->MovePoint(0, me->GetPositionX() + std::cos(me->GetOrientation()) * 5, me->GetPositionY() + std::sin(me->GetOrientation()) * 5, me->GetPositionZ());
430 orphan->AI()->Talk(TEXT_WOLVAR_ORPHAN_6);
431 timer = 5000;
432 break;
433 case 2:
434 orphan->SetFacingToObject(me);
435 orphan->AI()->Talk(TEXT_WOLVAR_ORPHAN_7);
436 timer = 5000;
437 break;
438 case 3:
439 Talk(TEXT_NESINGWARY_1);
440 timer = 5000;
441 break;
442 case 4:
443 orphan->AI()->Talk(TEXT_WOLVAR_ORPHAN_8);
444 timer = 5000;
445 break;
446 case 5:
447 orphan->AI()->Talk(TEXT_WOLVAR_ORPHAN_9);
448 timer = 5000;
449 break;
450 case 6:
451 orphan->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
452 player->GroupEventHappens(QUEST_THE_MIGHTY_HEMET_NESINGWARY, me);
453 Reset();
454 return;
455 }
456 ++phase;
457 }
458 else
459 timer -= diff;
460 }
461
462 void UpdateAI(uint32 diff) override
463 {
464 if (phase)
465 proceedCwEvent(diff);
466
467 if (!sayStep)
468 return;
469
470 if (sayTimer < diff)
471 {
472 Talk(SAY_HEMET_HADRIUS_TAMARA_1 + sayStep - 1);
473 sayTimer = 3000;
474 sayStep++;
475
476 if (sayStep > 3) // end
477 sayStep = 0;
478 }
479 else
480 sayTimer -= diff;
481 }
482
483 void SpellHit(Unit* caster, SpellInfo const* spellInfo) override
484 {
485 if (spellInfo->Id != SPELL_OFFER)
486 return;
487
488 Player* player = caster->ToPlayer();
489 if (!player)
490 return;
491
492 Quest const* quest = sObjectMgr->GetQuestTemplate(QUEST_TASTE_TEST);
493 if (!quest)
494 return;
495
496 QuestStatusMap::const_iterator itr = player->getQuestStatusMap().find(QUEST_TASTE_TEST);
497 if (itr->second.Status != QUEST_STATUS_INCOMPLETE)
498 return;
499
500 for (uint8 i = 0; i < 3; ++i)
501 {
502 if (uint32(quest->RequiredNpcOrGo[i]) != me->GetEntry())
503 continue;
504
505 if (itr->second.CreatureOrGOCount[i] != 0)
506 continue;
507
508 player->KilledMonsterCredit(me->GetEntry());
509 player->Say(SAY_OFFER, LANG_UNIVERSAL);
510 sayStep = 1;
511 break;
512 }
513 }
514
515 private:
516 uint16 sayTimer;
517 uint8 sayStep;
518 uint32 timer;
519 int8 phase;
520 ObjectGuid playerGUID;
521 ObjectGuid orphanGUID;
522 };

79class AreaTrigger_at_legion_teleporter : public AreaTriggerScript
80{
81 public:
82 AreaTrigger_at_legion_teleporter() : AreaTriggerScript("at_legion_teleporter") { }
83
84 bool OnTrigger(Player* player, AreaTriggerEntry const* /*trigger*/) override
85 {
86 if (player->IsAlive() && !player->IsInCombat())
87 {
88 if (player->GetTeam() == ALLIANCE && player->GetQuestRewardStatus(QUEST_GAINING_ACCESS_A))
89 {
90 player->CastSpell(player, SPELL_TELE_A_TO, false);
91 return true;
92 }
93
94 if (player->GetTeam() == HORDE && player->GetQuestRewardStatus(QUEST_GAINING_ACCESS_H))
95 {
96 player->CastSpell(player, SPELL_TELE_H_TO, false);
97 return true;
98 }
99
100 return false;
101 }
102 return false;
103 }

168class AreaTrigger_at_last_rites : public AreaTriggerScript
169{
170 public:
171 AreaTrigger_at_last_rites() : AreaTriggerScript("at_last_rites") { }
172
173 bool OnTrigger(Player* player, AreaTriggerEntry const* trigger) override
174 {
175 if (!(player->GetQuestStatus(QUEST_LAST_RITES) == QUEST_STATUS_INCOMPLETE ||
176 player->GetQuestStatus(QUEST_LAST_RITES) == QUEST_STATUS_COMPLETE ||
177 player->GetQuestStatus(QUEST_BREAKING_THROUGH) == QUEST_STATUS_INCOMPLETE ||
178 player->GetQuestStatus(QUEST_BREAKING_THROUGH) == QUEST_STATUS_COMPLETE))
179 return false;
180
181 WorldLocation pPosition;
182
183 switch (trigger->id)
184 {
185 case 5332:
186 case 5338:
187 pPosition = WorldLocation(571, 3733.68f, 3563.25f, 290.812f, 3.665192f);
188 break;
189 case 5334:
190 pPosition = WorldLocation(571, 3802.38f, 3585.95f, 49.5765f, 0.0f);
191 break;
192 case 5340:
193 if (player->GetQuestStatus(QUEST_LAST_RITES) == QUEST_STATUS_INCOMPLETE ||
194 player->GetQuestStatus(QUEST_LAST_RITES) == QUEST_STATUS_COMPLETE)
195 pPosition = WorldLocation(571, 3687.91f, 3577.28f, 473.342f);
196 else
197 pPosition = WorldLocation(571, 3739.38f, 3567.09f, 341.58f);
198 break;
199 default:
200 return false;
201 }
202
203 player->TeleportTo(pPosition);
204
205 return false;
206 }

292{
293 NPC_TAPPER_SWINDLEKEG = 24711,
294 NPC_IPFELKOFER_IRONKEG = 24710,
295
296 AT_BREWFEST_DUROTAR = 4829,
297 AT_BREWFEST_DUN_MOROGH = 4820,
298
299 SAY_WELCOME = 4,
300
301 AREATRIGGER_TALK_COOLDOWN = 5, // in seconds
302};
303
304class AreaTrigger_at_brewfest : public AreaTriggerScript
305{
306 public:
307 AreaTrigger_at_brewfest() : AreaTriggerScript("at_brewfest")
308 {
309 // Initialize for cooldown
310 _triggerTimes[AT_BREWFEST_DUROTAR] = _triggerTimes[AT_BREWFEST_DUN_MOROGH] = 0;
311 }
312
313 bool OnTrigger(Player* player, AreaTriggerEntry const* trigger) override
314 {
315 uint32 triggerId = trigger->id;
316 // Second trigger happened too early after first, skip for now
317 if (sWorld->GetGameTime() - _triggerTimes[triggerId] < AREATRIGGER_TALK_COOLDOWN)
318 return false;
319
320 switch (triggerId)
321 {
322 case AT_BREWFEST_DUROTAR:
323 if (Creature* tapper = player->FindNearestCreature(NPC_TAPPER_SWINDLEKEG, 20.0f))
324 tapper->AI()->Talk(SAY_WELCOME, player);
325 break;
326 case AT_BREWFEST_DUN_MOROGH:
327 if (Creature* ipfelkofer = player->FindNearestCreature(NPC_IPFELKOFER_IRONKEG, 20.0f))
328 ipfelkofer->AI()->Talk(SAY_WELCOME, player);
329 break;
330 default:
331 break;
332 }
333
334 _triggerTimes[triggerId] = sWorld->GetGameTime();
335 return false;
336 }
337
338 private:
339 std::map<uint32, time_t> _triggerTimes;
340};

348 SPELL_A52_NEURALYZER = 34400,
349 NPC_SPOTLIGHT = 19913,

351
352 AT_AREA_52_SOUTH = 4472,
353 AT_AREA_52_NORTH = 4466,
354 AT_AREA_52_WEST = 4471,
355 AT_AREA_52_EAST = 4422,
356};

358class AreaTrigger_at_area_52_entrance : public AreaTriggerScript
359{
360 public:
361 AreaTrigger_at_area_52_entrance() : AreaTriggerScript("at_area_52_entrance")
362 {
363 _triggerTimes[AT_AREA_52_SOUTH] = _triggerTimes[AT_AREA_52_NORTH] = _triggerTimes[AT_AREA_52_WEST] = _triggerTimes[AT_AREA_52_EAST] = 0;
364 }
365
366 bool OnTrigger(Player* player, AreaTriggerEntry const* trigger) override
367 {
368 float x = 0.0f, y = 0.0f, z = 0.0f;
369
370 if (!player->IsAlive())
371 return false;
372
373 uint32 triggerId = trigger->id;
374 if (sWorld->GetGameTime() - _triggerTimes[trigger->id] < SUMMON_COOLDOWN)
375 return false;
376
377 switch (triggerId)
378 {
379 case AT_AREA_52_EAST:
380 x = 3044.176f;
381 y = 3610.692f;
382 z = 143.61f;
383 break;
384 case AT_AREA_52_NORTH:
385 x = 3114.87f;
386 y = 3687.619f;
387 z = 143.62f;
388 break;
389 case AT_AREA_52_WEST:
390 x = 3017.79f;
391 y = 3746.806f;
392 z = 144.27f;
393 break;
394 case AT_AREA_52_SOUTH:
395 x = 2950.63f;
396 y = 3719.905f;
397 z = 143.33f;
398 break;
399 }
400
401 player->SummonCreature(NPC_SPOTLIGHT, x, y, z, 0.0f, TEMPSUMMON_TIMED_DESPAWN, 5000);
402 player->AddAura(SPELL_A52_NEURALYZER, player);
403 _triggerTimes[trigger->id] = sWorld->GetGameTime();
404 return false;
405 }
406
407 private:
408 std::map<uint32, time_t> _triggerTimes;

416{
417 QUEST_THE_LONESOME_WATCHER = 12877,

420 NPC_STORMFORGED_ERADICTOR = 29861,
421

425
426Position const stormforgedMonitorPosition = {6963.95f, 45.65f, 818.71f, 4.948f};
427Position const stormforgedEradictorPosition = {6983.18f, 7.15f, 806.33f, 2.228f};
428
429class AreaTrigger_at_frostgrips_hollow : public AreaTriggerScript
430{
431public:
432 AreaTrigger_at_frostgrips_hollow() : AreaTriggerScript("at_frostgrips_hollow")
433 {
434 stormforgedMonitorGUID.Clear();
435 stormforgedEradictorGUID.Clear();
436 }
437
438 bool OnTrigger(Player* player, AreaTriggerEntry const* /* trigger */) override
439 {
440 if (player->GetQuestStatus(QUEST_THE_LONESOME_WATCHER) != QUEST_STATUS_INCOMPLETE)
441 return false;
442
443 Creature* stormforgedMonitor = ObjectAccessor::GetCreature(*player, stormforgedMonitorGUID);
444 if (stormforgedMonitor)
445 return false;
446
447 Creature* stormforgedEradictor = ObjectAccessor::GetCreature(*player, stormforgedEradictorGUID);
448 if (stormforgedEradictor)
449 return false;
450
451 stormforgedMonitor = player->SummonCreature(NPC_STORMFORGED_MONITOR, stormforgedMonitorPosition, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
452 if (stormforgedMonitor)
453 {
454 stormforgedMonitorGUID = stormforgedMonitor->GetGUID();
455 stormforgedMonitor->SetWalk(false);
456 /// The npc would search an alternative way to get to the last waypoint without this unit state.
457 stormforgedMonitor->AddUnitState(UNIT_STATE_IGNORE_PATHFINDING);
458 stormforgedMonitor->GetMotionMaster()->MovePath(NPC_STORMFORGED_MONITOR * 100, false);
459 }
460
461 stormforgedEradictor = player->SummonCreature(NPC_STORMFORGED_ERADICTOR, stormforgedEradictorPosition, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
462 if (stormforgedEradictor)
463 {
464 stormforgedEradictorGUID = stormforgedEradictor->GetGUID();
465 stormforgedEradictor->GetMotionMaster()->MovePath(NPC_STORMFORGED_ERADICTOR * 100, false);
466 }
467
468 return true;
469 }
470
471private:
472 ObjectGuid stormforgedMonitorGUID;
473 ObjectGuid stormforgedEradictorGUID;
474};

42enum OnlyForFlight
43{
44 SPELL_ARCANE_CHARGES = 45072
45};
46
47class item_only_for_flight : public ItemScript
48{
49public:
50 item_only_for_flight() : ItemScript("item_only_for_flight") { }
51
52 bool OnUse(Player* player, Item* item, SpellCastTargets const& /*targets*/) override
53 {
54 uint32 itemId = item->GetEntry();
55 bool disabled = false;
56
57 //for special scripts
58 switch (itemId)
59 {
60 case 24538:
61 if (player->GetAreaId() != 3628)
62 disabled = true;
63 break;
64 case 34489:
65 if (player->GetZoneId() != 4080)
66 disabled = true;
67 break;
68 case 34475:
69 if (const SpellInfo* spellInfo = sSpellMgr->GetSpellInfo(SPELL_ARCANE_CHARGES))
70 Spell::SendCastResult(player, spellInfo, 1, SPELL_FAILED_NOT_ON_GROUND);
71 break;
72 }
73
74 // allow use in flight only
75 if (player->IsInFlight() && !disabled)
76 return false;
77
78 // error
79 player->SendEquipError(EQUIP_ERR_CANT_DO_RIGHT_NOW, item, NULL);
80 return true;
81 }
82};

221class item_pile_fake_furs : public ItemScript
222{
223public:
224 item_pile_fake_furs() : ItemScript("item_pile_fake_furs") { }
225
226 bool OnUse(Player* player, Item* /*item*/, SpellCastTargets const & /*targets*/) override
227 {
228 GameObject* go = NULL;
229 for (uint8 i = 0; i < CaribouTrapsNum; ++i)
230 {
231 go = player->FindNearestGameObject(CaribouTraps[i], 5.0f);
232 if (go)
233 break;
234 }
235
236 if (!go)
237 return false;
238
239 if (go->FindNearestCreature(NPC_NESINGWARY_TRAPPER, 10.0f, true) || go->FindNearestCreature(NPC_NESINGWARY_TRAPPER, 10.0f, false) || go->FindNearestGameObject(GO_HIGH_QUALITY_FUR, 2.0f))
240 return true;
241
242 float x, y, z;
243 go->GetClosePoint(x, y, z, go->GetObjectSize() / 3, 7.0f);
244 go->SummonGameObject(GO_HIGH_QUALITY_FUR, go->GetPositionX(), go->GetPositionY(), go->GetPositionZ(), 0, 0, 0, 0, 0, 1);
245 if (TempSummon* summon = player->SummonCreature(NPC_NESINGWARY_TRAPPER, x, y, z, go->GetOrientation(), TEMPSUMMON_DEAD_DESPAWN, 1000))
246 {
247 summon->SetVisible(false);
248 summon->SetReactState(REACT_PASSIVE);
249 summon->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
250 }
251 return false;
252 }

332class item_dehta_trap_smasher : public ItemScript
333{
334public:
335 item_dehta_trap_smasher() : ItemScript("item_dehta_trap_smasher") { }
336
337 bool OnUse(Player* player, Item* /*item*/, const SpellCastTargets & /*targets*/) override
338 {
339 if (player->GetQuestStatus(QUEST_CANNOT_HELP_THEMSELVES) != QUEST_STATUS_INCOMPLETE)
340 return false;
341
342 Creature* pMammoth = player->FindNearestCreature(NPC_TRAPPED_MAMMOTH_CALF, 5.0f);
343 if (!pMammoth)
344 return false;
345
346 GameObject* pTrap = NULL;
347 for (uint8 i = 0; i < MammothTrapsNum; ++i)
348 {
349 pTrap = player->FindNearestGameObject(MammothTraps[i], 11.0f);
350 if (pTrap)
351 {
352 pMammoth->AI()->DoAction(1);
353 pTrap->SetGoState(GO_STATE_READY);
354 player->KilledMonsterCredit(NPC_TRAPPED_MAMMOTH_CALF);
355 return true;
356 }
357 }
358 return false;
359 }

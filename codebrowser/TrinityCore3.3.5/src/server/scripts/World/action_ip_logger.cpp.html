89 /* It's impossible to log the account logout process out of character selection - shouldn't matter anyway,
90 * as ip doesn't change through playing (obviously).*/
91 // ACCOUNT_LOGOUT = 6
92 void AccountIPLogAction(uint32 accountId, IPLoggingTypes aType)
93 {
94 // Action IP Logger is only intialized if config is set up
95 // Else, this script isn't loaded in the first place: We require no config check.
96
97 // We declare all the required variables
98 uint32 playerGuid = accountId;
99 ObjectGuid::LowType characterGuid = 0;
100 std::string systemNote = "ERROR"; // "ERROR" is a placeholder here. We change it later.
101
102 // With this switch, we change systemNote so that we have a more accurate phrasing of what type it is.
103 // Avoids Magicnumbers in SQL table
104 switch (aType)
105 {
106 case ACCOUNT_LOGIN:
107 systemNote = "Logged on Successful AccountLogin";
108 break;
109 case ACCOUNT_FAIL_LOGIN:
110 systemNote = "Logged on Failed AccountLogin";
111 break;
112 case ACCOUNT_CHANGE_PW:
113 systemNote = "Logged on Successful Account Password Change";
114 break;
115 case ACCOUNT_CHANGE_PW_FAIL:
116 systemNote = "Logged on Failed Account Password Change";
117 break;
118 case ACCOUNT_CHANGE_EMAIL:
119 systemNote = "Logged on Successful Account Email Change";
120 break;
121 case ACCOUNT_CHANGE_EMAIL_FAIL:
122 systemNote = "Logged on Failed Account Email Change";
123 break;
124 /*case ACCOUNT_LOGOUT:
125 systemNote = "Logged on AccountLogout"; //Can not be logged
126 break;*/
127 // Neither should happen. Ever. Period. If it does, call Ghostbusters and all your local software defences to investigate.
128 case UNKNOWN_ACTION:
129 default:
130 systemNote = "ERROR! Unknown action!";
131 break;
132 }
133
134 // Once we have done everything, we can insert the new log.
135 // Seeing as the time differences should be minimal, we do not get unixtime and the timestamp right now;
136 // Rather, we let it be added with the SQL query.
137 if (aType != ACCOUNT_FAIL_LOGIN)
138 {
139 // As we can assume most account actions are NOT failed login, so this is the more accurate check.
140 // For those, we need last_ip...
141 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_ALDL_IP_LOGGING);
142
143 stmt->setUInt32(0, playerGuid);
144 stmt->setUInt32(1, characterGuid);
145 stmt->setUInt8(2, aType);
146 stmt->setUInt32(3, playerGuid);
147 stmt->setString(4, systemNote.c_str());
148 LoginDatabase.Execute(stmt);
149 }
150 else // ... but for failed login, we query last_attempt_ip from account table. Which we do with an unique query
151 {
152 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_FACL_IP_LOGGING);
153
154 stmt->setUInt32(0, playerGuid);
155 stmt->setUInt32(1, characterGuid);
156 stmt->setUInt8(2, aType);
157 stmt->setUInt32(3, playerGuid);
158 stmt->setString(4, systemNote.c_str());
159 LoginDatabase.Execute(stmt);
160 }
161 return;
162 }

196 /// Logs a number of actions done by players with an IP
197 void CharacterIPLogAction(Player* player, IPLoggingTypes aType)
198 {
199 // Action IP Logger is only intialized if config is set up
200 // Else, this script isn't loaded in the first place: We require no config check.
201
202 // We declare all the required variables
203 uint32 playerGuid = player->GetSession()->GetAccountId();
204 ObjectGuid::LowType characterGuid = player->GetGUID().GetCounter();
205 const std::string currentIp = player->GetSession()->GetRemoteAddress();
206 std::string systemNote = "ERROR"; // "ERROR" is a placeholder here. We change it...
207
208 // ... with this switch, so that we have a more accurate phrasing of what type it is
209 switch (aType)
210 {
211 case CHARACTER_CREATE:
212 systemNote = "Logged on CharacterCreate";
213 break;
214 case CHARACTER_LOGIN:
215 systemNote = "Logged on CharacterLogin";
216 break;
217 case CHARACTER_LOGOUT:
218 systemNote = "Logged on CharacterLogout";
219 break;
220 case CHARACTER_DELETE:
221 systemNote = "Logged on CharacterDelete";
222 break;
223 case CHARACTER_FAILED_DELETE:
224 systemNote = "Logged on Failed CharacterDelete";
225 break;
226 // Neither should happen. Ever. Period. If it does, call Mythbusters.
227 case UNKNOWN_ACTION:
228 default:
229 systemNote = "ERROR! Unknown action!";
230 break;
231 }
232
233 // Once we have done everything, we can insert the new log.
234 PreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_CHAR_IP_LOGGING);
235
236 stmt->setUInt32(0, playerGuid);
237 stmt->setUInt32(1, characterGuid);
238 stmt->setUInt8(2, aType);
239 stmt->setString(3, currentIp.c_str()); // We query the ip here.
240 stmt->setString(4, systemNote.c_str());
241 // Seeing as the time differences should be minimal, we do not get unixtime and the timestamp right now;
242 // Rather, we let it be added with the SQL query.
243
244 LoginDatabase.Execute(stmt);
245 return;
246 }

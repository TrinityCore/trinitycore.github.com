85/*
86 * ---
87 * --- Emerald Dragons : Base AI-structure used for all the Emerald dragons
88 * ---
89 */
90
91struct emerald_dragonAI : public WorldBossAI
92{
93 emerald_dragonAI(Creature* creature) : WorldBossAI(creature)
94 {
95 }
96
97 void Reset() override
98 {
99 WorldBossAI::Reset();
100 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
101 me->SetReactState(REACT_AGGRESSIVE);
102 DoCast(me, SPELL_MARK_OF_NATURE_AURA, true);
103 events.ScheduleEvent(EVENT_TAIL_SWEEP, 4000);
104 events.ScheduleEvent(EVENT_NOXIOUS_BREATH, urand(7500, 15000));
105 events.ScheduleEvent(EVENT_SEEPING_FOG, urand(12500, 20000));
106 }
107
108 // Target killed during encounter, mark them as suspectible for Aura Of Nature
109 void KilledUnit(Unit* who) override
110 {
111 if (who->GetTypeId() == TYPEID_PLAYER)
112 who->CastSpell(who, SPELL_MARK_OF_NATURE, true);
113 }
114
115 // Execute and reschedule base events shared between all Emerald Dragons
116 void ExecuteEvent(uint32 eventId) override
117 {
118 switch (eventId)
119 {
120 case EVENT_SEEPING_FOG:
121 // Seeping Fog appears only as "pairs", and only ONE pair at any given time!
122 // Despawntime is 2 minutes, so reschedule it for new cast after 2 minutes + a minor "random time" (30 seconds at max)
123 DoCast(me, SPELL_SEEPING_FOG_LEFT, true);
124 DoCast(me, SPELL_SEEPING_FOG_RIGHT, true);
125 events.ScheduleEvent(EVENT_SEEPING_FOG, urand(120000, 150000));
126 break;
127 case EVENT_NOXIOUS_BREATH:
128 // Noxious Breath is cast on random intervals, no less than 7.5 seconds between
129 DoCast(me, SPELL_NOXIOUS_BREATH);
130 events.ScheduleEvent(EVENT_NOXIOUS_BREATH, urand(7500, 15000));
131 break;
132 case EVENT_TAIL_SWEEP:
133 // Tail Sweep is cast every two seconds, no matter what goes on in front of the dragon
134 DoCast(me, SPELL_TAIL_SWEEP);
135 events.ScheduleEvent(EVENT_TAIL_SWEEP, 2000);
136 break;
137 }
138 }
139
140 void UpdateAI(uint32 diff) override
141 {
142 if (!UpdateVictim())
143 return;
144
145 events.Update(diff);
146
147 if (me->HasUnitState(UNIT_STATE_CASTING))
148 return;
149
150 while (uint32 eventId = events.ExecuteEvent())
151 ExecuteEvent(eventId);
152
153 if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0, -50.0f, true))
154 DoCast(target, SPELL_SUMMON_PLAYER);
155
156 DoMeleeAttackIfReady();
157 }
158};

164class npc_dream_fog : public CreatureScript
165{
166 public:
167 npc_dream_fog() : CreatureScript("npc_dream_fog") { }
168
169 struct npc_dream_fogAI : public ScriptedAI
170 {
171 npc_dream_fogAI(Creature* creature) : ScriptedAI(creature)
172 {
173 Initialize();
174 }
175
176 void Initialize()
177 {
178 _roamTimer = 0;
179 }
180
181 void Reset() override
182 {
183 Initialize();
184 }
185
186 void UpdateAI(uint32 diff) override
187 {
188 if (!UpdateVictim())
189 return;
190
191 if (!_roamTimer)
192 {
193 // Chase target, but don't attack - otherwise just roam around
194 if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
195 {
196 _roamTimer = urand(15000, 30000);
197 me->GetMotionMaster()->Clear(false);
198 me->GetMotionMaster()->MoveChase(target, 0.2f);
199 }
200 else
201 {
202 _roamTimer = 2500;
203 me->GetMotionMaster()->Clear(false);
204 me->GetMotionMaster()->MoveRandom(25.0f);
205 }
206 // Seeping fog movement is slow enough for a player to be able to walk backwards and still outpace it
207 me->SetWalk(true);
208 me->SetSpeed(MOVE_WALK, 0.75f);
209 }
210 else
211 _roamTimer -= diff;
212 }
213
214 private:
215 uint32 _roamTimer;
216 };
217
218 CreatureAI* GetAI(Creature* creature) const override
219 {
220 return new npc_dream_fogAI(creature);
221 }
222};

567class boss_taerar : public CreatureScript
568{
569 public:
570 boss_taerar() : CreatureScript("boss_taerar") { }
571
572 struct boss_taerarAI : public emerald_dragonAI
573 {
574 boss_taerarAI(Creature* creature) : emerald_dragonAI(creature)
575 {
576 Initialize();
577 }
578
579 void Initialize()
580 {
581 _stage = 1;
582 _shades = 0;
583 _banished = false;
584 _banishedTimer = 0;
585 }
586
587 void Reset() override
588 {
589 me->RemoveAurasDueToSpell(SPELL_SHADE);
590
591 Initialize();
592
593 emerald_dragonAI::Reset();
594 events.ScheduleEvent(EVENT_ARCANE_BLAST, 12000);
595 events.ScheduleEvent(EVENT_BELLOWING_ROAR, 30000);
596 }
597
598 void EnterCombat(Unit* who) override
599 {
600 Talk(SAY_TAERAR_AGGRO);
601 emerald_dragonAI::EnterCombat(who);
602 }
603
604 void SummonedCreatureDies(Creature* /*summon*/, Unit* /*killer*/) override
605 {
606 --_shades;
607 }
608
609 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/) override
610 {
611 // At 75, 50 or 25 percent health, we need to activate the shades and go "banished"
612 // Note: _stage holds the amount of times they have been summoned
613 if (!_banished && !HealthAbovePct(100 - 25 * _stage))
614 {
615 _banished = true;
616 _banishedTimer = 60000;
617
618 me->InterruptNonMeleeSpells(false);
619 DoStopAttack();
620
621 Talk(SAY_TAERAR_SUMMON_SHADES);
622
623 uint32 count = sizeof(TaerarShadeSpells) / sizeof(uint32);
624 for (uint32 i = 0; i < count; ++i)
625 DoCastVictim(TaerarShadeSpells[i], true);
626 _shades += count;
627
628 DoCast(SPELL_SHADE);
629 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
630 me->SetReactState(REACT_PASSIVE);
631
632 ++_stage;
633 }
634 }
635
636 void ExecuteEvent(uint32 eventId) override
637 {
638 switch (eventId)
639 {
640 case EVENT_ARCANE_BLAST:
641 DoCast(SPELL_ARCANE_BLAST);
642 events.ScheduleEvent(EVENT_ARCANE_BLAST, urand(7000, 12000));
643 break;
644 case EVENT_BELLOWING_ROAR:
645 DoCast(SPELL_BELLOWING_ROAR);
646 events.ScheduleEvent(EVENT_BELLOWING_ROAR, urand(20000, 30000));
647 break;
648 default:
649 emerald_dragonAI::ExecuteEvent(eventId);
650 break;
651 }
652 }
653
654 void UpdateAI(uint32 diff) override
655 {
656 if (!me->IsInCombat())
657 return;
658
659 if (_banished)
660 {
661 // If all three shades are dead, OR it has taken too long, end the current event and get Taerar back into business
662 if (_banishedTimer <= diff || !_shades)
663 {
664 _banished = false;
665
666 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
667 me->RemoveAurasDueToSpell(SPELL_SHADE);
668 me->SetReactState(REACT_AGGRESSIVE);
669 }
670 // _banishtimer has not expired, and we still have active shades:
671 else
672 _banishedTimer -= diff;
673
674 // Update the events before we return (handled under emerald_dragonAI::UpdateAI(diff); if we're not inside this check)
675 events.Update(diff);
676
677 return;
678 }
679
680 emerald_dragonAI::UpdateAI(diff);
681 }
682
683 private:
684 bool _banished; // used for shades activation testing
685 uint32 _banishedTimer; // counter for banishment timeout
686 uint8 _shades; // keep track of how many shades are dead
687 uint8 _stage; // check which "shade phase" we're at (75-50-25 percentage counters)
688 };
689
690 CreatureAI* GetAI(Creature* creature) const override
691 {
692 return new boss_taerarAI(creature);
693 }

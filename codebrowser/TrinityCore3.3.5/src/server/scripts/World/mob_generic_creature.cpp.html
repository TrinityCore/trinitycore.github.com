30#define GENERIC_CREATURE_COOLDOWN 5000
31
32class generic_creature : public CreatureScript
33{
34public:
35 generic_creature() : CreatureScript("generic_creature") { }
36
37 struct generic_creatureAI : public ScriptedAI
38 {
39 generic_creatureAI(Creature* creature) : ScriptedAI(creature)
40 {
41 Initialize();
42 }
43
44 void Initialize()
45 {
46 GlobalCooldown = 0;
47 BuffTimer = 0; //Rebuff as soon as we can
48 IsSelfRooted = false;
49 }
50
51 uint32 GlobalCooldown; //This variable acts like the global cooldown that players have (1.5 seconds)
52 uint32 BuffTimer; //This variable keeps track of buffs
53 bool IsSelfRooted;
54
55 void Reset() override
56 {
57 Initialize();
58 }
59
60 void EnterCombat(Unit* who) override
61 {
62 if (!me->IsWithinMeleeRange(who))
63 IsSelfRooted = true;
64 }
65
66 void UpdateAI(uint32 diff) override
67 {
68 //Always decrease our global cooldown first
69 if (GlobalCooldown > diff)
70 GlobalCooldown -= diff;
71 else GlobalCooldown = 0;
72
73 //Buff timer (only buff when we are alive and not in combat
74 if (!me->IsInCombat() && me->IsAlive())
75 {
76 if (BuffTimer <= diff)
77 {
78 //Find a spell that targets friendly and applies an aura (these are generally buffs)
79 SpellInfo const* info = SelectSpell(me, 0, 0, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_AURA);
80
81 if (info && !GlobalCooldown)
82 {
83 //Cast the buff spell
84 DoCastSpell(me, info);
85
86 //Set our global cooldown
87 GlobalCooldown = GENERIC_CREATURE_COOLDOWN;
88
89 //Set our timer to 10 minutes before rebuff
90 BuffTimer = 600000;
91 }//Try agian in 30 seconds
92 else BuffTimer = 30000;
93 } else BuffTimer -= diff;
94 }
95
96 //Return since we have no target
97 if (!UpdateVictim())
98 return;
99
100 //If we are within range melee the target
101 if (me->IsWithinMeleeRange(me->GetVictim()))
102 {
103 //Make sure our attack is ready and we arn't currently casting
104 if (me->isAttackReady() && !me->IsNonMeleeSpellCast(false))
105 {
106 bool Healing = false;
107 SpellInfo const* info = NULL;
108
109 //Select a healing spell if less than 30% hp
110 if (HealthBelowPct(30))
111 info = SelectSpell(me, 0, 0, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_HEALING);
112
113 //No healing spell available, select a hostile spell
114 if (info) Healing = true;
115 else info = SelectSpell(me->GetVictim(), 0, 0, SELECT_TARGET_ANY_ENEMY, 0, 0, 0, 0, SELECT_EFFECT_DONTCARE);
116
117 //50% chance if elite or higher, 20% chance if not, to replace our white hit with a spell
118 if (info && (rand32() % (me->GetCreatureTemplate()->rank > 1 ? 2 : 5) == 0) && !GlobalCooldown)
119 {
120 //Cast the spell
121 if (Healing)DoCastSpell(me, info);
122 else DoCastSpell(me->GetVictim(), info);
123
124 //Set our global cooldown
125 GlobalCooldown = GENERIC_CREATURE_COOLDOWN;
126 }
127 else me->AttackerStateUpdate(me->GetVictim());
128
129 me->resetAttackTimer();
130 }
131 }
132 else
133 {
134 //Only run this code if we arn't already casting
135 if (!me->IsNonMeleeSpellCast(false))
136 {
137 bool Healing = false;
138 SpellInfo const* info = NULL;
139
140 //Select a healing spell if less than 30% hp ONLY 33% of the time
141 if (HealthBelowPct(30) && rand32() % 3 == 0)
142 info = SelectSpell(me, 0, 0, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_HEALING);
143
144 //No healing spell available, See if we can cast a ranged spell (Range must be greater than ATTACK_DISTANCE)
145 if (info) Healing = true;
146 else info = SelectSpell(me->GetVictim(), 0, 0, SELECT_TARGET_ANY_ENEMY, 0, 0, NOMINAL_MELEE_RANGE, 0, SELECT_EFFECT_DONTCARE);
147
148 //Found a spell, check if we arn't on cooldown
149 if (info && !GlobalCooldown)
150 {
151 //If we are currently moving stop us and set the movement generator
152 if (!IsSelfRooted)
153 IsSelfRooted = true;
154
155 //Cast spell
156 if (Healing) DoCastSpell(me, info);
157 else DoCastSpell(me->GetVictim(), info);
158
159 //Set our global cooldown
160 GlobalCooldown = GENERIC_CREATURE_COOLDOWN;
161
162 }//If no spells available and we arn't moving run to target
163 else if (IsSelfRooted)
164 {
165 //Cancel our current spell and then allow movement agian
166 me->InterruptNonMeleeSpells(false);
167 IsSelfRooted = false;
168 }
169 }
170 }
171 }
172 };
173
174 CreatureAI* GetAI(Creature* creature) const override
175 {
176 return new generic_creatureAI(creature);
177 }
178};

180class trigger_periodic : public CreatureScript
181{
182public:
183 trigger_periodic() : CreatureScript("trigger_periodic") { }
184
185 struct trigger_periodicAI : public NullCreatureAI
186 {
187 trigger_periodicAI(Creature* creature) : NullCreatureAI(creature)
188 {
189 spell = me->m_spells[0] ? sSpellMgr->GetSpellInfo(me->m_spells[0]) : NULL;
190 interval = me->GetAttackTime(BASE_ATTACK);
191 timer = interval;
192 }
193
194 uint32 timer, interval;
195 const SpellInfo* spell;
196
197 void UpdateAI(uint32 diff) override
198 {
199 if (timer <= diff)
200 {
201 if (spell)
202 me->CastSpell(me, spell, true);
203 timer = interval;
204 }
205 else
206 timer -= diff;
207 }
208 };
209
210 CreatureAI* GetAI(Creature* creature) const override
211 {
212 return new trigger_periodicAI(creature);
213 }
214};

223 trigger_deathAI(Creature* creature) : NullCreatureAI(creature) { }
224 void JustDied(Unit* killer) override
225 {
226 if (me->m_spells[0])
227 me->CastSpell(killer, me->m_spells[0], true);
228 }
229 };
230
231 CreatureAI* GetAI(Creature* creature) const override
232 {
233 return new trigger_deathAI(creature);

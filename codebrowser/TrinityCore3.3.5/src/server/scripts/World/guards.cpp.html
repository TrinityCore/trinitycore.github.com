38enum GuardGeneric
39{
40 GENERIC_CREATURE_COOLDOWN = 5000,
41
42 SAY_GUARD_SIL_AGGRO = 0,
43

50class guard_generic : public CreatureScript
51{
52public:
53 guard_generic() : CreatureScript("guard_generic") { }

55 struct guard_genericAI : public GuardAI
56 {
57 guard_genericAI(Creature* creature) : GuardAI(creature)
58 {
59 Initialize();
60 }
61
62 void Initialize()
63 {
64 globalCooldown = 0;
65 buffTimer = 0;
66 }
67
68 void Reset() override
69 {
70 Initialize();
71 }
72
73 void EnterCombat(Unit* who) override
74 {
75 if (me->GetEntry() == NPC_CENARION_HOLD_INFANTRY)
76 Talk(SAY_GUARD_SIL_AGGRO, who);
77 if (SpellInfo const* spell = me->reachWithSpellAttack(who))
78 DoCast(who, spell->Id);
79 }
80
81 void UpdateAI(uint32 diff) override
82 {
83 //Always decrease our global cooldown first
84 if (globalCooldown > diff)
85 globalCooldown -= diff;
86 else
87 globalCooldown = 0;
88
89 //Buff timer (only buff when we are alive and not in combat
90 if (me->IsAlive() && !me->IsInCombat())
91 {
92 if (buffTimer <= diff)
93 {
94 //Find a spell that targets friendly and applies an aura (these are generally buffs)
95 SpellInfo const* info = SelectSpell(me, 0, 0, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_AURA);
96
97 if (info && !globalCooldown)
98 {
99 //Cast the buff spell
100 DoCast(me, info->Id);
101
102 //Set our global cooldown
103 globalCooldown = GENERIC_CREATURE_COOLDOWN;
104
105 //Set our timer to 10 minutes before rebuff
106 buffTimer = 600000;
107 } //Try again in 30 seconds
108 else buffTimer = 30000;
109 } else buffTimer -= diff;
110 }
111
112 //Return since we have no target
113 if (!UpdateVictim())
114 return;
115
116 // Make sure our attack is ready and we arn't currently casting
117 if (me->isAttackReady() && !me->IsNonMeleeSpellCast(false))
118 {
119 //If we are within range melee the target
120 if (me->IsWithinMeleeRange(me->GetVictim()))
121 {
122 bool healing = false;
123 SpellInfo const* info = NULL;
124
125 //Select a healing spell if less than 30% hp
126 if (me->HealthBelowPct(30))
127 info = SelectSpell(me, 0, 0, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_HEALING);
128
129 //No healing spell available, select a hostile spell
130 if (info)
131 healing = true;
132 else
133 info = SelectSpell(me->GetVictim(), 0, 0, SELECT_TARGET_ANY_ENEMY, 0, 0, 0, 0, SELECT_EFFECT_DONTCARE);
134
135 //20% chance to replace our white hit with a spell
136 if (info && urand(0, 99) < 20 && !globalCooldown)
137 {
138 //Cast the spell
139 if (healing)
140 DoCast(me, info->Id);
141 else
142 DoCastVictim(info->Id);
143
144 //Set our global cooldown
145 globalCooldown = GENERIC_CREATURE_COOLDOWN;
146 }
147 else
148 me->AttackerStateUpdate(me->GetVictim());
149
150 me->resetAttackTimer();
151 }
152 }
153 else
154 {
155 //Only run this code if we arn't already casting
156 if (!me->IsNonMeleeSpellCast(false))
157 {
158 bool healing = false;
159 SpellInfo const* info = NULL;
160
161 //Select a healing spell if less than 30% hp ONLY 33% of the time
162 if (me->HealthBelowPct(30) && 33 > urand(0, 99))
163 info = SelectSpell(me, 0, 0, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_HEALING);
164
165 //No healing spell available, See if we can cast a ranged spell (Range must be greater than ATTACK_DISTANCE)
166 if (info)
167 healing = true;
168 else
169 info = SelectSpell(me->GetVictim(), 0, 0, SELECT_TARGET_ANY_ENEMY, 0, 0, NOMINAL_MELEE_RANGE, 0, SELECT_EFFECT_DONTCARE);
170
171 //Found a spell, check if we arn't on cooldown
172 if (info && !globalCooldown)
173 {
174 //If we are currently moving stop us and set the movement generator
175 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() != IDLE_MOTION_TYPE)
176 {
177 me->GetMotionMaster()->Clear(false);
178 me->GetMotionMaster()->MoveIdle();
179 }
180
181 //Cast spell
182 if (healing)
183 DoCast(me, info->Id);
184 else
185 DoCastVictim(info->Id);
186
187 //Set our global cooldown
188 globalCooldown = GENERIC_CREATURE_COOLDOWN;
189 } //If no spells available and we arn't moving run to target
190 else if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() != CHASE_MOTION_TYPE)
191 {
192 //Cancel our current spell and then mutate new movement generator
193 me->InterruptNonMeleeSpells(false);
194 me->GetMotionMaster()->Clear(false);
195 me->GetMotionMaster()->MoveChase(me->GetVictim());
196 }
197 }
198 }
199
200 DoMeleeAttackIfReady();
201 }

203 void DoReplyToTextEmote(uint32 emote)
204 {
205 switch (emote)
206 {
207 case TEXT_EMOTE_KISS:
208 me->HandleEmoteCommand(EMOTE_ONESHOT_BOW);
209 break;
210
211 case TEXT_EMOTE_WAVE:
212 me->HandleEmoteCommand(EMOTE_ONESHOT_WAVE);
213 break;
214
215 case TEXT_EMOTE_SALUTE:
216 me->HandleEmoteCommand(EMOTE_ONESHOT_SALUTE);
217 break;
218
219 case TEXT_EMOTE_SHY:
220 me->HandleEmoteCommand(EMOTE_ONESHOT_FLEX);
221 break;
222
223 case TEXT_EMOTE_RUDE:
224 case TEXT_EMOTE_CHICKEN:
225 me->HandleEmoteCommand(EMOTE_ONESHOT_POINT);
226 break;
227 }
228 }
229
230 void ReceiveEmote(Player* player, uint32 textEmote) override
231 {
232 switch (me->GetEntry())
233 {
234 case NPC_STORMWIND_CITY_GUARD:
235 case NPC_STORMWIND_CITY_PATROLLER:
236 case NPC_ORGRIMMAR_GRUNT:
237 break;
238 default:
239 return;
240 }
241
242 if (!me->IsFriendlyTo(player))
243 return;
244
245 DoReplyToTextEmote(textEmote);
246 }
247
248 private:
249 uint32 globalCooldown;
250 uint32 buffTimer;
251 };

253 CreatureAI* GetAI(Creature* creature) const override
254 {
255 return new guard_genericAI(creature);
256 }

267class guard_shattrath_scryer : public CreatureScript
268{
269public:
270 guard_shattrath_scryer() : CreatureScript("guard_shattrath_scryer") { }
271
272 struct guard_shattrath_scryerAI : public GuardAI
273 {
274 guard_shattrath_scryerAI(Creature* creature) : GuardAI(creature)
275 {
276 Initialize();
277 }
278
279 void Initialize()
280 {
281 banishTimer = 5000;
282 exileTimer = 8500;
283 playerGUID.Clear();
284 canTeleport = false;
285 }
286
287 void Reset() override
288 {
289 Initialize();
290 }
291
292 void UpdateAI(uint32 diff) override
293 {
294 if (!UpdateVictim())
295 return;
296
297 if (canTeleport)
298 {
299 if (exileTimer <= diff)
300 {
301 if (Unit* temp = ObjectAccessor::GetUnit(*me, playerGUID))
302 {
303 temp->CastSpell(temp, SPELL_EXILE, true);
304 temp->CastSpell(temp, SPELL_BANISH_TELEPORT, true);
305 }
306 playerGUID.Clear();
307 exileTimer = 8500;
308 canTeleport = false;
309 } else exileTimer -= diff;
310 }
311 else if (banishTimer <= diff)
312 {
313 Unit* temp = me->GetVictim();
314 if (temp && temp->GetTypeId() == TYPEID_PLAYER)
315 {
316 DoCast(temp, SPELL_BANISHED_SHATTRATH_A);
317 banishTimer = 9000;
318 playerGUID = temp->GetGUID();
319 if (playerGUID)
320 canTeleport = true;
321 }
322 } else banishTimer -= diff;
323
324 DoMeleeAttackIfReady();
325 }
326
327 private:
328 uint32 exileTimer;
329 uint32 banishTimer;
330 ObjectGuid playerGUID;
331 bool canTeleport;
332 };
333
334 CreatureAI* GetAI(Creature* creature) const override
335 {
336 return new guard_shattrath_scryerAI(creature);
337 }

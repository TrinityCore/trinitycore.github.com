98/// Launch the Trinity server
99extern int main(int argc, char** argv)
100{
101 std::string configFile = _TRINITY_CORE_CONFIG;
102 std::string configService;
103
104 auto vm = GetConsoleArguments(argc, argv, configFile, configService);
105 // exit if help or version is enabled
106 if (vm.count("help") || vm.count("version"))
107 return 0;
108
109#ifdef _WIN32
110 if (configService.compare("install") == 0)
111 return WinServiceInstall() == true ? 0 : 1;
112 else if (configService.compare("uninstall") == 0)
113 return WinServiceUninstall() == true ? 0 : 1;
114 else if (configService.compare("run") == 0)
115 WinServiceRun();
116#endif
117
118 std::string configError;
119 if (!sConfigMgr->LoadInitial(configFile, configError))
120 {
121 printf("Error in config file: %s\n", configError.c_str());
122 return 1;
123 }
124
125 sLog->RegisterAppender<AppenderDB>();
126 // If logs are supposed to be handled async then we need to pass the io_service into the Log singleton
127 sLog->Initialize(sConfigMgr->GetBoolDefault("Log.Async.Enable", false) ? &_ioService : nullptr);
128
129 TC_LOG_INFO("server.worldserver", "%s (worldserver-daemon)", GitRevision::GetFullVersion());
130 TC_LOG_INFO("server.worldserver", "<Ctrl-C> to stop.\n");
131 TC_LOG_INFO("server.worldserver", " ______ __");
132 TC_LOG_INFO("server.worldserver", "/\\__ _\\ __ __/\\ \\__");
133 TC_LOG_INFO("server.worldserver", "\\/_/\\ \\/ _ __ /\\_\\ ___ /\\_\\ \\, _\\ __ __");
134 TC_LOG_INFO("server.worldserver", " \\ \\ \\/\\`'__\\/\\ \\ /' _ `\\/\\ \\ \\ \\/ /\\ \\/\\ \\");
135 TC_LOG_INFO("server.worldserver", " \\ \\ \\ \\ \\/ \\ \\ \\/\\ \\/\\ \\ \\ \\ \\ \\_\\ \\ \\_\\ \\");
136 TC_LOG_INFO("server.worldserver", " \\ \\_\\ \\_\\ \\ \\_\\ \\_\\ \\_\\ \\_\\ \\__\\\\/`____ \\");
137 TC_LOG_INFO("server.worldserver", " \\/_/\\/_/ \\/_/\\/_/\\/_/\\/_/\\/__/ `/___/> \\");
138 TC_LOG_INFO("server.worldserver", " C O R E /\\___/");
139 TC_LOG_INFO("server.worldserver", "http://TrinityCore.org \\/__/\n");
140 TC_LOG_INFO("server.worldserver", "Using configuration file %s.", configFile.c_str());
141 TC_LOG_INFO("server.worldserver", "Using SSL version: %s (library: %s)", OPENSSL_VERSION_TEXT, SSLeay_version(SSLEAY_VERSION));
142 TC_LOG_INFO("server.worldserver", "Using Boost version: %i.%i.%i", BOOST_VERSION / 100000, BOOST_VERSION / 100 % 1000, BOOST_VERSION % 100);
143
144 OpenSSLCrypto::threadsSetup();
145
146 // Seed the OpenSSL's PRNG here.
147 // That way it won't auto-seed when calling BigNumber::SetRand and slow down the first world login
148 BigNumber seed;
149 seed.SetRand(16 * 8);
150
151 /// worldserver PID file creation
152 std::string pidFile = sConfigMgr->GetStringDefault("PidFile", "");
153 if (!pidFile.empty())
154 {
155 if (uint32 pid = CreatePIDFile(pidFile))
156 TC_LOG_INFO("server.worldserver", "Daemon PID: %u\n", pid);
157 else
158 {
159 TC_LOG_ERROR("server.worldserver", "Cannot create PID file %s.\n", pidFile.c_str());
160 return 1;
161 }
162 }
163
164 // Set signal handlers (this must be done before starting io_service threads, because otherwise they would unblock and exit)
165 boost::asio::signal_set signals(_ioService, SIGINT, SIGTERM);
166#if PLATFORM == PLATFORM_WINDOWS
167 signals.add(SIGBREAK);
168#endif
169 signals.async_wait(SignalHandler);
170
171 // Start the Boost based thread pool
172 int numThreads = sConfigMgr->GetIntDefault("ThreadPool", 1);
173 std::vector<std::thread> threadPool;
174
175 if (numThreads < 1)
176 numThreads = 1;
177
178 for (int i = 0; i < numThreads; ++i)
179 threadPool.push_back(std::thread(boost::bind(&boost::asio::io_service::run, &_ioService)));
180
181 // Set process priority according to configuration settings
182 SetProcessPriority("server.worldserver");
183
184 // Start the databases
185 if (!StartDB())
186 {
187 ShutdownThreadPool(threadPool);
188 return 1;
189 }
190
191 // Set server offline (not connectable)
192 LoginDatabase.DirectPExecute("UPDATE realmlist SET flag = flag | %u WHERE id = '%d'", REALM_FLAG_OFFLINE, realm.Id.Realm);
193
194 LoadRealmInfo();
195
196 // Initialize the World
197 sWorld->SetInitialWorldSettings();
198
199 // Launch CliRunnable thread
200 std::thread* cliThread = nullptr;
201#ifdef _WIN32
202 if (sConfigMgr->GetBoolDefault("Console.Enable", true) && (m_ServiceStatus == -1)/* need disable console in service mode*/)
203#else
204 if (sConfigMgr->GetBoolDefault("Console.Enable", true))
205#endif
206 {
207 cliThread = new std::thread(CliThread);
208 }
209
210 // Start the Remote Access port (acceptor) if enabled
211 AsyncAcceptor* raAcceptor = nullptr;
212 if (sConfigMgr->GetBoolDefault("Ra.Enable", false))
213 raAcceptor = StartRaSocketAcceptor(_ioService);
214
215 // Start soap serving thread if enabled
216 std::thread* soapThread = nullptr;
217 if (sConfigMgr->GetBoolDefault("SOAP.Enabled", false))
218 {
219 soapThread = new std::thread(TCSoapThread, sConfigMgr->GetStringDefault("SOAP.IP", "127.0.0.1"), uint16(sConfigMgr->GetIntDefault("SOAP.Port", 7878)));
220 }
221
222 // Launch the worldserver listener socket
223 uint16 worldPort = uint16(sWorld->getIntConfig(CONFIG_PORT_WORLD));
224 std::string worldListener = sConfigMgr->GetStringDefault("BindIP", "0.0.0.0");
225
226 sWorldSocketMgr.StartNetwork(_ioService, worldListener, worldPort);
227
228 // Set server online (allow connecting now)
229 LoginDatabase.DirectPExecute("UPDATE realmlist SET flag = flag & ~%u, population = 0 WHERE id = '%u'", REALM_FLAG_OFFLINE, realm.Id.Realm);
230 realm.PopulationLevel = 0.0f;
231 realm.Flags = RealmFlags(realm.Flags & ~uint32(REALM_FLAG_OFFLINE));
232
233 // Start the freeze check callback cycle in 5 seconds (cycle itself is 1 sec)
234 if (int coreStuckTime = sConfigMgr->GetIntDefault("MaxCoreStuckTime", 0))
235 {
236 _maxCoreStuckTimeInMs = coreStuckTime * 1000;
237 _freezeCheckTimer.expires_from_now(boost::posix_time::seconds(5));
238 _freezeCheckTimer.async_wait(FreezeDetectorHandler);
239 TC_LOG_INFO("server.worldserver", "Starting up anti-freeze thread (%u seconds max stuck time)...", coreStuckTime);
240 }
241
242 TC_LOG_INFO("server.worldserver", "%s (worldserver-daemon) ready...", GitRevision::GetFullVersion());
243
244 sScriptMgr->OnStartup();
245
246 WorldUpdateLoop();
247
248 // Shutdown starts here
249 ShutdownThreadPool(threadPool);
250
251 sScriptMgr->OnShutdown();
252
253 sWorld->KickAll(); // save and kick all players
254 sWorld->UpdateSessions(1); // real players unload required UpdateSessions call
255
256 // unload battleground templates before different singletons destroyed
257 sBattlegroundMgr->DeleteAllBattlegrounds();
258
259 sWorldSocketMgr.StopNetwork();
260
261 sInstanceSaveMgr->Unload();
262 sOutdoorPvPMgr->Die();
263 sMapMgr->UnloadAll(); // unload all grids (including locked in memory)
264 sScriptMgr->Unload();
265
266 // set server offline
267 LoginDatabase.DirectPExecute("UPDATE realmlist SET flag = flag | %u WHERE id = '%d'", REALM_FLAG_OFFLINE, realm.Id.Realm);
268
269 // Clean up threads if any
270 if (soapThread != nullptr)
271 {
272 soapThread->join();
273 delete soapThread;
274 }
275
276 delete raAcceptor;
277
278 ///- Clean database before leaving
279 ClearOnlineAccounts();
280
281 StopDB();
282
283 TC_LOG_INFO("server.worldserver", "Halting process...");
284
285 ShutdownCLIThread(cliThread);
286
287 OpenSSLCrypto::threadsCleanup();
288
289 // 0 - normal shutdown
290 // 1 - shutdown at error
291 // 2 - restart command used, this code can be used by restarter for restart Trinityd
292
293 return World::GetExitCode();
294}

296void ShutdownCLIThread(std::thread* cliThread)
297{
298 if (cliThread != nullptr)
299 {
300#ifdef _WIN32
301 // First try to cancel any I/O in the CLI thread
302 if (!CancelSynchronousIo(cliThread->native_handle()))
303 {
304 // if CancelSynchronousIo() fails, print the error and try with old way
305 DWORD errorCode = GetLastError();
306 LPSTR errorBuffer;
307
308 DWORD formatReturnCode = FormatMessage(FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_IGNORE_INSERTS,
309 nullptr, errorCode, 0, (LPTSTR)&errorBuffer, 0, nullptr);
310 if (!formatReturnCode)
311 errorBuffer = "Unknown error";
312
313 TC_LOG_DEBUG("server.worldserver", "Error cancelling I/O of CliThread, error code %u, detail: %s",
314 uint32(errorCode), errorBuffer);
315 LocalFree(errorBuffer);
316
317 // send keyboard input to safely unblock the CLI thread
318 INPUT_RECORD b[4];
319 HANDLE hStdIn = GetStdHandle(STD_INPUT_HANDLE);
320 b[0].EventType = KEY_EVENT;
321 b[0].Event.KeyEvent.bKeyDown = TRUE;
322 b[0].Event.KeyEvent.uChar.AsciiChar = 'X';
323 b[0].Event.KeyEvent.wVirtualKeyCode = 'X';
324 b[0].Event.KeyEvent.wRepeatCount = 1;
325
326 b[1].EventType = KEY_EVENT;
327 b[1].Event.KeyEvent.bKeyDown = FALSE;
328 b[1].Event.KeyEvent.uChar.AsciiChar = 'X';
329 b[1].Event.KeyEvent.wVirtualKeyCode = 'X';
330 b[1].Event.KeyEvent.wRepeatCount = 1;
331
332 b[2].EventType = KEY_EVENT;
333 b[2].Event.KeyEvent.bKeyDown = TRUE;
334 b[2].Event.KeyEvent.dwControlKeyState = 0;
335 b[2].Event.KeyEvent.uChar.AsciiChar = '\r';
336 b[2].Event.KeyEvent.wVirtualKeyCode = VK_RETURN;
337 b[2].Event.KeyEvent.wRepeatCount = 1;
338 b[2].Event.KeyEvent.wVirtualScanCode = 0x1c;
339
340 b[3].EventType = KEY_EVENT;
341 b[3].Event.KeyEvent.bKeyDown = FALSE;
342 b[3].Event.KeyEvent.dwControlKeyState = 0;
343 b[3].Event.KeyEvent.uChar.AsciiChar = '\r';
344 b[3].Event.KeyEvent.wVirtualKeyCode = VK_RETURN;
345 b[3].Event.KeyEvent.wVirtualScanCode = 0x1c;
346 b[3].Event.KeyEvent.wRepeatCount = 1;
347 DWORD numb;
348 WriteConsoleInput(hStdIn, b, 4, &numb);
349 }
350#endif
351 cliThread->join();
352 delete cliThread;
353 }

368void WorldUpdateLoop()
369{
370 uint32 realCurrTime = 0;
371 uint32 realPrevTime = getMSTime();
372
373 uint32 prevSleepTime = 0; // used for balanced full tick time length near WORLD_SLEEP_CONST
374
375 ///- While we have not World::m_stopEvent, update the world
376 while (!World::IsStopped())
377 {
378 ++World::m_worldLoopCounter;
379 realCurrTime = getMSTime();
380
381 uint32 diff = getMSTimeDiff(realPrevTime, realCurrTime);
382
383 sWorld->Update(diff);
384 realPrevTime = realCurrTime;
385
386 // diff (D0) include time of previous sleep (d0) + tick time (t0)
387 // we want that next d1 + t1 == WORLD_SLEEP_CONST
388 // we can't know next t1 and then can use (t0 + d1) == WORLD_SLEEP_CONST requirement
389 // d1 = WORLD_SLEEP_CONST - t0 = WORLD_SLEEP_CONST - (D0 - d0) = WORLD_SLEEP_CONST + d0 - D0
390 if (diff <= WORLD_SLEEP_CONST + prevSleepTime)
391 {
392 prevSleepTime = WORLD_SLEEP_CONST + prevSleepTime - diff;
393
394 std::this_thread::sleep_for(std::chrono::milliseconds(prevSleepTime));
395 }
396 else
397 prevSleepTime = 0;
398
399#ifdef _WIN32
400 if (m_ServiceStatus == 0)
401 World::StopNow(SHUTDOWN_EXIT_CODE);
402
403 while (m_ServiceStatus == 2)
404 Sleep(1000);
405#endif
406 }
407}

415void FreezeDetectorHandler(const boost::system::error_code& error)
416{
417 if (!error)
418 {
419 uint32 curtime = getMSTime();
420
421 uint32 worldLoopCounter = World::m_worldLoopCounter;
422 if (_worldLoopCounter != worldLoopCounter)
423 {
424 _lastChangeMsTime = curtime;
425 _worldLoopCounter = worldLoopCounter;
426 }
427 // possible freeze
428 else if (getMSTimeDiff(_lastChangeMsTime, curtime) > _maxCoreStuckTimeInMs)
429 {
430 TC_LOG_ERROR("server.worldserver", "World Thread hangs, kicking out server!");
431 ABORT();
432 }
433
434 _freezeCheckTimer.expires_from_now(boost::posix_time::seconds(1));
435 _freezeCheckTimer.async_wait(FreezeDetectorHandler);
436 }

449bool LoadRealmInfo()
450{
451 boost::asio::ip::tcp::resolver resolver(_ioService);
452 boost::asio::ip::tcp::resolver::iterator end;
453
454 QueryResult result = LoginDatabase.PQuery("SELECT id, name, address, localAddress, localSubnetMask, port, icon, flag, timezone, allowedSecurityLevel, population, gamebuild FROM realmlist WHERE id = %u", realm.Id.Realm);
455 if (!result)
456 return false;
457
458 Field* fields = result->Fetch();
459 realm.Name = fields[1].GetString();
460 boost::asio::ip::tcp::resolver::query externalAddressQuery(ip::tcp::v4(), fields[2].GetString(), "");
461
462 boost::system::error_code ec;
463 boost::asio::ip::tcp::resolver::iterator endPoint = resolver.resolve(externalAddressQuery, ec);
464 if (endPoint == end || ec)
465 {
466 TC_LOG_ERROR("server.worldserver", "Could not resolve address %s", fields[2].GetString().c_str());
467 return false;
468 }
469
470 realm.ExternalAddress = (*endPoint).endpoint().address();
471
472 boost::asio::ip::tcp::resolver::query localAddressQuery(ip::tcp::v4(), fields[3].GetString(), "");
473 endPoint = resolver.resolve(localAddressQuery, ec);
474 if (endPoint == end || ec)
475 {
476 TC_LOG_ERROR("server.worldserver", "Could not resolve address %s", fields[3].GetString().c_str());
477 return false;
478 }
479
480 realm.LocalAddress = (*endPoint).endpoint().address();
481
482 boost::asio::ip::tcp::resolver::query localSubmaskQuery(ip::tcp::v4(), fields[4].GetString(), "");
483 endPoint = resolver.resolve(localSubmaskQuery, ec);
484 if (endPoint == end || ec)
485 {
486 TC_LOG_ERROR("server.worldserver", "Could not resolve address %s", fields[4].GetString().c_str());
487 return false;
488 }
489
490 realm.LocalSubnetMask = (*endPoint).endpoint().address();
491
492 realm.Port = fields[5].GetUInt16();
493 realm.Type = fields[6].GetUInt8();
494 realm.Flags = RealmFlags(fields[7].GetUInt8());
495 realm.Timezone = fields[8].GetUInt8();
496 realm.AllowedSecurityLevel = AccountTypes(fields[9].GetUInt8());
497 realm.PopulationLevel = fields[10].GetFloat();
498 realm.Build = fields[11].GetUInt32();
499 return true;
500}

502/// Initialize connection to the databases
503bool StartDB()
504{
505 MySQL::Library_Init();
506
507 // Load databases
508 DatabaseLoader loader("server.worldserver", DatabaseLoader::DATABASE_NONE);
509 loader
510 .AddDatabase(LoginDatabase, "Login")
511 .AddDatabase(CharacterDatabase, "Character")
512 .AddDatabase(WorldDatabase, "World");
513
514 if (!loader.Load())
515 return false;
516
517 ///- Get the realm Id from the configuration file
518 realm.Id.Realm = sConfigMgr->GetIntDefault("RealmID", 0);
519 if (!realm.Id.Realm)
520 {
521 TC_LOG_ERROR("server.worldserver", "Realm ID not defined in configuration file");
522 return false;
523 }
524
525 TC_LOG_INFO("server.worldserver", "Realm running as realm ID %d", realm.Id.Realm);
526
527 ///- Clean the database before starting
528 ClearOnlineAccounts();
529
530 ///- Insert version info into DB
531 WorldDatabase.PExecute("UPDATE version SET core_version = '%s', core_revision = '%s'", GitRevision::GetFullVersion(), GitRevision::GetHash()); // One-time query
532
533 sWorld->LoadDBVersion();
534
535 TC_LOG_INFO("server.worldserver", "Using World DB: %s", sWorld->GetDBVersion());
536 return true;
537}

548/// Clear 'online' status for all accounts with characters in this realm
549void ClearOnlineAccounts()
550{
551 // Reset online status for all accounts with characters on the current realm
552 LoginDatabase.DirectPExecute("UPDATE account SET online = 0 WHERE online > 0 AND id IN (SELECT acctid FROM realmcharacters WHERE realmid = %d)", realm.Id.Realm);
553
554 // Reset online status for all characters
555 CharacterDatabase.DirectExecute("UPDATE characters SET online = 0 WHERE online <> 0");
556
557 // Battleground instance ids reset at server restart
558 CharacterDatabase.DirectExecute("UPDATE character_battleground_data SET instanceId = 0");
559}

563variables_map GetConsoleArguments(int argc, char** argv, std::string& configFile, std::string& configService)
564{
565 // Silences warning about configService not be used if the OS is not Windows
566 (void)configService;
567
568 options_description all("Allowed options");
569 all.add_options()
570 ("help,h", "print usage message")
571 ("version,v", "print version build info")
572 ("config,c", value<std::string>(&configFile)->default_value(_TRINITY_CORE_CONFIG), "use <arg> as configuration file")
573 ;
574#ifdef _WIN32
575 options_description win("Windows platform specific options");
576 win.add_options()
577 ("service,s", value<std::string>(&configService)->default_value(""), "Windows service options: [install | uninstall]")
578 ;
579
580 all.add(win);
581#endif
582 variables_map vm;
583 try
584 {
585 store(command_line_parser(argc, argv).options(all).allow_unregistered().run(), vm);
586 notify(vm);
587 }
588 catch (std::exception& e) {
589 std::cerr << e.what() << "\n";
590 }
591
592 if (vm.count("help")) {
593 std::cout << all << "\n";
594 }
595 else if (vm.count("version"))
596 {
597 std::cout << GitRevision::GetFullVersion() << "\n";
598 }
599
600 return vm;
601}

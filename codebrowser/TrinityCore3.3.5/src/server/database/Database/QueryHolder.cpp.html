23bool SQLQueryHolder::SetQuery(size_t index, const char *sql)
24{
25 if (m_queries.size() <= index)
26 {
27 TC_LOG_ERROR("sql.sql", "Query index (%u) out of range (size: %u) for query: %s", uint32(index), (uint32)m_queries.size(), sql);
28 return false;
29 }
30
31 /// not executed yet, just stored (it's not called a holder for nothing)
32 SQLElementData element;
33 element.type = SQL_ELEMENT_RAW;
34 element.element.query = strdup(sql);
35
36 SQLResultSetUnion result;
37 result.qresult = NULL;
38
39 m_queries[index] = SQLResultPair(element, result);
40 return true;
41}

43bool SQLQueryHolder::SetPreparedQuery(size_t index, PreparedStatement* stmt)
44{
45 if (m_queries.size() <= index)
46 {
47 TC_LOG_ERROR("sql.sql", "Query index (%u) out of range (size: %u) for prepared statement", uint32(index), (uint32)m_queries.size());
48 return false;
49 }
50
51 /// not executed yet, just stored (it's not called a holder for nothing)
52 SQLElementData element;
53 element.type = SQL_ELEMENT_PREPARED;
54 element.element.stmt = stmt;
55
56 SQLResultSetUnion result;
57 result.presult = NULL;
58
59 m_queries[index] = SQLResultPair(element, result);
60 return true;
61}

63QueryResult SQLQueryHolder::GetResult(size_t index)
64{
65 // Don't call to this function if the index is of an ad-hoc statement
66 if (index < m_queries.size())
67 {
68 ResultSet* result = m_queries[index].second.qresult;
69 if (!result || !result->GetRowCount() || !result->NextRow())
70 return QueryResult(NULL);
71
72 return QueryResult(result);
73 }
74 else
75 return QueryResult(NULL);
76}
77
78PreparedQueryResult SQLQueryHolder::GetPreparedResult(size_t index)
79{
80 // Don't call to this function if the index is of a prepared statement
81 if (index < m_queries.size())
82 {
83 PreparedResultSet* result = m_queries[index].second.presult;
84 if (!result || !result->GetRowCount())
85 return PreparedQueryResult(NULL);
86
87 return PreparedQueryResult(result);
88 }
89 else
90 return PreparedQueryResult(NULL);
91}

93void SQLQueryHolder::SetResult(size_t index, ResultSet* result)
94{
95 if (result && !result->GetRowCount())
96 {
97 delete result;
98 result = NULL;
99 }
100
101 /// store the result in the holder
102 if (index < m_queries.size())
103 m_queries[index].second.qresult = result;
104}

106void SQLQueryHolder::SetPreparedResult(size_t index, PreparedResultSet* result)
107{
108 if (result && !result->GetRowCount())
109 {
110 delete result;
111 result = NULL;
112 }
113
114 /// store the result in the holder
115 if (index < m_queries.size())
116 m_queries[index].second.presult = result;
117}

119SQLQueryHolder::~SQLQueryHolder()
120{
121 for (size_t i = 0; i < m_queries.size(); i++)
122 {
123 /// if the result was never used, free the resources
124 /// results used already (getresult called) are expected to be deleted
125 if (SQLElementData* data = &m_queries[i].first)
126 {
127 switch (data->type)
128 {
129 case SQL_ELEMENT_RAW:
130 free((void*)(const_cast<char*>(data->element.query)));
131 break;
132 case SQL_ELEMENT_PREPARED:
133 delete data->element.stmt;
134 break;
135 }
136 }
137 }
138}

140void SQLQueryHolder::SetSize(size_t size)
141{
142 /// to optimize push_back, reserve the number of queries about to be executed
143 m_queries.resize(size);

152bool SQLQueryHolderTask::Execute()
153{
154 m_executed = true;
155
156 if (!m_holder)
157 return false;
158
159 /// we can do this, we are friends
160 std::vector<SQLQueryHolder::SQLResultPair> &queries = m_holder->m_queries;
161
162 for (size_t i = 0; i < queries.size(); i++)
163 {
164 /// execute all queries in the holder and pass the results
165 if (SQLElementData* data = &queries[i].first)
166 {
167 switch (data->type)
168 {
169 case SQL_ELEMENT_RAW:
170 {
171 char const* sql = data->element.query;
172 if (sql)
173 m_holder->SetResult(i, m_conn->Query(sql));
174 break;
175 }
176 case SQL_ELEMENT_PREPARED:
177 {
178 PreparedStatement* stmt = data->element.stmt;
179 if (stmt)
180 m_holder->SetPreparedResult(i, m_conn->Query(stmt));
181 break;
182 }
183 }
184 }
185 }
186
187 m_result.set_value(m_holder);
188 return true;
189}

28void PreparedStatement::BindParameters()
29{
30 ASSERT (m_stmt);
31
32 uint8 i = 0;
33 for (; i < statement_data.size(); i++)
34 {
35 switch (statement_data[i].type)
36 {
37 case TYPE_BOOL:
38 m_stmt->setBool(i, statement_data[i].data.boolean);
39 break;
40 case TYPE_UI8:
41 m_stmt->setUInt8(i, statement_data[i].data.ui8);
42 break;
43 case TYPE_UI16:
44 m_stmt->setUInt16(i, statement_data[i].data.ui16);
45 break;
46 case TYPE_UI32:
47 m_stmt->setUInt32(i, statement_data[i].data.ui32);
48 break;
49 case TYPE_I8:
50 m_stmt->setInt8(i, statement_data[i].data.i8);
51 break;
52 case TYPE_I16:
53 m_stmt->setInt16(i, statement_data[i].data.i16);
54 break;
55 case TYPE_I32:
56 m_stmt->setInt32(i, statement_data[i].data.i32);
57 break;
58 case TYPE_UI64:
59 m_stmt->setUInt64(i, statement_data[i].data.ui64);
60 break;
61 case TYPE_I64:
62 m_stmt->setInt64(i, statement_data[i].data.i64);
63 break;
64 case TYPE_FLOAT:
65 m_stmt->setFloat(i, statement_data[i].data.f);
66 break;
67 case TYPE_DOUBLE:
68 m_stmt->setDouble(i, statement_data[i].data.d);
69 break;
70 case TYPE_STRING:
71 m_stmt->setString(i, statement_data[i].str.c_str());
72 break;
73 case TYPE_NULL:
74 m_stmt->setNull(i);
75 break;
76 }
77 }
78 #ifdef _DEBUG
79 if (i < m_stmt->m_paramCount)
80 TC_LOG_WARN("sql.sql", "[WARNING]: BindParameters() for statement %u did not bind all allocated parameters", m_index);
81 #endif
82}

201MySQLPreparedStatement::MySQLPreparedStatement(MYSQL_STMT* stmt) :
202m_stmt(NULL),
203m_Mstmt(stmt),
204m_bind(NULL)
205{
206 /// Initialize variable parameters
207 m_paramCount = mysql_stmt_param_count(stmt);
208 m_paramsSet.assign(m_paramCount, false);
209 m_bind = new MYSQL_BIND[m_paramCount];
210 memset(m_bind, 0, sizeof(MYSQL_BIND)*m_paramCount);
211
212 /// "If set to 1, causes mysql_stmt_store_result() to update the metadata MYSQL_FIELD->max_length value."
213 my_bool bool_tmp = 1;
214 mysql_stmt_attr_set(stmt, STMT_ATTR_UPDATE_MAX_LENGTH, &bool_tmp);
215}
216
217MySQLPreparedStatement::~MySQLPreparedStatement()
218{
219 ClearParameters();
220 if (m_Mstmt->bind_result_done)
221 {
222 delete[] m_Mstmt->bind->length;
223 delete[] m_Mstmt->bind->is_null;
224 }
225 mysql_stmt_close(m_Mstmt);
226 delete[] m_bind;
227}
228
229void MySQLPreparedStatement::ClearParameters()
230{
231 for (uint32 i=0; i < m_paramCount; ++i)
232 {
233 delete m_bind[i].length;
234 m_bind[i].length = NULL;
235 delete[] (char*) m_bind[i].buffer;
236 m_bind[i].buffer = NULL;
237 m_paramsSet[i] = false;
238 }
239}

241static bool ParamenterIndexAssertFail(uint32 stmtIndex, uint8 index, uint32 paramCount)
242{
243 TC_LOG_ERROR("sql.driver", "Attempted to bind parameter %u%s on a PreparedStatement %u (statement has only %u parameters)", uint32(index) + 1, (index == 1 ? "st" : (index == 2 ? "nd" : (index == 3 ? "rd" : "nd"))), stmtIndex, paramCount);
244 return false;
245}

247//- Bind on mysql level
248bool MySQLPreparedStatement::CheckValidIndex(uint8 index)
249{
250 ASSERT(index < m_paramCount || ParamenterIndexAssertFail(m_stmt->m_index, index, m_paramCount));
251
252 if (m_paramsSet[index])
253 TC_LOG_WARN("sql.sql", "[WARNING] Prepared Statement (id: %u) trying to bind value on already bound index (%u).", m_stmt->m_index, index);
254 return true;
255}

342void MySQLPreparedStatement::setString(const uint8 index, const char* value)
343{
344 CheckValidIndex(index);
345 m_paramsSet[index] = true;
346 MYSQL_BIND* param = &m_bind[index];
347 size_t len = strlen(value) + 1;
348 param->buffer_type = MYSQL_TYPE_VAR_STRING;
349 delete [] static_cast<char *>(param->buffer);
350 param->buffer = new char[len];
351 param->buffer_length = len;
352 param->is_null_value = 0;
353 delete param->length;
354 param->length = new unsigned long(len-1);
355
356 memcpy(param->buffer, value, len);
357}

359void MySQLPreparedStatement::setNull(const uint8 index)
360{

362 m_paramsSet[index] = true;
363 MYSQL_BIND* param = &m_bind[index];
364 param->buffer_type = MYSQL_TYPE_NULL;
365 delete [] static_cast<char *>(param->buffer);
366 param->buffer = NULL;

368 param->is_null_value = 1;
369 delete param->length;
370 param->length = NULL;
371}

373void MySQLPreparedStatement::setValue(MYSQL_BIND* param, enum_field_types type, const void* value, uint32 len, bool isUnsigned)
374{
375 param->buffer_type = type;
376 delete [] static_cast<char *>(param->buffer);
377 param->buffer = new char[len];
378 param->buffer_length = 0;
379 param->is_null_value = 0;
380 param->length = NULL; // Only != NULL for strings
381 param->is_unsigned = isUnsigned;
382
383 memcpy(param->buffer, value, len);
384}

386std::string MySQLPreparedStatement::getQueryString(std::string const& sqlPattern) const
387{
388 std::string queryString = sqlPattern;
389
390 size_t pos = 0;
391 for (uint32 i = 0; i < m_stmt->statement_data.size(); i++)
392 {
393 pos = queryString.find('?', pos);
394 std::stringstream ss;
395
396 switch (m_stmt->statement_data[i].type)
397 {
398 case TYPE_BOOL:
399 ss << uint16(m_stmt->statement_data[i].data.boolean);
400 break;
401 case TYPE_UI8:
402 ss << uint16(m_stmt->statement_data[i].data.ui8); // stringstream will append a character with that code instead of numeric representation
403 break;
404 case TYPE_UI16:
405 ss << m_stmt->statement_data[i].data.ui16;
406 break;
407 case TYPE_UI32:
408 ss << m_stmt->statement_data[i].data.ui32;
409 break;
410 case TYPE_I8:
411 ss << int16(m_stmt->statement_data[i].data.i8); // stringstream will append a character with that code instead of numeric representation
412 break;
413 case TYPE_I16:
414 ss << m_stmt->statement_data[i].data.i16;
415 break;
416 case TYPE_I32:
417 ss << m_stmt->statement_data[i].data.i32;
418 break;
419 case TYPE_UI64:
420 ss << m_stmt->statement_data[i].data.ui64;
421 break;
422 case TYPE_I64:
423 ss << m_stmt->statement_data[i].data.i64;
424 break;
425 case TYPE_FLOAT:
426 ss << m_stmt->statement_data[i].data.f;
427 break;
428 case TYPE_DOUBLE:
429 ss << m_stmt->statement_data[i].data.d;
430 break;
431 case TYPE_STRING:
432 ss << '\'' << m_stmt->statement_data[i].str << '\'';
433 break;
434 case TYPE_NULL:
435 ss << "NULL";
436 break;
437 }
438
439 std::string replaceStr = ss.str();
440 queryString.replace(pos, 1, replaceStr);
441 pos += replaceStr.length();
442 }
443
444 return queryString;
445}

448PreparedStatementTask::PreparedStatementTask(PreparedStatement* stmt, bool async) :
449m_stmt(stmt), m_result(nullptr)
450{
451 m_has_result = async; // If it's async, then there's a result
452 if (async)
453 m_result = new PreparedQueryResultPromise();
454}
455
456PreparedStatementTask::~PreparedStatementTask()
457{
458 delete m_stmt;
459 if (m_has_result && m_result != nullptr)
460 delete m_result;
461}
462
463bool PreparedStatementTask::Execute()
464{
465 if (m_has_result)
466 {
467 PreparedResultSet* result = m_conn->Query(m_stmt);
468 if (!result || !result->GetRowCount())
469 {
470 delete result;
471 m_result->set_value(PreparedQueryResult(NULL));
472 return false;
473 }
474 m_result->set_value(PreparedQueryResult(result));
475 return true;
476 }
477
478 return m_conn->Execute(m_stmt);
479}

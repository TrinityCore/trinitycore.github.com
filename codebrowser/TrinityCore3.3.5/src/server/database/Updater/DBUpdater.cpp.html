39std::string DBUpdaterUtil::GetCorrectedMySQLExecutable()
40{
41 if (!corrected_path().empty())
42 return corrected_path();
43 else
44 return BuiltInConfig::GetMySQLExecutable();
45}
46
47bool DBUpdaterUtil::CheckExecutable()
48{
49 boost::filesystem::path exe(GetCorrectedMySQLExecutable());
50 if (!exists(exe))
51 {
52 exe.clear();
53
54 try
55 {
56 exe = search_path("mysql");
57 }
58 catch (std::runtime_error&)
59 {
60 }
61
62 if (!exe.empty() && exists(exe))
63 {
64 // Correct the path to the cli
65 corrected_path() = absolute(exe).generic_string();
66 return true;
67 }
68
69 TC_LOG_FATAL("sql.updates", "Didn't find executeable mysql binary at \'%s\' or in path, correct the path in the *.conf (\"Updates.MySqlCLIPath\").",
70 absolute(exe).generic_string().c_str());
71
72 return false;
73 }
74 return true;
75}

79 static std::string path;

176template<class T>
177bool DBUpdater<T>::Create(DatabaseWorkerPool<T>& pool)
178{
179 TC_LOG_INFO("sql.updates", "Database \"%s\" does not exist, do you want to create it? [yes (default) / no]: ",
180 pool.GetConnectionInfo()->database.c_str());
181
182 std::string answer;
183 std::getline(std::cin, answer);
184 if (!answer.empty() && !(answer.substr(0, 1) == "y"))
185 return false;
186
187 TC_LOG_INFO("sql.updates", "Creating database \"%s\"...", pool.GetConnectionInfo()->database.c_str());
188
189 // Path of temp file
190 static Path const temp("create_table.sql");
191
192 // Create temporary query to use external mysql cli
193 std::ofstream file(temp.generic_string());
194 if (!file.is_open())
195 {
196 TC_LOG_FATAL("sql.updates", "Failed to create temporary query file \"%s\"!", temp.generic_string().c_str());
197 return false;
198 }
199
200 file << "CREATE DATABASE `" << pool.GetConnectionInfo()->database << "` DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci\n\n";
201
202 file.close();
203
204 try
205 {
206 DBUpdater<T>::ApplyFile(pool, pool.GetConnectionInfo()->host, pool.GetConnectionInfo()->user, pool.GetConnectionInfo()->password,
207 pool.GetConnectionInfo()->port_or_socket, "", temp);
208 }
209 catch (UpdateException&)
210 {
211 TC_LOG_FATAL("sql.updates", "Failed to create database %s! Has the user `CREATE` priviliges?", pool.GetConnectionInfo()->database.c_str());
212 boost::filesystem::remove(temp);
213 return false;
214 }
215
216 TC_LOG_INFO("sql.updates", "Done.");
217 boost::filesystem::remove(temp);
218 return true;
219}

221template<class T>
222bool DBUpdater<T>::Update(DatabaseWorkerPool<T>& pool)
223{
224 if (!DBUpdaterUtil::CheckExecutable())
225 return false;
226
227 TC_LOG_INFO("sql.updates", "Updating %s database...", DBUpdater<T>::GetTableName().c_str());
228
229 Path const sourceDirectory(BuiltInConfig::GetSourceDirectory());
230
231 if (!is_directory(sourceDirectory))
232 {
233 TC_LOG_ERROR("sql.updates", "DBUpdater: Given source directory %s does not exist, skipped!", sourceDirectory.generic_string().c_str());
234 return false;
235 }
236
237 UpdateFetcher updateFetcher(sourceDirectory, [&](std::string const& query) { DBUpdater<T>::Apply(pool, query); },
238 [&](Path const& file) { DBUpdater<T>::ApplyFile(pool, file); },
239 [&](std::string const& query) -> QueryResult { return DBUpdater<T>::Retrieve(pool, query); });
240
241 UpdateResult result;
242 try
243 {
244 result = updateFetcher.Update(
245 sConfigMgr->GetBoolDefault("Updates.Redundancy", true),
246 sConfigMgr->GetBoolDefault("Updates.AllowRehash", true),
247 sConfigMgr->GetBoolDefault("Updates.ArchivedRedundancy", false),
248 sConfigMgr->GetIntDefault("Updates.CleanDeadRefMaxCount", 3));
249 }
250 catch (UpdateException&)
251 {
252 return false;
253 }
254
255 std::string const info = Trinity::StringFormat("Containing " SZFMTD " new and " SZFMTD " archived updates.",
256 result.recent, result.archived);
257
258 if (!result.updated)
259 TC_LOG_INFO("sql.updates", ">> %s database is up-to-date! %s", DBUpdater<T>::GetTableName().c_str(), info.c_str());
260 else
261 TC_LOG_INFO("sql.updates", ">> Applied " SZFMTD " %s. %s", result.updated, result.updated == 1 ? "query" : "queries", info.c_str());
262
263 return true;
264}

266template<class T>
267bool DBUpdater<T>::Populate(DatabaseWorkerPool<T>& pool)
268{
269 {
270 QueryResult const result = Retrieve(pool, "SHOW TABLES");
271 if (result && (result->GetRowCount() > 0))
272 return true;
273 }
274
275 if (!DBUpdaterUtil::CheckExecutable())
276 return false;
277
278 TC_LOG_INFO("sql.updates", "Database %s is empty, auto populating it...", DBUpdater<T>::GetTableName().c_str());
279
280 std::string const p = DBUpdater<T>::GetBaseFile();
281 if (p.empty())
282 {
283 TC_LOG_INFO("sql.updates", ">> No base file provided, skipped!");
284 return true;
285 }
286
287 Path const base(p);
288 if (!exists(base))
289 {
290 switch (DBUpdater<T>::GetBaseLocationType())
291 {
292 case LOCATION_REPOSITORY:
293 {
294 TC_LOG_ERROR("sql.updates", ">> Base file \"%s\" is missing, try to clone the source again.",
295 base.generic_string().c_str());
296
297 break;
298 }
299 case LOCATION_DOWNLOAD:
300 {
301 TC_LOG_ERROR("sql.updates", ">> File \"%s\" is missing, download it from \"https://github.com/TrinityCore/TrinityCore/releases\"" \
302 " and place it in your server directory.", base.filename().generic_string().c_str());
303 break;
304 }
305 }
306 return false;
307 }
308
309 // Update database
310 TC_LOG_INFO("sql.updates", ">> Applying \'%s\'...", base.generic_string().c_str());
311 try
312 {
313 ApplyFile(pool, base);
314 }
315 catch (UpdateException&)
316 {
317 return false;
318 }
319
320 TC_LOG_INFO("sql.updates", ">> Done!");
321 return true;
322}

343template<class T>
344void DBUpdater<T>::ApplyFile(DatabaseWorkerPool<T>& pool, std::string const& host, std::string const& user,
345 std::string const& password, std::string const& port_or_socket, std::string const& database, Path const& path)
346{
347 std::vector<std::string> args;
348 args.reserve(8);
349
350 // args[0] represents the program name
351 args.push_back("mysql");
352
353 // CLI Client connection info
354 args.push_back("-h" + host);
355 args.push_back("-u" + user);
356
357 if (!password.empty())
358 args.push_back("-p" + password);
359
360 // Check if we want to connect through ip or socket (Unix only)
361#ifdef _WIN32
362
363 args.push_back("-P" + port_or_socket);
364
365#else
366
367 if (!std::isdigit(port_or_socket[0]))
368 {
369 // We can't check here if host == "." because is named localhost if socket option is enabled
370 args.push_back("-P0");
371 args.push_back("--protocol=SOCKET");
372 args.push_back("-S" + port_or_socket);
373 }
374 else
375 // generic case
376 args.push_back("-P" + port_or_socket);
377
378#endif
379
380 // Set the default charset to utf8
381 args.push_back("--default-character-set=utf8");
382
383 // Set max allowed packet to 1 GB
384 args.push_back("--max-allowed-packet=1GB");
385
386 // Database
387 if (!database.empty())
388 args.push_back(database);
389
390 // ToDo: use the existing query in memory as virtual file if possible
391 file_descriptor_source source(path);
392
393 uint32 ret;
394 try
395 {
396 boost::process::pipe outPipe = create_pipe();
397 boost::process::pipe errPipe = create_pipe();
398
399 child c = execute(run_exe(
400 boost::filesystem::absolute(DBUpdaterUtil::GetCorrectedMySQLExecutable()).generic_string()),
401 set_args(args), bind_stdin(source), throw_on_error(),
402 bind_stdout(file_descriptor_sink(outPipe.sink, close_handle)),
403 bind_stderr(file_descriptor_sink(errPipe.sink, close_handle)));
404
405 file_descriptor_source mysqlOutfd(outPipe.source, close_handle);
406 file_descriptor_source mysqlErrfd(errPipe.source, close_handle);
407
408 stream<file_descriptor_source> mysqlOutStream(mysqlOutfd);
409 stream<file_descriptor_source> mysqlErrStream(mysqlErrfd);
410
411 std::stringstream out;
412 std::stringstream err;
413
414 copy(mysqlOutStream, out);
415 copy(mysqlErrStream, err);
416
417 TC_LOG_INFO("sql.updates", "%s", out.str().c_str());
418 TC_LOG_ERROR("sql.updates", "%s", err.str().c_str());
419
420 ret = wait_for_exit(c);
421 }
422 catch (boost::system::system_error&)
423 {
424 ret = EXIT_FAILURE;
425 }
426
427 source.close();
428
429 if (ret != EXIT_SUCCESS)
430 {
431 TC_LOG_FATAL("sql.updates", "Applying of file \'%s\' to database \'%s\' failed!" \
432 " If you are an user pull the latest revision from the repository. If you are a developer fix your sql query.",
433 path.generic_string().c_str(), pool.GetConnectionInfo()->database.c_str());
434
435 throw UpdateException("update failed");
436 }
437}

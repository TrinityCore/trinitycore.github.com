41bool FileLoader::loadFile(std::string const& fileName, bool log)
42{
43 free();
44 MPQFile mf(fileName.c_str());
45 if(mf.isEof())
46 {
47 if (log)
48 printf("No such file %s\n", fileName.c_str());
49 return false;
50 }
51
52 data_size = mf.getSize();
53
54 data = new uint8 [data_size];
55 mf.read(data, data_size);
56 mf.close();
57 if (prepareLoadedData())
58 return true;
59
60 printf("Error loading %s", fileName.c_str());
61 mf.close();
62 free();
63 return false;
64}
65
66bool FileLoader::prepareLoadedData()
67{
68 // Check version
69 version = (file_MVER *) data;
70 if (version->fcc != MverMagic.fcc)
71 return false;
72 if (version->ver != FILE_FORMAT_VERSION)
73 return false;
74 return true;
75}
76
77void FileLoader::free()
78{
79 if (data) delete[] data;
80 data = 0;
81 data_size = 0;
82 version = 0;
83}

35bool isHole(int holes, int i, int j)
36{
37 int testi = i / 2;
38 int testj = j / 4;
39 if(testi > 3) testi = 3;
40 if(testj > 3) testj = 3;
41 return (holes & holetab_h[testi] & holetab_v[testj]) != 0;
42}

66bool ADT_file::prepareLoadedData()
67{
68 // Check parent
69 if (!FileLoader::prepareLoadedData())
70 return false;
71
72 // Check and prepare MHDR
73 a_grid = (adt_MHDR *)(GetData()+8+version->size);
74 if (!a_grid->prepareLoadedData())
75 return false;
76
77 return true;
78}

80bool adt_MHDR::prepareLoadedData()
81{
82 if (fcc != MHDRMagic.fcc)
83 return false;
84
85 if (size != sizeof(adt_MHDR) - 8)
86 return false;
87
88 // Check and prepare MCIN
89 if (offsMCIN && !getMCIN()->prepareLoadedData())
90 return false;
91
92 // Check and prepare MH2O
93 if (offsMH2O && !getMH2O()->prepareLoadedData())
94 return false;
95
96 if (offsMFBO && flags & 1 && !getMFBO()->prepareLoadedData())
97 return false;
98
99 return true;
100}

102bool adt_MCIN::prepareLoadedData()
103{
104 if (fcc != MCINMagic.fcc)
105 return false;
106
107 // Check cells data
108 for (int i=0; i<ADT_CELLS_PER_GRID;i++)
109 for (int j=0; j<ADT_CELLS_PER_GRID;j++)
110 if (cells[i][j].offsMCNK && !getMCNK(i,j)->prepareLoadedData())
111 return false;
112
113 return true;
114}

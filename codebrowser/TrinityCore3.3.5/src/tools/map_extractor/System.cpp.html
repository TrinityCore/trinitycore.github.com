99void Usage(char* prg)
100{
101 printf(
102 "Usage:\n"\
103 "%s -[var] [value]\n"\
104 "-i set input path (max %d characters)\n"\
105 "-o set output path (max %d characters)\n"\
106 "-e extract only MAP(1)/DBC(2) - standard: both(3)\n"\
107 "-f height stored as int (less map size but lost some accuracy) 1 by default\n"\
108 "Example: %s -f 0 -i \"c:\\games\\game\"", prg, MAX_PATH_LENGTH - 1, MAX_PATH_LENGTH - 1, prg);
109 exit(1);
110}

112void HandleArgs(int argc, char * arg[])
113{
114 for(int c = 1; c < argc; ++c)
115 {
116 // i - input path
117 // o - output path
118 // e - extract only MAP(1)/DBC(2) - standard both(3)
119 // f - use float to int conversion
120 // h - limit minimum height
121 if(arg[c][0] != '-')
122 Usage(arg[0]);
123
124 switch(arg[c][1])
125 {
126 case 'i':
127 if (c + 1 < argc && strlen(arg[c + 1]) < MAX_PATH_LENGTH) // all ok
128 {
129 strncpy(input_path, arg[c++ + 1], MAX_PATH_LENGTH);
130 input_path[MAX_PATH_LENGTH - 1] = '\0';
131 }
132 else
133 Usage(arg[0]);
134 break;
135 case 'o':
136 if (c + 1 < argc && strlen(arg[c + 1]) < MAX_PATH_LENGTH) // all ok
137 {
138 strncpy(output_path, arg[c++ + 1], MAX_PATH_LENGTH);
139 output_path[MAX_PATH_LENGTH - 1] = '\0';
140 }
141 else
142 Usage(arg[0]);
143 break;
144 case 'f':
145 if(c + 1 < argc) // all ok
146 CONF_allow_float_to_int=atoi(arg[(c++) + 1])!=0;
147 else
148 Usage(arg[0]);
149 break;
150 case 'e':
151 if(c + 1 < argc) // all ok
152 {
153 CONF_extract=atoi(arg[(c++) + 1]);
154 if(!(CONF_extract > 0 && CONF_extract < 4))
155 Usage(arg[0]);
156 }
157 else
158 Usage(arg[0]);
159 break;
160 }
161 }
162}

164uint32 ReadBuild(int locale)
165{
166 // include build info file also
167 std::string filename = Trinity::StringFormat("component.wow-%s.txt", langs[locale]);
168 //printf("Read %s file... ", filename.c_str());
169
170 MPQFile m(filename.c_str());
171 if(m.isEof())
172 {
173 printf("Fatal error: Not found %s file!\n", filename.c_str());
174 exit(1);
175 }
176
177 std::string text = std::string(m.getPointer(), m.getSize());
178 m.close();
179
180 size_t pos = text.find("version=\"");
181 size_t pos1 = pos + strlen("version=\"");
182 size_t pos2 = text.find("\"",pos1);
183 if (pos == text.npos || pos2 == text.npos || pos1 >= pos2)
184 {
185 printf("Fatal error: Invalid %s file format!\n", filename.c_str());
186 exit(1);
187 }
188
189 std::string build_str = text.substr(pos1,pos2-pos1);
190
191 int build = atoi(build_str.c_str());
192 if (build <= 0)
193 {
194 printf("Fatal error: Invalid %s file format!\n", filename.c_str());
195 exit(1);
196 }
197
198 return build;
199}

201uint32 ReadMapDBC()
202{
203 printf("Read Map.dbc file... ");
204 DBCFile dbc("DBFilesClient\\Map.dbc");
205
206 if(!dbc.open())
207 {
208 printf("Fatal error: Invalid Map.dbc file format!\n");
209 exit(1);
210 }
211
212 size_t map_count = dbc.getRecordCount();
213 map_ids = new map_id[map_count];
214 for(uint32 x = 0; x < map_count; ++x)
215 {
216 map_ids[x].id = dbc.getRecord(x).getUInt(0);
217
218 const char* map_name = dbc.getRecord(x).getString(1);
219 size_t max_map_name_length = sizeof(map_ids[x].name);
220 if (strlen(map_name) >= max_map_name_length)
221 {
222 printf("Fatal error: Map name too long!\n");
223 exit(1);
224 }
225
226 strncpy(map_ids[x].name, map_name, max_map_name_length);
227 map_ids[x].name[max_map_name_length - 1] = '\0';
228 }
229 printf("Done! (" SZFMTD "maps loaded)\n", map_count);
230 return map_count;
231}

233void ReadLiquidTypeTableDBC()
234{
235 printf("Read LiquidType.dbc file...");
236 DBCFile dbc("DBFilesClient\\LiquidType.dbc");
237 if(!dbc.open())
238 {
239 printf("Fatal error: Invalid LiquidType.dbc file format!\n");
240 exit(1);
241 }
242
243 size_t liqTypeCount = dbc.getRecordCount();
244 size_t liqTypeMaxId = dbc.getMaxId();
245 LiqType = new uint16[liqTypeMaxId + 1];
246 memset(LiqType, 0xff, (liqTypeMaxId + 1) * sizeof(uint16));
247
248 for(uint32 x = 0; x < liqTypeCount; ++x)
249 LiqType[dbc.getRecord(x).getUInt(0)] = dbc.getRecord(x).getUInt(3);
250
251 printf("Done! (" SZFMTD " LiqTypes loaded)\n", liqTypeCount);
252}

354bool ConvertADT(std::string const& inputPath, std::string const& outputPath, int /*cell_y*/, int /*cell_x*/, uint32 build)
355{
356 ADT_file adt;
357
358 if (!adt.loadFile(inputPath))
359 return false;
360
361 adt_MCIN *cells = adt.a_grid->getMCIN();
362 if (!cells)
363 {
364 printf("Can't find cells in '%s'\n", inputPath.c_str());
365 return false;
366 }
367
368 memset(liquid_show, 0, sizeof(liquid_show));
369 memset(liquid_flags, 0, sizeof(liquid_flags));
370 memset(liquid_entry, 0, sizeof(liquid_entry));
371
372 // Prepare map header
373 map_fileheader map;
374 map.mapMagic = *reinterpret_cast<uint32 const*>(MAP_MAGIC);
375 map.versionMagic = *reinterpret_cast<uint32 const*>(MAP_VERSION_MAGIC);
376 map.buildMagic = build;
377
378 // Get area flags data
379 for (int i = 0; i < ADT_CELLS_PER_GRID; i++)
380 for (int j = 0; j < ADT_CELLS_PER_GRID; j++)
381 area_ids[i][j] = cells->getMCNK(i, j)->areaid;
382
383 //============================================
384 // Try pack area data
385 //============================================
386 bool fullAreaData = false;
387 uint32 areaId = area_ids[0][0];
388 for (int y = 0; y < ADT_CELLS_PER_GRID; ++y)
389 {
390 for (int x = 0; x < ADT_CELLS_PER_GRID; ++x)
391 {
392 if (area_ids[y][x] != areaId)
393 {
394 fullAreaData = true;
395 break;
396 }
397 }
398 }
399
400 map.areaMapOffset = sizeof(map);
401 map.areaMapSize = sizeof(map_areaHeader);
402
403 map_areaHeader areaHeader;
404 areaHeader.fourcc = *reinterpret_cast<uint32 const*>(MAP_AREA_MAGIC);
405 areaHeader.flags = 0;
406 if (fullAreaData)
407 {
408 areaHeader.gridArea = 0;
409 map.areaMapSize += sizeof(area_ids);
410 }
411 else
412 {
413 areaHeader.flags |= MAP_AREA_NO_AREA;
414 areaHeader.gridArea = static_cast<uint16>(areaId);
415 }
416
417 //
418 // Get Height map from grid
419 //
420 for (int i=0;i<ADT_CELLS_PER_GRID;i++)
421 {
422 for(int j=0;j<ADT_CELLS_PER_GRID;j++)
423 {
424 adt_MCNK * cell = cells->getMCNK(i,j);
425 if (!cell)
426 continue;
427 // Height values for triangles stored in order:
428 // 1 2 3 4 5 6 7 8 9
429 // 10 11 12 13 14 15 16 17
430 // 18 19 20 21 22 23 24 25 26
431 // 27 28 29 30 31 32 33 34
432 // . . . . . . . .
433 // For better get height values merge it to V9 and V8 map
434 // V9 height map:
435 // 1 2 3 4 5 6 7 8 9
436 // 18 19 20 21 22 23 24 25 26
437 // . . . . . . . .
438 // V8 height map:
439 // 10 11 12 13 14 15 16 17
440 // 27 28 29 30 31 32 33 34
441 // . . . . . . . .
442
443 // Set map height as grid height
444 for (int y=0; y <= ADT_CELL_SIZE; y++)
445 {
446 int cy = i*ADT_CELL_SIZE + y;
447 for (int x=0; x <= ADT_CELL_SIZE; x++)
448 {
449 int cx = j*ADT_CELL_SIZE + x;
450 V9[cy][cx]=cell->ypos;
451 }
452 }
453 for (int y=0; y < ADT_CELL_SIZE; y++)
454 {
455 int cy = i*ADT_CELL_SIZE + y;
456 for (int x=0; x < ADT_CELL_SIZE; x++)
457 {
458 int cx = j*ADT_CELL_SIZE + x;
459 V8[cy][cx]=cell->ypos;
460 }
461 }
462 // Get custom height
463 adt_MCVT *v = cell->getMCVT();
464 if (!v)
465 continue;
466 // get V9 height map
467 for (int y=0; y <= ADT_CELL_SIZE; y++)
468 {
469 int cy = i*ADT_CELL_SIZE + y;
470 for (int x=0; x <= ADT_CELL_SIZE; x++)
471 {
472 int cx = j*ADT_CELL_SIZE + x;
473 V9[cy][cx]+=v->height_map[y*(ADT_CELL_SIZE*2+1)+x];
474 }
475 }
476 // get V8 height map
477 for (int y=0; y < ADT_CELL_SIZE; y++)
478 {
479 int cy = i*ADT_CELL_SIZE + y;
480 for (int x=0; x < ADT_CELL_SIZE; x++)
481 {
482 int cx = j*ADT_CELL_SIZE + x;
483 V8[cy][cx]+=v->height_map[y*(ADT_CELL_SIZE*2+1)+ADT_CELL_SIZE+1+x];
484 }
485 }
486 }
487 }
488 //============================================
489 // Try pack height data
490 //============================================
491 float maxHeight = -20000;
492 float minHeight = 20000;
493 for (int y=0; y<ADT_GRID_SIZE; y++)
494 {
495 for(int x=0;x<ADT_GRID_SIZE;x++)
496 {
497 float h = V8[y][x];
498 if (maxHeight < h) maxHeight = h;
499 if (minHeight > h) minHeight = h;
500 }
501 }
502 for (int y=0; y<=ADT_GRID_SIZE; y++)
503 {
504 for(int x=0;x<=ADT_GRID_SIZE;x++)
505 {
506 float h = V9[y][x];
507 if (maxHeight < h) maxHeight = h;
508 if (minHeight > h) minHeight = h;
509 }
510 }
511
512 // Check for allow limit minimum height (not store height in deep ochean - allow save some memory)
513 if (CONF_allow_height_limit && minHeight < CONF_use_minHeight)
514 {
515 for (int y=0; y<ADT_GRID_SIZE; y++)
516 for(int x=0;x<ADT_GRID_SIZE;x++)
517 if (V8[y][x] < CONF_use_minHeight)
518 V8[y][x] = CONF_use_minHeight;
519 for (int y=0; y<=ADT_GRID_SIZE; y++)
520 for(int x=0;x<=ADT_GRID_SIZE;x++)
521 if (V9[y][x] < CONF_use_minHeight)
522 V9[y][x] = CONF_use_minHeight;
523 if (minHeight < CONF_use_minHeight)
524 minHeight = CONF_use_minHeight;
525 if (maxHeight < CONF_use_minHeight)
526 maxHeight = CONF_use_minHeight;
527 }
528
529 bool hasFlightBox = false;
530 if (adt_MFBO* mfbo = adt.a_grid->getMFBO())
531 {
532 memcpy(flight_box_max, &mfbo->max, sizeof(flight_box_max));
533 memcpy(flight_box_min, &mfbo->min, sizeof(flight_box_min));
534 hasFlightBox = true;
535 }
536
537 map.heightMapOffset = map.areaMapOffset + map.areaMapSize;
538 map.heightMapSize = sizeof(map_heightHeader);
539
540 map_heightHeader heightHeader;
541 heightHeader.fourcc = *reinterpret_cast<uint32 const*>(MAP_HEIGHT_MAGIC);
542 heightHeader.flags = 0;
543 heightHeader.gridHeight = minHeight;
544 heightHeader.gridMaxHeight = maxHeight;
545
546 if (maxHeight == minHeight)
547 heightHeader.flags |= MAP_HEIGHT_NO_HEIGHT;
548
549 // Not need store if flat surface
550 if (CONF_allow_float_to_int && (maxHeight - minHeight) < CONF_flat_height_delta_limit)
551 heightHeader.flags |= MAP_HEIGHT_NO_HEIGHT;
552
553 if (hasFlightBox)
554 {
555 heightHeader.flags |= MAP_HEIGHT_HAS_FLIGHT_BOUNDS;
556 map.heightMapSize += sizeof(flight_box_max) + sizeof(flight_box_min);
557 }
558
559 // Try store as packed in uint16 or uint8 values
560 if (!(heightHeader.flags & MAP_HEIGHT_NO_HEIGHT))
561 {
562 float step = 0;
563 // Try Store as uint values
564 if (CONF_allow_float_to_int)
565 {
566 float diff = maxHeight - minHeight;
567 if (diff < CONF_float_to_int8_limit) // As uint8 (max accuracy = CONF_float_to_int8_limit/256)
568 {
569 heightHeader.flags|=MAP_HEIGHT_AS_INT8;
570 step = selectUInt8StepStore(diff);
571 }
572 else if (diff<CONF_float_to_int16_limit) // As uint16 (max accuracy = CONF_float_to_int16_limit/65536)
573 {
574 heightHeader.flags|=MAP_HEIGHT_AS_INT16;
575 step = selectUInt16StepStore(diff);
576 }
577 }
578
579 // Pack it to int values if need
580 if (heightHeader.flags&MAP_HEIGHT_AS_INT8)
581 {
582 for (int y=0; y<ADT_GRID_SIZE; y++)
583 for(int x=0;x<ADT_GRID_SIZE;x++)
584 uint8_V8[y][x] = uint8((V8[y][x] - minHeight) * step + 0.5f);
585 for (int y=0; y<=ADT_GRID_SIZE; y++)
586 for(int x=0;x<=ADT_GRID_SIZE;x++)
587 uint8_V9[y][x] = uint8((V9[y][x] - minHeight) * step + 0.5f);
588 map.heightMapSize+= sizeof(uint8_V9) + sizeof(uint8_V8);
589 }
590 else if (heightHeader.flags&MAP_HEIGHT_AS_INT16)
591 {
592 for (int y=0; y<ADT_GRID_SIZE; y++)
593 for(int x=0;x<ADT_GRID_SIZE;x++)
594 uint16_V8[y][x] = uint16((V8[y][x] - minHeight) * step + 0.5f);
595 for (int y=0; y<=ADT_GRID_SIZE; y++)
596 for(int x=0;x<=ADT_GRID_SIZE;x++)
597 uint16_V9[y][x] = uint16((V9[y][x] - minHeight) * step + 0.5f);
598 map.heightMapSize+= sizeof(uint16_V9) + sizeof(uint16_V8);
599 }
600 else
601 map.heightMapSize+= sizeof(V9) + sizeof(V8);
602 }
603
604 // Get from MCLQ chunk (old)
605 for (int i = 0; i < ADT_CELLS_PER_GRID; i++)
606 {
607 for(int j = 0; j < ADT_CELLS_PER_GRID; j++)
608 {
609 adt_MCNK *cell = cells->getMCNK(i, j);
610 if (!cell)
611 continue;
612
613 adt_MCLQ *liquid = cell->getMCLQ();
614 int count = 0;
615 if (!liquid || cell->sizeMCLQ <= 8)
616 continue;
617
618 for (int y = 0; y < ADT_CELL_SIZE; y++)
619 {
620 int cy = i * ADT_CELL_SIZE + y;
621 for (int x = 0; x < ADT_CELL_SIZE; x++)
622 {
623 int cx = j * ADT_CELL_SIZE + x;
624 if (liquid->flags[y][x] != 0x0F)
625 {
626 liquid_show[cy][cx] = true;
627 if (liquid->flags[y][x] & (1<<7))
628 liquid_flags[i][j] |= MAP_LIQUID_TYPE_DARK_WATER;
629 ++count;
630 }
631 }
632 }
633
634 uint32 c_flag = cell->flags;
635 if (c_flag & (1<<2))
636 {
637 liquid_entry[i][j] = 1;
638 liquid_flags[i][j] |= MAP_LIQUID_TYPE_WATER; // water
639 }
640 if (c_flag & (1<<3))
641 {
642 liquid_entry[i][j] = 2;
643 liquid_flags[i][j] |= MAP_LIQUID_TYPE_OCEAN; // ocean
644 }
645 if (c_flag & (1<<4))
646 {
647 liquid_entry[i][j] = 3;
648 liquid_flags[i][j] |= MAP_LIQUID_TYPE_MAGMA; // magma/slime
649 }
650
651 if (!count && liquid_flags[i][j])
652 fprintf(stderr, "Wrong liquid detect in MCLQ chunk");
653
654 for (int y = 0; y <= ADT_CELL_SIZE; y++)
655 {
656 int cy = i * ADT_CELL_SIZE + y;
657 for (int x = 0; x <= ADT_CELL_SIZE; x++)
658 {
659 int cx = j * ADT_CELL_SIZE + x;
660 liquid_height[cy][cx] = liquid->liquid[y][x].height;
661 }
662 }
663 }
664 }
665
666 // Get liquid map for grid (in WOTLK used MH2O chunk)
667 adt_MH2O * h2o = adt.a_grid->getMH2O();
668 if (h2o)
669 {
670 for (int i = 0; i < ADT_CELLS_PER_GRID; i++)
671 {
672 for(int j = 0; j < ADT_CELLS_PER_GRID; j++)
673 {
674 adt_liquid_header *h = h2o->getLiquidData(i,j);
675 if (!h)
676 continue;
677
678 int count = 0;
679 uint64 show = h2o->getLiquidShowMap(h);
680 for (int y = 0; y < h->height; y++)
681 {
682 int cy = i * ADT_CELL_SIZE + y + h->yOffset;
683 for (int x = 0; x < h->width; x++)
684 {
685 int cx = j * ADT_CELL_SIZE + x + h->xOffset;
686 if (show & 1)
687 {
688 liquid_show[cy][cx] = true;
689 ++count;
690 }
691 show >>= 1;
692 }
693 }
694
695 liquid_entry[i][j] = h->liquidType;
696 switch (LiqType[h->liquidType])
697 {
698 case LIQUID_TYPE_WATER: liquid_flags[i][j] |= MAP_LIQUID_TYPE_WATER; break;
699 case LIQUID_TYPE_OCEAN: liquid_flags[i][j] |= MAP_LIQUID_TYPE_OCEAN; break;
700 case LIQUID_TYPE_MAGMA: liquid_flags[i][j] |= MAP_LIQUID_TYPE_MAGMA; break;
701 case LIQUID_TYPE_SLIME: liquid_flags[i][j] |= MAP_LIQUID_TYPE_SLIME; break;
702 default:
703 printf("\nCan't find Liquid type %u for map %s\nchunk %d,%d\n", h->liquidType, inputPath.c_str(), i, j);
704 break;
705 }
706 // Dark water detect
707 if (LiqType[h->liquidType] == LIQUID_TYPE_OCEAN)
708 {
709 uint8 *lm = h2o->getLiquidLightMap(h);
710 if (!lm)
711 liquid_flags[i][j] |= MAP_LIQUID_TYPE_DARK_WATER;
712 }
713
714 if (!count && liquid_flags[i][j])
715 printf("Wrong liquid detect in MH2O chunk");
716
717 float *height = h2o->getLiquidHeightMap(h);
718 int pos = 0;
719 for (int y=0; y<=h->height;y++)
720 {
721 int cy = i*ADT_CELL_SIZE + y + h->yOffset;
722 for (int x=0; x<= h->width; x++)
723 {
724 int cx = j*ADT_CELL_SIZE + x + h->xOffset;
725 if (height)
726 liquid_height[cy][cx] = height[pos];
727 else
728 liquid_height[cy][cx] = h->heightLevel1;
729 pos++;
730 }
731 }
732 }
733 }
734 }
735 //============================================
736 // Pack liquid data
737 //============================================
738 uint8 type = liquid_flags[0][0];
739 bool fullType = false;
740 for (int y=0;y<ADT_CELLS_PER_GRID;y++)
741 {
742 for(int x=0;x<ADT_CELLS_PER_GRID;x++)
743 {
744 if (liquid_flags[y][x]!=type)
745 {
746 fullType = true;
747 y = ADT_CELLS_PER_GRID;
748 break;
749 }
750 }
751 }
752
753 map_liquidHeader liquidHeader;
754
755 // no water data (if all grid have 0 liquid type)
756 if (type == 0 && !fullType)
757 {
758 // No liquid data
759 map.liquidMapOffset = 0;
760 map.liquidMapSize = 0;
761 }
762 else
763 {
764 int minX = 255, minY = 255;
765 int maxX = 0, maxY = 0;
766 maxHeight = -20000;
767 minHeight = 20000;
768 for (int y=0; y<ADT_GRID_SIZE; y++)
769 {
770 for(int x=0; x<ADT_GRID_SIZE; x++)
771 {
772 if (liquid_show[y][x])
773 {
774 if (minX > x) minX = x;
775 if (maxX < x) maxX = x;
776 if (minY > y) minY = y;
777 if (maxY < y) maxY = y;
778 float h = liquid_height[y][x];
779 if (maxHeight < h) maxHeight = h;
780 if (minHeight > h) minHeight = h;
781 }
782 else
783 liquid_height[y][x] = CONF_use_minHeight;
784 }
785 }
786 map.liquidMapOffset = map.heightMapOffset + map.heightMapSize;
787 map.liquidMapSize = sizeof(map_liquidHeader);
788 liquidHeader.fourcc = *reinterpret_cast<uint32 const*>(MAP_LIQUID_MAGIC);
789 liquidHeader.flags = 0;
790 liquidHeader.liquidType = 0;
791 liquidHeader.offsetX = minX;
792 liquidHeader.offsetY = minY;
793 liquidHeader.width = maxX - minX + 1 + 1;
794 liquidHeader.height = maxY - minY + 1 + 1;
795 liquidHeader.liquidLevel = minHeight;
796
797 if (maxHeight == minHeight)
798 liquidHeader.flags |= MAP_LIQUID_NO_HEIGHT;
799
800 // Not need store if flat surface
801 if (CONF_allow_float_to_int && (maxHeight - minHeight) < CONF_flat_liquid_delta_limit)
802 liquidHeader.flags |= MAP_LIQUID_NO_HEIGHT;
803
804 if (!fullType)
805 liquidHeader.flags |= MAP_LIQUID_NO_TYPE;
806
807 if (liquidHeader.flags & MAP_LIQUID_NO_TYPE)
808 liquidHeader.liquidType = type;
809 else
810 map.liquidMapSize += sizeof(liquid_entry) + sizeof(liquid_flags);
811
812 if (!(liquidHeader.flags & MAP_LIQUID_NO_HEIGHT))
813 map.liquidMapSize += sizeof(float)*liquidHeader.width*liquidHeader.height;
814 }
815
816 // map hole info
817 uint16 holes[ADT_CELLS_PER_GRID][ADT_CELLS_PER_GRID];
818
819 if (map.liquidMapOffset)
820 map.holesOffset = map.liquidMapOffset + map.liquidMapSize;
821 else
822 map.holesOffset = map.heightMapOffset + map.heightMapSize;
823
824 memset(holes, 0, sizeof(holes));
825 bool hasHoles = false;
826
827 for (int i = 0; i < ADT_CELLS_PER_GRID; ++i)
828 {
829 for (int j = 0; j < ADT_CELLS_PER_GRID; ++j)
830 {
831 adt_MCNK * cell = cells->getMCNK(i,j);
832 if (!cell)
833 continue;
834 holes[i][j] = cell->holes;
835 if (!hasHoles && cell->holes != 0)
836 hasHoles = true;
837 }
838 }
839
840 if (hasHoles)
841 map.holesSize = sizeof(holes);
842 else
843 map.holesSize = 0;
844
845 // Ok all data prepared - store it
846
847 std::ofstream outFile(outputPath, std::ofstream::out | std::ofstream::binary);
848 if (!outFile)
849 {
850 printf("Can't create the output file '%s'\n", outputPath.c_str());
851 return false;
852 }
853
854 outFile.write(reinterpret_cast<const char*>(&map), sizeof(map));
855 // Store area data
856 outFile.write(reinterpret_cast<const char*>(&areaHeader), sizeof(areaHeader));
857 if (!(areaHeader.flags & MAP_AREA_NO_AREA))
858 outFile.write(reinterpret_cast<const char*>(area_ids), sizeof(area_ids));
859
860 // Store height data
861 outFile.write(reinterpret_cast<const char*>(&heightHeader), sizeof(heightHeader));
862 if (!(heightHeader.flags & MAP_HEIGHT_NO_HEIGHT))
863 {
864 if (heightHeader.flags & MAP_HEIGHT_AS_INT16)
865 {
866 outFile.write(reinterpret_cast<const char*>(uint16_V9), sizeof(uint16_V9));
867 outFile.write(reinterpret_cast<const char*>(uint16_V8), sizeof(uint16_V8));
868 }
869 else if (heightHeader.flags & MAP_HEIGHT_AS_INT8)
870 {
871 outFile.write(reinterpret_cast<const char*>(uint8_V9), sizeof(uint8_V9));
872 outFile.write(reinterpret_cast<const char*>(uint8_V8), sizeof(uint8_V8));
873 }
874 else
875 {
876 outFile.write(reinterpret_cast<const char*>(V9), sizeof(V9));
877 outFile.write(reinterpret_cast<const char*>(V8), sizeof(V8));
878 }
879 }
880
881 if (heightHeader.flags & MAP_HEIGHT_HAS_FLIGHT_BOUNDS)
882 {
883 outFile.write(reinterpret_cast<char*>(flight_box_max), sizeof(flight_box_max));
884 outFile.write(reinterpret_cast<char*>(flight_box_min), sizeof(flight_box_min));
885 }
886
887 // Store liquid data if need
888 if (map.liquidMapOffset)
889 {
890 outFile.write(reinterpret_cast<const char*>(&liquidHeader), sizeof(liquidHeader));
891
892 if (!(liquidHeader.flags&MAP_LIQUID_NO_TYPE))
893 {
894 outFile.write(reinterpret_cast<const char*>(liquid_entry), sizeof(liquid_entry));
895 outFile.write(reinterpret_cast<const char*>(liquid_flags), sizeof(liquid_flags));
896 }
897
898 if (!(liquidHeader.flags&MAP_LIQUID_NO_HEIGHT))
899 {
900 for (int y = 0; y < liquidHeader.height; y++)
901 outFile.write(reinterpret_cast<const char*>(&liquid_height[y + liquidHeader.offsetY][liquidHeader.offsetX]), sizeof(float) * liquidHeader.width);
902 }
903 }
904
905 // store hole data
906 if (hasHoles)
907 outFile.write(reinterpret_cast<const char*>(holes), map.holesSize);
908
909 outFile.close();
910 return true;
911}

978void ExtractDBCFiles(int locale, bool basicLocale)
979{
980 printf("Extracting dbc files...\n");
981
982 std::set<std::string> dbcfiles;
983
984 // get DBC file list
985 for(ArchiveSet::iterator i = gOpenArchives.begin(); i != gOpenArchives.end();++i)
986 {
987 std::vector<std::string> files;
988 (*i)->GetFileListTo(files);
989 for (std::vector<std::string>::iterator iter = files.begin(); iter != files.end(); ++iter)
990 if (iter->rfind(".dbc") == iter->length() - strlen(".dbc"))
991 dbcfiles.insert(*iter);
992 }
993
994 std::string path = output_path;
995 path += "/dbc/";
996 CreateDir(path);
997 if(!basicLocale)
998 {
999 path += langs[locale];
1000 path += "/";
1001 CreateDir(path);
1002 }
1003
1004 // extract Build info file
1005 {
1006 std::string mpq_name = std::string("component.wow-") + langs[locale] + ".txt";
1007 std::string filename = path + mpq_name;
1008
1009 ExtractFile(mpq_name.c_str(), filename);
1010 }
1011
1012 // extract DBCs
1013 uint32 count = 0;
1014 for (std::set<std::string>::iterator iter = dbcfiles.begin(); iter != dbcfiles.end(); ++iter)
1015 {
1016 std::string filename = path;
1017 filename += (iter->c_str() + strlen("DBFilesClient\\"));
1018
1019 if (boost::filesystem::exists(filename))
1020 continue;
1021
1022 if (ExtractFile(iter->c_str(), filename))
1023 ++count;
1024 }
1025 printf("Extracted %u DBC files\n\n", count);
1026}

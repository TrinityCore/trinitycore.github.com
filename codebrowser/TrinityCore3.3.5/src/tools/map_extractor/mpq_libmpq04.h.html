40 void GetFileListTo(std::vector<std::string>& filelist) {
41 uint32_t filenum;
42 if(libmpq__file_number(mpq_a, "(listfile)", &filenum)) return;
43 libmpq__off_t size, transferred;
44 libmpq__file_size_unpacked(mpq_a, filenum, &size);
45
46 char *buffer = new char[size+1];
47 buffer[size] = '\0';
48
49 libmpq__file_read(mpq_a, filenum, (unsigned char*)buffer, size, &transferred);
50
51 char seps[] = "\n";
52 char *token;
53
54 token = strtok( buffer, seps );
55 uint32 counter = 0;
56 while ((token != NULL) && (counter < size)) {
57 //cout << token << endl;
58 token[strlen(token) - 1] = 0;
59 std::string s = token;
60 filelist.push_back(s);
61 counter += strlen(token) + 2;
62 token = strtok(NULL, seps);
63 }
64
65 delete[] buffer;
66 }
67};

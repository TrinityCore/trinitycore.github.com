25MPQArchive::MPQArchive(const char* filename)
26{
27 int result = libmpq__archive_open(&mpq_a, filename, -1);
28 printf("Opening %s\n", filename);
29 if(result) {
30 switch(result) {
31 case LIBMPQ_ERROR_OPEN :
32 printf("Error opening archive '%s': Does file really exist?\n", filename);
33 break;
34 case LIBMPQ_ERROR_FORMAT : /* bad file format */
35 printf("Error opening archive '%s': Bad file format\n", filename);
36 break;
37 case LIBMPQ_ERROR_SEEK : /* seeking in file failed */
38 printf("Error opening archive '%s': Seeking in file failed\n", filename);
39 break;
40 case LIBMPQ_ERROR_READ : /* Read error in archive */
41 printf("Error opening archive '%s': Read error in archive\n", filename);
42 break;
43 case LIBMPQ_ERROR_MALLOC : /* maybe not enough memory? :) */
44 printf("Error opening archive '%s': Maybe not enough memory\n", filename);
45 break;
46 default:
47 printf("Error opening archive '%s': Unknown error\n", filename);
48 break;
49 }
50 return;
51 }
52 gOpenArchives.push_front(this);
53}

61MPQFile::MPQFile(const char* filename):
62 eof(false),
63 buffer(0),
64 pointer(0),
65 size(0)
66{
67 for(ArchiveSet::iterator i=gOpenArchives.begin(); i!=gOpenArchives.end();++i)
68 {
69 mpq_archive *mpq_a = (*i)->mpq_a;
70
71 uint32_t filenum;
72 if(libmpq__file_number(mpq_a, filename, &filenum)) continue;
73 libmpq__off_t transferred;
74 libmpq__file_size_unpacked(mpq_a, filenum, &size);
75
76 // HACK: in patch.mpq some files don't want to open and give 1 for filesize
77 if (size<=1) {
78// printf("warning: file %s has size %d; cannot read.\n", filename, size);
79 eof = true;
80 buffer = 0;
81 return;
82 }
83 buffer = new char[size];
84
85 //libmpq_file_getdata
86 libmpq__file_read(mpq_a, filenum, (unsigned char*)buffer, size, &transferred);
87 /*libmpq_file_getdata(&mpq_a, hash, fileno, (unsigned char*)buffer);*/
88 return;
89
90 }
91 eof = true;
92 buffer = 0;
93}

95size_t MPQFile::read(void* dest, size_t bytes)
96{
97 if (eof) return 0;
98
99 size_t rpos = pointer + bytes;
100 if (rpos > size_t(size)) {
101 bytes = size - pointer;
102 eof = true;
103 }
104
105 memcpy(dest, &(buffer[pointer]), bytes);
106
107 pointer = rpos;
108
109 return bytes;
110}

112void MPQFile::seek(int offset)
113{
114 pointer = offset;
115 eof = (pointer >= size);
116}

118void MPQFile::seekRelative(int offset)
119{
120 pointer += offset;
121 eof = (pointer >= size);
122}

124void MPQFile::close()

38 struct MapTiles
39 {
40 MapTiles() : m_mapId(uint32(-1)), m_tiles(NULL) {}
41
42 MapTiles(uint32 id, std::set<uint32>* tiles) : m_mapId(id), m_tiles(tiles) {}
43 ~MapTiles() {}
44
45 uint32 m_mapId;
46 std::set<uint32>* m_tiles;
47
48 bool operator==(uint32 id)
49 {
50 return m_mapId == id;
51 }
52 };
53
54 typedef std::list<MapTiles> TileList;
55
56 struct Tile
57 {
58 Tile() : chf(NULL), solid(NULL), cset(NULL), pmesh(NULL), dmesh(NULL) {}
59 ~Tile()
60 {
61 rcFreeCompactHeightfield(chf);
62 rcFreeContourSet(cset);
63 rcFreeHeightField(solid);
64 rcFreePolyMesh(pmesh);
65 rcFreePolyMeshDetail(dmesh);
66 }
67 rcCompactHeightfield* chf;
68 rcHeightfield* solid;
69 rcContourSet* cset;
70 rcPolyMesh* pmesh;
71 rcPolyMeshDetail* dmesh;
72 };

74 class MapBuilder

76 public:
77 MapBuilder(float maxWalkableAngle = 70.f,
78 bool skipLiquid = false,
79 bool skipContinents = false,

81 bool skipBattlegrounds = false,
82 bool debugOutput = false,
83 bool bigBaseUnit = false,
84 const char* offMeshFilePath = NULL);
85
86 ~MapBuilder();
87
88 // builds all mmap tiles for the specified map id (ignores skip settings)
89 void buildMap(uint32 mapID);
90 void buildMeshFromFile(char* name);

93 void buildSingleTile(uint32 mapID, uint32 tileX, uint32 tileY);
94

96 void buildAllMaps(int threads);

98 void WorkerThread();

100 private:

109 // move map building

116 dtNavMesh* navMesh);

119 float* verts, int vertCount,

123 bool shouldSkipMap(uint32 mapID);

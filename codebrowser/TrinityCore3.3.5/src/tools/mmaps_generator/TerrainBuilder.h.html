61 struct MeshData
62 {
63 G3D::Array<float> solidVerts;
64 G3D::Array<int> solidTris;
65
66 G3D::Array<float> liquidVerts;
67 G3D::Array<int> liquidTris;
68 G3D::Array<uint8> liquidType;
69
70 // offmesh connection data

73 G3D::Array<unsigned char> offMeshConnectionDirs;

78 class TerrainBuilder
79 {
80 public:
81 TerrainBuilder(bool skipLiquid);
82 ~TerrainBuilder();
83
84 TerrainBuilder(const TerrainBuilder &tb) = delete;
85
86 void loadMap(uint32 mapID, uint32 tileX, uint32 tileY, MeshData &meshData);
87 bool loadVMap(uint32 mapID, uint32 tileX, uint32 tileY, MeshData &meshData);
88 void loadOffMeshConnections(uint32 mapID, uint32 tileX, uint32 tileY, MeshData &meshData, const char* offMeshFilePath);
89
90 bool usesLiquids() const { return !m_skipLiquid; }
91
92 // vert and triangle methods
93 static void transform(std::vector<G3D::Vector3> &original, std::vector<G3D::Vector3> &transformed,
94 float scale, G3D::Matrix3 &rotation, G3D::Vector3 &position);
95 static void copyVertices(std::vector<G3D::Vector3> &source, G3D::Array<float> &dest);
96 static void copyIndices(std::vector<VMAP::MeshTriangle> &source, G3D::Array<int> &dest, int offest, bool flip);
97 static void copyIndices(G3D::Array<int> &src, G3D::Array<int> &dest, int offset);
98 static void cleanVertices(G3D::Array<float> &verts, G3D::Array<int> &tris);
99 private:
100 /// Loads a portion of a map's terrain
101 bool loadMap(uint32 mapID, uint32 tileX, uint32 tileY, MeshData &meshData, Spot portion);
102
103 /// Sets loop variables for selecting only certain parts of a map's terrain
104 void getLoopVars(Spot portion, int &loopStart, int &loopEnd, int &loopInc);
105
106 /// Controls whether liquids are loaded
107 bool m_skipLiquid;
108
109 /// Get the vector coordinate for a specific position
110 void getHeightCoord(int index, Grid grid, float xOffset, float yOffset, float* coord, float* v);
111
112 /// Get the triangle's vector indices for a specific position
113 void getHeightTriangle(int square, Spot triangle, int* indices, bool liquid = false);
114
115 /// Determines if the specific position's triangles should be rendered
116 bool isHole(int square, const uint16 holes[16][16]);
117
118 /// Get the liquid vector coordinate for a specific position
119 void getLiquidCoord(int index, int index2, float xOffset, float yOffset, float* coord, float* v);
120
121 /// Get the liquid type for a specific position
122 uint8 getLiquidType(int square, const uint8 liquid_type[16][16]);
123 };

32bool DBCFile::open()
33{
34 MPQFile f(filename.c_str());
35
36 // Need some error checking, otherwise an unhandled exception error occurs
37 // if people screw with the data path.
38 if (f.isEof() == true)
39 return false;
40
41 unsigned char header[4];
42 unsigned int na,nb,es,ss;
43
44 f.read(header,4); // File Header
45
46 if (header[0]!='W' || header[1]!='D' || header[2]!='B' || header[3] != 'C')
47 {
48 f.close();
49 data = NULL;
50 printf("Critical Error: An error occured while trying to read the DBCFile %s.", filename.c_str());
51 return false;
52 }
53
54 //assert(header[0]=='W' && header[1]=='D' && header[2]=='B' && header[3] == 'C');
55
56 f.read(&na,4); // Number of records
57 f.read(&nb,4); // Number of fields
58 f.read(&es,4); // Size of a record
59 f.read(&ss,4); // String size
60
61 recordSize = es;
62 recordCount = na;
63 fieldCount = nb;
64 stringSize = ss;
65 //assert(fieldCount*4 == recordSize);
66 assert(fieldCount*4 >= recordSize);
67
68 data = new unsigned char[recordSize*recordCount+stringSize];
69 stringTable = data + recordSize*recordCount;
70 f.read(data,recordSize*recordCount+stringSize);
71 f.close();
72 return true;
73}

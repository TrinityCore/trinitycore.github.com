27bool ExtractSingleModel(std::string& fname)
28{
29 char * name = GetPlainName((char*)fname.c_str());
30 char * ext = GetExtension(name);
31
32 // < 3.1.0 ADT MMDX section store filename.mdx filenames for corresponded .m2 file
33 if (!strcmp(ext, ".mdx"))
34 {
35 // replace .mdx -> .m2
36 fname.erase(fname.length()-2,2);
37 fname.append("2");
38 }
39 // >= 3.1.0 ADT MMDX section store filename.m2 filenames for corresponded .m2 file
40 // nothing do
41
42 std::string output(szWorkDirWmo);
43 output += "/";
44 output += name;
45
46 if (FileExists(output.c_str()))
47 return true;
48
49 Model mdl(fname);
50 if (!mdl.open())
51 return false;
52
53 return mdl.ConvertToVMAPModel(output.c_str());
54}

56void ExtractGameobjectModels()
57{
58 printf("Extracting GameObject models...");
59 DBCFile dbc("DBFilesClient\\GameObjectDisplayInfo.dbc");
60 if(!dbc.open())
61 {
62 printf("Fatal error: Invalid GameObjectDisplayInfo.dbc file format!\n");
63 exit(1);
64 }
65
66 std::string basepath = szWorkDirWmo;
67 basepath += "/";
68 std::string path;
69
70 std::string modelListPath = basepath + "temp_gameobject_models";
71 FILE* model_list = fopen(modelListPath.c_str(), "wb");
72 if (!model_list)
73 {
74 printf("Fatal error: Could not open file %s\n", modelListPath.c_str());
75 return;
76 }
77
78 for (DBCFile::Iterator it = dbc.begin(); it != dbc.end(); ++it)
79 {
80 path = it->getString(1);
81
82 if (path.length() < 4)
83 continue;
84
85 fixnamen((char*)path.c_str(), path.size());
86 char * name = GetPlainName((char*)path.c_str());
87 fixname2(name, strlen(name));
88
89 char * ch_ext = GetExtension(name);
90 if (!ch_ext)
91 continue;
92
93 strToLower(ch_ext);
94
95 bool result = false;
96 if (!strcmp(ch_ext, ".wmo"))
97 {
98 result = ExtractSingleWmo(path);
99 }
100 else if (!strcmp(ch_ext, ".mdl"))
101 {
102 // TODO: extract .mdl files, if needed
103 continue;
104 }
105 else //if (!strcmp(ch_ext, ".mdx") || !strcmp(ch_ext, ".m2"))
106 {
107 result = ExtractSingleModel(path);
108 }
109
110 if (result)
111 {
112 uint32 displayId = it->getUInt(0);
113 uint32 path_length = strlen(name);
114 fwrite(&displayId, sizeof(uint32), 1, model_list);
115 fwrite(&path_length, sizeof(uint32), 1, model_list);
116 fwrite(name, sizeof(char), path_length, model_list);
117 }
118 }
119
120 fclose(model_list);
121
122 printf("Done!\n");
123}

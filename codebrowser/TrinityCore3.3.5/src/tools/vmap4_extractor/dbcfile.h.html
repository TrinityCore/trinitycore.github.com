57 // Iteration over database
58 class Iterator;
59 class Record
60 {
61 public:
62 Record& operator= (const Record& r)
63 {
64 file = r.file;
65 offset = r.offset;
66 return *this;
67 }
68 float getFloat(size_t field) const
69 {
70 assert(field < file.fieldCount);
71 return *reinterpret_cast<float*>(offset+field*4);
72 }
73 unsigned int getUInt(size_t field) const
74 {
75 assert(field < file.fieldCount);
76 return *reinterpret_cast<unsigned int*>(offset+(field*4));
77 }
78 int getInt(size_t field) const
79 {
80 assert(field < file.fieldCount);
81 return *reinterpret_cast<int*>(offset+field*4);
82 }
83 unsigned char getByte(size_t ofs) const
84 {
85 assert(ofs < file.recordSize);
86 return *reinterpret_cast<unsigned char*>(offset+ofs);
87 }
88 const char *getString(size_t field) const
89 {
90 assert(field < file.fieldCount);
91 size_t stringOffset = getUInt(field);
92 assert(stringOffset < file.stringSize);
93 //char * tmp = (char*)file.stringTable + stringOffset;
94 //unsigned char * tmp2 = file.stringTable + stringOffset;
95 return reinterpret_cast<char*>(file.stringTable + stringOffset);
96 }
97 private:
98 Record(DBCFile &file, unsigned char *offset): file(file), offset(offset) {}
99 DBCFile &file;
100 unsigned char *offset;
101
102 friend class DBCFile;
103 friend class Iterator;
104 };

106 /* Iterator that iterates over records */
107 class Iterator
108 {
109 public:
110 Iterator(DBCFile &file, unsigned char *offset):
111 record(file, offset) {}
112 /// Advance (prefix only)
113 Iterator & operator++() {
114 record.offset += record.file.recordSize;
115 return *this;
116 }
117 /// Return address of current instance
118 Record const & operator*() const { return record; }
119 const Record* operator->() const {
120 return &record;
121 }
122 /// Comparison
123 bool operator==(const Iterator &b) const
124 {
125 return record.offset == b.record.offset;
126 }
127 bool operator!=(const Iterator &b) const
128 {
129 return record.offset != b.record.offset;
130 }
131 private:
132 Record record;
133 };
134
135 // Get record by id
136 Record getRecord(size_t id);

138 Iterator begin();
139 /// Get begin iterator over records

151 unsigned char *data;
152 unsigned char *stringTable;

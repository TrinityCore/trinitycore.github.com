47void fixnamen(char* name, size_t len)
48{
49 for (size_t i = 0; i < len-3; i++)
50 {
51 if (i > 0 && name[i] >= 'A' && name[i] <= 'Z' && isalpha(name[i-1]))
52 name[i] |= 0x20;
53 else if ((i == 0 || !isalpha(name[i-1])) && name[i]>='a' && name[i]<='z')
54 name[i] &= ~0x20;
55 }
56 //extension in lowercase
57 for (size_t i = len - 3; i < len; i++)
58 name[i] |= 0x20;
59}
60
61void fixname2(char* name, size_t len)
62{
63 for (size_t i=0; i<len-3; i++)
64 {
65 if(name[i] == ' ')
66 name[i] = '_';
67 }
68}

82bool ADTFile::init(uint32 map_num, uint32 tileX, uint32 tileY)
83{
84 if(ADT.isEof ())
85 return false;
86
87 uint32 size;
88
89 std::string xMap;
90 std::string yMap;
91
92 Adtfilename.erase(Adtfilename.find(".adt"),4);
93 std::string TempMapNumber;
94 TempMapNumber = Adtfilename.substr(Adtfilename.length()-6,6);
95 xMap = TempMapNumber.substr(TempMapNumber.find("_")+1,(TempMapNumber.find_last_of("_")-1) - (TempMapNumber.find("_")));
96 yMap = TempMapNumber.substr(TempMapNumber.find_last_of("_")+1,(TempMapNumber.length()) - (TempMapNumber.find_last_of("_")));
97 Adtfilename.erase((Adtfilename.length()-xMap.length()-yMap.length()-2), (xMap.length()+yMap.length()+2));
98 //string AdtMapNumber = xMap + ' ' + yMap + ' ' + GetPlainName((char*)Adtfilename.c_str());
99 //printf("Processing map %s...\n", AdtMapNumber.c_str());
100 //printf("MapNumber = %s\n", TempMapNumber.c_str());
101 //printf("xMap = %s\n", xMap.c_str());
102 //printf("yMap = %s\n", yMap.c_str());
103
104 std::string dirname = std::string(szWorkDirWmo) + "/dir_bin";
105 FILE *dirfile;
106 dirfile = fopen(dirname.c_str(), "ab");
107 if(!dirfile)
108 {
109 printf("Can't open dirfile!'%s'\n", dirname.c_str());
110 return false;
111 }
112
113 while (!ADT.isEof())
114 {
115 char fourcc[5];
116 ADT.read(&fourcc,4);
117 ADT.read(&size, 4);
118 flipcc(fourcc);
119 fourcc[4] = 0;
120
121 size_t nextpos = ADT.getPos() + size;
122
123 if (!strcmp(fourcc,"MCIN"))
124 {
125 }
126 else if (!strcmp(fourcc,"MTEX"))
127 {
128 }
129 else if (!strcmp(fourcc,"MMDX"))
130 {
131 if (size)
132 {
133 char *buf = new char[size];
134 ADT.read(buf, size);
135 char *p=buf;
136 int t=0;
137 ModelInstansName = new std::string[size];
138 while (p<buf+size)
139 {
140 fixnamen(p,strlen(p));
141 char* s = GetPlainName(p);
142 fixname2(s,strlen(s));
143
144 ModelInstansName[t++] = s;
145
146 std::string path(p);
147 ExtractSingleModel(path);
148
149 p = p+strlen(p)+1;
150 }
151 delete[] buf;
152 }
153 }
154 else if (!strcmp(fourcc,"MWMO"))
155 {
156 if (size)
157 {
158 char* buf = new char[size];
159 ADT.read(buf, size);
160 char* p=buf;
161 int q = 0;
162 WmoInstansName = new std::string[size];
163 while (p<buf+size)
164 {
165 char* s = GetPlainName(p);
166 fixnamen(s, strlen(s));
167 fixname2(s, strlen(s));
168 p += strlen(p) + 1;
169 WmoInstansName[q++] = s;
170 }
171 delete[] buf;
172 }
173 }
174 //======================
175 else if (!strcmp(fourcc,"MDDF"))
176 {
177 if (size)
178 {
179 nMDX = (int)size / 36;
180 for (int i=0; i<nMDX; ++i)
181 {
182 uint32 id;
183 ADT.read(&id, 4);
184 ModelInstance inst(ADT,ModelInstansName[id].c_str(), map_num, tileX, tileY, dirfile);
185 }
186 delete[] ModelInstansName;
187 ModelInstansName = NULL;
188 }
189 }
190 else if (!strcmp(fourcc,"MODF"))
191 {
192 if (size)
193 {
194 nWMO = (int)size / 64;
195 for (int i=0; i<nWMO; ++i)
196 {
197 uint32 id;
198 ADT.read(&id, 4);
199 WMOInstance inst(ADT,WmoInstansName[id].c_str(), map_num, tileX, tileY, dirfile);
200 }
201 delete[] WmoInstansName;
202 WmoInstansName = NULL;
203 }
204 }
205 //======================
206 ADT.seek(nextpos);
207 }
208 ADT.close();
209 fclose(dirfile);
210 return true;
211}

95void ReadLiquidTypeTableDBC()
96{
97 printf("Read LiquidType.dbc file...");
98 DBCFile dbc("DBFilesClient\\LiquidType.dbc");
99 if(!dbc.open())
100 {
101 printf("Fatal error: Invalid LiquidType.dbc file format!\n");
102 exit(1);
103 }
104
105 size_t LiqType_count = dbc.getRecordCount();
106 size_t LiqType_maxid = dbc.getRecord(LiqType_count - 1).getUInt(0);
107 LiqType = new uint16[LiqType_maxid + 1];
108 memset(LiqType, 0xff, (LiqType_maxid + 1) * sizeof(uint16));
109
110 for(uint32 x = 0; x < LiqType_count; ++x)
111 LiqType[dbc.getRecord(x).getUInt(0)] = dbc.getRecord(x).getUInt(3);
112
113 printf("Done! (%u LiqTypes loaded)\n", (unsigned int)LiqType_count);
114}

116bool ExtractWmo()
117{
118 bool success = true;
119
120 //const char* ParsArchiveNames[] = {"patch-2.MPQ", "patch.MPQ", "common.MPQ", "expansion.MPQ"};
121
122 for (ArchiveSet::const_iterator ar_itr = gOpenArchives.begin(); ar_itr != gOpenArchives.end() && success; ++ar_itr)
123 {
124 std::vector<std::string> filelist;
125
126 (*ar_itr)->GetFileListTo(filelist);
127 for (std::vector<std::string>::iterator fname = filelist.begin(); fname != filelist.end() && success; ++fname)
128 {
129 if (fname->find(".wmo") != std::string::npos)
130 success = ExtractSingleWmo(*fname);
131 }
132 }
133
134 if (success)
135 printf("\nExtract wmo complete (No (fatal) errors)\n");
136
137 return success;
138}

140bool ExtractSingleWmo(std::string& fname)
141{
142 // Copy files from archive
143
144 char szLocalFile[1024];
145 const char * plain_name = GetPlainName(fname.c_str());
146 sprintf(szLocalFile, "%s/%s", szWorkDirWmo, plain_name);
147 fixnamen(szLocalFile,strlen(szLocalFile));
148
149 if (FileExists(szLocalFile))
150 return true;
151
152 int p = 0;
153 // Select root wmo files
154 char const* rchr = strrchr(plain_name, '_');
155 if (rchr != NULL)
156 {
157 char cpy[4];
158 memcpy(cpy, rchr, 4);
159 for (int i = 0; i < 4; ++i)
160 {
161 int m = cpy[i];
162 if (isdigit(m))
163 p++;
164 }
165 }
166
167 if (p == 3)
168 return true;
169
170 bool file_ok = true;
171 std::cout << "Extracting " << fname << std::endl;
172 WMORoot froot(fname);
173 if(!froot.open())
174 {
175 printf("Couldn't open RootWmo!!!\n");
176 return true;
177 }
178 FILE *output = fopen(szLocalFile,"wb");
179 if(!output)
180 {
181 printf("couldn't open %s for writing!\n", szLocalFile);
182 return false;
183 }
184 froot.ConvertToVMAPRootWmo(output);
185 int Wmo_nVertices = 0;
186 //printf("root has %d groups\n", froot->nGroups);
187 if (froot.nGroups !=0)
188 {
189 for (uint32 i = 0; i < froot.nGroups; ++i)
190 {
191 char temp[1024];
192 strncpy(temp, fname.c_str(), 1024);
193 temp[fname.length()-4] = 0;
194 char groupFileName[1024];
195 sprintf(groupFileName, "%s_%03u.wmo", temp, i);
196 //printf("Trying to open groupfile %s\n",groupFileName);
197
198 std::string s = groupFileName;
199 WMOGroup fgroup(s);
200 if(!fgroup.open())
201 {
202 printf("Could not open all Group file for: %s\n", plain_name);
203 file_ok = false;
204 break;
205 }
206
207 Wmo_nVertices += fgroup.ConvertToVMAPGroupWmo(output, &froot, preciseVectorData);
208 }
209 }
210
211 fseek(output, 8, SEEK_SET); // store the correct no of vertices
212 fwrite(&Wmo_nVertices,sizeof(int),1,output);
213 fclose(output);
214
215 // Delete the extracted file in the case of an error
216 if (!file_ok)
217 remove(szLocalFile);
218 return true;
219}

292bool fillArchiveNameVector(std::vector<std::string>& pArchiveNames)
293{
294 if(!hasInputPathParam)
295 getGamePath();
296
297 printf("\nGame path: %s\n", input_path);
298
299 char path[512];
300 std::string in_path(input_path);
301 std::vector<std::string> locales, searchLocales;
302
303 searchLocales.push_back("enGB");
304 searchLocales.push_back("enUS");
305 searchLocales.push_back("deDE");
306 searchLocales.push_back("esES");
307 searchLocales.push_back("frFR");
308 searchLocales.push_back("koKR");
309 searchLocales.push_back("zhCN");
310 searchLocales.push_back("zhTW");
311 searchLocales.push_back("enCN");
312 searchLocales.push_back("enTW");
313 searchLocales.push_back("esMX");
314 searchLocales.push_back("ruRU");
315
316 for (std::vector<std::string>::iterator i = searchLocales.begin(); i != searchLocales.end(); ++i)
317 {
318 std::string localePath = in_path + *i;
319 // check if locale exists:
320 struct stat status;
321 if (stat(localePath.c_str(), &status))
322 continue;
323 if ((status.st_mode & S_IFDIR) == 0)
324 continue;
325 printf("Found locale '%s'\n", i->c_str());
326 locales.push_back(*i);
327 }
328 printf("\n");
329
330 // open locale expansion and common files
331 printf("Adding data files from locale directories.\n");
332 for (std::vector<std::string>::iterator i = locales.begin(); i != locales.end(); ++i)
333 {
334 pArchiveNames.push_back(in_path + *i + "/locale-" + *i + ".MPQ");
335 pArchiveNames.push_back(in_path + *i + "/expansion-locale-" + *i + ".MPQ");
336 pArchiveNames.push_back(in_path + *i + "/lichking-locale-" + *i + ".MPQ");
337 }
338
339 // open expansion and common files
340 pArchiveNames.push_back(input_path + std::string("common.MPQ"));
341 pArchiveNames.push_back(input_path + std::string("common-2.MPQ"));
342 pArchiveNames.push_back(input_path + std::string("expansion.MPQ"));
343 pArchiveNames.push_back(input_path + std::string("lichking.MPQ"));
344
345 // now, scan for the patch levels in the core dir
346 printf("Scanning patch levels from data directory.\n");
347 sprintf(path, "%spatch", input_path);
348 if (!scan_patches(path, pArchiveNames))
349 return(false);
350
351 // now, scan for the patch levels in locale dirs
352 printf("Scanning patch levels from locale directories.\n");
353 bool foundOne = false;
354 for (std::vector<std::string>::iterator i = locales.begin(); i != locales.end(); ++i)
355 {
356 printf("Locale: %s\n", i->c_str());
357 sprintf(path, "%s%s/patch-%s", input_path, i->c_str(), i->c_str());
358 if(scan_patches(path, pArchiveNames))
359 foundOne = true;
360 }
361
362 printf("\n");
363
364 if(!foundOne)
365 {
366 printf("no locale found\n");
367 return false;
368 }
369
370 return true;
371}

373bool processArgv(int argc, char ** argv, const char *versionString)
374{
375 bool result = true;
376 hasInputPathParam = false;
377 preciseVectorData = false;
378
379 for(int i = 1; i < argc; ++i)
380 {
381 if(strcmp("-s",argv[i]) == 0)
382 {
383 preciseVectorData = false;
384 }
385 else if(strcmp("-d",argv[i]) == 0)
386 {
387 if((i+1)<argc)
388 {
389 hasInputPathParam = true;
390 strncpy(input_path, argv[i + 1], sizeof(input_path));
391 input_path[sizeof(input_path) - 1] = '\0';
392
393 if (input_path[strlen(input_path) - 1] != '\\' && input_path[strlen(input_path) - 1] != '/')
394 strcat(input_path, "/");
395 ++i;
396 }
397 else
398 {
399 result = false;
400 }
401 }
402 else if(strcmp("-?",argv[1]) == 0)
403 {
404 result = false;
405 }
406 else if(strcmp("-l",argv[i]) == 0)
407 {
408 preciseVectorData = true;
409 }
410 else
411 {
412 result = false;
413 break;
414 }
415 }
416 if(!result)
417 {
418 printf("Extract %s.\n",versionString);
419 printf("%s [-?][-s][-l][-d <path>]\n", argv[0]);
420 printf(" -s : (default) small size (data size optimization), ~500MB less vmap data.\n");
421 printf(" -l : large size, ~500MB more vmap data. (might contain more details)\n");
422 printf(" -d <path>: Path to the vector data source folder.\n");
423 printf(" -? : This message.\n");
424 }
425 return result;
426}

441 const char *versionString = "V4.00 2012_02";
442

444 if (!processArgv(argc, argv, versionString))

447 // some simple check if working dir is dirty
448 else

451 std::string sdir_bin = std::string(szWorkDirWmo) + "/dir_bin";
452 struct stat status;
453 if (!stat(sdir.c_str(), &status) || !stat(sdir_bin.c_str(), &status))
454 {
455 printf("Your output directory seems to be polluted, please use an empty directory!\n");
456 printf("<press return to exit>");
457 char garbage[2];
458 return scanf("%c", garbage);
459 }
460 }
461
462 printf("Extract %s. Beginning work ....\n",versionString);
463 //xxxxxxxxxxxxxxxxxxxxxxxxxxxxxx
464 // Create the working directory
465 if (mkdir(szWorkDirWmo
466#if defined(__linux__) || defined(__APPLE__)
467 , 0711
468#endif
469 ))
470 success = (errno == EEXIST);
471
472 // prepare archive name list
473 std::vector<std::string> archiveNames;
474 fillArchiveNameVector(archiveNames);
475 for (size_t i=0; i < archiveNames.size(); ++i)
476 {
477 MPQArchive *archive = new MPQArchive(archiveNames[i].c_str());
478 if (gOpenArchives.empty() || gOpenArchives.front() != archive)

480 }
481
482 if (gOpenArchives.empty())
483 {
484 printf("FATAL ERROR: None MPQ archive found by path '%s'. Use -d option with proper path.\n",input_path);

487 ReadLiquidTypeTableDBC();
488
489 // extract data
490 if (success)
491 success = ExtractWmo();
492
493 //xxxxxxxxxxxxxxxxxxxxxxxxxxxxxx
494 //map.dbc

496 {
497 DBCFile * dbc = new DBCFile("DBFilesClient\\Map.dbc");
498 if (!dbc->open())
499 {
500 delete dbc;
501 printf("FATAL ERROR: Map.dbc not found in data file.\n");
502 return 1;
503 }
504 map_count=dbc->getRecordCount ();
505 map_ids=new map_id[map_count];
506 for (unsigned int x=0;x<map_count;++x)
507 {
508 map_ids[x].id = dbc->getRecord(x).getUInt(0);
509
510 const char* map_name = dbc->getRecord(x).getString(1);
511 size_t max_map_name_length = sizeof(map_ids[x].name);
512 if (strlen(map_name) >= max_map_name_length)
513 {
514 delete dbc;
515 delete[] map_ids;
516 printf("FATAL ERROR: Map name too long.\n");
517 return 1;
518 }
519
520 strncpy(map_ids[x].name, map_name, max_map_name_length);
521 map_ids[x].name[max_map_name_length - 1] = '\0';
522 printf("Map - %s\n", map_ids[x].name);
523 }
524
525 delete dbc;
526 ParsMapFiles();
527 delete [] map_ids;
528 //nError = ERROR_SUCCESS;
529 // Extract models, listed in DameObjectDisplayInfo.dbc
530 ExtractGameobjectModels();
531 }
532
533 printf("\n");
534 if (!success)
535 {
536 printf("ERROR: Extract %s. Work NOT complete.\n Precise vector data=%d.\nPress any key.\n",versionString, preciseVectorData);
537 getchar();
538 }
539
540 printf("Extract %s. Work complete. No errors.\n",versionString);
541 delete [] LiqType;
542 return 0;
543}

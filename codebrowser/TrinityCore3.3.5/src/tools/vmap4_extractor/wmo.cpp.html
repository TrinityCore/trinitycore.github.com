148bool WMOGroup::open()
149{
150 MPQFile f(filename.c_str());
151 if(f.isEof ())
152 {
153 printf("No such file.\n");
154 return false;
155 }
156 uint32 size;
157 char fourcc[5];
158 while (!f.isEof())
159 {
160 f.read(fourcc,4);
161 f.read(&size, 4);
162 flipcc(fourcc);
163 if (!strcmp(fourcc,"MOGP"))//Fix sizeoff = Data size.
164 {
165 size = 68;
166 }
167 fourcc[4] = 0;
168 size_t nextpos = f.getPos() + size;
169 LiquEx_size = 0;
170 liquflags = 0;
171
172 if (!strcmp(fourcc,"MOGP"))//header
173 {
174 f.read(&groupName, 4);
175 f.read(&descGroupName, 4);
176 f.read(&mogpFlags, 4);
177 f.read(bbcorn1, 12);
178 f.read(bbcorn2, 12);
179 f.read(&moprIdx, 2);
180 f.read(&moprNItems, 2);
181 f.read(&nBatchA, 2);
182 f.read(&nBatchB, 2);
183 f.read(&nBatchC, 4);
184 f.read(&fogIdx, 4);
185 f.read(&liquidType, 4);
186 f.read(&groupWMOID,4);
187
188 }
189 else if (!strcmp(fourcc,"MOPY"))
190 {
191 MOPY = new char[size];
192 mopy_size = size;
193 nTriangles = (int)size / 2;
194 f.read(MOPY, size);
195 }
196 else if (!strcmp(fourcc,"MOVI"))
197 {
198 MOVI = new uint16[size/2];
199 f.read(MOVI, size);
200 }
201 else if (!strcmp(fourcc,"MOVT"))
202 {
203 MOVT = new float[size/4];
204 f.read(MOVT, size);
205 nVertices = (int)size / 12;
206 }
207 else if (!strcmp(fourcc,"MONR"))
208 {
209 }
210 else if (!strcmp(fourcc,"MOTV"))
211 {
212 }
213 else if (!strcmp(fourcc,"MOBA"))
214 {
215 MOBA = new uint16[size/2];
216 moba_size = size/2;
217 f.read(MOBA, size);
218 }
219 else if (!strcmp(fourcc,"MLIQ"))
220 {
221 liquflags |= 1;
222 hlq = new WMOLiquidHeader();
223 f.read(hlq, 0x1E);
224 LiquEx_size = sizeof(WMOLiquidVert) * hlq->xverts * hlq->yverts;
225 LiquEx = new WMOLiquidVert[hlq->xverts * hlq->yverts];
226 f.read(LiquEx, LiquEx_size);
227 int nLiquBytes = hlq->xtiles * hlq->ytiles;
228 LiquBytes = new char[nLiquBytes];
229 f.read(LiquBytes, nLiquBytes);
230
231 /* std::ofstream llog("Buildings/liquid.log", ios_base::out | ios_base::app);
232 llog << filename;
233 llog << "\nbbox: " << bbcorn1[0] << ", " << bbcorn1[1] << ", " << bbcorn1[2] << " | " << bbcorn2[0] << ", " << bbcorn2[1] << ", " << bbcorn2[2];
234 llog << "\nlpos: " << hlq->pos_x << ", " << hlq->pos_y << ", " << hlq->pos_z;
235 llog << "\nx-/yvert: " << hlq->xverts << "/" << hlq->yverts << " size: " << size << " expected size: " << 30 + hlq->xverts*hlq->yverts*8 + hlq->xtiles*hlq->ytiles << std::endl;
236 llog.close(); */
237 }
238 f.seek((int)nextpos);
239 }
240 f.close();
241 return true;
242}

244int WMOGroup::ConvertToVMAPGroupWmo(FILE *output, WMORoot *rootWMO, bool preciseVectorData)
245{
246 fwrite(&mogpFlags,sizeof(uint32),1,output);
247 fwrite(&groupWMOID,sizeof(uint32),1,output);
248 // group bound
249 fwrite(bbcorn1, sizeof(float), 3, output);
250 fwrite(bbcorn2, sizeof(float), 3, output);
251 fwrite(&liquflags,sizeof(uint32),1,output);
252 int nColTriangles = 0;
253 if (preciseVectorData)
254 {
255 char GRP[] = "GRP ";
256 fwrite(GRP,1,4,output);
257
258 int k = 0;
259 int moba_batch = moba_size/12;
260 MobaEx = new int[moba_batch*4];
261 for(int i=8; i<moba_size; i+=12)
262 {
263 MobaEx[k++] = MOBA[i];
264 }
265 int moba_size_grp = moba_batch*4+4;
266 fwrite(&moba_size_grp,4,1,output);
267 fwrite(&moba_batch,4,1,output);
268 fwrite(MobaEx,4,k,output);
269 delete [] MobaEx;
270
271 uint32 nIdexes = nTriangles * 3;
272
273 if(fwrite("INDX",4, 1, output) != 1)
274 {
275 printf("Error while writing file nbraches ID");
276 exit(0);
277 }
278 int wsize = sizeof(uint32) + sizeof(unsigned short) * nIdexes;
279 if(fwrite(&wsize, sizeof(int), 1, output) != 1)
280 {
281 printf("Error while writing file wsize");
282 // no need to exit?
283 }
284 if(fwrite(&nIdexes, sizeof(uint32), 1, output) != 1)
285 {
286 printf("Error while writing file nIndexes");
287 exit(0);
288 }
289 if(nIdexes >0)
290 {
291 if(fwrite(MOVI, sizeof(unsigned short), nIdexes, output) != nIdexes)
292 {
293 printf("Error while writing file indexarray");
294 exit(0);
295 }
296 }
297
298 if(fwrite("VERT",4, 1, output) != 1)
299 {
300 printf("Error while writing file nbraches ID");
301 exit(0);
302 }
303 wsize = sizeof(int) + sizeof(float) * 3 * nVertices;
304 if(fwrite(&wsize, sizeof(int), 1, output) != 1)
305 {
306 printf("Error while writing file wsize");
307 // no need to exit?
308 }
309 if(fwrite(&nVertices, sizeof(int), 1, output) != 1)
310 {
311 printf("Error while writing file nVertices");
312 exit(0);
313 }
314 if(nVertices >0)
315 {
316 if(fwrite(MOVT, sizeof(float)*3, nVertices, output) != nVertices)
317 {
318 printf("Error while writing file vectors");
319 exit(0);
320 }
321 }
322
323 nColTriangles = nTriangles;
324 }
325 else
326 {
327 char GRP[] = "GRP ";
328 fwrite(GRP,1,4,output);
329 int k = 0;
330 int moba_batch = moba_size/12;
331 MobaEx = new int[moba_batch*4];
332 for(int i=8; i<moba_size; i+=12)
333 {
334 MobaEx[k++] = MOBA[i];
335 }
336
337 int moba_size_grp = moba_batch*4+4;
338 fwrite(&moba_size_grp,4,1,output);
339 fwrite(&moba_batch,4,1,output);
340 fwrite(MobaEx,4,k,output);
341 delete [] MobaEx;
342
343 //-------INDX------------------------------------
344 //-------MOPY--------
345 MoviEx = new uint16[nTriangles*3]; // "worst case" size...
346 int *IndexRenum = new int[nVertices];
347 memset(IndexRenum, 0xFF, nVertices*sizeof(int));
348 for (int i=0; i<nTriangles; ++i)
349 {
350 // Skip no collision triangles
351 if (MOPY[2*i]&WMO_MATERIAL_NO_COLLISION ||
352 !(MOPY[2*i]&(WMO_MATERIAL_HINT|WMO_MATERIAL_COLLIDE_HIT)) )
353 continue;
354 // Use this triangle
355 for (int j=0; j<3; ++j)
356 {
357 IndexRenum[MOVI[3*i + j]] = 1;
358 MoviEx[3*nColTriangles + j] = MOVI[3*i + j];
359 }
360 ++nColTriangles;
361 }
362
363 // assign new vertex index numbers
364 int nColVertices = 0;
365 for (uint32 i=0; i<nVertices; ++i)
366 {
367 if (IndexRenum[i] == 1)
368 {
369 IndexRenum[i] = nColVertices;
370 ++nColVertices;
371 }
372 }
373
374 // translate triangle indices to new numbers
375 for (int i=0; i<3*nColTriangles; ++i)
376 {
377 assert(MoviEx[i] < nVertices);
378 MoviEx[i] = IndexRenum[MoviEx[i]];
379 }
380
381 // write triangle indices
382 int INDX[] = {0x58444E49, nColTriangles*6+4, nColTriangles*3};
383 fwrite(INDX,4,3,output);
384 fwrite(MoviEx,2,nColTriangles*3,output);
385
386 // write vertices
387 int VERT[] = {0x54524556, nColVertices*3*static_cast<int>(sizeof(float))+4, nColVertices};// "VERT"
388 int check = 3*nColVertices;
389 fwrite(VERT,4,3,output);
390 for (uint32 i=0; i<nVertices; ++i)
391 if(IndexRenum[i] >= 0)
392 check -= fwrite(MOVT+3*i, sizeof(float), 3, output);
393
394 assert(check==0);
395
396 delete [] MoviEx;
397 delete [] IndexRenum;
398 }
399
400 //------LIQU------------------------
401 if (LiquEx_size != 0)
402 {
403 int LIQU_h[] = {0x5551494C, static_cast<int>(sizeof(WMOLiquidHeader) + LiquEx_size) + hlq->xtiles*hlq->ytiles};// "LIQU"
404 fwrite(LIQU_h, 4, 2, output);
405
406 // according to WoW.Dev Wiki:
407 uint32 liquidEntry;
408 if (rootWMO->liquidType & 4)
409 liquidEntry = liquidType;
410 else if (liquidType == 15)
411 liquidEntry = 0;
412 else
413 liquidEntry = liquidType + 1;
414
415 if (!liquidEntry)
416 {
417 int v1; // edx@1
418 int v2; // eax@1
419
420 v1 = hlq->xtiles * hlq->ytiles;
421 v2 = 0;
422 if (v1 > 0)
423 {
424 while ((LiquBytes[v2] & 0xF) == 15)
425 {
426 ++v2;
427 if (v2 >= v1)
428 break;
429 }
430
431 if (v2 < v1 && (LiquBytes[v2] & 0xF) != 15)
432 liquidEntry = (LiquBytes[v2] & 0xF) + 1;
433 }
434 }
435
436 if (liquidEntry && liquidEntry < 21)
437 {
438 switch ((liquidEntry - 1) & 3)
439 {
440 case 0:
441 liquidEntry = ((mogpFlags & 0x80000) != 0) + 13;
442 break;
443 case 1:
444 liquidEntry = 14;
445 break;
446 case 2:
447 liquidEntry = 19;
448 break;
449 case 3:
450 liquidEntry = 20;
451 break;
452 }
453 }
454
455 hlq->type = liquidEntry;
456
457 /* std::ofstream llog("Buildings/liquid.log", ios_base::out | ios_base::app);
458 llog << filename;
459 llog << ":\nliquidEntry: " << liquidEntry << " type: " << hlq->type << " (root:" << rootWMO->liquidType << " group:" << liquidType << ")\n";
460 llog.close(); */
461
462 fwrite(hlq, sizeof(WMOLiquidHeader), 1, output);
463 // only need height values, the other values are unknown anyway
464 for (uint32 i = 0; i<LiquEx_size/sizeof(WMOLiquidVert); ++i)
465 fwrite(&LiquEx[i].height, sizeof(float), 1, output);
466 // todo: compress to bit field
467 fwrite(LiquBytes, 1, hlq->xtiles*hlq->ytiles, output);
468 }
469
470 return nColTriangles;
471}

484WMOInstance::WMOInstance(MPQFile& f, char const* WmoInstName, uint32 mapID, uint32 tileX, uint32 tileY, FILE* pDirfile)
485 : currx(0), curry(0), wmo(NULL), doodadset(0), pos(), indx(0), id(0), d2(0), d3(0)
486{
487 float ff[3];
488 f.read(&id, 4);
489 f.read(ff,12);
490 pos = Vec3D(ff[0],ff[1],ff[2]);
491 f.read(ff,12);
492 rot = Vec3D(ff[0],ff[1],ff[2]);
493 f.read(ff,12);
494 pos2 = Vec3D(ff[0],ff[1],ff[2]);
495 f.read(ff,12);
496 pos3 = Vec3D(ff[0],ff[1],ff[2]);
497 f.read(&d2,4);
498
499 uint16 trash,adtId;
500 f.read(&adtId,2);
501 f.read(&trash,2);
502
503 //-----------add_in _dir_file----------------
504
505 char tempname[512];
506 sprintf(tempname, "%s/%s", szWorkDirWmo, WmoInstName);
507 FILE *input;
508 input = fopen(tempname, "r+b");
509
510 if(!input)
511 {
512 printf("WMOInstance::WMOInstance: couldn't open %s\n", tempname);
513 return;
514 }
515
516 fseek(input, 8, SEEK_SET); // get the correct no of vertices
517 int nVertices;
518 int count = fread(&nVertices, sizeof (int), 1, input);
519 fclose(input);
520
521 if (count != 1 || nVertices == 0)
522 return;
523
524 float x,z;
525 x = pos.x;
526 z = pos.z;
527 if(x==0 && z == 0)
528 {
529 pos.x = 533.33333f*32;
530 pos.z = 533.33333f*32;
531 }
532 pos = fixCoords(pos);
533 pos2 = fixCoords(pos2);
534 pos3 = fixCoords(pos3);
535
536 float scale = 1.0f;
537 uint32 flags = MOD_HAS_BOUND;
538 if(tileX == 65 && tileY == 65) flags |= MOD_WORLDSPAWN;
539 //write mapID, tileX, tileY, Flags, ID, Pos, Rot, Scale, Bound_lo, Bound_hi, name
540 fwrite(&mapID, sizeof(uint32), 1, pDirfile);
541 fwrite(&tileX, sizeof(uint32), 1, pDirfile);
542 fwrite(&tileY, sizeof(uint32), 1, pDirfile);
543 fwrite(&flags, sizeof(uint32), 1, pDirfile);
544 fwrite(&adtId, sizeof(uint16), 1, pDirfile);
545 fwrite(&id, sizeof(uint32), 1, pDirfile);
546 fwrite(&pos, sizeof(float), 3, pDirfile);
547 fwrite(&rot, sizeof(float), 3, pDirfile);
548 fwrite(&scale, sizeof(float), 1, pDirfile);
549 fwrite(&pos2, sizeof(float), 3, pDirfile);
550 fwrite(&pos3, sizeof(float), 3, pDirfile);
551 uint32 nlen=strlen(WmoInstName);
552 fwrite(&nlen, sizeof(uint32), 1, pDirfile);
553 fwrite(WmoInstName, sizeof(char), nlen, pDirfile);
554
555 /* fprintf(pDirfile,"%s/%s %f,%f,%f_%f,%f,%f 1.0 %d %d %d,%d %d\n",
556 MapName,
557 WmoInstName,
558 (float) x, (float) pos.y, (float) z,
559 (float) rot.x, (float) rot.y, (float) rot.z,
560 nVertices,
561 realx1, realy1,
562 realx2, realy2
563 ); */

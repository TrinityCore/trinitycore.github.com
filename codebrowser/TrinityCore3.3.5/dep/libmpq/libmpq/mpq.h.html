45/* define errors. */
46#define LIBMPQ_ERROR_OPEN -1 /* open error on file. */
47#define LIBMPQ_ERROR_CLOSE -2 /* close error on file. */
48#define LIBMPQ_ERROR_SEEK -3 /* lseek error on file. */
49#define LIBMPQ_ERROR_READ -4 /* read error on file. */
50#define LIBMPQ_ERROR_WRITE -5 /* write error on file. */
51#define LIBMPQ_ERROR_MALLOC -6 /* memory allocation error. */
52#define LIBMPQ_ERROR_FORMAT -7 /* format errror. */
53#define LIBMPQ_ERROR_NOT_INITIALIZED -8 /* libmpq__init() wasn't called. */
54#define LIBMPQ_ERROR_SIZE -9 /* buffer size is to small. */
55#define LIBMPQ_ERROR_EXIST -10 /* file or block does not exist in archive. */
56#define LIBMPQ_ERROR_DECRYPT -11 /* we don't know the decryption seed. */
57#define LIBMPQ_ERROR_UNPACK -12 /* error on unpacking file. */

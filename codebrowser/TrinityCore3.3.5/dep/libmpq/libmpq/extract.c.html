49/* this function decompress a stream using huffman algorithm. */
50int32_t libmpq__decompress_huffman(uint8_t *in_buf, uint32_t in_size, uint8_t *out_buf, uint32_t out_size) {
51
52 (void)in_size;
53 /* TODO: make typdefs of this structs? */
54 /* some common variables. */
55 int32_t tb = 0;
56 struct huffman_tree_s *ht;
57 struct huffman_input_stream_s *is;
58
59 /* allocate memory for the huffman tree. */
60 if ((ht = malloc(sizeof(struct huffman_tree_s))) == NULL ||
61 (is = malloc(sizeof(struct huffman_input_stream_s))) == NULL) {
62
63 /* memory allocation problem. */
64 return LIBMPQ_ERROR_MALLOC;
65 }
66
67 /* cleanup structures. */
68 memset(ht, 0, sizeof(struct huffman_tree_s));
69 memset(is, 0, sizeof(struct huffman_input_stream_s));
70
71 /* initialize input stream. */
72 is->bit_buf = *(uint32_t *)in_buf;
73 in_buf += sizeof(int32_t);
74 is->in_buf = (uint8_t *)in_buf;
75 is->bits = 32;
76
77// TODO: add all the mallocs to init function and add function libmpq__huffman_tree_free() */
78// if ((result = libmpq__huffman_tree_init(ht, LIBMPQ_HUFF_DECOMPRESS)) < 0) {
79//
80// /* something on zlib initialization failed. */
81// return LIBMPQ_ERROR_UNPACK;
82// }
83
84 /* initialize the huffman tree for decompression. */
85 libmpq__huffman_tree_init(ht, LIBMPQ_HUFF_DECOMPRESS);
86
87 /* save the number of copied bytes. */
88 tb = libmpq__do_decompress_huffman(ht, is, out_buf, out_size);
89
90 /* free structures. */
91 free(is);
92 free(ht);
93
94 /* return transferred bytes. */
95 return tb;
96}

98/* this function decompress a stream using zlib algorithm. */
99int32_t libmpq__decompress_zlib(uint8_t *in_buf, uint32_t in_size, uint8_t *out_buf, uint32_t out_size) {
100
101 /* some common variables. */
102 int32_t result = 0;
103 int32_t tb = 0;
104 z_stream z;
105
106 /* fill the stream structure for zlib. */
107 z.next_in = (Bytef *)in_buf;
108 z.avail_in = (uInt)in_size;
109 z.total_in = in_size;
110 z.next_out = (Bytef *)out_buf;
111 z.avail_out = (uInt)out_size;
112 z.total_out = 0;
113 z.zalloc = NULL;
114 z.zfree = NULL;
115
116 /* initialize the decompression structure, storm.dll uses zlib version 1.1.3. */
117 if ((result = inflateInit(&z)) != Z_OK) {
118
119 /* something on zlib initialization failed. */
120 return result;
121 }
122
123 /* call zlib to decompress the data. */
124 if ((result = inflate(&z, Z_FINISH)) != Z_STREAM_END) {
125
126 /* something on zlib decompression failed. */
127 return result;
128 }
129
130 /* save transferred bytes. */
131 tb = z.total_out;
132
133 /* cleanup zlib. */
134 if ((result = inflateEnd(&z)) != Z_OK) {
135
136 /* something on zlib finalization failed. */
137 return result;
138 }
139
140 /* return transferred bytes. */
141 return tb;
142}

144/* this function decompress a stream using pkzip algorithm. */
145int32_t libmpq__decompress_pkzip(uint8_t *in_buf, uint32_t in_size, uint8_t *out_buf, uint32_t out_size) {
146
147 /* some common variables. */
148 int32_t tb = 0;
149 uint8_t *work_buf;
150 pkzip_data_s info;
151
152 /* allocate memory for pkzip data structure. */
153 if ((work_buf = malloc(sizeof(pkzip_cmp_s))) == NULL) {
154
155 /* memory allocation problem. */
156 return LIBMPQ_ERROR_MALLOC;
157 }
158
159 /* cleanup. */
160 memset(work_buf, 0, sizeof(pkzip_cmp_s));
161
162 /* fill data information structure. */
163 info.in_buf = in_buf;
164 info.in_pos = 0;
165 info.in_bytes = in_size;
166 info.out_buf = out_buf;
167 info.out_pos = 0;
168 info.max_out = out_size;
169
170 /* do the decompression. */
171 if ((tb = libmpq__do_decompress_pkzip(work_buf, &info)) < 0) {
172
173 /* free working buffer. */
174 free(work_buf);
175
176 /* something failed on pkzip decompression. */
177 return tb;
178 }
179
180 /* save transferred bytes. */
181 tb = info.out_pos;
182
183 /* free working buffer. */
184 free(work_buf);
185
186 /* return transferred bytes. */
187 return tb;
188}

190/* this function decompress a stream using bzip2 library. */
191int32_t libmpq__decompress_bzip2(uint8_t *in_buf, uint32_t in_size, uint8_t *out_buf, uint32_t out_size) {
192
193 /* some common variables. */
194 int32_t result = 0;
195 int32_t tb = 0;
196 bz_stream strm;
197
198 /* initialize the bzlib decompression. */
199 strm.bzalloc = NULL;
200 strm.bzfree = NULL;
201
202 /* initialize the structure. */
203 if ((result = BZ2_bzDecompressInit(&strm, 0, 0)) != BZ_OK) {
204
205 /* something on bzlib initialization failed. */
206 return result;
207 }
208
209 /* fill the stream structure for bzlib. */
210 strm.next_in = (char *)in_buf;
211 strm.avail_in = in_size;
212 strm.next_out = (char *)out_buf;
213 strm.avail_out = out_size;
214
215 /* do the decompression. */
216 while (BZ2_bzDecompress(&strm) != BZ_STREAM_END);
217
218 /* save transferred bytes. */
219 tb = strm.total_out_lo32;
220
221 /* cleanup of bzip stream. */
222 BZ2_bzDecompressEnd(&strm);
223
224 /* return transferred bytes. */
225 return tb;
226}

262/* this function decompress a stream using a combination of the other compression algorithm. */
263int32_t libmpq__decompress_multi(uint8_t *in_buf, uint32_t in_size, uint8_t *out_buf, uint32_t out_size) {
264
265 /* some common variables. */
266 int32_t tb = 0;
267 uint32_t count = 0;
268 uint32_t entries = (sizeof(dcmp_table) / sizeof(decompress_table_s));
269 uint8_t *temp_buf = NULL;
270 uint8_t *work_buf = 0;
271 uint8_t decompress_flag, decompress_unsupp;
272 uint32_t i;
273
274 /* get applied compression types. */
275 decompress_flag = decompress_unsupp = *in_buf++;
276
277 /* decrement data size. */
278 in_size--;
279
280 /* search decompression table type and get all types of compression. */
281 for (i = 0; i < entries; i++) {
282
283 /* check if have to apply this decompression. */
284 if (decompress_flag & dcmp_table[i].mask) {
285
286 /* increase counter for used compression algorithms. */
287 count++;
288 /* this algorithm is supported, remove from unsupp mask */
289 decompress_unsupp &= ~dcmp_table[i].mask;
290 }
291 }
292
293 /* check if there is some method unhandled. (e.g. compressed by future versions) */
294 if (decompress_unsupp) {
295
296 /* compression type is unknown and we need to implement it. :) */
297 return LIBMPQ_ERROR_UNPACK;
298 }
299
300 /* if multiple decompressions should be made, we need temporary buffer for the data. */
301 if (count > 1) {
302
303 /* allocate memory for temporary buffer. */
304 if ((temp_buf = malloc(out_size)) == NULL) {
305
306 /* memory allocation problem. */
307 return LIBMPQ_ERROR_MALLOC;
308 }
309
310 /* cleanup. */
311 memset(temp_buf, 0, out_size);
312 }
313
314 /* apply all decompressions. */
315 for (i = 0, count = 0; i < entries; i++) {
316
317 /* check if not used this kind of compression. */
318 if (decompress_flag & dcmp_table[i].mask) {
319
320 /* if multiple decompressions should be made, we need temporary buffer for the data. */
321 if (count == 0) {
322
323 /* use output buffer as working buffer. */
324 work_buf = out_buf;
325 } else {
326
327 /* use temporary buffer as working buffer. */
328 work_buf = temp_buf;
329 }
330
331 /* decompress buffer using corresponding function. */
332 if ((tb = dcmp_table[i].decompress(in_buf, in_size, work_buf, out_size)) < 0) {
333
334 /* free temporary buffer. */
335 free(temp_buf);
336
337 /* something on decompression failed. */
338 return tb;
339 }
340
341 /* move output size to source size for next compression. */
342 in_size = out_size;
343 in_buf = work_buf;
344
345 /* increase counter. */
346 count++;
347 }
348 }
349
350 /* if output buffer is not the same like target buffer, we have to copy data (this will happen on multiple decompressions). */
351 if (work_buf != out_buf) {
352
353 /* copy buffer. */
354 memcpy(out_buf, in_buf, out_size);
355 }
356
357 /* free temporary buffer. */
358 free(temp_buf);
359
360 /* return transferred bytes. */
361 return tb;
362}

10#ifndef BOOST_PROCESS_POSIX_TERMINATE_HPP
11#define BOOST_PROCESS_POSIX_TERMINATE_HPP
12
13#include <boost/process/config.hpp>
14#include <boost/system/error_code.hpp>
15#include <signal.h>
16
17namespace boost { namespace process { namespace posix {
18
19template <class Process>
20void terminate(const Process &p)
21{
22 if (::kill(p.pid, SIGKILL) == -1)
23 BOOST_PROCESS_THROW_LAST_SYSTEM_ERROR("kill(2) failed");
24}
25
26template <class Process>
27void terminate(const Process &p, boost::system::error_code &ec)
28{
29 if (::kill(p.pid, SIGKILL) == -1)
30 BOOST_PROCESS_RETURN_LAST_SYSTEM_ERROR(ec);
31 else
32 ec.clear();
33}

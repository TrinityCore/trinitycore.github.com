10#ifndef BOOST_PROCESS_POSIX_CREATE_PIPE_HPP
11#define BOOST_PROCESS_POSIX_CREATE_PIPE_HPP
12
13#include <boost/process/config.hpp>
14#include <boost/process/posix/pipe.hpp>
15#include <boost/system/error_code.hpp>
16#include <unistd.h>
17
18namespace boost { namespace process { namespace posix {
19
20inline pipe create_pipe()
21{
22 int fds[2];
23 if (::pipe(fds) == -1)
24 BOOST_PROCESS_THROW_LAST_SYSTEM_ERROR("pipe(2) failed");
25 return pipe(fds[0], fds[1]);
26}
27
28inline pipe create_pipe(boost::system::error_code &ec)
29{
30 int fds[2];
31 if (::pipe(fds) == -1)
32 BOOST_PROCESS_RETURN_LAST_SYSTEM_ERROR(ec);
33 else
34 ec.clear();
35 return pipe(fds[0], fds[1]);
36}

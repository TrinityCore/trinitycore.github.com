10#ifndef BOOST_PROCESS_POSIX_WAIT_FOR_EXIT_HPP
11#define BOOST_PROCESS_POSIX_WAIT_FOR_EXIT_HPP
12
13#include <boost/process/config.hpp>
14#include <boost/system/error_code.hpp>
15#include <sys/types.h>
16#include <sys/wait.h>
17
18namespace boost { namespace process { namespace posix {
19
20template <class Process>
21inline int wait_for_exit(const Process &p)
22{
23 pid_t ret;
24 int status;
25 do
26 {
27 ret = ::waitpid(p.pid, &status, 0);
28 } while ((ret == -1 && errno == EINTR) || (ret != -1 && !WIFEXITED(status)));
29 if (ret == -1)
30 BOOST_PROCESS_THROW_LAST_SYSTEM_ERROR("waitpid(2) failed");
31 return status;
32}
33
34template <class Process>
35inline int wait_for_exit(const Process &p, boost::system::error_code &ec)
36{
37 pid_t ret;
38 int status;
39 do
40 {
41 ret = ::waitpid(p.pid, &status, 0);
42 } while ((ret == -1 && errno == EINTR) || (ret != -1 && !WIFEXITED(status)));
43 if (ret == -1)
44 BOOST_PROCESS_RETURN_LAST_SYSTEM_ERROR(ec);
45 else
46 ec.clear();
47 return status;
48}

10#ifndef BOOST_PROCESS_POSIX_INITIALIZERS_RUN_EXE_HPP
11#define BOOST_PROCESS_POSIX_INITIALIZERS_RUN_EXE_HPP
12
13#include <boost/process/posix/initializers/initializer_base.hpp>
14#include <boost/filesystem.hpp>
15#include <boost/shared_array.hpp>
16#include <string>
17
18namespace boost { namespace process { namespace posix { namespace initializers {
19
20class run_exe_ : public initializer_base
21{
22public:
23 explicit run_exe_(const std::string &s) : s_(s), cmd_line_(new char*[2])
24 {
25 cmd_line_[0] = const_cast<char*>(s_.c_str());
26 cmd_line_[1] = 0;
27 }
28
29 template <class PosixExecutor>
30 void on_exec_setup(PosixExecutor &e) const
31 {
32 e.exe = s_.c_str();
33 if (!e.cmd_line)
34 e.cmd_line = cmd_line_.get();
35 }
36
37private:
38 std::string s_;
39 boost::shared_array<char*> cmd_line_;
40};
41
42inline run_exe_ run_exe(const char *s)
43{
44 return run_exe_(s);
45}
46
47inline run_exe_ run_exe(const std::string &s)
48{
49 return run_exe_(s);
50}
51
52inline run_exe_ run_exe(const boost::filesystem::path &p)
53{
54 return run_exe_(p.string());
55}
56
57}}}}
58
59#endif

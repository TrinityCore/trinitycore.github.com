10#ifndef BOOST_PROCESS_POSIX_INITIALIZERS_THROW_ON_ERROR_HPP
11#define BOOST_PROCESS_POSIX_INITIALIZERS_THROW_ON_ERROR_HPP
12
13#include <boost/process/config.hpp>
14#include <boost/process/posix/initializers/initializer_base.hpp>
15#include <boost/system/error_code.hpp>
16#include <boost/system/system_error.hpp>
17#include <unistd.h>
18#include <fcntl.h>
19#include <errno.h>
20
21namespace boost { namespace process { namespace posix { namespace initializers {
22
23class throw_on_error : public initializer_base
24{
25public:
26 template <class PosixExecutor>
27 void on_fork_setup(PosixExecutor&) const
28 {
29 if (::pipe(fds_) == -1)
30 BOOST_PROCESS_THROW_LAST_SYSTEM_ERROR("pipe(2) failed");
31 if (::fcntl(fds_[1], F_SETFD, FD_CLOEXEC) == -1)
32 {
33 int e = errno;
34 ::close(fds_[0]);
35 ::close(fds_[1]);
36 BOOST_PROCESS_THROW(boost::system::system_error(
37 boost::system::error_code(e, boost::system::system_category()),
38 BOOST_PROCESS_SOURCE_LOCATION "fcntl(2) failed"));
39 }
40 }
41
42 template <class PosixExecutor>
43 void on_fork_error(PosixExecutor&) const
44 {
45 int e = errno;
46 ::close(fds_[0]);
47 ::close(fds_[1]);
48 BOOST_PROCESS_THROW(boost::system::system_error(
49 boost::system::error_code(e, boost::system::system_category()),
50 BOOST_PROCESS_SOURCE_LOCATION "fork(2) failed"));
51 }
52
53 template <class PosixExecutor>
54 void on_fork_success(PosixExecutor&) const
55 {
56 ::close(fds_[1]);
57 int code;
58 if (::read(fds_[0], &code, sizeof(int)) > 0)
59 {
60 ::close(fds_[0]);
61 BOOST_PROCESS_THROW(boost::system::system_error(
62 boost::system::error_code(code,
63 boost::system::system_category()),
64 BOOST_PROCESS_SOURCE_LOCATION "execve(2) failed"));
65 }
66 ::close(fds_[0]);
67 }
68
69 template <class PosixExecutor>
70 void on_exec_setup(PosixExecutor&) const
71 {
72 ::close(fds_[0]);
73 }
74
75 template <class PosixExecutor>
76 void on_exec_error(PosixExecutor&) const
77 {
78 int e = errno;
79 while (::write(fds_[1], &e, sizeof(int)) == -1 && errno == EINTR)
80 ;
81 ::close(fds_[1]);
82 }
83
84private:
85 mutable int fds_[2];
86};

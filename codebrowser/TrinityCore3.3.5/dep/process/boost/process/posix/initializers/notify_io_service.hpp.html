10#ifndef BOOST_PROCESS_POSIX_INITIALIZERS_NOTIFY_IO_SERVICE_HPP
11#define BOOST_PROCESS_POSIX_INITIALIZERS_NOTIFY_IO_SERVICE_HPP
12
13#include <boost/process/config.hpp>
14#include <boost/process/posix/initializers/initializer_base.hpp>
15
16namespace boost { namespace process { namespace posix { namespace initializers {
17
18template <class IOService>
19class notify_io_service_ : public initializer_base
20{
21public:
22 explicit notify_io_service_(IOService &io_service) :
23 io_service_(io_service) {}
24
25 template <class PosixExecutor>
26 void on_fork_setup(PosixExecutor&) const
27 {
28 io_service_.notify_fork(IOService::fork_prepare);
29 }
30
31 template <class PosixExecutor>
32 void on_fork_success(PosixExecutor&) const
33 {
34 io_service_.notify_fork(IOService::fork_parent);
35 }
36
37 template <class PosixExecutor>
38 void on_exec_setup(PosixExecutor&) const
39 {
40 io_service_.notify_fork(IOService::fork_child);
41 }
42
43private:
44 IOService &io_service_;
45};
46
47template <class IOService>
48notify_io_service_<IOService> notify_io_service(IOService &io_service)
49{
50 return notify_io_service_<IOService>(io_service);
51}
52
53}}}}
54
55#endif

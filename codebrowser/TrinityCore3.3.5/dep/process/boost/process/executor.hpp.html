27/**
28 * Starts a program.
29 *
30 * boost::process::executor is a functor which calls the system functions
31 * to start a program. Before system functions are called it iterates
32 * over initializers and calls a member function passing a reference
33 * to itself as a parameter. Initializers get then a chance to setup
34 * the executor. If system functions fail boost::process::executor again
35 * iterates over initializers and calls another member function passing a
36 * reference to itself as a parameter. This gives initializers a
37 * chance to handle the error.
38 *
39 * \note Library users shouldn't need to use boost::process::executor.
40 * It is recommended to call boost::process::execute which uses
41 * boost::pocess::executor internally.
42 */
43struct executor
44{
45 /**
46 * Default constructor.
47 */
48 executor();
49
50 /**
51 * Starts a program.
52 *
53 * \tparam initializers define what and how the program is started
54 */
55 template <class Initializer, class... Initializers>
56 child operator()(const Initializer &initializer, const Initializers... &initializers);
57
58 ///\defgroup WindowsOnly Windows only.
59 ///@{
60
61 /**
62 * Program name.
63 *
64 * \remark <em>Windows only.</em>
65 */
66 LPCTSTR exe;
67
68 /**
69 * Command line.
70 *
71 * \remark <em>Windows only.</em>
72 */
73 LPTSTR cmd_line;
74
75 /**
76 * Process attributes.
77 *
78 * \remark <em>Windows only.</em>
79 */
80 LPSECURITY_ATTRIBUTES proc_attrs;
81
82 /**
83 * Thread attributes.
84 *
85 * \remark <em>Windows only.</em>
86 */
87 LPSECURITY_ATTRIBUTES thread_attrs;
88
89 /**
90 * Flag to inherit handles.
91 *
92 * \remark <em>Windows only.</em>
93 */
94 BOOL inherit_handles;
95
96 /**
97 * Creation flags.
98 *
99 * \remark <em>Windows only.</em>
100 */
101 DWORD creation_flags;
102
103 /**
104 * Environment variables.

106 * \remark <em>Windows only.</em>
107 */
108 LPVOID env;
109
110 /**
111 * Work directory.

113 * \remark <em>Windows only.</em>

115 LPCTSTR work_dir;

117 /**

120 * \remark <em>Windows only.</em>

125 * Startupinfoex structure.

127 * If this member variable is available, \c startup_info is a reference

132 STARTUPINFOEX startup_info_ex;
133
134 /**
135 * Process information.

138 * could be started successfully.
139 *
140 * \remark <em>Windows only.</em>
141 */
142 PROCESS_INFORMATION proc_info;

144 ///@}

161 char **cmd_line;

163 /**

168 char **env;

170 ///@}
171};

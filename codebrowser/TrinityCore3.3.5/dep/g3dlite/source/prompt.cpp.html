55
56 DialogTemplate(LPCSTR caption, DWORD style,
57 int x, int y, int w, int h,
58 LPCSTR font = NULL, WORD fontSize = 8) {
59
60 usedBufferLength = sizeof(DLGTEMPLATE);
61 totalBufferLength = usedBufferLength;
62
63 dialogTemplate = (DLGTEMPLATE*)malloc(totalBufferLength);
64
65 dialogTemplate->style = style;
66
67 if (font != NULL) {
68 dialogTemplate->style |= DS_SETFONT;
69 }
70
71 dialogTemplate->x = (short)x;
72 dialogTemplate->y = (short)y;
73 dialogTemplate->cx = (short)w;
74 dialogTemplate->cy = (short)h;
75 dialogTemplate->cdit = 0;
76
77 dialogTemplate->dwExtendedStyle = 0;
78
79 // The dialog box doesn't have a menu or a special class
80 AppendData("\0", 2);
81 AppendData("\0", 2);
82
83 // Add the dialog's caption to the template
84
85 AppendString(caption);
86
87 if (font != NULL) {
88 AppendData(&fontSize, sizeof(WORD));
89 AppendString(font);
90 }
91 }

197protected:
198
199 void AddStandardComponent(WORD type, LPCSTR caption, DWORD style, DWORD exStyle,
200 int x, int y, int w, int h, WORD id, LPSTR font = NULL, WORD fontSize = 8) {
201
202 DLGITEMTEMPLATE item;
203
204 // DWORD align the beginning of the component data
205
206 AlignData(sizeof(DWORD));
207
208 item.style = style;
209 if (font != NULL) {
210 item.style |= DS_SETFONT;
211 }
212 item.x = (short)x;
213 item.y = (short)y;
214 item.cx = (short)w;
215 item.cy = (short)h;
216 item.id = id;
217
218 item.dwExtendedStyle = exStyle;
219
220 AppendData(&item, sizeof(DLGITEMTEMPLATE));
221
222 WORD preType = 0xFFFF;
223
224 AppendData(&preType, sizeof(WORD));
225 AppendData(&type, sizeof(WORD));
226
227 AppendString(caption);
228
229 if (font != NULL) {
230 AppendData(&fontSize, sizeof(WORD));
231 AppendString(font);
232 }
233
234 // Increment the component count
235 dialogTemplate->cdit++;
236 }
237
238
239 void AlignData(int size) {
240
241 int paddingSize = usedBufferLength % size;
242
243 if (paddingSize != 0) {
244 EnsureSpace(paddingSize);
245 usedBufferLength += paddingSize;
246 }
247
248 }
249
250 void AppendString(LPCSTR string) {
251
252 int length = MultiByteToWideChar(CP_ACP, 0, string, -1, NULL, 0);
253
254 WCHAR* wideString = (WCHAR*)malloc(sizeof(WCHAR) * length);
255 MultiByteToWideChar(CP_ACP, 0, string, -1, wideString, length);
256
257 AppendData(wideString, length * sizeof(WCHAR));
258 free(wideString);
259
260 }
261
262 void AppendData(const void* data, int dataLength) {
263
264 EnsureSpace(dataLength);
265
266 memcpy((char*)dialogTemplate + usedBufferLength, data, dataLength);
267 usedBufferLength += dataLength;
268
269 }
270
271 void EnsureSpace(int length) {
272 if (length + usedBufferLength > totalBufferLength) {
273 totalBufferLength += length * 2;
274
275 void* newBuffer = malloc(totalBufferLength);
276 memcpy(newBuffer, dialogTemplate, usedBufferLength);
277
278 free(dialogTemplate);
279 dialogTemplate = (DLGTEMPLATE*)newBuffer;
280 }
281 }
282
283private:
284
285 DLGTEMPLATE* dialogTemplate;
286
287 int totalBufferLength;
288 int usedBufferLength;

304INT_PTR CALLBACK PromptDlgProc(HWND hDlg, UINT msg,
305 WPARAM wParam, LPARAM lParam) {
306 switch(msg) {
307 case WM_INITDIALOG:
308 {
309 PromptParams *params = (PromptParams*)lParam;
310 ::SetWindowTextA(::GetDlgItem(hDlg, IDC_MESSAGE), params->message);
311
312 ::SetFocus(::GetDlgItem(hDlg, IDC_BUTTON0));
313
314 SetWindowTextA(hDlg, params->title);
315
316 HFONT hfont =
317 CreateFontA(16, 0, 0, 0, FW_NORMAL,
318 FALSE, FALSE, FALSE,
319 ANSI_CHARSET,OUT_DEFAULT_PRECIS,CLIP_DEFAULT_PRECIS,
320 PROOF_QUALITY, FIXED_PITCH | FF_MODERN, "Courier New");
321
322 SendDlgItemMessage(hDlg, IDC_MESSAGE, WM_SETFONT, (WPARAM)hfont, MAKELPARAM(TRUE,0));
323
324
325 break;
326 }
327 case WM_COMMAND:
328 {
329 int choiceNumber = LOWORD(wParam) - IDC_BUTTON0;
330 if ((choiceNumber >= 0) && (choiceNumber < 10)) {
331 EndDialog(hDlg, choiceNumber);
332 return TRUE;
333 }
334 }
335
336 break;
337
338 case WM_NCDESTROY:
339 // Under SDL 1.2.6 we get a NCDESTROY message for no reason and the
340 // window is immediately closed. This is here to debug the problem.
341 (void)0;
342 break;
343
344 }
345
346 return FALSE;
347}

354/**
355 * Show a dialog prompt.
356 */
357static int guiPrompt(
358 const char* windowTitle,
359 const char* prompt,
360 const char** choice,
361 int numChoices) {
362
363 int width = 340;
364 int height = 220;
365
366 const int buttonSpacing = 2;
367 const int buttonWidth =
368 (width - buttonSpacing * 2 -
369 buttonSpacing * (numChoices - 1)) / numChoices;
370 const int buttonHeight = 13;
371
372
373 DialogTemplate dialogTemplate(
374 windowTitle,
375 WS_CAPTION | DS_CENTER | WS_SYSMENU,
376 10, 10, width, height,
377 "Tahoma");
378
379 dialogTemplate.AddEditBox(
380 "Edit", WS_VISIBLE | ES_READONLY | ES_OEMCONVERT | ES_MULTILINE | WS_TABSTOP, WS_EX_STATICEDGE,
381 2, 2, width - 4, height - buttonHeight - 7, IDC_MESSAGE);
382
383 int i;
384 for (i = 0; i < numChoices; ++i) {
385
386 int x = buttonSpacing + i * (buttonWidth + buttonSpacing);
387 int y = height - buttonHeight - buttonSpacing;
388
389 dialogTemplate.AddButton(choice[i], WS_VISIBLE | WS_TABSTOP, 0,
390 x, y, buttonWidth, buttonHeight, IDC_BUTTON0 + (WORD)i);
391
392 }
393
394 // Convert all single \n characters to \r\n for proper printing
395 int strLen = 0;
396 const char* pStr = prompt;
397
398 while (*pStr != '\0') {
399 if ((*pStr == '\n') && (pStr != prompt)) {
400 if (*(pStr - 1) != '\r') {
401 ++strLen;
402 }
403 }
404 ++strLen;
405 ++pStr;
406 }
407
408 char* newStr = (char*)malloc(strLen + 1);
409
410 const char* pStr2 = prompt;
411 char* pNew = newStr;
412
413 while (*pStr2 != '\0') {
414 if ((*pStr2 == '\n') && (pStr2 != prompt)) {
415 if (*(pStr2 - 1) != '\r') {
416 *pNew = '\r';
417 ++pNew;
418 }
419 }
420 *pNew = *pStr2;
421 ++pNew;
422 ++pStr2;
423 }
424
425 *pNew = '\0';
426
427 PromptParams params;
428 params.message = newStr;;
429 params.title = windowTitle;
430
431 HMODULE module = GetModuleHandle(0);
432 int ret = (int)DialogBoxIndirectParam(module, dialogTemplate, NULL, (DLGPROC) PromptDlgProc, (DWORD)&params);
433
434 free(newStr);
435
436 /*
437 For debugging when DialogBoxIndirectParam fails:
438
439 // The last error value. (Which is preserved across the call).
440 DWORD lastErr = GetLastError();
441
442 // The decoded message from FormatMessage
443 LPTSTR formatMsg = NULL;
444
445 if (NULL == formatMsg) {
446 FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER |
447 FORMAT_MESSAGE_IGNORE_INSERTS |
448 FORMAT_MESSAGE_FROM_SYSTEM,
449 NULL,
450 lastErr,
451 0,
452 (LPTSTR)&formatMsg,
453 0,
454 NULL);
455 }
456
457 // Make sure the message got translated into something.
458 LPTSTR realLastErr;
459 if (NULL != formatMsg) {
460 realLastErr = formatMsg;
461 } else {
462 realLastErr = "Last error code does not exist.";
463 }
464
465 // Get rid of the allocated memory from FormatMessage.
466 if (NULL != formatMsg) {
467 LocalFree((LPVOID)formatMsg);
468 }
469 */
470
471 return ret;
472}

478/**
479 * Show a prompt on stdout
480 */
481static int textPrompt(
482 const char* windowTitle,
483 const char* prompt,
484 const char** choice,
485 int numChoices) {
486
487 printf("\n___________________________________________________\n");
488 printf("%s\n", windowTitle);
489 printf("%s", prompt);
490
491 if (numChoices > 10) {
492 numChoices = 10;
493 }
494
495 int c = -1;
496 if (numChoices > 1) {
497 printf("\n");
498 printf("Choose an option by number:");
499
500 while ((c < 0) || (c >= numChoices)) {
501 printf("\n");
502
503 for (int i = 0; i < numChoices; ++i) {
504 if (numChoices <= 3) {
505 printf(" (%d) %s ", i, choice[i]);
506 } else {
507 printf(" (%d) %s\n", i, choice[i]);
508 }
509 }
510
511 printf("\n> ");
512 c = _getch() - '0';
513
514 if ((c < 0) || (c >= numChoices)) {
515 printf("'%d' is not a valid choice.", c);
516 } else {
517 printf("%d", c);
518 }
519 }
520
521 } else if (numChoices == 1) {
522
523 printf("\nPress any key for '%s'...", choice[0]);
524 _getch();
525 c = 0;
526
527 } else {
528
529 printf("\nPress any key...");
530 _getch();
531 c = 0;
532 }
533
534 printf("\n___________________________________________________\n");
535 return c;
536}

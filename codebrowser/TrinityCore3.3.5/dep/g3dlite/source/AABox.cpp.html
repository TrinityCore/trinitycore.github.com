25AABox::AABox(const Any& a) {
26 if (a.name() == "AABox::empty") {
27 *this = AABox::empty();
28 } else if (a.name() == "AABox::inf") {
29 *this = AABox::inf();
30 } else {
31 a.verifyName("AABox");
32 a.verifyType(Any::ARRAY);
33 if (a.size() == 1) {
34 *this = AABox(Point3(a[0]));
35 } else if (a.size() == 2) {
36 set(Point3(a[0]), Point3(a[1]));
37 } else {
38 a.verify(false, "AABox must recieve exactly 1 or two arguments.");
39 }
40 }
41}

44Any AABox::toAny() const {
45 if (isEmpty()) {
46 return Any(Any::ARRAY, "AABox::empty");
47 } else if (! isFinite()) {
48 return Any(Any::ARRAY, "AABox::inf");
49 } else {
50 Any a(Any::ARRAY, "AABox");
51 if (lo == hi) {
52 a.append(lo);
53 } else {
54 a.append(lo, hi);
55 }
56 return a;
57 }

112void AABox::split(const Vector3::Axis& axis, float location, AABox& low, AABox& high) const {
113 // Low, medium, and high along the chosen axis
114 float L = G3D::min(location, lo[axis]);
115 float M = G3D::min(G3D::max(location, lo[axis]), hi[axis]);
116 float H = G3D::max(location, hi[axis]);
117
118 // Copy over this box.
119 high = low = *this;
120
121 // Now move the split points along the special axis
122 low.lo[axis] = L;
123 low.hi[axis] = M;
124 high.lo[axis] = M;
125 high.hi[axis] = H;
126}

129Vector3 AABox::randomSurfacePoint() const {
130 Vector3 extent = hi - lo;
131 float aXY = extent.x * extent.y;
132 float aYZ = extent.y * extent.z;
133 float aZX = extent.z * extent.x;
134
135 float r = (float)uniformRandom(0.0f, aXY + aYZ + aZX);
136
137 // Choose evenly between positive and negative face planes
138 float d = ((float)uniformRandom(0, 1) < 0.5f) ? 0.0f : 1.0f;
139
140 // The probability of choosing a given face is proportional to
141 // its area.
142 if (r < aXY) {
143 return
144 lo +
145 Vector3(
146 (float)uniformRandom(0.0f, extent.x),
147 (float)uniformRandom(0.0f, extent.y),
148 d * extent.z);
149 } else if (r < aYZ) {
150 return
151 lo +
152 Vector3(
153 d * extent.x,
154 (float)uniformRandom(0, extent.y),
155 (float)uniformRandom(0, extent.z));
156 } else {
157 return
158 lo +
159 Vector3(
160 (float)uniformRandom(0, extent.x),
161 d * extent.y,
162 (float)uniformRandom(0, extent.z));
163 }
164}

175bool AABox::intersects(const AABox& other) const {
176 // Must be overlap along all three axes.
177 // Try to find a separating axis.
178
179 for (int a = 0; a < 3; ++a) {
180
181 // |--------|
182 // |------|
183
184 if ((lo[a] > other.hi[a]) ||
185 (hi[a] < other.lo[a])) {
186 return false;
187 }
188 }
189
190 return true;
191}

195bool AABox::culledBy
196 (const Array<Plane>& plane,
197 int& cullingPlane,
198 const uint32 _inMask,
199 uint32& childMask) const {
200
201 uint32 inMask = _inMask;
202 assert(plane.size() < 31);
203
204 childMask = 0;
205
206 const bool finite =
207 (abs(lo.x) < G3D::finf()) &&
208 (abs(hi.x) < G3D::finf()) &&
209 (abs(lo.y) < G3D::finf()) &&
210 (abs(hi.y) < G3D::finf()) &&
211 (abs(lo.z) < G3D::finf()) &&
212 (abs(hi.z) < G3D::finf());
213
214 // See if there is one plane for which all of the
215 // vertices are in the negative half space.
216 for (int p = 0; p < plane.size(); ++p) {
217
218 // Only test planes that are not masked
219 if ((inMask & 1) != 0) {
220
221 Vector3 corner;
222
223 int numContained = 0;
224 int v = 0;
225
226 // We can early-out only if we have found one point on each
227 // side of the plane (i.e. if we are straddling). That
228 // occurs when (numContained < v) && (numContained > 0)
229 for (v = 0; (v < 8) && ((numContained == v) || (numContained == 0)); ++v) {
230 // Unrolling these 3 if's into a switch decreases performance
231 // by about 2x
232 corner.x = (v & 1) ? hi.x : lo.x;
233 corner.y = (v & 2) ? hi.y : lo.y;
234 corner.z = (v & 4) ? hi.z : lo.z;
235
236 if (finite) { // this branch is highly predictable
237 if (plane[p].halfSpaceContainsFinite(corner)) {
238 ++numContained;
239 }
240 } else {
241 if (plane[p].halfSpaceContains(corner)) {
242 ++numContained;
243 }
244 }
245 }
246
247 if (numContained == 0) {
248 // Plane p culled the box
249 cullingPlane = p;
250
251 // The caller should not recurse into the children,
252 // since the parent is culled. If they do recurse,
253 // make them only test against this one plane, which
254 // will immediately cull the volume.
255 childMask = 1 << p;
256 return true;
257
258 } else if (numContained < v) {
259 // The bounding volume straddled the plane; we have
260 // to keep testing against this plane
261 childMask |= (1 << p);
262 }
263 }
264
265 // Move on to the next bit.
266 inMask = inMask >> 1;
267 }
268
269 // None of the planes could cull this box
270 cullingPlane = -1;
271 return false;
272}

275bool AABox::culledBy(
276 const Array<Plane>& plane,
277 int& cullingPlane,
278 const uint32 _inMask) const {
279
280 uint32 inMask = _inMask;
281 assert(plane.size() < 31);
282
283 const bool finite =
284 (abs(lo.x) < G3D::finf()) &&
285 (abs(hi.x) < G3D::finf()) &&
286 (abs(lo.y) < G3D::finf()) &&
287 (abs(hi.y) < G3D::finf()) &&
288 (abs(lo.z) < G3D::finf()) &&
289 (abs(hi.z) < G3D::finf());
290
291 // See if there is one plane for which all of the
292 // vertices are in the negative half space.
293 for (int p = 0; p < plane.size(); ++p) {
294
295 // Only test planes that are not masked
296 if ((inMask & 1) != 0) {
297
298 bool culled = true;
299 Vector3 corner;
300
301 int v;
302
303 // Assume this plane culls all points. See if there is a point
304 // not culled by the plane... early out when at least one point
305 // is in the positive half space.
306 for (v = 0; (v < 8) && culled; ++v) {
307
308 // Unrolling these 3 if's into a switch decreases performance
309 // by about 2x
310 corner.x = (v & 1) ? hi.x : lo.x;
311 corner.y = (v & 2) ? hi.y : lo.y;
312 corner.z = (v & 4) ? hi.z : lo.z;
313
314 if (finite) { // this branch is highly predictable
315 culled = ! plane[p].halfSpaceContainsFinite(corner);
316 } else {
317 culled = ! plane[p].halfSpaceContains(corner);
318 }
319 }
320
321 if (culled) {
322 // Plane p culled the box
323 cullingPlane = p;
324
325 return true;
326 }
327 }
328
329 // Move on to the next bit.
330 inMask = inMask >> 1;
331 }
332
333 // None of the planes could cull this box
334 cullingPlane = -1;
335 return false;
336}

345bool AABox::intersects(const Sphere& sphere) const {
346 double d = 0;
347
348 //find the square of the distance
349 //from the sphere to the box
350 for (int i = 0; i < 3; ++i) {
351 if (sphere.center[i] < lo[i]) {
352 d += square(sphere.center[i] - lo[i]);
353 } else if (sphere.center[i] > hi[i]) {
354 d += square(sphere.center[i] - hi[i]);
355 }
356 }
357
358 return d <= square(sphere.radius);
359}

361Vector3 AABox::corner(int index) const {
362
363 // default constructor inits all components to 0
364 Vector3 v;
365
366 switch (index)
367 {
368 case 0:
369 v.x = lo.x;
370 v.y = lo.y;
371 v.z = hi.z;
372 break;
373
374 case 1:
375 v.x = hi.x;
376 v.y = lo.y;
377 v.z = hi.z;
378 break;
379
380 case 2:
381 v.x = hi.x;
382 v.y = hi.y;
383 v.z = hi.z;
384 break;
385
386 case 3:
387 v.x = lo.x;
388 v.y = hi.y;
389 v.z = hi.z;
390 break;
391
392 case 4:
393 v.x = lo.x;
394 v.y = lo.y;
395 v.z = lo.z;
396 break;
397
398 case 5:
399 v.x = hi.x;
400 v.y = lo.y;
401 v.z = lo.z;
402 break;
403
404 case 6:
405 v.x = hi.x;
406 v.y = hi.y;
407 v.z = lo.z;
408 break;
409
410 case 7:
411 v.x = lo.x;
412 v.y = hi.y;
413 v.z = lo.z;
414 break;
415
416 default:
417 debugAssertM(false, "Invalid corner index");
418 break;
419 }
420
421 return v;
422}

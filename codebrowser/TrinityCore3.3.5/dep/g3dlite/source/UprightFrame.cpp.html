13UprightFrame::UprightFrame(const CoordinateFrame& cframe) {
14 Vector3 look = cframe.lookVector();
15
16 yaw = (float)(G3D::pi() + atan2(look.x, look.z));
17 pitch = asin(look.y);
18
19 translation = cframe.translation;
20}

49CoordinateFrame UprightFrame::toCoordinateFrame() const {
50 CoordinateFrame cframe;
51
52 Matrix3 P(Matrix3::fromAxisAngle(Vector3::unitX(), pitch));
53 Matrix3 Y(Matrix3::fromAxisAngle(Vector3::unitY(), yaw));
54
55 cframe.rotation = Y * P;
56 cframe.translation = translation;
57
58 return cframe;
59}

72void UprightFrame::unwrapYaw(UprightFrame* a, int N) {
73 // Use the first point to establish the wrapping convention
74 for (int i = 1; i < N; ++i) {
75 const float prev = a[i - 1].yaw;
76 float& cur = a[i].yaw;
77
78 // No two angles should be more than pi (i.e., 180-degrees) apart.
79 if (abs(cur - prev) > G3D::pi()) {
80 // These angles must have wrapped at zero, causing them
81 // to be interpolated the long way.
82
83 // Find canonical [0, 2pi] versions of these numbers
84 float p = (float)wrap(prev, twoPi());
85 float c = (float)wrap(cur, twoPi());
86
87 // Find the difference -pi < diff < pi between the current and previous values
88 float diff = c - p;
89 if (diff < -G3D::pi()) {
90 diff += (float)twoPi();
91 } else if (diff > G3D::pi()) {
92 diff -= (float)twoPi();
93 }
94
95 // Offset the current from the previous by the difference
96 // between them.
97 cur = prev + diff;
98 }
99 }
100}

121UprightSpline::UprightSpline(const Any& any) {
122 any.verifyName("UprightSpline");
123 any.verifyType(Any::TABLE);
124
125 extrapolationMode = any["extrapolationMode"];
126
127 const Any& controlsAny = any["control"];
128 controlsAny.verifyType(Any::ARRAY);
129
130 control.resize(controlsAny.length());
131 for (int controlIndex = 0; controlIndex < control.length(); ++controlIndex) {
132 control[controlIndex] = controlsAny[controlIndex];
133 }
134
135 const Any& timesAny = any["time"];
136 timesAny.verifyType(Any::ARRAY);
137
138 time.resize(timesAny.length());
139 for (int timeIndex = 0; timeIndex < time.length(); ++timeIndex) {
140 time[timeIndex] = timesAny[timeIndex];
141 }
142}

145Any UprightSpline::toAny(const std::string& myName) const {
146 Any any(Any::TABLE, myName);
147
148 any["extrapolationMode"] = extrapolationMode;
149
150 Any controlsAny(Any::ARRAY);
151 for (int controlIndex = 0; controlIndex < control.length(); ++controlIndex) {
152 controlsAny.append(control[controlIndex]);
153 }
154 any["control"] = controlsAny;
155
156 Any timesAny(Any::ARRAY);
157 for (int timeIndex = 0; timeIndex < time.length(); ++timeIndex) {
158 timesAny.append(Any(time[timeIndex]));
159 }
160 any["time"] = timesAny;
161
162 return any;

177void UprightSpline::serialize(class BinaryOutput& b) const {
178 b.writeInt32(extrapolationMode);
179
180 b.writeInt32(control.size());
181 for (int i = 0; i < control.size(); ++i) {
182 control[i].serialize(b);
183 }
184 b.writeInt32(time.size());
185 for (int i = 0; i < time.size(); ++i) {
186 b.writeFloat32(time[i]);
187 }
188}

191void UprightSpline::deserialize(class BinaryInput& b) {
192 extrapolationMode = SplineExtrapolationMode(b.readInt32());
193
194 control.resize(b.readInt32());
195 for (int i = 0; i < control.size(); ++i) {
196 control[i].deserialize(b);
197 }
198
199 if (b.hasMore()) {
200 time.resize(b.readInt32());
201 for (int i = 0; i < time.size(); ++i) {
202 time[i] = b.readFloat32();
203 }
204 debugAssert(time.size() == control.size());
205 } else {
206 // Import legacy path
207 time.resize(control.size());
208 for (int i = 0; i < time.size(); ++i) {
209 time[i] = (float)i;
210 }
211 }

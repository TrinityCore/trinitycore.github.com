19/**
20 Abstraction of memory management.
21 Default implementation uses G3D::System::malloc and is threadsafe.
22
23 \sa LargePoolMemoryManager, CRTMemoryManager, AlignedMemoryManager, AreaMemoryManager */
24class MemoryManager : public ReferenceCountedObject {
25protected:
26
27 MemoryManager();
28
29public:
30
31 typedef shared_ptr<class MemoryManager> Ref;
32
33 /** Return a pointer to \a s bytes of memory that are unused by
34 the rest of the program. The contents of the memory are
35 undefined */
36 virtual void* alloc(size_t s);
37
38 /** Invoke to declare that this memory will no longer be used by
39 the program. The memory manager is not required to actually
40 reuse or release this memory. */
41 virtual void free(void* ptr);
42
43 /** Returns true if this memory manager is threadsafe (i.e., alloc
44 and free can be called asychronously) */
45 virtual bool isThreadsafe() const;
46
47 /** Return the instance. There's only one instance of the default
48 MemoryManager; it is cached after the first creation. */
49 static MemoryManager::Ref create();
50};
51
52/**
53 Allocates memory on 16-byte boundaries.
54 \sa MemoryManager, CRTMemoryManager, AreaMemoryManager */
55class AlignedMemoryManager : public MemoryManager {
56protected:
57
58 AlignedMemoryManager();
59
60public:
61
62 typedef shared_ptr<class AlignedMemoryManager> Ref;
63
64
65 virtual void* alloc(size_t s);
66
67 virtual void free(void* ptr);
68
69 virtual bool isThreadsafe() const;
70
71 static AlignedMemoryManager::Ref create();
72};

74
75/** A MemoryManager implemented using the C runtime. Not recommended
76 for general use; this is largely for debugging. */
77class CRTMemoryManager : public MemoryManager {
78protected:
79 CRTMemoryManager();
80
81public:
82 typedef shared_ptr<class MemoryManager> Ref;
83 virtual void* alloc(size_t s);
84 virtual void free(void* ptr);
85 virtual bool isThreadsafe() const;
86
87 /** There's only one instance of this memory manager; it is
88 cached after the first creation. */
89 static CRTMemoryManager::Ref create();
90};

19namespace G3D {
20
21/** Thrown by TextInput, Any, and other parsers on unexpected input. */
22class ParseError {
23public:
24 enum {UNKNOWN = -1};
25
26 /** Empty means unknown */
27 std::string filename;
28
29 /** For a binary file, the location of the parse error. -1 if unknown.*/
30 int64 byte;
31
32 /** For a text file, the line number is the line number of start of token which caused the exception. 1 is
33 the first line of the file. -1 means unknown. Note that you can use
34 TextInput::Settings::startingLineNumberOffset to shift the effective line
35 number that is reported by that class.
36 */
37 int line;
38
39 /** Character number (in the line) of the start of the token which caused the
40 exception. 1 is the character in the line. May be -1 if unknown.
41 */
42 int character;
43
44 std::string message;
45
46 ParseError() : byte(UNKNOWN), line(UNKNOWN), character(UNKNOWN) {}
47
48 virtual ~ParseError() {}
49
50 ParseError(const std::string& f, int l, int c, const std::string& m) :
51 filename (f), byte(UNKNOWN), line(l), character(c), message(m) {}
52
53 ParseError(const std::string& f, int64 b, const std::string& m) :
54 filename (f), byte(b), line(UNKNOWN), character(UNKNOWN), message(m) {}
55
56 /** If information is known, ends in ": ", otherwise empty */
57 std::string formatFileInfo() const;
58};
59
60}
61
62#endif

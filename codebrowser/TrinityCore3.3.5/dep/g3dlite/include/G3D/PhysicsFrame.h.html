22namespace G3D {
23
24/**
25 An RT transformation using a quaternion; suitable for
26 physics integration.
27
28 This interface is in "Beta" and will change in the next release.
29 */
30class PhysicsFrame {
31public:
32
33 Quat rotation;
34
35 /**
36 Origin of this reference frame in its parent's frame.
37 */
38 Point3 translation;
39
40 /**
41 Initializes to the identity frame.
42 */
43 PhysicsFrame();
44
45 /**
46 Purely translational.
47 */
48 PhysicsFrame(const Vector3& translation) : translation(translation) {}
49 PhysicsFrame(const Quat& rot, const Vector3& translation) : rotation(rot), translation(translation) {}
50 PhysicsFrame(const Matrix3& rot, const Vector3& translation) : rotation(rot), translation(translation) {}
51 PhysicsFrame(const Matrix3& rot) : rotation(rot), translation(Vector3::zero()) {}
52 PhysicsFrame(const CoordinateFrame& coordinateFrame);
53
54 PhysicsFrame& operator=(const PhysicsFrame& p) {
55 rotation = p.rotation;
56 translation = p.translation;
57
58 return *this;
59 }
60
61 /**
62 - PhysicsFrame( [quat], [vec3] )
63 - Vector3( ... )
64 - CFrame( ... )
65 - CFrame::from...( ... )
66 */
67 PhysicsFrame(const class Any& any);
68
69 Any toAny() const;
70
71 /** Compose: create the transformation that is <I>other</I> followed by <I>this</I>.*/
72 PhysicsFrame operator*(const PhysicsFrame& other) const;
73
74 virtual ~PhysicsFrame() {}
75
76 /**
77 Linear interpolation (spherical linear for the rotations).
78 */
79 PhysicsFrame lerp(
80 const PhysicsFrame& other,
81 float alpha) const;
82
83 void deserialize(class BinaryInput& b);
84
85 void serialize(class BinaryOutput& b) const;
86
87 operator CFrame() const;
88
89 /** Multiplies both pieces by \a f; note that this will result in a non-unit
90 quaternion that needs to be normalized */
91 PhysicsFrame& operator*=(float f) {
92 rotation *= f;
93 translation *= f;
94 return *this;
95 }
96
97 /** Multiplies both pieces by \a f; note that this will result in a non-unit
98 quaternion that needs to be normalized */
99 PhysicsFrame operator*(float f) const {
100 return PhysicsFrame(rotation * f, translation * f);
101 }
102
103 PhysicsFrame operator+(const PhysicsFrame& f) const {
104 return PhysicsFrame(rotation + f.rotation, translation + f.translation);
105 }
106
107 PhysicsFrame& operator+=(const PhysicsFrame& f) {
108 rotation += f.rotation;
109 translation += f.translation;
110 return *this;
111 }
112
113 bool operator==(const PhysicsFrame& other) const {
114 return (translation == other.translation) &&
115 ((rotation == other.rotation) || (rotation == -other.rotation));
116 }
117
118 bool operator!=(const PhysicsFrame& other) const {
119 return ! ((*this) == other);
120 }
121
122};
123
124typedef PhysicsFrame PFrame;
125
126} // namespace
127
128#endif

31/**
32 A 4x4 matrix. Do not subclass. Data is initialized to 0 when default constructed.
33
34 \sa G3D::CoordinateFrame, G3D::Matrix3, G3D::Quat
35 */
36class Matrix4 {
37private:
38
39 float elt[4][4];
40
41 /**
42 Computes the determinant of the 3x3 matrix that lacks excludeRow

53public:
54
55 /** Must be in one of the following forms:
56 - Matrix4(#, #, # .... #)
57 - Matrix4::scale(#)
58 - Matrix4::scale(#, #, #)
59 - Matrix4::translation(#, #, #)
60 - Matrix4::identity()
61 - Matrix4::rollDegrees(#)
62 - Matrix4::pitchDegrees(#)
63 - Matrix4::yawDegrees(#)
64 */
65 explicit Matrix4(const Any& any);

92 static Matrix4 diagonal(float e00, float e11, float e22, float e33) {
93 return Matrix4(e00, 0, 0, 0,
94 0, e11, 0, 0,
95 0, 0, e22, 0,
96 0, 0, 0, e33);
97 }

99 /** Produces an RT transformation that nearly matches this Matrix4.

109 /** If this is a perspective projection matrix created by

122 double& farval,
123 float updirection = -1.0f) const;

125 inline float* operator[](int r) {

132 debugAssert(r >= 0);

141

143 return (const float*)&elt[0][0];
144 }
145
146 Matrix4 operator*(const Matrix4& other) const;
147 Matrix4 operator+(const Matrix4& other) const {
148 Matrix4 result;
149 for (int r = 0; r < 4; ++r) {
150 for (int c = 0; c < 4; ++c) {
151 result.elt[r][c] = elt[r][c] + other.elt[r][c];
152 }
153 }
154 return result;
155 }

238 /** 3D scale matrix */
239 inline static Matrix4 scale(const Vector3& v) {
240 return Matrix4(v.x, 0, 0, 0,
241 0, v.y, 0, 0,
242 0, 0, v.z, 0,
243 0, 0, 0, 1);
244 }

246 /** 3D scale matrix */
247 inline static Matrix4 scale(float x, float y, float z) {
248 return scale(Vector3(x, y, z));
249 }
250
251 /** 3D scale matrix */
252 inline static Matrix4 scale(float s) {
253 return scale(s,s,s);
254 }
255
256 /** 3D translation matrix */
257 inline static Matrix4 translation(const Vector3& v) {
258 return Matrix4(Matrix3::identity(), v);
259 }
260
261 inline static Matrix4 translation(float x, float y, float z) {
262 return Matrix4(Matrix3::identity(), Vector3(x, y, z));
263 }
264
265 /** Create a rotation matrix that rotates \a deg degrees around the Y axis */
266 inline static Matrix4 yawDegrees(float deg) {
267 return Matrix4(Matrix3::fromAxisAngle(Vector3::unitY(), toRadians(deg)));
268 }
269
270 inline static Matrix4 pitchDegrees(float deg) {
271 return Matrix4(Matrix3::fromAxisAngle(Vector3::unitX(), toRadians(deg)));
272 }
273
274 inline static Matrix4 rollDegrees(float deg) {
275 return Matrix4(Matrix3::fromAxisAngle(Vector3::unitZ(), toRadians(deg)));

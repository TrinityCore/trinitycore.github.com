315/** @def PRAGMA(expression)
316 \#pragma may not appear inside a macro, so this uses the pragma operator
317 to create an equivalent statement.*/
318#ifdef _MSC_VER
319// Microsoft's version http://msdn.microsoft.com/en-us/library/d9x1s805.aspx
320# define PRAGMA(x) __pragma(x)
321#else
322// C99 standard http://www.delorie.com/gnu/docs/gcc/cpp_45.html
323# define PRAGMA(x) _Pragma(#x)
324#endif

326/** \def G3D_BEGIN_PACKED_CLASS(byteAlign)
327 Switch to tight alignment.
328
329 \code
330 G3D_BEGIN_PACKED_CLASS(1)
331 ThreeBytes {
332 public:
333 uint8 a, b, c;
334 }
335 G3D_END_PACKED_CLASS(1)
336 \endcode
337
338
339 See G3D::Color3uint8 for an example.*/
340#ifdef __GNUC__
341# define G3D_BEGIN_PACKED_CLASS(byteAlign) class __attribute((__packed__))
342#elif defined(_MSC_VER)
343# define G3D_BEGIN_PACKED_CLASS(byteAlign) PRAGMA( pack(push, byteAlign) ) class
344#else
345# define G3D_BEGIN_PACKED_CLASS(byteAlign) class
346#endif
347
348/** \def G3D_END_PACKED_CLASS(byteAlign)
349 End switch to tight alignment
350
351 See G3D::Color3uint8 for an example.*/
352#ifdef __GNUC__
353# define G3D_END_PACKED_CLASS(byteAlign) __attribute((aligned(byteAlign))) ;
354#elif defined(_MSC_VER)
355# define G3D_END_PACKED_CLASS(byteAlign) ; PRAGMA( pack(pop) )
356#else
357# define G3D_END_PACKED_CLASS(byteAlign) ;
358#endif

386 /** Options for initG3D and initGLG3D. */
387 class G3DSpecification {
388 public:
389 /**
390 \brief Should G3D spawn its own network thread?
391
392 If true, G3D will spawn a thread for network management on the first invocation of G3D::NetServer::create or
393 G3D::NetConnection::connectToServer.
394
395 If false and networking is used, the application must explicitly invoke G3D::serviceNetwork() regularly to allow the network
396 code to run.
397
398 In either case, the network API is threadsafe.
399
400 Default: true.
401 */
402 bool threadedNetworking;
403
404 G3DSpecification() : threadedNetworking(true) {}
405
406 virtual ~G3DSpecification() {}
407 };

19namespace G3D {
20
21class Box;
22
23/** \see Projection */
24class Frustum {
25public:
26 class Face {
27 public:
28 /** Counter clockwise indices into vertexPos */
29 int vertexIndex[4];
30
31 /** The plane containing the face. */
32 Plane plane;
33 };
34
35 /** The vertices, in homogeneous space. The order is that of
36 the near face, starting from the (object space) +x,+y corner
37 and proceeding CCW from the camera's point of view; followed
38 by the far face also in CCW order.
39
40 If w == 0,
41 a vertex is at infinity. */
42 SmallArray<Vector4, 8> vertexPos;
43
44 /** The faces in the frustum. When the
45 far plane is at infinity, there are 5 faces,
46 otherwise there are 6. The faces are in the order
47 N,R,L,B,T,[F].
48 */
49 SmallArray<Face, 6> faceArray;
50
51 /** \param minObjectSpaceDepth Smallest value permitted for the near plane Z - far plane Z (e.g., to force finite bounds)*/
52 Box boundingBox(float minObjectSpaceDepth = finf()) const;
53};
54
55} // namespace G3D

23/** Prints to the common system log, log.txt, which is usually
24 in the working directory of the program. If your disk is
25 not writable or is slow, it will attempt to write to "c:/tmp/log.txt" or
26 "c:/temp/log.txt" on Windows systems instead.
27
28 Unlike printf or debugPrintf,
29 this function guarantees that all output is committed before it returns.
30 This is very useful for debugging a crash, which might hide the last few
31 buffered print statements otherwise.
32
33 Many G3D routines write useful warnings and debugging information to the
34 system log, which makes it a good first place to go when tracking down
35 a problem.
36 */
37void logPrintf(const char* fmt, ...);
38
39/** Does not flush the buffer; follow up with a logPrintf to force the flush. */
40void logLazyPrintf(const char* fmt, ...);
41
42/**
43 System log for debugging purposes. The first log opened
44 is the "common log" and can be accessed with the static
45 method common(). If you access common() and a common log
46 does not yet exist, one is created for you.
47 */
48class Log {
49private:
50
51 /**
52 Log messages go here.
53 */
54 FILE* logFile;
55
56 std::string filename;
57
58 static Log* commonLog;
59
60public:
61 int stripFromStackBottom;
62
63
64 /**
65 @param stripFromStackBottom Number of call stacks to strip from the
66 bottom of the stack when printing a trace. Useful for hiding
67 routines like "main" and "WinMain". If the specified file cannot
68 be opened for some reason, tries to open "c:/tmp/log.txt" or
69 "c:/temp/log.txt" instead.
70 */
71 Log(const std::string& filename = "log.txt",
72 int stripFromStackBottom = 0);
73
74 virtual ~Log();
75
76 /**
77 Returns the handle to the file log.
78 */
79 FILE* getFile() const;
80
81 /**
82 Marks the beginning of a logfile section.
83 */
84 void section(const std::string& s);
85
86 /**
87 Given arguments like printf, writes characters to the debug text overlay.
88 */
89 // We want G3D_CHECK_PRINTF_ARGS here, but that conflicts with the
90 // overload.
91 void __cdecl printf(const char* fmt, ...) G3D_CHECK_PRINTF_METHOD_ARGS;
92
93 void __cdecl vprintf(const char*, va_list argPtr) G3D_CHECK_VPRINTF_METHOD_ARGS;
94 /** Does not flush */
95 void __cdecl lazyvprintf(const char*, va_list argPtr) G3D_CHECK_VPRINTF_METHOD_ARGS;
96
97 static Log* common();
98
99 static std::string getCommonLogFilename();
100
101 void print(const std::string& s);
102
103
104 void println(const std::string& s);
105};

21/**
22 Convenient formatting of ASCII text written to a file.
23 <P>
24
25 The core writeString, writeNumber, and writeSymbol methods map to TextInput's
26 methods. Number and Symbol each print an additional space that is used to
27 separate adjacent tokens.
28
29 TextOutput::printf allows arbitrary text to be conveniently dumped
30 en-masse. Use [de]serialize(bool, TextOutput) and other overloads to read/write
31 primitive types in a standardized manner and
32
33 <P>
34 When a word-wrap line break occurs, all whitespace between words is replaced
35 with a single newline (the newline may be two characters-- see
36 G3D::TextOutput::Options::NewlineStyle). Word wrapping occurs against
37 the number of columns specified by Options::numColumns, <I>minus</I> the current
38 indent level.
39
40 Indenting adds the specified number of spaces immediately after a newline.
41 If a newline was followed by spaces in the original string, these are added
42 to the indent spaces. Indenting <B>will</B> indent blank lines and will leave
43 indents after the last newline of a file (if the indent level is non-zero at the end).
44
45 <P><B>Serialization/Marshalling</B>
46 <DT>Text serialization is accomplished using TextOutput by defining the pair of
47 methods:
48
49 <PRE>
50 void serialize(TextOutput& to) const;
51 void deserialize(TextInput& ti);
52 </PRE>
53
54 See also G3D::TextInput.
55
56 <P>
57 <B>BETA API</B>
58 <DT>This API is subject to change in future versions.
59 */
60class TextOutput {
61public:
62
63 class Settings {
64 public:
65 /**
66 WRAP_NONE Word wrapping is disabled
67 WRAP_WITHOUT_BREAKING Word-wrap, but don't break continuous lines that
68 are longer than numColumns (default)
69 WRAP_ALWAYS Wrap even if it means breaking a continuous line or
70 a quoted string.
71
72 Word wrapping is only allowed at whitespaces ('\\n', '\\r', '\\t', ' '); it
73 will not occur after commas, punctuation, minus signs, or any other characters
74 */
75 enum WordWrapMode {WRAP_NONE, WRAP_WITHOUT_BREAKING, WRAP_ALWAYS};
76
77 /** Defaults to WRAP_WITHOUT_BREAKING */
78 WordWrapMode wordWrap;
79
80 /** Is word-wrapping allowed to insert newlines inside double quotes?
81 Default: false */
82 bool allowWordWrapInsideDoubleQuotes;
83
84 /** Number of columns for word wrapping. Default: 8 */
85 int numColumns;
86
87 /** Number of spaces in each indent. Default: 4 */
88 int spacesPerIndent;
89
90 /** Style of newline used by word wrapping and by (optional) conversion.
91 default: Windows: NEWLINE_WINDOWS, Linux, OS X: NEWLINE_UNIX.
92 */
93 enum NewlineStyle {NEWLINE_WINDOWS, NEWLINE_UNIX};
94
95 NewlineStyle newlineStyle;
96
97 /** If true, all newlines are converted to NewlineStyle regardless of
98 how they start out. Default: true. */
99 bool convertNewlines;
100
101 /** Used by writeBoolean */
102 std::string trueSymbol;
103
104 /** Used by writeBoolean */
105 std::string falseSymbol;
106
107 Settings() :
108 wordWrap(WRAP_WITHOUT_BREAKING),
109 allowWordWrapInsideDoubleQuotes(false),
110 numColumns(80),
111 spacesPerIndent(4),
112 convertNewlines(true),
113 trueSymbol("true"),
114 falseSymbol("false") {
115 #ifdef G3D_WINDOWS
116 newlineStyle = NEWLINE_WINDOWS;
117 #else
118 newlineStyle = NEWLINE_UNIX;
119 #endif
120 }
121 };
122
123private:
124
125 /** Used by indentAndAppend to tell when we are writing the
126 first character of a new line.
127
128 So that push/popIndent work correctly, we cannot indent
129 immediately after writing a newline. Instead we must
130 indent on writing the first character <B>after</B> that
131 newline.
132 */
133 bool startingNewLine;
134
135 /** Number of characters at the end of the buffer since the last newline */
136 int currentColumn;
137
138 /** True if we have seen an open " and no close ".*/
139 bool inDQuote;
140
141 /** Empty if there is none */
142 std::string filename;
143
144 Array<char> data;
145
146 Settings option;
147
148 /** Number of indents to prepend before each line. Always set using setIndentLevel.*/
149 int indentLevel;
150
151 void setIndentLevel(int i);
152
153 /** Actual number of spaces to indent. */
154 int indentSpaces;
155
156 /** the newline character(s) */
157 std::string newline;
158
159 /** Starts at 1 */
160 int m_currentLine;
161
162 void setOptions(const Settings& _opt);
163
164 /** Converts to the desired newlines. Called from vprintf */
165 void convertNewlines(const std::string& in, std::string& out);
166
167 /** Called from vprintf */
168 void wordWrapIndentAppend(const std::string& str);
169
170 /** Appends the character to data, indenting whenever a newline is encountered.
171 Called from wordWrapIndentAppend */
172 void indentAppend(char c);

174public:
175
176 explicit TextOutput(const std::string& filename, const Settings& options = Settings());
177
178 /** Constructs a text output that can later be commited to a string instead of a file.*/
179 explicit TextOutput(const Settings& options = Settings());
180
181 /** Returns one plus the number of newlines written since the output was created. */
182 int line() const {
183 return m_currentLine;
184 }
185
186 /** Commit to the filename specified on the constructor.
187 <B>Not</B> called from the destructor; you must call
188 it yourself.
189 @param flush If true (default) the file is ready for reading when the method returns, otherwise
190 the method returns immediately and writes the file in the background.*/
191 void commit(bool flush = true);
192
193 /** Commits to this string */
194 void commitString(std::string& string);
195
196 /** Increase indent level by 1 */
197 void pushIndent();
198
199 void popIndent();
200
201 /** Produces a new string that contains the output */
202 std::string commitString();
203
204 /** Writes a quoted string. Special characters in the string (e.g., \\, \\t, \\n) are escaped so that
205 TextInput will produce the identical string on reading.*/
206 void writeString(const std::string& string);
207
208 void writeBoolean(bool b);
209
210 void writeNumber(double n);
211
212 void writeNumber(int n);
213
214 void writeNewline();
215 void writeNewlines(int numLines);
216
217 /** If the most recently written character was a space, remove it and return true. Can be called repeatedly to back up over multiple spaces. */
218 bool deleteSpace();
219
220 /** The symbol is written without quotes. Symbols are required to begin with a
221 letter or underscore and contain only letters, underscores, and numbers
222 or be a C++ symbol (e.g. "{", "(", "++", etc.)
223 so that they may be properly parsed by TextInput::readSymbol. Symbols are
224 printed with a trailing space.*/
225 void writeSymbol(const std::string& string);
226
227 void writeSymbol(char s);
228
229 /** Convenient idiom for writing multiple symbols in a row, e.g.
230 writeSymbols("name", "="); The empty symbols are not written.
231 */
232 void writeSymbols(
233 const std::string& a,
234 const std::string& b = "",
235 const std::string& c = "",
236 const std::string& d = "",
237 const std::string& e = "",
238 const std::string& f = "");
239
240 /** Normal printf conventions. Note that the output will be reformatted
241 for word-wrapping and newlines */
242 void __cdecl printf(const char* fmt, ...)
243 G3D_CHECK_PRINTF_METHOD_ARGS;
244
245 // Can't pass by reference because that confuses va_start
246 void __cdecl printf(const std::string fmt, ...);
247 void __cdecl vprintf(const char* fmt, va_list argPtr)
248 G3D_CHECK_VPRINTF_METHOD_ARGS;
249};

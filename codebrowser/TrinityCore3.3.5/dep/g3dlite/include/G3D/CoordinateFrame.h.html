38/**
39\brief A rigid body RT (rotation-translation) transformation.
40
41CoordinateFrame abstracts a 4x4 matrix that maps object space to world space:
42
43 v_world = C * v_object
44
45CoordinateFrame::rotation is the upper 3x3 submatrix, CoordinateFrame::translation
46is the right 3x1 column. The 4th row is always [0 0 0 1], so it isn't stored.
47So you don't have to remember which way the multiplication and transformation work,
48it provides explicit toWorldSpace and toObjectSpace methods. Also, points, vectors
49(directions), and surface normals transform differently, so they have separate methods.
50
51Some helper functions transform whole primitives like boxes in and out of object space.
52
53Convert to Matrix4 using CoordinateFrame::toMatrix4. You <I>can</I> construct a CoordinateFrame
54from a Matrix4 using Matrix4::approxCoordinateFrame, however, because a Matrix4 is more
55general than a CoordinateFrame, some information may be lost.
56
57\sa G3D::UprightFrame, G3D::PhysicsFrame, G3D::Matrix4, G3D::Quat
58*/

133 /**
134 Computes the inverse of this coordinate frame.
135 */
136 inline CoordinateFrame inverse() const {
137 CoordinateFrame out;
138 out.rotation = rotation.transpose();
139 out.translation = -out.rotation * translation;
140 return out;
141 }

158 /**
159 Returns the heading of the lookVector as an angle in radians relative to
160 the world -z axis. That is, a counter-clockwise heading where north (-z)
161 is 0 and west (-x) is PI/2.
162
163 Note that the heading ignores the Y axis, so an inverted
164 object has an inverted heading.
165 */
166 inline float getHeading() const {
167 Vector3 look = rotation.column(2);
168 float angle = -(float) atan2(-look.x, look.z);
169 return angle;
170 }

181 return this->inverse().toWorldSpace(v);
182 }
183
184 inline Vector4 toWorldSpace(const Vector4& v) const {
185 return Vector4(rotation * Vector3(v.x, v.y, v.z) + translation * v.w, v.w);
186 }
187
188 /**
189 Transforms the point into world space.
190 */
191 inline Point3 pointToWorldSpace(const Point3& v) const {
192 return Point3
193 (rotation[0][0] * v[0] + rotation[0][1] * v[1] + rotation[0][2] * v[2] + translation[0],
194 rotation[1][0] * v[0] + rotation[1][1] * v[1] + rotation[1][2] * v[2] + translation[1],
195 rotation[2][0] * v[0] + rotation[2][1] * v[1] + rotation[2][2] * v[2] + translation[2]);
196 }
197
198 /**
199 Transforms the point into object space. Assumes that the rotation matrix is orthonormal.
200 */
201 inline Point3 pointToObjectSpace(const Point3& v) const {
202 float p[3];
203 p[0] = v[0] - translation[0];
204 p[1] = v[1] - translation[1];
205 p[2] = v[2] - translation[2];
206 debugAssert(G3D::fuzzyEq(fabsf(rotation.determinant()), 1.0f));
207 return Point3(rotation[0][0] * p[0] + rotation[1][0] * p[1] + rotation[2][0] * p[2],
208 rotation[0][1] * p[0] + rotation[1][1] * p[1] + rotation[2][1] * p[2],
209 rotation[0][2] * p[0] + rotation[1][2] * p[1] + rotation[2][2] * p[2]);
210 }

281 CoordinateFrame operator*(const CoordinateFrame &other) const {
282 return CoordinateFrame(rotation * other.rotation,

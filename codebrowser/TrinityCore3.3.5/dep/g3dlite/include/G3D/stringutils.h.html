31/** Finds the index of the first '\\' or '/' character, starting at index \a start.
32 \sa G3D::findLastSlash, G3D::isSlash
33*/
34inline size_t findSlash(const std::string& f, size_t start = 0) {
35 size_t i = f.find('/', start);
36 size_t j = f.find('\\', start);
37 if ((i != std::string::npos) && (i < j)) {
38 return i;
39 } else {
40 return j;
41 }
42}

44/** \brief Returns the larger string index, ignoring std::string::npos. */
45inline size_t maxNotNPOS(size_t i, size_t j) {
46 if (i == std::string::npos) {
47 return j;
48 } else if (j == std::string::npos) {
49 return i;
50 } else {
51 return max(i, j);
52 }
53}
54
55/** Finds the index of the first '\\' or '/' character, starting at index \a start (if \a start is -1, starts at the end of the string).
56 \sa G3D::findSlash, G3D::isSlash
57 */
58inline size_t findLastSlash(const std::string& f, size_t start = std::string::npos) {
59 if (start == std::string::npos) {
60 start = f.length() - 1;
61 }
62
63 size_t i = f.rfind('/', start);
64 size_t j = f.rfind('\\', start);
65 return maxNotNPOS(i, j);
66}

183/** Number of new lines in the given string */
184inline int countNewlines(const std::string& s) {
185 int c = 0;
186 for (int i = 0; i < (int)s.size(); ++i) {
187 if (s[i] == '\n') {
188 ++c;
189 }
190 }
191 return c;
192}

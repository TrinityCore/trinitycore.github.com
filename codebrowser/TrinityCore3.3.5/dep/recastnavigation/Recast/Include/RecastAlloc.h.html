19#ifndef RECASTALLOC_H
20#define RECASTALLOC_H

22/// Provides hint values to the memory allocator on how long the
23/// memory is expected to be used.
24enum rcAllocHint
25{
26 RC_ALLOC_PERM, ///< Memory will persist after a function call.
27 RC_ALLOC_TEMP ///< Memory used temporarily within a function.
28};
29
30/// A memory allocation function.
31// @param[in] size The size, in bytes of memory, to allocate.
32// @param[in] rcAllocHint A hint to the allocator on how long the memory is expected to be in use.
33// @return A pointer to the beginning of the allocated memory block, or null if the allocation failed.
34/// @see rcAllocSetCustom
35typedef void* (rcAllocFunc)(int size, rcAllocHint hint);
36
37/// A memory deallocation function.
38/// @param[in] ptr A pointer to a memory block previously allocated using #rcAllocFunc.
39/// @see rcAllocSetCustom
40typedef void (rcFreeFunc)(void* ptr);
41
42/// Sets the base custom allocation functions to be used by Recast.
43/// @param[in] allocFunc The memory allocation function to be used by #rcAlloc
44/// @param[in] freeFunc The memory de-allocation function to be used by #rcFree
45void rcAllocSetCustom(rcAllocFunc *allocFunc, rcFreeFunc *freeFunc);
46
47/// Allocates a memory block.
48/// @param[in] size The size, in bytes of memory, to allocate.
49/// @param[in] hint A hint to the allocator on how long the memory is expected to be in use.
50/// @return A pointer to the beginning of the allocated memory block, or null if the allocation failed.
51/// @see rcFree
52void* rcAlloc(int size, rcAllocHint hint);
53
54/// Deallocates a memory block.
55/// @param[in] ptr A pointer to a memory block previously allocated using #rcAlloc.
56/// @see rcAlloc
57void rcFree(void* ptr);

60/// A simple dynamic array of integers.
61class rcIntArray
62{
63 int* m_data;
64 int m_size, m_cap;
65 inline rcIntArray(const rcIntArray&);
66 inline rcIntArray& operator=(const rcIntArray&);
67public:
68
69 /// Constructs an instance with an initial array size of zero.
70 inline rcIntArray() : m_data(0), m_size(0), m_cap(0) {}
71
72 /// Constructs an instance initialized to the specified size.
73 /// @param[in] n The initial size of the integer array.
74 inline rcIntArray(int n) : m_data(0), m_size(0), m_cap(0) { resize(n); }
75 inline ~rcIntArray() { rcFree(m_data); }
76
77 /// Specifies the new size of the integer array.
78 /// @param[in] n The new size of the integer array.
79 void resize(int n);
80
81 /// Push the specified integer onto the end of the array and increases the size by one.
82 /// @param[in] item The new value.
83 inline void push(int item) { resize(m_size+1); m_data[m_size-1] = item; }
84
85 /// Returns the value at the end of the array and reduces the size by one.
86 /// @return The value at the end of the array.
87 inline int pop() { if (m_size > 0) m_size--; return m_data[m_size]; }
88
89 /// The value at the specified array index.
90 /// @warning Does not provide overflow protection.
91 /// @param[in] i The index of the value.
92 inline const int& operator[](int i) const { return m_data[i]; }
93
94 /// The value at the specified array index.
95 /// @warning Does not provide overflow protection.
96 /// @param[in] i The index of the value.
97 inline int& operator[](int i) { return m_data[i]; }
98
99 /// The current size of the integer array.
100 inline int size() const { return m_size; }
101};

103/// A simple helper class used to delete an array when it goes out of scope.
104/// @note This class is rarely if ever used by the end user.
105template<class T> class rcScopedDelete
106{
107 T* ptr;
108 inline T* operator=(T* p);
109public:
110
111 /// Constructs an instance with a null pointer.
112 inline rcScopedDelete() : ptr(0) {}
113
114 /// Constructs an instance with the specified pointer.
115 /// @param[in] p An pointer to an allocated array.
116 inline rcScopedDelete(T* p) : ptr(p) {}
117 inline ~rcScopedDelete() { rcFree(ptr); }
118
119 /// The root array pointer.
120 /// @return The root array pointer.
121 inline operator T*() { return ptr; }
122};

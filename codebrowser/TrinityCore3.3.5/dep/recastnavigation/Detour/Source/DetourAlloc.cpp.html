19#include <stdlib.h>
20#include "DetourAlloc.h"
21
22static void *dtAllocDefault(int size, dtAllocHint)
23{
24 return malloc(size);
25}
26
27static void dtFreeDefault(void *ptr)
28{
29 free(ptr);
30}
31
32static dtAllocFunc* sAllocFunc = dtAllocDefault;
33static dtFreeFunc* sFreeFunc = dtFreeDefault;
34
35void dtAllocSetCustom(dtAllocFunc *allocFunc, dtFreeFunc *freeFunc)
36{
37 sAllocFunc = allocFunc ? allocFunc : dtAllocDefault;
38 sFreeFunc = freeFunc ? freeFunc : dtFreeDefault;
39}
40
41void* dtAlloc(int size, dtAllocHint hint)
42{
43 return sAllocFunc(size, hint);
44}
45
46void dtFree(void* ptr)
47{
48 if (ptr)
49 sFreeFunc(ptr);
50}
